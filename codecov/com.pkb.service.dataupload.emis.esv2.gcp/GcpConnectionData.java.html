package com.pkb.service.dataupload.emis.esv2.gcp;

import java.net.URI;
import java.net.URL;
import java.time.Duration;

import org.immutables.value.Value;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;

import com.pkb.service.dataupload.emis.esv2.ConnectionData;
import com.pkb.service.dataupload.emis.esv2.IEmisDownloadConnection;

@JsonDeserialize(as = ImmutableGcpConnectionData.class)
@Value.Immutable
public abstract class GcpConnectionData implements ConnectionData {
    @Value.Parameter
    @Override
    public abstract String uniqueTag();

    @Value.Parameter
    public abstract String keyId();

    @Value.Parameter
    public abstract String keySecret();

    @Value.Parameter
    public abstract URL apiUrl();

    @Value.Parameter
    public abstract String bucketName();

    @Value.Parameter
    public abstract Duration timeout();

    @Override
    public IEmisDownloadConnection connection() {
        return new GcpConnection(this);
    }

    @Override
    public URI uri() {
        return URI.create("gs://" + bucketName());
    }
}
