package com.pkb.api.internal.converter;

import com.pkb.api.internal.model.response.NonSensitivePersonInformationDTO;
import com.pkb.user.entity.PKBPerson;
import org.springframework.core.convert.converter.Converter;
import org.springframework.stereotype.Component;

@Component
public class PKBPersonToNonSensitivePersonInformationDTOConverter implements Converter<PKBPerson, NonSensitivePersonInformationDTO> {

    @Override
    public NonSensitivePersonInformationDTO convert(PKBPerson person) {
        NonSensitivePersonInformationDTO result = new NonSensitivePersonInformationDTO();
        result.setId(person.getId());
        result.setHumanUUID(person.getHumanUUID());
        result.setFirstName(person.getFirstName());
        result.setLastName(person.getLastName());
        result.setEmail(person.getEmail());
        result.setRecordFrozen(person.isFrozenByTeam());
        result.setUserStatus(person.getStatus());
        result.setUserType(person.getUserType());
        return result;
    }
}
