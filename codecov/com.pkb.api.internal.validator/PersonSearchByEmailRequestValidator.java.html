package com.pkb.api.internal.validator;

import com.pkb.api.internal.exception.RequestValidationException;
import com.pkb.api.internal.exception.RequestValidationExceptionBuilder;
import com.pkb.api.internal.model.requests.PersonSearcRequestDto;
import com.pkb.util.PKBEmailValidator;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;

import static org.apache.commons.collections4.CollectionUtils.isEmpty;

@Component
public class PersonSearchByEmailRequestValidator extends AbstractInternalRequestValidator {

    @Autowired
    private PKBEmailValidator emailValidator;

    public void validate(PersonSearcRequestDto request) throws RequestValidationException {
        RequestValidationExceptionBuilder exceptionBuilder = createExceptionBuilder();

        if (request == null) {
            exceptionBuilder.addGeneralError("email.search.request.null");
        } else {
            if (isEmpty(request.getValues())) {
                exceptionBuilder.addFieldError("values", request.getValues(), "email.search.request.empty.values");
            } else {
                validateEmailAddresses(request.getValues(), exceptionBuilder);
            }
        }
        exceptionBuilder.throwIfNeeded();
    }

    private void validateEmailAddresses(List<String> emails, RequestValidationExceptionBuilder exceptionBuilder) {
        emails.stream().filter(emailValidator::isNotValid)//
                .forEach(email -> exceptionBuilder.addFieldError("values", email, "email.invalid"));
    }
}
