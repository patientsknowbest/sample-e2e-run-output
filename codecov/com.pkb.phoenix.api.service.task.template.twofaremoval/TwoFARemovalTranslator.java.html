package com.pkb.phoenix.api.service.task.template.twofaremoval;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.phoenix.api.service.task.TaskService;
import com.pkb.phoenix.api.service.task.template.Translator;
import com.pkb.phoenix.repository.task.projection.Task;
import com.pkb.phoenix.shared.structures.request.ImmutableTwoFARemovalTaskRequest;
import com.pkb.phoenix.shared.structures.request.TwoFARemovalTaskRequest;
import com.pkb.phoenix.shared.structures.response.TwoFARemovalTaskResponse;
import reactor.util.context.ContextView;

import java.util.function.Function;

import static java.util.Optional.ofNullable;

public class TwoFARemovalTranslator implements Translator<TwoFARemovalParameters, TwoFARemovalTaskRequest, TwoFARemovalTaskResponse, TwoFARemovalResult> {
    private final DateTimeService dateTimeService;

    public TwoFARemovalTranslator(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }
    @Override
    public Function<Task<TwoFARemovalParameters>, TwoFARemovalTaskRequest> toRequest(ContextView context) {

        return twoFARemovalParametersTask -> {
            TwoFARemovalParameters twoFARemovalParameters = twoFARemovalParametersTask.getParameters();

            return ImmutableTwoFARemovalTaskRequest.builder()
                    .taskId(twoFARemovalParametersTask.getId())
                    .requestId(context.get(TaskService.TASK_ID).toString())
                    .freshDeskId(twoFARemovalParameters.fdId())
                    .email(twoFARemovalParameters.email())
                    .date(twoFARemovalParameters.date() == null ? dateTimeService.nowLocalDateTime().toString() : twoFARemovalParameters.date())
                    .build();
        };
    }

    @Override
    public Function<TwoFARemovalTaskResponse, TwoFARemovalResult> fromResponse(ContextView context) {
        return response ->
                ImmutableTwoFARemovalResult.builder()
                        .status(response.status())
                        .outcome(ofNullable(response.results()).orElse(""))
                        .build();
    }

    @Override
    public Class<TwoFARemovalTaskResponse> responseType() {
        return TwoFARemovalTaskResponse.class;
    }
}
