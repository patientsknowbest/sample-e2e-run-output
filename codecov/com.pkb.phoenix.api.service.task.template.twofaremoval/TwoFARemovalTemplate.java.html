package com.pkb.phoenix.api.service.task.template.twofaremoval;

import com.fasterxml.jackson.databind.ObjectReader;

import com.pkb.phoenix.api.service.freshdesk.FreshDeskService;

import static com.pkb.phoenix.api.service.task.TaskService.TASK_INSTANCE;
import static com.pkb.phoenix.api.service.task.template.TaskTemplateCode.TWO_FA_REMOVAL;
import static com.pkb.phoenix.api.service.tasktype.ValidationUtil.validateEmail;
import static com.pkb.phoenix.api.service.tasktype.ValidationUtil.validateFreshDeskId;
import static reactor.core.publisher.Mono.just;

public final class TwoFARemovalTemplate extends AbstractPostProcessingTaskTemplate<TwoFARemovalParameters, TwoFARemovalResult, TwoFARemovalResult.FdTwoFARemovalResult> {
    public static final String OUTBOUND_REQUEST_TWO_FA_REMOVAL = "direct:outboundRequestTwoFARemoval";

    private static final class BaseTwoFARemovalTemplate
            extends AbstractPubSubTaskTemplate<TwoFARemovalParameters, TwoFARemovalTaskRequest, TwoFARemovalTaskResponse, TwoFARemovalResult> {
        BaseTwoFARemovalTemplate(ReactiveRedisTemplate<String, String> redis,
                ProducerTemplate producer,
                DateTimeService dateTimeService,
                ObjectReader json) {
            super(TWO_FA_REMOVAL, TwoFARemovalParameters.class, new TwoFARemovalTranslator(dateTimeService), redis, producer, OUTBOUND_REQUEST_TWO_FA_REMOVAL, json);
        }

        @Override
        public List<String> validate(TwoFARemovalParameters parameters) {
            List<String> errors = new ArrayList<>();
            validateEmail(parameters.email()).ifPresent(errors::add);
            validateFreshDeskId(parameters.fdId()).ifPresent(errors::add);
            return errors;
        }
    }

    private final VelocityEngine templateEngine;
    private final FreshDeskService freshDeskService;

    public TwoFARemovalTemplate(ReactiveRedisTemplate<String, String> redis,
            ProducerTemplate producer,
            VelocityEngine templateEngine,
            FreshDeskService freshDeskService,
            DateTimeService dateTimeService,
            ObjectReader json) {
        super(new BaseTwoFARemovalTemplate(redis, producer, dateTimeService, json), json);
        this.templateEngine = templateEngine;
        this.freshDeskService = freshDeskService;
    }

    @Override
    protected Mono<TwoFARemovalResult.FdTwoFARemovalResult> postProcess(TwoFARemovalResult twoFARemovalResult, ContextView context) {
        var meta = context.<Meta> get(TaskService.META);
        var parameters = context.<Task<TwoFARemovalParameters>> get(TASK_INSTANCE).getParameters();

        return meta.session().user().fdApiKey()
                .map(ImmutableApiKey::of)
                .map(apiKey -> {
                    if (twoFARemovalResult.status().isSuccessful()) {
                        return freshDeskService.addNote(parameters.fdId(), renderMessage(parameters.email()), apiKey)
                                .then(freshDeskService.closeTicket(parameters.fdId(), apiKey))
                                .map($1 -> "Done");
                    }

                    return just("Skipped");
                })
                .orElseGet(() -> just("Skipped"))
                .onErrorResume($ -> just("Failed"))
                .map(status -> ImmutableFdTwoFARemovalResult.builder()
                        .outcome(twoFARemovalResult.outcome())
                        .fdStatus(status)
                        .status(twoFARemovalResult.status())
                        .build());

    }

    private String renderMessage(String email) {
        var template = templateEngine.getTemplate("twoFARemovalTemplate.vm");

        try (var writer = new StringWriter()) {
            var context = new VelocityContext();

            context.put("email", email);

            template.merge(context, writer);
            writer.flush();

            return writer.toString();
        } catch (IOException e) {
            throw new IllegalStateException(e);
        }
    }

}
