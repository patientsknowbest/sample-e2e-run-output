package com.pkb.aggregator.service.reference

import com.pkb.aggregator.R4IdType
import com.pkb.aggregator.R4Reference
import com.pkb.aggregator.extension.simpleUpstreamResourceId
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_APPOINTMENT
import com.pkb.aggregator.service.FHIR_TYPE_CONSENT
import com.pkb.aggregator.service.FHIR_TYPE_ENCOUNTER
import com.pkb.aggregator.service.MedicationIdentifierService
import com.pkb.aggregator.service.OrganizationIdentifierService
import com.pkb.aggregator.service.PatientIdentifierService
import com.pkb.aggregator.service.PractitionerIdentifierService
import com.pkb.aggregator.service.exception.UpstreamReferenceException
import com.pkb.aggregator.service.reference.extract.AppointmentExtractor
import com.pkb.aggregator.service.reference.extract.ConsentExtractor

import org.hl7.fhir.instance.model.api.IDomainResource
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

/**
 * A service for collecting references contained in resources, which will need translating to downstream references in
 * order for relationships to be maintained in the aggregated store
 */
@Service
class ReferenceExtractor(
    private val practitionerExtractors: List<PractitionerExtractor>,
    private val organizationExtractors: List<OrganizationExtractor>,
    private val medicationExtractors: List<MedicationExtractor>,
    private val encounterExtractors: List<EncounterExtractor>,
    private val appointmentExtractors: List<AppointmentExtractor>,
    private val patientExtractors: List<PatientExtractor>,
    private val consentExtractors: List<ConsentExtractor>,
    private val practitionerIdentifierService: PractitionerIdentifierService,
    private val organizationIdentifierService: OrganizationIdentifierService,
    private val medicationIdentifierService: MedicationIdentifierService,
    private val patientIdentifierService: PatientIdentifierService,
) {
    private val log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    data class IdentifierMap(
        val patients: MutableSet<UpstreamDownstreamIdPair>,
        val practitioners: MutableSet<UpstreamDownstreamIdPair>,
        val organizations: MutableSet<UpstreamDownstreamIdPair>,
        val encounters: MutableSet<UpstreamDownstreamIdPair>,
        val medications: MutableSet<UpstreamDownstreamIdPair>,
        val appointments: MutableSet<UpstreamDownstreamIdPair>,
        val consents: MutableSet<UpstreamDownstreamIdPair>,
    ) {
        fun getPatientReference(needle: UpstreamResource) = get(needle, patients)
        fun getPractitionerReference(needle: UpstreamResource) = get(needle, practitioners)
        fun getOrganizationReference(needle: UpstreamResource) = get(needle, organizations)
        fun getEncounterReference(needle: UpstreamResource) = get(needle, encounters)
        fun getMedicationReference(needle: UpstreamResource) = get(needle, medications)
        fun getAppointmentReference(needle: UpstreamResource) = get(needle, appointments)
        fun getConsentReference(needle: UpstreamResource) = get(needle, consents)

        private fun get(needle: UpstreamResource, haystack: Set<UpstreamDownstreamIdPair>) =
            haystack.find { it.upstream == needle }
                ?.downstreamId
                ?.let { R4Reference(it) }
                ?: throw UpstreamReferenceException(needle)

        fun findByDownstreamResourceId(resourceId: R4IdType) = when (val resourceType = resourceId.resourceType) {
            FHIR_TYPE_ENCOUNTER -> encounters.find { it.downstreamId == resourceId.asStringValue() }
            FHIR_TYPE_APPOINTMENT -> appointments.find { it.downstreamId == resourceId.asStringValue() }
            else -> throw RuntimeException("Unexpected resource type $resourceType")
        }
    }

    fun extract(resources: List<IDomainResource>) = IdentifierMap(
        patients = mapPatients(resources),
        practitioners = mapPractitioners(resources),
        organizations = mapOrganizations(resources),
        encounters = mapEncounters(resources),
        medications = mapMedications(resources),
        appointments = mapAppointments(resources),
        consents = mapConsents(resources),
    ).also {
        if (log.isDebugEnabled) {
            log.debug("Extracted references for resources {}: {}", resources.map { it.simpleUpstreamResourceId() }, it)
        }
    }

    private fun mapPatients(resources: List<IDomainResource>): MutableSet<UpstreamDownstreamIdPair> {

        val patients = patientExtractors.flatMap { it.extract(resources) }.ifEmpty { return mutableSetOf() }

        return patientIdentifierService.findByUpstreamResourceIdIn(patients.toSet())
            .map { UpstreamDownstreamIdPair(it.mapEntryKey.upstreamResource, it.downstreamResourceId) }
            .toMutableSet()
    }

    private fun mapOrganizations(resources: List<IDomainResource>): MutableSet<UpstreamDownstreamIdPair> {

        val organizationIds = organizationExtractors.flatMap { it.extract(resources) }.ifEmpty { return mutableSetOf() }

        return organizationIdentifierService.findByUpstreamResourceIdIn(organizationIds.toSet())
            .map { UpstreamDownstreamIdPair(it.mapEntryKey.upstreamResource, it.downstreamResourceId) }
            .toMutableSet()
    }

    private fun mapPractitioners(resources: List<IDomainResource>): MutableSet<UpstreamDownstreamIdPair> {

        val practitionerIds = practitionerExtractors.flatMap { it.extract(resources) }.ifEmpty { return mutableSetOf() }
        return practitionerIdentifierService.findByUpstreamResourceIdIn(practitionerIds.toSet())
            .map { UpstreamDownstreamIdPair(it.mapEntryKey.upstreamResource, it.downstreamResourceId) }
            .toMutableSet()
    }

    private fun mapMedications(resources: List<IDomainResource>): MutableSet<UpstreamDownstreamIdPair> {

        val medications = medicationExtractors.flatMap { it.extract(resources) }.ifEmpty { return mutableSetOf() }

        return medicationIdentifierService.findByUpstreamResourceIdIn(medications.toSet())
            .map { UpstreamDownstreamIdPair(it.mapEntryKey.upstreamResource, it.downstreamResourceId) }
            .toMutableSet()
    }

    private fun mapEncounters(resources: List<IDomainResource>) =
        encounterExtractors
            .flatMap { it.extract(resources) }
            .map { UpstreamDownstreamIdPair(it, "$FHIR_TYPE_ENCOUNTER/${it.toDownstreamResourceId()}") }
            .toMutableSet()

    private fun mapAppointments(resources: List<IDomainResource>) =
        appointmentExtractors
            .flatMap { it.extract(resources) }
            .map { UpstreamDownstreamIdPair(it, "$FHIR_TYPE_APPOINTMENT/${it.toDownstreamResourceId()}") }
            .toMutableSet()

    private fun mapConsents(resources: List<IDomainResource>) =
        consentExtractors
            .flatMap { it.extract(resources) }
            .map { UpstreamDownstreamIdPair(it, "$FHIR_TYPE_CONSENT/${it.toDownstreamResourceId()}") }
            .toMutableSet()
}
