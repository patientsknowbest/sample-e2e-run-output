package com.pkb.service.code;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.code.entity.Code;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.encounter.entity.Encounter;
import com.pkb.encounter.entity.Message;
import com.pkb.repository.legacy.LegacyCodeRepository;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;

import javax.annotation.Nullable;
import java.util.Collections;
import java.util.List;
import java.util.Map;
import java.util.Objects;
import java.util.Optional;
import java.util.stream.Collectors;

import static java.util.function.UnaryOperator.identity;
import static java.util.stream.Collectors.toMap;

public class CodeManager extends TransactionManager {

    private final LegacyCodeRepository repository;

    public CodeManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, LegacyCodeRepository repository) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.repository = repository;
    }

    public Optional<Code> getCode(String code) {
        var items = repository.getCodesByCode(code);
        if (items.isEmpty()) {
            return Optional.empty();
        }
        return Optional.of(items.get(0));
    }

    public List<Long> getCodeIds(List<String> codes) {
        return getCodeMap(codes)
                .values()
                .stream()
                .filter(Objects::nonNull)
                .map(Code::getId)
                .collect(Collectors.toList());
    }

    public void populateCodes(List<Encounter> encounters) {
        var messages = encounters.stream()
                .map(Encounter::getMessages)
                .flatMap(List::stream)
                .collect(Collectors.toList());
        var reasonCodeIds = messages.stream()
                .map(Message::getReasonCodeId)
                .filter(Objects::nonNull)
                .distinct()
                .collect(Collectors.toList());
        if (reasonCodeIds.isEmpty()) {
            return;
        }
        Map<Long, Code> codeMap = getCodeMapByCodeIds(reasonCodeIds);
        // add observation to each message record
        messages.forEach(msg -> Optional.ofNullable(msg.getReasonCodeId())
                .flatMap(codeId -> Optional.ofNullable(codeMap.get(codeId)))
                .ifPresent(code -> msg.setReasonCode(code.getCode())));
    }

    @VisibleForTesting
    Map<String, Code> getCodeMap(@Nullable List<String> codes) {
        if (codes == null || codes.isEmpty()) {
            return Collections.emptyMap();
        }
        return repository.getCodesByCodes(codes).stream().collect(toMap(Code::getCode, identity(), (v1, v2) -> v1));
    }

    @VisibleForTesting
    Map<Long, Code> getCodeMapByCodeIds(@Nullable List<Long> codeIds) {
        if ((codeIds == null) || codeIds.isEmpty()) {
            return Collections.emptyMap();
        }
        return repository.getCodesByCodeIds(codeIds).stream().collect(toMap(Code::getId, identity(), (v1, v2) -> v1));
    }
}
