package com.pkb.service.code;

import com.pkb.VersionDetails;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.coding.entity.CodingTranslation;
import com.pkb.coding.entity.CodingTranslationCodingFields;
import com.pkb.repository.CodingTranslationRepository;
import com.pkb.service.dataupload.emis.esv2.CodeSystems;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.DbUtil;
import org.jetbrains.annotations.NotNull;

import java.util.Collection;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.Set;
import java.util.stream.Collectors;

public class CodingTranslationService {

    private final CodingTranslationRepository repository;
    private final UUIDProvider uuidProvider;

    public CodingTranslationService(CodingTranslationRepository repository, UUIDProvider uuidProvider) {
        this.repository = repository;
        this.uuidProvider = uuidProvider;
    }

    public List<CodingTranslationCodingFields> getAllBySnomedCodesMappedToLoinc(Collection<String> snomedCodes) {
        return DbUtil.getInBatches(snomedCodes,
                batchOfCodes -> repository.getAllTranslationsByCodes(batchOfCodes, CodeSystems.SNOMED_CT, CodeSystems.LOINC));
    }

    public Optional<CodingTranslationCodingFields> getBySnomedCodeMappedToLoinc(String snomedCode) {
        return repository.getTranslationByCode(snomedCode, CodeSystems.SNOMED_CT, CodeSystems.LOINC);
    }

    public void ensureTranslationsExist(Map<String,String> sourceToTargetCode, String sourceCodingSystem, String targetCodingSystem, LoggedInEHRRequestContext requestContext) {
        if (sourceToTargetCode.isEmpty()) {
            return;
        }
        Set<String> codesRequiringTranslation = findCodesRequiringTranslation(sourceToTargetCode);

        VersionDetails vd = VersionDetails.of(requestContext);
        Set<CodingTranslation> newCodingTranslations = codesRequiringTranslation.stream()
                .map(sourceCode -> {
                    String targetCode = sourceToTargetCode.get(sourceCode);
                    return buildCodingTranslation(sourceCodingSystem, targetCodingSystem, vd, sourceCode, targetCode);
                })
                .collect(Collectors.toSet());
        repository.saveAllAndFlush(newCodingTranslations);
    }

    @NotNull
    private CodingTranslation buildCodingTranslation(String sourceCodingSystem, String targetCodingSystem, VersionDetails vd, String sourceCode, String targetCode) {
        CodingTranslation ct = new CodingTranslation(vd);
        ct.setId(uuidProvider.randomUUID());
        ct.setSourceCode(sourceCode);
        ct.setSourceCodingSystem(sourceCodingSystem);
        ct.setTargetCode(targetCode);
        ct.setTargetCodingSystem(targetCodingSystem);
        return ct;
    }

    @NotNull
    private Set<String> findCodesRequiringTranslation(Map<String, String> sourceToTargetCode) {
        Map<String, String> existingSourceToTargetTranslations = DbUtil.getInStreamOfBatches(sourceToTargetCode.keySet(),
                batchOfCodes -> repository.getAllTranslationsByCodes(batchOfCodes, CodeSystems.SNOMED_CT, CodeSystems.LOINC)
        ).collect(Collectors.toMap(CodingTranslationCodingFields::getSourceCode, CodingTranslationCodingFields::getTargetCode));

        return sourceToTargetCode.keySet().stream()
                .filter(sourceCode -> {
                    String existingTranslation = existingSourceToTargetTranslations.get(sourceCode);
                    String desiredTranslation = sourceToTargetCode.get(sourceCode);
                    if (existingTranslation == null) {
                        return true;
                    }
                    if (!existingTranslation.equals(desiredTranslation)) {
                        throw new IllegalStateException("Found different existing translation (" + existingTranslation + ") for code " + sourceCode);
                    }
                    return false;
                })
                .collect(Collectors.toSet());
    }
}
