package com.pkb.domain.provider;

import com.pkb.datamodel.PersonSpecialty;
import com.pkb.datamodel.entity.mapper.PersonSpecialtyMapper;
import com.pkb.entities.pub.PublicPersonSpecialty;
import com.pkb.repository.PersonSpecialtyRepository;
import org.springframework.transaction.annotation.Transactional;

import java.util.Optional;

public class PersonSpecialtyService {

    private final PersonSpecialtyRepository personSpecialtyRepository;
    private final PersonSpecialtyMapper mapper;

    public PersonSpecialtyService(PersonSpecialtyRepository personSpecialtyRepository, PersonSpecialtyMapper mapper) {
        this.personSpecialtyRepository = personSpecialtyRepository;
        this.mapper = mapper;
    }

    public Optional<PersonSpecialty> findPersonSpecialtyByPersonId(long personId) {
        return personSpecialtyRepository.findPersonSpecialtyByPersonId(personId)
                .map(mapper::entityToDatamodel);
    }

    private void delete(PublicPersonSpecialty publicPersonSpecialty) {
        personSpecialtyRepository.delete(publicPersonSpecialty);
    }

    private void save(PublicPersonSpecialty publicPersonSpecialty) {
        personSpecialtyRepository.save(publicPersonSpecialty);
    }

    @Transactional
    public void updatePersonSpecialties(long personId, Long selectedSpecialtyId) {

        Optional<PublicPersonSpecialty> existingSpecialty = personSpecialtyRepository.findPersonSpecialtyByPersonId(personId);
        if (selectedSpecialtyId == null) {
            existingSpecialty.ifPresent(this::delete);
            return;
        }

        Long existingSpecialtyId = existingSpecialty.map(PublicPersonSpecialty::getSpecialtyId).orElse(null);
        if (existingSpecialtyId == null || !existingSpecialtyId.equals(selectedSpecialtyId)) {
            PublicPersonSpecialty newPS = new PublicPersonSpecialty();
            newPS.setPersonId(personId);
            newPS.setSpecialtyId(selectedSpecialtyId);

            save(newPS);
            existingSpecialty.ifPresent(this::delete);
        }
    }
}
