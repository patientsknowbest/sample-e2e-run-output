package com.pkb.aggregator.integration.logging

import ch.qos.logback.classic.Level
import ch.qos.logback.classic.Logger
import ch.qos.logback.classic.turbo.TurboFilter
import ch.qos.logback.core.spi.FilterReply
import com.pkb.aggregator.service.exception.DownstreamResourceExistsException
import com.pkb.aggregator.service.exception.DownstreamVersionConflictException
import org.slf4j.Marker

/**
 * When any error defined here occurs, we allow it to propagate all the way up to the default handler so that the
 * google-pubsub component nacks the message (causing a redelivery). The error handler logs an error when it handles an
 * exception, which is behaviour we _usually_ want but in this case do not because it will trigger alerts for what is
 * effectively a normal, if rare, processing flow. Rather than reimplement the whole error handler to allow it to be
 * configured to not log specific errors, it is easier to just suppress the error with a filter.
 */
class ExpectedErrorTurboFilter : TurboFilter() {

    override fun decide(
        marker: Marker?,
        logger: Logger?,
        level: Level?,
        format: String?,
        params: Array<out Any>?,
        t: Throwable?
    ) =
        if (level == Level.ERROR && t is DownstreamVersionConflictException) {
            FilterReply.DENY
        } else if (level == Level.ERROR && t is DownstreamResourceExistsException) {
            FilterReply.DENY
        } else {
            // NEUTRAL means this filter has no opinion on whether this message gets logged or not
            FilterReply.NEUTRAL
        }
}
