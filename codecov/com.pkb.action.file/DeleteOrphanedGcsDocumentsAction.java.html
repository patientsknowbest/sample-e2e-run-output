package com.pkb.action.file;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.service.file.OrphanedGcsDocumentDeletionManager;
import org.springframework.beans.factory.annotation.Autowired;

import static io.vavr.API.Try;

public class DeleteOrphanedGcsDocumentsAction extends BaseAction {

    @Autowired
    private OrphanedGcsDocumentDeletionManager orphanedGcsDocumentDeletionManager;

    public String deleteOrphanedGcsDocuments() {
        return Try(() -> {
            orphanedGcsDocumentDeletionManager.deleteOrphanedGcsDocuments();
            return Action.SUCCESS;
        }).getOrElseGet(t -> Action.ERROR);
    }
}
