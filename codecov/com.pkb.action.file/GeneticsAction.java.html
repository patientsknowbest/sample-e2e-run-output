package com.pkb.action.file;

import com.pkb.file.FileDtoType;

/**
 * Shows genetic files uploaded in user account
 *
 * @author pravina
 *
 */
public class GeneticsAction extends MyFilesAction {

    private static final String TAB_NAME = "health";
    private static final String SUBTAB_NAME = "genetics";

    private String date;

    @Override
    public String getTab() {
        return TAB_NAME;
    }

    public String getSubTab() {
        return SUBTAB_NAME;
    }

    public String getDate() {
        return date;
    }

    public void setDate(String date) {
        this.date = date;
    }

    @Override
    protected FileDtoType defaultFileType() {
        return FileDtoType.GENETICS_FILE;
    }

}
