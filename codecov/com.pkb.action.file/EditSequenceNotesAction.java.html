package com.pkb.action.file;


import com.pkb.action.BaseAction;
import com.pkb.file.FileDtoType;
import com.pkb.file.entity.FileDTO;
import com.pkb.service.file.FileManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;

import java.time.Instant;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeParseException;
import java.util.Date;

/**
 * Edit a single genetics file's notes and/or date value (other fields and content are non-editable)
 *
 * @author pravina
 */
public class EditSequenceNotesAction extends BaseAction {

    private static final long serialVersionUID = 6591996549478820025L;

    private FileManager fileManager;

    private Long id;

    private FileDTO file;

    private String notes;

    private String date;

    private String tab = "health";

    private DateTimeFormatter dateFormat;

    @Override
    public String execute() {

        // parse date; format already checked somewhat in validate xml (check fully in validate() here?)
        Instant parsed = null;
        try {
            parsed = parseToInstantWithFormatter(getDate(), dateFormat);
        } catch (DateTimeParseException ignored) {
            addFieldError("date", getText("editFileAction.err.invalid_date"));
            return input();
        }

        long userId = loggedInUserUtil.getLoggedInUserId();

        PKBPerson patient = contextUserUtil.getContextUser();
        if (patient == null) {
            patient = userManager.getPKBPerson(loggedInUserUtil.getLoggedInUserId());
        }
        Long patientAccountId = userManager.getDefaultAccountId(patient.getId());

        // retrieve File and set notes

        var geneticsDTO = fileManager.getFile(getLoggedInEHRRequestContext(), getId(), false)
                .orElse(null);
        if (geneticsDTO == null) {
            addActionError(getText("editFileAction.err.already_deleted"));
            return SUCCESS; // redirect back to myFiles
        }
        geneticsDTO.setDate(Date.from(parsed));
        geneticsDTO.setNotes(notes);
        fileManager.saveFile(getEHRRequestContext(), geneticsDTO, patientAccountId, FileDtoType.GENETICS_FILE);

        addActionMessage(getText("editFileAction.msg.saved_successfully"));

        return SUCCESS;
    }

    public String cancel() {
        return SUCCESS;
    }

    @Override
    public String input() {

        // retrieve file and set in action
        file = fileManager.getFile(getLoggedInEHRRequestContext(), getId(), false)
                .orElse(null);
        setDate(dateToString(getFile().getDate(), dateFormat));
        setNotes(getFile().getNotes());

        return INPUT;
    }

    public FileManager getFileManager() {
        return fileManager;
    }

    public void setFileManager(FileManager fileManager) {
        this.fileManager = fileManager;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setTab(String tab) {
        this.tab = tab;
    }

    public String getTab() {
        return tab;
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public FileDTO getFile() {
        return file;
    }

    public void setFile(FileDTO file) {
        this.file = file;
    }

    public String getNotes() {
        return notes;
    }

    public void setNotes(String notes) {
        this.notes = notes;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public String getDate() {
        return date;
    }

    public void setDate(String date) {
        this.date = date;
    }

    @Override
    public void prepare() throws Exception {
        super.prepare();
        dateFormat = getGlobalDateInputFormatter();
    }
}
