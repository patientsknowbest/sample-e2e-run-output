package com.pkb.action.file;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.datamodel.DocumentMetadataIds;
import com.pkb.service.file.DocumentMigrationManager;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import static io.vavr.API.Try;
import static org.apache.commons.lang3.SerializationUtils.deserialize;

public class MigrateDocumentsAction extends BaseAction {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    private DocumentMigrationManager documentMigrationManager;

    public String migrateDocuments() {
        return Try(() -> {
            DocumentMetadataIds dto = deserialize(request.getInputStream());
            LOGGER.info("Migrating documents to GCS: {}", dto.getIds());
            documentMigrationManager.migrateDocument(dto.getIds());
            return Action.SUCCESS;
        }).getOrElseGet(t -> {
            LOGGER.error("Error migrating documents to GCS", t);
            return Action.ERROR;
        });
    }
}
