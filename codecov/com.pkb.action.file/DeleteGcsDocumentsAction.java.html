package com.pkb.action.file;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.service.file.GcsDocumentDeletionManager;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import static io.vavr.API.Try;

public class DeleteGcsDocumentsAction extends BaseAction {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    private GcsDocumentDeletionManager gcsDocumentDeletionManager;

    public String deleteGcsDocuments() {
        return Try(() -> {
            LOGGER.info("Deleting documents from GCS");
            gcsDocumentDeletionManager.deleteGcsDocuments();
            return Action.SUCCESS;
        }).getOrElseGet(t -> {
            LOGGER.error("Error deleting documents from GCS", t);
            return Action.ERROR;
        });
    }
}
