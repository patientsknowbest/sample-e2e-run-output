/**
 * Edit a single file's notes and/or date value (other fields and content are non-editable)
 *
 * @author robwhelan
 */
public class EditFileNotesAction extends ConsentCategoryBaseAction {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private static final long serialVersionUID = 6591996549478820025L;

    private final String tab = "files";
    private DateTimeFormatter dateFormat;

    private FileManager fileManager;

    // When editing data point privacy flags
    private Long dataPointId;

    private Long id;
    private FileDTO file;
    private String notes;
    private String date;

    @Override
    public String execute() {

        // parse date; format already checked somewhat in validate xml (check fully in validate() here?)
        Instant parsed = null;
        try {
            parsed = parseToInstantWithFormatter(getDate(), dateFormat);
        } catch (DateTimeParseException ignored) {
            addFieldError("date", getText("editFileAction.err.invalid_date"));
            return input();
        }

        // retrieve File and set notes

        var maybeFileDTO = fileManager.getFile(getLoggedInEHRRequestContext(), getId(), false);
        if (maybeFileDTO.isEmpty()) {
            addActionError(getText("editFileAction.err.already_deleted"));
            return SUCCESS; // redirect back to myFiles
        }
        FileDTO file = maybeFileDTO.get();

        var users = getContextAndLoggedInUsers();
        if (hasNotPermissionsToEditDatapoint(file, users.loggedInUser(), loggedInUserUtil.getLoggedInUserPrimaryTeam())) {
            return CANCEL;
        }
        file.setDate(Date.from(parsed));
        file.setNotes(notes);

        fileManager.saveFile(getEHRRequestContext(), file, getAccountId(users), FileDtoType.FILE);

        addActionMessage(getText("editFileAction.msg.saved_successfully"));

        return SUCCESS;
    }

    public String cancel() {
        return SUCCESS;
    }

    @Override
    public String input() {
        // retrieve file and set in action
        file = fileManager.getFile(getLoggedInEHRRequestContext(), getId(), false)
                .orElse(null);
        if (file == null) {
            return CANCEL;
        }
        if (hasNotPermissionsToEditDatapoint(file, loggedInUserUtil.getLoggedInUser(), loggedInUserUtil.getLoggedInUserPrimaryTeam())) {
            return CANCEL;
        }
        setDate(dateToString(file.getDate(), dateFormat));
        if (notes == null) {
            setNotes(file.getNotes());
        }
        disablePrivacyFlagEditForm();
        return INPUT;
    }

    public String editPrivacyForm() {
        var users = getContextAndLoggedInUsers();
        Supplier<Optional<FileDTO>> optionalFileDTOSupplier = fetchFile();
        return checkPermissionsAndDo((rc, f) -> fillPrivacyFlagsFrom(f), fetchFile(), INPUT, true, users);
    }

    private static final Map<String, MenuDataType> URL_TO_MENUDATA = ImmutableMap.<String, MenuDataType>builder()
            .put("editGeneticsFilePrivacy", MenuDataType.geneticsFile)
            .put("editFilePrivacy", MenuDataType.file)
            .build();

    @SkipValidation
    public String editPrivacySave() {
        var users = getContextAndLoggedInUsers(CONTACTS);
        MenuDataType type = getType(URL_TO_MENUDATA);
        contextUserId = contextUserUtil.getContextUserId();
        return checkPermissionsAndDo(
                (rc, f) -> privacyService.handlePrivacyChange(f, type, getNewFlag(), rc, getAccountId(users), users.contextUser(), users.loggedInUser()),
                fetchFile(), SUCCESS, false, users);
    }

    @SkipValidation
    public String editPrivacyCancel() {
        return CANCEL;
    }

    private long getAccountId(BaseAction.Users users) {
        return userManager.getDefaultAccountId(users.contextUser().getId());
    }

    Supplier<Optional<FileDTO>> fetchFile() {
        return () ->
                fileManager.getFile(getLoggedInEHRRequestContext(), dataPointId, false);
    }

    public FileManager getFileManager() {
        return fileManager;
    }

    public void setFileManager(FileManager fileManager) {
        this.fileManager = fileManager;
    }

    public String getTab() {
        return tab;
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public FileDTO getFile() {
        return file;
    }

    public void setFile(FileDTO file) {
        this.file = file;
    }

    public String getNotes() {
        return notes;
    }

    public void setNotes(String notes) {
        this.notes = notes;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public String getDate() {
        return date;
    }

    public void setDate(String date) {
        this.date = date;
    }

    public Long getDataPointId() {
        return dataPointId;
    }

    public void setDataPointId(Long dataPointId) {
        this.dataPointId = dataPointId;
    }

    @Override
    public void prepare() throws Exception {
        super.prepare();
        dateFormat = getGlobalDateInputFormatter();
    }

}
