package com.pkb.action.file;

import com.pkb.util.security.FileSecurityHandler;
import io.vavr.control.Either;
import org.apache.struts2.ServletActionContext;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.servlet.http.HttpServletResponse;

public interface FileSanitizerActionMixin {

    Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    default String sanitizeFile(String fileName, @NotNull byte[] data, String originalMediaType) {
        Either<String, String> checkResult = getFileSecurityHandler().runChecks(fileName, data, originalMediaType);
        if (checkResult.isLeft()) {
            // file upload UI displays why upload failed
            HttpServletResponse response = ServletActionContext.getResponse();
            response.setStatus(HttpServletResponse.SC_BAD_REQUEST);
            response.setHeader("Reason", checkResult.getLeft());
            addActionError(checkResult.getLeft());
            throw new RuntimeException(checkResult.getLeft());
        }
        return checkResult.get();
    }

    void addActionError(String threatError);

    @NotNull
    FileSecurityHandler getFileSecurityHandler();
}
