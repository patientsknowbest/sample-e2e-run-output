package com.pkb.api.util.mapper;

import com.google.common.collect.Maps;

import java.util.AbstractMap;
import java.util.Map;
import java.util.Map.Entry;
import java.util.Optional;

import static java.util.Optional.empty;
import static java.util.stream.Collectors.toMap;

/**
 * Base implementation for {@link Enum} to {@link Enum} mappings, where the relation is one to zero/one.
 *
 * @param <T>
 *            The {@link Enum} which has more values than the other.
 * @param <U>
 *            The {@link Enum} which has one to one mapping with the other {@link Enum}.
 */
public abstract class AbstractEnumMapper<T extends Enum<T>, U extends Enum<U>> {

    private final Map<T, Optional<U>> T_TO_U;
    private final Map<U, T> U_TO_T;

    public AbstractEnumMapper(Map<T, Optional<U>> map) {
        T_TO_U = Maps.immutableEnumMap(map);
        U_TO_T = Maps.immutableEnumMap(createInverseMap(map));
    }

    /**
     * Returns possible mapped value.
     *
     * @param input
     *            The {@link Enum} value to map.
     * @return {@link Enum} value if mapping exists, {@link Optional#empty()} otherwise.
     */
    public Optional<U> getPossibleValue(T input) {
        if (input == null) {
            return empty();
        }
        return T_TO_U.get(input);
    }

    /**
     * Returns mapped value.
     * Never returns null.
     *
     * @param input
     *            The {@link Enum} value to map.
     * @return The mapped {@link Enum} value.
     */
    public T getValue(U input) {
        return U_TO_T.get(input);
    }

    protected Map<U, T> createInverseMap(Map<T, Optional<U>> input) {
        return input.entrySet().stream().filter(o -> o.getValue().isPresent())
                .map(e -> new AbstractMap.SimpleEntry<>(e.getValue().get(), e.getKey()))
                .collect(toMap(Entry::getKey, Entry::getValue));
    }
}
