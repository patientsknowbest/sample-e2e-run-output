package com.pkb.api.util.mapper;

import com.pkb.api.v2.model.ApiEncounterStatus;
import com.pkb.encounter.entity.Encounter.EncounterStatus;
import org.springframework.stereotype.Component;

import java.util.EnumMap;
import java.util.Optional;

import static java.util.Optional.empty;

@Component
public class EncounterStatusMapper extends AbstractEnumMapper<ApiEncounterStatus, EncounterStatus> {

    public EncounterStatusMapper() {
        super(getMap());
    }

    private static EnumMap<ApiEncounterStatus, Optional<EncounterStatus>> getMap() {
        EnumMap<ApiEncounterStatus, Optional<EncounterStatus>> temporaryMap = new EnumMap<>(ApiEncounterStatus.class);
        temporaryMap.put(ApiEncounterStatus.ARRIVED, Optional.of(EncounterStatus.ARRIVED));
        temporaryMap.put(ApiEncounterStatus.BLANK, empty());
        temporaryMap.put(ApiEncounterStatus.CANCELLED, Optional.of(EncounterStatus.CANCELLED));
        temporaryMap.put(ApiEncounterStatus.FINISHED, Optional.of(EncounterStatus.FINISHED));
        temporaryMap.put(ApiEncounterStatus.IN_PROGRESS, Optional.of(EncounterStatus.IN_PROGRESS));
        temporaryMap.put(ApiEncounterStatus.ON_LEAVE, Optional.of(EncounterStatus.ON_LEAVE));
        temporaryMap.put(ApiEncounterStatus.PLANNED, Optional.of(EncounterStatus.PLANNED));
        temporaryMap.put(ApiEncounterStatus.UNRECOGNISED, empty());
        return temporaryMap;
    }

}
