package com.pkb.api.util.mapper;

import org.springframework.stereotype.Component;

import java.util.EnumMap;
import java.util.Optional;

import static java.util.Optional.empty;


@Component
public class ApiEncounterToEncounterMapper extends AbstractEnumMapper<ApiEncounterClass, EncounterClass> {

    public ApiEncounterToEncounterMapper() {
        super(getMap());
    }

    private static EnumMap<ApiEncounterClass, Optional<EncounterClass>> getMap() {

        temporaryMap.put(ApiEncounterClass.AMBULATORY, Optional.of(EncounterClass.AMBULATORY));
        temporaryMap.put(ApiEncounterClass.BLANK, empty());

        temporaryMap.put(ApiEncounterClass.EMERGENCY, Optional.of(EncounterClass.EMERGENCY));
        temporaryMap.put(ApiEncounterClass.FIELD, Optional.of(EncounterClass.FIELD));
        temporaryMap.put(ApiEncounterClass.HOME, Optional.of(EncounterClass.HOME));
        temporaryMap.put(ApiEncounterClass.IN_PATIENT, Optional.of(EncounterClass.IN_PATIENT));
        temporaryMap.put(ApiEncounterClass.OUT_PATIENT, Optional.of(EncounterClass.OUT_PATIENT));
        temporaryMap.put(ApiEncounterClass.OTHER, Optional.of(EncounterClass.OTHER));
        temporaryMap.put(ApiEncounterClass.VIRTUAL, Optional.of(EncounterClass.VIRTUAL));
        temporaryMap.put(ApiEncounterClass.UNRECOGNISED, empty());
        return temporaryMap;
    }
}
