package com.pkb.api.util.mapper;

import com.pkb.api.v2.model.ApiParticipant.ApiParticipantRole;
import com.pkb.app.dto.ParticipantDTO.ParticipantRole;
import org.springframework.stereotype.Component;

import java.util.EnumMap;
import java.util.Optional;

@Component
public class ParticipantRoleMapper extends AbstractEnumMapper<ApiParticipantRole, ParticipantRole> {

    public ParticipantRoleMapper() {
        super(getMap());
    }

    private static EnumMap<ApiParticipantRole, Optional<ParticipantRole>> getMap() {
        EnumMap<ApiParticipantRole, Optional<ParticipantRole>> temporaryMap = new EnumMap<>(ApiParticipantRole.class);
        for (ApiParticipantRole current : ApiParticipantRole.values()) {
            temporaryMap.put(current, Optional.ofNullable(current.getParticipantRole()));
        }
        return temporaryMap;
    }
}
