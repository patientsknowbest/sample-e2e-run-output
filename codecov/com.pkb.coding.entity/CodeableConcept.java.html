package com.pkb.coding.entity;

import com.google.common.base.MoreObjects;
import com.pkb.annotation.EHRField;
import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.stream.Collectors;

/**
 * May contain multiple Codings, and a single display text value
 *
 * @author robwhelan
 */
public class CodeableConcept implements Serializable {
    private static final long serialVersionUID = 1L;

    @EHRField
    private List<Coding> codings = new ArrayList<>();

    @EHRField
    private String displayText;

    public CodeableConcept() {
    }

    public CodeableConcept(Builder builder) {
        Collections.sort(builder.codings);
        this.codings = builder.codings;
        this.displayText = builder.displayText;
    }

    public static class Builder {
        private String displayText;
        private List<Coding> codings = new ArrayList<>();

        public Builder(String displayText) {
            this.displayText = displayText;
        }

        public Builder(Coding coding) {
            coding(coding);
        }

        public Builder() {
        }

        public Builder displayText(String displayText) {
            this.displayText = displayText;
            return this;
        }

        public Builder coding(Coding coding) {
            codings.add(coding);
            return this;
        }

        public CodeableConcept build() {
            return new CodeableConcept(this);
        }
    }

    public boolean hasAtLeastOneCode() {
        return codings.stream().map(Coding::getCode).anyMatch(StringUtils::isNotBlank);
    }

    public String getCodesForDisplay() {
        return codings.stream()
                .map(Coding::getCodeForDisplay)
                .filter(StringUtils::isNotBlank)
                .collect(Collectors.joining(", "));
    }

    public String getSimpleCodesForDisplay() {
        return codings.stream()
                .map(Coding::getSimpleCodeForDisplay)
                .filter(StringUtils::isNotBlank)
                .collect(Collectors.joining(", "));
    }

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("displayText", displayText)
                .add("codings", codings)
                .toString();
    }

    @Override
    public int hashCode() {
        int prime = 31;
        int result = 1;
        result = prime * result + ((codings == null) ? 0 : codings.hashCode());
        result = prime * result + ((displayText == null) ? 0 : displayText.toLowerCase().hashCode());
        return result;
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj) {
            return true;
        }
        if (obj == null) {
            return false;
        }
        if (getClass() != obj.getClass()) {
            return false;
        }
        CodeableConcept other = (CodeableConcept) obj;
        if (codings == null) {
            if (other.codings != null) {
                return false;
            }
        } else if (!codings.equals(other.codings)) {
            return false;
        }
        if (displayText == null) {
            return other.displayText == null;
        } else {
            return displayText.toLowerCase().equals(other.displayText.toLowerCase());
        }
    }

    public List<Coding> getCodings() {
        // avoid null list
        if (codings == null) {
            codings = new ArrayList<>();
        }
        return codings;
    }

    public void setCodings(List<Coding> codings) {
        // avoid null list
        if(codings == null) {
            this.codings = new ArrayList<>();
        } else {
            this.codings = new ArrayList<>(codings);
            Collections.sort(this.codings);
        }
    }

    /**
     * 2017-08-07: Temporary functionality. Until the proper UI changes have been made to supported coded data, we need to ensure that if
     * we have show something on the screen if we don't have a top level displayText, but we do have a display text in one of the codings.
     */
    public String getTextForDisplay() {
        if (StringUtils.isNotBlank(displayText)) {
            return displayText;
        } else {
            for (Coding coding : codings) {
                if (StringUtils.isNotBlank(coding.getDisplayText())) {
                    return coding.getDisplayText();
                }
            }
        }
        return "";
    }

    public String getDisplayText() {
        return displayText;
    }

    public void setDisplayText(String displayText) {
        this.displayText = displayText;
    }

    public Option<String> getSnomed() {
        return io.vavr.collection.List.ofAll(getCodings())
                .find(coding -> Coding.CODE_SYSTEM_SNOMED_CT.equals(coding.getCodeSystem()))
                .map(Coding::getCode);
    }
}
