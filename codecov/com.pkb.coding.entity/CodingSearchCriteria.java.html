package com.pkb.coding.entity;

import java.io.Serializable;

public class CodingSearchCriteria implements Serializable {

    private final String code;

    private final String codeSystem;

    private final boolean preferred;

    public CodingSearchCriteria(String code, String codeSystem, boolean preferred) {
        this.code = code;
        this.codeSystem = codeSystem;
        this.preferred = preferred;
    }

    public String getCode() {
        return code;
    }

    public String getCodeSystem() {
        return codeSystem;
    }

    public boolean isPreferred() {
        return preferred;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (!(o instanceof CodingSearchCriteria)) {
            return false;
        }

        CodingSearchCriteria that = (CodingSearchCriteria) o;

        if (code != null ? !code.equals(that.code) : that.code != null) {
            return false;
        }
        return codeSystem != null ? codeSystem.equals(that.codeSystem) : that.codeSystem == null;
    }

    @Override
    public int hashCode() {
        int result = code != null ? code.hashCode() : 0;
        result = 31 * result + (codeSystem != null ? codeSystem.hashCode() : 0);
        return result;
    }
}
