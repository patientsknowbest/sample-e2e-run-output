@Entity
@Table
public class CodingReceived implements Serializable {
    private static final long serialVersionUID = 1L;

    @Id
    @GenericGenerator(name = "seq_codingrecd_gen", strategy = "enhanced-sequence", parameters = {
            @org.hibernate.annotations.Parameter(name = "optimizer", value = OPTIMIZER_CANONICAL_NAME), // based on "pooled-lo"
            @org.hibernate.annotations.Parameter(name = "sequence_name", value = "seq_codingrecd"),
            @org.hibernate.annotations.Parameter(name = "increment_size", value = "50")
    })
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_codingrecd_gen")
    private Long id;

    @Column(name = "coding_data_type", nullable = false)
    @Enumerated(EnumType.STRING)
    private CodingDataType codingDataType;

    @Column(name = "org_id")
    private Long orgId;

    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    private Route route;

    @Column(nullable = false)
    private String code1;

    @Column(name = "code_system1")
    private String codeSystem1;

    @Column
    private String code2;

    @Column(name = "code_system2")
    private String codeSystem2;

    @Column(name = "orig_display_text")
    private String origDisplayText;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "codingmatch_id", referencedColumnName = "id")
    private CodingMatch codingMatch;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "codingmatch_master_id", referencedColumnName = "id")
    private CodingMatch codingMatchMaster;

    @Column(name = "created", insertable = false, updatable = false, nullable = false)
    private Instant created;

    public CodingReceived() {
    }

    public static Optional<CodingReceived> attemptCodingReceived(@NotNull CodingDataType codingDataType,
            @NotNull CodeableConcept codeableConcept,
            @NotNull EHRRequestContext requestContext) {
        Optional<CodingReceived> maybeCodingReceived = Optional.empty();

        int codingsSize = codeableConcept.getCodings().size();
        if ((codingsSize == 1 || codingsSize == 2) && codeableConcept.hasAtLeastOneCode()) {
            CodingReceived cr = new CodingReceived();
            cr.setCodingDataType(codingDataType);
            // TODO: What is the purpose of origDisplayText? If only for debugging, we should use getTextForDisplay instead.
            cr.setOrigDisplayText(codeableConcept.getDisplayText());

            Coding coding_1 = codeableConcept.getCodings().get(0);
            cr.setCode1(coding_1.getCode());
            cr.setCodeSystem1(coding_1.getCodeSystem());
            if (codingsSize > 1) {
                Coding coding_2 = codeableConcept.getCodings().get(1);
                if (StringUtils.isBlank(cr.getCode1())) {
                    cr.setCode1(coding_2.getCode());
                    cr.setCodeSystem1(coding_2.getCodeSystem());
                } else if (StringUtils.isNotBlank(coding_2.getCode())) {
                    cr.setCode2(coding_2.getCode());
                    cr.setCodeSystem2(coding_2.getCodeSystem());
                }
            }

            requestContext.getOrgId().ifPresent(cr::setOrgId);
            cr.setRoute(requestContext.getAccessRoute());

            maybeCodingReceived = Optional.of(cr);
        }

        return maybeCodingReceived;
    }

    public void applyCodingMatch(CodingMatch codingMatch) {
        setCodingMatch(codingMatch);

        CodingMatch firstAncestor = codingMatch;
        while (firstAncestor.getParent() != null) {
            firstAncestor = firstAncestor.getParent();
        }
        setCodingMatchMaster(firstAncestor);
    }

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("id", getId())
                .add("orgId", orgId)
                .add("route", route)
                .add("code1", code1)
                .add("codeSystem1", codeSystem1)
                .add("code2", code2)
                .add("codeSystem2", codeSystem2)
                .toString();
    }

    @Override
    public boolean equals(Object o) {

        if (this == o) {
            return true;
        }
        if (!(o instanceof CodingReceived)) {
            return false;
        }

        CodingReceived that = (CodingReceived) o;

        if (codingDataType != that.codingDataType) {
            return false;
        }
        if (orgId != null ? !orgId.equals(that.orgId) : that.orgId != null) {
            return false;
        }
        if (route != that.route) {
            return false;
        }
        if (!code1.equals(that.code1)) {
            return false;
        }
        if (codeSystem1 != null ? !codeSystem1.equals(that.codeSystem1) : that.codeSystem1 != null) {
            return false;
        }
        if (code2 != null ? !code2.equals(that.code2) : that.code2 != null) {
            return false;
        }
        return codeSystem2 != null ? codeSystem2.equals(that.codeSystem2) : that.codeSystem2 == null;
    }

    @Override
    public int hashCode() {
        int result = codingDataType != null ? codingDataType.hashCode() : 0;
        result = 31 * result + (orgId != null ? orgId.hashCode() : 0);
        result = 31 * result + route.hashCode();
        result = 31 * result + code1.hashCode();
        result = 31 * result + (codeSystem1 != null ? codeSystem1.hashCode() : 0);
        result = 31 * result + (code2 != null ? code2.hashCode() : 0);
        result = 31 * result + (codeSystem2 != null ? codeSystem2.hashCode() : 0);
        return result;
    }
