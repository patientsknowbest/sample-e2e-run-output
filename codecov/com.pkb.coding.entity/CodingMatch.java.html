package com.pkb.coding.entity;

import com.google.common.base.MoreObjects;
import com.pkb.VersionDetails;
import com.pkb.entities.enums.CodingDataType;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import org.apache.commons.lang3.StringUtils;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.io.Serializable;

/**
 * Matching rules for codings we receive; human-curated to store metadata here, like which
 * privacy flag to use for received data points and preferred text at display time
 *
 * @author robwhelan
 */
@Entity
@Table
public class CodingMatch implements Serializable {
    private static final long serialVersionUID = 1L;

    @Id
    @GenericGenerator(name = "seq_codingmatch_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "public"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_codingmatch"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_codingmatch_gen")
    private Long id;

    @Column(name = "coding_data_type", nullable = false)
    @Enumerated(EnumType.STRING)
    private CodingDataType codingDataType;

    @Column(nullable = false)
    private String code;

    @Column(name = "code_system")
    private String codeSystem;

    @Column(name = "display_text")
    private String displayText;

    @Column(nullable = false)
    private boolean exclude = false;

    @Column(name = "privacy_flag")
    @Enumerated(EnumType.STRING)
    private PrivacyFlag privacyFlag;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "parent_id", referencedColumnName = "id")
    private CodingMatch parent;

    @Embedded
    private VersionDetails versionDetails;

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected CodingMatch() {
        this.versionDetails = new VersionDetails();
    }

    public CodingMatch(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }

    public boolean isCodingSame(String code, String codeSystem) {
        return StringUtils.trimToEmpty(code).equals(StringUtils.trimToEmpty(this.code))
                && StringUtils.trimToEmpty(codeSystem).equals(StringUtils.trimToEmpty(this.codeSystem));
    }

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
            .add("id", getId())
            .add("code", code)
            .add("codeSystem", codeSystem)
            .add("exclude", exclude)
            .add("privacyFlag", privacyFlag)
            .toString();
    }

    @Override public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (!(o instanceof CodingMatch)) {
            return false;
        }

        CodingMatch that = (CodingMatch) o;

        if (codingDataType != that.codingDataType) {
            return false;
        }
        if (code != null ? !code.equals(that.code) : that.code != null) {
            return false;
        }
        if (codeSystem != null ? !codeSystem.equals(that.codeSystem) : that.codeSystem != null) {
            return false;
        }
        if (displayText != null ? !displayText.equals(that.displayText) : that.displayText != null) {
            return false;
        }
        if (exclude != that.exclude) {
            return false;
        }
        if (privacyFlag != that.privacyFlag) {
            return false;
        }
        if (parent != null ? !parent.equals(that.parent) : that.parent != null) {
            return false;
        }
        return true;
    }

    @Override public int hashCode() {
        int result = codingDataType != null ? codingDataType.hashCode() : 0;
        result = 31 * result + (code != null ? code.hashCode() : 0);
        result = 31 * result + (codeSystem != null ? codeSystem.hashCode() : 0);
        result = 31 * result + (displayText != null ? displayText.hashCode() : 0);
        result = 31 * result + (exclude ? 1 : 0);
        result = 31 * result + (privacyFlag != null ? privacyFlag.hashCode() : 0);
        result = 31 * result + (parent != null ? parent.hashCode() : 0);
        return result;
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public CodingDataType getCodingDataType() {
        return codingDataType;
    }

    public void setCodingDataType(CodingDataType codingDataType) {
        this.codingDataType = codingDataType;
    }

    public String getCode() {
        return code;
    }

    public void setCode(String code) {
        this.code = code;
    }

    public String getCodeSystem() {
        return codeSystem;
    }

    public void setCodeSystem(String codeSystem) {
        this.codeSystem = codeSystem;
    }

    public String getDisplayText() {
        return displayText;
    }

    public void setDisplayText(String displayText) {
        this.displayText = displayText;
    }

    public boolean isExclude() {
        return exclude;
    }

    public void setExclude(boolean exclude) {
        this.exclude = exclude;
    }

    public PrivacyFlag getPrivacyFlag() {
        return privacyFlag;
    }

    public void setPrivacyFlag(PrivacyFlag privacyFlag) {
        this.privacyFlag = privacyFlag;
    }

    public CodingMatch getParent() {
        return parent;
    }

    public void setParent(CodingMatch parent) {
        this.parent = parent;
    }

    public VersionDetails getVersionDetails() {
        return versionDetails;
    }

    public void setVersionDetails(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }
}
