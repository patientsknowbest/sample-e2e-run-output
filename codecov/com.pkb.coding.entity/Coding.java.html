package com.pkb.coding.entity;

import com.google.common.base.MoreObjects;
import com.pkb.annotation.EHRField;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;

import java.io.Serializable;
import java.util.Comparator;
import java.util.Objects;

/**
 * A single code with optional codeSystem and optional display text
 */
public class Coding implements Serializable, Comparable<Coding> {
    private static final long serialVersionUID = 1L;

    public static final String CODE_SYSTEM_SNOMED_CT = "SNOMED CT";

    @EHRField
    private String code;
    @EHRField
    private String codeSystem;
    @EHRField
    private String displayText;

    public Coding() {
    }

    public Coding(Builder builder) {
        this.code = builder.code;
        this.codeSystem = builder.codeSystem;
        this.displayText = builder.displayText;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (!(o instanceof Coding)) {
            return false;
        }
        Coding coding = (Coding) o;
        if (code != null && coding.code != null) {
            return StringUtils.equals(code, coding.code) && StringUtils.equals(codeSystem, coding.codeSystem);
        } else {
            return StringUtils.equals(displayText, coding.displayText);
        }
    }

    @Override
    public int hashCode() {
        return Objects.hash(code, codeSystem);
    }

    @Override
    public int compareTo(@NotNull Coding o) {
        return Comparator.comparing(Coding::getCodeSystem, Comparator.nullsFirst(String::compareTo))
                .thenComparing(Coding::getCode, Comparator.nullsFirst(String::compareTo))
                .compare(o, this);
    }

    /**
     * @return "code-system: code (display-text)"
     *         leave out empty/null fields
     *         ignore coding system if there's no code (not meaningful)
     *         skip parens around display text if there's no code
     */
    public String getCodeForDisplay() {
        String simpleCode = getSimpleCodeForDisplay();
        if (StringUtils.isNotBlank(simpleCode)) {
            if (StringUtils.isNotBlank(displayText)) {
                return simpleCode + " (" + displayText + ")";
            } else {
                return simpleCode;
            }
        } else {
            return StringUtils.trimToEmpty(displayText);
        }
    }

    /**
     * @return "code-system: code"
     *         leave out empty/null fields
     *         ignore coding system if there's no code (not meaningful)
     */
    public String getSimpleCodeForDisplay() {
        if (StringUtils.isBlank(code)) {
            return "";
        }

        if (StringUtils.isNotBlank(codeSystem)) {
            return codeSystem + ": " + code;
        } else {
            return StringUtils.trimToEmpty(code);
        }
    }

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("code", code)
                .add("codeSystem", codeSystem)
                .toString();
    }

    public String getCode() {
        return code;
    }

    public void setCode(String code) {
        this.code = code;
    }

    public String getCodeSystem() {
        return codeSystem;
    }

    public void setCodeSystem(String codeSystem) {
        this.codeSystem = codeSystem;
    }

    public String getDisplayText() {
        return displayText;
    }

    public void setDisplayText(String displayText) {
        this.displayText = displayText;
    }

    public static class Builder {
        private String code;
        private String codeSystem;
        private String displayText;

        public Builder(String code, String codeSystem) {
            this.code = code;
            this.codeSystem = codeSystem;
        }

        public Builder() {
        }

        public Builder code(String code) {
            this.code = code;
            return this;
        }

        public Builder codeSystem(String codeSystem) {
            this.codeSystem = codeSystem;
            return this;
        }

        public Builder displayText(String displayText) {
            this.displayText = displayText;
            return this;
        }

        public Coding build() {
            return new Coding(this);
        }
    }

}
