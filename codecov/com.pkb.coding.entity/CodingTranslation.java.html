package com.pkb.coding.entity;

import com.pkb.VersionDetails;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;
import java.util.Objects;
import java.util.UUID;

@Entity
@Table(schema = "public", name = "coding_translation")
public class CodingTranslation implements Serializable {

    @Id
    @Column(name = "id")
    @Type(type = "pg-uuid")
    private UUID id;

    @Column(name = "source_code")
    private String sourceCode;

    @Column(name = "source_coding_system")
    private String sourceCodingSystem;

    @Column(name = "target_code")
    private String targetCode;

    @Column(name = "target_coding_system")
    private String targetCodingSystem;

    @Embedded
    private VersionDetails versionDetails;

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected CodingTranslation() {
    }

    public CodingTranslation(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (o == null || getClass() != o.getClass()) {
            return false;
        }
        CodingTranslation that = (CodingTranslation) o;
        return sourceCode.equals(that.sourceCode) && sourceCodingSystem.equals(that.sourceCodingSystem) && targetCode.equals(that.targetCode) && targetCodingSystem.equals(that.targetCodingSystem);
    }

    @Override
    public int hashCode() {
        return Objects.hash(sourceCode, sourceCodingSystem, targetCode, targetCodingSystem);
    }

    @Override
    public String toString() {
        return "CodingTranslation{" +
                "id=" + id +
                ", sourceCode='" + sourceCode + '\'' +
                ", sourceCodingSystem='" + sourceCodingSystem + '\'' +
                ", targetCode='" + targetCode + '\'' +
                ", targetCodingSystem='" + targetCodingSystem + '\'' +
                ", versionDetails=" + versionDetails +
                '}';
    }

    public UUID getId() {
        return id;
    }

    public void setId(UUID id) {
        this.id = id;
    }

    public String getSourceCode() {
        return sourceCode;
    }

    public void setSourceCode(String sourceCode) {
        this.sourceCode = sourceCode;
    }

    public String getSourceCodingSystem() {
        return sourceCodingSystem;
    }

    public void setSourceCodingSystem(String sourceCodingSystem) {
        this.sourceCodingSystem = sourceCodingSystem;
    }

    public String getTargetCode() {
        return targetCode;
    }

    public void setTargetCode(String targetCode) {
        this.targetCode = targetCode;
    }

    public String getTargetCodingSystem() {
        return targetCodingSystem;
    }

    public void setTargetCodingSystem(String targetCodingSystem) {
        this.targetCodingSystem = targetCodingSystem;
    }

    public VersionDetails getVersionDetails() {
        return versionDetails;
    }

    public void setVersionDetails(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }
}
