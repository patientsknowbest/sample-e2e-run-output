package com.pkb.service.dataupload.processor;

import java.util.List;
import java.util.stream.Collectors;

import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.service.dataupload.processor.domain.UploadedDataPersistentState;
import com.pkb.service.dataupload.processor.domain.UploadedDataProcessingResult;
import com.pkb.service.dataupload.processor.parser.UploadedDataParser;
import com.pkb.service.dataupload.processor.persister.UploadedDataPersister;

public class DefaultUploadedDataProcessor<T, C extends UploadedDataProcessingContext> implements UploadedDataProcessor<T, C> {
    private final List<UploadedDataParser<T, C>> parserChain;
    private final UploadedDataPersister<T, C> persister;

    public DefaultUploadedDataProcessor(List<UploadedDataParser<T, C>> parserChain, UploadedDataPersister<T, C> persister) {
        this.parserChain = parserChain;
        this.persister = persister;
    }

    @Override
    public UploadedDataProcessingResult<T> processUploadedData(C context, UploadedDataDTO dto) {
        List<UploadedDataProcessingResult<T>> results = parserChain.stream().map(parser -> doProcessUploadedData(parser, context, dto)).collect(Collectors.toList());
        return results.get(results.size() - 1);
    }

    private UploadedDataProcessingResult<T> doProcessUploadedData(UploadedDataParser<T, C> parser, C context, UploadedDataDTO dto) {
        List<T> parsedUploadedData = parser.parse(context, dto);
        UploadedDataPersistentState persistentState = persister.persist(context, dto, parsedUploadedData);
        return new UploadedDataProcessingResult<>(persistentState, parsedUploadedData);
    }
}
