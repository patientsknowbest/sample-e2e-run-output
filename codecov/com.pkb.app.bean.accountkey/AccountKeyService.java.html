package com.pkb.app.bean.accountkey;

import com.google.common.collect.Table;
import com.pkb.app.dto.AccountSymmetricKeyDto;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.crypto.AccountKeys;
import com.pkb.crypto.AccountPrivateKey;
import org.jetbrains.annotations.NotNull;

import java.util.Map;
import java.util.Optional;
import java.util.Set;
import java.util.UUID;

public interface AccountKeyService {

    default @NotNull AccountKeys getAccountKeys(@NotNull LoggedInEHRRequestContext context, long accountId, UUID accountPublicId) {
        return tryGetAccountKeys(context, accountId, accountPublicId)
                .orElseThrow(() -> new IllegalStateException("failed getting account keys; account " + accountId + "; currentUserId " + context.getAccessingUserId()));
    }

    @NotNull Optional<AccountKeys> tryGetAccountKeys(@NotNull LoggedInEHRRequestContext context, long accountId, UUID accountPublicId);

    @NotNull
    default AccountPrivateKey getAccountPrivateKey(@NotNull LoggedInEHRRequestContext context, long accountId, UUID accountPublicId) {
        return getAccountKeys(context, accountId, accountPublicId).getPrivateKey();
    }


    /**
     * @return Account symmetric key paired with its ID.
     */
    @NotNull
    AccountSymmetricKeyDto getAccountSymmetricKeyForEncryption(@NotNull EHRRequestContext context, long accountId);

    /**
     * Request symmetric keys for multiple accounts.
     *
     * @return Account symmetric keys and IDs mapped by account IDs.
     */
    @NotNull
    Map<Long, AccountSymmetricKeyDto> getAccountSymmetricKeysForEncryption(@NotNull EHRRequestContext context, Set<Long> accountIds);

    @NotNull
    Table<EHRRequestContext, Long, AccountSymmetricKeyDto> getAccountSymmetricKeysForEncryption(Map<EHRRequestContext, Set<Long>> accountIdsByRequestContext);

    void notifyAccountObsolete(@NotNull EHRRequestContext context, long accountId);
}
