package com.pkb.app.bean.accountkey;

import com.google.common.collect.Table;
import com.pkb.app.bean.strategies.accountkey.AccountKeyLoadingStrategy;
import com.pkb.app.dto.AccountSymmetricKeyDto;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.crypto.AccountKeys;
import org.jetbrains.annotations.NotNull;

import javax.inject.Inject;
import java.util.Map;
import java.util.Optional;
import java.util.Set;
import java.util.UUID;

public class DefaultAccountKeyService implements AccountKeyService {

    @Inject
    private AccountKeyLoadingStrategy accountKeyLoadingStrategy;

    @Override
    public @NotNull Optional<AccountKeys> tryGetAccountKeys(@NotNull LoggedInEHRRequestContext context, long accountId, UUID accountPublicId) {
        return accountKeyLoadingStrategy.getAccountKeys(context, accountId, accountPublicId);
    }

    @Override
    @NotNull
    public AccountSymmetricKeyDto getAccountSymmetricKeyForEncryption(@NotNull EHRRequestContext context, long accountId) {
        return accountKeyLoadingStrategy.getOrCreateSymmetricKey(context, accountId);
    }

    @Override
    @NotNull
    public Map<Long, AccountSymmetricKeyDto> getAccountSymmetricKeysForEncryption(@NotNull EHRRequestContext context, Set<Long> accountIds) {
        return accountKeyLoadingStrategy.getOrCreateSymmetricKeys(context, accountIds);
    }

    @Override
    @NotNull
    public Table<EHRRequestContext, Long, AccountSymmetricKeyDto> getAccountSymmetricKeysForEncryption(Map<EHRRequestContext, Set<Long>> accountIdsByRequestContext) {
        return accountKeyLoadingStrategy.getOrCreateSymmetricKeys(accountIdsByRequestContext);
    }

    @Override
    public void notifyAccountObsolete(@NotNull EHRRequestContext context, long accountId) {
        accountKeyLoadingStrategy.reloadKeysForAccount(context, accountId);
    }
}
