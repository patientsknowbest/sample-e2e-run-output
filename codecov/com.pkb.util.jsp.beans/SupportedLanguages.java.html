package com.pkb.util.jsp.beans;

import com.opensymphony.xwork2.LocalizedTextProvider;
import com.opensymphony.xwork2.inject.Inject;
import com.pkb.util.SupportedLanguagesEnum;
import com.pkb.util.struts.ExtendedLocaleProviderFactory;
import org.apache.commons.lang3.LocaleUtils;

import java.util.Arrays;
import java.util.List;
import java.util.Locale;
import java.util.stream.Collectors;

/**
 * Defines a list of the supported languages by the web application. Adding a new language keep in mind that not every locale is supported by the underlying JVM. If the newly added
 * language doesn't work, extend list of custom languages defined in struts.xml (constant named {@link ExtendedLocaleProviderFactory#LOCALES}).
 */
public class SupportedLanguages {

    private LocalizedTextProvider textProvider;

    private List<Language> languages;

    @Inject
    public SupportedLanguages(LocalizedTextProvider textProvider) {
        this.textProvider = textProvider;

        languages = Arrays.stream(SupportedLanguagesEnum.values()).map(l -> createLanguage(l.getLocale(), l.getKey()))
                .collect(Collectors.toList());
    }

    public List<Language> getLanguages() {
        return languages;
    }

    private Language createLanguage(String locale, String key) {
        return new Language(locale, key, textProvider);
    }

    public static class Language {
        private Locale locale;
        private String key;
        private String localizedName;

        public Language(String locale, String key, LocalizedTextProvider textProvider) {
            super();
            this.locale = LocaleUtils.toLocale(locale);
            this.key = key;
            this.localizedName = textProvider.findDefaultText(key, this.locale);
        }

        public Locale getLocale() {
            return locale;
        }

        public String getKey() {
            return key;
        }

        public String getLocalizedName() {
            return localizedName;
        }
    }

}
