package com.pkb.emis.appointment.config;

import com.google.common.cache.Cache;
import com.google.common.cache.CacheBuilder;
import com.pkb.common.config.PhrConfig;
import com.pkb.domain.appointment.EmisAppointmentService;
import com.pkb.emis.appointment.model.EmisCachedSessionData;
import com.pkb.emis.appointment.service.EmisApiClient;
import com.pkb.emis.appointment.service.EmisAppointmentManager;
import com.pkb.emis.appointment.service.EmisServiceContext;
import com.pkb.emis.appointment.service.EmisSessionService;
import com.pkb.emis.appointment.service.SessionManagingEmisApiClient;
import com.pkb.integrations.emis.model.EmisJacksonFixes;
import org.apache.http.HttpHost;
import org.apache.http.client.config.RequestConfig;
import org.apache.http.impl.client.CloseableHttpClient;
import org.apache.http.impl.client.HttpClientBuilder;
import org.apache.http.ssl.SSLContextBuilder;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.http.client.HttpComponentsClientHttpRequestFactory;
import org.springframework.http.converter.json.MappingJackson2HttpMessageConverter;
import org.springframework.web.client.RestTemplate;

import javax.inject.Inject;
import java.io.File;

import static java.util.concurrent.TimeUnit.MINUTES;
import static org.apache.commons.lang3.StringUtils.isNotEmpty;

@Configuration
public class EmisAppointmentConfig {

    public static final long SESSION_DATA_CACHE_EXPIRATION_MINUTES = 20L;

    @Inject
    private PhrConfig config;

    @Bean
    public EmisAppointmentManager emisAppointmentManager(SessionManagingEmisApiClient emisApiClient) {
        return new EmisAppointmentManager(emisApiClient);
    }

    @Bean
    public EmisSessionService emisSessionService(@Qualifier("sessionDataCache") Cache<Long, EmisCachedSessionData> sessionDataCache, EmisAppointmentService appointmentService, EmisApiClient emisApiClient) {
        return new EmisSessionService(sessionDataCache, appointmentService, emisApiClient);
    }

    @Bean
    public EmisApiClient emisApiClient() {
        EmisServiceContext context = EmisServiceContext.newBuilder()
                .emisRestUrl(config.getEmisRestUrl())
                .xapiApplicationId(config.getEmisRestXapiApplicationId())
                .xapiVersion(config.getRestEmisXapiVersion()).build();
        return new EmisApiClient(context, restTemplate());
    }

    @Bean
    public SessionManagingEmisApiClient sessionManagingEmisApiClient(EmisApiClient emisApiClient, EmisSessionService sessionService, EmisAppointmentService appointmentService) {
        return new SessionManagingEmisApiClient(emisApiClient, sessionService, appointmentService);
    }

    private RestTemplate restTemplate() {
        try {
            SSLContextBuilder sslContextBuilder = SSLContextBuilder.create();
            if (isNotEmpty(config.getEmisRestKeystore())) {
                sslContextBuilder.loadKeyMaterial(new File(config.getEmisRestKeystore()).toURI().toURL(), "".toCharArray(), "".toCharArray());
            }

            RequestConfig.Builder requestConfigBuilder = RequestConfig.custom()
                    .setSocketTimeout(config.getEmisRestConnectTimeout())
                    .setConnectTimeout(config.getEmisRestConnectTimeout())
                    .setConnectionRequestTimeout(config.getEmisRestConnectTimeout());
            if (config.isProxyConfigured()) {
                requestConfigBuilder.setProxy(new HttpHost(config.getProxyHost(), config.getProxyPort()));
            }

            CloseableHttpClient client = HttpClientBuilder.create()
                    .setSSLContext(sslContextBuilder.build())
                    .setDefaultRequestConfig(requestConfigBuilder.build()).build();

            HttpComponentsClientHttpRequestFactory httpComponentsClientHttpRequestFactory = new EmisHttpComponentsClientHttpRequestFactory();
            httpComponentsClientHttpRequestFactory.setHttpClient(client);

            RestTemplate restTemplate = new RestTemplate(httpComponentsClientHttpRequestFactory);
            restTemplate.getMessageConverters().stream()
                    .filter(m -> m.getClass().getName().equals(MappingJackson2HttpMessageConverter.class.getName()))
                    .forEach(c -> EmisJacksonFixes.configureObjectMapper(((MappingJackson2HttpMessageConverter) c).getObjectMapper()));
            return restTemplate;
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }

    @Bean
    public Cache<Long, EmisCachedSessionData> sessionDataCache() {
        return CacheBuilder.newBuilder()
                .expireAfterWrite(SESSION_DATA_CACHE_EXPIRATION_MINUTES, MINUTES)
                .recordStats()
                .build();
    }
