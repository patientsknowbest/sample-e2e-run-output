package com.pkb.api.util.converter.v1;

import com.pkb.api.v1.model.ApiJournalEntry;
import com.pkb.diary.entity.DiaryEntry;
import org.springframework.core.convert.converter.Converter;

public class DiaryEntryToApiJournalEntryConverter implements Converter<DiaryEntry, ApiJournalEntry> {

    @Override
    public ApiJournalEntry convert(DiaryEntry input) {
        ApiJournalEntry entry = new ApiJournalEntry();
        entry.setAuthorId(input.getSource().getPersonId().toString());
        entry.setDate(input.getDiaryEntryTimestamp());
        entry.setEntryTextHtml(input.getEntryText());
        entry.setEntryTextPlain(input.getEntryText());
        entry.setId("" + input.getId());
        entry.setSubjectOfEntry(input.getSubjectOfEntry());
        return entry;
    }

}
