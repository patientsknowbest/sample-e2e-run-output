package com.pkb.api.util.converter.v1;

import com.pkb.api.v1.model.ApiContact;
import com.pkb.api.v1.model.ApiNationalId;
import com.pkb.api.v1.model.ApiUser;
import com.pkb.entities.enums.Gender;
import com.pkb.entities.enums.UserType;
import com.pkb.datamodel.Email;
import com.pkb.user.entity.NationalId;
import com.pkb.user.entity.PKBPerson;
import com.pkb.user.entity.PersonContact;
import org.apache.commons.lang3.StringUtils;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.core.convert.converter.Converter;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.stream.Collectors;

@Component
public class PkbPersonToApiUserConverter implements Converter<PKBPerson, ApiUser> {

    @Autowired
    private PersonContactToApiContactConverter personContactToApiContactConverter;

    @Override
    public ApiUser convert(PKBPerson person) {
        ApiUser result = new ApiUser();
        result.setId(person.getId());
        result.setDob(person.getDateOfBirthString());
        result.setEmail(person.findEmail().map(Email::address).getOrNull());
        result.setPhone(person.getPhone());
        result.setFirstName(person.getFirstName());
        result.setLastName(person.getLastName());
        result.setGenderMF(person.getGender() == 0 ? "f" : "m");
        result.setGender(Gender.toLetter(person.getGender()));
        result.setTitle(person.getTitle());

        result.setAddress1(person.getAddress1());
        result.setAddress2(person.getAddress2());
        result.setCity(person.getCity());
        result.setState(person.getState());
        result.setPostalCode(person.getPostalCode());
        result.setCountry(person.getCountry());
        result.setPatient(UserType.PATIENT == person.getUserType());

        result.setJobTitle(person.getJobTitle());
        result.setOrganizationName(person.getOrganizationName());
        result.setTeamName(person.getOrganizationUnit());
        result.setDateDeceased(person.getDeathTimestamp());

        if (result.isPatient()) {
            List<ApiNationalId> nationalIds = person.getNationalIds().stream()
                    .filter(nationalId -> StringUtils.isNotBlank(nationalId.getValue()))
                    .map(this::nationalIdToApiNationalIdConverter)
                    .collect(Collectors.toList());
            result.setNationalIds(nationalIds);
        }

        List<ApiContact> contacts = person.getContacts().stream().filter(PersonContact::isActualContact)
                .map(personContactToApiContactConverter::convert)
                .collect(Collectors.toList());
        result.setContacts(contacts);

        return result;
    }

    private ApiNationalId nationalIdToApiNationalIdConverter(NationalId id) {
        ApiNationalId result = new ApiNationalId();
        result.setName(id.getType().getName());
        result.setValue(id.getValueFormatted());
        result.setCountryCodes(id.getType().getPrimaryCountryCodes());
        return result;
    }
}
