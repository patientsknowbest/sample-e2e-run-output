package com.pkb.api.util.converter.v1;

import com.pkb.api.v1.model.ApiContact;
import com.pkb.entities.enums.ContactType;
import com.pkb.user.entity.PersonContact;
import org.springframework.core.convert.converter.Converter;
import org.springframework.stereotype.Component;

@Component
public class PersonContactToApiContactConverter implements Converter<PersonContact, ApiContact> {

    @Override
    public ApiContact convert(PersonContact contact) {
        ApiContact result = new ApiContact();
        result.setId(contact.getId());
        result.setConfirmed(contact.isConfirmed());
        result.setPrimary(contact.isPrimary());

        contact.getContactIfEmail().peek(email -> {
            result.setContactType(ContactType.EMAIL.toString());
            result.setContactValue(email.address());
        });

        contact.getContactIfPhone().peek(phone -> {
            result.setContactType(ContactType.PHONE.toString());
            result.setContactValue(phone.number());
        });

        contact.findAlsoContactPerson().peek(ignored -> {
            result.setContactType(ContactType.ALSO_CONTACT.toString());
        });

        return result;
    }

}
