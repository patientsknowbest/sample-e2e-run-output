package com.pkb.kms.service

import com.pkb.crypto.AccountPrivateKey
import com.pkb.crypto.UserPrivateKey
import com.pkb.kms.service.cache.OrgAccountKeyCacheService
import com.pkb.kms.shared.representation.KmsError
import com.pkb.kms.shared.representation.OrgRequester
import com.pkb.kms.shared.representation.PersonRequester
import io.vavr.control.Either
import io.vavr.kotlin.left
import io.vavr.kotlin.right
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class AccountPrivateKeyDecryptorService(private val orgAccountKeyCache: OrgAccountKeyCacheService) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    fun decryptAccountPrivateKeyForUser(userPrivateKey: UserPrivateKey,
                                        accountPath: AccountPath): Either<KmsError, AccountPrivateKey> =
        accountPath.orgAccountKey?.let { orgAccountKey: EncryptedOrgAccountKey ->
            orgAccountKeyCache.decryptOrgAccountKey(userPrivateKey, accountPath)
                    .map { it.accountPrivateKey.decryptAccountPrivateKey(accountPath.encryptedAccountPrivateKey) }
                    .peek {
                        log.debug("Requester ${accountPath.requester} decrypted keys for account " +
                                          "${accountPath.accountPublicId} via org(id=${orgAccountKey.orgId})")
                    }
        } ?: right<KmsError, AccountPrivateKey>(
                userPrivateKey
                        .decryptSymmetricKey(accountPath.encryptedSymmetricKey)
                        .decrypt(accountPath.encryptedAccountPrivateKey, null))
                .peek {
                    log.debug("Requester ${accountPath.requester} decrypted keys for account " +
                                      "${accountPath.accountPublicId} via direct access")
                }

    fun decryptAccountPrivateKeyNoUser(accountPath: AccountPath): Either<KmsError, AccountPrivateKey> =
        // MUST be an OrgRequester (which guarantees presence of org account key)
        when (accountPath.requester) {
            is PersonRequester -> left(KmsError.unexpectedRequesterType(accountPath.requester))
            is OrgRequester    -> orgAccountKeyCache.getCachedOrgAccountKey(accountPath.orgAccountKey!!.orgId)
                    .map { oak ->
                        oak.accountPrivateKey.decryptAccountPrivateKey(accountPath.encryptedAccountPrivateKey)
                    }
        }.peek {
            log.debug("Requester ${accountPath.requester} decrypted keys for account " +
                              "${accountPath.accountPublicId} via org(id=${accountPath.orgAccountKey!!.orgId})")
        }
}
