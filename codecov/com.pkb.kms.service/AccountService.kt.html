package com.pkb.kms.service

import com.pkb.crypto.AccountPrivateKey
import com.pkb.crypto.CryptoConstants
import com.pkb.crypto.dto.AccountKeysDTO
import com.pkb.crypto.dto.AccountPrivateKeyDTO
import com.pkb.crypto.javasecurityutil.KeyUtil
import com.pkb.entities.core.CoreAccount
import com.pkb.entities.embeddables.PublicKeyInfo
import com.pkb.kms.service.cache.AccountCacheService
import com.pkb.kms.service.cache.UserAccessibleAccountPrivateKeyCacheService
import com.pkb.kms.service.repository.OrgRepository
import com.pkb.kms.shared.representation.AccountIds
import com.pkb.kms.shared.representation.KmsError
import com.pkb.kms.shared.representation.OrgPublicId
import io.vavr.control.Either
import io.vavr.control.Option
import org.springframework.stereotype.Service
import org.springframework.transaction.annotation.Transactional
import org.springframework.transaction.interceptor.TransactionAspectSupport
import java.security.KeyPair

@Service
class AccountService(
    private val accountCacheService: AccountCacheService,
    private val orgRepository: OrgRepository,
    private val accessService: AccessService,
    private val userAccessibleAccountPrivateKeyCacheService: UserAccessibleAccountPrivateKeyCacheService
) {
    @Transactional
    fun createAccountAndGrantOrgAccess(orgId: OrgPublicId): Either<KmsError, AccountIds> {
        val keyPair = KeyUtil.generateNewKeyPair(CryptoConstants.RSA_SHORT_NAME, CryptoConstants.RSA_KEY_LENGTH)
        return Option.ofOptional(orgRepository.findByPublicId(orgId))
            .toEither(KmsError.orgNotFound(orgId))
            .flatMap { org ->
                accountCacheService.saveAccount(makeAccount(keyPair))
                    .peek { account ->
                        val accountKeysDTO = AccountKeysDTO.accountKeysDTO(account.publicId, AccountPrivateKeyDTO.accountPrivateKeyDTO(AccountPrivateKey.accountPrivateKey(keyPair.private)), emptyList())
                        accessService.createAndSaveAccountOrgRecord(accountKeysDTO, org, account)
                    }
            }
            .map { account -> AccountIds(account.id, account.publicId) }
            .peekLeft{
                try {
                    TransactionAspectSupport.currentTransactionStatus().setRollbackOnly()
                }
                catch(e: Exception) {
                    //Ignore
                }
            }
    }

    @Transactional
    fun createAccount(): Either<KmsError, AccountIds> {
        val keyPair = KeyUtil.generateNewKeyPair(CryptoConstants.RSA_SHORT_NAME, CryptoConstants.RSA_KEY_LENGTH)
        return accountCacheService.saveAccount(makeAccount(keyPair))
            .flatMap{ account  ->
                val accountKeysDTO = AccountKeysDTO.accountKeysDTO(account.publicId, AccountPrivateKeyDTO.accountPrivateKeyDTO(AccountPrivateKey.accountPrivateKey(keyPair.private)), emptyList())
                userAccessibleAccountPrivateKeyCacheService.addToCache(accountKeysDTO)
                    .map{ AccountIds(account.id, account.publicId)}
            }
    }

    private fun makeAccount(keyPair: KeyPair): CoreAccount {
        val account = CoreAccount()

        val publicKeyInfo = PublicKeyInfo()

        publicKeyInfo.x509EncodedKeySpec = keyPair.public.encoded
        publicKeyInfo.algorithm = keyPair.public.algorithm
        account.accountPublicKey = publicKeyInfo
        return account
    }

}
