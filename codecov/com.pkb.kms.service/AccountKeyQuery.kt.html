package com.pkb.kms.service

import com.pkb.crypto.UserPrivateKey
import com.pkb.crypto.dto.AccountKeysDTO
import com.pkb.kms.service.cache.PathCacheService
import com.pkb.kms.service.cache.UserPrivateKeyCacheService
import com.pkb.kms.service.extension.curried
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.KmsError
import com.pkb.kms.shared.representation.OrgRequester
import com.pkb.kms.shared.representation.PersonPublicId
import com.pkb.kms.shared.representation.PersonRequester
import com.pkb.kms.shared.representation.Requester
import io.vavr.control.Either
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles
import javax.transaction.Transactional

typealias UserKeyFn = (requesterPersonId: PersonPublicId) -> Either<KmsError, UserPrivateKey>
typealias KeyFn = (userPrivateKey: UserPrivateKey, accountPath: AccountPath) -> Either<KmsError, AccountKeysDTO>
typealias KeyFnNoUser = (accountPath: AccountPath) -> Either<KmsError, AccountKeysDTO>
typealias PathFn = (requester: Requester, accountPublicId: AccountPublicId) -> Either<KmsError, AccountPath>

@Service
class AccountKeyQueryProvider(private val accountKeyService: AccountKeyService,
                              private val pathCacheService: PathCacheService,
                              private val userPrivateKeyCache: UserPrivateKeyCacheService) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @Transactional
    fun query(accountPublicId: AccountPublicId, requester: Requester, forceVerify: Boolean) =
        forceVerify(forceVerify).execute(accountPublicId, requester)

    @Transactional
    fun query(accountPublicIds: List<AccountPublicId>,
              requester: Requester,
              forceVerify: Boolean): Either<KmsError, List<AccountKeysDTO>> =
        Either.sequenceRight(forceVerify(forceVerify).execute(accountPublicIds, requester))
                .map { it.toJavaList() }
                .peekLeft { log.debug("Batch request failed for requester $requester: $it") }

    private fun forceVerify(forceVerify: Boolean): AccountKeyQuery =
        if (forceVerify)
            AccountKeyQuery(
                    userPrivateKeyCache::getCachedUserPrivateKey,
                    accountKeyService::getAccountKeysInPathNoCache,
                    accountKeyService::getAccountKeysInPathNoUserNoCache,
                    pathCacheService::findPathNoCache)
        else
            AccountKeyQuery(
                    userPrivateKeyCache::getCachedUserPrivateKey,
                    accountKeyService::getAccountKeysInPath,
                    accountKeyService::getAccountKeysInPathNoUser,
                    pathCacheService::findPath)
}


private class AccountKeyQuery(private val userKeyFn: UserKeyFn,
                              private val keyFn: KeyFn,
                              private val keyFnNoUser: KeyFnNoUser,
                              private val pathFn: PathFn) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    fun execute(accountPublicIds: List<AccountPublicId>, requester: Requester) =
        accountPublicIds.map { this.execute(it, requester) }

    fun execute(accountPublicId: AccountPublicId, requester: Requester): Either<KmsError, AccountKeysDTO> =
        when (requester) {
            is PersonRequester -> executeWithPerson(requester, accountPublicId)
            is OrgRequester    -> executeWithOrg(requester, accountPublicId)
        }.peek { log.trace("Requester $requester accessed keys for account $accountPublicId") }

    private fun executeWithOrg(requester: Requester, accountPublicId: AccountPublicId) =
        pathFn(requester, accountPublicId)
                .flatMap { keyFnNoUser(it) }

    private fun executeWithPerson(requester: PersonRequester, accountPublicId: AccountPublicId) =
        keyFn.curried()
                .let { fn -> userKeyFn.invoke(requester.personPublicId).map(fn) }
                .flatMap { fn -> pathFn.invoke(requester, accountPublicId).flatMap(fn) }
}
