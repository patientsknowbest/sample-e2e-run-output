package com.pkb.kms.service

import com.pkb.common.datetime.DateTimeService
import com.pkb.crypto.CryptoConstants
import com.pkb.crypto.SymmetricKey.symmetricKey
import com.pkb.crypto.UserPrivateKey
import com.pkb.kms.service.entity.KmsNhsLoginCredentials
import com.pkb.kms.service.entity.KmsNhsLoginCredentialsChangeHistory
import com.pkb.kms.service.entity.KmsNhsLoginCredentialsCurrent
import com.pkb.kms.service.extension.sha256
import com.pkb.kms.service.repository.NhsLoginCredentialsChangeHistoryRepository
import com.pkb.kms.service.repository.NhsLoginCredentialsCurrentRepository
import com.pkb.kms.service.repository.NhsLoginCredentialsRepository
import com.pkb.kms.shared.representation.KmsError
import com.pkb.kms.shared.representation.PersonPublicId
import io.vavr.control.Either
import io.vavr.control.Option
import io.vavr.kotlin.right
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles
import java.util.UUID
import javax.transaction.Transactional

@Service
class NhsLoginService(private val nhsLoginCredentialsRepository: NhsLoginCredentialsRepository,
                      private val nhsLoginCredentialsCurrentRepository: NhsLoginCredentialsCurrentRepository,
                      private val nhsLoginCredentialsChangeHistoryRepository: NhsLoginCredentialsChangeHistoryRepository,
                      private val dateTimeService: DateTimeService,
                      private val uuidService: UUIDService) {
    /**
     * This has to be outside of userPrivateKeyService, so that calling
     * getCachedUserPrivateKey actually hits the cache;; because the caching is done via
     * interceptors, and these aren't invoked for internal method calls.
     */
    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @Transactional
    fun encryptAndSaveUserPrivateKeyWithNhsLogin(userPrivateKey: UserPrivateKey, personPublicId: UUID,
                                                 nhsLoginSecret: ByteArray): Either<KmsError, UUID>
    {
        val now = dateTimeService.now()

        // Hash the symmetric key so we can tell if it ever changes
        val symmetricKeyHash = nhsLoginSecret.sha256()

        // Save user private key encrypted with the secret
        val encryptedUserPrivateKey = symmetricKey(nhsLoginSecret, CryptoConstants.AES_SHORT_NAME)
                            .encrypt(userPrivateKey.keyBytes)

        return right(saveCredentials(KmsNhsLoginCredentials(null,
                            uuidService.uuid(),
                            personPublicId,
                            encryptedUserPrivateKey.encryptedBytes(),
                            userPrivateKey.algorithm,
                            CryptoConstants.AES_SHORT_NAME,
                            encryptedUserPrivateKey.nonce(),
                            symmetricKeyHash,
                            now)))
    }

    @Transactional
    fun saveCredentials(credentials: KmsNhsLoginCredentials): UUID {

        // Mark any existing credential as deleted
        val now = dateTimeService.now()
        expireCredentials(listOf(credentials.personPublicId))

        val saved = nhsLoginCredentialsRepository.save(credentials)

        // Find or create pointer record and update it to point to the new recovery record
        val current = findOrCreateCurrentPointer(saved)
                .apply {
                    nhsLoginCredentials = saved
                    updated = now
                }

        // Save and return public Id
        nhsLoginCredentialsCurrentRepository.save(current)
        log.info("Created nhs login credentials record for person {}", credentials.personPublicId)

        return saved.publicId
    }

    @Transactional
    fun expireCredentials(personPublicIds: List<PersonPublicId>): Either<KmsError, List<PersonPublicId>> {

        val now = dateTimeService.now()
        personPublicIds.forEach { personPublicId ->
                nhsLoginCredentialsCurrentRepository.findByPersonPublicId(personPublicId)
                        .peek {
                            // We're going to delete this pointer, so write a history entry
                            nhsLoginCredentialsChangeHistoryRepository.save(KmsNhsLoginCredentialsChangeHistory(
                                    personPublicId = personPublicId,
                                    nhsLoginCredentials = it.nhsLoginCredentials,
                                    from = it.updated,
                                    until = now))

                            nhsLoginCredentialsCurrentRepository.delete(it)
                            log.info("Deleted external password recovery record for person {}", personPublicId)
                        }
        }
        return right(personPublicIds)
    }

    @Transactional
    fun findCredentialsByPersonPublicId(personPublicId: UUID): Option<KmsNhsLoginCredentials> =
            nhsLoginCredentialsCurrentRepository.findByPersonPublicId(personPublicId).map{pointer -> pointer.nhsLoginCredentials }


    private fun findOrCreateCurrentPointer(nhsLoginCredentials: KmsNhsLoginCredentials) =
            nhsLoginCredentialsCurrentRepository.findByPersonPublicId(nhsLoginCredentials.personPublicId)
                    .peek {
                        // We're going to modify this pointer, so write a history entry
                        nhsLoginCredentialsChangeHistoryRepository.save(KmsNhsLoginCredentialsChangeHistory(
                                personPublicId = nhsLoginCredentials.personPublicId,
                                nhsLoginCredentials = it.nhsLoginCredentials,
                                from = it.updated,
                                until = nhsLoginCredentials.created))
                    }
                    .getOrElse {
                        // If there's no current pointer record, create one
                        KmsNhsLoginCredentialsCurrent(
                                personPublicId = nhsLoginCredentials.personPublicId,
                                nhsLoginCredentials = nhsLoginCredentials,
                                updated = nhsLoginCredentials.created)
                    }
}
