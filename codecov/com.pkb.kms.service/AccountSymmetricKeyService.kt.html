package com.pkb.kms.service

import com.pkb.crypto.document.EncryptedDocument
import com.pkb.crypto.document.PlaintextDocument
import com.pkb.crypto.dto.AccountSymmetricKeyDTO
import com.pkb.kms.service.cache.AccountSymmetricKeysCacheService
import com.pkb.kms.service.cache.UnauthenticatedAccountSymmetricKeyCacheService
import com.pkb.kms.service.extension.encryptWith
import com.pkb.kms.service.extension.recover
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.KmsError
import io.vavr.control.Either
import io.vavr.kotlin.left
import io.vavr.kotlin.right
import org.springframework.stereotype.Service

@Service
class AccountSymmetricKeyService(
    private val accountKeyService: AccountKeyService,
    private val accountSymmetricKeysCache: AccountSymmetricKeysCacheService,
    private val unauthenticatedAccountSymmetricKeyCache: UnauthenticatedAccountSymmetricKeyCacheService,
    metrics: MetricsHelper) {

    private val documentSizeSummary = metrics.summary(
            "document.size",
            "Sizes of documents sent for encryption")

    /**
     * Attempts to get a symmetric key for the specified account via any means possible
     *
     * First, checks if there are already cached symmetric keys for the account, and if so returns one of them.
     * If that fails, tries to access the account private key, which can then be used to load and decrypt any existing
     * symmetric keys from the database. If that also fails, creates a new one and persists it.
     */
    fun getAnyAccountSymmetricKey(accountPublicId: AccountPublicId): Either<KmsError, AccountSymmetricKeyDTO> =
        unauthenticatedAccountSymmetricKeyCache.getCachedAccountSymmetricKey(accountPublicId)
                .recover {
                    accountSymmetricKeysCache.getCachedAccountSymmetricKeys(accountPublicId)
                            .flatMap { getFirstSymmetricKey(it, accountPublicId)}
                }
                .recover {
                    accountKeyService.getAccountKeys(accountPublicId)
                            .flatMap { getFirstSymmetricKey(it.symmetricKeys, accountPublicId) }
                }
                .recover {
                    createAccountSymmetricKey(accountPublicId)
                }

    fun createAccountSymmetricKey(accountPublicId: AccountPublicId): Either<KmsError, AccountSymmetricKeyDTO> =
        unauthenticatedAccountSymmetricKeyCache.createAccountSymmetricKey(accountPublicId)
                .also {
                    // We just created a new key, so invalidate the keys cache in case some concurrent
                    // request populated it since we started this method
                    accountSymmetricKeysCache.evictKeysForAccount(accountPublicId)
                }

    fun encrypt(accountPublicId: AccountPublicId,
                docs: List<PlaintextDocument>): Either<KmsError, List<EncryptedDocument>> =
        getAnyAccountSymmetricKey(accountPublicId)
                .map { key: AccountSymmetricKeyDTO ->
                    docs.map { doc: PlaintextDocument ->
                        documentSizeSummary.record(doc.plainText().size.toDouble())
                        doc.encryptWith(key)
                    }
                }

    private fun getFirstSymmetricKey(keys: List<AccountSymmetricKeyDTO>,
                                     accountPublicId: AccountPublicId): Either<KmsError, AccountSymmetricKeyDTO> =
        if (keys.isNotEmpty()) {
            right(keys.first())
        } else {
            left(KmsError.symmetricKeyNotFound(accountPublicId))
        }

}
