package com.pkb.kms.service

import com.pkb.crypto.dto.UserPrivateKeyDTO
import com.pkb.kms.service.cache.UserPrivateKeyCacheService
import com.pkb.kms.service.extension.recoverWith
import com.pkb.kms.shared.representation.KmsError
import com.pkb.kms.shared.representation.PersonPublicId
import io.micrometer.core.instrument.MeterRegistry
import io.vavr.control.Either
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class LoginService(private val userPrivateKeyCache: UserPrivateKeyCacheService, 
                   private val meterRegistry: MeterRegistry) {
    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())
    
    fun login(userId: PersonPublicId, credentials: CharArray, forceVerify: Boolean): Either<KmsError, UserPrivateKeyDTO> = 
        userPrivateKeyCache.findUserPrivateKeyWithPassword(userId, credentials, forceVerify)
                .recoverWith {
                    // If login failed, try again but this time force-reload everything from the db
                    userPrivateKeyCache.findUserPrivateKeyWithPassword(userId, credentials, true)
                            .peek { meterRegistry.counter("login.success.after.reload").increment() }
                }
                .peek { 
                    log.debug("Person $userId with ${it.algorithm} private key logged in")
                }
                .map(UserPrivateKeyDTO::userPrivateKeyDTO)

    fun nhsLogin(userId: PersonPublicId, secret: ByteArray): Either<KmsError, UserPrivateKeyDTO> =
            userPrivateKeyCache.findUserPrivateKeyWithNhsLogin(userId, secret)
                    .peek {
                        log.debug("Person $userId with ${it.algorithm} private key logged in via NHS login")
                    }
                    .map(UserPrivateKeyDTO::userPrivateKeyDTO)

}
