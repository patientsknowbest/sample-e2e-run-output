package com.pkb.kms.service

import com.pkb.crypto.AccountPrivateKey
import com.pkb.crypto.AccountPrivateKey.accountPrivateKey
import com.pkb.crypto.UserPrivateKey
import com.pkb.crypto.dto.AccountKeysDTO
import com.pkb.crypto.dto.AccountPrivateKeyDTO
import com.pkb.crypto.dto.AccountPrivateKeyDTO.accountPrivateKeyDTO
import com.pkb.crypto.dto.AccountSymmetricKeyDTO
import com.pkb.kms.service.AccountKeysDTOBuilder.SymmetricKeyHandling.LOAD_ALLOW_CACHE
import com.pkb.kms.service.AccountKeysDTOBuilder.SymmetricKeyHandling.LOAD_NO_CACHE
import com.pkb.kms.service.AccountKeysDTOBuilder.SymmetricKeyHandling.SKIP
import com.pkb.kms.service.cache.AccountSymmetricKeysCacheService
import com.pkb.kms.service.extension.curried
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.KmsError
import io.vavr.control.Either
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class AccountKeysDTOBuilder(private val accountSymmetricKeysCache: AccountSymmetricKeysCacheService,
                            private val decryptorService: AccountPrivateKeyDecryptorService,
                            metrics: MetricsHelper) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    private val symmetricKeySummary = metrics.summary(
            "symmetric.key.count",
            "Number of symmetric keys per account")

    /**
     * Loading symmetric keys for an account is potentially quite expensive - it involves both a db query, which adds up when
     * loading large batches of account keys - and also a load of RSA decryptions. Many code paths - like org access grant - only
     * require the account private key, so chosing to skip loading symmetric keys can save significant time for a large batch.
     */
    fun makeAccountKeys(accountPublicId: AccountPublicId, accountPrivateKeyDTO: AccountPrivateKeyDTO, skipSymmetricKeys: Boolean = false) =
        doMakeAccountKeys(accountPublicId, if(skipSymmetricKeys) SKIP else LOAD_ALLOW_CACHE, accountPrivateKey(accountPrivateKeyDTO))

    fun makeAccountKeysFromPathNoCache(userPrivateKey: UserPrivateKey, accountPath: AccountPath) =
        makeAccountKeysFromPath(accountPath, LOAD_NO_CACHE) {
            decryptorService.decryptAccountPrivateKeyForUser(userPrivateKey, it)
        }

    fun makeAccountKeysFromPathNoUserNoCache(accountPath: AccountPath) =
        makeAccountKeysFromPath(accountPath, LOAD_NO_CACHE) {
            decryptorService.decryptAccountPrivateKeyNoUser(it)
        }

    @Suppress("ComplexRedundantLet")
    private fun makeAccountKeysFromPath(accountPath: AccountPath,
                                        symmetricKeyHandling: SymmetricKeyHandling,
                                        decryptFun: DecryptFun): Either<KmsError, AccountKeysDTO> =
        this::doMakeAccountKeys.curried()
                .let { fn -> fn(accountPath.accountPublicId) }
                .let { fn -> fn(symmetricKeyHandling) }
                .let { fn -> decryptFun(accountPath).map(fn) }

    private fun doMakeAccountKeys(accountPublicId: AccountPublicId,
                                  symmetricKeyHandling: SymmetricKeyHandling,
                                  accountPrivateKey: AccountPrivateKey): AccountKeysDTO {

        val symmetricKeyFn: (accountPublicId:AccountPublicId, accountPrivateKey: AccountPrivateKey) -> Either<KmsError, List<AccountSymmetricKeyDTO>> = when (symmetricKeyHandling) {
            LOAD_NO_CACHE -> accountSymmetricKeysCache::getAccountSymmetricKeysNoCache
            LOAD_ALLOW_CACHE -> accountSymmetricKeysCache::getAccountSymmetricKeys
            SKIP -> { _, _ -> Either.right(emptyList()) }
        }

        val symmetricKeys = symmetricKeyFn(accountPublicId, accountPrivateKey).getOrElseGet { listOf() }
        symmetricKeySummary.record(symmetricKeys.size.toDouble())

        return AccountKeysDTO.accountKeysDTO(
                accountPublicId,
                accountPrivateKeyDTO(accountPrivateKey.keyBytes, accountPrivateKey.algorithm),
                symmetricKeys)
    }

    private enum class SymmetricKeyHandling {
        /**
         * Force loading from the DB to ensure that all symmetric keys are definitely present
         */
        LOAD_NO_CACHE,

        /**
         * Return symmetric keys from cache if present, which may not be a complete list if more have been added after they were cached
         */
        LOAD_ALLOW_CACHE,

        /**
         * Skip loading symmetric keys altogether to save time
         */
        SKIP
    }
}
