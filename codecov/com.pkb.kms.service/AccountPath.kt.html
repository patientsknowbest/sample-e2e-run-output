package com.pkb.kms.service

import com.pkb.crypto.EncryptedAccountPrivateKey
import com.pkb.crypto.EncryptedSymmetricKey
import com.pkb.entities.core.CoreAccountOrg
import com.pkb.entities.core.UserAccountKeyLink
import com.pkb.kms.service.extension.encryptedAccountPrivateKey
import com.pkb.kms.service.extension.encryptedSymmetricKey
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.OrgRequester
import com.pkb.kms.shared.representation.PersonRequester
import com.pkb.kms.shared.representation.Requester
import java.io.Serializable

class AccountPath constructor(val requester: Requester,
                              val accountPublicId: AccountPublicId,
                              val encryptedAccountPrivateKey: EncryptedAccountPrivateKey,
                              val encryptedSymmetricKey: EncryptedSymmetricKey,
                              val orgAccountKey: EncryptedOrgAccountKey?) : Serializable {

    companion object {
        fun accountPath(requester: PersonRequester,
                        accountPublicId: AccountPublicId,
                        userAccountKeyLink: UserAccountKeyLink) =
            AccountPath(
                    requester = requester,
                    accountPublicId = accountPublicId,
                    encryptedAccountPrivateKey = userAccountKeyLink.accountPrivateKey.encryptedAccountPrivateKey(),
                    encryptedSymmetricKey = userAccountKeyLink.accountPrivateKey.encryptedSymmetricKey(),
                    orgAccountKey = userAccountKeyLink.orgAccountPrivateKey?.let {
                        val orgId = checkNotNull(userAccountKeyLink.orgId) {
                            "Org id must be set if org account key is present in $userAccountKeyLink"
                        }
                        EncryptedOrgAccountKey(orgId = orgId,
                                               encryptedAccountPrivateKey = it.encryptedAccountPrivateKey())
                    })

        fun accountPath(requester: OrgRequester, accountPublicId: AccountPublicId, accountOrg: CoreAccountOrg) =
            AccountPath(
                    requester = requester,
                    accountPublicId = accountPublicId,
                    encryptedAccountPrivateKey = accountOrg.accountPrivateKey.encryptedAccountPrivateKey(),
                    encryptedSymmetricKey = accountOrg.accountPrivateKey.encryptedSymmetricKey(),
                    orgAccountKey = EncryptedOrgAccountKey(orgId = accountOrg.orgId,
                                                           encryptedAccountPrivateKey = accountOrg.accountPrivateKey.encryptedAccountPrivateKey()))
    }

    override fun toString(): String {
        return "AccountPath(requester=$requester, accountPublicId=$accountPublicId)"
    }
}
