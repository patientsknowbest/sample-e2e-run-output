package com.pkb.kms.service

import com.pkb.crypto.SymmetricKey
import com.pkb.crypto.dto.DplKeyDTO
import com.pkb.kms.service.cache.AccountCacheService
import com.pkb.kms.service.cache.BatchDplKeyCacheService
import com.pkb.kms.service.cache.OrgCacheService
import com.pkb.kms.service.cache.PathCacheService
import com.pkb.kms.service.cache.TeamCacheService
import com.pkb.kms.service.cache.UserPrivateKeyCacheService
import com.pkb.kms.service.projection.TeamIds
import com.pkb.kms.shared.representation.AccountIds
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.DplKeyId
import com.pkb.kms.shared.representation.KmsError
import com.pkb.kms.shared.representation.OrgRequester
import com.pkb.kms.shared.representation.PersonPublicId
import com.pkb.kms.shared.representation.PersonRequester
import com.pkb.kms.shared.representation.Requester
import com.pkb.kms.shared.representation.TeamPublicId
import io.vavr.control.Either
import org.springframework.stereotype.Service

@Service
class DplKeyService(
    private val batchDplKeyCacheService: BatchDplKeyCacheService,
    private val accountCacheService: AccountCacheService,
    private val accountSymmetricKeyService: AccountSymmetricKeyService,
    private val accountKeyService: AccountKeyService,
    private val userPrivateKeyCacheService: UserPrivateKeyCacheService,
    private val pathCacheService: PathCacheService,
    private val orgCacheService: OrgCacheService,
    private val teamCacheService: TeamCacheService,
    private val uuidService: UUIDService,
) {
    fun createDplKey(storeForPeople: List<PersonPublicId>, storeForTeams: List<TeamPublicId>): Either<KmsError, DplKeyDTO> {
        val dplSymmetricKey = SymmetricKey.newInstance()
        val dplKeyId = uuidService.uuid()
        return Either.sequenceRight(getAccountIds(storeForPeople, storeForTeams).map {
            it.flatMap { accountIds ->
                saveDplKeyToAccount(dplSymmetricKey, dplKeyId, accountIds)
            }
        }).map { it.first() }
    }

    fun getKeysForUser(requester: Requester, keyIds: List<DplKeyId>, personPublicId: PersonPublicId): Either<KmsError, List<DplKeyDTO>> =
            getUserAccount(personPublicId).flatMap { getKeys(requester, keyIds, it) }


    fun getKeysForTeam(requester: Requester, keyIds: List<DplKeyId>, teamPublicID: TeamPublicId): Either<KmsError, List<DplKeyDTO>> =
            getTeamAccount(teamPublicID).flatMap { getKeys(requester, keyIds, it) }


    private fun getKeys(requester: Requester, keyIds: List<DplKeyId>, accountIds: AccountIds) =
            when (requester) {
                is PersonRequester -> getAccountKeysForUser(requester, accountIds.publicId)
                is OrgRequester -> getAccountKeysForOrg(requester, accountIds.publicId)
            }.flatMap {
                batchDplKeyCacheService.getDplKeysForAccount(accountIds, keyIds, it)
            }

    private fun getAccountIds(personList: List<PersonPublicId>, teamList: List<TeamPublicId>): List<Either<KmsError, AccountIds>> =
            personList.map(this::getUserAccount) + teamList.map(this::getTeamAccount)

    private fun saveDplKeyToAccount(symmetricKey: SymmetricKey, dplKeyId: DplKeyId, accountIds: AccountIds) =
            accountSymmetricKeyService.getAnyAccountSymmetricKey(accountIds.publicId).flatMap { accountKeys ->
                batchDplKeyCacheService.createDplKeyForAccount(symmetricKey, dplKeyId, accountIds.id, accountKeys)
            }

    private fun getTeamAccount(teamPublicID: TeamPublicId): Either<KmsError, AccountIds> =
            teamCacheService.findTeam(teamPublicID).map(TeamIds::teamAccountIds)

    private fun getUserAccount(personPublicId: PersonPublicId): Either<KmsError, AccountIds> =
            accountCacheService.findDefaultAccountForPerson(personPublicId)

    private fun getAccountKeysForOrg(requester: OrgRequester, accountPublicId: AccountPublicId) =
            orgCacheService.findOrg(requester.orgPublicId).flatMap { orgIds ->
                if (orgIds.orgAccountIds.publicId == accountPublicId) {
                    accountKeyService.getOrgAccountKeys(orgIds)
                } else {
                    pathCacheService.findPath(requester, accountPublicId).flatMap(accountKeyService::getAccountKeysInPathNoUser)
                }
            }

    private fun getAccountKeysForUser(requester: PersonRequester, accountPublicId: AccountPublicId) =
            userPrivateKeyCacheService.getCachedUserPrivateKey(requester.personPublicId).flatMap { upk ->
                pathCacheService.findPath(requester, accountPublicId)
                        .flatMap { path -> accountKeyService.getAccountKeysInPath(upk, path) }
            }
}
