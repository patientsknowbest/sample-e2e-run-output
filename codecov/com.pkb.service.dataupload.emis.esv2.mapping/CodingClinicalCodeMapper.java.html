package com.pkb.service.dataupload.emis.esv2.mapping;

import com.pkb.dataupload.emis.esv2.csv.EmisEsCodingClinicalCode;
import com.pkb.service.dataupload.emis.esv2.model.CodingClinicalCode;
import com.pkb.service.dataupload.emis.esv2.model.ImmutableCodingClinicalCode;

import java.time.ZoneId;
import java.util.Optional;

public class CodingClinicalCodeMapper
        implements LookupDataMapper<CodingClinicalCode, EmisEsCodingClinicalCode> {

    @Override
    public EmisEsCodingClinicalCode toJpaEntity(CodingClinicalCode object, ZoneId zoneId) {
        EmisEsCodingClinicalCode entity = new EmisEsCodingClinicalCode();
        entity.setCodeId(object.codeId());
        entity.setTerm(object.term());
        entity.setReadTermId(object.readTermId().orElse(null));
        entity.setSnomedCTConceptId(object.snomedCTConceptId());
        entity.setSnomedCTDescriptionId(object.snomedCTDescriptionId());
        entity.setNationalCode(object.nationalCode().orElse(null));
        entity.setNationalCodeCategory(object.nationalCodeCategory().orElse(null));
        entity.setNationalDescription(object.nationalDescription().orElse(null));
        entity.setEmisCodeCategoryDescription(object.emisCodeCategoryDescription().orElse(null));
        entity.setProcessingId(object.processingId());
        return entity;
    }

    @Override
    public CodingClinicalCode fromJpaEntity(EmisEsCodingClinicalCode entity, ZoneId zoneId) {
        return ImmutableCodingClinicalCode.builder()
                .codeId(entity.getCodeId())
                .emisCodeCategoryDescription(Optional.ofNullable(entity.getEmisCodeCategoryDescription()))
                .nationalCodeCategory(Optional.ofNullable(entity.getNationalCodeCategory()))
                .nationalCode(Optional.ofNullable(entity.getNationalCode()))
                .nationalDescription(Optional.ofNullable(entity.getNationalDescription()))
                .processingId(entity.getProcessingId())
                .readTermId(Optional.ofNullable(entity.getReadTermId()))
                .snomedCTConceptId(entity.getSnomedCTConceptId())
                .snomedCTDescriptionId(entity.getSnomedCTDescriptionId())
                .term(entity.getTerm())
                .build();
    }

}
