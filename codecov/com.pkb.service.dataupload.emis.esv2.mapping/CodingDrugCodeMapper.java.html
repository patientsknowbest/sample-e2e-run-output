package com.pkb.service.dataupload.emis.esv2.mapping;

import com.pkb.dataupload.emis.esv2.csv.EmisEsCodingDrugCode;
import com.pkb.service.dataupload.emis.esv2.model.CodingDrugCode;
import com.pkb.service.dataupload.emis.esv2.model.ImmutableCodingDrugCode;

import java.time.ZoneId;
import java.util.Optional;

public class CodingDrugCodeMapper
        implements LookupDataMapper<CodingDrugCode, EmisEsCodingDrugCode> {

    @Override
    public EmisEsCodingDrugCode toJpaEntity(CodingDrugCode object, ZoneId zoneId) {
        EmisEsCodingDrugCode entity = new EmisEsCodingDrugCode();
        entity.setCodeId(object.codeId());
        entity.setDmdProductCodeId(object.dmdProductCodeId().orElse(null));
        entity.setTerm(object.term());
        entity.setProcessingId(object.processingId());
        return entity;
    }

    @Override
    public CodingDrugCode fromJpaEntity(EmisEsCodingDrugCode entity, ZoneId zoneId) {
        return ImmutableCodingDrugCode.builder()
                .codeId(entity.getCodeId())
                .dmdProductCodeId(Optional.ofNullable(entity.getDmdProductCodeId()))
                .processingId(entity.getProcessingId())
                .term(entity.getTerm()).build();
    }

}
