package com.pkb.service.dataupload.emis.esv2.mapping;

import com.pkb.dataupload.emis.esv2.csv.EmisEsAdminPatient;
import com.pkb.service.dataupload.emis.es.util.EmisEsDateUtil;
import com.pkb.service.dataupload.emis.esv2.model.AdminPatient;
import com.pkb.service.dataupload.emis.esv2.model.ImmutableAdminPatient;

import java.time.LocalDate;
import java.time.ZoneId;
import java.util.Optional;

import static com.pkb.service.dataupload.emis.es.util.EmisEsDateUtil.DEFAULT_DATE_FORMATTER;
import static java.util.Optional.empty;
import static java.util.Optional.ofNullable;
import static org.apache.commons.lang3.StringUtils.isBlank;
import static org.apache.commons.lang3.StringUtils.trimToNull;

public class AdminPatientMapper
        implements LookupDataMapper<AdminPatient, EmisEsAdminPatient> {

    @Override
    public AdminPatient fromJpaEntity(EmisEsAdminPatient entity, ZoneId zoneId) {
        return ImmutableAdminPatient.builder()
                .patientGuid(entity.getPatientGuid())
                .organisationGuid(entity.getOrganisationGuid())
                .usualGpUserInRoleGuid(ofNullable(entity.getUsualGpUserInRoleGuid()))
                .sex(isBlank(entity.getSex()) ? Optional.empty() : Optional.of(entity.getSex()))
                .dateOfBirth(ofNullable(trimToNull(entity.getDateOfBirth())).map(ld -> LocalDate.parse(ld, DEFAULT_DATE_FORMATTER)))
                .dateOfDeath(ofNullable(entity.getDateOfDeath()).map(ld -> LocalDate.parse(ld, DEFAULT_DATE_FORMATTER)))
                .title(ofNullable(trimToNull(entity.getTitle())))
                .givenName(ofNullable(trimToNull(entity.getGivenName())))
                .middleNames(ofNullable(entity.getMiddleNames()))
                .surname(ofNullable(trimToNull(entity.getSurname())))
                .dateOfRegistration(entity.getDateOfRegistration() != null
                        ? Optional.of(EmisEsDateUtil.parseLocalDate(entity.getDateOfRegistration(), zoneId))
                        : Optional.empty())
                .nhsNumber(ofNullable(entity.getNhsNumber()))
                .patientNumber(ofNullable(entity.getPatientNumber()))
                .patientTypeDescription(ofNullable(entity.getPatientTypeDescription()))
                .dummyType(ofNullable(entity.getDummyType()))
                .houseNameFlatNumber(ofNullable(entity.getHouseNameFlatNumber()))
                .numberAndStreet(ofNullable(entity.getNumberAndStreet()))
                .village(ofNullable(entity.getVillage()))
                .town(ofNullable(entity.getTown()))
                .county(ofNullable(entity.getCounty()))
                .postcode(ofNullable(entity.getPostcode()))
                .residentialInstituteCode(ofNullable(entity.getResidentialInstituteCode()))
                .nhsNumberStatus(ofNullable(entity.getnHSNumberStatus()))
                .carerName(ofNullable(entity.getCarerName()))
                .carerRelation(ofNullable(entity.getCarerRelation()))
                .personGuid(entity.getPersonGuid())
                .dateOfDeactivation(entity.getDateOfDeactivation() != null
                        ? Optional.of(EmisEsDateUtil.parseLocalDate(entity.getDateOfDeactivation(), zoneId))
                        : empty())
                .deleted(entity.getDeleted())
                .spineSensitiveOrNullIfDeleted(ofNullable(entity.getSpineSensitive()))
                .isConfidential(ofNullable(entity.getIsConfidential()))
                .emailAddress(ofNullable(entity.getEmailAddress()))
                .homePhone(ofNullable(entity.getHomePhone()))
                .mobilePhone(ofNullable(entity.getMobilePhone()))
                .externalUsualGPGuid(ofNullable(entity.getExternalUsualGPGuid()))
                .externalUsualGP(ofNullable(entity.getExternalUsualGP()))
                .externalUsualGPOrganisation(ofNullable(entity.getExternalUsualGPOrganisation()))
                .processingId(entity.getProcessingId()).build();
    }

    @Override
    public EmisEsAdminPatient toJpaEntity(AdminPatient object, ZoneId zoneId) {
        EmisEsAdminPatient entity = new EmisEsAdminPatient();
        entity.setPatientGuid(object.patientGuid());
        entity.setOrganisationGuid(object.organisationGuid());
        entity.setUsualGpUserInRoleGuid(object.usualGpUserInRoleGuid().orElse(null));
        entity.setSex(object.sex().orElse(""));
        entity.setDateOfBirth(object.dateOfBirth().map(ld -> ld.format(DEFAULT_DATE_FORMATTER)).orElse(""));
        entity.setDateOfDeath(object.dateOfDeath().map(ld -> ld.format(DEFAULT_DATE_FORMATTER)).orElse(null));
        entity.setTitle(object.title().orElse(""));
        entity.setGivenName(object.givenName().orElse(""));
        entity.setMiddleNames(object.middleNames().orElse(null));
        entity.setSurname(object.surname().orElse(""));
        entity.setDateOfRegistration(object.dateOfRegistration().flatMap(ld -> EmisEsDateUtil.parseDate(ld, zoneId)).orElse(null));
        entity.setNhsNumber(object.nhsNumber().orElse(null));
        entity.setPatientNumber(object.patientNumber().orElse(null));
        entity.setPatientTypeDescription(object.patientTypeDescription().orElse(null));
        entity.setDummyType(object.dummyType().orElse(null));
        entity.setHouseNameFlatNumber(object.houseNameFlatNumber().orElse(null));
        entity.setNumberAndStreet(object.numberAndStreet().orElse(null));
        entity.setVillage(object.village().orElse(null));
        entity.setTown(object.town().orElse(null));
        entity.setCounty(object.county().orElse(null));
        entity.setPostcode(object.postcode().orElse(null));
        entity.setResidentialInstituteCode(object.residentialInstituteCode().orElse(null));
        entity.setnHSNumberStatus(object.nhsNumberStatus().orElse(null));
        entity.setCarerName(object.carerName().orElse(null));
        entity.setCarerRelation(object.carerRelation().orElse(null));
        entity.setPersonGuid(object.personGuid());
        entity.setDateOfDeactivation(object.dateOfDeactivation().flatMap(dod -> EmisEsDateUtil.parseDate(dod, zoneId)).orElse(null));
        entity.setDeleted(object.deleted().orElse(false));
        entity.setSpineSensitive(object.spineSensitiveOrNullIfDeleted().orElse(null));
        entity.setIsConfidential(object.isConfidential().orElse(null));
        entity.setEmailAddress(object.emailAddress().orElse(null));
        entity.setHomePhone(object.homePhone().orElse(null));
        entity.setMobilePhone(object.mobilePhone().orElse(null));
        entity.setExternalUsualGPGuid(object.externalUsualGPGuid().orElse(null));
        entity.setExternalUsualGP(object.externalUsualGP().orElse(null));
        entity.setExternalUsualGPOrganisation(object.externalUsualGPOrganisation().orElse(null));
        entity.setProcessingId(object.processingId());

        return entity;
    }

}
