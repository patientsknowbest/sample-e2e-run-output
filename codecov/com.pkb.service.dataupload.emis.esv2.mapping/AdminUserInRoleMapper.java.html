package com.pkb.service.dataupload.emis.esv2.mapping;

import com.pkb.dataupload.emis.esv2.csv.EmisEsAdminUserInRole;
import com.pkb.service.dataupload.emis.esv2.model.AdminUserInRole;
import com.pkb.service.dataupload.emis.esv2.model.ImmutableAdminUserInRole;

import java.time.ZoneId;
import java.util.Optional;

public class AdminUserInRoleMapper implements LookupDataMapper<AdminUserInRole, EmisEsAdminUserInRole> {
    @Override
    public EmisEsAdminUserInRole toJpaEntity(AdminUserInRole object, ZoneId zoneId) {
        EmisEsAdminUserInRole entity = new EmisEsAdminUserInRole(
                object.userInRoleGuid(),
                object.organisationGuid(),
                object.contractStartDate().atStartOfDay(zoneId).toInstant(),
                object.processingId());

        // optional attributes
        object.title().ifPresent(entity::setTitle);
        object.givenName().ifPresent(entity::setGivenName);
        object.surname().ifPresent(entity::setSurname);
        object.contractEndDate().ifPresent(ced -> entity.setContractEndDate(ced.atStartOfDay(zoneId).toInstant()));
        object.jobCategoryCode().ifPresent(entity::setJobCategoryCode);
        object.jobCategoryName().ifPresent(entity::setJobCategoryName);
        return entity;
    }

    @Override
    public AdminUserInRole fromJpaEntity(EmisEsAdminUserInRole entity, ZoneId zoneId) {
        return ImmutableAdminUserInRole.builder()
                .userInRoleGuid(entity.getUserInRoleGuid())
                .organisationGuid(entity.getOrganisationGuid())
                .title(Optional.ofNullable(entity.getTitle()))
                .givenName(Optional.ofNullable(entity.getGivenName()))
                .surname(Optional.ofNullable(entity.getSurname()))
                .jobCategoryCode(Optional.ofNullable(entity.getJobCategoryCode()))
                .jobCategoryName(Optional.ofNullable(entity.getJobCategoryName()))
                .contractStartDate(entity.getContractStartDate().atZone(zoneId).toLocalDate())
                .contractEndDate(Optional.ofNullable(entity.getContractEndDate()).map(ced -> ced.atZone(zoneId).toLocalDate()))
                .processingId(entity.getProcessingId())
                .build();
    }
}
