package com.pkb.service.dataupload.emis.esv2.mapping;

import com.pkb.dataupload.emis.esv2.csv.EmisEsAdminOrganisation;
import com.pkb.service.dataupload.emis.es.util.EmisEsDateUtil;
import com.pkb.service.dataupload.emis.esv2.model.AdminOrganisation;
import com.pkb.service.dataupload.emis.esv2.model.ImmutableAdminOrganisation;
import org.apache.commons.lang3.StringUtils;

import java.time.ZoneId;
import java.util.Optional;

import static java.util.Optional.empty;
import static java.util.Optional.ofNullable;
import static org.apache.commons.lang3.StringUtils.isBlank;

public class AdminOrganisationMapper
        implements LookupDataMapper<AdminOrganisation, EmisEsAdminOrganisation> {

    @Override
    public AdminOrganisation fromJpaEntity(EmisEsAdminOrganisation entity, ZoneId zoneId) {
        return ImmutableAdminOrganisation.builder()
                .organisationGuid(entity.getOrganisationGuid())
                .cdb(ofNullable(entity.getCdb()))
                .organisationName(ofNullable(StringUtils.trimToNull(entity.getOrganisationName())))
                .odsCode(ofNullable(entity.getOdsCode()))
                .parentOrganisationGuid(Optional.of(entity.getParentOrganisationGuid()).filter(StringUtils::isNotBlank))
                .ccgOrganisationGuid(ofNullable(entity.getCcgOrganisationGuid()))
                .organisationType(ofNullable(entity.getOrganisationType()))
                .openDate(entity.getOpenDate() != null ? EmisEsDateUtil.parseLocalDate(entity.getOpenDate(), zoneId) : null)
                .closeDate(
                        entity.getCloseDate() != null ? Optional.of(EmisEsDateUtil.parseLocalDate(entity.getCloseDate(), zoneId)) : empty())
                .mainLocationGuid(isBlank(entity.getMainLocationGuid()) ? Optional.empty() : Optional.of(entity.getMainLocationGuid()))
                .processingId(entity.getProcessingId()).build();
    }

    @Override
    public EmisEsAdminOrganisation toJpaEntity(AdminOrganisation object, ZoneId zoneId) {
        EmisEsAdminOrganisation entity = new EmisEsAdminOrganisation();
        entity.setOrganisationGuid(object.organisationGuid());
        entity.setCdb(object.cdb().orElse(null));
        entity.setOrganisationName(object.organisationName().orElse(""));
        entity.setOdsCode(object.odsCode().orElse(null));
        entity.setParentOrganisationGuid(object.parentOrganisationGuid().orElse(""));
        entity.setCcgOrganisationGuid(object.ccgOrganisationGuid().orElse(null));
        entity.setOrganisationType(object.organisationType().orElse(null));
        entity.setOpenDate(object.openDate() != null ? object.openDate().atStartOfDay(zoneId).toInstant() : null);
        //noinspection ConstantConditions
        entity.setCloseDate(object.closeDate()
                .map(ld -> ld.atStartOfDay(zoneId).toInstant())
                .orElse(null));
        entity.setMainLocationGuid(object.mainLocationGuid().orElse(""));
        entity.setProcessingId(object.processingId());
        return entity;
    }

}
