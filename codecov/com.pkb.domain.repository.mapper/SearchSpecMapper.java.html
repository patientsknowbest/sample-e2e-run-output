package com.pkb.domain.repository.mapper;

import com.pkb.domain.criteria.LowerBound;
import com.pkb.domain.criteria.TemporalRangeRestriction;
import com.pkb.domain.criteria.UpperBound;
import com.pkb.ehrdata.Filter;
import com.pkb.ehrdata.ImmutablePageFilter;
import com.pkb.ehrdata.PageFilter;
import io.vavr.collection.List;
import io.vavr.control.Option;

import java.time.LocalDate;

import static com.pkb.ehrdata.Filter.Operator.GREATER_OR_EQUAL;
import static com.pkb.ehrdata.Filter.Operator.LESS;
import static com.pkb.ehrdata.Filter.filter;
import static io.vavr.API.List;
import static java.lang.String.format;

public class SearchSpecMapper {

    public Option<PageFilter> mapPageFilter(Option<Integer> maybePageSize) {
        return maybePageSize.map(pageSize -> ImmutablePageFilter.builder()
                .startPosition(0)
                .count(pageSize)
                .build());
    }

    public List<Filter> mapEffectiveTimestamp(TemporalRangeRestriction<LocalDate> effectiveTimestamp) {
        return mapEffectiveTimestamp(effectiveTimestamp, "date01");
    }

    public List<Filter> mapEffectiveTimestamp(TemporalRangeRestriction<LocalDate> effectiveTimestamp, String dateField) {

        var filter1 = effectiveTimestamp.getLowerBound().map(filter -> {
            LocalDate value = filter.getValue();
            LowerBound constraint = filter.getConstraint();
            if (constraint == LowerBound.GREATER) { // [09.11.2018 00:00, ∞ ]
                return filter(dateField, GREATER_OR_EQUAL, value.plusDays(1).atStartOfDay());
            } else if (constraint == LowerBound.GREATER_OR_EQUAL) { // [08.11.2018 00:00, ∞ ]
                return filter(dateField, GREATER_OR_EQUAL, value.atStartOfDay());
            } else {
                throw new RuntimeException(format("Unsupported lower bound constraint [%s]", constraint.name()));
            }
        }).get();

        var filter2 = effectiveTimestamp.getUpperBound().map(filter -> {
            LocalDate value = filter.getValue();
            UpperBound constraint = filter.getConstraint();
            if (constraint == UpperBound.LESS) { // [∞ , 08.11.2018 00:00)
                return filter(dateField, LESS, value.atStartOfDay());
            } else if (constraint == UpperBound.LESS_OR_EQUAL) { // [∞ , 09.11.2018 00:00)
                return filter(dateField, LESS, value.plusDays(1).atStartOfDay());
            } else {
                throw new RuntimeException(format("Unsupported upper bound constraint [%s]", constraint.name()));
            }
        });

        return List(filter1, filter1);
    }
}
