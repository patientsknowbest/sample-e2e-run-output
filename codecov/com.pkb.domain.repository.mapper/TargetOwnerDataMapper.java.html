package com.pkb.domain.repository.mapper;

import com.pkb.datamodel.AccountKeys;
import com.pkb.datamodel.authorization.TargetOwnerData;
import com.pkb.datamodel.entity.mapper.AccountKeysMapper;
import com.pkb.datamodel.entity.mapper.AccountMapper;
import com.pkb.datamodel.entity.mapper.PatientConsentMapper;
import com.pkb.datamodel.entity.mapper.PersonMapper;
import com.pkb.datamodel.entity.mapper.TeamMapper;
import com.pkb.domain.repository.util.TargetOwnerDataHolder;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;

import static com.pkb.datamodel.authorization.ImmutableTargetOwnerData.targetOwnerData;
import static java.util.Objects.nonNull;

public class TargetOwnerDataMapper {
    private final PersonMapper personMapper;
    private final AccountMapper accountMapper;
    private final PatientConsentMapper consentMapper;
    private final TeamMapper teamMapper;
    private final AccountKeysMapper accountKeysMapper;

    public TargetOwnerDataMapper(PersonMapper personMapper,
                                 AccountMapper accountMapper,
                                 PatientConsentMapper consentMapper,
                                 TeamMapper teamMapper,
                                 AccountKeysMapper accountKeysMapper) {
        this.personMapper = personMapper;
        this.accountMapper = accountMapper;
        this.consentMapper = consentMapper;
        this.teamMapper = teamMapper;
        this.accountKeysMapper = accountKeysMapper;
    }
    
    public List<TargetOwnerData> map(List<TargetOwnerDataHolder> results) {
        return results.map(this::map);
    }

    public TargetOwnerData map(TargetOwnerDataHolder result) {
        return targetOwnerData()
                .person(personMapper.entityToDataModel(result.getPerson()))
                .account(accountMapper.entityToDataModel(result.getAccount()))
                .accountKeysReloader(result.getKeyReloader())
                .accountKeys(mapAccountKeys(result))
                .consent(Option.of(result.getConsent()).map(consentMapper::viewConsentToDataModel))
                .build();
    }

    private Option<AccountKeys> mapAccountKeys(@NotNull TargetOwnerDataHolder result) {
        return nonNull(result.getLink())
                ? Option.of(accountKeysMapper.map(
                    result.getUserOrAccountPrivateKey(),
                    result.getLink(),
                    result.getAccountSymmetricKeys()))
                : Option.none();
    }
}
