package com.pkb.action.diary;

import com.google.common.collect.ImmutableList;
import com.pkb.calendar.entity.AppointmentDTO;
import com.pkb.service.user.impl.PrimaryEmailContactManager;
import com.pkb.user.entity.PersonContact;
import com.pkb.user.entity.PrimaryEmailContactDTO;

import java.util.List;
import java.util.Set;
import java.util.regex.Pattern;

class AppointmentUtility {
    private static final Pattern DESCRIPTION_NEW_LINES = Pattern.compile("(?:\r\n|\n|\r)");

    static String prepareMultiline(String fieldValue) {
        if (fieldValue == null) {
            return "";
        } else {
            return DESCRIPTION_NEW_LINES.matcher(fieldValue).replaceAll("<br/>");
        }
    }

    static List<Long> getInvitedCarerIds(long patientId, AppointmentDTO dto, Set<Long> participants, PrimaryEmailContactManager primaryEmailContactManager) {
        PrimaryEmailContactDTO contactDto = primaryEmailContactManager.getPrimaryEmailContactDTO(patientId, dto);
        return contactDto.getAlsoContactPrimaryEmailContactList()
                .stream()
                .filter(PersonContact::isConfirmed)
                .map(contact -> contact.getPerson().getId())
                .filter(participants::contains)
                .collect(ImmutableList.toImmutableList());
    }

}
