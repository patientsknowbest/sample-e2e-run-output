package com.pkb.action.diary;

import com.pkb.datamodel.ui.UIDto;

public class ConfirmAppointmentRequestAction extends AbstractAppointmentRequestAction {

    private UIConfirmAppointmentRequestDto dto;

    @Override
    public String input() {
        dto = appointmentsService.confirmAppointmentRequest(ImmutableConfirmAppointmentRequestRequest.builder()
                .eHRRequestContext(getEHRRequestContext())
                .locale(getLocale())
                .uniqueId(getUniqueId())
                .requestType(getRequestType())
                .nhsStyle(sessionUtil.renderNhsStyle())
                .build());
        return getActionResult(dto);
    }

    public String finish() {
        UIDto finishDto = appointmentsService.finishAppointmentRequest(ImmutableFinishAppointmentRequestRequest.builder()
                .eHRRequestContext(getEHRRequestContext())
                .locale(getLocale())
                .uniqueId(getUniqueId())
                .requestType(getRequestType())
                .build());
        return getActionResult(finishDto);
    }

    public UIConfirmAppointmentRequestDto getDto() {
        return dto;
    }

    public void setDto(UIConfirmAppointmentRequestDto dto) {
        this.dto = dto;
    }
}
