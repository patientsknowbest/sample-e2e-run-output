package com.pkb.action.diary;

import com.pkb.action.NhsLoginAwareExternalRedirectAction;
import com.pkb.service.integration.AppointmentManagementPartnerService;
import com.pkb.util.UrlUtil;
import org.springframework.beans.factory.annotation.Autowired;

import java.net.URI;
import java.util.Map;
import java.util.UUID;

import static com.pkb.util.WebNamespace.DIARY;

public class AppointmentManagementRedirectAction extends NhsLoginAwareExternalRedirectAction {

    @Autowired
    AppointmentManagementPartnerService appointmentManagementPartnerService;

    private UUID appointmentUniqueId;

    @Override
    protected String amendRedirectUrl(String managementUrlString) {
        Long contextUserId = getLoggedInEHRRequestContext().getContextUserId().orElse(null);
        String callbackUrlvalue = UrlUtil.buildHref(config.getBaseURL(),
                DIARY,
                "listAppointments",
                Map.of("fromPartner", "true"), //so when can show a message that updates may not appear immediately
                contextUserId);

        return appointmentManagementPartnerService.validateAndAppendPkbCallback(managementUrlString, appointmentUniqueId, callbackUrlvalue)
                .map(URI::toString)
                .orElseThrow();
    }

    @Override
    public UUID getDataPointUniqueId() {
        return appointmentUniqueId;
    }

    @Override
    public void setDataPointUniqueId(UUID datapointUniqueId) {
        this.appointmentUniqueId = datapointUniqueId;
    }
}
