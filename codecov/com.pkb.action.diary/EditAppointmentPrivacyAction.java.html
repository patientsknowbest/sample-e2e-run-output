package com.pkb.action.diary;

import com.opensymphony.xwork2.Action;
import com.pkb.action.ConsentCategoryBaseAction;
import com.pkb.action.EditPrivacyService;
import com.pkb.action.ImmutableFormRequestAppointment;
import com.pkb.action.ImmutableSaveRequestAppointment;
import org.apache.struts2.interceptor.validation.SkipValidation;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.UUID;

import static com.pkb.action.DtoType.APPOINTMENT;

public class EditAppointmentPrivacyAction extends ConsentCategoryBaseAction {

    @Autowired
    private EditPrivacyService editPrivacyService;

    private UIPrivacyDto dto;

    private UUID uniqueId;

    private boolean returnToAppointment;

    public String editPrivacyForm() {
        if (uniqueId == null) {
            addActionError(getText("editAppointment.err.noId"));
            return Action.ERROR;
        }
        var request = ImmutableFormRequestAppointment.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .dtoType(APPOINTMENT)
                .dtoIdentifier(getUniqueId())
                .locale(getLocale())
                .nhsStyle(sessionUtil.renderNhsStyle())
                .returnToAppointment(returnToAppointment)
                .build();
        dto = editPrivacyService.showPrivacyForm(request);
        dto.getPrivacyFlags().forEach(this::fillPrivacyFlagsFrom);
        return getActionResult(dto);
    }

    @SkipValidation
    public String editPrivacySave() {
        var request = ImmutableSaveRequestAppointment.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .dtoType(APPOINTMENT)
                .dtoIdentifier(getUniqueId())
                .privacy(privacy)
                .locale(getLocale())
                .nhsStyle(sessionUtil.renderNhsStyle())
                .returnToAppointment(returnToAppointment)
                .build();
        dto = editPrivacyService.savePrivacy(request);
        return getActionResult(dto);
    }

    @SkipValidation
    public String editPrivacyCancel() {
        return returnToAppointment ? RETURN_TO_APPOINTMENT : RETURN_TO_APPOINTMENTS_LIST;
    }

    @Override
    public String getCustomStyle() {
        if (sessionUtil.getDynamicFeatures().renderNhsStyle()) {
            return NHS_STYLE;
        } else { return ""; }
    }

    public UUID getUniqueId() {
        return uniqueId;
    }

    public void setUniqueId(UUID uniqueId) {
        this.uniqueId = uniqueId;
    }

    public String getTab() {
        return dto.getTab();
    }

    public boolean isReturnToAppointment() {
        return returnToAppointment;
    }

    public void setReturnToAppointment(boolean returnToAppointment) {
        this.returnToAppointment = returnToAppointment;
    }
}
