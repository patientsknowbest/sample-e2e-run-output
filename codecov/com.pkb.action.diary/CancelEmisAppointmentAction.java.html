package com.pkb.action.diary;


import com.pkb.action.emis.pfs.EmisPfsStatsCollector;
import com.pkb.emis.appointment.service.EmisAppointmentManager;
import com.pkb.integrations.emis.model.EmisAppointment;
import com.pkb.integrations.emis.model.restcalls.EmisAppointmentCancelRequest;
import com.pkb.integrations.emis.model.restcalls.ImmutableEmisAppointmentCancelRequest;
import com.pkb.util.Constants;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;

import java.time.ZoneId;
import java.time.format.DateTimeFormatter;
import java.util.Locale;
import java.util.Optional;

class CancelEmisAppointmentAction extends BaseAppointmentAction {
    private static final DateTimeFormatter WEEKDAY_DAY_MONTH_YEAR = DateTimeFormatter.ofPattern("EEE d MMMM yyyy")
            .withZone(ZoneId.of("Europe/London")).withLocale(
                    Locale.ENGLISH);
    private static final DateTimeFormatter HOURS_MINUTES = DateTimeFormatter.ofPattern("HH:mm");
    private Long slotId;
    private String startDate;
    private String startTime;
    private String slotTypeName;
    private String clinicians;
    private String location;
    private String cancellationReasonText;

    public String getTab() {
        return "diary";
    }

    public String getSubTab() {
        return "calendar";
    }

    @Override
    public String input() {
        if (slotId != null && slotId > 0) {
            Optional<EmisAppointment> maybeAppointment = emisAppointmentManager.retrieveAppointments(getEmisContextHolder(), null, null)
                    .peekLeft(error -> {
                        //noinspection StrutsI18nError,SSBasedInspection
                        addActionError(emisMessageHelper.retrieveAppointmentsError(error, getLocale(), getEmisContextHolder()));
                    })
                    .map(result -> result.stream()
                            .filter(emisAppointment -> emisAppointment.getSlotId() == slotId)
                            .findFirst())
                    .getOrElse(Optional.empty());
            if (maybeAppointment.isPresent()) {
                EmisAppointment emisAppointment = maybeAppointment.get();
                startDate = getFormattedStartDateTime(emisAppointment, WEEKDAY_DAY_MONTH_YEAR);
                startTime = getFormattedStartDateTime(emisAppointment, HOURS_MINUTES);
                slotTypeName = emisAppointment.getSlotTypeName().orElse("");
                clinicians = EmisAppointmentManager.getCliniciansName(emisAppointment.getSession().getClinicians());
                location = emisAppointment.getSession().getLocation().getLocationName();
                return INPUT;
            }
        }

        addActionError(getText("err.emis.appointment.retrieval"));
        return CANCEL;
    }

    @NotNull
    private String getFormattedStartDateTime(EmisAppointment emisAppointment, DateTimeFormatter dateTimeFormatter) {
        return emisAppointment.getStartTime().atZone(Constants.APPLICATION_TZ).withZoneSameInstant(getZoneIdOfLoggedInUser()).format(dateTimeFormatter);
    }

    @Override
    public void validate() {
        if (StringUtils.isBlank(cancellationReasonText)) {
            addFieldError("cancellationReasonText", getText("cancelEmisAppointment.txt.error_cancel_reason"));
        }
    }

    @Override
    public String execute() {
        if (slotId != null && slotId > 0) {
            EmisAppointmentCancelRequest cancellationContext = ImmutableEmisAppointmentCancelRequest.builder()
                    .slotId(slotId)
                    .cancellationReason(cancellationReasonText)
                    .build();
            return emisAppointmentManager.cancelAnAppointment(getEmisContextHolder(), cancellationContext)
                    .map(result -> {
                        addActionMessage(getText("cancelEmisAppointment.txt.successfully_cancelled"));
                        emisAppointmentManager.getNationalPracticeCode(getEmisContextHolder())
                                .peek(EmisPfsStatsCollector::countCancelAppointmentPerPractice);
                        return SUCCESS;
                    })
                    .getOrElseGet(error -> {
                        //noinspection StrutsI18nError,SSBasedInspection
                        addActionError(emisMessageHelper.getEmisActionError(error, getLocale(), getEmisContextHolder()));
                        EmisPfsStatsCollector.countCancelAppointmentErrors(error);
                        return ERROR;
                    });

        }
        return SUCCESS;
    }

    public String cancel() {
        return CANCEL;
    }

    public Long getSlotId() {
        return slotId;
    }

    public void setSlotId(Long slotId) {
        this.slotId = slotId;
    }

    public String getStartDate() {
        return startDate;
    }

    public String getStartTime() {
        return startTime;
    }

    public String getSlotTypeName() {
        return slotTypeName;
    }

    public String getClinicians() {
        return clinicians;
    }

    public String getLocation() {
        return location;
    }

    public String getCancellationReasonText() {
        return cancellationReasonText;
    }

    public void setCancellationReasonText(String cancellationReasonText) {
        this.cancellationReasonText = cancellationReasonText;
    }
}
