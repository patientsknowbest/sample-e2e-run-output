package com.pkb.action.diary;

import com.pkb.action.BaseAction;
import com.pkb.action.diary.AppointmentsService.AppointmentRequestType;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.UUID;

public abstract class AbstractAppointmentRequestAction extends BaseAction {

    @Autowired
    AppointmentsService appointmentsService;

    private UUID uniqueId;

    private AppointmentRequestType requestType;

    private String tab = "diary";

    private String subTab = "calendar";

    public UUID getUniqueId() {
        return uniqueId;
    }

    public void setUniqueId(UUID uniqueId) {
        this.uniqueId = uniqueId;
    }

    public AppointmentRequestType getRequestType() {
        return requestType;
    }

    public void setRequestType(String requestType) {
        this.requestType = AppointmentRequestType.valueOf(requestType.toUpperCase());
    }

    @Override
    public String getCustomStyle() {
        return sessionUtil.renderNhsStyle() ? NHS_STYLE : PKB_STYLE;
    }

    public String getTab() {
        return tab;
    }

    public String getSubTab() {
        return subTab;
    }
}
