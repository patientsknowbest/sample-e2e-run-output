package com.pkb.action.diary;

import java.time.Instant;
import java.util.ArrayList;
import java.util.List;

import static org.apache.commons.lang3.StringUtils.EMPTY;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class UIAppointmentDetailsDto {

    protected String subject = EMPTY;
    private Instant startTimestamp;
    private String location = EMPTY;

    private String attendeesLabel;
    private List<UIAppointmentCarerDto> carers;
    protected List<String> participants = new ArrayList<>();
    protected boolean guestlistAvailable;
    private String timeFormat;

    public String getSubject() {
        return subject;
    }

    public void setSubject(String subject) {
        this.subject = subject;
    }

    public String getLocation() {
        return location;
    }

    public void setLocation(String location) {
        this.location = location;
    }

    public boolean isLocationAvailable() {
        return isNotBlank(location);
    }

    public Instant getStartTimestamp() {
        return startTimestamp;
    }

    public void setStartTimestamp(Instant startTimestamp) {
        this.startTimestamp = startTimestamp;
    }

    public List<UIAppointmentCarerDto> getCarers() {
        return carers;
    }

    public void setCarers(List<UIAppointmentCarerDto> carers) {
        this.carers = carers;
    }

    public List<String> getParticipants() {
        return participants;
    }

    public void setParticipants(List<String> participants) {
        this.participants = participants;
    }

    public boolean isGuestlistAvailable() {
        return guestlistAvailable;
    }

    public void setGuestlistAvailable(boolean guestlistAvailable) {
        this.guestlistAvailable = guestlistAvailable;
    }

    public String getAttendeesLabel() {
        return attendeesLabel;
    }

    public void setAttendeesLabel(String attendeesLabel) {
        this.attendeesLabel = attendeesLabel;
    }

    public String getTimeFormat() {
        return timeFormat;
    }

    public void setTimeFormat(String timeFormat) {
        this.timeFormat = timeFormat;
    }
}
