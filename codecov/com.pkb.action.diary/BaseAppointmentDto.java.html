package com.pkb.action.diary;

import com.pkb.datamodel.ui.UIDto;

import java.util.ArrayList;
import java.util.List;
import java.util.Optional;

import static com.opensymphony.xwork2.Action.SUCCESS;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class BaseAppointmentDto extends UIDto {

    protected String heading;
    protected String subject;
    protected boolean cancelled;
    protected boolean didNotAttend;
    protected String status;
    protected Optional<String> speciality;
    protected String location;
    protected Optional<String> source;
    protected Optional<String> typeCode;
    protected Optional<String> description;
    protected List<UIAppointmentCarerDto> carers;
    protected List<Long> participantIds;
    protected List<String> participants;
    protected Optional<String> appointmentRequestConversationHref;
    protected boolean guestlistAvailable;

    public BaseAppointmentDto(List<Long> patientId) {
        this.subject = "";
        this.cancelled = false;
        this.didNotAttend = false;
        this.status = "ACTIVE";
        this.speciality = Optional.empty();
        this.typeCode = Optional.empty();
        this.description = Optional.empty();
        this.source = Optional.empty();
        this.participantIds = new ArrayList<>(patientId);
        this.participants = new ArrayList<>();
        this.setActionResult(SUCCESS);
    }

    public String getHeading() {
        return heading;
    }

    public void setHeading(String heading) {
        this.heading = heading;
    }

    public String getSubject() {
        return subject;
    }

    public void setSubject(String subject) {
        this.subject = subject;
    }

    public boolean isCancelled() {
        return cancelled;
    }

    public void setCancelled(boolean cancelled) {
        this.cancelled = cancelled;
    }

    public boolean isDidNotAttend() {
        return didNotAttend;
    }

    public void setDidNotAttend(boolean didNotAttend) {
        this.didNotAttend = didNotAttend;
    }

    public String getStatus() {
        return status;
    }

    public void setStatus(String status) {
        this.status = status;
    }

    public Optional<String> getSpeciality() {
        return speciality;
    }

    public void setSpeciality(Optional<String> speciality) {
        this.speciality = speciality;
    }

    public String getLocation() {
        return location;
    }

    public void setLocation(String location) {
        this.location = location;
    }

    public boolean isLocationAvailable() {
        return isNotBlank(location);
    }

    public Optional<String> getSource() {
        return source;
    }

    public void setSource(Optional<String> source) {
        this.source = source;
    }

    public Optional<String> getTypeCode() {
        return typeCode;
    }

    public void setTypeCode(Optional<String> typeCode) {
        this.typeCode = typeCode;
    }

    public Optional<String> getDescription() {
        return description;
    }

    public void setDescription(Optional<String> description) {
        this.description = description;
    }

    public List<UIAppointmentCarerDto> getCarers() {
        return carers;
    }

    public void setCarers(List<UIAppointmentCarerDto> carers) {
        this.carers = carers;
    }

    public List<Long> getParticipantIds() {
        return participantIds;
    }

    public void setParticipantIds(List<Long> participantIds) {
        this.participantIds = participantIds;
    }

    public List<String> getParticipants() {
        return participants;
    }

    public void setParticipants(List<String> participants) {
        this.participants = participants;
    }

    public void setAppointmentRequestConversationHref(Optional<String> appointmentRequestConversationHref) {
        this.appointmentRequestConversationHref = appointmentRequestConversationHref;
    }

    public Optional<String> getAppointmentRequestConversationHref() {
        return appointmentRequestConversationHref;
    }

    public boolean isGuestlistAvailable() {
        return guestlistAvailable;
    }

    public void setGuestlistAvailable(boolean guestlistAvailable) {
        this.guestlistAvailable = guestlistAvailable;
    }
}
