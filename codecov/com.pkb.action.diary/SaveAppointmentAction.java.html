package com.pkb.action.diary;

import com.opensymphony.xwork2.Action;
import com.pkb.action.message.MessageBaseAction;
import org.apache.struts2.interceptor.validation.SkipValidation;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.UUID;

import static java.util.Optional.ofNullable;


public class SaveAppointmentAction extends MessageBaseAction {

    private static final String TAB_DIARY = "diary";
    private static final String TAB_SCHEDULE = "schedule";
    private static final String SUB_TAB = "calendar";

    @Autowired
    private AppointmentsService appointmentsService;

    private UIAppointmentDto dto;

    private UUID uniqueId;

    private boolean returnToAppointment;

    @Override
    public String input() {
        setDefaultFlag();
        if (dto == null) {
            var request = ImmutableCreateAppointmentFormRequest.builder()
                    .eHRRequestContext(getLoggedInEHRRequestContext())
                    .locale(getLocale())
                    .build();
            dto = appointmentsService.createAppointmentForm(request);
        } else {
            dto.setActionResult(SUCCESS);
        }
        return getActionResult(dto);
    }

    public String save() {
        var request = ImmutableSaveAppointmentRequest.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .dto(getDto())
                .activityUrl(getRequest().getRequestURL().toString())
                .privacy(ofNullable(getPrivacy()))
                .uniqueId(ofNullable(getUniqueId()))
                .locale(getLocale())
                .nhsStyle(sessionUtil.renderNhsStyle())
                .returnToAppointment(returnToAppointment)
                .build();
        dto = appointmentsService.saveAppointment(request);
        return getActionResult(dto);
    }

    @SkipValidation
    public String edit() {
        if (uniqueId == null) {
            addActionError(getText("editAppointment.err.noId"));
            return Action.ERROR;
        }
        var request = ImmutableEditAppointmentRequest.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .uniqueId(getUniqueId())
                .locale(getLocale())
                .build();
        dto = appointmentsService.editAppointment(request);
        String result = getActionResult(dto);
        if (Action.SUCCESS.equals(result) && dto.isNhsStyle()) {
            result = RETURN_TO_APPOINTMENT; // Special case for PHR-10370
        }
        return result;
    }

    public String cancel() {
        return returnToAppointment ? RETURN_TO_APPOINTMENT : RETURN_TO_APPOINTMENTS_LIST;
    }

    public UIAppointmentDto getDto() {
        return dto;
    }

    public void setDto(UIAppointmentDto dto) {
        this.dto = dto;
    }

    public String getTab() {
        return getLoggedInEHRRequestContext().isPatientAccount() ? TAB_DIARY : TAB_SCHEDULE;
    }

    public String getSubTab() {
        return SUB_TAB;
    }

    public UUID getUniqueId() {
        return uniqueId;
    }

    public void setUniqueId(UUID uniqueId) {
        this.uniqueId = uniqueId;
    }

    public boolean isReturnToAppointment() {
        return returnToAppointment;
    }

    public void setReturnToAppointment(boolean returnToAppointment) {
        this.returnToAppointment = returnToAppointment;
    }
}
