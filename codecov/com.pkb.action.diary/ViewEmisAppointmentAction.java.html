package com.pkb.action.diary;

import org.springframework.beans.factory.annotation.Autowired;

public class ViewEmisAppointmentAction extends BaseAppointmentAction {

    @Autowired
    private AppointmentsService appointmentsService;

    private UIAppointmentViewDto dto;

    private long slotId;

    @Override
    public String input() {
        var request = ImmutableViewEmisAppointmentRequest.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .emisSlotId(getSlotId())
                .locale(getLocale())
                .nhsStyle(sessionUtil.renderNhsStyle())
                .build();
        dto = appointmentsService.viewEmisAppointment(request);
        return getActionResult(dto);
    }

    public String cancel() {
        return CANCEL;
    }

    public UIAppointmentViewDto getDto() {
        return dto;
    }

    public long getSlotId() {
        return slotId;
    }

    public void setSlotId(long slotId) {
        this.slotId = slotId;
    }

    public String getTab() {
        return "diary";
    }

    public String getSubTab() {
        return "calendar";
    }

    @Override
    public String getCustomStyle() {
        return NO_HEADER_STYLE;
    }
}
