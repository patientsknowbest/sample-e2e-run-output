package com.pkb.action.diary;

public class IntroduceAppointmentRequestAction extends AbstractAppointmentRequestAction {

    private UIIntroduceAppointmentRequestDto dto;

    @Override
    public String input() {
        var request = ImmutableIntroduceAppointmentRequestRequest.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .uniqueId(getUniqueId())
                .locale(getLocale())
                .requestType(getRequestType())
                .nhsStyle(sessionUtil.renderNhsStyle())
                .build();
        dto = appointmentsService.introduceAppointmentRequest(request);
        return getActionResult(dto);
    }

    public UIIntroduceAppointmentRequestDto getDto() {
        return dto;
    }

    public void setDto(UIIntroduceAppointmentRequestDto dto) {
        this.dto = dto;
    }
}
