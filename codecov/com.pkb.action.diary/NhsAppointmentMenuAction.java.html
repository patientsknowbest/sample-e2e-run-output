package com.pkb.action.diary;

import com.pkb.action.BaseAction;
import com.pkb.dto.AppointmentActionsDto;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.UUID;

public class NhsAppointmentMenuAction extends BaseAction {

    @Autowired
    private AppointmentsService appointmentsService;

    private UUID uniqueId;

    private AppointmentActionsDto dto;

    private boolean returnToAppointment;

    @Override
    public String input() {
        var request = ImmutableViewAppointmentRequest.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .uniqueId(getUniqueId())
                .locale(getLocale())
                .nhsStyle(true)
                .build();
        dto = appointmentsService.appointmentActions(request);
        return SUCCESS;
    }

    public UUID getUniqueId() {
        return uniqueId;
    }

    public void setUniqueId(UUID uniqueId) {
        this.uniqueId = uniqueId;
    }

    public AppointmentActionsDto getDto() {
        return dto;
    }

    public void setDto(AppointmentActionsDto dto) {
        this.dto = dto;
    }

    @Override
    public String getCustomStyle() {
        //this page is only accessible when in nhs style
        return NHS_STYLE;
    }

    public boolean isReturnToAppointment() {
        return returnToAppointment;
    }

    public void setReturnToAppointment(boolean returnToAppointment) {
        this.returnToAppointment = returnToAppointment;
    }
}
