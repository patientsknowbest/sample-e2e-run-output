package com.pkb.action.diary;

import com.pkb.action.BaseAction;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.Objects;

import static com.pkb.action.BaseAction.NhsAppPage.APPOINTMENTS;


/**
 * Action class to list the patient's past and upcoming appointments
 *
 * Maps to listAppointments.jsp
 */
public class ListAppointmentsAction extends BaseAction {

    private static final String TAB_DIARY = "diary";
    private static final String TAB_SCHEDULE = "schedule";
    private static final String SUB_TAB = "calendar";

    private Integer upcomingPageCount;
    private Integer pastPageCount;
    private boolean fromPartner;

    private ListAppointmentDto dto;

    @Autowired
    private AppointmentsService appointmentsService;

    public String listAppointments() {
        var request = ImmutableListAppointmentRequest.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .locale(getLocale())
                .pastPageCount(getPastPageCount())
                .upcomingPageCount(getUpcomingPageCount())
                .isFromPartner(isFromPartner())
                .build();
        dto = appointmentsService.listAppointments(request);
        return getActionResult(dto);
    }

    @Override
    public String getNhsAppPage() {
        return APPOINTMENTS.name();
    }

    @Override
    public String getCustomStyle() {
        return MOBILE_OPTIMIZED_STYLE;
    }

    public ListAppointmentDto getDto() {
        return dto;
    }

    public String getTab() {
        return getLoggedInEHRRequestContext().isPatientAccount() ? TAB_DIARY : TAB_SCHEDULE;
    }

    public String getSubTab() {
        return SUB_TAB;
    }

    private int getUpcomingPageCount() {
        return Objects.requireNonNullElse(upcomingPageCount, 1);
    }

    public void setUpcomingPageCount(int upcomingPageCount) {
        this.upcomingPageCount = upcomingPageCount;
    }

    private int getPastPageCount() {
        return Objects.requireNonNullElse(pastPageCount, 1);
    }

    public void setPastPageCount(int pastPageCount) {
        this.pastPageCount = pastPageCount;
    }

    public boolean isFromPartner() {
        return fromPartner;
    }

    public void setFromPartner(boolean fromPartner) {
        this.fromPartner = fromPartner;
    }
}
