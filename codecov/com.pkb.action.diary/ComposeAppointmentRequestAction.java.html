package com.pkb.action.diary;

import com.pkb.action.message.MessageBaseAction;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.List;
import java.util.Optional;
import java.util.UUID;

public class ComposeAppointmentRequestAction extends MessageBaseAction {

    private static final String tab = "diary";
    private static final String subTab = "calendar";

    @Autowired
    private AppointmentsService appointmentsService;

    /* URL params */
    private AppointmentsService.AppointmentRequestType requestType;
    private UUID uniqueId;

    /* Fields for the JSP's */
    private List<String> recipientIds;
    private UIComposeAppointmentRequestDto dto;

    @Override
    public String input() {
        dto = appointmentsService.composeAppointmentRequestMessage(ImmutableComposeAppointmentRequestMessageRequest.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .requestType(getRequestType())
                .locale(getLocale())
                .uniqueId(getUniqueId())
                .dto(Optional.ofNullable(getDto()))
                .nhsStyle(sessionUtil.renderNhsStyle())
                .build());
        return getActionResult(dto);
    }

    public String reviewAppointmentRequestMessage() {
        AppointmentsService.ReviewAppointmentRequestMessageRequest reviewRequest = ImmutableReviewAppointmentRequestMessageRequest.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .requestType(getRequestType())
                .locale(getLocale())
                .uniqueId(getUniqueId())
                .dto(getDto())
                .nhsStyle(sessionUtil.renderNhsStyle())
                .build();
        dto = appointmentsService.reviewAppointmentRequestMessage(reviewRequest);
        if (dto.getResponseMeta().hasErrors()) {
            fillPrivacyFlagsFrom(dto.getPrivacy());
            input();
        }
        return getActionResult(dto);
    }

    public String sendAppointmentRequestMessage() {
        dto = appointmentsService.sendAppointmentRequestMessage(ImmutableSendAppointmentRequestMessageRequest.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .requestType(getRequestType())
                .locale(getLocale())
                .uniqueId(getUniqueId())
                .dto(getDto())
                .build());
        if (dto.getResponseMeta().hasErrors()) {
            reviewAppointmentRequestMessage();
        }
        return getActionResult(dto);
    }

    @Override
    public String getCustomStyle() {
        return sessionUtil.renderNhsStyle() ? NHS_STYLE : PKB_STYLE;
    }

    public UUID getUniqueId() {
        return uniqueId;
    }

    public void setUniqueId(UUID uniqueId) {
        this.uniqueId = uniqueId;
    }

    public AppointmentsService.AppointmentRequestType getRequestType() {
        return requestType;
    }

    public void setRequestType(String requestType) {
        this.requestType = AppointmentsService.AppointmentRequestType.valueOf(requestType.toUpperCase());
    }

    public String getTab() {
        return tab;
    }

    public String getSubTab() {
        return subTab;
    }

    public UIComposeAppointmentRequestDto getDto() {
        return dto;
    }

    public void setDto(UIComposeAppointmentRequestDto dto) {
        this.dto = dto;
    }

    public List<String> getRecipientIds() {
        return recipientIds;
    }

    public void setRecipientIds(List<String> recipientIds) {
        this.recipientIds = recipientIds;
    }
}
