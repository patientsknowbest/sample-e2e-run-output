package com.pkb.backgroundtasks.util

import org.slf4j.MDC
import org.springframework.stereotype.Component
import java.util.UUID

@Component
class RequestIdUtil {

    fun getRequestId(): String {
        return MDC.get(RequestIdFilter.REQUEST_ID) ?: UUID.randomUUID().toString()
    }


    /**
     * Helper method that can be used to wrap code that may run from a web request or may be triggered by some
     * background trigger (pubsub/scheduling) and so won't have a request id provided by the normal filter
     * mechanism; nevertheless we may still want to have a single, consistent request id for tracking stuff
     * in the logs
     */
    fun <T> ensureRequestId(toRun: () -> T) : T {
        val existingRequestId = MDC.get(RequestIdFilter.REQUEST_ID)
        if (existingRequestId == null) {
            val newRequestId = UUID.randomUUID().toString()
            MDC.put(RequestIdFilter.REQUEST_ID, newRequestId)
            try {
                return toRun.invoke()
            }
            finally {
                MDC.remove(RequestIdFilter.REQUEST_ID)
            }
        }

        return toRun.invoke()
    }
}
