package com.pkb.api.fhir.config;

import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.boot.context.properties.ConstructorBinding;
import org.springframework.boot.context.properties.bind.DefaultValue;
import org.springframework.cloud.context.config.annotation.RefreshScope;

@RefreshScope
@ConstructorBinding
@ConfigurationProperties(prefix = "hybrid.mail.api")
public class HybridMailProperties {


    private final int generalRequestLimit;
    private final int invitationTokenRequestLimit;

    public HybridMailProperties(
        @DefaultValue("5000") int generalRequestLimit,
        @DefaultValue("500") int invitationTokenRequestLimit
    ) {

        this.generalRequestLimit = generalRequestLimit;
        this.invitationTokenRequestLimit = invitationTokenRequestLimit;
    }

    public int getGeneralRequestLimit() {
        return generalRequestLimit;
    }

    public int getInvitationTokenRequestLimit() {
        return invitationTokenRequestLimit;
    }
}
