package com.pkb.api.fhir.config;

import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.boot.context.properties.ConstructorBinding;
import org.springframework.boot.context.properties.bind.DefaultValue;
import org.springframework.cloud.context.config.annotation.RefreshScope;

@RefreshScope
@ConstructorBinding
@ConfigurationProperties(prefix = "fhir.api")
public class FhirApiProperties {

    private final String url;
    private final boolean experimental;
    private final int observationMaxNumberOfResources;
    private final int appointmentMaxNumberOfResources;
    private final int communicationMaxNumberOfResources;
    private final int diagnosticReportMaxNumberOfResources;
    private final int encounterMaxNumberOfResources;
    private final int documentReferenceMaxNumberOfResources;
    private final int questionnaireMaxNumberOfResources;
    private final int questionnaireResponseMaxNumberOfResources;
    private final String questionnaireServiceUrl;
    private final int questionnaireRequestCreationBatchSize;
    private final int questionnaireRequestCreationBufferTimeoutMillis;
    private final int questionnaireRequestOperationMaxNumberOfTargets;

    public FhirApiProperties(
            @DefaultValue("https://my.patientsknowbest.com/fhir") String url,
            @DefaultValue("false") boolean experimental,
            @DefaultValue("60000") int observationMaxNumberOfResources,
            @DefaultValue("1000") int appointmentMaxNumberOfResources,
            @DefaultValue("1000") int communicationMaxNumberOfResources,
            @DefaultValue("1000") int diagnosticReportMaxNumberOfResources,
            @DefaultValue("1000") int encounterMaxNumberOfResources,
            @DefaultValue("10000") int documentReferenceMaxNumberOfResources,
            @DefaultValue("10000") int questionnaireMaxNumberOfResources,
            @DefaultValue("10000") int questionnaireResponseMaxNumberOfResources,
            @DefaultValue("http://questionnaireservice:8080") String questionnaireServiceUrl,
            @DefaultValue("100") int questionnaireRequestCreationBatchSize,
            @DefaultValue("500") int questionnaireRequestCreationBufferTimeoutMillis,
            @DefaultValue("500") int questionnaireRequestOperationMaxNumberOfTargets
    ) {
        this.url = url;
        this.experimental = experimental;
        this.observationMaxNumberOfResources = observationMaxNumberOfResources;
        this.appointmentMaxNumberOfResources = appointmentMaxNumberOfResources;
        this.communicationMaxNumberOfResources = communicationMaxNumberOfResources;
        this.diagnosticReportMaxNumberOfResources = diagnosticReportMaxNumberOfResources;
        this.encounterMaxNumberOfResources = encounterMaxNumberOfResources;
        this.documentReferenceMaxNumberOfResources = documentReferenceMaxNumberOfResources;
        this.questionnaireMaxNumberOfResources = questionnaireMaxNumberOfResources;
        this.questionnaireResponseMaxNumberOfResources = questionnaireResponseMaxNumberOfResources;
        this.questionnaireServiceUrl = questionnaireServiceUrl;
        this.questionnaireRequestCreationBatchSize = questionnaireRequestCreationBatchSize;
        this.questionnaireRequestCreationBufferTimeoutMillis = questionnaireRequestCreationBufferTimeoutMillis;
        this.questionnaireRequestOperationMaxNumberOfTargets = questionnaireRequestOperationMaxNumberOfTargets;
    }

    public String getUrl() {
        return url;
    }

    public boolean isExperimental() {
        return experimental;
    }

    public int getObservationMaxNumberOfResources() {
        return observationMaxNumberOfResources;
    }

    public int getAppointmentMaxNumberOfResources() {
        return appointmentMaxNumberOfResources;
    }

    public int getCommunicationMaxNumberOfResources() {
        return communicationMaxNumberOfResources;
    }

    public int getDiagnosticReportMaxNumberOfResources() {
        return diagnosticReportMaxNumberOfResources;
    }

    public int getEncounterMaxNumberOfResources() {
        return encounterMaxNumberOfResources;
    }

    public int getDocumentReferenceMaxNumberOfResources() {
        return documentReferenceMaxNumberOfResources;
    }

    public int getQuestionnaireMaxNumberOfResources() {
        return questionnaireMaxNumberOfResources;
    }

    public int getQuestionnaireResponseMaxNumberOfResources() {
        return questionnaireResponseMaxNumberOfResources;
    }

    public String getQuestionnaireServiceUrl() {
        return questionnaireServiceUrl;
    }
    
    public int getQuestionnaireRequestCreationBatchSize() {
        return questionnaireRequestCreationBatchSize;
    }
    
    public int getQuestionnaireRequestOperationMaxNumberOfTargets(){
        return questionnaireRequestOperationMaxNumberOfTargets;
    }
}
