package com.pkb.api.fhir.config;

import ca.uhn.fhir.context.FhirContext;
import ca.uhn.fhir.parser.IParser;
import ca.uhn.fhir.rest.server.HardcodedServerAddressStrategy;
import ca.uhn.fhir.rest.server.IServerAddressStrategy;
import com.pkb.api.fhir.FhirPatientCreationService;
import com.pkb.api.fhir.FhirPatientHL7PatientMapper;
import com.pkb.api.fhir.FhirPatientResourceService;
import com.pkb.api.fhir.PhrAuthenticationExtractingInterceptor;
import com.pkb.api.fhir.PhrFhirAuthenticationService;
import com.pkb.api.fhir.PhrPatientResourceProvider;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.oauth2.client.Oauth2Client;
import com.pkb.repository.FhirPatientResourceRepository;
import com.pkb.service.dataupload.hl7.HL7ApiHelper;
import com.pkb.service.dataupload.hl7.HL7TeamService;
import com.pkb.service.dataupload.impl.DataUploadManager;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.team.TeamService;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.user.impl.UserManager;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import static ca.uhn.fhir.context.FhirContext.forDstu3;

@Configuration
public class PhrFhirApiConfig {

    @Bean
    public PhrPatientResourceProvider phrPatientResourceProvider(FhirPatientCreationService fhirPatientCreationService,
                                                                 PhrFhirAuthenticationService phrFhirAuthenticationService) {
        return new PhrPatientResourceProvider(fhirPatientCreationService, phrFhirAuthenticationService);
    }

    @Bean
    public PhrFhirAuthenticationService phrFhirAuthentication(Oauth2Client oauth2Client) {
        return new PhrFhirAuthenticationService(oauth2Client);
    }

    @Bean
    public PhrAuthenticationExtractingInterceptor phrAuthenticationExtractingInterceptor() {
        return new PhrAuthenticationExtractingInterceptor();
    }

    @Bean
    public FhirContext fhirContext() {
        return forDstu3();
    }

    @Bean
    public IServerAddressStrategy serverAddressStrategy(PhrConfig config) {
        return new HardcodedServerAddressStrategy(config.getBaseURL() + config.phrFhirApiUrl());
    }

    @Bean
    public FhirPatientHL7PatientMapper fhirPatientHL7PatientMapper(DateTimeService dateTimeService, PhrConfig phrConfig) {
        return new FhirPatientHL7PatientMapper(dateTimeService);
    }

    @Bean
    public IParser fhirDstu3Parser() {
        return FhirContext.forDstu3().newJsonParser();
    }

    @Bean
    public FhirPatientCreationService fhirPatientCreationService (HL7TeamService hl7TeamService, FhirPatientHL7PatientMapper fhirPatientHL7PatientMapper,
                                                                  UserManager userManager, TeamUserManager teamUserManager,
                                                                  PatientConsentManager patientConsentManager, TeamService teamService,
                                                                  HL7ApiHelper apiHelper, DataUploadManager dataUploadManager,
                                                                  DateTimeService dateTimeService, IParser fhirDstu3Parser,
                                                                  FhirPatientResourceService fhirPatientResourceService) {
        return new FhirPatientCreationService(hl7TeamService, fhirPatientHL7PatientMapper, dateTimeService, userManager,
                teamUserManager, patientConsentManager, teamService, apiHelper, dataUploadManager, fhirDstu3Parser, fhirPatientResourceService);
    }

    @Bean
    public FhirPatientResourceService fhirPatientResourceService(FhirPatientResourceRepository fhirPatientResourceRepository) {
        return new FhirPatientResourceService(fhirPatientResourceRepository);
    }
}
