package com.pkb.api.fhir.config;

import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.boot.context.properties.ConstructorBinding;
import org.springframework.boot.context.properties.bind.DefaultValue;
import org.springframework.cloud.context.config.annotation.RefreshScope;

@RefreshScope
@ConstructorBinding
@ConfigurationProperties(prefix = "oauth2")
public class OAuth2Properties {

    private final String baseUrl;
    private final String phrClientUsername;
    private final String phrClientPassword;
    private final int phrClientTokenCacheExpireMinutes;

    public OAuth2Properties(
        @DefaultValue("http://oauth2-endpoint:8080") String baseUrl,
        @DefaultValue("f60604fa-9b40-461b-ae7d-0140d68f2fe4") String phrClientUsername,
        @DefaultValue("pass") String phrClientPassword,
        @DefaultValue("20") int phrClientTokenCacheExpireMinutes
    ) {

        this.baseUrl = baseUrl;
        this.phrClientUsername = phrClientUsername;
        this.phrClientPassword = phrClientPassword;
        this.phrClientTokenCacheExpireMinutes = phrClientTokenCacheExpireMinutes;
    }

    public String getBaseUrl() {
        return baseUrl;
    }

    public String getPhrClientUsername() {
        return phrClientUsername;
    }

    public String getPhrClientPassword() {
        return phrClientPassword;
    }

    public int getPhrClientTokenCacheExpireMinutes() {
        return phrClientTokenCacheExpireMinutes;
    }
}
