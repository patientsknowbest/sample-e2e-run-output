package com.pkb.questionnaireservice.service

import com.pkb.entities.enums.Route
import com.pkb.questionnaireservice.QuestionnaireId
import com.pkb.questionnaireservice.QuestionnaireRequestId
import com.pkb.questionnaireservice.entities.QuestionnaireRequestEntity
import com.pkb.questionnaireservice.mapper.mapRepositoryErrorOnSave
import com.pkb.questionnaireservice.mapper.mapResultsPageToSearchResults
import com.pkb.questionnaireservice.modelv2.QuestionnaireRequest
import com.pkb.questionnaireservice.modelv2.QuestionnaireServiceError
import com.pkb.questionnaireservice.modelv2.SearchResults
import com.pkb.questionnaireservice.repositories.ques.QuestionnaireRequestRepository
import io.vavr.control.Either
import io.vavr.control.Either.left
import io.vavr.control.Either.right
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.data.domain.Pageable
import org.springframework.stereotype.Service
import org.springframework.transaction.annotation.Transactional
import java.lang.invoke.MethodHandles

@Service
class QuestionnaireRequestService(
        private val questionnaireRequestRepository: QuestionnaireRequestRepository
) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    fun post(questionnaireRequest: QuestionnaireRequestEntity): Either<QuestionnaireServiceError, QuestionnaireRequestEntity> =
            if (questionnaireRequest.route != Route.FHIR_API && questionnaireRequest.route != Route.WEBAPP)
                // Only FHIR_API and WEBAPP are supported because they are the only two routes available to create questionnaire requests,
                // and the corresponding enum in the questionnaire service database will reject any other value
                // We have to make the distinction here because the error coming back from the database is a DataIntegrityViolation
                // and we can't distinguish between this case and a duplicate key exception
                Either.left(QuestionnaireServiceError.UnexpectedError("Route value ${questionnaireRequest.route} not supported"))
            else
                kotlin.runCatching {
                    saveQuestionnaireRequestIdentifierAndQuestionnaireRequest(questionnaireRequest)
                }.fold({ Either.right(it) },
                        { Either.left(mapRepositoryErrorOnSave(it, questionnaireRequest)) })

    fun postBatch(questionnaireRequests: List<QuestionnaireRequestEntity>): Either<QuestionnaireServiceError,List<QuestionnaireRequestEntity>> =
            kotlin.runCatching {
                questionnaireRequestRepository.saveAll(questionnaireRequests)
        }.fold({ right(it.toList().sortedWith(compareBy(QuestionnaireRequest::createdDate).thenBy(QuestionnaireRequest::patientId))) },
                   { left(QuestionnaireServiceError.UnexpectedError(it.message!!)) })

    fun get(id: QuestionnaireRequestId): Either<QuestionnaireServiceError, QuestionnaireRequestEntity> =
            questionnaireRequestRepository.findById(id)
                    .map { Either.right<QuestionnaireServiceError, QuestionnaireRequestEntity>(it) }
                    .orElseGet { Either.left(QuestionnaireServiceError.NotFoundError("QuestionnaireRequest with id $id not found")) }

    fun getByQuestionnaireIds(questionnaireIds: List<QuestionnaireId>, pageable: Pageable): Either<QuestionnaireServiceError, SearchResults<QuestionnaireRequestEntity>> =
            Either.right(mapResultsPageToSearchResults(questionnaireRequestRepository.findByQuestionnaireIds(questionnaireIds, pageable)))

    fun getNonResponders(
            questionnaireId: QuestionnaireId,
            pageable: Pageable
    ): Either<QuestionnaireServiceError, List<QuestionnaireRequestEntity>> {
        log.debug("Getting non-responders for questionnaire: {}", questionnaireId)
        val result = Either.right<QuestionnaireServiceError, List<QuestionnaireRequestEntity>>(
                questionnaireRequestRepository.getNonRespondersForQuestionnaires(listOf(questionnaireId))
        )
        log.debug("Finished getting non-responders for questionnaire: {}", questionnaireId)
        return result
    }

    @Transactional
    fun saveQuestionnaireRequestIdentifierAndQuestionnaireRequest(questRequest: QuestionnaireRequestEntity): QuestionnaireRequestEntity {
        return questionnaireRequestRepository.save(questRequest)
    }
}
