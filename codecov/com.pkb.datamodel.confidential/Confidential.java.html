package com.pkb.datamodel.confidential;

import org.apache.commons.lang3.StringUtils;
import org.immutables.value.Value;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.invoke.MethodHandles;

public abstract class Confidential<T> extends Wrapper<T> {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    /**
     * @deprecated Use {@link Confidential#value() instead}
     */
    @SuppressWarnings("DeprecatedIsStillUsed")
    @Value.Parameter
    @Deprecated
    public abstract T _value();

    /**
     * @return The confidential value
     */
    @Override
    public T value() {
        if (LOGGER.isDebugEnabled()) {
            LOGGER.debug("Unwrapping {} value from callsite {}", getClass().getSimpleName(),
                    new Throwable().getStackTrace()[1]);
        }

        //noinspection deprecation
        return _value();
    }

    @Override
    public String toString() {
        return getClass().getSimpleName() + "(<CONFIDENTIAL>)";
    }
}


@Value.Immutable
@Wrapped
abstract class _SensitiveString extends Confidential<String> {

    public boolean isEqualCaseInsensitive(SensitiveString other) {
        return StringUtils.equalsIgnoreCase(value(), other.value());
    }
}
