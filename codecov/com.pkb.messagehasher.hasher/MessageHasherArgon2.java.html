package com.pkb.messagehasher.hasher;

import com.pkb.messagehasher.MessageHasher;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.nio.charset.Charset;

@Service
public class MessageHasherArgon2 implements MessageHasher {

    static byte[] HASH_SALT_256 = "v02pkevUPsURTEy43XB6L43OC37qDVamplE1huI62fMZifSa8VvweCWU8IhSc93q43u9pAZalZADyb8Y6gJCNYg8KRd3RHjtiIzNdXi5OMWLZuzqxHK5VhSyIzz6TALOodjjwkvTMCcgvIpq2X2Vk18bS19MH8GXnA3osdbcGUqdq9abs8BQSWP3xpfdtoWa1WvfVVEUm8nWGZ84GK4JNDidi6IS0aOSb7GgfS3uMsvaA8e4wvylPESZPgkIIcOW".getBytes();

    @Autowired
    Argon2Advanced argon2;

    @Autowired
    HashConfig hashConfig;

    public MessageHasherArgon2() {
    }

    int suggestIterations() {
        return Argon2Helper.findIterations(
                argon2,
                hashConfig.getIterationFinderMaxMillis(),
                hashConfig.getMemoryUsage(),
                hashConfig.getParallelism()
        );
    }

    @Override
    public String generateHash(String messageContent) {
        return argon2.hash(
                hashConfig.getIterations(),
                hashConfig.getMemoryUsage(),
                hashConfig.getParallelism(),
                messageContent.toCharArray(),
                Charset.forName("UTF-8"),
                HASH_SALT_256
        );
    }
}
