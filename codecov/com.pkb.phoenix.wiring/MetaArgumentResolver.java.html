/**
 * This class makes it possible to declare resolvable {@link Meta} parameters to request controller methods.
 *
 * <pre>
 * <code>
 *
 *  {@literal @}RestController
 *  {@literal @}RequestMapping(value = "/phoenix/task")
 *   public class TaskController {
 *    {@literal @}PostMapping
 *     public DetailedTask createTask(@Valid @RequestBody CreateTask payload, Meta meta) {
 *       // ...
 *     }
 *   }
 * </code>
 * </pre>
 */
public class MetaArgumentResolver implements HandlerMethodArgumentResolver {
    public static class MetaArgumentResolverConfigurer implements WebMvcConfigurer {
        private final ObjectFactory<HttpSession> httpSessionFactory;
        private final String resource;
        private final UserService userService;

        public MetaArgumentResolverConfigurer(ObjectFactory<HttpSession> httpSessionFactory, String resource, UserService userService) {
            this.httpSessionFactory = httpSessionFactory;
            this.resource = resource;
            this.userService = userService;
        }

        @Override
        public void addArgumentResolvers(
                List<HandlerMethodArgumentResolver> resolvers) {
            resolvers.add(new MetaArgumentResolver(httpSessionFactory, resource, userService));
        }
    }

    private final ObjectFactory<HttpSession> httpSessionFactory;
    private final String resource;
    private final UserService userService;

    public MetaArgumentResolver(ObjectFactory<HttpSession> httpSessionFactory, String resource, UserService userService) {
        this.httpSessionFactory = httpSessionFactory;
        this.resource = resource;
        this.userService = userService;
    }

    @Override
    public boolean supportsParameter(MethodParameter parameter) {
        return parameter.getParameter().getType() == Meta.class;
    }

    @Override
    public Object resolveArgument(@NotNull MethodParameter parameter,
            ModelAndViewContainer mavContainer,
            @NotNull NativeWebRequest webRequest,
            WebDataBinderFactory binderFactory) throws Exception {

        var sessionId = ensureSessionIdExistsAndGetIt();
        var token = getToken();
        var keycloakId = ImmutableKeycloakId.of(token.getSubject());
        var roles = token.getResourceAccess().get(resource).getRoles()
                .stream()
                .map(role -> Match(role).of(
                        Case($(IG.code()), INFORMATION_GOVERNOR),
                        Case($(CS.code()), CUSTOMER_SUPPORT_SPECIALIST)))
                .collect(toUnmodifiableSet());

        var username = token.getName();

        var user = userService.findUserByKeycloakId(keycloakId)
                .map(registeredUser -> {
                    if (!registeredUser.name().equalsIgnoreCase(username)) {
                        return userService.updateExistingUser(registeredUser.internalId(), username, roles);
                    }
                    return registeredUser;
                })
                .orElseGet(() -> userService.createNewUser(username, keycloakId, roles));
        return meta(session(sessionId, user(user.internalId(), keycloakId.value(), user.name(), roles, user.fdApiKey())), randomUUID());
    }

    private UUID ensureSessionIdExistsAndGetIt() {
        var session = httpSessionFactory.getObject();
        var attribute = session.getAttribute("sid-phoenix");

        if (attribute == null) {
            session.setAttribute("sid-phoenix", randomUUID());
            attribute = session.getAttribute("sid-phoenix");
        }
        return UUID.fromString(attribute.toString());
    }

    @SuppressWarnings("unchecked")
    private AccessToken getToken() {
        //noinspection unchecked
        var keycloakPrincipal = (KeycloakPrincipal<RefreshableKeycloakSecurityContext>) SecurityContextHolder.getContext().getAuthentication().getPrincipal();

        return keycloakPrincipal.getKeycloakSecurityContext().getToken();
    }
}
