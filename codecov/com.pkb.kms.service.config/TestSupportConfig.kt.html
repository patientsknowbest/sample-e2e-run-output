package com.pkb.kms.service.config

import com.pkb.common.ClearableInternalState
import com.pkb.common.config.ConfigStorage
import com.pkb.common.config.KmsConfig
import com.pkb.common.datetime.DateTimeService
import com.pkb.common.testcontrol.config.TestControlServiceConfig
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import com.pkb.common.testlogging.DefaultDetailLoggingProvider
import com.pkb.common.testlogging.DetailLoggingProvider
import com.pkb.kms.service.testsupport.StateClearerHelper
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import org.springframework.context.annotation.Profile
import java.util.Optional


@Configuration
@Profile("test", "dev")
class TestSupportConfig {
    @Bean
    fun testLoggingService(): DetailLoggingProvider {
        return DefaultDetailLoggingProvider.INSTANCE
    }

    @Bean
    fun getTestSupportServiceConfig(config: KmsConfig, dateTimeService: DateTimeService,
                                    configStorage: ConfigStorage,
                                    stateClearerHelper: StateClearerHelper,
                                    testLoggingService: DetailLoggingProvider,
                                    pubSubNamespaceService: PubSubNamespaceService): TestControlServiceConfig {
        return TestControlServiceConfig(config.isTestControlEnabled,
                config.isTestControlRegistrationEnabled,
                config.testControlAppName,
                config.testControlCallbackUrl,
                config.testControlUrl,
                dateTimeService,
                configStorage,
                setOf<ClearableInternalState>(stateClearerHelper),
                testLoggingService,
                pubSubNamespaceService,
                Optional.empty())
    }
}
