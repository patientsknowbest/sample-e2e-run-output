package com.pkb.util.metrics;

import io.prometheus.client.Counter;

public class SpaMetrics {

    private static final Counter COUNTER = Counter.build()
            .name("single_page_application")
            .help("Single page application")
            .labelNames("module", "action")
            .register();


    public void patientBannerCalled() {
        COUNTER.labels("patient_banner", "called").inc();
    }
}
