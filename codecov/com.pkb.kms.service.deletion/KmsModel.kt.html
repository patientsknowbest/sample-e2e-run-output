package com.pkb.kms.service.deletion

import com.pkb.deletion.model.DatabaseTable
import com.pkb.deletion.model.ImmutablePkColumn
import com.pkb.deletion.model.PkColumn
import com.pkb.deletion.model.StandardInboundForeignKeyReference.owningFk
import com.pkb.kms.shared.representation.PersonPublicId


class KmsModel {

    companion object {
        private val NHS_LOGIN_CREDENTIALS_CHANGE_HISTORY: DatabaseTable<Long> = DatabaseTable.table("kms.nhs_login_credentials_change_history")
        private val NHS_LOGIN_CREDENTIALS_CURRENT: DatabaseTable<Long> = DatabaseTable.builderForLong().tableName("kms.nhs_login_credentials_current").pkColumn(
            ImmutablePkColumn.builder<Long>().columnTypeClass(
                Long::class.javaObjectType
            ).name("nhs_login_credentials_id").build()
        ).build()

        private val NHS_LOGIN_CREDENTIALS: DatabaseTable<Long> = DatabaseTable.table(
            "kms.nhs_login_credentials", listOf(
                owningFk(NHS_LOGIN_CREDENTIALS_CURRENT, "nhs_login_credentials_id"),
                owningFk(NHS_LOGIN_CREDENTIALS_CHANGE_HISTORY, "nhs_login_credentials_id")
            )
        )

        val PERSON: DatabaseTable<PersonPublicId> = DatabaseTable.builderFor(PersonPublicId::class.java).tableName("pkbperson")
            .pkColumn(
                PkColumn.pk(
                    PersonPublicId::class.java, listOf(
                        owningFk(NHS_LOGIN_CREDENTIALS, "person_public_id")
                    )
                )
            )
            .isVirtual(true)
            .build()
    }
}
