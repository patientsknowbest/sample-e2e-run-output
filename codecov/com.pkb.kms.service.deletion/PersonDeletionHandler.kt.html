package com.pkb.kms.service.deletion

import com.pkb.deletion.model.DatabaseTable
import com.pkb.deletion.model.DeleteOperationMessage
import com.pkb.deletion.model.DeleteOperationRequest
import com.pkb.deletion.model.DeletionRules
import com.pkb.deletion.model.DeletionType
import com.pkb.deletion.model.DeletionValidator
import com.pkb.deletion.model.EntityDeletionRequest
import com.pkb.deletion.model.ImmutableEntityDeletionRequest
import com.pkb.deletion.model.errors.DeletionServiceError
import com.pkb.deletion.service.AbstractDatabaseModelDeletionHandler
import com.pkb.deletion.service.DatabaseModelBasedDeletionService
import com.pkb.kms.shared.representation.PersonPublicId
import io.vavr.control.Either
import org.springframework.context.annotation.Profile
import org.springframework.stereotype.Component


@Component
@Profile("!unittest & !integration-test")
class PersonDeletionHandler(
    deletionService: DatabaseModelBasedDeletionService
) : AbstractDatabaseModelDeletionHandler<PersonPublicId>(deletionService, DeletionType.PERSON) {
    override fun getDeletionRules(entityDeletionRequest: EntityDeletionRequest<PersonPublicId>): DeletionRules {
        return object : DeletionRules {
            override fun <T> getValidator(table: DatabaseTable<T>): DeletionValidator<T> {
                return DeletionValidator.acceptAll(table)
            }
        }
    }

    override fun getEntityDeletionRequest(originalRequest: DeleteOperationRequest): Either<DeletionServiceError, EntityDeletionRequest<PersonPublicId>> =
        Either.right(
            ImmutableEntityDeletionRequest.builder<PersonPublicId>().from(originalRequest as DeleteOperationMessage)
            .rootEntityIds(originalRequest.entityIdMap.values)
            .rootTable(KmsModel.PERSON)
            .build())

}
