package com.pkb.service.file;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.File;
import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.IOException;

/**
 * Read from the given temp file, and delete it when done.
 *
 * @author robwhelan
 */
public class TempFileInputStream extends FileInputStream {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private File file;

    /**
     * Auto-cleanup (delete) file when it's fully read in.
     *
     * @param tempFile
     */
    public TempFileInputStream(File tempFile) throws FileNotFoundException {
        super(tempFile);
        file = tempFile;
    }

    public TempFileInputStream(String name) throws FileNotFoundException {
        this(new File(name));
    }

    @Override
    public void close() throws IOException {
        super.close();

        LOGGER.info("closing temp stream, so deleting file {}", file.getAbsolutePath());
        // try to delete the file after closing it
        try {
            if (!file.delete()) {
                LOGGER.error("couldn't delete up temp file after conversion");
            }
        } catch (Exception e) {
            LOGGER.error("non-failing exception: couldn't clean up temp file after conversion", e);
        }
    }
}
