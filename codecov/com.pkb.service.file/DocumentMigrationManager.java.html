package com.pkb.service.file;


import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.crypto.document.Document;
import com.pkb.entities.doc.DocDocumentMetadata;
import com.pkb.entities.enums.DocStorageBackend;
import com.pkb.service.document.DocumentManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import io.micrometer.core.instrument.Counter;
import io.micrometer.core.instrument.Metrics;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import org.hibernate.query.NativeQuery;
import org.hibernate.type.PostgresUUIDType;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;

import javax.persistence.EntityManager;
import javax.persistence.PersistenceContext;
import java.lang.invoke.MethodHandles;
import java.util.Collection;
import java.util.EnumMap;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Set;
import java.util.UUID;
import java.util.concurrent.CompletableFuture;
import java.util.concurrent.atomic.AtomicBoolean;
import java.util.function.Function;
import java.util.stream.Collectors;

import static java.util.stream.Collectors.groupingBy;
import static java.util.stream.Collectors.mapping;
import static java.util.stream.Collectors.toList;
import static java.util.stream.Collectors.toSet;

public class DocumentMigrationManager extends TransactionManager {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
    private static final AtomicBoolean MIGRATION_RUNNING = new AtomicBoolean(false);
    private static final String MIGRATION_SUCCESS_COUNTER_NAME = "phr_gcs_migration_success_counter";
    private static final String MIGRATION_ERROR_COUNTER_NAME = "phr_gcs_migration_error_counter";

    private static final Counter MIGRATION_SUCCESS_COUNTER = Counter.builder(MIGRATION_SUCCESS_COUNTER_NAME)
            .description("Successful migrations")
            .register(Metrics.globalRegistry);

    private static final Counter MIGRATION_ERROR_COUNTER = Counter.builder(MIGRATION_ERROR_COUNTER_NAME)
            .description("Failed migrations")
            .register(Metrics.globalRegistry);

    @PersistenceContext
    private EntityManager em;

    private final ThreadPoolTaskExecutor taskExecutor;
    private final DocumentManager documentManager;

    public DocumentMigrationManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService,
                                    UUIDProvider uuidProvider, ThreadPoolTaskExecutor taskExecutor, DocumentManager documentManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.taskExecutor = taskExecutor;
        this.documentManager = documentManager;
    }

    public void migrateDocument(Collection<UUID> ids) {
        if (ids.isEmpty()) {
            return;
        }

        if (MIGRATION_RUNNING.compareAndSet(false, true)) {
            try {
                List<UUID> filteredIds = em.createNativeQuery("SELECT d.id FROM doc.document_metadata d" +
                                " WHERE d.storage <> 'GCS'\\:\\:doc_storage_backend_type" +
                                " AND d.id IN (:ids)")
                        .setParameter("ids", ids)
                        .unwrap(NativeQuery.class)
                        .addScalar("id", PostgresUUIDType.INSTANCE)
                        .getResultList();

                Map<UUID, DocDocumentMetadata> metadataMap = em.createQuery("SELECT dm FROM DocDocumentMetadata dm WHERE dm.id IN (:ids)", DocDocumentMetadata.class)
                        .setParameter("ids", ids).getResultList().stream().collect(Collectors.toMap(DocDocumentMetadata::getId, Function.identity()));

                Map<UUID, String> dmIdToNewGcsId = parallelUploadToGCS(filteredIds);

                try {
                    batchMigrateDocuments(metadataMap, dmIdToNewGcsId);
                } catch (Exception e) {
                    if (!config.isBatchGcsMigrationFallbackEnabled()) {
                        LOGGER.error("Error while batch migrating document to GCS, fallback is disabled", e);
                        throw e;
                    }
                    LOGGER.error("Error while batch migrating document to GCS, switching to one-by-one migration for this batch", e);
                    migrateDocumentsOneByOne(metadataMap, dmIdToNewGcsId);
                }
            } finally {
                MIGRATION_RUNNING.set(false);
            }
        } else {
            LOGGER.info("Migrator already running, quit flow.");
        }
    }

    private Map<UUID, String> parallelUploadToGCS(List<UUID> filteredIds) {
        Map<UUID, Document> documentMap = documentManager.retrieve(filteredIds);

        List<Tuple2<UUID, CompletableFuture<String>>> futures = filteredIds.stream().map(id -> {
            CompletableFuture<String> future = taskExecutor.submitListenable(() -> {
                Document document = documentMap.get(id);
                return documentManager.createInGcs(document);
            }).completable();
            return Tuple.of(id, future);
        }).collect(toList());

        Map<UUID, String> dmIdToNewGcsId = new HashMap<>();
        futures.forEach(f -> {
            try {
                dmIdToNewGcsId.put(f._1, f._2.get());
            } catch (Exception e) {
                LOGGER.error("Error while executing thread to upload document to GCS: {}", f._1, e);
            }
        });

        return dmIdToNewGcsId;
    }

    private void batchMigrateDocuments(Map<UUID, DocDocumentMetadata> metadataMap, Map<UUID, String> dmIdToNewGcsId) {
        transactional(() -> {
            Map<DocStorageBackend, List<String>> idsToDelete = dmIdToNewGcsId.entrySet()
                    .stream()
                    .collect(groupingBy(k -> metadataMap.get(k.getKey()).getStorage(), () -> new EnumMap<>(DocStorageBackend.class), mapping(e -> metadataMap.get(e.getKey()).getIdInStorage(), toList())));

            StringBuilder batchUpdateQueryBuilder = new StringBuilder(100 + dmIdToNewGcsId.size() * 85);
            batchUpdateQueryBuilder.append("update doc.document_metadata AS dm ");
            batchUpdateQueryBuilder.append("SET storage = 'GCS'\\:\\:doc_storage_backend_type, ");
            batchUpdateQueryBuilder.append("    id_in_storage = c.new_id ");
            batchUpdateQueryBuilder.append("FROM ( VALUES");
            dmIdToNewGcsId.forEach((dmId, newId) -> batchUpdateQueryBuilder.append("('").append(newId).append("',").append("'").append(dmId).append("'\\:\\:uuid),"));
            if (batchUpdateQueryBuilder.charAt(batchUpdateQueryBuilder.length() - 1) == ',') {
                batchUpdateQueryBuilder.deleteCharAt(batchUpdateQueryBuilder.length() - 1);
            }
            batchUpdateQueryBuilder.append(") AS c(new_id,old_id) ");
            batchUpdateQueryBuilder.append("WHERE c.old_id = dm.id");

            int updated = em.createNativeQuery(batchUpdateQueryBuilder.toString()).executeUpdate();

            idsToDelete.keySet().forEach(storage -> {
                switch (storage) {
                    case GCS_FALLBACK:
                        Set<UUID> fallbackStoreIds = idsToDelete.get(storage).stream().map(s -> UUID.fromString(s)).collect(toSet());
                        em.createNativeQuery("DELETE FROM doc.gcs_fallback WHERE id IN (:ids)")
                                .setParameter("ids", fallbackStoreIds)
                                .executeUpdate();
                        break;
                    default:
                        throw new IllegalStateException(String.format("Documents (%s) were sent for migration, but not in a supported store: %s", idsToDelete.get(storage).stream().map(l -> l.toString()).collect(Collectors.joining(",")), storage));
                }
            });
            if (config.isBatchGcsMigrationTestFailEnabled()) {
                // Run bogus query
                em.createNativeQuery("DELETE FROM xy").executeUpdate();
            }
            Metrics.counter(MIGRATION_SUCCESS_COUNTER_NAME).increment(updated);
        });
    }

    private void migrateDocumentsOneByOne(Map<UUID, DocDocumentMetadata> metadataMap, Map<UUID, String> dmIdToNewGcsId) {
        dmIdToNewGcsId.forEach((id, newInternalId) -> {
            try {
                DocDocumentMetadata metadata = metadataMap.get(id);

                transactional(() -> {
                    em.createNativeQuery("UPDATE doc.document_metadata SET storage = 'GCS'\\:\\:doc_storage_backend_type, id_in_storage = :newId WHERE id = :id")
                            .setParameter("newId", newInternalId)
                            .setParameter("id", id)
                            .unwrap(NativeQuery.class)
                            .addScalar("id", PostgresUUIDType.INSTANCE)
                            .executeUpdate();

                    switch (metadata.getStorage()) {
                        case GCS_FALLBACK:
                            UUID fallbackStoreId = UUID.fromString(metadata.getIdInStorage());
                            em.createNativeQuery("DELETE FROM doc.gcs_fallback WHERE id = :id")
                                    .setParameter("id", fallbackStoreId)
                                    .executeUpdate();
                            break;
                        default:
                            throw new IllegalStateException(String.format("Document (%s) was sent for migration, but it's not in a supported store: %s", id.toString(), metadata.getStorage().toString()));
                    }

                    if (config.isGcsMigrationTestFailEnabled()) {
                        // Run bogus query
                        em.createNativeQuery("DELETE FROM xy").executeUpdate();
                    }

                    Metrics.counter(MIGRATION_SUCCESS_COUNTER_NAME).increment();
                });
            } catch (RuntimeException e) {
                LOGGER.error("Error while migrating document to GCS: {}", id, e);
                Metrics.counter(MIGRATION_ERROR_COUNTER_NAME).increment();

                try {
                    documentManager.deleteFromGcs(newInternalId);
                } catch (RuntimeException e2) {
                    LOGGER.error("Error while cleaning up document from GCS: {}", newInternalId, e2);
                }
            }
        });
    }
}
