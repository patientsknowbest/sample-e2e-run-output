package com.pkb.service.file;


import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.entities.doc.DocGcsDeleteQueue;
import com.pkb.repository.legacy.LegacyDocGcsDeleteQueueRepository;
import com.pkb.service.document.DocumentManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import io.micrometer.core.instrument.Metrics;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import io.vavr.control.Either;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.data.domain.Pageable;

import java.lang.invoke.MethodHandles;
import java.time.Instant;
import java.time.temporal.ChronoUnit;
import java.util.Collections;
import java.util.HashSet;
import java.util.List;
import java.util.Map;
import java.util.Set;
import java.util.concurrent.atomic.AtomicBoolean;
import java.util.function.Function;
import java.util.stream.Collectors;

public class GcsDocumentDeletionManager extends TransactionManager {
    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
    private static final AtomicBoolean DELETION_RUNNING = new AtomicBoolean(false);
    private static final String DELETE_SUCCESS_COUNTER_NAME = "phr_gcs_deletion_success_counter";
    private static final String DELETE_ERROR_COUNTER_NAME = "phr_gcs_deletion_error_counter";

    private final DocumentManager documentManager;
    private final LegacyDocGcsDeleteQueueRepository docGcsDeleteQueueRepository;

    public GcsDocumentDeletionManager(TolvenBeanFactory beanFactory,
                                      DateTimeService dateTimeService,
                                      UUIDProvider uuidProvider,
                                      PhrConfig config,
                                      DocumentManager documentManager,
                                      LegacyDocGcsDeleteQueueRepository docGcsDeleteQueueRepository) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.documentManager = documentManager;
        this.docGcsDeleteQueueRepository = docGcsDeleteQueueRepository;
    }

    public void deleteGcsDocuments() {
        if (DELETION_RUNNING.compareAndSet(false, true)) {
            try {
                Instant endDate = dateTimeService.now().minus(config.getDocumentGcsDeleteKeepForDays(), ChronoUnit.DAYS);
                int pageSize = config.getDocumentGcsDeleteBatchSize();
                int lastMaxId = 0;

                List<DocGcsDeleteQueue> queue = docGcsDeleteQueueRepository.getDeleteQueueBatch(endDate, lastMaxId, Pageable.ofSize(pageSize));

                while (!queue.isEmpty()) {
                    Map<String, DocGcsDeleteQueue> idMap = queue.stream().collect(Collectors.toMap(DocGcsDeleteQueue::getIdInStore, Function.identity()));
                    Set<String> internalIds = new HashSet<>(idMap.keySet());
                    LOGGER.info("Deleting documents from GCS: {}", internalIds);

                    List<Long> successfulQueueIds;
                    Map<String, String> failures;
                    try {
                        Map<String, Either<String, Boolean>> results = documentManager.deleteBlobsFromGcs(internalIds);

                        successfulQueueIds = results.entrySet().stream()
                                // we only care that we got a Boolean result
                                // false value means the document didn't exist, which we'll handle as a success
                                .filter(e -> e.getValue().isRight())
                                .map(e -> idMap.get(e.getKey()).getId())
                                .collect(Collectors.toList());

                        failures = results.entrySet().stream()
                                .filter(e -> e.getValue().isLeft())
                                .collect(Collectors.toMap(Map.Entry::getKey, e -> e.getValue().getLeft()));
                    } catch (Exception e) {
                        LOGGER.error("Error when calling GCS to delete documents", e);
                        successfulQueueIds = Collections.emptyList();
                        failures = queue.stream()
                                .map(entry -> Tuple.of(entry.getIdInStore(), ""))
                                .collect(Collectors.toMap(Tuple2::_1, Tuple2::_2));
                    }

                    Metrics.counter(DELETE_SUCCESS_COUNTER_NAME).increment(successfulQueueIds.size());
                    Metrics.counter(DELETE_ERROR_COUNTER_NAME).increment(failures.size());

                    if (!failures.isEmpty()) {
                        LOGGER.error("Failed to delete documents from GCS. Errors: {}",
                                failures.entrySet().stream().map(e -> e.getKey() + ": " + e.getValue()).collect(Collectors.toList()));
                    }

                    if (!successfulQueueIds.isEmpty()) {
                        List<Long> finalSuccessfulQueueIds = successfulQueueIds;
                        transactional(() -> {
                            try {
                                docGcsDeleteQueueRepository.deleteAllById(finalSuccessfulQueueIds);
                            } catch (Exception e) {
                                LOGGER.error("Error deleting items from GCS deletion queue: {}. Details: ", finalSuccessfulQueueIds, e);
                            }
                        });
                    }
                    queue = docGcsDeleteQueueRepository.getDeleteQueueBatch(endDate, queue.get(queue.size() - 1).getId(), Pageable.ofSize(pageSize));
                }
            } finally {
                DELETION_RUNNING.set(false);
            }
        } else {
            LOGGER.info("GCS deletion already running, quit flow.");
        }
    }
}
