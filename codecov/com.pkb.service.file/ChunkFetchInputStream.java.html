package com.pkb.service.file;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.document.entity.ChunkedDocDTO;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.IOException;
import java.io.InputStream;
import java.util.LinkedList;
import java.util.List;
import java.util.UUID;

import static java.util.Objects.requireNonNull;

/**
 * Only read each chunk from Tolven as it's needed, to send down via the response.
 * The buffer is initialized with the top-level file
 * The fileChunks queue
 *
 * @author robwhelan
 */
public class ChunkFetchInputStream extends InputStream {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private ChunkedDocManager chunkedDocManager;
    private LoggedInEHRRequestContext requestContext;

    private UUID baseFileId; // just for logging in case of error
    private int chunkSize; // for skip impl

    private LinkedList<UUID> nextChunkQueue;
    private byte[] buffer;
    private int length;
    private int pos;


    /**
     *
     * @param chunkedDocManager
     */
    public ChunkFetchInputStream(@NotNull LoggedInEHRRequestContext context, UUID parentDocId, ChunkedDocManager chunkedDocManager) {
        this.chunkedDocManager = requireNonNull(chunkedDocManager, "IChunkedDocManager cannot be null!");
        this.requestContext = requireNonNull(context, "EHRRequestContext cannot be null!");
        List<UUID> chunkIds = chunkedDocManager.getChunkIds(requireNonNull(parentDocId, "parentDocId cannot be null!"));
        this.baseFileId = parentDocId;

        // init buffer with the top-level file
        setVariables(chunkedDocManager.getChunk(requestContext, baseFileId));

        // set up queue of "what's next" for chunk IDs
        // pop will take from the front, so add to the end (we want FIFO)
        nextChunkQueue = new LinkedList<>();
        for (UUID chunkId : chunkIds) {
            nextChunkQueue.addLast(chunkId);
        }
    }

    @Override
    public int read() throws IOException {
        if (pos < length) {
            return (buffer[pos++] & 0xff);
        }

        // uh oh, buffer is used up!
        buffer = null; // release it
        if (nextChunkQueue.isEmpty()) {
            LOGGER.info("Download of attachment parentDocId: {} finished.", baseFileId);
            return -1; // the last chunk has been sent!
        }

        try {
            setVariables(chunkedDocManager.getChunk(requestContext, nextChunkQueue.pop()));
        } catch (Exception e) {
            LOGGER.error("Error fetching file chunks for file {}", baseFileId, e);
            // ending download
            return -1;
        }

        return (buffer[pos++] & 0xff);
    }

    @Override
    public long skip(long n) throws IOException {
        long current = n;
        UUID attachmentId = baseFileId;

        while (current > chunkSize) {
            current -= chunkSize;
            attachmentId = nextChunkQueue.pop();
        }

        setVariables(chunkedDocManager.getChunk(requestContext, attachmentId));

        return super.skip(current);
    }

    @Override
    public int available() throws IOException {
        return length - pos; // answer for the current chunk
    }

    @Override
    public void close() throws IOException {
        // free stuff
        chunkedDocManager = null;
        buffer = null;
    }

    public List<UUID> getNextChunkQueue() {
        return nextChunkQueue;
    }

    private void setVariables(ChunkedDocDTO chunk) {
        buffer = chunk.getContent();
        chunkSize = buffer.length; // size of first chunk
        length = buffer.length;
        pos = 0;
    }
}
