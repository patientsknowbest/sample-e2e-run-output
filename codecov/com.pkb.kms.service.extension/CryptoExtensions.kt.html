fun EncodedEncryptedPrivateKeyInfo.encryptedAccountPrivateKey(): EncryptedAccountPrivateKey {
    val encryptedSymmetricKey = encryptedSymmetricKey(encryptedSecretKey, secretKeyAlgorithm)
    return EncryptedAccountPrivateKey.encryptedAccountPrivateKey(
            encryptedPrivateKeyInfo,
            privateKeyAlgorithm,
            encryptedSymmetricKey,
            null)
}

fun EncodedEncryptedPrivateKeyInfo.encryptedSymmetricKey(): EncryptedSymmetricKey =
    encryptedSymmetricKey(encryptedSecretKey, secretKeyAlgorithm)

fun CoreAccount.publicKey(): AccountPublicKey =
    AccountPublicKey.accountPublicKey(accountPublicKey.x509EncodedKeySpec, accountPublicKey.algorithm)

fun CoreAccountSymmetricKey.encryptedSymmetricKeyBytes() = decodeBase64(encryptedSymmetricKeyBase64)

fun CoreAccountSymmetricKey.toEncryptedAccountSymmetricKey() =
    EncryptedAccountSymmetricKey(
            id = this.id,
            key = encryptedSymmetricKey(this.encryptedSymmetricKeyBytes(), this.symmetricKeyAlgorithm))

fun PlaintextDocument.encryptWith(accountSymmetricKeyDTO: AccountSymmetricKeyDTO): EncryptedDocument =
    ImmutableEncryptedDocument.builder()
            .from(symmetricKey(accountSymmetricKeyDTO).encrypt(this))
            .putMetadata("keyId", accountSymmetricKeyDTO.keyId.toString())
            .build()

fun ByteArray.sha256(): ByteArray = Hashing.sha256().hashBytes(this).asBytes()

fun SymmetricKey.encrypt(userPrivateKey: UserPrivateKey, nonce: ByteArray): EncryptedUserPrivateKey =
        // Default to one single iteration
        this.encrypt(userPrivateKey, nonce, 1)

fun SymmetricKey.decrypt(encryptedUserPrivateKey: EncryptedUserPrivateKey, nonce: ByteArray): UserPrivateKey =
        // Default to one single iteration
        this.decrypt(encryptedUserPrivateKey, nonce, 1)
