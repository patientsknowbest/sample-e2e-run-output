package com.pkb.kms.service.extension

import io.vavr.control.Either

fun <L, R> Either<L, R>.recover(mapper: () -> Either<L, R>): Either<L, R> {
    if (isLeft) {
        return mapper()
    }

    return this
}

fun <L, R> Either<L, R>.recoverWith(mapper: (L) -> Either<L, R>): Either<L, R> {
    if (isLeft) {
        return mapper(left)
    }

    return this
}
