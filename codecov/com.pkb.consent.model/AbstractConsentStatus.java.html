package com.pkb.consent.model;

import com.pkb.datamodel.consent.PatientConsentDTO;
import com.pkb.entities.enums.AccessingEntityType;
import org.immutables.value.Value.Check;
import org.immutables.value.Value.Default;
import org.immutables.value.Value.Derived;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;
import org.jetbrains.annotations.NotNull;

import java.io.Serializable;
import java.util.Optional;

@Immutable
@Style(get = { "is*", "get*" },
        passAnnotations = { NotNull.class },
        typeAbstract = {"Abstract*"},
        typeImmutable = "*")
public abstract class AbstractConsentStatus implements Serializable {

    public abstract Optional<PatientConsentDTO> getConsent();

    @Default
    public boolean isBreakTheGlassActive() {
        return false;
    }


    @Default
    public boolean isRequired() {
        return true;
    }

    @Default
    public boolean isOrgNetworkConsentPresent() {
        return false;
    }

    @Default
    public boolean isPatientSharingEnabled() {
        return true;
    }

    @Derived
    public boolean canViewData() {
        if (!isRequired()) {
            // No consent required - data can be viewed
            return true;
        }

        return isBreakTheGlassActive() || teamDataViewPossible() || getConsent().map(PatientConsentDTO::anyConsentGranted).orElse(false);
    }

    @Derived
    public boolean teamDataViewPossible() {
        return isTeamConsentPresent() || isOrgNetworkConsentPresent();
    }

    @Check
    protected void check() {
        if (!isPatientSharingEnabled() && isBreakTheGlassActive()) {
            throw new ConsentStatusException("BTG active for opted-out patient");
        }
    }

    private boolean isTeamConsentPresent() {
        return getConsent().map(consent -> consent.accessingEntityType() == AccessingEntityType.TEAM).orElse(false);
    }

    public static ConsentStatus noConsentRequired() {
        return ConsentStatus.builder().required(false).build();
    }

    public static ConsentStatus defaultConsent() { return ConsentStatus.builder().build(); }
}
