package com.pkb.action.test;

import com.pkb.symptom.entity.SymptomSeverity;
import org.jetbrains.annotations.Nullable;

import java.util.List;

import static org.apache.commons.lang3.ObjectUtils.notEqual;

/**
 * Helper class for retrieving data in severty-input.tagx
 */
public interface SymptomsListAware {

    List<String> getSymptomsResultSet2Value();

    @Nullable
    default SymptomSeverity getSymptomSeverity(int severity0to3) {
        return SymptomSeverity.getBy0to3(severity0to3);
    }

    default String getDefaultSeverityText(SymptomSeverity severity) {
        return severity.getName();
    }

    default boolean isAutoSaved(String severityId) {
        var symptomsResultSet2Value = getSymptomsResultSet2Value();
        return symptomsResultSet2Value != null && symptomsResultSet2Value.contains(severityId);
    }

    default boolean showAdditionalSeverityText(String severityText, String customSeverityText) {
        return notEqual(severityText, customSeverityText);
    }

    default String getSeverityValue(Long severity0to3, Long symptomId) {
        return String.join(",", String.valueOf(severity0to3), String.valueOf(symptomId));
    }

    default String getSeverityId(@Nullable Long teamId, String severityValue) {
        return teamId != null ? String.join(",", String.valueOf(teamId), severityValue) : severityValue;
    }
}
