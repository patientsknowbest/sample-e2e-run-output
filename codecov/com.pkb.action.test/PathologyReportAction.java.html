package com.pkb.action.test;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.action.support.PathologyReportTransformer;
import com.pkb.service.test.PathologyReportManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.test.entity.PathologyReportDTO;
import org.springframework.beans.factory.annotation.Autowired;

public class PathologyReportAction extends BaseAction {

    private static final long serialVersionUID = 1L;

    private Long pathologyReportId;
    private PathologyReportDTO pathologyReport;

    private PathologyReportManager pathologyReportManager;
    @Autowired
    private PathologyReportTransformer pathologyReportTransformer;

    private final String tab = "health";
    private final String subTab = "laboratory";

    public String viewPathologyReport() {
        long patientId = contextUserUtil.getContextOrLoggedInUserId();
        Long patientAccountId = userManager.getDefaultAccountId(patientId);
        pathologyReport = retrievePathologyReport(patientAccountId, pathologyReportId);
        return Action.SUCCESS;
    }

    private PathologyReportDTO retrievePathologyReport(Long patientAccountId, Long pathologyReportId) {
        return pathologyReportManager.getPathologyReport(getLoggedInEHRRequestContext(), pathologyReportId, patientAccountId)
                .map(t -> pathologyReportTransformer.transformPathologyReport(t)).orElse(null);
    }

    public Long getPathologyReportId() {
        return pathologyReportId;
    }

    public void setPathologyReportId(Long pathologyReportId) {
        this.pathologyReportId = pathologyReportId;
    }

    public PathologyReportDTO getPathologyReport() {
        return pathologyReport;
    }

    public void setPathologyReport(PathologyReportDTO pathologyReport) {
        this.pathologyReport = pathologyReport;
    }

    public PathologyReportManager getPathologyReportManager() {
        return pathologyReportManager;
    }

    public void setPathologyReportManager(
            PathologyReportManager pathologyReportManager) {
        this.pathologyReportManager = pathologyReportManager;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public String getTab() {
        return tab;
    }

    public String getSubTab() {
        return subTab;
    }
}
