package com.pkb.action.test;

import io.vavr.control.Option;
import io.vavr.control.Try;
import org.apache.commons.lang3.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.invoke.MethodHandles;
import java.util.Optional;

/**
 * This is an interface for Action classes and their views which can show a certain data type on page load.
 *
 * See e.g. measurement-input-list.tagx used in addMeasurements.jsp which show a certain measurement type if the user is
 * coming from the measurement history page
 */
public interface PreselectableAddDataPointPage {

    Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    Optional<String> getShowOnPageLoadDataPointId();

    default boolean showOnPageLoad(String dataPointId) {
        return getShowOnPageLoadDataPointId().filter(id -> id.equals(dataPointId)).isPresent();
    }

    default Optional<String> convertToShowOnPageLoadDataPointId(String dataTypeID) {
        return Option.of(dataTypeID)
                .filter(StringUtils::isNotBlank)
                .onEmpty(() -> LOGGER.info("Not showing data type on page load because supplied ID is blank"))
                .flatMap(dataTypeIDString -> Try.of(() -> Long.valueOf(dataTypeIDString))
                        .onFailure(throwable -> LOGGER.warn("Could not parse data type ID=[{}]", dataTypeID, throwable))
                        .toOption())
                .map(l -> l.toString())
                .toJavaOptional();
    }
}
