package com.pkb.action.test;

import com.pkb.test.entity.MeasurementDTO;
import com.pkb.test.entity.MeasurementType;
import com.pkb.util.DateUtil;

import java.time.ZoneId;
import java.util.Locale;

import static io.micrometer.core.instrument.util.StringUtils.isNotEmpty;

public class MeasurementUtil {

    private final DateUtil dateUtil;

    public MeasurementUtil(DateUtil dateUtil) {
        this.dateUtil = dateUtil;
    }

    public String measurementToString(MeasurementDTO measurement, MeasurementType type, Locale locale, ZoneId timeZone) {
        StringBuilder result = new StringBuilder();

        if (isNotEmpty(measurement.getValueFormatted())) {
            result.append(measurement.getValueFormatted() + " " + type.getFormattedUnitOrUnitAsDefault());
        }
        if (isNotEmpty(measurement.getValue2Formatted())) {
            if (isNotEmpty(measurement.getValueFormatted())) {
                result.append(" / ");
            }
            result.append(measurement.getValue2Formatted() + " " + type.getFormattedUnit2OrUnit2AsDefault());
        }
        result.append(" - " + dateUtil.getPKBFormattedDate(measurement.getMeasureDate()  == null ? null : measurement.getMeasureDate(), locale, timeZone));

        return result.toString();
    }
}
