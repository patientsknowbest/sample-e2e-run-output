package com.pkb.entities.enums.api;

import org.apache.commons.lang3.StringUtils;

import java.util.Arrays;

public enum ApiClientScope {
    PUBLIC, PATIENT, CLINICIAN,
    TEAMCOORD, PRIVACYOFFICER,
    /** @deprecated multi value scopes should no longer be used and will be removed. */
    @Deprecated
    @SuppressWarnings("DeprecatedIsStillUsed")
    PATIENT_OR_CLINICIAN,
    /** @deprecated multi value scopes should no longer be used and will be removed. */
    @Deprecated
    CLINICIAN_OR_TEAMCOORD,
    /** @deprecated multi value scopes should no longer be used and will be removed. */
    @Deprecated
    @SuppressWarnings("DeprecatedIsStillUsed")
    PATIENT_OR_CLINICIAN_OR_TEAMCOORD,
    EMERGENCY_CARE, TEAM_DATA_UPLOAD, SITE, SUPPORT,
    /** do not use but in place in conjunction with the {@link ApiClientScope#lookup(String)} to avoid using exceptions for flow control */
    NOT_VALID;

    public static ApiClientScope lookup(String valueToFind) {
        return Arrays.stream(ApiClientScope.values()).filter(value -> value.name().equals(StringUtils.upperCase(valueToFind))).findFirst().orElse(NOT_VALID);
    }
}
