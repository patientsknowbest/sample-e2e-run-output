package com.pkb.entities.enums.api;

public enum WebAppOtpStatus {
    /** This is the most recent OTP to be generated for this user, and it has not yet been used. */
    ACTIVE,
    /** A new OTP was generated for this user before this one was attempted to be used. */
    INACTIVE,
    /** This OTP was successfully used to login. */
    USED,
    /** The user tried to login with this OTP, but it had expired. */
    EXPIRED
}
