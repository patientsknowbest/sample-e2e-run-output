package com.pkb.aggregator.service.enrichment

import com.pkb.aggregator.Dstu3Extension
import com.pkb.aggregator.Dstu3StringType
import com.pkb.aggregator.R4Extension
import com.pkb.aggregator.R4StringType

fun createDstu3SourceExtension(sourceId: String, resourceId: String) = Dstu3Extension().apply {
    this.url = CanonicalExtensionURLs.upstreamTag()
    this.extension.add(Dstu3Extension().apply {
        this.url = CanonicalExtensionURLs.upstreamSourceId()
        this.setValue(Dstu3StringType(sourceId))
    })
    this.extension.add(Dstu3Extension().apply {
        this.url = CanonicalExtensionURLs.upstreamResourceId()
        this.setValue(Dstu3StringType(resourceId))
    })
}
