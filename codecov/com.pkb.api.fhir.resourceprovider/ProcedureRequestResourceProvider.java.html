package com.pkb.api.fhir.resourceprovider;

import ca.uhn.fhir.rest.annotation.IdParam;
import ca.uhn.fhir.rest.annotation.Read;
import ca.uhn.fhir.rest.api.server.RequestDetails;
import com.pkb.api.fhir.questionnaire.ProcedureRequestResourceService;
import com.pkb.api.fhir.service.AuthenticationService;
import org.hl7.fhir.dstu3.model.IdType;
import org.hl7.fhir.dstu3.model.ProcedureRequest;
import org.jetbrains.annotations.NotNull;

import java.util.function.Function;

import static com.pkb.api.fhir.util.Fhirs.notFoundDueToInvalidUUIDException;
import static com.pkb.api.fhir.util.Fhirs.uuid;

public class ProcedureRequestResourceProvider extends AbstractResourceProvider {

    private final AuthenticationService authenticationService;
    private final ProcedureRequestResourceService procedureRequestResourceService;

    public ProcedureRequestResourceProvider(
            @NotNull AuthenticationService authenticationService,
            @NotNull ProcedureRequestResourceService procedureRequestResourceService) {
        super(ProcedureRequest.class);
        this.authenticationService = authenticationService;
        this.procedureRequestResourceService = procedureRequestResourceService;
    }

    @Read
    public ProcedureRequest getResourceById(@IdParam IdType questionnaireRequestId, RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        return uuid(questionnaireRequestId)
                .mapLeft(error -> notFoundDueToInvalidUUIDException(questionnaireRequestId))
                .flatMap(q -> procedureRequestResourceService.getProcedureRequest(authenticatedIdentity, q))
                .getOrElseThrow(Function.identity());
    }
}
