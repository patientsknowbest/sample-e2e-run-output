import static org.slf4j.LoggerFactory.getLogger;

public class DiagnosticReportResourceProvider extends AbstractResourceProvider {

    private static final Logger LOGGER = getLogger(lookup().lookupClass());

    private static final String SP_CATEGORY_DESCRIPTION = "Required. Multi-value support: none. Modifier support: none.";
    private static final String SP_PATIENT_DESCRIPTION = "Required. Multi-value support: none. Modifier support: [<type>].";
    private static final String SP_DATE_DESCRIPTION = "Optional. Multi-value support: AND. Modifier support: none. Prefix support: [eq, gt, lt, ge, le].";

    private AuthenticationService authenticationService;
    private final DiagnosticReportResourceService diagnosticReportResourceService;

    private final ReferenceParamToUUIDConverter referenceParamConverter;
    private final DateRangeParamToDateRangeRestrictionConverter dateRangeParamToDateRangeRestrictionConverter;
    private final TokenParamToDiagnosticReportCategoryConverter diagnosticReportCategoryConverter;
    private final BundleMapper bundleMapper;

    public DiagnosticReportResourceProvider(
            AuthenticationService authenticationService,
            DiagnosticReportResourceService diagnosticReportResourceService,
            ReferenceParamToUUIDConverter referenceParamConverter,
            DateRangeParamToDateRangeRestrictionConverter dateRangeParamToDateRangeRestrictionConverter,
            TokenParamToDiagnosticReportCategoryConverter diagnosticReportCategoryConverter,
            BundleMapper bundleMapper) {
        super(DiagnosticReport.class);
        this.authenticationService = authenticationService;
        this.diagnosticReportResourceService = diagnosticReportResourceService;
        this.referenceParamConverter = referenceParamConverter;
        this.dateRangeParamToDateRangeRestrictionConverter = dateRangeParamToDateRangeRestrictionConverter;
        this.diagnosticReportCategoryConverter = diagnosticReportCategoryConverter;
        this.bundleMapper = bundleMapper;
    }

    @SuppressWarnings("Duplicates")
    @Search
    public Bundle findDiagnosticReport(
            @RequiredParam(name = SP_CATEGORY) @Description(value = SP_CATEGORY_DESCRIPTION) TokenParam category,
            @RequiredParam(name = SP_PATIENT) @Description(value = SP_PATIENT_DESCRIPTION) ReferenceParam patient,
            @OptionalParam(name = SP_DATE) @Description(value = SP_DATE_DESCRIPTION) DateRangeParam date,
            RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        NamedParam<TokenParam> categoryParam = new NamedParam<>(SP_CATEGORY, category, request.getParameters().get(SP_CATEGORY));
        NamedParam<ReferenceParam> patientParam = new NamedParam<>(SP_PATIENT, patient, request.getParameters().get(SP_PATIENT));
        NamedParam<DateRangeParam> dateParam = new NamedParam<>(SP_DATE, date, request.getParameters().get(SP_DATE));

        Seq<String> parameters = Seq(categoryParam.getName(), patientParam.getName(), dateParam.getName());

        // Early exit with empty bundle result if category is not valid
        Validation<Issues, Option<DiagnosticReportCategory>> issuesOrCategory = diagnosticReportCategoryConverter.convert(categoryParam);
        if (issuesOrCategory.isInvalid()) {
            LOGGER.warn(String.format("Invalid category sent. Diagnostics: [%s]", issuesOrCategory.getError()));
            return bundleMapper.toBundle(criteriaResult(0L), request, Some(parameters));
        }

        Validation<Issues, UUID> issuesOrPatientId = referenceParamConverter.convertRequired(patientParam, Patient);
        Validation<Issues, TemporalRangeRestriction<LocalDate>> issuesOrRestriction = dateRangeParamToDateRangeRestrictionConverter.convert(dateParam);

        Either<BaseServerResponseException, CriteriaResult<DiagnosticReport>> searchResult = Validation.combine(issuesOrPatientId, issuesOrRestriction, issuesOrCategory)
                .ap((patientId, restriction, cat ) -> buildSearchCriteria(authenticatedIdentity, patientId, restriction, cat))
                .map(diagnosticReportResourceService::search)
                .mapError(Issues::flatten)
                .fold(issues -> left(invalidRequestException(issues)), $ -> $.fold(Either::left, Either::right));

        return searchResult.fold(
                bundleMapper.toEmptyBundle(request, Some(parameters)),
                bundleMapper.toBundle(request, Some(parameters)));
    }

    private DiagnosticReportSearchCriteria buildSearchCriteria(AuthenticatedIdentity authenticatedIdentity,
                                                               UUID patientId,
                                                               TemporalRangeRestriction<LocalDate> effectiveTimestamp,
                                                               Option<DiagnosticReportCategory> category) {
        return ImmutableDiagnosticReportSearchCriteria.builder()
                .authenticatedIdentity(authenticatedIdentity)
                .patientId(patientId)
                .effectiveTimestamp(effectiveTimestamp)
                .category(category)
                .build();
    }
}
