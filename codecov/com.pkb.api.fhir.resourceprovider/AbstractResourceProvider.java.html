package com.pkb.api.fhir.resourceprovider;

import ca.uhn.fhir.rest.server.IResourceProvider;
import org.hl7.fhir.instance.model.api.IBaseResource;

public abstract class AbstractResourceProvider implements IResourceProvider {

    public static final String IP_REVERSE_INCLUDE = "_revinclude";

    private final Class<? extends IBaseResource> resourceType;

    @SuppressWarnings({ "WeakerAccess" })
    protected AbstractResourceProvider(Class<? extends IBaseResource> resourceType) {
        this.resourceType = resourceType;
    }

    @Override
    public final Class<? extends IBaseResource> getResourceType() {
        return resourceType;
    }
}
