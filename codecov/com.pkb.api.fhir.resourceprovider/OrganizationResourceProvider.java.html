package com.pkb.api.fhir.resourceprovider;

import ca.uhn.fhir.rest.annotation.IdParam;
import ca.uhn.fhir.rest.annotation.Operation;
import ca.uhn.fhir.rest.annotation.Read;
import ca.uhn.fhir.rest.annotation.ResourceParam;
import ca.uhn.fhir.rest.api.server.RequestDetails;
import com.google.common.collect.Lists;
import com.pkb.api.fhir.NationalIdToIdentifierMapper;
import com.pkb.api.fhir.org.LetterInvitationResourceService;
import com.pkb.api.fhir.org.OrganizationResourceService;
import com.pkb.api.fhir.service.AuthenticationService;
import com.pkb.datamodel.LetterInvitationTokenDTO;
import io.vavr.collection.List;
import org.hl7.fhir.dstu3.model.DateTimeType;
import org.hl7.fhir.dstu3.model.IdType;
import org.hl7.fhir.dstu3.model.Identifier;
import org.hl7.fhir.dstu3.model.Organization;
import org.hl7.fhir.dstu3.model.Parameters;
import org.hl7.fhir.dstu3.model.StringType;
import org.jetbrains.annotations.NotNull;

import java.util.Date;

import static com.pkb.api.fhir.util.Fhirs.notFoundDueToInvalidUUIDException;
import static com.pkb.api.fhir.util.Fhirs.uuid;
import static java.util.function.Function.identity;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity.WARNING;

public class OrganizationResourceProvider extends AbstractResourceProvider {
    public static final String GENERATE_INVITATION_TOKENS_OPERATION_NAME = "generate-invitation-tokens";

    private static final String PATIENT_IDENTIFIER_PART_NAME = "patientIdentifier";
    private final AuthenticationService authenticationService;
    private final OrganizationResourceService organizationResourceService;
    private final LetterInvitationResourceService letterInvitationResourceService;

    public OrganizationResourceProvider(AuthenticationService authenticationService,
                                        @NotNull OrganizationResourceService organizationResourceService,
                                        @NotNull LetterInvitationResourceService letterInvitationResourceService) {
        super(Organization.class);
        this.authenticationService = authenticationService;
        this.organizationResourceService = organizationResourceService;
        this.letterInvitationResourceService = letterInvitationResourceService;
    }

    @Read
    @SuppressWarnings({"WeakerAccess"})
    public Organization find(@IdParam IdType id, RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        return uuid(id.getIdPart(), WARNING)
                .mapLeft(error -> notFoundDueToInvalidUUIDException(id))
                .flatMap((orgId) -> organizationResourceService.findOrganization(authenticatedIdentity, orgId))
                .getOrElseThrow(identity());
    }

    @Operation(type = Parameters.class, name = GENERATE_INVITATION_TOKENS_OPERATION_NAME, idempotent = true)
    @SuppressWarnings({"WeakerAccess"})
    public Parameters generateInvitationTokens(@ResourceParam Parameters parameters, RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        return letterInvitationResourceService.getLetterInvitations(authenticatedIdentity, parameters)
                .map(this::createParameters)
                .getOrElseThrow(identity());
    }

    private Parameters createParameters(List<LetterInvitationTokenDTO> letterInvitationTokenDTOS) {
        Parameters parameters = new Parameters();
        letterInvitationTokenDTOS
                .forEach(dto -> parameters.addParameter(mapInvitationTokenDTOToParameter(dto)));
        return parameters;
    }

    private Parameters.ParametersParameterComponent mapInvitationTokenDTOToParameter(LetterInvitationTokenDTO invitationTokenDTO) {
        Identifier identifier = NationalIdToIdentifierMapper.toIdentifier(invitationTokenDTO.getValidNationalId());
        Parameters.ParametersParameterComponent identifierPart = new Parameters.ParametersParameterComponent();
        identifierPart.setName(PATIENT_IDENTIFIER_PART_NAME);
        identifierPart.setValue(identifier);

        Parameters.ParametersParameterComponent code = new Parameters.ParametersParameterComponent();
        code.setName("code");
        code.setValue(new StringType(invitationTokenDTO.getCode()));

        Parameters.ParametersParameterComponent token = new Parameters.ParametersParameterComponent();
        token.setName("token");
        token.setValue(new StringType(invitationTokenDTO.getToken()));

        Parameters.ParametersParameterComponent expiry = new Parameters.ParametersParameterComponent();
        expiry.setName("expiry");
        expiry.setValue(new DateTimeType(Date.from(invitationTokenDTO.getExpiry())));

        Parameters.ParametersParameterComponent component = new Parameters.ParametersParameterComponent();
        component.setName("patientTokens");
        component.setPart(Lists.newArrayList(identifierPart, code, token, expiry));

        return component;
    }
}
