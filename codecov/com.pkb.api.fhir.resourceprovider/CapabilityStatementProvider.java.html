package com.pkb.api.fhir.resourceprovider;

import ca.uhn.fhir.rest.annotation.IdParam;
import ca.uhn.fhir.rest.annotation.Metadata;
import ca.uhn.fhir.rest.annotation.Read;
import ca.uhn.fhir.rest.server.RestfulServer;
import ca.uhn.fhir.rest.server.exceptions.ResourceNotFoundException;
import ca.uhn.fhir.rest.server.servlet.ServletRequestDetails;
import com.pkb.common.datetime.DateTimeService;
import org.hl7.fhir.dstu3.hapi.rest.server.ServerCapabilityStatementProvider;
import org.hl7.fhir.dstu3.model.CapabilityStatement;
import org.hl7.fhir.dstu3.model.CodeType;
import org.hl7.fhir.dstu3.model.IdType;
import org.hl7.fhir.dstu3.model.OperationDefinition;
import org.hl7.fhir.dstu3.model.OperationOutcome;
import org.hl7.fhir.dstu3.model.UriType;

import javax.servlet.http.HttpServletRequest;
import java.util.Date;

import static ca.uhn.fhir.rest.api.Constants.CT_FHIR_JSON_NEW;
import static com.google.common.collect.ImmutableList.of;
import static com.pkb.domain.criteria.OperationOutcomeFactory.operationOutcome;
import static java.lang.String.format;
import static org.hl7.fhir.dstu3.model.Enumerations.PublicationStatus.ACTIVE;
import static org.hl7.fhir.dstu3.model.Observation.SP_CATEGORY;
import static org.hl7.fhir.dstu3.model.Observation.SP_DATE;
import static org.hl7.fhir.dstu3.model.Observation.SP_PATIENT;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity.ERROR;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueType.NOTFOUND;

public class CapabilityStatementProvider {
    private final RestfulServer server;
    private final ServerCapabilityStatementProvider serverCapabilityStatementProvider;
    private final boolean isExperimental;
    private final DateTimeService dateTimeService;

    public static CapabilityStatementProvider production(ServerCapabilityStatementProvider serverCapabilityStatementProvider, 
		    					 RestfulServer server,
							 DateTimeService dateTimeService) {
        return new CapabilityStatementProvider(serverCapabilityStatementProvider, false, server, dateTimeService);
    }

    public static CapabilityStatementProvider experimental(ServerCapabilityStatementProvider serverCapabilityStatementProvider, 
		    					   RestfulServer server,
							   DateTimeService dateTimeService) {
        return new CapabilityStatementProvider(serverCapabilityStatementProvider, true, server, dateTimeService);
    }

    private CapabilityStatementProvider(ServerCapabilityStatementProvider serverCapabilityStatementProvider, 
		    			boolean isExperimental, 
					RestfulServer server,
					DateTimeService dateTimeService) {
        this.serverCapabilityStatementProvider = serverCapabilityStatementProvider;
        this.isExperimental = isExperimental;
        this.server = server;
	    this.dateTimeService = dateTimeService;
    }

    public void setRestfulServer(RestfulServer server) {
        // RestfulServer#setServerConformanceProvider expects an instance that has this method.
    }

    @Metadata
    public CapabilityStatement getServerMetadata(HttpServletRequest request) {
        ServletRequestDetails servletRequestDetails = new ServletRequestDetails();
        servletRequestDetails.setServer(server);
        CapabilityStatement statement = serverCapabilityStatementProvider.getServerConformance(request, servletRequestDetails);
        statement.setUrl("http://fhir.patientsknowbest.com/capabilitystatement/facade");
        statement.setTitle("PKB Facade FHIR API");
        statement.setExperimental(isExperimental);
        statement.setStatus(ACTIVE);
        statement.setDate(Date.from(dateTimeService.now()));
        statement.setPublisher("Patients Know Best Ltd.");
        statement.setAcceptUnknown(CapabilityStatement.UnknownContentCode.NO);
        statement.setFormat(of(new CodeType(CT_FHIR_JSON_NEW)));
        statement.setSoftware(null);
        statement.setImplementationGuide(of(new UriType("http://fhir.patientsknowbest.com")));
        return statement;
    }

    @Read(type = OperationDefinition.class)
    public OperationDefinition readOperationDefinition(@IdParam IdType id) {
        OperationDefinition definition = null;
        try {
            ServletRequestDetails servletRequestDetails = new ServletRequestDetails();
            servletRequestDetails.setServer(server);
            definition = serverCapabilityStatementProvider.readOperationDefinition(id, servletRequestDetails);
            amendDefinitionAccordingToCustomRequirements(definition);
        } catch (ResourceNotFoundException e) {
            OperationOutcome outcome = operationOutcome()
                    .addIssue(new OperationOutcome.OperationOutcomeIssueComponent()
                            .setSeverity(ERROR)
                            .setCode(NOTFOUND)
                            .setDiagnostics(format("Could not find OperationDefinition with ID=[%s]", id.getIdPart())));
            e.setOperationOutcome(outcome);
            throw e;
        }
        return definition;
    }

    /**
     * The OperationDefinition resource is automatically generated based on annotations, e.g. see how
     * ObservationResourceProvider.findObservationGroup includes an Operation annotation. However, not all attributes
     * can be set within the annotation, so this method allows the CapabilityStatementProvider to insert the necessary
     * values before the OperationDefinition instance is returned.
     */
    private void amendDefinitionAccordingToCustomRequirements(OperationDefinition definition) {
        // Set the globally unique URL, and parameter descriptions
        if (ConsentResourceProvider.PURVIEW_OPERATION_NAME.equals(definition.getName())) {
            definition.setUrl("http://fhir.patientsknowbest.com/operation/purview");
        } else if (DocumentReferenceResourceProvider.READ_RECEIPT_QUERY_OPERATION_NAME.equals(definition.getName())) {
            definition.setUrl("http://fhir.patientsknowbest.com/operation/read-receipt-query");
        } else if (ObservationResourceProvider.GROUP_OPERATION_NAME.equals(definition.getName())) {
            definition.setUrl("http://fhir.patientsknowbest.com/operation/group");
            for(OperationDefinition.OperationDefinitionParameterComponent parameter : definition.getParameter()) {
                if(SP_CATEGORY.equals(parameter.getName())) {
                    parameter.setDocumentation(ObservationResourceProvider.SP_CATEGORY_DESCRIPTION);
                } else if(SP_PATIENT.equals(parameter.getName())) {
                    parameter.setDocumentation(ObservationResourceProvider.SP_PATIENT_DESCRIPTION);
                } else if(SP_DATE.equals(parameter.getName())) {
                    parameter.setDocumentation(ObservationResourceProvider.SP_DATE_DESCRIPTION);
                }
            }
        } else if (OrganizationResourceProvider.GENERATE_INVITATION_TOKENS_OPERATION_NAME.equals(definition.getName())) {
            definition.setUrl("http://fhir.patientsknowbest.com/operation/generate-invitation-tokens");
        } else if (PatientResourceProvider.PATIENT_DATE_OF_LAST_DATA_POINT_OPERATION_NAME.equals(definition.getName())) {
            definition.setUrl("http://fhir.patientsknowbest.com/operation/date-of-last-data-point");
        } else if (PatientResourceProvider.PATIENT_STATUS_QUERY_OPERATION_NAME.equals(definition.getName())) {
            definition.setUrl("http://fhir.patientsknowbest.com/operation/status-query");
        }
        // Set the status
        definition.setStatus(ACTIVE);
    }
}
