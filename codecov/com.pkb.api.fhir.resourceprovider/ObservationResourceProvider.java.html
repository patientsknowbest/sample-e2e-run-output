package com.pkb.api.fhir.resourceprovider;

import static com.pkb.api.fhir.util.Fhirs.invalidRequestException;
import static com.pkb.domain.criteria.CriteriaResult.emptyCriteriaResult;
import static io.vavr.API.Seq;
import static io.vavr.API.Some;
import static io.vavr.control.Either.left;
import static java.lang.invoke.MethodHandles.lookup;
import static org.hl7.fhir.dstu3.model.Observation.SP_CATEGORY;
import static org.hl7.fhir.dstu3.model.Observation.SP_DATE;
import static org.hl7.fhir.dstu3.model.Observation.SP_PATIENT;
import static org.hl7.fhir.dstu3.model.ResourceType.Patient;
import static org.slf4j.LoggerFactory.getLogger;

public class ObservationResourceProvider extends AbstractResourceProvider {
    private static final Logger LOGGER = getLogger(lookup().lookupClass());

    public static final String GROUP_OPERATION_NAME = "group";
    public static final String SP_CATEGORY_DESCRIPTION = "Required. Multi-value support: none. Modifier support: none.";
    public static final String SP_PATIENT_DESCRIPTION = "Required. Multi-value support: none. Modifier support: [<type>].";
    public static final String SP_DATE_DESCRIPTION = "Optional. Multi-value support: AND. Modifier support: none. Prefix support: [eq, gt, lt, ge, le].";

    private final AuthenticationService authenticationService;
    private final ObservationResourceService searchService;
    private final ReferenceParamToUUIDConverter referenceParamConverter;
    private final DateRangeParamToDateRangeRestrictionConverter dateRangeParamToDateRangeRestrictionConverter;
    private final TokenParamToObservationCategoryConverter observationCategoryConverter;
    private final BundleMapper bundleMapper;

    public ObservationResourceProvider(AuthenticationService authenticationService,
                                       ObservationResourceService searchService,
                                       ReferenceParamToUUIDConverter referenceParamConverter,
                                       DateRangeParamToDateRangeRestrictionConverter dateRangeParamToDateRangeRestrictionConverter,
                                       TokenParamToObservationCategoryConverter observationCategoryConverter,
                                       BundleMapper bundleMapper) {
        super(Observation.class);
        this.authenticationService = authenticationService;
        this.searchService = searchService;
        this.referenceParamConverter = referenceParamConverter;
        this.dateRangeParamToDateRangeRestrictionConverter = dateRangeParamToDateRangeRestrictionConverter;
        this.observationCategoryConverter = observationCategoryConverter;
        this.bundleMapper = bundleMapper;
    }

    @Search
    @SuppressWarnings({ "WeakerAccess" })
    public Bundle findObservation(
            @RequiredParam(name = SP_CATEGORY) @Description(value = SP_CATEGORY_DESCRIPTION) TokenParam category,
            @RequiredParam(name = SP_PATIENT) @Description(value = SP_PATIENT_DESCRIPTION) ReferenceParam patient,
            @OptionalParam(name = SP_DATE) @Description(value = SP_DATE_DESCRIPTION) DateRangeParam date,
            RequestDetails request) {
        return findObservation(category, patient, date, ResultFilter.ALL, request);
    }

    @Operation(type = Observation.class, name = GROUP_OPERATION_NAME, idempotent = true)
    @SuppressWarnings({ "WeakerAccess" })
    public Bundle findObservationGroup(
            @OperationParam(name = SP_CATEGORY, min = 1) TokenParam category,
            @OperationParam(name = SP_PATIENT, min = 1) ReferenceParam patient,
            @OperationParam(name = SP_DATE) DateRangeParam date,
            RequestDetails request) {
        return findObservation(category, patient, date, ResultFilter.ONE_PER_DAY, request);
    }


    private Bundle findObservation(TokenParam category,
                                   ReferenceParam patient,
                                   DateRangeParam date,
                                   ResultFilter resultFilter,
                                   RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        NamedParam<TokenParam> categoryParam = new NamedParam<>(SP_CATEGORY, category, request.getParameters().get(SP_CATEGORY));
        NamedParam<ReferenceParam> patientParam = new NamedParam<>(SP_PATIENT, patient, request.getParameters().get(SP_PATIENT));
        NamedParam<DateRangeParam> dateParam = new NamedParam<>(SP_DATE, date, request.getParameters().get(SP_DATE));

        Seq<String> parameters = Seq(categoryParam.getName(), patientParam.getName(), dateParam.getName());
        Validation<Issues, Option<ObservationCategory>> issuesOrCategory = observationCategoryConverter.convert(categoryParam);
        if (issuesOrCategory.isInvalid()) {
            LOGGER.warn(String.format("Invalid category sent. Diagnostics: [%s]", issuesOrCategory.getError()));
            return bundleMapper.toBundle(emptyCriteriaResult(), request, Some(parameters));
        }

        Validation<Issues, UUID> issuesOrPatientId = referenceParamConverter.convertRequired(patientParam, Patient);
        Validation<Issues, TemporalRangeRestriction<LocalDate>> issuesOrRestriction = dateRangeParamToDateRangeRestrictionConverter.convert(dateParam);
        Either<BaseServerResponseException, CriteriaResult<Observation>> searchResult =
            Validation.combine(issuesOrCategory, issuesOrPatientId, issuesOrRestriction)
                .ap((cat, id, restriction) -> this.buildSearchCriteria(authenticatedIdentity, cat, id, restriction, resultFilter))
                .map(searchService::search)
                .mapError(Issues::flatten)
                .fold(issues -> left(invalidRequestException(issues)), $ -> $.fold(Either::left, Either::right));

        return searchResult.fold(
                bundleMapper.toEmptyBundle(request, Some(parameters)),
                bundleMapper.toBundle(request, Some(parameters)));
    }

    private ObservationSearchCriteria buildSearchCriteria(AuthenticatedIdentity authenticatedIdentity,
                                                          Option<ObservationCategory> category,
                                                          UUID patientId,
                                                          TemporalRangeRestriction<LocalDate> effectiveTimestamp,
                                                          ResultFilter resultFilter) {
        return ImmutableObservationSearchCriteria.builder()
                .authenticatedIdentity(authenticatedIdentity)
                .patientId(patientId)
                .category(category)
                .effectiveTimestamp(effectiveTimestamp)
                .resultFilter(resultFilter)
                .build();
    }
}
