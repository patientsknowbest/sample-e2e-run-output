package com.pkb.api.fhir.resourceprovider;

import ca.uhn.fhir.model.api.annotation.Description;
import ca.uhn.fhir.rest.annotation.RequiredParam;
import ca.uhn.fhir.rest.annotation.Search;
import ca.uhn.fhir.rest.api.server.RequestDetails;
import ca.uhn.fhir.rest.param.StringParam;
import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import com.pkb.api.fhir.BundleMapper;
import com.pkb.api.fhir.converter.UrnUUIDConverter;
import com.pkb.api.fhir.org.NamingSystemResourceService;
import com.pkb.api.fhir.service.AuthenticationService;
import com.pkb.api.fhir.util.Issues;
import com.pkb.domain.criteria.FhirCriteriaResult;
import io.vavr.collection.Seq;
import io.vavr.control.Either;
import io.vavr.control.Option;
import io.vavr.control.Validation;
import org.hl7.fhir.dstu3.model.Bundle;
import org.hl7.fhir.dstu3.model.NamingSystem;

import java.util.UUID;

import static com.pkb.api.fhir.util.Fhirs.invalidRequestException;
import static io.vavr.API.Seq;
import static io.vavr.API.Some;
import static io.vavr.control.Either.left;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity.WARNING;

public class NamingSystemResourceProvider extends AbstractResourceProvider {
    private static final String VALUE = "value";
    private static final String SP_VALUE_DESCRIPTION = "Required. Multi-value support: none. Modifier support: [exact].";

    private final AuthenticationService authenticationService;
    private final NamingSystemResourceService resourceService;
    private final UrnUUIDConverter urnUUIDConverter;
    private final BundleMapper bundleMapper;

    public NamingSystemResourceProvider(AuthenticationService authenticationService,
                                        NamingSystemResourceService resourceService,
                                        UrnUUIDConverter urnUUIDConverter,
                                        BundleMapper bundleMapper) {
        super(NamingSystem.class);
        this.authenticationService = authenticationService;
        this.resourceService = resourceService;
        this.urnUUIDConverter = urnUUIDConverter;
        this.bundleMapper = bundleMapper;
    }

    @Search
    @SuppressWarnings({ "WeakerAccess" })
    public Bundle findNamingSystem(
            @RequiredParam(name = VALUE) @Description(value = SP_VALUE_DESCRIPTION) StringParam value,
            RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        Validation<Issues, UUID> issuesOrUUID = urnUUIDConverter.urnUuid(value, WARNING);
        Option<Seq<String>> parameters = Some(Seq(VALUE));
        Either<BaseServerResponseException, FhirCriteriaResult<NamingSystem>> result = issuesOrUUID
                .map(uuid -> resourceService.searchNamingSystem(authenticatedIdentity, uuid))
                .fold(issues -> left(invalidRequestException(issues)), $ -> $.fold(Either::left, Either::right));

        return result.fold(
        issues -> bundleMapper.toEmptyBundleOrThrowException(issues, request, parameters),
        output -> bundleMapper.toBundle(output, request, parameters));
    }
}
