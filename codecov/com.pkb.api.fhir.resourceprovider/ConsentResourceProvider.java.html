package com.pkb.api.fhir.resourceprovider;

import ca.uhn.fhir.model.api.annotation.Description;

import ca.uhn.fhir.rest.api.server.RequestDetails;
import ca.uhn.fhir.rest.param.ReferenceParam;
import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import com.pkb.api.fhir.BundleMapper;
import com.pkb.api.fhir.consent.ConsentResourceService;
import com.pkb.api.fhir.converter.ReferenceParamToUUIDConverter;
import com.pkb.api.fhir.service.AuthenticationService;
import com.pkb.api.fhir.util.Issues;
import com.pkb.api.fhir.util.NamedParam;
import com.pkb.api.fhir.util.NamedParameterFactory;
import com.pkb.domain.criteria.FhirCriteriaResult;
import io.vavr.collection.Seq;
import io.vavr.control.Either;
import io.vavr.control.Option;
import io.vavr.control.Validation;
import org.hl7.fhir.dstu3.model.Bundle;
import org.hl7.fhir.dstu3.model.Consent;
import org.hl7.fhir.dstu3.model.Observation;
import org.hl7.fhir.dstu3.model.Parameters;
import org.hl7.fhir.dstu3.model.Reference;

import java.util.UUID;

import static com.pkb.api.fhir.util.Fhirs.invalidRequestException;
import static io.vavr.API.Seq;
import static io.vavr.API.Some;
import static io.vavr.control.Either.left;
import static java.util.function.Function.identity;
import static org.hl7.fhir.dstu3.model.Consent.SP_PATIENT;
import static org.hl7.fhir.dstu3.model.ResourceType.Patient;

public class ConsentResourceProvider extends AbstractResourceProvider {
    public static final String PURVIEW_OPERATION_NAME = "purview";
    private static final String SP_PATIENT_DESCRIPTION = "Required. Multi-value support: none. Modifier support: [<type>].";

    private AuthenticationService authenticationService;
    private final ConsentResourceService service;
    private final BundleMapper bundleMapper;
    private final ReferenceParamToUUIDConverter referenceParamConverter;
    private final NamedParameterFactory namedParameterFactory;

    public ConsentResourceProvider(AuthenticationService authenticationService,
                                   ConsentResourceService service,
                                   BundleMapper bundleMapper,
                                   ReferenceParamToUUIDConverter referenceParamConverter,
                                   NamedParameterFactory namedParameterFactory) {
        super(Consent.class);
        this.authenticationService = authenticationService;
        this.service = service;
        this.bundleMapper = bundleMapper;
        this.referenceParamConverter = referenceParamConverter;
        this.namedParameterFactory = namedParameterFactory;
    }

    @Operation(type = Parameters.class, name = PURVIEW_OPERATION_NAME, idempotent = true, returnParameters = @OperationParam(name = PURVIEW_OPERATION_NAME, min = 1, max = 1, type = Reference.class))
    public Parameters purview(RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        Reference reference = service.purview(authenticatedIdentity)
                .getOrElseThrow(identity());

        Parameters parameters = new Parameters();
        parameters
                .addParameter()
                .setName("purview")
                .setValue(reference);

        return parameters;
    }

    @Search
    public Bundle getConsent(
            @RequiredParam(name = SP_PATIENT) @Description(value = SP_PATIENT_DESCRIPTION) ReferenceParam referenceToPatient,
            RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);

        NamedParam<ReferenceParam> patientNamedParam = namedParameterFactory
                .createNamedParam(referenceToPatient, Observation.SP_PATIENT, request);
        Validation<Issues, UUID> issuesOrLinkId = referenceParamConverter.convertRequired(patientNamedParam, Patient);
        Option<Seq<String>> parameters = Some(Seq(patientNamedParam.getName()));

        Either<BaseServerResponseException, FhirCriteriaResult<Consent>> searchResult = issuesOrLinkId.toEither()
                .map((linkId) -> service.findPatientConsents(authenticatedIdentity ,linkId))
                .fold(issues -> left(invalidRequestException(issues)), $ -> $.fold(Either::left, Either::right));

        return searchResult.fold(
                bundleMapper.toEmptyBundle(request, parameters),
                result -> bundleMapper.toBundle(result, request, parameters));
    }
}
