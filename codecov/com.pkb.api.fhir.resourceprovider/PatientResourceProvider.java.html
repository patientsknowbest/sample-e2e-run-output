package com.pkb.api.fhir.resourceprovider;

import ca.uhn.fhir.model.api.annotation.Description;
import ca.uhn.fhir.rest.annotation.IdParam;
import ca.uhn.fhir.rest.annotation.Operation;
import ca.uhn.fhir.rest.annotation.Read;
import ca.uhn.fhir.rest.annotation.RequiredParam;
import ca.uhn.fhir.rest.annotation.ResourceParam;
import ca.uhn.fhir.rest.annotation.Search;
import ca.uhn.fhir.rest.api.server.RequestDetails;
import ca.uhn.fhir.rest.param.TokenParam;
import com.pkb.api.fhir.BundleMapper;
import com.pkb.api.fhir.patient.PatientResourceService;
import com.pkb.api.fhir.service.AuthenticationService;
import com.pkb.api.fhir.util.NamedParam;
import io.vavr.collection.Seq;
import org.hl7.fhir.dstu3.model.Bundle;
import org.hl7.fhir.dstu3.model.IdType;
import org.hl7.fhir.dstu3.model.Parameters;
import org.hl7.fhir.dstu3.model.Patient;

import static com.pkb.api.fhir.util.Fhirs.notFoundDueToInvalidUUIDException;
import static com.pkb.api.fhir.util.Fhirs.uuid;
import static io.vavr.API.Seq;
import static io.vavr.API.Some;
import static java.util.function.Function.identity;
import static org.hl7.fhir.dstu3.model.Person.SP_IDENTIFIER;

public class PatientResourceProvider extends AbstractResourceProvider {
    private static final String SP_IDENTIFIER_DESCRIPTION = "Required. Multi-value support: none. Modifier support: none. Supported values: [https://fhir.nhs.uk/Id/nhs-number].";

    public static final String PATIENT_DATE_OF_LAST_DATA_POINT_OPERATION_NAME = "date-of-last-data-point";
    public static final String PATIENT_STATUS_QUERY_OPERATION_NAME = "status-query";

    private final AuthenticationService authenticationService;
    private final PatientResourceService patientResourceService;
    private final BundleMapper bundleMapper;

    public PatientResourceProvider(AuthenticationService authenticationService,
                                   PatientResourceService patientResourceService,
                                   BundleMapper bundleMapper) {
        super(Patient.class);
        this.authenticationService = authenticationService;
        this.patientResourceService = patientResourceService;
        this.bundleMapper = bundleMapper;
    }

    @Read
    @SuppressWarnings({ "WeakerAccess" })
    public Patient getResourceById(@IdParam IdType patientId, RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        return uuid(patientId)
                .mapLeft(error -> notFoundDueToInvalidUUIDException(patientId))
                .flatMap(id -> patientResourceService.findPatientDemographics(authenticatedIdentity, id))
                .getOrElseThrow(identity());
    }

    @Search
    @SuppressWarnings({ "WeakerAccess" })
    public Bundle searchPatient(
            @RequiredParam(name = SP_IDENTIFIER) @Description(value = SP_IDENTIFIER_DESCRIPTION) TokenParam token,
            RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        NamedParam<TokenParam> tokenParam = new NamedParam<>(SP_IDENTIFIER, token, request.getParameters().get(SP_IDENTIFIER));
        Seq<String> parameters = Seq(tokenParam.getName());

        return patientResourceService.searchForPatientByCodedIdentifier(authenticatedIdentity, token).fold(
                bundleMapper.toEmptyBundle(request, Some(parameters)),
                result -> bundleMapper.toBundle(result, request, Some(parameters)));
    }

    @Operation(type = Parameters.class, name = PATIENT_STATUS_QUERY_OPERATION_NAME, idempotent = true)
    public Parameters findStatus(@ResourceParam Parameters parameters, RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        return patientResourceService.findPatientStatuses(authenticatedIdentity, parameters)
                .getOrElseThrow(error -> error);
    }

    @Operation(type = Parameters.class, name = PATIENT_DATE_OF_LAST_DATA_POINT_OPERATION_NAME, idempotent = true)
    public Parameters getDateOfLastDataPoint(@ResourceParam Parameters parameters, RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        return patientResourceService.findLastDataPointDate(authenticatedIdentity, parameters)
                .getOrElseThrow(error -> error);
    }
}
