package com.pkb.api.fhir.resourceprovider;

import ca.uhn.fhir.model.api.annotation.Description;
import ca.uhn.fhir.rest.annotation.RequiredParam;
import ca.uhn.fhir.rest.annotation.Search;
import ca.uhn.fhir.rest.api.server.RequestDetails;
import ca.uhn.fhir.rest.param.ReferenceParam;
import com.pkb.api.fhir.BundleMapper;
import com.pkb.api.fhir.converter.ReferenceParamToUUIDConverter;
import com.pkb.api.fhir.person.PersonResourceService;
import com.pkb.api.fhir.service.AuthenticationService;
import com.pkb.api.fhir.util.Issues;
import com.pkb.api.fhir.util.NamedParam;
import io.vavr.collection.Seq;
import io.vavr.control.Option;
import io.vavr.control.Validation;
import org.hl7.fhir.dstu3.model.Bundle;
import org.hl7.fhir.dstu3.model.Person;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.pkb.api.fhir.util.Fhirs.invalidRequestException;
import static io.vavr.API.Seq;
import static io.vavr.API.Some;
import static org.hl7.fhir.dstu3.model.Person.SP_LINK;
import static org.hl7.fhir.dstu3.model.ResourceType.Practitioner;

public class PersonResourceProvider extends AbstractResourceProvider {
    private static final String SP_LINK_DESCRIPTION = "Required. Multi-value support: none. Modifier-support: [<type>].";

    private final AuthenticationService authenticationService;
    private final PersonResourceService personResourceService;
    private final BundleMapper bundleMapper;
    private final ReferenceParamToUUIDConverter referenceParamConverter;

    public PersonResourceProvider(AuthenticationService authenticationService,
                                  @NotNull PersonResourceService personResourceService,
                                  BundleMapper bundleMapper,
                                  ReferenceParamToUUIDConverter referenceParamConverter) {
        super(Person.class);
        this.authenticationService = authenticationService;
        this.personResourceService = personResourceService;
        this.bundleMapper = bundleMapper;
        this.referenceParamConverter = referenceParamConverter;
    }

    @Search
    @SuppressWarnings({ "WeakerAccess" })
    public Bundle searchPerson(
            @RequiredParam(name = SP_LINK) @Description(value = SP_LINK_DESCRIPTION) ReferenceParam referenceToUser,
            RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        NamedParam<ReferenceParam> linkParam = new NamedParam<>(SP_LINK, referenceToUser, request.getParameters().get(SP_LINK));
        Validation<Issues, UUID> issuesOrLinkId = referenceParamConverter.convertRequired(linkParam, Practitioner);
        Option<Seq<String>> parameters = Some(Seq(linkParam.getName()));
        return issuesOrLinkId.toEither()
                .map(linkId -> personResourceService.findPersonForPractitioner(authenticatedIdentity, linkId))
                .fold(
                        issues -> bundleMapper.toEmptyBundleOrThrowException(invalidRequestException(issues),request, parameters),
                        result -> bundleMapper.toBundle(result, request, parameters));

    }
}
