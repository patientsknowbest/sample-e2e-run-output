package com.pkb.api.fhir.resourceprovider;

import ca.uhn.fhir.rest.annotation.IdParam;
import ca.uhn.fhir.rest.annotation.Read;
import ca.uhn.fhir.rest.api.server.RequestDetails;
import com.pkb.api.fhir.practitioner.PractitionerResourceService;
import com.pkb.api.fhir.service.AuthenticationService;
import org.hl7.fhir.dstu3.model.IdType;
import org.hl7.fhir.dstu3.model.Practitioner;

import static com.pkb.api.fhir.util.Fhirs.notFoundDueToInvalidUUIDException;
import static com.pkb.api.fhir.util.Fhirs.uuid;
import static java.util.function.Function.identity;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity.WARNING;

public class PractitionerResourceProvider extends AbstractResourceProvider {
    private final AuthenticationService authenticationService;
    private final PractitionerResourceService practitionerResourceService;
    
    public PractitionerResourceProvider(AuthenticationService authenticationService, PractitionerResourceService practitionerResourceService) {
        super(Practitioner.class);
        this.authenticationService = authenticationService;
        this.practitionerResourceService = practitionerResourceService;
    }

    @Read
    @SuppressWarnings({ "WeakerAccess" })
    public Practitioner find(@IdParam IdType idType, RequestDetails request) {
        var authenticatedIdentity = authenticationService.authenticateOrThrow(request);
        return uuid(idType.getIdPart(), WARNING)
                .mapLeft(operationOutcome -> notFoundDueToInvalidUUIDException(idType))
                .flatMap(id -> practitionerResourceService.findPractitionerDemographics(authenticatedIdentity, id))
                .getOrElseThrow(identity());
    }
}
