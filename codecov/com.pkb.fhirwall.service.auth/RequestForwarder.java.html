package com.pkb.fhirwall.service.auth;

import com.pkb.fhirwall.domain.ResourceRepository;
import org.hl7.fhir.r4.model.Bundle;

import java.util.stream.Collectors;

public class RequestForwarder {

    private final ResourceRepository resourceRepository;

    public RequestForwarder(ResourceRepository resourceRepository) {
        this.resourceRepository = resourceRepository;
    }

    public Response forward(RequestDetails requestDetails) {
        var result = resourceRepository.searchResource(requestDetails.url());
        throwIfBundleHasMultipleResourceTypes(result, requestDetails.url());
        return ImmutableResponse.builder().resource(result).build();
    }

    /**
     * We only authorized one resource type for the search. Multi-resource-type requests are assumed to be blocked
     * by {@link com.pkb.fhirwall.service.controller.RequestValidator}. This is a safeguard: whenever it throws, it
     * means we overlooked something in the FHIR spec and we should implement blocking it.
     */
    private void throwIfBundleHasMultipleResourceTypes(Bundle bundle, String requestUrl) {
        var distinctResourceTypes = bundle.getEntry().stream()
                .map(entry -> entry.getResource().getResourceType())
                .distinct()
                .sorted()
                .collect(Collectors.toList());
        if (distinctResourceTypes.size() > 1) {
            throw new RuntimeException("Unexpected search result: bundle contains more than one kind of resource " +
                    "type. This request should have been filtered with a 400 response earlier.\n" +
                    "Offending request: " + requestUrl + "\n" +
                    "Result resource types: " + distinctResourceTypes);
        }
    }
}
