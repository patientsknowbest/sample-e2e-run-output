package com.pkb.fhirwall.service.auth;

import com.osohq.oso.Expression;
import com.osohq.oso.Variable;
import com.pkb.fhirwall.service.auth.oso.AuthBase;

import java.util.List;
import java.util.Optional;

import static com.osohq.oso.Operator.And;
import static com.osohq.oso.Operator.Dot;
import static com.osohq.oso.Operator.Isa;
import static com.osohq.oso.Operator.Unify;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static java.lang.String.format;
import static java.util.Optional.empty;
import static java.util.stream.Collectors.toList;

/**
 * Maps constraint expression from Oso engine to FHIR query parameters
 */
public class AuthzFilterMapper {

    List<String> toFilters(List<Expression> constraints) {
        return constraints.stream()
                .map(this::toFilter)
                .flatMap(Optional::stream)
                .collect(toList());
    }

    private Optional<String> toFilter(Expression expression) {
        return Match(expression.getOperator()).of(
                Case($(And), () -> handleAndExpression(expression)),
                Case($(Unify), () -> handleUnifyExpression(expression)),
                Case($(Dot), () -> handleDotExpression(expression)),
                Case($(Isa), ignored -> empty()),
                Case($(), unsupported -> {
                    throw new RuntimeException("Unsupported operator: " + unsupported);
                })
        );
    }

    /**
     * expression1 and expression2 -> filter1&filter2
     *
     * @param expression represents logical 'and' n-ary relation
     * @return filters concatenated with '&'
     */
    private Optional<String> handleAndExpression(Expression expression) {
        return expression.getArgs().stream()
                .filter(arg -> arg instanceof Expression)
                .map(arg -> toFilter((Expression) arg))
                .flatMap(Optional::stream)
                .reduce((filter1, filter2) -> filter1 + "&" + filter2);
    }

    /**
     * object = variable.field -> field=object.id
     * variable.field = object -> field=object.id
     *
     * @param expression represents 'is equal to` binary relation based on resource id
     * @return FHIR query where a field of the requested resource should be equal to the id of the auth object
     */
    private Optional<String> handleUnifyExpression(Expression expression) {
        if (expression.getArgs().size() != 2) {
            throw new RuntimeException("Unify expression must have 2 arguments! Number of args received: "
                    + expression.getArgs().size());
        }

        var firstOperand = expression.getArgs().get(0);
        var secondOperand = expression.getArgs().get(1);
        // object = variable.field -> field=object.id
        if (firstOperand instanceof AuthBase && secondOperand instanceof Expression) {
            return toFilter((Expression) secondOperand)
                    .map(field -> field + "=" + ((AuthBase) firstOperand).id());
        }
        // variable.field = object -> field=object.id
        if (firstOperand instanceof Expression && secondOperand instanceof AuthBase) {
            return toFilter((Expression) firstOperand)
                    .map(field -> field + "=" + ((AuthBase) secondOperand).id());
        }

        throw new RuntimeException(format("Unsupported operands for Unify expression! firstOperand: %s, secondOperand: %s",
                firstOperand, secondOperand));
    }

    /**
     * variable.field -> field
     *
     * @param expression represents a field of an unbound Oso variable
     * @return field of the variable
     */
    private Optional<String> handleDotExpression(Expression expression) {
        if (expression.getArgs().size() != 2) {
            throw new RuntimeException("Dot expression must have 2 arguments! Number of args received: "
                    + expression.getArgs().size());
        }

        var firstOperand = expression.getArgs().get(0);
        var secondOperand = expression.getArgs().get(1);
        if (firstOperand instanceof Variable && secondOperand instanceof String) {
            if (!firstOperand.toString().equals("_this")) {
                throw new RuntimeException("Unsupported Dot operand: " + firstOperand.toString());
            }

            return Optional.of((String) secondOperand);
        }

        throw new RuntimeException(format("Unsupported operands for Dot expression! firstOperand: %s, secondOperand: %s",
                firstOperand, secondOperand));
    }

}
