package com.pkb.fhirwall.service.auth;

import com.google.common.base.Charsets;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import org.springframework.http.server.ServletServerHttpRequest;
import org.springframework.util.LinkedMultiValueMap;
import org.springframework.util.MultiValueMap;
import org.springframework.web.util.UriComponentsBuilder;

import javax.servlet.http.HttpServletRequest;
import java.util.Arrays;
import java.util.Optional;

import static com.pkb.fhirwall.service.util.ResourceIdUtil.removeContext;
import static java.net.URLEncoder.encode;
import static java.util.regex.Pattern.compile;

/**
 * Appends additional filters from authorization rules to query parameters
 */
public class RequestTransformer {

    private final AccessLogger accessLogger;

    public RequestTransformer(AccessLogger accessLogger) {
        this.accessLogger = accessLogger;
    }

    public RequestDetails transform(HttpServletRequest request, AuthzResult authzResult) {
        var filterParameters = createMapFromFilters(authzResult);
        var newParameters = applyQueryTransformations(request, filterParameters);

        var newUrl = UriComponentsBuilder.fromPath(removeContext(request.getRequestURI()))
                .queryParams(newParameters)
                .build()
                .toUriString();

        accessLogger.logTransformedUrl(newUrl);

        return ImmutableRequestDetails.builder()
                .url(newUrl)
                .build();

    }

    private MultiValueMap<String, String> createMapFromFilters(AuthzResult authzResult) {
        var builder = UriComponentsBuilder.newInstance();
        authzResult.filters().forEach(builder::query);
        return builder.build().getQueryParams();
    }

    private MultiValueMap<String, String> applyQueryTransformations(
            HttpServletRequest request,
            MultiValueMap<String, String> newParameters) {

        var originalRequest = UriComponentsBuilder.fromHttpRequest(new ServletServerHttpRequest(request)).build();
        var originalParams = originalRequest.getQueryParams();
        var copy = new LinkedMultiValueMap<>(originalParams);
        copy.remove("_format");

        // If we have the patient resource id in the new parameters, we don't need the chained identifier search
        // any more so we'll remove it (check using encoded and unencoded string, just to be sure)
        if (newParameters.containsKey("patient")) {
            copy.remove("actor:Patient.identifier");
            copy.remove(encode("actor:Patient.identifier", Charsets.UTF_8));
        }

        var sortParam = generateSortParamIfNecessary(request, originalParams);


        var result = new LinkedMultiValueMap<String, String>();
        result.putAll(newParameters);
        result.putAll(copy);
        sortParam.ifPresent(param -> result.add(param._1, param._2));
        return result;
    }

    /*
     * Aidbox adds LIMIT to its queries to the underlying postgres instance, which can become very slow if you don't also
     * sort by something other than ID, because postgres just scans the table if the limit number is small.
     */
    private static Optional<Tuple2<String, String>> generateSortParamIfNecessary(HttpServletRequest request, MultiValueMap<String, String> originalParams) {
        var pattern = compile("\\/fhir\\/(\\w+)");
        var matcher = pattern.matcher(request.getRequestURI());
        if (matcher.find() && !originalParams.containsKey("_sort")) {
            var resource = matcher.group(1); //the 0th group is the entire string
            return Arrays.stream(SupportedResources.values())
                    .filter(supportedResource -> supportedResource.toString().equalsIgnoreCase(resource))
                    .findFirst().map(supportedResource -> Tuple.of("_sort", supportedResource.getDefaultSortField()));
        } else {
            return Optional.empty();
        }
    }
}
