package com.pkb.fhirwall.service.auth;

import ca.uhn.fhir.context.FhirContext;
import com.pkb.fhirwall.service.resource.ResourceEnricherManager;
import com.pkb.fhirwall.service.resource.wayfinder.ResourceRemovalManager;
import org.hl7.fhir.instance.model.api.IBaseResource;
import org.slf4j.Logger;

import javax.servlet.http.HttpServletResponse;
import java.io.IOException;

import static java.lang.invoke.MethodHandles.lookup;
import static javax.servlet.http.HttpServletResponse.SC_INTERNAL_SERVER_ERROR;
import static org.slf4j.LoggerFactory.getLogger;
import static org.springframework.http.HttpHeaders.CONTENT_TYPE;

public class ResponseTransformer {

    private static final Logger LOG = getLogger(lookup().lookupClass());

    private final FhirContext fhirContext;
    private final ResourceEnricherManager resourceEnricherManager;
    private final ResourceRemovalManager resourceRemovalManager;

    public ResponseTransformer(FhirContext fhirContext, ResourceEnricherManager resourceEnricherManager,
                               ResourceRemovalManager resourceRemovalManager) {
        this.fhirContext = fhirContext;
        this.resourceEnricherManager = resourceEnricherManager;
        this.resourceRemovalManager = resourceRemovalManager;
    }

    public void transform(Response fhirResponse, HttpServletResponse response, int status) {
        setResponseHeaderAndStatus(response, status);

        resourceEnricherManager.enrichResource(fhirResponse.resource());

        fhirResponse = resourceRemovalManager.applyRemovalRules(fhirResponse);

        writeResourceAsFhirJson(fhirResponse.resource(), response);
    }

    private void setResponseHeaderAndStatus(HttpServletResponse response, int status) {
        response.setStatus(status);
        response.setHeader(CONTENT_TYPE, "application/fhir+json");
    }

    private void writeResourceAsFhirJson(IBaseResource resource, HttpServletResponse response) {
        try {
            fhirContext.newJsonParser().encodeResourceToWriter(resource, response.getWriter());
        } catch (IOException e) {
            LOG.error("Unexpected error returning target resource: {}", e.getMessage());
            response.setStatus(SC_INTERNAL_SERVER_ERROR);
        }
    }
}
