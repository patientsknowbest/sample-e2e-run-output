package com.pkb.fhirwall.service.auth;

/*
 * The queries Aidbox constructs against postgres include a LIMIT clause, which can be really slow
 * if the table is big enough because postgres just scans the table instead of using the indexes.
 * It does not do this if you sort by something which is not ID, so we include a sort parameter if the client has not
 * supplied one.
 */
public enum SupportedResources {
    APPOINTMENT("-date");
    private final String defaultSortField;

    SupportedResources(String defaultSortField) {
        this.defaultSortField = defaultSortField;
    }

    public String getDefaultSortField() {
        return defaultSortField;
    }
}
