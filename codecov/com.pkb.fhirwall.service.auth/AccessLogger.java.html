package com.pkb.fhirwall.service.auth;

import org.slf4j.Logger;

import javax.servlet.http.HttpServletRequest;

import static java.lang.invoke.MethodHandles.lookup;
import static org.slf4j.LoggerFactory.getLogger;

public class AccessLogger {

    private static final Logger LOG = getLogger(lookup().lookupClass());

    public void logRequest(HttpServletRequest request) {
        LOG.info("Request for URL: {}", request.getRequestURL());
    }

    public void logTransformedUrl(String components) {
        LOG.info("Request for target FHIR: {}", components);
    }

    public void logError(String errorMessage) {
        LOG.error("Error {}", errorMessage);
    }

    public void logWarn(String errorMessage) {
        LOG.warn("Warn: {}", errorMessage);
    }
}
