package com.pkb.fhirwall.service.auth;

import org.hl7.fhir.exceptions.FHIRException;
import org.hl7.fhir.r4.model.ResourceType;
import org.jetbrains.annotations.NotNull;

import javax.servlet.http.HttpServletRequest;
import java.util.Optional;

public class AccessRequestParser {

    @NotNull
    public AccessRequest parse(HttpServletRequest request) {
        var subject = createSubjectFromRequest(request);
        var object = createObjectFromRequest(request);
        return ImmutableAccessRequest.builder()
                .subject(subject)
                .object(object)
                // constant for now, until we add support for other methods
                // later: HAPI FHIR must already implement this somewhere; it takes a HttpServletRequest and creates a
                // RequestDetails which includes the FHIR operation type.
                .operation(Operation.SEARCH)
                .build();
    }

    @NotNull
    private Subject createSubjectFromRequest(HttpServletRequest request) {
        return Optional.ofNullable(request.getAttribute("NHS_NUMBER"))
                .map(nhsNumber -> ImmutableSubject.builder()
                        .id(nhsNumber.toString())
                        .subjectIdType(SubjectIdType.NHS_NUMBER)
                        .build())
                .orElseThrow(() -> new RuntimeException("Only nhs number based authorization is supported"));
    }

    private ResourceType createObjectFromRequest(HttpServletRequest request) {
        var pathSegments = request.getRequestURI().split("/");
        if (pathSegments.length >= 3 && pathSegments[1].equals("fhir")) {
            var resourceCode = pathSegments[2];
            try {
                return ResourceType.fromCode(resourceCode);
            } catch (FHIRException ignored) {
                // unknown fhir type, handled by throwing uri details below
            }
        }
        throw new RuntimeException("Unexpected requestURI " + request.getRequestURI());
    }
}
