package com.pkb.fhirwall.service.auth;

import com.pkb.fhirwall.domain.PatientRepository;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.hl7.fhir.r4.model.Patient;

import static com.pkb.fhirwall.service.auth.Operation.SEARCH;
import static com.pkb.fhirwall.service.auth.PolicyInformationFailure.SUBJECT_NOT_FOUND;
import static com.pkb.fhirwall.service.auth.PolicyInformationFailure.UNSUPPORTED_OPERATION;
import static com.pkb.fhirwall.service.auth.SubjectIdType.NHS_NUMBER;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;

/**
 * Retrieves the data required for evaluating authorization rules for a FHIR resource.
 */
public class FhirPolicyInformationPoint implements PolicyInformationPoint {
    private final PatientRepository patientRepository;
    private final AuthzMapper authzMapper;

    public FhirPolicyInformationPoint(PatientRepository patientRepository, AuthzMapper authzMapper) {
        this.patientRepository = patientRepository;
        this.authzMapper = authzMapper;
    }

    @Override
    public Either<PolicyInformationFailure, ? extends PolicyInformation> fetchPolicyInformation(AccessRequest accessRequest) {
        return Match(accessRequest.operation()).of(
                Case($(SEARCH), op -> fetchSearchInformation(accessRequest)),
                Case($(), Either.left(UNSUPPORTED_OPERATION)));
    }

    private Either<PolicyInformationFailure, SearchPolicyInformation> fetchSearchInformation(AccessRequest accessRequest) {
        return fetchPatient(accessRequest)
                .map(authzMapper::mapWithConsent)
                .map(subject -> ImmutableSearchPolicyInformation.builder()
                        .subject(subject)
                        .object(accessRequest.object())
                        .build());
    }

    private Either<PolicyInformationFailure, Patient> fetchPatient(AccessRequest accessRequest) {
        return Match(accessRequest.subject().subjectIdType()).of(
                Case($(NHS_NUMBER), type -> findPatientByNhsNumber(accessRequest.subject().id())),
                Case($(), Either.left(UNSUPPORTED_OPERATION))
        );
    }

    private Either<PolicyInformationFailure, Patient> findPatientByNhsNumber(String nhsNumber) {
        var patients = patientRepository.findPatientByNhsNumber(nhsNumber);

        if (patients.size() > 1) {
            throw new RuntimeException("Multiple patients found for nhsNumber: " + nhsNumber);
        }

        return Option.ofOptional(patients.stream().findAny())
                .toEither(SUBJECT_NOT_FOUND);
    }
}
