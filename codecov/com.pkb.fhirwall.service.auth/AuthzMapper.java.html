package com.pkb.fhirwall.service.auth;

import com.pkb.fhirwall.domain.ConsentRepository;
import com.pkb.fhirwall.domain.PatientRepository;
import com.pkb.fhirwall.service.auth.oso.AppointmentAuth;
import com.pkb.fhirwall.service.auth.oso.AuthBase;
import com.pkb.fhirwall.service.auth.oso.PatientAuth;
import org.hl7.fhir.instance.model.api.IBaseResource;
import org.hl7.fhir.instance.model.api.IIdType;
import org.hl7.fhir.r4.model.Appointment;
import org.hl7.fhir.r4.model.Appointment.AppointmentParticipantComponent;
import org.hl7.fhir.r4.model.Bundle;
import org.hl7.fhir.r4.model.Bundle.BundleEntryComponent;
import org.hl7.fhir.r4.model.CodeableConcept;
import org.hl7.fhir.r4.model.Consent;
import org.hl7.fhir.r4.model.Patient;
import org.hl7.fhir.r4.model.Reference;

import java.util.Collection;
import java.util.List;

import static com.pkb.entities.enums.FhirConsentSource.OPTOUT;
import static com.pkb.entities.enums.FhirConsentSource.PATIENTCONSENT;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.Predicates.instanceOf;
import static java.lang.String.format;
import static java.util.Collections.emptyList;
import static java.util.Optional.empty;
import static java.util.stream.Collectors.toList;

/**
 * Maps FHIR resources to types suitable for consumption by authorization policies
 */
public class AuthzMapper {

    private static final String PATIENT = new Patient().fhirType();

    private final PatientRepository patientRepository;
    private final ConsentRepository consentRepository;

    public AuthzMapper(PatientRepository patientRepository, ConsentRepository consentRepository) {
        this.patientRepository = patientRepository;
        this.consentRepository = consentRepository;
    }

    public AuthBase mapWithConsent(Patient patient) {
        var consentsBundle = consentRepository.getConsentsByPatient(patient.getId());
        var consentResources = getConsentsByCode(consentsBundle,PATIENTCONSENT.name());
        var optOutResources = getConsentsByCode(consentsBundle, OPTOUT.name());
        if (optOutResources.size() > 1) {
            throw new RuntimeException("Multiple opt-out consents found for patient " + patient.getId());
        }

        return PatientAuth.of(patient, consentResources, optOutResources.stream().findFirst());
    }

    public AuthBase toAuth(IBaseResource resource) {
        return Match(resource).of(
                Case($(instanceOf(Patient.class)), patient -> PatientAuth.of(patient, emptyList(), empty())),
                Case($(instanceOf(Appointment.class)), this::mapAppointment),
                Case($(), o -> {
                    throw new RuntimeException("Unknown resource type: " + resource.fhirType());
                }));
    }

    private AppointmentAuth mapAppointment(Appointment appointment) {
        var patientIds = appointment.getParticipant().stream()
                .map(AppointmentParticipantComponent::getActor)
                .map(Reference::getReferenceElement)
                .filter(referenceElement -> referenceElement.getResourceType().equals(PATIENT))
                .map(IIdType::getIdPart)
                .collect(toList());
        if (patientIds.size() > 1) {
            throw new RuntimeException("Multiple patients are referenced from Appointment " + appointment.getId());
        }
        if (patientIds.isEmpty()) {
            return AppointmentAuth.withoutPatient(appointment);
        }

        return patientRepository.getPatientByFhirId(patientIds.get(0))
                .map(patient -> AppointmentAuth.of(appointment, patient))
                .orElseThrow(() -> new RuntimeException(
                        format("Patient %s not found for Appointment %s", patientIds.get(0), appointment.getId())));
    }

    private List<Consent> getConsentsByCode(Bundle bundle, String code) {
        return bundle.getEntry().stream()
                .map(BundleEntryComponent::getResource)
                .filter(resource -> resource instanceof Consent)
                .map(resource -> (Consent) resource)
                .filter(consent -> filterByCode(consent, code))
                .collect(toList());
    }

    private boolean filterByCode(Consent consent, String code) {
        return consent.getCategory().stream()
                .map(CodeableConcept::getCoding)
                .flatMap(Collection::stream)
                .anyMatch(coding -> code.equals(coding.getCode()));
    }
}
