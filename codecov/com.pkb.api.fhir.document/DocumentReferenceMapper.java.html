package com.pkb.api.fhir.document;

import com.pkb.api.fhir.CanonicalCodeSystemURLs;
import com.pkb.api.fhir.CanonicalExtensionURLs;
import com.pkb.api.fhir.PrivacyFlagsMapper;
import com.pkb.api.fhir.SpecialtyMapper;
import com.pkb.api.fhir.TemporalMapper;
import com.pkb.api.fhir.foundation.ReferenceMapper;
import com.pkb.datamodel.DocumentMessage;
import com.pkb.datamodel.Org;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.criteria.ImmutableCriteriaResult;
import io.vavr.collection.List;
import org.hl7.fhir.dstu3.model.Attachment;
import org.hl7.fhir.dstu3.model.CodeableConcept;
import org.hl7.fhir.dstu3.model.Coding;
import org.hl7.fhir.dstu3.model.DocumentReference;
import org.hl7.fhir.dstu3.model.DocumentReference.DocumentReferenceContentComponent;
import org.hl7.fhir.dstu3.model.Extension;
import org.hl7.fhir.dstu3.model.Identifier;
import org.hl7.fhir.dstu3.model.Meta;
import org.hl7.fhir.dstu3.model.Reference;
import org.jetbrains.annotations.NotNull;

import java.util.Collections;

import static com.pkb.api.fhir.DataAbsentReasonUtil.missingInstantType;
import static io.vavr.API.List;
import static org.hl7.fhir.dstu3.model.Enumerations.DocumentReferenceStatus.CURRENT;
import static org.hl7.fhir.dstu3.model.Enumerations.DocumentReferenceStatus.SUPERSEDED;

public class DocumentReferenceMapper {
    private final ReferenceMapper referenceMapper;
    private final TemporalMapper timeMapper;
    private final PrivacyFlagsMapper privacyFlagsMapper;
    private final SpecialtyMapper specialtyMapper;

    public DocumentReferenceMapper(ReferenceMapper referenceMapper,
                                   TemporalMapper timeMapper,
                                   PrivacyFlagsMapper privacyFlagsMapper,
                                   SpecialtyMapper specialtyMapper) {
        this.referenceMapper = referenceMapper;
        this.timeMapper = timeMapper;
        this.privacyFlagsMapper = privacyFlagsMapper;
        this.specialtyMapper = specialtyMapper;
    }

    public @NotNull CriteriaResult<DocumentReference> toCriteriaResultDocumentReferences(@NotNull List<DocumentMessage> messages) {
        return ImmutableCriteriaResult.<DocumentReference> builder()
                .addAllPage(messages.map(this::toDocumentReference))
                .total(messages.size())
                .build();
    }

    public @NotNull List<DocumentReference> toDocumentReferences(List<DocumentMessage> messages) {
        return messages.map(this::toDocumentReference);
    }

    public @NotNull DocumentReference toDocumentReference(DocumentMessage message) {
        DocumentReference documentReference = new DocumentReference();

        if (message.isCurrent()) {
            documentReference.setStatus(CURRENT);
        } else {
            documentReference.setStatus(SUPERSEDED);
        }

        documentReference.setId(message.getUniqueId().toString());
        documentReference.setType(toType(message.getType()));
        documentReference.setSubject(referenceMapper.toReferenceWithName(message.getPatient()));
        documentReference.setCreated(timeMapper.toDate(message.getCreated()));
        documentReference.setIndexedElement(message
                .getPersisted()
                .map(timeMapper::toInstantTypeInUTC)
                .getOrElse(missingInstantType()));

        documentReference.setContent(extractDocumentReferenceContentComponent(message).asJava());
        documentReference.setMeta(extractMetadata(message));
        message.getExternalMessageId().peek(emid -> documentReference.addIdentifier(getExternalMessageIdentifier(message, emid)));
        message.getSpecialty().peek(specialty -> documentReference.addExtension(specialtyMapper.toExtension(specialty)));
        return documentReference;
    }

    private CodeableConcept toType(DocumentMessage.DocumentType documentType) {
        CodeableConcept codeableConcept = new CodeableConcept();

        Coding coding = codeableConcept.addCoding().setSystem(CanonicalCodeSystemURLs.documentType());
        coding.setCode(documentType.name());
        return codeableConcept;
    }

    private List<DocumentReferenceContentComponent> extractDocumentReferenceContentComponent(DocumentMessage message) {
        DocumentReferenceContentComponent contentComponent = new DocumentReferenceContentComponent();
        Attachment att = new Attachment();
        att.setCreation(timeMapper.toDate(message.getCreated()));
        contentComponent.setAttachment(att);
        return List(contentComponent);
    }

    private Meta extractMetadata(DocumentMessage message) {

        Coding statusCode = new Coding();
        statusCode.setCode(message.getStatus().name());
        statusCode.setSystem(CanonicalCodeSystemURLs.documentStatus());

        Extension extension = new Extension();
        extension.setUrl(CanonicalExtensionURLs.documentStatus());
        extension.setValue(statusCode);

        Meta meta = new Meta();
        message.getPersisted().map(timeMapper::toDate).peek(meta::setLastUpdated);
        meta.setExtension(Collections.singletonList(extension));
        meta.setSecurity(privacyFlagsMapper.toCodingList(message.getPrivacyFlags()).asJava());

        return meta;
    }

    private Identifier getExternalMessageIdentifier(DocumentMessage message, String externalMessageId) {
        Org messageSourceOrg = message.getSource().getOrg()
                .getOrElseThrow(() -> new IllegalStateException("Org must be present when there is an externalMessageId"));
        Reference assigner = referenceMapper.toReference(messageSourceOrg);
        assigner.setDisplay(messageSourceOrg.getName());

        Identifier id = new Identifier();
        id.setAssigner(assigner);
        id.setValue(externalMessageId);

        return id;
    }

}
