package com.pkb.service.passwordreset;

import com.pkb.datamodel.Email;
import com.pkb.exception.MailException;
import com.pkb.service.emailmessage.ITemplateParamsFactory;
import com.pkb.service.emailmessage.impl.AddressInfo;
import com.pkb.service.emailmessage.impl.EmailManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.PKBConstants;
import org.jetbrains.annotations.NotNull;

import static com.pkb.entities.enums.Message.FORGOT_PASSWORD;

public class PasswordResetEmailSender {
    private final EmailManager emailManager;
    private final ITemplateParamsFactory templateParamsFactory;

    public PasswordResetEmailSender(
            @NotNull EmailManager emailManager,
            @NotNull ITemplateParamsFactory templateParamsFactory) {
        this.emailManager = emailManager;
        this.templateParamsFactory = templateParamsFactory;
    }

    public void sendPasswordResetLinkTo(PKBPerson user, String resetPasswordLink) {
        Email email = user.getEmail();
        if(email == null) {
            throw new RuntimeException("User has no email address, id-" + user.getId());
        }

        try {
            var properties = templateParamsFactory.createWithSiteProps()
                    .param(PKBConstants.TITLE, user.getTitle())
                    .param(PKBConstants.FIRST_NAME, user.getFirstName())
                    .param(PKBConstants.LAST_NAME, user.getLastName())
                    .param("resetPasswordLink", resetPasswordLink);

            emailManager.sendMessage(FORGOT_PASSWORD, user, new AddressInfo(email.toInternetAddress()), user, properties,
                    user.getLocale(), null, user.getId().toString(), null, null);
        } catch (MailException e) {
            throw new RuntimeException(e);
        }
    }
}
