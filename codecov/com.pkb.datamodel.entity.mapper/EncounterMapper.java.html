package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.AccountKeys;
import com.pkb.datamodel.AppMenuDataWithSourceObjects;
import com.pkb.datamodel.Encounter;
import com.pkb.datamodel.EncounterEvent;
import com.pkb.datamodel.ReferenceDatumCache;
import com.pkb.datamodel.crypto.CryptoService;
import com.pkb.datamodel.entity.ClinicalDataDeserializer;
import com.pkb.datamodel.user.Patient;
import com.pkb.entities.app.AppMenuData;
import com.pkb.entities.embeddables.SourceDetails;
import com.pkb.entities.enums.EncounterClass;
import com.pkb.entities.pub.PublicOrg;
import io.prometheus.client.Histogram;
import io.vavr.Function1;
import io.vavr.Function4;
import io.vavr.collection.List;
import io.vavr.collection.Map;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;

import javax.inject.Inject;
import java.util.UUID;

import static com.pkb.datamodel.ImmutableEncounter.encounter;
import static com.pkb.datamodel.ImmutableEncounterEvent.encounterEvent;

@Mapper
public abstract class EncounterMapper implements StringFieldMapper, TimedMapper {

    @Inject
    private SourceDetailsMapper sourceDetailsMapper;

    @Inject
    private PrivacyFlagsMapper privacyFlagsMapper;

    @Inject
    private LocationMapper locationMapper;

    @Inject
    private ParticipantMapper participantMapper;

    @Inject
    private CryptoService cryptoService;

    @Inject
    private ClinicalDataDeserializer dataDeserializer;

    @Inject
    private InstantMapper instantMapper;

    public List<Encounter> menuDatasToEncounters(
            @NotNull AccountKeys accountKeys,
            @NotNull List<AppMenuDataWithSourceObjects> menuData,
            @NotNull Patient patient,
            @NotNull ReferenceDatumCache referenceDatumCache) {
        Map<Long, byte[]> decryptedData = cryptoService.decrypt(accountKeys, menuData.map(AppMenuDataWithSourceObjects::getAppMenuData));
        return menuData
                .groupBy(wrapper -> wrapper.getAppMenuData().getUuid01()).toList()
                .map(e -> menuDatasToEncounter(e._1, e._2, patient, decryptedData, referenceDatumCache));
    }

    private Encounter menuDatasToEncounter(
            @NotNull UUID conversationId,
            @NotNull List<AppMenuDataWithSourceObjects> mds,
            @NotNull Patient patient,
            @NotNull Map<Long, byte[]> decryptedData,
            @NotNull ReferenceDatumCache referenceDatumCache) {

        Function1<AppMenuDataWithSourceObjects, EncounterEvent> toEncounterEvent = Function4.of(this::menuDataToEncounterEvent)
                .apply(patient)
                .apply(decryptedData)
                .apply(referenceDatumCache);

        try (Histogram.Timer ignored = MAP_TIMER.labels("encounter", "encounter").startTimer()) {
            return encounter()
                    .id(conversationId)
                    .addAllEvents(mds.map(toEncounterEvent).sortBy(EncounterEvent::timestamp))
                    .build();
        }
    }

    private EncounterEvent menuDataToEncounterEvent(
            Patient patient,
            Map<Long, byte[]> decryptedData,
            ReferenceDatumCache referenceDatumCache,
            AppMenuDataWithSourceObjects wrapper) {

        AppMenuData md = wrapper.getAppMenuData();

        try (Histogram.Timer ignored = MAP_TIMER.labels("encounter", "event").startTimer()) {
            byte[] plaintext = decryptedData.get(md.getId()).get();
            java.util.Map<String, Object> fields = dataDeserializer.deserialize(plaintext);
            java.util.Map<String, Object> location = (java.util.Map<String, Object>) fields.getOrDefault("location", null);
            java.util.List<java.util.Map<String, Object>> externalParticipants = (java.util.List<java.util.Map<String, Object>>) fields.getOrDefault("participantsNonPKB", null);
            Option<PublicOrg> org = Option.of(md.getSourceDetails()).map(SourceDetails::getOrg);
            return encounterEvent()
                    .id(md.getBaseFields().getUniqueId())
                    .patient(patient)
                    .persisted(instantMapper.mapToZonedDateTimeOption(md.getBaseFields().getPersistedDate()))
                    .timestamp(instantMapper.mapToZonedDateTime(md.getDate01()))
                    .accountId(md.getAccountId())
                    .privacyFlags(privacyFlagsMapper.entityToDataModel(md.getBaseFields()))
                    .source(sourceDetailsMapper.entityToDataModel(md.getSourceDetails(), wrapper.getSourcePerson().getOrNull()))
                    .encounterClass(EncounterClass.valueOf(md.getString09()))
                    .location(locationMapper.mapLocationFromFields(location))
                    .addAllExternalParticipants(participantMapper.mapExternalParticipants(externalParticipants))
                    .specialty(referenceDatumCache.lookup(org, Option.of((String)fields.get("hospitalServiceCode"))))
                    .build();
        }
    }
}
