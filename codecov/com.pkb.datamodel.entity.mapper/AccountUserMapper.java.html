package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.AccountUser;
import com.pkb.datamodel.ImmutableAccountUser;
import com.pkb.entities.core.CoreAccountUser;
import io.vavr.control.Option;
import org.mapstruct.Mapper;

@Mapper
public class AccountUserMapper {

    public AccountUser entityToDataModel(CoreAccountUser entity) {
        return ImmutableAccountUser.builder()
                .id(entity.getId())
                .defaultAccount(entity.isDefaultAccount())
                .accountId(entity.getAccountId())
                .status(entity.getStatus())
                .effectiveDate(entity.getEffectiveDate())
                .expirationDate(Option.of(entity.getEffectiveDate()))
                .accountPrivateKey(entity.getAccountPrivateKey())
                .userPersonId(entity.getUserPersonId())
                .build();
    }

    public CoreAccountUser dataModelToEntity(AccountUser dataModel) {
        CoreAccountUser result = new CoreAccountUser();
        result.setId(dataModel.getId());
        if (dataModel.getExpirationDate().isDefined()) {
            result.setExpirationDate(dataModel.getExpirationDate().get());
        }
        result.setEffectiveDate(dataModel.getEffectiveDate());
        result.setAccountPrivateKey(dataModel.getAccountPrivateKey());
        result.setAccountId(dataModel.getAccountId());
        result.setStatus(dataModel.getStatus());
        result.setDefaultAccount(dataModel.isDefaultAccount());
        result.setUserPersonId(dataModel.getUserPersonId());
        return result;
    }
}
