package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutableOrgLevelId;
import com.pkb.datamodel.OrgLevelId;
import com.pkb.entities.pub.PublicOrgLevelId;
import io.prometheus.client.Histogram;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.mapstruct.Mapper;

import javax.inject.Inject;
import java.time.ZoneOffset;
import java.time.ZonedDateTime;
import java.util.Objects;

@Mapper(uses = OrgLevelIdTypeMapper.class)
public abstract class OrgLevelIdMapper implements TimedMapper {
    @Inject
    private OrgLevelIdTypeMapper typeMapper;

    public OrgLevelId entityToDataModel(PublicOrgLevelId entity) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("org_level_id", "org_level_id").startTimer()) {
            return ImmutableOrgLevelId.builder()
                    .id(entity.getId())
                    .type(typeMapper.entityToDataModel(entity.getType()))
                    .build()
                    .withValue(entity.getValue())
                    .withEditDate(Option.of(entity.getEditDate()).map(it -> ZonedDateTime.ofInstant(it, ZoneOffset.UTC)));
        }
    }

    public List<OrgLevelId> entityToDataModel(List<PublicOrgLevelId> entities) {
        return entities.map(this::entityToDataModel);
    }

    /**
     * Maps {@link PublicOrgLevelId} entities to {@link OrgLevelId} domain objects, but excludes any that do not belong
     * to the specified org. Use this if you have overfetched org level ids and want to filter them.
     */
    public List<OrgLevelId> entityToDataModel(List<PublicOrgLevelId> entities, long orgId) {
        return entities
                .filter(olid -> Objects.equals(olid.getOrg().getId(), orgId))
                .map(this::entityToDataModel);
    }
}
