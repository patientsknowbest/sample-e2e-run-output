package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutableOrgLevelIdType;
import com.pkb.datamodel.OrgLevelIdType;
import com.pkb.entities.pub.PublicOrgLevelIdType;
import io.prometheus.client.Histogram;
import org.mapstruct.Mapper;

import javax.inject.Inject;

@Mapper(uses = OrgMapper.class)
public abstract class OrgLevelIdTypeMapper implements TimedMapper {

    @Inject
    private OrgMapper orgMapper;

    public OrgLevelIdType entityToDataModel(PublicOrgLevelIdType entity) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("org_level_id_type", "org_level_id_type").startTimer()) {
            return ImmutableOrgLevelIdType.builder()
                    .id(entity.getId())
                    .name(entity.getName())
                    .publicId(entity.getPublicId())
                    .assigningAuthority(entity.getHl7AssigningAuthority())
                    .typeCode(entity.getHl7TypeCode())
                    .org(orgMapper.entityToDataModel(entity.getOrg()))
                    .build();
        }
    }
}
