package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.authorization.ConsentedEntity;
import com.pkb.datamodel.consent.ImmutablePatientConsent;
import com.pkb.datamodel.consent.PatientConsent;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.entities.pub.PublicPKBPerson;
import com.pkb.entities.pub.PublicPatientConsent;
import com.pkb.entities.pub.PublicTeam;
import com.pkb.entities.pub.PublicViewConsent;
import io.prometheus.client.Histogram;
import io.vavr.collection.List;
import io.vavr.control.Either;
import org.mapstruct.Mapper;

import javax.inject.Inject;

import static com.pkb.entities.enums.AccessingEntityType.TEAM;
import static io.vavr.API.List;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

@Mapper
public abstract class PatientConsentMapper implements TimedMapper {

    @Inject
    private PersonMapper personMapper;

    @Inject
    private TeamWithOrgMapper teamWithOrgMapper;

    public PatientConsent patientConsentToDataModel(PublicPatientConsent patientConsent) {
        Either<PublicTeam, PublicPKBPerson> teamOrPerson = patientConsent.getAccessingEntityType() == TEAM ?
                left(patientConsent.getAccessingTeam()) :
                right(patientConsent.getAccessingPerson());

        try (Histogram.Timer ignored = MAP_TIMER.labels("patient_consent", "patient_consent").startTimer()) {
            return ImmutablePatientConsent.builder()
                    .publicId(patientConsent.getUniqueId())
                    .isProEditOnlyForChild(patientConsent.isProEditOnlyForChild())
                    .isDischarged(patientConsent.isDischarged())
                    .patient(personMapper.entityToPatient(patientConsent.getPatient()))
                    .consentedEntity(getConsentedEntity(teamOrPerson))
                    .build()
                    .withConsentsGranted(getConsentFlags(patientConsent));
        }
    }

    public PatientConsent viewConsentToDataModel(PublicViewConsent viewConsent) {
        Either<PublicTeam, PublicPKBPerson> teamOrPerson = viewConsent.getAccessingEntityType() == TEAM ?
                left(viewConsent.getAccessingTeam()) :
                right(viewConsent.getAccessingPerson());

        try (Histogram.Timer ignored = MAP_TIMER.labels("patient_consent", "view_consent").startTimer()) {
            return ImmutablePatientConsent.builder()
                    .publicId(viewConsent.getUniqueId())
                    .isProEditOnlyForChild(viewConsent.isProEditOnlyForChild())
                    .isDischarged(viewConsent.isDischarged())
                    .patient(personMapper.entityToPatient(viewConsent.getPatient()))
                    .consentedEntity(getConsentedEntity(teamOrPerson))
                    .build().withConsentsGranted(List.of(viewConsent.getPrivacyFlags()));
        }
    }

    private ConsentedEntity getConsentedEntity(Either<PublicTeam, PublicPKBPerson> teamOrPerson) {
        return teamOrPerson.fold(
                teamWithOrgMapper::entityToDataModel,
                personMapper::entityToDataModel);
    }

    private io.vavr.collection.List<PrivacyFlag> getConsentFlags(PublicPatientConsent patientConsent) {
        List<PrivacyFlag> consentsGranted = List();
        if (patientConsent.isGeneralHealthConsentGranted()) {
            consentsGranted = consentsGranted.append(PrivacyFlag.GENERAL);
        }
        if (patientConsent.isMentalHealthConsentGranted()) {
            consentsGranted = consentsGranted.append(PrivacyFlag.MENTAL_HEALTH);
        }
        if (patientConsent.isSexualHealthConsentGranted()) {
            consentsGranted = consentsGranted.append(PrivacyFlag.SEXUAL_HEALTH);
        }
        if (patientConsent.isSocialCareConsentGranted()) {
            consentsGranted = consentsGranted.append(PrivacyFlag.SOCIAL_CARE);
        }
        return consentsGranted;
    }

    public void setPersonMapper(PersonMapper personMapper) {
        this.personMapper = personMapper;
    }

    public void setTeamWithOrgMapper(TeamWithOrgMapper teamWithOrgMapper) {
        this.teamWithOrgMapper = teamWithOrgMapper;
    }
}
