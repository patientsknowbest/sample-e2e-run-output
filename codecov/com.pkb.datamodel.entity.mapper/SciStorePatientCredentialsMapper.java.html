package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutableSciStorePatientCredentials;
import com.pkb.datamodel.SciStorePatientCredentials;
import com.pkb.entities.pub.PublicSciStorePatientCredentials;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

@Mapper
public abstract class SciStorePatientCredentialsMapper implements OptionAwareMapper, TimedMapper {

    public @NotNull SciStorePatientCredentials entityToDataModel(@NotNull PublicSciStorePatientCredentials entity){
        return ImmutableSciStorePatientCredentials.builder()
                .pkbPatientId(entity.getPkbPatientId())
                .sciStorePatientId(entity.getSciStorePatientId())
                .sciStoreSubscriptionId(entity.getSciStoreSubscriptionId())
                .status(entity.getStatus())
                .build();
    }

    @Mapping(target = "id", ignore = true)
    abstract PublicSciStorePatientCredentials toPublicSciStorePatientCredentials(SciStorePatientCredentials sciStorePatientCredentials);

    public PublicSciStorePatientCredentials newDataModelToNewEntity(@NotNull SciStorePatientCredentials sciStorePatientCredentials){
        return toPublicSciStorePatientCredentials(sciStorePatientCredentials);
    }
}
