package com.pkb.datamodel.entity.mapper;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.ImmutableInvitationRepairJobExisting;
import com.pkb.datamodel.InvitationRepairJobExisting;
import com.pkb.datamodel.InvitationRepairJobNew;
import com.pkb.entities.pub.PublicInvitationRepairJob;
import io.vavr.control.Option;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

import javax.inject.Inject;

@Mapper
public abstract class InvitationRepairJobMapper implements TimedMapper, OptionAwareMapper {

    @Inject
    protected DateTimeService dateTimeService;

    @Mapping(target = "id", ignore = true) // assigned when persisted
    @Mapping(target = "created", expression = "java(dateTimeService.nowLocalDateTime())")
    @Mapping(target = "statusDate", expression = "java(dateTimeService.nowLocalDateTime())")
    public abstract PublicInvitationRepairJob dataModelToEntity(InvitationRepairJobNew job);

    public InvitationRepairJobExisting entityToDataModel(PublicInvitationRepairJob entity) {
        return ImmutableInvitationRepairJobExisting.builder()
                .id(entity.getId())
                .patientId(entity.getPatientId())
                .patientAccountPublicId(entity.getPatientAccountPublicId())
                .patientPublicId(entity.getPatientPublicId())
                .teamId(entity.getTeamId())
                .created(entity.getCreated())
                .status(entity.getStatus())
                .statusDate(entity.getStatusDate())
                .statusExtraText(Option.of(entity.getStatusExtraText()))
                .build();
    }
    
}
