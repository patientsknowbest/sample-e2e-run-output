package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.consent.ConsentExpiryDecision;
import com.pkb.datamodel.consent.ConsentExpiryDecisionExisting;
import com.pkb.datamodel.consent.ImmutableConsentExpiryDecisionExisting;
import com.pkb.entities.pub.PublicConsentExpiryDecision;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

@Mapper
public abstract class ConsentExpiryDecisionMapper implements OptionAwareMapper {

    public @NotNull ConsentExpiryDecisionExisting entityToDataModel(@NotNull PublicConsentExpiryDecision entity){
        return ImmutableConsentExpiryDecisionExisting.builder()
                .id(entity.getId())
                .consentId(entity.getConsentId())
                .patientAgeCriterion(Option.of(entity.getPatientAgeCriterion()))
                .expires(entity.isExpires())
                .enteredBy(entity.getEnteredBy())
                .enteredDate(entity.getEnteredDate())
                .consentDeletedAt(Option.of(entity.getConsentDeletedAt()))
                .reason(Option.of(entity.getReason()))
                .build();
    }

    @Mapping(target = "id")
    public abstract PublicConsentExpiryDecision dataModelToEntity(@NotNull ConsentExpiryDecisionExisting consentExpiryDecision);

    @Mapping(target = "id", ignore = true)
    public abstract PublicConsentExpiryDecision dataModelToEntity(@NotNull ConsentExpiryDecision consentExpiryDecision);
}
