package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutableSource;
import com.pkb.datamodel.Source;
import com.pkb.entities.embeddables.SourceDetails;
import com.pkb.entities.pub.PublicPKBPerson;
import io.prometheus.client.Histogram;
import org.jetbrains.annotations.Nullable;
import org.mapstruct.Mapper;

import javax.inject.Inject;

import static java.util.Objects.nonNull;

@Mapper(uses = {PersonMapper.class, TeamMapper.class, OrgMapper.class})
public abstract class SourceDetailsMapper implements OptionAwareMapper, TimedMapper {

    @Inject
    private PersonMapper personMapper;

    @Inject
    private TeamMapper teamMapper;

    @Inject
    private OrgMapper orgMapper;

    public Source entityToDataModel(@Nullable SourceDetails sourceDetails, PublicPKBPerson sourcePerson) {

        try (Histogram.Timer ignored = MAP_TIMER.labels("source_details", "source_details").startTimer()) {
            ImmutableSource.Builder builder = ImmutableSource.builder();

            if (nonNull(sourceDetails)) {
                if (sourceDetails.getTeam() != null) {
                    builder.team(teamMapper.entityToDataModel(sourceDetails.getTeam()));
                }
                if (sourceDetails.getOrg() != null) {
                    builder.org(orgMapper.entityToDataModel(sourceDetails.getOrg()));
                }
                if (sourceDetails.getResponsibleOrganisationIdentifier() != null) {
                    builder.responsibleOrganisationIdentifier(sourceDetails.getResponsibleOrganisationIdentifier());
                }
            }

            addPersonToSourceDetails(sourcePerson, builder);

            return builder.build();
        }
    }

    void addPersonToSourceDetails(PublicPKBPerson sourcePerson, ImmutableSource.Builder builder) {
        if (nonNull(sourcePerson)) {
            builder.person(personMapper.entityToDataModel(sourcePerson));
        }
    }
}
