package com.pkb.datamodel.entity.mapper;

import com.google.common.collect.Lists;
import com.pkb.datamodel.ClinicalCoding;
import com.pkb.datamodel.ImmutableClinicalCoding;
import com.pkb.datamodel.confidential.SensitiveString;
import org.apache.commons.collections4.CollectionUtils;
import org.jetbrains.annotations.Nullable;
import org.mapstruct.Mapper;

import java.util.HashMap;
import java.util.List;
import java.util.Map;

import static java.util.stream.Collectors.toList;

@Mapper
public abstract class ClinicalCodingMapper implements StringFieldMapper {

    public Map<String, Object> codingToMap(ClinicalCoding coding) {
        Map<String, Object> map = new HashMap<>();
        map.put("code", coding.getCode());
        map.put("codeSystem", coding.getCodeSystem().getOrElse((String)null));
        map.put("displayText", coding.getDisplayText().map(SensitiveString::value).getOrElse((String)null));
        return map;
    }

    public ClinicalCoding mapToCoding(Map<String, Object> map) {
        return ImmutableClinicalCoding.builder()
                .code(getString(map.get("code")))
                .build()
                .withCodeSystem(getOptionString(map.get("codeSystem")))
                .withDisplayText(getOptionSensitiveString(map.get("displayText")));
    }

    public List<ClinicalCoding> mapToCoding(@Nullable List<Map<String, Object>> maps) {
        if (CollectionUtils.isEmpty(maps)) {
            return Lists.newArrayList();
        }

        return maps.stream().map(this::mapToCoding).collect(toList());
    }

}
