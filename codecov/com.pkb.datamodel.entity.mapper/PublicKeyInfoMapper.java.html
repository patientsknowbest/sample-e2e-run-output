package com.pkb.datamodel.entity.mapper;

import com.pkb.crypto.AccountPublicKey;
import com.pkb.entities.embeddables.PublicKeyInfo;
import org.jetbrains.annotations.Nullable;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

import java.security.KeyFactory;
import java.security.PublicKey;
import java.security.spec.X509EncodedKeySpec;

import static com.pkb.crypto.AccountPublicKey.accountPublicKey;
import static io.vavr.API.unchecked;

@Mapper
public abstract class PublicKeyInfoMapper implements TimedMapper {

    @Mapping(source = "encoded", target = "x509EncodedKeySpec")
    public abstract PublicKeyInfo publicKeyToEntity(PublicKey publicKey);

    public @Nullable AccountPublicKey mapAccountPublicKey(PublicKeyInfo entity) {
        if (entity == null) {
            return null;
        }
        X509EncodedKeySpec keySpec = new X509EncodedKeySpec(entity.getX509EncodedKeySpec());
        return unchecked(() -> accountPublicKey(KeyFactory.getInstance(entity.getAlgorithm()).generatePublic(keySpec)))
                .get();

    }

    public PublicKeyInfo mapPublicKeyInfo(AccountPublicKey dataModel) {
        PublicKeyInfo result = new PublicKeyInfo();
        result.setAlgorithm(dataModel.getAlgorithm());
        result.setX509EncodedKeySpec(dataModel.getKeyBytes());
        return result;

    }
}
