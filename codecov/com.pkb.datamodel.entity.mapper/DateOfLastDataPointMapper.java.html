package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.projections.ImmutablePatientDateOfLastDataPoint;
import com.pkb.datamodel.projections.PatientDateOfLastDataPoint;
import com.pkb.entities.pub.PublicNationalId;
import io.prometheus.client.Histogram;
import io.vavr.control.Option;
import org.apache.commons.lang3.tuple.Pair;
import org.mapstruct.Mapper;

import javax.persistence.Tuple;
import java.time.Instant;
import java.util.UUID;

@Mapper
public class DateOfLastDataPointMapper implements TimedMapper {

    public PatientDateOfLastDataPoint mapToDataModelWithDate(Tuple tuple) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("date_of_last_data_point", "date").startTimer()) {
            return ImmutablePatientDateOfLastDataPoint.patientDateOfLastDataPoint()
                    .dateOfLastDataPoint(tuple.get(0, Instant.class))
                    .build();
        }
    }

    public PatientDateOfLastDataPoint mapToDataModelWithPublicId(UUID publicId, Pair<Long, Instant> persistedDateWithPersonId) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("date_of_last_data_point", "public_id").startTimer()) {
            return ImmutablePatientDateOfLastDataPoint.patientDateOfLastDataPoint()
                    .idValue(publicId)
                    .dateOfLastDataPoint(persistedDateWithPersonId.getRight())
                    .build();
        }
    }

    public PatientDateOfLastDataPoint mapToDataModelWithNationalId(PublicNationalId nationalId, Pair<Long, Instant> persistedDateWithPersonId) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("date_of_last_data_point", "national_id").startTimer()) {
            return ImmutablePatientDateOfLastDataPoint.patientDateOfLastDataPoint()
                    .validNationalId(nationalId == null ? Option.none() : Option.of(nationalId.getValidNationalId()))
                    .dateOfLastDataPoint(persistedDateWithPersonId.getRight())
                    .build();
        }
    }

}
