package com.pkb.datamodel.entity.mapper;

import com.pkb.crypto.AccountPrivateKey;
import com.pkb.crypto.EncryptedSymmetricKey;
import com.pkb.crypto.SymmetricKey;
import com.pkb.crypto.UserPrivateKey;
import com.pkb.crypto.dto.AccountSymmetricKeyDTO;
import com.pkb.datamodel.AccountKeys;
import com.pkb.datamodel.ImmutableAccountKeys;
import com.pkb.datamodel.crypto.CryptoService;
import com.pkb.entities.core.CoreAccountSymmetricKey;
import com.pkb.entities.core.UserAccountKeyLink;
import io.prometheus.client.Histogram;
import io.vavr.Function1;
import io.vavr.Function2;
import io.vavr.collection.List;
import io.vavr.collection.Map;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.mapstruct.Mapper;

import javax.inject.Inject;
import java.util.Base64;

import static com.pkb.crypto.EncryptedSymmetricKey.encryptedSymmetricKey;
import static com.pkb.crypto.dto.AccountSymmetricKeyDTO.accountSymmetricKeyDTO;

@Mapper
public abstract class AccountKeysMapper implements TimedMapper {

    @Inject
    private CryptoService cryptoService;

    public AccountKeys map(
            Either<UserPrivateKey, AccountPrivateKey> userOrAccountPrivateKey,
            @Nullable UserAccountKeyLink link,
            @NotNull List<CoreAccountSymmetricKey> encryptedSymmetricKeys) {

        try (Histogram.Timer ignored = MAP_TIMER.labels("account_keys", "user_account_key_link").startTimer()) {
            AccountPrivateKey accountPrivateKey = cryptoService.getAccountPrivateKey(userOrAccountPrivateKey, link);
            Function1<CoreAccountSymmetricKey, AccountSymmetricKeyDTO> decryptWithAccountPrivateKey
                    = Function2.of(this::mapSymmetricKey).apply(accountPrivateKey);

            Map<Long, SymmetricKey> symmetricKeys = encryptedSymmetricKeys
                    .map(decryptWithAccountPrivateKey)
                    .toMap(AccountSymmetricKeyDTO::getKeyId, SymmetricKey::symmetricKey);

            return ImmutableAccountKeys.builder()
                    .privateKey(accountPrivateKey)
                    .build()
                    .withSymmetricKeys(symmetricKeys);
        }
    }

    private AccountSymmetricKeyDTO mapSymmetricKey(AccountPrivateKey accountPrivateKey, CoreAccountSymmetricKey accountSymmetricKey) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("account_keys", "symmetric_key").startTimer()) {
            EncryptedSymmetricKey encryptedSymmetricKey = encryptedSymmetricKey(
                    Base64.getDecoder().decode(accountSymmetricKey.getEncryptedSymmetricKeyBase64()),
                    accountSymmetricKey.getSymmetricKeyAlgorithm());
            return accountSymmetricKeyDTO(accountSymmetricKey.getId(), accountPrivateKey.decryptSymmetricKey(encryptedSymmetricKey));
        }
    }
}
