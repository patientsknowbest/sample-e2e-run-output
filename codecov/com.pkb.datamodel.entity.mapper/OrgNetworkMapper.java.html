package com.pkb.datamodel.entity.mapper;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.ImmutableOrgNetworkExisting;
import com.pkb.datamodel.OrgNetworkExisting;
import com.pkb.datamodel.OrgNetworkNew;
import com.pkb.entities.pub.PublicOrg;
import com.pkb.entities.pub.PublicOrgNetwork;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.mapstruct.Named;
import org.mapstruct.ReportingPolicy;

import javax.inject.Inject;
import javax.persistence.EntityManager;
import java.util.Optional;

@Mapper(uses = OrgMapper.class, unmappedTargetPolicy = ReportingPolicy.IGNORE)
public abstract class OrgNetworkMapper implements TimedMapper {

    @Inject
    protected DateTimeService dateTimeService;

    @Inject
    protected OrgMapper orgMapper;

    @Inject
    protected EntityManager entityManager;

    @Mapping(target = "id", ignore = true) // assigned when persisted
    @Mapping(target = "createdDate", expression = "java(dateTimeService.now())")
    @Mapping(source = "leadingOrgId", target = "leadingOrg", qualifiedByName = "idToEntityReference")
    public abstract PublicOrgNetwork dataModelToEntity(OrgNetworkNew orgNetwork);

    @Named("idToEntityReference")
    public PublicOrg orgIdToOrgReference(long orgId) {
        return entityManager.getReference(PublicOrg.class, orgId);
    }

    public OrgNetworkExisting entityToDataModel(PublicOrgNetwork entity) {
        return ImmutableOrgNetworkExisting.builder()
                .id(entity.getId())
                .createdDate(entity.getCreatedDate())
                .createdByPersonId(entity.getCreatedByPersonId())
                .name(entity.getName())
                .leadingOrg(orgMapper.entityToDataModel(entity.getLeadingOrg()))
                .build();
    }

    public Optional<OrgNetworkExisting> entityToDataModel(Optional<PublicOrgNetwork> maybeEntity) {
        return maybeEntity.map(this::entityToDataModel);
    }

}
