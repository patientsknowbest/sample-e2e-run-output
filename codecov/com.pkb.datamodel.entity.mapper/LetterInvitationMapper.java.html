package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutableLetterInvitationExisting;
import com.pkb.datamodel.LetterInvitation;
import com.pkb.datamodel.LetterInvitationNew;
import com.pkb.entities.pub.PublicLetterInvitation;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

@Mapper
public abstract class LetterInvitationMapper implements OptionAwareMapper, TimedMapper {

    public @NotNull LetterInvitation entityToDataModel(@NotNull PublicLetterInvitation entity) {
        return ImmutableLetterInvitationExisting.builder()
                .id(entity.getId())
                .authorId(Option.of(entity.getAuthorId()))
                .claimedEmail(Option.of(entity.getClaimedEmail()))
                .code(entity.getCode())
                .created(entity.getCreated())
                .accountPrivateKey(entity.getAccountPrivateKey())
                .expiration(entity.getExpiration())
                .invalidated(entity.getInvalidated())
                .inviteeId(entity.getInviteeId())
                .teamId(Option.of(entity.getTeamId()))
                .verificationCode(Option.of(entity.getVerificationCode()))
                .build();
    }

    @Mapping(target = "id", ignore = true)
    @Mapping(target = "inviteeStatus", ignore = true)
    abstract PublicLetterInvitation toPublicLetterInvitation(LetterInvitationNew letterInvitation);

    public PublicLetterInvitation newDataModelToNewEntity(@NotNull LetterInvitationNew letterInvitation) {
        return toPublicLetterInvitation(letterInvitation);
    }
}
