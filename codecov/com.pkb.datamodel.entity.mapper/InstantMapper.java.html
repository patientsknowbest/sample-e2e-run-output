package com.pkb.datamodel.entity.mapper;

import io.vavr.control.Option;
import org.mapstruct.Mapper;

import javax.annotation.Nullable;
import java.time.Instant;
import java.time.ZoneOffset;
import java.time.ZonedDateTime;

@Mapper
public interface InstantMapper {
    default ZonedDateTime mapToZonedDateTime(Instant instant) {
        return ZonedDateTime.ofInstant(instant, ZoneOffset.UTC);
    }

    default Option<ZonedDateTime> mapToZonedDateTimeOption(@Nullable Instant instant) {
        if (instant == null) {
            return Option.none();
        }
        return Option.of(mapToZonedDateTime(instant));
    }
}
