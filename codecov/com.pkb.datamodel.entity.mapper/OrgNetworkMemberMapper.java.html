package com.pkb.datamodel.entity.mapper;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.ImmutableOrgNetworkMemberExisting;
import com.pkb.datamodel.OrgNetworkMemberExisting;
import com.pkb.datamodel.OrgNetworkMemberNew;
import com.pkb.entities.pub.PublicOrgNetwork;
import com.pkb.entities.pub.PublicOrgNetworkMember;
import io.vavr.control.Option;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.mapstruct.Named;
import org.mapstruct.ReportingPolicy;

import javax.inject.Inject;
import javax.persistence.EntityManager;

@Mapper(uses = OrgMapper.class, unmappedTargetPolicy = ReportingPolicy.IGNORE)
public abstract class OrgNetworkMemberMapper implements TimedMapper, OptionAwareMapper {

    @Inject
    protected DateTimeService dateTimeService;

    @Inject
    protected OrgMapper orgMapper;

    @Inject
    protected EntityManager entityManager;

    @Mapping(target = "id", ignore = true) // assigned when persisted
    @Mapping(target = "createdDate", expression = "java(dateTimeService.now())")
    @Mapping(source = "orgNetworkId", target = "orgNetwork", qualifiedByName = "orgNetworkIdToReference")
    public abstract PublicOrgNetworkMember dataModelToEntity(OrgNetworkMemberNew onm);

    @Named("orgNetworkIdToReference")
    PublicOrgNetwork orgNetworkIdToReference(long id) {
        return entityManager.getReference(PublicOrgNetwork.class, id);
    }


    public OrgNetworkMemberExisting entityToDataModel(PublicOrgNetworkMember entity) {
        return ImmutableOrgNetworkMemberExisting.builder()
                .id(entity.getId())
                .createdDate(entity.getCreatedDate())
                .createdByPersonId(entity.getCreatedByPersonId())
                .activatedByPersonId(Option.of(entity.getActivatedByPersonId()))
                .activatedDate(Option.of(entity.getActivatedDate()))
                .approvedByPersonId(Option.of(entity.getApprovedByPersonId()))
                .approvedDate(Option.of(entity.getApprovedDate()))
                .memberStatus(entity.getMemberStatus())
                .orgNetworkId(entity.getOrgNetwork().getId())
                .orgId(entity.getOrgId())
                .build();
    }

}
