package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutableSSOParameterEncryptionDetailsExisting;
import com.pkb.datamodel.SSOParameterEncryptionDetails;
import com.pkb.datamodel.SSOParameterEncryptionDetailsNew;
import com.pkb.entities.pub.PublicSSOParameterEncryptionDetails;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

@Mapper
public abstract class SSOParameterEncryptionDetailsMapper implements OptionAwareMapper, TimedMapper {

    public @NotNull SSOParameterEncryptionDetails entityToDataModel(@NotNull PublicSSOParameterEncryptionDetails entity){
        return ImmutableSSOParameterEncryptionDetailsExisting.builder()
                .id(entity.getId())
                .createdOn(entity.getCreatedOn())
                .externalId(entity.getExternalId())
                .encryptionKey(entity.getEncryptionKey())
                .initialisationVector(entity.getInitialisationVector())
                .build();
    }

    @Mapping(target = "id", ignore = true)
    abstract PublicSSOParameterEncryptionDetails toPublicSSOParameterEncryptionDetails(SSOParameterEncryptionDetailsNew ssoParameterEncryptionDetails);

    public PublicSSOParameterEncryptionDetails newDataModelToNewEntity(SSOParameterEncryptionDetailsNew ssoParameterEncryptionDetails){
        return toPublicSSOParameterEncryptionDetails(ssoParameterEncryptionDetails);
    }
}
