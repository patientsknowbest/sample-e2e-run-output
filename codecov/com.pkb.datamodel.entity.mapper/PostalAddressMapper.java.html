package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.user.ImmutablePostalAddress;
import com.pkb.datamodel.user.PostalAddress;
import com.pkb.entities.pub.PublicPKBPerson;
import io.prometheus.client.Histogram;
import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;
import org.mapstruct.Mapper;

@Mapper
public abstract class PostalAddressMapper implements TimedMapper {

    public PostalAddress personEntityToAddress(PublicPKBPerson entity) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("postal_address", "person").startTimer()) {
            return ImmutablePostalAddress.builder()
                    .address1(Option.of(entity.getAddress1()).filter(StringUtils::isNotBlank))
                    .address2(Option.of(entity.getAddress2()).filter(StringUtils::isNotBlank))
                    .city(Option.of(entity.getCity()).filter(StringUtils::isNotBlank))
                    .state(Option.of(entity.getState()).filter(StringUtils::isNotBlank))
                    .country(Option.of(entity.getCountry()).filter(StringUtils::isNotBlank))
                    .postCode(Option.of(entity.getPostalCode()).filter(StringUtils::isNotBlank))
                    .build();
        }
    }
}
