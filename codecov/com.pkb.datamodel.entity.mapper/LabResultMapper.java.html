package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.AppMenuDataWithSourceObjects;
import com.pkb.datamodel.ImmutableLabResult;
import com.pkb.datamodel.LabResult;
import com.pkb.entities.app.AppMenuData;
import io.prometheus.client.Histogram;
import io.vavr.collection.List;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;

import javax.inject.Inject;
import java.time.ZoneOffset;
import java.time.ZonedDateTime;

@Mapper
public abstract class LabResultMapper implements TimedMapper {

    @Inject
    private PrivacyFlagsMapper privacyFlagsMapper;

    @Inject
    private SourceDetailsMapper sourceDetailsMapper;

    public @NotNull List<LabResult> menuDataToLabResults(@NotNull List<AppMenuDataWithSourceObjects> wrapper) {
        return wrapper.map(this::menuDataToLabResult);
    }

    @NotNull
    private LabResult menuDataToLabResult(@NotNull AppMenuDataWithSourceObjects wrapper) {

        AppMenuData menuData = wrapper.getAppMenuData();

        try (Histogram.Timer ignored = MAP_TIMER.labels("lab_result", "lab_result").startTimer()) {
            return ImmutableLabResult
                    .builder()
                    .id(menuData.getId())
                    .uniqueId(menuData.getBaseFields().getUniqueId())
                    .privacyFlags(privacyFlagsMapper.entityToDataModel(menuData.getBaseFields()))
                    .source(sourceDetailsMapper.entityToDataModel(menuData.getSourceDetails(), wrapper.getSourcePerson().getOrNull()))
                    .accountId(menuData.getAccountId())
                    .effectiveTimestamp(date01ToZonedDateTime(menuData))
                    .build();
        }
    }

    private ZonedDateTime date01ToZonedDateTime(AppMenuData menuData) {
        return ZonedDateTime.ofInstant(menuData.getDate01(), ZoneOffset.UTC);
    }

}
