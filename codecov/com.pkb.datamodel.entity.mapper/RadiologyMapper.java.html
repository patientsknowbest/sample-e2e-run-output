package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.AppMenuDataWithSourceObjects;
import com.pkb.datamodel.ImmutableRadiologyImage;
import com.pkb.datamodel.RadiologyImage;
import com.pkb.entities.app.AppMenuData;
import io.prometheus.client.Histogram;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;

import javax.inject.Inject;

@Mapper
public abstract class RadiologyMapper implements TimedMapper {

    @Inject
    private PrivacyFlagsMapper privacyFlagsMapper;

    @Inject
    private SourceDetailsMapper sourceDetailsMapper;

    @Inject
    private InstantMapper instantMapper;

    @NotNull
    public RadiologyImage menuDataToRadiologyImage(@NotNull AppMenuDataWithSourceObjects wrapper) {

        AppMenuData menuData = wrapper.getAppMenuData();

        try (Histogram.Timer ignored = MAP_TIMER.labels("radiology", "radiology").startTimer()) {
            return ImmutableRadiologyImage.builder()
                    .id(menuData.getId())
                    .uniqueId(menuData.getBaseFields().getUniqueId())
                    .privacyFlags(privacyFlagsMapper.entityToDataModel(menuData.getBaseFields()))
                    .source(sourceDetailsMapper.entityToDataModel(menuData.getSourceDetails(), wrapper.getSourcePerson().getOrNull()))
                    .accountId(menuData.getAccountId())
                    .effectiveTimestamp(instantMapper.mapToZonedDateTime(menuData.getDate02()))
                    .persistedTimestamp(instantMapper.mapToZonedDateTime(menuData.getBaseFields().getPersistedDate()))
                    .build();
        }
    }
}
