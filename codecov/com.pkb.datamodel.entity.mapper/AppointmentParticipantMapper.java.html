package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.AccountKeys;
import com.pkb.datamodel.AppMenuDataWithSourceObjects;
import com.pkb.datamodel.AppointmentParticipant;
import com.pkb.datamodel.ImmutableAppointmentParticipant;
import com.pkb.datamodel.crypto.CryptoService;
import com.pkb.datamodel.entity.ClinicalDataDeserializer;
import com.pkb.entities.app.AppMenuData;
import io.prometheus.client.Histogram;
import io.vavr.collection.List;
import io.vavr.collection.Map;
import org.mapstruct.Mapper;

import javax.inject.Inject;

@Mapper
public class AppointmentParticipantMapper implements StringFieldMapper, TimedMapper {

    @Inject
    private CryptoService cryptoService;

    @Inject
    private ClinicalDataDeserializer dataDeserializer;

    public Map<Long, List<AppointmentParticipant>> menuDatasToAppointmentParticipants(
            AccountKeys accountKeys,
            io.vavr.collection.List<AppMenuDataWithSourceObjects> wrappers) {

        Map<Long, byte[]> decryptedData = cryptoService.decrypt(accountKeys, wrappers.map(AppMenuDataWithSourceObjects::getAppMenuData));
        return wrappers
                .map(md -> buildAppointmentParticipant(md.getAppMenuData(), decryptedData.get(md.getAppMenuData().getId()).get()))
                .groupBy(AppointmentParticipant::parentAppointmentPrivateId);
    }

    private AppointmentParticipant buildAppointmentParticipant(AppMenuData menuData, byte[] plaintext) {
        return dataDeserializer.deserialize(plaintext, fields -> {
            try (Histogram.Timer ignored = MAP_TIMER.labels("appointment", "participant").startTimer()) {
                return ImmutableAppointmentParticipant.builder()
                        .parentAppointmentPrivateId(menuData.getLong01())
                        .participantPrivateId(Long.parseLong((String) fields.get("participantId")))
                        .build();
            }
        });
    }
}
