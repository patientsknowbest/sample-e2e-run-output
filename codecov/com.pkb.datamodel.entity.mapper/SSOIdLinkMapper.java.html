package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutableSSOIdLinkExisting;
import com.pkb.datamodel.SSOIdLink;
import com.pkb.datamodel.SSOIdLinkNew;
import com.pkb.entities.pub.PublicSSOIdLink;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

@Mapper
public abstract class SSOIdLinkMapper implements OptionAwareMapper, TimedMapper{

    public @NotNull SSOIdLink entityToDataModel(@NotNull PublicSSOIdLink entity){
        return ImmutableSSOIdLinkExisting.builder()
                .id(entity.getId())
                .personId(entity.getPersonId())
                .ssoUsername(entity.getSsoUsername())
                .createdOn(entity.getCreatedOn())
                .ssoType(entity.getSsoType())
                .build();
    }
    
    @Mapping(target = "id", ignore = true)
    abstract PublicSSOIdLink toPublicSSOIdLink(SSOIdLinkNew ssoIdLink);

    public PublicSSOIdLink newDataModelToNewEntity(@NotNull SSOIdLinkNew ssoIdLink){
        return toPublicSSOIdLink(ssoIdLink);
    }
}
