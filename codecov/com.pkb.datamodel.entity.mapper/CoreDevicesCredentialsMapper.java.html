package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.CoreDevicesCredentials;
import com.pkb.datamodel.ImmutableCoreDevicesCredentials;
import com.pkb.entities.pub.PublicCoreDevicesCredentials;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

@Mapper
public abstract class CoreDevicesCredentialsMapper implements OptionAwareMapper, TimedMapper {

    @NotNull
    public CoreDevicesCredentials entityToDataModel(@NotNull PublicCoreDevicesCredentials entity){
        return ImmutableCoreDevicesCredentials.builder()
                .patientId(entity.getPatientId())
                .opaqueId(entity.getOpaqueId())
                .coreDevicesUserId(entity.getCoreDevicesUserId())
                .accessToken(Option.of(entity.getAccessToken()))
                .dateCreated(entity.getDateCreated())
                .build();
    }

    @Mapping(target = "id", ignore = true)
    abstract PublicCoreDevicesCredentials toPublicCoreDevicesCredentials(CoreDevicesCredentials coreDevicesCredentials);

    public PublicCoreDevicesCredentials newDataModelToNewEntity(@NotNull CoreDevicesCredentials coreDevicesCredentials){
        return toPublicCoreDevicesCredentials(coreDevicesCredentials);
    }
}
