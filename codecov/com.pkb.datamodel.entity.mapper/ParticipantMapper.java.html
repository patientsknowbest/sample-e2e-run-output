package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ExternalParticipant;
import com.pkb.datamodel.ImmutableExternalParticipant;
import com.pkb.datamodel.confidential.SensitiveString;
import io.prometheus.client.Histogram;
import io.vavr.collection.List;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.mapstruct.Mapper;

import java.util.EnumSet;
import java.util.Map;

import static com.pkb.datamodel.ImmutableExternalParticipant.externalParticipant;
import static io.vavr.collection.List.collector;
import static io.vavr.collection.List.empty;
import static org.apache.commons.collections4.CollectionUtils.isEmpty;

@Mapper
public abstract class ParticipantMapper implements TimedMapper {

    // https://patientsknowbest.slack.com/archives/C6S1R2T6D/p1548434913551400
    private EnumSet<ExternalParticipant.ParticipantRole> PERMISSABLE_ROLES = EnumSet.of(
            ExternalParticipant.ParticipantRole.ATTENDER,
            ExternalParticipant.ParticipantRole.CONSULTANT,
            ExternalParticipant.ParticipantRole.REFERRER);

    @NotNull
    List<ExternalParticipant> mapExternalParticipants(@Nullable java.util.List<java.util.Map<String, Object>> externalParticipantsFields) {

        if (isEmpty(externalParticipantsFields)) {
            return empty();
        }

        return externalParticipantsFields.stream()
                .filter(this::isPermissableRole)
                .map(this::mapExternalParticipantDTO)
                .collect(collector());
    }

    @NotNull
    List<ExternalParticipant> mapExternalParticipantDTOs(@Nullable java.util.List<java.util.Map<String, Object>> dtos) {
        if (isEmpty(dtos)) {
            return empty();
        }
        return List.ofAll(dtos).map(this::mapExternalParticipantDTO);
    }

    private boolean isPermissableRole(Map<String, Object> fields) {
        return PERMISSABLE_ROLES.contains(ExternalParticipant.ParticipantRole.valueOf((String) fields.get("role")));
    }

    @NotNull
    private ImmutableExternalParticipant mapExternalParticipantDTO(Map<String, Object> entry) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("participant", "external_participant").startTimer()) {
            return externalParticipant()
                    .name(SensitiveString.of((String) entry.get("nonpkbParticipant")))
                    .role(ExternalParticipant.ParticipantRole.valueOf((String) entry.get("role"))) // TODO: yuck!
                    .build();
        }
    }
}
