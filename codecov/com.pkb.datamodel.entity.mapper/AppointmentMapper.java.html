package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.AccountKeys;
import com.pkb.datamodel.AppMenuDataWithSourceObjects;
import com.pkb.datamodel.Appointment;
import com.pkb.datamodel.AppointmentParticipant;
import com.pkb.datamodel.ClinicalCoding;
import com.pkb.datamodel.CodeableConcept;
import com.pkb.datamodel.ImmutableAppointment;
import com.pkb.datamodel.ImmutableClinicalCoding;
import com.pkb.datamodel.ReferenceDatumCache;
import com.pkb.datamodel.confidential.SensitiveString;
import com.pkb.datamodel.crypto.CryptoService;
import com.pkb.datamodel.entity.ClinicalDataDeserializer;
import com.pkb.datamodel.user.Patient;
import com.pkb.datamodel.user.Person;
import com.pkb.entities.app.AppMenuData;
import com.pkb.entities.enums.AppointmentStatus;
import io.prometheus.client.Histogram;
import io.vavr.collection.List;
import io.vavr.collection.Map;
import io.vavr.control.Option;
import org.mapstruct.Mapper;

import javax.inject.Inject;

@Mapper
public class AppointmentMapper implements TimedMapper {

    @Inject
    private PrivacyFlagsMapper privacyFlagsMapper;

    @Inject
    private SourceDetailsMapper sourceDetailsMapper;

    @Inject
    private CryptoService cryptoService;

    @Inject
    private ClinicalDataDeserializer dataDeserializer;

    @Inject
    private InstantMapper instantMapper;

    public List<Appointment> mapToAppointments(List<AppMenuDataWithSourceObjects> appointmentMenuData,
                                               AccountKeys accountKeys,
                                               Map<Long, List<AppointmentParticipant>> participantsByAppointmentId,
                                               Map<Long, Person> personsById,
                                               ReferenceDatumCache specialtyCache) {
        Map<Long, byte[]> decryptedData = cryptoService.decrypt(accountKeys, appointmentMenuData.map(AppMenuDataWithSourceObjects::getAppMenuData));
        return appointmentMenuData
                .map(wrapper -> buildAppointment(wrapper,
                        decryptedData.get(wrapper.getAppMenuData().getId()).get(),
                        participantsByAppointmentId.get(wrapper.getAppMenuData().getId()).get(),
                        personsById,
                        specialtyCache));
    }

    private Appointment buildAppointment(AppMenuDataWithSourceObjects wrapper, byte[] plaintext,
            List<AppointmentParticipant> participants,
            Map<Long, Person> personsById,
            ReferenceDatumCache specialtyCache) {

        AppMenuData menuData = wrapper.getAppMenuData();

        try (Histogram.Timer ignored = MAP_TIMER.labels("appointment", "appointment").startTimer()) {
            return dataDeserializer.deserialize(plaintext, fields -> {
                Option<SensitiveString> location = getSensitiveStringFromField(fields, "location");
                Option<SensitiveString> subject = getSensitiveStringFromField(fields, "subject");
                Option<SensitiveString> description = getSensitiveStringFromField(fields, "description");
                Option<CodeableConcept> type = Option.none();
                if (menuData.getCode01Code() != null) {
                    ClinicalCoding typeCoding = ImmutableClinicalCoding.builder().code(menuData.getCode01Code())
                            .build().withCodeSystem(Option.of(menuData.getCode01CodeSystem()));
                    type = Option.of(CodeableConcept.of(typeCoding));
                }
                long patientId = Long.parseLong((String)fields.get("patientId"));
                Patient patient = (Patient) personsById.get(patientId).get();
                List<Person> pkbParticipants = participants.filter(p -> !p.participantPrivateId().equals(patientId))
                        .map(p -> personsById.get(p.participantPrivateId()))
                        .filter(Option::isDefined).map(Option::get);
                // Key "nonPkbParticipants" may explicitly map to `null` list, so calling getOrDefault doesn't help.
                List<SensitiveString> nonPkbParticipants = Option.of((java.util.List<String>)fields.get("nonPkbParticipants"))
                        .toList()
                        .flatMap(List::ofAll)
                        .map(SensitiveString::of);
                return ImmutableAppointment.builder()
                        .id(menuData.getId())
                        .accountId(menuData.getAccountId())
                        .source(sourceDetailsMapper.entityToDataModel(menuData.getSourceDetails(), wrapper.getSourcePerson().getOrNull()))
                        .privacyFlags(privacyFlagsMapper.entityToDataModel(menuData.getBaseFields()))
                        .uniqueId(menuData.getBaseFields().getUniqueId())
                        .persisted(instantMapper.mapToZonedDateTimeOption(menuData.getBaseFields().getPersistedDate()))
                        .accountId(menuData.getAccountId())
                        .status(AppointmentStatus.valueOf(menuData.getString02()))
                        .start(instantMapper.mapToZonedDateTime(menuData.getDate01()))
                        .end(instantMapper.mapToZonedDateTimeOption(menuData.getDate02()))
                        .location(location)
                        .subject(subject)
                        .type(type)
                        .description(description)
                        .specialty(specialtyCache
                                .lookup(Option.of(menuData.getSourceDetails().getOrg()), Option.of((String) fields.getOrDefault("hospitalServiceCode", null))))
                        .patient(patient)
                        .owner(patient) // Can this ever be different from patient?
                        .participants(pkbParticipants)
                        .nonPKBParticipants(nonPkbParticipants)
                        .build();
            });
        }
    }

    private Option<SensitiveString> getSensitiveStringFromField(java.util.Map<String, Object> fields, String fieldName) {
        return Option.of(fields.get(fieldName))
                .map(String.class::cast)
                .map(SensitiveString::of);
    }
}
