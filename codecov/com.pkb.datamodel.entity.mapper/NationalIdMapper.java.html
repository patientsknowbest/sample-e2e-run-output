package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutableNationalId;
import com.pkb.datamodel.NationalId;
import com.pkb.datamodel.NewNationalId;
import com.pkb.entities.pub.PublicNationalId;
import io.prometheus.client.Histogram;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.mapstruct.Mapper;
import org.mapstruct.ReportingPolicy;

import java.time.ZoneOffset;
import java.time.ZonedDateTime;

import static org.apache.commons.lang3.StringUtils.isNotBlank;

@Mapper(uses = PersonMapper.class, unmappedTargetPolicy = ReportingPolicy.IGNORE)
public abstract class NationalIdMapper implements TimedMapper {

    public abstract PublicNationalId dataModelToEntity(NewNationalId nationalId);

    public NationalId entityToDataModel(PublicNationalId entity) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("national_id", "national_id").startTimer()) {
            return ImmutableNationalId.builder()
                    .id(entity.getId())
                    .type(entity.getType())
                    .build()
                    .withValue(entity.getValue())
                    .withEditDate(Option.of(entity.getEditDate()).map(it -> ZonedDateTime.ofInstant(it, ZoneOffset.UTC)));
        }
    }

    public List<NationalId> entityToDataModel(List<PublicNationalId> entities) {
        return entities
                // FIXME: GDE-433
                .filter(nationalId -> isNotBlank(nationalId.getValue()))
                .map(this::entityToDataModel);
    }
}
