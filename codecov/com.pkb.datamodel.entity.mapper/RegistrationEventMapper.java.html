package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutableRegistrationEventExisting;
import com.pkb.datamodel.RegistrationEventExisting;
import com.pkb.datamodel.RegistrationEventNew;
import com.pkb.entities.pub.PublicRegistrationEvent;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

@Mapper(uses = {PersonMapper.class})
public abstract class RegistrationEventMapper implements TimedMapper {


    @Mapping(target = "id", ignore = true)
    public abstract PublicRegistrationEvent dataModelToEntity(RegistrationEventNew registrationEventNew);


    public @NotNull RegistrationEventExisting entityToDataModel(PublicRegistrationEvent entity) {
        return ImmutableRegistrationEventExisting.builder()
                .registrationMethod(entity.getRegistrationMethod())
                .registeredOn(entity.getRegisteredOn())
                .id(entity.getId())
                .invitationPublicId(Option.of(entity.getInvitationPublicId()))
                .invitationPrivateId(Option.of(entity.getInvitationPrivateId()))
                .registeredOnConfirmed(entity.isRegisteredOnConfirmed())
                .letterInvitationId(Option.of(entity.getLetterInvitationId()))
                .build();
    }

}
