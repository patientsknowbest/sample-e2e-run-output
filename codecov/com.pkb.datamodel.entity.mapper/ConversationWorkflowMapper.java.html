package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.messages.workflow.ConversationWorkflow;
import com.pkb.datamodel.messages.workflow.ConversationWorkflowExisting;
import com.pkb.datamodel.messages.workflow.ConversationWorkflowNew;
import com.pkb.datamodel.messages.workflow.ImmutableConversationWorkflowExisting;
import com.pkb.entities.pub.PublicConversationWorkflow;
import io.prometheus.client.Histogram;
import io.vavr.control.Option;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

import java.util.Optional;

@Mapper
public abstract class ConversationWorkflowMapper implements TimedMapper, OptionAwareMapper {

    public ConversationWorkflowExisting entityToDataModel(PublicConversationWorkflow entity) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("conversation_workflow", "conversation_workflow").startTimer()) {
            return ImmutableConversationWorkflowExisting.builder()
                    .conversationId(entity.getConversationId())
                    .teamId(entity.getTeamId())
                    .optimisticLockVersion(entity.getOptimisticLockVersion())
                    .archived(entity.isArchived())
                    .archiveChangeDate(Option.of(entity.getArchiveChangeDate()))
                    .assignedPersonId(Option.of(entity.getAssignedPersonId()))
                    .assignChangeDate(Option.of(entity.getAssignChangeDate()))
                    .build();
        }
    }

    public Optional<ConversationWorkflowExisting> entityToDataModel(Optional<PublicConversationWorkflow> maybeEntity) {
        return maybeEntity.map(this::entityToDataModel);
    }

    public <T> T map(Optional<T> t) {
        return t.orElse(null);
    }
    
    public PublicConversationWorkflow dataModelToEntity(ConversationWorkflow cw) {
        if (cw instanceof ConversationWorkflowNew) {
            return newDataModelToEntity((ConversationWorkflowNew) cw);
        }
        return existingDataModelToEntity((ConversationWorkflowExisting) cw);
    }

    @Mapping(target = "optimisticLockVersion", ignore = true)
    abstract PublicConversationWorkflow newDataModelToEntity(ConversationWorkflowNew cw);

    abstract PublicConversationWorkflow existingDataModelToEntity(ConversationWorkflowExisting cw);

}
