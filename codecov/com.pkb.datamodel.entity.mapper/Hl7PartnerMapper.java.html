package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.Hl7Partner;
import com.pkb.datamodel.ImmutableHl7Partner;
import com.pkb.entities.pub.PublicHl7Partner;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;

@Mapper
public abstract class Hl7PartnerMapper implements OptionAwareMapper, TimedMapper {
    public @NotNull Hl7Partner entityToDataModel(@NotNull PublicHl7Partner entity){
        return ImmutableHl7Partner.builder()
                .id(entity.getId())
                .code(entity.getCode())
                .ipWhitelist(entity.getIpWhitelist())
                .publicId(entity.getPublicId())
                .name(entity.getName())
                .timeZoneId(entity.getTimeZoneId())
                .hl7PasswordHash(Option.of(entity.getHl7PasswordHash()))
                .build();
    }

}
