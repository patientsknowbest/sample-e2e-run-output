package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutableReferenceDatum;
import com.pkb.datamodel.ReferenceDatum;
import com.pkb.entities.pub.PublicReferenceDatum;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;

import javax.inject.Inject;

@Mapper
public abstract class ReferenceDatumMapper implements OptionAwareMapper, TimedMapper {

    @Inject
    private OrgMapper orgMapper;

    public @NotNull ReferenceDatum entityToDataModel(@NotNull PublicReferenceDatum entity){
        return ImmutableReferenceDatum.builder()
                .org(orgMapper.entityToDataModel(entity.getOrg()))
                .dataType(entity.getDataType())
                .dataId(entity.getDataId())
                .dataText(entity.getDataText())
                .generalHealthConsentRequired(entity.getGeneralHealthConsentRequired())
                .mentalHealthConsentRequired(entity.getMentalHealthConsentRequired())
                .sexualHealthConsentRequired(entity.getSexualHealthConsentRequired())
                .socialCareConsentRequired(entity.getSocialCareConsentRequired())
                .publicId(entity.getPublicId())
                .build();
    }
}
