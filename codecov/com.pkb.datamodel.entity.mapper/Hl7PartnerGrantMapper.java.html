package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.Hl7PartnerGrant;
import com.pkb.datamodel.ImmutableHl7PartnerGrant;
import com.pkb.entities.pub.PublicHl7PartnerGrant;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;

import javax.inject.Inject;

@Mapper
public abstract class Hl7PartnerGrantMapper implements OptionAwareMapper, TimedMapper {

    @Inject
    private Hl7PartnerMapper hl7PartnerMapper;

    @Inject
    private OrgMapper orgMapper;


    public @NotNull Hl7PartnerGrant entityToDataModel(@NotNull PublicHl7PartnerGrant entity){
        return ImmutableHl7PartnerGrant.builder()
                .id(entity.getId())
                .autoCreateRecords(entity.isAutoCreateRecords())
                .checkNhsNumberStatus(entity.isCheckNhsNumberStatus())
                .created(entity.getCreated())
                .hl7Partner(hl7PartnerMapper.entityToDataModel(entity.getHl7Partner()))
                .newPatientTeamCode(Option.of(entity.getNewPatientTeamCode()))
                .org(orgMapper.entityToDataModel(entity.getOrg()))
                .sourceAlias(entity.getSourceAlias())
                .validatePidList(entity.getValidatePidList())
                .build();
    }


}
