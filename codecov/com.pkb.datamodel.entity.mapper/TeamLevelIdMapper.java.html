package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutableTeamLevelId;
import com.pkb.datamodel.TeamLevelId;
import com.pkb.entities.pub.PublicTeamLevelId;
import io.prometheus.client.Histogram;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.mapstruct.Mapper;

import javax.inject.Inject;
import java.time.ZoneOffset;
import java.time.ZonedDateTime;
import java.util.Objects;

@Mapper(uses = TeamLevelIdTypeMapper.class)
public abstract class TeamLevelIdMapper implements TimedMapper {

    @Inject
    private TeamLevelIdTypeMapper typeMapper;

    public TeamLevelId entityToDataModel(PublicTeamLevelId entity) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("team_level_id", "team_level_id").startTimer()) {
            return ImmutableTeamLevelId.builder()
                    .id(entity.getId())
                    .type(typeMapper.entityToDataModel(entity.getType()))
                    .build()
                    .withValue(entity.getValue())
                    .withEditDate(Option.of(entity.getEditDate()).map(it -> ZonedDateTime.ofInstant(it, ZoneOffset.UTC)));
        }
    }

    public List<TeamLevelId> entityToDataModel(List<PublicTeamLevelId> entities) {
        return entities.map(this::entityToDataModel);
    }

    /**
     * Maps {@link PublicTeamLevelId} entities to {@link TeamLevelId} domain objects, but excludes any that do not
     * belong to the specified team. Use this if you have overfetched team level ids and want to filter them.
     */
    public List<TeamLevelId> entityToDataModel(List<PublicTeamLevelId> entities, long teamId) {
        return entities
                .filter(tlid -> Objects.equals(tlid.getTeam().getId(), teamId))
                .map(this::entityToDataModel);
    }
}
