package com.pkb.datamodel.entity.mapper;

import com.google.common.collect.Maps;
import com.google.common.collect.Multimap;
import com.google.common.collect.MultimapBuilder;
import com.pkb.entities.enums.UserType;
import com.pkb.entities.enums.api.ApiClientScope;

import java.util.EnumMap;
import java.util.Map;

import static com.pkb.entities.enums.api.ApiClientScope.CLINICIAN;
import static com.pkb.entities.enums.api.ApiClientScope.CLINICIAN_OR_TEAMCOORD;
import static com.pkb.entities.enums.api.ApiClientScope.PATIENT;
import static com.pkb.entities.enums.api.ApiClientScope.PATIENT_OR_CLINICIAN;
import static com.pkb.entities.enums.api.ApiClientScope.PATIENT_OR_CLINICIAN_OR_TEAMCOORD;
import static com.pkb.entities.enums.api.ApiClientScope.PRIVACYOFFICER;
import static com.pkb.entities.enums.api.ApiClientScope.PUBLIC;
import static com.pkb.entities.enums.api.ApiClientScope.SITE;
import static com.pkb.entities.enums.api.ApiClientScope.SUPPORT;
import static com.pkb.entities.enums.api.ApiClientScope.TEAMCOORD;

/**
 * This class represents the mappings between {@link UserType}s and {@link Scope}s.
 */
public class UserTypeToScopeMapper {

    private static final Multimap<UserType, ApiClientScope> USERTYPE_IN_SCOPE_MAPPINGS;
    private static final Map<UserType, ApiClientScope> USERTYPE_TO_SCOPE_MAPPINGS;

    static {
        Multimap<UserType, ApiClientScope> multimap = MultimapBuilder.enumKeys(UserType.class).enumSetValues(ApiClientScope.class).build();
        multimap.put(UserType.PATIENT, PATIENT);
        multimap.put(UserType.PATIENT, PATIENT_OR_CLINICIAN);
        multimap.put(UserType.PATIENT, PATIENT_OR_CLINICIAN_OR_TEAMCOORD);
        multimap.put(UserType.REG_CLINICIAN, CLINICIAN);
        multimap.put(UserType.REG_CLINICIAN, CLINICIAN_OR_TEAMCOORD);
        multimap.put(UserType.REG_CLINICIAN, PATIENT_OR_CLINICIAN);
        multimap.put(UserType.REG_CLINICIAN, PATIENT_OR_CLINICIAN_OR_TEAMCOORD);
        multimap.put(UserType.INSTITUTE_ADMIN, TEAMCOORD);
        multimap.put(UserType.INSTITUTE_ADMIN, CLINICIAN_OR_TEAMCOORD);
        multimap.put(UserType.INSTITUTE_ADMIN, PATIENT_OR_CLINICIAN_OR_TEAMCOORD);
        multimap.put(UserType.INSTITUTE_ADMIN, SITE);
        multimap.put(UserType.TECH_SUPPORT, SUPPORT);
        multimap.put(UserType.SUPER_ADMIN, TEAMCOORD);
        multimap.put(UserType.EMPLOYEE, PUBLIC);
        multimap.put(UserType.ORG_COORD, PUBLIC);
        multimap.put(UserType.PRIVACY_OFFICER, PRIVACYOFFICER);
        multimap.put(UserType.PRIVACY_OFFICER, PUBLIC);

        USERTYPE_IN_SCOPE_MAPPINGS = multimap;

        Map<UserType, ApiClientScope> map = new EnumMap<>(UserType.class);
        map.put(UserType.INSTITUTE_ADMIN, TEAMCOORD);
        map.put(UserType.PRIVACY_OFFICER, PRIVACYOFFICER);
        map.put(UserType.REG_CLINICIAN, CLINICIAN);
        map.put(UserType.PATIENT, PATIENT);
        map.put(UserType.SUPER_ADMIN, TEAMCOORD);
        map.put(UserType.TECH_SUPPORT, SUPPORT);
        map.put(UserType.EMPLOYEE, PUBLIC);
        map.put(UserType.ORG_COORD, PUBLIC);

        USERTYPE_TO_SCOPE_MAPPINGS = Maps.immutableEnumMap(map);
    }

    /**
     * Determines weather the given {@link UserType} is allowed to access the given {@link Scope}.
     *
     * @param userType
     *            The type of user.
     * @param scope
     *            The scope to access.
     * @return <code>true</code> in case user is allowed to access given scope, <code>false</code> otherwise.
     */
    public boolean isUserTypeInScope(UserType userType, ApiClientScope scope) {
        return USERTYPE_IN_SCOPE_MAPPINGS.containsEntry(userType, scope);
    }

    /**
     * Returns the {@link Scope} for the given {@link UserType}.
     *
     * @param type
     *            The type of user.
     * @return The {@link Scope} the given type of user can access.
     */
    public ApiClientScope getScopeForUserType(UserType type) {
        return USERTYPE_TO_SCOPE_MAPPINGS.get(type);
    }
}
