package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutablePersonSpecialty;
import com.pkb.datamodel.PersonSpecialty;
import com.pkb.entities.pub.PublicPersonSpecialty;
import org.mapstruct.Mapper;

@Mapper
public class PersonSpecialtyMapper {

    public PersonSpecialty entityToDatamodel(PublicPersonSpecialty entity) {
        return ImmutablePersonSpecialty.builder()
                .id(entity.getId())
                .specialtyId(entity.getSpecialtyId())
                .personId(entity.getPersonId())
                .build();
    }
}
