package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.Account;
import com.pkb.datamodel.ImmutableAccount;
import com.pkb.entities.core.CoreAccount;
import org.mapstruct.Mapper;

import javax.inject.Inject;

@Mapper(uses = PublicKeyInfoMapper.class)
public abstract class AccountMapper implements TimedMapper {

    @Inject
    private PublicKeyInfoMapper publicKeyInfoMapper;

    public Account entityToDataModel(CoreAccount entity) {
        return ImmutableAccount.builder()
                .id(entity.getId())
                .publicId(entity.getPublicId())
                .publicKey(publicKeyInfoMapper.mapAccountPublicKey(entity.getAccountPublicKey()))
                .build();
    }

    public CoreAccount dataModelToEntity(Account dataModel) {
        CoreAccount result = new CoreAccount();
        result.setId(dataModel.getId());
        result.setPublicId(dataModel.getPublicId());
        result.setAccountPublicKey(publicKeyInfoMapper.mapPublicKeyInfo(dataModel.getPublicKey()));
        return result;
    }
}
