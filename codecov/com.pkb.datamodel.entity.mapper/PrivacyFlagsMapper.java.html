package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutablePrivacyFlags;
import com.pkb.datamodel.ImmutablePrivacyFlagsOverride;
import com.pkb.datamodel.PrivacyFlags;
import com.pkb.datamodel.PrivacyFlagsBase;
import com.pkb.datamodel.PrivacyFlagsOverride;
import com.pkb.entities.embeddables.BaseFields;
import com.pkb.entities.pub.PublicPrivacyFlags;
import io.prometheus.client.Histogram;
import io.vavr.control.Option;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

@Mapper
public abstract class PrivacyFlagsMapper implements OptionAwareMapper, TimedMapper {

    public PrivacyFlagsBase entityToDataModel(BaseFields entity) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("privacy_flags", "privacy_flags").startTimer()) {
            return entity.getLatestPrivacyFlags() == null ?
                    fromBaseFields(entity) :
                    fromPublicPrivacyFlags(entity.getLatestPrivacyFlags());
        }
    }

    public PrivacyFlagsOverride entityToDataModel(PublicPrivacyFlags entity) {
            return fromPublicPrivacyFlags(entity);
    }

    private PrivacyFlagsOverride fromPublicPrivacyFlags(PublicPrivacyFlags publicPrivacyFlags) {
        return ImmutablePrivacyFlagsOverride.builder()
                .id(Option.of(publicPrivacyFlags.getId()))
                .uniqueId(publicPrivacyFlags.getUniqueId())
                .changeDate(publicPrivacyFlags.getChangeDate())
                .sourcePersonId(Option.of(publicPrivacyFlags.getSourcePersonId()))
                .sourceTeamId(Option.of(publicPrivacyFlags.getSourceTeamId()))
                .sourceOrgId(Option.of(publicPrivacyFlags.getSourceOrgId()))
                .sourceText(Option.of(publicPrivacyFlags.getSourceText()))
                .generalRequired(publicPrivacyFlags.isRequireGeneral())
                .sexualHealthRequired(publicPrivacyFlags.isRequireSexualHealth())
                .mentalHealthRequired(publicPrivacyFlags.isRequireMentalHealth())
                .socialCareRequired(publicPrivacyFlags.isRequireSocialCare())
                .accessRoute(publicPrivacyFlags.getAccessRoute())
                .override(publicPrivacyFlags.isOverride())
                .build();
    }

    private PrivacyFlags fromBaseFields(BaseFields baseFields) {
        return ImmutablePrivacyFlags.builder()
                .generalRequired(baseFields.isRequireGeneral())
                .sexualHealthRequired(baseFields.isRequireSexualHealth())
                .mentalHealthRequired(baseFields.isRequireMentalHealth())
                .socialCareRequired(baseFields.isRequireSocialCare())
                .build();
    }

    @Mapping(target = "id", ignore = true)
    @Mapping(source = "generalRequired", target = "requireGeneral")
    @Mapping(source = "sexualHealthRequired", target = "requireSexualHealth")
    @Mapping(source = "mentalHealthRequired", target = "requireMentalHealth")
    @Mapping(source = "socialCareRequired", target = "requireSocialCare")
    public abstract PublicPrivacyFlags dataModelToEntity(PrivacyFlagsOverride model);

}
