package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.AttachmentMetadata;
import com.pkb.datamodel.NewUnreadableDocument;
import com.pkb.datamodel.UnreadableDocument;
import com.pkb.entities.doc.DocAttachment;
import com.pkb.entities.doc.DocUnreadableDocument;
import io.prometheus.client.Histogram;
import org.mapstruct.Mapper;
import org.mapstruct.ReportingPolicy;

import static com.pkb.datamodel.ImmutableUnreadableDocument.unreadableDocument;

@Mapper(uses = AttachmentMetadataMapper.class, unmappedTargetPolicy = ReportingPolicy.IGNORE)
public abstract class UnreadableDocumentMapper implements TimedMapper {

    public DocUnreadableDocument dataModelToEntity(NewUnreadableDocument unreadableDocument) {
        DocUnreadableDocument result = new DocUnreadableDocument();
        result.setReported(unreadableDocument.getReported());
        result.setPersonPublicId(unreadableDocument.getPersonPublicId());
        result.setExternalMessageId(unreadableDocument.getExternalMessageId());
        result.setAttachment(attachmentMetadataToDocAttachment(unreadableDocument.getAttachment()));
        var pdfReport = unreadableDocument.pdfValidationResult();
        result.setNumberOfPages(pdfReport.numberOfPages());
        result.setEncrypted(pdfReport.isEncrypted());
        result.setPDFa1b(pdfReport.isValidPDFa1b());
        result.setParseReport(pdfReport.parseReport());
        result.setMetadata(pdfReport.metadata());
        result.setClaimedFormat(pdfReport.claimedFormat());
        result.setNumberOfBytes(pdfReport.numberOfBytes());
        return result;
    }

    public UnreadableDocument entityToDataModel(DocUnreadableDocument entity) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("unreadabledocument", "unreadabledocument").startTimer()) {
            return unreadableDocument()
                    .id(entity.getId())
                    .reported(entity.getReported())
                    .build();
        }
    }

    protected DocAttachment attachmentMetadataToDocAttachment(AttachmentMetadata attachmentMetadata) {
        DocAttachment docAttachment = new DocAttachment();

        docAttachment.setId(attachmentMetadata.getId());
        docAttachment.setUploadTime(attachmentMetadata.getUploadTime());

        return docAttachment;
    }
}
