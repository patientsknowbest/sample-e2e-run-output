package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.consent.ImmutableTeamExport;
import com.pkb.datamodel.consent.TeamExport;
import com.pkb.datamodel.consent.TeamExportDTO;
import com.pkb.entities.enums.NationalIdType;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;

import java.util.Optional;

@Mapper
public class TeamExportMapper {

    public TeamExport entityToDatamodel(TeamExportDTO entity, Optional<NationalIdType> teamNationalIdType) {
        ImmutableTeamExport.Builder builder = ImmutableTeamExport.builder()
                .id(entity.getPersonId() == null ? "null" : String.valueOf(entity.getPersonId()))
                .address1(getStringValue(entity.getAddress1()))
                .address2(getStringValue(entity.getAddress2()))
                .city(getStringValue(entity.getCity()))
                .country(getStringValue(entity.getCountry()))
                .firstName(getStringValue(entity.getFirstName()))
                .lastName(getStringValue(entity.getLastName()));

        if (isSameCountry(entity, teamNationalIdType)) {
            builder.nationalIdentifier(getStringValue(entity.getNationalId()));
            builder.nationalIdentifierType(getStringValue(entity.getNationalIdType() == null ? "null" : entity.getNationalIdType().toString()));
        } else {
            builder.nationalIdentifier("null");
            builder.nationalIdentifierType("null");
        }

        return builder.olid(StringUtils.defaultString(entity.getOlid(), "null"))
                .phone(getStringValue(entity.getPhone()))
                .postalCode(getStringValue(entity.getPostalCode()))
                .primaryEmail(getStringValue(entity.getContactValue()))
                .state(getStringValue(entity.getState()))
                .title(getStringValue(entity.getTitle()))
                .consents(entity.getConsentIdList())
                .build();
    }

    @NotNull
    private Boolean isSameCountry(TeamExportDTO entity, Optional<NationalIdType> teamNationalIdType) {
        return teamNationalIdType.map(t ->
                entity.getNationalIdType() != null &&
                        t.areUsedInSameCountry(entity.getNationalIdType())).orElse(false);
    }

    private String getStringValue(String stringValueObject) {
        return stringValueObject != null && !stringValueObject.equals("") ? stringValueObject : "null";
    }
}
