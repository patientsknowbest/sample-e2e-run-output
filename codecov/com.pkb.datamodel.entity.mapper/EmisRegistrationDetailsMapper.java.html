package com.pkb.datamodel.entity.mapper;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.crypto.SymmetricKey;
import com.pkb.datamodel.EmisRegistrationDetailsDto;
import com.pkb.datamodel.EmisRegistrationDetailsNewDto;
import com.pkb.entities.embeddables.EncryptedWrapper;
import com.pkb.entities.emis_integration.EmisRegistrationDetails;
import io.vavr.Tuple2;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

import javax.inject.Inject;
import java.time.Instant;
import java.util.Base64;

@Mapper
public abstract class EmisRegistrationDetailsMapper extends EncryptedWrapperEntityMapper {

    @Inject
    private DateTimeService dateTimeService;


    abstract EmisRegistrationDetailsDto toEmisRegistrationDetailsDto(EmisRegistrationDetails entity, String accessIdentityGuid);

    public @NotNull EmisRegistrationDetailsDto entityToDataModel(@NotNull EmisRegistrationDetails entity, SymmetricKey symmetricKey) {
        EncryptedWrapper encryptedWrapper = entity.getAccessIdentityGuidEncryptedData();
        byte[] accessIdentityGuid = symmetricKey.decrypt(encryptedWrapper.getEncryptedData(), Base64.getDecoder().decode(encryptedWrapper.getNonceBase64()), encryptedWrapper.getCipherAlgorithm());

        return toEmisRegistrationDetailsDto(entity, new String(accessIdentityGuid));
    }

    abstract EmisRegistrationDetails toEmisRegistrationDetails(EmisRegistrationDetailsDto Dto, EncryptedWrapper accessIdentityGuidEncryptedData);

    public @NotNull EmisRegistrationDetails dataModelToEntity(EmisRegistrationDetailsDto emisRegistrationDetailsDto, Tuple2<Long, SymmetricKey> accountSymmetricKeyWithId) {
        EncryptedWrapper accessIdentityGuidEncryptedData = generateEncryptedWrapper(accountSymmetricKeyWithId, emisRegistrationDetailsDto.getAccessIdentityGuid());
        return toEmisRegistrationDetails(emisRegistrationDetailsDto, accessIdentityGuidEncryptedData);
    }

    @Mapping(source = "created", target = "created")
    @Mapping(source = "created", target = "lastModified")
    @Mapping(target = "id", ignore = true)
    abstract EmisRegistrationDetails toEmisRegistrationDetails(EmisRegistrationDetailsNewDto newDto, EncryptedWrapper accessIdentityGuidEncryptedData, Instant created);

    public @NotNull EmisRegistrationDetails newDataModelToNewEntity(EmisRegistrationDetailsNewDto emisRegistrationDetailsNewDto, Tuple2<Long, SymmetricKey> accountSymmetricKeyWithId) {
        EncryptedWrapper accessIdentityGuidEncryptedData = generateEncryptedWrapper(accountSymmetricKeyWithId, emisRegistrationDetailsNewDto.getAccessIdentityGuid());
        return toEmisRegistrationDetails(emisRegistrationDetailsNewDto, accessIdentityGuidEncryptedData, dateTimeService.now());
    }
}
