package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.AccountKeys;
import com.pkb.datamodel.AppMenuDataWithSourceObjects;
import com.pkb.datamodel.DocumentMessage;
import com.pkb.datamodel.ImmutableDocumentMessage;
import com.pkb.datamodel.ReferenceDatumCache;
import com.pkb.datamodel.crypto.CryptoService;
import com.pkb.datamodel.entity.ClinicalDataDeserializer;
import com.pkb.datamodel.user.Patient;
import com.pkb.entities.app.AppMenuData;
import io.prometheus.client.Histogram;
import io.vavr.collection.List;
import io.vavr.collection.Map;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;

import javax.inject.Inject;
import java.util.Objects;
import java.util.UUID;

import static java.util.Comparator.comparing;

@Mapper
public abstract class DocumentMessageMapper implements TimedMapper {
    @Inject
    private PrivacyFlagsMapper privacyFlagsMapper;

    @Inject
    private SourceDetailsMapper sourceDetailsMapper;

    @Inject
    private CryptoService cryptoService;

    @Inject
    private ClinicalDataDeserializer dataDeserializer;

    @Inject
    private InstantMapper instantMapper;

    @NotNull
    public List<DocumentMessage> menuDataToDocumentMessage(@NotNull List<AppMenuDataWithSourceObjects> menuData,
                                                           @NotNull Map<Long, AccountKeys> accountKeysByAccountId,
                                                           @NotNull ReferenceDatumCache specialtyCache,
                                                           @NotNull List<Patient> patients,
                                                           @NotNull List<AppMenuData> allMenuDataWithTheSameEncounterId) {
        Map<UUID, io.vavr.collection.List<AppMenuData>> allMenuDataWithEncounterIdAsKey = allMenuDataWithTheSameEncounterId
                .groupBy(AppMenuData::getUuid01);
        return menuData
                .map(menuDataWrapper -> menuDataToDocumentMessage(
                        menuDataWrapper,
                        tryDecrypt(menuDataWrapper.getAppMenuData(), accountKeysByAccountId),
                        patients
                                .filter(p -> menuDataWrapper.getAppMenuData().getString03().equals(p.getId().toString()))
                                .peek(),
                        findMostRecentlyCreatedData(allMenuDataWithEncounterIdAsKey.get(menuDataWrapper.getAppMenuData().getUuid01()).get())
                                .getOrElseThrow(() -> new IllegalStateException("No data"))
                                .getId(),
                        specialtyCache));
    }

    private Option<AppMenuData> findMostRecentlyCreatedData(List<AppMenuData> menuData) {
        return menuData.maxBy(comparing(data -> data.getBaseFields().getPersistedDate()));
    }

    private Option<byte[]> tryDecrypt(AppMenuData menuData, Map<Long, AccountKeys> accountKeysByAccountId) {
        return accountKeysByAccountId.get(menuData.getAccountId())
                .map(keys -> cryptoService.decrypt(keys, menuData));
    }

    private DocumentMessage menuDataToDocumentMessage(AppMenuDataWithSourceObjects menuDataWrapper, Option<byte[]> plaintext, @NotNull Patient patient,
            @NotNull Long latestId,
            @NotNull ReferenceDatumCache specialtyCache) {

        AppMenuData menuData = menuDataWrapper.getAppMenuData();

        try (Histogram.Timer ignored = MAP_TIMER.labels("document_message", "message").startTimer()) {
            return plaintext.map(pt -> initBuilderWithDecryptedData(menuData, pt, specialtyCache))
                    .getOrElse(ImmutableDocumentMessage::builder)
                    .id(menuData.getId())
                    .uniqueId(menuData.getBaseFields().getUniqueId())
                    .privacyFlags(privacyFlagsMapper.entityToDataModel(menuData.getBaseFields()))
                    .source(sourceDetailsMapper.entityToDataModel(menuData.getSourceDetails(), menuDataWrapper.getSourcePerson().getOrNull()))
                    .type(string14toDocumentType(menuData))
                    .patient(patient)
                    .encounter(menuData.getUuid01())
                    .externalMessageId(menuData.getString10())
                    .accountId(menuData.getAccountId())
                    .created(instantMapper.mapToZonedDateTime(menuData.getDate01()))
                    .persisted(instantMapper.mapToZonedDateTimeOption(menuData.getBaseFields().getPersistedDate()))
                    .current(Objects.equals(menuData.getId(), latestId))
                    .status(com.pkb.datamodel.Message.StatusType.valueOf(menuData.getString04()))
                    .build();
        }
    }

    private DocumentMessage.DocumentType string14toDocumentType(AppMenuData menuData) {
        return DocumentMessage.DocumentType.valueOf(menuData.getString14());
    }


    private ImmutableDocumentMessage.Builder initBuilderWithDecryptedData(AppMenuData menuData, byte[] plaintext,
            ReferenceDatumCache specialtyCache) {
        return dataDeserializer.deserialize(plaintext, fields ->
                ImmutableDocumentMessage.builder()
                        .specialty(specialtyCache.lookup(Option.of(menuData.getSourceDetails().getOrg()),
                                Option.of((String) fields.getOrDefault("hospitalServiceCode", null)))));
    }
}
