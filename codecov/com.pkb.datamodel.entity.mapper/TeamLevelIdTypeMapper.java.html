package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutableTeamLevelIdType;
import com.pkb.datamodel.TeamLevelIdType;
import com.pkb.entities.pub.PublicTeamLevelIdType;
import io.prometheus.client.Histogram;
import org.mapstruct.Mapper;

import javax.inject.Inject;

@Mapper(uses = TeamMapper.class)
public abstract class TeamLevelIdTypeMapper implements TimedMapper {

    @Inject
    private TeamMapper teamMapper;

    public TeamLevelIdType entityToDataModel(PublicTeamLevelIdType entity) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("team_level_id_type", "team_level_id_type").startTimer()) {
            return ImmutableTeamLevelIdType.builder()
                    .id(entity.getId())
                    .name(entity.getName())
                    .publicId(entity.getPublicId())
                    .assigningAuthority(entity.getHl7AssigningAuthority())
                    .typeCode(entity.getHl7TypeCode())
                    .team(teamMapper.entityToDataModel(entity.getTeam()))
                    .build();
        }
    }
}
