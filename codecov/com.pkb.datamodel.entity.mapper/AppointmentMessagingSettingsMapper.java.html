package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.AppointmentMessagingSettings;
import com.pkb.datamodel.ImmutableAppointmentMessagingSettings;
import com.pkb.entities.pub.PublicAppointmentMessagingSettings;
import org.mapstruct.Mapper;

import java.util.Optional;

@Mapper
public class AppointmentMessagingSettingsMapper {

    public AppointmentMessagingSettings entityToDataModel(PublicAppointmentMessagingSettings entity) {
        return ImmutableAppointmentMessagingSettings.builder()
                .orgId(entity.getOrgId())
                .appointmentMessagingEnabled(entity.isAppointmentMessagingEnabled())
                .appointmentMessagingTeamId(Optional.ofNullable(entity.getAppointmentMessagingTeamId()))
                .build();
    }

    public PublicAppointmentMessagingSettings dataModelToEntity(AppointmentMessagingSettings model) {
        PublicAppointmentMessagingSettings entity = new PublicAppointmentMessagingSettings();
        entity.setOrgId(model.getOrgId());
        entity.setAppointmentMessagingEnabled(model.isAppointmentMessagingEnabled());
        entity.setAppointmentMessagingTeamId(model.getAppointmentMessagingTeamId().orElse(null));
        return entity;
    }

}
