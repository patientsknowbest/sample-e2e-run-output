package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.ImmutablePatientDestroyCall;
import com.pkb.datamodel.PatientDestroyCall;
import com.pkb.entities.pub.PublicPatientDestroyCall;
import io.prometheus.client.Histogram;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

@Mapper
public abstract class PatientDestroyCallMapper implements TimedMapper {

    public PatientDestroyCall entityToDataModel(PublicPatientDestroyCall destroyCall) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("patient_destroy_call", "patient_destroy_call").startTimer()) {
            return ImmutablePatientDestroyCall.builder()
                    .patientId(destroyCall.getPatientId())
                    .callerOrgPublicId(destroyCall.getCallerOrgPublicId())
                    .calledAt(destroyCall.getCalledAt())
                    .build();
        }
    }

    @Mapping(target = "id", ignore = true)
    public abstract PublicPatientDestroyCall dataModelToNewEntity(PatientDestroyCall destroyCall);

}
