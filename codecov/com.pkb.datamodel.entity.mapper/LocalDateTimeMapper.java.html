package com.pkb.datamodel.entity.mapper;

import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;
import org.mapstruct.Mapper;

import javax.annotation.Nullable;
import java.time.Instant;
import java.time.LocalDate;
import java.time.ZoneId;
import java.time.format.DateTimeFormatter;

@Mapper
public interface LocalDateTimeMapper {
    DateTimeFormatter DEFAULT_DATE_FORMATTER = DateTimeFormatter.ofPattern("yyyy-MM-dd");

    default Option<LocalDate> getDeceasedDate(@Nullable Instant dod) {
        // The database is storing an Instant with no timezone information.
        // Assume the system default timezone for now (this is the same as what
        // happened when hibernate used LocalDate, except hibernate did it instead of us)
        return Option.of(dod).map(d -> d.atZone(ZoneId.systemDefault()).toLocalDate());
    }

    default Option<LocalDate> getDateOfBirth(@Nullable String dob) {
        return Option.of(dob)
                .filter(d -> StringUtils.isNotBlank(d))
                .map(dateString -> LocalDate.parse(dateString, DEFAULT_DATE_FORMATTER));
    }
}
