package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.BulkInvitationOutcome;
import com.pkb.datamodel.ImmutableBulkInvitationOutcome;
import com.pkb.entities.pub.PublicBulkInvitationOutcome;
import io.vavr.control.Option;
import org.mapstruct.Mapper;

@Mapper
public class BulkInvitationOutcomeMapper implements OptionAwareMapper {

    public BulkInvitationOutcome entityToDataModel(PublicBulkInvitationOutcome entity) {
        return ImmutableBulkInvitationOutcome.builder()
                .id(entity.getId())
                .uploadedAt(entity.getUploadedAt())
                .completedAt(Option.of(entity.getCompletedAt()))
                .uploadedById(entity.getUploadedById())
                .invitingTeamId(entity.getInvitingTeamId())
                .csvEnrichedWithOutcomes(Option.of(entity.getCsvEnrichedWithOutcomes()))
                .status(entity.getStatus())
                .build();
    }

    public PublicBulkInvitationOutcome dataModelToEntity(BulkInvitationOutcome dataModel) {
        PublicBulkInvitationOutcome result = new PublicBulkInvitationOutcome();
        result.setId(dataModel.getId());
        result.setCompletedAt(dataModel.getCompletedAt().getOrNull());
        result.setCsvEnrichedWithOutcomes(dataModel.getCsvEnrichedWithOutcomes().getOrNull());
        result.setInvitingTeamId(dataModel.getInvitingTeamId());
        result.setUploadedAt(dataModel.getUploadedAt());
        result.setStatus(dataModel.getStatus());
        result.setUploadedById(dataModel.getUploadedById());
        return result;
    }
}
