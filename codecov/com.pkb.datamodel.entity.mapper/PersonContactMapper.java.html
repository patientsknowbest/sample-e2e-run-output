package com.pkb.datamodel.entity.mapper;

import com.pkb.datamodel.contact.EmailContact;
import com.pkb.datamodel.contact.ImmutableEmailContact;
import com.pkb.datamodel.contact.NewPersonContact;
import com.pkb.datamodel.contact.PersonContact;
import com.pkb.datamodel.contact.PhoneContact;
import com.pkb.entities.pub.PublicPKBPerson;
import com.pkb.entities.pub.PublicPersonContact;
import io.prometheus.client.Histogram;
import io.vavr.collection.List;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;
import org.mapstruct.Mapper;
import org.mapstruct.ReportingPolicy;
import org.mapstruct.factory.Mappers;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import static com.pkb.datamodel.contact.ImmutablePhoneContact.builder;
import static com.pkb.datamodel.entity.mapper.ImmutableMappingError.mappingError;
import static java.text.MessageFormat.format;

@Mapper(uses = { OrgMapper.class, PersonMapper.class }, unmappedTargetPolicy = ReportingPolicy.IGNORE)
public abstract class PersonContactMapper implements TimedMapper {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    protected OrgMapper orgMapper = Mappers.getMapper(OrgMapper.class);

    public abstract PublicPersonContact dataModelToEntity(NewPersonContact newPersonContact);

    public Either<MappingError, PersonContact> entityToDataModel(PublicPersonContact entity) {
        switch (entity.getContactType()) {
            case EMAIL:
                return Either.right(toEmailContact(entity));
            case PHONE:
                return Either.left(mappingError(format("Not implemented mapper for {0}", entity)));
            default:
                return Either.left(mappingError(format("Unsupported contact type {0} on id {1}", entity.getContactType(), entity.getId())));
        }
    }

    /**
     * Warning! this method will ignore unknown contact types (it'll just log them)
     */
    public List<PersonContact> entityToDataModel(List<PublicPersonContact> entities) {
        return entities
                .map(this::entityToDataModel)
                .peek(i -> i.peekLeft(error -> LOGGER.warn(error.error())))
                .filter(Either::isRight)
                .map(Either::get);
    }

    private EmailContact toEmailContact(PublicPersonContact entity) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("email_contact", "contact").startTimer()) {
            return ImmutableEmailContact.builder()
                    .value(entity.getContactValue())
                    .id(entity.getId())
                    .confirmed(entity.isConfirmed())
                    .primary(entity.isPrimary())
                    .build();

        }
    }

    public Option<PhoneContact> personEntityToPhoneContact(PublicPKBPerson entity) {
        try (Histogram.Timer ignored = MAP_TIMER.labels("phone_contact", "person").startTimer()) {
            return Option.of(entity.getPhone())
                    .filter(StringUtils::isNotBlank)
                    .map(phone -> builder().value(phone).build());
        }
    }

}
