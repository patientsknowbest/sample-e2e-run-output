package com.pkb.domain.communication;

import com.pkb.authz.AuthorizationService;
import com.pkb.datamodel.Communication;
import com.pkb.domain.SynchUploadedDataService;
import com.pkb.domain.SystemUserTypeNotPermittedException;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.repository.CommunicationQuery;
import com.pkb.domain.repository.CommunicationRepository;
import io.vavr.control.Either;

import static com.pkb.entities.enums.UploadedDataDestination.ENCOUNTER;
import static io.vavr.control.Either.left;

public class CommunicationService {
    private final AuthorizationService authorizationService;
    private final CommunicationRepository communicationRepository;
    private final SynchUploadedDataService synchUploadedDataService;

    public CommunicationService(AuthorizationService authorizationService,
                                CommunicationRepository communicationRepository,
                                SynchUploadedDataService synchUploadedDataService) {
        this.authorizationService = authorizationService;
        this.communicationRepository = communicationRepository;
        this.synchUploadedDataService = synchUploadedDataService;
    }

    public Either<RuntimeException, CriteriaResult<Communication>> search(CommunicationQuery communicationQuery) {
        if (communicationQuery.authzData().inputs().authenticatedSystemUser().isDefined()) {
            return left(new SystemUserTypeNotPermittedException());
        }
        return authorizationService.authorize(communicationQuery.authzData())
                .checkSingleResult(() -> {
                    synchUploadedDataService.synchUploadedData(communicationQuery.authzData(), ENCOUNTER);
                    return communicationRepository.findCommunications(communicationQuery);
                });
    }
}
