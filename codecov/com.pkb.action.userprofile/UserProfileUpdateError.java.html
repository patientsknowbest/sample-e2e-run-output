package com.pkb.action.userprofile;

import com.pkb.common.base.Error;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

public class UserProfileUpdateError extends Error {

    public UserProfileUpdateError(@NotNull String errorCode, @NotNull String description, String... arguments) {
        super(errorCode, description, arguments);
    }

    @Override protected boolean canEqual(Object other) {
        return other instanceof UserProfileUpdateError;
    }
}

class DuplicateTeamLevelIdError extends UserProfileUpdateError {

    DuplicateTeamLevelIdError(@NotNull String teamLevelIdTypeName, @NotNull String teamLevelId,
            @Nullable String existingPersonTitle,
            @NotNull String existingPersonFirstName,
            @NotNull String existingPersonLastName) {
        super("manageprofile.duplicate.team.level.id", "Person already exists with ID",
                teamLevelIdTypeName, teamLevelId, existingPersonTitle, existingPersonFirstName,
                existingPersonLastName);
    }

}
