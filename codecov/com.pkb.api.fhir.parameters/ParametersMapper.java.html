package com.pkb.api.fhir.parameters;

import com.pkb.api.fhir.NationalIdToIdentifierMapper;
import com.pkb.datamodel.projections.PatientDateOfLastDataPoint;
import com.pkb.datamodel.projections.PatientStatus;
import com.pkb.deletion.model.DeleteRequestStatus;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.hl7.fhir.dstu3.model.BooleanType;
import org.hl7.fhir.dstu3.model.DateTimeType;
import org.hl7.fhir.dstu3.model.DateType;
import org.hl7.fhir.dstu3.model.IdType;
import org.hl7.fhir.dstu3.model.Identifier;
import org.hl7.fhir.dstu3.model.Parameters;
import org.hl7.fhir.dstu3.model.Resource;
import org.hl7.fhir.dstu3.model.StringType;
import org.hl7.fhir.dstu3.model.Type;
import org.jetbrains.annotations.NotNull;

import java.time.Instant;
import java.time.LocalDate;
import java.util.Date;
import java.util.TimeZone;
import java.util.UUID;
import java.util.function.Function;

import static ca.uhn.fhir.model.api.TemporalPrecisionEnum.SECOND;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.Predicates.instanceOf;

public class ParametersMapper {

    static final String PKB_ID_SYSTEM = "urn:uuid:cd97da65-0aa4-43a6-b695-cfd0c6f3267d";

    public static Parameters patientStatusToParameters(List<PatientStatus> patientStatuses) {
        Parameters parameters = new Parameters();
        patientStatuses.forEach(patientStatus -> {
            Parameters.ParametersParameterComponent parameterComponent = new Parameters.ParametersParameterComponent().setName("patientStatus");

            Identifier identifier = NationalIdToIdentifierMapper.toIdentifier(patientStatus.getValidNationalId());
            parameterComponent.addPart(createComponent("patientIdentifier", identifier));

            patientStatus.getRegistered().peek(registered -> parameterComponent.addPart(createParameter("registered", registered)));

            patientStatus.getDateOfBirth().peek(dateOfBirth -> parameterComponent.addPart(createParameter("dateOfBirth", dateOfBirth)));

            patientStatus.getDeceased().peek(deceased -> parameterComponent.addPart(createParameter("deceased", deceased)));

            patientStatus.getDiagnostics().peek(diagnostics -> parameterComponent.addPart(createParameter("diagnostics", diagnostics.name())));

            parameters.addParameter(parameterComponent);
        });
        return parameters;
    }

    public static Parameters dateOfLastDataPointToParameters(List<? extends PatientDateOfLastDataPoint> lastDataPoints) {
        Parameters parameters = new Parameters();
        lastDataPoints.forEach(lastDataPoint -> {
            Parameters.ParametersParameterComponent parameterComponent = getComponent(lastDataPoint);
            parameters.addParameter(parameterComponent);
        });
        return parameters;
    }

    public static Parameters deleteOperationIdToParameters(UUID operationId) {
        Parameters parameters = new Parameters();
        parameters.addParameter(createParameter("operationId", operationId.toString()));
        return parameters;
    }

    public static Parameters deleteOperationStatusParameters(UUID operationId, DeleteRequestStatus status) {
        Parameters parameters = new Parameters();
        parameters.addParameter(createParameter("operationId", operationId.toString()));
        parameters.addParameter(createParameter("status", status.toString()));
        return parameters;
    }

    public static Parameters dateOfLastDataPointToParameters(Option<PatientDateOfLastDataPoint> lastDataPoints) {
        Parameters parameters = new Parameters();
        lastDataPoints.map(lastDataPoint -> parameters.addParameter(getComponent(lastDataPoint)));
        return parameters;
    }

    @NotNull
    private static Parameters.ParametersParameterComponent getComponent(PatientDateOfLastDataPoint lastDataPoint) {
        Parameters.ParametersParameterComponent parameterComponent = new Parameters.ParametersParameterComponent().setName("lastDataPointsResult");

        Identifier identifier = lastDataPoint.getValidNationalId().map(NationalIdToIdentifierMapper::toIdentifier).getOrElse(new Identifier());
        lastDataPoint.getIdValue().peek(pkbId -> {
            identifier.setSystem(PKB_ID_SYSTEM);
            identifier.setValue(pkbId.toString());
        });
        if (identifier.hasSystem()) {
            parameterComponent.addPart(createComponent("patientIdentifier", identifier));
        }
        parameterComponent.addPart(createParameter("lastRecordUpdate", lastDataPoint.dateOfLastDataPoint()));
        return parameterComponent;
    }

    private static Parameters.ParametersParameterComponent createParameter(String name, Object value) {
        Type valueType = Match(value).of(
                Case($(instanceOf(String.class)), (Function<String, StringType>) StringType::new),
                Case($(instanceOf(Boolean.class)), (Function<Boolean, BooleanType>) BooleanType::new),
                Case($(instanceOf(LocalDate.class)), localDateValue -> new DateType(localDateValue.getYear(), localDateValue.getMonthValue() - 1, localDateValue.getDayOfMonth())),
                Case($(instanceOf(Instant.class)), instantValue -> new DateTimeType(Date.from(instantValue), SECOND, TimeZone.getTimeZone("UTC"))),
                Case($(instanceOf(UUID.class)), uuidValue -> new IdType(uuidValue.toString())), // Hapi can't return uuidType
                Case($(), () -> {
                    throw new RuntimeException(value.getClass() + " is not supported");
                }));
        return createComponent(name, valueType);
    }

    public static Parameters.ParametersParameterComponent createComponent(String name, Type valueType) {
        return new Parameters.ParametersParameterComponent()
                .setName(name)
                .setValue(valueType);
    }
    
    public static Parameters.ParametersParameterComponent createResource(String name, Resource resource){
        return new Parameters.ParametersParameterComponent()
                .setName(name)
                .setResource(resource);
    }
}
