package com.pkb.service.test;

import com.pkb.app.entity.EHRData;
import com.pkb.app.entity.EHRSearch;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.PKBFilter;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.exception.PKBException;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.test.entity.PathologyReportDTO;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import org.jetbrains.annotations.NotNull;

import java.util.List;
import java.util.Optional;

public class PathologyReportManager extends TransactionManager {

    public PathologyReportManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider) {
        super(config, beanFactory, dateTimeService, uuidProvider);
    }

    public Optional<PathologyReportDTO> getPathologyReport(@NotNull LoggedInEHRRequestContext requestContext, Long pathologyReportId, long accountId) {
        EHRSearch<PathologyReportDTO> ehrSearch = new EHRSearch<>(accountId, PathologyReportDTO.class,
                PathologyReportDTO.MS_PATH);
        ehrSearch.addFilter(new PKBFilter("id", PKBFilter.Operator.EQUAL, pathologyReportId));
        List<EHRData> ehrList = beanFactory.getEhrRemote().queryAndDecryptEHRData(ehrSearch, requestContext);
        List<PathologyReportDTO> reportList = beanFactory.getEhrRemote().populateDTOList(ehrList, PathologyReportDTO.class, requestContext);
        if ((reportList == null) || reportList.isEmpty()) {
            return Optional.empty();
        }
        return Optional.of(reportList.get(0));
    }

    public List<PathologyReportDTO> getPathologyReports(@NotNull LoggedInEHRRequestContext requestContext, long accountId) {
        try {
            List<PathologyReportDTO> pathologyReports = retrievePathologyReports(requestContext, accountId);

            // Order by *derived* date field
            pathologyReports.sort((dto1, dto2) -> {
                if ((dto1.getDate() == null) && (dto2.getDate() != null)) {
                    return 1;
                }
                if ((dto1.getDate() != null) && (dto2.getDate() == null)) {
                    return -1;
                }
                if ((dto1.getDate() == null) && (dto2.getDate() == null)) {
                    return 0;
                }
                return dto2.getDate().compareTo(dto1.getDate());
            });

            return pathologyReports;
        } catch (Exception exception) {
            throw new PKBException("Error while retrieving PathologyReports from account-" + accountId, exception);
        }
    }

    private List<PathologyReportDTO> retrievePathologyReports(@NotNull LoggedInEHRRequestContext requestContext, long accountId) {
        EHRSearch<PathologyReportDTO> ehrSearch = new EHRSearch<>(accountId, PathologyReportDTO.class,
                PathologyReportDTO.MS_PATH);
        ehrSearch.setLatestPerTypeFilter(EHRData.ENTERED_DATE, EHRData.UNIQUE_ID);
        ehrSearch.setOrderBy("date01", EHRSearch.OrderByDirection.Desc);
        List<EHRData> ehrList = beanFactory.getEhrRemote().queryAndDecryptEHRData(ehrSearch, requestContext);
        return beanFactory.getEhrRemote().populateDTOList(ehrList, PathologyReportDTO.class, requestContext);
    }
}
