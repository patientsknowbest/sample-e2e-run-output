    public Optional<TestHistoryDTO> getTestHistory(LoggedInEHRRequestContext requestContext, Long loincTestId, Long loincMappingId,
                                                   long patientUserId, IncludeDeletedResults deleted) {

        // retrieves ALL of the history(!) for the selected test or mapping
        TestHistoryFilter filter = new TestHistoryFilter(patientUserId);
        if (loincTestId != null) {
            filter.addLoincTestId(loincTestId);
        } else if (loincMappingId != null) {
            filter.addLoincMappingId(loincMappingId);
        } else {
            throw new IllegalArgumentException("both loincId and loincMappingId are null");
        }

        List<TestHistoryDTO> resultList;
        try {
            resultList = accessTestHistory(filter, requestContext, null, deleted);
        } catch (Exception e) {
            throw new PKBException("failed getting test history for " + patientUserId, e);
        }

        if (resultList.isEmpty()) {
            return empty();
        } else if (resultList.size() == 1) {
            return Optional.of(resultList.get(0));
        } else {
            // more than one?  this is a bug
            throw new PKBException(
                    "Expected single TestHistoryDTO for patient " + patientUserId + " but got " +
                            resultList.size() + " resultMenudataIds: " + getDebugInfo(resultList));
        }
    }

    private String getDebugInfo(List<TestHistoryDTO> historyList) {
        return historyList.stream().filter(dto -> isNotEmpty(dto.getTestResultList()))
                .map(dto -> dto.getTestResultList().stream().map(tr -> tr.getId().toString()).collect(joining(",", "[", "]")))
                .collect(joining(","));
    }

    private List<SavedIdsDTO> doSaveTestResults(List<TestResultDTO> testResults, long patientUserId, Function<Long, List<SavedIdsDTO>> testResultsSaver) {
        Long accountId = userManager.getDefaultAccountId(patientUserId);
        try {
            List<SavedIdsDTO> savedTestResults = testResultsSaver.apply(accountId);

            // update the uniqueId and ID values in the DTOs
            for (int i = 0; i < savedTestResults.size(); i++) {
                testResults.get(i).setId(savedTestResults.get(i).getId());
                testResults.get(i).getBaseFields().setUniqueId(savedTestResults.get(i).getUniqueId());
            }

            return savedTestResults;
        } catch (Exception e) {
            throw new PKBException(format("failed saving test results for %s", patientUserId), e);
        }
    }


    /**
     * Returns the single latest result for each requested test. The list of
     * {@code testIds} must be non-null and non-empty. The internal LOINC Id
     * will be used as the key to the returned {@code Map}.
     *
     * @param patientId      The patient whose results should be returned.
     * @param testIds        A non-null, non-empty list of internal LOINC Ids.
     * @param includeDeleted
     * @return The latest test results for each test in {@code testIds}.
     */
    public Map<Long, TestResultDTO> getLatestTestResults(@NotNull LoggedInEHRRequestContext requestContext, long patientId,
                                                         List<Long> testIds, IncludeDeletedResults deleted) {

        if ((testIds == null) || testIds.isEmpty()) {
            throw new IllegalArgumentException("A list of testIds must be provided");
        }

        TestHistoryFilter filter = new TestHistoryFilter(patientId);
        filter.limitRangeLatestOnly();
        filter.addLoincTestIdList(testIds);

        List<TestResultDTO> resultList = null;
        try {
            resultList = accessTestResults(filter, requestContext, deleted);
        } catch (Exception e) {
            throw new PKBException("Exception while getting latest test results for user " + patientId, e);
        }

        // put into a map (simple: loinc IDs will be unique)
        Map<Long, TestResultDTO> map = new HashMap<>();
        for (TestResultDTO result : resultList) {
            map.put(result.getLoincTestId(), result);
        }
        return map;
    }

    /**
     * This wrapper around the bean call is to make sure we load queued test results -- data loaded via the HL7 API
     * (which doesn't have decryption access, and thus can't use private mappings).
     *
     * @param testBean
     * @param filter
     * @return
     */
    private List<TestResultDTO> accessTestResults(TestHistoryFilter filter, @NotNull LoggedInEHRRequestContext requestContext, IncludeDeletedResults deleted) {

        try {
            synchUploadedData(Collections.singletonList(filter.getPatientId()), requestContext);

            long fromMS = System.currentTimeMillis();
            LOGGER.info("TIMING: accessTestHistory: accessTestResults for {}", filter.getPatientId());

            List<TestResultDTO> results = loincManager.getTestResults(requestContext, filter, null, deleted);

            LOGGER.info("TIMING: accessTestHistory({}): done getTestHistoryList for {}",
                    (System.currentTimeMillis() - fromMS), filter.getPatientId());
            return results;
        } catch (Exception e) {
            throw new PKBException("", e);
        }
    }

    /**
     * This wrapper around the bean call is to make sure we load queued test results -- data loaded via the HL7 API
     * (which doesn't have decryption access, and thus can't use private mappings).
     *
     * @param testBean
     * @param filter
     * @return
     */
    private List<TestHistoryDTO> accessTestHistory(
            TestHistoryFilter filter,
            LoggedInEHRRequestContext context,
            Integer latestResults,
            IncludeDeletedResults deleted) {

        try {
            synchUploadedData(Collections.singletonList(filter.getPatientId()), context);

            long fromMS = System.currentTimeMillis();
            LOGGER.info("TIMING: accessTestHistory: getTestHistoryList for {}", filter.getPatientId());

            List<TestHistoryDTO> results = loincManager.getTestHistoryList(filter, context, latestResults, deleted);

            filterTestResultsForGlasgowPatients(context, results);
            LOGGER.info("No of results after filtering results for glasgow patients-{}", results.size());

            LOGGER.info("TIMING: accessTestHistory({}): done getTestHistoryList for {}",
                    (System.currentTimeMillis() - fromMS), filter.getPatientId());

            return results;
        } catch (Exception e) {
            throw new PKBException("", e);
        }
    }

    // Horrible temporary kludge for SciStore 8/2/2016
    //
    // From https://pkbdev.atlassian.net/browse/PHR-460
    //
    //		If a test result comes from orgId=55508 (production glasgow org id)
    //		For patients do not show test results unless the LOINC code is one of:
    //			8123-2
    //			24467-3
    //			20447-9
    //
    // From https://pkbdev.atlassian.net/browse/PHR-2107
    //
    //     Extend the same kludge to all data with sourceText = 'SciStore'

    public static boolean isLoincHiddenFromGlasgowPatients(String loinc) {
        return !SCISTORE_LOINC_WHITELIST.contains(loinc);
    }

    private void filterTestResultsForGlasgowPatients(LoggedInEHRRequestContext context, List<TestHistoryDTO> results) {

        try {
            PKBPerson requestor = userManager.getPKBPerson(context.getAccessingUserId());
            if (requestor.isPatient()) {

                for (Iterator<TestHistoryDTO> historyIterator = results.iterator(); historyIterator.hasNext(); ) {
                    TestHistoryDTO history = historyIterator.next();

                    if ((history.getLoincTest() == null) || isLoincHiddenFromGlasgowPatients(history.getLoincTest().getLoinc())) {
                        history.getTestResultList().removeIf(resultDTO ->
                                SCISTORE_SOURCE_TEXT.equals(resultDTO.getSource().getText()));
                        if (history.getTestResultList().isEmpty()) {
                            historyIterator.remove();
                        }
                    }
                }
            }
        } catch (Exception e) {
            LOGGER.error("NON-FATAL error: failed filtering results for SCIStore", e);
        }
    }

    /**
     * Uses transactions, and commits after each UploadedData is processed... if this needs to be called from within a
     * transaction in the action, we'll need some fancier coding here (maybe programmatically merge the messages and submit
     * at once?).
     * <p>
     * Doesn't catch/handle any exceptions; just logs errors and sometimes lets them flow out (but more often just logs)
     */
    private void synchUploadedData(Collection<Long> patientIds, @NotNull LoggedInEHRRequestContext requestContext) {
        patientIds.forEach(patientId -> {
            LOGGER.info("TIMING: synchUploadedDate: starting for {}", patientId);
            StopWatch timer = new StopWatch();
            timer.start();

            LoggedInEHRRequestContext piggybackContext = requestContext.withPiggyback(true);
            processUploadedTestResult(piggybackContext, patientId);

            LOGGER.info("TIMING: synchUploadedDate ({}): claimed and handled uploaded data for {}", timer.getTime(), patientId);

            // Also sync any queued data from devices (uses its own transactions)
            coreDevicesManager.synchUploadedData(requestContext, patientId);

            LOGGER.info("TIMING: synchUploadedDate ({}): core devices synched for {}", timer.getTime(), patientIds);
        });
    }

    private void processUploadedTestResult(@NotNull LoggedInEHRRequestContext requestContext, Long patientId) {
        uploadedDataService.synchUploadedData(
                loggedInUploadedDataProcessingContext()
                        .requestContext(requestContext)
                        .patientId(patientId)
                        .destination(UploadedData.Destination.TEST_RESULT)
                        .build());
    }

    /**
     * Collects all test results in each of the provided histories and then updates their lab discipline based on a
     * matching ReferenceDatum record. If no match is found, the original value is left unmodified.
     * <p>
     * This is a display-time call, no database updates are made.
     *
     * @param testHistories The collection of test histories to update.
     */
    public void applyLabDisciplineReferenceDataMappingsToTestHistories(Collection<TestHistoryDTO> testHistories) {
        List<TestResultDTO> testResults = testHistories.stream()
                .map(TestHistoryDTO::getTestResultList)
                .flatMap(List::stream)
                .collect(Collectors.toList());
        Map<TestResultDTO, ReferenceDataManager.RefDatumOrgAndData> testsToRefDatumMap = testResults.stream()
                .filter(resultDTO -> resultDTO.getSource().getOrgId() != null && StringUtils.isNotBlank(resultDTO.getLabDiscipline()))
                .collect(Collectors.toMap(
                        Function.identity(),
                        result -> new ReferenceDataManager.RefDatumOrgAndData(result.getSource().getOrgId(), LAB_DISCIPLINE, result.getLabDiscipline())));
        Map<ReferenceDataManager.RefDatumOrgAndData, String> refDatumToTextMap = referenceDataManager.processReferenceDatumBatch(testsToRefDatumMap,
                Collections.singletonList(LAB_DISCIPLINE));
        testsToRefDatumMap.forEach((result, refDatum) -> result.setLabDiscipline(refDatumToTextMap.getOrDefault(refDatum, result.getLabDiscipline())));
    }

    /**
     * This is called from the new FHIR API
     * We haven't ported over the code to sync uploaded data yet, and it's out of scope for
     * the current implementation. However, it's not safe to go without it as the user could
     * be missing some test results from view.
     */
    @Override
    public void syncUploadedData(LoggedInEHRRequestContext requestContext, Collection<Long> targetPersonIds) {
        synchUploadedData(targetPersonIds, requestContext);
    }
