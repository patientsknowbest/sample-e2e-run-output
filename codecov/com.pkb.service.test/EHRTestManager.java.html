package com.pkb.service.test;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.coding.entity.CodingReceived;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.entities.enums.CodingDataType;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.service.coding.CodingManager;
import com.pkb.service.ehr.DeduplicatorManager;
import com.pkb.service.ehr.EhrDataPointManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.test.entity.TestResultDTO;
import com.pkb.util.tolven.TolvenBeanFactory;
import io.vavr.Function2;
import io.vavr.Tuple2;
import io.vavr.Tuple3;

import java.util.List;

public class EHRTestManager extends EhrDataPointManager<TestResultDTO> {


    private static final Function2<TestResultDTO, Long, Void> CODING_RECEIVED_ID_SETTER = (testResult, privateId) -> {
        testResult.setCodingReceivedId(privateId);
        return null;
    };

    private static final Function2<TestResultDTO, PrivacyFlag, Void> PRIVACY_FLAG_HANDLER = (testResult, privacyFlag) -> {
        testResult.getBaseFields().setPrivacyFlag(privacyFlag);
        return null;
    };


    public EHRTestManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, CodingManager codingManager, DeduplicatorManager deduplicator, Class<TestResultDTO> dtoClass, MenuDataType dataType) {
        super(config, beanFactory, dateTimeService, uuidProvider, codingManager, deduplicator, dtoClass, dataType);
    }

    @Override
    protected List<Tuple2<TestResultDTO, EHRRequestContext>> adjustCodings(List<Tuple2<TestResultDTO, EHRRequestContext>> data) {
        return doAdjustCodings(data, CodingDataType.TEST, TestResultDTO::getCoding,
                CODING_RECEIVED_ID_SETTER,
                PRIVACY_FLAG_HANDLER);
    }

    @Override
    protected List<Tuple2<TestResultDTO, EHRRequestContext>> adjustCodingsNoCodingLookup(List<Tuple3<TestResultDTO, CodingReceived, EHRRequestContext>> data) {
        return doAdjustCodingsNoCodingLookup(data, CODING_RECEIVED_ID_SETTER, PRIVACY_FLAG_HANDLER);
    }
}
