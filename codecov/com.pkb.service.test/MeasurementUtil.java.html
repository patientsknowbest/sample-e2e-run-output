package com.pkb.service.test;

import com.pkb.app.entity.SourceDetails;
import com.pkb.test.entity.MeasurementDTO;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.PKBServiceUtil;
import org.jetbrains.annotations.NotNull;

import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;

import static java.util.Objects.requireNonNull;

public class MeasurementUtil {

    private PKBServiceUtil pkbServiceUtil;

    public MeasurementUtil(PKBServiceUtil pkbServiceUtil) {
        this.pkbServiceUtil = pkbServiceUtil;
    }

    public String makeCalculatedBMISourceText(@NotNull MeasurementDTO weightDTO,
            @NotNull MeasurementDTO heightDTO,
            @NotNull PKBPerson pkbPerson) {
        SourceDetails weightSource = weightDTO.getSource();
        SourceDetails heightSource = heightDTO.getSource();
        Object[] args = {
                formatMeasureDateAtTime(weightDTO, pkbPerson), weightSource.getDisplayTextFull(), weightSource.getRoute(),
                formatMeasureDateAtTime(heightDTO, pkbPerson), heightSource.getDisplayTextFull(), heightSource.getRoute() };
        return pkbServiceUtil.getMessage("measurements.txt.calculated_bmi_source_text", args, pkbPerson.getLocale());
    }

    private String formatMeasureDateAtTime(MeasurementDTO measurementDTO, PKBPerson pkbPerson) {
        DateTimeFormatter dateAtTimeFormatter = pkbServiceUtil.getMeasurementDateAtTimeFormatter(pkbPerson.getLocale());
        LocalDateTime measureDate = measurementDTO.getMeasureDate()
                .atZone(requireNonNull(pkbPerson.getZoneId(), String.format("No ZoneId for user: %s", pkbPerson.getId())))
                .toLocalDateTime();
        return dateAtTimeFormatter.format(measureDate);
    }
}
