package com.pkb.apiservice;

import com.pkb.common.datetime.DateTimeService;
import io.jsonwebtoken.JwtBuilder;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.SignatureAlgorithm;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import javax.crypto.spec.SecretKeySpec;
import java.security.Key;
import java.util.Date;
import java.util.UUID;

import static com.pkb.util.Constants.APPLICATION_TZ;


@Component
public class BTGTokenService {

    public static final int BTG_TTL = 30;
    private static final String ISSUER = "PatientsKnowBest Inc.";

    @Autowired
    private DateTimeService dateTimeService;

    private final SignatureAlgorithm signatureAlgorithm;
    private final Key signKey;

    public BTGTokenService() {
        String secret = "QD=U7s2D+GMi9Qz8PMLHVYPxb+cPsu#Q"; //we could put this into an env or other such

        signatureAlgorithm = SignatureAlgorithm.HS256;
        signKey = new SecretKeySpec(secret.getBytes(), signatureAlgorithm.getJcaName());
    }

    /**
     * create BTG token using JWT
     *
     * @param patientId
     *            patient id
     * @return returns JWT token
     */
    public String create(long patientId) {
        return create(patientId, Date.from(dateTimeService.now()));
    }

    /**
     * @param patientId
     *            patient id
     * @param issueDate
     *            issueDate of token creation. expiry = issue date + TTL
     * @return returns JWT token
     */
    public String create(long patientId, Date issueDate) {
        //set the JWT claims
        JwtBuilder jwtBuilder = Jwts.builder()
                .setId(UUID.randomUUID().toString())
                .setExpiration(Date.from(issueDate.toInstant().atZone(APPLICATION_TZ).plusMinutes(BTG_TTL).toInstant()))
                .setSubject(Long.toString(patientId))
                .setIssuer(ISSUER)
                .signWith(signatureAlgorithm, signKey);

        return jwtBuilder.compact();
    }

    /**
     * completes silently if a token is valid. otherwise, it throws an exception
     */
    public void validate(String signedToken, long patientId) {
        Jwts.parser()
                .setClock(() -> Date.from(dateTimeService.now()))
                .setSigningKey(signKey)
                .requireIssuer(ISSUER)
                .requireSubject(Long.toString(patientId))
                .parseClaimsJws(signedToken).getBody();
    }

}
