package com.pkb.apiservice;

import com.pkb.api.error.ApiValidationError;
import com.pkb.api.util.converter.DocAttachToAttachmentInformationConverter;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.common.base.Error;
import com.pkb.common.base.ErrorBuilder;
import com.pkb.document.AttachmentDownloadStream;
import com.pkb.document.AttachmentInformation;
import com.pkb.document.entity.DocAttach;
import com.pkb.service.file.ChunkFetchInputStream;
import com.pkb.service.file.ChunkedDocManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.io.InputStream;
import java.util.Objects;
import java.util.Optional;
import java.util.UUID;

/**
 * Service to handle attachments.
 */
@Service
public class AttachmentService {

    public static final String ERROR_CODE_SHARING_DISABLED = "sharing.disabled";
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    private ChunkedDocManager chunkedDocManager;

    @Autowired
    private DocAttachToAttachmentInformationConverter docAttachToAttachmentInformationConverter;

    @Autowired
    private UserManager userManager;

    /**
     * Get the given attachment.
     * 
     * @param attachmentId
     *            The id of the attachment.
     * @param ehrRequestContext
     * @param actingPerson
     *            The {@link PKBPerson} performing the action.
     * @return {@link InputStream} which will return the requested attachment.
     */
    public Either<Error, AttachmentDownloadStream> getAttachment(Long attachmentId, @NotNull LoggedInEHRRequestContext ehrRequestContext, PKBPerson actingPerson) {
        Objects.requireNonNull(actingPerson, "Acting person cannot be null!");
        if (attachmentId == null) {
            throw new ApiValidationError("Invalid attachment id!");
        }
        Optional<DocAttach> maybeAttachmentDetails = chunkedDocManager.getAttachmentDetails(attachmentId);

        DocAttach attachmentDetails = maybeAttachmentDetails
                .orElseThrow(() -> new ApiValidationError("No attachment found for id: " + attachmentId));

        PKBPerson pkbPerson = userManager.getAccountOwner(attachmentDetails.getAccountId());

        boolean ownData = Optional.ofNullable(pkbPerson)
                .map(p -> actingPerson.getId().equals(p.getId()))
                .orElse(true);

        boolean sharingEnabled = Optional.ofNullable(pkbPerson)
                .map(PKBPerson::isSharingEnabled)
                .orElse(false);

        if (!sharingEnabled && !ownData) {
            return Either.left(ErrorBuilder.errorOf(ERROR_CODE_SHARING_DISABLED, "Sharing disabled for attachment account user."));
        }

        LoggedInEHRRequestContext decryptContext = ehrRequestContext;
        if (!ownData && pkbPerson != null) {
            decryptContext = ehrRequestContext.fillMissingContextUser(pkbPerson.getId());
        }

        UUID docId = attachmentDetails.getParentDocumentMetadataId() != null ? attachmentDetails.getParentDocumentMetadataId()
                : attachmentDetails.getAttachedDocumentMetadataId();

        ChunkFetchInputStream inputStream = new ChunkFetchInputStream(decryptContext, docId, chunkedDocManager);
        AttachmentInformation information = docAttachToAttachmentInformationConverter.convert(attachmentDetails);
        LOGGER.info("Attachment download started for id: {} by person id: {}", attachmentId, actingPerson.getId());
        return Either.right(new AttachmentDownloadStream(information, inputStream));
    }
}
