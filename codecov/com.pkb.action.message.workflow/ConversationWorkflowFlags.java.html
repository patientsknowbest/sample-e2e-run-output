package com.pkb.action.message.workflow;

import com.pkb.datamodel.DataModelStyle;
import org.immutables.value.Value;

@Value.Immutable
@DataModelStyle
public abstract class ConversationWorkflowFlags {

    public abstract boolean isArchivingEnabled();

    public abstract boolean isAssigningEnabled();

    @Value.Derived
    public boolean isWorkflowEnabled() {
        return isArchivingEnabled() || isAssigningEnabled();
    }
}
