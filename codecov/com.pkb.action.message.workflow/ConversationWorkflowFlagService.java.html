package com.pkb.action.message.workflow;

import com.google.common.collect.Sets;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.common.config.PhrConfig;
import com.pkb.entities.enums.UserType;

import java.util.Set;

import static com.pkb.entities.enums.UserType.REG_CLINICIAN;

public class ConversationWorkflowFlagService {

    private static final Set<UserType> WORKFLOW_ENABLED_USERS = Sets.immutableEnumSet(REG_CLINICIAN);

    private final PhrConfig config;

    public ConversationWorkflowFlagService(PhrConfig config) {
        this.config = config;
    }

    public ConversationWorkflowFlags getWorkflowFlags(LoggedInEHRRequestContext loggedInEHRRequestContext) {
        boolean workFlowEnabled = WORKFLOW_ENABLED_USERS.contains(loggedInEHRRequestContext.getAccessingUser().getUserType());
        boolean assignEnabled = workFlowEnabled && config.isConversationAssignEnabled() && loggedInEHRRequestContext.hasTeam() && loggedInEHRRequestContext.isUserInOwnAccount();
        boolean archiveEnabled = workFlowEnabled && config.isConversationArchiveEnabled() && loggedInEHRRequestContext.hasTeam() && loggedInEHRRequestContext.isUserInOwnAccount();

        return ImmutableConversationWorkflowFlags.builder()
                .archivingEnabled(archiveEnabled)
                .assigningEnabled(assignEnabled)
                .build();
    }
}
