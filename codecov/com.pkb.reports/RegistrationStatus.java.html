package com.pkb.reports;

import com.google.common.collect.ImmutableMap;

import java.util.Arrays;
import java.util.Optional;
import java.util.function.Function;

public enum RegistrationStatus {
    REGISTERED, NOT_REGISTERED;

    private final static ImmutableMap<String, RegistrationStatus> LOOKUP_MAP = Arrays.stream(RegistrationStatus.values())
            .collect(ImmutableMap.toImmutableMap(rs -> rs.toString(), Function.identity()));

    public static Optional<RegistrationStatus> getRegistrationStatus(String input) {
        return Optional.ofNullable(LOOKUP_MAP.get(input.trim()));
    }
}
