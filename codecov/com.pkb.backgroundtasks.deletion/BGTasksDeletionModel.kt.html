package com.pkb.backgroundtasks.deletion

import com.pkb.backgroundtasks.PersonPublicId
import com.pkb.deletion.model.DatabaseTable
import com.pkb.deletion.model.PkColumn
import com.pkb.deletion.model.PkColumn.pk
import com.pkb.deletion.model.SecondaryIdColumn.secondaryId
import com.pkb.deletion.model.StandardInboundForeignKeyReference.owningFk
import java.util.UUID


/**
 * Model for use with the deletion service - defines the entities related to a person to allow for
 * automatic hard deletion.
 */
class BGTasksDeletionModel {

    companion object {

        private val ORG_ACCESS_TASK_COMPLETE: DatabaseTable<UUID> = DatabaseTable.builderFor(UUID::class.java).tableName("bgtasks.org_access_task_complete")
            .pkColumn(pk(UUID::class.java, listOf()))
            .build()

        private val ORG_ACCESS_TASK_PENDING: DatabaseTable<UUID> = DatabaseTable.builderFor(UUID::class.java).tableName("bgtasks.org_access_task_pending")
            .pkColumn(
                pk(
                    UUID::class.java, listOf(
                        owningFk(ORG_ACCESS_TASK_COMPLETE, "id")
                    )
                )
            )
            .build()

        private val ACCOUNT_WITHOUT_KEY: DatabaseTable<UUID> = DatabaseTable.builderFor(UUID::class.java).tableName("bgtasks.account_without_key")
            .pkColumn(pk(UUID::class.java, listOf())).build()

        private val ACCOUNT_TASK_COMPLETE: DatabaseTable<UUID> = DatabaseTable.builderFor(UUID::class.java).tableName("bgtasks.account_task_complete_v2")
            .pkColumn(pk(UUID::class.java, listOf())).build()

        private val ACCOUNT_TASK_PENDING: DatabaseTable<UUID> = DatabaseTable.builderFor(UUID::class.java).tableName("bgtasks.account_task_pending_v2")
            .pkColumn(pk(UUID::class.java, listOf())).build()


        private val PERSON: DatabaseTable<PersonPublicId> = DatabaseTable.builderFor(PersonPublicId::class.java).tableName("pkbperson")
            .pkColumn(
                PkColumn.pk(
                    PersonPublicId::class.java, listOf(
                        owningFk(ORG_ACCESS_TASK_PENDING, "person_public_id")
                    )
                )
            )
            .isVirtual(true)
            .build()

        val ACCOUNT: DatabaseTable<Long> = DatabaseTable.builderForLong()
            .tableName("core.account")
            .pkColumn(
                pk(
                    listOf(
                        owningFk(PERSON, "defaultaccountid")
                    )
                )
            )
            .secondaryIdColumns(
                listOf(
                    secondaryId(
                        "public_id", UUID::class.java, listOf(
                            owningFk(ACCOUNT_WITHOUT_KEY, "id"),
                            owningFk(ACCOUNT_TASK_COMPLETE, "account_public_id"),
                            owningFk(ACCOUNT_TASK_PENDING, "account_public_id")
                        )
                    )
                )
            )
            .isVirtual(true)
            .build()

    }
}
