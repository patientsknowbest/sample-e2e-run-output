package com.pkb.aggregator.persistence.repository

import org.springframework.context.annotation.Profile
import org.springframework.stereotype.Component

interface DatabaseStateClearer {
    fun invoke(action: () -> Unit)
}

/**
 * By default we do nothing to the database on state clear. In production this should never happen (since test support
 * code is disabled), and in e2e tests the database resets are externally handled
 */
@Component
@Profile("!integration-test")
class DefaultDatabaseStateClearer : DatabaseStateClearer {
    override fun invoke(action: () -> Unit) {
        // Do nothing
    }
}

/**
 * When running integration tests we want to clear the database between tests
 */
@Component
@Profile("integration-test")
class TestDatabaseStateClearer : DatabaseStateClearer {
    override fun invoke(action: () -> Unit) {
        action()
    }
}
