package com.pkb.backgroundtasks.wayfinder.recordsservice.provider

import ca.uhn.fhir.rest.client.api.IGenericClient
import org.hl7.fhir.r4.model.Appointment
import org.hl7.fhir.r4.model.IdType
import org.hl7.fhir.r4.model.Identifier
import org.hl7.fhir.r4.model.Organization
import org.hl7.fhir.r4.model.Reference
import org.springframework.stereotype.Component
import java.util.Optional

/**
 * For a given appointment, derive its ODS code by using the source-organization extension present in the root.
 * For patient-entered appointments, this extension does not exist.
 */
@Component
class OdsCodeProvider(private val fhirStoreClient: IGenericClient) {

    fun findOdsCodeByAppointmentResource(resourceId: IdType): String? {
        val appointment = fhirStoreClient.read().resource(Appointment::class.java).withId(resourceId).execute()
        val sourceOrganizationExtension =  Optional.ofNullable(
            appointment.meta.getExtensionByUrl("http://fhir.patientsknowbest.com/structuredefinition/source-organization"))

        return if (sourceOrganizationExtension.isEmpty) {
            //This looks a little clunky, but we need to differentiate between no source org extension (good, normal),
            // and no ODS code for an org linked by the extension (bad, config issue).
            null
        } else {
            val sourceOrgUuid = sourceOrganizationExtension.map { ex -> (ex.value as? Reference)?.reference }
            sourceOrgUuid.map { uuid -> fhirStoreClient.read().resource(Organization::class.java).withId(uuid).execute() }
                .map { organisation -> organisation.identifier }
                .map { identifierList ->
                    identifierList.find<Identifier?> { id -> id?.system.equals("https://fhir.nhs.uk/Id/ods-organization-code") }
                }
                .map { ods -> ods?.value }.orElseThrow({ RuntimeException("Failed to find ODS code value for resource ${sourceOrgUuid.get()}") })
        }
    }
}
