package com.pkb.backgroundtasks.wayfinder.recordsservice.provider

import ca.uhn.fhir.rest.client.api.IGenericClient
import ca.uhn.fhir.rest.server.exceptions.ResourceGoneException
import com.pkb.entities.enums.NationalIdType.NHS_NUMBER
import com.pkb.entities.enums.ValidNationalId
import org.hl7.fhir.r4.model.Appointment
import org.hl7.fhir.r4.model.IdType
import org.hl7.fhir.r4.model.Patient
import org.springframework.stereotype.Component
import java.util.Optional

@Component
class NhsNumberProvider(private val fhirStoreClient: IGenericClient) {

    fun findNhsNumberByAppointmentResource(resourceId: IdType): Optional<ValidNationalId> {
        try {
            val appointment = fhirStoreClient.read().resource(Appointment::class.java).withId(resourceId).execute()
            val patientId = appointment.participant
                    .filter { part -> part.actor.hasReferenceElement() && part.actor.referenceElement.resourceType.equals("Patient") }
                    .map { patient -> patient.actor.referenceElement }
                    .single()

            val patient = fhirStoreClient.read().resource(Patient::class.java).withId(patientId).execute()
            val nhsNumbers = patient.identifier
                    .filter { id -> id.system.equals(NHS_NUMBER.fhirIdentifierSystem) }
                    .map { nhs -> nhs.value }
                    .toList()

            if (nhsNumbers.size > 1) {
                throw IllegalStateException("$patientId has more than one NHS number")
            }
            return NHS_NUMBER.getValidNationalIdAndType(nhsNumbers.singleOrNull())
        } catch (e: Exception) {
            when (e) {
                is ResourceGoneException -> {
                    // the resource has been deleted, no further action is required
                    return Optional.empty()
                }
                else -> throw RuntimeException("Failed to find NHS number for $resourceId - ${e.message}")
            }
        }
    }

}
