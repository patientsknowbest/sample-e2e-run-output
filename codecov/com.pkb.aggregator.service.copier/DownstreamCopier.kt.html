package com.pkb.aggregator.service.copier

import com.pkb.aggregator.service.reference.ReferenceExtractor
import org.hl7.fhir.instance.model.api.IDomainResource

/**
 * Interface defining copy operations for a resource.
 *
 * There are some resources with business identifiers that we can reliably aggregate across multiple resources -
 * examples include patients, organizations etc. Other resources, however, we can't safely aggregate because there's no
 * really reliable way to say for certain that Appointment A from upstream 1 is exactly the same as Appointment B from
 * upstream 2. We could implement rules to match start time, participants etc but such 'soft-matching' is risky. So
 * instead we simply copy these resources to the downstream without aggregating them at all - if it becomes common for
 * a copied resource to show up independently in multiple upstreams we'll implement a way for users to mark them as the
 * same, but not for this initial release.
 */
interface DownstreamCopier<out T : IDomainResource> {

    /**
     * Makes a copy of the given resources, with the references updated. Returns the copies, mapped by their upstream id
     */
    fun copy(resources: List<IDomainResource>, references: ReferenceExtractor.IdentifierMap): Map<String, T>

    fun copy(resource: IDomainResource, references: ReferenceExtractor.IdentifierMap) =
        copy(listOf(resource), references).values.single()

    /**
     * Returns true if this copier can handle the given resource, otherwise false
     */
    fun accepts(resource: IDomainResource): Boolean
}
