package com.pkb.api.v2.model;

import com.fasterxml.jackson.annotation.JsonCreator;
import com.fasterxml.jackson.annotation.JsonValue;

import java.util.Arrays;

import static com.pkb.util.EnumMatch.match;

public enum ValueType {
    UNKNOWN(""), TEXT("text"), NUMERIC("numeric"), VARIES("varies"), POSNEG("posneg"), UNRECOGNISED("unrecognised"); //reserved for unrecognised input from API

    private final String value;

    ValueType(String value) {
        this.value = value;

    }

    @JsonValue
    public String getValue() {
        return value;

    }

    @JsonCreator
    public static ValueType constructFromValue(String value) {

        return match(ValueType.values(), value, ValueType.UNKNOWN, ValueType.UNRECOGNISED);

    }

    public static String allowedValues() {
        StringBuilder values = new StringBuilder();

        for (ValueType candidate : ValueType.values()) {
            if (candidate != ValueType.UNRECOGNISED) {
                values.append(candidate.getValue());
            }

            values.append(",");
        }
        values.deleteCharAt(values.length() - 1);
        return values.toString();
    }

    public static boolean isLegal(ValueType type) {
        return Arrays.asList("text", "numeric", "varies", "posneg").contains(type.getValue());
    }

}
