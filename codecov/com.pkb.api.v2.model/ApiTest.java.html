package com.pkb.api.v2.model;

import com.fasterxml.jackson.annotation.JsonIgnore;
import com.fasterxml.jackson.annotation.JsonRootName;
import com.pkb.test.entity.TestResultType;

import java.io.Serializable;

/**
 * This represents a LOINC test. It contains details about the test and a list of results for the test.
 */
@JsonRootName(value = "apiTestV2", namespace = "v2-beta")
public class ApiTest implements Serializable {

    private static final long serialVersionUID = -1L;

    private Long testId;
    private String name;
    private String apiRef;
    private String code;
    private String codeSystem;
    private String formName;
    private String formCode;
    private ValueType expectedValueType;
    private String unit;
    private String onlineRef;

    @JsonIgnore
    public Long getTestId() {
        return testId;
    }

    public void setTestId(Long testId) {
        this.testId = testId;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getApiRef() {
        return apiRef;
    }

    public void setApiRef(String apiRef) {
        this.apiRef = apiRef;
    }

    public String getCode() {
        return code;
    }

    public void setCode(String code) {
        this.code = code;
    }

    public String getCodeSystem() {
        return codeSystem;
    }

    public void setCodeSystem(String codeSystem) {
        this.codeSystem = codeSystem;
    }

    public String getFormName() {
        return formName;
    }

    public void setFormName(String formName) {
        this.formName = formName;
    }

    public String getFormCode() {
        return formCode;
    }

    public void setFormCode(String formCode) {
        this.formCode = formCode;
    }

    public String getUnit() {
        return unit;
    }

    public void setUnit(String unit) {
        this.unit = unit;
    }

    public String getOnlineRef() {
        return onlineRef;
    }

    public void setOnlineRef(String onlineRef) {
        this.onlineRef = onlineRef;
    }

    public ValueType getExpectedValueType() {
        return expectedValueType;
    }

    public void setExpectedValueType(TestResultType.ValueType expectedValueType) {
        if (expectedValueType != null) {
            this.expectedValueType = ValueType.valueOf(expectedValueType.name());
        } else {
            this.expectedValueType = null;
        }
    }
}
