package com.pkb.api.v2.model;

import com.fasterxml.jackson.annotation.JsonCreator;
import com.fasterxml.jackson.annotation.JsonValue;

import static com.pkb.util.EnumMatch.match;

public enum ApiEncounterStatus {

    PLANNED("planned"), ARRIVED("arrived"), IN_PROGRESS("inprogress"), ON_LEAVE("onleave"), FINISHED("finished"), CANCELLED("cancelled"),
    BLANK(""),
    //reserved for unrecognised input from API:
    UNRECOGNISED("unrecognised");

    private final String value;

    ApiEncounterStatus(String value) {
        this.value = value;

    }

    @JsonValue
    public String getValue() {
        return value;

    }

    @JsonCreator
    public static ApiEncounterStatus constructFromValue(String value) {

        return match(ApiEncounterStatus.values(), value, ApiEncounterStatus.BLANK, ApiEncounterStatus.UNRECOGNISED);

    }

    public static String allowedValues() {
        StringBuilder values = new StringBuilder();
        for (ApiEncounterStatus candidate : ApiEncounterStatus.values()) {
            if (candidate != ApiEncounterStatus.UNRECOGNISED) {
                values.append(candidate.getValue());
            }

            values.append(",");
        }
        values.deleteCharAt(values.length() - 1);
        return values.toString();
    }
}
