package com.pkb.api.v2.model;

import com.fasterxml.jackson.annotation.JsonCreator;
import com.fasterxml.jackson.annotation.JsonValue;

import static com.pkb.util.EnumMatch.match;

public enum ApiEncounterClass {
    IN_PATIENT("inpatient"), OUT_PATIENT("outpatient"), AMBULATORY("ambulatory"), EMERGENCY("emergency"), HOME("home"), FIELD("field"),
    DAYTIME("daytime"), VIRTUAL("virtual"), OTHER("other"),

    BLANK(""),
    //reserved for unrecognised input from API:
    UNRECOGNISED("unrecognised");

    private final String value;

    ApiEncounterClass(String value) {
        this.value = value;

    }

    @JsonValue
    public String getValue() {
        return value;

    }

    @JsonCreator
    public static ApiEncounterClass constructFromValue(String value) {

        return match(ApiEncounterClass.values(), value, ApiEncounterClass.BLANK, ApiEncounterClass.UNRECOGNISED);

    }

    public static String allowedValues() {
        StringBuilder values = new StringBuilder();
        for (ApiEncounterClass candidate : ApiEncounterClass.values()) {
            if (candidate != ApiEncounterClass.UNRECOGNISED) {
                values.append(candidate.getValue());
            }

            values.append(",");
        }
        values.deleteCharAt(values.length() - 1);
        return values.toString();
    }

}
