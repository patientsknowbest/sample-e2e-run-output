package com.pkb.api.v2.model;

import io.swagger.annotations.ApiModel;

import javax.xml.bind.annotation.XmlRootElement;
import javax.xml.bind.annotation.XmlType;
import java.io.Serializable;
import java.util.Date;
import java.util.UUID;

@XmlRootElement(name = "V2SymptomReport", namespace = "v2-beta")
@XmlType(name = "V2SymptomReport", namespace = "v2-beta")

    private UUID id;

    private String apiRef;

    private Date reportDate;

    private String reportDateISO;

    private Integer severity0to3;// can be null

     * (readonly) the timestamp for when the diagnosis was persisted on the server
     */
    private Date persistedDate;

    // debugging util

        return "[SymptomReport: id " + id +

                ", " + severity0to3 + "]";
    }

    public String getReportDateISO() {
        return reportDateISO;

    public void setReportDateISO(String reportDateISO) {
        this.reportDateISO = reportDateISO;

    // getters/setters

    public UUID getId() {
        return id;
    }

    public void setId(UUID id) {
        this.id = id;
    }

    public Date getReportDate() {
        return reportDate;
    }

    public void setReportDate(Date reportDate) {
        this.reportDate = reportDate;
    }

    public String getApiRef() {
        return apiRef;
    }

    public void setApiRef(String symptomApiRef) {

    public Integer getSeverity0to3() {
        return severity0to3;
    }

    public void setSeverity0to3(Integer severity0to3) {
        this.severity0to3 = severity0to3;
    }

    public String getComments() {
        return comments;
    }

    public void setComments(String comments) {
        this.comments = comments;
    }

    public Date getEnteredDate() {
        return enteredDate;
    }

    public void setEnteredDate(Date enteredDate) {
        this.enteredDate = enteredDate;
    }

    public Date getPersistedDate() {
        return persistedDate;
    }

        this.persistedDate = persistedDate;

}
