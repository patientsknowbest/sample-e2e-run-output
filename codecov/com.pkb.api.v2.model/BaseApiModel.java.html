package com.pkb.api.v2.model;

import org.jetbrains.annotations.Nullable;

import java.util.Date;
import java.util.UUID;

public class BaseApiModel {

    protected static @Nullable Date copy(Date original) {
        if (original == null) {
            return null;
        }
        return new Date(original.getTime());
    }

    protected final Date enteredDate;

    protected final Date persistedDate;

    protected final UUID id;

    public BaseApiModel(BaseApiModelBuilder builder) {
        this(builder.id, builder.enteredDate, builder.persistedDate);
    }

    public BaseApiModel(UUID id, Date enteredDate, Date persistedDate) {
        this.id = id;
        this.enteredDate = copy(enteredDate);
        this.persistedDate = copy(persistedDate);
    }

    public Date getEnteredDate() {
        return copy(enteredDate);
    }

    public Date getPersistedDate() {
        return copy(persistedDate);
    }

    public UUID getId() {
        return id;
    }

    public boolean hasId() {
        return id != null;
    }

}
