package com.pkb.api.v2.model;

import com.fasterxml.jackson.annotation.JsonCreator;
import com.fasterxml.jackson.annotation.JsonValue;
import com.pkb.api.util.converter.ConversionException;
import com.pkb.app.dto.ParticipantDTO;
import com.pkb.app.dto.ParticipantDTO.ParticipantRole;
import com.pkb.user.entity.PKBPerson;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.Nullable;

import static com.pkb.util.EnumMatch.match;

public class ApiParticipant {

    private String pkbParticipantId;

    private String nonPkbParticipant;

    private ApiParticipantRole participantRole;

    private String firstName = "";

    private String lastName = "";

    private String title = "";

    private String teamName = "";

    private String jobTitle = "";

    public enum ApiParticipantRole {
        BLANK("", null), ADMITTER("admitter", ParticipantRole.ADMITTER), DISCHARGER("discharger", ParticipantRole.DISCHARGER),
        CONSULTANT("consultant", ParticipantRole.CONSULTANT), ATTENDER("attender", ParticipantRole.ATTENDER),
        REFERRER("referrer", ParticipantRole.REFERRER), CARER("carer", ParticipantRole.CARER), PATIENT("patient", ParticipantRole.PATIENT),
        //reserved for unrecognised input from API:
        UNRECOGNISED("unrecognised", null);

        private final String value;

        private final ParticipantRole participantRole;

        ApiParticipantRole(String value, ParticipantRole participantRole) {
            this.value = value;
            this.participantRole = participantRole;
        }

        @JsonValue
        public String getValue() {
            return value;

        }

        @JsonCreator
        public static ApiParticipantRole constructFromValue(String value) {

            return match(ApiParticipantRole.values(), value, ApiParticipantRole.BLANK, ApiParticipantRole.UNRECOGNISED);
        }

        public static String allowedValues() {
            StringBuilder values = new StringBuilder();
            for (ApiParticipantRole candidate : ApiParticipantRole.values()) {
                if (candidate != ApiParticipantRole.UNRECOGNISED) {
                    values.append(candidate.getValue());
                }

                values.append(",");
            }
            values.deleteCharAt(values.length() - 1);
            return values.toString();
        }

        public ParticipantRole getParticipantRole() {
            return participantRole;
        }

    }

    public String getPkbParticipantId() {
        return pkbParticipantId;
    }

    public void setPkbParticipantId(String pkbParticipantId) {
        this.pkbParticipantId = pkbParticipantId;
    }

    public String getNonPkbParticipant() {
        return nonPkbParticipant;
    }

    public void setNonPkbParticipant(String nonPkbParticipant) {
        this.nonPkbParticipant = nonPkbParticipant;
    }

    public ApiParticipantRole getParticipantRole() {
        return participantRole;
    }

    public void setParticipantRole(ApiParticipantRole participantRole) {
        this.participantRole = participantRole;
    }

    public String getFirstName() {
        return firstName;
    }

    public void setFirstName(String firstName) {
        this.firstName = firstName;
    }

    public String getLastName() {
        return lastName;
    }

    public void setLastName(String lastName) {
        this.lastName = lastName;
    }

    public String getTitle() {
        return title;
    }

    public void setTitle(String title) {
        this.title = title;
    }

    public String getTeamName() {
        return teamName;
    }

    public void setTeamName(String teamName) {
        this.teamName = teamName;
    }

    public String getJobTitle() {
        return jobTitle;
    }

    public void setJobTitle(String jobTitle) {
        this.jobTitle = jobTitle;
    }

    public static @Nullable ApiParticipant convert(ParticipantDTO dto) throws ConversionException {
        if (dto == null) {
            return null;
        }
        ApiParticipant api = new ApiParticipant();
        if (!StringUtils.isEmpty(dto.getPersonId())) {
            api.setPkbParticipantId(dto.getPersonId());
            if (dto.getPkbPerson() != null) {
                PKBPerson person = dto.getPkbPerson();
                api.setFirstName(person.getFirstName());
                api.setLastName(person.getLastName());
                api.setTitle(person.getTitle());
            }
        }
        api.setNonPkbParticipant(dto.getNonpkbParticipant());
        if (dto.getRole() != null) {
            for (ApiParticipantRole candidate : ApiParticipantRole.values()) {
                if (candidate.toString().equals(dto.getRole().toString())) {
                    api.setParticipantRole(candidate);
                }
            }
            if (api.getParticipantRole() == null) {
                throw new ConversionException("Couldn't map participant role " + dto.getRole().toString());
            }
        }

        return api;
    }

    // TODO : move this to converter interface for consistency later
    public static @Nullable ParticipantDTO convert(ApiParticipant api) throws ConversionException {
        if (api == null) {
            return null;
        }
        ParticipantDTO dto = new ParticipantDTO();
        if (api.getPkbParticipantId() != null) {
            dto.setPersonId(api.getPkbParticipantId());
        }

        if (!StringUtils.isEmpty(api.getNonPkbParticipant())) {
            dto.setNonpkbParticipant(api.getNonPkbParticipant());
        }

        if (api.getParticipantRole() != null) {
            for (ApiParticipantRole candidate : ApiParticipantRole.values()) {
                if (candidate.toString().equals(api.getParticipantRole().toString())) {
                    dto.setRole(candidate.getParticipantRole());
                }
            }
            if (api.getParticipantRole() == null) {
                throw new ConversionException("Couldn't map participant role " + dto.getRole().toString());
            }
        }

        return dto;
    }

}
