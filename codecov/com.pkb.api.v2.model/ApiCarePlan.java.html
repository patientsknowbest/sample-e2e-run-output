package com.pkb.api.v2.model;

import com.fasterxml.jackson.annotation.JsonCreator;
import com.fasterxml.jackson.annotation.JsonProperty;
import com.pkb.phplan.entity.PHPlan.ApprovalStatus;
import com.pkb.phplan.entity.PHPlan.EditableStatus;
import com.pkb.phplan.entity.PHPlan.PlanStatus;

import java.io.Serializable;
import java.util.Date;
import java.util.Objects;
import java.util.UUID;

/**
 * https://www.hl7.org/fhir/careplan.html
 *
 */
public class ApiCarePlan implements Serializable {

    private static final long serialVersionUID = -1L;

    private UUID id;

    private String subject; //FHIR attribute; patientid

    private Long version;

    private String name;

    private Date modified;//FHIR attribute ; When last updated

    private String lastEditedOnISO;

    private Date creationDate;

    private Date enteredDate;

    private Date persistedDate;

    private String description; //FHIR attribute

    private String note; //FHIR attribute; Comments about the plan

    private ApiParticipant creator;

    private ApiParticipant lastEditor;

    private PlanStatus status; // FHIR attribute

    private ApprovalStatus approvalStatus;

    private Long approverId;

    private Date reviewDate;

    private EditableStatus editableStatus;

    public ApiCarePlan() {
    }

    @JsonCreator
    public ApiCarePlan(@JsonProperty("subject") String subject, @JsonProperty("status") PlanStatus status) {
        this.subject = Objects.requireNonNull(subject, "Subject (PatientId) should be non-null");
        this.status = Objects.requireNonNull(status, "Status should be non-null");
    }

    public String getLastEditedOnISO() {
        return lastEditedOnISO;
    }

    public void setLastEditedOnISO(String lastEditedOnISO) {
        this.lastEditedOnISO = lastEditedOnISO;
    }

    // getters/setters

    public UUID getId() {
        return id;
    }

    public void setId(UUID id) {
        this.id = id;
    }

    public Long getVersion() {
        return version;
    }

    public void setVersion(Long version) {
        this.version = version;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public Date getModified() {
        return modified;
    }

    public void setModified(Date modified) {
        this.modified = modified;
    }

    public String getSubject() {
        return subject;
    }

    public Date getCreationDate() {
        return creationDate;
    }

    public void setCreationDate(Date creationDate) {
        this.creationDate = creationDate;
    }

    public String getDescription() {
        return description;
    }

    public void setDescription(String description) {
        this.description = description;
    }

    public Date getEnteredDate() {
        return enteredDate;
    }

    public void setEnteredDate(Date enteredDate) {
        this.enteredDate = enteredDate;
    }

    public Date getPersistedDate() {
        return persistedDate;
    }

    public void setPersistedDate(Date persistedDate) {
        this.persistedDate = persistedDate;
    }

    public String getNote() {
        return note;
    }

    public void setNote(String note) {
        this.note = note;
    }

    public PlanStatus getStatus() {
        return status;
    }

    public ApprovalStatus getApprovalStatus() {
        return approvalStatus;
    }

    public void setApprovalStatus(ApprovalStatus approvalStatus) {
        this.approvalStatus = approvalStatus;
    }

    public Long getApproverId() {
        return approverId;
    }

    public void setApproverId(Long approverId) {
        this.approverId = approverId;
    }

    public Date getReviewDate() {
        return reviewDate;
    }

    public void setReviewDate(Date reviewDate) {
        this.reviewDate = reviewDate;
    }

    public EditableStatus getEditableStatus() {
        return editableStatus;
    }

    public void setEditableStatus(EditableStatus editableStatus) {
        this.editableStatus = editableStatus;
    }

    public ApiParticipant getCreator() {
        return creator;
    }

    public void setCreator(ApiParticipant creator) {
        this.creator = creator;
    }

    public ApiParticipant getLastEditor() {
        return lastEditor;
    }

    public void setLastEditor(ApiParticipant lastEditor) {
        this.lastEditor = lastEditor;
    }

    public boolean hasId() {
        return id != null;
    }

}
