package com.pkb.api.v2.model;

import java.util.Date;
import java.util.List;
import java.util.UUID;

public class ApiAllergy {

    private UUID id;

    private String substanceCode;
    private String allergen;
    private Date onsetOfSymptoms;
    private Date enteredDate;
    private List<String> reactions;
    private String severity;
    private Date persistedDate;

    /**
     * Indicates whether this is an allergy the patient really has or not.
     *
     * note this is null to support API clients that don't know about it. If it is set as null, it will default to true
     */
    private Boolean patientIsAllergic = true;

    public Boolean getPatientIsAllergic() {
        return patientIsAllergic;
    }

    public void setPatientIsAllergic(Boolean patientIsAllergic) {
        this.patientIsAllergic = patientIsAllergic;
    }

    public UUID getId() {
        return id;
    }

    public void setId(UUID id) {
        this.id = id;
    }

    public String getSubstanceCode() {
        return substanceCode;
    }

    public void setSubstanceCode(String substanceCode) {
        this.substanceCode = substanceCode;
    }

    public String getAllergen() {
        return allergen;
    }

    public void setAllergen(String allergen) {
        this.allergen = allergen;
    }

    public Date getOnsetOfSymptoms() {
        return onsetOfSymptoms;
    }

    public void setOnsetOfSymptoms(Date onsetOfSymptoms) {
        this.onsetOfSymptoms = onsetOfSymptoms;
    }

    public Date getEnteredDate() {
        return enteredDate;
    }

    public void setEnteredDate(Date enteredDate) {
        this.enteredDate = enteredDate;
    }

    public List<String> getReactions() {
        return reactions;
    }

    public void setReactions(List<String> reactions) {
        this.reactions = reactions;
    }

    public String getSeverity() {
        return severity;
    }

    public void setSeverity(String severity) {
        this.severity = severity;
    }

    public Date getPersistedDate() {
        return persistedDate;
    }

    public void setPersistedDate(Date persistedDate) {
        this.persistedDate = persistedDate;
    }

}
