package com.pkb.integration.ods;

import java.util.Optional;

import javax.annotation.ParametersAreNonnullByDefault;

import org.hl7.fhir.dstu3.model.Organization;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import com.google.common.cache.LoadingCache;
import com.google.common.util.concurrent.UncheckedExecutionException;
import com.pkb.common.ClearableInternalState;

@ParametersAreNonnullByDefault
public class OdsService implements ClearableInternalState {

    private final LoadingCache<String, Optional<Organization>> orgCache;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    OdsService(LoadingCache<String, Optional<Organization>> orgCache) {
        this.orgCache = orgCache;
    }

    @NotNull
    public Optional<String> getPracticeName(String odsCode) {
        try {
            return orgCache.getUnchecked(odsCode).map(Organization::getName);
        } catch (UncheckedExecutionException e) {
            LOGGER.error("Error getting name for practice from ODS API for code {}", odsCode, e);
            return Optional.empty();
        }
    }

    @Override
    public void clearState() {
        orgCache.invalidateAll();
    }
}
