package com.pkb.deletion.routing;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.pkb.common.config.BaseConfig;
import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.deletion.model.DeleteOperationRequest;
import com.pkb.deletion.model.DeleteOperationResponse;
import com.pkb.deletion.model.DeletionServiceProperties;
import org.apache.camel.component.jackson.JacksonDataFormat;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.ObjectProvider;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Profile;
import org.springframework.stereotype.Component;

import static org.apache.commons.lang3.StringUtils.isBlank;

@Component
@Profile("!integration-test")
public class DeletionServiceRouting extends AbstractNamespaceAwareRouteBuilder {

    public static final String OPERATION_REQUEST = "direct:deleteOperationRequest";
    public static final String OPERATION_RESPONSE = "direct:deleteOperationResponse";

    private final PubSubNamespaceService pubSubNamespaceService;
    private final String serviceAccountKey;
    private final ObjectMapper objectMapper;
    private final DeletionServiceProperties properties;
    private final String pubSubProjectName;


    public DeletionServiceRouting(PubSubNamespaceService pubSubNamespaceService,
                                  @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey,
                                  ObjectMapper objectMapper,
                                  DeletionServiceProperties properties,
                                  ObjectProvider<BaseConfig> baseConfigProvider) {
        this.pubSubNamespaceService = pubSubNamespaceService;
        this.serviceAccountKey = serviceAccountKey;
        this.objectMapper = objectMapper;
        this.properties = properties;
        this.pubSubProjectName = baseConfigProvider.stream().findFirst().orElseThrow().getPubSubProjectName();
    }

    @Override
    public @NotNull PubSubNamespaceService camelNamespaceService() {
        return pubSubNamespaceService;
    }

    @Override
    public boolean throwExceptionForNamespaceMismatch() {
        return true;
    }

    @Override
    public void configure() throws Exception {
        from(DeletionServiceRouting.OPERATION_RESPONSE)
                .marshal(new JacksonDataFormat(objectMapper, DeleteOperationResponse.class)) // Sigh, have to do this manually because no Spring Boot autoconfig
                .convertBodyTo(byte[].class)
                .to(uri(pubSubProjectName, properties.getDeleteResponseTopicName()));


        from(uri(pubSubProjectName, properties.getDeleteRequestSubscriptionName()))
                .unmarshal(new JacksonDataFormat(objectMapper, DeleteOperationRequest.class))
                .to(DeletionServiceRouting.OPERATION_REQUEST);

    }

    private String uri(String pubSubProjectName, String topicName) {
        String serviceAccount = isBlank(serviceAccountKey) ? "" : String.format("?serviceAccountKey=file://%s", serviceAccountKey);
        return String.format("google-pubsub:%s:%s%s", pubSubProjectName, topicName, serviceAccount);
    }
}
