package com.pkb.service.dataupload.processor.parser.hl7.encounter;

import org.springframework.stereotype.Component;

import com.pkb.encounter.entity.Encounter;
import com.pkb.encounter.entity.EncounterEvent;
import com.pkb.exception.ApiCallMalformedException;
import com.pkb.service.dataupload.hl7.HL7ConnContext;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.field.HL7Code;
import com.pkb.service.dataupload.hl7.value.HL7Date;
import com.pkb.service.dataupload.processor.domain.UploadedEncounter;
import com.pkb.service.dataupload.processor.domain.UploadedEncounterRequestType;
import com.pkb.service.encounter.EncounterMessageHelper;
import com.pkb.user.entity.PKBPerson;

@Component
public class EncounterPatientScheduledAdmissionParser {

    /**
     * ADT^A05
     */
    public UploadedEncounter doProcess(HL7XmlDoc hl7, PKBPerson patient, Encounter encounter, EncounterEvent encounterEvent, HL7ConnContext hl7ConnContext,
            UploadedEncounterRequestType requestType) {
        HL7Code customAppointmentTypeCode = hl7.getCustomAppointmentTypeCode();

        // What is the scheduled date? In order of preference, pull from PV2.8, EVN.3, PV1.44, MSH-7 then error if not found in any
        HL7Date messageHL7Date = null;
        if ((hl7.getPV2() != null) && hl7.getPV2().getExpectedAdmitDateTime().isNotBlank()) {
            messageHL7Date = hl7.getPV2().getExpectedAdmitDateTime(); // PV2.8.1
        } else if ((hl7.getEVN() != null) && hl7.getEVN().getEventPlannedDate().isNotBlank()) {
            messageHL7Date = hl7.getEVN().getEventPlannedDate(); // EVN.3.1
        } else if (hl7.getPV1().isPresent() && hl7.getPV1().get().getAdmitDateTime().isNotBlank()) {
            messageHL7Date = hl7.getPV1().get().getAdmitDateTime(); // PV1.44.1
        } else if ((hl7.getMSH() != null) && hl7.getMSH().getDateTimeOfMessage().isNotBlank()) {
            messageHL7Date = hl7.getMSH().getDateTimeOfMessage(); // MSH-7
        } else {
            throw new ApiCallMalformedException("No patient admit scheduled date provided.");
        }

        encounterEvent.setMessageTimestamp(messageHL7Date.getInstant());
        encounterEvent.setSubject(EncounterMessageHelper.makeEncounterMessageSubject(encounterEvent.getEncounterClass()));
        encounterEvent.setContent(
                EncounterMessageHelper.makeEncounterMessageContentForScheduledAdmit(encounterEvent, patient, messageHL7Date.getUtcOffset()));
        UploadedEncounter result = new UploadedEncounter(encounterEvent, encounter, requestType);
        result.setCustomAppointmentTypeCode(customAppointmentTypeCode);
        result.setHL7ConnContext(hl7ConnContext);
        return result;
    }
}
