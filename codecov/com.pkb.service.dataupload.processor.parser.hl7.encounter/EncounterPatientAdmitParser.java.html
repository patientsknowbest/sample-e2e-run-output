package com.pkb.service.dataupload.processor.parser.hl7.encounter;

import org.springframework.stereotype.Component;

import com.pkb.encounter.entity.Encounter;
import com.pkb.encounter.entity.EncounterEvent;
import com.pkb.exception.ApiCallMalformedException;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.segment.HL7PatientVisit1;
import com.pkb.service.dataupload.hl7.value.HL7Date;
import com.pkb.service.dataupload.processor.domain.UploadedEncounter;
import com.pkb.service.dataupload.processor.domain.UploadedEncounterRequestType;
import com.pkb.service.encounter.EncounterMessageHelper;
import com.pkb.user.entity.PKBPerson;

@Component
public class EncounterPatientAdmitParser {

    /**
     * ADT^A01
     */
    public UploadedEncounter doProcess(HL7XmlDoc hl7, PKBPerson patient, Encounter encounter, EncounterEvent encounterEvent) {

        // Populate message with A01 specific fields
        HL7Date messageHL7Date = null;

        HL7PatientVisit1 pv1 = hl7.getPV1()
                .orElseThrow(() -> new ApiCallMalformedException("Cannot handle ADT^A01 message: PV1 segment missing"));
        if (pv1.getAdmitDateTime().isNotBlank()) {
            messageHL7Date = pv1.getAdmitDateTime();
        } else if (hl7.getMSH().getDateTimeOfMessage().isNotBlank()) { // fall back to pulling admit date from MSh-7
            messageHL7Date = hl7.getMSH().getDateTimeOfMessage();
        } else {
            throw new ApiCallMalformedException("No patient admit date provided.");
        }

        encounterEvent.setMessageTimestamp(messageHL7Date.getInstant());
        encounterEvent.setSubject(EncounterMessageHelper.makeEncounterMessageSubject(encounterEvent.getEncounterClass()));
        encounterEvent.setContent(EncounterMessageHelper.makeEncounterMessageContentA01(encounterEvent, patient, messageHL7Date.getUtcOffset()));

        return new UploadedEncounter(encounterEvent, encounter, UploadedEncounterRequestType.ADMIT);
    }
}
