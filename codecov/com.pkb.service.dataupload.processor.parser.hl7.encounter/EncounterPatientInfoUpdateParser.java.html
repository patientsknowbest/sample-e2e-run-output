package com.pkb.service.dataupload.processor.parser.hl7.encounter;

import java.util.ArrayList;
import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import com.pkb.encounter.entity.Encounter;
import com.pkb.encounter.entity.EncounterEvent;
import com.pkb.encounter.entity.Message;
import com.pkb.exception.ApiCallMalformedException;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.segment.HL7PatientVisit1;
import com.pkb.service.dataupload.hl7.value.HL7Date;
import com.pkb.service.dataupload.processor.domain.UploadedEncounter;
import com.pkb.service.dataupload.processor.domain.UploadedEncounterRequestType;
import com.pkb.service.dataupload.processor.support.EncounterMessageFilter;
import com.pkb.service.encounter.EncounterMessageHelper;
import com.pkb.user.entity.PKBPerson;

import io.prometheus.client.Counter;

@Component
public class EncounterPatientInfoUpdateParser {
    private static final Counter messageIgnores = Counter.build()
            .name("pkb_phr_hl7_message_ignores_count_encounter_sync")
            .help("Number of ignored messages")
            .labelNames("type")
            .register();
    private final EncounterMessageFilter messageFilter;

    @Autowired
    public EncounterPatientInfoUpdateParser(EncounterMessageFilter messageFilter) {
        this.messageFilter = messageFilter;
    }

    /**
     * ADT^A08
     * Allows:
     * <ul>
     * <li>Admit Date Update (A01 PV1-44)</li>
     * <li>Discharge Date update (A03 PV1-45)</li>
     * </ul>
     */
    public UploadedEncounter doProcess(HL7XmlDoc hl7, PKBPerson patient, Encounter encounter, EncounterEvent encounterEvent) {
        List<Message> messages = new ArrayList<>();
        if (encounter != null) {
            // This Encounter exists in PKB's database, PV1-19.1 external visit ID is known
            // Process information updates

            // pre-find an existing A01 to check if this is being updated
            List<Message> a01s = filterMessages(encounter.getMessages(), Message.MessageType.PATIENT_ADMIT);
            // pre-find an existing A03 to check if this is being updated
            List<Message> a03s = filterMessages(encounter.getMessages(), Message.MessageType.PATIENT_DISCHARGE);
            // Check what information we should update:

            HL7PatientVisit1 pv1 = hl7.getPV1()
                    .orElseThrow(() -> new ApiCallMalformedException("Cannot handle ADT^A08 message: PV1 segment missing"));

            // PV1.44 - Admit Date
            if (pv1.getAdmitDateTime().isProvided() && !a01s.isEmpty()) {
                if (a01s.size() > 1) {
                    throw new ApiCallMalformedException("Duplicate A01 admit records are present for this encounter.");
                } else {
                    Message admit = a01s.get(0);
                    if (!pv1.getAdmitDateTime().getInstant().equals(admit.getMessageTimestamp())) {
                        // XXX Create a new versioned instance of the admit message data point with the new value and delete the old
                        HL7Date admitHL7Date = pv1.getAdmitDateTime();
                        admit.setMessageTimestamp(admitHL7Date.getInstant());
                        admit.setSubject(EncounterMessageHelper.makeEncounterMessageSubject(admit.getEncounterClass()));
                        admit.setContent(
                                EncounterMessageHelper.makeEncounterMessageContentA01(admit, patient, admitHL7Date.getUtcOffset()));
                        //                        admit.setId(null); // XXX cause a new record to be created? Versioning?
                        messages.add(admit);
                    }
                }
            }

            // PV1.45 - Discharge Date
            if (pv1.getDischargeDateTime().isProvided() && !a03s.isEmpty()) {
                if (a03s.size() > 1) {
                    throw new ApiCallMalformedException("Duplicate A03 dischargerecords are present for this encounter.");
                } else {
                    Message discharge = a03s.get(0);
                    if (!pv1.getDischargeDateTime().getInstant().equals(discharge.getMessageTimestamp())) {
                        HL7Date dischargeHL7Date = pv1.getDischargeDateTime();
                        discharge.setMessageTimestamp(dischargeHL7Date.getInstant());
                        discharge.setSubject(EncounterMessageHelper.makeEncounterMessageSubject(discharge.getEncounterClass()));
                        discharge.setContent(
                                EncounterMessageHelper.makeEncounterMessageContentA03(discharge, patient, dischargeHL7Date.getUtcOffset()));
                        //                        discharge.setId(null); // XXX cause a new record to be created? Versioning?
                        messages.add(discharge);
                    }
                }
            }

            // XXX Include other information that can be updated via A08 event payload?

            HL7Date messageHL7Date = null;
            if ((hl7.getHL7EventType() != null) && hl7.getHL7EventType().getEventOccurredDate().isNotBlank()) {
                messageHL7Date = hl7.getHL7EventType().getEventOccurredDate(); // EVN-6
            } else if ((hl7.getHL7MessageHeader() != null) && hl7.getHL7MessageHeader().getDateTimeOfMessage().isNotBlank()) {
                messageHL7Date = hl7.getHL7MessageHeader().getDateTimeOfMessage(); // MSH-7
            } else {
                // do not continue if we cannot establish a timestamp for this message
                throw new RuntimeException("No Event occurred date (EVN-6 or MSH-7) provided in A08 update.");
            }

            encounterEvent.setMessageTimestamp(messageHL7Date.getInstant());
            encounterEvent.setSubject(EncounterMessageHelper.makeEncounterMessageSubject(encounterEvent.getEncounterClass()));
            encounterEvent.setContent(EncounterMessageHelper.makeEncounterMessageContentA08(encounterEvent, patient, messageHL7Date.getUtcOffset()));

        }else{
            messageIgnores.labels(hl7.getMessageType().getCode()).inc();
        }
        return new UploadedEncounter(encounterEvent, encounter, UploadedEncounterRequestType.INFORMATION_UPDATE, messages);
    }

    private List<Message> filterMessages(List<Message> messages, Message.MessageType messageType) {
        return messageFilter.filterMessages(messages, messageType);
    }
}
