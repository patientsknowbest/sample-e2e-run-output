package com.pkb.backgroundtasks

import com.pkb.backgroundtasks.util.RequestIdFilter.Companion.BACKGROUND_TASKS_REQUEST_HEADER
import io.micrometer.core.instrument.binder.okhttp3.OkHttpMetricsEventListener
import okhttp3.OkHttpClient
import okhttp3.RequestBody
import retrofit2.Call
import retrofit2.Retrofit
import retrofit2.create
import retrofit2.http.Body
import retrofit2.http.Header
import retrofit2.http.Headers
import retrofit2.http.POST

/**
 * The goal of this class is to provide a separated client with different timeouts.
 */
interface PhrSyncUploadedDataClient {
    companion object {
        private const val APPLICATION_HEADER = "Accept: application/octet-stream"
        fun phrSyncUploadedDataClient(baseUrl: String, client: OkHttpClient): PhrSyncUploadedDataClient =
                Retrofit.Builder()
                        .baseUrl(baseUrl)
                        .client(client)
                        .build()
                        .create()
    }

    @Headers(APPLICATION_HEADER, "${OkHttpMetricsEventListener.URI_PATTERN}: /synchUploadedData.action")
    @POST("/synchUploadedData.action")
    fun syncUploadedData(
            @Header(BACKGROUND_TASKS_REQUEST_HEADER) requestID: String,
            @Body request: RequestBody
    ): Call<Void>
}
