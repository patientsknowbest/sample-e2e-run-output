package com.pkb.backgroundtasks

import com.fasterxml.jackson.databind.ObjectMapper
import com.pkb.backgroundtasks.util.RequestIdFilter.Companion.BACKGROUND_TASKS_REQUEST_HEADER
import com.pkb.entities.enums.MenuDataType
import io.micrometer.core.instrument.binder.okhttp3.OkHttpMetricsEventListener
import okhttp3.OkHttpClient
import retrofit2.Call
import retrofit2.Retrofit
import retrofit2.converter.jackson.JacksonConverterFactory
import retrofit2.create
import retrofit2.http.Body
import retrofit2.http.GET
import retrofit2.http.Header
import retrofit2.http.Headers
import retrofit2.http.POST

interface PhrClient {

    companion object {
        private const val APPLICATION_JSON_HEADER = "Accept: application/json"
        fun phrClient(baseUrl: String, client: OkHttpClient, objectMapper: ObjectMapper): PhrClient =
            Retrofit.Builder()
                .baseUrl(baseUrl)
                .client(client)
                .addConverterFactory(JacksonConverterFactory.create(objectMapper))
                .build()
                .create()
    }

    @Headers(
        APPLICATION_JSON_HEADER,
        "${OkHttpMetricsEventListener.URI_PATTERN}: /activate/outOfAreaDeferredActivationNotification.action"
    )
    @POST("/activate/outOfAreaDeferredActivationNotification.action")
    fun notifyOutOfAreaPatientAccountActivation(
        @Header(BACKGROUND_TASKS_REQUEST_HEADER) requestID: String,
        @Body request: OutOfAreaActivationNotificationDTO
    ): Call<Void>

    @Headers(
        APPLICATION_JSON_HEADER,
        "${OkHttpMetricsEventListener.URI_PATTERN}: /activate/invitationActivationNotification.action"
    )
    @POST("/activate/invitationActivationNotification.action")
    fun notifyInvitationActivation(
        @Header(BACKGROUND_TASKS_REQUEST_HEADER) requestID: String,
        @Body request: InvitationActivationNotificationDTO
    ): Call<Void>


    @Headers(
        APPLICATION_JSON_HEADER,
        "${OkHttpMetricsEventListener.URI_PATTERN}: /restricted/migrations/ehrData"
    )
    @POST("/restricted/migrations/ehrData")
    fun triggerEhrMigrations(
        @Header(BACKGROUND_TASKS_REQUEST_HEADER) requestID: String,
        @Body request: EhrDataMigrationRequestDTO
    ): Call<Void>

    @Headers(
        APPLICATION_JSON_HEADER,
        "${OkHttpMetricsEventListener.URI_PATTERN}: /restricted/migrations/ehrMigrationVersions"
    )
    @GET("/restricted/migrations/ehrMigrationVersions")
    fun getEhrMigrationVersions(): Call<EhrMigrationVersionsDTO>
}

data class OutOfAreaActivationNotificationDTO(val patientPublicId: PersonPublicId)

data class InvitationActivationNotificationDTO(val patientId: Long, val teamId: Long)

data class EhrDataMigrationRequestDTO(val accountPrivateId: AccountPrivateId, val menuDataType: MenuDataType, val ehrDataIds: List<Long>)

data class EhrMigrationVersionsDTO(val versions: Map<MenuDataType, Int>)
