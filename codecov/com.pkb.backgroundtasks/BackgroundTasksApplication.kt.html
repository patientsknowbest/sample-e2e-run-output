package com.pkb.backgroundtasks

import com.pkb.common.spring.data.JpaKotlinRepositoryFactoryBean
import org.springframework.boot.SpringApplication
import org.springframework.boot.autoconfigure.SpringBootApplication
import org.springframework.boot.autoconfigure.domain.EntityScan
import org.springframework.data.jpa.repository.config.EnableJpaAuditing
import org.springframework.data.jpa.repository.config.EnableJpaRepositories

@SpringBootApplication
@EntityScan(value = ["com.pkb.entities", "com.pkb.backgroundtasks.entity",
    "com.pkb.backgroundtasks.wayfinder.recordsservice.entity"])
@EnableJpaRepositories("com.pkb.backgroundtasks.repository", "com.pkb.backgroundtasks.wayfinder.recordsservice.repository",
        repositoryFactoryBeanClass = JpaKotlinRepositoryFactoryBean::class)
@EnableJpaAuditing(dateTimeProviderRef = "dateTimeProvider")
class BackgroundTasksApplication {

    companion object {
        @JvmStatic
        fun main(args: Array<String>) {
            SpringApplication.run(BackgroundTasksApplication::class.java, *args)
        }
    }
}
