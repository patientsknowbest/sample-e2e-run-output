package com.pkb.datamodel.entity.mapper.context;

import org.jetbrains.annotations.Nullable;
import org.mapstruct.BeforeMapping;
import org.mapstruct.MappingTarget;
import org.slf4j.Logger;
import org.springframework.util.ReflectionUtils;

import java.lang.reflect.InvocationTargetException;
import java.util.IdentityHashMap;
import java.util.Map;

import static java.lang.invoke.MethodHandles.lookup;
import static org.slf4j.LoggerFactory.getLogger;

/**
 * Hibernate, mapstruct and immutables don't play nicely together. If you have two JPA entities that include
 * each other with @OneToMany and @ManyToOne mapstruct will end up in a loop and cause a stack overflow. The use
 * of immutables muddies the water slightly further. To overcome the issues mixing these libraries you have to use
 * a mapstruct context (https://mapstruct.org/documentation/stable/api/org/mapstruct/Context.html) that keeps track of
 * what it has already mapped. Additionally mapstruct does not like immutables so for this to work with them you have
 * to do an additional step when retrieving an instance of one.
 */
public class AvoidCyclicReferenceContext {

    private static final Logger LOGGER = getLogger(lookup().lookupClass());

    private Map<Object, Object> knownInstances = new IdentityHashMap<>();

    @BeforeMapping
    public <T> @Nullable T getMappedInstance(Object source) {
        var obj = knownInstances.get(source);
        var method = obj != null ? ReflectionUtils.findMethod(obj.getClass(), "build") : null;
        try {
            //noinspection unchecked
            return (T) (method != null ? method.invoke(obj) : obj);
        } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
            LOGGER.error("Unable to invoke immutables build method", e);
            return null;
        }
    }

    @BeforeMapping
    public void storeMappedInstance(Object source, @MappingTarget Object target) {
        knownInstances.put(source, target);
    }
}
