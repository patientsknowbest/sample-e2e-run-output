package com.pkb.service.phplan;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.phplan.entity.PHPlanTemplate;
import com.pkb.repository.legacy.LegacyPHPlanTemplateRepository;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import org.jetbrains.annotations.NotNull;

import java.util.Collection;
import java.util.Collections;
import java.util.List;
import java.util.Map;
import java.util.Optional;

import static java.util.function.Function.identity;
import static java.util.stream.Collectors.toMap;

public class LegacyPlanTemplateManager extends TransactionManager {

    private final LegacyPHPlanTemplateRepository phPlanTemplateRepository;

    public LegacyPlanTemplateManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService,
                                     UUIDProvider uuidProvider, LegacyPHPlanTemplateRepository phPlanTemplateRepository) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.phPlanTemplateRepository = phPlanTemplateRepository;
    }

    public List<PHPlanTemplate> findPHPlanTemplateList(Collection<Long> templateIds) {
        if (templateIds.isEmpty()) {
            return Collections.emptyList();
        }
        return phPlanTemplateRepository.findPHPlanTemplateList(templateIds);
    }

    public Map<Long, PHPlanTemplate> findPHPlanTemplateMap(@NotNull Collection<Long> templateIds) {
        List<PHPlanTemplate> result = findPHPlanTemplateList(templateIds);
        return result.stream().collect(toMap(com.pkb.phplan.entity.PHPlanTemplate::getId, identity()));
    }

    public Optional<PHPlanTemplate> getPHPlanTemplate(long templateId) {
        return phPlanTemplateRepository.findById(templateId);
    }

    public Optional<PHPlanTemplate> getPHPlanTemplateWithMetrics(long templateId) {
        return phPlanTemplateRepository.getPHPlanTemplateWithMetricsById(templateId);
    }

    public Optional<PHPlanTemplate> getPHPlanTemplateWithFieldsAndMetrics(long templateId) {
        return phPlanTemplateRepository.getPHPlanTemplateWithMetricsById(templateId)
                .map(planTemplate -> {
                    planTemplate.setFields(phPlanTemplateRepository.getPHPlanTemplateFieldsById(templateId));
                    return planTemplate;
                });
    }
}
