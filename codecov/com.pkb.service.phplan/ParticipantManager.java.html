package com.pkb.service.phplan;

import com.google.common.collect.Sets;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.domain.ParticipantService;
import com.pkb.encounter.entity.Message;
import com.pkb.institute.entity.Team;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import org.apache.commons.lang3.StringUtils;
import org.apache.commons.lang3.tuple.Pair;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.time.LocalDateTime;
import java.util.HashSet;
import java.util.List;
import java.util.Map;
import java.util.Set;
import java.util.UUID;


public class ParticipantManager extends TransactionManager {

    private final TeamUserManager teamUserManager;
    private final ParticipantService participantService;

    public ParticipantManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService,
                              UUIDProvider uuidProvider, TeamUserManager teamUserManager, ParticipantService participantService) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.teamUserManager = teamUserManager;
        this.participantService = participantService;
    }

    /**
     * Save the participant if there is not already a participant record.
     *
     * @param personId       the id of the PKBPerson who is a participant in the shared object
     * @param teamId         the teamId of the participant, or null if they are not in a team
     * @param uniqueId       the uniqueId of the shared object in app.menu_data
     */
    public void saveParticipant(long personId, Long teamId, UUID uniqueId) {
        participantService.getParticipant(personId, uniqueId, teamId)
                .ifPresentOrElse(p -> {},
                        () -> createParticipant(personId, teamId, uniqueId, dateTimeService.nowLocalDateTime()));
    }

    public void createParticipant(long personId, @Nullable Long teamId, @NotNull UUID uniqueId,
                                         @NotNull LocalDateTime addedDate) {
        PKBPerson participant = beanFactory.getPKBPersonBean().getPKBPerson(personId);
        participantService.saveParticipant(participant.getId(), uniqueId, addedDate, participant.isPatient() ? null : teamId);
    }

    public void saveParticipantsForMessage(Pair<Message, EHRRequestContext> pair) {
        saveParticipantsForMessage(pair.getValue(), pair.getKey());
    }
    /**
     * Save the participants of a message if they don't exist yet.
     *
     * @param requestContext the requestContext
     * @param message        the message to save participants to
     */
    public void saveParticipantsForMessage(EHRRequestContext requestContext, Message message) {

        transactional(() -> {
            Set<Long> existingParticipantIds = getParticipantIds(message.getFirstMessageUniqueId());
            Set<Long> participantIdsToSave = new HashSet<>();

            if (!StringUtils.isEmpty(message.getSenderId())) {
                participantIdsToSave.add(Long.parseLong(message.getSenderId()));
            }

            if (!StringUtils.isEmpty(message.getReceiverId())) {
                participantIdsToSave.add(Long.parseLong(message.getReceiverId()));
            }

            for (String participantId : message.getParticipants()) {
                participantIdsToSave.add(Long.parseLong(participantId));
            }

            saveNonExistingParticipants(requestContext, participantIdsToSave, message.getFirstMessageUniqueId(), existingParticipantIds);
        });
    }

    private Set<Long> getParticipantIds(UUID uniqueId) {
        return new HashSet<>(participantService.getPublicParticipantPersonIdsByUniqueId(uniqueId));
    }

    private void saveNonExistingParticipants(EHRRequestContext requestContext, Set<Long> personIds, UUID uniqueId, Set<Long> existingParticipantIds) {
        Set<Long> nonExistingIds = Sets.difference(personIds, existingParticipantIds).immutableCopy();

        if (!nonExistingIds.isEmpty()) {
            List<PKBPerson> pkbPersonList = beanFactory.getPKBPersonBean().findPKBPersonList(nonExistingIds);

            if (!pkbPersonList.isEmpty()) {
                Map<Long, Team> activePrimaryTeams = teamUserManager.findActivePrimaryTeams(requestContext, nonExistingIds);
                LocalDateTime localDateTime = dateTimeService.nowLocalDateTime();

                pkbPersonList.stream()
                        .map(person -> Tuple.of(person, activePrimaryTeams.get(person.getId())))
                        .map(personApt -> personApt._2 == null
                                ? new Tuple2<PKBPerson, Long>(personApt._1, null)
                                : Tuple.of(personApt._1, personApt._2.getId()))
                        .forEach(personTeamId -> createParticipant(personTeamId._1.getId(), personTeamId._2, uniqueId, localDateTime));
            }
        }
    }
}
