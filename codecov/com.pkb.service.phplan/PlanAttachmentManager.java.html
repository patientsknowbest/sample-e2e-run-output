public class PlanAttachmentManager extends TransactionManager {

    private enum AttachmentNatureQualifier {
        REAL_ATTACHMENT, PLAN_MAIN_CONTENT
    }

    private static final String PLAN_ATTACHMENT = "planAttachment";

    private final ChunkedDocManager chunkedDocManager;
    private final DocumentManager documentManager;

    public PlanAttachmentManager(PhrConfig config,
                                 TolvenBeanFactory beanFactory,
                                 DateTimeService dateTimeService,
                                 UUIDProvider uuidProvider,
                                 @Lazy ChunkedDocManager chunkedDocManager,
                                 DocumentManager documentManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.chunkedDocManager = chunkedDocManager;
        this.documentManager = documentManager;
    }

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    public List<PHPlanAttachment> getAttachments(Long planId, @NotNull LoggedInEHRRequestContext requestContext, Long accountId) {
        var result = getRealAttachments(singleton(planId), requestContext, singleton(accountId)).get(planId, accountId);
        if (result == null) {
            return emptyList();
        }
        return result;
    }

    public Long addAttachment(EHRRequestContext context, Long planEhrDataId, PHPlanAttachment planAttachment) {
        var attachmentId = planAttachment.getAttachmentId();
        if (attachmentId == null) {
            //new attachment; save the attachment and make the plan refer to new ID
            attachmentId = createAttachment(context, planEhrDataId, planAttachment);
        } else {
            attachmentId = createAttachmentReference(planEhrDataId, planAttachment);
        }
        // make sure these fields are set in the PhPlanAttachment:
        planAttachment.setPlanId(planEhrDataId);
        planAttachment.setAttachmentId(attachmentId);
        if (!planAttachment.getSource().isPopulated()) {
            throw new IllegalStateException("PHRZ-105,PHRZ-104: Source is null for data point!!! Fix code path to set it, and backfill DB!");
        }
        var ehrBean = beanFactory.getEhrRemote();
        var ed = ehrBean.populateEHRData(planAttachment, planAttachment.getAccountId(), MenuDataType.phPlanAttachment, context);
        ed.setDocumentMetadataId(planAttachment.getDocMetadataId());
        ed.generateNewRandomUniqueId();
        ehrBean.saveEHRData(context, ed);
        return attachmentId;

    }

    /**
     * Remove an attachment
     */
    void deleteAttachment(Long attachmentId) {
        chunkedDocManager.deleteAttachmentDetails(attachmentId);
    }

    private long createAttachment(EHRRequestContext requestContext, Long planId, PHPlanAttachment planAttachment) {
        try {
            var docId = planAttachment.getAttachment().getDocMetadataId();
            var userId = planAttachment.getUserId();
            if (docId == null) {
                // this pathway is for saving attachments to a plan where the attachment hasn't been uploaded async. eg saving a template attachment
                var doc = beanFactory.getEhrRemote().createDocument(
                        requestContext,
                        planAttachment.getContent(),
                        planAttachment.getAccountId(),
                        genericMetadata(
                                planAttachment.getMediaType(),
                                userId,
                                PLAN_ATTACHMENT).toJavaMap());
                docId = documentManager.create(doc);
            }
            var docAttachmentId = chunkedDocManager.createAttachment(planId, docId, planAttachment.getAttachment(), userId, planAttachment.getAccountId());
            // set this so it's accessible in the caller...
            planAttachment.getAttachment().setDocMetadataId(docId);
            return docAttachmentId;
        } catch (Exception e) {
            //noinspection deprecation
            throw new PKBPluginException("Exception while adding attachment to plan-" + planId, e);
        }
    }

    private long createAttachmentReference(Long planId, PHPlanAttachment planAttachment) {
        var docAttach = chunkedDocManager.getAttachmentDetails(planAttachment.getAttachmentId())
                .orElseThrow(() -> new IllegalStateException("Unable to create attachment reference due to missing attachment details with id " + planAttachment.getAttachmentId()));
        var attachedDocumentId = docAttach.getAttachedDocumentMetadataId();
        // create a new DocAttachment row
        var docAttachmentId = chunkedDocManager.createAttachment(planId,
                attachedDocumentId,
                planAttachment.getAttachment(),
                planAttachment.getUserId(),
                planAttachment.getAccountId());
        // set this so it's accessible in the caller...
        planAttachment.getAttachment().setDocMetadataId(attachedDocumentId);
        return docAttachmentId;
    }

    @Nullable
    public PHPlanAttachment getAttachmentContent(@NotNull LoggedInEHRRequestContext context, Long attachmentId) {
        try {
            var maybeDocAtt = chunkedDocManager.getAttachmentDetails(attachmentId);
            if (maybeDocAtt.isEmpty()) {
                return null;
            }
            var docAtt = maybeDocAtt.get();
            var attachedDocumentId = docAtt.getAttachedDocumentMetadataId();
            var doc = documentManager.retrieveOrThrow(attachedDocumentId);
            // decryption will fail unless we use the accountId of the document and the logged-in user id
            // user must also have access to that account
            long accountIdOfDoc = ACCOUNT_ID_PROVIDER.apply(doc);
            // throws DecryptionFailedException if that fails
            byte[] content = beanFactory.getEhrRemote().decryptDoc(doc, context).plainText();
            var planAttachmentData = getPKBAttachmentData(context, accountIdOfDoc, attachmentId);
            var attachment = new Attachment();
            attachment.setAccountId(accountIdOfDoc);

            AUTHOR_PERSON_ID_PROVIDER.apply(doc).ifPresent(attachment::setUserId);
            attachment.setId(attachmentId);
            attachment.setDocMetadataId(docAtt.getAttachedDocumentMetadataId());
            attachment.setFilename(docAtt.getDescription());
            attachment.setMediaType(MEDIA_TYPE_PROVIDER.apply(doc));
            attachment.setContent(content);
            Optional.ofNullable(planAttachmentData).ifPresent(planAttachment -> planAttachment.setAttachment(attachment));
            return planAttachmentData;
        } catch (Exception exception) {
            throw new RuntimeException("Exception while geting the contents for attachment id-" + attachmentId, exception);
        }
    }

    public Table<Long, Long, List<PHPlanAttachment>> getActionPlanContent(Collection<Long> planIds, @NotNull LoggedInEHRRequestContext context, Collection<Long> accountIds) {

    public Table<Long, Long, List<PHPlanAttachment>> getAttachments(Collection<Long> planIds, @NotNull LoggedInEHRRequestContext context, Collection<Long> accountIds) {

    private Table<Long, Long, List<PHPlanAttachment>> getRealAttachments(Collection<Long> planIds, @NotNull LoggedInEHRRequestContext context, Collection<Long> accountIds) {
        return getAttachments(planIds, context, accountIds, AttachmentNatureQualifier.REAL_ATTACHMENT);
    }

    private Table<Long, Long, List<PHPlanAttachment>> getAttachments(Collection<Long> planIds, @NotNull LoggedInEHRRequestContext context, Collection<Long> accountIds, AttachmentNatureQualifier attachmentNatureQualifier) {
        if (isEmpty(planIds) || isEmpty(accountIds)) {
            return ImmutableTable.of();
        }
        try {
            var attachmentsByPlanId = chunkedDocManager.findAttachmentsAsAttachment(planIds);
            var attachmentByAttachmentIdAndAccountId = getAttachmentMap(planIds, context, accountIds, attachmentNatureQualifier);
            Table<Long, Long, List<PHPlanAttachment>> result = ArrayTable.create(planIds, accountIds);

            attachmentsByPlanId.forEach((planId, value) ->
                    value.forEach(attachment -> {
                        PHPlanAttachment planAttachment = attachmentByAttachmentIdAndAccountId.get(attachment.getId(), attachment.getAccountId());
                        if (planAttachment == null) {
                            LOGGER.warn("Unable to find attachment corresponding to id-{} accountid-{} for planId={}", attachment.getId(), attachment.getAccountId(), planId);
                            // Pravina : this is by error in saving the plan due to an earlier  bug
                            // we choose to skip it instead of letting the whole plan load crash
                        } else {
                            planAttachment.setAttachment(attachment);
                            List<PHPlanAttachment> attachments = result.get(planId, attachment.getAccountId());
                            if (attachments == null) {
                                attachments = new LinkedList<>();
                                result.put(planId, attachment.getAccountId(), attachments);
                            }
                            attachments.add(planAttachment);
                        }
                    }));
            return result;
        } catch (Exception e) {
            //noinspection deprecation
            throw new PKBPluginException("Exception while getting PH plan list", e);
        }
    }

    private Table<Long, Long, PHPlanAttachment> getAttachmentMap(Collection<Long> planIds, @NotNull LoggedInEHRRequestContext context, Collection<Long> accountIds, AttachmentNatureQualifier attachmentNatureQualifier) {
        if (isEmpty(planIds) || isEmpty(accountIds)) {
            return ImmutableTable.of();
        }
        EHRSearch<PHPlanAttachment> search = new EHRSearch<>(accountIds, PHPlanAttachment.class, PHPlanAttachment.MS_PATH);
        search.addFilter(new PKBFilter(PHPlanAttachment.ATTACHMENT_PLAN_ID, PKBFilter.Operator.IN, planIds));
        var planAttachments = beanFactory.getEhrRemote().queryDecryptAndPopulateDataByAccountId(search, context);
        Predicate<RecordWithId<Long, PHPlanAttachment>> filterResults = null;
        switch (attachmentNatureQualifier) {
            case REAL_ATTACHMENT:
                filterResults = p -> !p.record().getCarePlanMainContent();
                break;
            case PLAN_MAIN_CONTENT:
                filterResults = p -> p.record().getCarePlanMainContent();
                break;
            default:
                throw new IllegalArgumentException(String.format("Unknown attachment nature qualifier: %s", attachmentNatureQualifier));
        }
        var attachmentIds = planAttachments.stream()
                .filter(filterResults)
                .map(r -> r.record().getAttachmentId())
                .collect(toSet());
        if (isEmpty(attachmentIds)) {
            return ImmutableTable.of();
        }
        Table<Long, Long, PHPlanAttachment> result = ArrayTable.create(attachmentIds, accountIds);
        planAttachments.stream()
                .filter(filterResults)
                .forEach(r -> result.put(r.record().getAttachmentId(), r.id(), r.record()));
        return result;
    }

    private @Nullable PHPlanAttachment getPKBAttachmentData(@NotNull LoggedInEHRRequestContext context, long accountId, Long attachmentId) {
        var search = new EHRSearch<>(accountId, PHPlanAttachment.class, PHPlanAttachment.MS_PATH);
        search.addFilter(new PKBFilter(PHPlanAttachment.ATTACHMENT_ID, PKBFilter.Operator.EQUAL, attachmentId));
        var planAttList = beanFactory.getEhrRemote().queryAndDecryptTypedData(search, context);
        if (planAttList.size() == 1) {
            // set the ID as *not* the ehrData.id, but the attachmentID (DocAttach.id)
            var planAtt = planAttList.get(0);
            planAtt.setId(planAtt.getAttachmentId());
            return planAtt;
        } else if (planAttList.isEmpty()) {
            return null;
        } else {
            throw new PKBPluginException("More than 1 attachment with same Id-" + attachmentId);
        }
    }
