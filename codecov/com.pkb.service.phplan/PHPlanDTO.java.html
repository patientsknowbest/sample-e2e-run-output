package com.pkb.service.phplan;

import com.pkb.datamodel.plantemplate.PHPlanTemplate;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.entities.enums.TemplateStatus;
import com.pkb.phplan.entity.PHPlan.ApprovalStatus;
import com.pkb.service.datapoint.DataPointDTO;
import org.apache.commons.lang3.NotImplementedException;
import org.jetbrains.annotations.NotNull;

import java.util.Date;
import java.util.UUID;

import static com.pkb.entities.enums.TemplateStatus.INDRAFT;

public class PHPlanDTO extends DataPointDTO {

    private Long id;
    private UUID uniqueId;
    private String name;
    private Date creationDate;
    private String creatorName;
    private String editorName;
    private Date reviewDate;

    private Date lastEditedOn;
    private Long creatorId;

    private TemplateStatus status;

    private ApprovalStatus approvalStatus;
    private String approverName;

    private boolean generalHealthConsentRequired;
    private boolean mentalHealthConsentRequired;
    private boolean sexualHealthConsentRequired;
    private boolean socialCareConsentRequired;

    private boolean isDeletableByCurrentUser;

    public static PHPlanDTO from(PHPlanTemplate template) {
        PHPlanDTO planDTO = new PHPlanDTO();
        planDTO.setId(template.getId());
        planDTO.setName(template.getName());
        return planDTO;
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public Date getCreationDate() {
        return creationDate;
    }

    public void setCreationDate(Date creationDate) {
        this.creationDate = creationDate;
    }

    public String getCreatorName() {
        return creatorName;
    }

    public void setCreatorName(String creatorName) {
        this.creatorName = creatorName;
    }

    public UUID getUniqueId() {
        return uniqueId;
    }

    public void setUniqueId(UUID uniqueId) {
        this.uniqueId = uniqueId;
    }

    public TemplateStatus getStatus() {
        return status;
    }

    public void setStatus(TemplateStatus status) {
        this.status = status;
    }

    public String getEditorName() {
        return editorName;
    }

    public void setEditorName(String editorName) {
        this.editorName = editorName;
    }

    public Date getLastEditedOn() {
        return lastEditedOn;
    }

    public void setLastEditedOn(Date lastEditedOn) {
        this.lastEditedOn = lastEditedOn;
    }

    public Date getReviewDate() {
        return reviewDate;
    }

    public void setReviewDate(Date reviewDate) {
        this.reviewDate = reviewDate;
    }

    public ApprovalStatus getApprovalStatus() {
        return approvalStatus;
    }

    public void setApprovalStatus(ApprovalStatus approvalStatus) {
        this.approvalStatus = approvalStatus;
    }

    public String getApproverName() {
        return approverName;
    }

    public void setApproverName(String approverName) {
        this.approverName = approverName;
    }

    public Long getCreatorId() {
        return creatorId;
    }

    public void setCreatorId(Long creatorId) {
        this.creatorId = creatorId;
    }

    public boolean isGeneralHealthConsentRequired() {
        return generalHealthConsentRequired;
    }

    public boolean isMentalHealthConsentRequired() {
        return mentalHealthConsentRequired;
    }

    public boolean isSexualHealthConsentRequired() {
        return sexualHealthConsentRequired;
    }

    public boolean isSocialCareConsentRequired() {
        return socialCareConsentRequired;
    }

    public boolean isDeletableByCurrentUser() {
        return isDeletableByCurrentUser;
    }

    public void setDeletableByCurrentUser(boolean deletableByCurrentUser) {
        isDeletableByCurrentUser = deletableByCurrentUser;
    }

    public PHPlanDTO setPrivacyFlag(@NotNull PrivacyFlag requiredFlag) {
        switch (requiredFlag) {
            case GENERAL:
                generalHealthConsentRequired = true;
                mentalHealthConsentRequired = false;
                sexualHealthConsentRequired = false;
                socialCareConsentRequired = false;
                return this;
            case MENTAL_HEALTH:
                generalHealthConsentRequired = false;
                mentalHealthConsentRequired = true;
                sexualHealthConsentRequired = false;
                socialCareConsentRequired = false;
                return this;
            case SEXUAL_HEALTH:
                generalHealthConsentRequired = false;
                mentalHealthConsentRequired = false;
                sexualHealthConsentRequired = true;
                socialCareConsentRequired = false;
                return this;
            case SOCIAL_CARE:
                generalHealthConsentRequired = false;
                mentalHealthConsentRequired = false;
                sexualHealthConsentRequired = false;
                socialCareConsentRequired = true;
                return this;
            default:
                throw new NotImplementedException("Privacy flag type " + requiredFlag + " not implemented yet!");
        }
    }

    public boolean isDraft() {
        return this.status == INDRAFT;
    }

    public boolean isNotEditedBefore() {
        return this.lastEditedOn == null;
    }

    public boolean isDraftNeverEdited() {
        return isDraft() && isNotEditedBefore();
    }
}
