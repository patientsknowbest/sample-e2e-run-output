package com.pkb.action.sso;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.crypto.util.RandomUtil;
import com.pkb.domain.SSOService;
import com.pkb.entities.enums.SSOType;
import com.pkb.sso.SsoMetrics;
import io.vavr.control.Try;
import org.springframework.beans.factory.annotation.Autowired;

import static com.pkb.entities.enums.SSOType.EMIS;
import static com.pkb.entities.enums.SSOType.SYSTMONE;
import static com.pkb.entities.enums.SSOType.UNKNOWN;
import static com.pkb.util.SSOConstants.SSO_AUTH_PASSWORD_LENGTH;
import static com.pkb.util.SSOConstants.SSO_AUTH_USERNAME_LENGTH;

public class CredentialsGenerationAction extends BaseAction {

    static final String SUCCESS_SYSTM_ONE = "successSystmOne";
    static final String SUCCESS_EMIS = "successEmis";

    protected SSOService ssoService;

    protected String ssoUsername;

    protected String ssoPassword;

    private boolean emisCredsExist = false;

    private boolean systmOneCredsExist = false;

    private String subTab = "generateCredentials";

    private String tab = "myaccount";

    private SSOType ssoType = UNKNOWN;

    private SsoMetrics ssoMetrics;

    @Override
    public String input() {
        var credsStatus = ssoService.getCredentialStatus(getLoggedInEHRRequestContext().getAccessingUserPublicId());
        emisCredsExist = credsStatus.contains(EMIS);
        systmOneCredsExist = credsStatus.contains(SYSTMONE);
        return Action.SUCCESS;
    }

    @Override
    public void validate() {
        if (ssoType == UNKNOWN) {
            addActionError(getText("generateCredsInput.err.unknown.sso.type", new String[] { ssoType.toString() }));
        }
    }


    @Override
    public String execute() throws Exception {
        ssoUsername = RandomUtil.randomString(SSO_AUTH_USERNAME_LENGTH);
        ssoPassword = RandomUtil.randomPassword(SSO_AUTH_PASSWORD_LENGTH);
        LoggedInEHRRequestContext loggedInEHRRequestContext = getLoggedInEHRRequestContext();
        ssoService.createRequiredSSOInfo(loggedInEHRRequestContext.getAccessingUserPublicId(), loggedInEHRRequestContext.getAccessingUserId(), ssoUsername, ssoPassword, ssoType);
        ssoMetrics.incrementCredentialsGenerated(loggedInUserUtil.getLoggedInUserPrimaryTeam().getCode(), ssoType);
        return ssoType == EMIS ? SUCCESS_EMIS : SUCCESS_SYSTM_ONE;
    }

    public String getSsoUsername() {
        return ssoUsername;
    }

    public String getSsoPassword() {
        return ssoPassword;
    }

    public boolean isCredsExist() { return emisCredsExist || systmOneCredsExist; }

    public String getTab() {
        return tab;
    }

    public void setTab(String tab) {
        this.tab = tab;
    }

    public String getSubTab() {
        return subTab;
    }

    public void setSubTab(String subTab) {
        this.subTab = subTab;
    }


    public void setSsoType(String ssoType) {
        this.ssoType = Try.of(() -> SSOType.valueOf(ssoType)).getOrElse(UNKNOWN);
    }

    public String getSsoType() {
        return ssoType.toString();
    }

    public boolean isSystmOneSsoEnabled() {
        return config.isSystmOneSsoEnabled();
    }

    @Autowired
    public void setSsoService(SSOService ssoService) {
        this.ssoService = ssoService;
    }

    @Autowired
    public void setSsoMetrics(SsoMetrics ssoMetrics) {
        this.ssoMetrics = ssoMetrics;
    }
}
