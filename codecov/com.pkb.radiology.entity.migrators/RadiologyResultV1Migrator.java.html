package com.pkb.radiology.entity.migrators;

import com.pkb.annotation.EHRVersionMigrator;
import com.pkb.app.entity.EHRData;
import com.pkb.radiology.entity.RadiologyResult;
import org.apache.commons.lang3.StringUtils;

/**
 * version 1 : migrate encrypted field dataEneterId to Source.personid
 */
public class RadiologyResultV1Migrator extends EHRVersionMigrator {

    public RadiologyResultV1Migrator() {
        super(1);
    }

    @Override
    public void migrate(EHRData ehr) {
        Object dataEntererIdObj = ehr.getEncryptedField(RadiologyResult.DATAENTERERID);
        if (dataEntererIdObj instanceof String) {
            // migrate it to source.personid field
            String dataEntererId = (String) dataEntererIdObj;
            if (!StringUtils.isEmpty(dataEntererId)) {
                ehr.setSourcePersonId(Long.parseLong(dataEntererId));
            }
        }
    }

}
