package com.pkb.radiology.entity.migrators;

import com.pkb.annotation.EHRVersionMigrator;
import com.pkb.app.entity.EHRData;
import com.pkb.radiology.entity.RadiologyResult;
import org.apache.commons.text.StringEscapeUtils;

import static org.apache.commons.lang3.StringUtils.isNotBlank;

/**
 * Description field was persisted with html escaping previously, without reverting that, it would be displayed in a double escaped form, as
 * the UI now properly escapes it.
 */
public class RadiologyResultV2Migrator extends EHRVersionMigrator {

    public RadiologyResultV2Migrator() {
        super(2);
    }

    @Override
    public void migrate(EHRData ehr) {
        Object descriptionField = ehr.getEncryptedField(RadiologyResult.DESCRIPTION);
        if (descriptionField instanceof String) {
            String description = (String) descriptionField;
            if (isNotBlank(description)) {
                ehr.setEncryptedField(RadiologyResult.DESCRIPTION, StringEscapeUtils.unescapeHtml4(description));
            }
        }
    }

}
