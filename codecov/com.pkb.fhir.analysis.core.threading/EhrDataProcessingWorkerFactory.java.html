package com.pkb.fhir.analysis.core.threading;

import com.pkb.fhir.analysis.configuration.AnalysisConfig;
import com.pkb.fhir.analysis.core.domain.EhrDataProcessingContext;
import com.pkb.fhir.analysis.process.check.AlreadyProcessedDataChecker;
import com.pkb.fhir.analysis.process.collect.EhrDataCollector;
import com.pkb.fhir.analysis.process.perform.EhrDataProcessor;

/**
 * This is a core part of the application.
 * Based on the selected process, the factory creates workers {@link EhrDataProcessorWorker} with the corresponding
 * collector-checker-processor implementations.
 * The 3 procedures are wired together for each implementation during configuration {@link AnalysisConfig#ehrDataProcessorWorkerFactoryMap).
 */
public class EhrDataProcessingWorkerFactory {
    private final EhrDataProcessor ehrDataProcessor;
    private final EhrDataCollector ehrDataCollector;
    private final AlreadyProcessedDataChecker processedDataChecker;

    public EhrDataProcessingWorkerFactory(EhrDataCollector ehrDataCollector, EhrDataProcessor ehrDataProcessor, AlreadyProcessedDataChecker processedDataChecker) {
        this.ehrDataProcessor = ehrDataProcessor;
        this.ehrDataCollector = ehrDataCollector;
        this.processedDataChecker = processedDataChecker;
    }

    public EhrDataProcessorWorker createEhrDataProcessorWorker(String processId, EhrDataProcessingContext context) {
        return new EhrDataProcessorWorker(processId, ehrDataProcessor, context, ehrDataCollector, processedDataChecker);
    }
}
