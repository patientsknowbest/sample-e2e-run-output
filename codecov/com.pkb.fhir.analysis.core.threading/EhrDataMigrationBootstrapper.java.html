package com.pkb.fhir.analysis.core.threading;

import com.pkb.fhir.analysis.core.domain.EhrDataProcessingContext;
import com.pkb.fhir.analysis.core.domain.EhrDataProcessingState;
import com.pkb.fhir.analysis.core.domain.MigrationParameters;
import com.pkb.fhir.analysis.process.domain.ProcessorEnum;
import com.pkb.fhir.analysis.ui.rest.MigrationController;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.time.Instant;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.concurrent.CompletableFuture;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.TimeUnit;

import static java.lang.invoke.MethodHandles.lookup;

/**
 * This is a core part of the application.
 * The migration parameters are passed in from the controller {@link MigrationController}.
 * By default, we don't override already processed data, the min and max menu_data ids are not required: we process all data if left empty.
 * It initializes the context, uses the worker factory and executes multiple workers. {@link EhrDataProcessorWorker}.
 */
public class EhrDataMigrationBootstrapper {
    private static final Logger LOG = LoggerFactory.getLogger(lookup().lookupClass());

    private final ExecutorServiceFactory executorServiceFactory;
    private final Map<ProcessorEnum, EhrDataProcessingWorkerFactory> ehrDataProcessorWorkerFactoryMap;
    private EhrDataProcessingContext context;
    private ExecutorService taskExecutor;
    private List<EhrDataProcessorWorker> workers = new ArrayList<>();

    public EhrDataMigrationBootstrapper(ExecutorServiceFactory executorServiceFactory, Map<ProcessorEnum, EhrDataProcessingWorkerFactory> ehrDataProcessorWorkerFactoryMap) {
        this.executorServiceFactory = executorServiceFactory;
        this.ehrDataProcessorWorkerFactoryMap = ehrDataProcessorWorkerFactoryMap;
        context = null;
    }

    public synchronized void bootstrapProcessorWorkers(MigrationParameters migrationParameters) {
        if (this.context != null) {
            throw new IllegalStateException("There is an ongoing procedure. Stop it first.");
        }
        applyDefaults(migrationParameters);
        if (tearDownProcesses()) {
            doBootstrapMigration(migrationParameters);
        } else {
            throw new RuntimeException("Failed to stop previous run");
        }

    }

    private void doBootstrapMigration(MigrationParameters migrationParameters) {
        Instant start = Instant.now();
        taskExecutor = executorServiceFactory.createExecutorService(migrationParameters.getMaxThreads());
        context = new EhrDataProcessingContext(migrationParameters, new EhrDataProcessingState(start, migrationParameters.getStartFromId()));

        LOG.info("Bootstrapping {} worker threads to do {}, each working with page size {} on menu data type {}", migrationParameters.getMaxThreads(), migrationParameters.getProcessorEnum(), migrationParameters.getDataPageSize(), migrationParameters.getMenuDataType());

        for (int i = 0; i < migrationParameters.getMaxThreads(); ++i) {
            createWorker(migrationParameters.getProcessorEnum(), i);
        }
        watchWorkersCompletion();
    }

    private void watchWorkersCompletion() {
        CompletableFuture.runAsync(() -> {
            try {
                taskExecutor.shutdown();
                taskExecutor.awaitTermination(9999, TimeUnit.DAYS);
            } catch (InterruptedException e) {
                LOG.error(e.getMessage(), e);
            }
        }).thenApply(i -> {
            context.getState().registerEnd(Instant.now());
            LOG.info("All process finished in {} seconds", context.getState().getRuntimeSeconds());
            return "";
        });
    }

    private void createWorker(ProcessorEnum processorEnum, int workerNumber) {
        String processId = String.format("ehr-data-processor-%d", workerNumber);
        context.getState().registerProcessingState(processId);
        EhrDataProcessorWorker ehrDataProcessorWorker = ehrDataProcessorWorkerFactoryMap.get(processorEnum).createEhrDataProcessorWorker(processId, context);
        workers.add(ehrDataProcessorWorker);
        taskExecutor.submit(ehrDataProcessorWorker);
    }

    private void applyDefaults(MigrationParameters migrationParameters) {
        if (migrationParameters.getMaxThreads() < 0) {
            migrationParameters.setMaxThreads(5);
        }
        if (migrationParameters.getDataPageSize() < 0 || migrationParameters.getDataPageSize() > 10000) {
            migrationParameters.setDataPageSize(1000);
        }
        if (migrationParameters.getStartFromId() == null) {
            migrationParameters.setStartFromId(-1L);
        }
    }

    public EhrDataProcessingContext getState() {
        return context;
    }

    public boolean tearDownProcesses() {
        shutdown();
        if (context == null || context.getState().isFinished()) {
            workers.clear();
            context = null;
            return true;
        } else {
            return false;
        }
    }

    public void shutdown() {
        if (taskExecutor != null) {
            workers.forEach(EhrDataProcessorWorker::cancel);
            LOG.info("Executor shut down. Processes cancelled.");
        }
    }

}
