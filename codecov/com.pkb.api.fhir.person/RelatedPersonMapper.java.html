package com.pkb.api.fhir.person;

import com.pkb.api.fhir.foundation.ReferenceMapper;
import com.pkb.datamodel.user.Patient;
import org.apache.commons.lang3.StringUtils;
import org.hl7.fhir.dstu3.model.HumanName;
import org.hl7.fhir.dstu3.model.RelatedPerson;
import org.jetbrains.annotations.NotNull;

public class RelatedPersonMapper {

    private ReferenceMapper referenceMapper;

    public RelatedPersonMapper(ReferenceMapper referenceMapper) {
        this.referenceMapper = referenceMapper;
    }

    public @NotNull RelatedPerson toRelatedPerson(@NotNull Patient patient, @NotNull Patient carer) {
        RelatedPerson relatedPerson = new RelatedPerson();
        relatedPerson.setPatient(referenceMapper.toReferenceWithName(patient));

        HumanName name = relatedPerson.addName();

        carer.demographics().getTitle().filter(StringUtils::isNotBlank).peek(name::addPrefix);
        carer.demographics().getFirstName().filter(StringUtils::isNotBlank).peek(name::addGiven);
        carer.demographics().getLastName().filter(StringUtils::isNotBlank).peek(name::setFamily);

        return relatedPerson;
    }
}
