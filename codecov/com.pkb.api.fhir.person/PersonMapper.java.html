package com.pkb.api.fhir.person;

import com.pkb.api.fhir.foundation.ReferenceMapper;
import com.pkb.datamodel.user.Professional;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.hl7.fhir.dstu3.model.Person;

public class PersonMapper {
    private final ReferenceMapper referenceMapper;

    public PersonMapper(ReferenceMapper referenceMapper) {
        this.referenceMapper = referenceMapper;
    }

    public Option<Person> toPerson(List<Professional> source) {

        if (source.isEmpty()) {
            return Option.none();
        }

        Person dest = new Person();

        // Quick hack 8/4/2019
        // PersonMapper is only used to map lists of pros who have multiple datamodel.Person records linked by a single humanuuid.
        // The id of the single resulting fhir.Person needs to be that humanuuid, which will be used in the fullUrl
        dest.setId(source.get(0).getHumanUUID().getOrElseThrow(IllegalStateException::new).toString());

        source.forEach(p -> {
            dest.addLink().setTarget(referenceMapper.toReferenceWithName(p));
        });

        return Option.of(dest);

    }
}
