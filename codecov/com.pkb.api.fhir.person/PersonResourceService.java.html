package com.pkb.api.fhir.person;

import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import ca.uhn.fhir.rest.server.exceptions.ResourceNotFoundException;
import com.pkb.api.fhir.AbstractResourceService;
import com.pkb.api.fhir.exception.ServerResponseExceptionMapper;
import com.pkb.authentication.AuthenticatedIdentity;
import com.pkb.authz.AuthorizationDataService;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.authz.data.AuthorizationInputs;
import com.pkb.datamodel.user.Professional;
import com.pkb.deletion.model.DeleteRequestStatus;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.criteria.FhirCriteriaResult;
import com.pkb.domain.criteria.ImmutableFhirCriteriaResult;
import com.pkb.domain.criteria.OperationOutcomeFactory;
import com.pkb.domain.person.PersonService;
import io.vavr.collection.Set;
import io.vavr.control.Either;
import org.hl7.fhir.dstu3.model.OperationOutcome;
import org.hl7.fhir.dstu3.model.Person;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.pkb.authz.ImmutableTargetMeta.targetMeta;
import static com.pkb.authz.action.Action.DELETE;
import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.context.Path.FHIR;
import static com.pkb.authz.data.ImmutableAuthorizationInputs.inputs;
import static io.vavr.API.None;
import static io.vavr.API.Set;
import static io.vavr.API.Some;
import static java.lang.String.format;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity.ERROR;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueType.NOTFOUND;

public class PersonResourceService extends AbstractResourceService<Person> {

    private final AuthorizationDataService authorizationDataService;
    private final PersonService personService;

    private final PersonMapper personMapper;

    public PersonResourceService(
            @NotNull AuthorizationDataService authorizationDataService,
            @NotNull PersonService personService,
            @NotNull PersonMapper personMapper,
            @NotNull ServerResponseExceptionMapper exceptionMapper) {
        super(exceptionMapper);
        this.authorizationDataService = authorizationDataService;
        this.personService = personService;
        this.personMapper = personMapper;
    }

    public FhirCriteriaResult<Person> findPersonForPractitioner(AuthenticatedIdentity authenticatedIdentity, @NotNull UUID publicId) {

        AuthorizationInputs authorizationInputs = inputs()
                .authenticatedIdentity(authenticatedIdentity)
                .action(SEARCH)
                .addTargets(targetMeta(Professional.class, Set(), Some(publicId), true))
                .path(FHIR)
                .build();

        AuthorizationData authorizationData = authorizationDataService.fetch(authorizationInputs);
        return personService.findAllProfessionalsForProfessional(authorizationData, publicId)
                .fold(ignored -> FhirCriteriaResult.empty(), this::mapResult);
    }

    private FhirCriteriaResult<Person> mapResult(CriteriaResult<Professional> result) {
        var people = personMapper.toPerson(result.getPage()).toList();
        return ImmutableFhirCriteriaResult.<Person>fhirCriteriaResult(people.size()).withPage(people);
    }


    public Either<BaseServerResponseException, UUID> requestDeletion(AuthenticatedIdentity authenticatedIdentity, Set<UUID> personIds, boolean forceDelete) {
        AuthorizationInputs authorizationInputs = inputs()
                .authenticatedIdentity(authenticatedIdentity)
                .action(DELETE)
                .addAllTargets(personIds.map(personId -> targetMeta(com.pkb.datamodel.user.Person.class, personIds.map(it -> it), None(), false)))
                .path(FHIR)
                .build();

        AuthorizationData authorizationData = authorizationDataService.fetch(authorizationInputs);
        return personService.requestDeletion(authorizationData, personIds, forceDelete)
                .mapLeft(err -> exceptionToFhir(err, Person.class));
    }

    public Either<BaseServerResponseException, DeleteRequestStatus> getDeleteOperationStatus(UUID requestId) {
        return personService.getDeleteRequestStatus(requestId)
                .toEither(() -> {
                    String message = format("Delete operation with id %1$s not found", requestId);
                    OperationOutcome outcome = OperationOutcomeFactory.operationOutcome()
                            .addIssue(new OperationOutcome.OperationOutcomeIssueComponent()
                                    .setSeverity(ERROR)
                                    .setCode(NOTFOUND)
                                    .setDiagnostics(message));

                    return new ResourceNotFoundException(message, outcome);
                });
    }

}
