package com.pkb.api.fhir.person;

import com.pkb.datamodel.user.Demographics;
import io.vavr.API;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.hl7.fhir.dstu3.model.HumanName;

public class HumanNameMapper {

    public Option<List<HumanName>> toHumanName(Demographics demographics) {
        return demographics.getFirstName()
                .flatMap(firstName -> demographics.getLastName()
                        .map(lastName -> {
                            //We don't make any judgement calls here, we add the firstName and middleName fields off the
                            // database exactly as they are stored as two separate indexes in the given names array.
                            HumanName humanName = new HumanName().setFamily(lastName).addGiven(firstName);

                            demographics.getMiddleName().peek(humanName::addGiven);
                            demographics.getTitle().peek(humanName::addPrefix);
                            return humanName;
                        }))
                .map(API::List);
    }
}
