package com.pkb.api.fhir.person;

import com.pkb.api.fhir.CanonicalExtensionURLs;
import com.pkb.api.fhir.NationalIdToIdentifierMapper;
import com.pkb.api.fhir.address.AddressMapper;
import com.pkb.api.fhir.contact.ContactPointMapper;
import com.pkb.datamodel.LocalId;
import com.pkb.datamodel.NationalId;
import com.pkb.datamodel.user.Demographics;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.criteria.FhirCriteriaResult;
import com.pkb.domain.criteria.ImmutableFhirCriteriaResult;
import io.vavr.collection.List;
import org.hl7.fhir.dstu3.model.BooleanType;
import org.hl7.fhir.dstu3.model.DateTimeType;
import org.hl7.fhir.dstu3.model.DateType;
import org.hl7.fhir.dstu3.model.Extension;
import org.hl7.fhir.dstu3.model.Identifier;
import org.hl7.fhir.dstu3.model.Patient;
import org.jetbrains.annotations.NotNull;

import java.time.Instant;
import java.time.ZoneOffset;
import java.util.Date;

import static ca.uhn.fhir.model.api.TemporalPrecisionEnum.DAY;

public class PatientMapper extends AbstractPersonMapper {

    private final ContactPointMapper contactPointMapper;
    private final AddressMapper addressMapper;
    private final HumanNameMapper nameMapper;

    public PatientMapper(ContactPointMapper contactPointMapper, AddressMapper addressMapper, HumanNameMapper nameMapper) {
        this.contactPointMapper = contactPointMapper;
        this.addressMapper = addressMapper;
        this.nameMapper = nameMapper;
    }

    public @NotNull Patient toPatient(@NotNull Demographics demographics) {
        Patient patient = new Patient();
        patient.setId(demographics.getPublicId().toString());
        nameMapper.toHumanName(demographics).peek(name -> patient.setName(name.asJava()));
        demographics.getDateOfBirth()
                .peek(dob -> {
                    Instant fhirInstant = dob.atStartOfDay(ZoneOffset.UTC).toInstant();
                    patient.setBirthDateElement(new DateType(Date.from(fhirInstant), DAY));
                });
        patient.setGender(convertGender(demographics.getGender()));
        demographics.getDeceasedDate()
                .peek(deceasedDate -> {
                    Instant fhirInstant = deceasedDate.atStartOfDay(ZoneOffset.UTC).toInstant();
                    patient.setDeceased(new DateTimeType(Date.from(fhirInstant), DAY));
                })
                .onEmpty(() -> patient.setDeceased(new BooleanType(demographics.isDeceased())));

        patient.setIdentifier(toFhirIdentifiers(demographics.getNationalId()).asJavaMutable());
        demographics.getOrgLevelId().forEach(n -> patient.addIdentifier(toFhirIdentifier(n)));
        demographics.getTeamLevelId().forEach(n -> patient.addIdentifier(toFhirIdentifier(n)));
        patient.addAddress(addressMapper.toAddress(demographics.getAddress()));
        contactPointMapper.toPersonContactPoints(List.narrow(demographics.getEmail()), List.narrow(demographics.getPhone()))
                .sorted(((contactPoint1, contactPoint2) -> contactPoint2.getSystem().compareTo(contactPoint1.getSystem())))
                .forEach(patient::addTelecom);
        patient.addExtension(new Extension()
                .setUrl(CanonicalExtensionURLs.sharingDisabled())
                .setValue(new BooleanType(demographics.isSharingDisabled())));

        return patient;
    }

    public @NotNull FhirCriteriaResult<Patient> toPatientList(@NotNull CriteriaResult<Demographics> demographics) {
        List<Patient> patients = demographics.getPage().map(this::toPatient);
        return ImmutableFhirCriteriaResult.<Patient>fhirCriteriaResult()
                .total(patients.size())
                .page(patients)
                .build();
    }

    private List<Identifier> toFhirIdentifiers(List<NationalId> nationalIdSet) {
        return nationalIdSet.map(NationalIdToIdentifierMapper::toIdentifier);
    }

    private Identifier toFhirIdentifier(LocalId<?, ?> id) {
        Identifier identifier = new Identifier();
        id.getValue().peek(identifier::setValue);
        identifier.setSystem("urn:uuid:" + id.getType().getPublicId());
        return identifier;
    }
}
