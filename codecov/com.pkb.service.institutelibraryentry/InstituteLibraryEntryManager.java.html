package com.pkb.service.institutelibraryentry;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.institute.entity.InstituteLibraryEntry;
import com.pkb.repository.legacy.LegacyInstituteLibraryEntryRepository;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;

import java.util.List;

public class InstituteLibraryEntryManager extends TransactionManager {

    private final LegacyInstituteLibraryEntryRepository instituteLibraryEntryRepository;

    public InstituteLibraryEntryManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider,
                                        LegacyInstituteLibraryEntryRepository instituteLibraryEntryRepository) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.instituteLibraryEntryRepository = instituteLibraryEntryRepository;
    }

    /**
     * Removes the specified {@code InstituteLibraryEntry} from
     * the specified team.
     *
     * @param teamCode The team for which the entry should be removed.
     * @param entryId  The ID of the entry to remove.
     * @return true - if, and only if, the delete was successful.
     */
    public boolean deleteLibraryEntry(long entryId) {
        boolean success = false;

        try {
            instituteLibraryEntryRepository.deleteById(entryId);
        } catch (Exception e) {
            throw new RuntimeException("Plugin exception deleting library entry: ", e);
        }
        return success;
    }

    /**
     * @param parentEntryId parent id
     * @return
     */
    public List<InstituteLibraryEntry> getLibraryEntryChildrenById(long parentEntryId) {
        List<InstituteLibraryEntry> entries;
        try {
            entries = instituteLibraryEntryRepository.findByParentId(parentEntryId);
        } catch (Exception e) {
            throw new RuntimeException("Plugin exception deleting library entry: ", e);
        }
        return entries;
    }

    /**
     * @param entryId db id of entry
     * @return library entry
     */
    public InstituteLibraryEntry getLibraryEntryById(long entryId) {
        InstituteLibraryEntry entry;
        try {
            entry = instituteLibraryEntryRepository.findById(entryId).orElse(null);
        } catch (Exception e) {
            throw new RuntimeException("Plugin exception deleting library entry: ", e);
        }
        return entry;
    }

    public Long updateLibraryEntry(InstituteLibraryEntry entry) {
        try {
            return instituteLibraryEntryRepository.saveAndFlush(entry).getId();
        } catch (Exception exception) {
            throw new RuntimeException("Error while updating library entry " + entry, exception);
        }
    }

    public Long createLibraryEntry(InstituteLibraryEntry entry) {
        try {
            return instituteLibraryEntryRepository.saveAndFlush(entry).getId();
        } catch (Exception exception) {
            throw new RuntimeException("Error while updating library entry " + entry, exception);
        }
    }
}
