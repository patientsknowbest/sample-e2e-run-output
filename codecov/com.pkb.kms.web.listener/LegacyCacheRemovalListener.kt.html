package com.pkb.kms.web.listener

import com.pkb.kms.service.cache.HasDeprecatedCaches
import com.pkb.kms.service.cache.KmsCacheManager
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.beans.factory.annotation.Autowired
import org.springframework.context.ApplicationListener
import org.springframework.context.event.ContextRefreshedEvent
import org.springframework.stereotype.Component
import java.lang.invoke.MethodHandles

@Component
class LegacyCacheRemovalListener(private val caches: List<HasDeprecatedCaches>)
    : ApplicationListener<ContextRefreshedEvent> {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @Autowired
    private lateinit var cacheManager: KmsCacheManager

    override fun onApplicationEvent(event: ContextRefreshedEvent) {
        // Fully delete the cache with the DEL command
        caches.flatMap { it.deprecatedCaches }
                .forEach { cacheName ->
                    cacheManager.deleteCache(cacheName).also {
                        log.info("Deleted $cacheName: $it")
                    }
                }
    }

}
