package com.pkb.patient;

import io.prometheus.client.Counter;
import org.jetbrains.annotations.Nullable;

import static org.apache.commons.lang3.StringUtils.defaultIfBlank;
import static org.apache.commons.lang3.StringUtils.trimToNull;

public class LetterInvitationMetrics {

    private static final String UNDEFINED_TEAM = "undefined_team";

    private static final Counter COUNTER = Counter.build()
            .name("letter_invitations")
            .help("Letter invitations")
            .labelNames("activity", "team")
            .register();

    private void inc(String activity, String teamCode) {
        COUNTER.labels(activity, defaultIfBlank(trimToNull(teamCode), UNDEFINED_TEAM).toLowerCase()).inc();
    }

    public void invitationGenerated(@Nullable String teamCode) {
        inc("generated", teamCode);
    }

    public void patientActivated(@Nullable String teamCode) {
        inc("activated", teamCode);
    }

    public void patientVerified(@Nullable String teamCode) {
        inc("verified", teamCode);
    }
}
