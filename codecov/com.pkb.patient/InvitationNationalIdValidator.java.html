package com.pkb.patient;

import com.pkb.entities.enums.NationalIdType;
import com.pkb.entities.enums.ValidNationalId;
import com.pkb.user.entity.NationalId;
import com.pkb.user.entity.PKBPerson;
import io.prometheus.client.Counter;
import org.jetbrains.annotations.NotNull;

import java.util.Optional;

public class InvitationNationalIdValidator {

    private static final Counter UPDATE_COUNTER = Counter.build()
            .name("pkb_phr_invitation_national_id_update_counter")
            .labelNames("type", "attempted")
            .help("Count of attempted updates of national identifier during invitation")
            .register();

    public boolean isTryingToUpdateNationalId(
            @NotNull PKBPerson existingPatient,
            @NotNull ValidNationalId nationalId) {
        return isTryingToUpdateNationalId(existingPatient, nationalId.value(), nationalId.type());
    }

    private boolean isTryingToUpdateNationalId(
            @NotNull PKBPerson existingPatient,
            @NotNull String newId,
            @NotNull NationalIdType idType) {

        Optional<NationalId> maybeExistingId = existingPatient.getNationalIdAnyTypeValidForSameCountryAsType(idType);

        if (maybeExistingId.isPresent()) {
            // If values do not match, this is an attempted update of an existing record
            boolean attemptedUpdate = !maybeExistingId.get().equalByValueAndType(newId, idType);
            UPDATE_COUNTER.labels("update", Boolean.toString(attemptedUpdate)).inc();
            return attemptedUpdate;
        }

        return false;
    }
}
