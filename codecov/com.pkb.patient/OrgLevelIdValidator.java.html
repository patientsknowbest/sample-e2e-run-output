package com.pkb.patient;

import com.pkb.VersionDetails;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.base.Error;
import com.pkb.institute.entity.Org;
import com.pkb.patient.invite.InvitationDto;
import com.pkb.user.entity.OrgLevelId;
import com.pkb.user.entity.OrgLevelIdType;
import io.vavr.control.Either;

import java.util.List;
import java.util.Optional;
import java.util.SortedSet;
import java.util.function.Function;

import static com.pkb.common.base.ErrorBuilder.error;
import static java.util.Collections.singletonList;
import static java.util.Optional.empty;

public class OrgLevelIdValidator {
    private static final String NO_ORG_LEVEL_ID_TYPE_FOUND_CODE = "patient.bulk.invite.no.org.level.id.type.found";
    private static final String NO_ORG_LEVEL_ID_TYPE_FOUND = "Org level id %s cannot be set because %s does not have any org level ID type";
    private static final String MORE_THAN_ONE_ORG_LEVEL_ID_TYPE_FOUND_CODE = "patient.bulk.invite.more.than.one.org.level.id.type.found";
    private static final String MORE_THAN_ONE_ORG_LEVEL_ID_TYPE_FOUND = "Org level id %s cannot be set because %s have more than one org level ID types";

    private final Org org;

    public OrgLevelIdValidator(Org org) {
        this.org = org;
    }

    public Either<List<Error>, Optional<OrgLevelId>> validate(InvitationDto invitation, EHRRequestContext requestContext) {
        return invitation.getInvitee().getInviteeIds().findOrgLevelId()
                .map(doValidate(requestContext))
                .orElse(Either.right(empty()));
    }

    private Function<String, Either<List<Error>, Optional<OrgLevelId>>> doValidate(EHRRequestContext requestContext) {
        return orgLevelIdValue -> {
            SortedSet<OrgLevelIdType> orgLevelIdTypes = org.getOrgLevelIdTypes();
            if (orgLevelIdTypes.size() == 1) {
                return Either.right(Optional.of(createOrgLevelId(orgLevelIdValue, orgLevelIdTypes.first(), requestContext)));
            } else if (orgLevelIdTypes.isEmpty()) {
                return Either.left(singletonList(
                        error(NO_ORG_LEVEL_ID_TYPE_FOUND_CODE, NO_ORG_LEVEL_ID_TYPE_FOUND).withArguments(
                                orgLevelIdValue,
                                org.getName()
                        ).build()));
            } else {
                return Either.left(singletonList(
                        error(MORE_THAN_ONE_ORG_LEVEL_ID_TYPE_FOUND_CODE, MORE_THAN_ONE_ORG_LEVEL_ID_TYPE_FOUND).withArguments(
                                orgLevelIdValue,
                                org.getName()
                        ).build()));
            }
        };
    }

    private OrgLevelId createOrgLevelId(String orgLevelIdValue, OrgLevelIdType orgLevelIdType, EHRRequestContext requestContext) {
        OrgLevelId orgLevelId = new OrgLevelId(VersionDetails.of(requestContext));
        orgLevelId.setValue(orgLevelIdValue);
        orgLevelId.setType(orgLevelIdType);
        orgLevelId.setOrg(org);
        return orgLevelId;
    }
}
