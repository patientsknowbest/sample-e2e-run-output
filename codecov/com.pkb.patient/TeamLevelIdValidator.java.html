package com.pkb.patient;

import com.pkb.VersionDetails;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.base.Error;
import com.pkb.institute.entity.Team;
import com.pkb.patient.invite.InvitationDto;
import com.pkb.user.entity.TeamLevelId;
import com.pkb.user.entity.TeamLevelIdType;
import io.vavr.control.Either;

import java.util.List;
import java.util.Optional;
import java.util.SortedSet;
import java.util.function.Function;

import static com.pkb.common.base.ErrorBuilder.error;
import static java.util.Collections.singletonList;
import static java.util.Optional.empty;

public class TeamLevelIdValidator {
    private static final String NO_TEAM_LEVEL_ID_TYPE_FOUND_CODE = "patient.bulk.invite.no.team.level.id.type.found";
    private static final String NO_TEAM_LEVEL_ID_TYPE_FOUND = "Team level id %s cannot be set because %s does not have any team level ID type";
    private static final String MORE_THAN_ONE_TEAM_LEVEL_ID_TYPE_FOUND_CODE = "patient.bulk.invite.more.than.one.team.level.id.type.found";
    private static final String MORE_THAN_ONE_TEAM_LEVEL_ID_TYPE_FOUND = "Team level id %s cannot be set because %s have more than one team level ID types";

    private final Team invitingTeam;

    public TeamLevelIdValidator(Team invitingTeam) {
        this.invitingTeam = invitingTeam;
    }

    public Either<List<Error>, Optional<TeamLevelId>> validate(InvitationDto invitation, EHRRequestContext requestContext) {
        return invitation.getInvitee().getInviteeIds().findTeamLevelId()
                .map(doValidate(invitation, requestContext))
                .orElse(Either.right(empty()));
    }

    private Function<String, Either<List<Error>, Optional<TeamLevelId>>> doValidate(InvitationDto invitation, EHRRequestContext requestContext) {
        return teamLevelIdValue -> {
            SortedSet<TeamLevelIdType> teamLevelIdTypes = invitingTeam.getTeamLevelIdTypes();
            if (teamLevelIdTypes.size() == 1) {
                return Either.right(Optional.of(createTeamLevelId(teamLevelIdValue, teamLevelIdTypes.first(), requestContext)));
            } else if (teamLevelIdTypes.isEmpty()) {
                return Either.left(singletonList(
                        error(NO_TEAM_LEVEL_ID_TYPE_FOUND_CODE, NO_TEAM_LEVEL_ID_TYPE_FOUND)
                                .withArguments(teamLevelIdValue, invitingTeam.getName())
                                .build()));
            } else {
                return Either.left(singletonList(
                        error(MORE_THAN_ONE_TEAM_LEVEL_ID_TYPE_FOUND_CODE, MORE_THAN_ONE_TEAM_LEVEL_ID_TYPE_FOUND)
                                .withArguments(teamLevelIdValue, invitingTeam.getName())
                                .build()));
            }
        };
    }

    private TeamLevelId createTeamLevelId(String teamLevelIdValue, TeamLevelIdType teamLevelIdType, EHRRequestContext requestContext) {
        TeamLevelId teamLevelId = new TeamLevelId(VersionDetails.of(requestContext));
        teamLevelId.setTeam(invitingTeam);
        teamLevelId.setValue(teamLevelIdValue);
        teamLevelId.setType(teamLevelIdType);
        return teamLevelId;
    }
}
