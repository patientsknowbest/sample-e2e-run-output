package com.pkb.questionnaireservice.config

import com.fasterxml.jackson.databind.ObjectMapper
import com.pkb.kms.client.core.Kms
import com.pkb.questionnaireservice.repositories.JpaKotlinRepositoryFactoryBean
import com.pkb.questionnaireservice.repositories.ques.EntityManagerResolver
import com.pkb.questionnaireservice.repositories.ques.QuestionnaireRepositoryCryptoHandler
import com.pkb.questionnaireservice.util.RequestIdUtils
import org.springframework.boot.autoconfigure.jdbc.DataSourceProperties
import org.springframework.boot.context.properties.ConfigurationProperties
import org.springframework.boot.orm.jpa.EntityManagerFactoryBuilder
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import org.springframework.context.annotation.Primary
import org.springframework.dao.annotation.PersistenceExceptionTranslationPostProcessor
import org.springframework.data.jpa.repository.config.EnableJpaRepositories
import org.springframework.orm.jpa.JpaTransactionManager
import org.springframework.orm.jpa.LocalContainerEntityManagerFactoryBean
import org.springframework.transaction.PlatformTransactionManager
import org.springframework.transaction.annotation.EnableTransactionManagement
import javax.persistence.EntityManagerFactory
import javax.sql.DataSource


@Configuration
@EnableTransactionManagement
@EnableJpaRepositories(
        basePackages = ["com.pkb.questionnaireservice.repositories.ques"],
        repositoryFactoryBeanClass = JpaKotlinRepositoryFactoryBean::class
)
class DatabaseConfig {

    @Bean
    @Primary
    @ConfigurationProperties("spring.datasource")
    fun dataSourceProperties(): DataSourceProperties = DataSourceProperties()

    @Primary
    @Bean
    fun dataSource(properties: DataSourceProperties): DataSource = properties.initializeDataSourceBuilder().build()

    @Primary
    @Bean
    fun entityManagerFactory(
            builder: EntityManagerFactoryBuilder,
            dataSource: DataSource
    ): LocalContainerEntityManagerFactoryBean =
            builder
                    .dataSource(dataSource)
                    .packages("com.pkb.questionnaireservice.entities")
                    .persistenceUnit("ques")
                    .build()

    @Primary
    @Bean
    fun transactionManager(entityManagerFactory: EntityManagerFactory
    ): PlatformTransactionManager = JpaTransactionManager(entityManagerFactory)


    @Bean
    fun cryptoHandler(kms: Kms, objectMapper: ObjectMapper, objectIdResolver: EntityManagerResolver, requestIdUtils: RequestIdUtils, questionnaireServiceProperties: QuestionnaireServiceProperties): QuestionnaireRepositoryCryptoHandler =
            QuestionnaireRepositoryCryptoHandler(kms, objectMapper, objectIdResolver, requestIdUtils, questionnaireServiceProperties)

    @Bean
    fun exceptionTranslation(): PersistenceExceptionTranslationPostProcessor =
            PersistenceExceptionTranslationPostProcessor()

    @Bean
    fun entityManagerIdResolver() = EntityManagerResolver()
}
