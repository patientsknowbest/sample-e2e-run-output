package com.pkb.questionnaireservice.config

import com.pkb.common.config.ConfigStorage
import com.pkb.common.config.ConfigurationPropertiesScopingPostProcessor
import com.pkb.common.config.QuestionnaireServiceConfig
import com.pkb.common.config.SpringBootConfigStorage
import com.pkb.common.datetime.DateTimeService
import com.pkb.common.datetime.DefaultDateTimeService
import com.pkb.common.datetime.FakeDateTimeService
import org.springframework.boot.context.properties.EnableConfigurationProperties
import org.springframework.cloud.context.scope.refresh.RefreshScope
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import org.springframework.core.annotation.Order
import org.springframework.core.env.ConfigurableEnvironment

@Configuration
@EnableConfigurationProperties(QuestionnaireServiceProperties::class)
class ConfigurationConfig {

    @Bean
    fun dateTimeService(config: QuestionnaireServiceConfig): DateTimeService {
        return if (config.isFakeDateTimeServiceEnabled) FakeDateTimeService() else DefaultDateTimeService()
    }

    @Bean
    fun configStorage(environment: ConfigurableEnvironment, refreshScope: RefreshScope): ConfigStorage {
        return SpringBootConfigStorage(environment, refreshScope)
    }

    @Bean
    fun config(configStorage: ConfigStorage): QuestionnaireServiceConfig {
       return QuestionnaireServiceConfig(configStorage)
    }

    companion object {
        @Bean
        @Order
        fun configurationPropertiesBeanScopeFixer() : ConfigurationPropertiesScopingPostProcessor = ConfigurationPropertiesScopingPostProcessor()
    }

}
