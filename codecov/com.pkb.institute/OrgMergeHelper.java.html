package com.pkb.institute;

import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.OrgSurvivingState;
import org.immutables.value.Value;

import java.util.Collections;
import java.util.List;
import java.util.Optional;
import java.util.Set;
import java.util.stream.Collectors;

import static org.springframework.util.CollectionUtils.isEmpty;

@Value.Immutable
public interface OrgMergeHelper {

    @Value.Parameter(order = 1)
    List<Org> getOrgs();

    @Value.Parameter(order = 2)
    String getOdsCode();

    @Value.Parameter(order = 3)
    Optional<OrgSurvivingState> getSurvivingState();

    @Value.Derived
    default Org getOrg() {
        if (isEmpty(getOrgs())) {
            throw new IllegalStateException("No org/orgs found with ODSCode: " + getOdsCode());
        }
        if (getOrgs().size() == 1) {
            return getOrgs().get(0);
        }
        if (getSurvivingState().isPresent()) {
            List<Org> survivors = getOrgs().stream()
                    .filter(org -> org.getId().equals(getSurvivingState().get().getOrgId()))
                    .collect(Collectors.toList());
            if(isEmpty(survivors)) {
                throw new IllegalStateException("There are no survivor org with ODSCode: " + getOdsCode());
            }
            if(survivors.size() > 1) {
                throw new IllegalStateException("There are multiple survivor orgs with ODSCode: " + getOdsCode() + " , orgId: " + getSurvivingState().get().getOrgId());
            }
            return survivors.get(0);
        }
        throw new IllegalStateException("There are no surviving state record with ODSCode: " + getOdsCode());
    }

    @Value.Derived
    default Set<String> getVictimOdsCodes() {
        if (getOrgs().size() == 1) {
            return Collections.emptySet();
        }
        return getOrgs().stream()
                .flatMap(org -> org.getOdsCodes()
                        .filter(ods -> !ods.equals(getOdsCode())))
                .collect(Collectors.toSet());
    }
}
