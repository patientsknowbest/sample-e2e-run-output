package com.pkb.app;

import com.pkb.app.entity.EHRData;
import com.pkb.crypto.decryptionhelper.dto.EncryptedDataWithSourceRecord;
import com.pkb.crypto.decryptionhelper.dto.ImmutableEncryptedDataWithSourceRecord;

import java.util.Optional;
import java.util.function.Function;

public class EHREncryptedDataDtoMapper implements Function<EHRData, EncryptedDataWithSourceRecord<EHRData>> {

    @Override
    public EncryptedDataWithSourceRecord<EHRData> apply(EHRData ehrData) {
        return ImmutableEncryptedDataWithSourceRecord.<EHRData>builder()
                .dataIdentifier("ehrdata:" + ehrData.getId())
                .encryptedData(ehrData.getEncryptedData())
                .secretKeyAlgorithm(Optional.ofNullable(ehrData.getSecretKeyAlgorithm()))
                .dESSecretKey(Optional.ofNullable(ehrData.getSecretKey()))
                .symmetricEncryptionKeyId(Optional.ofNullable(ehrData.getSymmetricEncryptionKeyId()))
                .nonceBase64(Optional.ofNullable(ehrData.getNonceBase64()))
                .sourceRecord(ehrData)
                .build();
    }
}
