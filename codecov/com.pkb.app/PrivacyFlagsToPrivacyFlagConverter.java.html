package com.pkb.app;

import com.pkb.entities.enums.PrivacyFlag;

import java.util.EnumSet;
import java.util.Optional;
import java.util.Set;
import java.util.stream.Stream;

public class PrivacyFlagsToPrivacyFlagConverter {

    public static Optional<PrivacyFlag> findPrivacyFlag(boolean general, boolean mental, boolean sexual, boolean social) {
        long flagSet = Stream.of(general,
                mental,
                sexual,
                social)
                .filter(c -> c)
                .count();
        if (flagSet > 1) {
            throw new IllegalStateException("More than one privacy flag is set");
        }

        if (general) {
            return Optional.of(PrivacyFlag.GENERAL);
        } else if (mental) {
            return Optional.of(PrivacyFlag.MENTAL_HEALTH);
        } else if (sexual) {
            return Optional.of(PrivacyFlag.SEXUAL_HEALTH);
        } else if (social) {
            return Optional.of(PrivacyFlag.SOCIAL_CARE);
        } else {
            return Optional.empty();
        }
    }

    public static Set<PrivacyFlag> getPrivacyFlags(boolean general, boolean mental, boolean sexual, boolean social) {
        Set<PrivacyFlag> result = EnumSet.noneOf(PrivacyFlag.class);
        if (general) {
            result.add(PrivacyFlag.GENERAL);
        }
        if (mental) {
            result.add(PrivacyFlag.MENTAL_HEALTH);
        }
        if (sexual) {
            result.add(PrivacyFlag.SEXUAL_HEALTH);
        }
        if (social) {
            result.add(PrivacyFlag.SOCIAL_CARE);
        }
        return result;
    }
}
