package com.pkb.encounter.workflow;

import static com.pkb.encounter.workflow.ArchivedStatus.ARCHIVED;

import org.jetbrains.annotations.Nullable;

public interface MessageWorkflowSupport {

    public default boolean isConversationArchived() {
        return getConversationWorkflow() != null && getConversationWorkflow().getArchivedStatus() == ARCHIVED;
    }

    public default boolean isConversationAssigned() {
        return getConversationWorkflow() != null && getConversationWorkflow().getAssignedPerson() != null;
    }

    @Nullable
    public abstract ConversationWorkflowDTO getConversationWorkflow();
}
