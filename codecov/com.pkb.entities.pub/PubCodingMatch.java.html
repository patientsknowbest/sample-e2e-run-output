package com.pkb.entities.pub;

import com.google.common.base.MoreObjects;
import com.pkb.entities.enums.CodingDataType;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;

@Entity
@Table(schema = "public", name = "codingmatch")
public class PubCodingMatch implements Serializable {

    private static final long serialVersionUID = 2775251973148171013L;

    @Id
    @GenericGenerator(name = "seq_codingmatch_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "public"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_codingmatch"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_codingmatch_gen")
    private Long id;

    @Column(name = "coding_data_type", nullable = false)
    @Enumerated(EnumType.STRING)
    private CodingDataType codingDataType;

    @Column(nullable = false)
    private String code;

    @Column(name = "code_system")
    private String codeSystem;

    @Column(name = "display_text")
    private String displayText;

    @Column(name = "privacy_flag")
    @Enumerated(EnumType.STRING)
    private PrivacyFlag privacyFlag;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "parent_id", referencedColumnName = "id")
    private PubCodingMatch parent;

    @Column(name = "created", insertable = false, updatable = false, nullable = false)
    private Instant created;

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .omitNullValues()
                .add("id", id)
                .add("codingDataType", codingDataType)
                .add("code", code)
                .add("codeSystem", codeSystem)
                .add("displayText", displayText)
                .add("privacyFlag", privacyFlag)
                .add("created", created)
                .toString();
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public CodingDataType getCodingDataType() {
        return codingDataType;
    }

    public void setCodingDataType(CodingDataType codingDataType) {
        this.codingDataType = codingDataType;
    }

    public String getCode() {
        return code;
    }

    public void setCode(String code) {
        this.code = code;
    }

    public String getCodeSystem() {
        return codeSystem;
    }

    public void setCodeSystem(String codeSystem) {
        this.codeSystem = codeSystem;
    }

    public String getDisplayText() {
        return displayText;
    }

    public void setDisplayText(String displayText) {
        this.displayText = displayText;
    }

    public PrivacyFlag getPrivacyFlag() {
        return privacyFlag;
    }

    public void setPrivacyFlag(PrivacyFlag privacyFlag) {
        this.privacyFlag = privacyFlag;
    }

    public PubCodingMatch getParent() {
        return parent;
    }

    public void setParent(PubCodingMatch parent) {
        this.parent = parent;
    }

    public Instant getCreated() {
        return created;
    }

    public void setCreated(Instant created) {
        this.created = created;
    }
}
