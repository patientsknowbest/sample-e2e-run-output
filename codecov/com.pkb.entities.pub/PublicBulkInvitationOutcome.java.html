package com.pkb.entities.pub;

import com.pkb.entities.enums.OperationOutcomeStatus;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.LocalDateTime;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;
import static javax.persistence.EnumType.STRING;

@Entity
@Table(schema = "public", name = "bulkinvitationoutcome")
public class PublicBulkInvitationOutcome implements Serializable {

    @Id
    @GenericGenerator(name = "seq_bulk_invitation_outcome_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "public"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_bulk_invitation_outcome"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.AUTO, generator = "seq_bulk_invitation_outcome_gen")
    private long id;

    @Column(nullable = false)
    private LocalDateTime uploadedAt;

    @Column
    private LocalDateTime completedAt;

    @Column(name = "uploadedby_id", nullable = false)
    private long uploadedById;

    @Column(name = "invitingteam_id", nullable = false)
    private long invitingTeamId;

    @Column
    private String csvEnrichedWithOutcomes;

    @Column(nullable = false)
    @Enumerated(STRING)
    private OperationOutcomeStatus status;

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public LocalDateTime getUploadedAt() {
        return uploadedAt;
    }

    public void setUploadedAt(LocalDateTime uploadedAt) {
        this.uploadedAt = uploadedAt;
    }

    public LocalDateTime getCompletedAt() {
        return completedAt;
    }

    public void setCompletedAt(LocalDateTime completedAt) {
        this.completedAt = completedAt;
    }

    public long getUploadedById() {
        return uploadedById;
    }

    public void setUploadedById(long uploadedById) {
        this.uploadedById = uploadedById;
    }

    public long getInvitingTeamId() {
        return invitingTeamId;
    }

    public void setInvitingTeamId(long invitingTeamId) {
        this.invitingTeamId = invitingTeamId;
    }

    public String getCsvEnrichedWithOutcomes() {
        return csvEnrichedWithOutcomes;
    }

    public void setCsvEnrichedWithOutcomes(String csvEnrichedWithOutcomes) {
        this.csvEnrichedWithOutcomes = csvEnrichedWithOutcomes;
    }

    public OperationOutcomeStatus getStatus() {
        return status;
    }

    public void setStatus(OperationOutcomeStatus status) {
        this.status = status;
    }
}
