package com.pkb.entities.pub;

import com.pkb.entities.enums.RegistrationMethod;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Parameter;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.Instant;
import java.util.UUID;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(schema = "public", name = "registration_event")
public class PublicRegistrationEvent {

    @Id
    @GenericGenerator(name = "registration_event_id_seq_gen", strategy = "enhanced-sequence",
            parameters = {
                    @Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @Parameter(
                            name = "schema",
                            value = "public"),
                    @Parameter(
                            name = "sequence_name",
                            value = "registration_event_id_seq"),
                    @Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "registration_event_id_seq_gen")
    private Long id;

    @Column(name = "person_id", nullable = false)
    private long personId;

    @Column(name = "registrationmethod", nullable = false)
    @Enumerated(EnumType.STRING)
    @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
    private RegistrationMethod registrationMethod;

    @Column(name = "registeredon", nullable = false)
    private Instant registeredOn;

    @SuppressWarnings("deprecation")
    @Deprecated // TODO remove in PHR-10242
    @Column(name = "invitation_id")
    private Long invitationPrivateId;

    @Column(name = "invitation_public_id")
    private UUID invitationPublicId;

    @Column(name = "letterinvitation_id")
    private Long letterInvitationId;

    @Column(name = "registeredon_confirmed")
    private boolean registeredOnConfirmed;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public long getPersonId() {
        return personId;
    }

    public void setPersonId(long personId) {
        this.personId = personId;
    }


    public RegistrationMethod getRegistrationMethod() {
        return registrationMethod;
    }

    public void setRegistrationMethod(RegistrationMethod registrationMethod) {
        this.registrationMethod = registrationMethod;
    }

    public Instant getRegisteredOn() {
        return registeredOn;
    }

    public void setRegisteredOn(Instant registeredOn) {
        this.registeredOn = registeredOn;
    }

    public UUID getInvitationPublicId() {
        return invitationPublicId;
    }

    public void setInvitationPublicId(UUID invitationPublicId) {
        this.invitationPublicId = invitationPublicId;
    }

    public Long getInvitationPrivateId() {
        return invitationPrivateId;
    }

    public void setInvitationPrivateId(Long invitationPrivateId) {
        this.invitationPrivateId = invitationPrivateId;
    }

    public Long getLetterInvitationId() {
        return letterInvitationId;
    }

    public void setLetterInvitationId(Long letterInvitationId) {
        this.letterInvitationId = letterInvitationId;
    }

    public boolean isRegisteredOnConfirmed() {
        return registeredOnConfirmed;
    }

    public void setRegisteredOnConfirmed(boolean registeredOnConfirmed) {
        this.registeredOnConfirmed = registeredOnConfirmed;
    }
}
