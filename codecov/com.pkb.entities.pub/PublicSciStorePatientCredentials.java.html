package com.pkb.entities.pub;

import org.hibernate.annotations.Cache;
import org.hibernate.annotations.CacheConcurrencyStrategy;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.persistence.UniqueConstraint;
import java.io.Serializable;

@Entity
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table(uniqueConstraints = { @UniqueConstraint(columnNames = { "pkbPatientId", "sciStorePatientId", "sciStoreSubscriptionId" }) },
        schema = "public", name = "scistorepatientcredentials")
public class PublicSciStorePatientCredentials implements Serializable {

    private static final long serialVersionUID = 1L;

    public static final String SUBSCR_ID_MISMATCHED = PublicSciStorePatientCredentials.Status.MISMATCHED.toString(); // need a placeholder; it's non-null
    public static final String CHI_NUMBER_UNKNOWN = PublicSciStorePatientCredentials.Status.CHI_NUMBER_UNKNOWN.toString();

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private long id;

    @Column(nullable = false)
    private long pkbPatientId;

    @Column(nullable = false)
    private String sciStorePatientId;

    @Column(nullable = false)
    private String sciStoreSubscriptionId;

    public enum Status {ACTIVE,INACTIVE,MISMATCHED,CHI_NUMBER_UNKNOWN}

    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    private Status status;


    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public long getPkbPatientId() {
        return pkbPatientId;
    }

    public void setPkbPatientId(long pkbPatientId) {
        this.pkbPatientId = pkbPatientId;
    }

    public String getSciStorePatientId() {
        return sciStorePatientId;
    }

    public void setSciStorePatientId(String sciStorePatientId) {
        this.sciStorePatientId = sciStorePatientId;
    }

    public String getSciStoreSubscriptionId() {
        return sciStoreSubscriptionId;
    }

    public void setSciStoreSubscriptionId(String sciStoreSubscriptionId) {
        this.sciStoreSubscriptionId = sciStoreSubscriptionId;
    }

    public Status getStatus() { return status; }

    public void setStatus(Status status) { this.status = status; }
}
