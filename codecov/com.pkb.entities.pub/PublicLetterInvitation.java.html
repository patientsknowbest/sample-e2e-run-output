package com.pkb.entities.pub;

import com.pkb.entities.embeddables.EncodedEncryptedPrivateKeyInfoWithSecretDeriv;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Parameter;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.persistence.Transient;
import java.io.Serializable;
import java.time.Instant;
import java.util.Objects;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(schema = "public", name = "letterinvitation")
public class PublicLetterInvitation implements Serializable {

    private static final long serialVersionUID = -1058040017115276995L;

    public static final int VERIFICATION_CODE_LENGTH = 36;
    private static final int CONTACT_VALUE_LENGTH = 255;
    private static final int CODE_LENGTH = 64;

    @Id
    @GenericGenerator(name = "letterinvitation_id_seq_gen", strategy = "enhanced-sequence",
            parameters = {
                    @Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @Parameter(
                            name = "schema",
                            value = "public"),
                    @Parameter(
                            name = "sequence_name",
                            value = "letterinvitation_id_seq"),
                    @Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(
            strategy = GenerationType.SEQUENCE,
            generator = "letterinvitation_id_seq_gen")
    private Long id;

    @Column(length = CODE_LENGTH, nullable = false, unique = true)
    private String code;

    @Column(nullable = false)
    private Instant expiration;

    @Column(nullable = false)
    private Instant created;

    @Column(name = "invitee_id", nullable = false)
    private Long inviteeId;

    @Column(nullable = false)
    private boolean invalidated;

    @Column(length = VERIFICATION_CODE_LENGTH, unique = true)
    private String verificationCode;

    @Column(length = CONTACT_VALUE_LENGTH)
    private String claimedEmail;

    @Column(name = "author_id")
    private Long authorId;

    @Column(name = "team_id")
    private Long teamId;

    @Embedded
    private EncodedEncryptedPrivateKeyInfoWithSecretDeriv accountPrivateKey;

    @Transient
    private Status inviteeStatus;

    public enum Status {
        NOT_INVITED_NOT_REGISTERED, INVITED_NOT_REGISTERED
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getCode() {
        return code;
    }

    public void setCode(String code) {
        this.code = code;
    }

    public Instant getExpiration() {
        return expiration;
    }

    public void setExpiration(Instant expiration) {
        this.expiration = expiration;
    }

    public Instant getCreated() {
        return created;
    }

    public void setCreated(Instant created) {
        this.created = created;
    }

    public Long getInviteeId() {
        return inviteeId;
    }

    public void setInviteeId(Long inviteeId) {
        this.inviteeId = inviteeId;
    }

    public boolean getInvalidated() {
        return invalidated;
    }

    public void setInvalidated(boolean invalidated) {
        this.invalidated = invalidated;
    }

    public String getVerificationCode() {
        return verificationCode;
    }

    public void setVerificationCode(String verificationCode) {
        this.verificationCode = verificationCode;
    }

    public String getClaimedEmail() {
        return claimedEmail;
    }

    public void setClaimedEmail(String claimedEmail) {
        this.claimedEmail = claimedEmail;
    }

    public Long getAuthorId() {
        return authorId;
    }

    public void setAuthorId(Long authorId) {
        this.authorId = authorId;
    }

    public Long getTeamId() {
        return teamId;
    }

    public void setTeamId(Long teamId) {
        this.teamId = teamId;
    }

    public EncodedEncryptedPrivateKeyInfoWithSecretDeriv getAccountPrivateKey() {
        return accountPrivateKey;
    }

    public void setAccountPrivateKey(EncodedEncryptedPrivateKeyInfoWithSecretDeriv accountPrivateKey) {
        this.accountPrivateKey = accountPrivateKey;
    }

    public Status getInviteeStatus() {
        return inviteeStatus;
    }

    public void setInviteeStatus(Status inviteeStatus) {
        this.inviteeStatus = inviteeStatus;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (!(o instanceof PublicLetterInvitation)) {
            return false;
        }
        PublicLetterInvitation that = (PublicLetterInvitation) o;
        return Objects.equals(id, that.id) &&
                Objects.equals(code, that.code) &&
                Objects.equals(expiration, that.expiration) &&
                Objects.equals(created, that.created) &&
                Objects.equals(inviteeId, that.inviteeId) &&
                Objects.equals(invalidated, that.invalidated) &&
                Objects.equals(verificationCode, that.verificationCode) &&
                Objects.equals(claimedEmail, that.claimedEmail) &&
                Objects.equals(authorId, that.authorId) &&
                Objects.equals(teamId, that.teamId) &&
                Objects.equals(accountPrivateKey, that.accountPrivateKey);
    }

    @Override
    public int hashCode() {
        return Objects.hash(id, code, expiration, created, inviteeId, invalidated, verificationCode, claimedEmail, authorId, teamId, accountPrivateKey);
    }
}
