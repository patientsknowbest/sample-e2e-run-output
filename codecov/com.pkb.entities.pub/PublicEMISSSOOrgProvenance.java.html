package com.pkb.entities.pub;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;
import java.util.Objects;

@Entity
@Table(schema = "public", name = "emisssoorgprovenance")
public class PublicEMISSSOOrgProvenance implements Serializable {

    private static final long serialVersionUID = -8497737647476933368L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private long id;

    @Column
    private String identifiers; //comma separated identifiers (these are for audit purpose only ; unuseful within PKB)

    @Column
    private String orgName;

    public PublicEMISSSOOrgProvenance() {
    }

    public PublicEMISSSOOrgProvenance(String identifiers, String orgName) {
        this.identifiers = identifiers;
        this.orgName = orgName;
    }

    public long getId() {
        return id;
    }

    public String getIdentifiers() {
        return identifiers;
    }

    public String getOrgName() {
        return orgName;
    }

    @Override
    public int hashCode() {
        int prime = 31;
        int result = 1;
        result = prime * result + Objects.hash(id);
        result = prime * result + ((identifiers == null) ? 0 : identifiers.hashCode());
        result = prime * result + ((orgName == null) ? 0 : orgName.hashCode());
        return result;
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj) {
            return true;
        }
        if (obj == null) {
            return false;
        }
        if (getClass() != obj.getClass()) {
            return false;
        }
        PublicEMISSSOOrgProvenance other = (PublicEMISSSOOrgProvenance) obj;
        if (identifiers == null) {
            if (other.identifiers != null) {
                return false;
            }
        } else if (!identifiers.equals(other.identifiers)) {
            return false;
        }
        if (orgName == null) {
            if (other.orgName != null) {
                return false;
            }
        } else if (!orgName.equals(other.orgName)) {
            return false;
        }
        return id == other.id;
    }
}
