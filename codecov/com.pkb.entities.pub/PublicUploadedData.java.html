package com.pkb.entities.pub;

import com.pkb.entities.enums.UploadedDataDestination;
import com.pkb.entities.enums.UploadedDataFormat;
import com.pkb.entities.enums.UploadedDataStatus;
import org.hibernate.annotations.DynamicInsert;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.time.Instant;
import java.util.UUID;

@Entity
@DynamicInsert
@Table(schema = "public", name = "uploadeddata")
public class PublicUploadedData {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column
    @Enumerated(EnumType.STRING)
    @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
    private UploadedDataDestination destination;

    @Column
    @Enumerated(EnumType.STRING)
    @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
    private UploadedDataFormat format;

    @Column(name = "embeddedid")
    private String embeddedId;

    @Column(name = "externalmessageid")
    private String externalMessageId;

    @Type(type = "pg-uuid")
    @Column(name = "document_metadata_id")
    private UUID documentMetadataId;

    @Column(name = "status")
    @Enumerated(EnumType.STRING)
    @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
    private UploadedDataStatus status;

    @Column
    private Instant created;

    @Column
    private Instant edited;

    @Column
    private Long updateTo;

    @Column(name = "destinationspecificdata")
    private String destinationSpecificData;

    @Column(name = "patientinfomismatchxml")
    private String patientInfoMismatchXml;

    @ManyToOne
    @JoinColumn(name = "sourceorg_id", referencedColumnName = "id")
    private PublicOrg sourceOrg;

    @ManyToOne
    @JoinColumn(name = "sourceteam_id", referencedColumnName = "id")
    private PublicTeam sourceTeam;

    @Column(name = "person_id")
    private Long personId;

    @ManyToOne
    @JoinColumn(name = "subsetofparentid", referencedColumnName = "id")
    private PublicUploadedData parent;

    @Column(name = "responsibleorganisationidentifier")
    private String responsibleOrganisationIdentifier;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public UploadedDataDestination getDestination() {
        return destination;
    }

    public void setDestination(UploadedDataDestination destination) {
        this.destination = destination;
    }

    public UploadedDataFormat getFormat() {
        return format;
    }

    public void setFormat(UploadedDataFormat format) {
        this.format = format;
    }

    public String getEmbeddedId() {
        return embeddedId;
    }

    public void setEmbeddedId(String embeddedId) {
        this.embeddedId = embeddedId;
    }

    public String getExternalMessageId() {
        return externalMessageId;
    }

    public void setExternalMessageId(String externalMessageId) {
        this.externalMessageId = externalMessageId;
    }

    public UUID getDocumentMetadataId() {
        return documentMetadataId;
    }

    public void setDocumentMetadataId(UUID documentMetadataId) {
        this.documentMetadataId = documentMetadataId;
    }

    public UploadedDataStatus getStatus() {
        return status;
    }

    public void setStatus(UploadedDataStatus status) {
        this.status = status;
    }

    public Instant getCreated() {
        return created;
    }

    public void setCreated(Instant created) {
        this.created = created;
    }

    public Instant getEdited() {
        return edited;
    }

    public void setEdited(Instant edited) {
        this.edited = edited;
    }

    public Long getUpdateTo() {
        return updateTo;
    }

    public void setUpdateTo(Long updateTo) {
        this.updateTo = updateTo;
    }

    public String getDestinationSpecificData() {
        return destinationSpecificData;
    }

    public void setDestinationSpecificData(String destinationSpecificData) {
        this.destinationSpecificData = destinationSpecificData;
    }

    public String getPatientInfoMismatchXml() {
        return patientInfoMismatchXml;
    }

    public void setPatientInfoMismatchXml(String patientInfoMismatchXml) {
        this.patientInfoMismatchXml = patientInfoMismatchXml;
    }

    public PublicOrg getSourceOrg() {
        return sourceOrg;
    }

    public void setSourceOrg(PublicOrg sourceOrg) {
        this.sourceOrg = sourceOrg;
    }

    public PublicTeam getSourceTeam() {
        return sourceTeam;
    }

    public void setSourceTeam(PublicTeam sourceTeam) {
        this.sourceTeam = sourceTeam;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }

    public PublicUploadedData getParent() {
        return parent;
    }

    public void setParent(PublicUploadedData parent) {
        this.parent = parent;
    }

    public String getResponsibleOrganisationIdentifier() {
        return responsibleOrganisationIdentifier;
    }

    public void setResponsibleOrganisationIdentifier(String responsibleOrganizationIdentifier) {
        this.responsibleOrganisationIdentifier = responsibleOrganizationIdentifier;
    }
}
