package com.pkb.entities.pub;

import com.pkb.entities.enums.PropertyName;
import com.pkb.entities.pub.converter.PropertyNameConverter;

import javax.persistence.Column;
import javax.persistence.Convert;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.OneToOne;
import javax.persistence.Table;

@Entity
@Table(schema = "public", name = "pkbpersonproperty")
public class PublicPKBPersonProperty {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    @Column(name = "id")
    private Long id;

    @Column(name = "propertyname", nullable = true)
    @Convert(converter = PropertyNameConverter.class)
    private PropertyName propertyName;

    @Column(name = "propertyvalue", nullable = true)
    private String propertyValue;

    @OneToOne
    @JoinColumn(name = "person_id", referencedColumnName = "id", nullable = true)
    private PublicPKBPerson person;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public PropertyName getPropertyName() {
        return propertyName;
    }

    public void setPropertyName(PropertyName propertyName) {
        this.propertyName = propertyName;
    }

    public String getPropertyValue() {
        return propertyValue;
    }

    public void setPropertyValue(String propertyValue) {
        this.propertyValue = propertyValue;
    }

    public PublicPKBPerson getPerson() {
        return person;
    }

    public void setPerson(PublicPKBPerson person) {
        this.person = person;
    }
}
