package com.pkb.entities.pub;

import com.pkb.entities.ids.ConversationWorkflowId;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.IdClass;
import javax.persistence.Table;
import javax.persistence.Version;
import java.time.Instant;
import java.util.UUID;

@Entity
@IdClass(ConversationWorkflowId.class)
@Table(name = "conversation_workflow", schema = "public")
public class PublicConversationWorkflow {

    @Id
    @Column(name = "conversation_id", nullable = false)
    @Type(type="pg-uuid")
    private UUID conversationId;

    @Id
    @Column(name = "team_id", nullable = false)
    private long teamId;

    @Column(name = "optimistic_lock_version", nullable = false)
    @Version
    private Integer optimisticLockVersion;

    @Column(name = "archived", nullable = false)
    private boolean archived;

    @Column(name = "archive_change_date")
    private Instant archiveChangeDate;

    @Column(name = "assigned_person_id")
    private Long assignedPersonId;

    @Column(name = "assign_change_date")
    private Instant assignChangeDate;

    public UUID getConversationId() {
        return conversationId;
    }

    public void setConversationId(UUID conversationId) {
        this.conversationId = conversationId;
    }

    public long getTeamId() {
        return teamId;
    }

    public void setTeamId(long teamId) {
        this.teamId = teamId;
    }

    public boolean isArchived() {
        return archived;
    }

    public void setArchived(boolean archived) {
        this.archived = archived;
    }

    public Integer getOptimisticLockVersion() {
        return optimisticLockVersion;
    }

    public void setOptimisticLockVersion(Integer optimisticLockVersion) {
        this.optimisticLockVersion = optimisticLockVersion;
    }

    public Instant getArchiveChangeDate() {
        return archiveChangeDate;
    }

    public void setArchiveChangeDate(Instant archiveChangeDate) {
        this.archiveChangeDate = archiveChangeDate;
    }

    public Long getAssignedPersonId() {
        return assignedPersonId;
    }

    public void setAssignedPersonId(Long assignedPersonId) {
        this.assignedPersonId = assignedPersonId;
    }

    public Instant getAssignChangeDate() {
        return assignChangeDate;
    }

    public void setAssignChangeDate(Instant assignChangeDate) {
        this.assignChangeDate = assignChangeDate;
    }
}
