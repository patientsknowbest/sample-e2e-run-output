package com.pkb.entities.pub;

import com.pkb.entities.enums.AdditionalConsentId;
import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;

@Entity
@Table(schema = "public", name = "patientadditionalconsent")
public class PublicPatientAdditionalConsent {

    @Id
    @GenericGenerator(name = "seq_pc_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "public"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_patientconsent"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_pc_gen")
    @Column(name = "id")
    private Long id;

    @Column(name = "additionalconsent", nullable = false)
    @Enumerated(EnumType.STRING)
    private AdditionalConsentId additionalConsent;

    @Column(name = "patientid", nullable = false)
    private long patientId;

    @Column(name = "teamid", nullable = false)
    private long teamId;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public AdditionalConsentId getAdditionalConsent() {
        return additionalConsent;
    }

    public void setAdditionalConsent(AdditionalConsentId additionalConsent) {
        this.additionalConsent = additionalConsent;
    }

    public long getPatientId() {
        return patientId;
    }

    public void setPatientId(long patientId) {
        this.patientId = patientId;
    }

    public long getTeamId() {
        return teamId;
    }

    public void setTeamId(long teamId) {
        this.teamId = teamId;
    }
}
