package com.pkb.entities.pub;

import com.pkb.entities.embeddables.FhirStoreSyncDetails;
import com.pkb.entities.enums.FhirStoreSyncStatus;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.Id;
import javax.persistence.Table;

/**
 * An archival table that represents the history of completed fhir_store_sync rows. See {@link PublicFhirStoreSync})
 * These are deleted from the main fhir_store_sync table on completion, success or error, to stop that
 * table from bloating too far, as we expect relatively heavy usage.
 */
@Entity
@Table(schema = "public", name = "fhir_store_sync_completed")
public class PublicFhirStoreSyncCompleted {

    //This is copied over from PublicFhirStoreSync, so we do not have a sequence generator for it.
    // We expose it into the hibernate for equality checks, but it isn't currently used by the app.
    @Id
    @SuppressWarnings("SSBasedInspection")
    private long id;

    @Embedded
    private FhirStoreSyncDetails syncDetails;

    @Column(name = "sync_status", nullable = false)
    @Enumerated(EnumType.STRING)
    private FhirStoreSyncStatus syncStatus;

    @Column(name = "details")
    private String details;

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public FhirStoreSyncDetails getSyncDetails() {
        return syncDetails;
    }

    public void setSyncDetails(FhirStoreSyncDetails syncDetails) {
        this.syncDetails = syncDetails;
    }

    public FhirStoreSyncStatus getSyncStatus() {
        return syncStatus;
    }

    public void setSyncStatus(FhirStoreSyncStatus syncStatus) {
        this.syncStatus = syncStatus;
    }

    public String getDetails() {
        return details;
    }

    public void setDetails(String details) {
        this.details = details;
    }

    @Override
    public int hashCode() {
        return getClass().hashCode();
    }

    @Override
    public boolean equals(Object o) {
        if (o instanceof PublicFhirStoreSyncCompleted) {
            return getId() == ((PublicFhirStoreSyncCompleted)o).getId();
        }
        return false;
    }
}
