package com.pkb.entities.pub;

import com.pkb.entities.embeddables.VersionDetails;
import com.pkb.entities.enums.CodingDataType;
import com.pkb.entities.enums.PrivacyFlag;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.Instant;

@Entity
@Table(name = "coding_match_import_new_records_temp_table")
public class CodingMatchImportNewRecordsTempTable {

    @Id
    private Long id;

    @Column(name = "coding_data_type", nullable = false)
    @Enumerated(EnumType.STRING)
    private CodingDataType codingDataType;

    @Column(nullable = false)
    private String code;

    @Column(name = "code_system")
    private String codeSystem;

    @Column(name = "display_text")
    private String displayText;

    @Column(nullable = false)
    private Instant created;

    @Column(nullable = false)
    private boolean exclude = false;

    @Column(name = "privacy_flag")
    @Enumerated(EnumType.STRING)
    private PrivacyFlag privacyFlag;

    @Column(name = "parent_id")
    private Long parentId;

    @Column(name = "validated")
    private boolean validated;

    @Embedded
    private VersionDetails versionDetails;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public CodingDataType getCodingDataType() {
        return codingDataType;
    }

    public void setCodingDataType(CodingDataType codingDataType) {
        this.codingDataType = codingDataType;
    }

    public String getCode() {
        return code;
    }

    public void setCode(String code) {
        this.code = code;
    }

    public String getCodeSystem() {
        return codeSystem;
    }

    public void setCodeSystem(String codeSystem) {
        this.codeSystem = codeSystem;
    }

    public String getDisplayText() {
        return displayText;
    }

    public void setDisplayText(String displayText) {
        this.displayText = displayText;
    }

    public boolean isExclude() {
        return exclude;
    }

    public void setExclude(boolean exclude) {
        this.exclude = exclude;
    }

    public PrivacyFlag getPrivacyFlag() {
        return privacyFlag;
    }

    public void setPrivacyFlag(PrivacyFlag privacyFlag) {
        this.privacyFlag = privacyFlag;
    }

    public Long getParentId() {
        return parentId;
    }

    public void setParentId(Long parentId) {
        this.parentId = parentId;
    }

    public VersionDetails getVersionDetails() {
        return versionDetails;
    }

    public void setVersionDetails(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }

    public boolean isValidated() {
        return validated;
    }

    public void setValidated(boolean validated) {
        this.validated = validated;
    }
}
