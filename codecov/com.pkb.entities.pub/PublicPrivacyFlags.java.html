package com.pkb.entities.pub;

import com.google.common.base.MoreObjects;
import com.pkb.entities.enums.Route;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Parameter;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;
import java.util.UUID;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(schema = "public", name = "privacy_flags")
public class PublicPrivacyFlags implements Serializable {

    private static final long serialVersionUID = -112171037981877383L;

    @Id
    @GenericGenerator(name = "privacy_flags_id_seq_gen", strategy = "enhanced-sequence",
            parameters = {
            @Parameter(
                    name = "optimizer",
                    value = OPTIMIZER_CANONICAL_NAME
            ), // based on "pooled-lo"
            @Parameter(
                    name = "schema",
                    value = "public"
            ),
            @Parameter(
                    name = "sequence_name",
                    value = "privacy_flags_id_seq"
            ),
            @Parameter(
                    name = "increment_size",
                    value = "50"
            )
    })
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "privacy_flags_id_seq_gen")
    private long id;

    @Column(nullable = false)
    @Type(type="pg-uuid")
    private UUID uniqueId;

    @Column(name = "change_date", nullable = false)
    private Instant changeDate;

    @Column(name="require_general", nullable = false)
    private boolean requireGeneral;

    @Column(name="require_sexual_health", nullable = false)
    private boolean requireSexualHealth;

    @Column(name="require_mental_health", nullable = false)
    private boolean requireMentalHealth;

    @Column(name="require_social_care", nullable = false)
    private boolean requireSocialCare;

    @Column(name = "source_person_id")
    private Long sourcePersonId;

    @Column(name = "source_team_id")
    private Long sourceTeamId;

    @Column(name = "source_org_id")
    private Long sourceOrgId;

    @Column(name = "source_text")
    private String sourceText;

    @Column(name = "access_route", nullable = false)
    @Enumerated(EnumType.STRING)
    private Route accessRoute;

    @Column(nullable = false)
    private boolean override;

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("requireGeneral", requireGeneral)
                .add("requireSexualHealth", requireSexualHealth)
                .add("requireMentalHealth", requireMentalHealth)
                .add("requireSocialCare", requireSocialCare)
                .toString();
    }

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public UUID getUniqueId() {
        return uniqueId;
    }

    public void setUniqueId(UUID uniqueId) {
        this.uniqueId = uniqueId;
    }

    public Instant getChangeDate() {
        return changeDate;
    }

    public void setChangeDate(Instant changeDate) {
        this.changeDate = changeDate;
    }

    public boolean isRequireGeneral() {
        return requireGeneral;
    }

    public void setRequireGeneral(boolean requireGeneral) {
        this.requireGeneral = requireGeneral;
    }

    public boolean isRequireMentalHealth() {
        return requireMentalHealth;
    }

    public void setRequireMentalHealth(boolean requireMentalHealth) {
        this.requireMentalHealth = requireMentalHealth;
    }

    public boolean isRequireSexualHealth() {
        return requireSexualHealth;
    }

    public void setRequireSexualHealth(boolean requireSexualHealth) {
        this.requireSexualHealth = requireSexualHealth;
    }

    public boolean isRequireSocialCare() {
        return requireSocialCare;
    }

    public void setRequireSocialCare(boolean requireSocialCare) {
        this.requireSocialCare = requireSocialCare;
    }

    public Long getSourcePersonId() {
        return sourcePersonId;
    }

    public void setSourcePersonId(Long sourcePersonId) {
        this.sourcePersonId = sourcePersonId;
    }

    public Long getSourceTeamId() {
        return sourceTeamId;
    }

    public void setSourceTeamId(Long sourceTeamId) {
        this.sourceTeamId = sourceTeamId;
    }

    public Long getSourceOrgId() {
        return sourceOrgId;
    }

    public void setSourceOrgId(Long sourceOrgId) {
        this.sourceOrgId = sourceOrgId;
    }

    public String getSourceText() {
        return sourceText;
    }

    public void setSourceText(String sourceText) {
        this.sourceText = sourceText;
    }

    public Route getAccessRoute() {
        return accessRoute;
    }

    public void setAccessRoute(Route accessRoute) {
        this.accessRoute = accessRoute;
    }

    public boolean isOverride() {
        return override;
    }

    public void setOverride(boolean override) {
        this.override = override;
    }
}
