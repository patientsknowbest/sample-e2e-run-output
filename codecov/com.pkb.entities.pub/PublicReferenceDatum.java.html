package com.pkb.entities.pub;

import com.pkb.entities.enums.ReferenceDataType;
import org.hibernate.annotations.Cache;
import org.hibernate.annotations.CacheConcurrencyStrategy;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.io.Serializable;
import java.util.UUID;

@Entity
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table(schema = "public", name = "referencedatum")
public class PublicReferenceDatum implements Serializable  {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private long id;

    //Changing the org is disallowed. Delete and recreate the ReferenceDatum if this is wrong.
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "org_id", referencedColumnName = "id", nullable = false, updatable = false)
    private PublicOrg org;

    //Changing this type is disallowed. Delete and recreate the ReferenceDatum if this is wrong.
    @Column(nullable = false, updatable = false)
    @Enumerated(EnumType.STRING)
    private ReferenceDataType dataType;

    @Column(nullable = false)
    private String dataId;

    @Column(nullable = false)
    private String dataText;

    @Column(name = "public_id", nullable = false, updatable = false, unique = true)
    @Type(type = "pg-uuid")
    private UUID publicId = UUID.randomUUID();

    @Column
    private boolean generalHealthConsentRequired;

    @Column
    private boolean mentalHealthConsentRequired;

    @Column
    private boolean sexualHealthConsentRequired;

    @Column
    private boolean socialCareConsentRequired;

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public PublicOrg getOrg() {
        return org;
    }

    public void setOrg(PublicOrg org) {
        this.org = org;
    }

    public ReferenceDataType getDataType() {
        return dataType;
    }

    public void setDataType(ReferenceDataType dataType) {
        this.dataType = dataType;
    }

    public String getDataId() {
        return dataId;
    }

    public void setDataId(String dataId) {
        this.dataId = dataId;
    }

    public String getDataText() {
        return dataText;
    }

    public void setDataText(String dataText) {
        this.dataText = dataText;
    }

    public boolean getGeneralHealthConsentRequired() {
        return generalHealthConsentRequired;
    }

    public void setGeneralHealthConsentRequired(boolean generalHealthConsentRequired) {
        this.generalHealthConsentRequired = generalHealthConsentRequired;
    }

    public boolean getMentalHealthConsentRequired() {
        return mentalHealthConsentRequired;
    }

    public void setMentalHealthConsentRequired(boolean mentalHealthConsentRequired) {
        this.mentalHealthConsentRequired = mentalHealthConsentRequired;
    }

    public boolean getSexualHealthConsentRequired() {
        return sexualHealthConsentRequired;
    }

    public void setSexualHealthConsentRequired(boolean sexualHealthConsentRequired) {
        this.sexualHealthConsentRequired = sexualHealthConsentRequired;
    }

    public boolean getSocialCareConsentRequired() {
        return socialCareConsentRequired;
    }

    public void setSocialCareConsentRequired(boolean socialCareConsentRequired) {
        this.socialCareConsentRequired = socialCareConsentRequired;
    }

    public UUID getPublicId() {
        return publicId;
    }

    public void setPublicId(UUID publicId) {
        this.publicId = publicId;
    }
}
