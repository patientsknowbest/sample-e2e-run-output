package com.pkb.entities.pub;

import com.pkb.entities.enums.TemplateStatus;

import javax.persistence.CascadeType;
import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.OneToMany;
import javax.persistence.Table;
import java.time.Instant;
import java.util.List;

import static javax.persistence.FetchType.LAZY;

@Entity
@Table(schema = "public", name = "phplantemplate")
public class PublicPHPlanTemplate {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column(name = "name")
    private String name;

    @Column(name = "creationdate", nullable = false)
    private Instant creationDate;

    @Column(name = "status", nullable = false)
    @Enumerated(EnumType.STRING)
    private TemplateStatus status;

    @Column(name = "creator_id")
    private Long creatorId;

    @Column(name = "institute_id", nullable = false)
    private Long instituteId;

    @Column(name = "reviewperioddays")
    private Long reviewPeriodDays;

    @Column(name = "requiregeneral")
    private boolean requireGeneral;

    @Column(name = "requiresexualhealth")
    private boolean requireSexualHealth;

    @Column(name = "requirementalhealth")
    private boolean requireMentalHealth;

    @Column(name = "requiresocialcare")
    private boolean requireSocialCare;

    @OneToMany(mappedBy = "planTemplate", fetch = LAZY, cascade = CascadeType.ALL)
    private List<PublicTemplateMetric> metrics;

    @OneToMany(mappedBy = "planTemplate", fetch = LAZY, cascade = CascadeType.ALL)
    private List<PublicTemplateField> fields;

    @OneToMany(mappedBy = "planTemplate", fetch = LAZY, cascade = CascadeType.ALL)
    private List<PublicTemplateAttachment> attachments;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public Instant getCreationDate() {
        return creationDate;
    }

    public void setCreationDate(Instant creationDate) {
        this.creationDate = creationDate;
    }

    public TemplateStatus getStatus() {
        return status;
    }

    public void setStatus(TemplateStatus status) {
        this.status = status;
    }

    public Long getCreatorId() {
        return creatorId;
    }

    public void setCreatorId(Long creatorId) {
        this.creatorId = creatorId;
    }

    public long getInstituteId() {
        return instituteId;
    }

    public void setInstituteId(long instituteId) {
        this.instituteId = instituteId;
    }

    public Long getReviewPeriodDays() {
        return reviewPeriodDays;
    }

    public void setReviewPeriodDays(Long reviewPeriodDays) {
        this.reviewPeriodDays = reviewPeriodDays;
    }

    public boolean isRequireGeneral() {
        return requireGeneral;
    }

    public void setRequireGeneral(boolean requireGeneral) {
        this.requireGeneral = requireGeneral;
    }

    public boolean isRequireSexualHealth() {
        return requireSexualHealth;
    }

    public void setRequireSexualHealth(boolean requireSexualHealth) {
        this.requireSexualHealth = requireSexualHealth;
    }

    public boolean isRequireMentalHealth() {
        return requireMentalHealth;
    }

    public void setRequireMentalHealth(boolean requireMentalHealth) {
        this.requireMentalHealth = requireMentalHealth;
    }

    public boolean isRequireSocialCare() {
        return requireSocialCare;
    }

    public void setRequireSocialCare(boolean requireSocialCare) {
        this.requireSocialCare = requireSocialCare;
    }

    public List<PublicTemplateMetric> getMetrics() {
        return metrics;
    }

    public void setMetrics(List<PublicTemplateMetric> metrics) {
        this.metrics = metrics;
    }

    public List<PublicTemplateField> getFields() {
        return fields;
    }

    public void setFields(List<PublicTemplateField> fields) {
        this.fields = fields;
    }

    public List<PublicTemplateAttachment> getAttachments() {
        return attachments;
    }

    public void setAttachments(List<PublicTemplateAttachment> attachments) {
        this.attachments = attachments;
    }
}
