package com.pkb.entities.pub;

import com.pkb.entities.enums.OrgExternalIdType;
import org.hibernate.annotations.Cache;
import org.hibernate.annotations.CacheConcurrencyStrategy;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;

@Entity
@Table(schema = "public", name = "orgexternalid")
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
public class PublicOrgExternalId {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    @SuppressWarnings("SSBasedInspection")
    private long id;

    @Column(name = "org_id")
    private long orgId;

    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    private OrgExternalIdType idType;

    @Column
    private String idValue;

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public long getOrgId() {
        return orgId;
    }

    public void setOrgId(long orgId) {
        this.orgId = orgId;
    }

    public OrgExternalIdType getIdType() {
        return idType;
    }

    public void setIdType(OrgExternalIdType idType) {
        this.idType = idType;
    }

    public String getIdValue() {
        return idValue;
    }

    public void setIdValue(String idValue) {
        this.idValue = idValue;
    }
}
