    private String skypeId;

    @Column

    private NotificationOption notificationOption;

    @Column
    private Instant lastEmailSent;

    @Column(name = "public_id", nullable = false, updatable = false, unique = true)
    @Type(type = "pg-uuid")
    private UUID publicId;

    @Embedded
    private VersionDetails versionDetails = new VersionDetails();

    @OneToOne(fetch = FetchType.EAGER, cascade = CascadeType.ALL)
    @JoinColumn(name = "id", referencedColumnName = "personid", insertable = false, updatable = false)

    private Long defaultAccountId;
