package com.pkb.entities.pub;

import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.OneToOne;
import javax.persistence.Table;
import java.time.Instant;

@Entity
@Table(schema = "public", name = "orgnetwork")
public class PublicOrgNetwork {

    @Id
    @GenericGenerator(name = "seq_orgnet_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "public"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_orgnet"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_orgnet_gen")
    @Column(name = "id")
    private Long id;

    @Column(name = "createddate", nullable = false)
    private Instant createdDate;

    @Column(name = "name", nullable = false)
    private String name;

    @Column(name = "createdbyperson_id", nullable = false)
    private Long createdByPersonId;

    @OneToOne
    @JoinColumn(name = "leadingorg_id", referencedColumnName = "id", nullable = false)
    private PublicOrg leadingOrg;

    public Instant getCreatedDate() {
        return createdDate;
    }

    public void setCreatedDate(Instant createdDate) {
        this.createdDate = createdDate;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Long getCreatedByPersonId() {
        return createdByPersonId;
    }

    public void setCreatedByPersonId(Long createdByPersonId) {
        this.createdByPersonId = createdByPersonId;
    }

    public PublicOrg getLeadingOrg() {
        return leadingOrg;
    }

    public void setLeadingOrg(PublicOrg leadingOrg) {
        this.leadingOrg = leadingOrg;
    }

}
