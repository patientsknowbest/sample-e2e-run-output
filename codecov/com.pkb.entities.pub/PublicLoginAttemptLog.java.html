package com.pkb.entities.pub;

import com.pkb.entities.enums.LoginAttemptLogResult;
import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.LocalDateTime;

@Entity
@Table(schema = "public", name = "loginattemptlog")
public class PublicLoginAttemptLog {

    @Id
    @GenericGenerator(name = "seq_ll_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "public"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_loginlog"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_ll_gen")
    @Column(name = "id")
    private long id;

    @Column(name = "badusername")
    private String badUserName;

    @Column(name = "result", nullable = false)
    @Enumerated(EnumType.STRING)
    private LoginAttemptLogResult result;

    @Column(name = "logintime", nullable = false)
    private LocalDateTime loginTime;

    @Column(name = "ipaddress", nullable = false)
    private String ipAddress;

    @Column(name = "personid")
    private Long personId;

    @Column(name = "subresult")
    private String subresult;

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public String getBadUserName() {
        return badUserName;
    }

    public void setBadUserName(String badUserName) {
        this.badUserName = badUserName;
    }

    public LoginAttemptLogResult getResult() {
        return result;
    }

    public void setResult(LoginAttemptLogResult result) {
        this.result = result;
    }

    public LocalDateTime getLoginTime() {
        return loginTime;
    }

    public void setLoginTime(LocalDateTime loginTime) {
        this.loginTime = loginTime;
    }

    public String getIpAddress() {
        return ipAddress;
    }

    public void setIpAddress(String ipAddress) {
        this.ipAddress = ipAddress;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }

    public String getSubresult() {
        return subresult;
    }

    public void setSubresult(String subresult) {
        this.subresult = subresult;
    }
}
