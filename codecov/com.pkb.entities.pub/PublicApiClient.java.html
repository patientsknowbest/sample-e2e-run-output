package com.pkb.entities.pub;

import com.pkb.entities.enums.api.ApiClientScope;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.OneToMany;
import javax.persistence.Table;
import java.util.List;

import static javax.persistence.FetchType.EAGER;

@Entity
@Table(schema = "public", name = "apiclient")
public class PublicApiClient {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    @Column(name = "id", nullable = false)
    private Long id;

    /** An API client entry might be disabled, even if present. See #3315 for more info. */
    @Column(name = "enabled")
    private boolean enabled;

    @Column(name = "ipwhitelist", length = 500)
    private String ipWhiteList;

    /**
     * e.g. 1 day (86400), 1 week (604800), 1 month (2592000), 1 year (31536000)
     * use 0 for a permanent session
     */
    @Column(name = "sessionexpiryseconds")
    private long sessionExpirySeconds;

    @Column(name = "tokenexpiryseconds")
    private long tokenExpirySeconds;

    @Column(name = "scopeallowed")
    @Enumerated(EnumType.STRING)
    private ApiClientScope scopeAllowed;

    /**
     * TODO: make longer and fix name
     * this is a space-separated list of URIs (or just one)
     */
    @Column(name = "redirectionendpointlist", length = 1000, nullable = false)
    private String redirectionEndpointList;

    @Column(name = "owner", length = 200)
    private String owner;

    @Column(name = "name", length = 200, nullable = false)
    private String name;

    /** this is the random key generated for the client to use when requesting authentication */
    @Column(name = "apiid", length = 32, nullable = false)
    private String apiId;

    @OneToMany(mappedBy = "apiClient", fetch = EAGER)
    private List<PublicPermittedGrantType> permittedGrantTypes;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public boolean isEnabled() {
        return enabled;
    }

    public void setEnabled(boolean enabled) {
        this.enabled = enabled;
    }

    public String getIpWhiteList() {
        return ipWhiteList;
    }

    public void setIpWhiteList(String ipWhiteList) {
        this.ipWhiteList = ipWhiteList;
    }

    public long getSessionExpirySeconds() {
        return sessionExpirySeconds;
    }

    public void setSessionExpirySeconds(long sessionExpirySeconds) {
        this.sessionExpirySeconds = sessionExpirySeconds;
    }

    public long getTokenExpirySeconds() {
        return tokenExpirySeconds;
    }

    public void setTokenExpirySeconds(long tokenExpirySeconds) {
        this.tokenExpirySeconds = tokenExpirySeconds;
    }

    /** @deprecated use {@link ApiClient#getScopesAllowed()} as the multi value scopes will be removed. */
    @Deprecated
    @SuppressWarnings("DeprecatedIsStillUsed")
    public ApiClientScope getScopeAllowed() {
        return scopeAllowed;
    }

    public void setScopeAllowed(ApiClientScope scopeAllowed) {
        this.scopeAllowed = scopeAllowed;
    }

    public String getRedirectionEndpointList() {
        return redirectionEndpointList;
    }

    public void setRedirectionEndpointList(String redirectionEndpointList) {
        this.redirectionEndpointList = redirectionEndpointList;
    }

    public String getOwner() {
        return owner;
    }

    public void setOwner(String owner) {
        this.owner = owner;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getApiId() {
        return apiId;
    }

    public void setApiId(String apiId) {
        this.apiId = apiId;
    }

    public List<PublicPermittedGrantType> getPermittedGrantTypes() {
        return permittedGrantTypes;
    }

    public void setPermittedGrantTypes(List<PublicPermittedGrantType> permittedGrantTypes) {
        this.permittedGrantTypes = permittedGrantTypes;
    }
}
