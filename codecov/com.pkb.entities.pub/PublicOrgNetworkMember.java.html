package com.pkb.entities.pub;

import com.pkb.entities.enums.OrgNetworkMemberStatus;
import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.time.Instant;

@Entity
@Table(schema = "public", name = "orgnetworkmember")
public class PublicOrgNetworkMember {

    @Id
    @GenericGenerator(name = "seq_orgnetmem_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "public"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_orgnetmem"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_orgnetmem_gen")
    private Long id;

    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    private OrgNetworkMemberStatus memberStatus;

    @ManyToOne(fetch = FetchType.LAZY, optional = false)
    @JoinColumn(name = "orgnetwork_id", referencedColumnName = "id", nullable = false)
    private PublicOrgNetwork orgNetwork;

    @Column(name = "org_id")
    private Long orgId;

    @Column(name = "createdbyperson_id")
    private Long createdByPersonId;

    @Column(insertable = false, updatable = false)
    private Instant createdDate;

    @Column(name = "approvedbyperson_id")
    private Long approvedByPersonId;

    @Column
    private Instant approvedDate;

    @Column(name = "activatedbyperson_id")
    private Long activatedByPersonId;

    @Column
    private Instant activatedDate;

    public PublicOrgNetworkMember() {
    }

    public void approve(long approvedById, Instant approved) {
        setMemberStatus(OrgNetworkMemberStatus.PENDING_ACTIVATION);
        setApprovedDate(approved);
        setApprovedByPersonId(approvedById);
    }

    public void beginActivation(long activatedByPersonId, Instant activated) {
        setMemberStatus(OrgNetworkMemberStatus.ACTIVATING);
        setActivatedDate(activated);
        setActivatedByPersonId(activatedByPersonId);
    }

    public void completeActivation() {
        if (memberStatus != OrgNetworkMemberStatus.ACTIVATING) {
            throw new IllegalStateException("Tried to complete activation for org network member " + id + " that was in state: " + memberStatus);
        }
        setMemberStatus(OrgNetworkMemberStatus.ACTIVE);
    }


    public boolean isActivated() {
        return OrgNetworkMemberStatus.ACTIVE == this.getMemberStatus();
    }

    @Override
    public String toString() {
        return "OrgNetworkMember{"
                + "id=" + id
                + ", memberStatus=" + memberStatus
                + ", orgNetwork=" + orgNetwork.getId()
                + ", org=" + orgId
                + ", activatedByPersonId=" + (activatedByPersonId == null ? "null" : activatedByPersonId)
                + ", activatedDate=" + activatedDate
                + ", createdByPerson=" + createdByPersonId
                + ", createdDate=" + createdDate
                + '}';
    }

    // getters/setters
    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public OrgNetworkMemberStatus getMemberStatus() {
        return memberStatus;
    }

    public void setMemberStatus(OrgNetworkMemberStatus memberStatus) {
        this.memberStatus = memberStatus;
    }

    public PublicOrgNetwork getOrgNetwork() {
        return orgNetwork;
    }

    public void setOrgNetwork(PublicOrgNetwork orgNetwork) {
        this.orgNetwork = orgNetwork;
    }

    public long getOrgId() {
        return orgId;
    }

    public void setOrgId(long orgId) {
        this.orgId = orgId;
    }

    public Instant getActivatedDate() {
        return activatedDate;
    }

    public void setActivatedDate(Instant activatedDate) {
        this.activatedDate = activatedDate;
    }

    public Instant getCreatedDate() {
        return createdDate;
    }

    public void setCreatedDate(Instant createdDate) {
        this.createdDate = createdDate;
    }

    public Instant getApprovedDate() {
        return approvedDate;
    }

    public void setApprovedDate(Instant approvedDate) {
        this.approvedDate = approvedDate;
    }

    public Long getCreatedByPersonId() {
        return createdByPersonId;
    }

    public void setCreatedByPersonId(Long createdByPersonId) {
        this.createdByPersonId = createdByPersonId;
    }

    public Long getApprovedByPersonId() {
        return approvedByPersonId;
    }

    public void setApprovedByPersonId(Long approvedByPersonId) {
        this.approvedByPersonId = approvedByPersonId;
    }

    public Long getActivatedByPersonId() {
        return activatedByPersonId;
    }

    public void setActivatedByPersonId(Long activatedByPersonId) {
        this.activatedByPersonId = activatedByPersonId;
    }

}
