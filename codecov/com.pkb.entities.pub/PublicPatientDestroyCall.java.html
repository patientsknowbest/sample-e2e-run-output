package com.pkb.entities.pub;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.Instant;
import java.util.UUID;

@Entity
@Table(schema = "public", name = "patient_destroy_call")
public class PublicPatientDestroyCall {

    @Id
    @GeneratedValue
    private Long id;

    @Column(name = "patient_id")
    private Long patientId;

    @Column(name = "caller_org_public_id")
    private UUID callerOrgPublicId;

    @Column(name = "called_at")
    private Instant calledAt;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Long getPatientId() {
        return patientId;
    }

    public void setPatientId(Long patientId) {
        this.patientId = patientId;
    }

    public UUID getCallerOrgPublicId() {
        return callerOrgPublicId;
    }

    public void setCallerOrgPublicId(UUID callerOrgPublicId) {
        this.callerOrgPublicId = callerOrgPublicId;
    }

    public Instant getCalledAt() {
        return calledAt;
    }

    public void setCalledAt(Instant calledAt) {
        this.calledAt = calledAt;
    }

}
