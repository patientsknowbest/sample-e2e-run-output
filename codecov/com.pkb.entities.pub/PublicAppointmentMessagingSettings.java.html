package com.pkb.entities.pub;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.Table;

@Entity
@Table(schema = "public", name = "appointment_messaging_settings")
public class PublicAppointmentMessagingSettings {

    @Id
    @Column(name="org_id")
    private long orgId;

    @Column(name="appointment_messaging_enabled")
    private boolean appointmentMessagingEnabled;

    @Column(name="appointment_messaging_team_id")
    private Long appointmentMessagingTeamId;

    public long getOrgId() {
        return orgId;
    }

    public void setOrgId(long orgId) {
        this.orgId = orgId;
    }

    public boolean isAppointmentMessagingEnabled() {
        return appointmentMessagingEnabled;
    }

    public void setAppointmentMessagingEnabled(boolean teamMessagingEnabled) {
        this.appointmentMessagingEnabled = teamMessagingEnabled;
    }

    public Long getAppointmentMessagingTeamId() {
        return appointmentMessagingTeamId;
    }

    public void setAppointmentMessagingTeamId(Long appointmentMessagingTeamId) {
        this.appointmentMessagingTeamId = appointmentMessagingTeamId;
    }
}
