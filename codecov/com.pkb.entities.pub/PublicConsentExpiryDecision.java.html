package com.pkb.entities.pub;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;

/**
 * A {@code ConsentExpiryDecision} record is created whenever the carer consent settings are saved for a carer (probably
 * parent) of a child being younger than 16 years. The {@code expires} field can be false (which is a bit misleading),
 * and it is a record of that the checkbox was explicitly checked, in other words, maintenance of the consent was explicitly
 * requested by a user (probably professional or teamcoord). If there is no expiry record for a {@link PatientConsent}, then
 * it inherently means that the consent should be deleted by default when the child reaches 16.
 *
 * So the consent is deleted (and access is revoked) when the {@code expires} flag is {@code true} OR there is no expiry record
 * present in the database.
 *
 * The {@code expires} flag is {@code true} if the consent settings form has been saved with the "Maintain Access" checkbox being
 * unchecked, so the teamcoord explicitly made a decision about the expiry. If there is no expiry record for a {@link PatientConsent}
 * then it means the consent settings weren't touched, and in this case we revoke the access of the carer.
 *
 * So we can say that an instance of this class represents an explicitly made decision about what should happen when the
 * child reaches 16.
 *
 * The related scheduled tasks are {@code ChildBirthdayEmailSender} and {@code ExpiredConsentRemoved}.
 *
 * @author erosb
 */
@Entity
@Table(name = "consentexpirydecision")
public class PublicConsentExpiryDecision implements Serializable {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private long id;

    @Column(name = "consent_id", nullable=false)
    private long consentId;

    /**
     * The consent becomes invalid after the patient reaches this age (the age is stored in years).
     */
    @Column
    private Integer patientAgeCriterion;

    @Column
    private boolean expires;

    @Column(name = "enteredby", nullable=false)
    private long enteredBy;

    @Column(nullable=false)
    private Instant enteredDate;

    @Column(name = "consent_deleted_at")
    private Instant consentDeletedAt;

    @Column
    private String reason;

    public long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public long getConsentId() {
        return consentId;
    }

    public void setConsentId(long consentId) {
        this.consentId = consentId;
    }

    public Integer getPatientAgeCriterion() {
        return patientAgeCriterion;
    }

    public void setPatientAgeCriterion(Integer patientAgeCriterion) {
        this.patientAgeCriterion = patientAgeCriterion;
    }

    public long getEnteredBy() {
        return enteredBy;
    }

    public void setEnteredBy(long enteredBy) {
        this.enteredBy = enteredBy;
    }

    public Instant getEnteredDate() {
        return enteredDate;
    }

    public void setEnteredDate(Instant enteredDate) {
        this.enteredDate = enteredDate;
    }

    public boolean isExpires() {
        return expires;
    }

    public void setExpires(boolean expires) {
        this.expires = expires;
    }

    public Instant getConsentDeletedAt() {
        return consentDeletedAt;
    }

    public void setConsentDeletedAt(Instant deletedAt) {
        this.consentDeletedAt = deletedAt;
    }

    public String getReason() {
        return reason;
    }

    public void setReason(String reason) {
        this.reason = reason;
    }

}
