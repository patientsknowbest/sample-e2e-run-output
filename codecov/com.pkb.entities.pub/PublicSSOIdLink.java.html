package com.pkb.entities.pub;

import com.pkb.entities.enums.SSOType;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Parameter;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;
import java.util.Objects;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(schema = "public", name = "sso_id_link")
public class PublicSSOIdLink implements Serializable {

    private static final long serialVersionUID = 7048243141047025662L;

    @Id
    @GenericGenerator(name = "sso_id_link_id_seq_gen", strategy = "enhanced-sequence",
            parameters = {
                    @Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @Parameter(
                            name = "schema",
                            value = "public"),
                    @Parameter(
                            name = "sequence_name",
                            value = "sso_id_link_id_seq"),
                    @Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(
            strategy = GenerationType.SEQUENCE,
            generator = "sso_id_link_id_seq_gen")
    private Long id;

    @Column(name = "person_id", nullable = false)
    private Long personId;

    @Column(name = "sso_username", nullable = false)
    private String ssoUsername;

    @Column(name = "created_on", nullable = false)
    private Instant createdOn;

    @Column(name = "sso_type", nullable = false)
    @Enumerated(EnumType.STRING)
    private SSOType ssoType;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }

    public String getSsoUsername() {
        return ssoUsername;
    }

    public void setSsoUsername(String ssoUsername) {
        this.ssoUsername = ssoUsername;
    }

    public Instant getCreatedOn() {
        return createdOn;
    }

    public void setCreatedOn(Instant createdOn) {
        this.createdOn = createdOn;
    }

    public SSOType getSsoType() {
        return ssoType;
    }

    public void setSsoType(SSOType ssoType) {
        this.ssoType = ssoType;
    }

    @Override
    public int hashCode() {
        return Objects.hash(getId(), getPersonId(), getSsoUsername(), getCreatedOn(), getSsoType());
    }

    @Override public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (o == null || getClass() != o.getClass()) {
            return false;
        }
        PublicSSOIdLink other = (PublicSSOIdLink) o;
        return Objects.equals(getId(), other.getId()) &&
                Objects.equals(getPersonId(), other.getPersonId()) &&
                Objects.equals(getSsoUsername(), other.getSsoUsername()) &&
                Objects.equals(getCreatedOn(), other.getCreatedOn()) &&
                getSsoType() == other.getSsoType();
    }
}
