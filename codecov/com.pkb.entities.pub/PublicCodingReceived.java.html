package com.pkb.entities.pub;

import com.google.common.base.MoreObjects;
import com.pkb.entities.enums.CodingDataType;
import com.pkb.entities.enums.Route;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(schema = "public", name = "codingreceived")
public class PublicCodingReceived implements Serializable {

    private static final long serialVersionUID = 1284515989162741626L;

    @Id
    @GenericGenerator(name = "seq_codingrecd_gen", strategy = "enhanced-sequence",
    parameters = {
            @org.hibernate.annotations.Parameter(
                    name = "optimizer",
                    value = OPTIMIZER_CANONICAL_NAME), // based on "pooled-lo"
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_codingrecd"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(
            strategy = GenerationType.SEQUENCE,
            generator = "seq_codingrecd_gen")
    private Long id;

    @Column(name = "coding_data_type", nullable = false)
    @Enumerated(EnumType.STRING)
    private CodingDataType codingDataType;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "org_id", referencedColumnName = "id")
    private PublicOrg org;

    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    private Route route;

    @Column(nullable = false)
    private String code1;

    @Column(name = "code_system1")
    private String codeSystem1;

    @Column
    private String code2;

    @Column(name = "code_system2")
    private String codeSystem2;

    @Column(name = "orig_display_text")
    private String origDisplayText;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "codingmatch_id", referencedColumnName = "id")
    private PubCodingMatch codingMatch;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "codingmatch_master_id", referencedColumnName = "id")
    private PubCodingMatch codingMatchMaster;

    @Column(name = "created", insertable = false, updatable = false, nullable = false)
    private Instant created;

    public PublicCodingReceived() {
    }

    public PublicCodingReceived(CodingDataType codingDataType, PublicOrg org, Route route, String origDisplayText) {
        this.codingDataType = codingDataType;
        this.org = org;
        this.route = route;
        this.origDisplayText = origDisplayText;
    }

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("id", id)
                .add("codingDataType", codingDataType)
                .add("org", org)
                .add("route", route)
                .add("code1", code1)
                .add("codeSystem1", codeSystem1)
                .add("code2", code2)
                .add("codeSystem2", codeSystem2)
                .add("origDisplayText", origDisplayText)
                .add("created", created)
                .toString();
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public CodingDataType getCodingDataType() {
        return codingDataType;
    }

    public void setCodingDataType(CodingDataType codingDataType) {
        this.codingDataType = codingDataType;
    }

    public PublicOrg getOrg() {
        return org;
    }

    public void setOrg(PublicOrg org) {
        this.org = org;
    }

    public Route getRoute() {
        return route;
    }

    public void setRoute(Route route) {
        this.route = route;
    }

    public String getCode1() {
        return code1;
    }

    public void setCode1(String code1) {
        this.code1 = code1;
    }

    public String getCodeSystem1() {
        return codeSystem1;
    }

    public void setCodeSystem1(String codeSystem1) {
        this.codeSystem1 = codeSystem1;
    }

    public String getCode2() {
        return code2;
    }

    public void setCode2(String code2) {
        this.code2 = code2;
    }

    public String getCodeSystem2() {
        return codeSystem2;
    }

    public void setCodeSystem2(String codeSystem2) {
        this.codeSystem2 = codeSystem2;
    }

    public String getOrigDisplayText() {
        return origDisplayText;
    }

    public void setOrigDisplayText(String origDisplayText) {
        this.origDisplayText = origDisplayText;
    }

    public PubCodingMatch getCodingMatch() {
        return codingMatch;
    }

    public void setCodingMatch(PubCodingMatch codingMatch) {
        this.codingMatch = codingMatch;
    }

    public PubCodingMatch getCodingMatchMaster() {
        return codingMatchMaster;
    }

    public void setCodingMatchMaster(PubCodingMatch codingMatchMaster) {
        this.codingMatchMaster = codingMatchMaster;
    }

    public Instant getCreated() {
        return created;
    }

    public void setCreated(Instant created) {
        this.created = created;
    }
}
