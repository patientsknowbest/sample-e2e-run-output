package com.pkb.entities.pub;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.time.Instant;

@Entity
@Table(schema = "public", name = "apiauthrefreshhistory")
public class PublicApiAuthRefreshHistory {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    @Column(name = "id")
    private Long id;

    @Column(name = "refreshtokenpart", nullable = false)
    private String refreshTokenPart;

    @Column(name = "issued", nullable = false)
    private Instant issued;

    @Column(name = "used", nullable = false)
    private Instant used;

    @ManyToOne
    @JoinColumn(name = "session_id", referencedColumnName = "id", nullable = false)
    private PublicApiAuthSession session;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getRefreshTokenPart() {
        return refreshTokenPart;
    }

    public void setRefreshTokenPart(String refreshTokenPart) {
        this.refreshTokenPart = refreshTokenPart;
    }

    public Instant getIssued() {
        return issued;
    }

    public void setIssued(Instant issued) {
        this.issued = issued;
    }

    public Instant getUsed() {
        return used;
    }

    public void setUsed(Instant used) {
        this.used = used;
    }

    public PublicApiAuthSession getSession() {
        return session;
    }

    public void setSession(PublicApiAuthSession session) {
        this.session = session;
    }
}
