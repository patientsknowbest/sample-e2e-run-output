package com.pkb.entities.pub;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.Table;
import java.util.Base64;

@Entity
@Table(schema = "public", name = "user_credentials")
public class PublicUserCredentials {

    @Id
    @Column(name = "user_id")
    private Long userId;

    @Column(name = "certificate_base64", nullable = false)
    private String certificateBase64;

    @Column(name = "pkcs12_base64", nullable = false)
    private String pkcs12Base64;

    public String getCertificateBase64() {
        return certificateBase64;
    }

    public byte[] getCertificateBytes() {
        return Base64.getDecoder().decode(certificateBase64);
    }

    public void setCertificateBase64(String certificateBase64) {
        this.certificateBase64 = certificateBase64;
    }

    public String getPkcs12Base64() {
        return pkcs12Base64;
    }

    public void setPkcs12Base64(String pkcs12Base64) {
        this.pkcs12Base64 = pkcs12Base64;
    }

    public Long getUserId() {
        return userId;
    }

    public void setUserId(Long userId) {
        this.userId = userId;
    }
}
