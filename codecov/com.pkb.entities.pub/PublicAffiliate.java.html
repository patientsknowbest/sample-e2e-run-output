package com.pkb.entities.pub;

import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(schema = "public", name = "affiliate")
public class PublicAffiliate implements Serializable {
    private static final long serialVersionUID = 1L;

    @Id
    @GenericGenerator(name = "seq_affiliate_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "public"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_affiliate"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.AUTO, generator = "seq_affiliate_gen")
    private long id;

    @Column(nullable = false)
    private long teamId;

    @Column(nullable = false)
    private long affiliateTeamId;

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public long getTeamId() {
        return teamId;
    }

    public void setTeamId(long teamId) {
        this.teamId = teamId;
    }

    public long getAffiliateTeamId() {
        return affiliateTeamId;
    }

    public void setAffiliateTeamId(long affiliateTeamId) {
        this.affiliateTeamId = affiliateTeamId;
    }
}
