package com.pkb.entities.pub;

import com.google.common.base.MoreObjects;
import com.pkb.datamodel.NationalIdSupport;
import com.pkb.entities.embeddables.VersionDetails;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.entities.enums.ValidNationalId;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;

import static org.apache.commons.lang3.StringUtils.trimToNull;

@Entity
@Table(schema = "public", name = "nationalid")
public class PublicNationalId implements Serializable, Comparable<PublicNationalId>, NationalIdSupport {

    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "person_id")
    private PublicPKBPerson person;

    @Enumerated(EnumType.STRING)
    private NationalIdType type;

    @Column
    private String value;

    @Column
    private Instant editDate;

    @Embedded
    private VersionDetails versionDetails = new VersionDetails();

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("id", id)
                .add("type", type)
                .add("value", value)
                .toString();
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public PublicPKBPerson getPerson() {
        return person;
    }

    public void setPerson(PublicPKBPerson person) {
        this.person = person;
    }

    public NationalIdType getType() {
        return type;
    }

    public void setType(NationalIdType type) {
        this.type = type;
    }

    public String getValue() {
        return value;
    }

    public void setValue(String value) {
        this.value = value;
    }

    public Instant getEditDate() {
        return editDate;
    }

    public void setEditDate(Instant editDate) {
        this.editDate = editDate;
    }

    public VersionDetails getVersionDetails() {
        return versionDetails;
    }

    public void setVersionDetails(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }

    public void setValidNationalId(ValidNationalId validNationalId) {
        setType(validNationalId.type());
        setValue(validNationalId.value());
    }

    @Override
    public ValidNationalId getValidNationalId() {
        //noinspection deprecation
        return NationalIdType.deprecated_buildFromDatabaseValues(type, value);
    }

    @Override
    public int compareTo(@NotNull PublicNationalId o) {
        if (type == o.getType()) {
            int compare = person.getId().compareTo(o.getPerson().getId());
            if (compare == 0) {
                // value on nationalid can be null if you remove it from a person's record. We maintain the person link,
                // but null out the value (unlike for org/team levelids where we null out the person link but keep the value).
                // Then if you enter another national id, we populate the null value with what you've entered.
                return StringUtils.trimToEmpty(value).compareTo(StringUtils.trimToEmpty(o.getValue()));
            }
        }
        return type.compareTo(o.getType());
    }

    @Override
    public boolean equals(Object o) {
        if (!(o instanceof PublicNationalId)) {
            return false;
        }

        PublicNationalId nid = (PublicNationalId) o;
        return this.type == nid.type &&
                StringUtils.trimToEmpty(nid.value).equals( StringUtils.trimToEmpty(this.value));
    }

    @Override
    public int hashCode() {
        String data = "type[" + type + "]" +
                "value[" + trimToNull(value) + "]";
        return data.hashCode();
    }
}
