package com.pkb.entities.pub;

import com.pkb.entities.enums.AccessingEntityType;
import com.pkb.entities.enums.PrivacyFlag;
import com.vladmihalcea.hibernate.type.array.EnumArrayType;
import org.hibernate.annotations.Parameter;
import org.hibernate.annotations.Type;
import org.hibernate.annotations.TypeDef;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.util.UUID;

/**
 * This entity is backed by a VIEW and is therefore read-only.
 * It is a convenience link to check consent from one person to another
 * without having to add joins to instituteuser, etc.
 * Similar to what {@link com.pkb.entities.core.UserAccountKeyLink} does for decryption access checks
 */
@Entity
@Table(schema = "public", name = "view_consent_v2")
@TypeDef(name = "privacy-flags-array", typeClass = EnumArrayType.class, parameters = {
        @Parameter(name = EnumArrayType.SQL_ARRAY_TYPE, value = "privacy_flag")}
)
public class PublicViewConsent {
    @Id
    @Column(name = "id")
    private Long id;

    @Column(name = "uniqueid")
    @Type(type="pg-uuid")
    private UUID uniqueId;

    @Type(type = "privacy-flags-array")
    @Column(name = "privacy_flags")
    private PrivacyFlag[] privacyFlags;

    @Column(name = "discharged")
    private boolean discharged;

    @Column(name = "proeditonlyforchild")
    private boolean proEditOnlyForChild;

    @Column(name = "patient_public_id")
    @Type(type = "pg-uuid")
    private UUID patientPublicId;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "patient_id")
    private PublicPKBPerson patient;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "accessing_person_id")
    private PublicPKBPerson accessingPerson; //TODO (PHR-8393) - Split this entity

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "accessing_team_id")
    private PublicTeam accessingTeam;

    @Column(name = "accessingentitytype")
    @Enumerated(EnumType.STRING)
    private AccessingEntityType accessingEntityType;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public UUID getUniqueId() {
        return uniqueId;
    }

    public void setUniqueId(UUID uniqueId) {
        this.uniqueId = uniqueId;
    }

    public PrivacyFlag[] getPrivacyFlags() {
        return privacyFlags;
    }

    public void setPrivacyFlags(PrivacyFlag[] privacyFlags) {
        this.privacyFlags = privacyFlags;
    }

    public boolean isDischarged() {
        return discharged;
    }

    public void setDischarged(boolean discharged) {
        this.discharged = discharged;
    }

    public boolean isProEditOnlyForChild() {
        return proEditOnlyForChild;
    }

    public void setProEditOnlyForChild(boolean proEditOnlyForChild) {
        this.proEditOnlyForChild = proEditOnlyForChild;
    }

    public UUID getPatientPublicId() {
        return patientPublicId;
    }

    public void setPatientPublicId(UUID patientPublicId) {
        this.patientPublicId = patientPublicId;
    }

    public PublicPKBPerson getPatient() {
        return patient;
    }

    public void setPatient(PublicPKBPerson patient) {
        this.patient = patient;
    }

    public PublicPKBPerson getAccessingPerson() {
        return accessingPerson;
    }

    public void setAccessingPerson(PublicPKBPerson accessingPerson) {
        this.accessingPerson = accessingPerson;
    }

    public PublicTeam getAccessingTeam() {
        return accessingTeam;
    }

    public void setAccessingTeam(PublicTeam accessingTeam) {
        this.accessingTeam = accessingTeam;
    }

    public AccessingEntityType getAccessingEntityType() {
        return accessingEntityType;
    }

    public void setAccessingEntityType(AccessingEntityType accessingEntityType) {
        this.accessingEntityType = accessingEntityType;
    }
}
