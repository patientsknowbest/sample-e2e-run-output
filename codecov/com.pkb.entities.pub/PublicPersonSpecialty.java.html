package com.pkb.entities.pub;

import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;

@Entity
@Table(schema = "public", name = "person_specialty")
public class PublicPersonSpecialty {
    private static final long serialVersionUID = 1L;

    @Id
    @GenericGenerator(name = "seq_prov", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "public"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_person_specialty"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_prov")
    private long id;

    @Column(name = "person_id")
    private long personId;

    /**
     * The specialty.
     */
    @Column(name = "specialty_id")
    private long specialtyId;

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public long getPersonId() {
        return personId;
    }

    public void setPersonId(long personId) {
        this.personId = personId;
    }

    public long getSpecialtyId() {
        return specialtyId;
    }

    public void setSpecialtyId(long specialtyId) {
        this.specialtyId = specialtyId;
    }

    @Override
    public boolean equals(Object obj) {
        if (!(obj instanceof PublicPersonSpecialty)) {
            return false;
        }
        PublicPersonSpecialty other = (PublicPersonSpecialty) obj;
        return other.getSpecialtyId() == this.specialtyId
                && other.getPersonId() == this.personId
                && other.getId() == this.id;
    }

    @Override
    public int hashCode() {
        return Long.valueOf(id).hashCode();
    }
}
