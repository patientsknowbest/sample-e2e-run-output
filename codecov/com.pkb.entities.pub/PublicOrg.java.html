package com.pkb.entities.pub;

import com.google.common.base.MoreObjects;
import com.pkb.entities.core.CoreAccount;
import org.hibernate.annotations.DynamicInsert;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.OneToMany;
import javax.persistence.OneToOne;
import javax.persistence.Table;
import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;
import java.util.Objects;
import java.util.UUID;

@Entity
@Table(schema = "public", name = "org")
@DynamicInsert // If a field is null, let postgres generate default
public class PublicOrg implements Serializable {

    private static final long serialVersionUID = -8846943925798526387L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    @Column(nullable = false)
    private Long id;

    @OneToMany(mappedBy = "org")
    private List<PublicTeam> teams;

    @Column(nullable = false)
    private String name;

    @Column(nullable = false, unique = true)
    private String code;

    @Column
    private String crmLink;

    @Column
    private String dataUploadIpWhitelist;

    @Column
    private String hl7UploadValidatePidList;

    @Column
    private String hl7NewPatientTeamCode;

    @Column
    private String hl7UploadSourceWhitelist;

    @Column(nullable = false)
    private String country;

    @Column(nullable = false)
    private boolean hl7AutoCreateRecords;

    @Column
    private boolean hl7CheckNhsNumberStatus;

    @Column(length = 1000)
    private String emailFooterText;

    @Column(length = 100, nullable = false)
    private String timeZoneId;

    @Column
    private String logoFileName;

    @OneToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "orgAccountId", nullable = false)
    private CoreAccount orgAccount;

    @Column(name = "public_id", nullable = false, updatable = false, unique = true)
    @Type(type = "pg-uuid")
    private UUID publicId;

    /**
     * Image in the Org email footer.
     */
    @Column(length = 100)
    private String emailFooterImageFilename;

    @Column(name = "hl7_password_hash")
    private String hl7PasswordHash;

    @Column(columnDefinition = "boolean default false", nullable = false)
    private boolean patientViewSuspended;

    @Column(name = "org_type") //Unfortunately nullable due to historical data. Populate over time
    private String orgType;

    @Column(columnDefinition = "boolean default true", nullable = false)
    private boolean active = true;

    public PublicOrg() {
        teams = new ArrayList<>();
    }

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .omitNullValues()
                .add("id", id)
                .add("name", name)
                .add("code", code)
                .toString();
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public List<PublicTeam> getTeams() {
        return teams;
    }

    public void setTeams(List<PublicTeam> teams) {
        this.teams = teams;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getCode() {
        return code;
    }

    public void setCode(String code) {
        this.code = code;
    }

    public String getCrmLink() {
        return crmLink;
    }

    public void setCrmLink(String crmLink) {
        this.crmLink = crmLink;
    }

    public String getDataUploadIpWhitelist() {
        return dataUploadIpWhitelist;
    }

    public void setDataUploadIpWhitelist(String dataUploadIpWhitelist) {
        this.dataUploadIpWhitelist = dataUploadIpWhitelist;
    }

    public String getHl7UploadValidatePidList() {
        return hl7UploadValidatePidList;
    }

    public void setHl7UploadValidatePidList(String hl7UploadValidatePidList) {
        this.hl7UploadValidatePidList = hl7UploadValidatePidList;
    }

    public String getHl7NewPatientTeamCode() {
        return hl7NewPatientTeamCode;
    }

    public void setHl7NewPatientTeamCode(String hl7NewPatientTeamCode) {
        this.hl7NewPatientTeamCode = hl7NewPatientTeamCode;
    }

    public String getHl7UploadSourceWhitelist() {
        return hl7UploadSourceWhitelist;
    }

    public void setHl7UploadSourceWhitelist(String hl7UploadSourceWhitelist) {
        this.hl7UploadSourceWhitelist = hl7UploadSourceWhitelist;
    }

    public String getCountry() {
        return country;
    }

    public void setCountry(String country) {
        this.country = country;
    }

    public boolean isHl7AutoCreateRecords() {
        return hl7AutoCreateRecords;
    }

    public void setHl7AutoCreateRecords(boolean hl7AutoCreateRecords) {
        this.hl7AutoCreateRecords = hl7AutoCreateRecords;
    }

    public boolean isHl7CheckNhsNumberStatus() {
        return hl7CheckNhsNumberStatus;
    }

    public void setHl7CheckNhsNumberStatus(boolean hl7CheckNhsNumberStatus) {
        this.hl7CheckNhsNumberStatus = hl7CheckNhsNumberStatus;
    }

    public String getEmailFooterText() {
        return emailFooterText;
    }

    public void setEmailFooterText(String emailFooterText) {
        this.emailFooterText = emailFooterText;
    }

    public String getTimeZoneId() {
        return timeZoneId;
    }

    public void setTimeZoneId(String timeZoneId) {
        this.timeZoneId = timeZoneId;
    }

    public String getLogoFileName() {
        return logoFileName;
    }

    public void setLogoFileName(String logoFileName) {
        this.logoFileName = logoFileName;
    }

    public CoreAccount getOrgAccount() {
        return orgAccount;
    }

    public void setOrgAccount(CoreAccount orgAccount) {
        this.orgAccount = orgAccount;
    }

    public UUID getPublicId() {
        return publicId;
    }

    public void setPublicId(UUID publicId) {
        this.publicId = publicId;
    }

    public String getEmailFooterImageFilename() {
        return emailFooterImageFilename;
    }

    public void setEmailFooterImageFilename(String footerLogoFileName) {
        this.emailFooterImageFilename = footerLogoFileName;
    }

    public String getHl7PasswordHash() {
        return hl7PasswordHash;
    }

    public void setHl7PasswordHash(String hl7PasswordHash) {
        this.hl7PasswordHash = hl7PasswordHash;
    }

    public boolean isPatientViewSuspended() {
        return patientViewSuspended;
    }

    public void setPatientViewSuspended(boolean patientViewSuspended) {
        this.patientViewSuspended = patientViewSuspended;
    }

    public String getOrgType() {
        return orgType;
    }

    public void setOrgType(String orgType) {
        this.orgType = orgType;
    }

    public boolean isActive() {
        return active;
    }

    public void setActive(boolean active) {
        this.active = active;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (!(o instanceof PublicOrg)) {
            return false;
        }
        PublicOrg publicOrg = (PublicOrg) o;
        return Objects.equals(code, publicOrg.code);
    }

    @Override
    public int hashCode() {
        return Objects.hash(code);
    }
}
