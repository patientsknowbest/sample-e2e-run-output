package com.pkb.entities.pub;

import com.pkb.entities.enums.api.ApiAuthSessionStatus;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.OneToMany;
import javax.persistence.Table;
import java.time.Instant;
import java.util.Set;

import static javax.persistence.CascadeType.ALL;
import static javax.persistence.FetchType.EAGER;

/**
 * A user's authorize session with the REST API on a given client.
 * Complications:
 * we have to store the user's password in the database, encrypted.
 * to decrypt it, we'll require a decryption key
 * we really do NOT want that decryption key stored in the database as well.
 * We GET the password once, when the user approves the app (here we return an auth code via redirect)
 * We NEED the password in three places:
 * app sends auth code; we return access token and refresh token
 * app sends access token (to get access)
 * app sends refresh token (to get new access token)
 *
 * So ALL of those three things must include, at the time, the decryption key, and we shouldn't store any of those in the database
 * We'll have to build them in segments, so we can do lookups on the first part and decrypt with the second part (which we won't store).
 *
 *
 *
 * @author robwhelan
 */
@Entity
@Table(schema = "public", name = "apiauthsession")
public class PublicApiAuthSession {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    @Column(name = "id")
    private Long id;

    @Column(name = "sessionid", nullable = false)
    private String sessionId;

    @Column(name = "status", nullable = false)
    @Enumerated(EnumType.STRING)
    private ApiAuthSessionStatus status;

    @Column(name = "redirectionendpoint", nullable = false)
    private String redirectionEndpoint;

    @Column(name = "scope", nullable = false)
    private String scope;

    @Column(name = "state")
    private String state;

    @Column(name = "authcodepart")
    private String authCodePart;

    @Column(name = "authcodeissued")
    private Instant authCodeIssued;

    @Column(name = "accesstokenpart")
    private String accessTokenPart;

    @Column(name = "accesstokenissued")
    private Instant accessTokenIssued;

    @Column(name = "refreshtokenpart")
    private String refreshTokenPart;

    @Column(name = "iterationcount")
    private int iterationCount;

    @Column(name = "encrypted_password_inline")
    private byte[] encryptedPasswordInline;

    @Column(name = "salt_inline")
    private byte[] saltInline;

    @Column(name = "person_id")
    private Long personId;

    @ManyToOne(fetch = EAGER)
    @JoinColumn(name = "client_id", referencedColumnName = "id", nullable = false)
    private PublicApiClient client;

    @OneToMany(mappedBy = "session", fetch = EAGER, cascade = ALL)
    private Set<PublicApiAuthRefreshHistory> refreshHistorySet;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getSessionId() {
        return sessionId;
    }

    public void setSessionId(String sessionId) {
        this.sessionId = sessionId;
    }

    public ApiAuthSessionStatus getStatus() {
        return status;
    }

    public void setStatus(ApiAuthSessionStatus status) {
        this.status = status;
    }

    public String getRedirectionEndpoint() {
        return redirectionEndpoint;
    }

    public void setRedirectionEndpoint(String redirectionEndpoint) {
        this.redirectionEndpoint = redirectionEndpoint;
    }

    public String getScope() {
        return scope;
    }

    public void setScope(String scope) {
        this.scope = scope;
    }

    public String getState() {
        return state;
    }

    public void setState(String state) {
        this.state = state;
    }

    public String getAuthCodePart() {
        return authCodePart;
    }

    public void setAuthCodePart(String authCodePart) {
        this.authCodePart = authCodePart;
    }

    public Instant getAuthCodeIssued() {
        return authCodeIssued;
    }

    public void setAuthCodeIssued(Instant authCodeIssued) {
        this.authCodeIssued = authCodeIssued;
    }

    public String getAccessTokenPart() {
        return accessTokenPart;
    }

    public void setAccessTokenPart(String accessTokenPart) {
        this.accessTokenPart = accessTokenPart;
    }

    public Instant getAccessTokenIssued() {
        return accessTokenIssued;
    }

    public void setAccessTokenIssued(Instant accessTokenIssued) {
        this.accessTokenIssued = accessTokenIssued;
    }

    public String getRefreshTokenPart() {
        return refreshTokenPart;
    }

    public void setRefreshTokenPart(String refreshTokenPart) {
        this.refreshTokenPart = refreshTokenPart;
    }

    public Integer getIterationCount() {
        return iterationCount;
    }

    public void setIterationCount(Integer iterationCount) {
        this.iterationCount = iterationCount;
    }

    public byte[] getEncryptedPasswordInline() {
        return encryptedPasswordInline;
    }

    public void setEncryptedPasswordInline(byte[] encryptedPasswordInline) {
        this.encryptedPasswordInline = encryptedPasswordInline;
    }

    public byte[] getSaltInline() {
        return saltInline;
    }

    public void setSaltInline(byte[] saltInline) {
        this.saltInline = saltInline;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }

    public PublicApiClient getClient() {
        return client;
    }

    public void setClient(PublicApiClient client) {
        this.client = client;
    }

    public Set<PublicApiAuthRefreshHistory> getRefreshHistorySet() {
        return refreshHistorySet;
    }

    public void setRefreshHistorySet(Set<PublicApiAuthRefreshHistory> refreshHistorySet) {
        this.refreshHistorySet = refreshHistorySet;
    }
}
