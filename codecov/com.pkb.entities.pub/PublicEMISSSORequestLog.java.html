package com.pkb.entities.pub;

import javax.persistence.CascadeType;
import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.OneToOne;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;
import java.util.Objects;

import static com.google.common.base.Preconditions.checkNotNull;

@Entity
@Table(schema = "public", name = "emisssorequestlog")
public class PublicEMISSSORequestLog implements Serializable {

    private static final long serialVersionUID = -9203667685472902297L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private long id;

    @Column
    private String requestId; // as received from EMIS SessionRequest

    @Column
    private Instant requestDate;

    // FetchType.EAGER bcoz this record will not be normally read
    // but if we want to read it (for audit), we want to read it all
    @OneToOne(fetch = FetchType.EAGER, cascade= CascadeType.ALL)
    @JoinColumn(name = "user_provenance_id", referencedColumnName = "id")
    private PublicEMISSSOUserProvenance userProvenance;

    @OneToOne (fetch = FetchType.EAGER, cascade=CascadeType.ALL)
    @JoinColumn(name = "app_provenance_id", referencedColumnName = "id")
    private PublicEMISSSOApplicationProvenance applicationProvenance;

    @OneToOne (fetch = FetchType.EAGER, cascade=CascadeType.ALL)
    @JoinColumn(name = "org_provenance_id", referencedColumnName = "id")
    private PublicEMISSSOOrgProvenance orgProvenance;

    @Column
    private String emisSSOUsername;

    public PublicEMISSSORequestLog() {
    }

    public PublicEMISSSORequestLog(String requestId, Instant requestDate, String emisSSOUsername) {
        this.requestId = checkNotNull(requestId, "requestId cannot be null");
        this.requestDate = checkNotNull(requestDate, "requestDate cannot be null");
        this.emisSSOUsername = checkNotNull(emisSSOUsername, "emisSSOUsername cannot be null");
    }

    public long getId() {
        return id;
    }

    public String getRequestId() {
        return requestId;
    }

    public Instant getRequestDate() {
        return requestDate;
    }

    public PublicEMISSSOUserProvenance getUserProvenance() {
        return userProvenance;
    }

    public void setUserProvenance(PublicEMISSSOUserProvenance userProvenance) {
        this.userProvenance = userProvenance;
    }

    public PublicEMISSSOApplicationProvenance getApplicationProvenance() {
        return applicationProvenance;
    }

    public void setApplicationProvenance(PublicEMISSSOApplicationProvenance applicationProvenance) {
        this.applicationProvenance = applicationProvenance;
    }

    public PublicEMISSSOOrgProvenance getOrgProvenance() {
        return orgProvenance;
    }

    public void setOrgProvenance(PublicEMISSSOOrgProvenance orgProvenance) {
        this.orgProvenance = orgProvenance;
    }

    public String getEmisSSOUsername() {
        return emisSSOUsername;
    }

    public void setEmisSSOUsername(String emisSSOUsername) {
        this.emisSSOUsername = emisSSOUsername;
    }


    @Override
    public int hashCode() {
        int prime = 31;
        int result = 1;
        result = prime * result + ((applicationProvenance == null) ? 0 : applicationProvenance.hashCode());
        result = prime * result + ((emisSSOUsername == null) ? 0 : emisSSOUsername.hashCode());
        result = prime * result + Objects.hash(id);
        result = prime * result + ((orgProvenance == null) ? 0 : orgProvenance.hashCode());
        result = prime * result + ((requestDate == null) ? 0 : requestDate.hashCode());
        result = prime * result + ((userProvenance == null) ? 0 : userProvenance.hashCode());
        return result;
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj) {
            return true;
        }
        if (obj == null) {
            return false;
        }
        if (getClass() != obj.getClass()) {
            return false;
        }
        PublicEMISSSORequestLog other = (PublicEMISSSORequestLog) obj;
        if (applicationProvenance == null) {
            if (other.applicationProvenance != null) {
                return false;
            }
        } else if (!applicationProvenance.equals(other.applicationProvenance)) {
            return false;
        }
        if (emisSSOUsername == null) {
            if (other.emisSSOUsername != null) {
                return false;
            }
        } else if (!emisSSOUsername.equals(other.emisSSOUsername)) {
            return false;
        }
        if (orgProvenance == null) {
            if (other.orgProvenance != null) {
                return false;
            }
        } else if (!orgProvenance.equals(other.orgProvenance)) {
            return false;
        }
        if (requestDate == null) {
            if (other.requestDate != null) {
                return false;
            }
        } else if (!requestDate.equals(other.requestDate)) {
            return false;
        }
        if (userProvenance == null) {
            return other.userProvenance == null;
        } else if (!userProvenance.equals(other.userProvenance)){
            return false;
        }
        return id == other.id;
    }
}
