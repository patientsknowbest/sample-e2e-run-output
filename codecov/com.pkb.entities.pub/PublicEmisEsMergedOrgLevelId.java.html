package com.pkb.entities.pub;

import com.google.common.base.MoreObjects;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Parameter;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

/**
 * This class represents primary ID that will be used if a person has multiple patient IDs
 */
@Entity
@Table(schema = "public", name = "emis_es_merged_org_level_id")
public class PublicEmisEsMergedOrgLevelId implements Serializable {

    private static final long serialVersionUID = 1L;

    @Id
    @GenericGenerator(name = "emis_es_merged_org_level_id_id_seq_gen", strategy = "enhanced-sequence", parameters = {
            @Parameter(name = "optimizer", value = OPTIMIZER_CANONICAL_NAME), // based on "pooled-lo"
            @Parameter(name = "sequence_name", value = "emis_es_merged_org_level_id_id_seq"),
            @Parameter(name = "increment_size", value = "50")
    })
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "emis_es_merged_org_level_id_id_seq_gen")
    private Long id;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "org_id", referencedColumnName = "id")
    private PublicOrg org;

    @Column(name = "merged_value", nullable = false)
    private Integer mergedValue;

    @Column(name = "primary_value", nullable = false)
    private Integer primaryValue;

    @Column(name = "added_date", nullable = false)
    private Instant addedDate;

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("id", id)
                .add("mergedValue", mergedValue)
                .add("primaryValue", primaryValue)
                .add("addedDate", addedDate)
                .toString();
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public PublicOrg getOrg() {
        return org;
    }

    public void setOrg(PublicOrg org) {
        this.org = org;
    }

    public Integer getMergedValue() {
        return mergedValue;
    }

    public void setMergedValue(Integer mergedValue) {
        this.mergedValue = mergedValue;
    }

    public Integer getPrimaryValue() {
        return primaryValue;
    }

    public void setPrimaryValue(Integer primaryValue) {
        this.primaryValue = primaryValue;
    }

    public Instant getAddedDate() {
        return addedDate;
    }

    public void setAddedDate(Instant addedDate) {
        this.addedDate = addedDate;
    }
}
