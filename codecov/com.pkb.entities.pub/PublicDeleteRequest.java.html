package com.pkb.entities.pub;

import org.hibernate.annotations.Type;
import org.springframework.data.domain.Persistable;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.PostLoad;
import javax.persistence.PostPersist;
import javax.persistence.Table;
import javax.persistence.Transient;
import java.time.Instant;
import java.util.Objects;
import java.util.UUID;

@Entity
@Table(name="delete_request", schema="public")
public class PublicDeleteRequest implements Persistable<UUID> {

    @Id
    private UUID id;

    @Column(name = "create_response_json")
    @Type(type = "jsonb")
    private String createResponseJson;

    @Column(name = "execute_response_json")
    @Type(type = "jsonb")
    private String executeResponseJson;

    @Column(name = "rollback_response_json")
    @Type(type = "jsonb")
    private String rollbackResponseJson;

    @SuppressWarnings("unused")
    @Column(name="created_date")
    private Instant createdDate;

    @Column(name="create_response_received_date")
    private Instant createResponseReceivedDate;

    @Column(name="execute_response_received_date")
    private Instant executeResponseReceivedDate;

    @Column(name="rollback_response_received_date")
    private Instant rollbackResponseReceivedDate;

    @Transient
    private boolean persisted = false;

    PublicDeleteRequest() {}

    public PublicDeleteRequest(UUID id, Instant createdDate) {
        this.id = id;
        this.createdDate = createdDate;
    }

    @Override
    public UUID getId() {
        return id;
    }


    public String getCreateResponseJson() {
        return createResponseJson;
    }

    public void setCreateResponseJson(String createResponseJson) {
        this.createResponseJson = createResponseJson;
    }

    public String getExecuteResponseJson() {
        return executeResponseJson;
    }

    public void setExecuteResponseJson(String executeResponseJson) {
        this.executeResponseJson = executeResponseJson;
    }

    public String getRollbackResponseJson() {
        return rollbackResponseJson;
    }

    public void setRollbackResponseJson(String rollbackResponseJson) {
        this.rollbackResponseJson = rollbackResponseJson;
    }

    public Instant getCreatedDate() {
        return createdDate;
    }

    public Instant getCreateResponseReceivedDate() {
        return createResponseReceivedDate;
    }

    public void setCreateResponseReceivedDate(Instant createResponseReceivedDate) {
        this.createResponseReceivedDate = createResponseReceivedDate;
    }

    public Instant getExecuteResponseReceivedDate() {
        return executeResponseReceivedDate;
    }

    public void setExecuteResponseReceivedDate(Instant executeResponseReceivedDate) {
        this.executeResponseReceivedDate = executeResponseReceivedDate;
    }

    public Instant getRollbackResponseReceivedDate() {
        return rollbackResponseReceivedDate;
    }

    public void setRollbackResponseReceivedDate(Instant rollbackResponseReceivedDate) {
        this.rollbackResponseReceivedDate = rollbackResponseReceivedDate;
    }

    @Override
    public boolean isNew() {
        return !persisted;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (!(o instanceof PublicDeleteRequest)) {
            return false;
        }
        PublicDeleteRequest that = (PublicDeleteRequest) o;
        return Objects.equals(id, that.id);
    }

    @Override
    public int hashCode() {
        return Objects.hash(id);
    }

    @PostLoad
    @PostPersist
    private void setPersisted() {
        this.persisted = true;
    }
}
