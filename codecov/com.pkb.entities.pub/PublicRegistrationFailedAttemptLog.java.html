package com.pkb.entities.pub;

import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Parameter;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.Instant;
import java.util.UUID;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@SuppressWarnings("unused")
@Entity
@Table(schema = "public", name = "registration_failed_attempt_log")
public class PublicRegistrationFailedAttemptLog {

    public enum RegistrationField {
        DATE_OF_BIRTH
    }

    @Id
    @GenericGenerator(name = "registration_failed_attempt_log_id_seq_gen", strategy = "enhanced-sequence",
            parameters = {
                    @Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @Parameter(
                            name = "schema",
                            value = "public"),
                    @Parameter(
                            name = "sequence_name",
                            value = "registration_failed_attempt_log_id_seq"),
                    @Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(
            strategy = GenerationType.SEQUENCE,
            generator = "registration_failed_attempt_log_id_seq_gen")
    private Long id;

    @Column(name = "person_public_id", updatable = false, nullable = false)
    private UUID personPublicId;

    @Column(name = "registration_time", updatable = false, nullable = false)
    private Instant registrationTime;

    @Column(name = "registration_field", updatable = false, nullable = false)
    @Enumerated(EnumType.STRING)
    private RegistrationField registrationField;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public UUID getPersonPublicId() {
        return personPublicId;
    }

    public void setPersonPublicId(UUID personPublicId) {
        this.personPublicId = personPublicId;
    }

    public Instant getRegistrationTime() {
        return registrationTime;
    }

    public void setRegistrationTime(Instant registrationTime) {
        this.registrationTime = registrationTime;
    }

    public RegistrationField getRegistrationField() {
        return registrationField;
    }

    public void setRegistrationField(RegistrationField registrationField) {
        this.registrationField = registrationField;
    }
}
