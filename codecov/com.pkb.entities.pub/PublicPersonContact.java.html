package com.pkb.entities.pub;

import com.pkb.entities.embeddables.VersionDetails;
import com.pkb.entities.enums.ContactType;
import org.hibernate.annotations.Type;
import org.jetbrains.annotations.NotNull;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.io.Serializable;
import java.util.UUID;

import static com.pkb.entities.enums.ContactType.ALSO_CONTACT;

@Entity
@Table(schema = "public", name = "personcontact")
public class PublicPersonContact implements Serializable, Comparable<PublicPersonContact> {

    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column(name = "public_id", nullable = false, updatable = false, unique = true)
    @Type(type = "pg-uuid")
    private UUID publicId;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "person_id", referencedColumnName = "id", nullable = false)
    private PublicPKBPerson person;

    @Column
    @Enumerated(EnumType.STRING)
    private ContactType contactType;

    @Column
    private String contactValue;

    @Column
    private boolean confirmed;

    @Column
    private boolean isPrimary;

    /**
     * Carer of a patient {@link PublicPersonContact#person}
     */
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "ref_person_id", referencedColumnName = "id")
    private PublicPKBPerson refPerson;

    @Embedded
    private VersionDetails versionDetails = new VersionDetails();

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public UUID getPublicId() {
        return publicId;
    }

    public void setPublicId(UUID publicId) {
        this.publicId = publicId;
    }

    public PublicPKBPerson getPerson() {
        return person;
    }

    public void setPerson(PublicPKBPerson person) {
        this.person = person;
    }

    public boolean isConfirmed() {
        return confirmed;
    }

    public void setConfirmed(boolean confirmed) {
        this.confirmed = confirmed;
    }

    public ContactType getContactType() {
        return contactType;
    }

    public void setContactType(ContactType contactType) {
        this.contactType = contactType;
    }

    public String getContactValue() {
        return contactValue;
    }

    public void setContactValue(String contactValue) {
        this.contactValue = (contactValue == null) ? null : contactValue.trim().toLowerCase();
    }

    public boolean isPrimary() {
        return isPrimary;
    }

    public boolean getIsPrimary() {
        return isPrimary;
    }

    public void setPrimary(boolean isPrimary) {
        this.isPrimary = isPrimary;
    }

    public PublicPKBPerson getRefPerson() {
        return refPerson;
    }

    public void setRefPerson(PublicPKBPerson refPerson) {
        this.refPerson = refPerson;
    }

    public VersionDetails getVersionDetails() {
        return versionDetails;
    }

    public void setVersionDetails(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }

    @Override
    public int compareTo(@NotNull PublicPersonContact o) {
        if (contactType == o.getContactType()) {
            if (contactType == ALSO_CONTACT) {
                //Paranoid data quality check. These contacts don't have contactValue set.
                return (refPerson != null ? refPerson.getId() : Long.valueOf(0))
                        .compareTo(o.refPerson != null ? o.refPerson.getId() : Long.valueOf(0));
            }
            return contactValue.compareTo(o.getContactValue());
        }
        return contactType.compareTo(o.getContactType());
    }

    @Override
    public int hashCode() {
        return getClass().hashCode();
    }

    @Override
    public boolean equals(Object o) {
        if (o instanceof PublicPersonContact) {
            return getId() != null && getId().equals(((PublicPersonContact)o).getId());
        }
        return false;
    }
}
