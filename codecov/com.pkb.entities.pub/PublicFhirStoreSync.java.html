package com.pkb.entities.pub;

import com.pkb.entities.embeddables.FhirStoreSyncDetails;

import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.Table;

import static com.pkb.entities.enums.FhirStoreSyncOperationType.CREATE;
import static com.pkb.entities.enums.FhirStoreSyncOperationType.DELETE;
import static com.pkb.entities.enums.FhirStoreSyncOperationType.UPDATE;

/**
 * Not created through the application, only through SQL triggers which execute add_to_fhir_store_sync()
 * A record of inserts, updates and deletes on a subset of tables. This information is polled so it can be
 * transformed to FHIR format and sent into the FHIR API. This allows us to maintain the two in sync as we
 * transition over.
 *
 * Once a row is completed, it is moved to fhir_store_sync_completed (see {@link PublicFhirStoreSyncCompleted})
 */
@Entity
@Table(schema = "public", name = "fhir_store_sync")
public class PublicFhirStoreSync {

    //This is only ever inserted into via triggers, so having a dedicated sequence generator doesn't make much sense.
    @Id
    @SuppressWarnings("SSBasedInspection")
    private long id;

    @Embedded
    private FhirStoreSyncDetails syncDetails = new FhirStoreSyncDetails();

    public boolean isCreateOperation() {
        return getSyncDetails().getOperationType() == CREATE;
    }

    public boolean isUpdateOperation() {
        return getSyncDetails().getOperationType() == UPDATE;
    }

    public boolean isDeleteOperation() {
        return getSyncDetails().getOperationType() == DELETE;
    }

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public FhirStoreSyncDetails getSyncDetails() {
        return syncDetails;
    }

    public void setSyncDetails(FhirStoreSyncDetails syncDetails) {
        this.syncDetails = syncDetails;
    }

    @Override
    public int hashCode() {
        return getClass().hashCode();
    }

    @Override
    public boolean equals(Object o) {
        if (o instanceof PublicFhirStoreSync) {
            return getId() == ((PublicFhirStoreSync)o).getId();
        }
        return false;
    }
}
