package com.pkb.entities.pub;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.time.Instant;

@Entity
@Table(schema = "public", name = "hl7partner_grant")
public class PublicHl7PartnerGrant {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "hl7partner_id", referencedColumnName = "id", nullable = false)
    private PublicHl7Partner hl7Partner;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "org_id", referencedColumnName = "id", nullable = false)
    private PublicOrg org;

    @Column(name = "created", updatable = false, nullable = false)
    private Instant created;

    @Column(name = "validatepidlist", nullable = false)
    private String validatePidList;

    @Column(name = "autocreaterecords", nullable = false)
    private boolean autoCreateRecords;

    @Column(name = "checknhsnumberstatus", nullable = false)
    private boolean checkNhsNumberStatus;

    @Column(name = "newpatientteamcode")
    private String newPatientTeamCode;

    @Column(name = "sourcealias", nullable = false)
    private String sourceAlias;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public PublicHl7Partner getHl7Partner() {
        return hl7Partner;
    }

    public void setHl7Partner(PublicHl7Partner hl7Partner) {
        this.hl7Partner = hl7Partner;
    }

    public PublicOrg getOrg() {
        return org;
    }

    public void setOrg(PublicOrg org) {
        this.org = org;
    }

    public Instant getCreated() {
        return created;
    }

    public void setCreated(Instant created) {
        this.created = created;
    }

    public String getValidatePidList() {
        return validatePidList;
    }

    public void setValidatePidList(String validatePidList) {
        this.validatePidList = validatePidList;
    }

    public boolean isAutoCreateRecords() {
        return autoCreateRecords;
    }

    public void setAutoCreateRecords(boolean autoCreateRecords) {
        this.autoCreateRecords = autoCreateRecords;
    }

    public boolean isCheckNhsNumberStatus() {
        return checkNhsNumberStatus;
    }

    public void setCheckNhsNumberStatus(boolean checkNhsNumberStatus) {
        this.checkNhsNumberStatus = checkNhsNumberStatus;
    }

    public String getNewPatientTeamCode() {
        return newPatientTeamCode;
    }

    public void setNewPatientTeamCode(String newPatientTeamCode) {
        this.newPatientTeamCode = newPatientTeamCode;
    }

    public String getSourceAlias() {
        return sourceAlias;
    }

    public void setSourceAlias(String sourceAlias) {
        this.sourceAlias = sourceAlias;
    }

}
