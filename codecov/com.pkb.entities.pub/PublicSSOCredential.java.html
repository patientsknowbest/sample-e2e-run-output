package com.pkb.entities.pub;

import com.pkb.entities.embeddables.EncryptedWrapper;
import com.pkb.entities.enums.SSOType;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Parameter;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;
import java.util.Objects;
import java.util.UUID;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(schema = "public", name = "sso_credential")
public class PublicSSOCredential implements Serializable {

    private static final long serialVersionUID = -1003808896427811571L;

    @Id
    @GenericGenerator(name = "sso_credential_id_seq_gen", strategy = "enhanced-sequence",
            parameters = {
                    @Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @Parameter(
                            name = "schema",
                            value = "public"),
                    @Parameter(
                            name = "sequence_name",
                            value = "sso_credential_id_seq"),
                    @Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(
            strategy = GenerationType.SEQUENCE,
            generator = "sso_credential_id_seq_gen")
    private Long id;

    @Column(name = "created_on", nullable = false)
    private Instant createdOn;

    @Column(name = "person_public_id", nullable = false)
    @Type(type = "pg-uuid")
    private UUID personPublicId;

    @Embedded
    private EncryptedWrapper ssoPasswordEncryptedData;

    @Column(name = "sso_type", nullable = false)
    @Enumerated(EnumType.STRING)
    private SSOType ssoType;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Instant getCreatedOn() {
        return createdOn;
    }

    public void setCreatedOn(Instant createdOn) {
        this.createdOn = createdOn;
    }

    public EncryptedWrapper getSsoPasswordEncryptedData() {
        return ssoPasswordEncryptedData;
    }

    public void setSsoPasswordEncryptedData(EncryptedWrapper ssoPasswordEncryptedData) {
        this.ssoPasswordEncryptedData = ssoPasswordEncryptedData;
    }

    public UUID getPersonPublicId() {
        return personPublicId;
    }

    public void setPersonPublicId(UUID personPublicId) {
        this.personPublicId = personPublicId;
    }

    public SSOType getSsoType() {
        return ssoType;
    }

    public void setSsoType(SSOType ssoType) {
        this.ssoType = ssoType;
    }

    @Override public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (o == null || getClass() != o.getClass()) {
            return false;
        }
        PublicSSOCredential that = (PublicSSOCredential) o;
        return Objects.equals(getId(), that.getId()) &&
                Objects.equals(getCreatedOn(), that.getCreatedOn()) &&
                Objects.equals(getSsoPasswordEncryptedData(), that.getSsoPasswordEncryptedData()) &&
                Objects.equals(getPersonPublicId(), that.getPersonPublicId()) &&
                getSsoType() == that.getSsoType();
    }

    @Override public int hashCode() {
        return Objects.hash(getId(), getCreatedOn(), getSsoPasswordEncryptedData(), getPersonPublicId(), getSsoType());
    }
}
