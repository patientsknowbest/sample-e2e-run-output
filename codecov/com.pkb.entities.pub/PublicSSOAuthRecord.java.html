package com.pkb.entities.pub;

import com.pkb.entities.enums.SSOType;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Parameter;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.PostLoad;
import javax.persistence.Table;
import javax.persistence.Transient;
import java.io.Serializable;
import java.time.Instant;
import java.util.Base64;
import java.util.Objects;
import java.util.UUID;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(schema = "public", name = "sso_auth_record")
public class PublicSSOAuthRecord implements Serializable {

    private static final long serialVersionUID = -7818892160131186810L;

    @Id
    @GenericGenerator(name = "sso_auth_record_id_seq_gen", strategy = "enhanced-sequence",
            parameters = {
                    @Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @Parameter(
                            name = "schema",
                            value = "public"),
                    @Parameter(
                            name = "sequence_name",
                            value = "sso_auth_record_id_seq"),
                    @Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(
            strategy = GenerationType.SEQUENCE,
            generator = "sso_auth_record_id_seq_gen")
    private long id;

    @Column(name = "created_on", nullable = false)
    private Instant createdOn;

    @Column(name = "person_public_id", nullable = false)
    @Type(type = "pg-uuid")
    private UUID personPublicId;

    @Column(name = "pkb_key_base64")
    private String pkbKeyBase64;

    @Column(name = "salt_base64", nullable = false)
    private String saltBase64;

    @Column(name = "encrypted_private_key_base64", nullable = false)
    private String encryptedPrivateKeyBase64;

    @Column(name = "iteration_count", nullable = false)
    private int iterationCount;

    @Column(name = "sso_type", nullable = false)
    @Enumerated(EnumType.STRING)
    private SSOType ssoType;

    @Transient
    private byte[] pkbKey;

    @Transient
    private byte[] salt;

    @Transient
    private byte[] encryptedPrivateKey;

    @PostLoad
    public void decodeBase64Fields() {
        pkbKey = pkbKeyBase64 != null ? Base64.getDecoder().decode(pkbKeyBase64) : null;
        salt = Base64.getDecoder().decode(saltBase64);
        encryptedPrivateKey = Base64.getDecoder().decode(encryptedPrivateKeyBase64);
    }

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public Instant getCreatedOn() {
        return createdOn;
    }

    public void setCreatedOn(Instant createdOn) {
        this.createdOn = createdOn;
    }

    public UUID getPersonPublicId() {
        return personPublicId;
    }

    public void setPersonPublicId(UUID personPublicId) {
        this.personPublicId = personPublicId;
    }

    public String getPkbKeyBase64() {
        return pkbKeyBase64;
    }

    public void setPkbKeyBase64(String pkbKeyBase64) {
        this.pkbKeyBase64 = pkbKeyBase64;
    }

    public String getSaltBase64() {
        return saltBase64;
    }

    public void setSaltBase64(String saltBase64) {
        this.saltBase64 = saltBase64;
    }

    public String getEncryptedPrivateKeyBase64() {
        return encryptedPrivateKeyBase64;
    }

    public void setEncryptedPrivateKeyBase64(String encryptedPrivateKeyBase64) {
        this.encryptedPrivateKeyBase64 = encryptedPrivateKeyBase64;
    }

    public int getIterationCount() {
        return iterationCount;
    }

    public void setIterationCount(int iterationCount) {
        this.iterationCount = iterationCount;
    }

    public SSOType getSsoType() {
        return ssoType;
    }

    public void setSsoType(SSOType ssoType) {
        this.ssoType = ssoType;
    }

    public byte[] getPkbKey() {
        return pkbKey;
    }

    public void setPkbKey(byte[] pkbKey) {
        this.pkbKey = pkbKey;
    }

    public byte[] getSalt() {
        return salt;
    }

    public void setSalt(byte[] salt) {
        this.salt = salt;
    }

    public byte[] getEncryptedPrivateKey() {
        return encryptedPrivateKey;
    }

    public void setEncryptedPrivateKey(byte[] encryptedPrivateKey) {
        this.encryptedPrivateKey = encryptedPrivateKey;
    }

    @Override
    public int hashCode() {
        return Objects.hash(getId(), getCreatedOn(), getPersonPublicId(), getPkbKeyBase64(), getSaltBase64(),
                getEncryptedPrivateKeyBase64(), getIterationCount(), getSsoType());
    }

    @Override public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (o == null || getClass() != o.getClass()) {
            return false;
        }
        PublicSSOAuthRecord other = (PublicSSOAuthRecord) o;
        return getId() == other.getId() &&
                Objects.equals(getCreatedOn(), other.getCreatedOn()) &&
                Objects.equals(getPersonPublicId(), other.getPersonPublicId()) &&
                Objects.equals(getPkbKeyBase64(), other.getPkbKeyBase64()) &&
                Objects.equals(getSaltBase64(), other.getSaltBase64()) &&
                Objects.equals(getEncryptedPrivateKeyBase64(), other.getEncryptedPrivateKeyBase64()) &&
                getIterationCount() == other.getIterationCount() &&
                getSsoType() == other.getSsoType();
    }
}
