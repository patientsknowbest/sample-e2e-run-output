package com.pkb.entities.pub;

import com.pkb.entities.enums.api.GrantType;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.io.Serializable;

@Entity
@Table(schema = "public", name = "oauth2_permitted_grant_types")
public class PublicPermittedGrantType implements Serializable {

    /** Note: when using Hibernate a composite-id class, such as this, must implement Serializable */

    @Id
    @Column(name = "grant_type_name", nullable = false)
    @Enumerated(EnumType.STRING)
    private GrantType grantType;

    @Id
    @ManyToOne(optional = false)
    @JoinColumn(name = "apiclient_id", referencedColumnName = "id", nullable = false)
    private PublicApiClient apiClient;

    public GrantType getGrantType() {
        return grantType;
    }

    public void setGrantType(GrantType grantType) {
        this.grantType = grantType;
    }

    public PublicApiClient getApiClient() {
        return apiClient;
    }

    public void setApiClient(PublicApiClient apiClient) {
        this.apiClient = apiClient;
    }
}
