package com.pkb.entities.pub;

import com.pkb.entities.enums.AdditionalConsentId;
import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;

/**
 *
 * @author Russ Gray <russ.gray@patientsknowbest.com>
 */
@Entity
@Table(schema = "public", name = "teamadditionalconsent")
public class PublicTeamAdditionalConsent implements Serializable {

    private static final long serialVersionUID = 1891348592707989225L;

    @Id
    @GenericGenerator(name = "seq_pc_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "public"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_patientconsent"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_pc_gen")
    private long id;

    @Column(nullable = false)
    private long teamId;

    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    private AdditionalConsentId additionalConsent;

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public long getTeamId() {
        return teamId;
    }

    public void setTeamId(long teamId) {
        this.teamId = teamId;
    }

    public AdditionalConsentId getAdditionalConsent() {
        return additionalConsent;
    }

    public void setAdditionalConsent(AdditionalConsentId additionalConsentId) {
        this.additionalConsent = additionalConsentId;
    }
}
