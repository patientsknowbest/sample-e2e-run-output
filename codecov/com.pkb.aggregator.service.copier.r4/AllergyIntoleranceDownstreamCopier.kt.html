package com.pkb.aggregator.service.copier.r4

import com.pkb.aggregator.R4AllergyIntolerance
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.aggregation.updateResourceId
import com.pkb.aggregator.service.copier.DownstreamCopier
import com.pkb.aggregator.service.reference.ReferenceExtractor.IdentifierMap
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

@Service
class AllergyIntoleranceDownstreamCopier : DownstreamCopier<R4AllergyIntolerance> {

    override fun accepts(resource: IDomainResource) = R4AllergyIntolerance::class.isInstance(resource)

    override fun copy(resources: List<IDomainResource>, references: IdentifierMap): Map<String, R4AllergyIntolerance> {

        val allergies = resources.mapNotNull { it as? R4AllergyIntolerance }
        val downstreamAllergies = allergies
            .map { it.copy() }
            .onEach { updateReferences(it, references) }

        return updateResourceId(downstreamAllergies)
    }

    private fun updateReferences(allergyIntolerance: R4AllergyIntolerance, references: IdentifierMap) {
        allergyIntolerance.patient = createPatientReference(allergyIntolerance, references)
        allergyIntolerance.encounter = createEncounterReference(allergyIntolerance, references)
    }

    private fun createPatientReference(
        allergyIntolerance: R4AllergyIntolerance,
        references: IdentifierMap,
    ) =
        if (allergyIntolerance.hasPatient()) {
            val source = allergyIntolerance.getSourcedReference()
            val needle = UpstreamResource(source.upstreamSource, allergyIntolerance.patient.reference)
            references.getPatientReference(needle)
        } else null

    private fun createEncounterReference(
        allergyIntolerance: R4AllergyIntolerance,
        references: IdentifierMap,
    ) =
        if (allergyIntolerance.hasEncounter()) {
            val source = allergyIntolerance.getSourcedReference()
            val needle = UpstreamResource(source.upstreamSource, allergyIntolerance.encounter.reference)
            references.getEncounterReference(needle)
        } else null
}
