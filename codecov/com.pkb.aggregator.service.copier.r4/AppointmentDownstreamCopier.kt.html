package com.pkb.aggregator.service.copier.r4

import com.pkb.aggregator.R4Appointment
import com.pkb.aggregator.R4Reference
import com.pkb.aggregator.configuration.PhrMigrationConfiguration
import com.pkb.aggregator.configuration.PhrMigrationConfiguration.Companion.PRIVACY_OVERRIDE_ORG
import com.pkb.aggregator.configuration.PhrMigrationConfiguration.Companion.SOURCE_ORG
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_PATIENT
import com.pkb.aggregator.service.FHIR_TYPE_PRACTITIONER
import com.pkb.aggregator.service.aggregation.updateResourceId
import com.pkb.aggregator.service.copier.DownstreamCopier
import com.pkb.aggregator.service.reference.ReferenceExtractor.IdentifierMap
import org.hl7.fhir.instance.model.api.IDomainResource
import org.hl7.fhir.r4.model.Appointment
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class AppointmentDownstreamCopier(private val phrMigrationConfiguration: PhrMigrationConfiguration) :
    DownstreamCopier<R4Appointment> {

    private val log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun accepts(resource: IDomainResource) = R4Appointment::class.isInstance(resource)

    override fun copy(resources: List<IDomainResource>, references: IdentifierMap) =
        resources.mapNotNull { it as? R4Appointment }
            .map { it.copy() }
            .onEach { updateReferences(it, references) }
            .let { updateResourceId(it) }

    private fun updateReferences(appointment: R4Appointment, references: IdentifierMap) {
        appointment.participant = createParticipantReferences(appointment, references)
        appointment.meta = createMetaSourceOrganizationReferences(appointment, references)
    }

    private fun createParticipantReferences(
        appointment: R4Appointment,
        references: IdentifierMap,
    ): List<Appointment.AppointmentParticipantComponent> {
        val source = appointment.getSourcedReference()
        return appointment.participant
            .mapNotNull { participant ->
                if (participant.actor.hasReference()) {
                    // Translate the reference
                    UpstreamResource(source.upstreamSource, participant.actor.reference).let { needle ->
                        when (val resourceType = participant.actor.referenceElement.resourceType) {
                            FHIR_TYPE_PRACTITIONER -> references.getPractitionerReference(needle)
                            FHIR_TYPE_PATIENT -> references.getPatientReference(needle)
                            else -> throw RuntimeException("Unexpected participant actor type $resourceType")
                        }
                    }.let { downStreamReference ->
                        participant.copy().apply {
                            actor = downStreamReference.copy()
                        }
                    }
                } else {
                    // No reference in the participant actor, probably just a display value - so copy it
                    participant.copy()
                }
            }
    }

    /**
     * Translates organization references found in extensions to the meta element of an Appointment
     */
    private fun createMetaSourceOrganizationReferences(
        appointment: R4Appointment,
        references: IdentifierMap,
    ) = if (appointment.hasMeta()) {
        val source = appointment.getSourcedReference()
        val meta = appointment.meta.copy()
        if (meta.hasExtension()) {
            meta.getExtensionByUrl(phrMigrationConfiguration.extensionSystems[SOURCE_ORG])?.let {
                val needle = UpstreamResource(source.upstreamSource, (it.value as R4Reference).reference)
                it.setValue(references.getOrganizationReference(needle))
            }
            meta.getExtensionByUrl(phrMigrationConfiguration.extensionSystems[PRIVACY_OVERRIDE_ORG])?.let {
                val needle = UpstreamResource(source.upstreamSource, (it.value as R4Reference).reference)
                it.setValue(references.getOrganizationReference(needle))
            }
        }
        meta
    } else {
        null
    }
}
