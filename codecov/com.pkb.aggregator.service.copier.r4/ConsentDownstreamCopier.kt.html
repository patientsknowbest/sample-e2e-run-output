package com.pkb.aggregator.service.copier.r4

import com.pkb.aggregator.R4Consent
import com.pkb.aggregator.R4ConsentProvisionActorComponent
import com.pkb.aggregator.R4Reference
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_ORGANIZATION
import com.pkb.aggregator.service.FHIR_TYPE_PRACTITIONER
import com.pkb.aggregator.service.aggregation.updateResourceId
import com.pkb.aggregator.service.copier.DownstreamCopier
import com.pkb.aggregator.service.reference.ReferenceExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.hl7.fhir.r4.model.Reference
import org.springframework.stereotype.Service


@Service
class ConsentDownstreamCopier : DownstreamCopier<R4Consent> {

    override fun accepts(resource: IDomainResource) = R4Consent::class.isInstance(resource)

    override fun copy(resources: List<IDomainResource>, references: ReferenceExtractor.IdentifierMap): Map<String, R4Consent> =
            resources.filterIsInstance<R4Consent>()
                    .map { it.copy() }
                    .onEach { updateReferences(it, references) }
                    .let { updateResourceId(it) }

    private fun updateReferences(consent: R4Consent, references: ReferenceExtractor.IdentifierMap) {
        if (consent.hasPatient()) {
            consent.patient = createPatientReference(consent, references)
        }
        if (consent.hasOrganization()) {
            consent.organization = createOrganizationReference(consent, references)
        }
        if (consent.hasProvision()) {
            consent.provision?.actor = createActorReference(consent, references)
        }
    }

    private fun createPatientReference(consent: R4Consent, references: ReferenceExtractor.IdentifierMap): R4Reference {
        val source = consent.getSourcedReference()
        val needle = UpstreamResource(source.upstreamSource, consent.patient.reference)
        return references.getPatientReference(needle)
    }

    private fun createOrganizationReference(consent: R4Consent, references: ReferenceExtractor.IdentifierMap)
            : List<Reference> {
        val source = consent.getSourcedReference()

        return consent.organization.map { organization ->
            if (organization.hasReference()) {
                val needle = UpstreamResource(source.upstreamSource, organization.reference)
                references.getOrganizationReference(needle)
            } else {
                organization.copy()
            }
        }
    }

    private fun createActorReference(consent: R4Consent, references: ReferenceExtractor.IdentifierMap)
            : List<R4ConsentProvisionActorComponent>? {

        val source = consent.getSourcedReference();
        return consent.provision.actor
                .mapNotNull { actor ->
                    if (actor.hasReference()) {
                        UpstreamResource(source.upstreamSource, actor.reference.reference).let { needle ->
                            when (val resourceType = actor.reference.referenceElement.resourceType) {
                                FHIR_TYPE_PRACTITIONER -> references.getPractitionerReference(needle)
                                FHIR_TYPE_ORGANIZATION -> references.getOrganizationReference(needle)
                                else -> throw RuntimeException("Unexpected type $resourceType")
                            }
                        }.let { downstreamReference ->
                            actor.copy().apply {
                                reference = downstreamReference.copy()
                            }
                        }
                    } else {
                        actor.copy()
                    }
                }
    }
}
