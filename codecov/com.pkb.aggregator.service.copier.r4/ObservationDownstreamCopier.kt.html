package com.pkb.aggregator.service.copier.r4

import com.pkb.aggregator.R4Observation
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_PATIENT
import com.pkb.aggregator.service.aggregation.updateResourceId
import com.pkb.aggregator.service.copier.DownstreamCopier
import com.pkb.aggregator.service.reference.ReferenceExtractor.IdentifierMap
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

@Service
class ObservationDownstreamCopier : DownstreamCopier<R4Observation> {

    override fun accepts(resource: IDomainResource) = R4Observation::class.isInstance(resource)

    override fun copy(resources: List<IDomainResource>, references: IdentifierMap): Map<String, R4Observation> {

        val downstreamObservations = resources.mapNotNull { it as? R4Observation }
            .map { it.copy() }
            .onEach { updateReferences(it, references) }

        return updateResourceId(downstreamObservations)
    }

    private fun updateReferences(observation: R4Observation, references: IdentifierMap) {
        observation.subject = createSubjectReference(observation, references)
        observation.encounter = createEncounterReference(observation, references)
    }

    private fun createSubjectReference(observation: R4Observation, references: IdentifierMap) =
        if (observation.hasSubject() && observation.subject.referenceElement.resourceType == FHIR_TYPE_PATIENT) {
            val source = observation.getSourcedReference()
            val needle = UpstreamResource(source.upstreamSource, observation.subject.reference)
            references.getPatientReference(needle)
        } else {
            null
        }

    private fun createEncounterReference(observation: R4Observation, references: IdentifierMap) =
        if (observation.hasEncounter()) {
            val source = observation.getSourcedReference()
            val needle = UpstreamResource(source.upstreamSource, observation.encounter.reference)
            references.getEncounterReference(needle)
        } else {
            null
        }
}
