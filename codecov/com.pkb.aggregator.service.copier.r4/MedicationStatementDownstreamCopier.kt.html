package com.pkb.aggregator.service.copier.r4

import com.pkb.aggregator.R4MedicationStatement
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_MEDICATION
import com.pkb.aggregator.service.FHIR_TYPE_PATIENT
import com.pkb.aggregator.service.aggregation.updateResourceId
import com.pkb.aggregator.service.copier.DownstreamCopier
import com.pkb.aggregator.service.reference.ReferenceExtractor.IdentifierMap
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

@Service
class MedicationStatementDownstreamCopier : DownstreamCopier<R4MedicationStatement> {

    override fun accepts(resource: IDomainResource) = R4MedicationStatement::class.isInstance(resource)

    override fun copy(
        resources: List<IDomainResource>,
        references: IdentifierMap,
    ): Map<String, R4MedicationStatement> {

        // Make copies of each resource and set correct downstream references
        val downstreamMedicationStatements = resources.mapNotNull { it as? R4MedicationStatement }
            .map { it.copy() }
            .onEach { updateReferences(it, references) }

        // Return new resources, keyed by original upstream resource ids
        return updateResourceId(downstreamMedicationStatements)
    }

    private fun updateReferences(
        medicationStatement: R4MedicationStatement,
        references: IdentifierMap,
    ) {
        medicationStatement.subject = createSubjectReference(medicationStatement, references)
        medicationStatement.medication = createMedicationReference(medicationStatement, references)
    }

    private fun createSubjectReference(medicationStatement: R4MedicationStatement, references: IdentifierMap) =
        if (medicationStatement.hasSubject() && medicationStatement.subject.referenceElement.resourceType == FHIR_TYPE_PATIENT) {
            val source = medicationStatement.getSourcedReference()
            val needle = UpstreamResource(source.upstreamSource, medicationStatement.subject.reference)
            references.getPatientReference(needle)
        } else {
            null
        }

    private fun createMedicationReference(medicationStatement: R4MedicationStatement, references: IdentifierMap) =
        if (medicationStatement.hasMedication() && medicationStatement.medicationReference.referenceElement.resourceType == FHIR_TYPE_MEDICATION) {
            val source = medicationStatement.getSourcedReference()
            val needle = UpstreamResource(source.upstreamSource, medicationStatement.medicationReference.reference)
            references.getMedicationReference(needle)
        } else {
            null
        }
}
