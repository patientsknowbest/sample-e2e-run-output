package com.pkb.aggregator.service.copier.r4

import com.pkb.aggregator.R4CodeSystem
import com.pkb.aggregator.service.aggregation.updateResourceId
import com.pkb.aggregator.service.copier.DownstreamCopier
import com.pkb.aggregator.service.reference.ReferenceExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

@Service
class CodeSystemDownstreamCopier : DownstreamCopier<R4CodeSystem> {

    override fun accepts(resource: IDomainResource) = R4CodeSystem::class.isInstance(resource)

    override fun copy(resources: List<IDomainResource>, references: ReferenceExtractor.IdentifierMap): Map<String, R4CodeSystem> {

        val codeSystems = resources.mapNotNull { it as? R4CodeSystem }
        val downstreamCodeSystems = codeSystems
            .map { it.copy() }

        return updateResourceId(downstreamCodeSystems)
    }
}
