package com.pkb.aggregator.service.copier.r4

import com.pkb.aggregator.R4Extension
import com.pkb.aggregator.R4Provenance
import com.pkb.aggregator.R4Reference
import com.pkb.aggregator.configuration.PhrMigrationConfiguration
import com.pkb.aggregator.configuration.PhrMigrationConfiguration.Companion.SOURCE_ORG
import com.pkb.aggregator.configuration.PhrMigrationConfiguration.Companion.SOURCE_PERSON
import com.pkb.aggregator.configuration.PhrMigrationConfiguration.Companion.SOURCE_TEAM
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_CONSENT
import com.pkb.aggregator.service.FHIR_TYPE_ORGANIZATION
import com.pkb.aggregator.service.FHIR_TYPE_PATIENT
import com.pkb.aggregator.service.FHIR_TYPE_PRACTITIONER
import com.pkb.aggregator.service.aggregation.updateResourceId
import com.pkb.aggregator.service.copier.DownstreamCopier
import com.pkb.aggregator.service.reference.ReferenceExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.hl7.fhir.r4.model.Provenance
import org.hl7.fhir.r4.model.Reference
import org.springframework.stereotype.Service

@Service
class ProvenanceDownstreamCopier(
    private val migrationConfiguration: PhrMigrationConfiguration,
) : DownstreamCopier<R4Provenance> {

    override fun accepts(resource: IDomainResource) = R4Provenance::class.isInstance(resource)

    override fun copy(
        resources: List<IDomainResource>,
        references: ReferenceExtractor.IdentifierMap
    ): Map<String, R4Provenance> =
        resources.mapNotNull { it as? R4Provenance }
            .map { it.copy() }
            .onEach { updateReferences(it, references) }
            .let { updateResourceId(it) }

    private fun updateReferences(provenance: R4Provenance, references: ReferenceExtractor.IdentifierMap) {
        provenance.target = createTargetReferences(provenance, references)
        provenance.extension = createExtensions(provenance, references)
        provenance.agent = createAgent(provenance, references)
    }

    private fun createTargetReferences(
        provenance: R4Provenance,
        references: ReferenceExtractor.IdentifierMap
    ): List<Reference> {
        val source = provenance.getSourcedReference()

        return provenance.target.map { target ->
            if (target.hasReference()) {
                UpstreamResource(source.upstreamSource, target.reference).let { needle ->
                    when (val resourceType = target.referenceElement.resourceType) {
                        FHIR_TYPE_CONSENT -> references.getConsentReference(needle)
                        FHIR_TYPE_PATIENT -> references.getPatientReference(needle)
                        FHIR_TYPE_PRACTITIONER -> references.getPractitionerReference(needle)
                        else -> throw RuntimeException("Unexpected target type $resourceType")
                    }
                }
            } else {
                target.copy()
            }
        }
    }

    private fun createExtensions(
        provenance: R4Provenance,
        references: ReferenceExtractor.IdentifierMap
    ): List<R4Extension>? {
        val source = provenance.getSourcedReference()

        return provenance.extension.map {
            when (it.url) {
                migrationConfiguration.extensionSystems[PhrMigrationConfiguration.CONNECTING_ORG] -> {
                    it.copy().apply {
                        setValue(references.getOrganizationReference(UpstreamResource(
                            source.upstreamSource,
                            (it.value as Reference).reference
                        )))
                    }
                }
                else -> it.copy()
            }
        }
    }

    private fun createAgent(
        provenance: R4Provenance,
        references: ReferenceExtractor.IdentifierMap
    ): List<Provenance.ProvenanceAgentComponent>? {
        val source = provenance.getSourcedReference()

        return provenance.agent.map { agent ->
            agent.copy().apply {
                who?.reference?.let {
                    who = UpstreamResource(source.upstreamSource, it).let { needle ->
                        when (val resourceType = who.referenceElement.resourceType) {
                            FHIR_TYPE_ORGANIZATION -> references.getOrganizationReference(needle)
                            FHIR_TYPE_PATIENT -> references.getPatientReference(needle)
                            FHIR_TYPE_PRACTITIONER -> references.getPractitionerReference(needle)
                            else -> throw RuntimeException("Unexpected target type $resourceType")
                        }
                    }
                    onBehalfOf = who.copy()
                }
                // inside .map, return or return@map would return to the outer map, hence we use the 'extension' label
                onBehalfOf.extension = agent.onBehalfOf.extension.map extension@{
                    val reference = (it.value as? R4Reference)
                        ?: return@extension it.copy()

                    when (it.url) {
                        migrationConfiguration.extensionSystems[SOURCE_ORG],
                        migrationConfiguration.extensionSystems[SOURCE_TEAM] -> {
                            val needle = UpstreamResource(source.upstreamSource, reference.reference)
                            it.copy().apply {
                                setValue(references.getOrganizationReference(needle))
                            }
                        }
                        // TODO PHR-10107 copy reference to Patient or Practitioner
                        // We are not doing anything with these since we can't tell right now if this is a patient
                        // or a practitioner
                        migrationConfiguration.extensionSystems[SOURCE_PERSON] -> it.copy()
                        else -> {
                            throw RuntimeException("Unexpected reference type in extension, url: ${it.url}")
                        }
                    }
                }
            }
        }
    }
}
