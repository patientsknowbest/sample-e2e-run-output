package com.pkb.aggregator.service.copier.r4

import com.pkb.aggregator.R4Condition
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_PATIENT
import com.pkb.aggregator.service.aggregation.updateResourceId
import com.pkb.aggregator.service.copier.DownstreamCopier
import com.pkb.aggregator.service.reference.ReferenceExtractor.IdentifierMap
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

@Service
class ConditionDownstreamCopier : DownstreamCopier<R4Condition> {

    override fun accepts(resource: IDomainResource) = R4Condition::class.isInstance(resource)

    override fun copy(resources: List<IDomainResource>, references: IdentifierMap): Map<String, R4Condition> {

        val downstreamConditions = resources.mapNotNull { it as? R4Condition }
            .map { it.copy() }
            .onEach { updateReferences(it, references) }

        return updateResourceId(downstreamConditions)
    }

    private fun updateReferences(condition: R4Condition, references: IdentifierMap) {
        condition.subject = createSubjectReference(condition, references)
        condition.encounter = createEncounterReference(condition, references)
    }

    private fun createSubjectReference(condition: R4Condition, references: IdentifierMap) =
        if (condition.hasSubject() && condition.subject.referenceElement.resourceType == FHIR_TYPE_PATIENT) {
            val source = condition.getSourcedReference()
            val needle = UpstreamResource(source.upstreamSource, condition.subject.reference)
            references.getPatientReference(needle)
        } else {
            null
        }

    private fun createEncounterReference(condition: R4Condition, references: IdentifierMap) =
        if (condition.hasEncounter()) {
            val source = condition.getSourcedReference()
            val needle = UpstreamResource(source.upstreamSource, condition.encounter.reference)
            references.getEncounterReference(needle)
        } else {
            null
        }
}
