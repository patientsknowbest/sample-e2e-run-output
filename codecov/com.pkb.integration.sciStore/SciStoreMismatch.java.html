package com.pkb.integration.sciStore;

import com.pkb.user.entity.PKBPerson;
import uk.nhs.scot.show.isd.scistore.FindPatientItem;

import static com.pkb.user.entity.PKBPerson.DOB_FORMATTER;

public class SciStoreMismatch {

    private String CHINumber;

    private PKBPerson pkbPatient;

    private FindPatientItem sciStorePatient;

    public String getCHINumber() {
        return CHINumber;
    }

    public void setCHINumber(String CHINumber) {
        this.CHINumber = CHINumber;
    }

    public PKBPerson getPkbPatient() {
        return pkbPatient;
    }

    public void setPkbPatient(PKBPerson pkbPatient) {
        this.pkbPatient = pkbPatient;
    }

    public FindPatientItem getSciStorePatient() {
        return sciStorePatient;
    }

    public void setSciStorePatient(FindPatientItem sciStorePatient) {
        this.sciStorePatient = sciStorePatient;
    }

    public String getSciStorePatientDob() {
        return DOB_FORMATTER.format(sciStorePatient.getDateOfBirth().toGregorianCalendar().toInstant());
    }

}
