package com.pkb.integration.sciStore;

import java.util.List;

import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;

import uk.nhs.scot.show.isd.scistore.InvestigationReportMessageType;

@Component
public class SciStoreModelUtils {

    public static String getUID(InvestigationReportMessageType report) {
        return report.getDiscipline().trim() + "/" + report.getReportIdentifier().trim();
    }

    public static String paragraph(List<String> comments) { // TODO - should we use html or just ctrl chars?
        StringBuilder sb = new StringBuilder();
        for (String comment: comments) {
            sb.append(StringUtils.stripEnd(comment, null/*default: strips whitespace*/));
            sb.append("\n");
        }
        return sb.toString();
    }
}
