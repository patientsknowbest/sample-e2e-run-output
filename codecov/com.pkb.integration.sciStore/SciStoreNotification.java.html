package com.pkb.integration.sciStore;

import com.pkb.entities.enums.ValidNationalId;

public class SciStoreNotification {

    private String CHINumber;
    private String recordKey;

    public String getCHINumber() {
        return CHINumber;
    }

    public void setCHINumber(ValidNationalId nationalId) {
        CHINumber = nationalId.getValue();
    }

    public void setCHINumber(String cHINumber) {
        CHINumber = cHINumber;
    }

    public String getRecordKey() {
        return recordKey;
    }

    public void setRecordKey(String recordKey) {
        this.recordKey = recordKey;
    }

    public static SciStoreNotificationBuilder builder() {
        return new SciStoreNotificationBuilder();
    }

    public static class SciStoreNotificationBuilder {
        private SciStoreNotification entity;

        public SciStoreNotificationBuilder() {
            entity = new SciStoreNotification();
        }

        public SciStoreNotificationBuilder withCHINumber(String cHINumber) {
            entity.setCHINumber(cHINumber);
            return this;
        }

        public SciStoreNotificationBuilder withRecordKey(String recordKey) {
            entity.setRecordKey(recordKey);
            return this;
        }

        public SciStoreNotification build() {
            return entity;
        }
    }

}
