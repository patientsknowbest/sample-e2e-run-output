package com.pkb.integration.sciStore;

// Subset of FindPatientItem message

import javax.xml.datatype.DatatypeConfigurationException;
import javax.xml.datatype.DatatypeFactory;

import uk.nhs.scot.show.isd.scistore.FindPatientItem;

public class SciStorePatient {

    private String sciStoreId;
    private String CHINumber;

    private String name;
    private String givenName;
    private String middleName;
    private String familyName;
    private String sex;
    private String dobString;
    private String address;
    private String postcode;

    private static final DatatypeFactory datatypeFactory;

    static {
        try {
            datatypeFactory = DatatypeFactory.newInstance();
        } catch (DatatypeConfigurationException e) {
            throw new RuntimeException(e);
        }
    }

    public SciStorePatient() {
    }

    // Helpers until we get rid of this class completely

    public SciStorePatient(FindPatientItem source) {
        this.sciStoreId = source.getPatientID();
        this.CHINumber = source.getCHI();
        this.name = source.getName();
        this.givenName = source.getGivenName();
        this.middleName = source.getMiddleName();
        this.familyName = source.getFamilyName();
        this.sex = source.getSex();
        this.dobString = source.getDateOfBirth().toXMLFormat();
        this.address = source.getAddress();
        this.postcode = source.getPostCode();
    }

    public FindPatientItem toPatientItem() {
        FindPatientItem patientItem = new FindPatientItem();
        patientItem.setCHI(CHINumber);
        patientItem.setFamilyName(familyName);
        patientItem.setPostCode(postcode);
        patientItem.setPatientID(sciStoreId);
        patientItem.setMiddleName(middleName);
        patientItem.setGivenName(givenName);
        patientItem.setSex(sex);
        patientItem.setDateOfBirth(datatypeFactory.newXMLGregorianCalendar(dobString));
        return patientItem;
    }
