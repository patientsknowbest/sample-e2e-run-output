package com.pkb.integration.sciStore;

import java.util.Date;

public class SciStoreTestSummary {

    private String discipline;
    private String testReportId; // Id used to fetch the report from scistore
    private String testReportIdentifier; // discipline+"/"+testReportIdentifier is the same across all version of this report
    private Date dateSampled;
    private String description;
    private boolean normal;

    public String getTestReportId() {
        return testReportId;
    }

    public void setTestReportId(String testReportId) {
        this.testReportId = testReportId;
    }

    public String getDiscipline() { return discipline; }

    public void setDiscipline(String discipline) { this.discipline = discipline; }

    public Date getDateSampled() {
        return dateSampled;
    }

    public void setDateSampled(Date dateSampled) {
        this.dateSampled = dateSampled;
    }

    public String getDescription() {
        return description;
    }

    public void setDescription(String description) {
        this.description = description;
    }

    public boolean isNormal() {
        return normal;
    }

    public void setNormal(boolean normal) {
        this.normal = normal;
    }

    public String getTestReportIdentifier() {
        return testReportIdentifier;
    }

    public void setTestReportIdentifier(String testReportIdentifier) {
        this.testReportIdentifier = testReportIdentifier;
    }

    public static SciStoreTestSummaryBuilder builder() {
        return new SciStoreTestSummaryBuilder();
    }

    public static class SciStoreTestSummaryBuilder {
        private SciStoreTestSummary entity;

        public SciStoreTestSummaryBuilder() {
            entity = new SciStoreTestSummary();
        }

        public SciStoreTestSummaryBuilder withDiscipline(String discipline) {
            entity.setDiscipline(discipline);
            return this;
        }

        public SciStoreTestSummaryBuilder withTestReportId(String testReportId) {
            entity.setTestReportId(testReportId);
            return this;
        }

        public SciStoreTestSummaryBuilder withTestReportIdentifier(String testReportIdentifier) {
            entity.setTestReportIdentifier(testReportIdentifier);
            return this;
        }

        public SciStoreTestSummaryBuilder withDateSampled(Date dateSampled) {
            entity.setDateSampled(dateSampled);
            return this;
        }

        public SciStoreTestSummaryBuilder withDescription(String description) {
            entity.setDescription(description);
            return this;
        }

        public SciStoreTestSummaryBuilder withNormal(boolean normal) {
            entity.setNormal(normal);
            return this;
        }

        public SciStoreTestSummary build() {
            return entity;
        }
    }
