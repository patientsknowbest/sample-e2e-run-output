package com.pkb.phoenix.api.wiring.config;

import com.pkb.common.config.ConfigStorage;
import com.pkb.common.testcontrol.services.DefaultPubSubNamespaceService;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.phoenix.api.service.task.template.TaskResponseInboundRouteBuilder;
import com.pkb.phoenix.api.service.task.template.emaildeletion.TaskResponseListener;
import org.apache.camel.CamelContext;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class TaskPubSubConfig {

    @Bean
    public PubSubNamespaceService pubsubNamespaceService(CamelContext camelContext) {
        return new DefaultPubSubNamespaceService(camelContext);
    }

    @Bean
    public PhoenixPubSubConfig pubSubConfig(ConfigStorage configStorage) {
        return () -> configStorage;
    }

    @Bean
    TaskResponseInboundRouteBuilder taskResponseRouteBuilder(PubSubNamespaceService namespaceService,
                                                             PhoenixPubSubConfig config,
                                                             TaskResponseListener processor,
                                                             @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey) {
        return new TaskResponseInboundRouteBuilder(namespaceService, config, processor, serviceAccountKey);
    }
}
