package com.pkb.entities.doc;

import com.google.common.base.MoreObjects;
import com.pkb.entities.app.AppMenuData;
import com.pkb.entities.core.CoreAccount;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Type;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;
import java.util.UUID;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(schema = "doc", name = "attachment")
public class DocAttachment implements Serializable {

    private static final long serialVersionUID = -6064121547855039216L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Id
    @GenericGenerator(name = "seq_doc_gen", strategy = "enhanced-sequence", parameters = {
            @org.hibernate.annotations.Parameter(name = "optimizer", value = OPTIMIZER_CANONICAL_NAME),
            @org.hibernate.annotations.Parameter(name = "schema", value = "doc"),
            @org.hibernate.annotations.Parameter(name = "sequence_name", value = "seq_doc"),
            @org.hibernate.annotations.Parameter(name = "increment_size", value = "50")
    })
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_doc_gen")
    private Long id;

    @JoinColumn(name = "account_id")
    @ManyToOne(fetch = FetchType.LAZY)
    private CoreAccount account;

    @Type(type = "pg-uuid")
    @Column(name = "parent_document_metadata_id")
    private UUID parentDocumentMetadataId;

    @JoinColumn(name = "menudata_id")
    @ManyToOne(fetch = FetchType.LAZY)
    private AppMenuData menuData;

    @Type(type = "pg-uuid")
    @Column(name = "attached_document_metadata_id")
    private UUID attachedDocumentMetadataId;

    @Column
    private String description;

    @Column(name = "upload_time")
    private Instant uploadTime;

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("id", id)
                .add("description", description)
                .add("uploadTime", uploadTime)
                .toString();
    }

    public static long getSerialVersionUID() {
        return serialVersionUID;
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public CoreAccount getAccount() {
        return account;
    }

    public void setAccount(CoreAccount account) {
        this.account = account;
    }

    public UUID getParentDocumentMetadataId() {
        return parentDocumentMetadataId;
    }

    public void setParentDocumentMetadataId(UUID parentDocumentMetadataId) {
        this.parentDocumentMetadataId = parentDocumentMetadataId;
    }

    public AppMenuData getMenuData() {
        return menuData;
    }

    public void setMenuData(AppMenuData menuData) {
        this.menuData = menuData;
    }

    public UUID getAttachedDocumentMetadataId() {
        return attachedDocumentMetadataId;
    }

    public void setAttachedDocumentMetadataId(UUID attachedDocumentMetadataId) {
        this.attachedDocumentMetadataId = attachedDocumentMetadataId;
    }

    public String getDescription() {
        return description;
    }

    public void setDescription(String description) {
        this.description = description;
    }

    public Instant getUploadTime() {
        return uploadTime;
    }

    public void setUploadTime(Instant uploadTime) {
        this.uploadTime = uploadTime;
    }
}
