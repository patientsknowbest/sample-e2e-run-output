package com.pkb.entities.doc;

import com.pkb.datamodel.Email;
import com.pkb.entities.embeddables.EncodedEncryptedPrivateKeyInfoWithSecretDeriv;
import com.pkb.entities.enums.InvitationType;
import com.pkb.entities.enums.PrivacyFlag;
import org.hibernate.annotations.Type;
import org.jetbrains.annotations.Nullable;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.Instant;
import java.util.Collection;
import java.util.EnumSet;
import java.util.Optional;
import java.util.UUID;

import static com.pkb.datamodel.ImmutableDisclosingEmail.disclosingEmail;

@Entity
@Table(schema = "doc", name = "invitation")
public class DocInvitation {

    @Id
    @SuppressWarnings("deprecation")
    @Deprecated // TODO remove in PHR-10242
    @GeneratedValue(strategy = GenerationType.AUTO)
    @Column(name = "id", nullable = false)
    private Long id;

    //@Id
    @Column(name = "public_id", nullable = false, updatable = false, unique = true)
    @Type(type = "pg-uuid")
    private UUID publicId;


    @Column(name = "title")
    private String invitationType;


    @Column(name = "created")
    private Instant created;


    @Column(name = "status")
    private String status;

    @Column(name = "target_email")
    private String targetEmail;

    @Column(name = "authorPerson_id")
    private Long authorPersonId;

    @Column(name = "targetPerson_id")
    private Long targetPersonId;

    @Column(name = "account_id")
    private Long accountId;

    /**
     * This can be null. If it's null, the invitation will not create any sort of consent
     * record when it is accepted. Reasons why this might be desirable include: a) this is a
     * legacy invitation where consent was created at the point of invitation or b) this
     * is one of the invitations that shouldn't really be an invitation - like an invitation to
     * a clinician to upgrade their account.
     */
    @Column(name="consent_bits")
    private Integer consentBits;

    @Embedded
    private EncodedEncryptedPrivateKeyInfoWithSecretDeriv accountPrivateKey;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public UUID getPublicId() {
        return publicId;
    }

    public void setPublicId(UUID publicId) {
        this.publicId = publicId;
    }


    public String getDocInvitationType() {
        return invitationType;
    }

    public void setDocInvitationType(InvitationType invitationType, String addUserId) {
        this.invitationType = invitationType.description + "#UTA=" + addUserId;
    }


    public Instant getCreated() {
        return created;
    }

    public void setCreated(Instant created) {
        this.created = created;
    }


    public String getStatus() {
        return status;
    }

    public void setStatus(String status) {
        this.status = status;
    }

    public @Nullable Email getTargetEmail() {
        if (targetEmail != null && !targetEmail.isBlank()) {
            return disclosingEmail(targetEmail);
        } else {
            return null;
        }
    }

    public void setTargetEmail(Email targetEmail) {
        this.targetEmail = targetEmail.address();
    }


    public EncodedEncryptedPrivateKeyInfoWithSecretDeriv getAccountPrivateKey() {
        return accountPrivateKey;
    }

    public void setAccountPrivateKey(EncodedEncryptedPrivateKeyInfoWithSecretDeriv accountPrivateKey) {
        this.accountPrivateKey = accountPrivateKey;
    }

    public Long getAuthorPersonId() {
        return authorPersonId;
    }

    public void setAuthorPersonId(Long authorPersonId) {
        this.authorPersonId = authorPersonId;
    }

    public Long getTargetPersonId() {
        return targetPersonId;
    }

    public void setTargetPersonId(Long targetPersonId) {
        this.targetPersonId = targetPersonId;
    }

    public Long getAccountId() {
        return accountId;
    }

    public void setAccountId(Long accountId) {
        this.accountId = accountId;
    }

    public Optional<EnumSet<PrivacyFlag>> getGrantedConsents() {
        return Optional.ofNullable(this.consentBits).map(PrivacyFlag::fromBitFlags);
    }

    public void setGrantedConsents(@Nullable Collection<PrivacyFlag> consents) {
        consentBits = Optional.ofNullable(consents).map(PrivacyFlag::toBitFlags).orElse(null);
    }

}
