package com.pkb.entities.doc;

import com.pkb.entities.enums.DocStorageBackend;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.Instant;
import java.util.UUID;

@Entity
@Table(schema = "doc", name = "document_metadata")
public class DocDocumentMetadata {

    @Id
    @Column(name = "id")
    @Type(type = "pg-uuid")
    private UUID id;

    @Column(name = "storage")
    @Enumerated(EnumType.STRING)
    @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
    private DocStorageBackend storage;

    @Column(name = "id_in_storage")
    private String idInStorage;

    @Column(name = "account_id")
    private long accountId;

    @Column(name = "author_person_id")
    private Long authorPersonId;

    @Column(name = "media_type")
    private String mediaType;

    @Column(name = "finalsubmittime")
    private Instant finalSubmitTime;

    @Column(name = "encrypted_secret_key_inline")
    private byte[] encryptedKeyInline;

    @Column(name = "secret_key_algorithm")
    private String algorithm;

    @Column(name = "symmetric_encryption_key_id")
    private Long symmetricEncryptionKeyId;

    @Column(name = "nonce")
    private String nonce;

    @Column(name = "datapoint_level_key_id")
    @Type(type = "pg-uuid")
    private UUID datapointLevelKeyId;

    @Column(name = "datapoint_level_encrypted")
    private boolean datapointLevelEncrypted;

    public UUID getId() {
        return id;
    }

    public void setId(UUID id) {
        this.id = id;
    }

    public DocStorageBackend getStorage() {
        return storage;
    }

    public void setStorage(DocStorageBackend storage) {
        this.storage = storage;
    }

    public String getIdInStorage() {
        return idInStorage;
    }

    public void setIdInStorage(String idInStorage) {
        this.idInStorage = idInStorage;
    }

    public long getAccountId() {
        return accountId;
    }

    public void setAccountId(long accountId) {
        this.accountId = accountId;
    }

    public Long getAuthorPersonId() {
        return authorPersonId;
    }

    public void setAuthorPersonId(Long authorPersonId) {
        this.authorPersonId = authorPersonId;
    }

    public String getMediaType() {
        return mediaType;
    }

    public void setMediaType(String mediaType) {
        this.mediaType = mediaType;
    }

    public Instant getFinalSubmitTime() {
        return finalSubmitTime;
    }

    public void setFinalSubmitTime(Instant finalSubmitTime) {
        this.finalSubmitTime = finalSubmitTime;
    }

    public byte[] getEncryptedKeyInline() {
        return encryptedKeyInline;
    }

    public void setEncryptedKeyInline(byte[] encryptedKeyInline) {
        this.encryptedKeyInline = encryptedKeyInline;
    }

    public String getAlgorithm() {
        return algorithm;
    }

    public void setAlgorithm(String algorithm) {
        this.algorithm = algorithm;
    }

    public Long getSymmetricEncryptionKeyId() {
        return symmetricEncryptionKeyId;
    }

    public void setSymmetricEncryptionKeyId(Long symmetricEncryptionKeyId) {
        this.symmetricEncryptionKeyId = symmetricEncryptionKeyId;
    }

    public String getNonce() {
        return nonce;
    }

    public void setNonce(String nonceBase64) {
        this.nonce = nonceBase64;
    }

    public UUID getDatapointLevelKeyId() {
        return datapointLevelKeyId;
    }

    public void setDatapointLevelKeyId(UUID datapointLevelKeyId) {
        this.datapointLevelKeyId = datapointLevelKeyId;
    }

    public boolean isDatapointLevelEncrypted() {
        return datapointLevelEncrypted;
    }

    public void setDatapointLevelEncrypted(boolean datapointLevelEncrypted) {
        this.datapointLevelEncrypted = datapointLevelEncrypted;
    }
}
