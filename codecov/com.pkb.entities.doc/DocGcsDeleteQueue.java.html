package com.pkb.entities.doc;

import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(schema = "doc", name = "gcs_delete_queue")
public class DocGcsDeleteQueue implements Serializable {

    @Id
    @GenericGenerator(name = "seq_gcs_delete_queue_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "doc"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "gcs_delete_queue_id_seq"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(
            strategy = GenerationType.SEQUENCE,
            generator = "seq_gcs_delete_queue_gen")
    private Long id;

    @Column(name = "id_in_store")
    private String idInStore;

    @Column(name = "deleted_date")
    private Instant deletedDate;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getIdInStore() {
        return idInStore;
    }

    public void setIdInStore(String idInStore) {
        this.idInStore = idInStore;
    }

    public Instant getDeletedDate() {
        return deletedDate;
    }

    public void setDeletedDate(Instant deletedDate) {
        this.deletedDate = deletedDate;
    }
}
