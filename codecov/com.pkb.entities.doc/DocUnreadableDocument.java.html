package com.pkb.entities.doc;

import com.google.common.base.MoreObjects;
import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import com.vladmihalcea.hibernate.type.json.JsonBinaryType;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Type;
import org.hibernate.annotations.TypeDef;
import org.jetbrains.annotations.NotNull;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.OneToOne;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;
import java.util.UUID;

@Entity
@Table(schema = "doc", name = "unreadabledocument")
@TypeDef(
        name = "jsonb",
        typeClass = JsonBinaryType.class
)
public class DocUnreadableDocument implements Serializable {

    @Id
    @GenericGenerator(name = "seq_doc_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "doc"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_doc"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_doc_gen")
    private Long id;

    @Column(name = "person_public_id", nullable = false)
    private UUID personPublicId;

    @Column(name = "external_message_id")
    private String externalMessageId;

    @JoinColumn(name = "attachment_id", nullable = false)
    @OneToOne(fetch = FetchType.LAZY)
    private DocAttachment attachment;

    @Column(nullable = false)
    private Instant reported;

    @Column(name = "number_of_bytes", nullable = false)
    long numberOfBytes;

    @Column(name="number_of_pages", nullable = false)
    long numberOfPages;

    @Column(name="claimed_format", nullable = false)
    String claimedFormat;

    @Column(name="is_encrypted", nullable = false)
    boolean isEncrypted;

    @Column(name="is_pdfa1b", nullable = false)
    boolean isPDFa1b;

    @Column(name="parse_report", nullable = false)
    String parseReport;

    @Type(type="jsonb")
    @Column(name="metadata", columnDefinition="jsonb", nullable=false)
    String metadata;

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("id", id)
                .add("personPublicId", personPublicId)
                .add("externalMessageId", externalMessageId)
                .add("reported", reported)
                .add("numberOfBytes", numberOfBytes)
                .add("numberOfPages", numberOfPages)
                .add("claimedFormat",claimedFormat)
                .add("isEncrypted", isEncrypted)
                .add("isPDFa1b",isPDFa1b)
                .add("parseReport",parseReport)
                .add("metadata",metadata)
                .toString();
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public UUID getPersonPublicId() {
        return personPublicId;
    }

    public void setPersonPublicId(UUID personPublicId) {
        this.personPublicId = personPublicId;
    }

    public String getExternalMessageId() {
        return externalMessageId;
    }

    public void setExternalMessageId(String externalMessageId) {
        this.externalMessageId = externalMessageId;
    }

    public DocAttachment getAttachment() {
        return attachment;
    }

    public void setAttachment(DocAttachment attachment) {
        this.attachment = attachment;
    }

    public Instant getReported() {
        return reported;
    }

    public void setReported(Instant reported) {
        this.reported = reported;
    }

    public long getNumberOfBytes() {
        return numberOfBytes;
    }

    public void setNumberOfBytes(long numberOfBytes) {
        this.numberOfBytes = numberOfBytes;
    }

    public long getNumberOfPages() {
        return numberOfPages;
    }

    public void setNumberOfPages(long numberOfPages) {
        this.numberOfPages = numberOfPages;
    }

    public @NotNull String getClaimedFormat() {
        return claimedFormat;
    }

    public void setClaimedFormat(@NotNull String claimedFormat) {
        this.claimedFormat = claimedFormat;
    }

    public boolean isEncrypted() {
        return isEncrypted;
    }

    public void setEncrypted(boolean encrypted) {
        isEncrypted = encrypted;
    }

    public boolean isPDFa1b() {
        return isPDFa1b;
    }

    public void setPDFa1b(boolean PDFa1b) {
        isPDFa1b = PDFa1b;
    }

    public @NotNull String getParseReport() {
        return parseReport;
    }

    public void setParseReport(@NotNull String parseReport) {
        this.parseReport = parseReport;
    }

    public @NotNull String getMetadata() {
        return metadata;
    }

    public void setMetadata(@NotNull String metadata) {
        this.metadata = metadata;
    }
}
