package com.pkb.api.v1.model;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.fasterxml.jackson.databind.ser.std.ToStringSerializer;
import org.apache.commons.lang3.ArrayUtils;
import org.jetbrains.annotations.Nullable;

import javax.xml.bind.annotation.XmlRootElement;
import java.io.Serializable;
import java.util.ArrayList;
import java.util.Date;
import java.util.List;

/**
 * A treatment plan; id is unique per plan (shared by all versions). Fields are label and text. Current labels are ACTION_PLAN, GREEN,
 * AMBER, RED.
 */
@XmlRootElement(name = "TreatmentPlan")
public class ApiTreatmentPlan implements Serializable {

    private static final long serialVersionUID = -1L;

    private String id;

    private Long patientId;

    private Long version;

    private String name;

    private Date lastEditedOn;

    private Date creationDate;

    private Date enteredDate;

    private Date persistedDate;

    private String lastEditedOnISO;

    private String description; //FHIR attribute

    private String note; //FHIR attribute; Comments about the plan

    private List<Field> fields;

    private List<ApiAttachment> attachments;

    private List<ApiMetric> symptoms;

    private List<ApiMetric> measurements;

    private List<ApiMetric> tests;

    public List<ApiAttachment> getAttachments() {
        return attachments;
    }

    public void setAttachments(List<ApiAttachment> attachments) {
        this.attachments = attachments;
    }

    public List<ApiMetric> getSymptoms() {
        return symptoms;
    }

    public void setSymptoms(List<ApiMetric> symptoms) {
        this.symptoms = symptoms;
    }

    public List<ApiMetric> getMeasurements() {
        return measurements;
    }

    public void setMeasurements(List<ApiMetric> measurements) {
        this.measurements = measurements;
    }

    public List<ApiMetric> getTests() {
        return tests;
    }

    public void setTests(List<ApiMetric> tests) {
        this.tests = tests;
    }

    // debugging util
    @Override
    public String toString() {

        // return "treatmentPlan";
        String s = "[ApiTreatmentPlan \"" + name + "\": id " + id +
                " (version " + getVersion() + ", patientId " + patientId +
                " edit @ " + getLastEditedOnISO() +
                "), fields " + ArrayUtils.toString(fields.toArray()) + "]";
        return s;
    }

    public void addField(String label, String text) {
        if (fields == null) {
            fields = new ArrayList<>();
        }

        Field f = new Field();
        f.setLabel(label);
        f.setText(text);

        fields.add(f);
    }

    public @Nullable Field fieldForLabel(String label) {
        if (fields == null) {
            return null;
        }

        for (Field f : fields) {
            if (label.equals(f.getLabel())) {
                return f;
            }
        }
        return null; // not found
    }

    public String getLastEditedOnISO() {
        return lastEditedOnISO;
    }

    public void setLastEditedOnISO(String lastEditedOnISO) {
        this.lastEditedOnISO = lastEditedOnISO;
    }

    public static class Field {
        private String label;
        private String text;

        @Override
        public String toString() {
            return "[Field (" + label + "): " + text + "]";
        }

        public String getLabel() {
            return label;
        }

        public void setLabel(String label) {
            this.label = label;
        }

        public String getText() {
            return text;
        }

        public void setText(String text) {
            this.text = text;
        }
    }

    // getters/setters

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public Long getVersion() {
        return version;
    }

    public void setVersion(Long version) {
        this.version = version;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public Date getLastEditedOn() {
        return lastEditedOn;
    }

    public void setLastEditedOn(Date lastEditedOn) {
        this.lastEditedOn = lastEditedOn;
    }

    public List<Field> getFields() {
        return fields;
    }

    public void setFields(List<Field> fields) {
        this.fields = fields;
    }

    @JsonSerialize(using= ToStringSerializer.class)
    public Long getPatientId() {
        return patientId;
    }

    @JsonDeserialize(as = Long.class)
    public void setPatientId(Long patientId) {
        this.patientId = patientId;
    }

    public Date getCreationDate() {
        return creationDate;
    }

    public void setCreationDate(Date creationDate) {
        this.creationDate = creationDate;
    }

    public String getDescription() {
        return description;
    }

    public void setDescription(String description) {
        this.description = description;
    }

    public Date getEnteredDate() {
        return enteredDate;
    }

    public void setEnteredDate(Date enteredDate) {
        this.enteredDate = enteredDate;
    }

    public Date getPersistedDate() {
        return persistedDate;
    }

    public void setPersistedDate(Date persistedDate) {
        this.persistedDate = persistedDate;
    }

    public String getNote() {
        return note;
    }

    public void setNote(String note) {
        this.note = note;
    }

}
