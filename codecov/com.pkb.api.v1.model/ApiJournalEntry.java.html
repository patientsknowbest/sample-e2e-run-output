package com.pkb.api.v1.model;

import java.util.Date;

/**
 * Represents a journal entry. The term "journal" is used instead of "diary"
 * for clarity. See #1521 for more information.
 *
 * @author Tom
 *
 */
public class ApiJournalEntry {

    private String id;

    private Date date;

    private String authorId;

    private String subjectOfEntry;

    private String entryTextPlain;

    private String entryTextHtml;

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public Date getDate() {
        return date;
    }

    public void setDate(Date date) {
        this.date = date;
    }

    public String getEntryTextPlain() {
        return entryTextPlain;
    }

    public void setEntryTextPlain(String entryTextPlain) {
        this.entryTextPlain = entryTextPlain;
    }

    public String getEntryTextHtml() {
        return entryTextHtml;
    }

    public void setEntryTextHtml(String entryTextHtml) {
        this.entryTextHtml = entryTextHtml;
    }

    public String getSubjectOfEntry() {
        return subjectOfEntry;
    }

    public void setSubjectOfEntry(String subjectOfEntry) {
        this.subjectOfEntry = subjectOfEntry;
    }

    public String getAuthorId() {
        return authorId;
    }

    public void setAuthorId(String authorId) {
        this.authorId = authorId;
    }
}
