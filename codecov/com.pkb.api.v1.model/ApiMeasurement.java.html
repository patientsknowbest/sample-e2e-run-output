package com.pkb.api.v1.model;

import com.fasterxml.jackson.annotation.JsonProperty;
import com.pkb.test.entity.PredefinedMeasurementType;
import org.jetbrains.annotations.Nullable;

import javax.xml.bind.annotation.XmlRootElement;
import java.io.Serializable;
import java.time.LocalDate;

@XmlRootElement(name = "Measurement")
public class ApiMeasurement implements Serializable {

    private static final long serialVersionUID = -1L;

    private String id;

    private String measurementTypeApiRef;

    private LocalDate measureDate;

    private String measureDateISO;

    private Double value;

    private Double value2;

    // debugging util
    @Override
    public String toString() {
        String value2String = (value2 == null) ? "" : "/" + value2;
        return "[Measurement: id " + id +
                ", type " + measurementTypeApiRef + " @ " + measureDate +
                ", " + value + value2String + "]";
    }

    protected @Nullable String getValueFormatted() {
        PredefinedMeasurementType mt = PredefinedMeasurementType.getByApiRef(measurementTypeApiRef);
        if (mt != null) {
            return mt.getType().format(value);
        } else {
            return null;
        }
    }

    protected @Nullable String getValue2Formatted() {
        PredefinedMeasurementType mt = PredefinedMeasurementType.getByApiRef(measurementTypeApiRef);
        if (mt != null) {
            return mt.getType().format2(value, value2);
        } else {
            return null;
        }
    }

    public String getMeasureDateISO() {
        return measureDateISO;
    }

    public void setMeasureDateISO(String measureDateISO) {
        this.measureDateISO = measureDateISO;
    }

    // getters/setters

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public LocalDate getMeasureDate() {
        return measureDate;
    }

    public void setMeasureDate(LocalDate measureDate) {
        this.measureDate = measureDate;
    }

    public Double getValue() {
        return value;
    }

    public void setValue(Double value) {
        this.value = value;
    }

    public Double getValue2() {
        return value2;
    }

    public void setValue2(Double value2) {
        this.value2 = value2;
    }

    @Deprecated
    @JsonProperty("apiRef")
    public String getMeasurementTypeApiRef() {
        return measurementTypeApiRef;
    }

    @Deprecated
    public void setMeasurementTypeApiRef(String measurementTypeApiRef) {
        this.measurementTypeApiRef = measurementTypeApiRef;
    }
}
