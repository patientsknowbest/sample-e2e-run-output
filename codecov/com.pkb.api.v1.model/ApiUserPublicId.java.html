package com.pkb.api.v1.model;

import javax.xml.bind.annotation.XmlRootElement;
import java.util.UUID;

/**
 * https://pkbdev.atlassian.net/browse/PHR-7653
 * ApiUser is extended with publicId used in /v1/users/self/patients
 */
@XmlRootElement(name = "User")
public class ApiUserPublicId extends ApiUser {

    public ApiUserPublicId(ApiUser user) {
        this.setId(user.getId());
        this.setDob(user.getDob());
        this.setEmail(user.getEmail());
        this.setPhone(user.getPhone());
        this.setFirstName(user.getFirstName());
        this.setLastName(user.getLastName());
        this.setGenderMF(user.getGenderMF());
        this.setGender(user.getGender());
        this.setTitle(user.getTitle());
        this.setDateDeceased(user.getDateDeceased());
        this.setAddress1(user.getAddress1());
        this.setAddress2(user.getAddress2());
        this.setCity(user.getCity());
        this.setState(user.getState());
        this.setPostalCode(user.getPostalCode());
        this.setCountry(user.getCountry());
        this.setPatient(user.isPatient());
        this.setJobTitle(user.getJobTitle());
        this.setOrganizationName(user.getOrganizationName());
        this.setTeamName(user.getTeamName());
        this.setHeightCm(user.getHeightCm());
        this.setHeightFeet(user.getHeightFeet());
        this.setHeightInches(user.getHeightInches());
        this.setHeightInchesTotal(user.getHeightInchesTotal());
        this.setWeightKg(user.getWeightKg());
        this.setWeightLbs(user.getWeightLbs());
        this.setContacts(user.getContacts());
    }

    public ApiUserPublicId() {
    }

    private UUID publicId;

    public UUID getPublicId() {
        return publicId;
    }

    public void setPublicId(UUID publicId) {
        this.publicId = publicId;
    }
}
