package com.pkb.api.v1.model;

import io.swagger.annotations.ApiModelProperty;

import javax.xml.bind.annotation.XmlRootElement;
import java.io.Serializable;
import java.util.List;

/**
 * This represents a LOINC test. It contains details about the test and a list of results for the test.
 *
 *
 */
@XmlRootElement(name = "Test")
public class ApiTest implements Serializable {

    private static final long serialVersionUID = -1L;

    @ApiModelProperty(value = "A PKB-assigned ID which identifies this particular test type.")
    private String id;

    @ApiModelProperty(value = "The test code.")
    private String code;

    @ApiModelProperty(value = "A human readable name for this test.")
    private String name;

    @ApiModelProperty(value = "A list of results for this test.")
    private List<ApiTestResult> testResults;

    @ApiModelProperty(value = "The units in which the result was measured.")
    private String unit;

    @ApiModelProperty(value = "The coding system from which the test code was taken.")
    private String codeSystem;

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public String getCode() {
        return code;
    }

    public void setCode(String loincCode) {
        this.code = loincCode;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public List<ApiTestResult> getTestResults() {
        return testResults;
    }

    public void setTestResults(List<ApiTestResult> testResults) {
        this.testResults = testResults;
    }

    public String getUnit() {
        return unit;
    }

    public void setUnit(String unit) {
        this.unit = unit;
    }

    public String getCodeSystem() {
        return codeSystem;
    }

    public void setCodeSystem(String codeSystem) {
        this.codeSystem = codeSystem;
    }

}
