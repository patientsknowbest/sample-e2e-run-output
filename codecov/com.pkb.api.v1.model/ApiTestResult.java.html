package com.pkb.api.v1.model;

import io.swagger.annotations.ApiModelProperty;

import javax.xml.bind.annotation.XmlRootElement;
import java.io.Serializable;
import java.util.Date;

/**
 * Represents a result of a loinc test (ie an individual result at a particular date and time).
 */
@XmlRootElement(name = "TestResult")
public class ApiTestResult implements Serializable {

    private static final long serialVersionUID = 1L;

    @ApiModelProperty(value = "A PKB-assigned ID which identifies this particular test result.")
    private String id;

    @ApiModelProperty(value = "The clinically relevant timestamp of the result.")
    private Date resultDate;

    @ApiModelProperty(value = "The value of the result. It may be either numeric or textual.")
    private String value;

    @ApiModelProperty(value = "Any comments accompanying the result value. If this result was a single textual report then there will be only comments (no value).")
    private String comments;

    @ApiModelProperty(value = "The upper limit of the reference range, if the range was numeric.")
    private Double rangeHigh;

    @ApiModelProperty(value = "The lower limit of the reference range, if the range was numeric.")
    private Double rangeLow;

    public Date getResultDate() {
        return resultDate;
    }

    public void setResultDate(Date resultDate) {
        this.resultDate = resultDate;
    }

    public String getValue() {
        return value;
    }

    public void setValue(String value) {
        this.value = value;
    }

    public Double getRangeHigh() {
        return rangeHigh;
    }

    public void setRangeHigh(Double rangeHigh) {
        this.rangeHigh = rangeHigh;
    }

    public Double getRangeLow() {
        return rangeLow;
    }

    public void setRangeLow(Double rangeLow) {
        this.rangeLow = rangeLow;
    }

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public String getComments() {
        return comments;
    }

    public void setComments(String comments) {
        this.comments = comments;
    }

}
