package com.pkb.api.v1.model;

import java.io.Serializable;

/**
 * This class is a lightweight version of users for use with messaging. It contains the minimum needed for most messaging functionality.
 *
 * The ID should match the ID of the equivalent ApiUser so that the API can be used to get more details on these Api MessageUsers are
 * required
 *
 */
public class ApiMessageUser implements Serializable {

    private static final long serialVersionUID = 1L;

    private String id;
    private String firstName = "";
    private String lastName = "";
    private String title = "";
    private String teamName = "";
    private String jobTitle = "";

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public String getFirstName() {
        return firstName;
    }

    public void setFirstName(String name) {
        this.firstName = name;
    }

    public String getTitle() {
        return title;
    }

    public void setTitle(String title) {
        this.title = title;
    }

    public String getTeamName() {
        return teamName;
    }

    public void setTeamName(String teamName) {
        this.teamName = teamName;
    }

    public String getJobTitle() {
        return jobTitle;
    }

    public void setJobTitle(String jobTitle) {
        this.jobTitle = jobTitle;
    }

    public String getLastName() {
        return lastName;
    }

    public void setLastName(String lastName) {
        this.lastName = lastName;
    }

}
