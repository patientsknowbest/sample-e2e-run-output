package com.pkb.api.v1.model;

import javax.xml.bind.annotation.XmlRootElement;
import java.io.Serializable;
import java.util.List;

@XmlRootElement(name = "Conversation")
public class ApiConversation implements Serializable {

    private static final long serialVersionUID = -1L;

    private String id;
    private List<ApiMessage> messages;
    private List<ApiMessageUser> participants;
    private Long latestMessageId;
    private Long messageCount;

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public List<ApiMessage> getMessages() {
        return messages;
    }

    public void setMessages(List<ApiMessage> messages) {
        this.messages = messages;
    }

    public List<ApiMessageUser> getParticipants() {
        return participants;
    }

    public void setParticipants(List<ApiMessageUser> participants) {
        this.participants = participants;
    }

    public Long getLatestMessageId() {
        return latestMessageId;
    }

    public void setLatestMessageId(Long id) {
        this.latestMessageId = id;
    }

    public Long getMessageCount() {
        return messageCount;
    }

    public void setMessageCount(Long messageCount) {
        this.messageCount = messageCount;
    }

}
