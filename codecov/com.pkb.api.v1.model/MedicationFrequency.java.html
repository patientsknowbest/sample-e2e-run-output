package com.pkb.api.v1.model;

import com.fasterxml.jackson.annotation.JsonCreator;
import com.fasterxml.jackson.annotation.JsonValue;
import org.jetbrains.annotations.Nullable;

import static com.pkb.util.EnumMatch.match;

public enum MedicationFrequency {

    BLANK("", ""), ONCE_A_DAY("od", "once_a_day"), TWICE_A_DAY("bd", "twice_a_day"), THREE_TIMES_A_DAY("tds", "three_times_a_day"),
    FOUR_TIMES_A_DAY("qds", "four_times_a_day"), EVERY_4_HOURS("q4h", "every_four_hours"), EVERY_OTHER_DAY("qod", "every_other_day"),
    TAKE_AS_NEEDED("prn", "take_as_needed"), TAKE_AS_DIRECTED("ud", "take_as_directed"),
    //reserved for unrecognised input from API:
    UNRECOGNISED("", "unrecognised");

    private final String value;
    private String code;

    MedicationFrequency(String code, String value) {
        this.value = value;
        this.code = code;

    }

    public String getCode() {
        return code;
    }

    @Nullable
    public static MedicationFrequency getFromCode(String code) {
        for (MedicationFrequency frequency : values()) {
            if (frequency.code.equals(code)) {
                return frequency;
            }
        }
        return null;
    }

    @JsonValue
    public String getValue() {
        return value;

    }

    @JsonCreator
    public static MedicationFrequency constructFromValue(String value) {

        return match(MedicationFrequency.values(), value, MedicationFrequency.BLANK, MedicationFrequency.UNRECOGNISED);

    }

    public static String allowedValues() {
        StringBuilder values = new StringBuilder();
        for (MedicationFrequency candidate : MedicationFrequency.values()) {
            if (candidate != MedicationFrequency.UNRECOGNISED) {
                values.append(candidate.getValue());
            }

            values.append(",");
        }
        values.deleteCharAt(values.length() - 1);
        return values.toString();
    }

}
