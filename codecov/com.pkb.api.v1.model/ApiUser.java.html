    /**
     * Indicates whether this user has at least one ID, of any type, with a
     * non-blank value.
     *
     * @return
     */
    public boolean hasAnId() {
        if ((nationalIds != null) && !nationalIds.isEmpty()) {
            for (ApiNationalId nid : nationalIds) {
                if (StringUtils.isNotBlank(nid.getValue())) {
                    return true;
                }
            }
        }
        if ((orgLevelIds != null) && !orgLevelIds.isEmpty()) {
            for (ApiOrgLevelId oid : orgLevelIds) {
                if (StringUtils.isNotBlank(oid.getValue())) {
                    return true;
                }
            }
        }
        if ((teamLevelIds != null) && !teamLevelIds.isEmpty()) {
            for (ApiTeamLevelId tid : teamLevelIds) {
                if (StringUtils.isNotBlank(tid.getValue())) {
                    return true;
                }
            }
        }
        return false;
    }

    // setters for conversions

    public void setHeightValuesCm(Double heightValueCm) {
        if (heightValueCm == null) {
            this.heightCm = null;
            this.heightFeet = null;
            this.heightInches = null;
            this.heightInchesTotal = null;
        } else {
            this.heightCm = integerFormat.format(heightValueCm);
            double inches = heightValueCm / 2.54d;

            this.heightInchesTotal = integerFormat.format(inches);
            // convert back to int for truncating arithmetic
            int inchesInt = Integer.parseInt(heightInchesTotal);

            this.heightFeet = integerFormat.format(inchesInt / 12);
            this.heightInches = integerFormat.format(inchesInt % 12);
        }
    }

    public void setWeightValuesKg(Double weightValueKg) {

        if (weightValueKg == null) {
            this.weightKg = null;
            this.weightLbs = null;
        } else {
            // kg to lbs: value*0.45359237d;
            double lbs = weightValueKg / 0.45359237d;
            this.weightKg = onePlaceValueFormat.format(weightValueKg);
            this.weightLbs = onePlaceValueFormat.format(lbs);
        }
    }

    public void setDob(String dob) {

            // It is not the responsibility of this class to know whether this field is mandatory. Only try to parse if present.
            if (StringUtils.isNotBlank(dob)) {
                DateTimeFormatter dtf = DateTimeFormatter.ofPattern("yyyy-MM-dd");

            throw new RuntimeException("Provided DOB value must be in format: yyyy-MM-dd", dtpe);
        }

        this.dob = dob;
    }
