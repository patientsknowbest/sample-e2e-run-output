package com.pkb.api.v1.model;

/**
 * For referencing another API data type, e.g. a lab result metric inside a care plan.
 * <p>
 * See {@link TemplateMetric} for more information.
 *
 * @author russell
 */
public class ApiMetric {

    private String name;
    private String id;
    private String apiRef;

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public String getApiRef() {
        return apiRef;
    }

    public void setApiRef(String apiRef) {
        this.apiRef = apiRef;
    }
}
