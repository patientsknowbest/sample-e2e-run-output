package com.pkb.kms.web.controller

import com.pkb.kms.service.AccountSymmetricKeyService
import com.pkb.kms.shared.representation.EncryptionRequest
import com.pkb.kms.shared.representation.EncryptionResponse
import com.pkb.kms.web.ErrorException
import io.micrometer.core.annotation.Timed
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.RequestBody
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("encryption")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class EncryptionController(private val accountSymmetricKeyService: AccountSymmetricKeyService) {

    @PostMapping
    fun encrypt(@RequestBody encryptionRequest: EncryptionRequest): EncryptionResponse {
        val documents =
            accountSymmetricKeyService.encrypt(encryptionRequest.accountPublicId, encryptionRequest.documents)
                    .getOrElseThrow { err -> ErrorException(err) }

        return EncryptionResponse(documents, accountPublicId = encryptionRequest.accountPublicId)
    }
}
