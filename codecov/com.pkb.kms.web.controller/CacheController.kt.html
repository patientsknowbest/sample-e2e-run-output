package com.pkb.kms.web.controller

import com.pkb.common.ClearableInternalState
import com.pkb.kms.service.testsupport.StateClearerHelper
import org.springframework.context.annotation.Profile
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.DeleteMapping
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("cache")
@Profile("test", "dev")
class CacheController(private val caches: List<ClearableInternalState>) {

    @DeleteMapping
    fun clearAllCaches(): ResponseEntity<Void> {
        // Skip StateClearerHelper as it does stuff beyond simply emptying caches
        caches.filterNot { it is StateClearerHelper }.map { it.clearState() }
        return ResponseEntity(HttpStatus.NO_CONTENT)
    }
}
