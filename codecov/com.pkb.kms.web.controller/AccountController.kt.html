package com.pkb.kms.web.controller


import com.pkb.kms.service.AccountService
import com.pkb.kms.service.cache.AccountCacheService
import com.pkb.kms.shared.representation.AccountIds
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.web.ErrorException
import io.micrometer.core.annotation.Timed
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.DeleteMapping
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RestController
import java.lang.invoke.MethodHandles

@RestController
@RequestMapping("accounts")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class AccountController(private val accountCacheService: AccountCacheService, private val accountService: AccountService) {
    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())
    @PostMapping("create")
    fun createAccount(): ResponseEntity<AccountIds> =
        accountService.createAccount()
            .map {
                log.info("Created account ${it.publicId}")
                ResponseEntity<AccountIds>(it, HttpStatus.CREATED)
            }
            .getOrElseThrow { err -> ErrorException(err) }

    @DeleteMapping("cache")
    fun clearCache(): ResponseEntity<Void> {
        accountCacheService.clearState()
        return ResponseEntity(HttpStatus.NO_CONTENT)
    }

    @DeleteMapping("cache/{account_public_id}")
    fun evictAccount(@PathVariable("account_public_id") accountPublicId: AccountPublicId): ResponseEntity<Void> {
        accountCacheService.evictAccount(accountPublicId)
        return ResponseEntity(HttpStatus.NO_CONTENT)
    }
}
