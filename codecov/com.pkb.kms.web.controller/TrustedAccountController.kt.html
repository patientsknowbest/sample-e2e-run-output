package com.pkb.kms.web.controller

import com.pkb.kms.service.AccountService
import com.pkb.kms.shared.representation.AccountIds
import com.pkb.kms.shared.representation.OrgPublicId
import com.pkb.kms.web.ErrorException
import io.micrometer.core.annotation.Timed
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RestController
import java.lang.invoke.MethodHandles

@RestController
@RequestMapping("trusted/accounts")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class TrustedAccountController(private val accountService: AccountService) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @PostMapping("create_and_grant_org_access/{org_id}")
    fun createAndGrantOrgAccess(@PathVariable("org_id") orgId: OrgPublicId): ResponseEntity<AccountIds> =
        accountService.createAccountAndGrantOrgAccess(orgId)
            .map {
                log.info("Created account ${it.publicId} for org ${orgId}")
                ResponseEntity<AccountIds>(it, HttpStatus.CREATED)
            }
            .getOrElseThrow { err -> ErrorException(err) }

}
