package com.pkb.kms.web.controller

import com.pkb.crypto.dto.UserPrivateKeyDTO
import com.pkb.crypto.dto.UserPrivateKeyDTO.userPrivateKeyDTO
import com.pkb.entities.enums.SSOType
import com.pkb.kms.service.ExternalPasswordRecoveryService
import com.pkb.kms.service.NhsLoginService
import com.pkb.kms.service.SSOService
import com.pkb.kms.service.cache.UserPrivateKeyCacheService
import com.pkb.kms.shared.representation.NhsLoginRequest
import com.pkb.kms.shared.representation.PersonPublicId
import com.pkb.kms.shared.representation.SsoCredentialsCreateRequest
import com.pkb.kms.web.ErrorException
import io.micrometer.core.annotation.Timed
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.*
import javax.validation.Valid

@RestController
@RequestMapping("credentials")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class CredentialsController(private val userPrivateKeyCache: UserPrivateKeyCacheService,
                            private val nhsLoginService: NhsLoginService,
                            private val externalPasswordRecoveryService: ExternalPasswordRecoveryService,
                            private val ssoService: SSOService) {

    @PostMapping("nhs")
    fun addNhsLoginCredentials(@Valid @RequestBody nhsLoginRequest: NhsLoginRequest): ResponseEntity<Void> =
        userPrivateKeyCache.getCachedUserPrivateKey(nhsLoginRequest.userId)
                .map { userPrivateKey ->
                    nhsLoginService.encryptAndSaveUserPrivateKeyWithNhsLogin(userPrivateKey,nhsLoginRequest.userId, nhsLoginRequest.secret)
                            .map { ResponseEntity<Void>(HttpStatus.NO_CONTENT) }
                            .getOrElseThrow { err -> ErrorException(err) }}
                .getOrElseThrow { err -> ErrorException(err) }

    @DeleteMapping("nhs/{person_public_ids}")
    fun expireNhsLoginCredentials(@PathVariable("person_public_ids") personPublicIds: List<PersonPublicId>): ResponseEntity<Void> =
            nhsLoginService.expireCredentials(personPublicIds)
                    .map { ResponseEntity<Void>(HttpStatus.NO_CONTENT) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @PostMapping("external_recovery/{person_public_id}")
    fun createExternalRecoveryRecord(@PathVariable("person_public_id") personPublicId: PersonPublicId): ResponseEntity<Void> =
            externalPasswordRecoveryService.encryptUserPrivateKey(personPublicId)
                    .flatMap { externalPasswordRecoveryService.createNewRecoveryRecord(personPublicId, it) }
                    .map { ResponseEntity<Void>(HttpStatus.NO_CONTENT) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @GetMapping("external_recovery/{person_public_id}")
    fun recoverUserKey(@PathVariable("person_public_id") personPublicId: PersonPublicId): UserPrivateKeyDTO =
            externalPasswordRecoveryService.findRecoveryRecord(personPublicId)
                    .map { userPrivateKeyDTO(it) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @DeleteMapping("external_recovery/{person_public_id}")
    fun deleteExternalRecoveryRecord(@PathVariable("person_public_id") personPublicId: PersonPublicId): ResponseEntity<Void> {
        externalPasswordRecoveryService.expireRecoveryRecord(personPublicId)
        return ResponseEntity<Void>(HttpStatus.NO_CONTENT)
    }

    @PostMapping("sso/credential")
    fun createSSOCredential(@RequestBody request: SsoCredentialsCreateRequest): ResponseEntity<Void> =
        ssoService.createSSOCredential(request.userId, request.key, request.ssoType)
            .map  { ResponseEntity<Void>(HttpStatus.NO_CONTENT) }
            .getOrElseThrow { err -> ErrorException(err) }

    @PostMapping("sso/auth_record")
    fun createSSOAuthRecord(@RequestBody request: SsoCredentialsCreateRequest): ResponseEntity<Void> =
        ssoService.createSSOAuthRecord(request.userId, request.key, request.ssoType)
            .map  { ResponseEntity<Void>(HttpStatus.NO_CONTENT) }
            .getOrElseThrow { err -> ErrorException(err) }

    @GetMapping("sso/auth_record_exists/{person_public_id}")
    fun ssoAuthRecordExists(@PathVariable("person_public_id") personPublicId: PersonPublicId, @RequestParam("ssoType") ssoType: SSOType): Boolean =
        ssoService.ssoAuthRecordExistsFor(personPublicId, ssoType)


    @GetMapping("sso/status/{person_public_id}")
    fun credentialStatus(@PathVariable("person_public_id") personPublicId: PersonPublicId): Set<SSOType> =
        ssoService.credentialStatus(personPublicId)


    @GetMapping("sso/emis_key_valid/{person_public_id}")
    fun emisKeyValid(@PathVariable("person_public_id") personPublicId: PersonPublicId, @RequestParam("password_hash") passwordHash:String): Boolean =
        ssoService.emisKeyValid(personPublicId, passwordHash)

}
