package com.pkb.kms.web.controller

import com.pkb.crypto.dto.AccountPrivateKeyDTO
import com.pkb.kms.service.AccountKeyService
import com.pkb.kms.service.cache.AccountSymmetricKeysCacheService.Companion.ACCOUNT_SYMMETRIC_KEYS_CACHE_NAME
import com.pkb.kms.service.cache.DplKeyCacheService
import com.pkb.kms.service.cache.KmsCacheManager
import com.pkb.kms.service.cache.OrgAccessibleAccountPrivateKeyCacheService.Companion.ORG_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME
import com.pkb.kms.service.cache.UnauthenticatedAccountSymmetricKeyCacheService.Companion.ACCOUNT_SYMMETRIC_KEY_CACHE_NAME
import com.pkb.kms.service.cache.UserAccessibleAccountPrivateKeyCacheService.Companion.USER_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME
import com.pkb.kms.shared.representation.AccountPrivateId
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.DplKeyId
import com.pkb.kms.shared.representation.KmsError
import io.vavr.control.Either
import org.springframework.context.annotation.Profile
import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RequestParam
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("kmstest")
@Profile("test", "dev")
class TestController(private val cacheManager: KmsCacheManager,
                     private val accountKeyService: AccountKeyService,
                     private val dplKeyCacheService: DplKeyCacheService) {

    @GetMapping("throw")
    fun throwUnhandledException(@RequestParam message: String): Nothing = throw RuntimeException(message)


    @GetMapping("account_keys/cache/status/{account_public_ids}")
    fun getAccountKeyCacheStatus(@PathVariable("account_public_ids") accountPublicIds: List<AccountPublicId>): Map<String, Set<AccountPublicId>> {

        val statusFn: (String, List<AccountPublicId>) -> Set<AccountPublicId> = { cacheName, ids ->
            cacheManager.getAll<AccountPublicId, Either<KmsError, AccountPrivateKeyDTO>>(cacheName, ids)
                    .let { it.keys.intersect(ids) }
        }

        return mapOf(
                "org" to statusFn(ORG_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME, accountPublicIds),
                "user" to statusFn(USER_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME, accountPublicIds)
        )
    }

    @GetMapping("symmetric_keys/cache/status/{account_public_ids}")
    fun getAccountSymmetricKeyCacheStatus(@PathVariable("account_public_ids") accountPublicIds: List<AccountPublicId>): Map<String, Set<AccountPublicId>> {

        val statusFn: (String, List<AccountPublicId>) -> Set<AccountPublicId> = { cacheName, ids ->
            cacheManager.getAll<AccountPublicId, Either<KmsError, AccountPrivateKeyDTO>>(cacheName, ids)
                    .let { it.keys.intersect(ids) }
        }

        return mapOf(
                "single" to statusFn(ACCOUNT_SYMMETRIC_KEY_CACHE_NAME, accountPublicIds),
                "all" to statusFn(ACCOUNT_SYMMETRIC_KEYS_CACHE_NAME, accountPublicIds)
        )
    }

    @GetMapping("dpl_keys/cache/status/{dpl_key_ids}", params = ["account_id"])
    fun getDplKeyCacheStatus(@PathVariable("dpl_key_ids") dplKeyIds: Set<DplKeyId>,
                             @RequestParam("account_id") accountId: AccountPrivateId): List<DplKeyId> =
            dplKeyCacheService.getDplKeysForAccountsFromCache(dplKeyIds.map { Pair(accountId, it) })
                    .second.map { it.id }
}
