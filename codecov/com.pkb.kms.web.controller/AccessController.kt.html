package com.pkb.kms.web.controller

import com.pkb.kms.service.AccessService
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.GrantOrgAccessToAccountRequest
import com.pkb.kms.shared.representation.GrantUserAccessToAccountRequest
import com.pkb.kms.shared.representation.OrgPublicId
import com.pkb.kms.web.ErrorException
import io.micrometer.core.annotation.Timed
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.RequestBody
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RestController
import java.lang.invoke.MethodHandles

@RestController
@RequestMapping("access")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class AccessController(private val accessService: AccessService) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @PostMapping("account/self")
    fun grantUserAccessToAccount(@RequestBody grantRequest: GrantUserAccessToAccountRequest): ResponseEntity<Void> =
            accessService.grantAccountAccessToAccountOwner(grantRequest.accountPublicId, grantRequest.targetPersonPublicId)
                    .map {
                        log.info("Granted access to account ${grantRequest.accountPublicId} for person ${grantRequest.targetPersonPublicId}")
                        ResponseEntity<Void>(HttpStatus.NO_CONTENT)
                    }
                    .getOrElseThrow { err -> ErrorException(err) }
    
    @PostMapping("account/{account_id}/org/{org_id}")
    fun grantOrgAccess(@PathVariable("account_id") accountPublicId: AccountPublicId,
                       @PathVariable("org_id") orgPublicId: OrgPublicId): ResponseEntity<Void> =
            accessService.grantOrgAccess(accountPublicId, orgPublicId)
                    .map {
                        log.info("Granted access to account {} for org {}", accountPublicId, orgPublicId)
                        ResponseEntity<Void>(HttpStatus.NO_CONTENT)
                    }
                    .getOrElseThrow { err -> ErrorException(err) }


    @PostMapping("account")
    fun grantOrgAccessMultiple(@RequestBody grantOrgAccessToAccountRequest: GrantOrgAccessToAccountRequest): ResponseEntity<List<AccountPublicId>> =
        accessService.grantOrgAccess(grantOrgAccessToAccountRequest.accountPublicIds, grantOrgAccessToAccountRequest.orgPublicId)
            .map {
                ResponseEntity(it, HttpStatus.CREATED)
            }
            .getOrElseThrow { err -> ErrorException(err) }
}
