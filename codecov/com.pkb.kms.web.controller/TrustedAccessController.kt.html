package com.pkb.kms.web.controller

import com.pkb.kms.service.AccessService
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.GrantUserAccessToAccountRequest
import com.pkb.kms.shared.representation.PersonPublicId
import com.pkb.kms.web.ErrorException
import io.micrometer.core.annotation.Timed
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.DeleteMapping
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.RequestBody
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RestController
import java.lang.invoke.MethodHandles

@RestController
@RequestMapping("trusted/access")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class TrustedAccessController(private val accessService: AccessService) {
    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @PostMapping("account")
    fun grantUserAccessToAccount(@RequestBody grantRequest: GrantUserAccessToAccountRequest): ResponseEntity<Void> =
        accessService.grantAccountAccessToUser(grantRequest.accountPublicId, grantRequest.targetPersonPublicId)
            .map {
                log.info("Granted access to account ${grantRequest.accountPublicId} for person ${grantRequest.targetPersonPublicId}")
                ResponseEntity<Void>(HttpStatus.NO_CONTENT)
            }
            .getOrElseThrow { err -> ErrorException(err) }

    @DeleteMapping("/account/{account_id}/{person_public_id}")
    fun revokeUserAccessToAccount(@PathVariable("account_id") accountPublicId: AccountPublicId, @PathVariable("person_public_id") personPublicId: PersonPublicId): ResponseEntity<Void> =
        accessService.revokeAccountAccessForUser(accountPublicId, personPublicId)
            .map {
                log.info("Revoked access to account $accountPublicId for person $personPublicId")
                ResponseEntity<Void>(HttpStatus.NO_CONTENT)
            }
            .getOrElseThrow { err -> ErrorException(err) }
}
