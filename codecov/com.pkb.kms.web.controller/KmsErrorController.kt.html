package com.pkb.kms.web.controller

import com.pkb.kms.shared.representation.KmsError
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.boot.web.servlet.error.ErrorController
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.stereotype.Controller
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.ResponseBody
import java.lang.invoke.MethodHandles
import javax.servlet.http.HttpServletRequest

/**
 * https://github.com/spring-projects/spring-boot/issues/3980#issuecomment-142071601
 */
@Suppress("MVCPathVariableInspection")
@Controller
@RequestMapping("\${error.path:/error}")
class KmsErrorController : ErrorController {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())


    @RequestMapping
    @ResponseBody
    fun error(request: HttpServletRequest): ResponseEntity<Any?>? {

        val status = getStatus(request)
        val uri = getUri(request)
        log.error("$status: $uri")

        val kmsErrorType = when (status) {
            HttpStatus.NOT_FOUND -> KmsError.Type.NOT_FOUND
            HttpStatus.UNAUTHORIZED, HttpStatus.FORBIDDEN -> KmsError.Type.NOT_AUTHORISED
            else -> KmsError.Type.UNEXPECTED
        }

        return ResponseEntity(KmsError("error.path", uri, kmsErrorType), status)
    }

    private fun getStatus(request: HttpServletRequest): HttpStatus {
        val statusCode = request.getAttribute("javax.servlet.error.status_code") as? Int
                ?: return HttpStatus.INTERNAL_SERVER_ERROR
        return HttpStatus.valueOf(statusCode)
    }

    private fun getUri(request: HttpServletRequest): String {
        return request.getAttribute("javax.servlet.error.request_uri") as? String
                ?: "unknown"
    }
}
