package com.pkb.kms.web.controller

import com.pkb.kms.service.AccountKeyService
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.web.ErrorException
import io.micrometer.core.annotation.Timed
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RestController
import java.lang.invoke.MethodHandles

@RestController
@RequestMapping("trusted/account_keys")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class TrustedAccountKeyController(private val accountKeyService: AccountKeyService) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    /**
     * Retrieves account keys associated with the accounts specified by [accountPublicIds]
     *
     * This method does _not_ require the accessing user to be specified. It is intended to be used by backend services
     * and not as part of an interactive user session via PHR.
     *
     * If the keys are not cached, we try any means available to reach them, such as looking up all org accounts that
     * have a copy of the key and examining the org account key cache to see if any of those keys are available, in
     * which case they are used to decrypt and return the target account keys.
     */
    @GetMapping("{account_public_ids}")
    fun getAccountKeys(@PathVariable("account_public_ids") accountPublicIds: List<AccountPublicId>) =
            accountPublicIds.map { accountPublicId: AccountPublicId ->
                accountKeyService.getAccountKeys(accountPublicId)
                        .peek { log.info("Account keys for account $accountPublicId accessed by trusted client") }
                        .getOrElseThrow { err -> ErrorException(err) }
            }


    @GetMapping("/ignore_missing/{account_public_ids}")
    fun getAccountKeysIgnoreMissing(@PathVariable("account_public_ids") accountPublicIds: List<AccountPublicId>) =
            accountKeyService.getAccountKeys(accountPublicIds)
}
