package com.pkb.kms.web.controller

import com.pkb.kms.service.cache.KmsCacheManager
import com.pkb.kms.service.cache.OrgAccountKeyCacheService
import com.pkb.kms.service.cache.OrgAccountKeyCacheService.Companion.ORG_ACCOUNT_PRIVATE_KEY_CACHE_NAME
import io.micrometer.core.annotation.Timed
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.DeleteMapping
import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("org_account_keys")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class OrgAccountKeyController(private val orgAccountKeyCacheService: OrgAccountKeyCacheService,
                              private val cacheManager: KmsCacheManager) {

    /**
     * Retrieves a list of all org ids where the org account key is currently available in the cache
     */
    @GetMapping("cache/list")
    fun listCachedOrgAccounts(): Set<Long> =
            cacheManager.getAllKeysInCache(ORG_ACCOUNT_PRIVATE_KEY_CACHE_NAME, Long::class)

    @DeleteMapping("cache")
    fun clearCache(): ResponseEntity<Void> {
        orgAccountKeyCacheService.clearState()
        return ResponseEntity(HttpStatus.NO_CONTENT)
    }

    @DeleteMapping("cache/{org_id}")
    fun evictOrgAccountKey(@PathVariable("org_id") orgId: Long): ResponseEntity<Void> {
        orgAccountKeyCacheService.evictOrgAccountKey(orgId)
        return ResponseEntity(HttpStatus.NO_CONTENT)
    }
}
