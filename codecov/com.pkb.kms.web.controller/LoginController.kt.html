package com.pkb.kms.web.controller

import com.pkb.crypto.dto.UserPrivateKeyDTO
import com.pkb.kms.service.LoginService
import com.pkb.kms.service.SSOService
import com.pkb.kms.shared.representation.LoginRequest
import com.pkb.kms.shared.representation.NhsLoginRequest
import com.pkb.kms.shared.representation.SSOLoginRequest
import com.pkb.kms.web.ErrorException
import io.micrometer.core.annotation.Timed
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.RequestBody
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("login")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class LoginController(private val loginService: LoginService, private val ssoService: SSOService) {
    /**
     * The default login mechanism is password, so both /login and /login/password are directed to this endpoint
     */
    @PostMapping(value = ["", "password"])
    fun login(@RequestBody loginRequest: LoginRequest): ResponseEntity<Void> =
            loginService.login(loginRequest.userId, loginRequest.credentials, loginRequest.forceVerify)
                    .map { ResponseEntity<Void>(HttpStatus.NO_CONTENT) }
                    .getOrElseThrow { err -> throw ErrorException(err) }

    @PostMapping("nhs")
    fun nhsLogin(@RequestBody nhsLoginRequest: NhsLoginRequest): UserPrivateKeyDTO =
            loginService.nhsLogin(nhsLoginRequest.userId, nhsLoginRequest.secret)
                    .getOrElseThrow { err -> ErrorException(err) }


    @PostMapping("sso")
    fun ssoLogin(@RequestBody loginRequest: SSOLoginRequest): UserPrivateKeyDTO =
        ssoService.ssoLogin(loginRequest.personPublicId, loginRequest.keyString, loginRequest.ssoType)
            .getOrElseThrow { err -> throw ErrorException(err) }
}
