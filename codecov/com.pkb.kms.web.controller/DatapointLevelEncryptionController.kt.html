package com.pkb.kms.web.controller

import com.pkb.crypto.dto.DplKeyDTO
import com.pkb.kms.service.DplKeyService
import com.pkb.kms.shared.representation.DplKeyId
import com.pkb.kms.shared.representation.DplKeyRequest
import com.pkb.kms.shared.representation.OrgPublicId
import com.pkb.kms.shared.representation.PersonPublicId
import com.pkb.kms.shared.representation.Requester
import com.pkb.kms.shared.representation.TeamPublicId
import com.pkb.kms.web.ErrorException
import io.micrometer.core.annotation.Timed
import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.RequestBody
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RequestParam
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("dpl_keys")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class DatapointLevelEncryptionController(private val dplKeyService: DplKeyService) {


    @PostMapping()
    fun createDplKey(@RequestBody dplKeyRequest: DplKeyRequest): DplKeyDTO =
            dplKeyService.createDplKey(dplKeyRequest.storeForPeople, dplKeyRequest.storeForTeams)
                    .getOrElseThrow { err -> ErrorException(err) }

    @GetMapping("/{key_ids}", params = ["target_person_id"])
    fun getDplKeys(@RequestParam("ref_org_id") requesterOrgPublicId: OrgPublicId?,
                   @RequestParam("ref_person_id") requesterPersonPublicId: PersonPublicId?,
                   @RequestParam("target_person_id") targetPersonId: PersonPublicId,
                   @PathVariable("key_ids") keyIds: List<DplKeyId>): List<DplKeyDTO> =
            Requester.derive(requesterPersonPublicId, requesterOrgPublicId).flatMap { dplKeyService.getKeysForUser(it, keyIds, targetPersonId) }
                    .getOrElseThrow { err -> ErrorException(err) }


    @GetMapping("/{key_ids}", params = ["target_team_id"])
    fun getDplKeysForTeam(@RequestParam("ref_org_id") requesterOrgPublicId: OrgPublicId?,
                          @RequestParam("ref_person_id") requesterPersonPublicId: PersonPublicId?,
                          @RequestParam("target_team_id") targetTeamId: TeamPublicId,
                          @PathVariable("key_ids") keyIds: List<DplKeyId>): List<DplKeyDTO> =
            Requester.derive(requesterPersonPublicId, requesterOrgPublicId).flatMap { dplKeyService.getKeysForTeam(it, keyIds, targetTeamId) }
                    .getOrElseThrow { err -> ErrorException(err) }

}
