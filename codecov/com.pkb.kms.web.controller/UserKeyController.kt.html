package com.pkb.kms.web.controller

import com.pkb.kms.service.cache.UserCacheService
import com.pkb.kms.service.cache.UserPrivateKeyCacheService
import com.pkb.kms.shared.representation.PersonPublicId
import io.micrometer.core.annotation.Timed
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.*

@RestController
@RequestMapping("user_keys")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class UserKeyController(
    private val userPrivateKeyCache: UserPrivateKeyCacheService,
    private val userCache: UserCacheService
) {

    @DeleteMapping("cache")
    fun clearCache(): ResponseEntity<Void> {
        userPrivateKeyCache.clearState()
        userCache.clearState()
        return ResponseEntity(HttpStatus.NO_CONTENT)
    }

    @DeleteMapping("cache/{person_public_id}")
    fun evictUserKey(@PathVariable("person_public_id") personPublicId: PersonPublicId): ResponseEntity<Void> {
        userPrivateKeyCache.evictUserPrivateKey(personPublicId)
        userCache.evictUserCredentials(personPublicId)
        return ResponseEntity(HttpStatus.NO_CONTENT)
    }

    @GetMapping("cache/status/{user_public_ids}")
    fun isUserPrivateKeysCached(@PathVariable("user_public_ids") personIds: List<PersonPublicId>): Set<PersonPublicId> =
        personIds.filter { userPrivateKeyCache.getCachedUserPrivateKey(it).isRight }.toSet()

    @GetMapping("/{user_public_id}/exists")
    fun userKeyExist(@PathVariable("user_public_id") publicId: PersonPublicId): Boolean {
        return userCache.findPerson(publicId)
            .flatMap { userCache.findCredentials(it) }
            .fold({ false }, { true })
    }
}
