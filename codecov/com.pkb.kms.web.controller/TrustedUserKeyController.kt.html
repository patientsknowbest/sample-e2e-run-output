package com.pkb.kms.web.controller

import com.pkb.crypto.dto.UserPrivateKeyDTO
import com.pkb.kms.service.cache.UserPrivateKeyCacheService
import com.pkb.kms.shared.representation.PersonPublicId
import com.pkb.kms.web.ErrorException
import io.micrometer.core.annotation.Timed
import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("trusted/user_keys")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class TrustedUserKeyController(private val userPrivateKeyCache: UserPrivateKeyCacheService) {

    @GetMapping("{person_public_id}")
    fun getUserKey(@PathVariable("person_public_id") personPublicId: PersonPublicId): UserPrivateKeyDTO =
        userPrivateKeyCache.getCachedUserPrivateKey(personPublicId)
            .map { UserPrivateKeyDTO.userPrivateKeyDTO(it) }
            .getOrElseThrow { err -> ErrorException(err) }
}
