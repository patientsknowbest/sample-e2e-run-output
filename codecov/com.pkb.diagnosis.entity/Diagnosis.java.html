package com.pkb.diagnosis.entity;

import com.pkb.annotation.EHRField;
import com.pkb.annotation.EHRField.QueryField;
import com.pkb.annotation.EHRMigration;
import com.pkb.annotation.EncryptedFieldMigration;
import com.pkb.annotation.Migrate;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;
import com.pkb.app.interfaces.DataPointWithStartDate;
import com.pkb.app.interfaces.EncryptedDataPoint;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.coding.entity.CodeableConcept;
import com.pkb.entities.enums.ConditionCategory;
import com.pkb.entities.enums.DiagnosisClinicalStatus;
import com.pkb.entities.enums.MenuDataType;
import org.jetbrains.annotations.Nullable;

import java.time.Instant;
import java.util.Arrays;
import java.util.Date;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.stream.Collectors;

import static com.pkb.annotation.EHRField.ValueDTOType.NESTED;

/**
 * Diagnosis record for a patient
 *
 * @author pravina
 * @since Dec 04 2014 : Migrated to using EHRBean ; This record does not rely on
 * Tolven mappings defined in pkb.application anymore
 * <p>
 * Attributes and nomenclature from the FHIR Condition - http://www.hl7.org/implement/standards/fhir/condition.html
 */
@EncryptedFieldMigration({
        @Migrate(fromEncrypted = "authorLdapUID", toQueryField = QueryField.sourcePersonId)})
@EHRMigration(migratorClass = DiagnosisMigrator.class)
public class Diagnosis implements IBaseDTO, EncryptedDataPoint, DataPointWithStartDate {

    private static final long serialVersionUID = 1L;

    public static final MenuDataType MS_PATH = MenuDataType.diagnosis;
    static final String DIAGNOSIS_DIAGNOSIS = "diagnosis";
    public static final String DIAGNOSIS_CATEGORY = "category";
    public static final String DIAGNOSIS_DIAGNOSIS_CODE_ID = "diagnosisCodeId";

    @EHRField(queryField = QueryField.id)
    private Long id;

    @EHRField(queryField = QueryField.date01)
    private Date startDate;

    @EHRField(queryField = QueryField.date02)
    private Date endDate;

    @EHRField(queryField = QueryField.date03)
    private Date dateAsserted; // When first detected / suspected/ entered

    @EHRField(valueDtoType = NESTED)
    private CodeableConcept diagnosis;

    @EHRField(queryField = QueryField.long01)
    private Long diagnosisCodingReceivedId;

    @EHRField
    private String details;

    @EHRField(queryField = QueryField.accountId)
    private Long patientAccountId;

    /**
     * defaults to PROBLEM_LIST_ITEM: these are displayed in the UI
     */
    @EHRField(queryField = QueryField.string05)
    private ConditionCategory category = ConditionCategory.PROBLEM_LIST_ITEM;

    @EHRField
    private DiagnosisClinicalStatus clinicalStatus;

    private SourceDetails source;

    private final DTOBaseFields baseFields = new DTOBaseFields();

    /**
     * This is an old field and should be purged, but default to true for now and remove all references.
     * Once migration has been successful and we've confirmed there are no more where this is set to FALSE, we can
     * just null out the entire field.
     */
    @EHRField(queryField = QueryField.boolean01)
    private Boolean patientHasCondition = Boolean.TRUE;

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected Diagnosis() {
        this.source = new SourceDetails();
    }

    public Diagnosis(SourceDetails source) {
        this.source = source;
    }

    @Override
    public CodeableConcept code() {
        return diagnosis;
    }

    @Nullable
    @Override
    public Instant startDate() {
        return startDate == null ? null : startDate.toInstant();
    }

    @Override
    public Long getId() {
        return id;
    }

    @Override
    public void setId(Long id) {
        this.id = id;
    }

    public Date getStartDate() {
        return startDate;
    }

    public void setStartDate(Date startDate) {
        this.startDate = startDate;
    }

    public Date getEndDate() {
        return endDate;
    }

    public void setEndDate(Date endDate) {
        this.endDate = endDate;
    }

    public CodeableConcept getDiagnosis() {
        return diagnosis;
    }

    public void setDiagnosis(CodeableConcept diagnosis) {
        this.diagnosis = diagnosis;
    }

    public Long getDiagnosisCodingReceivedId() {
        return diagnosisCodingReceivedId;
    }

    public void setDiagnosisCodingReceivedId(Long diagnosisCodingReceivedId) {
        this.diagnosisCodingReceivedId = diagnosisCodingReceivedId;
    }

    @Override
    public Long getPatientAccountId() {
        return patientAccountId;
    }

    public void setPatientAccountId(Long patientAccountId) {
        this.patientAccountId = patientAccountId;
    }

    public String getDetails() {
        return details;
    }

    public void setDetails(String details) {
        this.details = details;
    }

    @Override
    public String toString() {
        return "Condition{" + "id=" + id + ", startDate=" + startDate + ", endDate=" + endDate + ", diagnosis=" + diagnosis
                + ", details=" + details + ", patientAccountId=" + patientAccountId
                + ", source=" + source + ", baseFields=" + baseFields + '}';
    }

    public DiagnosisClinicalStatus getClinicalStatus() {
        return clinicalStatus;
    }

    public void setClinicalStatus(DiagnosisClinicalStatus clinicalStatus) {
        this.clinicalStatus = clinicalStatus;
    }

    public Boolean getPatientHasCondition() {
        return patientHasCondition;
    }

    public void setPatientHasCondition(Boolean patientHasCondition) {
        this.patientHasCondition = patientHasCondition;
    }

    @Override
    public SourceDetails getSource() {
        return source;
    }

    @Override
    public DTOBaseFields getBaseFields() {
        return baseFields;
    }

    public Date getDateAsserted() {
        return dateAsserted;
    }

    public void setDateAsserted(Date dateAsserted) {
        this.dateAsserted = dateAsserted;
    }

    /**
     * The only values supported (initial impl): PROBLEM_LIST_ITEM and null.
     * Only problem list items are displayed in the UI; the items with no category set are from EMIS ES
     * (we're planning to show those in an updated UI in the future).
     */
    public ConditionCategory getCategory() {
        return category;
    }

    /**
     * Future (when FHIR API catches up): CodeableConcept, store/read via Coded Data flows, and support query on it
     * For now: only PROBLEM_LIST_ITEM or unset (no data). We only display problem list items, and by default all new
     * Diagnoses are problem list items.
     */
    public void setCategory(ConditionCategory category) {
        this.category = category;
    }

    @Override
    public MenuDataType getDataType() {
        return MS_PATH;
    }

    public String getClinicalCodes() {

        LinkedHashMap<String, List<CodeableConcept>> map = new LinkedHashMap<>();
        map.put("", Arrays.asList(diagnosis));

        return map.entrySet().stream().flatMap(e -> e.getValue()
                .stream()
                .filter(p -> p != null && !p.getSimpleCodesForDisplay().isEmpty())
                .map(v -> v.getSimpleCodesForDisplay() + e.getKey()))
                .collect(Collectors.joining(", "));
    }

    public void setSource(SourceDetails sourceDetails) {
        this.source = sourceDetails;
    }
}
