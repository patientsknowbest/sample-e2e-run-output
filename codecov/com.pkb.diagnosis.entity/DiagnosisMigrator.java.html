package com.pkb.diagnosis.entity;

import com.pkb.PrivateIdMigrator;
import com.pkb.annotation.EHRMigratorBase;
import com.pkb.annotation.EHRVersionMigrator;
import com.pkb.app.entity.DTOMapping;
import com.pkb.app.entity.EHRData;
import com.pkb.coding.entity.CodeableConcept;
import com.pkb.util.ValueMappingUtil;

public class DiagnosisMigrator extends EHRMigratorBase {


    public static final String FIELD_NAME = "diagnosis";

    private static final Migrators MIGRATORS = Migrators.of(
            EHRVersionMigrator.of(DiagnosisMigrator::migrateStringFieldToCodeableConcept, 1),
            PrivateIdMigrator.migratorFor(Diagnosis.class, "diagnosisCodingReceivedId", "string04", 2)
    );


    @Override
    public Migrators getMigrators() {
        return MIGRATORS;
    }

    /**
     * Current encrypted value for fieldName must be String or null.
     * Convert to CodeableConcept with string value as display text.
     *
     * @param ehr The EHRData to be migrated
     */
    private static void migrateStringFieldToCodeableConcept(EHRData ehr) {
        Object valueObj = ehr.getEncryptedField(FIELD_NAME);
        if (valueObj == null) {
            return;
        } else if (!(valueObj instanceof String)) {
            throw new IllegalStateException("Encrypted value for " + FIELD_NAME + ": expected String, found " + valueObj.getClass());
        }

        CodeableConcept cc = new CodeableConcept();
        cc.setDisplayText((String) valueObj);
        ehr.setEncryptedField(FIELD_NAME, toMap(cc));
    }

    private static Object toMap(CodeableConcept cc) {
        DTOMapping<Diagnosis> diagnosisMapping = DTOMapping.get(Diagnosis.class);
        DTOMapping<?> dtoMapping = diagnosisMapping.getNestedDtoMapping(Diagnosis.DIAGNOSIS_DIAGNOSIS);
        return ValueMappingUtil.prepValueForEHRData(cc, dtoMapping);
    }
}
