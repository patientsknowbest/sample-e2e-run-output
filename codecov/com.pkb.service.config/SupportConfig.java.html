package com.pkb.service.config;

import com.pkb.common.config.ConfigStorage;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.domain.AccountUserService;
import com.pkb.domain.RegistrationEventService;
import com.pkb.domain.duplicate.person.PersonService;
import com.pkb.repository.SupportRepository;
import com.pkb.repository.SupportTempBackupTableRepository;
import com.pkb.service.support.DataCollectionOutboundRouteBuilder;
import com.pkb.service.support.DataCollectionService;
import com.pkb.service.support.EmailDeletionOutboundRouteBuilder;
import com.pkb.service.support.EmailDeletionService;
import com.pkb.service.support.MessageDeletionOutboundRouterBuilder;
import com.pkb.service.support.MessageDeletionService;
import com.pkb.service.support.PhoenixPubSubConfig;
import com.pkb.service.support.SupportTaskInboundRequestRouteBuilder;
import com.pkb.service.support.SupportToolTwoFARemovalService;
import com.pkb.service.support.TwoFARemovalOutboundRouteBuilder;
import com.pkb.service.twofactor.TwoFactorAuthenticationService;
import com.pkb.service.user.impl.UserManager;
import org.apache.camel.ProducerTemplate;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Profile;
import org.springframework.jdbc.core.JdbcTemplate;

import javax.sql.DataSource;

@Configuration
public class SupportConfig {

    @Bean
    public PhoenixPubSubConfig pubSubConfig(ConfigStorage configStorage) {
        return () -> configStorage;
    }

    @Bean
    @Profile("!unittest")
    public SupportTaskInboundRequestRouteBuilder supportTaskInboundRequestRouteBuilder(PubSubNamespaceService pubSubNamespaceService,
                                                                                       PhoenixPubSubConfig config,
                                                                                       @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey) {
        return new SupportTaskInboundRequestRouteBuilder(pubSubNamespaceService, config, serviceAccountKey);
    }

    @Bean
    DataCollectionService dataCollectionService(ProducerTemplate producerTemplate, PersonService personService, AccountUserService accountUserService, RegistrationEventService registrationEventService) {
        return new DataCollectionService(producerTemplate, personService, accountUserService, registrationEventService);
    }

    @Bean
    @Profile("!unittest")
    DataCollectionOutboundRouteBuilder dataCollectionOutboundRouteBuilder(PubSubNamespaceService pubSubNamespaceService,
                                                                          PhoenixPubSubConfig config,
                                                                          @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey) {
        return new DataCollectionOutboundRouteBuilder(pubSubNamespaceService, config, serviceAccountKey);
    }


    @Bean
    @Profile("!unittest")
    MessageDeletionOutboundRouterBuilder messageDeletionOutboundRouterBuilder(PubSubNamespaceService pubSubNamespaceService,
                                                                              PhoenixPubSubConfig config,
                                                                              @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey) {
        return new MessageDeletionOutboundRouterBuilder(pubSubNamespaceService, config, serviceAccountKey);
    }

    @Bean
    MessageDeletionService messageDeletionService(ProducerTemplate producerTemplate, SupportRepository supportRepository) {
        return new MessageDeletionService(producerTemplate, supportRepository);
    }

    @Bean
    @Profile("!unittest")
    EmailDeletionOutboundRouteBuilder emailDeletionOutboundRouteBuilder(PhoenixPubSubConfig config,
                                                                        PubSubNamespaceService namespaceService,
                                                                        @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey) {
        return new EmailDeletionOutboundRouteBuilder(config, namespaceService, serviceAccountKey);
    }

    @Bean
    EmailDeletionService emailDeletionService(ProducerTemplate producerTemplate, SupportRepository supportRepository) {
        return new EmailDeletionService(producerTemplate, supportRepository);
    }

    @Bean
    SupportToolTwoFARemovalService twoFARemovalService(ProducerTemplate producerTemplate,
                                                       TwoFactorAuthenticationService twoFactorAuthenticationService,
                                                       PersonService personService,
                                                       UserManager userManager,
                                                       DateTimeService dateTimeService) {
        return new SupportToolTwoFARemovalService(producerTemplate, twoFactorAuthenticationService, personService, userManager, dateTimeService);
    }

    @Bean
    @Profile("!unittest")
    TwoFARemovalOutboundRouteBuilder twoFARemovalOutboundRouteBuilder(PubSubNamespaceService pubSubNamespaceService,
                                                                      PhoenixPubSubConfig config,
                                                                      @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey) {
        return new TwoFARemovalOutboundRouteBuilder(pubSubNamespaceService, config, serviceAccountKey);
    }

    @Bean
    public JdbcTemplate jdbcTemplate(DataSource dataSource) {
        return new JdbcTemplate(dataSource);
    }

    @Bean
    public SupportRepository supportRepository(JdbcTemplate jdbcTemplate, SupportTempBackupTableRepository supportTempBackupTableRepository) {
        return new SupportRepository(jdbcTemplate, supportTempBackupTableRepository);
    }

}
