package com.pkb.domain.routing;

import com.pkb.common.config.BaseConfig;
import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.deletion.model.DeleteOperationRequest;
import com.pkb.deletion.model.DeleteOperationResponse;
import com.pkb.deletion.model.DeletionServiceProperties;
import com.pkb.domain.config.DomainServiceProperties;
import org.apache.camel.LoggingLevel;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

import static org.apache.commons.lang3.StringUtils.isBlank;

@Component
public class DeleteRequestRouteBuilder extends AbstractNamespaceAwareRouteBuilder {
    public static final String OPERATION_REQUEST = "direct:deleteOperationRequest";
    public static final String OPERATION_RESPONSE = "direct:deleteOperationResponse";
    public static final String HANDLE_OPERATION_REQUEST = "direct:handleDeleteOperationRequest";

    public static final String AGGREGATE_OPERATION_RESPONSE = "direct:aggregateDeleteOperationResponse";
    public static final String DELETE_OPERATION_RESPONSE_AGGREGATED = "direct:deleteOperationResponseAggregated";

    private final PubSubNamespaceService pubSubNamespaceService;
    private final String serviceAccountKey;
    private final BaseConfig config;
    private final DomainServiceProperties properties;
    private final DeletionServiceProperties deletionServiceProperties;

    public DeleteRequestRouteBuilder(PubSubNamespaceService pubSubNamespaceService,
                                     @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey,
                                     BaseConfig config,
                                     DomainServiceProperties properties,
                                     DeletionServiceProperties  deletionServiceProperties) {
        this.pubSubNamespaceService = pubSubNamespaceService;
        this.serviceAccountKey = serviceAccountKey;
        this.config = config;
        this.properties = properties;
        this.deletionServiceProperties = deletionServiceProperties;
    }

    @Override
    public @NotNull PubSubNamespaceService camelNamespaceService() {
        return pubSubNamespaceService;
    }

    @Override
    public boolean throwExceptionForNamespaceMismatch() {
        return true;
    }

    @Override
    public void configure() throws Exception {
        from(OPERATION_REQUEST)
                .marshal().json(DeleteOperationRequest.class)
                .convertBodyTo(byte[].class)
                .to(uri(config.getPubSubProjectName(), properties.getDeleteRequestTopicName()));

        //In principle we could just call these directly, or add it to the camel routes above as an intermediate processing step,
        //but I want to keep the PubSubAggregator stuff decoupled from the rest of the PersonDeletion stuff so
        //that it could in theory be pulled out into a separate component somewhere
        from(uri(config.getPubSubProjectName(), deletionServiceProperties.getDeleteRequestSubscriptionName()))
                .unmarshal().json(DeleteOperationRequest.class)
                .log(LoggingLevel.INFO, "Received message on request subscription")
                .to(HANDLE_OPERATION_REQUEST);

        from(uri(config.getPubSubProjectName(), properties.getDeleteResponseSubscriptionName()))
                .unmarshal().json(DeleteOperationResponse.class)
                .to(OPERATION_RESPONSE);

        //At some point other things in the system may want to take an interest in the completion of a delete process, in which case
        //this should be routed via a pubsub topic rather than just a direct internal camel route within this service
        from(DELETE_OPERATION_RESPONSE_AGGREGATED)
                .to(AGGREGATE_OPERATION_RESPONSE);
    }

    private String uri(String pubSubProjectName, String topicName) {
        String serviceAccount = isBlank(serviceAccountKey) ? "" : String.format("?serviceAccountKey=file://%s", serviceAccountKey);
        return String.format("google-pubsub:%s:%s%s", pubSubProjectName, topicName, serviceAccount);
    }
}
