package com.pkb.domain.routing;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.domain.pubsub.payload.GrantAccessToOrgRequest;
import com.pkb.domain.pubsub.payload.GrantOrgAccessRequest;
import org.jetbrains.annotations.NotNull;

import static org.apache.commons.lang3.StringUtils.isBlank;

public class CryptoAccessGrantRequestRouteBuilder extends AbstractNamespaceAwareRouteBuilder {

    private final PubSubNamespaceService pubSubNamespaceService;
    private final String serviceAccountKey;
    private final PhrConfig config;

    public static final String GRANT_ACCESS_TO_ORG_REQUEST = "direct:grantAccessToOrgRequest";
    public static final String GRANT_ORG_ACCESS_REQUEST = "direct:grantOrgAccessRequest";

    public CryptoAccessGrantRequestRouteBuilder(PubSubNamespaceService pubSubNamespaceService,
                                                String serviceAccountKey,
                                                PhrConfig config) {
        this.pubSubNamespaceService = pubSubNamespaceService;
        this.serviceAccountKey = serviceAccountKey;
        this.config = config;
    }

    @Override
    public @NotNull PubSubNamespaceService camelNamespaceService() {
        return pubSubNamespaceService;
    }

    @Override
    public boolean throwExceptionForNamespaceMismatch() {
        return true;
    }

    @Override
    public void configure() throws Exception {
        from(CryptoAccessGrantRequestRouteBuilder.GRANT_ACCESS_TO_ORG_REQUEST)
                .marshal().avro(GrantAccessToOrgRequest.getClassSchema())
                .convertBodyTo(byte[].class)
                .to(uri(config.getPubSubProjectName(), config.getGrantAccessToOrgRequestTopic()));

        from(CryptoAccessGrantRequestRouteBuilder.GRANT_ORG_ACCESS_REQUEST)
                .marshal().avro(GrantOrgAccessRequest.getClassSchema())
                .convertBodyTo(byte[].class)
                .to(uri(config.getPubSubProjectName(), config.getGrantOrgAccessRequestTopic()));
    }

    private String uri(String pubSubProjectName, String topicName) {
        String serviceAccount = isBlank(serviceAccountKey) ? "" : String.format("?serviceAccountKey=file://%s", serviceAccountKey);
        return String.format("google-pubsub:%s:%s%s", pubSubProjectName, topicName, serviceAccount);
    }
}
