package com.pkb.service.reminders;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.entities.enums.PropertyName;
import com.pkb.exception.PKBException;
import com.pkb.reminders.ReminderRemote;
import com.pkb.reminders.dto.EHRMessageDTO;
import com.pkb.service.user.impl.UserManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;

import java.time.Instant;
import java.util.List;

public class ReminderManager extends TransactionManager {

    private final UserManager userManager;

    public ReminderManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, UserManager userManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.userManager = userManager;
    }

    /**
     * gets a batch of email addresses and first names needed to send reminder emails.
     *
     * The batch will return all active patients with valid email addresess who havent' opted out
     *
     * @return a list of strings. Each string contains [email,firstname]
     */
    public List<PKBPerson> getSymptomReminderEmailAddressBatch(int batchSize, long minPersonId) {

        try {
            ReminderRemote reminderBean = getBeanFactory().getReminderBean();
            return reminderBean.getSymptomReminderEmailAddressBatch(batchSize, minPersonId);
        } catch (Exception e) {
            throw new PKBException("Failed to get email reminders for symptoms email", e);
        }
    }

    /**
     * Gets a batch of patients needed to send unread message reminder emails.
     */
    public List<EHRMessageDTO> getUnreadMessageReminderEmailAddressBatch(int batchSize, int offset) {
        ReminderRemote reminderBean = getBeanFactory().getReminderBean();
        return reminderBean.getUnreadMessageReminderEmailAddressBatch(batchSize, offset, dateTimeService.yesterday());
    }

    /**
     * Gets a batch of patients needed to send unread document (HL7) reminder emails.
     */
    public List<EHRMessageDTO> getUnreadDocumentReminderEmailAddressBatch(int batchSize, int offset) {
        ReminderRemote reminderBean = getBeanFactory().getReminderBean();
        return reminderBean.getUnreadDocumentReminderEmailAddressBatch(batchSize, offset, dateTimeService.yesterday());
    }

    /**
     * Gets a list of patient records needed to send child birthday reminder emails.
     *
     * This will return:
     * active patients
     * with carers
     * with birthdays in the given date range
     * who do NOT have a PKBPersonProperty set "true" with the given name
     * 
     * Side effect: for each patient returned, assume we'll email successfully (so ADD the property now)
     */
    public List<PKBPerson> claimChildBirthdayEmailRecipients(Instant birthdayFrom, Instant birthdayTo, PropertyName reminderPropertyName) {
        try {
            ReminderRemote reminderBean = getBeanFactory().getReminderBean();
            return reminderBean.claimChildBirthdayEmailRecipients(birthdayFrom, birthdayTo, reminderPropertyName);
        } catch (Exception e) {
            throw new PKBException("Failed to get children records for birthday reminder email", e);
        }
    }

    /**
     * Tries to opts out the specified person and returns a boolean to indicate whether the action occurred
     *
     * Will return false if the person isn't a patient or if the person has already opted out to indicate that no action has been taken
     *
     * @param personId
     * @return
     */
    public boolean optOutSymptomReminder(Long personId) {

        PKBPerson person = userManager.getPKBPerson(personId);
        if (!person.isPatient()) {
            return false;
        }

        try {
            ReminderRemote reminderBean = getBeanFactory().getReminderBean();
            return reminderBean.optOut(person);
        } catch (Exception e) {
            throw new PKBException("Failed to opt out person id:" + personId, e);
        }

    }

    /**
     * Tries to opt in the specified person and returns a boolean to indicate whether the action occurred
     *
     * Will return false if the person isn't a patient or if the person has already opted in to indicate that no action has been taken
     *
     * @param personId
     * @return
     */
    public boolean optInSymptomReminder(Long personId) {
        PKBPerson person = userManager.getPKBPerson(personId);
        if (!person.isPatient()) {
            return false;
        }

        try {
            ReminderRemote reminderBean = getBeanFactory().getReminderBean();
            return reminderBean.optIn(person);
        } catch (Exception e) {
            throw new PKBException("Failed to opt out person id:" + personId, e);
        }

    }

    /**
     * Returns whether the specified person is opted in
     *
     * @param personId
     * @return
     */
    public boolean isOptedIn(Long personId) {
        try {
            ReminderRemote reminderBean = getBeanFactory().getReminderBean();
            return reminderBean.isOptedIn(personId);
        } catch (Exception e) {
            throw new PKBException("Failed to opt out person id:" + personId, e);
        }
    }

}
