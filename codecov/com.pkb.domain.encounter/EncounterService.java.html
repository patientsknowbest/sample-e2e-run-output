package com.pkb.domain.encounter;

import com.pkb.authz.AuthorizationService;
import com.pkb.datamodel.Encounter;
import com.pkb.domain.SynchUploadedDataService;
import com.pkb.domain.SystemUserTypeNotPermittedException;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.criteria.ImmutableCriteriaResult;
import com.pkb.domain.repository.EncounterQuery;
import com.pkb.domain.repository.EncounterRepository;
import com.pkb.domain.repository.ImmutableEncounterQuery;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import static com.pkb.entities.enums.UploadedDataDestination.ENCOUNTER;
import static io.vavr.control.Either.left;

public class EncounterService {
    private final AuthorizationService authorizationService;
    private final EncounterRepository encounterRepository;
    private final SynchUploadedDataService synchUploadedDataService;

    public EncounterService(AuthorizationService authorizationService,
                            EncounterRepository encounterRepository,
                            SynchUploadedDataService synchUploadedDataService) {
        this.authorizationService = authorizationService;
        this.encounterRepository = encounterRepository;
        this.synchUploadedDataService = synchUploadedDataService;
    }

    public @NotNull Either<RuntimeException, CriteriaResult<Encounter>> findEncounters(@NotNull EncounterCriteria criteria) {
        if (criteria.authzData().inputs().authenticatedSystemUser().isDefined()) {
            return left(new SystemUserTypeNotPermittedException());
        }
        return authorizationService.authorize(criteria.authzData())
                .checkSingleResult(() -> {
                    synchUploadedDataService.synchUploadedData(criteria.authzData(), ENCOUNTER);
                    EncounterQuery query = ImmutableEncounterQuery
                            .builder()
                            .authData(criteria.authzData())
                            .dateRestriction(criteria.getDateRestriction())
                            .maxNumberOfResources(criteria.getPageSize())
                            .build();

                    CriteriaResult<Encounter> result = encounterRepository.findBy(query);

                    return ImmutableCriteriaResult.<Encounter> builder()
                            .addAllPage(result.getPage())
                            .total(result.getTotal())
                            .build();
                });
    }
}
