package com.pkb.domain.deletion;

import com.pkb.deletion.model.DeleteOperationMessage;
import com.pkb.deletion.model.DeleteOperationRequest;
import com.pkb.deletion.model.DeletionRules;
import com.pkb.deletion.model.DeletionType;
import com.pkb.deletion.model.EntityDeletionRequest;
import com.pkb.deletion.model.ImmutableEntityDeletionRequest;
import com.pkb.deletion.model.errors.DeletionServiceError;
import com.pkb.deletion.model.errors.ImmutableGeneralValidationFailure;
import com.pkb.deletion.service.AbstractDatabaseModelDeletionHandler;
import com.pkb.deletion.service.DatabaseModelBasedDeletionService;
import com.pkb.repository.PersonRepository;
import io.vavr.control.Either;
import org.springframework.stereotype.Component;

import java.util.List;

/**
 * Handler that configures the behaviour of PHR for deleting PKBPerson entities.
 * See {@link AbstractDatabaseModelDeletionHandler} for more details.
 */
@Component
public class PersonDeletionHandler extends AbstractDatabaseModelDeletionHandler<Long> {

    private final PersonRepository personRepository;

    public PersonDeletionHandler(PersonRepository personRepository, DatabaseModelBasedDeletionService databaseModelBasedDeletionService) {
        super(databaseModelBasedDeletionService, DeletionType.PERSON);
        this.personRepository = personRepository;
    }

    @Override
    protected DeletionRules getDeletionRules(EntityDeletionRequest<Long> entityDeletionRequest) {
        return new PersonDeletionRules(entityDeletionRequest);
    }

    @Override
    protected Either<DeletionServiceError, EntityDeletionRequest<Long>> getEntityDeletionRequest(DeleteOperationRequest originalRequest) {
        List<Long> accountIds = personRepository.findAccountIdsByPersonIds(originalRequest.getEntityIdMap().keySet());
        if (accountIds.size() != originalRequest.getEntityIdMap().size()) {
            String message = "Some of person ids supplied for deletion either do not exist or do not have a valid account record";
            return Either.left(ImmutableGeneralValidationFailure.builder().requestId(originalRequest.getRequestId()).message(message).build());
        }



        EntityDeletionRequest<Long> accountDeleteRequest = ImmutableEntityDeletionRequest.<Long>builder()
                .from((DeleteOperationMessage) originalRequest)
                .rootTable(PhrModel.ACCOUNT)
                .rootEntityIds(accountIds)
                .build();
        return Either.right(accountDeleteRequest);
    }

}
