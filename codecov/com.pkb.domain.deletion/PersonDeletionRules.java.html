package com.pkb.domain.deletion;

import com.pkb.deletion.model.DatabaseTable;
import com.pkb.deletion.model.DeletionRules;
import com.pkb.deletion.model.DeletionValidator;
import com.pkb.deletion.model.EntityDeletionRequest;
import com.pkb.deletion.model.errors.TableValidationFailure;
import com.pkb.deletion.repository.DeleteOperationRepository;
import io.vavr.control.Either;

import java.util.Map;
import java.util.Set;
import java.util.UUID;

/**
 * Implementation of {@link DeletionRules} for PKBPerson
 * objects. See interface docs for more details.
 */
public class PersonDeletionRules implements DeletionRules {

    private final EntityDeletionRequest<Long> request;


    PersonDeletionRules(EntityDeletionRequest<Long> request) {
        this.request = request;
    }

    private static final Set<DatabaseTable<?>> MUST_BE_EMPTY = Set.of(
        PhrModel.TEAM,
        PhrModel.ORG,
        PhrModel.BULK_INVITATION_OUTCOME,
        PhrModel.ORG_NETWORK,
        PhrModel.ORG_NETWORK_MEMBER,
        PhrModel.PH_PLAN_TEMPLATE
    );

    private static final Set<DatabaseTable<?>> MUST_BE_EMPTY_IF_NOT_FORCE = Set.of(
            PhrModel.MENU_DATA
    );

    private static final Set<DatabaseTable<?>> NOT_GREATER_THAN_ORIGINAL_DELETION_LIST = Set.of(
      PhrModel.ACCOUNT
    );

    @Override
    public <T> DeletionValidator<T> getValidator(DatabaseTable<T> table) {

        if(MUST_BE_EMPTY.contains(table)) {
            return DeletionValidator.validateDeleteCount("Not allowed to delete from [" + table.getTableName() + "]. Ids:\n", count -> count == 0, table);
        }

        if (!request.isForceDelete() && MUST_BE_EMPTY_IF_NOT_FORCE.contains(table)) {
            return DeletionValidator.validateDeleteCount("Not allowed to delete from [" + table.getTableName() + "] without force delete. Ids:\n", count -> count == 0, table);
        }

        if(NOT_GREATER_THAN_ORIGINAL_DELETION_LIST.contains(table)) {
            return DeletionValidator.listNotBiggerThan(request.getEntityIdMap().size(), table.getTableName() + " can't have more entries than original list of persons to delete. Ids:\n", table);
        }

        if(PhrModel.PERSON.equals(table)) {
            return DeletionValidator.validateFilteredCount("Mustn't delete persons that weren't in original list. Extra person ids:\n",
                    " id_to_delete::bigint NOT IN (:ids)", count -> count == 0, Map.of("ids", request.getEntityIdMap().keySet()), table);
        }

        if(PhrModel.MENU_DATA.equals(table)) {
            return new DeletionValidator<>(table) {
                @Override
                public Either<TableValidationFailure, Integer> validate(UUID requestId, UUID operationId, DeleteOperationRepository queryExecutor) {
                    String schema = queryExecutor.getDeletionServiceSchema();
                    String sql = "SELECT md.id from app.menu_data md INNER JOIN " + schema + ".delete_operation_id doi ON doi.operation_id = '" + operationId + "' AND md.id = doi.id_to_delete::bigint WHERE account_id NOT IN (:accountIds);";
                    return DeletionValidator.convertErrorIdList(requestId, operationId, "Tried to delete the following records from menu_data that belong to users other than those specified for deletion",
                            table, queryExecutor.selectList(sql, Map.of("accountIds", request.getRootEntityIds()), table.getPkClass()));
                }
            };
        }

        return DeletionValidator.acceptAll(table);
    }
}
