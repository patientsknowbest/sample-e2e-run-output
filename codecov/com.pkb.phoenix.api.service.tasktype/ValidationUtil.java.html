package com.pkb.phoenix.api.service.tasktype;

import javax.mail.internet.AddressException;
import javax.mail.internet.InternetAddress;
import java.util.Collection;
import java.util.List;
import java.util.Optional;
import java.util.UUID;
import java.util.function.Supplier;
import java.util.regex.Pattern;

import static org.apache.commons.lang3.StringUtils.containsAny;
import static org.apache.commons.lang3.StringUtils.isBlank;

public class ValidationUtil {

    private static final Pattern ZERO_WIDTH_SPACE_AND_NO_BREAK_SPACE_UNICODE_CHARACTERS = Pattern.compile("[\u200B\u200C\u200D\uFEFF\u00A0]");

    public static Optional<String> validateEmail(String email) {
        final String valueTypeDesc = "Email address";

        return firstPresent(List.of(
                () -> checkBlank(email, valueTypeDesc).get(),
                () -> checkForWhiteSpaces(email, valueTypeDesc).get(),
                () -> checkIfHasZeroWidthOrNonBreakingSpaceCharacters(email, valueTypeDesc).get(),
                () -> checkIfEmailIsValid(email).get()));
    }

    public static Optional<String> validatePrivateId(String privateId) {
        final String valueTypeDesc = "Private Id";
        return firstPresent(List.of(
                () -> checkBlank(privateId, valueTypeDesc).get(),
                () -> checkForWhiteSpaces(privateId, valueTypeDesc).get(),
                () -> checkIfHasZeroWidthOrNonBreakingSpaceCharacters(privateId, valueTypeDesc).get(),
                () -> checkIfValueIsANumber(privateId).get()));
    }

    public static Optional<String> validateFreshDeskId(String fdId) {
        final String valueTypeDesc = "FreshDesk Ticket Id";
        return firstPresent(List.of(
                () -> checkBlank(fdId, valueTypeDesc).get(),
                () -> checkForWhiteSpaces(fdId, valueTypeDesc).get(),
                () -> checkIfHasZeroWidthOrNonBreakingSpaceCharacters(fdId, valueTypeDesc).get(),
                () -> checkFreshDeskId(fdId).get()));
    }

    public static Optional<String> validateMessageId(String messageId) {
        final String valueTypeDesc = "Message Id";
        return firstPresent(List.of(
                () -> checkBlank(messageId, valueTypeDesc).get(),
                () -> checkIfHasZeroWidthOrNonBreakingSpaceCharacters(messageId, valueTypeDesc).get(),
                () -> checkIsUUID(messageId).get()));
    }

    public static Optional<String> validateApiKey(String apiKey) {
        final String valueTypeDesc = "API key";
        return firstPresent(List.of(
                () -> checkBlank(apiKey, valueTypeDesc).get(),
                () -> checkIfHasZeroWidthOrNonBreakingSpaceCharacters(apiKey, valueTypeDesc).get(),
                () -> checkForWhiteSpaces(apiKey, valueTypeDesc).get()));
    }

    public static Optional<String> validateNationalID(String apiKey) {
        final String valueTypeDesc = "National Id";
        return firstPresent(List.of(
                () -> checkBlank(apiKey, valueTypeDesc).get(),
                () -> checkIfHasZeroWidthOrNonBreakingSpaceCharacters(apiKey, valueTypeDesc).get(),
                () -> checkForWhiteSpaces(apiKey, valueTypeDesc).get()));
    }

    private static Supplier<Optional<String>> checkBlank(String value, String valueTypeDesc) {
        if (isBlank(value)) {
            return () -> Optional.of(String.format("%s is mandatory!", valueTypeDesc));
        }
        return Optional::empty;
    }

    private static Supplier<Optional<String>> checkIsUUID(String value) {
        try {
            UUID.fromString(value);
            return Optional::empty;
        } catch (IllegalArgumentException ignored) {
            return () -> Optional.of(String.format("[%s] is not a valid Message Id!", value));
        }
    }

    private static Supplier<Optional<String>> checkForWhiteSpaces(String value, String valueTypeDesc) {
        if (containsAny(value, ' ')) {
            return () -> Optional.of(String.format("%1$s is invalid. Please remove all the whitespaces from the %1$s!", valueTypeDesc));
        }
        return Optional::empty;
    }

    private static Supplier<Optional<String>> checkIfEmailIsValid(String email) {
        try {
            InternetAddress emailAddr = new InternetAddress(email);
            emailAddr.validate();
        } catch (AddressException ex) {
            return () -> Optional.of(String.format("Email address=[%s] is not valid! Cause=[%s]", email, ex.getMessage()));
        }
        return Optional::empty;
    }

    private static Supplier<Optional<String>> checkIfValueIsANumber(String valueToBeValidated) {
        try {
            Long.parseLong(valueToBeValidated);
            return Optional::empty;
        } catch (NumberFormatException ignored) {
            return () -> Optional.of(String.format("Entered value=[%s] is not valid!", valueToBeValidated));
        }
    }

    private static Supplier<Optional<String>> checkIfHasZeroWidthOrNonBreakingSpaceCharacters(String valueToValidate, String valueTypeDesc) {
        if (ZERO_WIDTH_SPACE_AND_NO_BREAK_SPACE_UNICODE_CHARACTERS.matcher(valueToValidate).find()) {
            return () -> Optional.of(String.format("%s=[%s] is not valid! Cause= The value contains " +
                    "at least one invisible character. Please remove it!", valueTypeDesc, valueToValidate));
        }
        return Optional::empty;
    }

    private static Supplier<Optional<String>> checkFreshDeskId(String value) {
        //First we need to remove the FD_ prefix so that multiple underscores can be validated in the next step.
        var fdId = value.toUpperCase().replace("FD_", "");

        //Check if FD ID contains more than one underscore and if so an error message will be returned.
        long count = fdId.chars().filter(ch -> ch == '_').count();
        if (count > 1) {
            return () -> Optional.of("FreshDesk Ticket Id cannot contain more than one underscore!");
        }

        //Now the underscore can be removed.
        fdId = fdId.replace("_", "");
        try {
            Long.parseLong(fdId);
            return Optional::empty;
        } catch (NumberFormatException ignored) {
            return () -> Optional.of("FreshDesk Ticket Id is not valid!");
        }
    }

    private static <T> Optional<T> firstPresent(Collection<Supplier<Optional<T>>> opts) {
        return opts.stream()
                .map(Supplier::get)
                .filter(Optional::isPresent)
                .map(Optional::get)
                .findFirst();
    }
}
