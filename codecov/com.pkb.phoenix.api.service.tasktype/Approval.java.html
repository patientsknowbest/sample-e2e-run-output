package com.pkb.phoenix.api.service.tasktype;

public enum Approval {
    /**
     * Approval is not required (e.g.: getting some basic info about Patients)
     */
    NONE,
    /**
     * Permanently approved (e.g.: IG granted auto approval for task type, like CIE message deletion)
     */
    SIMPLE,
    /**
     * Per instance based approval is need (e.g.: standard - non CIE - message deletion.
     */
    SINGLE
}
