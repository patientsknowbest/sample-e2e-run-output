package com.pkb.fhir.migration.configuration;

import ca.uhn.fhir.context.FhirContext;
import ca.uhn.fhir.rest.api.EncodingEnum;
import ca.uhn.fhir.rest.client.api.IGenericClient;
import ca.uhn.fhir.rest.client.api.IHttpRequest;
import ca.uhn.fhir.rest.client.interceptor.BearerTokenAuthInterceptor;
import com.pkb.fhir.common.configuration.ApplicationConfig;
import org.keycloak.OAuth2Constants;
import org.keycloak.admin.client.Keycloak;
import org.keycloak.admin.client.KeycloakBuilder;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class FhirStoreConfig {

    @Bean
    public FhirContext fhirContext() {
        return FhirContext.forR4();
    }

    @Bean
    public IGenericClient fhirClient(FhirContext fhirContext, ApplicationConfig applicationConfig) {
        Keycloak keycloak = KeycloakBuilder.builder()
                .serverUrl(applicationConfig.getKeycloakUrl() + "/auth")
                .realm(applicationConfig.getKeycloakRealm())
                .grantType(OAuth2Constants.CLIENT_CREDENTIALS)
                .clientId(applicationConfig.getKeycloakClientId())
                .clientSecret(applicationConfig.getKeycloakClientSecret())
                .build();
        IGenericClient client = fhirContext.newRestfulGenericClient(applicationConfig.getAidboxUrl() + "/fhir");
        client.registerInterceptor(new BearerTokenAuthInterceptor("accessToken") {
            @Override
            public void interceptRequest(IHttpRequest theRequest) {
                theRequest.addHeader("Authorization", "Bearer " + keycloak.tokenManager().getAccessTokenString());
            }
        });
        client.setEncoding(EncodingEnum.JSON);
        return client;
    }
}
