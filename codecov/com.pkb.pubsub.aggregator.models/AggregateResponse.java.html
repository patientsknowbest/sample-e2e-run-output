package com.pkb.pubsub.aggregator.models;

import org.immutables.value.Value;

import java.util.List;
import java.util.Set;
import java.util.UUID;
import java.util.stream.Collectors;

public interface AggregateResponse<T extends AggregateResponse<T, R>, R extends AggregatableResponse<R, T>> {
    List<R> getResponses();

    Set<String> getRequiredResponseServices();

    @Value.Default
    default boolean isTimedOut() {
        return false;
    }


    @Value.Default
    default UUID getAggregatableRequestId() {
        return getResponses().stream().findFirst().map(AggregatableResponse::getAggregatableRequestId).orElseThrow();
    }

    @Value.Derived
    default Status getStatus() {
        boolean isComplete = getResponses().stream().map(AggregatableResponse::getRespondingService).collect(Collectors.toList()).containsAll(getRequiredResponseServices());
        if (isComplete) {
            return Status.COMPLETE;
        }
        return isTimedOut() ? Status.TIMED_OUT : Status.INCOMPLETE;
    }

    @Value.NonAttribute
    T combine(T other);


    public static enum Status {INCOMPLETE, COMPLETE, TIMED_OUT}
}
