@Path("/v2-beta/users")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Component
@Api(tags = { "Users" }, value = "User & Team management V2 service")
public class V2UserService {

    private final static Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private TeamManager teamMgr;

    private TeamUserManager teamUserMgr;

    private UserManager userManager;

    private UserApiHelper userApiHelper;

    private RestUtil util;

    private PkbPersonToApiUserConverter pkbPersonToApiUserConverter;

    @Autowired
    public void setUtil(RestUtil util) {
        this.util = util;
    }

    @Autowired
    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    @Autowired
    public void setTeamMgr(TeamManager teamMgr) {
        this.teamMgr = teamMgr;
    }

    @Autowired
    public void setTeamUserMgr(TeamUserManager teamUserMgr) {
        this.teamUserMgr = teamUserMgr;
    }

    @Autowired
    public void setUserApiHelper(UserApiHelper userApiHelper) {
        this.userApiHelper = userApiHelper;
    }

    @Autowired
    public void setPkbPersonToApiUserConverter(PkbPersonToApiUserConverter pkbPersonToApiUserConverter) {
        this.pkbPersonToApiUserConverter = pkbPersonToApiUserConverter;
    }

    @GET
    @Path("/nationalIdTypes")
    @Timed
    @RequiresScope(scopes = { PUBLIC })
    @ApiOperation(value = "Scope: PUBLIC", notes = "This call returns all supported NationalIdTypes", response = NationalIdType.class, code = 200)
    public ApiReadResponse<NationalIdType> getApiRefsNational(
            @ApiParam(value = "Auth Token in the form Bearer <my-token>") @HeaderParam(value = "Authorization") String authToken) {
        return new ApiReadResponse<>(NationalIdType.values());
    }

    @GET
    @Path("/byNationalId/{nationalId}/{nationalIdType}")
    @Timed(hideParams = "nationalId")
    @RequiresScope(scopes = { PATIENT, CLINICIAN, TEAMCOORD, SITE })
    @ApiOperation(value = "Scope: PATIENT, CLINICIAN, TEAMCOORD", notes = "Given a national number, this call provides a way to retrieve a user", response = ApiUser.class)
    public Response getNationalNumberByPatientId(
            @ApiParam(value = "a national ID such as an NHS Number") @PathParam("nationalId") String nationalId,
            @ApiParam(value = "Type of Id. See /nationalIdTypes for options") @PathParam("nationalIdType") NationalIdType nationalIdType) {
        if (Strings.isNullOrEmpty(nationalId)) {
            String message = "nationalId cannot be null";
            LOGGER.error(message);
            throw new WebApplicationException(message, Response.Status.BAD_REQUEST);
        }

        PKBPerson.Lazy[] lazies = new PKBPerson.Lazy[] { PKBPerson.Lazy.NATIONAL_AND_LOCAL_IDS,
                PKBPerson.Lazy.CONTACTS, PKBPerson.Lazy.CONTACTS_DEEP, PKBPerson.Lazy.PROPERTIES };

        Option<PKBPerson> maybePerson = Optional.ofNullable(nationalIdType)
                .flatMap(n -> n.getValidNationalIdAndType(nationalId))
                .map(vnid -> userManager.getPKBPersonByNationalId(vnid, lazies))
                .map(e ->  PersonAmbiguityHandler.getMaybePersonOrNone(e,"nationalId"))
                .orElse(Option.none());

        if (maybePerson.isDefined()) {
            util.getCommonParams(maybePerson.get().getId());
            return Response.ok(pkbPersonToApiUserConverter.convert(maybePerson.get())).status(Response.Status.OK).build();
        } else {
            ApiReadResponse<String> response = new ApiReadResponse<>();
            response.setResults(emptyList());
            return Response.ok(response).status(Response.Status.OK).build();
        }

    }
}
