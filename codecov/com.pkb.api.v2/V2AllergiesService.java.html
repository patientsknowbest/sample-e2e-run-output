package com.pkb.api.v2;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.aspect.annotation.Timed;
import com.pkb.api.error.ApiValidationError;
import com.pkb.api.util.PKBApiHelper;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.api.util.v2.AllergiesApiHelper;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.api.util.v2.exception.SharingDisabledException;
import com.pkb.api.v1.model.ApiReadResponse;
import com.pkb.api.v2.model.ApiAllergy;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import io.swagger.annotations.ApiParam;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.PathParam;
import javax.ws.rs.Produces;
import javax.ws.rs.QueryParam;
import java.util.List;

import static com.pkb.entities.enums.api.ApiClientScope.CLINICIAN;
import static com.pkb.entities.enums.api.ApiClientScope.PATIENT;

@Path("/v2-beta/allergies")
@Produces("application/json")
@Consumes("application/json")
@Component
@Api(tags = { "Allergies" }, value = "Manage access to the allergies service")
public class V2AllergiesService {

    private UserManager userManager;

    private AllergiesApiHelper allergiesApiHelper;

    private RestUtil util;

    @Autowired
    private void setUtil(RestUtil util) {
        this.util = util;
    }

    @Autowired
    public void setAllergiesApiHelper(AllergiesApiHelper allergiesApiHelper) {
        this.allergiesApiHelper = allergiesApiHelper;
    }

    @Autowired
    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    /**
     * Returns the up to date list of apiRefs supported by the system
     * <p/>
     * apiRefs are used to specify the correct coding of an allergy. They can also be used to apply a default name
     * to an allergy
     * <p/>
     * <br/>
     * <a href="http://dev.patientsknowbest.com/home/rest-api/scopes"><b>Scope</b></a>: PATIENT, CLINICIAN
     *
     * @return
     */
    @GET
    @Path("/apiRefs")
    @Timed
    @RequiresScope(scopes = { PATIENT, CLINICIAN })
    @ApiOperation(value = "Scope: PATIENT, CLINICIAN", notes = "This operation is no longer supported.")
    public List<String> getApiRefs() {
        throw new ApiValidationError("Allergy apiRefs are no longer supported");
    }

    @GET
    @Path("/forPatient/{patientId}")
    @Timed(hideParams = "patientId")
    @RequiresScope(scopes = CLINICIAN)
    @ApiOperation(value = "Scope: CLINICIAN", notes = "Returns allergies for the specified patient. "
            + "Please see the <a href=\"http://dev.patientsknowbest.com/home/data-model#TOC-Allergy\">[[Allergy]]</a> section of our "
            + "Data Model for more information.")
    public ApiReadResponse<ApiAllergy> getAllergiesForPatient(
            @ApiParam("The PKB ID of the patient whose allergies should be returned.") @PathParam("patientId") long patientId,
            @ApiParam("This parameter is not currently supported.") @QueryParam("updatedSince") String updatedSinceString,
            @ApiParam("This parameter is not currently supported.") @QueryParam("apiRef") List<String> apiRefs,
            @ApiParam("This parameter is not currently supported.") @QueryParam("includeNegativeAllergies") boolean includeNegativeAllergies) {
        PKBCommonApiParams params = util.getCommonParams(patientId,  true);
        PKBPerson loggedInUser = params.getPerson();
        try {
            if (!apiRefs.isEmpty()) {
                throw new ApiValidationError("Allergy search by apiRef is no longer supported");
            }

            return PKBApiHelper.buildReadResponse(allergiesApiHelper.getAllergiesForPatient(params.getRequestContext(), patientId,
                    loggedInUser, includeNegativeAllergies));
        } catch (SharingDisabledException sde) {
            throw sde; // is handled by exception mapper
        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error getting allergies", e);
        }
    }
}
