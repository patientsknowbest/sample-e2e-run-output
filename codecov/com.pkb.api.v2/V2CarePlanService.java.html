/**
 * REST Service to create /update / fetch careplans for a patient
 * Supported
 * - Empty template plans
 * - All root level information in care plans
 * <p>
 * Not supported
 * - Fields : ACTION , GREEN , AMBER , RED
 * - Plan templates
 * - Attachments
 *
 * @author pravina
 */
@Path("/v2-beta/careplans")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Component
@Api(tags = {"Careplans"}, value = "Manage access to the careplans service")
public class V2CarePlanService {

    static final String V1_TREATMENT_PLANS_PATH = "/v2-beta/careplans";

    @Autowired
    private PlanManager planManager;

    @Autowired
    private UserManager userManager;

    @Autowired
    private RestUtil util;

    @Autowired
    private CarePlanConverter carePlanConverter;

    @GET
    @Path("/{patientId}")
    @Timed(hideParams = "patientId")
    @RequiresScope(scopes = {CLINICIAN, TEAMCOORD})
    @ApiOperation(value = "Scope: CLINICIAN, TEAMCOORD", notes = "Get careplans for the patient given by {patientId}, optionally linked to another data point",
            response = ApiCarePlan.class, responseContainer = "List")
    public ApiReadResponse<ApiCarePlan> getCarePlansForPatient(
            @PathParam("patientId") String patientId) throws ConversionException {
        PKBCommonApiParams params = util.getCommonParams(Long.parseLong(patientId), true);
        PKBPerson loggedInUser = params.getPerson();
        return getCarePlans(patientId, params, loggedInUser);
    }

    private ApiReadResponse<ApiCarePlan> getCarePlans(String patientId, PKBCommonApiParams params, PKBPerson loggedInUser) throws ConversionException {
        Long patientAccountId = userManager.getDefaultAccountId(Long.parseLong(patientId));
        List<PHPlan> plans = planManager.getPHPlans(params.getRequestContext(), loggedInUser.getId(), patientAccountId, 0, 0,
                Lazy.CREATOR_DETAILS, Lazy.LAST_EDITOR_DEATILS);
        List<ApiCarePlan> carePlans = carePlanConverter.toApiModel(plans);
        return PKBApiHelper.buildReadResponse(carePlans);
    }
}
