package com.pkb.api.v2;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.aspect.annotation.Timed;
import com.pkb.api.error.ApiValidationError;
import com.pkb.api.util.PKBApiHelper;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.api.util.v2.TestsApiHelper;
import com.pkb.api.v1.model.ApiReadResponse;
import com.pkb.api.v2.model.ApiTestResult;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.DataDelayService;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import io.swagger.annotations.ApiParam;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import javax.servlet.http.HttpServletRequest;
import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.QueryParam;
import javax.ws.rs.core.Context;
import javax.ws.rs.core.MediaType;
import java.util.List;
import java.util.stream.Collectors;

import static com.pkb.entities.enums.api.ApiClientScope.PATIENT;

@Path("/v2-beta/tests")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Component
@Api(tags = {"Tests"}, value = "Manage access to the test service")
public class V2TestService {

    @Autowired
    private TestsApiHelper testsApiHelper;

    @Autowired
    private RestUtil util;

    @Autowired
    private DataDelayService dataDelayService;

    @Context
    private HttpServletRequest httpRequest;

    @GET
    @Timed
    @Path("/results/")
    @RequiresScope(scopes = {PATIENT})
    @ApiOperation(value = "Scope: PATIENT", notes =
            "Allows a patient to search for their test results based on one or more provided " +
                    "<a href=\"http://loinc.org\">LOINC codes</a>. Optionally, set latestOnly to true to retrieve only " +
                    "the most recent test result for each matching test type. " +
                    "Test results that have a non-expired delay will be omitted from the search results. " +
                    "Note that in PKB a test type is defined not just by the code but also by the units (this ensures " +
                    "all results for a given type can be graphed together safely). As such, searching by a single LOINC " +
                    "code might match more than one PKB test type, and so more than one result might be returned even " +
                    "with latestOnly set to true. " +
                    "Since this endpoint searches by LOINC code, test results that are not mapped to a supported LOINC " +
                    "code cannot be retrieved using this API call. " +
                    "See our guidance on the <a href=\"https://deploy.patientsknowbest.com/integrations/HL7/Path/LOINC\">" +
                    "Benefits of LOINC mapping</a> for more information.",
            response = ApiTestResult.class,
            responseContainer = "List")
    public ApiReadResponse<ApiTestResult> getTestResults(
            @ApiParam(value = "One or more LOINC codes (repeat the parameter using & to provide multiple values). The " +
                    "search will only return results for test types associated with one of the provided codes. " +
                    "Example: ?loincCode=14743-9&loincCode=704-7",
                    example = "704-7",
                    required = true,
                    allowMultiple = true)
            @QueryParam("loincCode") List<String> loincCodes,
            @ApiParam(value = "If set to true then for each test type which matches the search criteria only the latest " +
                    "result will be returned. Defaults to false.",
                    example = "false")
            @QueryParam("latestOnly") Boolean latestOnly) {
        PKBCommonApiParams params = util.getCommonParams();
        if (loincCodes.isEmpty()) {
            throw new ApiValidationError("Please specify one or more loincs you want results for");
        }
        PKBPerson callingUser = params.getPerson();
        List<ApiTestResult> testResults = testsApiHelper.getTestResults(params.getRequestContext(), loincCodes, callingUser.getId(), Boolean.TRUE.equals(latestOnly));
        List<ApiTestResult> testResultsFilteredForDelays = testResults.stream().filter(x ->
                dataDelayService.findWhenBecomesVisibleForUserType(
                        (x.getResultDate() == null) ? null : x.getResultDate().toInstant(),
                        x.getDelayDisplayDays(),
                        callingUser.getUserType()).isEmpty()
        ).collect(Collectors.toList());
        return PKBApiHelper.buildReadResponse(testResultsFilteredForDelays);
    }
}
