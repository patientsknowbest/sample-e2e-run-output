/**
 * Service for getting, adding, updating and deleting diagnoses
 *
 * @author russell
 */
@Path("/v2-beta/diagnoses")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Component
@Api(tags = { "Diagnoses" }, value = "Manage access to the diagnoses service")
public class V2DiagnosisService {

    private DiagnosisManager diagnosisManager;

    private UserManager userManager;

    private PatientConsentManager userAccessManager;

    private DateTimeService dateTimeService;

    @Context
    private HttpServletRequest httpRequest;

    private RestUtil util;

    @Autowired
    private void setUtil(RestUtil util) {
        this.util = util;
    }

    private V2DiagnosisApiHelper getDiagnosisApiHelper() {
        return new V2DiagnosisApiHelper(diagnosisManager, userAccessManager, userManager, dateTimeService);

    }

    @Autowired
    public void setDiagnosisManager(DiagnosisManager diagnosisManager) {
        this.diagnosisManager = diagnosisManager;
    }

    @Autowired
    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    @Autowired
    public void setUserAccessManager(PatientConsentManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }

    @Autowired
    public void setDateTimeService(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    @GET
    @Path("/apiRefs")
    @ApiOperation(value = "Scope: PATIENT, CLINICIAN", notes = "This operation is no longer supported.")
    @Timed
    @RequiresScope(scopes = { PATIENT, CLINICIAN })
    public List<String> getApiRefs(
            @HeaderParam(value = "Authorization") String authToken) {
        DiagnosisApiRef[] refs = DiagnosisApiRef.values();
        ArrayList<String> refStrings = new ArrayList<>();
        for (DiagnosisApiRef ref : refs) {
            refStrings.add(ref.getApiRef());
        }
        return refStrings;
    }

    @GET
    @Path("/forPatient/{patientId}")
    @Timed(hideParams = "patientId")
    @RequiresScope(scopes = CLINICIAN)
    @ApiOperation(value = "Scope: CLINICIAN", notes = "Returns diagnoses for the specified patient. "
            + "Please see the <a href=\"http://dev.patientsknowbest.com/home/data-model#TOC-Diagnosis\">[[Diagnosis]]</a> section of our "
            + "Data Model for more information.")
    public ApiReadResponse<ApiDiagnosis> getDiagnosesForPatient(
            @ApiParam("The PKB ID of the patient whose diagnoses should be returned.") @PathParam("patientId") long patientId,
            @ApiParam("If provided, only diagnoses where <a href=\"http://dev.patientsknowbest.com/home/data-model#TOC-Data-Point\">"
                    + "[[Diagnosis.Entered Timestamp]]</a> is greater than the provided value will be returned.") @QueryParam("updatedSince") String updatedSinceString,
            @ApiParam("This parameter is not currently supported.") @QueryParam("apiRef") List<String> apiRefs,
            @ApiParam("This parameter is not currently supported.") @QueryParam("includeNegativeConditions") boolean includeNegativeConditions) {

        PKBCommonApiParams params = util.getCommonParams(patientId, true);
        PKBPerson loggedInUser = params.getPerson();
        ApiReadResponse<ApiDiagnosis> results = new ApiReadResponse<>();
        try {
            for (String apiRef : apiRefs) {
                if (DiagnosisApiRef.getCodeByApiRef(apiRef) == null) {
                    throw new ApiValidationError("Unrecognised apiRef - " + apiRef);
                }
            }
            Instant updatedSinceDate = null;
            if ((updatedSinceString != null) && !updatedSinceString.isEmpty()) {
                updatedSinceDate = DatatypeConverter.parseDateTime(updatedSinceString).toInstant();
            }
            results.setResults(getDiagnosisApiHelper().getDiagnoses(params.getRequestContext(),
                    apiRefs, patientId, loggedInUser, updatedSinceDate, includeNegativeConditions));
        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error getting diagnoses", e);
        }
        return results;
    }

}
