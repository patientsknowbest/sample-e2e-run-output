package com.pkb.reminders.dto;

import com.pkb.entities.enums.Route;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.user.entity.PKBPerson;
import io.vavr.control.Option;
import org.immutables.value.Value;
import org.jetbrains.annotations.Nullable;

import java.io.Serializable;
import java.time.LocalDateTime;
import java.util.UUID;

/**
 * Simple DTO to transfer message related data from app.menu_data.
 */
@Value.Immutable
public interface EHRMessageDTO extends Serializable {
    long getMessageId();

    PKBPerson getReceiver();

    Route getRoute();

    LocalDateTime getReceived();

    UUID getConversationId();

    Option<Org> getOrg();

    @Nullable Team getTeam();

    @Value.Default
    default String getSourceTitle() {
        return "";
    }

    @Value.Default
    default String getSourceFirstName() {
        return "";
    }

    @Value.Default
    default String getSourceLastName() {
        return "";
    }

    @Value.Default
    default String getSourceOrg() {
        return "";
    }
}
