package com.pkb.service.dataupload.hl7.processor;

import com.pkb.common.config.PhrConfig;
import com.pkb.exception.ApiCallMalformedException;
import com.pkb.reports.A19PatientStatusService;
import com.pkb.reports.A19Record;
import com.pkb.reports.RegistrationStatus;
import com.pkb.repository.legacy.CustomizedLegacyInstituteUserRepository;
import com.pkb.service.dataupload.hl7.HL7ConnContext;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.response.A19Response;
import com.pkb.service.dataupload.hl7.response.StaA19Response;
import com.pkb.service.dataupload.hl7.response.StandardA19Response;
import com.pkb.service.dataupload.hl7.segment.HL7QueryDetail;
import com.pkb.service.dataupload.hl7.segment.HL7QueryFilter;
import com.pkb.service.dataupload.hl7.value.HL7Date;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.user.entity.PKBPerson;
import io.vavr.Tuple2;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.Nullable;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.w3c.dom.Element;
import org.w3c.dom.NodeList;

import java.lang.invoke.MethodHandles;
import java.time.ZonedDateTime;
import java.util.Arrays;
import java.util.EnumSet;
import java.util.List;
import java.util.Optional;
import java.util.UUID;

import static com.pkb.util.Constants.POSTGRES_DATE_NEGATIVE_INFINITY_ZONED_UTC;
import static org.apache.commons.lang3.StringUtils.isEmpty;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class QryA19Processor {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private final TeamUserManager teamUserManager;
    private final A19PatientStatusService a19PatientStatusService;
    private final PhrConfig config;
    private final UUIDProvider uuidProvider;

    public QryA19Processor(TeamUserManager teamUserManager, A19PatientStatusService a19PatientStatusService, PhrConfig config, UUIDProvider uuidProvider) {
        this.teamUserManager = teamUserManager;
        this.config = config;
        this.a19PatientStatusService = a19PatientStatusService;
        this.uuidProvider = uuidProvider;
    }

    public String process(HL7XmlDoc hl7, HL7ConnContext hl7ConnContext, String nowInHL7Format) {

        // optional
        Optional<HL7QueryFilter> filter = parseQueryFilter(hl7);

        // This is required
        HL7QueryDetail detail = parseQueryDetail(hl7)
                .orElseThrow(() -> new ApiCallMalformedException("Required QRD segment is missing"));

        return process(detail, filter, hl7, hl7ConnContext, nowInHL7Format);
    }

    private String process(HL7QueryDetail detail, Optional<HL7QueryFilter> filter, HL7XmlDoc hl7, HL7ConnContext hl7ConnContext,
                           String nowInHL7Format) {

        switch (detail.getQueryFilterType().getString()) {
            case HL7QueryDetail.REGISTRATION_STATUS:
                return processSta(filter, hl7ConnContext, detail, hl7, nowInHL7Format);
            default:
                return processNonSta(detail, filter.orElse(null), hl7ConnContext, hl7, nowInHL7Format);
        }
    }

    private String processNonSta(HL7QueryDetail detail, HL7QueryFilter filter, HL7ConnContext hl7ConnContext,
                                 HL7XmlDoc hl7, String nowInHL7Format) {

        ZonedDateTime startDate = null, endDate = null;

        if (filter != null) {
            startDate = getStartTimestampElseNegativeInfinity(filter.getStartTimestamp());
            endDate = filter.getEndTimestamp().getZonedDateTime();
        }

        // include all patients if the query filter type is "APN"; otherwise include only patients w/ IDs
        boolean patientsWithIdsOnly = (!HL7QueryDetail.DEMOGRAPHICS_BY_NAME.equals(detail.getQueryFilterType().getString()));
        String patientId = detail.getPatientId().getString();
        boolean localIdFlag = HL7QueryDetail.ID_TYPE_LOCAL.equals(detail.getPatientIdType().getString());
        List<PKBPerson> personList = getPersonList(hl7ConnContext, startDate, endDate,
                patientId, localIdFlag, patientsWithIdsOnly);
        return buildMessage(personList, detail, filter, hl7, nowInHL7Format, hl7ConnContext);
    }

    private String processSta(Optional<HL7QueryFilter> filter, HL7ConnContext hl7ConnContext,
                              HL7QueryDetail detail, HL7XmlDoc hl7, String nowInHL7Format) {
        ZonedDateTime startDate = filter.map(f -> getStartTimestampElseNegativeInfinity(f.getStartTimestamp())).orElse(null);

        UUID lastSeenPublicId = filter.map(f -> f.getLastSeenId().getString()).filter(StringUtils::isNotBlank).map(UUID::fromString).orElse(null);
        String registrationStatusString = detail.getRegisteredStatus().getString();
        RegistrationStatus registrationStatus = null;
        if (isNotBlank(registrationStatusString)) {
            registrationStatus = RegistrationStatus.getRegistrationStatus(registrationStatusString)
                    .orElseThrow(() -> new ApiCallMalformedException("QRD-11.1 must be one of these values: " + Arrays.toString(RegistrationStatus.values())));
        }
        Tuple2<Boolean, List<A19Record>> personsWithRegStatus = getPersonsWithRegStatus(hl7ConnContext, startDate,
                detail.getPageSize().getString(), detail.getPageSizeDetail().getString(),
                registrationStatus, lastSeenPublicId);
        return buildMessageSTA(personsWithRegStatus, detail, filter, hl7, nowInHL7Format, hl7ConnContext);
    }

    private Tuple2<Boolean, List<A19Record>> getPersonsWithRegStatus(
            HL7ConnContext hl7ConnContext, @Nullable ZonedDateTime queryStartDate,
            String pageSize, String pageSizeDetail, @Nullable RegistrationStatus registeredStatus, @Nullable UUID lastSeenPublicId) {

        if (!isEmpty(pageSize) && !"RD".equals(pageSizeDetail)) {
            throw new ApiCallMalformedException("QRD-7.2 must be 'RD'");
        }

        if (hl7ConnContext.hasSourceTeam()) {
            throw new ApiCallMalformedException("Team-level STA queries not supported");
        } else {
            return a19PatientStatusService.getOrgPatientsWithRegStatusForDataUpload(
                    hl7ConnContext.getSourceOrg().getId(), queryStartDate,
                    getPageSize(pageSize), registeredStatus,
                    lastSeenPublicId);
        }
    }

    private int getPageSize(String requestedPageSize) {
        try {

            int systemPageSize = config.getHl7QryA19PageSize();

            if (isEmpty(requestedPageSize)) {
                return systemPageSize;
            }

            int requestVal = Integer.parseInt(requestedPageSize);

            if (requestVal <= 0) {
                throw new ApiCallMalformedException("QRD-7.1 must be a positive integer");
            }
            return Math.min(requestVal, systemPageSize);
        } catch (NumberFormatException nfe) {
            throw new ApiCallMalformedException("QRD-7.1 must be a positive integer", nfe);
        }
    }

    private List<PKBPerson> getPersonList(HL7ConnContext hl7ConnContext, @Nullable ZonedDateTime startDate,
                                          @Nullable ZonedDateTime endDate, String patientId,
                                          boolean localIdFlag, boolean patientsWithIdsOnly) {
        if (hl7ConnContext.hasSourceTeam()) {
            return teamUserManager.getTeamPatientsForDataUpload(
                    hl7ConnContext.getSourceTeam().orElse(null),
                    startDate, endDate, patientId, localIdFlag, patientsWithIdsOnly,
                    EnumSet.noneOf(CustomizedLegacyInstituteUserRepository.DataUploadOption.class));
        } else {
            return teamUserManager.getOrgPatientsForDataUpload(hl7ConnContext.getEHRRequestContext(),
                    hl7ConnContext.getSourceOrg(),
                    startDate, endDate, patientId, localIdFlag, patientsWithIdsOnly);
        }
    }

    private String buildMessage(List<PKBPerson> personList, HL7QueryDetail detail, HL7QueryFilter filter, HL7XmlDoc hl7,
                                String timestamp, HL7ConnContext hl7ConnContext) {

        A19Response response = new StandardA19Response(hl7, timestamp, detail, filter, hl7ConnContext, personList);

        doLogging(response, personList.size());

        return response.getFullResponse();
    }

    private String buildMessageSTA(Tuple2<Boolean, List<A19Record>> personsWithRegStatus,
                                   HL7QueryDetail detail, Optional<HL7QueryFilter> filter, HL7XmlDoc hl7,
                                   String timestamp, HL7ConnContext hl7ConnContext) {

        A19Response response = new StaA19Response(hl7, timestamp, detail, filter, hl7ConnContext, personsWithRegStatus, uuidProvider);

        doLogging(response, personsWithRegStatus._2.size());

        return response.getFullResponse();
    }

    private void doLogging(A19Response response, int size) {
        if (config.isDisplayOfSensitiveErrorInformationEnabled()) {
            LOGGER.info("SENDING RESPONSE: {}", response);
        } else {
            String responseBeforePatientData = response.getNonSensitiveResponse(); // for prod logging
            LOGGER.info("SENDING RESPONSE: {} [{} patients listed]", responseBeforePatientData, size);
        }
    }

    private Optional<HL7QueryDetail> parseQueryDetail(HL7XmlDoc hl7) {

        NodeList nodeList = hl7.getRoot().getElementsByTagName("QRD");

        if ((nodeList != null) && (nodeList.getLength() > 0)) {
            HL7QueryDetail dto = new HL7QueryDetail(hl7.getMessageProperties());
            dto.parseQrd((Element) nodeList.item(0));
            return Optional.of(dto);
        }
        return Optional.empty();
    }

    private Optional<HL7QueryFilter> parseQueryFilter(HL7XmlDoc hl7) {

        NodeList nodeList = hl7.getRoot().getElementsByTagName("QRF");

        if ((nodeList != null) && (nodeList.getLength() > 0)) {
            HL7QueryFilter dto = new HL7QueryFilter(hl7.getMessageProperties());
            dto.parseQrf((Element) nodeList.item(0));
            return Optional.of(dto);
        }
        return Optional.empty();
    }

    private ZonedDateTime getStartTimestampElseNegativeInfinity(HL7Date startTimestamp) {
        if (startTimestamp.getZonedDateTime() == null) {
            return POSTGRES_DATE_NEGATIVE_INFINITY_ZONED_UTC;
        }
        return startTimestamp.getZonedDateTime();
    }
}
