package com.pkb.service.emailmessage.sender.impl;

import com.pkb.common.config.PhrConfig;
import com.pkb.exception.MessageException;
import com.pkb.service.emailmessage.sender.ITemplateHandler;
import com.pkb.util.security.RelaxedHtmlMessageFilter;
import org.apache.velocity.Template;
import org.apache.velocity.VelocityContext;
import org.apache.velocity.app.Velocity;
import org.apache.velocity.runtime.RuntimeConstants;
import org.apache.velocity.tools.generic.DateTool;
import org.apache.velocity.tools.generic.EscapeTool;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.io.StringWriter;
import java.lang.invoke.MethodHandles;
import java.util.Locale;
import java.util.Map;
import java.util.Properties;
import java.util.TimeZone;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class VelocityTemplateHandler
        implements ITemplateHandler {
    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private static final String TEMPLATE_EXTENSION = ".vm";
    private static final Pattern VELOCITY_FILE_EXTENSION_PATTERN = Pattern.compile(".vm", Pattern.LITERAL);

    private Map<String, String> props = null;

    private boolean initialized = false;

    private PhrConfig config;

    private void init() {
        if (initialized) {
            return;
        }

        Properties properties = new Properties();
        // set defaults
        properties.setProperty(RuntimeConstants.RESOURCE_LOADER, "class");
        properties.setProperty("class.resource.loader.class",
                "org.apache.velocity.runtime.resource.loader.ClasspathResourceLoader");
        properties.put(RuntimeConstants.RUNTIME_LOG_NAME, "velocity");
        // settings that make it compatible with 1.7
        properties.put(RuntimeConstants.CONVERSION_HANDLER_CLASS, "none");
        properties.put(RuntimeConstants.SPACE_GOBBLING, "bc");
        properties.put(RuntimeConstants.CHECK_EMPTY_OBJECTS, "false");
        if (config.isVelocityRuntimeReferenceStrictEnabled()) {
            properties.put("runtime.references.strict", true);
        }

        if ((props != null) && !props.isEmpty()) {
            properties.putAll(props);
        }

        try {
            Velocity.init(properties);
            initialized = true;
        } catch (Exception e) {
            LOGGER.error("Error initializing Velocity Engine", e);
        }
    }

    public VelocityTemplateHandler(Map props, PhrConfig config) {
        this.props = props;
        this.config = config;
        init();
    }

    @Override
    public String process(Map<String, Object> content, String templateFileName, Locale locale) throws MessageException {
        StringWriter stringWriter = null;
        try {
            init();
            VelocityContext context = new VelocityContext();

            String fullTemplateFileName = ensureExtension(templateFileName);
            String[] propertiesFileNames = new String[] { VELOCITY_FILE_EXTENSION_PATTERN.matcher(fullTemplateFileName).replaceAll(Matcher.quoteReplacement("")),
                    "templates/email/signatureHtml", "templates/email/signatureTxt" };

            if (content != null) {
                for (Object o : content.entrySet()) {
                    Map.Entry pairs = (Map.Entry) o;
                    context.put(pairs.getKey().toString(), pairs.getValue());
                }
            }
            context.put("dateTool", new DateTool());
            context.put("timeZone", TimeZone.getDefault());
            context.put("locale", Locale.US); // note: we're not using velocity's built-in i18n
            context.put("text", new MessageI18nTool(propertiesFileNames, locale));
            context.put("esc", new EscapeTool());
            context.put("vmMethods", VelocityContextMethods.class);

            Template template = Velocity.getTemplate(fullTemplateFileName);
            stringWriter = new StringWriter();
            template.merge(context, stringWriter);
            stringWriter.flush();
            stringWriter.close();
        } catch (Exception ex) {
            throw new MessageException("Unable to process message for template " + templateFileName + " in locale: " + locale.toLanguageTag(), ex);
        }
        return (stringWriter.toString());
    }

    private String ensureExtension(String templateFileName) {
        if (!templateFileName.endsWith(TEMPLATE_EXTENSION)) {
            templateFileName += TEMPLATE_EXTENSION;
        }

        return templateFileName;
    }

    @Component
    public static class VelocityContextMethods {

        private static RelaxedHtmlMessageFilter FILTER;

        @Autowired
        public VelocityContextMethods(RelaxedHtmlMessageFilter htmlMessageFilter) {
            this.FILTER = htmlMessageFilter;
        }

        public static String filterHtml(String input) {
            return FILTER.filter(input);
        }
    }
}
