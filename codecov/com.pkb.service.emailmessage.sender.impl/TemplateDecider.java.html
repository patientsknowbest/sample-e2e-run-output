package com.pkb.service.emailmessage.sender.impl;

import com.pkb.entities.enums.Message;
import com.pkb.service.emailmessage.sender.ITemplateDecider;

import java.util.Map;

import static com.google.common.base.Preconditions.checkNotNull;
import static java.lang.String.format;

public class TemplateDecider implements ITemplateDecider {
    private static final String NO_SUCH_MAPPING_DEFINED = "There is not message -> template mapping defined for channel: '%s'";

    private Map<String, Map<Message, String>> templateMap;

    @Override
    public String getTemplate(Message message, String channel) {
        Map<Message, String> actualMap = checkNotNull(templateMap.get(channel), format(NO_SUCH_MAPPING_DEFINED, channel));

        return actualMap.get(message);
    }

    /**
     * @param templateMap The templateMap to set.
     */
    public void setTemplateMap(Map<String, Map<Message, String>> templateMap) {
        this.templateMap = templateMap;
    }

}
