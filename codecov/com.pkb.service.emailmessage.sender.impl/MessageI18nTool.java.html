package com.pkb.service.emailmessage.sender.impl;

import java.util.List;
import java.util.Locale;

import org.apache.commons.lang3.ArrayUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.context.support.ResourceBundleMessageSource;

public class MessageI18nTool {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private static final String DEFAULT_STRING = "i18n_lookup_failed";
    private ResourceBundleMessageSource messageSource;
    private String bundleBaseNames = null; // for debugging
    private Locale locale;

    public MessageI18nTool(String[] resourceBundleBaseNames, Locale locale) {
        messageSource = new ResourceBundleMessageSource();
        messageSource.setBasenames(resourceBundleBaseNames);
        this.bundleBaseNames = ArrayUtils.toString(resourceBundleBaseNames);
        this.locale = locale;
    }

    public String exec(String code) {
        if ((code == null) || code.isEmpty()) {
            throw new IllegalArgumentException("Empty code");
        }

        String result = messageSource.getMessage(code, null, DEFAULT_STRING, locale);
        if (DEFAULT_STRING.equals(result)) {
            LOGGER.error("failed i18n lookup for code {}; {}, locale {}", code, bundleBaseNames, locale);
            result = code;
        }
        return result;
    }

    public String exec(String code, List arguments) {
        if ((code == null) || code.isEmpty()) {
            throw new IllegalArgumentException("Empty code");
        }

        String result = messageSource.getMessage(code, arguments.toArray(), DEFAULT_STRING, locale);
        if (DEFAULT_STRING.equals(result)) {
            LOGGER.error("failed i18n lookup for code {}; {}, locale {}", code, bundleBaseNames, locale);
            result = code + " " + formatArgs(arguments);
        }
        return result;
    }

    // just put them in a list and leave out nulls
    private static String formatArgs(List args) {
        String list = "";
        if ((args == null) || args.isEmpty()) {
            return list;
        }

        for (Object arg : args) {
            if (arg == null) {
                list += "<null>, ";
            } else {
                list += arg + ", ";
            }
        }

        if (list.endsWith(", ")) {
            list = list.substring(0, list.length() - 2); // trim off trailing comma
        }

        return list;
    }
}
