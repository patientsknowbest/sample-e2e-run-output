package com.pkb.service.emailmessage.sender.impl;

import com.pkb.common.config.PhrConfig;
import com.pkb.document.entity.Attachment;
import com.pkb.exception.MessageException;
import com.pkb.service.emailmessage.sender.IMessageDispatcher;
import com.pkb.util.PKBConstants;
import com.pkb.util.XMLUtil;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.core.io.ByteArrayResource;
import org.springframework.mail.MailException;
import org.springframework.mail.javamail.JavaMailSender;
import org.springframework.mail.javamail.MimeMessageHelper;

import javax.mail.MessagingException;
import javax.mail.internet.MimeMessage;
import java.io.UnsupportedEncodingException;
import java.net.URLDecoder;
import java.util.Collection;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

import static java.nio.charset.StandardCharsets.UTF_8;

public class MessageDispatcher implements IMessageDispatcher {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private JavaMailSender mailSender;

    private PhrConfig config;

    @Override
    public void dispatchMessage(ChannelProfile sender, ChannelProfile receiver,
                                Collection<ChannelProfile> ccTo,
                                String message,
                                Map<String, String> subjectDecorator, List<Attachment> attachments) throws MessageException {
        try {
            MimeMessage mimeMessage = buildEmailMessage(message, sender, receiver, ccTo, subjectDecorator, attachments);
            mailSender.send(mimeMessage);
        } catch (MailException exception) {
            LOGGER.error("Exception while dispatching message", exception);
            throw new MessageException("Exception while dispatching message", exception);
        }
    }

    private MimeMessage buildEmailMessage(String message, ChannelProfile sender, ChannelProfile receiver,
                                          Collection<ChannelProfile> ccTo,
                                          Map<String, String> subjectDecorator, List<Attachment> attchments)
            throws MessageException {

        Map<String, Class> aliases = new HashMap<>();
        aliases.put("message", EmailMessageDTO.class);
        EmailMessageDTO emailMessageDTO = (EmailMessageDTO) XMLUtil.unmarshall(message, aliases);
        MimeMessageHelper messageHelper = null;
        try {
            String plainTextMessage = URLDecoder.decode(emailMessageDTO.getMessageText(), UTF_8.name());
            String subject = emailMessageDTO.getMessageTitle().trim();
            if (subjectDecorator != null) {
                String prefix = subjectDecorator.get(PKBConstants.SUBJECT_PREFIX);
                if (prefix != null) {
                    subject = "For [" + prefix + "] " + subject;
                }
            }
            MimeMessage mimeMessage = mailSender.createMimeMessage();

            messageHelper = new MimeMessageHelper(mimeMessage,
                    true, UTF_8.name());
            if (config.isEmailLoggingEnabled()) {
                LOGGER.info("Subject {}", subject);
                LOGGER.info("text con {}", plainTextMessage);
            }
            messageHelper.setTo(receiver.getHandle());
            messageHelper.setFrom(sender.getHandle());
            messageHelper.setText(plainTextMessage, emailMessageDTO.getHtmlMessageText());
            messageHelper.setSubject(subject);
            String[] ccAddrs = ccTo.stream().map(ChannelProfile::getHandle).toArray(String[]::new);
            LOGGER.info("cc address count: {}", ccAddrs.length);
            messageHelper.setCc(ccAddrs);
            addAttachments(attchments, messageHelper);

        } catch (UnsupportedEncodingException | MessagingException uex) {
            LOGGER.error("Unable to build message ", uex);
            throw new MessageException("Unable to build meesage ", uex);
        }
        return messageHelper.getMimeMessage();

    }

    private void addAttachments(List<Attachment> attchments, MimeMessageHelper messageHelper) throws MessagingException {
        ByteArrayResource byteArrayResource = null;

        if ((attchments != null) && !attchments.isEmpty()) {
            for (Attachment att : attchments) {
                byteArrayResource = new ByteArrayResource(att.getContent());
                messageHelper.addAttachment(att.getFilename(), byteArrayResource, att.getMediaType());
            }
        }
    }

    public JavaMailSender getMailSender() {
        return mailSender;
    }

    public void setMailSender(JavaMailSender mailSender) {
        this.mailSender = mailSender;
    }

    public void setConfig(PhrConfig config) {
        this.config = config;
    }
}
