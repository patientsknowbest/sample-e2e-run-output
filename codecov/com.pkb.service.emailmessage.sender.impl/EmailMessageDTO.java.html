package com.pkb.service.emailmessage.sender.impl;

/**
 * This class represents Generic message with title and content
 */

public class EmailMessageDTO {

    private String htmlMessageText;

    private String messageTitle;

    private String messageText;

    public String getHtmlMessageText() {
        return htmlMessageText;
    }

    public void setHtmlMessageText(String htmlMessageText) {
        this.htmlMessageText = htmlMessageText;
    }

    public String getMessageTitle() {
        return messageTitle;
    }

    public void setMessageTitle(String messageTitle) {
        this.messageTitle = messageTitle;
    }

    public String getMessageText() {
        return messageText;
    }

    public void setMessageText(String messageText) {
        this.messageText = messageText;
    }
}
