package com.pkb.api.fhir;

import ca.uhn.fhir.context.FhirContext;
import ca.uhn.fhir.rest.api.EncodingEnum;
import ca.uhn.fhir.rest.server.IResourceProvider;
import ca.uhn.fhir.rest.server.IServerAddressStrategy;
import ca.uhn.fhir.rest.server.RestfulServer;
import ca.uhn.fhir.rest.server.interceptor.IServerInterceptor;
import com.pkb.api.fhir.config.FhirApiProperties;
import com.pkb.api.fhir.interceptors.AuthenticationExceptionHandlingInterceptor;
import com.pkb.api.fhir.interceptors.AuthenticationExtractingInterceptor;
import com.pkb.api.fhir.interceptors.ContentTypeInterceptor;
import com.pkb.api.fhir.interceptors.ExceptionLoggingInterceptor;
import com.pkb.api.fhir.interceptors.MetricsInterceptor;
import com.pkb.api.fhir.interceptors.RequestIdInterceptor;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.util.CorrelationIdUtil;
import io.vavr.collection.List;
import org.hl7.fhir.dstu3.hapi.rest.server.ServerCapabilityStatementProvider;

import javax.servlet.ServletException;
import javax.servlet.http.HttpServletRequest;
import java.util.UUID;

import static com.pkb.api.fhir.resourceprovider.CapabilityStatementProvider.experimental;
import static com.pkb.api.fhir.resourceprovider.CapabilityStatementProvider.production;
import static io.vavr.API.List;

public class FhirApiServer extends RestfulServer {
    private static final long serialVersionUID = 1L;
    static final String FHIR_API_URL_PATTERN = "/fhir/*";
    private final List<IServerInterceptor> interceptors;
    private final FhirApiProperties fhirConfig;
    private final List<IResourceProvider> resourceProviders;
    private final DateTimeService dateTimeService;
    private final IServerAddressStrategy serverAddressStrategy;
    private final AuthenticationExtractingInterceptor authenticationExtractingInterceptor;
    private final CorrelationIdUtil correlationIdUtil;


    public FhirApiServer(
            FhirContext fhirContext,
            FhirApiProperties fhirConfig,
            AuthenticationExtractingInterceptor authenticationExtractingInterceptor,
            MetricsInterceptor metricsInterceptor,
            AuthenticationExceptionHandlingInterceptor authenticationExceptionHandlingInterceptor,
            ExceptionLoggingInterceptor exceptionLoggingInterceptor,
            ContentTypeInterceptor contentTypeInterceptor,
            RequestIdInterceptor requestIdInterceptor,
            List<IResourceProvider> resourceProviders,
            DateTimeService dateTimeService,
            IServerAddressStrategy serverAddressStrategy, CorrelationIdUtil correlationIdUtil) {
        super(fhirContext);
        this.authenticationExtractingInterceptor = authenticationExtractingInterceptor;
        this.correlationIdUtil = correlationIdUtil;
        // Interceptors are explictly injected so we can control the order
        this.interceptors = List(
                authenticationExtractingInterceptor,
                metricsInterceptor,
                authenticationExceptionHandlingInterceptor,
                exceptionLoggingInterceptor,
                contentTypeInterceptor,
                requestIdInterceptor
        );
        this.fhirConfig = fhirConfig;
        this.resourceProviders = resourceProviders;
	    this.dateTimeService = dateTimeService;
        this.serverAddressStrategy = serverAddressStrategy;
    }

    @Override
    protected String getOrCreateRequestId(HttpServletRequest theRequest) {
        //want to override the default hapi behaviour so that we use UUIDs for requestIds
        UUID pkbRequestId = correlationIdUtil.getCorrelationIdFromRequest(theRequest);
        correlationIdUtil.put(pkbRequestId);
        return pkbRequestId.toString();
    }

    @Override
    protected void initialize() throws ServletException {
        super.initialize();
        setServerAddressStrategy(serverAddressStrategy);

        setDefaultResponseEncoding(EncodingEnum.JSON);

        setImplementationDescription("Patients Know Best");

        getInterceptorService().registerInterceptors(interceptors.asJava());

        setResourceProviders(resourceProviders.asJava());

        setServerConformanceProvider(fhirConfig.isExperimental()
                ? experimental(new ServerCapabilityStatementProvider(), this, dateTimeService)
                : production(new ServerCapabilityStatementProvider(), this, dateTimeService));
    }
}
