package com.pkb.api.fhir;

import com.pkb.entities.enums.NationalIdType;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;

import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;

public class FhirCodeSystemToNationalIdTypeMapper {

    public @NotNull Option<NationalIdType> toNationalIdType(@NotNull String fhirCodeSystem) {
        return Match(fhirCodeSystem).of(
                Case($("https://fhir.nhs.uk/Id/nhs-number"), Option.of(NationalIdType.NHS_NUMBER)),
                Case($("http://fhir.nl/fhir/NamingSystem/bsn"), Option.of(NationalIdType.BSN_NL)),
                Case($("http://fhir.de/NamingSystem/gkv/kvid-10"), Option.of(NationalIdType.KVN_DE)),
                Case($("urn:oid:2.16.840.1.113883.2.1.3.2.4.16.53"), Option.of(NationalIdType.CHI_NUMBER)),
                Case($("urn:uuid:1479257a-ce8f-482a-89fa-429ac066c158"), Option.of(NationalIdType.H_AND_C_NUMBER)),
                Case($(), Option.none())
        );
    }

}
