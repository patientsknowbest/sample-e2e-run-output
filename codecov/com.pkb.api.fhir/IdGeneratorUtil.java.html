package com.pkb.api.fhir;

import io.vavr.API;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import io.vavr.collection.List;

import java.util.UUID;
import java.util.function.Function;

import static io.vavr.API.Tuple;
import static java.util.UUID.nameUUIDFromBytes;

public class IdGeneratorUtil {

    /**
     * Generates unique, stable identifiers for the elements of the given list with the help of
     * {@link UUID#nameUUIDFromBytes(byte[])} based on the elements' generator function
     *
     * @param objects
     *          the list of objects with a correct hashCode implementation
     *          we want to generate unique IDs for
     * @param generator
     *          generator function that tries to supply uniqueness to the objects
     * @return the list of unique IDs in the same order as the entities in the list
     */
    public static <T, R> List<Tuple2<T, String>> generateIdsFor(List<T> objects, Function<T, R> generator) {
        var objAndUuidTuples = objects.map(object -> createObjectAndUuidTupleFrom(object, generator));
        var counts = API.<String, Integer>Map();
        var ret = API.<Tuple2<T, String>>List();
        for (var objAndUuid : objAndUuidTuples) {
            var count = counts.getOrElse(objAndUuid._2, 1);
            ret = ret.append(Tuple(objAndUuid._1, objAndUuid._2 + "-" + count));
            counts = counts.put(objAndUuid._2, count + 1);
        }
        return ret;
    }

    /**
     * Generates a unique and stable identifier for the given element based on a generator function e.g. hashCode
     * @param object
     *          object with correct hashCode implementation we intend to generate unique ID for
     * @param generator
     *          generator function that tries to supply uniqueness to the objects
     * @return unique ID for the received object
     */
    public static <T, R> String generateIdFor(T object, Function<T, R> generator) {
        return generateUUIDFor(object, generator) + "-1";
    }

    private static <T, R> Tuple2<T, String> createObjectAndUuidTupleFrom(T object, Function<T, R> generator) {
        return Tuple.of(object, generateUUIDFor(object, generator));
    }

    private static <T, R> String generateUUIDFor(T object, Function<T, R> generator) {
        return nameUUIDFromBytes(generator.apply(object).toString().getBytes()).toString();
    }
}
