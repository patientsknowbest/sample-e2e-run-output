package com.pkb.api.fhir;

import com.pkb.entities.temp.FhirPatientResource;
import com.pkb.repository.FhirPatientResourceRepository;
import org.jetbrains.annotations.NotNull;

import javax.transaction.Transactional;
import java.util.UUID;

public class FhirPatientResourceService {

    private final FhirPatientResourceRepository patientResourceRepository;

    public FhirPatientResourceService(FhirPatientResourceRepository patientResourceRepository) {
        this.patientResourceRepository = patientResourceRepository;
    }

    private void save(String resource, long personId, UUID orgPublicId) {
        FhirPatientResource fhirPatientResource = new FhirPatientResource();

        fhirPatientResource.setOrgPublicId(orgPublicId);
        fhirPatientResource.setPersonId(personId);
        fhirPatientResource.setPatientResource(resource);

        patientResourceRepository.save(fhirPatientResource);
    }

    @Transactional
    public void addOrUpdatePatientResource(@NotNull String resource, long personId, @NotNull UUID orgPublicId) {
        boolean isFound = patientResourceRepository.existsByPersonIdAndOrgPublicId(personId, orgPublicId);

        if (isFound) {
            patientResourceRepository.updatePatientResource(resource, personId, orgPublicId);
        } else {
            save(resource, personId, orgPublicId);
        }
    }

}
