package com.pkb.api.fhir;

import com.pkb.datamodel.PrivacyFlagsBase;
import com.pkb.entities.enums.PrivacyFlag;
import io.vavr.API;
import io.vavr.collection.List;
import org.hl7.fhir.dstu3.model.Coding;
import org.jetbrains.annotations.NotNull;

public class PrivacyFlagsMapper {

    public @NotNull List<Coding> toCodingList(@NotNull PrivacyFlagsBase privacyFlags) {
        List<Coding> privacyCodes = API.List();
        if (privacyFlags.isGeneralRequired()) {
            privacyCodes = privacyCodes.append(toCoding(PrivacyFlag.GENERAL));
        }
        if (privacyFlags.isMentalHealthRequired()) {
            privacyCodes = privacyCodes.append(toCoding(PrivacyFlag.MENTAL_HEALTH));
        }
        if (privacyFlags.isSexualHealthRequired()) {
            privacyCodes = privacyCodes.append(toCoding(PrivacyFlag.SEXUAL_HEALTH));
        }
        if (privacyFlags.isSocialCareRequired()) {
            privacyCodes = privacyCodes.append(toCoding(PrivacyFlag.SOCIAL_CARE));
        }
        return privacyCodes;
    }

    public @NotNull Coding toCoding(@NotNull PrivacyFlag pf) {
        Coding coding = new Coding();
        coding.setCode(toCode(pf));
        coding.setSystem(CanonicalCodeSystemURLs.privacyLabel());
        return coding;
    }

    private @NotNull String toCode(@NotNull PrivacyFlag privacyFlag) {
        switch (privacyFlag) {
        case GENERAL:
            return "GENERAL_HEALTH";
        default:
            return privacyFlag.name();
        }
    }

}
