package com.pkb.api.fhir;

import static org.apache.http.HttpStatus.SC_OK;

import java.util.UUID;

import javax.servlet.http.HttpServletResponse;

import org.hl7.fhir.dstu3.model.Patient;
import org.hl7.fhir.instance.model.api.IBaseResource;

import ca.uhn.fhir.rest.annotation.Operation;
import ca.uhn.fhir.rest.annotation.ResourceParam;
import ca.uhn.fhir.rest.api.server.RequestDetails;
import ca.uhn.fhir.rest.server.IResourceProvider;
import io.prometheus.client.Counter;

public class PhrPatientResourceProvider implements IResourceProvider {

    private static final Counter FHIR_DEMOGRAPHICS_API_USAGE = Counter.build()
            .name("pkb_phr_fhir_demographics_api_usage")
            .labelNames("requests") // total, success
            .help("Number of FHIR messages adding/updating patient demographics")
            .register();

    private final FhirPatientCreationService fhirPatientCreationService;
    private final PhrFhirAuthenticationService phrFhirAuthenticationService;

    public PhrPatientResourceProvider(FhirPatientCreationService fhirPatientCreationService,
                                      PhrFhirAuthenticationService phrFhirAuthenticationService) {
        this.fhirPatientCreationService = fhirPatientCreationService;
        this.phrFhirAuthenticationService = phrFhirAuthenticationService;
    }

    @Override
    public Class<? extends IBaseResource> getResourceType() {
        return Patient.class;
    }

    @SuppressWarnings("unused")
    @Operation(name = "$process-message", idempotent = true, manualResponse = true)
    public void processMessage(@ResourceParam Patient patient, RequestDetails request, HttpServletResponse httpServletResponse) {
        FHIR_DEMOGRAPHICS_API_USAGE.labels("total").inc();

        // We don't actually utilise the result of the authentication, i.e. the account keys, as we do not conduct subsequent authz.
        // An authentication failure will throw an appropriate exception and OperationOutcome out of this method for a response.
        phrFhirAuthenticationService.authenticate(request);

        fhirPatientCreationService.processPatient(patient, UUID.fromString(request.getHeader("X-Org-Public-Id")), UUID.randomUUID())
                .mapLeft(serverResponseException -> {
                    // Throw the processing error to return a response to the client
                    throw serverResponseException;
                });

        FHIR_DEMOGRAPHICS_API_USAGE.labels("success").inc();

        // 200 response for success
        httpServletResponse.setContentType("text/plain");
        httpServletResponse.setStatus(SC_OK);
    }
}
