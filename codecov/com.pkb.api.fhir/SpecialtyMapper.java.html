package com.pkb.api.fhir;

import com.pkb.datamodel.ClinicalCoding;
import org.hl7.fhir.dstu3.model.Coding;
import org.hl7.fhir.dstu3.model.Extension;
import org.jetbrains.annotations.NotNull;

import static com.pkb.api.fhir.CanonicalExtensionURLs.specialty;

public class SpecialtyMapper {

    public @NotNull Extension toExtension(@NotNull ClinicalCoding specialty) {

        Coding coding = new Coding();
        coding.setCode(specialty.getCode());
        specialty.getDisplayText().peek(d -> coding.setDisplay(d.value()));

        Extension extension = new Extension()
                .setUrl(specialty())
                .setValue(coding);

        return extension;
    }
}
