package com.pkb.api.fhir;

import ca.uhn.fhir.rest.api.server.RequestDetails;
import ca.uhn.fhir.rest.server.exceptions.InvalidRequestException;
import com.pkb.oauth2.client.Oauth2Client;
import com.pkb.oauth2.client.model.ThirdPartyClientJwtVerificationRequest;
import org.hl7.fhir.dstu3.model.CodeableConcept;
import org.hl7.fhir.dstu3.model.Coding;
import org.hl7.fhir.dstu3.model.OperationOutcome;

import java.util.Optional;
import java.util.UUID;

import static com.pkb.api.fhir.PhrAuthenticationExtractingInterceptor.USER_DATA_AUTHN_COOKIE;
import static com.pkb.api.fhir.PhrAuthenticationExtractingInterceptor.USER_DATA_AUTHN_HEADER;
import static com.pkb.oauth2.client.model.ImmutableThirdPartyClientJwtVerificationRequest.thirdPartyClientJwtVerificationRequest;
import static io.vavr.API.Try;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity.ERROR;

/**
 * PURPOSE
 * A trimmed down version of the authentication flow we have in our separate fhir-api service, purely for authenticating
 * the demographics API represented by PhrFhirApiServer.
 *
 * This flow mandates client credentials and therefore requires:
 * 1. An x-org-public-id in the header of the message
 * 2. A bearer access token to be provided for authentication
 *
 * Other mechanisms of authentication are not supported.
 *
 * DISCLAIMER
 * A good bit of this auth layer is copied from the fhir-api code, which doesn't make me super happy.
 * There was a broad understanding with this restructured approach discussed by Martin, Steven and myself that
 * there would by necessity be a duplicate of this auth layer as part of the proposal. I've kept it as minimal as possible.
 *
 * This entire flow is temporary until FHIRWall is up (~October). At that point the FHIR endpoint will be that of the
 * FHIR server backed by the FHIR data store, the auth will be in the FHIRWall layer, and most of this frontend facing
 * code can be deleted. All that will remain will be the mapping to the HL7Patient DTO, and that will be interacted
 * with probably via pub-sub. So it's not long term. We have contractural reasons why we need to deliver the demographics
 * component of this API earlier than the FHIR data store/FHIRWall layer, otherwise a good chunk of this would not have be done.
 */
public class PhrFhirAuthenticationService {

    private static final String ORG_PUBLIC_ID_HEADER_NAME = "X-Org-Public-Id";
    private static final String AUTHENTICATION_SCHEMA = "bearer";

    private final Oauth2Client oauth2Client;

    public PhrFhirAuthenticationService(Oauth2Client oauth2Client) {
        this.oauth2Client = oauth2Client;
    }

    public void authenticate(RequestDetails request) {
        findOrgPublicIdHeader(request).ifPresentOrElse(orgId -> loginWithJwt(extractAccessToken(request), orgId), () -> {
                    throw new InvalidRequestException("x-org-public-id not present", getBadAuthenticationOperationOutcome());
                });
    }

    private Optional<UUID> findOrgPublicIdHeader(RequestDetails request) {
        return Try(() -> request.getHeader(ORG_PUBLIC_ID_HEADER_NAME))
                .map(UUID::fromString)
                .toJavaOptional();
    }

    private void loginWithJwt(String clientJwt, UUID orgPublicId) {
        ThirdPartyClientJwtVerificationRequest thirdPartyClientJwtVerificationRequest = thirdPartyClientJwtVerificationRequest()
                .orgPublicId(orgPublicId)
                .clientJWT(clientJwt)
                .build();
        oauth2Client.verifyThirdPartyClientJwt(thirdPartyClientJwtVerificationRequest)
                .mapLeft(error -> {
                    throw new InvalidRequestException(error.getDescription(), getBadAuthenticationOperationOutcome());
                });
    }

    private String extractAccessToken(RequestDetails request) {
        String header = (String)request.getUserData().get(USER_DATA_AUTHN_HEADER);
        String tokenFromHeader = null;
        if (header != null) {
            if (header.toLowerCase().startsWith(AUTHENTICATION_SCHEMA)) {
                tokenFromHeader = header.substring(AUTHENTICATION_SCHEMA.length()).trim();
            } else {
                throw new InvalidRequestException("invalid authorization header", getBadAuthenticationOperationOutcome());
            }
        }
        String tokenFromCookie = (String) request.getUserData().get(USER_DATA_AUTHN_COOKIE);
        if (tokenFromHeader != null && tokenFromCookie != null) {
            throw new InvalidRequestException("the Authorization header xor the ui-exchange cookie should be passed", getBadAuthenticationOperationOutcome());
        }
        else if (tokenFromHeader == null && tokenFromCookie == null) {
            throw new InvalidRequestException("no authentication provided", getBadAuthenticationOperationOutcome());
        }
        return tokenFromCookie != null ? tokenFromCookie : tokenFromHeader;
    }

    private OperationOutcome getBadAuthenticationOperationOutcome() {
        OperationOutcome operationOutcome = new OperationOutcome();
        operationOutcome.setId(UUID.randomUUID().toString());
        operationOutcome.addIssue()
                .setSeverity(ERROR)
                .setDiagnostics("You did not provide a valid access token.")
                .setCode(OperationOutcome.IssueType.SECURITY)
                .setDetails(new CodeableConcept()
                        .addCoding(new Coding()
                                .setCode("BAD_AUTHENTICATION")
                                .setSystem("http://fhir.patientsknowbest.com/codesystem/response-code")
                                .setDisplay("Valid access token not provided")));
        return operationOutcome;
    }
}
