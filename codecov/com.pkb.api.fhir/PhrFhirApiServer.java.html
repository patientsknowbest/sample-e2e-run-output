package com.pkb.api.fhir;

import ca.uhn.fhir.context.FhirContext;
import ca.uhn.fhir.rest.api.EncodingEnum;
import ca.uhn.fhir.rest.server.IServerAddressStrategy;
import ca.uhn.fhir.rest.server.RestfulServer;
import com.pkb.util.PKBApplicationContext;

import javax.servlet.ServletException;
import java.util.List;

/**
 * A FHIR server endpoint embedded within the PHR application.
 * A transitory step allowing for patient creation via FHIR using the existing HL7 stack.
 * Deliberately designed to not be easily expandable, this is slated for destruction as we move forward with FHIR data store/FHIRWall
 */
public class PhrFhirApiServer extends RestfulServer {

    private final PhrPatientResourceProvider phrPatientResourceProvider;
    private final IServerAddressStrategy serverAddressStrategy;
    private final PhrAuthenticationExtractingInterceptor phrAuthenticationExtractingInterceptor;

    public PhrFhirApiServer() {
        //Servlets in wf can't utilise constructor injection, WELD falls over. Probably because we're using CDI and spring
        // DI at the same time, which is the root cause of many of our problems.
        super(PKBApplicationContext.getBean(FhirContext.class));
        phrPatientResourceProvider = PKBApplicationContext.getBean(PhrPatientResourceProvider.class);
        serverAddressStrategy = PKBApplicationContext.getBean(IServerAddressStrategy.class);
        phrAuthenticationExtractingInterceptor = PKBApplicationContext.getBean(PhrAuthenticationExtractingInterceptor.class);
    }

    @Override
    protected void initialize() throws ServletException {
        super.initialize();
        setServerAddressStrategy(serverAddressStrategy);
        setDefaultResponseEncoding(EncodingEnum.JSON);
        setImplementationDescription("Patients Know Best");

        getInterceptorService().registerInterceptors(List.of(phrAuthenticationExtractingInterceptor));

        setResourceProviders(List.of(phrPatientResourceProvider));
    }
}
