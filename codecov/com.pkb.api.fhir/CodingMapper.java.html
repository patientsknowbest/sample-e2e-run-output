package com.pkb.api.fhir;

import com.pkb.datamodel.ClinicalCoding;
import com.pkb.datamodel.confidential.SensitiveString;
import org.hl7.fhir.dstu3.model.Coding;
import org.mapstruct.AfterMapping;
import org.mapstruct.Mapper;
import org.mapstruct.MappingTarget;
import org.mapstruct.ReportingPolicy;

@Mapper(unmappedTargetPolicy = ReportingPolicy.IGNORE)
public abstract class CodingMapper {

    public abstract Coding mapCoding(ClinicalCoding clinicalCoding);

    public abstract Coding mapCoding(String code, String system);

    @AfterMapping
    protected void mapCodeSystem(ClinicalCoding clinicalCoding, @MappingTarget Coding result) {
        clinicalCoding.getCodeSystem().peek(result::setSystem);
    }

    @AfterMapping
    protected void mapDisplayText(ClinicalCoding clinicalCoding, @MappingTarget Coding result) {
        clinicalCoding.getDisplayText()
                .map(SensitiveString::value)
                .peek(result::setDisplay);
    }
}
