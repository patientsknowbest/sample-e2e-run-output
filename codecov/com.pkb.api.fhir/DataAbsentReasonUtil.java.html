package com.pkb.api.fhir;

import org.hl7.fhir.dstu3.model.CodeType;
import org.hl7.fhir.dstu3.model.Extension;
import org.hl7.fhir.dstu3.model.InstantType;

public class DataAbsentReasonUtil {

    /**
     * Returns an Extension which can be added to an otherwise mandatory element when the data is unknown.
     * 
     * See: http://hl7.org/fhir/STU3/extensibility.html#2.5.0.2.2
     * 
     * @return An Extension to be used when otherwise mandatory data is absent because it is unknown.
     */
    public static Extension getDataAbsentExtension() {
        Extension ext = new Extension();
        ext.setUrl("http://hl7.org/fhir/StructureDefinition/data-absent-reason");
        CodeType ct = new CodeType();
        ct.setValue("unknown");
        ext.setValue(ct);
        return ext;
    }

    public static InstantType missingInstantType() {
        InstantType it = new InstantType();
        it.addExtension(DataAbsentReasonUtil.getDataAbsentExtension());
        return it;
    }
}
