package com.pkb.api.fhir;

import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import ca.uhn.fhir.rest.server.exceptions.InternalErrorException;
import com.pkb.api.fhir.exception.ServerResponseExceptionMapper;
import com.pkb.authz.NotAuthorizedException;
import com.pkb.authz.NotFoundException;
import com.pkb.domain.SystemUserTypeNotPermittedException;
import com.pkb.domain.criteria.CriteriaResult;
import io.vavr.Function2;
import io.vavr.collection.List;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.apache.commons.lang3.exception.ExceptionUtils;
import org.hl7.fhir.dstu3.model.BaseResource;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.slf4j.Logger;

import java.util.function.Consumer;
import java.util.function.Function;
import java.util.function.Supplier;

import static com.google.common.base.Predicates.instanceOf;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.List;
import static io.vavr.API.Match;
import static io.vavr.API.run;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;
import static org.slf4j.LoggerFactory.getLogger;

public abstract class AbstractResourceService<R extends BaseResource> {
    protected final ServerResponseExceptionMapper exceptionMapper;

    protected AbstractResourceService(@NotNull ServerResponseExceptionMapper exceptionMapper) {
        this.exceptionMapper = exceptionMapper;
    }

    protected <T> Function<Throwable, Either<BaseServerResponseException, T>> minimalFlow(@NotNull Class<R> resourceType) {
        return cause -> left(exceptionMapper.minimalFlow(cause, resourceType));
    }

    protected Consumer<RuntimeException> logAuthorizationError(Class<R> resource) {
        return cause -> {
            Logger logger = getLogger(resource);
            Match(cause).of(
                    Case($(instanceOf(NotFoundException.class)), Function2.of(this::logAuthorizationFailureMessage).apply(logger)),
                    Case($(instanceOf(NotAuthorizedException.class)), Function2.of(this::logAuthorizationFailureMessage).apply(logger)),
                    Case($(), e -> run(() -> logger.error("Unexpected authorization failure", e))));
        };
    }

    protected Function<Throwable, Either<BaseServerResponseException, Option<R>>> recoverWhenResourceSearched() {
        return cause -> Match(cause).of(
                Case($(instanceOf(VirtualMachineError.class)), ExceptionUtils::rethrow),
                Case($(instanceOf(NotFoundException.class)), right(Option.none())),
                Case($(instanceOf(NotAuthorizedException.class)), right(Option.none())),
                Case($(instanceOf(Throwable.class)), this::internalError));
    }

    protected BaseServerResponseException exceptionToFhir(Throwable throwable, Class<R> resourceType) {
        return Match(throwable).of(
                Case($(instanceOf(SystemUserTypeNotPermittedException.class)), nae -> toSystemUserTypeNotPermitted()),
                Case($(instanceOf(NotFoundException.class)), exceptionMapper.toNotFound(resourceType)),
                Case($(instanceOf(NotAuthorizedException.class)), nae -> exceptionMapper.notAuthorized(NotAuthorizedException.class.cast(nae))),
                Case($(), ex -> new InternalErrorException(throwable)));
    }

    protected Function<Throwable, Either<BaseServerResponseException, List<R>>> recoverWhenResourcesSearched() {
        return cause -> Match(cause).of(
                Case($(instanceOf(VirtualMachineError.class)), ExceptionUtils::rethrow),
                Case($(instanceOf(NotFoundException.class)), right(List())),
                Case($(instanceOf(NotAuthorizedException.class)), right(List())),
                Case($(instanceOf(Throwable.class)), this::internalError));
    }

    protected Function<Throwable, Either<BaseServerResponseException, CriteriaResult<R>>> recoverWhenPaginatedResourcesSearched() {
        return recoverWhenPaginatedResourcesSearched(CriteriaResult::emptyCriteriaResult);
    }

    protected <C> Function<Throwable, Either<BaseServerResponseException, C>> recoverWhenPaginatedResourcesSearched(Supplier<C> emptyCriteria) {
        return cause -> Match(cause).of(
                Case($(instanceOf(VirtualMachineError.class)), ExceptionUtils::rethrow),
                Case($(instanceOf(SystemUserTypeNotPermittedException.class)), ExceptionUtils::rethrow),
                Case($(instanceOf(NotFoundException.class)), right(emptyCriteria.get())),
                Case($(instanceOf(NotAuthorizedException.class)), c -> right(emptyCriteria.get())),
                Case($(instanceOf(Throwable.class)), this::internalError));
    }

    @Nullable
    private Void logAuthorizationFailureMessage(Logger logger, RuntimeException e) {
        run(() -> logger.info("Authorization failure: {}", e.getMessage()));
        return null;
    }

    protected BaseServerResponseException toSystemUserTypeNotPermitted() {
        return exceptionMapper.systemUserTypeNotPermitted();
    }

    private <E> Either<BaseServerResponseException, E> internalError(Throwable reason) {
        return Either.left(new InternalErrorException(reason));
    }
}
