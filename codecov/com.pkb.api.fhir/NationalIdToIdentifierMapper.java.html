package com.pkb.api.fhir;

import com.pkb.datamodel.NationalId;
import com.pkb.entities.enums.ValidNationalId;
import org.hl7.fhir.dstu3.model.Identifier;
import org.jetbrains.annotations.NotNull;

public class NationalIdToIdentifierMapper {

    public static Identifier toIdentifier(@NotNull ValidNationalId validNationalId) {
        Identifier identifier = new Identifier();
        identifier.setSystem(validNationalId.type().getFhirIdentifierSystem());
        identifier.setValue(validNationalId.value());
        return identifier;
    }

    public static Identifier toIdentifier(@NotNull NationalId nationalId) {
        Identifier identifier = new Identifier();
        identifier.setSystem(nationalId.getType().getFhirIdentifierSystem());
        nationalId.getValue().peek(identifier::setValue);
        return identifier;
    }


}
