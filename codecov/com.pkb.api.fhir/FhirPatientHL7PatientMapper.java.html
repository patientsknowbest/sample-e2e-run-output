package com.pkb.api.fhir;

import static com.pkb.entities.enums.NationalIdType.NHS_NUMBER;

import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;
import java.util.Date;
import java.util.List;
import java.util.Optional;
import java.util.TimeZone;
import java.util.stream.Collectors;

import org.hl7.fhir.dstu3.model.Address;
import org.hl7.fhir.dstu3.model.BooleanType;
import org.hl7.fhir.dstu3.model.CodeableConcept;
import org.hl7.fhir.dstu3.model.Coding;
import org.hl7.fhir.dstu3.model.ContactPoint;
import org.hl7.fhir.dstu3.model.HumanName;
import org.hl7.fhir.dstu3.model.Identifier;
import org.hl7.fhir.dstu3.model.Patient;
import org.hl7.fhir.dstu3.model.Period;
import org.hl7.fhir.dstu3.model.PrimitiveType;
import org.hl7.fhir.dstu3.model.StringType;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.jooq.tools.StringUtils;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.ImmutableDisclosingEmail;
import com.pkb.exception.ApiCallMalformedException;
import com.pkb.service.dataupload.hl7.HL7ConnContext;
import com.pkb.service.dataupload.hl7.field.HL7Code;
import com.pkb.service.dataupload.hl7.segment.HL7Patient;
import com.pkb.service.dataupload.hl7.value.HL7Date;

import ca.uhn.fhir.util.ElementUtil;

/**
 * Does a unidirectional mapping from a STU3 (with some CareConnect extensions) Patient resource to HL7Patient
 * DTO so we can hook into the HL7 registration and patient creation flows from the FHIR API.
 *
 * This is initially a limited subset of the fields possible within FHIR.
 */
public class FhirPatientHL7PatientMapper {

    private final DateTimeService dateTimeService;

    public FhirPatientHL7PatientMapper(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    @NotNull
    HL7Patient map(@NotNull Patient fhirPatient, @NotNull HL7ConnContext hl7ConnContext) {
        //We have validated and enforced only one name element in the resource
        HumanName name = fhirPatient.getNameFirstRep();

        HL7Patient hl7 = new HL7Patient(null);

        //We dispose of anything apart from the first title
        hl7.setTitle(name.getPrefix().stream().findFirst().map(PrimitiveType::getValue).orElse(""));
        hl7.setFirstName(getValueOrThrow(name.getGiven().stream().map(StringType::getValueNotNull)
                .collect(Collectors.joining(" ")).trim(), "Given name"));
        hl7.setLastName(getValueOrThrow(name.getFamily(), "Family name"));

        hl7.setGender(fhirPatient.getGender() == null ? "" : fhirPatient.getGender().toCode());
        setDob(fhirPatient, hl7);
        setDeathIndicators(fhirPatient, hl7);

        processNationalId(fhirPatient, hl7ConnContext, hl7);

        processAddresses(fhirPatient, hl7);
        processEmails(fhirPatient, hl7);
        processPhones(fhirPatient, hl7);
        processLanguage(fhirPatient, hl7);

        return hl7;
    }

    private void processNationalId(@NotNull Patient fhirPatient, @NotNull HL7ConnContext hl7ConnContext, @NotNull HL7Patient hl7) {
        Identifier nhsIdentifier = fhirPatient.getIdentifier().stream()
                .filter(identifier -> identifier.getSystem().equals(NHS_NUMBER.getFhirIdentifierSystem()))
                .findFirst()
                .orElseThrow(() -> new RuntimeException("No NHS number identifier is present"));

        hl7.createAndAddNationalId(hl7ConnContext, dateTimeService, nhsIdentifier.getValue(), "", NHS_NUMBER);
    }

    private void processLanguage(@NotNull Patient fhirPatient, @NotNull HL7Patient hl7) {
        String languageId = fhirPatient.getExtension().stream()
                .filter(extension -> extension.getUrl() != null
                        && extension.getUrl().equals("https://fhir.hl7.org.uk/STU3/StructureDefinition/Extension-CareConnect-NHSCommunication-1")
                        && extension.hasValue())
                .map(extension -> ((CodeableConcept) extension.getValue()).getCoding().stream()
                        .filter(coding -> coding.hasSystem() && coding.getSystem().equals("https://fhir.hl7.org.uk/STU3/CodeSystem/CareConnect-HumanLanguage-1"))
                        .findFirst()
                        .map(Coding::getCode).orElse(""))
                .findFirst().orElse("");

        //If the language code is empty, this will be deleted
        HL7Code languageCode = new HL7Code();
        languageCode.setId(languageId);
        hl7.setLanguageCode(languageCode);
    }

    private void processAddresses(@NotNull Patient fhirPatient, @NotNull HL7Patient hl7) {
        List<Address> addresses = fhirPatient.getAddress()
                .stream()
                .filter(address -> {
                    Period period = address.getPeriod();
                    return period.getEnd() == null || !period.hasEnd() || period.getEnd().after(Date.from(dateTimeService.now()));
                })
                .collect(Collectors.toList());

        //Addresses: we can only store one. We prioritise home > work > temp seems reasonable, then fall back on whatever.
        // If there are multiple unended for a given type, just get the first in the array, we can't do any better.
        Optional<Address> addressToUse = addresses.stream().filter(address -> address.getUse() == Address.AddressUse.HOME && addressHasData(address)).findFirst()
                .or(() -> addresses.stream().filter(address -> address.getUse() == Address.AddressUse.WORK && addressHasData(address)).findFirst())
                .or(() -> addresses.stream().filter(address -> address.getUse() == Address.AddressUse.TEMP && addressHasData(address)).findFirst())
                .or(() -> addresses.stream().filter(this::addressHasData).findFirst()); //If we don't have any of these, just use whatever they've provided as a fallback
        addressToUse.ifPresent(address -> {
            hl7.setAddress1(getAddressLine(address, 0));
            hl7.setAddress2(getAddressLine(address, 1));

            hl7.setCity(getValueOrEmpty(address.getCity()));
            hl7.setState(getValueOrEmpty(address.getState()));
            hl7.setZip(getValueOrEmpty(address.getPostalCode()));
            hl7.setCountry3char(getValueOrEmpty(address.getCountry()));
        });
    }

    private boolean addressHasData(Address address) {
        return !ElementUtil.isEmpty(address.getLine(), address.getCity(), address.getState(), address.getPostalCode(), address.getCountry());
    }

    private void processPhones(@NotNull Patient fhirPatient, @NotNull HL7Patient hl7) {
        Optional<ContactPoint> homePhone = getContactPointForSystemAndUse(fhirPatient,
                ContactPoint.ContactPointSystem.PHONE, ContactPoint.ContactPointUse.HOME);
        homePhone.ifPresentOrElse(phone -> hl7.setHomePhone(phone.getValue()),
                () -> hl7.setHomePhone(""));

        Optional<ContactPoint> mobilePhone = getContactPointForSystemAndUse(fhirPatient,
                ContactPoint.ContactPointSystem.PHONE, ContactPoint.ContactPointUse.MOBILE);
        mobilePhone.ifPresentOrElse(phone -> hl7.setMobilePhone(phone.getValue()),
                () -> hl7.setMobilePhone(""));

        Optional<ContactPoint> workPhone = getContactPointForSystemAndUse(fhirPatient,
                ContactPoint.ContactPointSystem.PHONE, ContactPoint.ContactPointUse.WORK);
        workPhone.ifPresentOrElse(phone -> hl7.setWorkPhone(phone.getValue()),
                () -> hl7.setWorkPhone(""));

        //If we don't have any of these, just use whatever they've provided as the home as a fallback
        if (homePhone.isEmpty() && mobilePhone.isEmpty() && workPhone.isEmpty()) {
            Optional<ContactPoint> anyPhone = getContactPointForSystemAndUse(fhirPatient, ContactPoint.ContactPointSystem.PHONE, null);
            anyPhone.ifPresent(phone -> hl7.setHomePhone(phone.getValue()));
        }
    }

    private void processEmails(@NotNull Patient fhirPatient, @NotNull HL7Patient hl7) {
        Optional<String> homeEmailValue = getContactPointForSystemAndUse(fhirPatient,
                ContactPoint.ContactPointSystem.EMAIL, ContactPoint.ContactPointUse.HOME)
                        .map(ContactPoint::getValue);

        homeEmailValue.ifPresent(emailValue -> hl7.setHomeEmail(ImmutableDisclosingEmail.disclosingEmail(emailValue)));

        Optional<String> workEmailValue = getContactPointForSystemAndUse(fhirPatient,
                ContactPoint.ContactPointSystem.EMAIL, ContactPoint.ContactPointUse.WORK)
                        .map(ContactPoint::getValue);

        workEmailValue.ifPresent(emailValue -> hl7.setWorkEmail(ImmutableDisclosingEmail.disclosingEmail(emailValue)));

        //If we don't have an email of either of these types, just use whatever they've provided as the home as a fallback
        if (homeEmailValue.isEmpty() && workEmailValue.isEmpty()) {
            Optional<String> anyEmailValue = getContactPointForSystemAndUse(fhirPatient, ContactPoint.ContactPointSystem.EMAIL, null)
                    .map(ContactPoint::getValue);
            anyEmailValue.ifPresent(emailValue -> hl7.setHomeEmail(ImmutableDisclosingEmail.disclosingEmail(emailValue)));
        }
    }

    private void setDob(@NotNull Patient fhirPatient, @NotNull HL7Patient hl7) {
        if (fhirPatient.getBirthDate() != null) {
            ZoneId zoneId = ZoneId.of(TimeZone.getDefault().getID());
            //The HAPI Fhir server uses the timezone supplied with the date if there is one, otherwise it defaults
            //to the system timezone to create a Date object. You can't send timezones with birthdays, so we know 
            //it will have used the system timezone
            var formatter = DateTimeFormatter.BASIC_ISO_DATE; //eg 20090714
            var birthdateAsZonedDateTime = ZonedDateTime.ofInstant(fhirPatient.getBirthDate().toInstant(), zoneId);

            hl7.setDob(new HL7Date(birthdateAsZonedDateTime.toLocalDate().format(formatter), birthdateAsZonedDateTime));
        } else {
            throw new ApiCallMalformedException("Birth date cannot be null");
        }
    }

    @NotNull
    private Optional<ContactPoint> getContactPointForSystemAndUse(@NotNull Patient fhirPatient, @NotNull ContactPoint.ContactPointSystem system,
            @Nullable ContactPoint.ContactPointUse use) {
        return fhirPatient.getTelecom()
                .stream()
                .filter(telecom -> telecom.getSystem() == system
                        && telecom.getUse() == use)
                .filter(telecom -> {
                    Period period = telecom.getPeriod();
                    return period.getEnd() == null || !period.hasEnd() || period.getEnd().after(Date.from(dateTimeService.now()));
                })
                .filter(telecom -> !StringUtils.isBlank(telecom.getValue()))
                .findFirst(); //Multiple unended for a given type? Just get the first in the array.
    }

    private void setDeathIndicators(@NotNull Patient fhirPatient, @NotNull HL7Patient hl7) {
        //The deceased field can be either a DateTimeType, in which case it has a death timestamp, or BooleanType
        // If unset, it is null, and neither.
        if (fhirPatient.hasDeceasedDateTimeType()) {
            var suppliedTimezone = fhirPatient.getDeceasedDateTimeType().getTimeZone();
            //If no timezone has been suppplied, use the configured one
            //HAPI Fhir server tries to convert the date supplied into a java Date object. It uses the time zone supplied,
            //and if no time zone was supplied then it defaults to the system timezone.
            ZoneId zoneId = suppliedTimezone != null ? suppliedTimezone.toZoneId() : ZoneId.of(TimeZone.getDefault().getID());
            var deceasedDate = fhirPatient.getDeceasedDateTimeType().getValue();
            var deceasedZonedDateTime = ZonedDateTime.ofInstant(deceasedDate.toInstant(), zoneId);
            DateTimeFormatter hl7ValueFormatter = DateTimeFormatter.ofPattern("yyyyMMddHHmmssZ").withZone(zoneId);
            hl7.setDeathTimestamp(new HL7Date(
                    deceasedZonedDateTime.format(hl7ValueFormatter),
                    deceasedZonedDateTime));
        } else if (fhirPatient.hasDeceasedBooleanType()) {
            BooleanType deceased = fhirPatient.getDeceasedBooleanType();
            String deathIndicator = deceased != null && deceased.getValue() != null && deceased.getValue() ? "Y" : "N";
            hl7.setDeathIndicatorYN(deathIndicator);
        }
        //The absence of death does not resurrect someone who has previously been marked as dead. Only an affirmative
        // provision of the death flag in a false state is sufficient to reverse this, consistent with HL7 behaviour.
    }

    @NotNull
    private String getAddressLine(@NotNull Address address, int index) {
        List<StringType> lines = address.getLine();
        return lines.size() > index ? lines.get(index).toString() : "";
    }

    @NotNull
    private String getValueOrEmpty(@Nullable String value) {
        return value != null ? value : "";
    }

    @NotNull
    private String getValueOrThrow(@Nullable String value, @NotNull String valueDescription) {
        if (StringUtils.isBlank(value)) {
            throw new ApiCallMalformedException(valueDescription + " cannot be null or whitespace");
        } else {
            return value;
        }
    }
}
