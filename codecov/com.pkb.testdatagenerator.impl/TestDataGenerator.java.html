package com.pkb.testdatagenerator.impl;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.entities.enums.Route;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.test.LoincManager;
import com.pkb.test.entity.LoincMapping;
import com.pkb.test.entity.LoincTest;
import com.pkb.test.entity.MeasurementDTO;
import com.pkb.test.entity.PredefinedMeasurementType;
import com.pkb.test.entity.TestResultDTO;
import com.pkb.test.entity.TestResultType;
import com.pkb.testdatagenerator.ITestDataGenerator;
import com.pkb.util.tolven.TolvenBeanFactory;
import org.jetbrains.annotations.NotNull;

import java.time.LocalDate;
import java.time.ZoneOffset;
import java.time.ZonedDateTime;
import java.util.ArrayList;
import java.util.List;
import java.util.Random;

public class TestDataGenerator implements ITestDataGenerator {

    TolvenBeanFactory tolvenBeanFactory;
    DateTimeService dateTimeService;
    TeamUserManager teamUserManager;
    LoincManager loincManager;

    public TestDataGenerator(TolvenBeanFactory tolvenBeanFactory, DateTimeService dateTimeService, TeamUserManager teamUserManager, LoincManager loincManager) {
        this.tolvenBeanFactory = tolvenBeanFactory;
        this.dateTimeService = dateTimeService;
        this.teamUserManager = teamUserManager;
        this.loincManager = loincManager;
    }

    @Override
    public List<MeasurementDTO> generateMeasurements(@NotNull LoggedInEHRRequestContext requestContext, int numResults) {
        Random randomGenerator = new Random();

        List<MeasurementDTO> results = new ArrayList<>();

        for (int t = 0; t < numResults; ++t) {
            MeasurementDTO dto = new MeasurementDTO(new SourceDetails(requestContext));
            dto.setMeasureDate(LocalDate.of(2005, 1, 1).plusDays(randomGenerator.nextInt(3650)).atStartOfDay().toInstant(
                    ZoneOffset.UTC));
            dto.setMeasurementCode(PredefinedMeasurementType.BLOOD_PRESSURE.code().getOrNull());
            dto.setMeasurementType(PredefinedMeasurementType.BLOOD_PRESSURE);
            dto.setValue(80.0 * randomGenerator.nextDouble() + 80.0);
            dto.setValue2(80.0 * randomGenerator.nextDouble());
            results.add(dto);
        }
        return results;
    }

    /**
     * Generate random results (quality to improve)
     *
     * @param patientId
     * @param numResults
     */
    @Override
    public List<TestResultDTO> generateLoincTestData(@NotNull LoggedInEHRRequestContext requestContext, long patientId, int numResults) {
        Random randomGenerator = new Random();
        List<LoincTest> allLoincTests = loincManager.getAllLoincTests();
        int nlt = allLoincTests.size();
        ZonedDateTime zdt = dateTimeService.nowZonedDateTime();
        List<TestResultDTO> results = new ArrayList<>();
        for (int t = 0; t < numResults; ++t) {
            LoincTest loinc = allLoincTests.get(randomGenerator.nextInt(nlt));
            TestResultDTO dto = new TestResultDTO(new SourceDetails(requestContext));
            dto.setLoincTestId(loinc.getId());
            setRandomTestData(dto, requestContext.getAccessingUserId(), "Random LOINC test generator", randomGenerator, zdt);
            results.add(dto);
        }
        return results;
    }

    /**
     * Generate random results (quality to improve)
     *
     * @param patientId
     * @param numResults
     */
    @Override
    public List<TestResultDTO> generateMappingTestData(@NotNull LoggedInEHRRequestContext requestContext, long patientId, int numResults) {
        Team team = teamUserManager.getUserTeams(requestContext, patientId, true/*activeOnly*/).get(0);
        Org org = team.getOrg();

        Random randomGenerator = new Random();

        ZonedDateTime zdt = dateTimeService.nowZonedDateTime();
        List<TestResultDTO> results = new ArrayList<>();

        for (int t = 0; t < numResults; ++t) {
            TestResultDTO dto = new TestResultDTO(new SourceDetails(requestContext));

            String codeSystem = "TestDataGen";
            String unit = "g/L";
            String randCode = "Random-" + randomGenerator.nextInt(50); // limit range

            TestResultType type = loincManager.getTestResultType(org.getId(), randCode, codeSystem, unit);

            if (type == null) {
                LoincMapping lm = new LoincMapping();
                lm.setOrg(org);
                lm.setTeam(team);
                lm.setTestCodingSystem(codeSystem);
                lm.setTestCode(randCode);
                lm.setName(randCode);
                lm.setUnit(unit);

                Long newMappingId = loincManager.addLoincMapping(lm);
                dto.setLoincMappingId(newMappingId);
            } else {
                LoincMapping lm = (LoincMapping) type;
                dto.setLoincMappingId(lm.getId());

                // not likely, is it
                if (lm.getLoincTest() != null) {
                    dto.setLoincTestId(lm.getLoincTest().getId());
                }
            }

            setRandomTestData(dto, requestContext.getAccessingUserId(), "Random mapping test generator", randomGenerator, zdt);
            results.add(dto);
        }
        // go the scistore way for now: save one at a time and get lots of private maps to merge!
        return results;
    }

    private void setRandomTestData(TestResultDTO dto, long sourcePersonId, String sourceText, Random randomGenerator, ZonedDateTime zdt) {
        dto.setComments("Random test data");
        dto.setValue(toThreeDecimals(100.0 * randomGenerator.nextFloat()));
        dto.setRangeLow(0.0);
        dto.setRangeHigh(100.0);
        ZonedDateTime randomZdt = zdt.withYear(dateTimeService.nowZonedDateTime().getYear() - randomGenerator.nextInt(10)).withDayOfYear(randomGenerator.nextInt(364) + 1);
        dto.setTestDate(randomZdt.toInstant());
        dto.getSource().init(sourcePersonId, null, null, sourceText, Route.WEBAPP, null);
    }

    private double toThreeDecimals(double value) {
        return Math.floor(value * 1000) / 1000;
    }
}
