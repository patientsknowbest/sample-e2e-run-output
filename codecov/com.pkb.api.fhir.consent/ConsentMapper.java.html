package com.pkb.api.fhir.consent;

import com.pkb.api.fhir.PrivacyFlagsMapper;
import com.pkb.api.fhir.foundation.ReferenceMapper;
import com.pkb.api.fhir.person.RelatedPersonMapper;
import com.pkb.datamodel.authorization.ConsentedEntity;
import com.pkb.datamodel.consent.PatientConsent;
import com.pkb.datamodel.user.Patient;
import com.pkb.entities.enums.PrivacyFlag;
import io.vavr.collection.List;
import org.hl7.fhir.dstu3.model.CodeableConcept;
import org.hl7.fhir.dstu3.model.Coding;
import org.hl7.fhir.dstu3.model.Consent;
import org.hl7.fhir.dstu3.model.DomainResource;
import org.hl7.fhir.dstu3.model.Reference;
import org.hl7.fhir.dstu3.model.RelatedPerson;

import static com.pkb.api.fhir.Authority.authority;
import static com.pkb.api.fhir.IdGeneratorUtil.generateIdFor;
import static java.util.Collections.singletonList;

public class ConsentMapper {
    private final ReferenceMapper referenceMapper;
    private final PrivacyFlagsMapper privacyFlagsMapper;
    private final RelatedPersonMapper relatedPersonMapper;

    public ConsentMapper(ReferenceMapper referenceMapper, PrivacyFlagsMapper privacyFlagsMapper, RelatedPersonMapper relatedPersonMapper) {
        this.referenceMapper = referenceMapper;
        this.privacyFlagsMapper = privacyFlagsMapper;
        this.relatedPersonMapper = relatedPersonMapper;
    }

    public List<Consent> toConsentList(List<PatientConsent> patientConsents) {
        return patientConsents.map(this::toConsent);
    }

    public Consent toConsent(PatientConsent patientConsent) {
        Patient patient = patientConsent.getPatient();

        Consent consent = new Consent();
        consent.setId(patientConsent.getPublicId().toString());
        consent.setStatus(patientConsent.isDischarged() ? Consent.ConsentState.INACTIVE : Consent.ConsentState.ACTIVE);
        consent.setPatient(referenceMapper.toReferenceWithName(patient));
        consent.setActor(singletonList(toConsentActorComponent(patientConsent.getConsentedEntity(), patient, consent)));
        consent.setExcept(patientConsent.getConsentsGranted().map(this::toExceptComponent).asJava());

        Consent.ConsentPolicyComponent cpp = new Consent.ConsentPolicyComponent();
        cpp.setAuthority(authority());
        consent.setPolicy(singletonList(cpp));

        return consent;
    }

    private Consent.ConsentActorComponent toConsentActorComponent(ConsentedEntity actor, Patient patient, DomainResource resource) {
        Consent.ConsentActorComponent cac = new Consent.ConsentActorComponent();
        cac.setRole(getCodeForCONSENTER());

        if (actor instanceof Patient) {
            Patient carer = (Patient) actor;
            RelatedPerson relatedPerson = relatedPersonMapper.toRelatedPerson(patient, carer);
            relatedPerson.setId("#carer-" + generateIdFor(carer, Patient::toString));
            resource.addContained(relatedPerson);

            Reference reference = new Reference();
            reference.setReference(relatedPerson.getId());
            reference.setDisplay(referenceMapper.toDisplay(carer));
            cac.setReference(reference);
        } else {
            cac.setReference(referenceMapper.toReferenceWithName(actor));
        }
        return cac;
    }

    private CodeableConcept getCodeForCONSENTER() {
        Coding coding = new Coding();
        coding.setCode("CONSENTER");
        coding.setSystem("http://hl7.org/fhir/v3/RoleCode");

        CodeableConcept codeableConcept = new CodeableConcept();
        codeableConcept.setCoding(singletonList(coding));

        return codeableConcept;
    }

    private Consent.ExceptComponent toExceptComponent(PrivacyFlag pf) {
        Consent.ExceptComponent exceptComponent = new Consent.ExceptComponent();
        exceptComponent.setType(Consent.ConsentExceptType.PERMIT);
        exceptComponent.setSecurityLabel(singletonList(privacyFlagsMapper.toCoding(pf)));
        return exceptComponent;
    }

}
