package com.pkb.api.fhir.consent;

import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;

import com.pkb.authz.AuthorizationDataService;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.authz.data.AuthorizationInputs;
import com.pkb.datamodel.consent.PatientConsent;
import com.pkb.datamodel.consent.Purview;
import com.pkb.domain.ConsentService;
import com.pkb.domain.SystemUserTypeNotPermittedException;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.criteria.FhirCriteriaResult;
import com.pkb.domain.criteria.ImmutableFhirCriteriaResult;

import io.vavr.control.Either;
import org.hl7.fhir.dstu3.model.Consent;
import org.hl7.fhir.dstu3.model.Reference;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.google.common.base.Predicates.instanceOf;
import static com.pkb.authz.ImmutableTargetMeta.targetMeta;
import static com.pkb.authz.action.Action.READ;
import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.context.Path.FHIR;
import static com.pkb.authz.data.ImmutableAuthorizationInputs.inputs;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.API.None;
import static io.vavr.API.Set;
import static io.vavr.API.Some;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

public class ConsentResourceService extends AbstractResourceService<Consent> {
    private final ConsentService service;
    private final ReferenceMapper referenceMapper;
    private final ConsentMapper consentMapper;
    private final AuthorizationDataService authorizationDataService;

    public ConsentResourceService(ConsentService service,
                                  ConsentMapper mapper,
                                  ReferenceMapper referenceMapper,
                                  ServerResponseExceptionMapper exceptionMapper,
                                  AuthorizationDataService authorizationDataService) {
        super(exceptionMapper);
        this.service = service;
        this.referenceMapper = referenceMapper;
        this.consentMapper = mapper;
        this.authorizationDataService = authorizationDataService;
    }

    public @NotNull Either<BaseServerResponseException, Reference> purview(AuthenticatedIdentity authenticatedIdentity) {

        AuthorizationInputs authorizationInputs = inputs()
                .authenticatedIdentity(authenticatedIdentity)
                .action(READ)
                .addTargets(targetMeta(Purview.class, Set(), None(), true))
                .path(FHIR)
                .build();

        if (authorizationInputs.authenticatedSystemUser().isDefined()) {
            return left(toSystemUserTypeNotPermitted());
        }

        AuthorizationData authorizationData = authorizationDataService.fetch(authorizationInputs);

        return service.determinePurview(authorizationData)
                .map(referenceMapper::toReference)
                .mapLeft(exception -> exceptionToFhir(exception, Consent.class));
    }

    public @NotNull Either<BaseServerResponseException, FhirCriteriaResult<Consent>> findPatientConsents(AuthenticatedIdentity authenticatedIdentity, UUID patientPublicId) {
        AuthorizationInputs authorizationInputs = inputs()
                .authenticatedIdentity(authenticatedIdentity)
                .action(SEARCH)
                .addTargets(targetMeta(PatientConsent.class, Set(), Some(patientPublicId), true))
                .path(FHIR)
                .build();

        AuthorizationData authorizationData = authorizationDataService.fetch(authorizationInputs);

        return service.findConsents(authorizationData, patientPublicId)
                .fold(error -> Match(error).of(
                        Case($(instanceOf(SystemUserTypeNotPermittedException.class)), nae -> left(toSystemUserTypeNotPermitted())),
                        Case($(), right(FhirCriteriaResult.empty()))),
                        result -> right(mapConsents(result)));
    }

    private FhirCriteriaResult<Consent> mapConsents(CriteriaResult<PatientConsent> result) {
        List<Consent> consents = result.getPage().map(consentMapper::toConsent);
        return ImmutableFhirCriteriaResult.<Consent>fhirCriteriaResult()
                .total(result.getTotal())
                .page(consents)
                .build();
    }
}
