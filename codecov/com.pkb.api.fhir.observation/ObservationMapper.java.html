package com.pkb.api.fhir.observation;

import com.pkb.api.fhir.TemporalMapper;
import com.pkb.datamodel.LabResult;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.criteria.ImmutableCriteriaResult;
import org.hl7.fhir.dstu3.model.Observation;

import static com.pkb.api.fhir.DataAbsentReasonUtil.getDataAbsentExtension;
import static org.hl7.fhir.dstu3.model.Observation.ObservationStatus.UNKNOWN;

public class ObservationMapper {

    private TemporalMapper temporalMapper;

    public ObservationMapper(TemporalMapper temporalMapper) {
        this.temporalMapper = temporalMapper;
    }

    public CriteriaResult<Observation> toObservations(CriteriaResult<LabResult> labResults) {

        return ImmutableCriteriaResult.<Observation> builder()
                .addAllPage(labResults.getPage()
                        .map(this::toObservation))
                .total(labResults.getTotal())
                .build();
    }

    public Observation toObservation(LabResult labResult) {
        Observation observation = new Observation();
        observation.setId(labResult.getUniqueId().toString());
        observation.setStatus(UNKNOWN);
        observation.getCode().addExtension(getDataAbsentExtension());
        observation.setEffective(temporalMapper.toDateTimeType(labResult.getEffectiveTimestamp()));
        return observation;
    }
}
