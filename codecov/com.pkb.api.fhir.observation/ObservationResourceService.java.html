package com.pkb.api.fhir.observation;

import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import com.pkb.api.fhir.AbstractResourceService;
import com.pkb.api.fhir.config.FhirApiProperties;
import com.pkb.api.fhir.exception.ServerResponseExceptionMapper;
import com.pkb.authz.AuthorizationDataService;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.datamodel.LabResult;
import com.pkb.domain.SystemUserTypeNotPermittedException;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.labresult.ImmutableLabResultCriteria;
import com.pkb.domain.labresult.LabResultCriteria;
import com.pkb.domain.labresult.LabResultService;
import io.vavr.control.Either;
import org.hl7.fhir.dstu3.model.Observation;
import org.jetbrains.annotations.NotNull;

import static com.google.common.base.Predicates.instanceOf;
import static com.pkb.authz.ImmutableTargetMeta.targetMeta;
import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.context.Path.FHIR;
import static com.pkb.authz.data.ImmutableAuthorizationInputs.inputs;
import static com.pkb.domain.criteria.CriteriaResult.emptyCriteriaResult;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.API.Set;
import static io.vavr.API.Some;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

public class ObservationResourceService extends AbstractResourceService<Observation> {
    private final AuthorizationDataService authorizationDataService;
    private final LabResultService labResultService;
    private final ObservationMapper observationMapper;
    private final FhirApiProperties fhirConfig;

    public ObservationResourceService(AuthorizationDataService authorizationDataService,
                                      ServerResponseExceptionMapper exceptionMapper,
                                      ObservationMapper observationMapper,
                                      LabResultService labResultService,
                                      FhirApiProperties fhirConfig) {
        super(exceptionMapper);
        this.observationMapper = observationMapper;
        this.labResultService = labResultService;
        this.fhirConfig = fhirConfig;
        this.authorizationDataService = authorizationDataService;
    }

    public @NotNull Either<BaseServerResponseException, CriteriaResult<Observation>> search(@NotNull ObservationSearchCriteria criteria) {
        AuthorizationData authzData = authorizationDataService.fetch(inputs()
                .path(FHIR)
                .authenticatedIdentity(criteria.authenticatedIdentity())
                .action(SEARCH)
                .addTargets(targetMeta(LabResult.class, Set(), Some(criteria.patientId()), true))
                .build());
        // All authz exceptions are ignored and we just return an empty result for search operations.
        // Except for SystemUserTypeNotPermittedException, which returns an error response to the caller
        // https://pkbdev.atlassian.net/wiki/spaces/TECHNOLOGY/pages/135921753/FHIR+API+Roadmap+Observation
        return labResultService.search(authzData, mapToLabResultCriteria(criteria))
                .fold(error -> Match(error).of(
                        Case($(instanceOf(SystemUserTypeNotPermittedException.class)), nae -> left(toSystemUserTypeNotPermitted())),
                        Case($(), right(emptyCriteriaResult()))),
                        appointments -> right(observationMapper.toObservations(appointments)));
    }

    private LabResultCriteria mapToLabResultCriteria(ObservationSearchCriteria criteria) {
        return ImmutableLabResultCriteria.builder()
                .patientPublicId(criteria.patientId())
                .effectiveTimestamp(criteria.effectiveTimestamp())
                .resultFilter(criteria.resultFilter())
                .pageSize(fhirConfig.getObservationMaxNumberOfResources())
                .build();
    }

}
