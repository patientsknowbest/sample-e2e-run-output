package com.pkb.pubsub.aggregator.entity;

import com.vladmihalcea.hibernate.type.json.JsonBinaryType;
import org.hibernate.annotations.Type;
import org.hibernate.annotations.TypeDef;

import javax.persistence.Column;
import javax.persistence.Embeddable;
import java.time.Instant;
import java.util.Objects;

@Embeddable
@TypeDef(name = "jsonb", typeClass = JsonBinaryType.class)
public class PublicAggregatableResponse {

    @Column(name="service_name")
    private String serviceName;

    @Column(name="message_json")
    @Type(type = "jsonb")
    private String messageJson;

    @Column(name="created_date")
    private Instant createdDate;

    protected PublicAggregatableResponse() {
    }

    public String getServiceName() {
        return serviceName;
    }

    public String getMessageJson() {
        return messageJson;
    }

    public Instant getCreatedDate() {
        return createdDate;
    }

    @Override
    public String toString() {
        return "PublicAggregatableResponse{" +
                ", serviceName='" + serviceName + '\'' +
                '}';
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (!(o instanceof PublicAggregatableResponse)) {
            return false;
        }
        PublicAggregatableResponse that = (PublicAggregatableResponse) o;
        return serviceName.equals(that.serviceName);
    }

    @Override
    public int hashCode() {
        return Objects.hash(serviceName);
    }

}
