package com.pkb.pubsub.aggregator.entity;

import com.pkb.pubsub.aggregator.models.AggregateResponse;
import com.vladmihalcea.hibernate.type.json.JsonBinaryType;
import org.hibernate.annotations.Type;
import org.hibernate.annotations.TypeDef;
import org.jetbrains.annotations.NotNull;

import javax.persistence.CollectionTable;
import javax.persistence.Column;
import javax.persistence.ElementCollection;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.OrderBy;
import javax.persistence.Table;
import java.time.Instant;
import java.util.LinkedHashSet;
import java.util.Objects;
import java.util.Set;
import java.util.UUID;

@Entity
@Table(name = "aggregatable_request", schema = "public")
@TypeDef(name = "jsonb", typeClass = JsonBinaryType.class)
public class PublicAggregatableRequest {

    @Id
    private UUID id;

    @ElementCollection(fetch = FetchType.EAGER)
    @CollectionTable(name = "aggregatable_response", joinColumns = {@JoinColumn(name = "request_id")})
    @OrderBy("createdDate")
    private Set<PublicAggregatableResponse> responses = new LinkedHashSet<>();

    @Column(name = "message_json")
    @Type(type = "jsonb")
    private String messageJson;

    @SuppressWarnings("unused")
    @Column(name = "created_date")
    private Instant createdDate;

    @Column(name = "status")
    @Enumerated(EnumType.STRING)
    private AggregateResponse.Status status = AggregateResponse.Status.INCOMPLETE;

    PublicAggregatableRequest() {
    }

    public UUID getId() {
        return id;
    }

    @NotNull
    public Set<PublicAggregatableResponse> getResponses() {
        return responses;
    }

    public String getMessageJson() {
        return messageJson;
    }

    public Instant getCreatedDate() {
        return createdDate;
    }

    public AggregateResponse.Status getStatus() {
        return status;
    }

    public void setStatus(AggregateResponse.Status status) {
        this.status = status;
    }

    @Override
    public String toString() {
        return "PublicAggregatableRequest{" +
                "id=" + id +
                '}';
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (!(o instanceof PublicAggregatableRequest)) {
            return false;
        }
        PublicAggregatableRequest that = (PublicAggregatableRequest) o;
        return Objects.equals(id, that.id);
    }

    @Override
    public int hashCode() {
        return Objects.hash(id);
    }

}
