package com.pkb.api.fhir.contact;

import com.pkb.datamodel.contact.PersonContact;
import com.pkb.entities.enums.ContactType;
import io.vavr.collection.List;
import org.hl7.fhir.dstu3.model.ContactPoint;

import static java.text.MessageFormat.format;
import static org.hl7.fhir.dstu3.model.ContactPoint.ContactPointSystem.EMAIL;
import static org.hl7.fhir.dstu3.model.ContactPoint.ContactPointSystem.PHONE;

public class ContactPointMapper {

    private List<ContactPoint> toPersonContactPoints(List<PersonContact> contacts) {
        return contacts.map(this::toContactPoint);
    }

    public List<ContactPoint> toPersonContactPoints(List<PersonContact> contacts, List<PersonContact> otherContacts) {
        return toPersonContactPoints(contacts.appendAll(otherContacts));
    }

    public ContactPoint toContactPoint(PersonContact contact) {
        return toContactPoint(contact.getType(), contact.getValue());
    }

    private ContactPoint toContactPoint(ContactType type, String value) {
        ContactPoint contactPoint = new ContactPoint();
        switch (type) {
            case EMAIL:
                contactPoint.setSystem(EMAIL);
                break;
            case PHONE:
                contactPoint.setSystem(PHONE);
                break;
            default:
                throw new RuntimeException(format("Unsupported contact type {0}", type.name()));
        }
        contactPoint.setValue(value);
        return contactPoint;
    }
}
