package com.pkb.service.dataupload.emis.esv2.converters;

import com.pkb.service.dataupload.emis.esv2.EmisLsEntry;
import com.pkb.service.dataupload.emis.esv2.exception.EmisFilePatternException;
import com.pkb.service.dataupload.emis.esv2.model.EmisFile;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileType;
import com.pkb.service.dataupload.emis.esv2.model.ImmutableEmisFile;
import org.jetbrains.annotations.NotNull;
import org.springframework.core.convert.converter.Converter;

import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;

import static org.apache.commons.lang3.StringUtils.endsWithIgnoreCase;
import static org.apache.commons.lang3.StringUtils.removeEndIgnoreCase;
import static org.apache.commons.lang3.StringUtils.split;

/**
 * Parses file name like delta_37602_Admin_Patient_20211104002203_4AFE1410-126A-4A0D-BCE7-839AC06B8213.csv.gpg
 */
public class EmisLsEntryToEmisFileConverter implements Converter<EmisLsEntry, EmisFile> {

    private static final String EMIS_NOT_ENCRYPTED_FILE_EXTENSION = ".csv";
    private static final String EMIS_ENCRYPTED_FILE_EXTENSION = ".csv.gpg";
    private static final String FIELD_SEPARATOR = "_";
    private static final int NUMBER_OF_FIELDS_IN_FILENAME = 6;
    private static final int DELTA_OR_BULK_INDEX = 0;
    private static final int TYPE_BEGIN_INDEX = 2;
    private static final int TYPE_END_INDEX = 3;
    private static final int TIMESTAMP_INDEX = 4;
    private static final int GUID_INDEX = 5;
    private static final DateTimeFormatter TIMESTAMP_FORMATTER = DateTimeFormatter.ofPattern("yyyyMMddHHmmss");
    private static final String BULK = "Bulk";
    private static final String DELTA = "Delta";

    private final StringToEmisFileTypeConverter stringToEmisFileTypeConverter;

    public EmisLsEntryToEmisFileConverter(@NotNull StringToEmisFileTypeConverter stringToEmisFileTypeConverter) {
        this.stringToEmisFileTypeConverter = stringToEmisFileTypeConverter;
    }

    @Override
    @NotNull
    public EmisFile convert(EmisLsEntry lsEntry) {
        String[] fileNamePieces = split(lsEntry.getFilename(), FIELD_SEPARATOR);
        if (fileNamePieces == null || fileNamePieces.length != NUMBER_OF_FIELDS_IN_FILENAME) {
            throw new EmisFilePatternException("Filename can't be split into required fields!");
        }

        ImmutableEmisFile.Builder builder = processFileNameParts(fileNamePieces);
        builder.size(lsEntry.size());
        builder.relativePath(lsEntry.getRelativePath());
        return builder.build();
    }

    private ImmutableEmisFile.Builder processFileNameParts(String[] pieces) {
        return ImmutableEmisFile.builder()
                .isBulk(isBulk(pieces[DELTA_OR_BULK_INDEX]))
                .timeStamp(LocalDateTime.parse(pieces[TIMESTAMP_INDEX], TIMESTAMP_FORMATTER))
                .type(getFileType(pieces[TYPE_BEGIN_INDEX] + FIELD_SEPARATOR + pieces[TYPE_END_INDEX]))
                .guid(removeCSVFileExtension(pieces[GUID_INDEX]));
    }

    private boolean isBulk(String segment) {
        if (segment.equalsIgnoreCase(BULK)) {
            return true;
        } else if (segment.equalsIgnoreCase(DELTA)) {
            return false;
        } else {
            throw new EmisFilePatternException("Unexpected Bulk or Delta segment " + segment);
        }
    }

    private String removeCSVFileExtension(String input) {
        if (endsWithIgnoreCase(input, EMIS_ENCRYPTED_FILE_EXTENSION)) {
            return removeEndIgnoreCase(input, EMIS_ENCRYPTED_FILE_EXTENSION);
        } else if (endsWithIgnoreCase(input, EMIS_NOT_ENCRYPTED_FILE_EXTENSION)) {
            return removeEndIgnoreCase(input, EMIS_NOT_ENCRYPTED_FILE_EXTENSION);
        } else {
            throw new EmisFilePatternException("Not a CSV file!");
        }
    }

    private EmisFileType getFileType(String input) {
        return stringToEmisFileTypeConverter.convert(input).orElseThrow(() -> new EmisFilePatternException("Unsupported EMIS file type!"));
    }

}
