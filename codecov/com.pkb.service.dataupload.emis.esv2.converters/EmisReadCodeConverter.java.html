package com.pkb.service.dataupload.emis.esv2.converters;

import static com.pkb.service.dataupload.emis.esv2.model.Constants.CODE_SUFFIX_KC60;
import static com.pkb.service.dataupload.emis.esv2.model.Constants.CODE_SUFFIX_SHHAPT;
import static com.pkb.service.dataupload.emis.esv2.model.Constants.EMIS_READ_CODE_SEPARATOR;
import static com.pkb.service.dataupload.emis.esv2.model.Constants.READ_V2_CODE_LENGTH;
import static com.pkb.service.dataupload.emis.esv2.model.Constants.READ_V2_CODE_PAD_VALUE;

import java.util.Optional;

import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.springframework.core.convert.converter.Converter;

public class EmisReadCodeConverter implements Converter<String, Optional<String>> {
    @Override
    public @NotNull Optional<String> convert(@Nullable String source) {
        return Optional.ofNullable(source)
                .filter(r -> !StringUtils.endsWithAny(r, CODE_SUFFIX_KC60, CODE_SUFFIX_SHHAPT))
                .map(r -> r.split(EMIS_READ_CODE_SEPARATOR, 2)[0])
                .filter(StringUtils::isNotBlank)
                .filter(r -> r.length() <= READ_V2_CODE_LENGTH)
                .map(r -> StringUtils.rightPad(r, READ_V2_CODE_LENGTH, READ_V2_CODE_PAD_VALUE));
    }
}
