package com.pkb.service.dataupload.emis.esv2.converters;

import static org.apache.commons.lang3.StringUtils.isNotBlank;

import java.util.Map;
import java.util.Optional;

import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.springframework.core.convert.converter.Converter;

import com.google.common.collect.ImmutableMap;
import com.google.common.collect.ImmutableMap.Builder;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileType;

public class StringToEmisFileTypeConverter implements Converter<String, Optional<EmisFileType>> {

    private static final Map<String, EmisFileType> mappings;

    static {
        Builder<String, EmisFileType> builder = ImmutableMap.builder();
        for (EmisFileType actual : EmisFileType.values()) {
            builder.put(actual.getFileNamePart(), actual);
        }
        mappings = builder.build();
    }

    @Override
    @NotNull
    public Optional<EmisFileType> convert(@Nullable String source) {
        Optional<EmisFileType> result = Optional.empty();
        if (isNotBlank(source)) {
            result = Optional.ofNullable(mappings.get(source));
        }
        return result;
    }
}
