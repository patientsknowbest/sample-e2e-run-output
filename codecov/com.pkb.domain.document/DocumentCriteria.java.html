package com.pkb.domain.document;

import com.pkb.authz.data.AuthorizationData;
import com.pkb.domain.criteria.Constraint;
import com.pkb.domain.criteria.PageSized;
import com.pkb.domain.criteria.Restriction;
import com.pkb.domain.criteria.TemporalRangeRestriction;
import io.vavr.control.Option;
import org.immutables.value.Value;
import org.immutables.vavr.encodings.VavrEncodingEnabled;

import java.time.LocalDate;
import java.util.UUID;

@Value.Immutable
@VavrEncodingEnabled
public abstract class DocumentCriteria implements PageSized {
    public abstract AuthorizationData authData();

    public abstract Option<UUID> getEncounterId();

    public abstract Option<Restriction<Constraint, LocalDate>> getIndexed();

    public abstract Option<TemporalRangeRestriction<LocalDate>> getCreated();

    public abstract Option<String> getDocumentStatus();

    public abstract Option<UUID> getSourceOrganisationId();

    public abstract int capsize();
}
