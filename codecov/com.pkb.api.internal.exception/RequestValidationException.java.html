package com.pkb.api.internal.exception;

import org.springframework.http.HttpStatus;
import org.springframework.validation.FieldError;
import org.springframework.validation.ObjectError;

import java.util.List;

public class RequestValidationException extends InternalApiException {

    private final List<ObjectError> generalErrors;
    private final List<FieldError> fieldErrors;
    private HttpStatus statusCode;

    public RequestValidationException(List<ObjectError> generalErrors, List<FieldError> fieldErrors, HttpStatus statusCode) {
        this.generalErrors = generalErrors;
        this.fieldErrors = fieldErrors;
        this.statusCode = statusCode;
    }

    public List<ObjectError> getGeneralErrors() {
        return generalErrors;
    }

    public List<FieldError> getFieldErrors() {
        return fieldErrors;
    }

    public HttpStatus getStatusCode() {
        return statusCode;
    }
}
