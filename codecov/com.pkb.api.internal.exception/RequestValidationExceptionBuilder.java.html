package com.pkb.api.internal.exception;

import com.google.common.base.Optional;
import com.pkb.common.StatefulBuilder;
import org.springframework.http.HttpStatus;
import org.springframework.validation.FieldError;
import org.springframework.validation.ObjectError;

import java.util.ArrayList;
import java.util.List;

import static org.apache.commons.collections4.CollectionUtils.isNotEmpty;
import static org.apache.commons.lang3.ArrayUtils.EMPTY_OBJECT_ARRAY;

/**
 * Builder for {@link RequestValidationException}.
 */
public class RequestValidationExceptionBuilder
        implements StatefulBuilder<Optional<RequestValidationException>, RequestValidationExceptionBuilder> {

    private final List<ObjectError> generalErrors = new ArrayList<>();
    private final List<FieldError> fieldErrors = new ArrayList<>();
    private HttpStatus statusCode = HttpStatus.BAD_REQUEST;

    /**
     * Builds an {@link Optional} {@link RequestValidationException}.
     * 
     * @return {@link Optional#absent()} in case there are no errors, an {@link RequestValidationException} otherwise.
     */
    @Override
    public Optional<RequestValidationException> build() {
        Optional<RequestValidationException> result = Optional.absent();
        if (hasAnyError()) {
            result = Optional.of(new RequestValidationException(generalErrors, fieldErrors, statusCode));
        }
        return result;
    }

    /**
     * Throws the built {@link RequestValidationException} in case there are any error set.
     * 
     * @throws RequestValidationException
     */
    public void throwIfNeeded() throws RequestValidationException {
        if (hasAnyError()) {
            throw new RequestValidationException(generalErrors, fieldErrors, statusCode);
        }
    }

    public HttpStatus getStatusCode() {
        return statusCode;
    }

    public void setStatusCode(HttpStatus statusCode) {
        this.statusCode = statusCode;
    }

    /**
     * Adds a general error to the list.
     * These errors are not specific to a field in the request, but a general request error.
     * 
     * @param code
     *            I18N message code.
     * @param args
     *            Arguments for I18N message.
     */
    public void addGeneralError(String code, Object[] args) {
        generalErrors.add(new ObjectError("", new String[] { code }, args, ""));
    }

    /**
     * Adds a general error to the list.
     * These errors are not specific to a field in the request, but a general request error.
     * 
     * @param code
     *            I18N message code.
     */
    public void addGeneralError(String code) {
        addGeneralError(code, EMPTY_OBJECT_ARRAY);
    }

    /**
     * Adds a field specific error to the list.
     * These errors are specific to a field in the request.
     * 
     * @param fieldName
     *            The name of the field
     * @param rejectedValue
     *            The value of the field
     * @param code
     *            I18N message code.
     * @param args
     *            Arguments for I18N message.
     */
    public void addFieldError(String fieldName, Object rejectedValue, String code, Object[] args) {
        fieldErrors.add(new FieldError("", fieldName, rejectedValue, false, new String[] { code }, args, ""));
    }

    /**
     * Adds a field specific error to the list.
     * These errors are specific to a field in the request.
     * 
     * @param fieldName
     *            The name of the field
     * @param rejectedValue
     *            The value of the field
     * @param code
     *            I18N message code.
     */
    public void addFieldError(String fieldName, Object rejectedValue, String code) {
        addFieldError(fieldName, rejectedValue, code, EMPTY_OBJECT_ARRAY);
    }

    private boolean hasAnyError() {
        return isNotEmpty(generalErrors) || isNotEmpty(fieldErrors);
    }
}
