package com.pkb.aggregator.persistence.entity.embeddable

import java.io.Serializable
import java.util.UUID
import javax.persistence.Column
import javax.persistence.Embeddable

/**
 * Combines an upstream id (e.g. 'org-1') with an upstream resource id (e.g. 'Patient/123') to create a namespaced
 * reference that can reliably identify a resource without worrying about id clashes should another upstream manage to
 * create a resource with an identical id
 */
@Embeddable
data class UpstreamResource(
    @Column(name = "upstream_source")
    val upstreamSource: String,

    @Column(name = "upstream_resource_id")
    val upstreamResourceId: String,
) : Serializable {
    fun toDownstreamResourceId(): String {
        val bytes = "pkb-internal-$upstreamSource-$upstreamResourceId".toByteArray()
        return UUID.nameUUIDFromBytes(bytes).toString()
    }
}
