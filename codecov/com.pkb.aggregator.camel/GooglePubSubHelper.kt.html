package com.pkb.aggregator.camel

import com.pkb.aggregator.camel.routing.dto.UpstreamChangeNotification
import com.pkb.aggregator.extension.valueAsMutableMapOf
import com.pkb.aggregator.service.exception.DownstreamResourceExistsException
import com.pkb.aggregator.service.exception.DownstreamVersionConflictException
import org.apache.camel.Body
import org.apache.camel.Exchange
import org.apache.camel.Headers
import org.apache.camel.component.google.pubsub.GooglePubsubConstants
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class GooglePubSubHelper {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    companion object {

        const val PUBSUB_ACTION_CREATE_RESOURCE = "CreateResource"
        const val PUBSUB_ACTION_UPDATE_RESOURCE = "UpdateResource"
        const val PUBSUB_ACTION_DELETE_RESOURCE = "DeleteResource"

        // These are the attribute names defined by the Google PubSub integration. See
        // https://cloud.google.com/healthcare-api/docs/concepts/pubsub
        const val ATTRIBUTE_ACTION = "action"
        const val ATTRIBUTE_RESOURCE_TYPE = "resourceType"

        private val CHANGE_NOTIFICATION_ATTRIBUTES = setOf(ATTRIBUTE_RESOURCE_TYPE, ATTRIBUTE_ACTION)
        private const val DELIVERY_COUNT_ATTRIBUTE_KEY = "googclient_deliveryattempt"
    }

    fun parseIncomingMessage(
        @Body body: ByteArray,
        @Headers headers: MutableMap<String, Any>,
    ): UpstreamChangeNotification {
        val attributes = getChangeNotificationAttributes(headers)
        val notification = UpstreamChangeNotification(String(body), attributes)

        log.info(
            "Received {} notification for {} (delivery attempt {})",
            notification.attributes[ATTRIBUTE_ACTION],
            notification.data,
            getDeliveryCount(headers)
        )
        return notification
    }

    fun checkForErrorRequiringRedelivery(exchange: Exchange) {
        // Some errors require a pubsub redelivery rather than a camel retry-from-point-of-failure. Such errors are
        // detected here and the exception re-thrown so that the pubsub message is nack'd
        when (val exception = exchange.getProperty(Exchange.EXCEPTION_CAUGHT)) {
            is DownstreamVersionConflictException -> {
                // Log an info-level message so as not to trigger any alerts, since this 'error' is just a retry
                log.info("Version conflict, will retry: {} {} (upstream={}, downstream={})",
                    exception.code,
                    exception.message,
                    exchange.getProperty(SOURCED_RESOURCE_PROPERTY),
                    exchange.getProperty(DOWNSTREAM_RESOURCE_ID_PROPERTY)
                )
                throw exception
            }
            is DownstreamResourceExistsException -> {
                log.info("Downstream resource unexpectedly existed during POST, will retry: {} (upstream={}, downstream={})",
                    exception.message,
                    exchange.getProperty(SOURCED_RESOURCE_PROPERTY),
                    exchange.getProperty(DOWNSTREAM_RESOURCE_ID_PROPERTY)
                )
                throw exception
            }
        }
    }

    private fun getChangeNotificationAttributes(headers: MutableMap<String, Any>) =
        headers.valueAsMutableMapOf<String, String>(GooglePubsubConstants.ATTRIBUTES)
            .filterKeys { CHANGE_NOTIFICATION_ATTRIBUTES.contains(it) }
            .toMutableMap()

    private fun getDeliveryCount(headers: Map<String, Any>) =
        headers.valueAsMutableMapOf<String, String>(GooglePubsubConstants.ATTRIBUTES)
            .getOrDefault(DELIVERY_COUNT_ATTRIBUTE_KEY, "1")
            .toInt()
}
