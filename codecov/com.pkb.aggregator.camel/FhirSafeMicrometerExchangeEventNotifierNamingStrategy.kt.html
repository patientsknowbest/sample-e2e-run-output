package com.pkb.aggregator.camel

import io.micrometer.core.instrument.Tags
import org.apache.camel.Endpoint
import org.apache.camel.Exchange
import org.apache.camel.component.micrometer.MicrometerConstants
import org.apache.camel.component.micrometer.MicrometerConstants.DEFAULT_CAMEL_EXCHANGE_EVENT_METER_NAME
import org.apache.camel.component.micrometer.eventnotifier.MicrometerEventNotifierService
import org.apache.camel.component.micrometer.eventnotifier.MicrometerExchangeEventNotifierNamingStrategy
import org.apache.camel.spi.CamelEvent
import org.springframework.stereotype.Component

/**
 * Safe (non-leaky, low-cardinality) implementation of [MicrometerExchangeEventNotifierNamingStrategy].
 *
 * Detects camel-fhir endpoints and uses `endpointBaseUri` as the endpoint name (i.e. strips off any query params as
 * they may leak authc info and can be high-cardinality). Uses default implementation for other endpoints.
 */
@Component
class FhirSafeMicrometerExchangeEventNotifierNamingStrategy : MicrometerExchangeEventNotifierNamingStrategy {

    override fun getName(exchange: Exchange, endpoint: Endpoint) = DEFAULT_CAMEL_EXCHANGE_EVENT_METER_NAME

    /**
     * If endpoint is a fhir request,
     */
    override fun getTags(event: CamelEvent.ExchangeEvent, endpoint: Endpoint): Tags =
        if (endpoint is org.apache.camel.component.fhir.FhirEndpoint) {
            Tags.of(
                MicrometerConstants.CAMEL_CONTEXT_TAG, event.exchange.context.name,
                MicrometerConstants.SERVICE_NAME, MicrometerEventNotifierService::class.java.simpleName,
                MicrometerConstants.EVENT_TYPE_TAG, event.javaClass.simpleName,
                MicrometerConstants.ENDPOINT_NAME, endpoint.endpointBaseUri,
                MicrometerConstants.FAILED_TAG, event.exchange.isFailed.toString(),
            )
        } else {
            super.getTags(event, endpoint)
        }
}
