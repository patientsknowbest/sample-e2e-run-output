package com.pkb.aggregator.camel

import com.pkb.aggregator.R4IdType
import org.apache.camel.ExchangeProperty
import org.apache.camel.Headers
import org.springframework.stereotype.Service

@Service
class CamelFhirHelper {

    /**
     * Sets the url parameter for the next FHIR operation using the message header override, per
     * https://camel.apache.org/components/3.17.x/fhir-component.html#_api_search_method_searchByUrl
     */
    fun prepareDownstreamSearchById(
        @ExchangeProperty(DOWNSTREAM_RESOURCE_ID_PROPERTY) resourceId: String,
        @Headers headers: MutableMap<String, Any>,
    ) {
        with (R4IdType(resourceId)) {
            headers[CAMELFHIR_URL_HEADER] = "${this.resourceType}?id=${this.idPart}"
        }
    }
}
