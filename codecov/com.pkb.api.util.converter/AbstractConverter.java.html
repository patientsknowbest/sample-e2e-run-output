package com.pkb.api.util.converter;

import com.pkb.app.entity.EHRRequestContext;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import java.util.stream.Collectors;

import static java.util.Collections.emptyList;

public abstract class AbstractConverter<A, E> implements IConverter<A, E> {

    @Override
    public List<A> toApiModel(Collection<E> dtoList) throws ConversionException {
        if ((dtoList == null) || dtoList.isEmpty()) {
            return emptyList();
        }
        List<A> rval = new ArrayList<>(dtoList.size());
        rval.addAll(dtoList.stream().map(this::toApiModel).collect(Collectors.toList()));
        return rval;
    }

    @Override
    public List<E> toDTO(Collection<A> apiModelList, EHRRequestContext requestContext) throws ConversionException {
        if ((apiModelList == null) || apiModelList.isEmpty()) {
            return emptyList();
        }
        List<E> rval = new ArrayList<>(apiModelList.size());
        rval.addAll(apiModelList.stream().map(e -> toDTO(e, requestContext)).collect(Collectors.toList()));
        return rval;
    }

}
