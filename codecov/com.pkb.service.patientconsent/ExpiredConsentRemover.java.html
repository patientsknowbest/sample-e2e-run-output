package com.pkb.service.patientconsent;

import com.pkb.datamodel.Email;
import com.pkb.datamodel.consent.PatientConsentDTO;
import com.pkb.entities.enums.CarerRemovalAge;
import com.pkb.entities.enums.Message;
import com.pkb.exception.MailException;
import com.pkb.service.emailmessage.IAddressInfoProvider;
import com.pkb.service.emailmessage.ITemplateParamsFactory;
import com.pkb.service.emailmessage.impl.EmailManager;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.PKBConstants;
import io.prometheus.client.Counter;
import io.prometheus.client.Summary;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.List;
import java.util.Map;
import java.util.Optional;

import static java.util.Collections.emptyList;
import static java.util.Objects.requireNonNull;


public class ExpiredConsentRemover {

    private static final Summary METRICS = Summary.build()
            .name("pkb_phr_schedtask_expiredconsentremover_duration")
            .help("duration of the removal of carer consents of children reaching age limit")
            .labelNames("age")
            .register();

    private static final Counter COUNTER = Counter.build()
            .name("pkb_phr_schedtask_expiredconsentremover_count")
            .help("count of notifications sent about carer consents expiring due to children growing up")
            .labelNames("age")
            .register();

    private final Logger logger;

    private final UserAccessManager userAccessManager;

    private final EmailManager emailManager;

    private final UserManager userManager;

    private final ITemplateParamsFactory templateParamsFactory;

    private final IAddressInfoProvider addressInfoProvider;

    private final PatientConsentManager consentManager;


    public ExpiredConsentRemover(UserAccessManager userAccessManager,
                                 EmailManager emailManager,
                                 UserManager userManager,
                                 ITemplateParamsFactory templateParamsFactory,
                                 IAddressInfoProvider addressInfoProvider,
                                 PatientConsentManager consentManager) {
        this(userAccessManager, emailManager, userManager, templateParamsFactory,
                addressInfoProvider, LoggerFactory.getLogger(ExpiredConsentRemover.class), consentManager);
    }

    public ExpiredConsentRemover(UserAccessManager userAccessManager,
                                 EmailManager emailManager,
                                 UserManager userManager,
                                 ITemplateParamsFactory templateParamsFactory,
                                 IAddressInfoProvider addressInfoProvider,
                                 Logger logger,
                                 PatientConsentManager consentManager) {
        this.userAccessManager = requireNonNull(userAccessManager, "userAccessManager cannot be null");
        this.emailManager = requireNonNull(emailManager, "emailManager cannot be null");
        this.userManager = requireNonNull(userManager, "userManager cannot be null");
        this.templateParamsFactory = requireNonNull(templateParamsFactory, "templateParamsFactory cannot be null");
        this.addressInfoProvider = requireNonNull(addressInfoProvider, "addressInfoProvider cannot be null");
        this.logger = requireNonNull(logger, "logger cannot be null");
        this.consentManager = consentManager;
    }

    public synchronized void run(CarerRemovalAge age) {
        @SuppressWarnings("AccessToStaticFieldLockedOnInstance")
        Summary.Timer timer = METRICS.labels(String.valueOf(age)).startTimer();
        List<PatientConsentDTO> expiredConsents = emptyList();
        try {
            logger.info("entering ExpiredConsentRemover");
            expiredConsents = consentManager.getExpiredConsents(age);
            logger.info("deleting {} expired consents for age {}", expiredConsents.size(), age.getValue());
            for (PatientConsentDTO cons : expiredConsents) {
                PKBPerson carer = userManager.getPKBPerson(cons.requirePersonId(), PKBPerson.Lazy.CONTACTS);
                PKBPerson patient = userManager.getPKBPerson(cons.patientPersonId(), PKBPerson.Lazy.CONTACTS, PKBPerson.Lazy.PROPERTIES);
                consentManager.expireConsent(cons);
                userManager.setBirthdayProperty(patient, age.getRemovalPropertyName());

                sendNotificationsAboutConsentExpiry(carer, patient, age);
                logger.info("revoked access of carer {} from the account of patient {}", carer.getId(), patient.getId());
            }
        } finally {
            COUNTER.labels(String.valueOf(age.getValue())).inc(expiredConsents.size());
            timer.observeDuration();
        }
    }

    private void sendNotificationsAboutConsentExpiry(PKBPerson carer, PKBPerson patient, CarerRemovalAge age) {
        Map<String, Object> properties = templateParamsFactory.createWithSiteProps();
        properties.put("patient", patient);
        properties.put("carer", carer);
        properties.put(PKBConstants.PATIENT_AGE, age.getValue());

        sendNotification(patient, properties, Message.CHILD_CARER_CONSENT_EXPIRY_TO_PATIENT);
        properties.remove(PKBConstants.SUBJECT_PREFIX);
        sendNotification(carer, properties, Message.CHILD_CARER_CONSENT_EXPIRY_TO_CARER);
    }

    private void sendNotification(PKBPerson person, Map<String, Object> properties, Message message) {
        logger.debug("sending notification email to {}", person.findEmail().map(Email::address).getOrNull());

        Optional.ofNullable(person.getEmail()).ifPresentOrElse(email -> {
            try {
                emailManager.sendSystemMessage(message, null, addressInfoProvider.getContact(email).getOrNull(),
                        person, properties, person.getLocale(), null, null, null);
            } catch (MailException e) {
                logger.warn("Failed to send email about revoked access to person - {}", person.getId(), e);
            }
        }, () -> logger.warn("Primary email not found for person {}", person.getId()));
    }
}
