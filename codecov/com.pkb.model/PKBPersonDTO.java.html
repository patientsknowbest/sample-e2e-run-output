    /**
     * strip out non-functional formatting: parens, space, dash, dot leave + to
     * indicate country code
     */
    public String getPhoneClean() {
        if (isBlank(phone)) {
            return phone;
        }

        String whitelist = "1234567890+";

        StringBuilder clean = new StringBuilder();
        for (int i = 0; i < phone.length(); i++) {
            char c = phone.charAt(i);
            if (StringUtils.contains(whitelist, c)) {
                clean.append(c);
            }
        }
        return clean.toString();
    }

    public static PKBPersonDTO of(
            String title,
            String firstName,
            String lastName,
            Email emailId,
            UserType userType,
            String languageCode,
            String timezone) {

        PKBPersonDTO personDTO = new PKBPersonDTO();
        personDTO.setTitle(title);
        personDTO.setFirstName(firstName);
        personDTO.setLastName(lastName);
        personDTO.setUserType(userType);
        personDTO.setUserName(emailId);
        personDTO.setEmailId(emailId);
        personDTO.setStatus(UserStatus.CREATED);
        personDTO.setPassword(RandomUtil.randomPassword(10));

        // unlikely -- if we can verify it's impossible (and unit tested to stay that way), remove this
        if (isBlank(personDTO.getPassword())) {
            throw new IllegalStateException("Password generation failed; " + personDTO.getPassword());
        }

        // fields that can't be null when setting up the auth keys & LDAP record
        personDTO.setDateOfBirthString("1900-01-01");
        personDTO.setLanguageCode(languageCode);
        personDTO.setTimeZoneId(timezone);
        return personDTO;
    }
