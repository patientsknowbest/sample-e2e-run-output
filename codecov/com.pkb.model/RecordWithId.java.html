package com.pkb.model;

import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import java.io.Serializable;

/**
 * Converted from an interface as this currently needs to be serializable.
 */
@Value.Immutable
public abstract class RecordWithId<T, D> implements Serializable {

    @Value.Parameter
    public abstract T id();

    @Value.Parameter
    public abstract D record();

    public <R> RecordWithId<T, R> switchRecord(@NotNull R input) {
        return ImmutableRecordWithId.of(id(), input);
    }

    public static <M, N> RecordWithId<M, N> of(@NotNull M id, @NotNull N record) {
        return ImmutableRecordWithId.of(id, record);
    }
}
