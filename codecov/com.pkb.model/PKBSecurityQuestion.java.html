/**
 * This class represents a security question; A list of security questions will
 * be asked to a user to back up the password
 *
 * @author pravinam
 */
public class PKBSecurityQuestion {

    private Long id;

    private String question;

    public PKBSecurityQuestion() {
    }

    public Long getId() {
        return id;
    }

    public String getQuestion() {
        return question;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public void setQuestion(String question) {
        this.question = question;
    }

}
