package com.pkb.integration.coredevices;

import com.pkb.datamodel.DataModelStyle;
import org.immutables.value.Value;

@Value.Immutable
@DataModelStyle
public interface Notification extends Comparable<Notification> {

    String getCoreDevicesUserId();
    String getCategory();
    String getSource();
    String getOriginalJson();

    @Override
    default int compareTo(Notification n) {
        int cf = getCoreDevicesUserId().compareTo(n.getCoreDevicesUserId());
        if (cf != 0) {
            return cf;
        }

        int cs = getCategory().compareTo(n.getCategory());
        if (cs != 0) {
            return cs;
        }

        return getSource().compareTo(n.getSource());
    }

}
