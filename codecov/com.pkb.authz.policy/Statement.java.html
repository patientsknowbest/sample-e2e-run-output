package com.pkb.authz.policy;

import com.pkb.authz.action.Action;
import com.pkb.authz.policy.condition.Condition;
import com.pkb.authz.policy.precondition.ActionMatchPrecondition;
import com.pkb.authz.policy.precondition.Precondition;
import io.vavr.collection.List;
import io.vavr.collection.Set;
import org.immutables.value.Value.Check;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;
import org.immutables.vavr.encodings.VavrEncodingEnabled;
import org.jetbrains.annotations.NotNull;

import static com.google.common.base.Preconditions.checkState;
import static com.pkb.authz.policy.DefaultPolicy.policy;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

/**
 * Part of the fluent API for defining policies. This class is the backing object for
 * {@link PolicyBuilder}
 */
@Immutable
@Style(passAnnotations = { NotNull.class })
@VavrEncodingEnabled
@SuppressWarnings("varargs")
public interface Statement {
    /**
     * Mandatory part of a policy defintion.
     * It has to be a non blank unique string that describes the policy.
     * Ideally, each and every description should be also unique to help explain
     * why a given action is allowed/denied.
     */
    @NotNull String description();

    /**
     * @return the conditions determining if this policy is applicable or not.
     * Usually will contain an action, user and object precondition
     */
    @NotNull List<Precondition> preconditions();

    /**
     * @return The conditions to be tested as part of the policy
     */
    @NotNull List<Condition> conditions();

    /**
     * The result of this policy. Usually {@link Effect#GRANT}. If there are no applicable policies allowing a
     * GRANT, then authorization is denied. If there are 2 policies that evaluate and one is GRANT and the other is
     * REVOKE, then authorization is denied.
     */
    @NotNull Effect effect();

    /**
     * You should normally obtain an instance of this via PolicyBuilder, which does not allow these conditions to occur.
     */
    @Check
    default void validate() {
        checkState(isNotBlank(description()), "Policy description cannot be blank");
        checkState(!actions().isEmpty(), "Policy must have at least one action precondition");
    }

    @NotNull
    default Policy definePolicy() {
        return policy(this);
    }

    @NotNull
    default Set<Action> actions() {
        return preconditions()
                .filter(ActionMatchPrecondition.class::isInstance)
                .map(ActionMatchPrecondition.class::cast)
                .map(ActionMatchPrecondition::getAction)
                .toSet();
    }
}
