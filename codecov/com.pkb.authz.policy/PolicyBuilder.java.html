package com.pkb.authz.policy;

import com.pkb.authentication.principal.user.AuthenticatedUser;
import com.pkb.authz.policy.condition.Condition;

import static com.pkb.authz.policy.PolicyBuilderAdditionalConditionsStage.createPolicyBuilderAdditionalConditionsStage;
import static com.pkb.authz.policy.PolicyBuilderFinalizer.createPolicyBuilderFinalizer;
import static com.pkb.authz.policy.precondition.Preconditions.objectTypePrecondition;
import static com.pkb.authz.policy.precondition.Preconditions.userTypePrecondition;

/**
 * Fluent API for building {@link Policy}s.
 * The resulting code should look a lot like the rules in the documentation, e.g:
 * <a href="https://pkbdev.atlassian.net/wiki/spaces/TECHNOLOGY/pages/136020093/Patient">Patient</a>
 */
public class PolicyBuilder {
    private ImmutableStatement.Builder builder = ImmutableStatement.builder();

    private PolicyBuilder() {}

    public static PolicyBuilder policyBuilder() {
        return new PolicyBuilder();
    }

    public PolicyBuilder describeAs(String description) {
        builder.description(description);
        return this;
    }

    public PolicyBuilder forObjectType(Class<?> objectType) {
        builder.addPreconditions(objectTypePrecondition(objectType));
        return this;
    }

    public PolicyBuilder forUserType(Class<? extends AuthenticatedUser> userType) {
        builder.addPreconditions(userTypePrecondition(userType));
        return this;
    }

    public PolicyBuilderAdditionalConditionsStage when(Condition condition) {
        builder.addConditions(condition);
        return createPolicyBuilderAdditionalConditionsStage(builder);
    }

    public PolicyBuilderFinalizer then() {
        return createPolicyBuilderFinalizer(builder);
    }
}
