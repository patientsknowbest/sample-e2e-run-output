package com.pkb.authz.policy;

import com.pkb.authz.condition.ConditionFailure;
import com.pkb.authz.permission.ImmutableTextualJustification;
import com.pkb.authz.permission.Justification;
import io.vavr.collection.List;
import io.vavr.collection.Set;
import io.vavr.control.Validation;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.JustificationUtils.merge;
import static com.pkb.authz.JustificationUtils.textual;
import static com.pkb.authz.policy.ImmutablePolicyCheckResult.policyCheckResult;
import static com.pkb.authz.policy.util.ConditionCheckResultMerger.conditionCheckResultMerger;
import static io.vavr.control.Either.right;

/**
 * A dynamically built policy.
 * Get one of these by using {@link PolicyBuilder} fluent API.
 */
public final class DefaultPolicy implements Policy {
    private final Statement statement;

    private DefaultPolicy(@NotNull Statement statement) {
        this.statement = statement;
    }

    @Override
    public @NotNull String description() {
        return statement.description();
    }

    @Override
    public boolean isApplicable(Facts facts) {
        return statement.preconditions()
                .forAll(precondition -> precondition.test(facts));
    }

    @Override
    public @NotNull Validation<Set<ConditionFailure>, PolicyCheckResult> evaluate(@NotNull Facts facts) {
        return statement.conditions()
                .map(condition -> condition.test(facts))
                .reduceOption(conditionCheckResultMerger())
                .getOrElse(right(List.of(ImmutableTextualJustification.justification("Policy has no conditions"))))
                .map(justifications -> (PolicyCheckResult)policyCheckResult()
                        .effect(statement.effect())
                        .actions(statement.actions())
                        .justifications(merge(policyAllows(), justifications))
                        .build())
                .toValidation();
    }

    private List<Justification> policyAllows() {
        return List.of(textual(statement.description()));
    }

    public static Policy policy(Statement definition) {
        return new DefaultPolicy(definition);
    }

    @Override
    public String toString() {
        return description();
    }
}
