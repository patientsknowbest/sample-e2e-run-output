package com.pkb.action.dto;

import java.util.Map;

import com.pkb.notification.entity.Activity;
import com.pkb.notification.entity.ActivityProperty;

public final class ActivityDTOBuilder {
    private Activity activity;
    private String dateInText;
    private Map<ActivityProperty, Object> actionProperties;

    private ActivityDTOBuilder() {
    }

    public static ActivityDTOBuilder anActivityDTO() {
        return new ActivityDTOBuilder();
    }

    public ActivityDTOBuilder withActivity(Activity activity) {
        this.activity = activity;
        return this;
    }

    public ActivityDTOBuilder withDateInText(String dateInText) {
        this.dateInText = dateInText;
        return this;
    }

    public ActivityDTOBuilder withActionProperties(Map<ActivityProperty, Object> actionProperties) {
        this.actionProperties = actionProperties;
        return this;
    }

    public ActivityDTO build() {
        ActivityDTO activityDTO = new ActivityDTO();
        activityDTO.setActivity(activity);
        activityDTO.setDateInText(dateInText);
        activityDTO.setActionProperties(actionProperties);
        return activityDTO;
    }
}
