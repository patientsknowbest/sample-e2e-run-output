package com.pkb.action.dto;

import java.io.Serializable;
import java.util.Map;

import com.pkb.entities.enums.AppointmentStatus;
import com.pkb.notification.entity.Activity;
import com.pkb.notification.entity.ActivityProperty;

/**
 * DTO to carry display related information for an activity
 *
 * @author pravina
 *
 */
public class ActivityDTO implements Serializable {

    private Activity activity;

    private String dateInText;

    private Map<ActivityProperty, Object> actionProperties;

    public Activity getActivity() {
        return activity;
    }

    public void setActivity(Activity activity) {
        this.activity = activity;
    }

    public String getDateInText() {
        return dateInText;
    }

    public void setDateInText(String dateInText) {
        this.dateInText = dateInText;
    }

    public Map<ActivityProperty, Object> getActionProperties() {
        return actionProperties;
    }

    public void setActionProperties(Map<ActivityProperty, Object> actionProperties) {
        this.actionProperties = actionProperties;
    }

    public Object getActionProperty(ActivityProperty property){
        return this.actionProperties.get(property);
    }

    public void setActionProperty(ActivityProperty property, Object value){
        this.actionProperties.put(property, value);
    }

    public boolean isLabTestResult(){
        return this.activity.getAction() == Activity.Action.UPLOADED_LAB_RESULTS;
    }

    public boolean isAppointment(){
        return this.activity.getAction() == Activity.Action.INVITED_TO_APPOINTMENT;
    }

    public boolean isMessage(){
        return this.activity.getAction() == Activity.Action.SENT_MESSAGE;
    }

    public boolean isCancelledAppointment() { return this.activity.getAction() == Activity.Action.INVITED_TO_APPOINTMENT &&
            this.getActionProperty(ActivityProperty.STATUS) == AppointmentStatus.CANCELLED; }

    public boolean isDidNotAttendAppointment() {return this.activity.getAction() == Activity.Action.INVITED_TO_APPOINTMENT &&
            this.getActionProperty(ActivityProperty.STATUS) == AppointmentStatus.DNA; }
}
