package com.pkb.fhir.analysis.configuration;

import com.pkb.fhir.analysis.persistence.targetdb.repository.DecryptedEhrDataRepository;
import com.pkb.fhir.analysis.process.check.AlreadyDecryptedDataChecker;
import com.pkb.fhir.analysis.process.collect.EncryptedEhrDataCollector;
import com.pkb.fhir.analysis.process.perform.AnalysisEncryptedEhrDataDecryptor;
import com.pkb.fhir.analysis.storage.EhrDataStorage;
import com.pkb.fhir.common.decrypt.RetryingEhrDataDecryptor;
import com.pkb.fhir.common.repository.MenuDataRepository;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Primary;

@Configuration
public class AnalysisDecryptorConfig {

    @Bean
    @Primary
    public EncryptedEhrDataCollector encryptedEhrDataCollector(MenuDataRepository menuDataRepository) {
        return new EncryptedEhrDataCollector(menuDataRepository);
    }

    @Bean
    public EhrDataStorage ehrDataStorage(DecryptedEhrDataRepository decryptedEhrDataRepository) {
        return new EhrDataStorage(decryptedEhrDataRepository);
    }

    @Bean
    public AnalysisEncryptedEhrDataDecryptor encryptedEhrDataDecryptor(MenuDataRepository menuDataRepository,
                                                                       RetryingEhrDataDecryptor ehrDataDecryptor,
                                                                       EhrDataStorage ehrDataStorage) {
        return new AnalysisEncryptedEhrDataDecryptor(menuDataRepository, ehrDataDecryptor, ehrDataStorage);
    }

    @Bean
    public AlreadyDecryptedDataChecker alreadyDecryptedDataChecker(DecryptedEhrDataRepository decryptedEhrDataRepository) {
        return new AlreadyDecryptedDataChecker(decryptedEhrDataRepository);
    }
}
