package com.pkb.fhir.analysis.configuration;

import com.pkb.fhir.analysis.core.threading.EhrDataMigrationBootstrapper;
import com.pkb.fhir.analysis.core.threading.EhrDataProcessingWorkerFactory;
import com.pkb.fhir.analysis.core.threading.ExecutorServiceFactory;
import com.pkb.fhir.analysis.process.check.AlreadyDecryptedDataChecker;
import com.pkb.fhir.analysis.process.check.AlreadyTransformedDataChecker;
import com.pkb.fhir.analysis.process.check.AlreadyValidatedDataChecker;
import com.pkb.fhir.analysis.process.collect.DecryptedEhrDataCollector;
import com.pkb.fhir.analysis.process.collect.EncryptedEhrDataCollector;
import com.pkb.fhir.analysis.process.collect.TransformedEhrDataCollector;
import com.pkb.fhir.analysis.process.domain.ProcessorEnum;
import com.pkb.fhir.analysis.process.perform.AnalysisDecryptedEhrDataTransformer;
import com.pkb.fhir.analysis.process.perform.AnalysisEncryptedEhrDataDecryptor;
import com.pkb.fhir.analysis.process.perform.TransformedEhrDataValidator;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.EnumMap;
import java.util.Map;

@Configuration
public class AnalysisConfig {

    @Bean
    public Map<ProcessorEnum, EhrDataProcessingWorkerFactory> ehrDataProcessorWorkerFactoryMap(
            // collector
            EncryptedEhrDataCollector encryptedSourceEhrDataCollector,
            DecryptedEhrDataCollector decryptedSourceEhrDataCollector,
            TransformedEhrDataCollector transformedSourceEhrDataCollector,
            // processor
            AnalysisEncryptedEhrDataDecryptor encryptedEhrDataDecryptor,
            AnalysisDecryptedEhrDataTransformer analysisDecryptedEhrDataTransformer,
            TransformedEhrDataValidator transformedEhrDataValidator,
            // checker
            AlreadyDecryptedDataChecker alreadyDecryptedDataChecker,
            AlreadyTransformedDataChecker alreadyTransformedDataChecker,
            AlreadyValidatedDataChecker alreadyValidatedDataChecker
    ) {
        EnumMap<ProcessorEnum, EhrDataProcessingWorkerFactory> result = new EnumMap<>(ProcessorEnum.class);
        result.put(ProcessorEnum.DECRYPT, new EhrDataProcessingWorkerFactory(
                encryptedSourceEhrDataCollector,
                encryptedEhrDataDecryptor,
                alreadyDecryptedDataChecker
        ));
        result.put(ProcessorEnum.TRANSFORM, new EhrDataProcessingWorkerFactory(
                decryptedSourceEhrDataCollector,
                analysisDecryptedEhrDataTransformer,
                alreadyTransformedDataChecker
        ));
        result.put(ProcessorEnum.VALIDATE, new EhrDataProcessingWorkerFactory(
                transformedSourceEhrDataCollector,
                transformedEhrDataValidator,
                alreadyValidatedDataChecker
        ));
        return result;
    }

    @Bean
    public ExecutorServiceFactory executorServiceFactory() {
        return new ExecutorServiceFactory();
    }

    @Bean
    public EhrDataMigrationBootstrapper ehrDataMigrationBootstrapper(ExecutorServiceFactory executorServiceFactory,
                                                                     Map<ProcessorEnum, EhrDataProcessingWorkerFactory> ehrDataProcessorWorkerFactoryMap) {
        return new EhrDataMigrationBootstrapper(executorServiceFactory, ehrDataProcessorWorkerFactoryMap);
    }
}
