package com.pkb.kms.service.mapper;

import com.pkb.crypto.EncryptedSymmetricKey;
import com.pkb.crypto.SymmetricKey;
import com.pkb.crypto.dto.AccountSymmetricKeyDTO;
import com.pkb.entities.core.CoreAccount;
import com.pkb.entities.core.CoreAccountSymmetricKey;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;

import java.util.Base64;

import static com.pkb.crypto.dto.AccountSymmetricKeyDTO.accountSymmetricKeyDTO;

@Mapper
public abstract class SymmetricKeyMapper {

    @Mapping(target = "id", ignore = true)
    @Mapping(target = "encryptedSymmetricKeyBase64", source = "symmetricKey.encryptedKeyBytes")
    @Mapping(target = "symmetricKeyAlgorithm", source = "symmetricKey.algorithm")
    public abstract CoreAccountSymmetricKey mapNewEntity(EncryptedSymmetricKey symmetricKey, CoreAccount account);

    public AccountSymmetricKeyDTO map(SymmetricKey symmetricKey, CoreAccountSymmetricKey entity) {
        return accountSymmetricKeyDTO(entity.getId(), symmetricKey);
    }

    String mapBytesToBase64(byte[] value) {
        return Base64.getEncoder().encodeToString(value);
    }
}
