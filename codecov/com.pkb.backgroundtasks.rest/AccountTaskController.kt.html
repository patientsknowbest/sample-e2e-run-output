package com.pkb.backgroundtasks.rest

import com.pkb.backgroundtasks.AccountPublicId
import com.pkb.backgroundtasks.entity.AccountTaskDetails
import com.pkb.backgroundtasks.projection.TaskStatusContainer
import com.pkb.backgroundtasks.service.AccountTaskService
import org.springframework.web.bind.annotation.DeleteMapping
import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.PutMapping
import org.springframework.web.bind.annotation.RequestBody
import org.springframework.web.bind.annotation.RestController
import java.util.UUID

/**
 * This controller is intended for internal use for other parts of the stack needing
 * to enqueue account operations that require a key from the KMS that may not be available
 * in the calling context; and also to check on the status of those tasks; and to
 * run any outstanding tasks for a particular account synchronously
 */
@RestController
class AccountTaskController(private val accountTaskService: AccountTaskService) {

    /**
     * Run any outstanding  tasks for a given account immediately. Returns the status of any tasks
     * that were run.
     */
    @PostMapping("/accountTask/{accountPublicId}")
    fun runTasks(@PathVariable accountPublicId: AccountPublicId): TaskStatusContainer =
        accountTaskService.triggerTasks(accountPublicId).let { TaskStatusContainer(accountPublicId, it) }

    /**
     * Triggers a pass through all outstanding tasks if one is not already running.
     * Async operation, returns immediately.
     *
     * Generally not that useful in production, as there's already a scheduled task that does this every 5 minutes;
     * any component in production that wants to force immediate processing of account tasks is almost certainly
     * better off calling the synchornous account-specific method above.
     */
    @PostMapping("/accountTask")
    fun triggerAllTasks() =
        accountTaskService.triggerAllPendingTasks()

    /**
     * Enqueues an operation to be performed on an account using keys available in the KMS. The most recently
     * enqueued operations are performed first, triggered every 5 minutes.  If it's critical that a task
     * should be performed ASAP, clients should call the runTasks endpoint above for the account to force
     * it to happen immediately.
     */
    @PutMapping("/accountTask/{accountPublicId}")
    fun <T : AccountTaskDetails<T>> enqueueTask(
        @PathVariable accountPublicId: AccountPublicId,
        @RequestBody accountTaskDetails: T
    ): UUID =
        accountTaskService.enqueueTask(accountPublicId, accountTaskDetails)

    /**
     * Returns the status of all tasks that have ever been performed for a specific account
     * in order of esecution
     */
    @GetMapping("/accountTask/{accountPublicId}")
    fun getTaskStatus(@PathVariable accountPublicId: AccountPublicId): TaskStatusContainer =
        accountTaskService.getTaskStatuses(accountPublicId)
            .let { TaskStatusContainer(accountPublicId, it) }

    @GetMapping("/test/accountTask")
    fun getAllTaskStatuses() =
        accountTaskService.TEST_ONLY_getAllTaskStatuses()

    /**
     * TESTING ONLY: Do not use in production. Deletes *all* tasks for every account.
     */
    @DeleteMapping("/test/accountTask")
    fun deleteAllTasks() =
        accountTaskService.TEST_ONLY_deleteAllTasks()

    /**
     * TESTING ONLY Deletes all tasks (including those already executed) for a
     * given account.
     */
    @DeleteMapping("/test/accountTask/{accountPublicId}")
    fun deleteAllTasks(@PathVariable accountPublicId: AccountPublicId) =
        accountTaskService.TEST_ONLY_deleteTasksFor(accountPublicId)
}
