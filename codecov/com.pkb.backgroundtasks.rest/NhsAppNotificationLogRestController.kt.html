package com.pkb.backgroundtasks.rest

import com.pkb.backgroundtasks.service.NhsAppNotificationLogService
import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RestController

@RestController
@ConditionalOnProperty(name = ["backgroundtasks.nhs-notification-retry.enabled"], havingValue = "true")
class NhsAppNotificationLogRestController(private val nhsAppNotificationLogService: NhsAppNotificationLogService) {

    @RequestMapping("/truncateNHSAppNotificationLog")
    fun truncateNHSAppNotificationLog() {
        nhsAppNotificationLogService.truncateLog()
    }
}
