package com.pkb.entities.nhs;

import com.pkb.entities.enums.NationalIdType;
import com.pkb.entities.enums.ValidNationalId;
import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import com.vladmihalcea.hibernate.type.json.JsonBinaryType;
import org.hibernate.annotations.DynamicInsert;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Type;
import org.hibernate.annotations.TypeDef;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.Instant;

import static com.pkb.entities.enums.NationalIdType.NHS_NUMBER;
import static java.lang.String.format;

@Entity
@Table(schema = "nhs", name = "nhs_notification_log")
@TypeDef(
        name = "jsonb",
        typeClass = JsonBinaryType.class
)
@DynamicInsert
public class PublicNhsAppNotificationLog {

    @Id
    @GenericGenerator(name = "nhs_notification_log_id_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "nhs"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "nhs_notification_log_id_seq"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "nhs_notification_log_id_gen")
    private long id;

    @Column(name = "target_nhs_number", nullable = false)
    private String targetNhsNumber;

    @Column(name = "message", nullable=false)
    private String message;

    @Column(name = "request_id", nullable=false)
    private String requestId;

    @Column(name = "created", nullable=false)
    private Instant created;

    @Column(name = "notification_trigger", nullable = false)
    private String notificationTrigger;

    @Column(name = "success", nullable=false)
    private boolean success;

    @Column(name = "processed", nullable=false)
    private Instant processed;

    @Type(type = "jsonb")
    @Column(name="error_message", columnDefinition="jsonb")
    private String errorMessage;

    public long getId() {
        return id;
    }

    public ValidNationalId getTargetNhsNumber() {
        return NationalIdType.NHS_NUMBER.getValidNationalIdAndType(targetNhsNumber)
                .orElseThrow(() -> new RuntimeException(format("Invalid NHS number '%s' in nhs_notification_log",targetNhsNumber)));
    }

    public void setTargetNhsNumber(ValidNationalId targetNhsNumber) {
        this.targetNhsNumber = NHS_NUMBER.getValidNationalIdAndType(targetNhsNumber.getValue())
                .map(ValidNationalId::getValue)
                .orElseThrow(() -> new IllegalArgumentException(format("Logging '%s' to nhs_notification_log", targetNhsNumber.getType().name())));
    }

    public String getMessage() {
        return message;
    }

    public void setMessage(String message) {
        this.message = message;
    }

    public String getRequestId() { return requestId; }

    public void setRequestId(String requestId) { this.requestId = requestId; }

    public Instant getCreated() {
        return created;
    }

    public void setCreated(Instant created) {
        this.created = created;
    }

    public String getNotificationTrigger() {
        return notificationTrigger;
    }

    public void setNotificationTrigger(String notificationTrigger) {
        this.notificationTrigger = notificationTrigger;
    }

    public boolean isSuccess() {
        return success;
    }

    public void setSuccess(boolean success) {
        this.success = success;
    }

    public Instant getProcessed() {
        return processed;
    }

    public void setProcessed(Instant processed) {
        this.processed = processed;
    }

    public String getErrorMessage() {
        return errorMessage;
    }

    public void setErrorMessage(String errorMessage) {
        this.errorMessage = errorMessage;
    }
}
