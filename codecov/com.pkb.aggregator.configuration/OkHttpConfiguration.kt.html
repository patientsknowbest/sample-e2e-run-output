package com.pkb.aggregator.configuration

import io.micrometer.core.instrument.Metrics
import io.micrometer.core.instrument.binder.okhttp3.OkHttpMetricsEventListener
import okhttp3.Credentials
import okhttp3.OkHttpClient
import okhttp3.logging.HttpLoggingInterceptor
import org.keycloak.admin.client.Keycloak
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import java.net.URI

@Configuration
class OkHttpConfiguration {

    @Bean
    fun okhttp(
        metricsEventListener: OkHttpMetricsEventListener,
        properties: OkHttpProperties,
        keycloak: Keycloak?,
        fhirConfiguration: FhirConfiguration,
    ): OkHttpClient {

        val builder = OkHttpClient.Builder()
            .eventListener(metricsEventListener)
            .connectTimeout(properties.connectTimeout)
            .readTimeout(properties.readTimeout)
            .writeTimeout(properties.writeTimeout)
            .retryOnConnectionFailure(properties.retryOnConnectionFailure)

        // Add authentication interceptor
        builder.addInterceptor {
            val authcHeader = keycloak?.let { "Bearer ${it.tokenManager().accessTokenString}" }
                ?: basicCredentials(fhirConfiguration, it.request().url.toUri())

            it.proceed(
                it.request()
                    .newBuilder()
                    .header("Authorization", authcHeader)
                    .build()
            )
        }

        // Add logging interceptor
        if (fhirConfiguration.downstream.endpoint.printTrace) {
            builder.addInterceptor(HttpLoggingInterceptor().apply {
                level = HttpLoggingInterceptor.Level.BODY
            })
        }

        return builder.build()
    }

    private fun basicCredentials(fhirConfiguration: FhirConfiguration, serverUri: URI): String {
        val endpoint = fhirConfiguration.findEndpointByUrl(serverUri)
        return Credentials.basic(endpoint.username!!, endpoint.password!!)
    }

    @Bean
    fun okhttpMetricsEventListener() =
        OkHttpMetricsEventListener.builder(Metrics.globalRegistry, "okhttp.requests").build()
}
