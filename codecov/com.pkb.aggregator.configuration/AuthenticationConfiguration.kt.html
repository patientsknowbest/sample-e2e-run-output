package com.pkb.aggregator.configuration

import org.keycloak.OAuth2Constants.CLIENT_CREDENTIALS
import org.keycloak.admin.client.Keycloak
import org.keycloak.admin.client.KeycloakBuilder
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration

@Configuration
class AuthenticationConfiguration {

    @Bean
    fun keycloakClient(keycloakConfiguration: KeycloakConfiguration): Keycloak? =
        if (keycloakConfiguration.enableBasicAuth) {
            null
        } else {
            KeycloakBuilder.builder()
                .serverUrl(keycloakConfiguration.authServerUrl + "/auth")
                .realm(keycloakConfiguration.realm)
                .grantType(CLIENT_CREDENTIALS)
                .clientId(keycloakConfiguration.resource)
                .clientSecret(keycloakConfiguration.credentials!!.secret)
                .build()
        }
}
