package com.pkb.aggregator.configuration

import org.springframework.boot.context.properties.ConfigurationProperties
import org.springframework.cloud.context.config.annotation.RefreshScope
import org.springframework.context.annotation.Configuration
import java.net.URI

@ConfigurationProperties("fhir")
@Configuration
@RefreshScope
class FhirConfiguration {
    var upstream: FhirUpstreamConfiguration = FhirUpstreamConfiguration()
    var downstream: FhirDownstreamConfiguration = FhirDownstreamConfiguration()

    fun findEndpointByUrl(serverUri: URI) = upstream.endpoints.plus(downstream.endpoint).find {
        URI(it.serverUrl).authority == serverUri.authority
    } ?: throw RuntimeException("Cannot find endpoint configuration for $serverUri")
}

class FhirUpstreamConfiguration {
    lateinit var endpoints: List<FhirEndpoint>
    var pageSize: Int = 500
}

class FhirDownstreamConfiguration {
    var endpoint: FhirEndpoint = FhirEndpoint()
}

class FhirEndpoint {
    var username: String? = null
    var password: String? = null
    var printTrace: Boolean = false

    lateinit var serverUrl: String
    lateinit var fhirVersion: String
    lateinit var sourceId: String
}
