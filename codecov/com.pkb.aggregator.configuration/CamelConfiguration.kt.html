package com.pkb.aggregator.configuration

import com.pkb.aggregator.camel.FhirSafeMicrometerExchangeEventNotifierNamingStrategy
import io.micrometer.core.instrument.MeterRegistry
import org.apache.camel.CamelContext
import org.apache.camel.component.micrometer.eventnotifier.MicrometerExchangeEventNotifier
import org.apache.camel.component.micrometer.eventnotifier.MicrometerRouteEventNotifier
import org.apache.camel.component.micrometer.messagehistory.MicrometerMessageHistoryFactory
import org.apache.camel.component.micrometer.routepolicy.MicrometerRoutePolicyFactory
import org.apache.camel.component.micrometer.spi.InstrumentedThreadPoolFactory
import org.apache.camel.spring.boot.CamelContextConfiguration
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration


@Configuration
class CamelConfiguration {

    @Bean
    fun instrumentedThreadPoolFactory(registry: MeterRegistry) = InstrumentedThreadPoolFactory(registry)

    @Bean
    fun micrometerRoutePolicyFactory(registry: MeterRegistry) = MicrometerRoutePolicyFactory().apply {
        meterRegistry = registry
    }

    @Bean
    fun micrometerMessageHistoryFactory(registry: MeterRegistry) = MicrometerMessageHistoryFactory().apply {
        meterRegistry = registry
    }

    @Bean
    fun micrometerExchangeEventNotifier(strategy: FhirSafeMicrometerExchangeEventNotifierNamingStrategy) =
        MicrometerExchangeEventNotifier().apply {
            meterRegistry = meterRegistry
            namingStrategy = strategy
        }

    @Bean
    fun micrometerRouteEventNotifier() =
        MicrometerRouteEventNotifier().apply {
            meterRegistry = meterRegistry
        }

    @Bean
    fun contextConfiguration(
        micrometerExchangeEventNotifier: MicrometerExchangeEventNotifier,
        micrometerRouteEventNotifier: MicrometerRouteEventNotifier,
    ): CamelContextConfiguration {
        return object : CamelContextConfiguration {
            override fun beforeApplicationStart(context: CamelContext) {
                context.managementStrategy.addEventNotifier(micrometerExchangeEventNotifier)
                context.managementStrategy.addEventNotifier(micrometerRouteEventNotifier)
            }

            override fun afterApplicationStart(camelContext: CamelContext) {}
        }
    }
}
