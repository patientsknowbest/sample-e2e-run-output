package com.pkb.questionnaireservice.mapper

import com.pkb.questionnaireservice.modelv2.BaseModel
import com.pkb.questionnaireservice.modelv2.QuestionnaireServiceError
import org.springframework.dao.DataIntegrityViolationException

fun <E: BaseModel> mapRepositoryErrorOnSave(error: Throwable, objectFailedSave: E): QuestionnaireServiceError =
        when (error) {
            // I think this should be DuplicateKeyException, but for some reason that is _not_ thrown
            //  in the combination of postgres + hibernate + spring data JPA that we're using.
            is DataIntegrityViolationException -> QuestionnaireServiceError.ConflictError("${objectFailedSave::class.java.simpleName} with id ${objectFailedSave.id} already exists!")
            else -> QuestionnaireServiceError.UnexpectedError(error.message!!)
        }
