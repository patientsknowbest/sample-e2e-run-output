package com.pkb.questionnaireservice.mapper

import com.pkb.questionnaireservice.modelv2.QuestionnaireServiceError

fun com.pkb.kms.shared.representation.KmsError.mapKmsError(): QuestionnaireServiceError{
    return when (this.type) {
        com.pkb.kms.shared.representation.KmsError.Type.NOT_AUTHORISED -> QuestionnaireServiceError.NotAuthorizedError(this.description)
        else -> QuestionnaireServiceError.UnexpectedError(this.description)
    }
}
