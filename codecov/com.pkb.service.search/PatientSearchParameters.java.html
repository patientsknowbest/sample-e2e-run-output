package com.pkb.service.search;

import com.google.common.base.MoreObjects;
import com.google.common.collect.ImmutableMap;
import com.pkb.dto.PatientSearchParametersDTO;
import org.apache.commons.lang3.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Arrays;
import java.util.Map;
import java.util.Optional;
import java.util.function.Function;

import static java.util.stream.Collectors.toMap;

/**
 *
 * @author Russ Gray
 */
public class PatientSearchParameters {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    public enum SearchIdentifierType {

        NATIONAL_ID("national"),
        TEAM_LEVEL_ID("team"),
        ORG_LEVEL_ID("org");

        private final String text;

        // Idea from Effective Java 2nd Ed, Item 30
        private static final Map<String, SearchIdentifierType> ENUM_MAP;

        static {
            ENUM_MAP = ImmutableMap.copyOf(
                    Arrays.stream(SearchIdentifierType.values())
                    .collect(toMap(
                                    SearchIdentifierType::toString,
                                    Function.identity())));
        }

        SearchIdentifierType(String text) {
            this.text = text;
        }

        @Override
        public String toString() {
            return text;
        }

        public String getText() {
            return text;
        }

        public static Optional<SearchIdentifierType> fromString(String value) {
            return Optional.of(ENUM_MAP.get(value));
        }

        public static boolean contains(String value) {
            return ENUM_MAP.keySet().contains(value);
        }
    }

    private final SearchIdentifierType identifierType;
    private final String identifier;

    private final String name;
    private final Integer dateOfBirthYear;
    private final Integer dateOfBirthMonth;
    private final Integer dateOfBirthDay;

    public SearchIdentifierType getIdentifierType() {
        return identifierType;
    }

    public String getIdentifier() {
        return identifier;
    }

    public String getName() {
        return name;
    }

    public Integer getDateOfBirthYear() {
        return dateOfBirthYear;
    }

    public Integer getDateOfBirthMonth() {
        return dateOfBirthMonth;
    }

    public Integer getDateOfBirthDay() {
        return dateOfBirthDay;
    }

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .omitNullValues()
                .add("identifierType", identifierType)
                .add("identifier", identifier)
                .add("name", name)
                .add("dateOfBirthYear", dateOfBirthYear)
                .add("dateOfBirthMonth", dateOfBirthMonth)
                .add("dateOfBirthDay", dateOfBirthDay)
                .toString();
    }

    public static class Builder {

        private SearchIdentifierType identifierType;
        private String identifier;
        private String name;
        private Integer dateOfBirthYear;
        private Integer dateOfBirthMonth;
        private Integer dateOfBirthDay;

        private Builder() {
        }

        public Builder identifierType(SearchIdentifierType value) {
            this.identifierType = value;
            return this;
        }

        public Builder identifierTypeString(String value) {
            return identifierType(SearchIdentifierType.fromString(value).orElseGet(() -> {
                SearchIdentifierType defaultValue = SearchIdentifierType.NATIONAL_ID;
                LOGGER.warn("Unknown search type {}, defaulting to {}", value, defaultValue);
                return defaultValue;
            }));
        }

        public Builder identifier(String value) {
            this.identifier = StringUtils.defaultIfEmpty(value, null);
            return this;
        }

        public Builder name(String value) {
            this.name = StringUtils.defaultIfEmpty(value, null);
            return this;
        }

        public Builder dateOfBirthYear(Integer value) {
            this.dateOfBirthYear = value;
            return this;
        }

        public Builder dateOfBirthMonth(Integer value) {
            this.dateOfBirthMonth = value;
            return this;
        }

        public Builder dateOfBirthDay(Integer value) {
            this.dateOfBirthDay = value;
            return this;
        }

        public Builder convertFrom(PatientSearchParametersDTO dto) {
            return this
                    .identifierTypeString(dto.getSearchIdType())
                    .identifier(dto.getSearchId())
                    .name(dto.getSearchName())
                    .dateOfBirthYear(StringUtils.isBlank(dto.getSearchDobYear()) ? null : Integer.parseInt(dto.getSearchDobYear()))
                    .dateOfBirthMonth(StringUtils.isBlank(dto.getSearchDobMonth()) ? null : Integer.parseInt(dto.getSearchDobMonth()))
                    .dateOfBirthDay(StringUtils.isBlank(dto.getSearchDobDay()) ? null : Integer.parseInt(dto.getSearchDobDay()));
        }

        public PatientSearchParameters build() {
            return new com.pkb.service.search.PatientSearchParameters(
                    identifierType,
                    identifier,
                    name,
                    dateOfBirthYear,
                    dateOfBirthMonth,
                    dateOfBirthDay);
        }
    }

    public static PatientSearchParameters.Builder builder() {
        return new PatientSearchParameters.Builder();
    }

    private PatientSearchParameters(
            SearchIdentifierType identifierType,
            String identifier,
            String name,
            Integer dateOfBirthYear,
            Integer dateOfBirthMonth,
            Integer dateOfBirthDay) {
        this.identifierType = identifierType;
        this.identifier = identifier;
        this.name = name;
        this.dateOfBirthYear = dateOfBirthYear;
        this.dateOfBirthMonth = dateOfBirthMonth;
        this.dateOfBirthDay = dateOfBirthDay;
    }

}
