package com.pkb.common.config;

/**
 * In production environments we want to redact PID as much as possible from logs and other tools that we use to diagnose and/or operate our services.
 * While in other cases we need more info (e.g.: in test environments or when debugging something on prod).
 * This interface is meant to be expanded and add more and more options.
 */
public interface SensitiveInformationSupport {
    enum EmailSensitivityLevel {
        /** Signals that emails should not be used as plain values in logs, metrics, etc... */
        OBFUSCATE,
        /** Allows emails to be used as plain text anywhere. */
        DISCLOSE
    }

    EmailSensitivityLevel emailSensitivityLevel();
}
