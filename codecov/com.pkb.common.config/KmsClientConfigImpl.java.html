package com.pkb.common.config;

import com.pkb.kms.client.core.KmsClientConfig;
import org.jetbrains.annotations.NotNull;

public class KmsClientConfigImpl implements KmsClientConfig {

    private ConfigStorage storage;

    public KmsClientConfigImpl(ConfigStorage storage) {
        this.storage = storage;
    }

    @NotNull
    @Override
    public String getClientId() {
        return getConfigStorage().getString("kms.clientId", "phr");
    }

    @Override
    public ConfigStorage getConfigStorage() {
        return storage;
    }
}
