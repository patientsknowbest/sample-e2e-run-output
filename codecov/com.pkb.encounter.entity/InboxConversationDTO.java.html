package com.pkb.encounter.entity;

import com.pkb.api.internal.messaging.MessagingPKBPersonDTO;
import com.pkb.encounter.workflow.ConversationWorkflowDTO;
import com.pkb.encounter.workflow.MessageWorkflowSupport;
import com.pkb.entities.enums.EncounterClass;

import javax.annotation.Nullable;
import java.io.Serializable;
import java.util.List;
import java.util.UUID;

/**
 * Holds the conversation data needed to display the inbox.
 * TODO: merge with other Message/Conversation DTOs (Message, PKBMessageDTO, Conversation)
 *
 * @author robwhelan
 */
public class InboxConversationDTO implements MessageWorkflowSupport, Serializable {
    private static final long serialVersionUID = 1L;

    private Long messageCount;

    private UUID conversationId;

    private MessagingPKBPersonDTO patient;

    private MessagingPKBPersonDTO latestSender;

    private Message latestMessage;

    //TODO: remove it when migration is stable
    private List<String> participants; // only for migration testing purpose

    private EncounterClass encounterClass;

    private MessagingPKBPersonDTO originalSender;

    private Message originalMessage;

    private MessageSenderDTO latestMessageSenderDetails;

    private MessageSenderDTO originalMessageSenderDetails;

    private String specialty;

    private boolean hidePrivacyLabels;

    @Nullable
    private ConversationWorkflowDTO conversationWorkflow;

    @Override
    public String toString() {
        StringBuilder buffer = new StringBuilder();
        buffer.append(" Conversation Id: " + conversationId);
        buffer.append(", MessageCount: " + messageCount);
        buffer.append(", Encounter Class: " + encounterClass);
        buffer.append(", Latest Sender: [" + latestSender + "]");
        buffer.append(", Latest Message: [" + latestMessage + "]");
        buffer.append(", Latest Message Sender DTO: [" + latestMessageSenderDetails + "]");
        buffer.append(", Original Sender: [" + originalSender + "]");
        buffer.append(", Original Message: [" + originalMessage + "]");
        buffer.append(", Original Message Sender DTO: [" + originalMessageSenderDetails + "]");
        buffer.append(", Conversation Workflow DTO: [" + conversationWorkflow + "]");
        buffer.append("\n");
        return buffer.toString();
    }

    public Message getLatestMessage() {
        return latestMessage;
    }

    public void setLatestMessage(Message latestMessage) {
        this.latestMessage = latestMessage;
    }

    public Long getMessageCount() {
        return messageCount;
    }

    public void setMessageCount(Long messageCount) {
        this.messageCount = messageCount;
    }

    public UUID getConversationId() {
        return conversationId;
    }

    public void setConversationId(UUID conversationId) {
        this.conversationId = conversationId;
    }

    public MessagingPKBPersonDTO getPatient() {
        return patient;
    }

    public void setPatient(MessagingPKBPersonDTO patient) {
        this.patient = patient;
    }

    public MessagingPKBPersonDTO getLatestSender() {
        return latestSender;
    }

    public void setLatestSender(MessagingPKBPersonDTO latestSender) {
        this.latestSender = latestSender;
    }

    public List<String> getParticipants() {
        return participants;
    }

    public void setParticipants(List<String> participants) {
        this.participants = participants;
    }

    public EncounterClass getEncounterClass() {
        return encounterClass;
    }

    public void setEncounterClass(EncounterClass encounterClass) {
        this.encounterClass = encounterClass;
    }

    public MessagingPKBPersonDTO getOriginalSender() {
        return originalSender;
    }

    public void setOriginalSender(MessagingPKBPersonDTO originalSender) {
        this.originalSender = originalSender;
    }

    public Message getOriginalMessage() {
        return originalMessage;
    }

    public void setOriginalMessage(Message originalMessage) {
        this.originalMessage = originalMessage;
    }

    public MessageSenderDTO getLatestMessageSenderDetails() {
        return latestMessageSenderDetails;
    }

    public void setLatestMessageSenderDetails(MessageSenderDTO latestMessageSenderDetails) {
        this.latestMessageSenderDetails = latestMessageSenderDetails;
    }

    public MessageSenderDTO getOriginalMessageSenderDetails() {
        return originalMessageSenderDetails;
    }

    public void setOriginalMessageSenderDetails(MessageSenderDTO originalMessageSenderDetails) {
        this.originalMessageSenderDetails = originalMessageSenderDetails;
    }

    public String getSpecialty() {
        return specialty;
    }

    public void setSpecialty(String specialty) {
        this.specialty = specialty;
    }

    public boolean isHidePrivacyLabels() {
        return hidePrivacyLabels;
    }

    public void setHidePrivacyLabels(boolean hidePrivacyLabels) {
        this.hidePrivacyLabels = hidePrivacyLabels;
    }

    @Override
    public ConversationWorkflowDTO getConversationWorkflow() {
        return conversationWorkflow;
    }

    public void setConversationWorkflow(ConversationWorkflowDTO conversationWorkflow) {
        this.conversationWorkflow = conversationWorkflow;
    }
}
