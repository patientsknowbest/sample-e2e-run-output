package com.pkb.encounter.entity;

import com.pkb.api.internal.messaging.MessagingPKBPersonDTO;

import java.io.Serializable;

/**
 * Encapsulates the information that identifies the sender of a Message.
 *
 * <p>
 * This is a first implementation of this approach and is intended
 * to consolidate the details of the various sources of messages.
 * This object is populated from the various locations the source
 * information is located in - different depending on the sender of the message.
 * </p>
 *
 * TODO: peer review of validity of this approach to solving this problem.
 *
 * <p>
 * Possible originators of a Message in a conversation include:
 * </p>
 * <table border="1">
 * <tr>
 * <th>Sender</th>
 * <th>Info</th>
 * </tr>
 * <tr>
 * <td>Clinician</td>
 * <td><code>senderId</code> will be set but <code>source</code> instance will be null</td>
 * </tr>
 * <tr>
 * <td>Patient</td>
 * <td>Same as for clinician</td>
 * </tr>
 * <tr>
 * <td>HL7 Message</td>
 * <td><code>senderId</code> will be null but <code>source</code>
 * instance will contain either org or team value.</td>
 * </tr>
 * </table>
 *
 *
 * <p>
 * </p>
 *
 * @author johnspencepkb
 *
 */
public class MessageSenderDTO implements Serializable {

    /**
     *
     */
    private static final long serialVersionUID = 642079505333922030L;

    /**
     * The PKBPerson that sent this message, if in fact
     * it was a PKBPerson that sent the message. Will be blank for example
     * if message was created from receipt of HL7 message from an org or team.
     */
    private MessagingPKBPersonDTO sender;

    /**
     * The org ID the sender of this message belongs to.
     */
    private Long orgId;

    /**
     * Convenience field to make orgName available if we have an orgId.
     */
    private String orgName;

    /**
     * If the originator of this message was part of a team, then
     * the teamId will be set. If the teamId is set, then we will also
     * set the orgId as a team is always part of an Org but not necessarily
     * the reverse is true.
     */
    private Long teamId;

    /**
     * Convenience field to make teamName available if we have an teamId.
     */
    private String teamName;

    public MessagingPKBPersonDTO getSender() {
        return sender;
    }

    public void setSender(MessagingPKBPersonDTO sender) {
        this.sender = sender;
    }

    public Long getOrgId() {
        return orgId;
    }

    public void setOrgId(Long orgId) {
        this.orgId = orgId;
    }

    public String getOrgName() {
        return orgName;
    }

    public void setOrgName(String orgName) {
        this.orgName = orgName;
    }

    public Long getTeamId() {
        return teamId;
    }

    public void setTeamId(Long teamId) {
        this.teamId = teamId;
    }

    public String getTeamName() {
        return teamName;
    }

    public void setTeamName(String teamName) {
        this.teamName = teamName;
    }

    public boolean isRegClinician() {
        return (sender != null) && (sender.isPro());
    }

}
