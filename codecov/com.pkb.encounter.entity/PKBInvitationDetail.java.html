package com.pkb.encounter.entity;

import com.pkb.annotation.EHRField;
import com.pkb.annotation.EHRField.QueryField;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;
import com.pkb.app.interfaces.IBaseDTO;

import java.io.Serializable;
import java.util.UUID;

// We no longer store invitation details as xml within an encrypted string field
// Instead they are stored as a separate menudata item of datatype='invitationDetail''
//@XmlAccessorType(XmlAccessType.FIELD)
//@XmlType(name = "PKBInvitationDetail")

public class PKBInvitationDetail implements IBaseDTO, Serializable {

    private static final long serialVersionUID = 1L;

    public static final String USER_TO_ADD = "UTA=";

    public static final String PH_FIELD_INVITATION_ID = "invitationPrivateId";

    @EHRField(queryField = QueryField.id)
    protected Long id;

    @EHRField(queryField = QueryField.uuid01)
    protected UUID invitationPublicId;

    @SuppressWarnings("deprecation")
    @Deprecated // TODO remove in PHR-10242
    @EHRField(queryField = QueryField.long01)
    protected Long invitationPrivateId;

    @EHRField(queryField = QueryField.documentMetadataId)
    private UUID pwdAccessDocumentMetadataId;

    @EHRField
    protected String content;

    @EHRField
    protected String instituteCode;

    @EHRField
    protected String addUserId; // The user to be added amongst the inviter and invitee

    private final DTOBaseFields baseFields = new DTOBaseFields();
    private final SourceDetails source;
    private String teamName; // Kludge 6/1/2016, set in GetInvitationsAction.toDTO() for listInvitations.jsp

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected PKBInvitationDetail() {
        this.source = new SourceDetails();
    }

    public PKBInvitationDetail(SourceDetails sourceDetails) {
        this.source = sourceDetails;
    }

    @Override
    public Long getId() {
        return id;
    }

    @Override
    public void setId(Long id) {
        this.id = id;
    }

    public UUID getInvitationPublicId() {
        return this.invitationPublicId;
    }

    public void setInvitationPublicId(UUID value) {
        this.invitationPublicId = value;
    }

    public Long getInvitationPrivateId() {
        return this.invitationPrivateId;
    }

    public void setInvitationPrivateId(Long value) {
        this.invitationPrivateId = value;
    }

    public UUID getPwdAccessDocumentMetadataId() {
        return pwdAccessDocumentMetadataId;
    }

    public void setPwdAccessDocumentMetadataId(UUID pwdAccessDocumentMetadataId) {
        this.pwdAccessDocumentMetadataId = pwdAccessDocumentMetadataId;
    }

    public String getContent() {
        return content;
    }

    public void setContent(String content) {
        this.content = content;
    }

    public String getInstituteCode() {
        return instituteCode;
    }

    public void setInstituteCode(String instituteCode) {
        this.instituteCode = instituteCode;
    }

    public String getAddUserId() {
        return addUserId;
    }

    // TODO PHR-11035 - https://pkbdev.atlassian.net/browse/PHR-11035
    // INVITATION_TO_PATIENT and PATIENT_INVITATION_AS_CAREGIVER no longer need addUserId to distinguish between them
    public void setAddUserId(String addUserId) {
        this.addUserId = addUserId;
    }

    @Override
    public SourceDetails getSource() {
        return source;
    }

    @Override
    public DTOBaseFields getBaseFields() {
        return baseFields;
    }

    public String getTeamName() {
        return teamName;
    }

    public void setTeamName(String teamName) {
        this.teamName = teamName;
    }
}
