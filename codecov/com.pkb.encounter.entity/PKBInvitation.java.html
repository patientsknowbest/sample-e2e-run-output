package com.pkb.encounter.entity;

import com.pkb.datamodel.Email;
import com.pkb.entities.enums.InvitationStatus;
import com.pkb.entities.enums.InvitationType;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.user.entity.PKBPerson;
import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;

import javax.annotation.Nullable;
import java.io.Serializable;
import java.time.Instant;
import java.util.EnumSet;
import java.util.Optional;
import java.util.UUID;

/**
 * This class represents a invitation sent from a patient to a Clinician
 *
 * @author pravinam
 */
public class PKBInvitation implements Serializable {
    private static final long serialVersionUID = 1L;

    private Long id;

    private UUID publicId;

    private Long accountId;

    private Long authorUserId;

    private PKBPerson targetUser;

    private String invitationType;

    private Instant created;

    private InvitationStatus invitationStatus;

    private PKBInvitationDetail invitationDetail;

    private Email targetEmail;

    private boolean accountKeyRequired;
    private String accountKeyAccessToken;

    private EnumSet<PrivacyFlag> grantedConsents;

    /** side-effect! Cleans/trims (removing any space chars anywhere in the token) as part of making sure we have a usable token */
    public void setAccountKeyAccessToken(String accountKeyAccessToken) {
        this.accountKeyAccessToken = StringUtils.remove(accountKeyAccessToken, " ");
    }


    public PKBInvitationDetail getInvitationDetail() {
        return invitationDetail;
    }

    public void setInvitationDetail(PKBInvitationDetail invitationDetail) {
        this.invitationDetail = invitationDetail;
    }

    public PKBPerson getTargetUser() {
        return targetUser;
    }

    public Option<PKBPerson> findTargetUser() {
        return Option.of(targetUser);
    }

    public void setTargetUser(PKBPerson targetUser) {
        this.targetUser = targetUser;
    }

    public InvitationType getInvitationType() { return InvitationType.invitationConverter(invitationType); }

    public void setInvitationType(InvitationType invitationType) {this.invitationType = invitationType.description;}

    public void setInvitationType(String invitationType) {this.invitationType = invitationType; }

    public Instant getCreated() {
        return created;
    }

    public void setCreated(Instant created) {
        this.created = created;
    }

    public Long getAccountId() {
        return accountId;
    }

    public void setAccountId(Long accountId) {
        this.accountId = accountId;
    }

    public Long getAuthorUserId() {
        return authorUserId;
    }

    public void setAuthorUserId(Long authorUserId) {
        this.authorUserId = authorUserId;
    }

    public InvitationStatus getInvitationStatus() {
        return invitationStatus;
    }

    public void setInvitationStatus(InvitationStatus invitationStatus) {
        this.invitationStatus = invitationStatus;
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public UUID getPublicId() {
        return publicId;
    }

    public void setPublicId(UUID publicId) {
        this.publicId = publicId;
    }

    public Email getTargetEmail() {
        return findTargetEmail().getOrNull();
    }

    public Option<Email> findTargetEmail() {
        return Option.of(targetEmail);
    }

    public void setTargetEmail(Email targetEmail) {
        this.targetEmail = targetEmail;
    }

    @Override
    public String toString() {
        return "To:" + targetEmail +
                "Type of invitation:" + invitationType +
                "Sent:" + created;
    }


    public boolean isAccountKeyRequired() {
        return accountKeyRequired;
    }

    public void setAccountKeyRequired(boolean accountKeyRequired) {
        this.accountKeyRequired = accountKeyRequired;
    }

    public String getAccountKeyAccessToken() {
        return accountKeyAccessToken;
    }

    public Optional<EnumSet<PrivacyFlag>> getGrantedConsents() {
        return Optional.ofNullable(grantedConsents);
    }

    public void setGrantedConsents(@Nullable EnumSet<PrivacyFlag> grantedConsents) {
        this.grantedConsents = grantedConsents;
    }
}
