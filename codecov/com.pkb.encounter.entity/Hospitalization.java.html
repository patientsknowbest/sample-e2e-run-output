package com.pkb.encounter.entity;

import java.io.Serializable;
import java.util.Date;

import com.pkb.annotation.EHRField;

/**
 * An hospitalization instance in an Encounter
 * http://www.hl7.org/implement/standards/fhir/encounter.html
 *
 * @author pravina
 *
 */
public class Hospitalization implements Serializable {

    /**
     *
     */
    private static final long serialVersionUID = 911546648379141710L;

    @EHRField
    private Date admissionDate;

    @EHRField
    private Date dischargeDate;

    @EHRField
    private GeographicLocation dischargeDestination;

    public Date getAdmissionDate() {
        return admissionDate;
    }

    public void setAdmissionDate(Date admissionDate) {
        this.admissionDate = admissionDate;
    }

    public Date getDischargeDate() {
        return dischargeDate;
    }

    public void setDischargeDate(Date dischargeDate) {
        this.dischargeDate = dischargeDate;
    }

    public GeographicLocation getDischargeDestination() {
        return dischargeDestination;
    }

    public void setDischargeDestination(GeographicLocation dischargeDestination) {
        this.dischargeDestination = dischargeDestination;
    }
}
