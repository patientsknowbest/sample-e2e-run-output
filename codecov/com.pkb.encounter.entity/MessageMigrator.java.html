package com.pkb.encounter.entity;

import com.pkb.PrivateIdMigrator;
import com.pkb.annotation.EHRMigratorBase;
import com.pkb.common.datamigrator.Html4XmlUnescaper;
import com.pkb.encounter.entity.migrators.MessageV1Migrator;
import com.pkb.encounter.entity.migrators.MessageV2Migrator;
import com.pkb.encounter.entity.migrators.MessageV3Migrator;
import com.pkb.encounter.entity.migrators.MessageV4Migrator;

public class MessageMigrator extends EHRMigratorBase {

    private static final Migrators MIGRATORS = Migrators.of(
            new MessageV1Migrator(),
            new MessageV2Migrator(new Html4XmlUnescaper()),
            new MessageV3Migrator(),
            new MessageV4Migrator(),
            PrivateIdMigrator.migratorFor(Message.class, "messageTypeId", "string11", 5),
            PrivateIdMigrator.migratorFor(Message.class, "reasonCodeId", "string13",  6));

    @Override
    public Migrators getMigrators() {
        return MIGRATORS;
    }

}
