package com.pkb.encounter.entity;

/**
 * Codes taken from http://www.snomedbrowser.com/Codes/Details/308021002
 *
 * @author pravina
 *
 */
public enum LocationType {

    ANTENATAL_CLINIC("185226003", "SNOMED-CT", "Seen in antenatal clinic"),
    WARD("225746001", "SNOMED-CT", "Ward (environment)"),
    HOSPITAL_CASUALTY("185210004", "SNOMED-CT", "Seen in hospital casualty"),
    HEALTH_CENTER("185203005", "SNOMED-CT", "Seen in Health Center"),
    DOMESTIC_ENVIRONMENT("261770009", "SNOMED-CT", "Domestic (environment)"),
    GENERAL_PRACTITIONER_SURGERY("185202000", "SNOMED-CT", "Seen in general practitioner surgery"),
    ICU("309904001", "SNOMED-CT", "Intensive care unit"),
    DELIVERY_SUITE("413964002", "SNOMED-CT", "Delivery suite"),
    THEATRE("257715004", "SNOMED-CT", "Theatre"),
    POSTNATAL_CLINIC("185266004", "SNOMED-CT", "Seen in postnatal clinic");

    private String code;

    private String codeSystemName;

    private String description;

    LocationType(String code, String codeSystemName, String description) {
        this.code = code;
        this.codeSystemName = codeSystemName;
        this.description = description;
    }

    public String getCode() {
        return code;
    }

    public void setCode(String code) {
        this.code = code;
    }

    public String getCodeSystemName() {
        return codeSystemName;
    }

    public void setCodeSystemName(String codeSystemName) {
        this.codeSystemName = codeSystemName;
    }

    public String getDescription() {
        return description;
    }

    public void setDescription(String description) {
        this.description = description;
    }

}
