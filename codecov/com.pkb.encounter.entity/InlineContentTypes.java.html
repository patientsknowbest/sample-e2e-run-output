package com.pkb.encounter.entity;

/**
 * @deprecated this is only used for a few special cases,
 * like embedding IMH results in messages
 * (dead feature, still in some old data)
 */
@Deprecated
public interface InlineContentTypes {

    String MSG_QUESTIONNAIRE = "SURVEY";
    String MSG_INLINE_CALL = "CALL";
    String MSG_INLINE_MED_APPROVED = "MED_APPROVED";
    String MSG_INLINE_MED_FOR_APPROVAL = "MED_FOR_APPROVAL";
    String MSG_INLINE_MED = "MED";

    String getInlineContentType();

    default boolean isSurvey() {
        return MSG_QUESTIONNAIRE.equals(getInlineContentType());
    }

    default boolean isCall() {
        return MSG_INLINE_CALL.equals(getInlineContentType());
    }

    default boolean isMedApproved() {
        return MSG_INLINE_MED_APPROVED.equals(getInlineContentType());
    }

    default boolean isMedForApproval() {
        return MSG_INLINE_MED_FOR_APPROVAL.equals(getInlineContentType());
    }

    default boolean isMed() {
        return MSG_INLINE_MED.equals(getInlineContentType());
    }

    default boolean isInlineContentSet() {
        return getInlineContentType() != null;
    }
}
