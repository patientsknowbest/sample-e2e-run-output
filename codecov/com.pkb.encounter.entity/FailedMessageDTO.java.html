package com.pkb.encounter.entity;

import com.pkb.annotation.EHRField;
import com.pkb.annotation.EHRField.QueryField;
import com.pkb.entities.enums.MenuDataType;

import java.io.Serializable;
import java.util.UUID;

/**
 * A DTO class to capture the message which could not be delivered
 *
 * @author pravina
 *
 */
public class FailedMessageDTO implements Serializable {

    /**
     *
     */
    private static final long serialVersionUID = -4940407179176650400L;

    public static final MenuDataType MS_PATH = MenuDataType.failedMessage;

    @EHRField(queryField = QueryField.uuid01)
    private UUID conversationId;

    @EHRField(queryField = QueryField.long01)
    private Long messageId;

    @EHRField
    private String receiverId;

    @EHRField(queryField = QueryField.uniqueId)
    private UUID uniqueId;

    public UUID getConversationId() {
        return conversationId;
    }

    public void setConversationId(UUID conversationId) {
        this.conversationId = conversationId;
    }

    public Long getMessageId() {
        return messageId;
    }

    public void setMessageId(Long messageId) {
        this.messageId = messageId;
    }

    public String getReceiverId() {
        return receiverId;
    }

    public void setReceiverId(String receiverId) {
        this.receiverId = receiverId;
    }

    public UUID getUniqueId() {
        return uniqueId;
    }

    public void setUniqueId(UUID uniqueId) {
        this.uniqueId = uniqueId;
    }
}
