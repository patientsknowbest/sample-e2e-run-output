package com.pkb.encounter.entity;

import java.io.Serializable;
import java.util.Date;
import java.util.UUID;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.persistence.Temporal;
import javax.persistence.TemporalType;

import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Type;

import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;

@Entity
@Table(schema = "public", name = "encounter_appointment_link")
public class EncounterToAppointmentLink implements Serializable {
    private static final long serialVersionUID = 2L;

    @Id
    @GenericGenerator(name = "seq_app_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "app"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_app"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_app_gen")
    private long id;

    @Column(name = "encounter_unique_id")
    @Type(type = "pg-uuid")
    private UUID encounterUniqueId;

    @Column(name = "appointment_unique_id")
    @Type(type = "pg-uuid")
    private UUID appointmentUniqueId;

    @Column(name = "account_id")
    private long accountId;

    @Temporal(TemporalType.TIMESTAMP)
    @Column
    private Date persistedDate;

    public EncounterToAppointmentLink() {}

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public long getAccountId() {
        return accountId;
    }

    public void setAccountId(long accountId) {
        this.accountId = accountId;
    }

    public Date getPersistedDate() {
        return persistedDate;
    }

    public void setPersistedDate(Date persistedDate) {
        this.persistedDate = persistedDate;
    }

    public UUID getEncounterUniqueId() {
        return encounterUniqueId;
    }

    public void setEncounterUniqueId(UUID leftUniqueId) {
        this.encounterUniqueId = leftUniqueId;
    }

    public UUID getAppointmentUniqueId() {
        return appointmentUniqueId;
    }

    public void setAppointmentUniqueId(UUID rightUniqueId) {
        this.appointmentUniqueId = rightUniqueId;
    }
}
