package com.pkb.api.internal.messaging;

import com.pkb.entities.enums.Gender;
import com.pkb.user.entity.PKBPerson;
import com.pkb.user.entity.PKBPersonHelper;
import org.jetbrains.annotations.Nullable;
import org.springframework.core.convert.converter.Converter;

import java.time.LocalDate;
import java.util.Optional;

public class PKBPersonToMessagingPKBPersonDTOConverter implements Converter<PKBPerson, MessagingPKBPersonDTO> {

    private PKBPersonHelper personHelper;

    // needed for the moment by CDI
    public PKBPersonToMessagingPKBPersonDTOConverter() {

    }

    public PKBPersonToMessagingPKBPersonDTOConverter(PKBPersonHelper personHelper) {
        this.personHelper = personHelper;
    }

    @Override
    public MessagingPKBPersonDTO convert(@Nullable PKBPerson source) {
        if (source == null) {
            return null;
        }
        ImmutableMessagingPKBPersonDTO.Builder personDtoBuilder = ImmutableMessagingPKBPersonDTO.builder();
        personDtoBuilder.id(source.getId());
        personDtoBuilder.userType(source.getUserType());
        if (source.getFirstName() != null) {
            personDtoBuilder.firstName(source.getFirstName());
        }
        if (source.getLastName() != null) {
            personDtoBuilder.lastName(source.getLastName());
        }
        if (source.getJobTitle() != null) {
            personDtoBuilder.jobTitle(source.getJobTitle());
        }
        if (source.getTitle() != null) {
            personDtoBuilder.title(source.getTitle());
        }
        Optional<LocalDate> maybeDob = source.findDateOfBirth();
        if (maybeDob.isPresent()) {
            personDtoBuilder.age(personHelper.calculateAge(source));
        }
        personDtoBuilder.gender(Gender.getById(source.getGender()));
        return personDtoBuilder.build();
    }

    @Override
    public <U> Converter<PKBPerson, U> andThen(Converter<? super MessagingPKBPersonDTO, ? extends U> after) {
        return Converter.super.andThen(after);
    }
}
