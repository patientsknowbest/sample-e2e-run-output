package com.pkb.api.internal.messaging;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.pkb.app.entity.AbstractPersonRoles;
import com.pkb.app.entity.Genders;
import org.immutables.value.Value;

import java.io.Serializable;
import java.util.Optional;

@Value.Immutable
@JsonSerialize(as = ImmutableMessagingPKBPersonDTO.class)
@JsonDeserialize(as = ImmutableMessagingPKBPersonDTO.class)
public abstract class MessagingPKBPersonDTO extends AbstractPersonRoles implements Serializable, Genders {
    public abstract long getId();
    public abstract Optional<String> getFirstName();
    public abstract Optional<String> getLastName();
    public abstract Optional<String> getJobTitle();
    public abstract Optional<String> getTitle();
    public abstract Optional<Integer> getAge();

    public String getNameWithoutTitle() {
        return getFirstName().orElse("") + " " + getLastName().orElse("");
    }

    public String getInitials() {
        String firstInitial = getFirstName().map(t -> t.substring(0,1)).orElse("");
        String lastInitial = getLastName().map(t -> t.substring(0,1)).orElse("");
        return firstInitial + lastInitial;
    }
}
