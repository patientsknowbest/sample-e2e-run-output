package com.pkb.plan.facade;

import com.opensymphony.xwork2.ActionContext;
import com.pkb.plan.support.PrintablePlanTransformer;
import com.pkb.plan.support.PrintableTextCsvPrinter;
import com.pkb.print.domain.PrintableText;
import com.pkb.service.phplan.export.ExportPlanRequestData;
import com.pkb.service.phplan.export.ExportablePatientPlans;
import com.pkb.service.phplan.export.facade.PlanExportFacade;
import com.pkb.util.async.AsyncTaskRunner;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.io.StringWriter;
import java.util.ArrayList;
import java.util.List;

/**
 * Responsible for providing features of a facade to a view, designed to be specific for the view.
 * Provides a way to share view-specific logic between actions (or similar clients that provides backend of a view).
 *
 * View facades should not call other view facades.
 * Normally view facades are orchestrating (feature)facades and/or providing data to view in a way it is easy to render.
 */
@Component
public class PlanExportViewFacade {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    private AsyncTaskRunner asyncTaskRunner;
    @Autowired
    private PlanExportFacade planExportFacade;
    @Autowired
    private PrintablePlanTransformer printablePlanTransformer;
    @Autowired
    private PrintableTextCsvPrinter csvPrinter;

    public void exportPlans(ExportPlanRequestData exportPlanRequestData) {
        asyncTaskRunner.executeInThread(() -> doExportPlans(exportPlanRequestData));
    }

    private void doExportPlans(ExportPlanRequestData exportPlanRequestData) {
        try {
            ActionContext.setContext(exportPlanRequestData.actionContext());
            ExportablePatientPlans exportablePatientPlans = planExportFacade.exportPlans(exportPlanRequestData);
            List<List<PrintableText>> printablePlanExport = new ArrayList<>();
            printablePlanExport.add(collectPrintableHeaders(exportPlanRequestData, exportablePatientPlans));
            printablePlanExport.addAll(transformPrintablePlans(exportPlanRequestData, exportablePatientPlans));
            StringWriter exportedContent = csvPrinter.print(printablePlanExport, exportPlanRequestData.locale());
            planExportFacade.finalizeExport(exportablePatientPlans, exportPlanRequestData, exportedContent);

        } catch (Exception e) {
            LOGGER.error("Exeception while exporting plans by user-{} from template-{}", exportPlanRequestData.loggedInUser(), exportPlanRequestData.templateId(), e);
        }
    }

    private List<PrintableText> collectPrintableHeaders(ExportPlanRequestData exportPlanRequestData, ExportablePatientPlans exportablePatientPlans) throws Exception {
        List<PrintableText> result;
        if (exportPlanRequestData.exportAll()) {
            result = printablePlanTransformer.getCSVHeaderAll();
        } else {
            result = printablePlanTransformer.getCSVHeaderLatest(exportablePatientPlans);
        }
        return result;
    }

    private List<List<PrintableText>> transformPrintablePlans(ExportPlanRequestData exportPlanRequestData, ExportablePatientPlans exportablePatientPlans) {
        List<List<PrintableText>> result;
        if (exportPlanRequestData.exportAll()) {
            result = printablePlanTransformer.transformCSVExportAll(exportablePatientPlans);
        } else {
            result = printablePlanTransformer.transformCSVExportLatest(exportablePatientPlans);
        }
        return result;
    }
}
