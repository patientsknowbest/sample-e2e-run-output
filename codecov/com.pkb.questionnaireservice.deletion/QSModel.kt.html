package com.pkb.questionnaireservice.deletion

import com.pkb.deletion.model.DatabaseTable
import com.pkb.deletion.model.ImmutablePkColumn
import com.pkb.deletion.model.PkColumn.pk
import com.pkb.deletion.model.StandardInboundForeignKeyReference.owningFk
import com.pkb.deletion.model.StandardInboundForeignKeyReference.standardFk
import com.pkb.questionnaireservice.PersonPublicId
import com.pkb.questionnaireservice.QuestionnaireId
import com.pkb.questionnaireservice.QuestionnaireRequestId
import com.pkb.questionnaireservice.QuestionnaireResponseId
import java.util.UUID

class QSModel {

    companion object {

        private val QUESTIONNAIRE_RESPONSE: DatabaseTable<QuestionnaireResponseId> = DatabaseTable.builderFor(QuestionnaireResponseId::class.java).tableName("ques.questionnaire_response")
            .pkColumn(pk(QuestionnaireResponseId::class.java, emptyList())).build()

        private val QUESTIONNAIRE_REQUEST: DatabaseTable<QuestionnaireRequestId> =
            DatabaseTable.builderFor(QuestionnaireRequestId::class.java).tableName("ques.questionnaire_request")
            .pkColumn(
                pk(QuestionnaireRequestId::class.java, listOf(
                owningFk(QUESTIONNAIRE_RESPONSE, "questionnaire_request_id")
            ))
            ).build()

        private val QUESTIONNAIRE_TEAM: DatabaseTable<UUID> = DatabaseTable.builderFor(UUID::class.java).tableName("ques.questionnaire_team")
            .pkColumn(ImmutablePkColumn.builder<UUID>().name("questionnaire_id").columnTypeClass(UUID::class.java).build()).build()

        private val QUESTIONNAIRE: DatabaseTable<QuestionnaireId> = DatabaseTable.builderFor(QuestionnaireId::class.java).tableName("ques.questionnaire")
            .pkColumn(
                pk(QuestionnaireId::class.java, listOf(
                    owningFk(QUESTIONNAIRE_REQUEST, "questionnaire_id"),
                    owningFk(QUESTIONNAIRE_TEAM, "questionnaire_id")
                ))
            ).build()

        val PERSON: DatabaseTable<PersonPublicId> = DatabaseTable.builderFor(PersonPublicId::class.java).tableName("pkbperson")
            .pkColumn(
                pk(PersonPublicId::class.java, listOf(
                owningFk(QUESTIONNAIRE_REQUEST, "patient_id"),
                standardFk(QUESTIONNAIRE_REQUEST, "requestor_person_id"),
                standardFk(QUESTIONNAIRE_RESPONSE, "completed_by"),
                standardFk(QUESTIONNAIRE, "creator_person_id")
            ))
            )
            .isVirtual(true)
            .build()
    }
}
