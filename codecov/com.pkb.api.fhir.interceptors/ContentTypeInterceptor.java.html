package com.pkb.api.fhir.interceptors;

import ca.uhn.fhir.rest.server.interceptor.InterceptorAdapter;
import com.google.common.base.Splitter;
import io.vavr.collection.List;
import io.vavr.collection.Set;
import io.vavr.control.Option;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.io.UncheckedIOException;

import static com.google.common.base.CharMatcher.anyOf;
import static io.vavr.API.Set;
import static javax.servlet.http.HttpServletResponse.SC_UNSUPPORTED_MEDIA_TYPE;
import static org.apache.commons.lang3.StringUtils.isBlank;

public class ContentTypeInterceptor extends InterceptorAdapter {
    private static final int SECURITY_ELEMENT_LIMIT = 30;
    private static final Splitter ACCEPT_HEADER_SPLITTER = Splitter.on(anyOf(",;"))
            .trimResults()
            .omitEmptyStrings()
            .limit(SECURITY_ELEMENT_LIMIT);

    private static final Set<String> SUPPORTED_CONTENT_TYPES = Set(
            "*/*", // If no Accept: header is provided, this is what we see.
            "json",
            "application/json",
            "application/fhir+json",
            "application/json+fhir");

    @Override
    public boolean incomingRequestPreProcessed(HttpServletRequest theRequest, HttpServletResponse theResponse) {
        String contentTypeString = Option.of(theRequest.getParameter("_format"))
                .getOrElse(theRequest.getHeader("Accept"));

        if (isBlank(contentTypeString)) {
            return true;
        }
        List<String> contentTypes = List.ofAll(ACCEPT_HEADER_SPLITTER.splitToList(contentTypeString));

        Option<String> maybeSupported = contentTypes
                .map(String::toLowerCase)
                .filter(SUPPORTED_CONTENT_TYPES::contains)
                .headOption();

        boolean isAccepted = maybeSupported.isDefined();
        if (!isAccepted) {
            try {
                theResponse.sendError(SC_UNSUPPORTED_MEDIA_TYPE);
            } catch (IOException e) {
                throw new UncheckedIOException(e);
            }
        }
        return isAccepted;
    }
}
