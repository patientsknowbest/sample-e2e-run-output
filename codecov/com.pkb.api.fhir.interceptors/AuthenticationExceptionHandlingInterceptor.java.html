package com.pkb.api.fhir.interceptors;

import ca.uhn.fhir.rest.api.server.RequestDetails;
import ca.uhn.fhir.rest.server.exceptions.AuthenticationException;
import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import ca.uhn.fhir.rest.server.interceptor.ExceptionHandlingInterceptor;
import ca.uhn.fhir.rest.server.interceptor.InterceptorAdapter;
import com.pkb.common.util.FrameFilter;

import javax.servlet.ServletException;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;

public class AuthenticationExceptionHandlingInterceptor extends InterceptorAdapter {
    private static final ExceptionHandlingInterceptor DEFAULT_EXCEPTION_HANDLER = new ExceptionHandlingInterceptor();

    @Override
    public boolean handleException(
            RequestDetails requestDetails,
            BaseServerResponseException exception,
            HttpServletRequest request,
            HttpServletResponse response) throws ServletException, IOException {
        
        boolean shouldContinue = super.handleException(requestDetails, (BaseServerResponseException)FrameFilter.filter(exception), request, response);

        if (AuthenticationException.class.isInstance(exception)) {
            shouldContinue = DEFAULT_EXCEPTION_HANDLER.handleException(requestDetails, (BaseServerResponseException)FrameFilter.filter(exception), request, response);
        }

        return shouldContinue;
    }
}
