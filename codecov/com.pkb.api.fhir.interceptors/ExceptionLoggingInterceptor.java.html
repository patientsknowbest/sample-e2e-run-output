package com.pkb.api.fhir.interceptors;

import ca.uhn.fhir.rest.api.server.RequestDetails;
import ca.uhn.fhir.rest.server.exceptions.AuthenticationException;
import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import ca.uhn.fhir.rest.server.exceptions.ForbiddenOperationException;
import ca.uhn.fhir.rest.server.exceptions.InternalErrorException;
import ca.uhn.fhir.rest.server.exceptions.ResourceNotFoundException;
import ca.uhn.fhir.rest.server.interceptor.InterceptorAdapter;
import com.google.common.collect.ImmutableSet;
import com.pkb.common.util.FrameFilter;
import org.slf4j.Logger;

import javax.servlet.ServletException;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.util.stream.Stream;

import static java.lang.invoke.MethodHandles.lookup;
import static java.util.stream.Collectors.toSet;
import static org.slf4j.LoggerFactory.getLogger;

public class ExceptionLoggingInterceptor extends InterceptorAdapter {
    private static final Logger LOG = getLogger(lookup().lookupClass());
    private ImmutableSet<String> fhirExceptionsToLogWithDetails = ImmutableSet.of(
            InternalErrorException.class.getCanonicalName()
    );
    private ImmutableSet<String> fhirExceptionsToLogWithMinimalDetails = ImmutableSet.copyOf(Stream.of(
            AuthenticationException.class,
            ResourceNotFoundException.class,
            ForbiddenOperationException.class)
            .map(Class::getCanonicalName)
            .collect(toSet())
    );

    @Override
    public boolean handleException(
            RequestDetails requestDetails,
            BaseServerResponseException exception,
            HttpServletRequest servletRequest,
            HttpServletResponse servletResponse) throws ServletException, IOException {
        boolean b = super.handleException(requestDetails, exception, servletRequest, servletResponse);
        String canonicalName = exception.getClass().getCanonicalName();
        if (fhirExceptionsToLogWithDetails.contains(canonicalName)) {
            LOG.error("Error occurred while processing request. Stacktrace is:", FrameFilter.filter(exception));
        } else if (fhirExceptionsToLogWithMinimalDetails.contains(canonicalName)) {
            LOG.error("Error occurred while processing request. Message is: {}", exception.getMessage());
        }
        return b;
    }
}
