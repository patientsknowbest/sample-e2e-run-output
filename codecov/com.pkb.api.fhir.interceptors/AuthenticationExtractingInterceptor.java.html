package com.pkb.api.fhir.interceptors;

import ca.uhn.fhir.rest.api.server.RequestDetails;
import ca.uhn.fhir.rest.server.interceptor.InterceptorAdapter;
import io.vavr.control.Option;

import javax.servlet.http.Cookie;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.util.Arrays;

import static com.google.common.net.HttpHeaders.AUTHORIZATION;

/**
 * Extract authentication information fron the raw http servlet request 
 * and put it in the UserData instead, so we don't need to inject the raw 
 * HttpServletRequest into every request method.
 */
public class AuthenticationExtractingInterceptor extends InterceptorAdapter {
    public static final String USER_DATA_AUTHN_HEADER = "AUTHN_HEADER";
    public static final String USER_DATA_AUTHN_COOKIE = "AUTHN_COOKIE";
    @Override
    public boolean incomingRequestPostProcessed(RequestDetails details, HttpServletRequest theRequest, HttpServletResponse theResponse) {
        String authorizationHeader = theRequest.getHeader(AUTHORIZATION);
        details.getUserData().put(USER_DATA_AUTHN_HEADER, authorizationHeader);
        Arrays.stream(Option.of(theRequest.getCookies()).getOrElse(new Cookie[]{}))
                .filter(cookie -> cookie.getName().equalsIgnoreCase("ui-exchange"))
                .findFirst()
                .ifPresent(cookie -> details.getUserData().put(USER_DATA_AUTHN_COOKIE, cookie.getValue()));
        return super.incomingRequestPreProcessed(theRequest, theResponse);
    }
}
