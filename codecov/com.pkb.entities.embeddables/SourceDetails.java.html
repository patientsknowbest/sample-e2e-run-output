package com.pkb.entities.embeddables;

import com.google.common.base.MoreObjects;
import com.pkb.entities.pub.PublicOrg;
import com.pkb.entities.pub.PublicTeam;
import org.hibernate.annotations.JoinFormula;

import javax.persistence.Column;
import javax.persistence.Embeddable;
import javax.persistence.FetchType;
import javax.persistence.ManyToOne;
import java.io.Serializable;

@Embeddable
public class SourceDetails implements Serializable {

	private static final long serialVersionUID = 9032869466513593975L;

	@Column(name = "sourcePersonId")
	private Long personId;

	@Column(name = "sourceTeamId")
	private Long teamId;

	@ManyToOne(fetch = FetchType.LAZY)
	@JoinFormula(value = "sourceTeamId", referencedColumnName = "id")
	private PublicTeam team;

	@Column(name = "sourceOrgId")
	private Long orgId;

	@ManyToOne(fetch = FetchType.LAZY)
	@JoinFormula(value = "sourceOrgId", referencedColumnName = "id")
	private PublicOrg org;

	@Column(name = "responsibleOrganisationIdentifier")
	private String responsibleOrganisationIdentifier;

	@Override
	public String toString() {
		return MoreObjects.toStringHelper(this)
				.omitNullValues()
				.add("personId", personId)
				.add("team", team)
				.add("org", org)
				.add("responsibleOrganisationIdentifier", responsibleOrganisationIdentifier)
				.toString();
	}

	public Long getPersonId() {
		return personId;
	}

	public void setPersonId(long personId) {
		this.personId = personId;
	}

	public PublicTeam getTeam() {
		return team;
	}

	public void setTeam(PublicTeam team) {
		this.team = team;
		if (team == null) {
			teamId = null;
		} else {
			teamId = team.getId();
		}
	}

	public PublicOrg getOrg() {
		return org;
	}

	public void setOrg(PublicOrg org) {
		this.org = org;
		if (org == null) {
			orgId = null;
		} else {
			orgId = org.getId();
		}
	}

	public String getResponsibleOrganisationIdentifier() {
		return responsibleOrganisationIdentifier;
	}

	public void setResponsibleOrganisationIdentifier(String responsibleOrganisationIdentifier) {
		this.responsibleOrganisationIdentifier = responsibleOrganisationIdentifier;
	}
}
