package com.pkb.entities.embeddables;

import javax.persistence.Column;
import javax.persistence.Embeddable;

@Embeddable
public class EncodedEncryptedPrivateKeyInfoWithSecretDeriv extends EncodedEncryptedPrivateKeyInfo {

    @Column(name = "secret_key_deriv_algorithm")
    private String secretKeyDerivAlgorithm;

    @Column(name = "secret_key_deriv_data_length")
    private Integer secretKeyDerivDataLength;

    @Column(name = "secret_key_deriv_salt_inline")
    private byte[] secretKeyDerivSaltInline;

    @Column(name = "secret_key_deriv_iterations")
    private Integer secretKeyDerivIterations;

    public EncodedEncryptedPrivateKeyInfoWithSecretDeriv(){}

    public EncodedEncryptedPrivateKeyInfoWithSecretDeriv(byte[] encryptedPrivateKeyInfo, int privateKeySize, String privateKeyAlgorithm,
                                                         byte[] encryptedSecretKey, String secretKeyAlgorithm,
                                                         String secretKeyDerivAlgorithm, Integer secretKeyDerivDataLength,
                                                         byte[] secretKeyDerivSalt, Integer secretKeyDerivIterations) {
        super(encryptedPrivateKeyInfo, privateKeySize, privateKeyAlgorithm, encryptedSecretKey, secretKeyAlgorithm);
        this.secretKeyDerivAlgorithm = secretKeyDerivAlgorithm;
        this.secretKeyDerivDataLength = secretKeyDerivDataLength;
        this.secretKeyDerivSaltInline = secretKeyDerivSalt;
        this.secretKeyDerivIterations = secretKeyDerivIterations;
    }

    public String getSecretKeyDerivAlgorithm() {
        return secretKeyDerivAlgorithm;
    }

    public void setSecretKeyDerivAlgorithm(String secretKeyDerivAlgorithm) {
        this.secretKeyDerivAlgorithm = secretKeyDerivAlgorithm;
    }

    public Integer getSecretKeyDerivDataLength() {
        return secretKeyDerivDataLength;
    }

    public void setSecretKeyDerivDataLength(Integer secretKeyDerivDataLength) {
        this.secretKeyDerivDataLength = secretKeyDerivDataLength;
    }

    public byte[] getSecretKeyDerivSaltInline() {
        return secretKeyDerivSaltInline;
    }

    public void setSecretKeyDerivSaltInline(byte[] secretKeyDerivSaltInline) {
        this.secretKeyDerivSaltInline = secretKeyDerivSaltInline;
    }

    public Integer getSecretKeyDerivIterations() {
        return secretKeyDerivIterations;
    }

    public void setSecretKeyDerivIterations(Integer secretKeyDerivIterations) {
        this.secretKeyDerivIterations = secretKeyDerivIterations;
    }
}
