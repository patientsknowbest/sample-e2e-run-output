package com.pkb.entities.embeddables;

import com.google.common.base.MoreObjects;

import javax.persistence.Column;
import javax.persistence.MappedSuperclass;
import java.io.Serializable;

@MappedSuperclass
public class EncodedEncryptedPrivateKeyInfo implements Serializable {

    private static final long serialVersionUID = 2380725520417074804L;

    @Column(name = "encoded_encrypted_private_key_info_inline")
    private byte[] encryptedPrivateKeyInfoInline;

    @Column(name = "private_key_size")
    private int privateKeySize;

    @Column(name = "private_key_algorithm")
    private String privateKeyAlgorithm;

    @Column(name = "encrypted_secret_key_inline")
    private byte[] encryptedSecretKeyInline;

    @Column(name = "secret_key_algorithm")
    private String secretKeyAlgorithm;

    public EncodedEncryptedPrivateKeyInfo() {
    }

    public EncodedEncryptedPrivateKeyInfo(byte[] encryptedPrivateKeyInfo, int privateKeySize, String privateKeyAlgorithm, byte[] encryptedSecretKey, String secretKeyAlgorithm) {
        this.encryptedPrivateKeyInfoInline = encryptedPrivateKeyInfo;
        this.privateKeySize = privateKeySize;
        this.privateKeyAlgorithm = privateKeyAlgorithm;
        this.encryptedSecretKeyInline = encryptedSecretKey;
        this.secretKeyAlgorithm = secretKeyAlgorithm;
    }

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("privateKeySize", privateKeySize)
                .add("privateKeyAlgorithm", privateKeyAlgorithm)
                .add("secretKeyAlgorithm", secretKeyAlgorithm)
                .toString();
    }

    public byte[] getEncryptedPrivateKeyInfo() {
        return encryptedPrivateKeyInfoInline;
    }

    public void setEncryptedPrivateKeyInfo(byte[] encryptedPrivateKeyInfo) {
        this.encryptedPrivateKeyInfoInline = encryptedPrivateKeyInfo;
    }

    public int getPrivateKeySize() {
        return privateKeySize;
    }

    public void setPrivateKeySize(int privateKeySize) {
        this.privateKeySize = privateKeySize;
    }

    public String getPrivateKeyAlgorithm() {
        return privateKeyAlgorithm;
    }

    public void setPrivateKeyAlgorithm(String privateKeyAlgorithm) {
        this.privateKeyAlgorithm = privateKeyAlgorithm;
    }

    public byte[] getEncryptedSecretKey() {
        return encryptedSecretKeyInline;
    }

    public void setEncryptedSecretKey(byte[] encryptedSecretKey) {
        this.encryptedSecretKeyInline = encryptedSecretKey;
    }

    public String getSecretKeyAlgorithm() {
        return secretKeyAlgorithm;
    }

    public void setSecretKeyAlgorithm(String secretKeyAlgorithm) {
        this.secretKeyAlgorithm = secretKeyAlgorithm;
    }
}
