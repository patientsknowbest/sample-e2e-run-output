package com.pkb.entities.embeddables;

import com.pkb.entities.enums.FhirStoreSyncOperationType;
import com.pkb.entities.enums.FhirStoreSyncResourceTarget;
import com.pkb.entities.enums.FhirStoreSyncSourceTable;

import javax.persistence.Column;
import javax.persistence.Embeddable;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import java.time.Instant;
import java.util.UUID;

@Embeddable
public class FhirStoreSyncDetails {

    @Column(name = "source_table", nullable = false)
    @Enumerated(EnumType.STRING)
    private FhirStoreSyncSourceTable sourceTable;

    @Column(name = "source_table_id", nullable = false)
    private long sourceTableId;

    @Column(name = "source_table_public_id")
    private UUID sourceTablePublicId;

    //Some tables collapse into other resources. This links them to their parent, e.g. personcontact goes into
    // Patient/Practitioner rather than its own resource. This is a long and not a UUID as these tables refer to each
    // other by primary key in the current datamodel.
    @Column(name = "source_table_parent_id")
    private Long sourceTableParentId;

    @Column(name = "operation_type", nullable = false)
    @Enumerated(EnumType.STRING)
    private FhirStoreSyncOperationType operationType;

    @Column(name = "resource_target", nullable = false)
    @Enumerated(EnumType.STRING)
    private FhirStoreSyncResourceTarget resourceTarget;

    @Column(name = "sync_status_timestamp", nullable = false)
    private Instant syncStatusTimestamp;

    public FhirStoreSyncSourceTable getSourceTable() {
        return sourceTable;
    }

    public void setSourceTable(FhirStoreSyncSourceTable sourceTable) {
        this.sourceTable = sourceTable;
    }

    public long getSourceTableId() {
        return sourceTableId;
    }

    public void setSourceTableId(long sourceTableId) {
        this.sourceTableId = sourceTableId;
    }

    public UUID getSourceTablePublicId() {
        return sourceTablePublicId;
    }

    public void setSourceTablePublicId(UUID sourceTablePublicId) {
        this.sourceTablePublicId = sourceTablePublicId;
    }

    public Long getSourceTableParentId() {
        return sourceTableParentId;
    }

    public void setSourceTableParentId(Long sourceTableParentId) {
        this.sourceTableParentId = sourceTableParentId;
    }

    public Instant getSyncStatusTimestamp() {
        return syncStatusTimestamp;
    }

    public void setSyncStatusTimestamp(Instant syncStatusTimestamp) {
        this.syncStatusTimestamp = syncStatusTimestamp;
    }

    public FhirStoreSyncResourceTarget getResourceTarget() {
        return resourceTarget;
    }

    public void setResourceTarget(FhirStoreSyncResourceTarget resourceTarget) {
        this.resourceTarget = resourceTarget;
    }

    public FhirStoreSyncOperationType getOperationType() {
        return operationType;
    }

    public void setOperationType(FhirStoreSyncOperationType operationType) {
        this.operationType = operationType;
    }

    public boolean isConsent() {
        return getResourceTarget() == FhirStoreSyncResourceTarget.CONSENT;
    }

    public boolean isPractitionerRole() {
        return getResourceTarget() == FhirStoreSyncResourceTarget.PRACTITIONER_ROLE;
    }

    public boolean isPatient() {
        return getResourceTarget() == FhirStoreSyncResourceTarget.PATIENT;
    }
}
