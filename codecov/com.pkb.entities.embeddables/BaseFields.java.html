package com.pkb.entities.embeddables;

import com.google.common.base.MoreObjects;
import com.pkb.entities.pub.PublicPrivacyFlags;
import org.hibernate.annotations.JoinFormula;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Embeddable;
import javax.persistence.FetchType;
import javax.persistence.ManyToOne;
import java.io.Serializable;
import java.time.Instant;
import java.util.UUID;

@Embeddable
public class BaseFields implements Serializable {

    private static final long serialVersionUID = 8025805780771016605L;

    @Column
    private Instant enteredDate;

    @Column
    private Instant persistedDate;

    @Column
    private Long uploadedDataId;

    @Column
    private boolean requireGeneral;

    @Column
    private boolean requireMentalHealth;

    @Column
    private boolean requireSexualHealth;

    @Column
    private boolean requireSocialCare;

    @ManyToOne(fetch = FetchType.EAGER)
    @JoinFormula("(SELECT pf.id FROM privacy_flags pf WHERE pf.uniqueid = uniqueid ORDER BY pf.override DESC, pf.change_date DESC, id DESC LIMIT 1)")
    private PublicPrivacyFlags latestPrivacyFlags;

    @Column
    @Type(type = "pg-uuid")
    private UUID uniqueId;

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("enteredDate", enteredDate)
                .add("persistedDate", persistedDate)
                .add("uploadedDataId", uploadedDataId)
                .add("requireGeneral", requireGeneral)
                .add("requireMentalHealth", requireMentalHealth)
                .add("requireSexualHealth", requireSexualHealth)
                .add("requireSocialCare", requireSocialCare)
                .add("latestPrivacyFlags",latestPrivacyFlags)
                .add("uniqueId", uniqueId)
                .toString();
    }

    public Instant getEnteredDate() {
        return enteredDate;
    }

    public void setEnteredDate(Instant enteredDate) {
        this.enteredDate = enteredDate;
    }

    public Instant getPersistedDate() {
        return persistedDate;
    }

    public void setPersistedDate(Instant persistedDate) {
        this.persistedDate = persistedDate;
    }

    public Long getUploadedDataId() {
        return uploadedDataId;
    }

    public void setUploadedDataId(Long uploadedDataId) {
        this.uploadedDataId = uploadedDataId;
    }


    public boolean isRequireGeneral() {
        return requireGeneral;
    }

    public void setRequireGeneral(boolean requireGeneral) {
        this.requireGeneral = requireGeneral;
    }

    public boolean isRequireMentalHealth() {
        return requireMentalHealth;
    }

    public void setRequireMentalHealth(boolean requireMentalHealth) {
        this.requireMentalHealth = requireMentalHealth;
    }

    public boolean isRequireSexualHealth() {
        return requireSexualHealth;
    }

    public void setRequireSexualHealth(boolean requireSexualHealth) {
        this.requireSexualHealth = requireSexualHealth;
    }

    public boolean isRequireSocialCare() {
        return requireSocialCare;
    }

    public void setRequireSocialCare(boolean requireSocialCare) {
        this.requireSocialCare = requireSocialCare;
    }

    public UUID getUniqueId() {
        return uniqueId;
    }

    public void setUniqueId(UUID uniqueId) {
        this.uniqueId = uniqueId;
    }

    public PublicPrivacyFlags getLatestPrivacyFlags() {
        return latestPrivacyFlags;
    }

    public void setLatestPrivacyFlags(PublicPrivacyFlags latestPrivacyFlags) {
        this.latestPrivacyFlags = latestPrivacyFlags;
    }
}
