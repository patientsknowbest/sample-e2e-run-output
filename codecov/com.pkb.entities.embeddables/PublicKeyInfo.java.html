package com.pkb.entities.embeddables;

import com.google.common.base.MoreObjects;

import javax.persistence.Column;
import javax.persistence.Embeddable;
import java.io.Serializable;

@Embeddable
public class PublicKeyInfo implements Serializable {

    private static final long serialVersionUID = -6845536103623117743L;

    @Column(name = "x509_encoded__key_spec_inline")
    private byte[] x509EncodedKeySpecInline;

    @Column(name = "public_key_algorithm")
    private String algorithm;

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("algorithm", algorithm)
                .toString();
    }

    public byte[] getX509EncodedKeySpec() {
        return x509EncodedKeySpecInline;
    }

    public void setX509EncodedKeySpec(byte[] x509EncodedKeySpec) {
        this.x509EncodedKeySpecInline = x509EncodedKeySpec;
    }

    public String getAlgorithm() {
        return algorithm;
    }

    public void setAlgorithm(String algorithm) {
        this.algorithm = algorithm;
    }
}
