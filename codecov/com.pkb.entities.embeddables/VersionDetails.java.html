package com.pkb.entities.embeddables;

import com.google.common.base.MoreObjects;
import com.pkb.entities.enums.Route;

import javax.persistence.Column;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.JoinColumn;
import javax.persistence.MappedSuperclass;
import java.io.Serializable;
import java.time.Instant;

@MappedSuperclass
public class VersionDetails implements Serializable {

    private static final long serialVersionUID = -536716664941554361L;

    @Column(name = "sourcepersonid")
    private Long sourcePersonId;

    @JoinColumn(name = "sourceteamid")
    private Long sourceTeamId;

    @JoinColumn(name = "sourceorgid")
    private Long sourceOrgId;

    @Column
    private String sourceText;

    @Column
    @Enumerated(EnumType.STRING)
    private Route accessRoute;

    @Column
    private Instant enteredDate;

    @Column(insertable = false, updatable = false)
    private Instant created;

    @Column(insertable = false)
    private Instant versionPersisted;

    @Column
    private String responsibleOrganisationIdentifier;

    @Column
    private Long connectingOrgId;

    @Column(name = "hl7PartnerId")
    private Long partnerId;

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("sourceText", sourceText)
                .add("accessRoute", accessRoute)
                .add("enteredDate", enteredDate)
                .add("created", created)
                .add("versionPersisted", versionPersisted)
                .add("connectingOrgId", connectingOrgId)
                .add("partnerId", partnerId)
                .toString();
    }

    public Long getSourcePersonId() {
        return sourcePersonId;
    }

    public void setSourcePersonId(Long sourcePersonId) {
        this.sourcePersonId = sourcePersonId;
    }

    public Long getSourceTeamId() {
        return sourceTeamId;
    }

    public void setSourceTeamId(Long sourceTeamId) {
        this.sourceTeamId = sourceTeamId;
    }

    public Long getSourceOrgId() {
        return sourceOrgId;
    }

    public void setSourceOrgId(Long sourceOrgId) {
        this.sourceOrgId = sourceOrgId;
    }

    public String getSourceText() {
        return sourceText;
    }

    public void setSourceText(String sourceText) {
        this.sourceText = sourceText;
    }

    public Route getAccessRoute() {
        return accessRoute;
    }

    public void setAccessRoute(Route accessRoute) {
        this.accessRoute = accessRoute;
    }

    public Instant getEnteredDate() {
        return enteredDate;
    }

    public void setEnteredDate(Instant enteredDate) {
        this.enteredDate = enteredDate;
    }

    public Instant getCreated() {
        return created;
    }

    public void setCreated(Instant created) {
        this.created = created;
    }

    public Instant getVersionPersisted() {
        return versionPersisted;
    }

    public void setVersionPersisted(Instant versionPersisted) {
        this.versionPersisted = versionPersisted;
    }

    public String getResponsibleOrganisationIdentifier() { return responsibleOrganisationIdentifier; }

    public void setResponsibleOrganisationIdentifier(String responsibleOrganisationIdentifier) {
        this.responsibleOrganisationIdentifier = responsibleOrganisationIdentifier;
    }

    public Long getConnectingOrgId() {
        return connectingOrgId;
    }

    public void setConnectingOrgId(Long connectingOrgId) {
        this.connectingOrgId = connectingOrgId;
    }

    public Long getPartnerId() {
        return partnerId;
    }

    public void setPartnerId(Long partnerId) {
        this.partnerId = partnerId;
    }
}
