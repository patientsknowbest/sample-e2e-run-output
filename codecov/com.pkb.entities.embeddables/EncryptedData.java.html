package com.pkb.entities.embeddables;

import javax.persistence.Column;
import javax.persistence.Embeddable;
import java.io.Serializable;

/**
 * Represents a blob of encrypted data paired with the secret key needed to decrypt it. The secret key is also
 * encrypted, and will need to be decrypted using a private key based on context (typically the private key of the
 * account this encrypted data belongs to)
 *
 * TODO: Consider merging this with {@link com.pkb.entities.embeddables.EncodedEncryptedPrivateKeyInfo}
 *
 */
@Embeddable
public class EncryptedData implements Serializable {

    private static final long serialVersionUID = -7545554493110265676L;

    @Column(name = "content_inline")
    private byte[] cipherTextInline;

    @Column(name = "encrypted_secret_key_inline")
    private byte[] encryptedSecretKeyInline;

    @Column(name = "secret_key_algorithm")
    private String secretKeyAlgorithm;

    public byte[] getCiphertext() {
        return cipherTextInline;
    }

    public void setCiphertext(byte[] ciphertext) {
        this.cipherTextInline = ciphertext;
    }

    public byte[] getEncryptedSecretKey() {
        return encryptedSecretKeyInline;
    }

    public void setEncryptedSecretKey(byte[] encryptedSecretKey) {
        this.encryptedSecretKeyInline = encryptedSecretKey;
    }

    public String getSecretKeyAlgorithm() {
        return secretKeyAlgorithm;
    }

    public void setSecretKeyAlgorithm(String secretKeyAlgorithm) {
        this.secretKeyAlgorithm = secretKeyAlgorithm;
    }
}
