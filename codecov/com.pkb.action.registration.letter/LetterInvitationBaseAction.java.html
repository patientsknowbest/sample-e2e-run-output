package com.pkb.action.registration.letter;


import com.pkb.action.BaseAction;
import com.pkb.domain.duplicate.TeamDomainService;
import org.slf4j.Logger;
import org.springframework.beans.factory.annotation.Autowired;

import java.lang.invoke.MethodHandles;

import static com.pkb.action.registration.letter.ImmutableTeamDetailsDTO.teamDetails;
import static org.slf4j.LoggerFactory.getLogger;

public class LetterInvitationBaseAction extends BaseAction {

    private static final Logger LOGGER = getLogger(MethodHandles.lookup().lookupClass());

    static final String ACTIVATION_ERROR = "activationError";

    @Autowired
    private TeamDomainService teamDomainService;

    private TeamDetailsDTO teamDetails;

    public TeamDetailsDTO getTeamDetails() {
        return teamDetails;
    }

    boolean lookupBrandDetails(String teamCode) {
        if (teamCode != null) {
            teamDetails = teamDomainService.findTeamWithOrgByTeamCode(teamCode)
                    .peekLeft(error -> LOGGER.info(error.getDescription()))
                    .map(teamWithOrg -> teamDetails(teamWithOrg.getName(), teamCode, teamWithOrg.getOrg().getName(), teamWithOrg.getTeamLogoPath(), teamWithOrg.getOrg().getCode()))
                    .getOrNull();
        }
        return teamDetails != null;
    }

}
