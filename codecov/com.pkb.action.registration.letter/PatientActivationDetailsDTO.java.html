package com.pkb.action.registration.letter;

import com.pkb.action.registration.AbstractPatientActivationDTO;
import com.pkb.util.validation.date.CompositeDate.DayPart;
import com.pkb.util.validation.date.CompositeDate.MonthPart;
import com.pkb.util.validation.date.CompositeDate.YearPart;
import com.pkb.util.validation.date.CompositeDateOfBirth;
import org.apache.struts.beanvalidation.constraints.FieldMatch;

import javax.validation.constraints.Email;
import javax.validation.constraints.Min;
import javax.validation.constraints.NotBlank;
import javax.validation.constraints.NotNull;

import static com.pkb.action.registration.AbstractPatientActivationDTO.PASSWORDS_MUST_MATCH;

@CompositeDateOfBirth
@FieldMatch.List({
        @FieldMatch(first = "password", second = "passwordConfirmation", message = PASSWORDS_MUST_MATCH),
        @FieldMatch(first = "email", second = "emailConfirmation", message = PatientActivationDetailsDTO.EMAILS_MUST_MATCH)
})
public class PatientActivationDetailsDTO extends AbstractPatientActivationDTO {
    static final String EMAILS_MUST_MATCH = "emails.must.match";
    static final String EMAIL_MUST_MATCH_WITH_EMAIL_CONFIRMATION = "email.and.its.confirmation.should.match";

    @NotBlank(message = "decryptToken.might.not.be.blank")
    private String decryptToken;
    @NotBlank(message = "invitationCode.might.not.be.blank")
    private String invitationCode;

    @YearPart
    @Min(value = 1, message = "yearOfBirth.has.invalid.value.selected")
    private int yearOfBirth;
    @MonthPart
    @Min(value = 1, message = "monthOfBirth.has.invalid.value.selected")
    private short monthOfBirth;
    @DayPart
    @Min(value = 1, message = "dayOfBirth.has.invalid.value.selected")
    private short dayOfBirth;

    @NotNull(message = "confirmation.email.cannot.be.blank")
    private com.pkb.datamodel.Email emailConfirmation;

    public String getDecryptToken() {
        return decryptToken;
    }

    public void setDecryptToken(String decryptToken) {
        this.decryptToken = decryptToken;
    }

    public String getInvitationCode() {
        return invitationCode;
    }

    public void setInvitationCode(String invitationCode) {
        this.invitationCode = invitationCode;
    }

    public int getYearOfBirth() {
        return yearOfBirth;
    }

    public void setYearOfBirth(int yearOfBirth) {
        this.yearOfBirth = yearOfBirth;
    }

    public short getMonthOfBirth() {
        return monthOfBirth;
    }

    public void setMonthOfBirth(short monthOfBirth) {
        this.monthOfBirth = monthOfBirth;
    }

    public short getDayOfBirth() {
        return dayOfBirth;
    }

    public void setDayOfBirth(short dayOfBirth) {
        this.dayOfBirth = dayOfBirth;
    }

    public com.pkb.datamodel.Email getEmailConfirmation() {
        return emailConfirmation;
    }

    public void setEmailConfirmation(@Email(message = "confirmation.email.is.not.valid") com.pkb.datamodel.Email emailConfirmation) {
        this.emailConfirmation = emailConfirmation;
    }
}
