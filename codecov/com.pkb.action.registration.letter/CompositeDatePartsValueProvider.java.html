package com.pkb.action.registration.letter;

import com.google.common.collect.ImmutableMap;
import com.pkb.common.datetime.DateTimeService;
import org.jetbrains.annotations.NotNull;

import java.util.Map;
import java.util.function.Function;

import static com.pkb.PKBConstants.EARLIEST_ACCEPTED_YEAR_OF_BIRTH;
import static io.vavr.collection.Stream.rangeClosed;
import static io.vavr.collection.Stream.rangeClosedBy;
import static java.util.function.Function.identity;

public class CompositeDatePartsValueProvider {
    private final DateTimeService dateTimeService;

    private static final Map<String, String> DAYS = rangeClosed(1, 31)
            .map(dayIndex -> Integer.toString(dayIndex))
            .toLinkedMap(identity(), identity())
            .toJavaMap();


    private static final Map<Integer, String> MONTHS = ImmutableMap.<Integer, String>builder()
            .put(1, "global.txt.january")
            .put(2, "global.txt.february")
            .put(3, "global.txt.march")
            .put(4, "global.txt.april")
            .put(5, "global.txt.may")
            .put(6, "global.txt.june")
            .put(7, "global.txt.july")
            .put(8, "global.txt.august")
            .put(9, "global.txt.september")
            .put(10, "global.txt.october")
            .put(11, "global.txt.november")
            .put(12, "global.txt.december")
            .build();

    public CompositeDatePartsValueProvider(@NotNull DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    public Map<String, String> getDays() {
        return DAYS;
    }

    public Map<String, String> getMonths() {
        return getMonths(identity());
    }

    public Map<String, String> getMonths(Function<String, String> translator) {
        return rangeClosed(1, 12)
                .toLinkedMap(index -> Integer.toString(index), index -> translator.apply(MONTHS.get(index)))
                .toJavaMap();
    }

    public Map<String, String> getYears() {
        int currentYear = dateTimeService.today().getYear();
        return rangeClosedBy(currentYear, EARLIEST_ACCEPTED_YEAR_OF_BIRTH, -1)
                .map(dayIndex -> Integer.toString(dayIndex))
                .toLinkedMap(identity(), identity())
                .toJavaMap();
    }
}
