package com.pkb.action.registration.letter;

import com.pkb.dto.LetterInvitationDTO;
import com.pkb.entities.enums.Route;
import com.pkb.patient.PatientActivationManager;
import io.vavr.control.Option;
import org.apache.struts.beanvalidation.constraints.ValidationGroup;

import javax.inject.Inject;
import javax.validation.constraints.Email;
import javax.validation.constraints.NotBlank;
import javax.validation.groups.Default;

public class FinishLetterInvitedPatientActivationAction extends LetterInvitationBaseAction {
    private static final String VERIFICATION_CODE = "verificationCode";
    private static final String LOGIN_SUBMIT = "loginSubmit";

    @NotBlank(message = "verification.code.must.not.be.blank")
    private String verificationCode;
    private String clue;

    @NotBlank(message = "email.cannot.be.blank", groups = ValidationGroups.FormSubmission.class)
    @Email(message = "email.is.not.valid", groups = ValidationGroups.FormSubmission.class)
    private String j_username;

    @NotBlank(message = "password.cannot.be.blank", groups = ValidationGroups.FormSubmission.class)
    private String j_password;

    private Long inviteeId;

    @Inject
    protected PatientActivationManager patientActivationManager;

    public String displayForm() {
        recordValidationErrors();
        if (getFieldErrors().containsKey(VERIFICATION_CODE)) {
            promoteFieldLevelToActionLevelError(VERIFICATION_CODE);
            return ACTIVATION_ERROR;
        }

        lookupBrandDetails(clue);
        LetterInvitationDTO invitation = patientActivationManager.findInvitationForActivationByVerificationCode(verificationCode);
        if(invitation.isInvalidated()) {
            j_username = invitation.findClaimedEmail().map(com.pkb.datamodel.Email::address).getOrNull();
            return LOGIN;

        } else {
            prePopulateEmail(invitation.getClaimedEmail());
            inviteeId = patientActivationManager.getInvitedPersonIdByVerificationCode(verificationCode).orElse(null);

            return INPUT;
        }
    }

    @ValidationGroup(value = {Default.class, ValidationGroups.FormSubmission.class})
    public String activatePatient() {
        recordValidationErrors();
        if (isRequestNotValid()) {
            return displayForm();
        }

        patientActivationManager.verifyEmail(getEHRRequestContext().withAccessRoute(Route.WEBAPP), getVerificationCode(), j_password.toCharArray());
        lookupBrandDetails(clue);
        return LOGIN_SUBMIT;
    }

    public String getJ_username() {
        return j_username;
    }

    public void setJ_username(String j_username) {
        this.j_username = j_username;
    }

    public String getJ_password() {
        return j_password;
    }

    public void setJ_password(String j_password) {
        this.j_password = j_password;
    }

    public String getVerificationCode() {
        return verificationCode;
    }

    public void setVerificationCode(String verificationCode) {
        this.verificationCode = verificationCode;
    }

    public String getClue() {
        return clue;
    }

    public void setClue(String clue) {
        this.clue = clue;
    }

    private void prePopulateEmail(com.pkb.datamodel.Email emailUsedForActivation) {
        setJ_username(Option.of(emailUsedForActivation).map(com.pkb.datamodel.Email::address).getOrNull());
    }

    public Long getInviteeId() {
        return inviteeId;
    }

    public void setInviteeId(Long inviteeId) {
        this.inviteeId = inviteeId;
    }
}
