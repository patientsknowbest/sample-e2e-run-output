package com.pkb.action.registration.letter;

import com.pkb.patient.PatientActivationManager;
import com.pkb.util.validation.stereotypes.Number;
import org.slf4j.Logger;

import javax.inject.Inject;
import java.lang.invoke.MethodHandles;

import com.pkb.datamodel.Email;
import static org.slf4j.LoggerFactory.getLogger;


public class InvitationCodeAcceptedAction extends LetterInvitationBaseAction {
    private static final Logger LOGGER = getLogger(MethodHandles.lookup().lookupClass());

    private static final String INVITATION_CLUE = "invitationClue";

    // In-Out Parameters
    @Number(message = "invitation.clue.might.not.be.empty")
    private String invitationClue;
    private String clue;
    private Long inviteeId;

    // Out Parameters
    private Email email;
    @Inject
    private PatientActivationManager patientActivationManager;

    public String verificationCodeSent() {
        if (getFieldErrors().containsKey(INVITATION_CLUE)) {
            LOGGER.error("Invalid Invitation ID in request: invitationClue=[{}]. Cannot display Letter invitation Step 1 confirmation page.", invitationClue);
            promoteFieldLevelToActionLevelError(INVITATION_CLUE);
            return ACTIVATION_ERROR;
        }

        lookupBrandDetails(clue);
        email = patientActivationManager.findLetterInvitationById(Long.parseLong(invitationClue)).getClaimedEmail();

        return SUCCESS;
    }

    public String getInvitationClue() {
        return invitationClue;
    }

    public void setInvitationClue(String invitationClue) {
        this.invitationClue = invitationClue;
    }

    public String getClue() {
        return clue;
    }

    public void setClue(String clue) {
        this.clue = clue;
    }

    public Email getEmail() {
        return email;
    }

    public Long getInviteeId() {
        return inviteeId;
    }

    public void setInviteeId(Long inviteeId) {
        this.inviteeId = inviteeId;
    }
}
