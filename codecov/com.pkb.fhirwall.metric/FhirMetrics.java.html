package com.pkb.fhirwall.metric;

import io.prometheus.client.Counter;

public class FhirMetrics {

    // total requests by route
    private static final Counter FHIRWALL_REQUEST_COUNTER = Counter.build()
            .name("pkb_fhirwall_requests")
            .labelNames("route")
            .help("Requests gone through FhirController")
            .register();

    // total errors by type
    private static final Counter FHIRWALL_ERROR_COUNTER = Counter.build()
            .name("pkb_fhirwall_errors")
            .labelNames("errorType")
            .help("Number of Errors in FhirController by errorType")
            .register();

    public void incrementMetadataCounter() {
        FHIRWALL_REQUEST_COUNTER.labels("metadata").inc();
    }

    public void incrementReadResourceCounter() {
        FHIRWALL_REQUEST_COUNTER.labels("read_resource").inc();
    }

    public void incrementBadRequestCounter() {
        FHIRWALL_ERROR_COUNTER.labels("bad_request").inc();
    }

    public void incrementUnauthorizedCounter() {
        FHIRWALL_ERROR_COUNTER.labels("unauthorized").inc();
    }

    public void incrementNotFoundCounter() {
        FHIRWALL_ERROR_COUNTER.labels("not_found").inc();
    }
}
