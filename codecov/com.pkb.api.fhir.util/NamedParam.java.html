package com.pkb.api.fhir.util;

import io.vavr.collection.List;
import org.jetbrains.annotations.NotNull;

import javax.annotation.Nullable;
import java.util.Objects;

import static io.vavr.API.List;

public class NamedParam<T> {
    private final String name;
    private final T param;
    private final List<String> originalValues;

    public NamedParam(@NotNull String name, @Nullable T param, @Nullable String originalValue) {
        this(name, param, List(originalValue));
    }

    public NamedParam(@NotNull String name, @Nullable T param, @Nullable String[] originalValues) {
        this(name, param, originalValues == null ? List() : List(originalValues));
    }

    public NamedParam(@NotNull String name, @Nullable T param, @NotNull List<String> originalValues) {
        this.name = name;
        this.param = param;
        this.originalValues = originalValues.filter(Objects::nonNull);
    }

    public String getName() {
        return name;
    }

    public @Nullable T getParam() {
        return param;
    }

    public List<String> getOriginalValues() {
        return originalValues;
    }

    public boolean successfullyParsed() {
        return param != null;
    }
}
