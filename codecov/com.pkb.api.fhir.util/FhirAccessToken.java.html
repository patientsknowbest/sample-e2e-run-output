package com.pkb.api.fhir.util;

import ca.uhn.fhir.rest.api.server.RequestDetails;
import ca.uhn.fhir.rest.server.exceptions.AuthenticationException;
import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import ca.uhn.fhir.rest.server.exceptions.InvalidRequestException;
import io.vavr.control.Either;
import org.hl7.fhir.dstu3.model.CodeableConcept;
import org.hl7.fhir.dstu3.model.Coding;
import org.hl7.fhir.dstu3.model.OperationOutcome;

import static com.pkb.api.fhir.CanonicalCodeSystemURLs.responseCode;
import static com.pkb.api.fhir.interceptors.AuthenticationExtractingInterceptor.USER_DATA_AUTHN_COOKIE;
import static com.pkb.api.fhir.interceptors.AuthenticationExtractingInterceptor.USER_DATA_AUTHN_HEADER;
import static com.pkb.domain.criteria.OperationOutcomeFactory.operationOutcome;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity.ERROR;

public class FhirAccessToken {

    private static final String AUTHENTICATION_SCHEMA = "bearer";

    public static Either<BaseServerResponseException, String> extractAccessToken(RequestDetails request) {
        String header = (String)request.getUserData().get(USER_DATA_AUTHN_HEADER);
        String tokenFromHeader = null;
        if (header != null) {
            if (header.toLowerCase().startsWith(AUTHENTICATION_SCHEMA)) {
                tokenFromHeader = header.substring(AUTHENTICATION_SCHEMA.length()).trim();
            } else {
                AuthenticationException exc = new AuthenticationException("invalid Authorization header");
                exc.setOperationOutcome(badAuthentication());
                return Either.left(exc);
            }
        }
        String tokenFromCookie = (String) request.getUserData().get(USER_DATA_AUTHN_COOKIE);
        if (!(tokenFromHeader == null || tokenFromCookie == null)) {
            return Either.left(new InvalidRequestException("the Authorization header xor the ui-exchange cookie should be passed"));
        } else if (tokenFromCookie != null) {
            return Either.right(tokenFromCookie);
        } else if (tokenFromHeader != null) {
            return Either.right(tokenFromHeader);
        } else {
            AuthenticationException exc = new AuthenticationException("no authentication provided");
            exc.setOperationOutcome(badAuthentication());
            return Either.left(exc);
        }
    }

    public static OperationOutcome badAuthentication() {
        return createErrorOutcome(
                "BAD_AUTHENTICATION",
                "Valid access token not provided",
                "You did not provide a valid access token.");
    }

    public static OperationOutcome createErrorOutcome(String code, String display, String diagnostics) {
        OperationOutcome operationOutcome = operationOutcome();
        operationOutcome.addIssue()
                .setSeverity(ERROR)
                .setDiagnostics(diagnostics)
                .setCode(OperationOutcome.IssueType.SECURITY)
                .setDetails(new CodeableConcept()
                        .addCoding(new Coding()
                                .setCode(code)
                                .setSystem(responseCode())
                                .setDisplay(display)));
        return operationOutcome;
    }

}
