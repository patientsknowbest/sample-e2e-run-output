package com.pkb.api.fhir.util;

import ca.uhn.fhir.rest.param.ReferenceParam;
import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import ca.uhn.fhir.rest.server.exceptions.InternalErrorException;
import ca.uhn.fhir.rest.server.exceptions.InvalidRequestException;
import ca.uhn.fhir.rest.server.exceptions.ResourceNotFoundException;
import io.vavr.collection.List;
import io.vavr.control.Either;
import org.hl7.fhir.dstu3.model.DateType;
import org.hl7.fhir.dstu3.model.IdType;
import org.hl7.fhir.dstu3.model.OperationOutcome;
import org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity;
import org.hl7.fhir.dstu3.model.OperationOutcome.OperationOutcomeIssueComponent;
import org.hl7.fhir.dstu3.model.ResourceType;
import org.jetbrains.annotations.NotNull;

import java.time.LocalDate;
import java.util.UUID;

import static com.pkb.domain.criteria.OperationOutcomeFactory.operationOutcome;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;
import static java.lang.String.format;
import static java.util.stream.Collectors.joining;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity.ERROR;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity.WARNING;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueType.NOTFOUND;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueType.PROCESSING;

@FunctionalInterface
interface ServerExceptionConstructor {
    BaseServerResponseException createServerException(OperationOutcome outcome, String message);
}

public final class Fhirs {
    private static final String PREFIX = "urn:uuid:";

    private Fhirs() {
    }

    public static void checkReferenceType(@NotNull ReferenceParam reference, @NotNull ResourceType typeWanted) {
        String typeWantedAsString = typeWanted.toString();
        if (!reference.hasResourceType()) {
            throw new InvalidRequestException(format("ReferenceParam expected resource-type=[%s] but got none", typeWantedAsString));
        }
        if (!reference.getResourceType().equals(typeWantedAsString)) {
            throw new InvalidRequestException(format("ReferenceParam expected resource-type=[%s] but got [%s]", typeWantedAsString, reference.getResourceType()));
        }
    }

    /**
     * Either converts supplied value into a {@link UUID} or raises an {@link InvalidRequestException}.
     *
     * @param id
     *            {@link IdType} containing String to be converted into a UUID.
     * @return InvalidRequestException when input is not a valid {@link UUID} {@link OperationOutcome} with severity set to
     *         {@link IssueSeverity#ERROR ERROR}
     */
    public static @NotNull Either<OperationOutcome, UUID> uuid(@NotNull IdType id) {
        return uuid(id.getIdPart());
    }

    /**
     * Either converts supplied value into a {@link UUID} or raises an {@link InvalidRequestException}.
     *
     * @param id
     *            {@link ReferenceParam} containing String to be converted into a UUID.
     * @return InvalidRequestException when input is not a valid {@link UUID} {@link OperationOutcome} with severity set to
     *         {@link IssueSeverity#ERROR ERROR}
     */
    public static @NotNull Either<OperationOutcome, UUID> uuid(@NotNull ReferenceParam id) {
        return uuid(id.getIdPart());
    }

    /**
     * Either converts supplied value into a {@link UUID} or raises an {@link InvalidRequestException}.
     *
     * @param id
     *            String to be converted into a UUID.
     * @return InvalidRequestException when input is not a valid {@link UUID} {@link OperationOutcome} with severity set to
     *         {@link IssueSeverity#ERROR ERROR}
     */
    public static @NotNull Either<OperationOutcome, UUID> uuid(@NotNull String id) {
        return uuid(id, ERROR);
    }

    public static @NotNull Either<OperationOutcome, UUID> uuid(@NotNull String id, @NotNull IssueSeverity severity) {
        try {
            return right(UUID.fromString(id));
        } catch (IllegalArgumentException ignored) {
            String diagnostics = format("Supplied value=[%s] is not a valid UUID", id);
            return left(operationOutcome()
                    .addIssue(new OperationOutcomeIssueComponent()
                            .setSeverity(severity)
                            .setCode(PROCESSING)
                            .setDiagnostics(diagnostics)));
        }
    }

    public static @NotNull DateType toFhirLocalDate(@NotNull LocalDate localDate) {
        return new DateType(localDate.getYear(), localDate.getMonth().getValue() - 1, localDate.getDayOfMonth());
    }

    public static @NotNull BaseServerResponseException invalidRequestExceptionConstructor(@NotNull OperationOutcome outcome, @NotNull String message) {
        return new InvalidRequestException(message, outcome);
    }

    public static @NotNull BaseServerResponseException internalErrorExceptionConstructor(@NotNull OperationOutcome outcome, @NotNull String message) {
        return new InternalErrorException(message, outcome);
    }

    public static @NotNull BaseServerResponseException resourceNotFoundExceptionConstructor(@NotNull OperationOutcome outcome, @NotNull String message) {
        return new ResourceNotFoundException(message, outcome);
    }

    public static @NotNull BaseServerResponseException serverResponseException(List<OperationOutcomeIssueComponent> issues, ServerExceptionConstructor constructor) {
        OperationOutcome outcome = operationOutcome();
        issues.forEach(outcome::addIssue);
        String message = outcome.getIssue().stream()
                .map(OperationOutcomeIssueComponent::getDiagnostics)
                .collect(joining(","));
        return constructor.createServerException(outcome, message);
    }

    public static @NotNull BaseServerResponseException invalidRequestException(Issues issues) {
        return serverResponseException(issues.get(), Fhirs::invalidRequestExceptionConstructor);
    }

    public static @NotNull BaseServerResponseException internalErrorException(Issues issues) {
        return serverResponseException(issues.get(), Fhirs::internalErrorExceptionConstructor);
    }

    public static @NotNull BaseServerResponseException resourceNotFoundException(Issues issues) {
        return serverResponseException(issues.get(), Fhirs::resourceNotFoundExceptionConstructor);
    } 

    public static @NotNull BaseServerResponseException notFoundDueToInvalidUUIDException(IdType id) {
        OperationOutcome outcome = operationOutcome()
                .addIssue(new OperationOutcomeIssueComponent()
                        .setSeverity(ERROR)
                        .setCode(NOTFOUND)
                        .setDiagnostics("Could not find " + id.getResourceType() + " with ID=[" + id.getIdPart() + "]"))
                .addIssue(new OperationOutcomeIssueComponent()
                        .setSeverity(WARNING)
                        .setCode(PROCESSING)
                        .setDiagnostics("Not a valid UUID: {" + id.getIdPart() + "}"));
        return new ResourceNotFoundException("Not a valid UUID: {" + id.getIdPart() + "}", outcome);
    }
}
