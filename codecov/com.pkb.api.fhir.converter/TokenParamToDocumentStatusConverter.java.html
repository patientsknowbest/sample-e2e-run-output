package com.pkb.api.fhir.converter;

import ca.uhn.fhir.rest.param.TokenParam;
import com.pkb.api.fhir.documentreference.DocumentReferenceDocumentStatus;
import com.pkb.api.fhir.util.Issues;
import com.pkb.api.fhir.util.NamedParam;
import io.vavr.control.Option;
import io.vavr.control.Validation;
import org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity;
import org.jetbrains.annotations.NotNull;

import java.util.stream.Stream;

import static com.pkb.api.fhir.CanonicalCodeSystemURLs.documentStatus;
import static com.pkb.api.fhir.util.Issues.errorFrom;
import static com.pkb.api.fhir.util.Issues.issueFrom;
import static io.vavr.control.Validation.invalid;
import static io.vavr.control.Validation.valid;
import static java.lang.String.format;
import static java.util.function.Function.identity;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class TokenParamToDocumentStatusConverter {

    public @NotNull Validation<Issues, Option<DocumentReferenceDocumentStatus>> convert(@NotNull NamedParam<TokenParam> tokenParam) {
        if (!tokenParam.successfullyParsed()) {
            return valid(Option.none());
        } else if (tokenParam.getParam().getModifier() != null || tokenParam.getParam().getMissing() != null) {
            return invalid(errorFrom(format("Modifier [%s] is not supported for parameter [%s].", tokenParam.getParam().getModifier() != null ? tokenParam.getParam().getModifier() : "MISSING", tokenParam.getName())));
        }

        return Validation.combine(validateValue(tokenParam.getParam().getValue()), validateSystem(tokenParam.getParam().getSystem()))
                .ap((maybeDocumentStatus, maybeSystem) -> Validation.<Issues, Option<DocumentReferenceDocumentStatus>> valid(maybeDocumentStatus))
                .mapError(Issues::flatten)
                .fold(Validation::invalid, identity());
    }

    private Validation<Issues, Option<DocumentReferenceDocumentStatus>> validateValue(String value) {
        if (isNotBlank(value)) {
            return Stream.of(DocumentReferenceDocumentStatus.values())
                    .filter(documentStatus -> documentStatus.name().equalsIgnoreCase(value))
                    .findFirst()
                    .map(documentStatus -> Validation.<Issues, Option<DocumentReferenceDocumentStatus>> valid(Option.of(documentStatus)))
                    .orElseGet(() -> invalid(issueFrom(format("The document-status does not support the value [%s].", value), IssueSeverity.INFORMATION)));
        }
        return valid(Option.none());
    }

    private Validation<Issues, Option<String>> validateSystem(String system) {
        if (system == null || documentStatus().equals(system)) {
            return valid(Option.of(system));
        }
        return invalid(issueFrom(format("System does not support the value [%s].", system), IssueSeverity.INFORMATION));
    }

}
