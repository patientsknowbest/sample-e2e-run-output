package com.pkb.api.fhir.converter;

import ca.uhn.fhir.rest.param.TokenParam;
import com.pkb.api.fhir.diagnosticreport.DiagnosticReportCategory;
import com.pkb.api.fhir.util.Issues;
import com.pkb.api.fhir.util.NamedParam;
import io.vavr.control.Option;
import io.vavr.control.Validation;
import org.jetbrains.annotations.NotNull;

import java.util.stream.Stream;

import static com.pkb.api.fhir.CanonicalCodeSystemURLs.diagnosticReportCategory;
import static com.pkb.api.fhir.util.Issues.errorFrom;
import static io.vavr.control.Validation.invalid;
import static io.vavr.control.Validation.valid;
import static java.lang.String.format;
import static java.util.function.Function.identity;
import static org.apache.commons.lang3.StringUtils.isBlank;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class TokenParamToDiagnosticReportCategoryConverter {

    public @NotNull Validation<Issues, Option<DiagnosticReportCategory>> convert(@NotNull NamedParam<TokenParam> tokenParam) {
        if (!tokenParam.successfullyParsed()) {
            return valid(Option.none());
        }

        return Validation.combine(validateCode(tokenParam.getParam().getValue()), validateSystem(tokenParam.getParam().getSystem()))
                .ap(this::validate)
                .mapError(Issues::flatten)
                .fold(Validation::invalid, identity());
    }

    private Validation<Issues, Option<DiagnosticReportCategory>> validateCode(String code) {
        if (isNotBlank(code)) {
            return Stream.of(DiagnosticReportCategory.values())
                    .filter(category -> category.name().equalsIgnoreCase(code))
                    .findFirst()
                    .map(category -> Validation.<Issues, Option<DiagnosticReportCategory>>valid(Option.of(category)))
                    .orElseGet(() -> invalid(errorFrom(format("DiagnosticReport category does not support the value [%s].", code))));
        }
        return valid(Option.none());
    }

    private Validation<Issues, Option<String>> validateSystem(String system) {
        if (system == null || diagnosticReportCategory().equals(system)) {
            return valid(Option.of(system));
        }
        if (isBlank(system)) {
            return invalid(errorFrom("System does not support an empty value."));
        }
        return invalid(errorFrom(format("System does not support the value [%s].", system)));
    }

    private Validation<Issues, Option<DiagnosticReportCategory>> validate(@NotNull Option<DiagnosticReportCategory> maybeDiagnosticReport,
                                                                            @NotNull Option<String> maybeSystem) {

        if (maybeDiagnosticReport.isDefined() || maybeSystem.isDefined()) {
            return Validation.valid(maybeDiagnosticReport);
        } else {
            return Validation.invalid(errorFrom("The diagnostic report category and the system are missing."));
        }
    }
}
