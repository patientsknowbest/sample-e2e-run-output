package com.pkb.api.fhir.converter;

import ca.uhn.fhir.rest.param.StringParam;
import com.pkb.api.fhir.util.Issues;
import io.vavr.control.Validation;
import org.hl7.fhir.dstu3.model.OperationOutcome;
import org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity;

import java.util.UUID;

import static io.vavr.API.List;
import static io.vavr.control.Validation.invalid;
import static java.lang.String.format;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity.ERROR;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueType.PROCESSING;

public class UrnUUIDConverter {
    private static final String PREFIX = "urn:uuid:";
    private static final String VALUE = "value";

    private final UUIDConverter uuidConverter;

    public UrnUUIDConverter(UUIDConverter uuidConverter) {
        this.uuidConverter = uuidConverter;
    }

    public Validation<Issues, UUID> urnUuid(String value) {
        return urnUuid(value, ERROR);
    }

    public Validation<Issues, UUID> urnUuid(String value, IssueSeverity severity) {
        return value.toLowerCase().startsWith(PREFIX) ? //
                uuidConverter.uuid(value.substring(PREFIX.length()), severity) : //
                invalid(Issues.of(List(new OperationOutcome.OperationOutcomeIssueComponent()
                        .setSeverity(severity)
                        .setCode(PROCESSING)
                        .setDiagnostics(format("Supplied value=[%s] is not a valid URN", value)))));
    }

    public Validation<Issues, UUID> urnUuid(StringParam stringParam) {
        return urnUuid(stringParam, ERROR);
    }

    public Validation<Issues, UUID> urnUuid(StringParam stringParam, IssueSeverity severity) {
        return stringParam.isExact() ?
                urnUuid(stringParam.getValueNotNull().trim(), severity) :
                invalid(Issues.of(
                    List(new OperationOutcome.OperationOutcomeIssueComponent()
                        .setSeverity(ERROR)
                        .setCode(PROCESSING)
                        .setDiagnostics(format("Search parameter [%s] must be requested with modifier [exact].", VALUE)))));
    }
}
