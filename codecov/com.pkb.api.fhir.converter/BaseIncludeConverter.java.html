package com.pkb.api.fhir.converter;

import ca.uhn.fhir.model.api.Include;
import com.pkb.api.fhir.util.IncludeResource;
import com.pkb.api.fhir.util.Issues;
import io.vavr.Value;
import io.vavr.collection.List;
import io.vavr.control.Either;
import io.vavr.control.Validation;

public abstract class BaseIncludeConverter {

    public abstract Validation<Issues, IncludeResource> convert(Include include);

    public Validation<Issues, List<IncludeResource>> convert(List<Include> includes) {
        // Annoyingly, Validation.sequence accepts the wrong type parameter, so go to Either & back...
        return Validation.fromEither(Either.sequence(includes
                .map(this::convert)
                .map(Validation::toEither))
                .mapLeft(Issues::combine))
                .map(Value::toList);
    }
}
