package com.pkb.api.fhir.converter;

import ca.uhn.fhir.rest.param.TokenParam;
import com.pkb.api.fhir.observation.ObservationCategory;
import com.pkb.api.fhir.util.Issues;
import com.pkb.api.fhir.util.NamedParam;
import io.vavr.control.Option;
import io.vavr.control.Validation;
import org.jetbrains.annotations.NotNull;

import java.util.stream.Stream;

import static com.pkb.api.fhir.CanonicalCodeSystemURLs.observationCategory;
import static com.pkb.api.fhir.util.Issues.errorFrom;
import static io.vavr.control.Validation.invalid;
import static io.vavr.control.Validation.valid;
import static java.lang.String.format;
import static java.util.function.Function.identity;
import static org.apache.commons.lang3.StringUtils.isBlank;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class TokenParamToObservationCategoryConverter {

    public @NotNull Validation<Issues, Option<ObservationCategory>> convert(@NotNull NamedParam<TokenParam> tokenParam) {
        if (!tokenParam.successfullyParsed()) {
            return valid(Option.none());
        }
        return Validation.combine(validateCode(tokenParam.getParam().getValue()), validateSystem(tokenParam.getParam().getSystem()))
                .ap((maybeObservation, maybeSystem) -> {
                    if (maybeObservation.isDefined() || maybeSystem.isDefined()) {
                        return Validation.<Issues, Option<ObservationCategory>> valid(maybeObservation);
                    } else {
                        return Validation.<Issues, Option<ObservationCategory>> invalid(errorFrom("The observation category and the system are missing."));
                    }
                })
                .mapError(Issues::flatten)
                .fold(Validation::invalid, identity());
    }

    private Validation<Issues, Option<ObservationCategory>> validateCode(String code) {
        if (isNotBlank(code)) {
            return Stream.of(ObservationCategory.values())
                    .filter(category -> category.name().equalsIgnoreCase(code))
                    .findFirst()
                    .map(category -> Validation.<Issues, Option<ObservationCategory>> valid(Option.of(category)))
                    .orElseGet(() -> invalid(errorFrom(format("Observation category does not support the value [%s].", code))));
        }
        return valid(Option.none());
    }

    private Validation<Issues, Option<String>> validateSystem(String system) {
        if (system == null || observationCategory().equals(system)) {
            return valid(Option.of(system));
        }
        if (isBlank(system)) {
            return invalid(errorFrom(format("System does not support an empty value.")));
        }
        return invalid(errorFrom(format("System does not support the value [%s].", system)));
    }

}
