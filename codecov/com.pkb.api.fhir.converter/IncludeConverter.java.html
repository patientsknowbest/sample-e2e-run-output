package com.pkb.api.fhir.converter;

import ca.uhn.fhir.model.api.Include;
import com.pkb.api.fhir.util.IncludeResource;
import com.pkb.api.fhir.util.Issues;
import io.vavr.control.Validation;

import static com.pkb.api.fhir.util.Issues.errorFrom;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.control.Validation.invalid;
import static io.vavr.control.Validation.valid;

public class IncludeConverter extends BaseIncludeConverter {
    public static final String PARAM_NAME = "_include";
    public static final String QUESTIONNAIRE_RESPONSE_SUBJECT_PATIENT_STR = "QuestionnaireResponse:subject:Patient";
    public static final String QUESTIONNAIRE_RESPONSE_BASED_ON_STR = "QuestionnaireResponse:basedOn:ProcedureRequest";
    private static final Include QUESTIONNAIRE_SUBJECT_PATIENT = new Include(QUESTIONNAIRE_RESPONSE_SUBJECT_PATIENT_STR);
    private static final Include QUESTIONNAIRE_BASED_ON_PROCEDURE_REQUEST = new Include(QUESTIONNAIRE_RESPONSE_BASED_ON_STR);

    @Override
    public Validation<Issues, IncludeResource> convert(Include include) {
        return Match(include).of(
                Case($(QUESTIONNAIRE_SUBJECT_PATIENT), valid(IncludeResource.SUBJECT_PATIENT)),
                Case($(QUESTIONNAIRE_BASED_ON_PROCEDURE_REQUEST), valid(IncludeResource.BASED_ON_PROCEDURE_REQUEST)),
                // HAPI actually validates these for us, and returns a correctly formatted error
                // Leaving this here in case that stops happening for some reason.
                Case($(), invalid(errorFrom("Unsupported parameter " + include.getValue()))));
    }
}
