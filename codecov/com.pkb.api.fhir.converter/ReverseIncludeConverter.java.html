package com.pkb.api.fhir.converter;

import ca.uhn.fhir.model.api.Include;
import com.pkb.api.fhir.util.IncludeResource;
import com.pkb.api.fhir.util.Issues;
import io.vavr.control.Validation;

import static com.pkb.api.fhir.util.IncludeResource.PROVENANCE;
import static com.pkb.api.fhir.util.Issues.errorFrom;
import static io.vavr.control.Validation.invalid;
import static io.vavr.control.Validation.valid;

public class ReverseIncludeConverter extends BaseIncludeConverter {
    public static final String PARAM_NAME = "_revinclude";
    public static final String PROVENANCE_TARGET_STR = "Provenance:target";
    private static final Include PROVENANCE_TARGET = new Include(PROVENANCE_TARGET_STR);

    @Override
    public Validation<Issues, IncludeResource> convert(Include include) {
        if (include.equals(PROVENANCE_TARGET)) {
            return valid(PROVENANCE);
        } else {
            // HAPI actually validates these for us, and returns a correctly formatted error
            // Leaving this here in case that stops happening for some reason.
            return invalid(errorFrom("Unsupported parameter " + include.getValue()));
        }
    }

}
