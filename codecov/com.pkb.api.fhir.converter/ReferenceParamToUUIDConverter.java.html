package com.pkb.api.fhir.converter;

import ca.uhn.fhir.rest.param.ReferenceParam;
import com.pkb.api.fhir.util.Issues;
import com.pkb.api.fhir.util.NamedParam;
import io.vavr.control.Option;
import io.vavr.control.Validation;
import org.hl7.fhir.dstu3.model.ResourceType;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.pkb.api.fhir.util.Issues.errorFrom;
import static io.vavr.control.Validation.invalid;
import static io.vavr.control.Validation.valid;
import static java.lang.String.format;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity.ERROR;
import static org.hl7.fhir.dstu3.model.OperationOutcome.IssueSeverity.WARNING;

public class ReferenceParamToUUIDConverter {

    public @NotNull Validation<Issues, Option<UUID>> convert(@NotNull NamedParam<ReferenceParam> reference, @NotNull ResourceType typeWanted) {
        if (!reference.successfullyParsed()) {
            return valid(Option.none());
        }
        return convertRequired(reference, typeWanted).map(Option::of);
    }

    public @NotNull Validation<Issues, UUID> convertRequired(@NotNull NamedParam<ReferenceParam> reference, @NotNull ResourceType typeWanted) {
        Issues issues = checkReferenceType(reference, typeWanted);
        if (issues.isEmpty()) {
            return toUUID(reference.getParam().getIdPart());
        } else {
            return invalid(issues);
        }
    }

    public static  @NotNull Validation<Issues, UUID> toUUID(@NotNull String id) {
        try {
            return valid(UUID.fromString(id));
        } catch (IllegalArgumentException ignored) {
            String diagnostics = format("Supplied value=[%s] is not a valid UUID", id);
            return invalid(Issues.issueFrom(diagnostics, WARNING));
        }
    }

    private Issues checkReferenceType(NamedParam<ReferenceParam> referenceParam, ResourceType typeWanted) {
        Issues issues = Issues.empty();
        ReferenceParam reference = referenceParam.getParam();
        String typeWantedAsString = typeWanted.toString();
        if (reference.getMissing() != null) {
            return errorFrom(format("Modifier [MISSING] is not supported for parameter [%s].", referenceParam.getName()));
        }

        if (!reference.hasResourceType()) {
            issues = issues.add(format("ReferenceParam expected resource-type=[%s] but got none", typeWantedAsString), ERROR);
        } else if (!reference.getResourceType().equals(typeWantedAsString)) {
            issues = issues.add(format("ReferenceParam expected resource-type=[%s] but got [%s]", typeWantedAsString, reference.getResourceType()), ERROR);
        }
        return issues;
    }
}
