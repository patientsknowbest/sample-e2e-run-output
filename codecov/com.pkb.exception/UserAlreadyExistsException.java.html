package com.pkb.exception;

import com.pkb.model.PKBPersonDTO;

/**
 * Exception thrown during validation of user email while registering a user
 *
 * @author pravinam
 *
 */
public class UserAlreadyExistsException extends PKBException {

    private static final long serialVersionUID = 1L;

    private PKBPersonDTO user;

    /**
     * Defines UserAlreadyExistsException object with {@link PKBPersonDTO}
     *
     * @param user
     */
    public UserAlreadyExistsException(PKBPersonDTO user) {
        this.user = user;
    }

    /**
     * Defines UserAlreadyExistsException object with error message and {@link PKBPersonDTO}
     *
     * @param message
     * @param user
     */
    public UserAlreadyExistsException(String message, PKBPersonDTO user) {
        super(message);
        this.user = user;
    }

    /**
     * Defines UserAlreadyExistsException object with {@link PKBPersonDTO} and original exceptoin
     *
     * @param cause
     * @param user
     */
    public UserAlreadyExistsException(Throwable cause, PKBPersonDTO user) {
        super(cause);
        this.user = user;
    }

    /**
     * Defines UserAlreadyExistsException object with error message and original exception
     *
     * @param message
     * @param cause
     * @param user
     */
    public UserAlreadyExistsException(String message, Throwable cause,
            PKBPersonDTO user) {
        super(message, cause);
        this.user = user;
    }

    public PKBPersonDTO getUser() {
        return this.user;
    }
}
