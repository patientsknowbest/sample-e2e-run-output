package com.pkb.exception;

/**
 * Represents a general PKB exception
 * This exception is thrown by all service layer classes, which deals with tolven API.
 *
 * @author pravinam
 * @deprecated Usage of PKBException is discouraged, Use appropriate RuntimeException instead.
 */
@Deprecated
public class PKBException extends RuntimeException {

    private static final long serialVersionUID = 1L;

    /**
     * Defines PKBException object with default properties
     */
    public PKBException() {
        super();
    }

    /**
     * Defines PKBException object with error message
     *
     * @param message
     *            error message
     */
    public PKBException(String message) {
        super(message);
    }

    /**
     * Defines PKBException object with original exception
     *
     * @param cause
     */
    public PKBException(Throwable cause) {
        super(cause);
    }

    /**
     * Defines PKBException object with original Exception and error message
     *
     * @param message
     *            error message
     * @param cause
     *            original exception
     */
    public PKBException(String message, Throwable cause) {
        super(message, cause);
    }
}
