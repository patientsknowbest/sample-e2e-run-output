package com.pkb.exception;

public class MessageException extends RuntimeException {

    /**
     * This exception is thrown by the EmailManager while sending emails
     */

    /** The Constant serialVersionUID. */
    private static final long serialVersionUID = -2955826821766922622L;

    /**
     * Defines MessageException object with default properties.
     */
    public MessageException() {
        super();
    }

    /**
     * Defines MessageException object with error message
     *
     * @param message
     *            the message
     */
    public MessageException(String message) {
        super(message);
    }

    /**
     * Defines MessageException object with original exception
     *
     * @param cause
     *            Original Exception object
     *
     */
    public MessageException(Throwable cause) {
        super(cause);
    }

    /**
     * Instantiates a new builds the exception.
     *
     * @param message
     *            error message
     * @param cause
     *            Original Exception object
     */
    public MessageException(String message, Throwable cause) {
        super(message, cause);
    }

}
