package com.pkb.exception;

/**
 * This {@code Exception} indicates that multiple patient identifiers were used together, but where the identifiers match more than one
 * medical record.
 * 
 * Captures the initial match and first conflict -- there may be other conflicts in the data, this only captures the first for debugging
 * 
 * @see UserManager.lookupSinglePatient().
 */
public class ConflictingPatientIdentifiersException extends RuntimeException {

    private static final long serialVersionUID = 1L;
    private long matchedPatientId;
    private long firstConflictPatientId;

    public ConflictingPatientIdentifiersException(String message, long matchedPatientId, long firstConflictPatientId) {
        super(message);
        this.matchedPatientId = matchedPatientId;
        this.firstConflictPatientId = firstConflictPatientId;
    }

    public long getMatchedPatientId() {
        return matchedPatientId;
    }

    public long getFirstConflictPatientId() {
        return firstConflictPatientId;
    }
}
