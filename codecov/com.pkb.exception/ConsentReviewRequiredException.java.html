package com.pkb.exception;

/**
 * This exception is conditionally thrown if an action is executed when a
 * consent review is pending (as set on PKBPersonProperty). Struts will redirect
 * to the review wizard when this exception bubbles up.
 *
 * @author Russ Gray
 */
public class ConsentReviewRequiredException extends PKBException {
    private static final long serialVersionUID = -2753594404612710315L;

    public ConsentReviewRequiredException() {
    }

    public ConsentReviewRequiredException(String message) {
        super(message);
    }

    public ConsentReviewRequiredException(Throwable cause) {
        super(cause);
    }

    public ConsentReviewRequiredException(String message, Throwable cause) {
        super(message, cause);
    }
}
