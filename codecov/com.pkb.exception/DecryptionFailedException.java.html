package com.pkb.exception;

/**
 * Throw this if decryption failed -- usually indicate an attempt at abusing the system, or (more likely) a problem in our
 * code that's trying to link things together.
 *
 * @author robwhelan
 */
public class DecryptionFailedException extends PKBPluginException {

    private static final long serialVersionUID = 1L;

    public DecryptionFailedException(String message) {
        super(message);
    }

    public DecryptionFailedException(String message, Throwable cause) {
        super(message, cause);
    }

}
