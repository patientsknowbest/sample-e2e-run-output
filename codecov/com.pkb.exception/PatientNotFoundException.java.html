package com.pkb.exception;

/**
 * This {@code Exception} indicates that a patient lookup failed to find a patient. It is not always an error for a patient lookup to fail,
 * but this is to be used when the lookup should have succeeded. For example, it is always an error to fail to find a patient by PKB ID;
 * there is no reason a PKB ID-based lookup would be performed if the patient was not expected to be found.
 * 
 * @see UserManager.lookupSinglePatient().
 */
public class PatientNotFoundException extends RuntimeException {

    private static final long serialVersionUID = 1L;

    public PatientNotFoundException(String message) {
        super(message);
    }

    public PatientNotFoundException(String message, Throwable cause) {
        super(message, cause);
    }
}
