package com.pkb.exception;

import java.sql.SQLException;

/**
 * Represents a general exception thrown by beans in PKB plugin
 * Also; if the "cause" is an SQLException, don't save as cause -- it'll be lost in deserialization.
 * Extract that exception's info.
 *
 * @author pravinam
 * @author robwhelan
 */
@Deprecated
public class PKBPluginException extends RuntimeException {

    private static final long serialVersionUID = 1L;

    public PKBPluginException(String message) {
        super(message);
    }

    public PKBPluginException(String message, Throwable cause) {
        super(message, makeCauseSafe(cause));
    }

    private static Throwable makeCauseSafe(Throwable cause) {
        if (containsUnsafeCause(cause)) {
            return rewrapUnsafeCause(cause);
        } else {
            return cause;
        }
    }

    /* future fancy version: re-create each exception class properly (reflection) */
    private static Throwable rewrapUnsafeCause(Throwable t) {
        if (t == null) {
            return t;
        }

        // nicely rewrap where we can
        if (t instanceof SQLException) {
            SQLException sqle = (SQLException) t;
            Throwable cause = sqle.getCause();
            SQLException safeSqle = new SQLException("(PKB-rewrapped " + t.getClass().getName() + ") " + sqle.getMessage(),
                    sqle.getSQLState(), sqle.getErrorCode(),
                    rewrapUnsafeCause(cause));
            safeSqle.setStackTrace(sqle.getStackTrace());
            // rewrap next exception as well, recursively...
            safeSqle.setNextException((SQLException) rewrapUnsafeCause(sqle.getNextException()));
            return safeSqle;
        } else {
            // just wrap in a plain Throwable
            Throwable cause = t.getCause();
            Throwable safeT = new Throwable("(PKB-rewrapped " + t.getClass().getName() + ") " + t.getMessage(),
                    rewrapUnsafeCause(cause));
            safeT.setStackTrace(t.getStackTrace());
            return safeT;
        }
    }

    private static boolean containsUnsafeCause(Throwable t) {
        while (t != null) {
            if (t.getClass().getName().equals("org.postgresql.util.PSQLException")) {
                return true;
            }

            // SQLExceptions have their own methods for exception cause chains...
            if (t instanceof SQLException) {
                if (containsUnsafeCause(((SQLException) t).getNextException())) {
                    return true;
                }
            }

            t = t.getCause();
        }
        return false;
    }

}
