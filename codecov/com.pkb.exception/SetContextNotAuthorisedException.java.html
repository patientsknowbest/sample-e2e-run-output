package com.pkb.exception;

/**
 * This {@code Exception} represents an attempt to switch into a context for a
 * user for whom the currently logged in user does not have permission. In
 * practice, the most likely cause is that the user forgot to log out before
 * clicking on a link for a different user.
 */
public class SetContextNotAuthorisedException extends PKBException {

    private static final long serialVersionUID = 1L;

    public SetContextNotAuthorisedException() {
    }

    public SetContextNotAuthorisedException(String msg) {
        super(msg);
    }
}
