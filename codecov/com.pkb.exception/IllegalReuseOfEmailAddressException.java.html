package com.pkb.exception;

/**
 * This exception is thrown when we find multiple PKBPersons with the same
 * email address, when we were expecting to find a maximum of 1.
 *
 * A clinician may have multiple linked personas with the same email, and soon
 * a coord can too. A patient cannot, neither can a PKB employee.
 *
 * @author Russ Gray
 */
public class IllegalReuseOfEmailAddressException extends PKBException {
    private static final long serialVersionUID = -6820080767097331754L;

    public IllegalReuseOfEmailAddressException() {
    }

    public IllegalReuseOfEmailAddressException(String message) {
        super(message);
    }

    public IllegalReuseOfEmailAddressException(Throwable cause) {
        super(cause);
    }

    public IllegalReuseOfEmailAddressException(String message, Throwable cause) {
        super(message, cause);
    }
}
