package com.pkb.exception;

/**
 * Exception thrown when email address of a user is invalid
 *
 * @author pravinam
 *
 */
public class InvalidEmailAddressException extends PKBException {

    /** The Constant serialVersionUID. */
    private static final long serialVersionUID = -2955826821766922622L;

    /**
     * Defined InvalidEmailAddressException Object with default properties
     */
    public InvalidEmailAddressException() {
        super();
    }

    /**
     * Defined InvalidEmailAddressException Object with original exception
     *
     * @param cause
     *            Original exception
     *
     */
    public InvalidEmailAddressException(Throwable cause) {
        super(cause);
    }

    /**
     * Instantiates a new builds the exception.
     *
     * @param message
     *            error message
     * @param cause
     *            origianal exception
     *            the cause
     */
    public InvalidEmailAddressException(String message, Throwable cause) {
        super(message, cause);
    }

}
