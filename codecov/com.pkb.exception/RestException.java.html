package com.pkb.exception;

import javax.ws.rs.WebApplicationException;
import javax.ws.rs.core.Response;

/**
 * Represents a general PKB REST exception
 *
 * @author vorekoya
 */
public class RestException extends WebApplicationException {

    private String additionalMessage = "";

    public RestException() {
        super();
    }

    public RestException(Throwable cause) {
        super(cause);
    }

    public RestException(Throwable cause, String message) {
        super(cause);
        this.additionalMessage = message;
    }

    public RestException(Throwable cause, Response.Status status, String message) {
        super(cause, status);
        this.additionalMessage = message;
    }

    public String getAdditionalMessage() {
        return additionalMessage;
    }

    public void setAdditionalMessage(String message) {
        this.additionalMessage = message;
    }
}
