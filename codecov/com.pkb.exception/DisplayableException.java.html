package com.pkb.exception;

/**
 * Instead of an error page -- these exceptions will show on the (appropriate) dashboard, with a message, warning, or error message
 * (as selected below).
 *
 * You may not have access to getText() when throwing this, so set i18n property & variables here for a display-time call to
 * getText(textProperty, textArgs).
 *
 * Handling: struts.xml catches this exception and redirects to RedirectHomeAction.
 *
 * @author robwhelan
 */
public class DisplayableException extends PKBException {

    private static final long serialVersionUID = 1L;

    public enum DisplayAs {
        message, error
    }

    private DisplayAs displayAs;

    private String textProperty;
    private String[] textArgs;

    public DisplayableException(String textProperty, DisplayAs displayAs) {
        this(textProperty, null/*textArgs*/, displayAs);
    }

    public DisplayableException(String textProperty, String[] textArgs, DisplayAs displayAs) {
        super();
        this.textProperty = textProperty;
        this.textArgs = textArgs;
        this.displayAs = displayAs;
    }

    // add when needed: constructors to pass through  cause and message for logging

    public DisplayAs getDisplayAs() {
        return displayAs;
    }

    public String getTextProperty() {
        return textProperty;
    }

    public String[] getTextArgs() {
        return textArgs;
    }
}
