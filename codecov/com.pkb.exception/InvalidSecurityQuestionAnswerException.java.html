package com.pkb.exception;

/**
 * Exception to be thrown if the security question or answer provided by the user is incorrect
 *
 * @author pravinam
 *
 */
public class InvalidSecurityQuestionAnswerException extends PKBException {

    private static final long serialVersionUID = 1L;

    private long userId;

    public InvalidSecurityQuestionAnswerException(long userId) {
        this.userId = userId;
    }

    public InvalidSecurityQuestionAnswerException(String message, long userId) {
        super(message);
        this.userId = userId;
    }

    public InvalidSecurityQuestionAnswerException(long userId, Throwable throwable) {
        super(throwable);
        this.userId = userId;
    }

    public long getUserId() {
        return userId;
    }

    public void setUserId(long userId) {
        this.userId = userId;
    }

}
