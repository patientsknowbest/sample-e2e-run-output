package com.pkb.action.prescriptions;


import com.pkb.action.BaseAction;
import com.pkb.domain.appointment.EmisAppointmentService;
import com.pkb.user.entity.PKBPerson;
import org.springframework.beans.factory.annotation.Autowired;

import static org.apache.commons.collections4.CollectionUtils.isNotEmpty;

public class PrescriptionsAction extends BaseAction {

    private boolean registeredForGpPrescriptions = false;

    private EmisAppointmentService emisAppointmentService;

    @Override
    public void prepare() throws Exception {
        super.prepare();

        var users = getContextAndLoggedInUsers();
        PKBPerson contextUser = users.contextUser();
        registeredForGpPrescriptions = isRegisteredForPrescriptionOrdering(contextUser);
    }

    private boolean isRegisteredForPrescriptionOrdering(PKBPerson person) {
        return person.isPatient() && isNotEmpty(emisAppointmentService.findEmisRegistrationDetailsByPersonId(getEmisContextHolder()));
    }

    public String getTab() {
        return "treatments";
    }

    public String getSubTab() {
        return "prescriptions";
    }

    public String viewPrescriptions() {
        return SUCCESS;
    }

    public boolean isRegisteredForGpPrescriptions() {
        return registeredForGpPrescriptions;
    }

    @Autowired
    public void setEmisAppointmentService(EmisAppointmentService emisAppointmentService) {
        this.emisAppointmentService = emisAppointmentService;
    }
}
