package com.pkb.domain.labresult;

import com.google.common.base.Preconditions;
import com.pkb.authz.AuthorizationService;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.datamodel.LabResult;
import com.pkb.domain.SynchUploadedDataService;
import com.pkb.domain.SystemUserTypeNotPermittedException;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.criteria.ImmutableCriteriaResult;
import com.pkb.domain.repository.ImmutableLabResultQuery;
import com.pkb.domain.repository.LabResultQuery;
import com.pkb.domain.repository.LabResultRepository;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.pkb.entities.enums.UploadedDataDestination.TEST_RESULT;
import static io.vavr.control.Either.left;

public class LabResultService {
    private final LabResultRepository labResultRepository;
    private final AuthorizationService authorizationService;
    private final SynchUploadedDataService synchUploadedDataService;

    public LabResultService(LabResultRepository labResultRepository, AuthorizationService authorizationService, SynchUploadedDataService synchUploadedDataService) {
        this.labResultRepository = labResultRepository;
        this.authorizationService = authorizationService;
        this.synchUploadedDataService = synchUploadedDataService;
    }

    public @NotNull Either<RuntimeException, CriteriaResult<LabResult>> search(@NotNull AuthorizationData authzData,
                                                                               @NotNull LabResultCriteria criteria) {
        // Sanity check
        authzData.singleTargetData().peek(requestedPerson -> {
            UUID publicIdAuthData = requestedPerson.person().getPublicId();
            UUID publicIdRequested = criteria.getPatientPublicId();
            Preconditions.checkArgument(publicIdAuthData.equals(publicIdRequested),
                    "Requested lab results for %s but have authorization data for %s", publicIdRequested, publicIdAuthData);
        });

        if (authzData.inputs().authenticatedSystemUser().isDefined()) {
            return left(new SystemUserTypeNotPermittedException());
        }

        LabResultQuery query = ImmutableLabResultQuery.builder()
                .authData(authzData)
                .effectiveTimestamp(criteria.getEffectiveTimestamp())
                .resultFilter(criteria.resultFilter())
                .build();

        return authorizationService.authorize(authzData)
                .checkSingleResult(() -> {
                    synchUploadedDataService.synchUploadedData(authzData, TEST_RESULT);
                    return ImmutableCriteriaResult.<LabResult>builder()
                            .addAllPage(labResultRepository.findBy(query, criteria.getPageSize()))
                            .total(labResultRepository.count(query))
                            .build();
        });
    }
}
