package com.pkb.fhirwall.service.resource.wayfinder;

import com.pkb.fhirwall.config.FhirwallConfig;
import org.hl7.fhir.instance.model.api.IBaseResource;
import org.hl7.fhir.r4.model.Appointment;
import org.hl7.fhir.r4.model.Reference;
import org.hl7.fhir.r4.model.ResourceType;
import org.jetbrains.annotations.NotNull;

import java.util.Collection;
import java.util.List;
import java.util.Optional;
import java.util.stream.Collectors;

/**
 * A Wayfinder specific concept: remove Appointments for orgs which are not onboarded to Wayfinder due to IG.
 */
public class AppointmentResourceRemover {

    private final FhirwallConfig config;

    public AppointmentResourceRemover(FhirwallConfig config) {
        this.config = config;
    }

    void applyRemovalRules(Collection<IBaseResource> bundleResources) {
        bundleResources.removeIf(appointment -> !appointmentIsForWayfinderOnboardedTrust((Appointment) appointment));
    }

    private boolean appointmentIsForWayfinderOnboardedTrust(Appointment appointment) {
        return getLocationParticipants(appointment).stream().findFirst().map(participantComponent -> {
            String odsCode = participantComponent.getActor().getIdentifier().getValue();
            return getWayfinderOnboardedOdsCodes().contains(odsCode);
        }).orElse(false); //No location participants == self-entered appointment. DO NOT return these
    }

    @NotNull
    private List<Appointment.AppointmentParticipantComponent> getLocationParticipants(Appointment appointment) {
        return appointment.getParticipant()
                .stream().filter(participant -> {
                    Optional<Reference> actorReference = Optional.ofNullable(participant.getActor());
                    Optional<String> actorType = actorReference.map(Reference::getType);
                    return actorType.map(type -> type.equals(ResourceType.Location.toString())).orElse(false);
                })
                .collect(Collectors.toList());
    }

    private List<String> getWayfinderOnboardedOdsCodes() {
        return config.getWayfinderOnboardedOdsCodes();
    }
}
