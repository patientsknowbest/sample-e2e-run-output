package com.pkb.fhirwall.service.resource.wayfinder;

import com.pkb.fhirwall.service.auth.ImmutableResponse;
import com.pkb.fhirwall.service.auth.Response;
import com.pkb.fhirwall.service.util.BundleService;
import org.hl7.fhir.instance.model.api.IBaseResource;
import org.hl7.fhir.r4.model.Bundle;
import org.hl7.fhir.r4.model.Resource;
import org.hl7.fhir.r4.model.ResourceType;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.List;
import java.util.Objects;
import java.util.Optional;
import java.util.stream.Collectors;

import static com.pkb.fhirwall.service.util.ResourceIdUtil.BASE_RESOURCE_COMPARATOR_ID;

/**
 * Handles the removal of certain elements from the responses we send back based on over-arching business rules,
 * rather than authorization.
 *
 * Splitting this out from authz, enrichment etc. to keep it nice and transparent as to what's happening and why.
 *
 * Bundle lists are immutable (the content of the list itself is not), so to effectively do this you have to construct
 * a new Bundle and add these resources into it as entry components, which is frustrating.
 */
public class ResourceRemovalManager {

    private AppointmentResourceRemover appointmentResourceRemover;
    private BundleService bundleService;

    public ResourceRemovalManager(AppointmentResourceRemover appointmentResourceRemover,
                                  BundleService bundleService) {
        this.appointmentResourceRemover = appointmentResourceRemover;
        this.bundleService = bundleService;
    }

    public Response applyRemovalRules(Response originalResponse) {
        IBaseResource resource = originalResponse.resource();
        if (resource instanceof Bundle) {
            var bundleResources = bundleService.getResourcesFromBundleGroupingByFhirType((Bundle)resource);

            Optional<Response> newResponse = Optional.ofNullable(bundleResources.get(ResourceType.Appointment.toString())).map(appointmentList -> {
                //Apply the removal rules just to appointments for now
                ArrayList<IBaseResource> resourceListCopy = new ArrayList<>(appointmentList);
                appointmentResourceRemover.applyRemovalRules(resourceListCopy);

                //Add all other resources into the now stripped-down resource list to construct the new Bundle
                // Sort because hash maps are not deterministic as to how they bucket their values
                List<String> keySet = new ArrayList<>(bundleResources.keySet());
                Collections.sort(keySet);
                keySet.stream()
                        .filter(fhirType -> !Objects.equals(fhirType, ResourceType.Appointment.toString()))
                        .forEach(fhirTypeNotAppointment -> resourceListCopy.addAll(bundleResources.get(fhirTypeNotAppointment)));

                return ImmutableResponse.builder().resource(createNewBundleForResources((Bundle)originalResponse.resource(), resourceListCopy)).build();
            });
            return newResponse.orElse(originalResponse);
        }
        //No error here, we can pass through this for e.g. CapabilityStatements
        return originalResponse;
    }

    /**
     * As bundles are immutable, we need to construct a new one from scratch.
     * There are some compromises we're making here: pagination will be wonky, so I'm removing it entirely from the
     * new Bundle. This is fine because this removal is a Wayfinder-specific concept, and they never intend to use it.
     */
    private Bundle createNewBundleForResources(Bundle originalBundle, Collection<IBaseResource> resourceListCopy) {
        Bundle newBundle = new Bundle();
        newBundle.setTotal(resourceListCopy.size());
        newBundle.setType(originalBundle.getType());

        var originalBundleEntries = originalBundle.getEntry().stream()
                .collect(Collectors.groupingBy(entry -> entry.getResource().getId()));

        //A given search should return the bundle in a predictable way. Makes testing non-flaky as well.
        // This is possibly not required given we've sorted the key set, but seems like a safe thing to do anyway.
        List<IBaseResource> resourceList = new ArrayList<>(resourceListCopy);
        resourceList.sort(BASE_RESOURCE_COMPARATOR_ID);

        resourceListCopy.forEach(resource -> {
            Bundle.BundleEntryComponent originalEntryComponent = originalBundleEntries.get(resource.getIdElement().getValue()).stream()
                    .findFirst()
                    //Don't see how this is possible
                    .orElseThrow(() -> new RuntimeException("Failed to find original bundle entry for id " + resource.getIdElement().getValue()));

            Bundle.BundleEntryComponent entryComponent = new Bundle.BundleEntryComponent();
            entryComponent.setResource((Resource)resource);
            entryComponent.setLink(originalEntryComponent.getLink());
            entryComponent.setFullUrl(originalEntryComponent.getFullUrl());

            newBundle.addEntry(entryComponent);
        });

        return newBundle;
    }
}
