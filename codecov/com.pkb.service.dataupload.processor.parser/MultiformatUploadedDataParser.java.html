package com.pkb.service.dataupload.processor.parser;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import java.util.Map;

import com.pkb.dataupload.entity.UploadedData;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.service.dataupload.processor.UploadedDataProcessingContext;

public class MultiformatUploadedDataParser<T, C extends UploadedDataProcessingContext> extends AbstractUploadedDataParser<T, C> {
    private Map<UploadedData.Format, UploadedDataParser<T, C>> parserMapping;

    public MultiformatUploadedDataParser(Map<UploadedData.Format, UploadedDataParser<T, C>> parserMapping) {
        super(new ArrayList<>(parserMapping.keySet()));
        this.parserMapping = parserMapping;
    }

    @Override
    public Collection<UploadedData.Format> getSupportedFormat() {
        return parserMapping.keySet();
    }

    @Override
    protected List<T> doParse(C context, UploadedDataDTO dto) {
        UploadedDataParser<T, C> parser = parserMapping.get(dto.getFormat());
        if (parser == null) {
            throw new IllegalStateException(
                    String.format("No parser found for format %s; Uploaded data id: %s", dto.getFormat(), dto.getId()));
        }
        return parser.parse(context, dto);
    }
}
