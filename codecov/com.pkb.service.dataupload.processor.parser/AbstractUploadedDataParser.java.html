package com.pkb.service.dataupload.processor.parser;

import com.pkb.dataupload.entity.UploadedData;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.service.dataupload.processor.UploadedDataProcessingContext;

import java.util.Collection;
import java.util.Collections;
import java.util.List;

public abstract class AbstractUploadedDataParser<T, C extends UploadedDataProcessingContext> implements UploadedDataParser<T, C> {
    private List<UploadedData.Format> supportedFormat;

    AbstractUploadedDataParser(List<UploadedData.Format> supportedFormat) {
        this.supportedFormat = supportedFormat;
    }

    public AbstractUploadedDataParser(UploadedData.Format supportedFormat) {
        this.supportedFormat = Collections.singletonList(supportedFormat);
    }

    @Override
    public List<T> parse(C context, UploadedDataDTO dto) {
        List<T> result;
        if (supportedFormat.contains(dto.getFormat())) {
            result = doParse(context, dto);
        } else {
            throw new IllegalArgumentException(
                    String.format("Unexpected uploaded data type for format %s; Uploaded data id: %s", dto.getFormat(), dto.getId()));
        }
        return result;
    }

    @Override
    public Collection<UploadedData.Format> getSupportedFormat() {
        return supportedFormat;
    }

    protected abstract List<T> doParse(C context, UploadedDataDTO dto);

}
