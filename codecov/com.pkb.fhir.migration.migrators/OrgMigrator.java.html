package com.pkb.fhir.migration.migrators;

import static com.pkb.entities.enums.FhirStoreSyncSourceTable.ORG;
import static com.pkb.entities.enums.FhirStoreSyncSourceTable.ORGEXTERNALID;
import static com.pkb.fhir.migration.migrators.BaseFhirMigrator.ResourceAndOperation.FhirServerOperation.CREATE_OR_UPDATE;
import static com.pkb.fhir.migration.migrators.BaseFhirMigrator.ResourceAndOperation.FhirServerOperation.DELETE;
import static io.vavr.control.Either.right;

public class OrgMigrator extends BaseFhirMigrator {

    //See https://dev.patientsknowbest.com/home/fhir-api/identifier-systems
    private static final String ORG_CODE_SYSTEM_URL = "http://fhir.patientsknowbest.com/id/org-code";
    private static final String ORG_WEBMD_CODE_SYSTEM_URL = "http://fhir.patientsknowbest.com/id/org-webmd-code";
    private static final String ORG_LEGACY_PRIMARY_KEY_SYSTEM_URL = "http://fhir.patientsknowbest.com/id/org-legacy-identifier";
    private static final String ORGEXTERNALID_LEGACY_PRIMARY_KEY_SYSTEM_URL = "http://fhir.patientsknowbest.com/id/orgexternalid-legacy-identifier";

    // See https://dev.patientsknowbest.com/home/fhir-api/structuredefinition
    private static final String ORG_COUNTRY_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-country";
    private static final String ORG_CRMLINK_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-crmlink";
    private static final String ORG_DATA_UPLOAD_IP_WHITELIST_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-data-upload-ip-whitelist";
    private static final String ORG_EMAIL_IMAGE_FOOTER_FILENAME_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-email-image-footer-filename";
    private static final String ORG_EMAIL_FOOTER_TEXT_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-email-footer-text";
    private static final String ORG_HL7_AUTO_CREATE_RECORDS_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-hl7-auto-create-records";
    private static final String ORG_HL7_CHECK_NHS_NUMBERS_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-hl7-check-nhs_numbers";
    private static final String ORG_HL7_NEW_PATIENT_TEAM_CODE_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-hl7-new-patient-team-code";
    private static final String ORG_HL7_UPLOAD_SOURCE_WHITELIST_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-hl7-upload-source-whitelist";
    private static final String ORG_HL7_UPLOAD_VALIDATE_PID_LIST_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-hl7-validate-pid-list";
    private static final String ORG_LOGO_FILE_NAME_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-logo-file-name";
    private static final String ORG_ACCOUNT_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-account";
    private static final String ORG_PATIENT_VIEW_SUSPENDED_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/org-patient-view-suspended";
    private static final String ORG_EXTERNAL_ID_TYPE = "http://fhir.patientsknowbest.com/structuredefinition/org-external-id-type";

    private final OrgExternalIdService orgExternalIdService;

    public OrgMigrator(PersonService personService, TeamService teamService, OrgService orgService,
                       Hl7PartnerService hl7PartnerService, OrgExternalIdService orgExternalIdService) {
        super(personService, teamService, orgService, hl7PartnerService);
        this.orgExternalIdService = orgExternalIdService;
    }

    @Override
    Either<MigrationException, List<ResourceAndOperation>> migrate(@NotNull PublicFhirStoreSync fhirStoreSync) {
        if (fhirStoreSync.isDeleteOperation() && fhirStoreSync.getSyncDetails().getSourceTable() == ORG) {
            Organization toDelete = new Organization();
            IdType idType = new IdType(ResourceType.Organization.name(),
                    String.valueOf(fhirStoreSync.getSyncDetails().getSourceTablePublicId()));
            toDelete.setId(idType);
            return right(ResourceAndOperation.all(List.of(toDelete), DELETE));
        }

        return Option.ofOptional(getPublicOrg(fhirStoreSync))
            .toEither(() -> (MigrationException)new MigrationEntityNotFoundException("Failed to sync org to fhir store for source table "
                    + fhirStoreSync.getSyncDetails().getSourceTable() +
                    " and source id " + fhirStoreSync.getSyncDetails().getSourceTableId()))
            .map(org -> ResourceAndOperation.all(List.of(transformOrgToFhir(org)), CREATE_OR_UPDATE));
    }

    @NotNull
    private Optional<PublicOrg> getPublicOrg(@NotNull PublicFhirStoreSync fhirStoreSync) {
        FhirStoreSyncSourceTable sourceTable = fhirStoreSync.getSyncDetails().getSourceTable();
        if (sourceTable == ORG) {
            return orgService.findOptionalByIdFetchOrgAccount(fhirStoreSync.getSyncDetails().getSourceTableId());
        }
        else if (sourceTable == ORGEXTERNALID) {
            //Sub-element within the org
            return orgService.findOptionalByIdFetchOrgAccount(fhirStoreSync.getSyncDetails().getSourceTableParentId());
        }
        else  {
            throw new MigrationException("Trying to process unexpected source table [" + sourceTable + "] within the Org migrator");
        }
    }

    private Organization transformOrgToFhir(@NotNull PublicOrg org) {
        Organization organization = new Organization();
        organization.setId(String.valueOf(org.getPublicId()));

        Optional.ofNullable(org.getName()).filter(StringUtils::isNotBlank).ifPresent(organization::setName);

        //Various org identififers
        Optional.ofNullable(org.getCode())
                .map(orgCode -> {
                    Identifier identifier = new Identifier();
                    identifier.setSystem(ORG_CODE_SYSTEM_URL);
                    identifier.setValue(org.getCode());
                    return identifier;
                }).ifPresent(organization::addIdentifier);

        organization.addIdentifier(getPkbIdentifier(org.getId(), ORG_LEGACY_PRIMARY_KEY_SYSTEM_URL));

        orgExternalIdService.findByOrgId(org.getId())
                .forEach(externalId ->  {
                    Identifier identifier = new Identifier();
                    identifier.setSystem(getSystemForOrgExternalId(externalId));
                    identifier.setValue(externalId.getIdValue());
                    addExtension(externalId.getId(), ORGEXTERNALID_LEGACY_PRIMARY_KEY_SYSTEM_URL, identifier);
                    organization.addIdentifier(identifier);
                });

        //We're now going to actually populate this information on the database!
        organization.setActive(org.isActive());

        //We need to differentiate orgs from teams, they will both be Organization resources. Teams have a nice category,
        // orgs don't really: they're not all healthcare providers, which is the most obvious one.
        // Business and success are categorising these on UK prod at least. Fall back on OTHER where this isn't done
        String orgTypeCode = org.getOrgType();
        Coding coding = new Coding();
        coding.setSystem("http://hl7.org/fhir/organization-type");
        coding.setCode(orgTypeCode == null ? OrganizationType.OTHER.toCode() : orgTypeCode);
        CodeableConcept organizationType = new CodeableConcept();
        organizationType.setCoding(List.of(coding));

        organization.addType(organizationType);

        addExtensions(org, organization);
        return organization;
    }

    private String getSystemForOrgExternalId(PublicOrgExternalId externalId) {

        switch (externalId.getIdType()) {
            case ODS:
                return "https://fhir.nhs.uk/Id/ods-organization-code";
            default:
                throw new RuntimeException("Unexpected OrgExternalId type " + externalId.getIdType());
        }
    }

    private void addExtensions(PublicOrg org, Organization organization) {
        addExtension(org.getCountry(), ORG_COUNTRY_EXTENSION_URL, organization);
        addExtension(org.getCrmLink(), ORG_CRMLINK_EXTENSION_URL, organization);
        addExtension(org.getDataUploadIpWhitelist(), ORG_DATA_UPLOAD_IP_WHITELIST_EXTENSION_URL, organization);
        addExtension(org.getEmailFooterImageFilename(), ORG_EMAIL_IMAGE_FOOTER_FILENAME_EXTENSION_URL, organization);
        addExtension(org.getEmailFooterText(), ORG_EMAIL_FOOTER_TEXT_EXTENSION_URL, organization);
        addExtension(org.isHl7AutoCreateRecords(), ORG_HL7_AUTO_CREATE_RECORDS_EXTENSION_URL, organization);
        addExtension(org.isHl7CheckNhsNumberStatus(), ORG_HL7_CHECK_NHS_NUMBERS_EXTENSION_URL, organization);
        addExtension(org.getHl7NewPatientTeamCode(), ORG_HL7_NEW_PATIENT_TEAM_CODE_EXTENSION_URL, organization);
        addExtension(org.getHl7UploadSourceWhitelist(), ORG_HL7_UPLOAD_SOURCE_WHITELIST_EXTENSION_URL, organization);
        addExtension(org.getHl7UploadValidatePidList(), ORG_HL7_UPLOAD_VALIDATE_PID_LIST_EXTENSION_URL, organization);
        addExtension(org.getLogoFileName(), ORG_LOGO_FILE_NAME_EXTENSION_URL, organization);

        organization.addExtension(getExtensionWithReferenceIdentifier(ORG_ACCOUNT_EXTENSION_URL, org.getOrgAccount().getPublicId()));

        addExtension(org.isPatientViewSuspended(), ORG_PATIENT_VIEW_SUSPENDED_EXTENSION_URL, organization);
        addExtension(org.getTimeZoneId(), TIME_ZONE_ID_EXTENSION_URL, organization);
    }
}
