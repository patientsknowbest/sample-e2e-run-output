package com.pkb.fhir.migration.migrators;

import com.pkb.entities.pub.PublicFhirStoreSync;
import com.pkb.entities.pub.PublicPKBPerson;
import com.pkb.entities.pub.PublicPatientOptOut;
import com.pkb.fhir.common.exception.MigrationEntityNotFoundException;
import com.pkb.fhir.common.exception.MigrationException;
import com.pkb.fhir.common.service.OrgService;
import com.pkb.fhir.common.service.PersonService;
import com.pkb.fhir.common.service.TeamService;
import com.pkb.fhir.migration.service.BaseMenuDataService;
import com.pkb.fhir.migration.service.Hl7PartnerService;
import com.pkb.fhir.migration.service.PatientOptOutService;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.hl7.fhir.instance.model.api.IBaseResource;
import org.hl7.fhir.r4.model.CodeableConcept;
import org.hl7.fhir.r4.model.Coding;
import org.hl7.fhir.r4.model.Consent;
import org.hl7.fhir.r4.model.IdType;
import org.hl7.fhir.r4.model.Patient;
import org.hl7.fhir.r4.model.Provenance;
import org.hl7.fhir.r4.model.Reference;
import org.hl7.fhir.r4.model.ResourceType;
import org.jetbrains.annotations.NotNull;

import java.util.List;
import java.util.Optional;

import static com.pkb.entities.enums.FhirConsentSource.OPTOUT;
import static com.pkb.entities.enums.FhirStoreSyncSourceTable.PATIENTOPTOUT;
import static com.pkb.fhir.migration.migrators.BaseFhirMigrator.ResourceAndOperation.FhirServerOperation.CREATE_OR_UPDATE;
import static com.pkb.fhir.migration.migrators.BaseFhirMigrator.ResourceAndOperation.FhirServerOperation.DELETE;
import static io.vavr.control.Either.right;

/**
 * PatientOptOut entities get mapped to a Consent FHIR resource, of category "PatientOptOut". These point
 * at the Patient, but do not point at any Organization, as they are a global property.
 *
 * A PatientOptOut opts a patient out from all sharing of data if marked as sharingEnabled == false.
 * If inactive, i.e. sharingEnabled == true, it does nothing.
 */
public class PatientOptOutMigrator extends BaseFhirMigrator {

    //See https://dev.patientsknowbest.com/home/fhir-api/identifier-systems
    private static final String PATIENTOPTOUT_LEGACY_PRIMARY_KEY_SYSTEM_URL = "http://fhir.patientsknowbest.com/id/patientoptout-legacy-identifier";

    //Source information extensions
    private static final String DOCUMENTATION_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/documentation";

    private final PatientOptOutService patientOptOutService;
    private final BaseMenuDataService menuDataService;

    public PatientOptOutMigrator(PersonService personService, TeamService teamService, OrgService orgService,
                                 Hl7PartnerService hl7PartnerService, PatientOptOutService patientOptOutService,
                                 BaseMenuDataService menuDataService) {
        super(personService, teamService, orgService, hl7PartnerService);
        this.patientOptOutService = patientOptOutService;
        this.menuDataService = menuDataService;
    }

    @Override
    Either<MigrationException, List<ResourceAndOperation>> migrate(@NotNull PublicFhirStoreSync fhirStoreSync) {
        if (fhirStoreSync.isDeleteOperation()) {
            String consentResourceUuid = getResourceUuidFromTableSource(String.valueOf(fhirStoreSync.getSyncDetails().getSourceTablePublicId()),
                    fhirStoreSync.getSyncDetails().getSourceTable());
            Provenance provenanceToDelete = new Provenance();

            IdType idTypeProvenance = new IdType(ResourceType.Provenance.name(),
                    getProvenanceUuidFromResourceTarget(consentResourceUuid, ResourceType.Consent));
            provenanceToDelete.setId(idTypeProvenance);

            Consent consentToDelete = new Consent();
            IdType idTypeConsent = new IdType(ResourceType.Consent.name(), consentResourceUuid);
            consentToDelete.setId(idTypeConsent);
            return right(ResourceAndOperation.all(List.of(provenanceToDelete, consentToDelete), DELETE));
        }

        return Option.ofOptional(patientOptOutService.findById(fhirStoreSync.getSyncDetails().getSourceTableId()))
                .toEither(() -> (MigrationException)new MigrationEntityNotFoundException("Failed to sync patientoptout to fhir store for source id "
                        + fhirStoreSync.getSyncDetails().getSourceTableId()))
                .map(patientOptOut -> ResourceAndOperation.all(transformPatientOptOutToFhir(patientOptOut), CREATE_OR_UPDATE));
    }

    private List<IBaseResource> transformPatientOptOutToFhir(PublicPatientOptOut patientOptOut) {
        Consent consent = new Consent();
        consent.setId(getResourceUuidFromTableSource(String.valueOf(patientOptOut.getPublicId()), PATIENTOPTOUT));
        consent.addIdentifier(getPkbIdentifier(patientOptOut.getPersonId(), PATIENTOPTOUT_LEGACY_PRIMARY_KEY_SYSTEM_URL));
        setMeta(patientOptOut.getVersionDetails().getVersionPersisted(), consent);
        Consent.ConsentPolicyComponent policyComponent = new Consent.ConsentPolicyComponent();
        policyComponent.setAuthority("http://www.patientsknowbest.com"); //As per current specs
        consent.addPolicy(policyComponent);

        addConsentScope(consent);

        consent.setPatient(new Reference(new Patient().setId(String.valueOf(personService.findPersonOptional(patientOptOut.getPersonId())
                .map(PublicPKBPerson::getPublicId)
                .orElseThrow(() -> new MigrationEntityNotFoundException("No linked person for patient opt out " + patientOptOut.getPersonId()))))));

        Coding coding = new Coding();
        coding.setSystem(CONSENT_CATEGORY_URL);
        coding.setCode(OPTOUT.name());
        CodeableConcept consentCategory = new CodeableConcept();
        consentCategory.setCoding(List.of(coding));
        consent.setCategory(List.of(consentCategory));

        //This is an opt OUT so if it's ACTIVE then you're OPTING OUT.
        // This ties into our general opt-in modelling for Consents where they represent a policy which, by default,
        // indicates that you cannot see anything. Anything you can see, i.e. individual privacy labels, are modelled
        // as exceptions to that policy.
        // See https://pkbdev.atlassian.net/wiki/spaces/TECHNOLOGY/pages/136118372/Consent for more info
        consent.setStatus(patientOptOut.isSharingEnabled() ? Consent.ConsentState.INACTIVE : Consent.ConsentState.ACTIVE);

        Optional.ofNullable(patientOptOut.getDocumentationId())
                .flatMap(menuDataService::findByIdNotDeleted)
                .ifPresent(menuData -> consent.addExtension(getExtensionWithReferenceIdentifier(DOCUMENTATION_EXTENSION_URL, menuData.getUniqueId())));


        return List.of(consent, getProvenance(patientOptOut.getVersionDetails(), consent));
    }
}
