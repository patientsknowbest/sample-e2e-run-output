package com.pkb.fhir.migration.migrators;

import ca.uhn.fhir.rest.client.api.IGenericClient;

import static com.pkb.entities.enums.FhirStoreSyncStatus.ERROR;
import static com.pkb.entities.enums.FhirStoreSyncStatus.ERROR_SILENTLY;
import static com.pkb.entities.enums.FhirStoreSyncStatus.SKIPPED;
import static com.pkb.entities.enums.FhirStoreSyncStatus.SYNCED;
import static com.pkb.fhir.migration.migrators.BaseFhirMigrator.ResourceAndOperation.FhirServerOperation.CREATE_OR_UPDATE;
import static com.pkb.fhir.migration.migrators.BaseFhirMigrator.ResourceAndOperation.FhirServerOperation.DELETE;

/**
 * Is called into from {@link com.pkb.migration.fhir.FhirStoreSyncRunnerService} to perform a transformation and
 * sync to the FHIR store.
 *
 * Subsequently calls into the relevant FHIR migrator for the entity being migrated, and gets a List of resources back.
 * These are then sent to the migration FHIR server as create/updates or deletes.
 */
public class MigrationManager {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private final PKBPersonMigrator pkbPersonMigrator;
    private final TeamMigrator teamMigrator;
    private final OrgMigrator orgMigrator;
    private final InstituteUserMigrator instituteUserMigrator;
    private final PatientOptOutMigrator patientOptOutMigrator;
    private final PatientConsentMigrator patientConsentMigrator;
    private final MenuDataMigrator menuDataMigrator;
    private final ReferenceDatumMigrator referenceDatumMigrator;
    private final EhrmigratorConfig config;
    private final IGenericClient client;

    public MigrationManager(PKBPersonMigrator pkbPersonMigrator,
                            TeamMigrator teamMigrator,
                            OrgMigrator orgMigrator,
                            InstituteUserMigrator instituteUserMigrator,
                            PatientOptOutMigrator patientOptOutMigrator,
                            PatientConsentMigrator patientConsentMigrator,
                            MenuDataMigrator menuDataMigrator,
                            ReferenceDatumMigrator referenceDatumMigrator,
                            EhrmigratorConfig config,
                            IGenericClient client) {
        this.pkbPersonMigrator = pkbPersonMigrator;
        this.teamMigrator = teamMigrator;
        this.orgMigrator = orgMigrator;
        this.instituteUserMigrator = instituteUserMigrator;
        this.patientOptOutMigrator = patientOptOutMigrator;
        this.patientConsentMigrator = patientConsentMigrator;
        this.menuDataMigrator = menuDataMigrator;
        this.referenceDatumMigrator = referenceDatumMigrator;
        this.config = config;
        this.client = client;
    }

    public MigrationOutcome migrateEntity(PublicFhirStoreSync fhirStoreSync) {
        try {
            BaseFhirMigrator migrator = getMigratorForEntity(fhirStoreSync.getSyncDetails().getSourceTable());
            Either<MigrationException, List<BaseFhirMigrator.ResourceAndOperation>> result = migrator.migrate(fhirStoreSync);
            MigrationOutcome migrationOutcome = result.fold(error ->  Match(error).of(
                            Case($(Predicates.instanceOf(MigrationEntityNotFoundException.class)),
                                    ImmutableMigrationOutcome.migrationOutcome(SKIPPED, Optional.of(error))),
                            Case($(), ImmutableMigrationOutcome.migrationOutcome(ERROR, Optional.of(error)))),
                    this::sendToFhirServer);

            if (migrationOutcome.status() == ERROR) {
                LOGGER.error("Error when migrating entity of type {} and id {}; fhir_store_sync id is {}", fhirStoreSync.getSyncDetails().getSourceTable(),
                        fhirStoreSync.getSyncDetails().getSourceTableId(), fhirStoreSync.getId());
            }
            return migrationOutcome;
        } catch (MigrationEntityNotFoundException | DecryptionKeysNotFoundException e) {
            return ImmutableMigrationOutcome.migrationOutcome(SKIPPED, Optional.of(e));
        } catch (Exception e) {
            LOGGER.error("Error when migrating entity of type {} and id {}; fhir_store_sync id is {}", fhirStoreSync.getSyncDetails().getSourceTable(),
                    fhirStoreSync.getSyncDetails().getSourceTableId(), fhirStoreSync.getId(), e);
            return ImmutableMigrationOutcome.migrationOutcome(ERROR, Optional.of(e));
        }
    }

    private MigrationOutcome sendToFhirServer(List<BaseFhirMigrator.ResourceAndOperation> operations) {
        var opToResources = io.vavr.collection.List.ofAll(operations)
                .groupBy(BaseFhirMigrator.ResourceAndOperation::operation)
                .map((op, resourcesForOp) -> Tuple(op, resourcesForOp.map(BaseFhirMigrator.ResourceAndOperation::resource)))
                // Updates must be processed before deletes to allow references to be cleared
                .toSortedMap(Ordering.explicit(CREATE_OR_UPDATE, DELETE), k -> k);

        for (var operationResourceList : opToResources) {
            MigrationOutcome migrationOutcome = Match(operationResourceList._1).of(
                Case($(CREATE_OR_UPDATE), () -> sendUpdateToFhirServer(operationResourceList._2.toJavaList())),
                Case($(DELETE), () -> sendDeleteToFhirServer(operationResourceList._2.toJavaList())));
            if (migrationOutcome.status() == ERROR || migrationOutcome.status() == ERROR_SILENTLY) {
                return migrationOutcome;
            }
        }

        return ImmutableMigrationOutcome.migrationOutcome(SYNCED, Optional.empty());
    }

    @NotNull
    private BaseFhirMigrator getMigratorForEntity(FhirStoreSyncSourceTable sourceTable) {
        switch (sourceTable) {
            case PKBPERSON:
            case PERSONCONTACT:
            case NATIONALID:
            case ORGLEVELID:
            case TEAMLEVELID:
            case PERSON_GUID:
                return pkbPersonMigrator;
            case TEAM:
                return teamMigrator;
            case ORG:
            case ORGEXTERNALID:
                return orgMigrator;
            case INSTITUTEUSER:
                return instituteUserMigrator;
            case PATIENTOPTOUT:
                return patientOptOutMigrator;
            case PATIENTCONSENT:
                return patientConsentMigrator;
            case MENU_DATA:
                return menuDataMigrator;
            case REFERENCEDATUM:
                return referenceDatumMigrator;
            default:
                throw new IllegalArgumentException("Trying to migrate entity to FHIR for unexpected class: " + sourceTable);
        }
    }

    private MigrationOutcome sendUpdateToFhirServer(@NotNull List<IBaseResource> resources) {
        try {
            for (IBaseResource resource : resources) {
                //Update as create is fine in aidbox
                MethodOutcome response = client.update().resource(resource).execute();
                if (config.getFhirSyncDebugLoggingEnabled()) {
                    LOGGER.info("Sent a CREATE/UPDATE of type {} with id {} to fhir server. Created was: {}", resource.getClass(),
                            resource.getIdElement().getIdPart(), (response.getCreated() != null && response.getCreated()));
                }
            }
            return ImmutableMigrationOutcome.migrationOutcome(SYNCED, Optional.empty());
        } catch (FhirClientConnectionException | ForbiddenOperationException e) {
            // There is little point alerting us to these. They happen sporadically while the FHIR server is under high or low load, and
            // we retry indefinitely so there's literally no consequence. They always succeed on a retry, and we've never seen them
            // anywhere other than the migrator. This is a cheap way of reducing the noise in our alerts channel.
            // If this were to persistantly fail, we would have this flagged via other metrics such as a low processing rate, or a
            // fhir_store_sync queue that was backing up.
            LOGGER.warn("Unexpected CONNECTION ERROR sending resources to FHIR server for create/update. See underneath for more details", e);
            resources.forEach(resource -> LOGGER.info("Resource in connection error! Type {} with id {}", resource.getClass(),
                    resource.getIdElement().getIdPart()));
            return ImmutableMigrationOutcome.migrationOutcome(ERROR_SILENTLY, Optional.of(e));
        } catch (Exception e) {
            LOGGER.error("Unexpected ERROR sending resources to FHIR server for create/update. See underneath for more details", e);
            resources.forEach(resource -> LOGGER.info("Resource in error! Type {} with id {}", resource.getClass(), resource.getIdElement().getIdPart()));
            return ImmutableMigrationOutcome.migrationOutcome(ERROR, Optional.of(e));
        }
    }

    private MigrationOutcome sendDeleteToFhirServer(@NotNull List<IBaseResource> resources) {
        //Could do resourceById but this forces an explicit construction of the resource so you can't
        // e.g. pass in the wrong id with a resource type.
        try {
            for (IBaseResource resource : resources) {
                client.delete().resource(resource).execute();
                if (config.getFhirSyncDebugLoggingEnabled()) {
                    LOGGER.info("Sent a DELETE of type {} with id {} to fhir server", resource.getClass(),
                            resource.getIdElement().getIdPart());
                }
            }
            return ImmutableMigrationOutcome.migrationOutcome(SYNCED, Optional.empty());
        } catch (FhirClientConnectionException | ForbiddenOperationException e) {
            // There is little point alerting us to these. They happen sporadically while the FHIR server is under high or low load, and
            // we retry indefinitely so there's literally no consequence. They always succeed on a retry, and we've never seen them
            // anywhere other than the migrator. This is a cheap way of reducing the noise in our alerts channel.
            // If this were to persistantly fail, we would have this flagged via other metrics such as a low processing rate, or a
            // fhir_store_sync queue that was backing up.
            LOGGER.warn("Unexpected CONNECTION ERROR sending resources to FHIR server for deletion. See underneath for more details", e);
            resources.forEach(resource -> LOGGER.info("Resource in connection error! Type {} with id {}", resource.getClass(),
                    resource.getIdElement().getIdPart()));
            return ImmutableMigrationOutcome.migrationOutcome(ERROR_SILENTLY, Optional.of(e));
        } catch (Exception e) {
            LOGGER.error("Unexpected ERROR sending resources to FHIR server for deletion. See underneath for more details", e);
            resources.forEach(resource -> LOGGER.info("Resource in error! Type {} with id {}", resource.getClass(), resource.getIdElement().getIdPart()));
            return ImmutableMigrationOutcome.migrationOutcome(ERROR, Optional.of(e));
        }
    }

    @Value.Immutable
    @Value.Style(allParameters = true, of = "migrationOutcome")
    public interface MigrationOutcome {
        FhirStoreSyncStatus status();
        Optional<Exception> exception();
    }
}
