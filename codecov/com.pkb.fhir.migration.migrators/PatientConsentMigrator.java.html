import static com.pkb.entities.enums.PrivacyFlag.SOCIAL_CARE;
import static com.pkb.fhir.migration.migrators.BaseFhirMigrator.ResourceAndOperation.FhirServerOperation.CREATE_OR_UPDATE;

/**
 * PatientConsents become Consent entities, easy enough.
 * The default policy assumption for PKB consent is that it is opt IN. Therefore, a Consent, by default, allows
 * the target actor of the Consent to see nothing at all. If a Patient has granted various types of privacy label access,
 * these are recorded as security labels on the provision of the Consent.
 *
 * If the policy is inactive, the target actor similarly does not have the right to see anything.
 *
 * See https://pkbdev.atlassian.net/wiki/spaces/TECHNOLOGY/pages/136118372/Consent for futher information in the older
 * STU3 context, and https://pkbdev.atlassian.net/wiki/spaces/TECHNOLOGY/pages/3265921029/Consent+in+R4+FHIR for
 * R4-specific discussions.
 */
public class PatientConsentMigrator extends BaseFhirMigrator {

    //See https://dev.patientsknowbest.com/home/fhir-api/identifier-systems
    private static final String PATIENTCONSENT_LEGACY_PRIMARY_KEY_SYSTEM_URL = "http://fhir.patientsknowbest.com/id/patientconsent-legacy-identifier";

    //See https://dev.patientsknowbest.com/home/fhir-api/structuredefinition
    private static final String CONSENT_ADDED_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/consent-added";
    //This one is confusing. Essentially it means "only a professional may edit the sharing of this patient, not the carer"
    private static final String PRO_EDIT_ONLY_FOR_CHILD_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/only-professionals-may-edit";

    private ConsentService consentService;

    public PatientConsentMigrator(PersonService personService, TeamService teamService, OrgService orgService,
                                  Hl7PartnerService hl7PartnerService, ConsentService consentService) {
        super(personService, teamService, orgService, hl7PartnerService);
        this.consentService = consentService;
    }

    @Override
    Either<MigrationException, List<ResourceAndOperation>> migrate(@NotNull PublicFhirStoreSync fhirStoreSync) {
        if (fhirStoreSync.isDeleteOperation()) {
            String consentResourceUuid = getResourceUuidFromTableSource(String.valueOf(fhirStoreSync.getSyncDetails().getSourceTablePublicId()),
                    fhirStoreSync.getSyncDetails().getSourceTable());
            DomainResource toDelete = new Consent();
            IdType idTypeConsent = new IdType(ResourceType.Consent.name(), consentResourceUuid);
            toDelete.setId(idTypeConsent);
            return right(ResourceAndOperation.all(List.of(toDelete), DELETE));
        }

        return Option.ofOptional(consentService.findOptionalByIdFetchLazies(fhirStoreSync.getSyncDetails().getSourceTableId()))
            .toEither(() -> (MigrationException)new MigrationEntityNotFoundException("Failed to sync patientconsent to fhir store for source id "
                                                                                             + fhirStoreSync.getSyncDetails().getSourceTableId()))
            .map(patientConsent -> ResourceAndOperation.all(List.of(transformPatientConsentToFhir(patientConsent)), CREATE_OR_UPDATE));
    }

    @NotNull
    private Consent transformPatientConsentToFhir(@NotNull PublicPatientConsent patientConsent) {
        Consent consent = new Consent();
        consent.setId(getResourceUuidFromTableSource(String.valueOf(patientConsent.getUniqueId()), FhirStoreSyncSourceTable.PATIENTCONSENT));
        consent.addIdentifier(getPkbIdentifier(patientConsent.getId(), PATIENTCONSENT_LEGACY_PRIMARY_KEY_SYSTEM_URL));
        Consent.ConsentPolicyComponent policyComponent = new Consent.ConsentPolicyComponent();
        policyComponent.setAuthority("http://www.patientsknowbest.com"); //As per current specs
        consent.addPolicy(policyComponent);

        //There's a lot of bad data around in this field
        if (!isNegativeInfinity(patientConsent.getUpdated())) {
            setMeta(patientConsent.getUpdated(), consent);
        }

        addConsentScope(consent);

        consent.setPatient(new Reference(new Patient().setId(String.valueOf(patientConsent.getPatient().getPublicId()))));
        consent.setStatus(patientConsent.isDischarged() ? Consent.ConsentState.INACTIVE : Consent.ConsentState.ACTIVE);

        Coding coding = new Coding();
        coding.setSystem(CONSENT_CATEGORY_URL);
        coding.setCode(PATIENTCONSENT.name());
        CodeableConcept consentCategory = new CodeableConcept();
        consentCategory.setCoding(List.of(coding));
        consent.setCategory(List.of(consentCategory));

        // A Consent has a provision, which defines the constraints to the base policy. This includes the link to the
        // consented entity, as well as, in our case, security labels defining the scope of the provision.
        // Our base policy is to deny access to everything, then the provision allows for specific viewing of certain
        // security categories.
        Consent.provisionComponent provisionComponent = new Consent.provisionComponent();
        provisionComponent.setType(Consent.ConsentProvisionType.PERMIT); //Because these security labels are OPT IN.
        consent.setProvision(provisionComponent);

        // A Consent points at a team, individual or carer. Team is an Organization, individual is a Patient or Practitioner,
        // and a carer is a RelatedPerson.
        Consent.provisionActorComponent actorComponent = new Consent.provisionActorComponent();
        actorComponent.setRole(getRoleForConsentActor());
        actorComponent.setReference(getReferenceForConsentActor(patientConsent));

        provisionComponent.addActor(actorComponent);

        // Security labels align very well with our privacy labels. These represent categories of data the Consent
        // allows the consented entity (actor reference) to see.
        addSecurityLabelComponents(patientConsent, consent);

        addExtension(patientConsent.isProEditOnlyForChild(), PRO_EDIT_ONLY_FOR_CHILD_EXTENSION_URL, consent);

        //There's a lot of bad data around in this field as well
        if (!isNegativeInfinity(patientConsent.getAdded())) {
            addExtension(patientConsent.getAdded(), CONSENT_ADDED_EXTENSION_URL, consent);
        }

        return consent;
    }

    private CodeableConcept getRoleForConsentActor() {
        Coding coding = new Coding();
        coding.setCode("CONSENTER");
        coding.setSystem("http://hl7.org/fhir/v3/RoleCode"); //See https://www.hl7.org/fhir/STU3/v3/RoleCode/cs.html

        CodeableConcept codeableConcept = new CodeableConcept();
        codeableConcept.setCoding(singletonList(coding));
        return codeableConcept;
    }

    private void addSecurityLabelComponents(@NotNull PublicPatientConsent patientConsent, Consent consent) {
        if (patientConsent.isGeneralHealthConsentGranted()) {
            addSecurityLabelComponent(consent, GENERAL);
        }
        if (patientConsent.isMentalHealthConsentGranted()) {
            addSecurityLabelComponent(consent, MENTAL_HEALTH);
        }
        if (patientConsent.isSexualHealthConsentGranted()) {
            addSecurityLabelComponent(consent, SEXUAL_HEALTH);
        }
        if (patientConsent.isSocialCareConsentGranted()) {
            addSecurityLabelComponent(consent, SOCIAL_CARE);
        }
    }

    private void addSecurityLabelComponent(Consent consent, PrivacyFlag privacyFlag) {
        Coding coding = new Coding();
        coding.setCode(privacyFlag.getExternalDesc());

        // We use our own system here to give us flexibility in the future, and also because there's no "social care"
        // value in the set. Bit of a shame, but there you go.
        coding.setSystem(PRIVACY_LABEL_CODESYSTEM_URL);

        consent.getProvision().addSecurityLabel(coding);
    }

    @NotNull
    private Reference getReferenceForConsentActor(@NotNull PublicPatientConsent patientConsent) {
        switch (patientConsent.getAccessingEntityType()) {
            case TEAM:
                return createActorReferenceForTeam(patientConsent);
            case INDIVIDUAL:
                return createActorReferenceForPkbPerson(patientConsent, new Practitioner());
            case CARER:
                //TODO (PHR-9694) - This requires carers migrating beforehand
                //return createActorReferenceForPkbPerson(patientConsent, new RelatedPerson());
                throw new MigrationException("Not migrating consent for carer yet. See PHR-9694");
            default:
                throw new MigrationException("Unexpected accessingentitytype for patient consent [" + patientConsent.getAccessingEntityType() + "]");
        }
    }

    @NotNull
    private Reference createActorReferenceForTeam(@NotNull PublicPatientConsent patientConsent) {
        Organization team = new Organization();
        UUID teamPublicId = Optional.ofNullable(patientConsent.getAccessingTeam())
                .map(PublicTeam::getPublicId)
                .orElseThrow(() -> new MigrationEntityNotFoundException("Patient consent "
                        + patientConsent.getId() + " of accessing entity type team, but no team for this id"));
        IdType idTypeTeam = new IdType(ResourceType.Organization.name(), String.valueOf(teamPublicId));
        team.setId(idTypeTeam);
        return new Reference(team);
    }

    @NotNull
    private Reference createActorReferenceForPkbPerson(@NotNull PublicPatientConsent patientConsent, @NotNull DomainResource resource) {
        UUID carerPublicId = Optional.ofNullable(patientConsent.getAccessingPerson())
                .map(PublicPKBPerson::getPublicId)
                .orElseThrow(() -> new MigrationEntityNotFoundException("Patient consent "
                        + patientConsent.getId() + " of accessing entity type " + patientConsent.getAccessingEntityType() + ", but no linked entity for this id"));
        IdType idTypeCarer = new IdType(resource.getResourceType().name(), String.valueOf(carerPublicId));
        resource.setId(idTypeCarer);
        return new Reference(resource);
    }
}
