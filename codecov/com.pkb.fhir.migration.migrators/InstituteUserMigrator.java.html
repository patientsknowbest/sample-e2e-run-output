package com.pkb.fhir.migration.migrators;

import com.pkb.entities.enums.SponsorshipStatus;
import com.pkb.entities.pub.PublicFhirStoreSync;
import com.pkb.entities.pub.PublicInstituteUser;
import com.pkb.entities.pub.PublicPKBPerson;
import com.pkb.entities.pub.PublicTeam;
import com.pkb.fhir.common.exception.MigrationEntityNotFoundException;
import com.pkb.fhir.common.exception.MigrationException;
import com.pkb.fhir.common.service.OrgService;
import com.pkb.fhir.common.service.PersonService;
import com.pkb.fhir.common.service.TeamService;
import com.pkb.fhir.migration.service.Hl7PartnerService;
import com.pkb.fhir.migration.service.InstituteUserService;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.hl7.fhir.r4.model.CodeableConcept;
import org.hl7.fhir.r4.model.Coding;
import org.hl7.fhir.r4.model.Consent;
import org.hl7.fhir.r4.model.DomainResource;
import org.hl7.fhir.r4.model.IdType;
import org.hl7.fhir.r4.model.Organization;
import org.hl7.fhir.r4.model.Patient;
import org.hl7.fhir.r4.model.Practitioner;
import org.hl7.fhir.r4.model.PractitionerRole;
import org.hl7.fhir.r4.model.Reference;
import org.hl7.fhir.r4.model.ResourceType;
import org.jetbrains.annotations.NotNull;

import java.util.List;
import java.util.Optional;
import java.util.UUID;

import static com.pkb.entities.enums.FhirConsentSource.INSTITUTEUSER;
import static com.pkb.entities.enums.SponsorshipStatus.ACTIVE;
import static com.pkb.fhir.migration.migrators.BaseFhirMigrator.ResourceAndOperation.FhirServerOperation.CREATE_OR_UPDATE;
import static com.pkb.fhir.migration.migrators.BaseFhirMigrator.ResourceAndOperation.FhirServerOperation.DELETE;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

public class InstituteUserMigrator extends BaseFhirMigrator {

    //See https://dev.patientsknowbest.com/home/fhir-api/identifier-systems
    private static final String INSTITUTEUSER_LEGACY_PRIMARY_KEY_SYSTEM_URL = "http://fhir.patientsknowbest.com/id/instituteuser-legacy-identifier";

    // See https://dev.patientsknowbest.com/home/fhir-api/structuredefinition
    private static final String INSTITUTEUSER_SPONSORSHIP_STATUS_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/instituteuser-sponsorship-status";
    private static final String INSTITUTEUSER_STATUS_UPDATED_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/instituteuser-status-updated";
    private static final String INSTITUTEUSER_CONTACT_OPTIONS_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/instituteuser-contact-options";
    private static final String INSTITUTEUSER_USER_TYPE_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/instituteuser-types";


    private final InstituteUserService instituteUserService;

    public InstituteUserMigrator(PersonService personService, TeamService teamService, OrgService orgService,
                                 Hl7PartnerService hl7PartnerService, InstituteUserService instituteUserService) {
        super(personService, teamService, orgService, hl7PartnerService);
        this.instituteUserService = instituteUserService;
    }

    /**
     * A single Practitioner PKBPerson can only have 1 active instituteuser row, but we create ones which are
     * invited, inactive etc. to maintain full audit and functionality.
     *
     * A given human being can be active in multiple teams by having multiple different pkbpersons. Whee!
     *
     * For Patients, these become Consent links between the Organization and the Patient.
     */
    @Override
    Either<MigrationException, List<ResourceAndOperation>> migrate(@NotNull PublicFhirStoreSync fhirStoreSync) {
        if (fhirStoreSync.isDeleteOperation()) {
            DomainResource toDelete;
            if (fhirStoreSync.getSyncDetails().isPractitionerRole()) {
                toDelete = new PractitionerRole();
                IdType idTypePractitionerRole = new IdType(ResourceType.PractitionerRole.name(),
                        String.valueOf(fhirStoreSync.getSyncDetails().getSourceTablePublicId()));
                toDelete.setId(idTypePractitionerRole);
            }
            else if (fhirStoreSync.getSyncDetails().isConsent()) {
                toDelete = new Consent();
                IdType idTypeConsent = new IdType(ResourceType.Consent.name(),
                        String.valueOf(fhirStoreSync.getSyncDetails().getSourceTablePublicId()));
                toDelete.setId(idTypeConsent);
            }
            else  {
                return left(new MigrationException("Unexpected resource target [" + fhirStoreSync.getSyncDetails().getResourceTarget()
                        + "] for institute user migration with sync: " + fhirStoreSync.getId()));
            }
            return right(ResourceAndOperation.all(List.of(toDelete), DELETE));
        }

        return Option.ofOptional(instituteUserService.findOptionalByIdFetchTeam(fhirStoreSync.getSyncDetails().getSourceTableId()))
                .toEither(() -> (MigrationException)new MigrationEntityNotFoundException("Failed to sync instituteuser to fhir store for source id "
                        + fhirStoreSync.getSyncDetails().getSourceTableId()))
                .map(instituteUser -> ResourceAndOperation.all(List.of(transformInstituteUserToFhir(fhirStoreSync,
                        instituteUser)), CREATE_OR_UPDATE));
    }

    @NotNull
    private DomainResource transformInstituteUserToFhir(@NotNull PublicFhirStoreSync sync, @NotNull PublicInstituteUser instituteUser) {
        if (sync.getSyncDetails().isPractitionerRole()) {
            return transformInstituteUserToPractitionerRole(instituteUser);
        } else if (sync.getSyncDetails().isConsent()) {
            return transformInstituteUserToConsent(instituteUser);
        } else {
            throw new MigrationException("Unexpected resource target [" + sync.getSyncDetails().getResourceTarget()
                    + "] for institute user migration with sync: " + sync.getId());
        }
    }

    @NotNull
    private PractitionerRole transformInstituteUserToPractitionerRole(@NotNull PublicInstituteUser instituteUser) {
        PractitionerRole practitionerRole = new PractitionerRole();
        practitionerRole.setId(String.valueOf(instituteUser.getPublicId()));
        practitionerRole.addIdentifier(getPkbIdentifier(instituteUser.getId(), INSTITUTEUSER_LEGACY_PRIMARY_KEY_SYSTEM_URL));

        UUID personPublicId = getPersonPublicId(instituteUser);
        practitionerRole.setPractitioner(new Reference(new Practitioner().setId(String.valueOf(personPublicId))));

        UUID teamPublicId = getTeamPublicId(instituteUser);
        practitionerRole.setOrganization(new Reference(new Organization().setId(String.valueOf(teamPublicId))));

        //We're going to do two things here: store this in the active boolean, and preserve the specific value in
        // an extension.
        Optional.ofNullable(instituteUser.getSponsorshipStatus()).ifPresent(sponsorshipStatus -> {
            practitionerRole.setActive(sponsorshipStatus == ACTIVE);
            addExtension(sponsorshipStatus.name(), INSTITUTEUSER_SPONSORSHIP_STATUS_EXTENSION_URL, practitionerRole);
        });

        addCommonExtensions(instituteUser, practitionerRole);

        return practitionerRole;
    }

    private UUID getPersonPublicId(@NotNull PublicInstituteUser instituteUser) {
        return personService.findPersonOptional(instituteUser.getPersonId())
                .map(PublicPKBPerson::getPublicId)
                .orElseThrow(() -> new MigrationEntityNotFoundException("No linked person for institute_user " + instituteUser.getId()));
    }

    private UUID getTeamPublicId(@NotNull PublicInstituteUser instituteUser) {
        Optional<PublicTeam> team = Optional.ofNullable(instituteUser.getTeam());
        return team.map(PublicTeam::getPublicId)
                .orElseThrow(() -> new MigrationEntityNotFoundException("No linked team for institute_user " + instituteUser.getId()));
    }

    @NotNull
    private Consent transformInstituteUserToConsent(@NotNull PublicInstituteUser instituteUser) {
        Consent consent = new Consent();
        consent.setId(String.valueOf(instituteUser.getPublicId()));
        consent.addIdentifier(getPkbIdentifier(instituteUser.getId(), INSTITUTEUSER_LEGACY_PRIMARY_KEY_SYSTEM_URL));

        addConsentScope(consent);

        UUID personPublicId = getPersonPublicId(instituteUser);
        consent.setPatient(new Reference(new Patient().setId(String.valueOf(personPublicId))));

        UUID teamPublicId = getTeamPublicId(instituteUser);
        consent.addOrganization(new Reference(new Organization().setId(String.valueOf(teamPublicId))));

        Coding coding = new Coding();
        coding.setSystem(CONSENT_CATEGORY_URL);
        coding.setCode(INSTITUTEUSER.name());
        CodeableConcept consentCategory = new CodeableConcept();
        consentCategory.setCoding(List.of(coding));
        consent.setCategory(List.of(consentCategory));

        //We're going to do two things here: store this as the status, and preserve the specific value in
        // an extension.
        Optional.ofNullable(instituteUser.getSponsorshipStatus()).ifPresent(sponsorshipStatus -> {
            setConsentStatus(consent, sponsorshipStatus);
            addExtension(sponsorshipStatus.name(), INSTITUTEUSER_SPONSORSHIP_STATUS_EXTENSION_URL, consent);
        });

        addCommonExtensions(instituteUser, consent);
        return consent;
    }

    void setConsentStatus(Consent consent, SponsorshipStatus sponsorshipStatus) {
        switch (sponsorshipStatus) {
            case ACTIVE:
                consent.setStatus(Consent.ConsentState.ACTIVE);
                return;
            case INACTIVE:
                consent.setStatus(Consent.ConsentState.INACTIVE);
                return;
            case INVITED:
            case INVITED_TO_UPGRADE:
                consent.setStatus(Consent.ConsentState.PROPOSED);
                return;
            default:
                throw new MigrationException("Unexpected sponsorship status type: " + sponsorshipStatus);
        }
    }

    private void addCommonExtensions(@NotNull PublicInstituteUser instituteUser, @NotNull DomainResource resource) {
        addExtension(instituteUser.getStatusUpdatedOn(), INSTITUTEUSER_STATUS_UPDATED_EXTENSION_URL, resource);

        Optional.ofNullable(instituteUser.getContactable())
                .ifPresent(contactOptions -> addExtension(contactOptions.name(), INSTITUTEUSER_CONTACT_OPTIONS_EXTENSION_URL, resource));

        Optional.ofNullable(instituteUser.getUserType())
                .ifPresent(userType -> addExtension(userType.name(), INSTITUTEUSER_USER_TYPE_EXTENSION_URL, resource));
    }
}
