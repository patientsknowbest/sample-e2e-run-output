    public TeamMigrator(PersonService personService, TeamService teamService, OrgService orgService,

    @Override
    Either<MigrationException, List<ResourceAndOperation>> migrate(@NotNull PublicFhirStoreSync fhirStoreSync) {
        if (fhirStoreSync.isDeleteOperation()) {
            Organization toDelete = new Organization();
            IdType idType = new IdType(ResourceType.Organization.name(),
                    String.valueOf(fhirStoreSync.getSyncDetails().getSourceTablePublicId()));
            toDelete.setId(idType);
            return right(ResourceAndOperation.all(List.of(toDelete), DELETE));
        }

        return Option.ofOptional(teamService.findById(fhirStoreSync.getSyncDetails().getSourceTableId()))
                .toEither(() -> (MigrationException)new MigrationEntityNotFoundException("Failed to sync team to fhir store for source id "
                        + fhirStoreSync.getSyncDetails().getSourceTableId()))
                .map(team -> ResourceAndOperation.all(List.of(transformTeamToFhir(team)), CREATE_OR_UPDATE));
    }

    private Organization transformTeamToFhir(@NotNull PublicTeam team) {
        Organization organization = new Organization();
        organization.setId(String.valueOf(team.getPublicId()));
        organization.addIdentifier(getPkbIdentifier(team.getId(), TEAM_LEGACY_PRIMARY_KEY_SYSTEM_URL));

        Optional.ofNullable(team.getName()).filter(StringUtils::isNotBlank).ifPresent(organization::setName);

        Optional.ofNullable(team.getCode()).filter(StringUtils::isNotBlank)
                .ifPresent(teamCode -> {
                    Identifier identifier = new Identifier();
                    identifier.setSystem(TEAM_CODE_SYSTEM_URL);
                    identifier.setValue(teamCode);
                    organization.addIdentifier(identifier);
                });

        //We don't actually have this concept in our current data model, but I think it would be handy in the future.
        // We have teams that stop using PKB and at the moment that's just hidden knowledge.
        organization.setActive(true);

        Coding coding = new Coding();
        coding.setSystem("http://hl7.org/fhir/organization-type");
        coding.setCode(OrganizationType.TEAM.toCode());
        CodeableConcept organizationType = new CodeableConcept();
        organizationType.setCoding(List.of(coding));

        organization.addType(organizationType);

        organization.setPartOf(new Reference(new Organization().setId(String.valueOf(team.getOrg().getPublicId()))));

        addExtensions(team, organization);
        return organization;
    }
