package com.pkb.fhir.migration.migrators;

import ca.uhn.fhir.context.FhirContext;
import com.google.common.annotations.VisibleForTesting;
import com.pkb.entities.enums.FhirStoreSyncResourceTarget;
import com.pkb.entities.pub.PublicFhirStoreSync;
import com.pkb.fhir.common.decrypt.DecryptedEhrData;
import com.pkb.fhir.common.exception.MigrationException;
import com.pkb.fhir.common.service.OrgService;
import com.pkb.fhir.common.service.PersonService;
import com.pkb.fhir.common.service.TeamService;
import com.pkb.fhir.migration.decrypt.MigrationEncryptedEhrDataDecryptor;
import com.pkb.fhir.migration.service.Hl7PartnerService;
import com.pkb.fhir.migration.transformation.MigrationDecryptedEhrDataTransformer;
import com.pkb.fhir.migration.transformation.TransformedEhrData;
import io.vavr.control.Either;
import org.hl7.fhir.r4.model.AllergyIntolerance;
import org.hl7.fhir.r4.model.Appointment;
import org.hl7.fhir.r4.model.Condition;
import org.hl7.fhir.r4.model.IdType;
import org.hl7.fhir.r4.model.Resource;
import org.jetbrains.annotations.NotNull;

import java.util.List;

import static com.pkb.fhir.migration.migrators.BaseFhirMigrator.ResourceAndOperation.FhirServerOperation.CREATE_OR_UPDATE;
import static com.pkb.fhir.migration.migrators.BaseFhirMigrator.ResourceAndOperation.FhirServerOperation.DELETE;
import static io.vavr.control.Either.right;

public class MenuDataMigrator extends BaseFhirMigrator {
    private final MigrationEncryptedEhrDataDecryptor decryptor;
    private final MigrationDecryptedEhrDataTransformer transformer;
    private final FhirContext fhirContext;

    public MenuDataMigrator(PersonService personService, TeamService teamService, OrgService orgService,
                            Hl7PartnerService hl7PartnerService,
                            MigrationEncryptedEhrDataDecryptor decryptor, MigrationDecryptedEhrDataTransformer transformer,
                            FhirContext fhirContext) {
        super(personService, teamService, orgService, hl7PartnerService);
        this.decryptor = decryptor;
        this.transformer = transformer;
        this.fhirContext = fhirContext;
    }

    @Override
    public Either<MigrationException, List<ResourceAndOperation>> migrate(@NotNull PublicFhirStoreSync fhirStoreSync) {
        Resource resource = getResourceForMenuData(fhirStoreSync.getSyncDetails().getResourceTarget());

        if (fhirStoreSync.isDeleteOperation()) {
            String resourceId = String.valueOf(fhirStoreSync.getSyncDetails().getSourceTablePublicId());
            IdType idType = new IdType(resource.getResourceType().name(), resourceId);
            resource.setId(idType);
            return right(ResourceAndOperation.all(List.of(resource), DELETE));
        }

        long sourceId = fhirStoreSync.getSyncDetails().getSourceTableId();
        Resource newResource = retrieveAndTransformMenuData(sourceId, resource.getClass());
        return Either.right(ResourceAndOperation.all(List.of(newResource), CREATE_OR_UPDATE));
    }

    @VisibleForTesting
    Resource getResourceForMenuData(FhirStoreSyncResourceTarget resourceTarget) {
        switch (resourceTarget) {
            case ALLERGY:
                return new AllergyIntolerance();
            case APPOINTMENT:
                return new Appointment();
            case DIAGNOSIS:
                    return new Condition();
            default:
                throw new IllegalArgumentException("Trying to get FHIR resource for unexpected data type: " + resourceTarget);
        }
    }

    private Resource retrieveAndTransformMenuData(long menuDataId, Class<? extends Resource> resourceClass) {
        DecryptedEhrData decryptionResult = decryptor.process(menuDataId);
        TransformedEhrData transformationResult = transformer.process(decryptionResult);
        return fhirContext.newJsonParser().parseResource(resourceClass, transformationResult.getContent());
    }
}
