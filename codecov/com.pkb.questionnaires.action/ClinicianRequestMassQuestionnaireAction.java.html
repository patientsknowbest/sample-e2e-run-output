package com.pkb.questionnaires.action;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.institute.entity.Team;
import com.pkb.questionnaireservice.modelv2.QuestionnaireTeam;
import com.pkb.service.questionnaire.QuestionnaireManager;
import com.pkb.service.team.TeamUserManager;

import java.util.List;
import java.util.UUID;

public class ClinicianRequestMassQuestionnaireAction extends QuestionnaireBaseAction {
    private TeamUserManager teamUserManager;
    private QuestionnaireManager questionnaireManager;

    private UUID questionnaireId;
    private List<QuestionnaireTeam> questionnaires;
    private String questionnaireName;

    public String sendMassQuestionnaire() {
        long loggedInUserId = loggedInUserUtil.getLoggedInUserId();
        Team primaryTeam = loggedInUserUtil.getLoggedInUserPrimaryTeam();
        if (!teamUserManager.isActiveTeamUser(loggedInUserId) || primaryTeam == null) {
            return UNPAID;
        }
        
        LoggedInEHRRequestContext loggedInEHRRequestContext = getLoggedInEHRRequestContext();
        return questionnaireManager
                .requestMassQuestionnaire(loggedInEHRRequestContext, userManager.getPKBPerson(loggedInUserId), primaryTeam, questionnaireId)
                .fold(this::handleMassQuestionnaireRequestFailure, it -> handleMassQuestionnaireRequestSuccess(primaryTeam));
    }

    public String cancel(){
        return CANCEL;
    }

    private String handleMassQuestionnaireRequestSuccess(Team team) {
        addActionMessage(getText("requestMassQuestionnaireAction.txt.massRequest_emailed", new String[] { team.getName() }));
        return SEARCH;
    }

    public String startMassQuestionnaire() {
        Long loggedInUserId = loggedInUserUtil.getLoggedInUserId();
        Team primaryTeam = loggedInUserUtil.getLoggedInUserPrimaryTeam();
        // Currently, a questionnaire can only be requested by a clinician
        if (!teamUserManager.isActiveTeamUser(loggedInUserId) || primaryTeam == null) {
            return UNPAID;
        }
        return questionnaireManager.getQuestionnaireTeamListForTeam(primaryTeam.getPublicId(), false)
                .peek(ques -> questionnaires = ques)
                .fold(this::handleMassQuestionnaireRequestFailure, it -> SUCCESS);
    }

    public String confirmStartMassQuestionnaire() {
        if (questionnaireId != null) {
            return questionnaireManager.getQuestionnaireById(questionnaireId)
                    .peek(ques -> questionnaireName = ques.getName())
                    .fold(this::handleMassQuestionnaireRequestFailure, it -> SUCCESS);
        }
        return SUCCESS;
    }

    public UUID getQuestionnaireId() {
        return questionnaireId;
    }

    public void setQuestionnaireId(UUID questionnaireId) {
        this.questionnaireId = questionnaireId;
    }

    public List<QuestionnaireTeam> getQuestionnaires() {
        return questionnaires;
    }

    public void setQuestionnaires(List<QuestionnaireTeam> questionnaires) {
        this.questionnaires = questionnaires;
    }

    public String getQuestionnaireName() {
        return questionnaireName;
    }

    public void setQuestionnaireName(String questionnaireName) {
        this.questionnaireName = questionnaireName;
    }

    public QuestionnaireManager getQuestionnaireManager() {
        return questionnaireManager;
    }

    public void setQuestionnaireManager(QuestionnaireManager questionnaireManager) {
        this.questionnaireManager = questionnaireManager;
    }

    public TeamUserManager getTeamUserManager() {
        return teamUserManager;
    }

    public void setTeamUserManager(TeamUserManager teamUserManager) {
        this.teamUserManager = teamUserManager;
    }
}
