package com.pkb.questionnaires.action;

import com.pkb.action.BaseAction;
import com.pkb.service.questionnaire.QuestionnaireException;
import com.pkb.service.questionnaire.QuestionnaireFieldException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.IOException;
import java.io.Writer;
import java.util.regex.Pattern;

public class QuestionnaireBaseAction extends BaseAction {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final Pattern FILENAME_REPLACE_PATTERN = Pattern.compile("\\W+");

    static final String UNPAID = "unpaid";
    static final String SEARCH = "search";

    protected void handleQuestionnaireException(QuestionnaireException e) {
        if (e instanceof QuestionnaireFieldException) {
            addFieldError(((QuestionnaireFieldException) e).errorField, getText(e.errorProperty));
        } else {
            addActionError(getText(e.errorProperty, e.getErrorTextParams()));
        }
    }

    protected String handleMassQuestionnaireRequestFailure(QuestionnaireException error) {
        handleQuestionnaireException(error);
        return ERROR;
    }

    Writer getCsvWriter(String filename) throws IOException {
        getResponse().setContentType("text/csv");
        getResponse().setCharacterEncoding("utf-8");
        getResponse().addHeader("Content-Disposition", "attachment;filename=\"" + sanitiseFilename(filename) + ".csv\"");
        return getResponse().getWriter();
    }

    private String sanitiseFilename(String filename) {
        return FILENAME_REPLACE_PATTERN.matcher(filename).replaceAll("_");
    }
}
