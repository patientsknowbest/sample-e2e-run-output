package com.pkb.questionnaires.action;

import com.opensymphony.xwork2.Action;
import com.pkb.institute.entity.Team;
import com.pkb.service.questionnaire.QuestionnaireException;
import com.pkb.service.questionnaire.QuestionnaireManager;
import com.pkb.service.team.TeamManager;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.Nullable;

import java.io.File;
import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.io.Reader;
import java.util.List;
import java.util.UUID;

import static io.vavr.API.unchecked;
import static java.nio.charset.StandardCharsets.UTF_8;

public class AdminQuestionnaireAction extends QuestionnaireBaseAction {

    private static final long serialVersionUID = 1;

    private String questionnaireName;

    private File qaFile;

    private String instituteCode;

    private UUID questionnaireId;

    private String contentType;

    private int contentLength;

    private InputStream inputStream;

    private QuestionnaireManager questionnaireManager;

    private TeamManager teamManager;

    public String getForm() {
        return Action.SUCCESS;
    }

    public String cancel() {
        return Action.SUCCESS;
    }

    public String addQuestionnaire() throws FileNotFoundException {

        if ((questionnaireName == null) || questionnaireName.isEmpty()) {
            addFieldError("questionnaireName", getText("questionnaire.name.missing"));
            return Action.INPUT;
        }

        if ((qaFile == null) || !qaFile.exists()) {
            addFieldError("qaFile", getText("questionnaire.file.missing"));
            return Action.INPUT;
        }

        Team team = teamManager.getTeamByCode(instituteCode); // Not in session for superadmin?
        PKBPerson creator = loggedInUserUtil.getLoggedInPerson();

        Reader reader = new InputStreamReader(new FileInputStream(qaFile), UTF_8);
        // this closes the reader in a finally block:
        return questionnaireManager.createQuestionnaireFromCSV(questionnaireName, reader, creator, team)
                .fold(this::handleImportErrors, questionnaireTeam -> Action.SUCCESS);
    }

    private String handleImportErrors(List<QuestionnaireException> excList) {
        excList.forEach(this::handleQuestionnaireException);
        return Action.INPUT;
    }

    public @Nullable String exportQuestionnaire() throws IOException {
        return questionnaireManager.getQuestionnaireById(questionnaireId)
                .peek(quest -> questionnaireManager.getQuestionnaireAsCSV(quest, unchecked(() -> getCsvWriter(quest.getName()))))
                .fold(this::handleExportQuestionnaireFailure, it -> handleExportQuestionnaireSuccess());
    }

    private @Nullable String handleExportQuestionnaireSuccess() {
        return null;
    }

    private @Nullable String handleExportQuestionnaireFailure(QuestionnaireException e) {
        handleQuestionnaireException(e);
        return ERROR;
    }

    public String getQuestionnaireName() {
        return questionnaireName;
    }

    public void setQuestionnaireName(String questionnaireName) {
        this.questionnaireName = questionnaireName;
    }

    public File getQaFile() {
        return qaFile;
    }

    public void setQaFile(File qaFile) {
        this.qaFile = qaFile;
    }

    public QuestionnaireManager getQuestionnaireManager() {
        return questionnaireManager;
    }

    public void setQuestionnaireManager(QuestionnaireManager questionnaireManager) {
        this.questionnaireManager = questionnaireManager;
    }

    public String getInstituteCode() {
        return instituteCode;
    }

    public void setInstituteCode(String instituteCode) {
        this.instituteCode = instituteCode;
    }

    public TeamManager getTeamManager() {
        return teamManager;
    }

    public void setTeamManager(TeamManager teamManager) {
        this.teamManager = teamManager;
    }

    public static long getSerialversionuid() {
        return serialVersionUID;
    }

    public UUID getQuestionnaireId() {
        return questionnaireId;
    }

    public void setQuestionnaireId(UUID questionnaireId) {
        this.questionnaireId = questionnaireId;
    }

    public String getContentType() {
        return contentType;
    }

    public int getContentLength() {
        return contentLength;
    }

    public InputStream getInputStream() {
        return inputStream;
    }

    public void setContentType(String contentType) {
        this.contentType = contentType;
    }

    public void setContentLength(int contentLength) {
        this.contentLength = contentLength;
    }

    public void setInputStream(InputStream inputStream) {
        this.inputStream = inputStream;
    }

}
