package com.pkb.questionnaires.action;

import com.opensymphony.xwork2.Action;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.institute.entity.Team;
import com.pkb.questionnaireservice.modelv2.QuestionnaireResponseCount;
import com.pkb.questionnaireservice.modelv2.QuestionnaireTeam;
import com.pkb.service.questionnaire.ImmutableQuestionnaireReportCsvExportFilter;
import com.pkb.service.questionnaire.QuestionnaireManager;
import com.pkb.service.questionnaire.QuestionnaireReportCsvExportFilter;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.service.user.impl.UserManager;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.IOException;
import java.lang.invoke.MethodHandles;
import java.util.ArrayList;
import java.util.List;
import java.util.UUID;
import java.util.stream.Collectors;

import static io.vavr.API.unchecked;

public class QuestionnaireResponsesAction extends QuestionnaireBaseAction {

    private static final long serialVersionUID = 1L;
    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private QuestionnaireManager questionnaireManager;
    private List<QuestionnaireTeam> questionnaires;
    private List<QuestionnaireResponseCount> hiddenQuestionnaires;
    private List<QuestionnaireResponseCount> visibleQuestionnaires;
    private String nameToDownload;
    private UUID idToDownload;

    private String nationalIdName;
    private int idWanted;

    private UserAccessManager userAccessManager;

    public String getTab() {
        return "questionnaires";
    }

    public String getNameToDownload() {
        return nameToDownload;
    }

    public void setNameToDownload(String nameToDownload) {
        this.nameToDownload = nameToDownload;
    }

    public UUID getIdToDownload() {
        return idToDownload;
    }

    public void setIdToDownload(UUID idToDownload) {
        this.idToDownload = idToDownload;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public void setQuestionnaires(List<QuestionnaireTeam> questionnaires) {
        this.questionnaires = questionnaires;
    }

    public List<QuestionnaireTeam> getQuestionnaires() {
        return questionnaires;
    }

    public QuestionnaireManager getQuestionnaireManager() {
        return questionnaireManager;
    }

    public void setQuestionnaireManager(QuestionnaireManager questionnaireManager) {
        this.questionnaireManager = questionnaireManager;
    }

    public String viewQuestionnaireResponses() {
        LOGGER.debug("Getting questionnaire...");
        Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();

        if (team == null) {
            questionnaires = null;
        } else {
            getNationalIdName(team);
            var maybeQuestionnaireResponseCounts = questionnaireManager.getQuestionnaireResponseCountsForTeam(team.getPublicId());
            return maybeQuestionnaireResponseCounts
                    .peek(questionnaireResponseCounts -> {
                        questionnaires = questionnaireResponseCounts.stream().map(QuestionnaireResponseCount::getQuestionnaireTeam).collect(Collectors.toList());
                        java.util.Map<Boolean, List<QuestionnaireResponseCount>> questionnairesVisibilityMap = maybeQuestionnaireResponseCounts.get().stream()
                                .collect(Collectors.partitioningBy(qt -> qt.getQuestionnaireTeam().getHidden()));
                        hiddenQuestionnaires = new ArrayList<>(questionnairesVisibilityMap.get(Boolean.TRUE));
                        visibleQuestionnaires = new ArrayList<>(questionnairesVisibilityMap.get(Boolean.FALSE));
                    }).fold(this::handleMassQuestionnaireRequestFailure, it -> SUCCESS);
        }

        return Action.SUCCESS;
    }

    private void getNationalIdName(Team team) {
        nationalIdName = NationalIdType.getPrimaryNationalIdTypeFromCountryCode(team.getCountry())
                .map(NationalIdType::getName)
                .orElse("");
    }

    public @Nullable String downloadResponses() throws IOException {
        Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();
        QuestionnaireReportCsvExportFilter questionnaireReportCsvExportFilter = getFilter();

        return questionnaireManager.getQuestionnaireResultsAsCSV(team, idToDownload, questionnaireReportCsvExportFilter, unchecked(() -> getCsvWriter(nameToDownload)),
                        getLoggedInEHRRequestContext().getAccessingUserPublicId())
                .fold(this::handleMassQuestionnaireRequestFailure, it -> null);
    }

    @NotNull
    private QuestionnaireReportCsvExportFilter getFilter() {
        ImmutableQuestionnaireReportCsvExportFilter.Builder filter = ImmutableQuestionnaireReportCsvExportFilter.builder();
        switch (idWanted) {
        case 1:
            filter.includeNationalId(true)
                    .includeOrgIds(false)
                    .includeRequestor(true);
            break;
        case 2:
            filter.includeNationalId(false)
                    .includeOrgIds(true)
                    .includeRequestor(true);
            break;
        case 4:
            filter.includeNationalId(false)
                    .includeOrgIds(false)
                    .includeRequestor(false);
            break;
        }
        return filter.build();
    }

    public UserAccessManager getUserAccessManager() {
        return userAccessManager;
    }

    public void setUserAccessManager(UserAccessManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }

    public String getNationalIdName() {
        return nationalIdName;
    }

    public void setNationalIdName(String nationalIdName) {
        this.nationalIdName = nationalIdName;
    }

    public int getIdWanted() {
        return idWanted;
    }

    public void setIdWanted(int idWanted) {
        this.idWanted = idWanted;
    }

    public List<QuestionnaireResponseCount> getHiddenQuestionnaires() {
        return hiddenQuestionnaires;
    }

    public List<QuestionnaireResponseCount> getVisibleQuestionnaires() {
        return visibleQuestionnaires;
    }
}
