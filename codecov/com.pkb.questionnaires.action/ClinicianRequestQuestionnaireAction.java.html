package com.pkb.questionnaires.action;

import com.opensymphony.xwork2.Action;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.institute.entity.Team;
import com.pkb.service.questionnaire.QuestionnaireException;
import com.pkb.service.questionnaire.QuestionnaireManager;
import com.pkb.service.team.TeamUserManager;
import com.pkb.user.entity.PKBPerson;

import java.util.UUID;

public class ClinicianRequestQuestionnaireAction extends QuestionnaireBaseAction {
    private TeamUserManager teamUserManager;
    private QuestionnaireManager questionnaireManager;

    private String recipientId;
    private UUID questionnaireId;
    private UUID questionnaireRequestId;
    private String patientIdString;
    private String previousURL;

    @SuppressWarnings("unused")
    public String sendIndividualQuestionnaire() {
        previousURL = request.getHeader("referer");
        long loggedInUserId = loggedInUserUtil.getLoggedInUserId();
        Team primaryTeam = loggedInUserUtil.getLoggedInUserPrimaryTeam();

        if (!teamUserManager.isActiveTeamUser(loggedInUserId) || primaryTeam == null) {
            return UNPAID;
        }

        LoggedInEHRRequestContext loggedInEHRRequestContext = getLoggedInEHRRequestContext();
        PKBPerson selectedPatient = userManager.getPKBPerson(Long.parseLong(recipientId.trim()), PKBPerson.Lazy.NATIONAL_AND_LOCAL_IDS);
        return questionnaireManager.sendQuestionnaireRequest(selectedPatient, questionnaireId, loggedInUserId,
                primaryTeam, loggedInEHRRequestContext)
                .fold(this::handleIndividualQuestionnaireRequestFailure, it -> handleIndividualQuestionnaireRequestSuccess(selectedPatient));
    }

    private String handleIndividualQuestionnaireRequestSuccess(PKBPerson selectedPatient) {
        addActionMessage(getText("requestOrFillQuestionaire.patientRequest.emailed",
                new String[] { selectedPatient.getFirstName() + " " + selectedPatient.getLastName() }));
        return (contextUserId == null) ? SEARCH : SUCCESS;
    }

    private String handleIndividualQuestionnaireRequestFailure(QuestionnaireException error) {
        handleQuestionnaireException(error);
        return ERROR;
    }

    @SuppressWarnings("unused")
    public String remindPatient() {
        questionnaireManager.remindPatientAboutQuestionnaire(loggedInUserUtil.getLoggedInUserId(), Long.parseLong(patientIdString),
                questionnaireRequestId, loggedInUserUtil.getLoggedInUserPrimaryTeam(getEHRRequestContext()));
        // We need to set this contextUserId because we are redirecting to getIbox.action and need the value for this action
        contextUserId = loggedInUserUtil.getLoggedInUserId();
        addActionMessage(getText("requestOrFillQuestionaire.patientReminder.emailed"));
        return Action.SUCCESS;
    }

    public TeamUserManager getTeamUserManager() {
        return teamUserManager;
    }

    public void setTeamUserManager(TeamUserManager teamUserManager) {
        this.teamUserManager = teamUserManager;
    }

    public QuestionnaireManager getQuestionnaireManager() {
        return questionnaireManager;
    }

    public void setQuestionnaireManager(QuestionnaireManager questionnaireManager) {
        this.questionnaireManager = questionnaireManager;
    }

    public String getRecipientId() {
        return recipientId;
    }

    public void setRecipientId(String recipientId) {
        this.recipientId = recipientId;
    }

    public UUID getQuestionnaireId() {
        return questionnaireId;
    }

    public void setQuestionnaireId(UUID questionnaireId) {
        this.questionnaireId = questionnaireId;
    }

    public UUID getQuestionnaireRequestId() {
        return questionnaireRequestId;
    }

    public void setQuestionnaireRequestId(UUID questionnaireRequestId) {
        this.questionnaireRequestId = questionnaireRequestId;
    }

    public String getPatientIdString() {
        return patientIdString;
    }

    public void setPatientIdString(String patientIdString) {
        this.patientIdString = patientIdString;
    }

    public String getPreviousURL() {
        return previousURL;
    }
}
