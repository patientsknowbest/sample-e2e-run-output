package com.pkb.crypto.decryptionhelper;

import com.pkb.common.ehrdata.serialization.EHRDataDeserializer;
import com.pkb.crypto.decryptionhelper.dto.EncryptedDataWithSourceRecord;
import com.pkb.crypto.decryptionhelper.dto.ImmutableEncryptedDataWithSourceRecord;
import com.pkb.crypto.decryptionhelper.provider.AccountPrivateKeyProvider;
import com.pkb.crypto.decryptionhelper.provider.SymmetricKeyProvider;
import com.pkb.crypto.decryptionhelper.provider.accountkeys.AccountKeysDTOSymmetricKeyProvider;
import com.pkb.crypto.decryptionhelper.provider.accountkeys.AccountKeysProviderFactory;
import com.pkb.crypto.dto.AccountKeysDTO;
import com.pkb.entities.app.AppMenuData;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import org.jetbrains.annotations.NotNull;

import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.function.Function;

import static java.util.stream.Collectors.toMap;

/**
 * Simple wrapper around the decryption helper with a pre-configured mapper.
 * This allows you to easily decrpyt xml01 on a menu data row without worrying about if it's AES/DES/....
 */
@SuppressWarnings("WeakerAccess")
public class DecryptionHelperAppMenuData {

    final static Function<AppMenuData, EncryptedDataWithSourceRecord<AppMenuData>> APP_MENU_DATA_MAPPER = (amd) -> ImmutableEncryptedDataWithSourceRecord.<AppMenuData>builder()
            .dataIdentifier("amd:" + amd.getId())
            .encryptedData(amd.getXml01())
            .secretKeyAlgorithm(Optional.ofNullable(amd.getSecretKeyAlgorithm()))
            .dESSecretKey(Optional.ofNullable(amd.getSecretKey()))
            .symmetricEncryptionKeyId(Optional.ofNullable(amd.getSymmetricEncryptionKeyId()))
            .nonceBase64(Optional.ofNullable(amd.getNonceBase64()))
            .sourceRecord(amd)
            .build();
    private final EHRDataDeserializer deserializer;

    private DecryptionHelper helper;

    private AccountKeysProviderFactory<AppMenuData> accountKeysProviderFactory;

    public DecryptionHelperAppMenuData(DecryptionHelper helper,
                                       AccountKeysProviderFactory<AppMenuData> accountKeysProviderFactory,
                                       EHRDataDeserializer deserializer) {
        this.helper = helper;
        this.accountKeysProviderFactory = accountKeysProviderFactory;
        this.deserializer = deserializer;
    }

    /**
     * Bulk version of {@link #decryptAppMenuDataWithAccountKeys(AppMenuData, AccountKeysDTO)}
     *
     * @return A map of the decrypted content keyed by the menudata id
     */
    @NotNull
    public Map<Long, byte[]> decryptAppMenuDataWithAccountKeys(@NotNull List<AppMenuData> entities, @NotNull AccountKeysDTO accountKeysDTO) {

        return decryptAppMenuData(entities,
                accountKeysProviderFactory.symmetricKeyFromAccountKeysProvider(accountKeysDTO),
                accountKeysProviderFactory.accountPrivateKeyFromAccountKeysProvider(accountKeysDTO));
    }

    /**
     * Bulk version of {@link #decryptAppMenuData(AppMenuData, SymmetricKeyProvider, AccountPrivateKeyProvider)}
     *
     * @return A map of the decrypted content keyed by the menudata id
     */
    @NotNull
    public Map<Long, byte[]> decryptAppMenuData(@NotNull List<AppMenuData> entities,
                                                @NotNull SymmetricKeyProvider<AppMenuData> symmetricKeyProvider,
                                                @NotNull AccountPrivateKeyProvider<AppMenuData> accountPrivateKeyProvider) {

        return helper.decryptBulk(entities, symmetricKeyProvider, accountPrivateKeyProvider, APP_MENU_DATA_MAPPER)
                .entrySet()
                .stream()
                .map(entry -> {
                    Long menuDataId = entry
                            .getKey()
                            .getSourceRecord()
                            .orElseThrow(() -> new IllegalStateException("Source record missing from helper"))
                            .getId();

                    return Tuple.of(menuDataId, entry.getValue());
                }).collect(toMap(Tuple2::_1, Tuple2::_2));
    }

    /**
     * Decrypt this menu data record with the accountKeysDTO (e.g queried from KMS).
     *
     * @throws IllegalStateException with {@link AccountKeysDTOSymmetricKeyProvider#ERROR_ACCOUNT_KEYS_MISSING_SYMMETRIC_KEY} if any symmetric keys are missing.
     */
    public byte[] decryptAppMenuDataWithAccountKeys(@NotNull AppMenuData entity, @NotNull AccountKeysDTO accountKeysDTO) {
        return decryptAppMenuData(entity,
                accountKeysProviderFactory.symmetricKeyFromAccountKeysProvider(accountKeysDTO),
                accountKeysProviderFactory.accountPrivateKeyFromAccountKeysProvider(accountKeysDTO));
    }


    /**
     * Decrypt this menu data record with the accountKeysDTO (e.g queried from KMS) and then deserialize to a map of fields
     *
     * @throws IllegalStateException with {@link AccountKeysDTOSymmetricKeyProvider#ERROR_ACCOUNT_KEYS_MISSING_SYMMETRIC_KEY} if any symmetric keys are missing.
     */
    @NotNull
    public Map<String, Object> decryptAndDeserializeAppMenuDataWithAccountKeys(@NotNull AppMenuData entity, @NotNull AccountKeysDTO accountKeysDTO) {
        return deserializer.deserialize(decryptAppMenuData(entity,
                accountKeysProviderFactory.symmetricKeyFromAccountKeysProvider(accountKeysDTO),
                accountKeysProviderFactory.accountPrivateKeyFromAccountKeysProvider(accountKeysDTO)));
    }

    /**
     * Decrypt this menu data record with custom secret providers.
     * See {@link SymmetricKeyProvider}, {@link AccountPrivateKeyProvider}
     */
    public byte[] decryptAppMenuData(@NotNull AppMenuData entity,
                                     @NotNull SymmetricKeyProvider<AppMenuData> symmetricKeyProvider,
                                     @NotNull AccountPrivateKeyProvider<AppMenuData> accountPrivateKeyProvider) {
        return helper.decrypt(entity, symmetricKeyProvider, accountPrivateKeyProvider, APP_MENU_DATA_MAPPER);
    }


}
