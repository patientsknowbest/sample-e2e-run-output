package com.pkb.domain.duplicate;

import com.pkb.datamodel.entity.mapper.PersonDemographicsMapper;
import com.pkb.datamodel.user.Demographics;
import com.pkb.entities.pub.PublicOrgLevelId;
import com.pkb.entities.pub.PublicPKBPerson;
import com.pkb.entities.pub.PublicTeamLevelId;
import com.pkb.repository.OrgLevelIdRepository;
import com.pkb.repository.PersonRepository;
import com.pkb.repository.TeamLevelIdRepository;
import io.vavr.collection.List;
import io.vavr.collection.Map;
import org.jetbrains.annotations.NotNull;

import java.util.ArrayList;
import java.util.Collection;
import java.util.stream.Collector;

import static org.apache.commons.collections4.CollectionUtils.isEmpty;

public class PatientDemographicsService {

    private final PersonRepository personRepository;
    private final OrgLevelIdRepository orgLevelIdRepository;
    private final TeamLevelIdRepository teamLevelIdRepository;

    private final PersonDemographicsMapper mapper;

    public PatientDemographicsService(PersonRepository personRepository,
                                      OrgLevelIdRepository orgLevelIdRepository, TeamLevelIdRepository teamLevelIdRepository, PersonDemographicsMapper mapper) {
        this.personRepository = personRepository;
        this.orgLevelIdRepository = orgLevelIdRepository;
        this.teamLevelIdRepository = teamLevelIdRepository;
        this.mapper = mapper;
    }

    public <R, A> R findByIds(@NotNull Collection<Long> personIds, @NotNull Collector<? super Demographics, A, R> collector) {
        if (isEmpty(personIds)) {
            return new ArrayList<Demographics>().stream().collect(collector);
        }

        java.util.List<PublicPKBPerson> persons = personRepository.findPersonsByListOfIds(personIds);
        Map<Long, List<PublicOrgLevelId>> orgLevelIdsByPersonId = orgLevelIdRepository.findAllByPersonId(personIds)
                .groupBy(PublicOrgLevelId::getPersonId);
        Map<Long, List<PublicTeamLevelId>> teamLevelIdsByPersonId = teamLevelIdRepository.findAllByPersonId(personIds).groupBy(PublicTeamLevelId::getPersonId);
        return persons.stream()
                .map(person -> mapper.entityToDataModel(person, orgLevelIdsByPersonId.get(person.getId()), teamLevelIdsByPersonId.get(person.getId())))
                .collect(collector);
    }

}
