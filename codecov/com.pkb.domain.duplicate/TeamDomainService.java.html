package com.pkb.domain.duplicate;

import com.pkb.common.base.Error;
import com.pkb.datamodel.entity.mapper.TeamWithConsentsMapper;
import com.pkb.datamodel.entity.mapper.TeamWithOrgMapper;
import com.pkb.datamodel.team.TeamWithConsents;
import com.pkb.datamodel.team.TeamWithOrg;
import com.pkb.entities.pub.PublicTeam;
import com.pkb.repository.TeamRepository;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.Optional;

import static com.pkb.common.base.ErrorBuilder.errorOf;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class TeamDomainService {
    private static final String TEAM_CODE_CANNOT_BE_EMPTY = "Team Code cannot be empty.";
    private static final String NO_TEAM_BY_CODE_FOUND = "No Team found by team code=[%s]";
    private static final String TEAM_CODE_CANNOT_BE_EMPTY_CODE = "team.code.cannot.be.empty";
    private static final String NO_TEAM_BY_CODE_FOUND_CODE = "no.team.found.by.team.code";

    private final TeamRepository teamRepository;
    private final TeamWithOrgMapper teamWithOrgMapper;
    private final TeamWithConsentsMapper teamWithConsentsMapper;

    public TeamDomainService(TeamRepository teamRepository,
                             TeamWithOrgMapper teamWithOrgMapper,
                             TeamWithConsentsMapper teamWithConsentsMapper) {
        this.teamRepository = teamRepository;
        this.teamWithOrgMapper = teamWithOrgMapper;
        this.teamWithConsentsMapper = teamWithConsentsMapper;
    }

    public Optional<PublicTeam> findById(long id){
        return teamRepository.findById(id);
    }

    public Either<Error, TeamWithOrg> findTeamWithOrgByTeamCode(@NotNull String teamCode) {
        return validateTeamCode(teamCode)
                .flatMap(this::findExistingTeamWithOrgByCode);
    }

    public Optional<TeamWithOrg> findVerifyingTeamForPersonId(long personId) {
        return teamRepository.findVerifyingTeamForPersonId(personId)
                .map(teamWithOrgMapper::entityToDataModel);
    }
    public Optional<TeamWithConsents> findTeamWithConsentsByCode(String teamCode) {
        return teamRepository
                .findOptionalByCode(teamCode)
                .map(teamWithConsentsMapper::entityToDatamodel);
    }

    private Either<Error, String> validateTeamCode(@NotNull String teamCode) {
        return isNotBlank(teamCode) ? right(teamCode) : left(errorOf(TEAM_CODE_CANNOT_BE_EMPTY_CODE, TEAM_CODE_CANNOT_BE_EMPTY));
    }

    private Either<Error, TeamWithOrg> findExistingTeamWithOrgByCode(String teamCode) {
        return teamRepository.findOptionalByCode(teamCode)
                .map(teamWithOrgMapper::entityToDataModel)
                .map(Either::<Error, TeamWithOrg>right)
                .orElseGet(() -> left(errorOf(NO_TEAM_BY_CODE_FOUND_CODE, NO_TEAM_BY_CODE_FOUND, teamCode)));
    }
}
