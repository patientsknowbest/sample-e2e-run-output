package com.pkb.entities.core;

import com.google.common.base.MoreObjects;
import com.pkb.entities.embeddables.EncodedEncryptedPrivateKeyInfo;
import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import org.hibernate.annotations.DynamicInsert;
import org.hibernate.annotations.DynamicUpdate;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;

@Entity
@Table(schema = "core", name = "account_org")
@DynamicInsert
@DynamicUpdate
public class CoreAccountOrg implements Serializable {

    private static final long serialVersionUID = 1L;

    @Id
    @GenericGenerator(name = "seq_cao_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "core"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_accountorg"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_cao_gen")
    private Long id;

    @Column(name = "account_id", nullable = false, updatable = false)
    private Long accountId;

    @Column(name = "org_id", nullable = false)
    private long orgId;

    @Embedded
    private EncodedEncryptedPrivateKeyInfo accountPrivateKey;

    private Instant created;

    @Column(name = "sourceorg_id", updatable = false)
    private Long sourceOrgId;

    public CoreAccountOrg() {
        
    }

    public Long getId() {
        return id;
    }

    public void setId(Long val) {
        this.id = val;
    }

    public Long getSourceOrgId() {
        return sourceOrgId;
    }

    public void setSourceOrgId(Long sourceOrgId) {
        this.sourceOrgId = sourceOrgId;
    }

    public Long getAccountId() {
        return accountId;
    }

    public void setAccountId(Long accountId) {
        this.accountId = accountId;
    }

    public long getOrgId() {
        return orgId;
    }

    public void setOrgId(long orgId) {
        this.orgId = orgId;
    }

    public EncodedEncryptedPrivateKeyInfo getAccountPrivateKey() {
        return accountPrivateKey;
    }

    public void setAccountPrivateKey(EncodedEncryptedPrivateKeyInfo accountPrivateKey) {
        this.accountPrivateKey = accountPrivateKey;
    }

    public Instant getCreated() {
        return created;
    }

    public void setCreated(Instant created) {
        this.created = created;
    }

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("id", id)
                .toString();
    }
}
