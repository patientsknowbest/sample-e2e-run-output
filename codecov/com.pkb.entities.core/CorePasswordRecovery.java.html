package com.pkb.entities.core;

import com.pkb.entities.enums.core.PasswordRecoveryStatus;
import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;

@Entity
@Table(schema = "core", name = "password_recovery")
public class CorePasswordRecovery implements Serializable {

    private static final long serialVersionUID = 1L;

    @Column
    private Instant creation;

    @Column(name = "encrypted_password_inline")
    private byte[] encryptedPasswordInline;

    @Id
    @GenericGenerator(name = "seq_core_gen", strategy = "enhanced-sequence", parameters = {
            @org.hibernate.annotations.Parameter(name = "optimizer", value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
            @org.hibernate.annotations.Parameter(name = "schema", value = "core"),
            @org.hibernate.annotations.Parameter(name = "sequence_name", value = "seq_core"),
            @org.hibernate.annotations.Parameter(name = "increment_size", value = "50")
    })
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_core_gen")
    private long id;

    @Column(name = "pbe_key_iteration_count")
    private int iterationCount;

    @Column(name = "security_question")
    private String securityQuestion;

    @Column(name = "password_purpose")
    private String passwordPurpose;

    @Column(name = "salt_inline")
    private byte[] saltInline;

    @Column
    @Enumerated(EnumType.STRING)
    private PasswordRecoveryStatus status;

    @Column(name = "person_id")
    private long personId;

    public Instant getCreation() {
        return creation;
    }

    public byte[] getEncryptedPassword() {
        return encryptedPasswordInline;
    }

    public Long getId() {
        return id;
    }

    public int getIterationCount() {
        return iterationCount;
    }

    public String getSecurityQuestion() {
        return securityQuestion;
    }

    public String getPasswordPurpose() {
        return passwordPurpose;
    }

    public byte[] getSalt() {
        return saltInline;
    }

    public PasswordRecoveryStatus getStatus() {
        return status;
    }

    public void setCreation(Instant creation) {
        this.creation = creation;
    }

    public void setEncryptedPassword(byte[] encryptedPassword) {
        this.encryptedPasswordInline = encryptedPassword;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public void setIterationCount(int iterationCount) {
        this.iterationCount = iterationCount;
    }

    public void setSecurityQuestion(String securityQuestion) {
        this.securityQuestion = securityQuestion;
    }

    public void setPasswordPurpose(String passwordPurpose) {
        this.passwordPurpose = passwordPurpose;
    }

    public void setSalt(byte[] salt) {
        this.saltInline = salt;
    }

    public void setStatus(PasswordRecoveryStatus status) {
        this.status = status;
    }

    public long getPersonId() {
        return personId;
    }

    public void setPersonId(long personId) {
        this.personId = personId;
    }
}
