package com.pkb.entities.core;

import com.google.common.base.MoreObjects;
import com.pkb.entities.embeddables.EncodedEncryptedPrivateKeyInfo;
import com.pkb.entities.embeddables.UserAccountKeyLinkId;

import javax.persistence.AttributeOverride;
import javax.persistence.AttributeOverrides;
import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.IdClass;
import javax.persistence.Table;
import java.io.Serializable;

/**
 * Represents a link between a user (e.g. a pro) and an account (typically
 * belonging to a patient). For any given user, if you have their user private
 * key and can obtain an instance of this entity for a given account, then you
 * can use the user private key to decrypt the account private key. If the org
 * private key fields are populated, you may need to decrypt the org private key
 * first, and then use that to decrypt the account private key.
 */
@Entity
@IdClass(UserAccountKeyLinkId.class)
@Table(schema = "core", name = "view_account_key_links_2")
public class UserAccountKeyLink implements Serializable {

    private static final long serialVersionUID = -5588037640267737061L;

    @Id
    @Column(name = "access_type")
    private String accessType;

    @Id
    @Column(name = "person_id")
    private Long personId;

    @Id
    @Column(name = "account_id")
    private Long accountId;

    @Column(name = "org_id")
    private Long orgId;

    @Embedded
    @AttributeOverrides({
        @AttributeOverride(name = "encryptedPrivateKeyInfoInline", column = @Column(name = "encrypted_account_private_key_inline")),
        @AttributeOverride(name = "privateKeySize", column = @Column(name = "account_private_key_size")),
        @AttributeOverride(name = "privateKeyAlgorithm", column = @Column(name = "account_private_key_algorithm")),
        @AttributeOverride(name = "encryptedSecretKeyInline", column = @Column(name = "account_key_encrypted_secret_key_inline")),
        @AttributeOverride(name = "secretKeyAlgorithm", column = @Column(name = "account_key_secret_key_algorithm")),
    })
    private EncodedEncryptedPrivateKeyInfo accountPrivateKey;

    @Embedded
    @AttributeOverrides({
        @AttributeOverride(name = "encryptedPrivateKeyInfoInline", column = @Column(name = "encrypted_org_private_key_inline")),
        @AttributeOverride(name = "privateKeySize", column = @Column(name = "org_private_key_size")),
        @AttributeOverride(name = "privateKeyAlgorithm", column = @Column(name = "org_private_key_algorithm")),
        @AttributeOverride(name = "encryptedSecretKeyInline", column = @Column(name = "org_key_encrypted_secret_key_inline")),
        @AttributeOverride(name = "secretKeyAlgorithm", column = @Column(name = "org_key_secret_key_algorithm")),
    })
    private EncodedEncryptedPrivateKeyInfo orgAccountPrivateKey;

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(this)
                .add("accessType", accessType)
                .add("personId", personId)
                .add("accountId", accountId)
                .add("orgId", orgId)
                .add("accountPrivateKey", accountPrivateKey)
                .add("orgAccountPrivateKey", orgAccountPrivateKey)
                .toString();
    }

    public String getAccessType() {
        return accessType;
    }

    public void setAccessType(String accessType) {
        this.accessType = accessType;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }

    public Long getAccountId() {
        return accountId;
    }

    public void setAccountId(Long accountId) {
        this.accountId = accountId;
    }

    public Long getOrgId() {
        return orgId;
    }

    public void setOrgId(Long orgId) {
        this.orgId = orgId;
    }

    public EncodedEncryptedPrivateKeyInfo getAccountPrivateKey() {
        return accountPrivateKey;
    }

    public void setAccountPrivateKey(EncodedEncryptedPrivateKeyInfo accountPrivateKey) {
        this.accountPrivateKey = accountPrivateKey;
    }

    public EncodedEncryptedPrivateKeyInfo getOrgAccountPrivateKey() {
        return orgAccountPrivateKey;
    }

    public void setOrgAccountPrivateKey(EncodedEncryptedPrivateKeyInfo orgAccountPrivateKey) {
        this.orgAccountPrivateKey = orgAccountPrivateKey;
    }
}
