package com.pkb.authz.policy.condition;

import com.pkb.authz.TargetMeta;
import com.pkb.authz.condition.ConditionFailure;
import com.pkb.authz.permission.Justification;
import com.pkb.authz.policy.Facts;
import com.pkb.datamodel.authorization.TargetOwnerData;
import io.vavr.collection.List;
import io.vavr.collection.Set;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.condition.ImmutableConditionFailure.conditionFailure;
import static io.vavr.API.Set;
import static io.vavr.control.Either.left;

/**
 * Convenient way of defining {@link Condition}s for target with owner
 * 
 * @param <O>
 *            the type of object you are testing on
 */
@FunctionalInterface
public interface TargetAndOwnerCondition<O> extends TypedCondition<O> {
    @Override
    default @NotNull Either<Set<ConditionFailure>, List<Justification>> testTyped(@NotNull Facts<O> facts) {


        return facts.findTargetOwnerData()
                .map(targetOwnerData -> test(facts.targetMeta(), targetOwnerData))
                .getOrElse(left(Set(conditionFailure("Condition cannot be evaluated as target owner is not present"))));
    }



    @NotNull
    Either<Set<ConditionFailure>, List<Justification>> test(@NotNull TargetMeta<O> targetMeta, @NotNull TargetOwnerData targetOwnerData);
}
