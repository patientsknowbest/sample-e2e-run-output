package com.pkb.authz.policy.condition;

import com.pkb.authz.JustificationUtils;
import com.pkb.authz.condition.ConditionFailure;
import com.pkb.authz.permission.Justification;
import com.pkb.authz.policy.Facts;
import io.vavr.API;
import io.vavr.collection.List;
import io.vavr.collection.Set;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import static io.vavr.control.Either.narrow;

/**
 * Conditions are the tests in a policy.
 */
@FunctionalInterface
public interface Condition {
    @NotNull Either<Set<ConditionFailure>, List<Justification>> test(@NotNull Facts<?> facts);

    default Condition or(Condition other) {
        return (facts) -> narrow(
                test(facts).swap()
                        .flatMap(firstTestFailures -> other.test(facts).swap()
                                .map(secondTestFailures -> API.<ConditionFailure>Set()
                                        .addAll(firstTestFailures)
                                        .addAll(secondTestFailures)))
                        .swap());
    }

    default Condition and(Condition other) {
        return (facts) -> narrow(
                test(facts)
                        .flatMap(firstTestSuccess -> other.test(facts)
                                .map(secondSucess -> JustificationUtils.merge(firstTestSuccess, secondSucess))
                                )
                        );
    }

}
