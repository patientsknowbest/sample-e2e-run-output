package com.pkb.authz.policy.condition;

import com.pkb.authentication.principal.user.AuthenticatedUser;
import com.pkb.authz.condition.ConditionFailure;
import com.pkb.authz.permission.Justification;
import com.pkb.authz.policy.Facts;
import io.vavr.collection.List;
import io.vavr.collection.Set;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

/**
 * Convenient way of defining {@link Condition}s for user
 */
@FunctionalInterface
public interface ActorCondition extends Condition {
    @Override
    default @NotNull Either<Set<ConditionFailure>, List<Justification>> test(@NotNull Facts facts) {
        return test(facts.authenticatedUser());
    }

    @NotNull Either<Set<ConditionFailure>, List<Justification>> test(@NotNull AuthenticatedUser user);
}
