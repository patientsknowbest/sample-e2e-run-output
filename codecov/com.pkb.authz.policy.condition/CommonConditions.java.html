package com.pkb.authz.policy.condition;

import com.pkb.authentication.principal.team.TeamWithOrgTeams;
import com.pkb.authentication.principal.user.AuthenticatedIndividualProfessional;
import com.pkb.authentication.principal.user.AuthenticatedPatient;
import com.pkb.authentication.principal.user.AuthenticatedPerson;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authentication.principal.user.AuthenticatedTeamProfessional;
import com.pkb.authentication.principal.user.AuthenticatedUser;
import com.pkb.datamodel.PubliclyIdentifiable;
import com.pkb.datamodel.authorization.TargetOwnerData;
import com.pkb.datamodel.consent.PatientConsent;
import com.pkb.datamodel.team.TeamWithOrg;
import com.pkb.datamodel.user.Patient;
import io.vavr.collection.List;
import io.vavr.control.Option;

import java.util.Objects;
import java.util.UUID;

import static com.pkb.authz.JustificationUtils.textual;
import static com.pkb.authz.condition.ImmutableConditionFailure.conditionFailure;
import static com.pkb.authz.condition.ImmutableNoAssociation.noAssociation;
import static com.pkb.authz.condition.ImmutableNotFound.notFound;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.API.Set;
import static io.vavr.API.Some;
import static io.vavr.Predicates.instanceOf;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.narrow;
import static io.vavr.control.Either.right;

/**
 * Conditions which may be shared among many policies.
 */
public class CommonConditions {

    public static ActorAndTargetAndOwnerCondition isSamePerson() {
        return (actor, target, targetOwnerData) ->
                narrow(Some(actor)
                        .filter(AuthenticatedPerson.class::isInstance)
                        .map(AuthenticatedPerson.class::cast)
                        .filter(actorAsPerson ->
                                actorAsPerson.getId().equals(targetOwnerData.person().getId()) ||
                                (actorAsPerson.getHumanUUID().isDefined() && Objects.equals(actorAsPerson.getHumanUUID(), targetOwnerData.person().getHumanUUID())))
                        .map(ignored -> List.of(textual("Actor is the same person")))
                        .toEither(Set(noAssociation("Actor is not the same person"))));
    }

    public static ActorAndTargetCondition actorIsProfessionalInOrg() {
        return (actor, target) ->
            narrow(Some(actor)
                .filter(AuthenticatedTeamProfessional.class::isInstance)
                .map(AuthenticatedTeamProfessional.class::cast)
                .filter(actorAsTeamProfessional ->
                        ((TeamWithOrgTeams) actorAsTeamProfessional.getTeam()).getTeamsOfOrg()
                                .map(PubliclyIdentifiable::getPublicId)
                                .contains(target.requireUUID()))
                    .map(ignored -> List.of(textual("Actor is professional in org")))
                .toEither(Set(notFound(target.requireUUID(), TeamWithOrg.class))));
    }

    public static ActorAndTargetAndOwnerCondition actorHasDecryptionAccessToOwner() {
        return (actor, target, targetOwnerData) -> targetOwnerData.actorHasDecryptionAccess() ?
                right(List.of(textual("Actor has decryption access to owner"))) :
                left(Set(noAssociation("Actor does not have decryption access")));
    }

    public static ActorCondition actorIsNotAccessFrozen() {
        return (actor) ->
            Some(actor)
                .filter(AuthenticatedPatient.class::isInstance)
                .map(AuthenticatedPatient.class::cast)
                .map(AuthenticatedPatient::isAccessFrozen)
                .getOrElse(false) ?
                    left(Set(conditionFailure("Access is frozen"))) :
                    right(List.of(textual("Access is not frozen")));
    }

    public static ActorAndTargetAndOwnerCondition actorIsOwner() {
        return (actor, target, targetOwnerData) ->
                isActorTheOwnerOfTheData(actor, targetOwnerData.person().getPublicId()) ?
                        right(List.of(textual("Actor is the owner of the data"))) :
                        left(Set(noAssociation("Actor is not the owner of the data")));
    }

    private static boolean isActorTheOwnerOfTheData(AuthenticatedUser authenticatedUser, UUID ownerId) {
        Option<UUID> optionalActorId = Match(authenticatedUser).of(
                Case($(instanceOf(AuthenticatedUser.class)), e -> Option.of(e.getPublicId())),
                Case($(instanceOf(AuthenticatedSystemUser.class)), Option.none()));

        return optionalActorId
                .map(actorId -> actorId.equals(ownerId))
                .getOrElse(false);
    }

    public static TargetAndOwnerCondition ownerIsNotSharingDisabled() {
        return (target, targetOwnerData) ->
            asPatient(targetOwnerData)
                    .map(Patient::isSharingDisabled)
                    .getOrElse(false) ?
                left(Set(conditionFailure("Sharing is disabled"))) :
                    right(List.of(textual("Sharing is not disabled")));
    }

    public static TargetAndOwnerCondition ownerIsNotAccessFrozen() {
        return (target, targetOwnerData) ->
            asPatient(targetOwnerData)
                    .map(Patient::isAccessFrozen)
                    .getOrElse(false) ?
                left(Set(conditionFailure("Access to the requested data is frozen"))) :
                    right(List.of(textual("Access is not frozen")));
    }

    public static ActorAndTargetAndOwnerCondition actorHasConsent() {
        return (actor, target, targetOwnerData) ->
                Some(actor)
                        .filter(user -> user instanceof AuthenticatedPatient || user instanceof AuthenticatedIndividualProfessional)
                        .map($ -> targetOwnerData.consent().map(PatientConsent::isAtLeastOneConsentGranted).getOrElse(false))
                        .getOrElse(false) ?
                        right(List.of(textual("Actor is linked with consentable user"))) :
                        left(Set(conditionFailure("Actor is not linked with consentable user")));
    }

    private static Option<Patient> asPatient(TargetOwnerData targetOwnerData) {
        return Some(targetOwnerData.person())
                .filter(Patient.class::isInstance)
                .map(Patient.class::cast);
    }
}
