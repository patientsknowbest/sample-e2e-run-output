package com.pkb.config;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.datatype.guava.GuavaModule;
import com.fasterxml.jackson.datatype.jdk8.Jdk8Module;
import com.fasterxml.jackson.datatype.jsr310.JavaTimeModule;
import com.pkb.common.config.BaseConfig;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.deletion.config.DeletionServiceLiquibaseConfiguration;
import com.pkb.deletion.model.DeletionServiceProperties;
import com.pkb.deletion.routing.DeletionServiceRouting;
import io.micrometer.core.instrument.MeterRegistry;
import io.micrometer.core.instrument.Metrics;
import org.springframework.beans.factory.ObjectProvider;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.ComponentScan;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Import;
import org.springframework.context.annotation.Profile;
import org.springframework.context.annotation.PropertySource;

/**
 * Because PHR is not yet a Spring Boot application, we can't use the deletion service
 * autoconfiguration module. This class replicates most of what that does with a few
 * tweaks to make it work in PHR-land. In particular, we can't rely on Spring Boot
 * to configure up a Jackson ObjectMapper based on modules exposed as beans, so
 * we make our own, and we can't use nice Spring Boot configuration properties
 *
 * Remove once phr is spring-ified properly: PHR-11233
 */
@Configuration
@Import(DeletionServiceLiquibaseConfiguration.class)
@ComponentScan({"com.pkb.domain.deletion", "com.pkb.deletion.service",  "com.pkb.deletion.repository"})
@PropertySource("classpath:/deletion-service.properties")
@Profile("!unittest")
public class PhrDeletionServiceConfiguration {

    @Bean
    public DeletionServiceProperties getDeletionServiceProperties(PhrConfig config) {
        return new PhrConfigBackedDeletionServiceProperties(config);
    }

    @Bean(name = "deletionServiceObjectMapper")
    public ObjectMapper getDeletionServiceObjectMapper() {
        ObjectMapper objectMapper = new ObjectMapper();
        objectMapper.registerModule(new Jdk8Module());
        objectMapper.registerModule(new GuavaModule());
        objectMapper.registerModule(new JavaTimeModule());
        return objectMapper;
    }

    @Bean
    public MeterRegistry getMeterRegistry() {
        return Metrics.globalRegistry;
    }

    @Bean
    public DeletionServiceRouting deletionServiceRouting(PubSubNamespaceService pubSubNamespaceService, @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey, @Qualifier("deletionServiceObjectMapper") ObjectMapper objectMapper, DeletionServiceProperties deletionServiceProperties, ObjectProvider<BaseConfig> baseConfigObjectProvider) {
        return new DeletionServiceRouting(pubSubNamespaceService, serviceAccountKey, objectMapper, deletionServiceProperties, baseConfigObjectProvider);
    }

}
