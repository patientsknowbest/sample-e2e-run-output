package com.pkb.config;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.repository.ReactiveAccountRepository;
import io.r2dbc.postgresql.PostgresqlConnectionConfiguration;
import io.r2dbc.postgresql.PostgresqlConnectionFactory;
import io.r2dbc.spi.ConnectionFactory;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Profile;
import org.springframework.core.env.Environment;
import org.springframework.data.auditing.DateTimeProvider;
import org.springframework.data.jpa.repository.config.EnableJpaAuditing;
import org.springframework.data.jpa.repository.config.EnableJpaRepositories;
import org.springframework.jndi.JndiTemplate;
import org.springframework.orm.jpa.LocalContainerEntityManagerFactoryBean;
import org.springframework.orm.jpa.vendor.HibernateJpaVendorAdapter;
import org.springframework.r2dbc.core.DatabaseClient;
import org.springframework.transaction.jta.JtaTransactionManager;

import javax.naming.NamingException;
import javax.sql.DataSource;
import java.util.Optional;
import java.util.Properties;

@Configuration
@Profile("!unittest")
@EnableJpaRepositories(value = {
        "com.pkb.repository.internal",
        "com.pkb.repository",
        "com.pkb.externalid.repository",
        "com.pkb.service.dataupload.emis.esv2.repository"
}, entityManagerFactoryRef = "springEntityManagerFactory", transactionManagerRef = "springTransactionManager")
@EnableJpaAuditing(dateTimeProviderRef = "dateTimeProvider")
public class DomainDataConfig {

    @Bean
    public LocalContainerEntityManagerFactoryBean springEntityManagerFactory(@Qualifier("springDataSource") DataSource dataSource, Environment env) {
        var properties = new Properties();
        properties.put("hibernate.dialect", "com.pkb.hibernate.PostgreSQLPkbDialect");
        properties.put("hibernate.jdbc.batch_size", env.getProperty("HIBERNATE_JDBC_BATCH_SIZE","10"));
        properties.put("hibernate.order_inserts", env.getProperty("HIBERNATE_ORDER_INSERTS","false"));
        properties.put("hibernate.order_updates", env.getProperty("HIBERNATE_ORDER_UPDATES","false"));
        var emf = new LocalContainerEntityManagerFactoryBean();
        emf.setPackagesToScan("com.pkb.entities", "com.pkb.**.entity", "com.pkb.dataupload.emis.esv2", "org.tolven.core.entity", "org.tolven.security.key");
        emf.setJpaVendorAdapter(new HibernateJpaVendorAdapter());
        emf.setJpaProperties(properties);
        emf.setPersistenceUnitName("spring");
        emf.setJtaDataSource(dataSource);
        return emf;
    }

    @Bean
    public DateTimeProvider dateTimeProvider(DateTimeService dateTimeService) {
        return () -> Optional.of(dateTimeService.now());
    }

    @Bean
    public DataSource springDataSource() throws NamingException {
        return (DataSource) new JndiTemplate().lookup("java:/DefaultDS");
    }

    @Bean
    public JtaTransactionManager springTransactionManager() {
        return new JtaTransactionManager();
    }

    @Bean
    public ConnectionFactory connectionFactory(Environment env) {
            return new PostgresqlConnectionFactory(
                    PostgresqlConnectionConfiguration.builder()
                            .host(env.getRequiredProperty("PHR_DATABASE_HOST"))
                            .port(env.getProperty("PHR_DATABASE_PORT", Integer.class, 5432))
                            .database(env.getProperty("PHR_DATABASE_SCHEMA_NAME", "postgres"))
                            .username(env.getRequiredProperty("PHR_DATABASE_USERNAME"))
                            .password(env.getRequiredProperty("PHR_DATABASE_PASSWORD"))
                            .build()
            );

    }

    @Bean
    public DatabaseClient databaseClient(ConnectionFactory connectionFactory) {
        return DatabaseClient.builder()
                .connectionFactory(connectionFactory)
                .namedParameters(true)
                .build();
    }

    @Bean
    ReactiveAccountRepository reactiveAccountRepository (DatabaseClient databaseClient) {
        return new ReactiveAccountRepository(databaseClient);
    }
}
