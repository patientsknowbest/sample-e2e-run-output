package com.pkb.questionnaireservice.client

import com.pkb.questionnaireservice.PersonPublicId
import com.pkb.questionnaireservice.QuestionnaireId
import com.pkb.questionnaireservice.modelv2.QuestionnaireRequest
import com.pkb.questionnaireservice.modelv2.QuestionnaireResponse
import org.springframework.http.MediaType
import org.springframework.web.reactive.function.client.WebClient
import reactor.adapter.JdkFlowAdapter
import java.util.UUID
import java.util.concurrent.Flow

interface QuestionnaireResponseClient {
    fun getResponsesForQuestionnaire(requestId: UUID, questionnaireId: QuestionnaireId, loggedInUserId: PersonPublicId): Flow.Publisher<QuestionnaireResponse>
    fun getNonRespondersForQuestionnaire(requestId: UUID, questionnaireId: QuestionnaireId): Flow.Publisher<QuestionnaireRequest>
}


class QuestionnaireResponseClientImpl(private val webClient: WebClient) : QuestionnaireResponseClient{

    override fun getResponsesForQuestionnaire(requestId: UUID, questionnaireId: QuestionnaireId, loggedInUserId: PersonPublicId): Flow.Publisher<QuestionnaireResponse> =
            webClient.get().uri {
                it.path("/questionnaire-response/all/{questionnaireId}")
                        .queryParam("user_id", loggedInUserId)
                        .build(questionnaireId)

            }
                    //We have to use the deprecated value until we upgrade Spring in PHR
                    .accept(MediaType.APPLICATION_NDJSON)
                    .header("X-Request-ID", requestId.toString())
                    .retrieve()
                    .bodyToFlux(QuestionnaireResponse::class.java)
                    .let(JdkFlowAdapter::publisherToFlowPublisher)


    override fun getNonRespondersForQuestionnaire(requestId: UUID, questionnaireId: QuestionnaireId): Flow.Publisher<QuestionnaireRequest> =
        webClient.get().uri {
            it.path("/questionnaire-request/non-responders/all/{questionnaireId}").build(questionnaireId)
        }
            .header("X-Request-ID", requestId.toString())
            .retrieve()
            .bodyToFlux(QuestionnaireRequest::class.java)
            .let(JdkFlowAdapter::publisherToFlowPublisher)

}
