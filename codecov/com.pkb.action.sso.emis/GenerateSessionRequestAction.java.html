package com.pkb.action.sso.emis;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.crypto.SSOCryptoHelper;
import com.pkb.datamodel.Email;
import com.pkb.datamodel.SSOIdLink;
import com.pkb.domain.SSOService;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.entities.enums.SSOType;
import com.pkb.entities.enums.ValidNationalId;
import com.pkb.service.user.impl.UserManager;
import com.pkb.sso.emis.SessionRequestGenerator;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.PersonAmbiguityHandler;
import com.pkb.util.SessionRequestUtil;
import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import uk.co.e_mis.integration.portalsdk.entities.SessionRequest;

import javax.xml.bind.JAXBException;
import javax.xml.datatype.DatatypeConfigurationException;
import java.util.Objects;
import java.util.Optional;
import java.util.stream.Stream;

/**
 * Generate SessionRequest XMl to be used for simulating SSO request from EMIS client for manual testing * monitoring
 */
public class GenerateSessionRequestAction extends BaseAction {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private String patientNSHNumber;

    private String ssoUserName;

    private String ssoPassword;

    private String sessionRequestXml;

    private SSOService ssoService;

    private PKBPerson testPatient;

    private PKBPerson testProfessional;

    private SessionRequest sessionRequest;

    private SSOCryptoHelper ssoCryptoHelper;

    public String generateSessionRequest() throws DatatypeConfigurationException {
        if (!isTestServer()) {
            // only allow monitoring user on production server to use this feature
            if (getMonitoringProfSSOUsername().map(s -> !ssoUserName.equals(s))
                    .orElse(true)) {
                addActionError(getText("sso.emis.err.manual_testing_disabled"));
                return Action.ERROR;
            }
        }

        if (!validateInput()) {
            return Action.ERROR;
        }

        SessionRequestGenerator sessionRequestGenerator = new SessionRequestGenerator(dateTimeService, ssoCryptoHelper);
        SessionRequestUtil sessionRequestUtil = new SessionRequestUtil();
        ValidNationalId patientNationalId = NationalIdType.NHS_NUMBER.getValidNationalIdAndType(patientNSHNumber).orElseThrow();
        try {
            sessionRequest = sessionRequestGenerator.generate(testProfessional, testPatient, ssoUserName, ssoPassword, patientNationalId);
            sessionRequestXml = sessionRequestUtil
                    .toXml(sessionRequest);
        } catch (JAXBException ignored) {
            addActionError(getText("sso.emis.err.unable_to_generate_session_request_xml"));
            return Action.ERROR;
        }
        return Action.SUCCESS;
    }

    private Optional<String> getMonitoringProfSSOUsername() {
        Email monitoringProfEmail = getConfig().getEMISSSOMonitoringProfEmail();
        try {
            PKBPerson monitoringProf = userManager.getSinglePersonByEmail(monitoringProfEmail);
            return (ssoService.getSSOIdLinkForPersonIdAndSSOType(monitoringProf.getId(), SSOType.EMIS)
                    .map(SSOIdLink::getSsoUsername));
        } catch (Exception e) {
            LOGGER.info("Error while getting SSOUsername for EMIS SSO monitoring professional {}", monitoringProfEmail, e);
            return Optional.empty();
        }
    }

    private boolean validateInput() {
        boolean valid = true;

        if (Stream.of(patientNSHNumber, ssoUserName, ssoPassword).anyMatch(Objects::isNull)) {
            addActionError(getText("sso.emis.err.missing_input_fields"));
            return false;
        }

        // clean up
        patientNSHNumber = patientNSHNumber.trim();
        ssoUserName = ssoUserName.trim();
        ssoPassword = ssoPassword.trim();

        if (Stream.of(patientNSHNumber, ssoUserName, ssoPassword).anyMatch(StringUtils::isBlank)) {
            addActionError(getText("sso.emis.err.missing_input_fields"));
            return false;
        }

        Option<PKBPerson> maybePerson = NationalIdType.NHS_NUMBER.getValidNationalIdAndType(patientNSHNumber)
                .map(vnid -> userManager.getPKBPersonByNationalId(vnid, PKBPerson.Lazy.CONTACTS))
                .map(e -> PersonAmbiguityHandler.getMaybePersonOrNone(e, "nationalId"))
                .orElse(Option.none());

        if (maybePerson.isDefined()) {
            testPatient = maybePerson.get();
        } else {
            addActionError(getText("sso.emis.err.patient_nhs_not_found", new String[]{patientNSHNumber}));
        }

        testProfessional = ssoService.getSSOIdLinkForSSOUsernameAndSSOType(ssoUserName, SSOType.EMIS)
                .map(ssoIdLink -> userManager.getPKBPerson(ssoIdLink.getPersonId(), PKBPerson.Lazy.CONTACTS))
                .orElse(null);

        if (testProfessional == null) {
            addActionError(getText("sso.emis.err.professional_ssousername_not_found", new String[]{ssoUserName}));
        }
        if (!getActionErrors().isEmpty()) {
            valid = false;
        }
        return valid;
    }

    public String getSsoUserName() {
        return ssoUserName;
    }

    public void setSsoUserName(String ssoUserName) {
        this.ssoUserName = ssoUserName;
    }

    public String getSsoPassword() {
        return ssoPassword;
    }

    public void setSsoPassword(String ssoPassword) {
        this.ssoPassword = ssoPassword;
    }

    public String getSessionRequestXml() {
        return sessionRequestXml;
    }

    public void setSessionRequestXml(String sessionRequestXml) {
        this.sessionRequestXml = sessionRequestXml;
    }

    public String getPatientNSHNumber() {
        return patientNSHNumber;
    }

    public void setPatientNSHNumber(String patientNSHNumber) {
        this.patientNSHNumber = patientNSHNumber;
    }

    @Autowired
    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    @Autowired
    public void setSsoService(SSOService ssoService) {
        this.ssoService = ssoService;
    }

    @Autowired
    public void setSsoCryptoHelper(SSOCryptoHelper ssoCryptoHelper) {
        this.ssoCryptoHelper = ssoCryptoHelper;
    }

    public SessionRequest getSessionRequest() {
        return sessionRequest;
    }
}
