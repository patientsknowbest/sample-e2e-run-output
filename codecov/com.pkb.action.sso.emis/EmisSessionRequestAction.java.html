package com.pkb.action.sso.emis;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.action.BasePingAction;
import com.pkb.domain.SSOService;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.entities.enums.SSOType;
import com.pkb.service.sso.impl.emis.EMISSSOAuthManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.sso.SsoMetrics;
import com.pkb.sso.emis.EMISSessionRequestValidator;
import com.pkb.sso.emis.EMISSessionRequestValidator.SessionRequestErrorCode;
import com.pkb.sso.emis.SessionRequestConverter;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.PersonAmbiguityHandler;
import com.pkb.util.SessionRequestUtil;
import io.prometheus.client.Summary;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Autowired;
import uk.co.e_mis.integration.portalsdk.entities.SessionRequest;

import javax.ws.rs.WebApplicationException;
import java.io.IOException;
import java.util.List;

import static com.pkb.util.PKBWebUtil.CONTEXT_USER_ID;

/**
 * Service to intercept SSO requests from EMIS client
 * <p>
 * Function :
 * <ul>
 * <li>Intercept the emis request from EMIS client</li>
 * <li>Validate the SessionRequest xml posted and flag errors if necessary. E.g
 * malformed XML</li>
 * <li>Authenticate the requesting user (professional) If all vital information
 * is available in SessionRequest like - patient details , Portal Id for PKB ,
 * request provenance etc</li>
 * <li>Redirect to patient summary page if authentication is successful</li>
 * <li>Log the SSO request in PKB database for audit purpose</li>
 * </ul>
 *
 * @author pravina
 */
public class EmisSessionRequestAction extends BaseAction {

    private static final long serialVersionUID = 4614644885276096123L;

    static final String FIRST_SSO_REQUEST = "firstSSORequestPage";

    static final String PING_EMIS_SESSION_REQUEST_SERVICE = "pingEmisSessionRequestService";
    static final String EMIS_SESSION_REQUEST_PING_RESULT = "emisSessionRequestPingResult";

    private SSOService ssoService;
    private EMISSSOAuthManager ssoAuthManager;
    private SsoMetrics ssoMetrics;

    private String emisSSOKey; // As sent in session request

    private String emisSSOUserName; // As sent in session request

    private String emisPasswordHash; // As sent in session request

    private SessionRequestUtil sessionRequestUtil = new SessionRequestUtil();

    private EMISSessionRequestValidator emisSessionRequestValidator = new EMISSessionRequestValidator();

    private String pingResult = BasePingAction.PingStatus.FAIL.toString(); // result of ping set up for health check of portal service

    public String processSessionRequest() {

        // Remove the contextUserId from the old session as part of the launch action. We have to do this as the EMIS
        // client appears to reuse the session. Before the spring security refactor we were getting away with this by
        // subverting validation preventing a patient switch by overriding this attribute as part of the launch action.
        // Now we should explicitly clear it down so there can't be any cross-contamination when doing portal loads
        // in the EMIS client. Invalidation of the entire session causes lots of knock-on problems so we've rejected
        // it as a solution.
        request.getSession().removeAttribute(CONTEXT_USER_ID);

        Summary.Timer timer = ssoMetrics.startTimerForProcessingSessionRequest(SSOType.EMIS);
        try {
            SessionRequest emisSessionRequest;
            try {
                emisSessionRequest = parseAndValidateSessionRequest();
                if (hasActionErrors()) {
                    return Action.ERROR;
                }
            } catch (IOException | WebApplicationException ignored) {
                return handleBadRequest("ssoEMISError.err.bad_request");
            }

            emisSSOUserName = emisSessionRequest.getProvenance().getUser().getExternalCredentials().getUsername();
            emisPasswordHash = emisSessionRequest.getProvenance().getUser().getExternalCredentials().getPassword().getValue();
            String[] credentials = emisPasswordHash.split(":");
            emisSSOKey = credentials[2];

            if (emisSSOUserName.equals(PING_EMIS_SESSION_REQUEST_SERVICE)) {
                pingResult = BasePingAction.PingStatus.OKAY.toString();
                return EMIS_SESSION_REQUEST_PING_RESULT;
            }

            String patientNHSNumber = emisSessionRequest.getPatient().getIdentifier().get(0).getValue();
            Option<PKBPerson> maybePerson = NationalIdType.NHS_NUMBER.getValidNationalIdAndType(patientNHSNumber)
                    .map(userManager::getPKBPersonByNationalId)
                    .map(e -> PersonAmbiguityHandler.getMaybePersonOrNone(e, "nationalId"))
                    .orElse(Option.none());

            if (maybePerson.isEmpty()) {
                return handleBadRequest("ssoEMISError.err.patient_not_found");
            }

            logEMISSSORequest(emisSessionRequest);
            contextUserId = maybePerson.get().getId();

            if (ssoService.authRecordExists(emisSSOUserName, SSOType.EMIS)) {
                return Action.SUCCESS;
            }

            return processRequestWithoutValidSSOAuth();

        } finally {
            timer.observeDuration();
        }

    }

    @NotNull
    private String handleBadRequest(String errorCode) {
        addActionError(getText(errorCode));
        ssoMetrics.incrementUnSuccessfulSessionRequest(SSOType.EMIS, errorCode);

        return Action.ERROR;
    }

    private void logEMISSSORequest(SessionRequest emisSessionRequest) {
        SessionRequestConverter sessionRequestConverter = new SessionRequestConverter(emisSessionRequest, emisSSOUserName, dateTimeService);
        ssoService.saveEMISSSORequestLog(sessionRequestConverter.getEMISSSORequestLog());
    }

    private String processRequestWithoutValidSSOAuth() {
        // Either valid sso auth record doesn't exist or
        // SSO creds were not created from PKB UI
        return ssoService.getSSOIdLinkForSSOUsernameAndSSOType(emisSSOUserName, SSOType.EMIS)
                .map(ssoIdLink -> {
                    // First SSO request so redirect to login page to accept PKB creds
                    return FIRST_SSO_REQUEST;
                }).orElseGet(() -> {
                    // professional hasn't created SSO creds from PKB UI ; raise error message informing the same
                    return handleBadRequest("ssoEMISError.err.pkb_emis_link_absent");
                });
    }

    /**
     * Parse SessionRequest XML & validate it
     */
    private SessionRequest parseAndValidateSessionRequest() throws IOException {
        SessionRequest emisSessionRequest = sessionRequestUtil.parseSessionRequest(getRequest().getReader());
        List<SessionRequestErrorCode> errorCodes = emisSessionRequestValidator.validate(emisSessionRequest);
        if (!errorCodes.isEmpty()) {
            errorCodes.forEach(code -> addActionError(getText(code.getI18nCode())));
        }
        return emisSessionRequest;
    }

    @Autowired
    public void setSsoService(SSOService ssoService) {
        this.ssoService = ssoService;
    }

    @Autowired
    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    @Autowired
    public void setSsoAuthManager(EMISSSOAuthManager ssoAuthManager) {
        this.ssoAuthManager = ssoAuthManager;
    }

    @Autowired
    public void setSsoMetrics(SsoMetrics ssoMetrics) {
        this.ssoMetrics = ssoMetrics;
    }

    public String getEmisSSOKey() {
        return emisSSOKey;
    }

    public void setEmisSSOKey(String emisSSOKey) {
        this.emisSSOKey = emisSSOKey;
    }

    public String getEmisSSOUserName() {
        return emisSSOUserName;
    }

    public void setEmisSSOUserName(String emisSSOUserName) {
        this.emisSSOUserName = emisSSOUserName;
    }

    public String getEmisPasswordHash() {
        return emisPasswordHash;
    }

    public void setEmisPasswordHash(String emisPasswordHash) {
        this.emisPasswordHash = emisPasswordHash;
    }

    public String getPingResult() {
        return pingResult;
    }

    public void setPingResult(String pingResult) {
        this.pingResult = pingResult;
    }

}
