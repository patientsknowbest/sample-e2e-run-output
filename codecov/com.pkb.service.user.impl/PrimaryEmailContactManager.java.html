package com.pkb.service.user.impl;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.consent.RequiresConsent;
import com.pkb.entities.enums.AccessingEntityType;
import com.pkb.entities.enums.UserStatus;
import com.pkb.entities.enums.UserType;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.user.entity.PersonContact;
import com.pkb.user.entity.PrimaryEmailContactDTO;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import io.vavr.control.Option;

import java.util.ArrayList;

import static io.vavr.collection.List.ofAll;

public class PrimaryEmailContactManager  extends TransactionManager {

    private final PatientConsentManager patientConsentManager;
    private final PersonContactManager personContactManager;

    public PrimaryEmailContactManager(PhrConfig config, TolvenBeanFactory beanFactory,
                                      DateTimeService dateTimeService, UUIDProvider uuidProvider,
                                      PatientConsentManager patientConsentManager,
                                      PersonContactManager personContactManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.patientConsentManager = patientConsentManager;
        this.personContactManager = personContactManager;
    }

    /**
     * Get the primary email contact dto for the user. If a requestContext and consentRequiredOrNull are both specified,
     * also_contacts will not be included as appropriate to the consentRequired (ie if, for example,  a user has
     * sexualHealth permission and a data point requires it). Otherwise also_contacts won't be included
     *
     * @param userId The user whose primary email contact information should be retrieved
     * @return A {@link PersonEmailContactDTO} object representing primary email contact information for this user
     */
    public PrimaryEmailContactDTO getPrimaryEmailContactDTO(long userId, RequiresConsent consentRequiredOrNull) {
        try {
            var dto = new PrimaryEmailContactDTO();
            var all = personContactManager.getContactsWithRefPersonInOrder(userId);
            var alsoContactPrimaryEmailContacts = new ArrayList<PersonContact>();
            for (PersonContact pc : all) {
                if (consentRequiredOrNull != null) {
                    //noinspection deprecation
                    pc.findAlsoContactPerson()
                            .filter(refPerson ->
                                    UserStatus.DEAD != refPerson.getStatus()
                                            && refPerson.getUserType() == UserType.PATIENT)
                            .flatMap(refPerson ->
                                    Option.ofOptional(patientConsentManager.findPatientConsent(
                                                    userId,
                                                    refPerson.getId(), AccessingEntityType.CARER))
                                            .filter(consent -> !consent.discharged() && consent.isEntityVisible(consentRequiredOrNull))
                                            .flatMap(ignored -> personContactManager.findPrimaryContact(refPerson.getId())))
                            .filter(refPersonPrimary -> refPersonPrimary.getContactIfPrimaryEmail().isDefined())
                            .peek(alsoContactPrimaryEmailContacts::add);
                }
                if (pc.getContactIfPrimaryEmail().isDefined()) {
                    dto = new PrimaryEmailContactDTO(pc);
                }
            }
            // PHR-4579: we need to deduplicate until the underlying data is cleaned up
            dto.setAlsoContactPrimaryEmailContactList(ofAll(alsoContactPrimaryEmailContacts)
                    .distinctBy(PersonContact::getId)
                    .asJava());
            return dto;
        } catch (Exception e) {
            throw new RuntimeException("Exception while getting primary contact DTO for user-" + userId, e);
        }
    }
}
