package com.pkb.service.user.impl;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.BulkInvitationOutcome;
import com.pkb.domain.BulkInvitationOutcomeService;
import com.pkb.entities.enums.OperationOutcomeStatus;
import com.pkb.institute.entity.Team;
import com.pkb.service.user.BulkInvitationOutcomesManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import org.jetbrains.annotations.NotNull;

import java.time.LocalDateTime;
import java.util.List;
import java.util.Optional;
import java.util.function.Function;

import static com.pkb.entities.enums.OperationOutcomeStatus.FAILED;
import static com.pkb.entities.enums.OperationOutcomeStatus.FAILED_FOR_UNEXPECTED_REASON;
import static com.pkb.entities.enums.OperationOutcomeStatus.IN_PROGRESS;
import static com.pkb.entities.enums.OperationOutcomeStatus.PARTIALLY_SUCCEEDED;
import static com.pkb.entities.enums.OperationOutcomeStatus.SUCCEEDED;

public class RemoteAwareBulkInvitationOutcomesManager extends TransactionManager implements BulkInvitationOutcomesManager {

    private final BulkInvitationOutcomeService bulkInvitationOutcomeService;

    public RemoteAwareBulkInvitationOutcomesManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, BulkInvitationOutcomeService bulkInvitationOutcomeService) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.bulkInvitationOutcomeService = bulkInvitationOutcomeService;
    }

    @Override
    public @NotNull BulkInvitationOutcome registerBulkInvitationStart(@NotNull PKBPerson initiator, @NotNull Team invitingTeam, @NotNull LocalDateTime initiatedAt) {
        return wrapExceptions(outcomesRemote -> outcomesRemote.createOutcome(initiator.getId(), invitingTeam.getId(), initiatedAt, IN_PROGRESS));
    }

    @Override
    public @NotNull BulkInvitationOutcome markAsFinished(@NotNull BulkInvitationOutcome outcome, @NotNull String csvEnrichedWithOutcomes, int successCount, int failureCount) {
        BulkInvitationOutcome result;
        if (successCount > 0) {
            if (failureCount > 0) {
                result = updateOutcome(outcome, csvEnrichedWithOutcomes, PARTIALLY_SUCCEEDED);
            } else {
                result = updateOutcome(outcome, csvEnrichedWithOutcomes, SUCCEEDED);
            }
        } else if (failureCount > 0) {
            result = updateOutcome(outcome, csvEnrichedWithOutcomes, FAILED);
        } else {
            result = updateOutcome(outcome, csvEnrichedWithOutcomes, FAILED_FOR_UNEXPECTED_REASON);
        }
        return result;
    }

    @Override
    public @NotNull BulkInvitationOutcome markAsFailed(@NotNull PKBPerson initiator, @NotNull Team invitingTeam, @NotNull LocalDateTime initiatedAt) {
        return wrapExceptions(outcomesRemote -> {
            Optional<BulkInvitationOutcome> outcome = outcomesRemote.markAsFailed(initiator.getId(), invitingTeam.getId(), initiatedAt);
            if (outcome.isEmpty()) {
                return outcomesRemote.createOutcome(initiator.getId(), invitingTeam.getId(), initiatedAt, FAILED_FOR_UNEXPECTED_REASON);
            }
            return outcome.get();
        });
    }

    @Override
    public @NotNull List<BulkInvitationOutcome> findLastInvitationOutcomesForTeam(long teamId) {
        return wrapExceptions(bulkInvitationOutcomeService -> bulkInvitationOutcomeService.findLastInvitationOutcomesForTeam(teamId, config.getMaxBulkInvitationOutcomeReport()));
    }

    @Override
    public @NotNull Optional<String> findOutcomeById(long teamId, long reportId) {
        return wrapExceptions(remote -> remote.findOutcome(teamId, reportId));
    }

    // ========== Helper methods below. ==========

    private BulkInvitationOutcome updateOutcome(BulkInvitationOutcome outcomeId, String csvEnrichedWithOutcomes, OperationOutcomeStatus status) {
        return wrapExceptions(outcomesRemote -> outcomesRemote.updateOutcome(outcomeId, csvEnrichedWithOutcomes, status));
    }

    private <T> T wrapExceptions(Function<BulkInvitationOutcomeService, T> mapping) {
        try {
            return mapping.apply(bulkInvitationOutcomeService);
        } catch (Exception cause) {
            throw new IllegalStateException(cause);
        }
    }
}
