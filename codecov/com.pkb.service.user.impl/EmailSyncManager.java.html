package com.pkb.service.user.impl;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.Email;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.user.PKBPersonRemote;
import com.pkb.user.entity.PKBPerson;
import com.pkb.user.entity.PersonContact;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.invoke.MethodHandles;
import java.util.ArrayList;
import java.util.Collection;
import java.util.Comparator;
import java.util.Objects;
import java.util.Optional;
import java.util.UUID;
import java.util.function.Predicate;

/**
 * This class handles the synchronization of email addresses across multiple
 * accounts of the same person which are linked via the humanUUID.
 */
public class EmailSyncManager extends TransactionManager {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
    private static final Predicate<PersonContact> notConfirmed = (PersonContact contact) -> !contact.isConfirmed();
    private static final Predicate<PersonContact> emailType = (PersonContact contact) -> contact.getContactIfEmail().isDefined();

    private final PersonContactManager contactManager;

    public EmailSyncManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, PersonContactManager contactManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.contactManager = contactManager;
    }

    /**
     * Synchronize every person's email addresses that has multiple accounts
     * 
     * @param requestContext
     *            {@link EHRRequestContext}
     */
    public void syncEveryPersonWithMultipleAccounts(@NotNull EHRRequestContext requestContext) {
        PKBPersonRemote personRemote = beanFactory.getPKBPersonBean();
        syncHumanUUIDList(requestContext, List.ofAll(personRemote.getHumanUUIDListForMultiplePKBPerson()));
    }

    /**
     * Synchronize a single person's email addresses that has multiple accounts
     * 
     * @param requestContext
     *            {@link EHRRequestContext}
     * @param person
     *            The person with multiple accounts
     */
    public void syncSinglePersonWithMultipleAccounts(@NotNull EHRRequestContext requestContext, @NotNull PKBPerson person) {
        syncHumanUUIDList(requestContext, List.of(person.getHumanUUID()));
    }

    /**
     * Synchronize a person's email addresses given a list of humanUUIDs
     * 
     * @param requestContext
     *            {@link EHRRequestContext}
     * @param uuids
     *            The list of UUIDs
     */
    public void syncHumanUUIDList(@NotNull EHRRequestContext requestContext, @NotNull List<String> uuids) {
        PKBPersonRemote personRemote = beanFactory.getPKBPersonBean();
        uuids.filter(Objects::nonNull).distinct().forEach(uuid -> {
            LOGGER.info("Started synchronizing email addresses for-{}", uuid);
            List<PKBPerson> accounts = List.ofAll(personRemote.getPKBPersonListByHumanUUID(uuid, PKBPerson.Lazy.CONTACTS));
            List<PersonContact> contacts = accounts.flatMap(PKBPerson::getContacts).filter(emailType);
            Optional<PersonContact> primaryContact = getMostRecentPrimaryEmailContact(contacts);
            List<PersonContact> confirmedContacts = getConfirmedEmailContacts(contacts);
            accounts.forEach(account -> {
                LOGGER.info("Starting processing email addresses for-{}", account.getId());
                java.util.List<PersonContact> changes = new ArrayList<>();
                primaryContact.ifPresent(address -> changes.addAll(syncPrimary(account, address).asJava()));
                changes.addAll(syncConfirmed(account, confirmedContacts).asJava());
                if (!changes.isEmpty()) {
                    try {
                        beginTransaction();
                        contactManager.saveContactList(requestContext, changes);
                        commitTransaction();
                    } finally {
                        rollbackIfNotCommitted();
                    }
                }
                LOGGER.info("Finished processing email addresses for-{}", account.getId());
            });
            LOGGER.info("Finished synchronizing email addresses for-{}", uuid);
        });
    }

    protected List<PersonContact> syncPrimary(@NotNull PKBPerson person, @NotNull PersonContact primaryContact) {
        return primaryContact.getContactIfEmail().map(email ->
        {
            java.util.List<PersonContact> changes = new ArrayList<>();
            person.getContacts().stream().filter(PersonContact::isPrimary).forEach(contact -> {
                Email value = contact.getContactIfEmail().getOrNull();
                if (value != null && email.address().compareToIgnoreCase(value.address()) != 0) {
                    // Only need to unset the primary flag and make sure the email address is lowercase
                    contact.setPrimary(false);
                    contact.setContactAsEmail(value);
                    changes.add(contact);
                    LOGGER.info("Changing-{}-to no longer be primary email address for-{}", value.address(), person.getId());
                }
            });
            Optional<PersonContact> found = findContactWithSameEmailAddress(person.getContacts(), email);
            if (found.isPresent()) {
                PersonContact contact = found.get();
                if (!contact.isPrimary()) {
                    changes.add(copyOrCreateContact(person, primaryContact, contact, true));
                    LOGGER.info("Changing-{}-to be primary email address for-{}", contact.getContactIfEmail().get().address(), person.getId());
                }
            } else {
                changes.add(copyOrCreateContact(person, primaryContact, null, true));
                LOGGER.info("Adding-{}-as primary email address for-{}", email, person.getId());
            }
            
            return List.ofAll(changes);
        }).getOrElse(() -> List.empty());
    }

    protected List<PersonContact> syncConfirmed(@NotNull PKBPerson person, @NotNull List<PersonContact> confirmedContacts) {
        java.util.List<PersonContact> changes = new ArrayList<>();
        person.getContacts().stream().filter(notConfirmed).forEach(personContact -> confirmedContacts.forEach(confirmed -> {
            if (personContact.contactValueEquals(confirmed)) {
                changes.add(copyOrCreateContact(person, confirmed, personContact, personContact.isPrimary()));
                LOGGER.info("Changing-{}-to be confirmed for-{}", personContact, person.getId());
            }
        }));
        return List.ofAll(changes);
    }

    protected Optional<PersonContact> getMostRecentPrimaryEmailContact(@NotNull List<PersonContact> contacts) {
        return contacts
                .filter(contact -> contact.getContactIfPrimaryEmail().isDefined())
                .maxBy(Comparator.comparingLong(contact -> contact.getVersionDetails().getEnteredDate().getTime()))
                .toJavaOptional();
    }

    protected List<PersonContact> getConfirmedEmailContacts(@NotNull List<PersonContact> contacts) {
        return contacts.filter(contact -> contact.isConfirmed() && contact.getContactIfEmail().isDefined());
    }

    protected Optional<PersonContact> findContactWithSameEmailAddress(@NotNull Collection<PersonContact> contacts, @NotNull Email address) {
        var target = Option.of(address);
        return contacts
                .stream()
                .filter(contact -> contact.getContactIfEmail().equals(target))
                .findAny();
    }

    protected PersonContact copyOrCreateContact(@NotNull PKBPerson person, @NotNull PersonContact from, PersonContact to, boolean isPrimary) {
        PersonContact result = to;
        if (result == null) {
            result = new PersonContact();
            result.setPublicId(UUID.randomUUID());
        }
        result.setPrimary(isPrimary);
        result.setConfirmed(from.isConfirmed());
        from.getContactIfEmail().peek(result::setContactAsEmail);
        from.findAlsoContactPerson().peek(result::setToAlsoContact);
        from.getContactIfEmail().peek(result::setContactAsEmail);
        from.getContactIfPhone().peek(result::setContactAsPhone);
        result.setPerson(person);
        result.setVersionDetails(from.getVersionDetails());
        return result;
    }
}
