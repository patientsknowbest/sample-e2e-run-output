package com.pkb.service.user.impl;

import com.pkb.app.entity.PatientAuditLog;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.repository.legacy.LegacyAuditLogRepository;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import org.springframework.data.domain.PageRequest;

import java.util.List;

public class AuditLogManager extends TransactionManager {
    private final LegacyAuditLogRepository auditLogRepository;

    public AuditLogManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, LegacyAuditLogRepository auditLogRepository) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.auditLogRepository = auditLogRepository;
    }

    public List<PatientAuditLog> getAuditLogForPatient(long patientId, int offset, int count, boolean hideOwnAccess) {
        int page = count > 0 ? offset/count : 0;
        var pageRequest = PageRequest.of(page, count);
        if (hideOwnAccess) {
            return auditLogRepository.getPatientAuditLogsHideSelf(patientId, pageRequest);
        }
        else {
            return auditLogRepository.getAllPatientAuditLogs(patientId, pageRequest);
        }
    }
}
