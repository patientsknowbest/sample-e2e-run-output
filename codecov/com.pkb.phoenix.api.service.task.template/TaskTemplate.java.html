package com.pkb.phoenix.api.service.task.template;

import com.pkb.phoenix.repository.task.projection.ImmutableTask;
import com.pkb.phoenix.repository.task.projection.Task;
import reactor.core.publisher.Mono;

import java.util.List;

/**
 * A {@link TaskTemplate} captures common bits of a task like:
 * - its code (used to identify),
 * - parameter validation,
 * - how it should be executed.
 *
 * It's really like a blue print which needs user input (parameters) to get going.
 *
 * @param <PARAMETER> - Data structure of parameters for a specific task instance.
 * @param <RESULT> - Result data structure (when a task instance is executed).
 */
public interface TaskTemplate<PARAMETER, RESULT> {
    /**
     * A task template is identified by a code.
     */
    TaskTemplateCode code();

    List<String> validate(PARAMETER parameters);

    Mono<RESULT> execute(Task<PARAMETER> parameters);

    /**
     * This method must exists for technical reasons as long as
     * the DAO layer is not capable of returning parameters with the correct generic type.
     */
    default Task<PARAMETER> adapt(Task<String> task) {
        return ImmutableTask.<PARAMETER>builder()
                .id(task.getId())
                .typeCode(task.getTypeCode())
                .parameters(adapt(task.getParameters()))
                .events(task.events())
                .latestEvent(task.getLatestEvent())
                .build();
    }

    /**
     * In long term, the logic encapsulated in this method should move down to the DAO layer.
     * @param parameters - task instance parameters represented as raw string
     * @return task instance parameters as a POJO.
     */
    PARAMETER adapt(String parameters);
}
