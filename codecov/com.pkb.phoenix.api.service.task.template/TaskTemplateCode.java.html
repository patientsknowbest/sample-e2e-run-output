package com.pkb.phoenix.api.service.task.template;

import com.pkb.phoenix.api.service.Parameter;
import com.pkb.phoenix.api.service.task.template.datagathering.patient.DataCollectionParameters;
import com.pkb.phoenix.api.service.task.template.datagathering.patient.PatientDetails;
import com.pkb.phoenix.api.service.task.template.emaildeletion.EmailDeletionParameters;
import com.pkb.phoenix.api.service.task.template.emaildeletion.EmailDeletionResult.FdEmailDeletionResult;
import com.pkb.phoenix.api.service.task.template.messagedeletion.MessageDeletionsParameters;
import com.pkb.phoenix.api.service.task.template.messagedeletion.MessageDeletionsResult.ExtendedMessageDeletionsResult;
import com.pkb.phoenix.api.service.task.template.twofaremoval.TwoFARemovalParameters;
import com.pkb.phoenix.api.service.task.template.twofaremoval.TwoFARemovalResult.FdTwoFARemovalResult;

import static com.pkb.phoenix.api.service.tasktype.Approval.NONE;
import static com.pkb.phoenix.api.service.tasktype.Approval.SIMPLE;
import static com.pkb.phoenix.api.service.tasktype.Approval.SINGLE;

/**
 * Represents the mapping of task types as used by the different task templates to the codes used in the api, approval required, parameter and result types
 *
 * Don't delete from this enum once released and at least one task instance was created in production.
 */
public enum TaskTemplateCode {
    EMAIL_DEL(NONE, EmailDeletionParameters.class, FdEmailDeletionResult.class, "EMAIL_DEL"),
    PATIENT_DATA(NONE, DataCollectionParameters.class, PatientDetails.class, "PATIENT_DATA"),
    MSGD_CIE(SIMPLE, MessageDeletionsParameters.class, ExtendedMessageDeletionsResult.class, "MSGD_CIE"),
    MSGD_GEN(SINGLE, MessageDeletionsParameters.class, ExtendedMessageDeletionsResult.class, "MSGD_GEN"),
    TWO_FA_REMOVAL(SIMPLE, TwoFARemovalParameters.class, FdTwoFARemovalResult.class, "TWO_FA_REMOVAL");

    private final Approval approval;
    private final Class<? extends Parameter> type;
    private final Class<?> resultType;
    private final String apiCode;

    TaskTemplateCode(Approval approval, Class<? extends Parameter> type, Class<?> resultType, String apiCode) {
        this.approval = approval;
        this.type = type;
        this.resultType = resultType;
        this.apiCode = apiCode;
    }

    public String apiCode() {
        return apiCode;
    }

    public Approval approval() {
        return approval;
    }

    public Class<? extends Parameter> parameterType() {
        return type;
    }

    public Class<?> resultType() {
        return resultType;
    }
}
