package com.pkb.phoenix.api.service.task.template;

import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.phoenix.api.service.task.template.emaildeletion.TaskResponseListener;
import com.pkb.phoenix.api.wiring.config.PhoenixPubSubConfig;
import com.pkb.phoenix.shared.structures.response.TaskResponse;
import org.apache.camel.component.jackson.JacksonDataFormat;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import static com.pkb.phoenix.api.service.task.template.SupportToolPubSubUtil.uri;

public class TaskResponseInboundRouteBuilder extends AbstractNamespaceAwareRouteBuilder {

    private final PubSubNamespaceService pubSubNamespaceService;
    private final PhoenixPubSubConfig config;
    private final TaskResponseListener processor;
    private final @Nullable String serviceAccountKey;

    public TaskResponseInboundRouteBuilder(PubSubNamespaceService pubSubNamespaceService,
                                           PhoenixPubSubConfig config,
                                           TaskResponseListener processor,
                                           @Nullable String serviceAccountKey) {
        this.pubSubNamespaceService = pubSubNamespaceService;
        this.config = config;
        this.processor = processor;
        this.serviceAccountKey = serviceAccountKey;
    }

    @Override
    public @NotNull PubSubNamespaceService camelNamespaceService() {
        return pubSubNamespaceService;
    }

    @Override
    public boolean throwExceptionForNamespaceMismatch() {
        return true;
    }

    @Override
    public void configure() {
        from(uri(config.getPubSubProjectName(), config.taskCompletedSubscriptionName(), serviceAccountKey))
                .unmarshal(new JacksonDataFormat(TaskResponse.class))
                .process(processor)
                .onException(Throwable.class).maximumRedeliveries(0);
    }
}
