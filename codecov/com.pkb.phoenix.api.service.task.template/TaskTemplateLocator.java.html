package com.pkb.phoenix.api.service.task.template;

import org.jetbrains.annotations.NotNull;

import java.util.Map;
import java.util.Optional;

import static java.lang.String.format;
import static java.util.Optional.ofNullable;

public class TaskTemplateLocator {
    private final Map<TaskTemplateCode, TaskTemplate<?, ?>> templates;

    public TaskTemplateLocator(Map<TaskTemplateCode, TaskTemplate<?, ?>> templates) {
        this.templates = templates;
    }

    @SuppressWarnings("unchecked")
    public <P, R> Optional<TaskTemplate<P, R>> find(@NotNull TaskTemplateCode taskTemplateCode) {
        //noinspection unchecked
        return ofNullable((TaskTemplate<P, R>) templates.get(taskTemplateCode));
    }

    public <P, R> TaskTemplate<P, R> get(@NotNull TaskTemplateCode code) {
        return this.<P, R>find(code)
                .orElseThrow(() -> new IllegalArgumentException(format("No Task Type is found with code=[%s]", code)));    }
}
