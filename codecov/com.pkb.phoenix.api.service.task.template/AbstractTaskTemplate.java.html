package com.pkb.phoenix.api.service.task.template;

import com.fasterxml.jackson.databind.ObjectReader;

import java.io.IOException;

public abstract class AbstractTaskTemplate<PARAMETER, RESULT> implements TaskTemplate<PARAMETER, RESULT> {
    private final TaskTemplateCode code;
    final Class<PARAMETER> parameterType;
    protected final ObjectReader json;

    public AbstractTaskTemplate(TaskTemplateCode code, Class<PARAMETER> parameterType, ObjectReader json) {
        this.code = code;
        this.parameterType = parameterType;
        this.json = json;
    }

    @Override
    public final TaskTemplateCode code() {
        return code;
    }

    @Override
    public final PARAMETER adapt(String parameters) {
        return deserializeTaskResponse(parameters, parameterType);
    }

    <T> T deserializeTaskResponse(String data, Class<T> type) {
        try {
            return json.readValue(data, type);
        } catch (IOException e) {
            throw new IllegalStateException(e);
        }
    }
}
