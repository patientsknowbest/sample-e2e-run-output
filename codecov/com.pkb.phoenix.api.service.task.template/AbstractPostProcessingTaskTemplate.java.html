package com.pkb.phoenix.api.service.task.template;

import com.fasterxml.jackson.databind.ObjectReader;
import com.pkb.phoenix.repository.task.projection.Task;
import reactor.core.publisher.Mono;
import reactor.util.context.ContextView;

import java.util.List;

/**
 * Use this type when some transformation needs to be done after
 * a task was carried out (aka we received a response).
 * This implementation was chosen over a template method with
 * default NOOP implementation to allow returning a different result types
 * (which might add/remove some field or even create a completely different structure).
 *
 * @param <PARAMETER>           Task Parameter Type
 * @param <INTERMEDIARY_RESULT> Intermediary Result Type
 * @param <RESULT>              Task Result Type
 */
public abstract class AbstractPostProcessingTaskTemplate<PARAMETER, INTERMEDIARY_RESULT, RESULT> extends AbstractTaskTemplate<PARAMETER, RESULT> {
    private final TaskTemplate<PARAMETER, INTERMEDIARY_RESULT> delegate;

    public AbstractPostProcessingTaskTemplate(AbstractTaskTemplate<PARAMETER, INTERMEDIARY_RESULT> delegate, ObjectReader json) {
        super(delegate.code(), delegate.parameterType, json);
        this.delegate = delegate;
    }

    @Override
    public List<String> validate(PARAMETER parameters) {
        return delegate.validate(parameters);
    }

    @Override
    public Mono<RESULT> execute(Task<PARAMETER> task) {
        return delegate.execute(task)
                .transformDeferredContextual((resultMono, context) -> resultMono
                        .flatMap(result -> this.postProcess(result, context)));
    }

    abstract protected Mono<RESULT> postProcess(INTERMEDIARY_RESULT result, ContextView context);
}
