package com.pkb.service.security.twofactor;

import org.apache.http.client.utils.URIBuilder;
import org.springframework.stereotype.Component;

import static com.google.common.base.Preconditions.checkArgument;
import static org.apache.commons.lang3.StringUtils.isNotEmpty;

/**
 * Generates content for QR code used for two factor authentication (2FA).
 */
@Component
public class TotpUrlGenerator {

    private static final String TWO_FACTOR_ISSUER = "PatientsKnowBest";

    /**
     * Generates TOTP URL for two factor authentication (2FA).
     * The issuer will be 'PatientsKnowBest'.
     *
     * @param accountName
     *            The account the secret belongs to. Should be the email address of the user.
     * @param secret
     *            The shared secret used for 2FA.
     * @return The TOTP URL as {@link String}.
     */
    public String getOtpAuthTotpURL(String accountName, String secret) {
        checkArgument(isNotEmpty(accountName), "Account name cannot be empty or null!");
        checkArgument(isNotEmpty(secret), "Secret cannot be empty or null!");
        URIBuilder uri = new URIBuilder().setScheme("otpauth").setHost("totp")
                .setPath(formatLabel(TWO_FACTOR_ISSUER, accountName)).setParameter("secret", secret)
                .setParameter("issuer", TWO_FACTOR_ISSUER);
        return uri.toString();
    }

    private String formatLabel(String issuer, String accountName) {
        StringBuilder sb = new StringBuilder();
        sb.append(issuer);
        sb.append(":");
        sb.append(accountName);
        return sb.toString();
    }
}
