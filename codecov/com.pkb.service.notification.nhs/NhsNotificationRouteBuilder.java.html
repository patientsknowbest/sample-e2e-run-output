package com.pkb.service.notification.nhs;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.service.notification.nhs.pubsub.payload.NhsApiNotificationMessage;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Profile;
import org.springframework.stereotype.Component;

import static org.apache.commons.lang3.StringUtils.isBlank;

@Profile("!unittest")
@Component
public class NhsNotificationRouteBuilder extends AbstractNamespaceAwareRouteBuilder {

    public static final String QUEUE_NHS_NOTIFICATION = "direct:queueNhsNotification";
    public static final String PROCESS_NHS_NOTIFICATION = "direct:processNhsNotification";

    private final PubSubNamespaceService pubSubNamespaceService;
    private final String serviceAccountKey;
    private final PhrConfig config;

    public NhsNotificationRouteBuilder(PubSubNamespaceService pubSubNamespaceService,
                                       @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey,
                                       PhrConfig config) {
        this.pubSubNamespaceService = pubSubNamespaceService;
        this.serviceAccountKey = serviceAccountKey;
        this.config = config;
    }

    @Override
    public @NotNull PubSubNamespaceService camelNamespaceService() {
        return pubSubNamespaceService;
    }

    @Override
    public boolean throwExceptionForNamespaceMismatch() {
        return true;
    }

    @Override
    public void configure() throws Exception {

        if (config.isNhsAppNotificationEnabled()) {
            from(NhsNotificationRouteBuilder.QUEUE_NHS_NOTIFICATION)
                    .marshal().avro(NhsApiNotificationMessage.getClassSchema())
                    .convertBodyTo(byte[].class)
                    .to(uri(config.getPubSubProjectName(), config.getNhsAppNotificationPubsubTopic()));

            from(uri(config.getPubSubProjectName(), config.getNhsAppNotificationPubsubSubscription()))
                    .unmarshal().avro(NhsApiNotificationMessage.getClassSchema())
                    .to(NhsNotificationRouteBuilder.PROCESS_NHS_NOTIFICATION);
        }
    }

    private String uri(String pubSubProjectName, String topicName) {
        String serviceAccount = isBlank(serviceAccountKey) ? "" : String.format("?serviceAccountKey=file://%s", serviceAccountKey);
        return String.format("google-pubsub:%s:%s%s", pubSubProjectName, topicName, serviceAccount);
    }
}
