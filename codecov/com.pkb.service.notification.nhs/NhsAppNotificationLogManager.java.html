package com.pkb.service.notification.nhs;

import com.pkb.domain.nhs.NhsAppNotificationLogService;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.entities.enums.ValidNationalId;
import com.pkb.service.notification.nhs.pubsub.payload.NhsApiNotificationMessage;

import java.time.Instant;

import static java.lang.String.format;

public class NhsAppNotificationLogManager {

    private final NhsAppNotificationLogService notificationLogService;

    public NhsAppNotificationLogManager(NhsAppNotificationLogService notificationLogService) {
        this.notificationLogService = notificationLogService;
    }

    public void reportSuccess(NhsApiNotificationMessage message) {
        Instant created = Instant.parse(message.getCreatedDate()); // Sigh
        notificationLogService.reportSuccess(validateNhsNumber(message.getNhsNumber()), message.getTrigger().toString(), message.getMessage(), message.getRequestId(),created);
    }

    public void reportFailure(NhsApiNotificationMessage message, String errorResponse) {
        Instant created = Instant.parse(message.getCreatedDate());
        notificationLogService.reportFailure(validateNhsNumber(message.getNhsNumber()), message.getTrigger().toString(), message.getMessage(), message.getRequestId(),created, errorResponse);
    }

    private ValidNationalId validateNhsNumber(String nhsNumberString) {
        return NationalIdType.NHS_NUMBER.getValidNationalIdAndType(nhsNumberString)
                .orElseThrow(() -> new RuntimeException(format("Invalid NHS number '%s' reported to nhs_notification_log",nhsNumberString)));
    }

}
