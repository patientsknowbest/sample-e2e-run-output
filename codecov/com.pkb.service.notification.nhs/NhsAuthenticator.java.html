package com.pkb.service.notification.nhs;

import com.pkb.common.config.PhrConfig;
import com.pkb.integration.nhs.model.response.TokenResponse;
import okhttp3.Authenticator;
import okhttp3.Request;
import okhttp3.Response;
import okhttp3.Route;
import org.jetbrains.annotations.Nullable;

import java.io.IOException;
import java.util.Optional;
import java.util.function.Supplier;

public class NhsAuthenticator implements Authenticator {

    static final String HEADER_RETRY_COUNT = "X-InternalRetryCount";

    private final Supplier<TokenResponse> tokenSupplier;
    private final PhrConfig config;

    public NhsAuthenticator(Supplier<TokenResponse> tokenSupplier, PhrConfig config) {
        this.tokenSupplier = tokenSupplier;
        this.config = config;
    }

    @Nullable
    @Override
    public Request authenticate(Route route, Response response) throws IOException {
        if (isUnauthorised(response)) {

            int retry = getRetryCount(response);

            if (retry < config.getNhsApiTokenRefreshRetry()) {
                String token = tokenSupplier.get().getAccessToken();
                return response.request().newBuilder()
                        .header("Authorization", "Bearer " + token)
                        .header(HEADER_RETRY_COUNT, String.valueOf(retry + 1))
                        .build();
            }
        }
        return null;
    }

    private boolean isUnauthorised(Response response) {
        return response.code() == 401;
    }

    private int getRetryCount(Response response) {
        try {
            return Optional.ofNullable(response.request().header(HEADER_RETRY_COUNT))
                    .map(Integer::parseInt)
                    .orElse(0);
        } catch (NumberFormatException ignored) {
            return 0;
        }
    }
}
