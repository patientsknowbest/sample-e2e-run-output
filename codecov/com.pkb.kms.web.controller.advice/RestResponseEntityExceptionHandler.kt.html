package com.pkb.kms.web.controller.advice

import com.google.common.annotations.VisibleForTesting
import com.pkb.common.util.FrameFilter
import com.pkb.kms.shared.representation.KmsError
import com.pkb.kms.web.ErrorException
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.http.HttpHeaders
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.http.converter.HttpMessageNotReadableException
import org.springframework.orm.ObjectOptimisticLockingFailureException
import org.springframework.validation.FieldError
import org.springframework.web.bind.MethodArgumentNotValidException
import org.springframework.web.bind.annotation.ExceptionHandler
import org.springframework.web.bind.annotation.RestControllerAdvice
import org.springframework.web.context.request.WebRequest
import org.springframework.web.servlet.mvc.method.annotation.ResponseEntityExceptionHandler
import java.lang.invoke.MethodHandles
import java.lang.reflect.UndeclaredThrowableException

// http://www.baeldung.com/exception-handling-for-rest-with-spring#controlleradvice

@Suppress("unused")
@RestControllerAdvice
class RestResponseEntityExceptionHandler : ResponseEntityExceptionHandler() {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    /**
     * Specific error handler for KmsError exceptions
     */
    @ExceptionHandler(value = [ErrorException::class])
    @VisibleForTesting
    fun handle(ex: ErrorException, request: WebRequest): ResponseEntity<Any> {

        logException(ex)
        val status = when (ex.error.type) {
            KmsError.Type.NOT_FOUND -> HttpStatus.NOT_FOUND
            KmsError.Type.NOT_AUTHORISED -> HttpStatus.UNAUTHORIZED
            KmsError.Type.BAD_REQUEST -> HttpStatus.BAD_REQUEST
            KmsError.Type.CONFLICT -> HttpStatus.CONFLICT
            KmsError.Type.UNEXPECTED -> HttpStatus.INTERNAL_SERVER_ERROR
        }
        return handle(ex.error, status, request)
    }

    @ExceptionHandler(value = [ObjectOptimisticLockingFailureException::class])
    fun handle(ex: ObjectOptimisticLockingFailureException, request: WebRequest): ResponseEntity<Any> {
        logException(ex)
        val description = ex.message ?: "Unknown conflict"
        return handle(
                KmsError("conflict", description, KmsError.Type.CONFLICT),
                HttpStatus.CONFLICT,
                request)
    }

    /**
     * Catch unhandled exceptions, log them, and wrap them as KmsError with type UNEXPECTED
     */
    @ExceptionHandler(value = [Exception::class])
    fun handle(ex: Exception, request: WebRequest): ResponseEntity<Any> {

        logException(ex)
        val description = when (ex) {
            is UndeclaredThrowableException -> ex.undeclaredThrowable?.message ?: "Unknown UndeclaredThrowableException"
            else -> ex.message ?: "Unknown exception"
        }
        return handle(
                KmsError("unhandled.exception", description, KmsError.Type.UNEXPECTED),
                HttpStatus.INTERNAL_SERVER_ERROR,
                request)
    }

    override fun handleHttpMessageNotReadable(ex: HttpMessageNotReadableException,
                                              headers: HttpHeaders,
                                              status: HttpStatus,
                                              request: WebRequest): ResponseEntity<Any> {
        logException(ex)
        return handle(
                KmsError("not.readable", ex.message ?: "Unknown reason", KmsError.Type.BAD_REQUEST),
                status,
                request)
    }

    override fun handleMethodArgumentNotValid(ex: MethodArgumentNotValidException,
                                              headers: HttpHeaders,
                                              status: HttpStatus,
                                              request: WebRequest): ResponseEntity<Any> {
        logException(ex)
        val description = ex.bindingResult.allErrors.joinToString(",") {
            when (it) {
                is FieldError -> "${it.objectName}.${it.field}: ${it.defaultMessage}"
                else -> "${it.defaultMessage}"
            }
        }

        return handle(
                KmsError("argument.invalid", description, KmsError.Type.BAD_REQUEST),
                status,
                request)
    }

    private fun handle(err: KmsError, status: HttpStatus, request: WebRequest) =
            handleExceptionInternal(ErrorException(err), err, HttpHeaders(), status, request)

    private fun logException(ex: Exception) = when (ex) {
        is ErrorException -> logErrorException(ex)
        else -> log.error("Unexpected exception", FrameFilter.filter(ex))
    }

    private fun logErrorException(ex: ErrorException) = when (ex.error.type) {
        KmsError.Type.UNEXPECTED -> log.error("${ex.error.error} (${ex.error.type}) - ${ex.error.description}")
        KmsError.Type.NOT_FOUND -> log.debug("${ex.error.error} (${ex.error.type}) - ${ex.error.description}")
        else -> log.info("${ex.error.error} (${ex.error.type}) - ${ex.error.description}")
    }
}
