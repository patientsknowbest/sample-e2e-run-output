package com.pkb.emis.appointment.service;

import com.google.common.collect.ImmutableMap;
import com.pkb.integrations.emis.model.restcalls.EmisApiError;
import io.vavr.CheckedFunction0;
import io.vavr.control.Try;
import org.springframework.http.HttpStatus;
import org.springframework.web.client.HttpStatusCodeException;

import java.util.LinkedHashMap;
import java.util.Map;
import java.util.Optional;

final class EmisErrorExtractor implements ErrorCode.ErrorExtractor {

    private EmisApiClient emisApiClient;
    private final HttpStatusCodeException error;
    private boolean isErrorParsed = false;
    private EmisApiError parsed;
    private boolean isMapParsed = false;
    private Map<String, Object> map;

    EmisErrorExtractor(EmisApiClient emisApiClient, HttpStatusCodeException error) {
        this.emisApiClient = emisApiClient;
        this.error = error;
    }

    @Override
    public Optional<EmisApiError> getAsErrorObject() {
        CheckedFunction0<EmisApiError> caller = () -> emisApiClient.getObjectMapper().readValue(getAsString(), EmisApiError.class);
        if (!isErrorParsed) {
            parsed = Try.of(caller).getOrNull();
            isErrorParsed = true;
        }
        return Optional.ofNullable(parsed);
    }

    @Override
    public String getAsString() {
        return error.getResponseBodyAsString();
    }

    @Override
    public Map<String, Object> getAsMap() {
        CheckedFunction0<Map> caller = () -> emisApiClient.getObjectMapper().readValue(getAsString(), LinkedHashMap.class);
        if (!isMapParsed) {
            //noinspection unchecked
            map = Try.of(caller).getOrElse(ImmutableMap.of());
            isMapParsed = true;
        }
        return map;
    }

    @Override
    public HttpStatus getStatus() {
        return error.getStatusCode();
    }
}
