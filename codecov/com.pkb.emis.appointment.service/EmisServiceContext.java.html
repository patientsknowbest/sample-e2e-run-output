package com.pkb.emis.appointment.service;

public class EmisServiceContext {

    private final String emisRestUrl;
    private final String xapiApplicationId;
    private final String xapiVersion;

    private EmisServiceContext(Builder builder) {
        emisRestUrl = builder.emisRestUrl;
        xapiApplicationId = builder.xapiApplicationId;
        xapiVersion = builder.xapiVersion;
    }

    public static Builder newBuilder() {
        return new Builder();
    }

    String getEmisRestUrl() {
        return emisRestUrl;
    }

    String getXapiApplicationId() {
        return xapiApplicationId;
    }

    String getXapiVersion() {
        return xapiVersion;
    }

    public static final class Builder {
        private String emisRestUrl;
        private String xapiApplicationId;
        private String xapiVersion;

        private Builder() {
        }

        public Builder emisRestUrl(String val) {
            emisRestUrl = val;
            return this;
        }

        public Builder xapiApplicationId(String val) {
            xapiApplicationId = val;
            return this;
        }

        public Builder xapiVersion(String val) {
            xapiVersion = val;
            return this;
        }

        public EmisServiceContext build() {
            return new EmisServiceContext(this);
        }
    }
}
