package com.pkb.emis.appointment.service;

import com.google.common.cache.Cache;
import com.pkb.common.ClearableInternalState;
import com.pkb.datamodel.EmisRegistrationDetailsDto;
import com.pkb.repository.util.EmisContextHolder;
import com.pkb.domain.appointment.EmisAppointmentService;
import com.pkb.emis.appointment.model.EmisCachedSessionData;
import com.pkb.integrations.emis.model.restcalls.EmisSessionCreateRequest;
import com.pkb.integrations.emis.model.restcalls.ImmutableEmisSessionCreateRequest;
import io.vavr.control.Either;
import io.vavr.control.Try;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.invoke.MethodHandles;
import java.util.List;

import static com.pkb.emis.appointment.service.ErrorCode.NO_CACHED_ODS;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

public class EmisSessionService implements ClearableInternalState {

    private final Cache<Long, EmisCachedSessionData> sessionCache;

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private final EmisAppointmentService emisAppointmentService;
    private final EmisApiClient endpointService;

    public EmisSessionService(Cache<Long, EmisCachedSessionData> sessionDataCache, EmisAppointmentService emisAppointmentService, EmisApiClient endpointService) {
        this.sessionCache = sessionDataCache;
        this.emisAppointmentService = emisAppointmentService;
        this.endpointService = endpointService;
    }

    @NotNull
    Either<ErrorCode, SessionData> getSessionData(EmisContextHolder emisContextHolder) {
        EmisCachedSessionData emisCachedSessionData = sessionCache.getIfPresent(emisContextHolder.personId());
        if (emisCachedSessionData == null) {
            return Try.of(() -> load(emisContextHolder))
                    .map(EmisCachedSessionData::getSessionData)
                    .map(Either::<ErrorCode, SessionData> right)
                    .recover(EmisApiException.class, emisApiException -> left(emisApiException.getErrorCode()))
                    .get();
        }
        return convert(emisCachedSessionData);
    }

    private Either<ErrorCode, SessionData> convert(EmisCachedSessionData sessionDataTuple3) {
        return Try.of(() -> sessionDataTuple3)
                .map(EmisCachedSessionData::getSessionData)
                .map(Either::<ErrorCode, SessionData> right)
                .get();
    }

    public EmisCachedSessionData load(EmisContextHolder emisContextHolder) {
        List<EmisRegistrationDetailsDto> details = emisAppointmentService.findEmisRegistrationDetailsByPersonId(emisContextHolder);
        if (details.isEmpty()) {
            throw new EmisApiException(ErrorCode.MISSING_LINK_TO_GP);
        } else {
            //Currently we only support a single active record per patient. It's not 100% clear under
            //what circumstances, if any, a patient could have multiple registrations with EMIS. For
            //the time being taking the first should be fine
            // Alert is added to Papertrail; please keep that in sync with this error message: https://papertrailapp.com/searches/71866212/edit
            if (details.size() > 1) {
                LOGGER.error("EMIS-CACHE expecting to retrieve a single entry for personId={}, but retrieved {}", emisContextHolder.personId(), details.size());
            }
            EmisRegistrationDetailsDto registrationDetail = details.get(0);
            EmisSessionCreateRequest request = ImmutableEmisSessionCreateRequest.builder()
                    .nationalPracticeCode(registrationDetail.getOdsCode())
                    .accessIdentityGuid(registrationDetail.getAccessIdentityGuid())
                    .build();
            Either<ErrorCode, SessionData> sessionEither = endpointService.startEndUserAndServiceUserSessions(request);

            EmisCachedSessionData result = new EmisCachedSessionData(registrationDetail.getAccessIdentityGuid(), registrationDetail.getOdsCode(),
                    sessionEither.getOrElseThrow(EmisApiException::new));
            sessionCache.put(emisContextHolder.personId(), result);
            return result;
        }
    }

    void reloadSession(EmisContextHolder emisContextHolder) {
        sessionCache.invalidate(emisContextHolder.personId());
        load(emisContextHolder);
    }

    public void removeSession(long personId) {
        sessionCache.invalidate(personId);
    }

    Either<ErrorCode, String> getODSCodeFromCache(Long personId) {
        EmisCachedSessionData emisCachedSessionData = sessionCache.getIfPresent(personId);
        if (emisCachedSessionData == null) {
            return left(NO_CACHED_ODS);
        }
        return right(emisCachedSessionData.getNationalPracticeCode());
    }

    @Override
    public void clearState() {
        sessionCache.invalidateAll();
    }

}
