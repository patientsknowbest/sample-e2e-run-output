package com.pkb.emis.appointment.service;

import com.google.common.collect.ImmutableList;
import com.pkb.datamodel.EmisRegistrationDetailsDto;
import com.pkb.datamodel.EmisRegistrationDetailsNewDto;
import com.pkb.datamodel.ImmutableEmisRegistrationDetailsNewDto;
import com.pkb.repository.util.EmisContextHolder;
import com.pkb.domain.appointment.EmisAppointmentService;
import com.pkb.integrations.emis.model.restcalls.EmisAccountLinkRequest;
import com.pkb.integrations.emis.model.restcalls.EmisAccountLinkResponse;
import com.pkb.integrations.emis.model.restcalls.EmisInSessionRequest;
import com.pkb.integrations.emis.model.restcalls.EmisRequest;
import com.pkb.integrations.emis.model.restcalls.EmisResponse;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.http.HttpHeaders;

import java.lang.invoke.MethodHandles;
import java.util.List;
import java.util.function.Function;
import java.util.function.Supplier;

import static com.pkb.emis.appointment.service.ErrorCode.EMIS_SESSION_EXPIRED;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

public class SessionManagingEmisApiClient {
    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
    private final EmisApiClient emisApiClient;
    private final EmisSessionService sessionService;
    private final EmisAppointmentService emisAppointmentService;

    public SessionManagingEmisApiClient(EmisApiClient emisApiClient, EmisSessionService sessionService, EmisAppointmentService emisAppointmentService) {
        this.emisApiClient = emisApiClient;
        this.sessionService = sessionService;
        this.emisAppointmentService = emisAppointmentService;
    }

    Either<ErrorCode, EmisAccountLinkResponse> linkToGPPractice(EmisContextHolder emisContextHolder, EmisAccountLinkRequest request, ImmutableList<ErrorCode> codes) {
        return emisApiClient.startEndUserSession()
                .flatMap(endUserSessionData -> {
                    HttpHeaders headers = emisApiClient.getHeadersForEndUserSession(endUserSessionData.getEndUserSessionId());
                    return emisApiClient.callExchange(request, headers, codes);
                })
                .map(linkResponse -> {
                    sessionService.removeSession(emisContextHolder.personId());
                    EmisAccountLinkRequest.LinkageDetails linkageDetails = request.getLinkageDetails();
                    EmisRegistrationDetailsNewDto details = ImmutableEmisRegistrationDetailsNewDto.builder()
                            .personId(emisContextHolder.personId())
                            .odsCode(linkageDetails.getNationalPracticeCode())
                            .accessIdentityGuid(linkResponse.getAccessIdentityGuid())
                            .deleted(false)
                            .build();
                    emisAppointmentService.persistNewRegistrationDetails(emisContextHolder, details);
                    return linkResponse;
                });
    }

    Either<ErrorCode, SessionData> getSessionInfo(EmisContextHolder emisContextHolder) {
        return sessionService.getSessionData(emisContextHolder);
    }

    <T extends EmisResponse, U extends EmisInSessionRequest<T, U>> Either<ErrorCode, T> makeCallWithRetryInSession(EmisContextHolder emisContextHolder, U request, ImmutableList<ErrorCode> codes) {
        return makeCallWithRetryInSessionInternal(emisContextHolder, sessionData -> request.withUserPatientLinkToken(sessionData.getUserPatientLinkToken()), codes);
    }

    <T extends EmisResponse, U extends EmisRequest<T>> Either<ErrorCode, T> makeCallWithRetryInSession(EmisContextHolder emisContextHolder, U request, ImmutableList<ErrorCode> codes) {
        return makeCallWithRetryInSessionInternal(emisContextHolder, sessionData -> request, codes);
    }

    private <T extends EmisResponse, U extends EmisRequest<T>> Either<ErrorCode, T> makeCallWithRetryInSessionInternal(EmisContextHolder emisContextHolder, Function<SessionData, U> requestProvider,
                                                                                                                       ImmutableList<ErrorCode> codes) {
        Supplier<Either<ErrorCode, T>> caller = () -> sessionService.getSessionData(emisContextHolder)
                .flatMap(sessionData -> emisApiClient.callExchange(
                        requestProvider.apply(sessionData),
                        getHeadersWithSession(sessionData),
                        codes));

        return caller.get().fold(error -> handleSessionExpiry(emisContextHolder, caller, error), Either::right);
    }

    private <T> Either<ErrorCode, T> handleSessionExpiry(EmisContextHolder emisContextHolder, Supplier<Either<ErrorCode, T>> apiCaller, ErrorCode originalError) {
        if (originalError == EMIS_SESSION_EXPIRED) {
            LOGGER.info("EMIS-API-SESSION: Emis session expired for patient with id {}, trying to refresh session", emisContextHolder.personId());
            sessionService.reloadSession(emisContextHolder);
            return apiCaller.get();
        }
        return left(originalError);
    }

    @NotNull
    private HttpHeaders getHeadersWithSession(SessionData sessionData) {
        HttpHeaders headers = emisApiClient.getHeadersForEndUserSession(sessionData.getEndUserSessionId());
        headers.set("X-API-SessionId", sessionData.getSessionId());
        return headers;
    }

    Either<ErrorCode, String> getNationalPracticeCode(EmisContextHolder emisContextHolder) {
        return sessionService.getODSCodeFromCache(emisContextHolder.personId())
                .fold(error -> getNationalPracticeCodeFromDB(emisContextHolder), Either::right);
    }

    @NotNull
    private Either<ErrorCode, String> getNationalPracticeCodeFromDB(EmisContextHolder emisContextHolder) {
        List<EmisRegistrationDetailsDto> emisAppointmentRegistrationDetails = emisAppointmentService.findEmisRegistrationDetailsByPersonId(emisContextHolder);
        if (emisAppointmentRegistrationDetails.isEmpty()) {
            LOGGER.error("Failed to find EMIS registration details record for patient with id {}", emisContextHolder.personId());
            return left(ErrorCode.INTERNAL_ERROR);
        } else {
            return right(emisAppointmentRegistrationDetails.get(0).getOdsCode());
        }
    }
}
