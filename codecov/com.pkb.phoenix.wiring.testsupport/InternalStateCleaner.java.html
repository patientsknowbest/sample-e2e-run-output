package com.pkb.phoenix.wiring.testsupport;

import com.pkb.common.ClearableInternalState;
import org.springframework.data.redis.core.ReactiveRedisTemplate;

import static org.apache.commons.lang3.StringUtils.stripEnd;
import static org.springframework.data.redis.core.ScanOptions.scanOptions;

public class InternalStateCleaner implements ClearableInternalState {
    private final ReactiveRedisTemplate<String, String> redis;
    private final String pattern;
    private final int batchSize;

    public InternalStateCleaner(ReactiveRedisTemplate<String, String> redis, String prefix, int batchSize) {
        this.redis = redis;
        // https://redis.io/commands/scan#the-match-option
        // We want to find all values that have a certain prefix, we add * to turn prefix into a regex pattern.
        this.pattern = stripEnd(prefix, "*") + "*";
        this.batchSize = batchSize;
    }

    @Override
    public void clearState() {
        redis.scan(scanOptions().match(pattern).build())
                .buffer(batchSize)
                .flatMap(keys -> redis.delete(keys.toArray(String[]::new)))
                .blockLast();
    }
}
