package com.pkb.phoenix.wiring.testsupport;

import com.pkb.common.config.BaseConfig;
import com.pkb.common.config.ConfigStorage;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.common.testcontrol.config.ITestControlServiceConfig;
import com.pkb.common.testcontrol.config.TestControlServiceConfig;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.common.testlogging.DefaultDetailLoggingProvider;
import com.pkb.common.testlogging.DetailLoggingProvider;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Profile;
import org.springframework.data.redis.core.ReactiveRedisTemplate;

import java.util.Optional;
import java.util.Set;

@Configuration
@Profile("e2e | integration-test")
class TestSupportConfig {
    @Bean
    DetailLoggingProvider loggingProvider() {
        return DefaultDetailLoggingProvider.INSTANCE;
    }

    @Bean
    InternalStateCleaner internalStateCleaner(ReactiveRedisTemplate<String, String> redis,
                                              @Value("${phoenix.api.redis.key.prefix}") String prefix,
                                              @Value("${phoenix.api.testcontrol.redis.deletion.batch.size}") int batchSize) {
        return new InternalStateCleaner(redis, prefix, batchSize);
    }

    @Bean
    TestControlServiceConfig getTestSupportServiceConfig(BaseConfig baseConfig,
                                                         DateTimeService dateTimeService,
                                                         ConfigStorage configStorage,
                                                         InternalStateCleaner internalStateCleaner,
                                                         DetailLoggingProvider testLoggingService,
                                                         PubSubNamespaceService pubSubNamespaceService) {
        return new TestControlServiceConfig(baseConfig.isTestControlEnabled(),
                baseConfig.isTestControlRegistrationEnabled(),
                baseConfig.getTestControlAppName(),
                baseConfig.getTestControlCallbackUrl(),
                baseConfig.getTestControlUrl(),
                dateTimeService,
                configStorage,
                Set.of(internalStateCleaner),
                testLoggingService,
                pubSubNamespaceService,
                Optional.empty());
    }

    @Bean
    TestControlRouteBuilder camelPubSubRoutePhoenix(ITestControlServiceConfig config) {
        return new TestControlRouteBuilder(config);
    }
}
