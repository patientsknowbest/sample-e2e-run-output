package com.pkb.action.message;

import com.google.common.collect.ImmutableMap;
import com.google.common.collect.Maps;
import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.entities.enums.EncounterClass;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import java.time.Instant;
import java.time.format.DateTimeFormatter;
import java.util.EnumMap;
import java.util.Locale;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.ConcurrentMap;

import static com.pkb.action.BaseAction.NhsAppPage.MESSAGES;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class GetMessagesAction extends BaseAction {

    private static final long serialVersionUID = 1L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private InboxService inboxService;
    private EncounterClassLocalizationHelper encounterClassLocalizationHelper;

    private int offset;
    private int pageSize;
    private WorkflowFilter workflowFilter;
    private String filterPatientId;
    private String filterFromDate;
    private String filterToDate;

    private InboxDto inboxDto;
    private LoggedInEHRRequestContext loggedinContext;

    private static final ImmutableMap<EncounterClass, String> ENCOUNTER_TO_CSS_CLASS_MAP = Maps.immutableEnumMap(ImmutableMap.<EncounterClass, String> builder()
            .put(EncounterClass.IN_PATIENT, "a")
            .put(EncounterClass.OUT_PATIENT, "b")
            .put(EncounterClass.AMBULATORY, "c")
            .put(EncounterClass.EMERGENCY, "d")
            .put(EncounterClass.HOME, "e")
            .put(EncounterClass.FIELD, "f")
            .put(EncounterClass.DAYTIME, "g")
            .put(EncounterClass.VIRTUAL, "h")
            .build());

    private static final ConcurrentMap<Locale, Map<WorkflowFilter, String>> WORKFLOW_FILTER_MAPS = new ConcurrentHashMap<>();

    @Override
    public void prepare() throws Exception {
        loggedinContext = getLoggedInEHRRequestContext();
    }

    /**
     * Gets inbox messages for logged in user
     */
    @Override
    public String execute() {
        setNhsAppPage(MESSAGES.name());

        if (offset <= 0) {
            offset = 1;
        }

        try {
            this.pageSize = getConfig().getInboxPageSize();

            DateTimeFormatter dateFormat = getGlobalDateInputFormatter();
            Instant filterFrom = parseDate(dateFormat, filterFromDate);
            Instant filterTo = parseDate(dateFormat, filterToDate);

            if ((filterFrom != null) && (filterTo != null) && filterFrom.isAfter(filterTo)) {
                addActionError(getText("listDiscussions.txt.filter_from_cannot_be_later_than_to"));
                return Action.ERROR;
            }

            boolean isAbleToStartQuestionnaire = loggedInUserUtil.isLoggedInUserTeamProfessional() || loggedInUserUtil.isIdentityVerifiedPatientOrOnBehalfOfOne(
                    contextUserUtil.getContextUser());
            ImmutableGetInboxRequest.Builder getInboxRequestBuilder = ImmutableGetInboxRequest.builder()
                    .loggedInEHRRequestContext(loggedinContext)
                    .offset(offset)
                    .locale(getLocale())
                    .ableToStartQuestionnaire(isAbleToStartQuestionnaire);

            if (workflowFilter != null) {
                getInboxRequestBuilder.workflowFilter(workflowFilter);
            }
            if (isNotBlank(filterPatientId)) {
                getInboxRequestBuilder.filterPatientId(filterPatientId);
            }
            if (filterFrom != null) {
                getInboxRequestBuilder.filterFrom(filterFrom);
            }
            if (filterTo != null) {
                getInboxRequestBuilder.filterTo(filterTo);
            }

            inboxDto = inboxService.getInbox(getInboxRequestBuilder.build(), this::getText, getLocale());
            return getActionResult(inboxDto);
        } catch (Exception exception) {
            LOGGER.error("Error while getting messages ", exception);
            addActionError(getText("listDiscussions.txt.error_getting_messages"));
            return Action.ERROR;
        }
    }

    private Instant parseDate(DateTimeFormatter dateFormat, String date) {
        if (date != null) {
            try {
                return parseToInstantWithFormatter(date, dateFormat);
            } catch (Exception ignored) {
                // ignored
            }
        }

        return null;
    }

        return inboxDto;

        this.inboxDto = inboxDto;

        return workflowFilter;

        this.workflowFilter = workflowFilter;
    }

    public Map<EncounterClass, String> getEncounterToCssClassMap() {
        return ENCOUNTER_TO_CSS_CLASS_MAP;
    }

    public Map<WorkflowFilter, String> getWorkflowFilterMap() {
        return WORKFLOW_FILTER_MAPS.computeIfAbsent(getLocale(), this::computeWorkflowMap);
    }

    private Map<WorkflowFilter, String> computeWorkflowMap(Locale locale) {
        EnumMap<WorkflowFilter, String> result = new EnumMap<>(WorkflowFilter.class);
        result.put(WorkflowFilter.ACTIVE, getText("message.workflow.filter.option.one.text"));
        result.put(WorkflowFilter.ARCHIVED, getText("message.workflow.filter.option.two.text"));
        result.put(WorkflowFilter.ASSIGNED, getText("message.workflow.filter.option.three.text"));
        return Maps.immutableEnumMap(result);
    }

    public Map<EncounterClass, String> getEncounterToTextMap() {
        return encounterClassLocalizationHelper.getLocalizedMap(getLocale(), (key, locale) -> getText(key));
    }

    public LoggedInEHRRequestContext getLoggedinContext() {
        return loggedinContext;
    }

    public void setLoggedinContext(LoggedInEHRRequestContext loggedinContext) {
        this.loggedinContext = loggedinContext;
    }
}
