    @Override
    public String execute() {

        LoggedInEHRRequestContext loggedInEHRRequestContext = getLoggedInEHRRequestContext();

        String result;
        try {

            boolean isCaregiver = (contextUserUtil.getContextUserId() != null)
                    && loggedInEHRRequestContext.isPatient();
            long userId = loggedInUserUtil.getLoggedInUserId();

            // check if non-institute clinician
            nonInstituteClinician = !(teamUserManager.isTeamUser(userId))
                    && loggedInEHRRequestContext.isPro();
            if (validateInputs()) {

                // If the user is discussing a PHPlan embed a reference to it at the
                // end of the message, and make sure that every recipient has their own
                // copy

                if (TOPIC_PHPLAN.equals(discussTopic)) {
                    message = message + "[PLAN=" + selectedPlanUniqueId + "," + selectedPlanVersion + "]";

                    long patientId = contextUserUtil.getContextOrLoggedInUserId();
                    long patientAccountId = userManager.getDefaultAccountId(patientId);
                    PHPlan pHPlan = planManager.getPHPlanForAccountId(selectedPlanUniqueId, loggedInEHRRequestContext, patientAccountId, false/*includeDeleted*/);

                    // VITALLY IMPORTANT to clear the ids, otherwise planManager will try to overwrite the
                    // original objects with the new person's encryption key, causing no end of confusion

                    Long thisPlanId = pHPlan.getId();
                    pHPlan.setId(null);

                    if (recipientId != null) {
                        for (String recId : recipientId) {
                            long receiverId = Long.parseLong(recId);
                            Long recAccountId = userManager.getDefaultAccountId(receiverId);
                            if (!planManager.userHasPlan(getEHRRequestContext(), receiverId, pHPlan.getBaseFields().getUniqueId())) {
                                Long recPlanId = planManager.savePHPlan(getEHRRequestContext(), recAccountId, pHPlan);
                                planManager.copyAttachments(loggedInEHRRequestContext, userId, thisPlanId, receiverId, recPlanId);
                            }
                        }
                    }
                }

                attachments = generateAttachmentsFromInputs();
                // Add the patient id as well
                if ((recipientId == null) || recipientId.isEmpty()) {
                    recipientId = new ArrayList<>();
                }
                if (patientId != null) {
                    recipientId.add(patientId.toString());
                }
                if (colleagueId != null) {
                    recipientId.add(colleagueId.toString());
                }
                PKBPerson user = userManager.getPKBPerson(userId);
                recipientId.add(user.getId().toString());

                Long onBehalfAuthorId = null;
                if (isCaregiver) {
                    onBehalfAuthorId = loggedInUserUtil.getLoggedInUserId();
                }
                Message messageObj = buildMessage(onBehalfAuthorId, recipientId);

                messageObj.initializeUniqueIds(conversationId == null);

                encounterManager.saveMessage(loggedInEHRRequestContext, loggedInUserUtil.getLoggedInUserId(),
                        messageObj, attachments, contextUserUtil.getContextUserId(), true);

                tbmSelectedTeam
                        .forEach(teamId -> TBM_SELECTED_TEAMS.labels(teamId).inc());

                // success msg for the user
                addActionMessage(getText("composeMessageAction.txt.message_sent"));
                result = TOPIC_REDIRECTS.getOrDefault(discussTopic, Action.SUCCESS);
            } else {
                result = INPUT;
                // save the uploaded files because else call to input will wipe it out

                List<Attachment> uploadedFiles = generateAttachmentsFromInputs();
                input();
                attachments = uploadedFiles;

            }
        } catch (Exception ex) {
            result = ERROR;
            addActionError(getText("composeMessageAction.err.could_not_save"));
            LOGGER.error("Unable to send/save message ", ex);
        }

        LOGGER.info("Exiting execute");
        return result;
    }

    private boolean validateInputs() {
        boolean flag = true;

        if (isEmptyStrutsInput(recipientId)) {
            recipientId = null;
        }

        if (((recipientId == null) || recipientId.isEmpty()) && (patientId == null) && (colleagueId == null)) {
            flag = false;
            addFieldError("recipientId", getText("composeMessageAction.err.select_recipient"));
        }

        if ((this.subject == null) || (this.subject.trim().length() < 1)) {
            flag = false;
            addFieldError("subject",
                    getText("composeMessageAction.err.invalid_subject_length"));
        }

        if ((this.message == null) || (this.message.trim().length() < 1)) {
            flag = false;
            addFieldError("message",
                    getText("composeMessageAction.err.invalid_message_length"));
        }

        if ((this.message == null)
                || (this.message.trim().length() > PKBConstants.MSG_CONTENT_CHAR_LIMIT)) {
            flag = false;
            addFieldError("message",
                    getText("composeMessageAction.err.message_too_long"));
        }

        if (nonInstituteClinician) {
            flag = validateChosenClinicians();
        }

        return flag;
    }

    public String autoSaveMessage() {

        LOGGER.info("autoSaveMessage:start");
        if (json == null) {
            LOGGER.info("autoSaveMessage: json data null. nothing to save. exiting");
            autoSaveResult = new ByteArrayInputStream(AUTO_SAVE_RESET.getBytes(UTF_8));
            return Action.SUCCESS;
        }

        String composerId = Long.toString(loggedInUserUtil.getLoggedInUserId());

        try {
            MessageAutoSaveModel model = MESSAGE_PATCH_READER.readValue(json);

            return autoSaveService.prepareMessageForAutoSave(getLoggedInEHRRequestContext(), model)
                    .map(message -> {
                        message.setSenderId(composerId);

                        // force draft status, and conversationId (if there isn't one already)
                        message.setMessageStatus(MessageStatus.INDRAFT);

                        // the JavaScript puts patientId and clinicianIds into the "participants" list in the message
                        List<String> participantIds = message.getParticipants();
                        participantIds.add(composerId);

                        if (!message.getSource().isPopulated()) {
                            message.getSource().init(getEHRRequestContext());
                        }

                        message.getBaseFields().setDefaultGeneralConsentIfNoneSet();

                        autoSaveResult = encounterManager.transactional(() -> {
                            Long msgId = encounterManager.saveMessageNewConversation(getLoggedInEHRRequestContext(), message, null /*attachments*/);
                            byte[] resultArray = ("" + message.getEncounterUniqueId() + "," + msgId).getBytes(UTF_8);
                            return new ByteArrayInputStream(resultArray);
                        });
                        return Action.SUCCESS;
                    })
                    .mapLeft(error -> {
                        autoSaveResult = new ByteArrayInputStream("error".getBytes());
                        return Action.ERROR;
                    })
                    .fold(identity(), identity());

        } catch (Exception e) {
            LOGGER.error("problem with JSON autosave for {}", composerId, e);
            autoSaveResult = new ByteArrayInputStream(Action.ERROR.getBytes(UTF_8));
        }
        LOGGER.info("autoSaveMessage:end");
        return Action.SUCCESS;
    }

    private Message buildMessage(Long onBehalfAuthorId, List<String> participants) {
        Long referredPatient = null;

        if (patientId != null) {
            // clinician is sending message to patient
            referredPatient = patientId;
        }
        boolean isCaregiver = (contextUserUtil.getContextUserId() != null)
                && getLoggedInEHRRequestContext().isPatient();
        Long composerId = loggedInUserUtil.getLoggedInUserId();
        // patient is logged in and sending the message
        if (getLoggedInEHRRequestContext().isPatient()) {
            referredPatient = composerId;
            if (referredPatientId != null) {
                referredPatient = referredPatientId;
            }
        }
        if (isCaregiver) {
            // patient referred in the conversation is the context user
            referredPatient = contextUserUtil.getContextUserId();
        }

        if (messageId == null) {
            conversationId = UUID.randomUUID();
        }

        Message messageObject = encounterManager.createMessage(Objects.toString(referredPatient, null),
                conversationId,
                messageId == null,
                "" + composerId,
                subject,
                message,
                PrivacyFlag.GENERAL,
                getEHRRequestContext());
        messageObject.setId(messageId);
        messageObject.setPrivateMessage(privateMessage);

        messageObject.setOnBehalfAuthorId(Objects.toString(onBehalfAuthorId, null));

        List<String> serializableList = new LinkedList<>();
        // Struts own list doesn't unserialize!
        serializableList.addAll(participants);
        messageObject.setParticipants(serializableList);

        // Save consent on the message. This will be copied into the Encounter
        // object on subsequent loads by EHREncounterBean.populateConversation
        fillPrivacyFlagIn(messageObject);

        return messageObject;
    }

    private boolean validateChosenClinicians() {
        List<PKBPerson> careTeam = new ArrayList<>();
        boolean valid = true;
        // if no clinicians chosen ; no validation required
        if ((recipientId == null) || recipientId.isEmpty()) {
            return valid;
        }
        PKBPerson patient = userManager.getPKBPerson(patientId);
        List<PKBPerson> clinicians = teamUserManager.getCliniciansWithIndivAccess(patientId,
                ContactOptions.Messages_Only);
        careTeam.addAll(clinicians);
        List<PKBPerson> carers = patientConsentManager.getCaregivers(patientId);
        careTeam.addAll(carers);

        //Pravina : Why is this check required for a NI Clinician ?
        for (String chosenId : recipientId) {
            valid = false;
            for (PKBPerson clinician : careTeam) {
                // Check if the recipient belongs to patient
                if (chosenId.equals(clinician.getIdString())) {
                    valid = true;
                    break;
                }
            }
            if (!valid) {
                // Chosen clinician is not assigned to patient
                PKBPerson clinician = userManager.getPKBPerson(Long.parseLong(chosenId));
                // Throw error message
                List<Object> args = new ArrayList<>();
                args.add(0, patient.getTitle() + " " + patient.getFirstName() + " " + patient.getLastName());
                String errClinicianName = clinician.getTitle() + " " + clinician.getFirstName() + " " + clinician.getLastName();
                args.add(1, errClinicianName);
                args.add(2, errClinicianName);
                addFieldError("recipientId",
                        getText("composeMessage.invalid.clinician.for.patient", args));
                break;
            }
        }
        return valid;
    }

    @Override
    public String input() {
        // if another feature is passing attachments to this form, display them
        if (isNotBlank(getDiscussTopic())) {
            message = discussMessage;
            attachments = discussAttachments;
        }

        // if selectedId was passed in, merge that into our recipientId and patientId here
        // the struts tags in the JSP will auto-check the boxes based on what's in recipientId and (possibly) patientId
        if (getSelectedId() != null) {
            if (recipientId == null) {
                recipientId = new ArrayList<>();
            }
            recipientId.add(getSelectedId().toString());
            patientId = getSelectedId();
        }

        // Add the colleague id in the recipient list
        if (getColleagueId() != null) {
            if (recipientId == null) {
                recipientId = new ArrayList<>();
            }
            recipientId.add(getColleagueId().toString());
        }

        // if patient is logged in and sending the message
        if (getLoggedInEHRRequestContext().isPatient()) {
            referredPatientId = loggedInUserUtil.getLoggedInUserId();
            referredPatient = loggedInUserUtil.getLoggedInUser(PKBPerson.Lazy.PROPERTIES);
        } else {
            teamNationalIdType = teamUserManager.findPrimaryTeam(getEHRRequestContext(), loggedInUserUtil.getLoggedInUserId()).map(Team::getCountry)
                    .flatMap(NationalIdType::getPrimaryNationalIdTypeFromCountryCode).orElse(null);

        }
        // Get the recipient id from the session
        // TODO: For messaging refacor: We're currently using multiple variables (contextRecipientId, carerId, patientId) to get the recipient.
        //  This should always use contextUserId in the future, and we should use getContextAndLoggedInUsers()
        Long contextRecipientId = sessionUtil.getContextRecipientId();
        if (contextRecipientId != null) {
            patientId = contextRecipientId;
            //referred patient is the context user
            referredPatientId = patientId;
            // Remove the recipient id from the session
            sessionUtil.clearContextRecipientId();
        }

        if (patientId != null) {
            patient = userManager.getPKBPerson(patientId, PKBPerson.Lazy.PROPERTIES);
            referredPatientId = patientId;
            referredPatient = patient;
        }

        if (careeId != null) {
            caree = userManager.getPKBPerson(careeId, PKBPerson.Lazy.PROPERTIES);
            referredPatientId = careeId;
            referredPatient = caree;
            patientId = careeId;
            patient = caree;

        }

        if (colleagueId != null) {
            colleague = userManager.getPKBPerson(colleagueId, PKBPerson.Lazy.PROPERTIES);
        }
        setDefaultFlag();
        // setup lists and so on:
        return reloadForm();
    }

    @Override
    public String reloadForm() {
        // New flattened processing logic!!
        //
        // If we have a patient -- context user, or we're sending a message about a patient, or current user is patient: let's show:
        //   patient's clinicians grouped by team
        //   then non-team clinicians
        //   then carers
        // Special case: current user is the patient: show dependents
        //
        // Temporary tweak: current team isn't migrated? Team clinician? Also show all clinicians on the current team (as colleagues)
        //
        // special case: if consultTeamId is set, populate that list & display it
        //
        // If we DON'T have a patient (clinician to clinician message), let's show:
        //    just colleagues (others on same team)

        LoggedInEHRRequestContext loggedInEHRRequestContext = getLoggedInEHRRequestContext();

        columnSize = getConfig().getRecipientColumnSize();
        //attachments may be already populated if passed in for a discussion
        if (attachments == null) {
            attachments = generateAttachmentsFromInputs();
        }

        PKBPerson currentUser = loggedInUserUtil.getLoggedInUser(PKBPerson.Lazy.PROPERTIES);
        LoggedInEHRRequestContext ehrRequestContextSingleAccess = null;
        Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam(getEHRRequestContext());
        if (!currentUser.isPatient() && referredPatient != null) {
            ehrRequestContextSingleAccess = getEHRRequestContextSingleAccess(referredPatient.getId(), team);
        }

        try {
            ZoneId zoneId = loggedInUserUtil.getLoggedInUserZoneId();
            zoneId = zoneId == null ? Constants.APPLICATION_TZ : zoneId;
            participantSelectorParams = participantSelectorService.setupParticipantSelectorParams(TOPIC_CONSULT_TEAM.equals(discussTopic), consultTeamId,
                    currentUser, referredPatient, isLoggedInUserProfessional(), ContactOptions.Messages_Only, team,
                    loggedInEHRRequestContext, ehrRequestContextSingleAccess,
                    DateTimeFormatter.ofPattern("dd MMM yy").withLocale(getLocale()).withZone(zoneId));
            consultTeam = participantSelectorParams.getConsultTeam();
            carerList = participantSelectorParams.getCarerList();
        } catch (Exception e) {
            LOGGER.error("unable to load accepted cliniciants for {}", referredPatientId, e);
            addActionError(getText("error.load.accepted.clinicians"));
        }

        return INPUT;
    }
