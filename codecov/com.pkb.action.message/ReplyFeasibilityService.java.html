package com.pkb.action.message;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.service.encounter.impl.tolven.EncounterManager;
import com.pkb.service.questionnaire.QuestionnaireManager;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.user.impl.UserManager;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.Set;
import java.util.UUID;

import static io.vavr.control.Either.left;
import static java.lang.String.format;
import static java.util.stream.Collectors.toSet;

public class ReplyFeasibilityService {

    private final EncounterManager encounterManager;
    private final QuestionnaireManager questionnaireManager;
    private final TeamUserManager teamUserManager;
    private final UserManager userManager;

    public ReplyFeasibilityService(EncounterManager encounterManager,
                                   QuestionnaireManager questionnaireManager,
                                   TeamUserManager teamUserManager,
                                   UserManager userManager) {
        this.encounterManager = encounterManager;
        this.questionnaireManager = questionnaireManager;
        this.teamUserManager = teamUserManager;
        this.userManager = userManager;
    }

    /**
     * Checks if a message/conversation based on a logged-in request context.
     * We are considering exceptions for use cases that are not logical: if the context/accessing user is not provided of an account
     * and/or the conversation does not exist.
     * Any other use case will be checked by {@link #isFeasibleToBeReplied(UUID, Set) the feasaibility to be replied method}
     * @param context - logged-in request context
     * @param conversationId - UUID of the conversation
     * @return either a RuntimeException for ilogical cases or if the message is available for reply
     */
    Either<RuntimeException, Boolean> isFeasibleToBeReplied(@NotNull LoggedInEHRRequestContext context, UUID conversationId) {
        var userId = context.getContextOrAccessingUserId();
        var accountId = userManager.getDefaultAccountId(userId);
        if (accountId == null) {
            return left(new RuntimeException(format("User %s doesn't have account id", userId)));
        }
        return encounterManager.getOptionalConversation(context, conversationId, userId, accountId)
                .map(conversation -> {
                    Set<Long> participantIds = conversation.getParticipants().stream().map(Long::valueOf).collect(toSet());
                    return Either.<RuntimeException, Boolean>right(isFeasibleToBeReplied(conversationId, participantIds));
                })
                .orElse(left(new RuntimeException(format("Conversation %s does not exist", conversationId))));
    }

    /**
     * Checks if a message/conversation can be replied based on a set of participants.
     * We want to stop users from being able to reply to any conversation that is a questionnaire _and_ at least one of the participants
     * is not contactable by messages
     *
     * @param conversationId - UUID of the conversation
     * @param participantIds - set of participant IDs to be verified for contact availability
     * @return if the message is available for reply
     */
    boolean isFeasibleToBeReplied(UUID conversationId, Set<Long> participantIds) {
        if (participantIds.isEmpty()) {
            return true;
        }

        if (!questionnaireManager.isQuestionnaireRequest(conversationId)) {
            return true;
        }

        return !teamUserManager.existsUserNotContactableByMessages(participantIds);
    }
}
