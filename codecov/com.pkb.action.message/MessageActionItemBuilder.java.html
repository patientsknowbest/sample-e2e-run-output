package com.pkb.action.message;

import com.google.common.collect.ImmutableMap;
import com.pkb.dto.ImmutableUrlDto;
import com.pkb.dto.PKBMessageDTO;
import com.pkb.dto.UrlDto;
import com.pkb.encounter.entity.Message;
import com.pkb.mappers.EntityDependency;
import com.pkb.service.datapoint.DataWithUIPermissions;
import com.pkb.util.UrlUtil;
import org.springframework.context.MessageSource;

import java.util.Optional;

import static com.pkb.util.WebNamespace.AUTH;
import static java.util.Objects.requireNonNull;

/**
 * Builds the actions the user can take on a single message.
 */
public class MessageActionItemBuilder {

    private final MessageSource messageSource;

    public MessageActionItemBuilder(MessageSource messageSource) {
        this.messageSource = messageSource;
    }

    Optional<UrlDto> entityToDto(DataWithUIPermissions<Message> dto, EntityDependency<PKBMessageDTO> dependency) {
        if (dto.isPrivacyFlagEditable()) {
            Message message = dto.getData();
            return Optional.of(ImmutableUrlDto.builder()
                    .href(UrlUtil.buildHref(AUTH, "editEncounterPrivacyForm", ImmutableMap.of(
                                    "encounterEventUniqueId", message.getBaseFields().getUniqueId().toString(),
                                    "conversationId", requireNonNull(message.getEncounterUniqueId()).toString()),
                            dependency.loggedInRequestContext().getContextOrAccessingUserId()))
                    .content(messageSource.getMessage("consentcategory.txt.edit", null, dependency.getLocale()))
                    .build());
        }
        return Optional.empty();
    }
}
