package com.pkb.action.message;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.pkb.action.message.workflow.ConversationWorkflowFlagsSupport;
import com.pkb.datamodel.DataModelStyle;
import com.pkb.datamodel.ui.UIDto;
import com.pkb.dto.DiscussionParticipantsDTO;
import com.pkb.dto.FormActionItemDto;
import com.pkb.dto.PKBMessageDTO;
import com.pkb.encounter.entity.Encounter;
import com.pkb.encounter.workflow.MessageWorkflowSupport;
import com.pkb.service.datapoint.DataWithEncounterRules;
import com.pkb.service.datapoint.DataWithUIPermissions;
import org.apache.commons.collections.CollectionUtils;
import org.immutables.value.Value;
import org.jetbrains.annotations.Nullable;

import java.util.List;

@Value.Immutable
@DataModelStyle
@JsonSerialize(as = ImmutableMessageScreenDto.class)
@JsonDeserialize(as = ImmutableMessageScreenDto.class)
public abstract class MessageScreenDto extends UIDto
        implements MessageWorkflowSupport, ConversationWorkflowFlagsSupport {

    public abstract List<FormActionItemDto> actionItems();

    @Nullable
    public abstract DataWithEncounterRules<DataWithUIPermissions<Encounter>> conversation();

    public abstract boolean isReplyDisabled();

    @Nullable
    public abstract String draftMessageId(); //Used to be called messageId

    @Nullable
    public abstract String subject();

    @Nullable
    public abstract String message();

    @Nullable
    public abstract String recipientId();

    @Nullable
    public abstract String initialSubject();

    public abstract boolean isCurrentParticipant();

    @Nullable
    public abstract DiscussionParticipantsDTO participants();

    public abstract boolean isHidePrivacyLabels();

    public abstract List<PKBMessageDTO> messageDTOS();

    public abstract boolean privateMessage();

    public boolean hasMessages() {
        return CollectionUtils.isNotEmpty(messageDTOS());
    }
}
