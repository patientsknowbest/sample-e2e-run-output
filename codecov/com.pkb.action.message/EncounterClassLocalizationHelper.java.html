package com.pkb.action.message;

import com.google.common.collect.ImmutableMap;
import com.google.common.collect.Maps;
import com.pkb.action.AbstractLocalizationHelper;
import com.pkb.entities.enums.EncounterClass;

import java.util.Arrays;
import java.util.EnumMap;
import java.util.Locale;
import java.util.function.BiFunction;

public class EncounterClassLocalizationHelper extends AbstractLocalizationHelper<EncounterClass> {

    @Override
    protected ImmutableMap<EncounterClass, String> computeMap(Locale locale, BiFunction<String, Locale, String> localizedStringSupplier) {
        EnumMap<EncounterClass, String> result = new EnumMap<>(EncounterClass.class);
        Arrays.stream(EncounterClass.values()).forEach(ec -> result.put(ec, localizedStringSupplier.apply("listDiscussions.txt." + ec, locale)));
        return Maps.immutableEnumMap(result);
    }
}
