    @Override
    public String execute() {
        LOGGER.info("Entering execute");
        String result = Action.SUCCESS;
        loggedinContext = getLoggedInEHRRequestContext();

        if (!attemptReply(conversationId)) {
            LOGGER.warn("Attempt to reply in a conversation that is not feasible to be replied - {}", conversationId);
            return ERROR;
        }

        try {
            if (handover) {
                if (privateMessage) {
                    throw new RuntimeException(
                            "Attempting to handover a discussion despite privateMessage flag. The message has not been sent.");
                }
                return "handover";
            } else if (addParticipant) {
                if (privateMessage && !loggedinContext.isUserInOwnAccount()) {
                    throw new RuntimeException(
                            "Attempting to bring in a new user despite privateMessage flag and not being the patient. The message has not been sent.");
                }
                return "addParticipant";
            } else {
                if (validateInputs(loggedinContext)) {

                    if (StringUtils.isNotBlank(inlineContentId) &&
                            (MSG_INLINE_MED.equals(contentType)
                                    || MSG_INLINE_MED_APPROVED.equals(contentType)
                                    || MSG_INLINE_MED_FOR_APPROVAL.equals(contentType))) {
                        message = message + "\n[MED=" + inlineContentId + "]";
                    }

                    if (MSG_INLINE_CALL.equals(contentType)) {
                        message = message + "\n[CALL=#]";
                    }

                    attachments = generateAttachmentsFromInputs();

                    Long msgId = null;
                    if ((messageId != null) && !messageId.isEmpty()) {
                        msgId = Long.parseLong(messageId);
                    }
                    Message messageObject = buildMessage(loggedinContext, msgId, MessageStatus.UNREAD);

                    sendReplyMessage(messageObject, attachments, conversationId);
                    // success msg for the user
                    addActionMessage(getText("replyMessageAction.txt.message_sent"));
                    //clear the message contents if any
                    message = "";
                } else {
                    result = Action.INPUT;
                }
            }
        } catch (Exception exception) {
            addActionError(getText("replyMessageAction.err.could_not_save"));
            LOGGER.error("Error while sending reply ", exception);
            result = ERROR;
        }
        LOGGER.info("Exiting execute result : {}", result);
        return result;
    }

    public String reload() {
        attachments = generateAttachmentsFromInputs();
        return Action.SUCCESS;
    }

    private boolean validateInputs(@NotNull LoggedInEHRRequestContext requestContext) {
        boolean flag = true;

        String errorMessage = "";
        if ((this.message == null) || (this.message.trim().length() < 1)
                || TYPE_MESSAGE_HERE.equals(this.message.trim())) {
            this.messageRequired = true;
            errorMessage = getText("error.empty.message");
            flag = false;
        }

        if ((this.message == null)
                || (this.message.trim().length() > PKBConstants.MSG_CONTENT_CHAR_LIMIT)) {
            this.invalidMessageLength = true;
            flag = false;
            if ((this.message == null) || this.message.isEmpty()) {
                errorMessage = getText("error.empty.message");
            } else {
                errorMessage = getText("message.too.long");
            }
        }

        // NOTE : Error message had to be handled this way because this action redirects to
        // showConversation which will lose the action & fields errors chain action will not work either
        // and neither does message interceptor to store the and retrieve the errors message
        // because it causes session to be null in showConversation action which fails the flow
        if (!errorMessage.isEmpty()) {
            session.put("errorMessage", errorMessage);
        }

        long composerId = loggedInUserUtil.getLoggedInUserId();
        try {
            long accountId;
            if (contextUserUtil.getContextUser() != null) {
                accountId = messageWebBean.getDefaultAccountId(contextUserUtil.getContextUserId());
            } else {
                accountId = messageWebBean.getDefaultAccountId(composerId);
            }
            var conversation = encounterManager.getOptionalConversation(requestContext.withoutAccessLog(),
                    conversationId,
                    composerId,
                    accountId)
                    .orElseThrow(() -> new IllegalStateException(format("Unable to find conversation uuid=%s", conversationId)));
            // If the latest message in the conversation is in INDRAFT
            // check for participants
            Message latestMessage = conversation.getLatestMessage();
            if (latestMessage.getMessageStatus() == MessageStatus.INDRAFT) {
                List<String> fullParticipants = conversation.getParticipants();
                if ((fullParticipants == null) || fullParticipants.isEmpty()
                        || (fullParticipants.size() == 1 /*sender himself*/)) {
                    flag = false;
                    addActionError(getText("replyMessageAction.err.select_recipient"));
                }
            }
        } catch (RuntimeException e) {
            flag = false;
            LOGGER.error("Error while validating no of participants while replying to message", e);
        }

        return flag;
    }

    public String autoSaveMessageReply() {
        LOGGER.info("autoSaveMessageReply:start");
        try {
            loggedinContext = getLoggedInEHRRequestContext();
            MessageAutoSaveModel model = MESSAGE_PATCH_READER.readValue(json);

            if (!attemptReply(model.getEncounterUniqueId())) {
                LOGGER.warn("Attempt to auto-save reply in a conversation that is not feasible to be replied - {}", model.getEncounterUniqueId());
                return ERROR;
            }

            return autoSaveService.prepareMessageForAutoSave(loggedinContext, model)
                    .map(jsonMessage -> {
                        LOGGER.info("messageId={}; conversationId={}", jsonMessage.getId(), jsonMessage.getEncounterUniqueId());
                        jsonMessage.getSource().init(loggedinContext);
                        Long messageId = sendReplyMessage(jsonMessage, null/*no attachments*/, jsonMessage.getEncounterUniqueId());
                        // conversation id will not be null but messageId cannot always have a value in reply screen
                        byte[] resultArray = ("" + jsonMessage.getEncounterUniqueId() + "," + messageId).getBytes(UTF_8);
                        autoSaveResult = new ByteArrayInputStream(resultArray);
                        return Action.SUCCESS;
                    })
                    .mapLeft(error -> {
                        LOGGER.error("problem with JSON autosave for {} , error: {}", loggedInUserUtil.getLoggedInUserId(), error);
                        autoSaveResult = new ByteArrayInputStream(ERROR.getBytes(UTF_8));
                        return ERROR;
                    })
                    .fold(identity(), identity());
        } catch (Exception e) {
            LOGGER.error("problem with JSON autosave for {}", loggedInUserUtil.getLoggedInUserId(), e);
            autoSaveResult = new ByteArrayInputStream(ERROR.getBytes(UTF_8));
            return ERROR;
        } finally {
            LOGGER.info("autoSaveMessageReply:end");
        }
    }

    private Message buildMessage(EHRRequestContext requestContext, Long messageId, MessageStatus messageStatus) {
        Message messageObj = encounterManager.createMessage(patientId,
                conversationId,
                false,
                null,
                StringEscapeUtils.unescapeHtml4(subject),
                message,
                PrivacyFlag.GENERAL,
                requestContext);
        messageObj.setPrivateMessage(privateMessage);
        messageObj.setId(messageId);
        messageObj.setMessageStatus(messageStatus);

        return messageObj;
    }

    private boolean attemptReply(UUID conversationId) {
        return replyFeasibilityService.isFeasibleToBeReplied(loggedinContext.withoutAccessLog(), conversationId)
                .fold(e -> false, identity());
    }
