package com.pkb.action.message;

import java.util.stream.Collectors;

public class HandoverConversationFormAction extends BaseAction {

    /**
     *
     */
    private static final long serialVersionUID = 1L;

    // Received from discussionCenter.jsp
    private UUID conversationId;

    // Returned to handoverDiscussion.jsp
    private List<PKBPerson> fellowClinicians;
    private boolean isOneClinician;
    private int columnSize;

    // Received from handoverDiscussion.jsp
    private Long patientId;
    private Long clinicianId;
    private String subject;
    private String messageToClinician;

    private IMessageWebBean messageWebBean;
    private EncounterManager encounterManager;
    private TeamUserManager teamUserManager;
    private PatientConsentManager patientConsentManager;

    private IUserWebBean userWebBean;

    @Override
    public String execute() {

        LoggedInEHRRequestContext loggedInEHRRequestContext = getLoggedInEHRRequestContext();
        Long userId = loggedInUserUtil.getLoggedInUserId();
        Long accountId = loggedInUserUtil.getLoggedInUserAccountId();

        // Read the top message of the conversation to find out who the patient is

        var conversation = encounterManager.getOptionalConversation(loggedInEHRRequestContext, conversationId, userId, accountId);
        var topMessage = conversation.map(Encounter::getLatestMessage);
        messageWebBean.expandInlineContents(loggedInEHRRequestContext, userId, accountId, topMessage.orElse(null), false);


        patientId = null;
        if (topMessage.isPresent()) {
            for (String participant : topMessage.get().getParticipants()) {
                PKBPerson person = userManager.getPKBPerson(Long.parseLong(participant));
                if (person.isPatient()) {
                    patientId = Long.parseLong(participant);
                    break;
                }
            }
        }
        if (patientId == null) {
            addActionError(getText("handoverDiscussion.err.no_patients"));
            return Action.INPUT;
        }

        subject = topMessage.map(Message::getSubject).orElse("");

        fellowClinicians = new LinkedList<>();
        Set<Long> fellowClinicianIds = new TreeSet<>();

        patientConsentManager.getConsentLinks(getEHRRequestContext(), patientId, false, EnumSet.of(AccessingEntityType.INDIVIDUAL))
                .getIndividuals().stream().filter(clinician -> !clinician.getId().equals(userId))
                .forEach(clinician -> {
                    fellowClinicians.add(clinician);
                    fellowClinicianIds.add(clinician.getId());
                });

        fellowClinicians
                .addAll(teamUserManager.getColleagues(getEHRRequestContext(), userId, ContactOptions.Messages_Only)
                        .stream()
                        .filter(clinician -> !fellowClinicianIds.contains(clinician.getId()))
                        .collect(Collectors.toList()));

        if (fellowClinicians.isEmpty()) {
            addActionError(getText("handoverDiscussion.err.no_clinicians"));
            return Action.INPUT;
        }
        isOneClinician = fellowClinicians.size() == 1;
        columnSize = getConfig().getRecipientColumnSize();

        clinicianId = fellowClinicians.get(0).getId();

        return Action.SUCCESS;
    }

    public String doHandover() {
        long userId = loggedInUserUtil.getLoggedInUserId();
        Long accountId = messageWebBean.getDefaultAccountId(userId);
        Long newAccountId = messageWebBean.getDefaultAccountId(clinicianId);

        encounterManager.handoverConversation(getLoggedInEHRRequestContext(), conversationId, messageToClinician, userId, accountId,
                clinicianId, newAccountId, patientId);

        addActionMessage(getText("handoverConversationAction.msg.handover_success"));
        return Action.SUCCESS;
    }
