package com.pkb.action.message;

import com.fasterxml.jackson.annotation.JsonProperty;
import com.pkb.encounter.entity.Message;

import java.util.List;
import java.util.UUID;

import static java.util.stream.Collectors.toList;

/**
 * This class represents a patch / partial update to a message. Used by the message autosave feature,
 * it models the message properties sent in the autosave ajax call's request json.
 */
public class MessageAutoSaveModel {

    @JsonProperty("id")
    private Long messageId;

    private UUID encounterUniqueId;

    private String subject;

    private String content;

    private List<Long> participants;

    private Long patientId;

    private String receiverId;

    private Boolean privateMessage;

    /**
     * Patches the {@code original} message with {@code this} (populates the changed properties).
     *
     * @return same instance as original
     */
    public Message applyTo(Message original) {
        if (subject != null) {
            original.setSubject(subject);
        }
        if (content != null) {
            original.setContent(content);
        }
        if (participants != null) {
            original.setParticipants(participants.stream().map(Object::toString).collect(toList()));
        }
        if (patientId != null) {
            original.setPatientId(patientId.toString());
        }
        if (receiverId != null) {
            original.setReceiverId(receiverId);
        }
        if (privateMessage != null) {
            original.setPrivateMessage(privateMessage);
        }
        return original;
    }

    public Long getMessageId() {
        return messageId;
    }

    public void setMessageId(Long messageId) {
        this.messageId = messageId;
    }

    public UUID getEncounterUniqueId() {
        return encounterUniqueId;
    }

    public void setEncounterUniqueId(UUID encounterUniqueId) {
        this.encounterUniqueId = encounterUniqueId;
    }

    public String getSubject() {
        return subject;
    }

    public void setSubject(String subject) {
        this.subject = subject;
    }

    public String getContent() {
        return content;
    }

    public void setContent(String content) {
        this.content = content;
    }

    public List<Long> getParticipants() {
        return participants;
    }

    public void setParticipants(List<Long> participants) {
        this.participants = participants;
    }

    public Long getPatientId() {
        return patientId;
    }

    public void setPatientId(Long patientId) {
        this.patientId = patientId;
    }

    public boolean isFirstDraft() {
        return messageId == null;
    }

    public Boolean isPrivateMessage() {
        return privateMessage;
    }

    public void setPrivateMessage(Boolean privateMessage) {
        this.privateMessage = privateMessage;
    }

    public String getReceiverId() {
        return receiverId;
    }

    public void setReceiverId(String receiverId) {
        this.receiverId = receiverId;
    }

}
