package com.pkb.action.message;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import org.immutables.value.Value;

@Value.Immutable
@JsonSerialize(as = ImmutableOffsetPagingInfo.class)
@JsonDeserialize(as = ImmutableOffsetPagingInfo.class)
public abstract class OffsetPagingInfo {

    @Value.Parameter
    public abstract long numberOfElements();

    @Value.Parameter
    public abstract int pageSize();

    @Value.Parameter
    public abstract long offset();

    @Value.Derived
    public long nextOffset() {
        return offset() + pageSize();
    }

    @Value.Derived
    public long previousOffset() {
        return offset() - pageSize();
    }

    @Value.Derived
    public long firstOffset() {
        return 0;
    }

    @Value.Derived
    public long lastOffset() {
        return numberOfElements() - pageSize() + 1;
    }

    @Value.Derived
    public boolean isPreviousPageAvailable() {
        return pageSize() < offset();
    }

    @Value.Derived
    public boolean isNextPageAvailable() {
        return numberOfElements() > offset() + pageSize() - 1;
    }

    @Value.Derived
    public boolean isLastPage() {
        return numberOfElements() < offset() + pageSize();
    }
}
