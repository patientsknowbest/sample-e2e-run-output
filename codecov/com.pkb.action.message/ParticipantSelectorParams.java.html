package com.pkb.action.message;

import com.pkb.dto.PersonAvailabilityDTO;
import com.pkb.dto.PersonDto;
import com.pkb.dto.TeamCliniciansDTO;
import com.pkb.dto.TeamDTO;

import java.util.ArrayList;
import java.util.List;

import static org.apache.commons.collections4.CollectionUtils.isNotEmpty;

public class ParticipantSelectorParams {
    private TeamDTO consultTeam;
    private String userCarersLabel;
    private List<PersonAvailabilityDTO> consultTeamList = new ArrayList<>();
    private List<TeamCliniciansDTO> cliniciansByTeam = new ArrayList<>();
    private List<PersonAvailabilityDTO> nonTeamClinicians = new ArrayList<>();
    private List<PersonAvailabilityDTO> colleagues = new ArrayList<>();
    private List<PersonDto> carerList = new ArrayList<>();

    public TeamDTO getConsultTeam() {
        return consultTeam;
    }

    public void setConsultTeam(TeamDTO consultTeam) {
        this.consultTeam = consultTeam;
    }

    public String getUserCarersLabel() {
        return userCarersLabel;
    }

    public void setUserCarersLabel(String userCarersLabel) {
        this.userCarersLabel = userCarersLabel;
    }

    public List<PersonAvailabilityDTO> getConsultTeamList() {
        return consultTeamList;
    }

    public void setConsultTeamList(List<PersonAvailabilityDTO> consultTeamList) {
        this.consultTeamList = consultTeamList;
    }

    public List<TeamCliniciansDTO> getCliniciansByTeam() {
        return cliniciansByTeam;
    }

    public void setCliniciansByTeam(List<TeamCliniciansDTO> cliniciansByTeam) {
        this.cliniciansByTeam = cliniciansByTeam;
    }

    public List<PersonAvailabilityDTO> getNonTeamClinicians() {
        return nonTeamClinicians;
    }

    public void setNonTeamClinicians(List<PersonAvailabilityDTO> nonTeamClinicians) {
        this.nonTeamClinicians = nonTeamClinicians;
    }

    public List<PersonAvailabilityDTO> getColleagues() {
        return colleagues;
    }

    public void setColleagues(List<PersonAvailabilityDTO> colleagues) {
        this.colleagues = colleagues;
    }

    public List<PersonDto> getCarerList() {
        return carerList;
    }

    public void setCarerList(List<PersonDto> carerList) {
        this.carerList = carerList;
    }

    public boolean potentialRecipientExists() {
        return isNotEmpty(consultTeamList) ||
                cliniciansByTeam.stream().anyMatch(c -> isNotEmpty(c.getCliniciansList())) ||
                isNotEmpty(nonTeamClinicians) ||
                isNotEmpty(colleagues) ||
                isNotEmpty(carerList);
    }
}
