package com.pkb.service.diagnosis;

import com.pkb.app.dto.PageFilterDTO;
import com.pkb.app.entity.EHRData;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.EHRSearch;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.PKBFilter;
import com.pkb.coding.entity.CodingReceived;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.data.EHRRemote;
import com.pkb.diagnosis.entity.Diagnosis;
import com.pkb.entities.enums.CodingDataType;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.exception.PKBPluginException;
import com.pkb.service.coding.CodingManager;
import com.pkb.service.ehr.DeduplicatorManager;
import com.pkb.service.ehr.MassUpdatableEhrDataPointManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import io.vavr.Function2;
import io.vavr.Tuple2;
import io.vavr.Tuple3;
import org.jetbrains.annotations.NotNull;

import java.time.Instant;
import java.util.List;

import static org.apache.commons.collections4.CollectionUtils.isNotEmpty;

public class EHRDiagnosisManager extends MassUpdatableEhrDataPointManager<Diagnosis> {

    private static final String PATIENT_HAS_CONDITION = "boolean01";

    private static final Function2<Diagnosis, Long, Void> CODING_RECEIVED_ID_SETTER = (diagnosis, privateId) -> {
        diagnosis.setDiagnosisCodingReceivedId(privateId);
        return null;
    };

    private static final Function2<Diagnosis, PrivacyFlag, Void> PRIVACY_FLAG_HANDLER = (diagnosis, privacyFlag) -> {
        diagnosis.getBaseFields().setPrivacyFlag(privacyFlag);
        return null;
    };

    public EHRDiagnosisManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, CodingManager codingManager, DeduplicatorManager deduplicator, Class<Diagnosis> dtoClass, MenuDataType dataType) {
        super(config, beanFactory, dateTimeService, uuidProvider, codingManager, deduplicator, dtoClass, dataType);
    }

    public List<Diagnosis> getDiagnoses(@NotNull LoggedInEHRRequestContext requestContext, long patientAccountId, PageFilterDTO pageFilterOrNull,
                                        boolean includeNegativeConditions, boolean includeNonProblems) {
        EHRRemote ehrBean = beanFactory.getEhrRemote();
        EHRSearch<Diagnosis> ehrSearch = new EHRSearch<>(patientAccountId, Diagnosis.class, Diagnosis.MS_PATH);
        ehrSearch.setLatestPerTypeFilter(EHRData.ENTERED_DATE, EHRData.UNIQUE_ID);
        addNegativeConditionFilter(includeNegativeConditions, ehrSearch);
        addProblemsOnlyFilter(includeNonProblems, ehrSearch);
        addPageFilter(pageFilterOrNull, ehrSearch);
        return ehrBean.queryDecryptAndPopulateData(ehrSearch, requestContext);
    }

    private void addNegativeConditionFilter(boolean includeNegativeConditions, EHRSearch search) {
        if (!includeNegativeConditions) {
            search.addFilter(new PKBFilter(PATIENT_HAS_CONDITION, PKBFilter.Operator.TRUE_AND_IS_NOT_NULL, null, false/*applyWhenChoosingLatest*/));
        }
    }

    private void addProblemsOnlyFilter(boolean includeNonProblems, EHRSearch ehrSearch) {
        if (!includeNonProblems) {
            ehrSearch.addFilter(new PKBFilter(Diagnosis.DIAGNOSIS_CATEGORY, PKBFilter.Operator.IS_NOT_NULL, null, false/*applyWhenChoosingLatest*/));
        }
    }

    private void addPageFilter(PageFilterDTO pageFilterOrNull, EHRSearch search) {
        if (pageFilterOrNull != null) {
            search.setResultsMaxReturned(pageFilterOrNull.getPageSize());
            search.setResultsStartPosition(pageFilterOrNull.getOffset());
        }
    }

    public List<Diagnosis> getDiagnoses(@NotNull LoggedInEHRRequestContext requestContext, long patientAccountId, List<Long> diagnosisIds,
                                        boolean includeNegativeConditions) {
        try {
            EHRRemote ehrBean = beanFactory.getEhrRemote();
            EHRSearch<Diagnosis> ehrSearch = new EHRSearch<>(patientAccountId, Diagnosis.class, Diagnosis.MS_PATH);
            ehrSearch.addFilter(new PKBFilter("id", PKBFilter.Operator.IN, diagnosisIds));
            ehrSearch.setLatestPerTypeFilter(EHRData.ENTERED_DATE, EHRData.UNIQUE_ID);
            addNegativeConditionFilter(includeNegativeConditions, ehrSearch);
            return ehrBean.queryDecryptAndPopulateData(ehrSearch, requestContext);
        } catch (Exception e) {
            throw new PKBPluginException("Exception while fetching diagnosis record for patient account Id -" + patientAccountId, e);
        }
    }

    public List<Diagnosis> getDiagnoses(long patientAccountId, List<Long> diagnosisCodeIdsOrEmpty, Instant updatedSinceDate,
                                        @NotNull LoggedInEHRRequestContext requestContext,
                                        boolean includeNegativeConditions, boolean includeNonProblems) {
        EHRRemote ehrBean = beanFactory.getEhrRemote();
        EHRSearch<Diagnosis> ehrSearch = new EHRSearch<>(patientAccountId, Diagnosis.class, Diagnosis.MS_PATH);
        if (updatedSinceDate != null) {
            ehrSearch.addFilter(new PKBFilter(EHRData.ENTERED_DATE, PKBFilter.Operator.GREATER, updatedSinceDate));
        }
        ehrSearch.setLatestPerTypeFilter(EHRData.ENTERED_DATE, EHRData.UNIQUE_ID);
        if (isNotEmpty(diagnosisCodeIdsOrEmpty)) {
            ehrSearch.addFilter(new PKBFilter(Diagnosis.DIAGNOSIS_DIAGNOSIS_CODE_ID, PKBFilter.Operator.IN, diagnosisCodeIdsOrEmpty));
        }
        addNegativeConditionFilter(includeNegativeConditions, ehrSearch);
        addProblemsOnlyFilter(includeNonProblems, ehrSearch);
        return ehrBean.queryDecryptAndPopulateData(ehrSearch, requestContext);
    }

    @Override
    protected List<Tuple2<Diagnosis, EHRRequestContext>> adjustCodings(List<Tuple2<Diagnosis, EHRRequestContext>> data) {
        return doAdjustCodings(data, CodingDataType.DIAGNOSIS, Diagnosis::getDiagnosis,
                CODING_RECEIVED_ID_SETTER,
                PRIVACY_FLAG_HANDLER);
    }

    @Override
    protected List<Tuple2<Diagnosis, EHRRequestContext>> adjustCodingsNoCodingLookup(List<Tuple3<Diagnosis, CodingReceived, EHRRequestContext>> data) {
        return doAdjustCodingsNoCodingLookup(data, CODING_RECEIVED_ID_SETTER, PRIVACY_FLAG_HANDLER);
    }

}
