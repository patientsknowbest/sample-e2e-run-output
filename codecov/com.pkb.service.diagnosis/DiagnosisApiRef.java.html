    private static Map<String, DiagnosisApiRef> apiRefMap = new HashMap<>();

    // assume for now many to many mapping
    private static Map<String, String> codeToApiRefMap = new HashMap<>();

    static {
        // mapping of apirefs to code details
        DiagnosisApiRef[] diagnosisApiRefs = DiagnosisApiRef.values();
        for (DiagnosisApiRef diagnosisApiRef : diagnosisApiRefs) {
            DiagnosisApiRef apiRef = apiRefMap.get(diagnosisApiRef.getApiRef());
            if (apiRef == null) {
                apiRefMap.put(diagnosisApiRef.getApiRef(), diagnosisApiRef);
                codeToApiRefMap.put(diagnosisApiRef.getCode(), diagnosisApiRef.getApiRef());
            }
        }

    }

    public static String getCodeByApiRef(String apiRefStr) {
        String apiRef = null;
        DiagnosisApiRef diagnosisApiRef = apiRefMap.get(apiRefStr);
        if (diagnosisApiRef != null) {
            apiRef = diagnosisApiRef.getCode();
        }
        return apiRef;
    }

    public static List<String> getCodesByApiRef(List<String> apiRefList) {
        List<String> codes = new ArrayList<>();
        for (String apiRefStr : apiRefList) {
            DiagnosisApiRef apiRef = apiRefMap.get(apiRefStr);
            if (apiRef != null) {
                codes.add(apiRef.getCode());
            }
        }
        return codes;
    }

    public static String getApiRefByCode(String code) {
        return codeToApiRefMap.get(code);
    }

    public static DiagnosisApiRef getByApiRef(String apiRef) {
        return apiRefMap.get(apiRef);
    }
