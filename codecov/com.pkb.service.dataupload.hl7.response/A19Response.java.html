package com.pkb.service.dataupload.hl7.response;

import com.pkb.datamodel.user.Demographics;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.service.dataupload.hl7.HL7ConnContext;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.segment.HL7QueryDetail;
import com.pkb.service.dataupload.hl7.segment.HL7QueryFilter;
import com.pkb.user.entity.PKBPerson;
import io.vavr.Tuple4;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.Nullable;

import java.time.LocalDate;
import java.util.Collections;
import java.util.Optional;
import java.util.Set;
import java.util.regex.Pattern;

import static com.pkb.PKBConstants.DATE_OF_BIRTH_FORMATTER;
import static java.util.stream.Collectors.toSet;
import static org.apache.commons.lang3.StringUtils.EMPTY;
import static org.apache.commons.lang3.StringUtils.isBlank;
import static org.apache.commons.lang3.StringUtils.isNotBlank;
import static org.apache.commons.lang3.StringUtils.trim;

public abstract class A19Response {

    private static final Pattern DASH_PATTERN = Pattern.compile("-");
    public static final char HL7_VALUE_SEPARATOR_CHAR = '~';

    protected HL7QueryDetail detail;
    private HL7QueryFilter filter;
    private HL7XmlDoc hl7;
    private String timestamp;
    private HL7ConnContext hl7ConnContext;

    public A19Response(HL7XmlDoc hl7, String timestamp, HL7QueryDetail detail,
                       @Nullable HL7QueryFilter filter, HL7ConnContext hl7ConnContext) {
        this.hl7 = hl7;
        this.timestamp = timestamp;
        this.detail = detail;
        this.filter = filter;
        this.hl7ConnContext = hl7ConnContext;
    }

    public String getNonSensitiveResponse() {
        String response = "MSH|^~\\&|HL7API|PKB|"
                + hl7.getSendingApp().toHl7() + "|" + hl7.getSendingFacility()
                + "|" + timestamp + "||ADR^A19|PKB-" + timestamp
                + "|P|2.4||||||||"
                + "\nMSA|AA|" + hl7.getMessageIdString()
                + "\n" + detail.toHl7();

        if (filter != null) {
            response += "\n" + filter.toHl7();
        }
        return response;
    }

    public String getFullResponse() {
        StringBuilder response = new StringBuilder(getNonSensitiveResponse());
        Org org = hl7ConnContext.getSourceOrg();
        Team team = hl7ConnContext.getSourceTeam().orElse(null);
        NationalIdType expectedType = hl7ConnContext.getNationalIdType().orElse(null);

        addPersonSpecificDetails(response, org, team, expectedType);

        response.append("\n");
        return response.toString();
    }

    abstract void addPersonSpecificDetails(StringBuilder response, Org org, Team team, NationalIdType expectedType);

    protected String getAltIdValues(PKBPerson person, Org org, Team team) {
        String orgLevelAltIdValues = getOrgLevelAltIdValues(detail.getQueryFilterType().getString(), person, org);

        String altIdValues = appendTeamLevelAltIdValues(
                orgLevelAltIdValues,
                team, org, person);

        return altIdValues;
    }

    protected String getAltIdValues(Demographics personDemographics, Org org, Team team) {
        StringBuilder orgLevelAltIdValuesBuilder = getOrgLevelAltIdValues(detail.getQueryFilterType().getString(), personDemographics, org);
        appendTeamLevelAltIdValues(orgLevelAltIdValuesBuilder, team, org, personDemographics);
        return orgLevelAltIdValuesBuilder.toString();
    }

    protected String getNIdValue(NationalIdType expectedType, PKBPerson person) {
        if (expectedType != null) {
            return person.getNationalIdAnyTypeValidForSameCountryAsType(expectedType)
                    .filter(nid -> isNotBlank(nid.getValue()))
                    .map(nid -> nid.getValue() + "^^^" + nid.getType().getHl7IdAuthority() + "^" + nid.getType().getHl7IdType())
                    .orElse(EMPTY);
        }
        return EMPTY;
    }

    protected String getNIdValue(NationalIdType expectedType, Demographics personDemographics) {
        if (expectedType != null) {
            return personDemographics.getNationalIdByTypeInCountry(expectedType)
                    .filter(nid -> isNotBlank(nid.getValue().getOrElse(EMPTY)))
                    .map(nid -> nid.getValue().getOrElse(EMPTY) + "^^^" + nid.getType().getHl7IdAuthority() + "^" + nid.getType().getHl7IdType())
                    .getOrElse(EMPTY);
        }
        return EMPTY;
    }

    protected StringBuilder getOrgLevelAltIdValues(String queryFilterType, Demographics personDemographics, Org org) {
        // If this is an org-level connection, or if team-level
        // connection but not an"ID_RESULTS_ONLY" query, then
        // append OrgLevelId values.
        if (!hl7ConnContext.hasSourceTeam() || !HL7QueryDetail.ID_RESULTS_ONLY.equals(queryFilterType)) {
            StringBuilder altIdValuesBuilder = new StringBuilder();
            personDemographics.getOrgLevelId()
                    .filter(olid -> org.getId().equals(olid.getType().getOrg().getId()))
                    .sorted()
                    .forEach(olid -> appendOrgLevelId(org.getCode(), olid.getValue().getOrElse((String) null), altIdValuesBuilder, olid.getType().getAssigningAuthority(),
                            olid.getType().getTypeCode()));
            removeTrailingRepetitionCharacter(altIdValuesBuilder);
            return altIdValuesBuilder;
        }
        return new StringBuilder();
    }

    protected String getOrgLevelAltIdValues(String queryFilterType, PKBPerson person, Org org) {
        // If this is an org-level connection, or if team-level
        // connection but not an"ID_RESULTS_ONLY" query, then
        // append OrgLevelId values.
        if (!hl7ConnContext.hasSourceTeam() || !HL7QueryDetail.ID_RESULTS_ONLY.equals(queryFilterType)) {
            if (person.getOrgLevelIds() != null) {
                StringBuilder altIdValuesBuilder = new StringBuilder();
                person.getOrgLevelIds().stream()
                        .filter(olid -> olid.getOrg().getId().equals(org.getId()))
                        .forEach(olid -> appendOrgLevelId(org.getCode(), olid.getValue(), altIdValuesBuilder, olid.getType().getHl7AssigningAuthority(),
                                olid.getType().getHl7TypeCode()));
                removeTrailingRepetitionCharacter(altIdValuesBuilder);
                return altIdValuesBuilder.toString();
            }
        }
        return EMPTY;
    }

    protected void appendOrgLevelId(String orgCode, String orgLevelIdValue, StringBuilder altIdValuesBuilder, String assigningAuthority, String typeCode) {
        String aa = assigningAuthority;
        String tc = typeCode;
        if (isBlank(aa)) {
            aa = HL7XmlDoc.hl7Escape(orgCode);
            tc = "MR";
        }
        altIdValuesBuilder.append(orgLevelIdValue).append("^^^").append(aa).append("^").append(tc).append("~");
    }

    protected void appendTeamLevelAltIdValues(StringBuilder altIdValuesBuilder, Team team, Org org, Demographics personDemographics) {
        Set<Long> teamIdList = getTeamIdList(team, org);
        appendSeparatorCharWhenNeeded(altIdValuesBuilder);
        personDemographics.getTeamLevelId()
                .filter(tlid -> teamIdList.contains(tlid.getType().getTeam().getId()))
                .sorted()
                .forEach(tlid -> appendTeamLevelId(org.getCode(), tlid.getType().getTeam().getCode(), tlid.getValue().getOrElse((String) null), altIdValuesBuilder,
                        tlid.getType().getAssigningAuthority(),
                        tlid.getType().getTypeCode()));
        removeTrailingRepetitionCharacter(altIdValuesBuilder);
    }

    protected String appendTeamLevelAltIdValues(String altIdValues, Team team, Org org, PKBPerson person) {
        Set<Long> teamIdList = getTeamIdList(team, org);
        StringBuilder altIdValuesBuilder = new StringBuilder(altIdValues);
        appendSeparatorCharWhenNeeded(altIdValuesBuilder);
        if (person.getTeamLevelIds() != null) {
            person.getTeamLevelIds().stream()
                    .filter(tid -> teamIdList.contains(tid.getTeam().getId()))
                    .forEach(tlid -> appendTeamLevelId(org.getCode(), tlid.getTeam().getCode(), tlid.getValue(), altIdValuesBuilder, tlid.getType().getHl7AssigningAuthority(),
                            tlid.getType().getHl7TypeCode()));
            removeTrailingRepetitionCharacter(altIdValuesBuilder);
        }
        return altIdValuesBuilder.toString();
    }

    private void appendSeparatorCharWhenNeeded(StringBuilder builder) {
        if (0 < builder.length()) {
            builder.append(HL7_VALUE_SEPARATOR_CHAR);
        }
    }

    private void removeTrailingRepetitionCharacter(StringBuilder builder) {
        if (0 < builder.length()) {
            if (HL7_VALUE_SEPARATOR_CHAR == builder.charAt(builder.length() - 1)) {
                builder.deleteCharAt(builder.length() - 1);
            }
        }
    }

    private Set<Long> getTeamIdList(Team team, Org org) {
        // Append all TeamLevelId values. If this is an org-level
        // connection, then include IDs for all Teams.
        if (hl7ConnContext.hasSourceTeam()) {
            return Collections.singleton(team.getId());
        } else {
            return org.getTeams().stream().map(Team::getId).collect(toSet());
        }
    }

    private void appendTeamLevelId(String orgCode, String teamCode, String teamLevelValue, StringBuilder altIdValuesBuilder, String assigningAuthority, String typeCode) {
        String aa = assigningAuthority;
        String tc = typeCode;
        if (isBlank(aa)) {
            aa = HL7XmlDoc.hl7Escape(orgCode + ":" + teamCode);
            tc = "MR";
        }
        altIdValuesBuilder.append(teamLevelValue).append("^^^").append(aa).append("^").append(tc).append("~");
    }

    protected Tuple4<String, String, String, String> getDemographics(PKBPerson p, String queryFilterType) {
        // don't fill in demographic info if they are avoiding it (for privacy reasons)
        if (HL7QueryDetail.ID_RESULTS_ONLY.equals(queryFilterType)) {
            return new Tuple4<>(EMPTY, EMPTY, EMPTY, EMPTY);
        }
        String lName = HL7XmlDoc.hl7Escape(p.getLastName());
        String fName = HL7XmlDoc.hl7Escape(p.getFirstName());
        String dob = getHL7FormatDobString(p.findDateOfBirth());
        String address = EMPTY;
        // address: for now, just postal code if they have one
        if (isNotBlank(p.getPostalCode())) {
            address = "^^^^" + trim(p.getPostalCode());
        }
        return new Tuple4<>(lName, fName, dob, address);
    }

    protected Tuple4<String, String, String, String> getDemographics(Demographics p, String queryFilterType) {
        // don't fill in demographic info if they are avoiding it (for privacy reasons)
        if (HL7QueryDetail.ID_RESULTS_ONLY.equals(queryFilterType)) {
            return new Tuple4<>(EMPTY, EMPTY, EMPTY, EMPTY);
        }
        String lName = p.getLastName().map(HL7XmlDoc::hl7Escape).getOrElse(EMPTY);
        String fName = p.getFirstName().map(HL7XmlDoc::hl7Escape).getOrElse(EMPTY);
        String dob = getHL7FormatDobString(p.getDateOfBirth().toJavaOptional());
        // address: for now, just postal code if they have one
        String address = p.getAddress().getPostCode().map(pc -> "^^^^" + trim(pc)).getOrElse(EMPTY);
        return new Tuple4<>(lName, fName, dob, address);
    }

    private String getHL7FormatDobString(Optional<LocalDate> optionalDob) {
        String dobString = optionalDob.map(DATE_OF_BIRTH_FORMATTER::format).map(StringUtils::trimToEmpty).orElse(EMPTY);
        return DASH_PATTERN.matcher(dobString).replaceAll("");
    }
}
