package com.pkb.service.dataupload.hl7.response;

import java.util.List;

import com.pkb.entities.enums.NationalIdType;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.service.dataupload.hl7.HL7ConnContext;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.segment.HL7QueryDetail;
import com.pkb.service.dataupload.hl7.segment.HL7QueryFilter;
import com.pkb.user.entity.PKBPerson;

import io.vavr.Tuple4;

public class StandardA19Response extends A19Response {

    private List<PKBPerson> personList;

    public StandardA19Response(HL7XmlDoc hl7, String timestamp, HL7QueryDetail detail,
                               HL7QueryFilter filter, HL7ConnContext hl7ConnContext, List<PKBPerson> personList) {
        super(hl7, timestamp, detail, filter, hl7ConnContext);
        this.personList = personList;
    }

    @Override
    void addPersonSpecificDetails(StringBuilder response, Org org, Team team, NationalIdType expectedType) {

        for (PKBPerson person : personList) {

            String altIdValues = getAltIdValues(person, org, team);

            Tuple4<String,String,String,String> demographics =
                    getDemographics(person, detail.getQueryFilterType().getString());

            String nIdValue = getNIdValue(expectedType, person);

            response.append("\nPID||");

            response.append(nIdValue).append("|")
                    .append(nIdValue).append("|").append(altIdValues)
                    .append("|")
                    .append(demographics._1)
                    .append("^").append(demographics._2)
                    .append("||").append(demographics._3)
                    .append("||||").append(demographics._4);
        }
    }
}
