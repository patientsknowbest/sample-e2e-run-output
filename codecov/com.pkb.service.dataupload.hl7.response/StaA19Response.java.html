package com.pkb.service.dataupload.hl7.response;

import com.pkb.datamodel.user.Demographics;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.reports.A19Record;
import com.pkb.service.dataupload.hl7.HL7ConnContext;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.segment.HL7QueryDetail;
import com.pkb.service.dataupload.hl7.segment.HL7QueryFilter;
import com.pkb.service.uuid.UUIDProvider;
import io.vavr.Tuple2;
import io.vavr.Tuple4;

import java.time.ZoneId;
import java.time.format.DateTimeFormatter;
import java.util.List;
import java.util.Locale;
import java.util.Optional;

import static com.pkb.util.Constants.isNegativeInfinity;
import static org.apache.commons.lang3.StringUtils.isNotEmpty;

public class StaA19Response extends A19Response {
    private final static DateTimeFormatter HL7_ZID_SEGMENT_FORMATTER = DateTimeFormatter.ofPattern("yyyyMMddHHmmss.SSSZ").withLocale(Locale.UK)
            .withZone(ZoneId.of("UTC"));

    private Tuple2<Boolean, List<A19Record>> personsWithRegStatus;
    private final UUIDProvider uuidProvider;

    public StaA19Response(HL7XmlDoc hl7, String timestamp, HL7QueryDetail detail, Optional<HL7QueryFilter> filter,
                          HL7ConnContext hl7ConnContext, Tuple2<Boolean, List<A19Record>> personsWithRegStatus, UUIDProvider uuidProvider) {
        super(hl7, timestamp, detail, filter.orElse(null), hl7ConnContext);
        this.personsWithRegStatus = personsWithRegStatus;
        this.uuidProvider = uuidProvider;
    }

    @Override
    void addPersonSpecificDetails(StringBuilder response, Org org, Team team, NationalIdType expectedType) {

        boolean resultTruncated = personsWithRegStatus._1;
        for (A19Record a19Record : personsWithRegStatus._2) {

            Demographics personDemographics = a19Record.getDemographics();
            String altIdValues = getAltIdValues(personDemographics, org, team);

            Tuple4<String, String, String, String> demographics = getDemographics(personDemographics, detail.getQueryFilterType().getString());

            String nIdValue = getNIdValue(expectedType, personDemographics);
            response.append("\nPID||");

            response.append("|").append(nIdValue);
            if (isNotEmpty(altIdValues)) {
                response.append("~").append(altIdValues);
            }
            response.append("||");

            response.append(demographics._1)
                    .append("^").append(demographics._2)
                    .append("||").append(demographics._3)
                    .append("||||").append(demographics._4)
                    .append("|");

            response.append("\nZID|")
                    .append(a19Record.getPublicId()).append("|")
                    .append(a19Record.getRegistrationStatus().toString()).append("|");

            if (isNegativeInfinity(a19Record.getLastUpdated())) {
                // Append nothing for postgres -infinity "magic" value
            } else {
                response.append(HL7_ZID_SEGMENT_FORMATTER.format(a19Record.getLastUpdated()));
            }
        }

        if (resultTruncated) {
            response.append("\nDSC|").append(uuidProvider.randomUUID());
        }
    }
}
