package com.pkb.phoenix.api.service.task;

import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;

import java.time.LocalDateTime;
import java.util.List;
import java.util.Optional;

/**
 * The model used by our TaskService to represent an instance of a task. (ie a task that has been created, approved, rejected, executed etc by a user)
 * @param <PARAMETER> - the type of parameters accepted by the task
 */

@Style(stagedBuilder = true)
public interface TaskInstance<PARAMETER> {
    @Immutable
    interface TaskType {
        String code();
    }

    @Immutable
    interface Actor {
        int id();

        String name();

        boolean isIg();
    }

    @Immutable
    interface Event {
        int id();

        Actor actor();

        String event();

        LocalDateTime at();

        Optional<String> note();
    }

    @Immutable
    abstract class AbstractTaskInstance<P> implements TaskInstance<P> {
    }

    int id();

    Status status();

    PARAMETER parameters();

    TaskType type();

    Event mostRecentEvent();

    List<Event> events();
}
