package com.pkb.phoenix.api.service.task;

import com.pkb.phoenix.api.service.task.TaskInstance.Event;
import com.pkb.phoenix.repository.task.projection.Task;
import com.pkb.phoenix.repository.taskeventlog.projection.EventWithActorDetails;
import org.mapstruct.Mapper;

import static com.pkb.phoenix.api.service.task.Status.DONE;
import static com.pkb.phoenix.api.service.task.Status.FAILED;
import static com.pkb.phoenix.api.service.task.TaskService.toStatus;
import static java.util.stream.Collectors.toUnmodifiableList;

@Mapper
public abstract class TaskMapper {

    // PNX-118
    public final <P> FailedTask<P> buildModel(Task<P> task, Throwable cause) {
        return ImmutableFailedTask.<P>builder()
                .id(task.getId())
                .status(FAILED)
                .parameters(task.getParameters())
                .type(ImmutableTaskType.builder().code(task.getTypeCode()).build())
                .mostRecentEvent(buildModel(task.getLatestEvent()))
                .addAllEvents(task.events().stream().map(this::buildModel).collect(toUnmodifiableList()))
                .build();
    }

    // PNX-118
    public final <P, R> CompletedTask<P, R> buildModel(Task<P> task, R result) {
        return ImmutableCompletedTask.<P, R>builder()
                .id(task.getId())
                .type(ImmutableTaskType.builder().code(task.getTypeCode()).build())
                .parameters(task.getParameters())
                .status(DONE)
                .result(result)
                .mostRecentEvent(buildModel(task.getLatestEvent()))
                .addAllEvents(task.events().stream().map(this::buildModel).collect(toUnmodifiableList()))
                .build();
    }

    public final <P> TaskInstance<P> buildModel(Task<P> task) {
        return ImmutableTaskInstance.<P>builder()
                .id(task.getId())
                .status(toStatus(task.getLatestEvent().getEvent()))
                .parameters(task.getParameters())
                .type(ImmutableTaskType.builder().code(task.getTypeCode()).build())
                .mostRecentEvent(buildModel(task.getLatestEvent()))
                .addAllEvents(task.events().stream().map(this::buildModel).collect(toUnmodifiableList()))
                .build();
    }

    public final Event buildModel(EventWithActorDetails eventWithUser) {
        var actor = ImmutableActor.builder()
                .id(eventWithUser.getActorId())
                .name(eventWithUser.getActorName())
                .isIg(eventWithUser.isActorIg())
                .build();
        return ImmutableEvent.builder()
                .id(eventWithUser.eventId())
                .actor(actor)
                .event(eventWithUser.getEvent().name())
                .at(eventWithUser.getAt())
                .note(eventWithUser.getNote())
                .build();
    }
}
