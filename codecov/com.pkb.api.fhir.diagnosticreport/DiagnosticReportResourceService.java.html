package com.pkb.api.fhir.diagnosticreport;

import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;

import com.pkb.api.fhir.config.FhirApiProperties;
import com.pkb.api.fhir.exception.ServerResponseExceptionMapper;
import com.pkb.authz.AuthorizationDataService;
import com.pkb.authz.data.AuthorizationData;

import com.pkb.datamodel.RadiologyImage;
import com.pkb.domain.SystemUserTypeNotPermittedException;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.radiology.ImmutableRadiologyCriteria;
import com.pkb.domain.radiology.RadiologyCriteria;
import com.pkb.domain.radiology.RadiologyService;
import io.vavr.control.Either;

import static com.google.common.base.Predicates.instanceOf;
import static com.pkb.authz.ImmutableTargetMeta.targetMeta;
import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.context.Path.FHIR;
import static com.pkb.authz.data.ImmutableAuthorizationInputs.inputs;
import static com.pkb.domain.criteria.CriteriaResult.emptyCriteriaResult;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.API.Set;
import static io.vavr.API.Some;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

public class DiagnosticReportResourceService extends AbstractResourceService<DiagnosticReport> {
    private final DiagnosticReportMapper diagnosticReportMapper;
    private final RadiologyService radiologyService;
    private final AuthorizationDataService authorizationDataService;
    private final FhirApiProperties fhirConfig;

    public DiagnosticReportResourceService(@NotNull ServerResponseExceptionMapper exceptionMapper,
                                           @NotNull DiagnosticReportMapper diagnosticReportMapper,
                                           @NotNull RadiologyService radiologyService,
                                           @NotNull AuthorizationDataService authorizationDataService,
                                           FhirApiProperties fhirConfig) {
        super(exceptionMapper);
        this.diagnosticReportMapper = diagnosticReportMapper;
        this.radiologyService = radiologyService;
        this.authorizationDataService = authorizationDataService;
        this.fhirConfig = fhirConfig;
    }

    public Either<BaseServerResponseException, CriteriaResult<DiagnosticReport>> search(@NotNull DiagnosticReportSearchCriteria searchCriteria) {
        AuthorizationInputs inputs = inputs()
                .authenticatedIdentity(searchCriteria.authenticatedIdentity())
                .action(SEARCH)
                .addTargets(targetMeta(RadiologyImage.class, Set(), Some(searchCriteria.patientId()), true))
                .path(FHIR)
                .build();

        AuthorizationData authzData = authorizationDataService.fetch(inputs);

        RadiologyCriteria criteria = ImmutableRadiologyCriteria.builder()
                .authzData(authzData)
                .patientPublicId(searchCriteria.patientId())
                .effectiveTimestamp(searchCriteria.effectiveTimestamp())
                .pageSize(fhirConfig.getDiagnosticReportMaxNumberOfResources())
                .build();

        return radiologyService.search(criteria)
                .peekLeft(logAuthorizationError(DiagnosticReport.class))
                .fold(error -> Match(error).of(
                        Case($(instanceOf(SystemUserTypeNotPermittedException.class)), nae -> left(toSystemUserTypeNotPermitted())),
                        Case($(), right(emptyCriteriaResult()))),
                        diagnosticReports -> right(diagnosticReportMapper.toDiagnosticReports(diagnosticReports)));
    }
}
