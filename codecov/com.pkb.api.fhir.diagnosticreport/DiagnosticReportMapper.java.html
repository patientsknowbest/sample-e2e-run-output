package com.pkb.api.fhir.diagnosticreport;

import com.pkb.api.fhir.PrivacyFlagsMapper;
import com.pkb.api.fhir.TemporalMapper;
import com.pkb.datamodel.RadiologyImage;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.criteria.ImmutableCriteriaResult;
import org.hl7.fhir.dstu3.model.DiagnosticReport;
import org.hl7.fhir.dstu3.model.Meta;
import org.jetbrains.annotations.NotNull;

import static com.pkb.api.fhir.DataAbsentReasonUtil.getDataAbsentExtension;

public class DiagnosticReportMapper {
    private final TemporalMapper temporalMapper;
    private final PrivacyFlagsMapper privacyFlagsMapper;

    public DiagnosticReportMapper(TemporalMapper temporalMapper, PrivacyFlagsMapper privacyFlagsMapper) {
        this.temporalMapper = temporalMapper;
        this.privacyFlagsMapper = privacyFlagsMapper;
    }

    public @NotNull CriteriaResult<DiagnosticReport> toDiagnosticReports(@NotNull CriteriaResult<RadiologyImage> radiologyImages) {
        return ImmutableCriteriaResult.<DiagnosticReport> builder()
                .addAllPage(radiologyImages.getPage().map(this::toDiagnosticReport))
                .total(radiologyImages.getTotal())
                .build();
    }

    // TODO: generate this with mapstruct
    public @NotNull DiagnosticReport toDiagnosticReport(@NotNull RadiologyImage radiologyImage) {
        DiagnosticReport report = new DiagnosticReport();
        report.setId(radiologyImage.getUniqueId().toString());
        report.setStatus(DiagnosticReport.DiagnosticReportStatus.UNKNOWN);
        report.getCode().addExtension(getDataAbsentExtension());
        report.setEffective(temporalMapper.toDateTimeType(radiologyImage.getEffectiveTimestamp()));

        Meta meta = new Meta();
        meta.setSecurity(privacyFlagsMapper.toCodingList(radiologyImage.getPrivacyFlags()).asJava());
        meta.setLastUpdated(temporalMapper.toDate(radiologyImage.getPersistedTimestamp()));
        report.setMeta(meta);

        return report;
    }
}
