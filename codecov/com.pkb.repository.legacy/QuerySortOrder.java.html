package com.pkb.repository.legacy;

/**
 * This exists as a copy of the class in the com.pkb.query.domain package in pkb-plugin\n
 * In time, all of the EJBs will be removed, and once that happens the pkb-plugin project\n
 * will also be removed. At some point even further in the future, we'll move to pure Spring\n
 * repos and then maybe this class will disappear completely.
 */
public class QuerySortOrder {
    public enum SortOrderDirection {
        ASC, DESC
    }

    private SortOrderDirection direction;
    private String fieldName;

    public QuerySortOrder(SortOrderDirection direction, String fieldName) {
        this.direction = direction;
        this.fieldName = fieldName;
    }

    public SortOrderDirection getDirection() {
        return direction;
    }

    public void setDirection(SortOrderDirection direction) {
        this.direction = direction;
    }

    public String getFieldName() {
        return fieldName;
    }

    public void setFieldName(String fieldName) {
        this.fieldName = fieldName;
    }
}
