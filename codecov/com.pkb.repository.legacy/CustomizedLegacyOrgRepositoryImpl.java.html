package com.pkb.repository.legacy;

import com.google.common.collect.Lists;
import com.pkb.entities.enums.OrgExternalIdType;
import com.pkb.institute.entity.Org;
import com.pkb.util.DbUtil;
import org.apache.commons.lang3.ArrayUtils;
import org.jetbrains.annotations.Nullable;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import javax.persistence.EntityManager;
import javax.persistence.TypedQuery;
import java.util.Collection;
import java.util.Collections;
import java.util.List;
import java.util.Optional;
import java.util.UUID;

import static com.pkb.util.DbUtil.getExactlyOneUniqueResultOrNull;
import static java.util.Collections.emptyList;
import static org.apache.commons.collections4.CollectionUtils.isEmpty;

public class CustomizedLegacyOrgRepositoryImpl implements CustomizedLegacyOrgRepository {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    private EntityManager em;

    private StringBuilder getOrgQueryBuilder(Org.Lazy[] fields) {
        StringBuilder queryBuilder = new StringBuilder("SELECT DISTINCT o FROM Org o ");
        if (ArrayUtils.contains(fields, Org.Lazy.TEAMS)) {
            queryBuilder.append(" LEFT JOIN FETCH o.teams AS orgTeams");
            queryBuilder.append(" LEFT JOIN FETCH orgTeams.teamLevelIdTypes");
        }

        if (ArrayUtils.contains(fields, Org.Lazy.EXTERNAL_IDS)) {
            queryBuilder.append(" LEFT JOIN FETCH o.externalIds ");
        }
        return queryBuilder;
    }

    @Override
    public Optional<Org> getOrg(Long orgId, Org.Lazy... fields) {
        if (orgId == null) {
            return Optional.empty();
        }
        StringBuilder orgQueryBuilder = getOrgQueryBuilder(fields);
        orgQueryBuilder.append(" WHERE o.id = :orgId");
        TypedQuery<Org> query = em.createQuery(orgQueryBuilder.toString(), Org.class);
        query.setParameter("orgId", orgId);
        return Optional.ofNullable(getExactlyOneUniqueResultOrNull(query));
    }

    @Override
    public List<Org> getOrgs(Collection<Long> orgIds, Org.Lazy... fields) {
        if (isEmpty(orgIds)) {
            return emptyList();
        }
        StringBuilder orgQueryBuilder = getOrgQueryBuilder(fields);
        orgQueryBuilder.append(" WHERE o.id IN (:orgIds)");
        TypedQuery<Org> query = em.createQuery(orgQueryBuilder.toString(), Org.class);
        query.setParameter("orgIds", Lists.newArrayList(orgIds));
        return query.getResultList();
    }

    @Override
    public Optional<Org> getOrgByCode(String orgCode, Org.Lazy... fields) {
        if (orgCode == null) {
            return Optional.empty();
        }
        StringBuilder orgQueryBuilder = getOrgQueryBuilder(fields);
        orgQueryBuilder.append(" WHERE o.code = :code");
        TypedQuery<Org> query = em.createQuery(orgQueryBuilder.toString(), Org.class);
        query.setParameter("code", orgCode);
        return Optional.ofNullable(getExactlyOneUniqueResultOrNull(query));
    }

    @Override
    public List<Org> getOrgsByCodes(Collection<String> orgCodes, Org.Lazy... fields) {
        if (isEmpty(orgCodes)) {
            return emptyList();
        }
        var queryString = getOrgQueryBuilder(fields)
                .append(" WHERE o.code IN (:orgCodes)")
                .toString();
        return DbUtil.getInBatches(orgCodes, batch -> em.createQuery(queryString, Org.class)
                .setParameter("orgCodes", batch)
                .getResultList());
    }

    @Override
    public Optional<Org> getOrgByOrgExternalIdType(OrgExternalIdType idType, String idValue, Org.Lazy... fields) {
        List<Org> items = getOrgsByOrgExternalIdType(idType, Collections.singleton(idValue), fields);
        if (items.isEmpty()) {
            return Optional.empty();
        }
        if (items.size() > 1) {
            LOGGER.error("Found multiple orgs for idType {} and value {}", idType, idValue);
        }
        return Optional.ofNullable(items.get(0));
    }

    @Override
    public List<Org> getOrgsByOrgExternalIdType(OrgExternalIdType idType, Collection<String> idValues, Org.Lazy... fields) {
        String select = "SELECT ex.org.id FROM OrgExternalId ex" +
                " WHERE ex.idType = :idType" +
                "   AND ex.idValue IN (:idValues)";

        TypedQuery<Long> query = em.createQuery(select, Long.class);
        query.setParameter("idType", idType);
        query.setParameter("idValues", idValues);

        return getOrgs(query.getResultList(), fields);
    }

    @Override
    @Nullable
    public Optional<Org> getOrgByPublicId(UUID publicId, Org.Lazy... fields) {
        StringBuilder orgQueryBuilder = getOrgQueryBuilder(fields);
        orgQueryBuilder.append(" WHERE o.publicId = (:publicId)");
        TypedQuery<Org> query = em.createQuery(orgQueryBuilder.toString(), Org.class);
        query.setParameter("publicId", publicId);
        return Optional.ofNullable(getExactlyOneUniqueResultOrNull(query));
    }

    @Override
    public List<Org> getOrgList(Org.Lazy... fields) {
        StringBuilder orgQueryBuilder = getOrgQueryBuilder(fields);
        orgQueryBuilder.append(" ORDER BY o.name");
        TypedQuery<Org> query = em.createQuery(orgQueryBuilder.toString(), Org.class);
        return query.getResultList();
    }
}
