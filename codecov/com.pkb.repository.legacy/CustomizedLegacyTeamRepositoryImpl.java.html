package com.pkb.repository.legacy;

import com.google.common.collect.Lists;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.entities.enums.SponsorshipStatus;
import com.pkb.institute.entity.Team;
import com.pkb.util.DbUtil;
import org.apache.commons.lang3.ArrayUtils;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.transaction.annotation.Transactional;

import javax.persistence.EntityManager;
import javax.persistence.Query;
import javax.persistence.TypedQuery;
import java.util.Collection;
import java.util.Collections;
import java.util.List;
import java.util.UUID;

import static com.pkb.util.DbUtil.getExactlyOneUniqueResultOrNull;
import static java.util.stream.Collectors.toSet;

public class CustomizedLegacyTeamRepositoryImpl implements CustomizedLegacyTeamRepository {

    @Autowired
    EntityManager em;

    @Override
    public List<Team> getTeamsWithActiveAlarms(long patientId) {
        try {
            TypedQuery<Team> query = em.createQuery(
                    "SELECT t from Team t, EHRData ed "
                            + "WHERE ed.accountId = t.teamAccountId "
                            + "AND ed.dataType = '" + MenuDataType.symptomAlarm + "' "
                            + "AND ed.string01 = :patientId "
                            + "AND ed.deleted = FALSE",
                    Team.class);
            query.setParameter("patientId", Long.toString(patientId));
            return query.getResultList();
        } catch (Exception e) {
            throw new RuntimeException("Error while fetching teams with active alarms for patient-" + patientId, e);
        }
    }

    private StringBuilder getInstituteQueryBuilder(Team.Lazy[] fields) {
        StringBuilder queryBuilder = new StringBuilder("SELECT DISTINCT t FROM Team t ");
        if (ArrayUtils.contains(fields, Team.Lazy.LIBRARY)) {
            queryBuilder.append(" LEFT JOIN FETCH t.libraryEntries ");
        }
        if (ArrayUtils.contains(fields, Team.Lazy.SYMPTOMS)) {
            queryBuilder.append(" LEFT JOIN FETCH t.symptomsToMonitor ");
        }
        if (ArrayUtils.contains(fields, Team.Lazy.ORG_EXT_IDS)) {
            queryBuilder.append(" LEFT JOIN FETCH t.org o LEFT JOIN FETCH o.externalIds ");
        }
        return queryBuilder;
    }

    @Override
    @Nullable
    public Team getInstitute(Long teamId, Team.Lazy... fields) {
        StringBuilder instituteQueryBuilder = getInstituteQueryBuilder(fields);
        instituteQueryBuilder.append(" WHERE t.id = :teamId");
        TypedQuery<Team> query = em.createQuery(instituteQueryBuilder.toString(), Team.class);
        query.setParameter("teamId", teamId);
        return getExactlyOneUniqueResultOrNull(query);
    }

    @Override
    public List<Team> getInstitutes(Collection<Long> teamId, Team.Lazy... fields) {
        StringBuilder instituteQueryBuilder = getInstituteQueryBuilder(fields);
        instituteQueryBuilder.append(" WHERE t.id IN (:teamId)");
        TypedQuery<Team> query = em.createQuery(instituteQueryBuilder.toString(), Team.class);
        query.setParameter("teamId", Lists.newArrayList(teamId));
        return query.getResultList();
    }

    @Override
    @Nullable
    public Team getInstituteByCode(String teamCode, Team.Lazy... fields) {
        StringBuilder instituteQueryBuilder = getInstituteQueryBuilder(fields)
                .append(" WHERE t.code = :teamCode");
        TypedQuery<Team> query = em.createQuery(instituteQueryBuilder.toString(), Team.class);
        //get a case insensitive match of the code
        query.setParameter("teamCode", teamCode.toLowerCase());
        return getExactlyOneUniqueResultOrNull(query);
    }

    @Override
    public List<Team> getTeamsByCode(Collection<String> teamCodes, Team.Lazy... fields) {
        //get a case insensitive match of the code
        var teamCodeSet = teamCodes.stream().filter(StringUtils::isNotBlank).map(String::toLowerCase).collect(toSet());
        String teamQuery = getInstituteQueryBuilder(fields).append(" WHERE t.code IN (:teamCodes)").toString();
        return DbUtil.getInBatches(teamCodeSet, batch ->
                em.createQuery(teamQuery, Team.class).setParameter("teamCodes", batch).getResultList());
    }

    @Override
    @NotNull
    public Team getInstituteByPublicId(UUID publicId) {
        return em.createQuery("SELECT t FROM Team t JOIN FETCH t.org WHERE t.publicId = :publicId", Team.class)
                .setParameter("publicId", publicId)
                .getSingleResult();
    }

    @Override
    public List<UUID> getTeamPublicIdsByIds(List<Long> teamIds) {
        return em.createQuery("SELECT t.publicId FROM Team t WHERE t.id IN :teamIds", UUID.class)
                .setParameter("teamIds", teamIds)
                .getResultList();
    }
    
    @Override
    public List<Team> getTeamsByPublicIds(Collection<UUID> publicIds) {
        if (publicIds.isEmpty()){
            return Collections.emptyList();
        }
        return em.createQuery("SELECT t FROM Team t JOIN FETCH t.org WHERE t.publicId IN (:publicIds)", Team.class)
                .setParameter("publicIds", publicIds)
                .getResultList();
    }

    @Override
    public UUID getTeamPublicIdByTeamCode(String teamCode) {
        return em.createQuery("SELECT t.publicId FROM Team t WHERE t.code = :teamCode", UUID.class)
                .setParameter("teamCode", teamCode)
                .getSingleResult();
    }

    @Override
    @Transactional
    public void deleteSymptomsToMonitor(Team team) {
        if (team.getSymptomsToMonitor() != null) {
            String hack = "DELETE FROM TeamSymptom WHERE institute_id = :teamId";
            Query query = em.createQuery(hack);
            query.setParameter("teamId", team.getId());
            query.executeUpdate();
        }
    }

    @Override
    public List<Team> getUserInstitutesByStatus(long userId, SponsorshipStatus status) {
        String select = "SELECT iu.institute FROM InstituteUserEntity iu WHERE iu.personId = :userId " +
                "AND iu.sponsorshipStatus = :status";
        TypedQuery<Team> query = em.createQuery(select, Team.class);
        query.setParameter("userId", userId);
        query.setParameter("status", status);
        return query.getResultList();
    }
}
