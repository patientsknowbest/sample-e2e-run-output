package com.pkb.repository.legacy;

import com.pkb.entities.enums.MenuDataType;
import com.pkb.util.DbUtil;
import io.vavr.control.Option;
import org.springframework.beans.factory.annotation.Autowired;

import javax.persistence.EntityManager;
import javax.persistence.NoResultException;
import javax.persistence.Query;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.Collection;
import java.util.Optional;

import static java.util.stream.Collectors.toSet;

public class CustomizedLegacySymptomReportRepositoryImpl implements CustomizedLegacySymptomReportRepository {

    @Autowired
    private EntityManager em;

    @Override
    public void deleteSymptomAlarms(long teamAccountId, Collection<Long> patientIds) {
        DbUtil.executeInBatches(patientIds, batch -> {
            Query query = em.createNativeQuery(
                    "DELETE FROM app.menu_data AS md "
                            + " WHERE md.account_id = :teamAccountId "
                            + " AND md.string01 IN (:patientIds) "
                            + " AND md.datatype = '" + MenuDataType.symptomAlarm.name() + "' ");

            var batchOfIdStrings = batch.stream()
                    .map(l -> l.toString())
                    .collect(toSet());

            query.setParameter("teamAccountId", teamAccountId);
            query.setParameter("patientIds", batchOfIdStrings);
            return query.executeUpdate();
        });
    }

    @Override
    public Optional<LocalDate> getDateOfLatestSymptom(long patientAccountId) {
        LocalDate result = null;
        try {
            Query query = em.createNativeQuery("SELECT MAX(md.date01) "
                    + " FROM app.Menu_Data md"
                    + " WHERE md.account_id = :accountId "
                    + " AND   md.datatype = '" + MenuDataType.symptomReport.name() + "' ");
            query.setParameter("accountId", patientAccountId);
            result = Option.of((java.sql.Timestamp) query.getSingleResult())
                    .map(java.sql.Timestamp::toLocalDateTime)
                    .map(LocalDateTime::toLocalDate)
                    .getOrNull();
        } catch (NoResultException ignored) {
            // ignore: this will just return null.
        } catch (Exception e) {
            throw new RuntimeException("Exception getting date of latest symptom for patient account: " + patientAccountId, e);
        }

        return Optional.ofNullable(result);
    }
}
