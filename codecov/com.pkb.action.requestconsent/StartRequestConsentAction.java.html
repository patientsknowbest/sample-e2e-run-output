public class StartRequestConsentAction extends BaseAction {

    private static final long serialVersionUID = -6096996647435744531L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    // Custom responses
    @VisibleForTesting
    static final String CANCEL = "cancel";
    @VisibleForTesting
    static final String GRANTED_COORD = "grantedCoord";
    @VisibleForTesting
    static final String GRANTED = "granted";

    private UserWebBean userWebBean;
    private UserAccessManager userAccessManager;
    private PKBEmailMessageManager pkbEmailMessageManager;

    private Long patientId;
    private PatientConsent patientConsent;
    private String errorMessage;

    public String view() {
        Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();

        // Prepopulate consent with team defaults
        patientConsent = new PatientConsent();
        patientConsent.setGeneralHealthConsentGranted(team.isDefaultRequestGeneral());
        patientConsent.setSexualHealthConsentGranted(team.isDefaultRequestSexualHealth());
        patientConsent.setMentalHealthConsentGranted(team.isDefaultRequestMentalHealth());
        patientConsent.setSocialCareConsentGranted(team.isDefaultRequestSocialCare());

        return Action.SUCCESS;
    }

    public String cancel() {
        contextUserUtil.clearContextUser();
        return CANCEL;
    }

    public String grantConsent() {
        try {

            EHRRequestContext ehrRequestContext = getEHRRequestContext();
            Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam(ehrRequestContext);

            // Setup reasons - EXPLICIT in this case, because the patient is present
            PatientConsentReason reasons = new PatientConsentReason(new SourceDetails(ehrRequestContext));
            reasons.setAllReasons(ConsentReason.EXPLICIT_CONSENT, "");
            reasons.trackConsentCategories(patientConsent);
            patientConsent.setReasons(reasons);

            // TODO: Detect if clinician is part of a team
            userWebBean.updateOrCreateTeamConsentRecord(ehrRequestContext, patientId, team.getId(), patientConsent);

            PKBPerson patient = userManager.getPKBPerson(patientId, PKBPerson.Lazy.CONTACTS);

            PKBPerson loggedInUser = loggedInUserUtil.getLoggedInUser();

            pkbEmailMessageManager.sendTeamAccessNotificationToPatient(ACCESS_GRANT,
                    ImmutableEmailInfo.builder()
                            .recipient(patient)
                            .sender(loggedInUser)
                            .team(team)
                            .includeSender(true)
                            .build(),
                    ImmutableConsentSettingsDTO.builder()
                            .generalHealthConsentGranted(patientConsent.isGeneralHealthConsentGranted())
                            .sexualHealthConsentGranted(patientConsent.isSexualHealthConsentGranted())
                            .mentalHealthConsentGranted(patientConsent.isMentalHealthConsentGranted())
                            .socialCareConsentGranted(patientConsent.isSocialCareConsentGranted())
                            .proEditOnlyForChild(patientConsent.isProEditOnlyForChild())
                            .discharged(patientConsent.isDischarged())
                            .build());

            if (UserType.INSTITUTE_ADMIN == loggedInUserUtil.getLoggedInUserType()) {
                return GRANTED_COORD;
            }
            return GRANTED;

        } catch (PKBException ex) {
            LOGGER.error("failed to create consent record", ex);
            return Action.ERROR;
        }
    }

    public Long getPatientId() {
        return patientId;
    }

    public void setPatientId(Long patientId) {
        this.patientId = patientId;
    }

    public PatientConsent getPatientConsent() {
        return patientConsent;
    }

    public void setPatientConsent(PatientConsent patientConsent) {
        this.patientConsent = patientConsent;
    }

    public String getErrorMessage() {
        return errorMessage;
    }

    public void setErrorMessage(String errorMessage) {
        this.errorMessage = errorMessage;
    }

    public UserWebBean getUserWebBean() {
        return userWebBean;
    }

    public void setUserWebBean(UserWebBean userWebBean) {
        this.userWebBean = userWebBean;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public UserAccessManager getUserAccessManager() {
        return userAccessManager;
    }

    public void setUserAccessManager(UserAccessManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }

    public PKBEmailMessageManager getPkbEmailMessageManager() {
        return pkbEmailMessageManager;
    }

    public void setPkbEmailMessageManager(PKBEmailMessageManager pkbEmailMessageManager) {
        this.pkbEmailMessageManager = pkbEmailMessageManager;
    }

}
