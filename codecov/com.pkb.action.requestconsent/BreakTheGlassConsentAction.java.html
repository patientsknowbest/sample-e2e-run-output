package com.pkb.action.requestconsent;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.bean.impl.UserWebBean;
import com.pkb.institute.entity.Team;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;

/**
 *
 * @author Russ Gray <russ.gray@patientsknowbest.com>
 */
public class BreakTheGlassConsentAction extends BaseAction {

    private static final long serialVersionUID = 4626846534279861369L;

    private UserWebBean userWebBean;

    private UserAccessManager userAccessManager;
    private PKBEmailMessageManager pkbEmailMessageManager;

    // Custom responses
    private static final String GRANTED = "granted";
    private static final String CANCEL = "cancel";

    private Long patientId;

    public String view() {
        return Action.SUCCESS;
    }

    public String smash() {
        EHRRequestContext ehrRequestContext = getEHRRequestContext();
        if (userWebBean.breakTheGlass(ehrRequestContext, patientId, null)) {

            PKBPerson patient = userManager.getPKBPerson(patientId, PKBPerson.Lazy.CONTACTS);
            Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam(ehrRequestContext);
            PKBPerson loggedInUser = loggedInUserUtil.getLoggedInUser();

            pkbEmailMessageManager.sendBreakTheGlassNotificationToPatient(loggedInUser, patient, team);

            return GRANTED;
        } else {
            return Action.ERROR;
        }
    }

    public String cancel() {
        return CANCEL;
    }

    public Long getPatientId() {
        return patientId;
    }

    public void setPatientId(Long patientId) {
        this.patientId = patientId;
    }

    public void setUserWebBean(UserWebBean userWebBean) {
        this.userWebBean = userWebBean;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public UserAccessManager getUserAccessManager() {
        return userAccessManager;
    }

    public void setUserAccessManager(UserAccessManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }

    public PKBEmailMessageManager getPkbEmailMessageManager() {
        return pkbEmailMessageManager;
    }

    public void setPkbEmailMessageManager(PKBEmailMessageManager pkbEmailMessageManager) {
        this.pkbEmailMessageManager = pkbEmailMessageManager;
    }
}
