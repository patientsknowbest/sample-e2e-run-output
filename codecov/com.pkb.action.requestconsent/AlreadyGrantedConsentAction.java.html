public class AlreadyGrantedConsentAction extends BaseAction {

    private static final long serialVersionUID = 6093555651441361092L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    // Custom responses
    @VisibleForTesting
    static final String GRANTED = "granted";
    @VisibleForTesting
    static final String GRANTED_COORD = "grantedCoord";
    @VisibleForTesting
    static final String CANCEL = "cancel";

    private UserAccessManager userAccessManager;
    private PKBEmailMessageManager pkbEmailMessageManager;

    public static class GrantedConsentReason {

        private String consentReason;
        private String text;

        public GrantedConsentReason(String consentReason, String text) {
            this.consentReason = consentReason;
            this.text = text;
        }

        public String getConsentReason() {
            return consentReason;
        }

        public void setConsentReason(String consentReason) {
            this.consentReason = consentReason;
        }

        public String getText() {
            return text;
        }

        public void setText(String text) {
            this.text = text;
        }

    }

    private UserWebBean userWebBean;

    private Long patientId;
    private String reason;
    private PatientConsent patientConsent;
    private String errorMessage;

    private List<GrantedConsentReason> alreadyGrantedReasons;

    public String view() {
        Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();

        // Prepopulate consent with team defaults
        patientConsent = new PatientConsent();
        patientConsent.setGeneralHealthConsentGranted(team.isDefaultRequestGeneral());
        patientConsent.setSexualHealthConsentGranted(team.isDefaultRequestSexualHealth());
        patientConsent.setMentalHealthConsentGranted(team.isDefaultRequestMentalHealth());
        patientConsent.setSocialCareConsentGranted(team.isDefaultRequestSocialCare());

        return Action.SUCCESS;
    }

    public String cancel() {
        contextUserUtil.clearContextUser();
        return CANCEL;
    }

    public String grantConsent() {
        try {
            if (StringUtils.isBlank(reason)) {
                errorMessage = getText("alreadyGranted.err.reason");
                return Action.SUCCESS;
            }

            // Set up the reasons (will only be saved for categories that changed)
            ConsentReason selectedReason = ConsentReason.valueOf(reason);
            if (patientConsent.getReasons() == null) {
                patientConsent.setReasons(new PatientConsentReason(new SourceDetails(getEHRRequestContext())));
            }
            patientConsent.getReasons().setAllReasons(selectedReason, "");
            patientConsent.getReasons().trackConsentCategories(patientConsent);
            patientConsent.setAdded(dateTimeService.now());

            userManager.transactional(() -> {
                // TODO: Detect if clinician is part of a team
                Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam(getEHRRequestContext());
                userWebBean.updateOrCreateTeamConsentRecord(getEHRRequestContext(), patientId, team.getId(), patientConsent);


                PKBPerson patient = userManager.getPKBPerson(patientId, PKBPerson.Lazy.CONTACTS);
                PKBPerson loggedInUser = loggedInUserUtil.getLoggedInUser();

                pkbEmailMessageManager.sendTeamAccessNotificationToPatient(ACCESS_GRANT,
                        ImmutableEmailInfo.builder()
                                .recipient(patient)
                                .sender(loggedInUser)
                                .team(team)
                                .includeSender(true)
                                .build(),
                        ImmutableConsentSettingsDTO.builder()
                                .generalHealthConsentGranted(patientConsent.isGeneralHealthConsentGranted())
                                .sexualHealthConsentGranted(patientConsent.isSexualHealthConsentGranted())
                                .mentalHealthConsentGranted(patientConsent.isMentalHealthConsentGranted())
                                .socialCareConsentGranted(patientConsent.isSocialCareConsentGranted())
                                .proEditOnlyForChild(patientConsent.isProEditOnlyForChild())
                                .discharged(patientConsent.isDischarged())
                                .build());
            });

            if (UserType.INSTITUTE_ADMIN == loggedInUserUtil.getLoggedInUserType()) {
                return GRANTED_COORD;
            }

            return GRANTED;

        } catch (PKBException ex) {
            LOGGER.error("failed to create consent record", ex);
            return Action.ERROR;
        } finally {
            userManager.rollbackIfNotCommitted();
        }

    }

    @Override
    public void prepare() throws Exception {
        super.prepare();

        alreadyGrantedReasons = Lists.newArrayList(
                new GrantedConsentReason(ConsentReason.EXPLICIT_CONSENT.name(),
                        getText("alreadyGranted.txt.reason.explicit")),
                new GrantedConsentReason(ConsentReason.REFERRAL.name(), getText("alreadyGranted.txt.reason.referral")),
                new GrantedConsentReason(ConsentReason.POA.name(), getText("alreadyGranted.txt.reason.poa")),
