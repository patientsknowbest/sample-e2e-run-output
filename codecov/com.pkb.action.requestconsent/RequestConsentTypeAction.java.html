package com.pkb.action.requestconsent;

import com.google.common.collect.Lists;
import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.service.user.impl.UserAccessManager;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.invoke.MethodHandles;
import java.util.List;
import java.util.function.Function;

public class RequestConsentTypeAction extends BaseAction {

    private static final long serialVersionUID = 2035697273377016124L;
    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    // Custom responses
    private static final String CANCEL = "cancel";

    public enum ConsentTypeRequested {

        START_CONSENT, ALREADY_GRANTED, BREAK_GLASS
    }

    public static class AccessType {

        private String consentTypeRequested;
        private String text;

        public AccessType(ConsentTypeRequested consentTypeRequested, String text) {
            this.consentTypeRequested = consentTypeRequested.name();
            this.text = text;
        }

        public String getConsentTypeRequested() {
            return consentTypeRequested;
        }

        public void setConsentTypeRequested(String consentTypeRequested) {
            this.consentTypeRequested = consentTypeRequested;
        }

        public String getText() {
            return text;
        }

        public void setText(String text) {
            this.text = text;
        }
    }

    private Long patientId;
    private String accessType;
    private List<AccessType> accessTypes;

    private UserAccessManager userAccessManager;

    private List<AccessType> generalAccessTypes;

    private List<AccessType> coordinatorAccessTypes;

    public String view() {
        patientId = contextUserUtil.getContextUserId();
        accessTypes = generalAccessTypes;

        if (getLoggedInEHRRequestContext().isTeamCoordinator()) {
            accessTypes = coordinatorAccessTypes;
        }
        return Action.SUCCESS;
    }

    public String cancel() {
        contextUserUtil.clearContextUser();
        return CANCEL;
    }

    public String select() {
        ConsentTypeRequested consentTypeRequested = ConsentTypeRequested.valueOf(accessType);
        switch (consentTypeRequested) {
        case START_CONSENT:
            return "startRequestConsent";
        case ALREADY_GRANTED:
            return "alreadyGranted";
        case BREAK_GLASS:
            return "breakTheGlass";
        }
        return Action.ERROR;
    }

    public Long getPatientId() {
        return patientId;
    }

    public void setPatientId(Long patientId) {
        this.patientId = patientId;
    }

    public String getAccessType() {
        return accessType;
    }

    public void setAccessType(String accessType) {
        this.accessType = accessType;
    }

    public List<AccessType> getAccessTypes() {
        return accessTypes;
    }

    public void setUserAccessManager(UserAccessManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }

    public Function<AccessType, String> getConsentTypeRequestedGetter() {
        return AccessType::getConsentTypeRequested;
    }

    public Function<AccessType, String> getTextGetter() {
        return AccessType::getText;
    }

    @Override
    public void prepare() throws Exception {
        super.prepare();
        generalAccessTypes = Lists.newArrayList(
                new AccessType(ConsentTypeRequested.START_CONSENT, getText("requestConsentType.txt.type.start_consent")),
                new AccessType(ConsentTypeRequested.ALREADY_GRANTED, getText("requestConsentType.txt.type.already_have_consent")),
                new AccessType(ConsentTypeRequested.BREAK_GLASS, getText("requestConsentType.txt.type.break_the_glass")));

        coordinatorAccessTypes = Lists.newArrayList(
                new AccessType(ConsentTypeRequested.START_CONSENT, getText("requestConsentType.txt.type.start_consent")),
                new AccessType(ConsentTypeRequested.ALREADY_GRANTED, getText("requestConsentType.txt.type.already_have_consent")));
    }

}
