package com.pkb.authz.policy.util;

import com.pkb.authz.JustificationUtils;
import com.pkb.authz.condition.ConditionFailure;
import com.pkb.authz.permission.Justification;
import io.vavr.collection.List;
import io.vavr.collection.Set;
import io.vavr.control.Either;

import java.util.function.BinaryOperator;

import static io.vavr.API.Seq;

/**
 * See {@link PermissionsCollectors}
 */
public final class ConditionCheckResultMerger implements BinaryOperator<Either<Set<ConditionFailure>, List<Justification>>> {
    private static class ConditionCheckResultMergerHolder {
        private static final ConditionCheckResultMerger INSTANCE = new ConditionCheckResultMerger();
    }

    public static BinaryOperator<Either<Set<ConditionFailure>, List<Justification>>> conditionCheckResultMerger() {
        return ConditionCheckResultMergerHolder.INSTANCE;
    }

    private ConditionCheckResultMerger() {
    }

    @Override
    public Either<Set<ConditionFailure>, List<Justification>> apply(
            Either<Set<ConditionFailure>, List<Justification>> result,
            Either<Set<ConditionFailure>, List<Justification>> anotherResult) {

        return Either.sequence(Seq(result, anotherResult))
                .mapLeft(conditionFailureSets -> conditionFailureSets.reduce(Set::addAll))
                .map(justificationLists -> justificationLists.reduce(JustificationUtils::merge));
    }
}
