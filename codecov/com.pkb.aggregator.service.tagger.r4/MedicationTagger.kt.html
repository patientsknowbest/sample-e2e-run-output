package com.pkb.aggregator.service.tagger.r4

import com.pkb.aggregator.R4Element
import com.pkb.aggregator.R4Medication
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

@Service
class MedicationTagger : R4ResourceTagger<R4Medication> {

    override fun getTaggableElements(resource: R4Medication) = buildList<R4Element> {

        if (resource.hasIdentifier()) {
            addAll(resource.identifier)
        }

        if (resource.hasCode()) {
            addAll(resource.code.coding)
        }

        if (resource.hasForm()) {
            addAll(resource.form.coding)
        }

        if (resource.hasAmount()) {
            add(resource.amount)
        }

        if (resource.hasIngredient()) {
            addAll(resource.ingredient)
        }

        if (resource.hasBatch()) {
            add(resource.batch)
        }
    }

    override fun filter(resources: List<IDomainResource>) = resources.mapNotNull { it as? R4Medication }
}
