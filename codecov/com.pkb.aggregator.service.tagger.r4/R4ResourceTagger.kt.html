package com.pkb.aggregator.service.tagger.r4

import com.pkb.aggregator.R4DomainResource
import com.pkb.aggregator.R4Element
import com.pkb.aggregator.extension.findSingleR4UpstreamTagExtension
import com.pkb.aggregator.extension.tagWithSource
import com.pkb.aggregator.service.tagger.ResourceTagger
import org.hl7.fhir.instance.model.api.IDomainResource

interface R4ResourceTagger<T : R4DomainResource> : ResourceTagger {

    override fun tag(resources: List<IDomainResource>) {
        filter(resources)
            .forEach { resource ->
                val upstreamTag = resource.findSingleR4UpstreamTagExtension()

                getTaggableElements(resource).forEach {
                    it.tagWithSource(upstreamTag.copy())
                }
            }
    }

    fun getTaggableElements(resource: T): List<R4Element> = listOf()

    fun filter(resources: List<IDomainResource>): List<T>
}
