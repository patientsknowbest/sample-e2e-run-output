package com.pkb.aggregator.service.tagger.r4

import com.pkb.aggregator.R4Element
import com.pkb.aggregator.R4Practitioner
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

@Service
class PractitionerTagger : R4ResourceTagger<R4Practitioner> {

    override fun getTaggableElements(resource: R4Practitioner) = buildList<R4Element> {

        if (resource.hasIdentifier()) {
            addAll(resource.identifier)
        }

        if (resource.hasName()) {
            addAll(resource.name)
        }

        if (resource.hasTelecom()) {
            addAll(resource.telecom)
        }

        if (resource.hasAddress()) {
            addAll(resource.address)
        }

        if (resource.hasPhoto()) {
            addAll(resource.photo)
        }

        if (resource.hasQualification()) {
            addAll(resource.qualification)
        }

        if (resource.hasCommunication()) {
            addAll(resource.communication)
        }
    }

    override fun filter(resources: List<IDomainResource>) =
        resources.mapNotNull { it as? R4Practitioner }

}
