package com.pkb.aggregator.service.tagger.r4

import com.pkb.aggregator.R4Element
import com.pkb.aggregator.R4Organization
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

@Service
class OrganizationTagger : R4ResourceTagger<R4Organization> {

    override fun getTaggableElements(resource: R4Organization) = buildList<R4Element> {

        if (resource.hasIdentifier()) {
            addAll(resource.identifier)
        }

        if (resource.hasType()) {
            addAll(resource.type)
        }

        if (resource.hasTelecom()) {
            addAll(resource.telecom)
        }

        if (resource.hasAddress()) {
            addAll(resource.address)
        }

        if (resource.hasPartOf()) {
            add(resource.partOf)
        }

        if (resource.hasContact()) {
            addAll(resource.contact)
        }
    }

    override fun filter(resources: List<IDomainResource>) =
        resources.mapNotNull { it as? R4Organization }
}
