package com.pkb.aggregator.service.tagger.r4

import com.pkb.aggregator.R4Element
import com.pkb.aggregator.R4Patient
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

@Service
class PatientTagger : R4ResourceTagger<R4Patient> {

    override fun getTaggableElements(resource: R4Patient): List<R4Element> = buildList {

        if (resource.hasIdentifier()) {
            addAll(resource.identifier)
        }

        if (resource.hasName()) {
            addAll(resource.name)
        }

        if (resource.hasTelecom()) {
            addAll(resource.telecom)
        }

        if (resource.hasAddress()) {
            addAll(resource.address)
        }

        if (resource.hasMaritalStatus()) {
            addAll(resource.maritalStatus.coding)
        }

        if (resource.hasPhoto()) {
            addAll(resource.photo)
        }

        if (resource.hasContact()) {
            addAll(resource.contact)
        }

        if (resource.hasCommunication()) {
            addAll(resource.communication)
        }

        if (resource.hasGeneralPractitioner()) {
            addAll(resource.generalPractitioner)
        }

        if (resource.hasManagingOrganization()) {
            add(resource.managingOrganization)
        }

        if (resource.hasLink()) {
            addAll(resource.link)
        }
    }

    override fun filter(resources: List<IDomainResource>) = resources.mapNotNull { it as? R4Patient }
}
