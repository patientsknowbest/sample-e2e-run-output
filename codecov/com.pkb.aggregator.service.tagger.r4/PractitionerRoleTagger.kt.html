package com.pkb.aggregator.service.tagger.r4

import com.pkb.aggregator.R4Element
import com.pkb.aggregator.R4PractitionerRole
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

@Service
class PractitionerRoleTagger : R4ResourceTagger<R4PractitionerRole> {

    override fun getTaggableElements(resource: R4PractitionerRole) = buildList<R4Element> {

        if (resource.hasIdentifier()) {
            addAll(resource.identifier)
        }

        if (resource.hasPeriod()) {
            add(resource.period)
        }

        if (resource.hasPractitioner()) {
            add(resource.practitioner)
        }

        if (resource.hasOrganization()) {
            add(resource.organization)
        }

        if (resource.hasCode()) {
            addAll(resource.code.flatMap { it.coding })
        }

        if (resource.hasSpecialty()) {
            addAll(resource.specialty.flatMap { it.coding })
        }

        if (resource.hasLocation()) {
            addAll(resource.location)
        }

        if (resource.hasHealthcareService()) {
            addAll(resource.healthcareService)
        }

        if (resource.hasTelecom()) {
            addAll(resource.telecom)
        }

        if (resource.hasAvailableTime()) {
            addAll(resource.availableTime)
        }

        if (resource.hasNotAvailable()) {
            addAll(resource.notAvailable)
        }

        if (resource.hasEndpoint()) {
            addAll(resource.endpoint)
        }
    }

    override fun filter(resources: List<IDomainResource>) = resources.mapNotNull { it as? R4PractitionerRole }
}
