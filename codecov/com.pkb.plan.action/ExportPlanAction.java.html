package com.pkb.plan.action;

import com.opensymphony.xwork2.ActionContext;
import com.pkb.action.BaseAction;
import com.pkb.institute.entity.Team;
import com.pkb.plan.facade.PlanExportViewFacade;
import com.pkb.service.phplan.PHPlanDTO;
import com.pkb.service.phplan.PlanTemplateManager;
import com.pkb.service.phplan.export.ImmutableExportPlanRequestData;

import java.util.ArrayList;
import java.util.List;
import java.util.stream.Collectors;

import static java.util.Collections.emptyList;

public class ExportPlanAction extends BaseAction {

    private Long templateId;
    private String exportType;
    private List<PHPlanDTO> plans;
    private PlanTemplateManager planTemplateManager;
    private PlanExportViewFacade planExportViewFacade;

    @Override
    public String input() {
        Team userPrimaryTeam = loggedInUserUtil.getLoggedInUserPrimaryTeam();
        plans = (userPrimaryTeam != null) ? planTemplateManager.getTemplateList(userPrimaryTeam.getId()).stream()
                .map(PHPlanDTO::from)
                .collect(Collectors.toList())
                : new ArrayList<>();

        return SUCCESS;
    }

    @Override
    public String execute() {
        try {
            planExportViewFacade.exportPlans(ImmutableExportPlanRequestData.builder()
                    .actionContext(ActionContext.getContext())
                    .requestContext(getLoggedInEHRRequestContext())
                    .loggedInUser(loggedInUserUtil.getLoggedInPerson())
                    .templateId(templateId)
                    .teamId(loggedInUserUtil.getLoggedInUserPrimaryTeam().getId())
                    .locale(getLocale())
                    .exportAll("All".equals(exportType))
                    .addAllSelectedClinicians(emptyList())
                    .build());
            addActionMessage(getText("plan.export.initiated"));
        } catch (Exception e) {
            addActionError(getText("err.plan.export.by.clinician"));
            LOGGER.error("Exception while exporting plan for template-{}", templateId, e);
            return INPUT;
        }
        return SUCCESS;
    }

    public String cancel() {
        return CANCEL;
    }

    public PlanExportViewFacade getPlanExportViewFacade() {
        return planExportViewFacade;
    }

    public void setPlanExportViewFacade(PlanExportViewFacade planExportViewFacade) {
        this.planExportViewFacade = planExportViewFacade;
    }

    public Long getTemplateId() {
        return templateId;
    }

    public void setTemplateId(Long templateId) {
        this.templateId = templateId;
    }

    public String getExportType() {
        return exportType;
    }

    public void setExportType(String exportType) {
        this.exportType = exportType;
    }

    public List<PHPlanDTO> getPlans() {
        return plans;
    }

    public void setPlans(List<PHPlanDTO> plans) {
        this.plans = plans;
    }

    public PlanTemplateManager getPlanTemplateManager() {
        return planTemplateManager;
    }

    public void setPlanTemplateManager(PlanTemplateManager planTemplateManager) {
        this.planTemplateManager = planTemplateManager;
    }
}
