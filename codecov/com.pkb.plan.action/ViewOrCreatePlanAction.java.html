package com.pkb.plan.action;

import com.google.common.collect.Lists;
import com.pkb.action.BaseAction;

import com.pkb.service.phplan.PlanManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import org.apache.commons.lang3.StringUtils;

import java.util.List;
import java.util.UUID;

public class ViewOrCreatePlanAction extends BaseAction {

    private static final long serialVersionUID = 1L;

    // /auth/viewPlan.action?planId=98562009&contextUserId=10409

    // edit draft? /auth/managePlans!editPlan.action?selectedPlanUniqueId=85e9e814-547c-4c5e-800f-e2ea3bc761f7&contextUserId=52

    // if none found, make a new one (name: hard-code to "Think Ahead") and open for editing.

    //    <result name="viewExisting" type="redirectAction">viewPlan?planId=%{planId}&amp;contextUserId=%{contextUserId}</result>
    //    <result name="editExistingDraft" type="redirectAction">editPlan?selectedPlanUniqueId=%{selectedPlanUniqueId}&amp;contextUserId=%{contextUserId}</result>
    //    <result name="createNew" type="redirectAction">createPlan?patientId=%{patientId}&amp;planTemplateId=%{planTemplateId}&amp;planName=%{planName}&amp;contextUserId=%{contextUserId}</result>

    // to view existing
    private Long planId;

    // to edit existing
    private UUID selectedPlanUniqueId;

    // to create a plan
    private Long patientId;
    private String planName;
    private Long planTemplateId;

    // private Long contextUserId;  in BaseAction

    private PlanManager planManager;

    @Override
    public String execute() {
        PKBPerson patient = contextUserUtil.getContextUser();
        if (patient == null) {
            patient = userManager.getPKBPerson(loggedInUserUtil.getLoggedInUserId());
        }

        patientId = patient.getId();
        contextUserId = patient.getId();

        List<PHPlan> plans = planManager.getPHPlansFromTemplate(getLoggedInEHRRequestContext(), patient.getId(), planTemplateId, false/*activeOnly*/, patient.getDefaultAccountId());
        if (plans.isEmpty()) {
            // create a new one
            planName = "Think Ahead";
            // already set: planTemplateId, patientId, and contextUserId
            return "createNew";
        } else {
            // latest is first in the list -- see bean method
            PHPlan latestPlan = plans.get(0);
            switch (latestPlan.getStatus()) {
                case ACTIVE:
                    planId = latestPlan.getId();
                    return "viewExisting";
                case INDRAFT:
                    selectedPlanUniqueId = latestPlan.getBaseFields().getUniqueId();
                    return "editExistingDraft";
                default:
                    //dunno!
                    addActionError(getText("err.plan.open.unexpected.status", Lists.newArrayList(latestPlan.getStatus())));
                    return ERROR;
            }
        }
    }

    public Long getPlanId() {
        return planId;
    }

    public void setPlanId(Long planId) {
        this.planId = planId;
    }

    public Long getPatientId() {
        return patientId;
    }

    public void setPatientId(Long patientId) {

    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public String getPlanName() {
        return planName;
    }

    public void setPlanName(String planName) {
        this.planName = planName;
    }

    public void setPlanManager(PlanManager planManager) {
        this.planManager = planManager;
    }

    public Long getPlanTemplateId() {
        return planTemplateId;
    }

    //    public void setPlanTemplateId(Long planTemplateId) {
    //        this.planTemplateId = planTemplateId;
    //    }
    // Workaround struts bug - if the value - which should be irrelevant - is in Arabic the id comes back as a String instead of a Long
    public void setPlanTemplateId(String value) {
        if (!StringUtils.isEmpty(value)) {
            planTemplateId = Long.parseLong(value);
        }
    }

    public UUID getSelectedPlanUniqueId() {
        return selectedPlanUniqueId;
    }

}
