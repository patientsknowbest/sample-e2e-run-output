package com.pkb.plan.action;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.action.BaseAction;
import com.pkb.exception.PKBException;
import com.pkb.phplan.entity.PlanExportDTO;
import com.pkb.service.phplan.PlanExportManager;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.ByteArrayInputStream;
import java.io.InputStream;
import java.util.List;

/**
 * Action class to list , download plan exports
 *
 * @author pravina
 *
 */
public class ManagePlanExportsAction extends BaseAction {

    /**
     *
     */
    private static final long serialVersionUID = 1L;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private PlanExportManager planExportManager;

    private List<PlanExportDTO> planExports;

    private Long planExportId;

    private String contentType;

    private int contentLength;

    private InputStream inputStream;

    private String fileName;

    public String browse() {
        try {
            planExports = planExportManager.getPlanExports(getLoggedInEHRRequestContext(), loggedInUserUtil.getLoggedInUserId());
        } catch (Exception e) {
            LOGGER.error("Error while fetching plan exports", e);
            addActionError(getText("managePlanExportsAction.err.error_fetching_exports"));
        }
        return SUCCESS;
    }

    public String downloadPlanExport() {
        try {
            LOGGER.info("Enter - downloadPlanExport");
            long loggedInUserId = loggedInUserUtil.getLoggedInUserId();
            PlanExportDTO planExportDTO = planExportManager.getPlanExportWithContent(getLoggedInEHRRequestContext(), loggedInUserId,
                    planExportId);
            fileName = getCsvFileName(planExportDTO.getName(),planExportDTO.getMediaType());
            contentType = planExportDTO.getMediaType() + ";charset=utf-8";
            contentLength = planExportDTO.getContent().length;
            inputStream = new ByteArrayInputStream(planExportDTO.getContent());

        } catch (Exception exception) {
            // can't download; let struts return the error page (after logging the user id, browser, etc.)
            throw new PKBException("failed getting plan export for id " + planExportId, exception);
        }
        return "success";
    }

    @VisibleForTesting
    public String getCsvFileName(String baseName,String mediaType) {
        // PHR-7939  - some browsers need ".csv" but older exports do not have it
        if ("text/csv".equals(mediaType) && !baseName.endsWith(".csv")) {
            return baseName + ".csv";
        }
        return baseName;
    }

    public boolean isExportPlanButtonVisible() {
        return !loggedInUserUtil.isLoggedInUserIndividualProfessional();
    }

    public PlanExportManager getPlanExportManager() {
        return planExportManager;
    }

    public void setPlanExportManager(PlanExportManager planExportManager) {
        this.planExportManager = planExportManager;
    }

    public List<PlanExportDTO> getPlanExports() {
        return planExports;
    }

    public void setPlanExports(List<PlanExportDTO> planExports) {
        this.planExports = planExports;
    }

    public Long getPlanExportId() {
        return planExportId;
    }

    public void setPlanExportId(Long planExportId) {
        this.planExportId = planExportId;
    }

    public String getContentType() {
        return contentType;
    }

    public void setContentType(String contentType) {
        this.contentType = contentType;
    }

    public int getContentLength() {
        return contentLength;
    }

    public void setContentLength(int contentLength) {
        this.contentLength = contentLength;
    }

    public InputStream getInputStream() {
        return inputStream;
    }

    public void setInputStream(InputStream inputStream) {
        this.inputStream = inputStream;
    }

    public String getFileName() {
        return fileName;
    }

    public void setFileName(String fileName) {
        this.fileName = fileName;
    }

}
