package com.pkb.phoenix.repository.taskeventlog;

import com.pkb.phoenix.enums.TaskEvent;
import com.pkb.phoenix.tables.daos.TaskEventLogDao;
import org.jooq.CommonTableExpression;
import org.jooq.Configuration;
import org.jooq.Field;
import org.jooq.Name;
import org.jooq.Record;

import java.util.Optional;

import static com.pkb.phoenix.Tables.TASK_EVENT_LOG;
import static org.jooq.impl.DSL.asterisk;
import static org.jooq.impl.DSL.inline;
import static org.jooq.impl.DSL.name;
import static org.jooq.impl.DSL.partitionBy;
import static org.jooq.impl.DSL.rowNumber;
import static org.jooq.impl.DSL.select;
import static org.jooq.impl.DSL.with;

public class TaskEventLogJooqRepository extends TaskEventLogDao implements TaskEventLogRepository {
    private static final Name MOST_RECENT_EVENT = name("most_recent_event");
    private static final Field<Integer> RANK = rowNumber().over(partitionBy(TASK_EVENT_LOG.TASK).orderBy(TASK_EVENT_LOG.AT.desc(), TASK_EVENT_LOG.ID.desc())).as("rank");

    private static final CommonTableExpression<Record> SORTED_EVENTS = name("sorted_events").as(
            select(TASK_EVENT_LOG.asterisk(), RANK)
                            .from(TASK_EVENT_LOG));

    public static final CommonTableExpression<Record> MOST_RECENT_EVENTS = MOST_RECENT_EVENT.as(
            with(SORTED_EVENTS)
                    .select(asterisk())
                    .from(SORTED_EVENTS)
                    .where(inline(1).eq(SORTED_EVENTS.field("rank", Integer.class))));


    private static CommonTableExpression<Record> sortedEventsFor(int taskId) {
        return name("sorted_events").as(
                select(TASK_EVENT_LOG.asterisk(), RANK)
                        .from(TASK_EVENT_LOG)
                        .where(TASK_EVENT_LOG.TASK.eq(inline(taskId))));
    }

    public static CommonTableExpression<Record> latestEventFor(int taskId) {
        var sortedEventsTask = sortedEventsFor(taskId);

        return MOST_RECENT_EVENT.as(
                with(sortedEventsTask)
                        .select(sortedEventsTask.fields())
                        .from(sortedEventsTask)
                        .where(inline(1).eq(sortedEventsTask.field("rank", Integer.class))));
    }

    public TaskEventLogJooqRepository(Configuration configuration) {
        super(configuration);
    }

    @Override
    public Optional<TaskEvent> findLatestEventFor(int taskId) {
        var latestEventFor = latestEventFor(taskId);

        var event = TASK_EVENT_LOG.as(MOST_RECENT_EVENT).EVENT;

        return ctx().with(latestEventFor)
                .select(event)
                .from(latestEventFor)
                .fetchOptional()
                .map(record -> record.get(event));
    }
}
