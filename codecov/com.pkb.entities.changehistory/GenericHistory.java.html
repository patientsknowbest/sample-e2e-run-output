package com.pkb.entities.changehistory;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.SequenceGenerator;
import javax.persistence.Table;
import java.time.Instant;

@Entity
@Table(schema = "change_history", name = "generic_history")
public class GenericHistory {

    @Id
    @SequenceGenerator(name = "change_history.generic_history_generichistoryid_seq", sequenceName = "change_history.generic_history_generichistoryid_seq")
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "change_history.generic_history_generichistoryid_seq")
    @Column(name = "generichistoryid")
    private Long genericHistoryId;

    @Column(name = "schemaname", nullable = false)
    private String schemaName;

    @Column(name = "tablename", nullable = false)
    private String tableName;

    @Column(name = "relid", nullable = false)
    private String relId;

    @Column(name = "actiontstamptx", nullable = false)
    private Instant actionTimestampTx;

    @Column(name = "actiontstampstm", nullable = false)
    private Instant actionTimestampStm;

    @Column(name = "actiontstampclk", nullable = false)
    private Instant actionTimestampClk;

    @Column(name = "transactionid")
    private Long transactionId;

    @Column(name = "clientaddr")
    private String clientAddress;

    @Column(name = "clientport")
    private Integer clientPort;

    @Column(name = "clientquery")
    private String clientQuery;

    @Column(name = "action")
    private char action;

        this.relId = relId;
    }

        return actionTimestampTx;

        this.actionTimestampTx = actionTimestampTx;

    public Instant getActionTimestampStm() {
        return actionTimestampStm;

    public void setActionTimestampStm(Instant actionTimestampStm) {
        this.actionTimestampStm = actionTimestampStm;
