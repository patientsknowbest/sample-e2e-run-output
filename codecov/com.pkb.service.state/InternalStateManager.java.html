package com.pkb.service.state;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import com.pkb.service.encounter.impl.tolven.EncounterManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;

public class InternalStateManager extends TransactionManager {

    private final EncounterManager encounterManager;

    public InternalStateManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, EncounterManager encounterManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.encounterManager = encounterManager;
    }

    public void clearState() {
        ResettablePooledLoOptimizer.clearStateGlobal();
        encounterManager.clearLoincIdState();
    }
}
