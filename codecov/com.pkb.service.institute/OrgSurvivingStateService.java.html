package com.pkb.service.institute;


import com.pkb.VersionDetails;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.entities.enums.OrgExternalIdType;
import com.pkb.institute.entity.OrgSurvivingState;
import com.pkb.repository.OrgSurvivingStateRepository;
import com.pkb.service.uuid.UUIDProvider;

import java.util.Collection;
import java.util.List;
import java.util.Optional;

public class OrgSurvivingStateService {

    private final OrgSurvivingStateRepository repository;

    private final DateTimeService dateTimeService;

    private final UUIDProvider uuidProvider;

    public OrgSurvivingStateService(OrgSurvivingStateRepository repository, DateTimeService dateTimeService, UUIDProvider uuidProvider) {
        this.repository = repository;
        this.dateTimeService = dateTimeService;
        this.uuidProvider = uuidProvider;
    }

    public Optional<OrgSurvivingState> getSurvivorStateByOdsCode(String odsCode) {
        return repository.getByExternalIdValueAndExternalIdTypeAndIsSurvivorTrue(odsCode, OrgExternalIdType.ODS);
    }

    public OrgSurvivingState save(Long orgId, String type, String idValue, Boolean isSurvivor, String description, EHRRequestContext context) {
        OrgSurvivingState entity = new OrgSurvivingState();
        entity.setId(uuidProvider.randomUUID());
        entity.setOrgId(orgId);
        entity.setExternalIdType(OrgExternalIdType.valueOf(type));
        entity.setExternalIdValue(idValue);
        entity.setSurvivor(isSurvivor);
        entity.setDescription(description);
        entity.setVersionDetails(VersionDetails.of(context, dateTimeService.now()));
        return repository.save(entity);
    }

    public List<OrgSurvivingState> getAllSurvivorStateByOdsCodes(Collection<String> odsCodes) {
        return repository.getAllByExternalIdValueInAndExternalIdTypeAndIsSurvivorTrue(odsCodes, OrgExternalIdType.ODS);
    }

}
