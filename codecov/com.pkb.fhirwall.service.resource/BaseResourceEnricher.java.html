package com.pkb.fhirwall.service.resource;

import com.pkb.fhirwall.domain.ResourceRepository;
import org.hl7.fhir.instance.model.api.IBaseResource;

import java.util.List;

/**
 * This doesn't belong in FHIRWALL really but it's here for now. It'll eventually be extracted out to a transformation
 * layer of some kind. Maybe something exciting like Spring Cloud Data Flow, who knows.
 *
 * Sometimes resources returned to callers need to have information present which we don't want to store natively, i.e.
 * denormalised info. Enrich the resource at this point with that information.
 *
 * Future impls may do this on a customer-by-customer basis, but right now only adding universally useful stuff.
 */
public abstract class BaseResourceEnricher {

    ResourceRepository resourceRepository;

    BaseResourceEnricher(ResourceRepository resourceRepository) {
        this.resourceRepository = resourceRepository;
    }

    public abstract void enrichResources(List<IBaseResource> resource);
}
