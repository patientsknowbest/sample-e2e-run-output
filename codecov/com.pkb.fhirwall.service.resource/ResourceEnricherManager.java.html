package com.pkb.fhirwall.service.resource;

import ca.uhn.fhir.context.FhirContext;
import com.google.common.annotations.VisibleForTesting;
import com.pkb.fhirwall.service.util.BundleService;
import org.hl7.fhir.instance.model.api.IBase;
import org.hl7.fhir.instance.model.api.IBaseResource;
import org.hl7.fhir.r4.model.Bundle;
import org.hl7.fhir.r4.model.ResourceType;

import java.util.List;

/**
 * Handles delegation of resource list enrichment to the enricher of the appropriate type
 */
public class ResourceEnricherManager {

    private AppointmentResourceEnricher appointmentResourceEnricher;
    private FhirContext fhirContext;
    private BundleService bundleService;

    public ResourceEnricherManager(AppointmentResourceEnricher appointmentResourceEnricher, FhirContext fhirContext,
                                   BundleService bundleService) {
        this.appointmentResourceEnricher = appointmentResourceEnricher;
        this.fhirContext = fhirContext;
        this.bundleService = bundleService;
    }

    public void enrichResource(IBaseResource resource) {
        if (resource instanceof Bundle) {
            var bundleResources = bundleService.getResourcesFromBundleGroupingByFhirType((Bundle)resource);
            bundleResources.keySet().forEach(fhirType -> enrichResourceList(bundleResources.get(fhirType)));
        }
        //No error here, we can pass through this for e.g. CapabilityStatements
    }

    @VisibleForTesting
    void enrichResourceList(List<IBaseResource> bundleResource) {
        bundleResource.stream().findFirst().map(IBase::fhirType).ifPresent(fhirType -> {
            if (!bundleResource.stream().allMatch(resource -> resource.fhirType().equals(fhirType))) {
                //This is super paranoid and currently impossible.
                throw new RuntimeException("Passed a bundle of resources for fhir type " + fhirType + ", but not all the resources are of this type");
            }
            if (fhirType.equals(ResourceType.Appointment.toString())) {
                appointmentResourceEnricher.enrichResources(bundleResource);
            }
            //No error here, there are loads of resources we won't do anything with
        });
    }
}
