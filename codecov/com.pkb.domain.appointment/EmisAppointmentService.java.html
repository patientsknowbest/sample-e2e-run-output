package com.pkb.domain.appointment;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.crypto.SymmetricKey;
import com.pkb.datamodel.EmisRegistrationDetailsDto;
import com.pkb.datamodel.EmisRegistrationDetailsNewDto;
import com.pkb.datamodel.entity.mapper.EmisRegistrationDetailsMapper;
import com.pkb.domain.AccountSymmetricKeyService;
import com.pkb.entities.embeddables.EncryptedWrapper;
import com.pkb.entities.emis_integration.EmisRegistrationDetails;
import com.pkb.repository.EmisRegistrationDetailsRepository;
import com.pkb.repository.util.EmisContextHolder;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import io.vavr.control.Option;
import org.jetbrains.annotations.Nullable;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.transaction.Transactional;
import java.util.List;
import java.util.stream.Collectors;

public class EmisAppointmentService {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final EmisRegistrationDetailsRepository emisRegistrationDetailsRepository;
    private final AccountSymmetricKeyService accountSymmetricKeyService;
    private final EmisRegistrationDetailsMapper emisRegistrationDetailsMapper;
    private final DateTimeService dateTimeService;

    public EmisAppointmentService(EmisRegistrationDetailsRepository emisRegistrationDetailsRepository,
                                  AccountSymmetricKeyService accountSymmetricKeyService,
                                  EmisRegistrationDetailsMapper emisRegistrationDetailsMapper,
                                  DateTimeService dateTimeService) {
        this.emisRegistrationDetailsRepository = emisRegistrationDetailsRepository;
        this.accountSymmetricKeyService = accountSymmetricKeyService;
        this.emisRegistrationDetailsMapper = emisRegistrationDetailsMapper;
        this.dateTimeService = dateTimeService;
    }

    public List<EmisRegistrationDetailsDto> findEmisRegistrationDetailsByPersonId(EmisContextHolder emisContextHolder) {
        return findEmisRegistrationDetailsByPersonId(emisContextHolder, false);
    }

    List<EmisRegistrationDetailsDto> findDeletedEmisRegistrationDetailsByPersonId(EmisContextHolder emisContextHolder) {
        return findEmisRegistrationDetailsByPersonId(emisContextHolder, true);
    }

    boolean isPersonRegisteredToBookAppointments(EmisContextHolder emisContextHolder) {
        return emisRegistrationDetailsRepository.countOfEmisRegistrationDetailsForPersonId(emisContextHolder.personId(), false) > 0;
    }

    @Transactional
    public EmisRegistrationDetailsDto persistNewRegistrationDetails(EmisContextHolder emisContextHolder, EmisRegistrationDetailsNewDto newDto) {
        emisRegistrationDetailsRepository.logicallyDeleteExistingRegistrationDetailsFor(newDto.getPersonId(), dateTimeService.now());
        EmisRegistrationDetailsDto response = create(emisContextHolder, newDto);
        LOGGER.info("Persisted new registration details for PersonId {}", newDto.getPersonId());
        return response;
    }

    private List<EmisRegistrationDetailsDto> findEmisRegistrationDetailsByPersonId(EmisContextHolder emisContextHolder, boolean deleted) {
        return emisRegistrationDetailsRepository.findEmisRegistrationDetailsByPersonId(emisContextHolder.personId(), deleted).stream()
                .map(emisRegistrationDetails -> emisRegistrationDetailsMapper.entityToDataModel(emisRegistrationDetails, accountSymmetricKeyService.getAccountKeys(emisContextHolder.accountId(), emisContextHolder.actorPersonId())
                        .findSymmetricKeyById(emisRegistrationDetails.getAccessIdentityGuidEncryptedData().getSymmetricEncryptionKeyId())
                        .getOrElseThrow(() -> new RuntimeException("No symmetric key found for id:" + emisRegistrationDetails.getAccessIdentityGuidEncryptedData().getSymmetricEncryptionKeyId()))))
                .collect(Collectors.toList());
    }

    @VisibleForTesting
    EmisRegistrationDetailsDto create(EmisContextHolder emisContextHolder, EmisRegistrationDetailsNewDto newDto) {
        Tuple2<Long, SymmetricKey> symmetricKeyWithId = getSymmetricKey(emisContextHolder, null);

        EmisRegistrationDetails newEntity = emisRegistrationDetailsMapper.newDataModelToNewEntity(newDto, symmetricKeyWithId);
        newEntity = emisRegistrationDetailsRepository.save(newEntity);
        return emisRegistrationDetailsMapper.entityToDataModel(newEntity, symmetricKeyWithId._2);
    }

    private Tuple2<Long, SymmetricKey> getSymmetricKey(EmisContextHolder emisContextHolder, @Nullable Long id) {
        return Option.of(id)
                .map(idValue -> {
                    long symmetricKeyId = emisRegistrationDetailsRepository.findById(idValue)
                            .map(EmisRegistrationDetails::getAccessIdentityGuidEncryptedData)
                            .map(EncryptedWrapper::getSymmetricEncryptionKeyId)
                            .orElseThrow(() -> new RuntimeException("Could not find symmetric key id for EmisRegistrationDetails: " + idValue));
                    return accountSymmetricKeyService.getAccountKeys(emisContextHolder.accountId(), emisContextHolder.actorPersonId()).findSymmetricKeyById(symmetricKeyId)
                            .map(symmetricKey -> Tuple.of(symmetricKeyId, symmetricKey))
                            .getOrElseThrow(() -> new RuntimeException("No symmetric key found for id:" + symmetricKeyId));
                })
                .getOrElse(() -> accountSymmetricKeyService.getAccountKeys(emisContextHolder.accountId(), emisContextHolder.actorPersonId()).getSymmetricKeys()
                        .headOption()
                        .getOrElse(() -> accountSymmetricKeyService.generateSymmetricKey(emisContextHolder.accountId())));
    }
}
