package com.pkb.domain.appointment;

import com.pkb.authz.AuthorizationService;
import com.pkb.datamodel.Appointment;
import com.pkb.domain.SynchUploadedDataService;
import com.pkb.domain.SystemUserTypeNotPermittedException;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.repository.AppointmentQuery;
import com.pkb.domain.repository.AppointmentRepository;
import io.vavr.control.Either;

import static com.pkb.entities.enums.UploadedDataDestination.APPOINTMENT;
import static io.vavr.control.Either.left;

public class AppointmentService {
    private final AppointmentRepository appointmentRepository;
    private final AuthorizationService authorizationService;
    private final SynchUploadedDataService synchUploadedDataService;

    public AppointmentService(AppointmentRepository appointmentRepository,
                              AuthorizationService authorizationService,
                              SynchUploadedDataService synchUploadedDataService) {
        this.appointmentRepository = appointmentRepository;
        this.authorizationService = authorizationService;
        this.synchUploadedDataService = synchUploadedDataService;
    }

    public Either<RuntimeException, CriteriaResult<Appointment>> search(AppointmentQuery appointmentQuery) {
        if (appointmentQuery.authData().inputs().authenticatedSystemUser().isDefined()) {
            // We don't allow system users to read appointments
            return left(new SystemUserTypeNotPermittedException());
        }
        return authorizationService.authorize(appointmentQuery.authData())
                .checkSingleResult(() -> {
                    synchUploadedDataService.synchUploadedData(appointmentQuery.authData(), APPOINTMENT);
                    return appointmentRepository.findAppointments(appointmentQuery);
                });
    }
}
