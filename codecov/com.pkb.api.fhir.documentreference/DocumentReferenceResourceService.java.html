public class DocumentReferenceResourceService extends AbstractResourceService<DocumentReference> {

    private final DocumentService service;
    private final DocumentReferenceMapper mapper;
    private final ProvenanceMapper provenanceMapper;
    private final AuthorizationDataService authorizationDataService;
    private final AuthorizationService authorizationService;
    private final FhirApiProperties fhirConfig;
    private final HybridMailProperties hybridMailProperties;

    public DocumentReferenceResourceService(
            DocumentService service,
            DocumentReferenceMapper mapper,
            ProvenanceMapper provenanceMapper,
            ServerResponseExceptionMapper exceptionMapper,
            AuthorizationDataService authorizationDataService,
            AuthorizationService authorizationService,
            FhirApiProperties fhirConfig,
            HybridMailProperties hybridMailProperties) {
        super(exceptionMapper);
        this.service = service;
        this.mapper = mapper;
        this.provenanceMapper = provenanceMapper;
        this.authorizationDataService = authorizationDataService;
        this.authorizationService = authorizationService;
        this.fhirConfig = fhirConfig;
        this.hybridMailProperties = hybridMailProperties;
    }

    public @NotNull Either<BaseServerResponseException, FhirCriteriaResult<DocumentReference>> search(
            @NotNull DocumentReferenceSearchCriteria criteria,
            @NotNull List<IncludeResource> includeResources) {
        return Try.of(() -> service.findDocuments(mapToDocumentCriteria(criteria)).map(mapToResult(includeResources))
                .getOrElseThrow(identity()))
                .recover(recoverWhenPaginatedResourcesSearched(FhirCriteriaResult::empty))
                .toEither()
                .fold(error -> Match(error).of(
                        Case($(instanceOf(SystemUserTypeNotPermittedException.class)), nae -> left(toSystemUserTypeNotPermitted())),
                        Case($(), left(new InternalErrorException(error)))),
                        identity());
    }

    public @NotNull Either<BaseServerResponseException, List<ReadReceiptDTO>> searchDocumentStatus(AuthenticatedIdentity authenticatedIdentity, @NotNull List<String> externalMessageIds) {
        if (externalMessageIds.size() > hybridMailProperties.getGeneralRequestLimit()) {
            String diagnostics = format("The number of requested read receipts exceeds the maximum number allowed: %d", hybridMailProperties.getGeneralRequestLimit());
            return left(invalidRequestException(Issues.errorFrom(diagnostics, TOOCOSTLY)));
        } else {
            AuthorizationData authorizationData = getAuthorizationData(authenticatedIdentity);
            return authorizationService.authorize(authorizationData)
                    .checkSingleResult(() -> {
                        AuthenticatedSystemUser systemUser = authorizationData.inputs().requireAuthenticatedSystemUser();
                        return service.findReadReceiptsByExternalId(externalMessageIds, systemUser.getOrgId())
                                .getOrElseThrow(identity());
                    }).bimap(InternalErrorException::new, this::mapToReadReceipt);
        }
    }

    private AuthorizationData getAuthorizationData(AuthenticatedIdentity authenticatedIdentity) {
        AuthorizationInputs authInputs = inputs()
                .authenticatedIdentity(authenticatedIdentity)
                .action(SEARCH)
                .addTargets(targetMeta(ReadReceipt.class, Set(), None(), false))
                .path(FHIR)
                .build();
        return authorizationDataService.fetch(authInputs);
    }

    private List<ReadReceiptDTO> mapToReadReceipt(List<ReadReceipt> list) {
        return list
                .map(doc -> ImmutableReadReceiptDTO.builder()
                        .externalDocumentId(doc.getExternalMessageId())
                        .status(doc.getStatus())
                        .unreadableDocument(doc.hasUnreadableDocument()).build());
    }

    private DocumentCriteria mapToDocumentCriteria(DocumentReferenceSearchCriteria criteria) {
        AuthorizationData authzData = authorizationDataService.fetch(
                inputs()
                        .path(FHIR)
                        .authenticatedIdentity(criteria.authenticatedIdentity())
                        .action(SEARCH)
                        .addTargets(targetMeta(DocumentMessage.class, Set(), criteria.patientId(), true))
                        .build());

        return ImmutableDocumentCriteria.builder()
                .authData(authzData)
                .indexed(criteria.indexed())
                .created(criteria.created())
                .documentStatus(criteria.documentStatus().map(DocumentReferenceDocumentStatus::name))
                .sourceOrganisationId(criteria.sourceOrganisationId())
                .capsize(fhirConfig.getDocumentReferenceMaxNumberOfResources())
                .build();
    }

    private Function<List<DocumentMessage>, Either<BaseServerResponseException, FhirCriteriaResult<DocumentReference>>> mapToResult(List<IncludeResource> includeResources) {
        return messages -> {
            List<DocumentReference> documentReferences = mapper.toDocumentReferences(messages);
            List<Provenance> provenances = includeResources.contains(PROVENANCE) ? messages.map(provenanceMapper::toProvenance) : List();
            return right(ImmutableFhirCriteriaResult.<DocumentReference> fhirCriteriaResult(documentReferences.size())
                    .withPage(documentReferences)
                    .withIncludes(List.narrow(provenances)));
        };
    }
}
