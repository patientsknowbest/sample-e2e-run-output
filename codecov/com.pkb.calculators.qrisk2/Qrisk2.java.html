package com.pkb.calculators.qrisk2;

public class Qrisk2 {

    // Removed everything except the enums 2016-06-29

    public enum Ethnicity {

        NOT_RECORDED(0, "Not recorded"), BRITISH(1, "British"), IRISH(2, "Irish"), OTHER_WHITE(3, "Other white"),
        WHITE_AND_BLACK_CARIBBEAN(4, "White and black Caribbean"), WHITE_AND_BLACK_AFRICAN(5, "White and black African"),
        WHITE_AND_ASIAN(6, "White and Asian"), OTHER_MIXED(7, "Other mixed"), INDIAN(8, "Indian"), PAKISTANI(9, "Pakistani"),
        BANGLADESHI(10, "Bangladeshi"), OTHER_ASIAN(11, "Other Asian"), CARIBBEAN(12, "Caribbean"), BLACK_AFRICAN(13, "Black African"),
        OTHER_BLACK(14, "Other black");

        private int code;
        private String text;

        Ethnicity(int code, String text) {
            this.code = code;
            this.text = text;
        }

        public int getCode() {
            return code;
        }

        public String getText() {
            return text;
        }

        public static int getCode(String t) {
            for (Ethnicity e : values()) {
                if (e.text.equals(t)) {
                    return e.code;
                }
            }
            return 0;

        }
    }

}
