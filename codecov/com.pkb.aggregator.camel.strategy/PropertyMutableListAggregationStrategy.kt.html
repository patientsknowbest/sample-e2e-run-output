package com.pkb.aggregator.camel.strategy

import com.pkb.aggregator.extension.asMutableListOfType
import org.apache.camel.AggregationStrategy
import org.apache.camel.Exchange

/**
 * Aggregates two exchanges by iterating through all exchange properties on the first exchange and, if the property
 * value is a list, copying/appending every value in it to the corresponding property on the second exchange
 *
 * This allows two separate routes (fhir requests, in our case) to each collect a list of resources independently, and
 * then have them combined so that the next part of the route sees all the resources in the expected property.
 *
 * Specific properties can be optionally ignored by specifying them in the [ignoredProperties] list.
 *
 * Expected to be used with Camel's
 * [Multicast](https://camel.apache.org/components/3.17.x/eips/multicast-eip.html#_aggregating) component
 */
internal class PropertyMutableListAggregationStrategy(private val ignoredProperties: List<String> = listOf())
    : AggregationStrategy {

    override fun aggregate(oldExchange: Exchange?, newExchange: Exchange): Exchange {

        (oldExchange?.properties ?: emptyMap())
            .forEach { (property, value) ->
                // Only aggregate lists from non-ignored properties
                if (!ignoredProperties.contains(property) && value is MutableList<*>) {
                    val other = newExchange.properties[property]
                    if (other == null) {
                        newExchange.properties[property] = value
                    } else {
                        val accumulated = (other as? MutableList<*>)?.asMutableListOfType<Any>()
                        if (accumulated != null) {
                            value.filterNotNull().forEach { accumulated.add(it) }
                        }
                    }
                }
            }

        return newExchange
    }
}
