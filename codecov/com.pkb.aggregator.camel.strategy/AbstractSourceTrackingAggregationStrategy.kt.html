package com.pkb.aggregator.camel.strategy

import com.pkb.aggregator.configuration.FhirConfiguration
import org.apache.camel.AggregationStrategy
import org.apache.camel.Exchange
import org.apache.http.client.utils.URLEncodedUtils
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import java.lang.invoke.MethodHandles

abstract class AbstractSourceTrackingAggregationStrategy<T>(
    private val fhirConfiguration: FhirConfiguration
) : AggregationStrategy {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun aggregate(oldExchange: Exchange?, newExchange: Exchange): Exchange {
        val body = convertBody(newExchange.message.body)
        val recipientListEndpoint = newExchange.getProperty(Exchange.RECIPIENT_LIST_ENDPOINT).toString()
        val sourceId = findSourceId(recipientListEndpoint)

        val taggedBody = tagBody(body, sourceId)

        return if (oldExchange == null) {
            newExchange.message.body = mutableListOf(taggedBody)
            newExchange
        } else {
            val list = convertBodyList(oldExchange.message.body)
            list.add(taggedBody)
            oldExchange
        }
    }

    abstract fun convertBody(body: Any): T

    abstract fun convertBodyList(body: Any): MutableList<T>

    abstract fun tagBody(body: T, sourceId: String): T

    private fun findSourceId(recipientListEndpoint: String): String {
        val serverUrl = findServerUrl(recipientListEndpoint)
            ?: throw RuntimeException("Could not find serverUrl in receipient list endpoint $recipientListEndpoint")
        return fhirConfiguration.upstream.endpoints.firstOrNull { it.serverUrl == serverUrl }?.sourceId
            ?: throw RuntimeException("Could not find source id for $serverUrl")
    }

    private fun findServerUrl(recipientListEndpoint: String) =
        URLEncodedUtils.parse(recipientListEndpoint, Charsets.UTF_8)
            .find { it.name == "serverUrl" }
            ?.value
}
