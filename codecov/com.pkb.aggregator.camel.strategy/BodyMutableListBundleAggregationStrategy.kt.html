package com.pkb.aggregator.camel.strategy

import com.pkb.aggregator.Dstu3Bundle
import com.pkb.aggregator.R4Bundle
import com.pkb.aggregator.R4DomainResource
import com.pkb.aggregator.configuration.FhirConfiguration
import com.pkb.aggregator.extension.asMutableListOfType
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.extension.tagWithSource
import com.pkb.aggregator.extension.toR4
import com.pkb.aggregator.service.enrichment.createR4SourceExtension
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import org.hl7.fhir.instance.model.api.IBaseBundle
import org.springframework.stereotype.Component

/**
 * Handles aggregating exchanges (from e.g. a recipient list route) where the body is a list of fhir bundles
 */
@Component
class BodyMutableListBundleAggregationStrategy(
    fhirConfiguration: FhirConfiguration,
) : AbstractSourceTrackingAggregationStrategy<IBaseBundle>(fhirConfiguration) {

    override fun convertBody(body: Any): IBaseBundle = when (body) {
        is IBaseBundle -> body
        else -> throw UnexpectedTypeException(body)
    }

    override fun convertBodyList(body: Any): MutableList<IBaseBundle> = when (body) {
        is MutableList<*> -> body.asMutableListOfType()
        else -> throw UnexpectedTypeException(body)
    }

    override fun tagBody(body: IBaseBundle, sourceId: String) = when (body) {
        is Dstu3Bundle -> tagR4Bundle(body.toR4(), sourceId)
        is R4Bundle -> tagR4Bundle(body, sourceId)
        else -> throw UnexpectedTypeException(body)
    }

    private fun tagR4Bundle(bundle: R4Bundle, sourceId: String): R4Bundle {
        // Tag every resource in the bundle by adding an extension with source info
        bundle.entry.forEach {
            val resource = it.resource as? R4DomainResource
                ?: throw UnexpectedTypeException("Cannot set extension on resource type", it.resource)
            val sourceExtension = createR4SourceExtension(sourceId, resource.simpleResourceId())
            resource.tagWithSource(sourceExtension)
        }
        return bundle
    }
}
