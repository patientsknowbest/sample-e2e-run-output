package com.pkb.fhirwall.domain;

import org.hl7.fhir.r4.model.Bundle;

import static java.lang.String.format;

public class ConsentRepository {
    // include a sort to make query performant
    // See https://pkbdev.atlassian.net/browse/PHR-11253
    private static final String CONSENT_BY_PATIENT = "Consent?patient=%s&_sort=-date";

    private final ResourceRepository resourceRepository;

    public ConsentRepository(ResourceRepository resourceRepository) {
        this.resourceRepository = resourceRepository;
    }

    public Bundle getConsentsByPatient(String patientId) {
        return resourceRepository.searchResource(format(CONSENT_BY_PATIENT, patientId));
    }
}
