package com.pkb.fhirwall.domain;

import ca.uhn.fhir.context.FhirContext;
import ca.uhn.fhir.util.BundleUtil;
import org.hl7.fhir.r4.model.Patient;

import java.util.List;
import java.util.Optional;
import java.util.stream.Collectors;

import static java.lang.String.format;

/**
 * This repository for accessing Patient related resources in a FHIR store
 */
public class PatientRepository {

    //Best to include a sort, otherwise the underlying postgres can assume that it should scan the patient table because aidbox includes a LIMIT in the query.
    // See https://pkbdev.atlassian.net/browse/PHR-11253
    private static final String PATIENT_BY_NHS_NUMBER = "Patient?identifier=https://fhir.nhs.uk/Id/nhs-number|%s&_sort=name";
    private static final String PATIENT_BY_IDENTIFIER = "Patient?identifier=%s";

    private final ResourceRepository resourceRepository;
    private final FhirContext fhirContext;

    public PatientRepository(ResourceRepository resourceRepository, FhirContext fhirContext) {
        this.resourceRepository = resourceRepository;
        this.fhirContext = fhirContext;
    }

    public Optional<Patient> getPatientByIdentifier(String identifier) {
        var searchResult = resourceRepository.searchResource(format(PATIENT_BY_IDENTIFIER, identifier));
        if (searchResult.hasEntry()) {
            // Should be only one entry and that's Patient resource
            if (searchResult.getTotal() > 1) {
                throw new RuntimeException("More than one patient found by identifier: " + identifier);
            }
            var maybePatientResource = searchResult.getEntry().get(0).getResource();
            if (maybePatientResource instanceof Patient) {
                return Optional.of((Patient) maybePatientResource);
            }
            throw new RuntimeException("Not Patient resource found by identifier: " + identifier);
        }

        return Optional.empty();
    }

    public List<Patient> findPatientByNhsNumber(String nhsNumber) {
        var patientBundle = resourceRepository.searchResource(format(PATIENT_BY_NHS_NUMBER, nhsNumber));
        return BundleUtil.toListOfResources(fhirContext, patientBundle)
                .stream()
                .map(patient -> (Patient) patient)
                .collect(Collectors.toList());
    }

    public Optional<Patient> getPatientByFhirId(String patientId) {
        return resourceRepository.getResourceByFhirId("Patient/" + patientId, Patient.class);
    }

}
