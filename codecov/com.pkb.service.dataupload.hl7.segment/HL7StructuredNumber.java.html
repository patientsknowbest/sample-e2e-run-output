package com.pkb.service.dataupload.hl7.segment;

import static java.lang.String.format;

import java.util.Arrays;

import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import com.pkb.exception.ApiCallMalformedException;

import io.vavr.Tuple;
import io.vavr.control.Either;
import io.vavr.control.Try;

@Value.Immutable
@Value.Style(strictBuilder = true)
public interface HL7StructuredNumber {

    static @NotNull Either<ApiCallMalformedException, HL7StructuredNumber> parseRawSegments(@NotNull String rawOperator, @NotNull String rawNum) {
        return Arrays.stream(HL7ComparatorOperator.values())
                .filter(op -> op.toString().equals(rawOperator))
                .findAny()
                .map(op -> Tuple.of(op, rawNum))
                .map(tuple -> tuple.apply(HL7StructuredNumber::parseNumber))
                .orElseGet(() -> Either.left(new ApiCallMalformedException(format("failed to parse segment [%s] as structured number operator", rawOperator))));
    }

    static Either<ApiCallMalformedException, HL7StructuredNumber> parseNumber(HL7ComparatorOperator op, String rawNumber) {
        return Try.of(() -> Double.valueOf(rawNumber))
                .<HL7StructuredNumber>map(numberValue -> ImmutableHL7StructuredNumber.builder()
                        .comparator(op)
                        .value(numberValue)
                        .build())
                .toEither(new ApiCallMalformedException(format("failed to parse input [%s] as number", rawNumber)));
    }

    Double value();

    HL7ComparatorOperator comparator();

}
