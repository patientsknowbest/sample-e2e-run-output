package com.pkb.service.dataupload.hl7.segment;

import com.pkb.app.interfaces.DataPointWithStartDate;
import com.pkb.coding.entity.CodeableConcept;
import com.pkb.exception.ApiCallMalformedException;
import com.pkb.service.dataupload.hl7.HL7MessageProperties;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.field.HL7Code;
import com.pkb.service.dataupload.hl7.field.HL7PersonRef;
import com.pkb.service.dataupload.hl7.value.HL7Date;
import com.pkb.service.dataupload.hl7.value.HL7String;
import org.jetbrains.annotations.Nullable;
import org.w3c.dom.Element;

import java.io.Serializable;
import java.time.Instant;

/**
 * Parses a DG1 segment. See HL7 v2.4 Chapter 6.
 *
 * @author robwhelan
 */
public class HL7Diagnosis implements DataPointWithStartDate, Serializable {

    private static final long serialVersionUID = 1L;

    private HL7Code diagnosisCode = new HL7Code();
    private HL7Date diagnosisTimestamp = new HL7Date();
    private HL7PersonRef diagnosingClinician = new HL7PersonRef();

    private HL7MessageProperties messageProperties;

    public HL7Diagnosis(HL7MessageProperties messageProperties) {
        this.messageProperties = messageProperties;
    }

    @Override
    public CodeableConcept code() {
        return diagnosisCode.asCodeableConcept();
    }

    @Nullable
    @Override
    public Instant startDate() {
        return diagnosisTimestamp.getInstant();
    }

    public void parseDg1(Element dg1) throws ApiCallMalformedException {
        String setId = HL7XmlDoc.getHl7Text(dg1, "DG1.1.1").getString();
        String trace = "DG1[" + setId + "]";

        diagnosisCode.parseCe(dg1, "DG1", 3);
        HL7String dgTsText = HL7XmlDoc.getHl7Text(dg1, "DG1.5.1");
        diagnosisTimestamp = HL7XmlDoc.parseDateOrTimestamp(dgTsText, trace + ".5.1", messageProperties);

        diagnosingClinician.parseXcn(dg1, "DG1", 16);
    }

    public HL7Code getDiagnosisCode() {
        return diagnosisCode;
    }

    public void setDiagnosisCode(HL7Code diagnosisCode) {
        this.diagnosisCode = diagnosisCode;
    }

    public HL7Date getDiagnosisTimestamp() {
        return diagnosisTimestamp;
    }

    public void setDiagnosisTimestamp(HL7Date diagnosisTimestamp) {
        this.diagnosisTimestamp = diagnosisTimestamp;
    }

    public HL7PersonRef getDiagnosingClinician() {
        return diagnosingClinician;
    }

    public void setDiagnosingClinician(HL7PersonRef diagnosingClinician) {
        this.diagnosingClinician = diagnosingClinician;
    }
}
