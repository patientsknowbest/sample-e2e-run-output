/**
 * Parses an OBR segment. See HL7 v2.4 Chapter 7.
 * <p>
 * Info from the ORC (optional), OBR (required), plus lists of the OrderGroup-level NTEs (optional) and OBXs
 * (plus each OBX may have its own NTE list).
 * <p>
 * YAGNI #1: ORC and OBR fields are NOT stored as separate objects for now, in the interest of simplicity.
 * We can easily migrate to a deeper object graph when/if needed.
 *
 * @author robwhelan
 */
public class HL7OrderGroup implements Serializable {

    private static final Logger LOGGER = LoggerFactory.getLogger(HL7OrderGroup.class);

    private static final long serialVersionUID = 1L;

    private static final ImmutableSet<String> OBX_STATUS_ACCEPT_VALUES = ImmutableSet.of("F", "C");
    private static final ImmutableSet<String> OBX_STATUS_SKIP_VALUES = ImmutableSet.of("I", "O", "P", "X");

    private HL7String orcFillerOrderNumber = new HL7String();
    private HL7PersonLocation entererLocation = new HL7PersonLocation();

    private HL7String obrSetId = new HL7String();
    private HL7Date obrTimestamp = new HL7Date();
    private HL7Date obrSpecimenReceivedTimestamp = new HL7Date();
    private HL7String obrFillerOrderNumber = new HL7String();
    private HL7Code obrServiceCode = new HL7Code();
    private HL7PersonRef reqBy = new HL7PersonRef();
    private HL7String obrLabDiscipline = new HL7String();
    private HL7String obrResultStatus = new HL7String();

    private List<HL7String> obrNoteList = new ArrayList<>();
    private List<HL7Observation> observationList = new ArrayList<>();

    private HL7MessageProperties messageProperties;

    /**
     * some ORU^R01 messages may contain a mix of data types and actions (deletion vs. save) - this allows a single parsing run
     * then different methods can review not-yet-processed groups
     */
    private boolean processed = false;

    public HL7OrderGroup(HL7MessageProperties messageProperties) {
        this.messageProperties = messageProperties;
    }

    public void parseOrc(Element orc) {
        this.orcFillerOrderNumber = HL7XmlDoc.getHl7Text(orc, "ORC.3.1");
        this.entererLocation.parsePl(orc, "ORC", 13);
    }

    public void parseObr(Element obr) throws ApiCallMalformedException {
        obrSetId = HL7XmlDoc.getHl7Text(obr, "OBR.1.1");

        // may need to get observation timestamp from OBR record (not always in each OBX...)
        obrTimestamp = HL7XmlDoc.parseDateOrTimestamp(HL7XmlDoc.getHl7Text(obr, "OBR.7.1"), "OBR[" + obrSetId.getString() + "].7.1",
                messageProperties);
        obrSpecimenReceivedTimestamp = HL7XmlDoc.parseDateOrTimestamp(HL7XmlDoc.getHl7Text(obr, "OBR.14.1"), "OBR.14.1", messageProperties);
        obrFillerOrderNumber = HL7XmlDoc.getHl7Text(obr, "OBR.3.1");
        obrServiceCode.parseCe(obr, "OBR", 4);
        reqBy.parseXcn(obr, "OBR", 16);
        obrLabDiscipline = HL7XmlDoc.getHl7Text(obr, "OBR.24.1");
        obrResultStatus = HL7XmlDoc.getHl7Text(obr, "OBR.25.1");
    }

    public void parseNte(Element nte) {
        if (observationList.isEmpty()) {
            obrNoteList.addAll(HL7XmlDoc.getHl7TextList(nte, "NTE.3.1", true/*unescape*/, false/*trim*/));
        } else {
            HL7Observation latestObservation = observationList.get(observationList.size() - 1);
            latestObservation.parseNte(nte);
        }
    }

    public void parseObx(Element obx) throws ApiCallMalformedException {
        new HL7ObservationParser().parse(messageProperties, obx, obrSetId.getString())
                .ifPresent(observation -> {
                    if (validateObx(observation)) {
                        observationList.add(observation);
                    } else {
                        // Silently skip this OBX segment
                    }
                });
    }

    /**
     * Returns true if the observation should be accepted, or false if it should be silently ignored.
     * If the message should be rejected then an exception is thrown.
     *
     * @param observation The observation to inspect
     * @return true - if the observation should be accepted; false - otherwise
     */
    public static boolean validateObx(HL7Observation observation) {
        return validateObxValueType(observation.getValueType())
                && validateObxStatus(observation.getResultStatus());
    }

    /**
     * Returns true if the value type has been provided.
     * <p>
     * A blank value will cause an exception to be thrown.
     *
     * @param value The OBX value type from the message
     * @return true - if the value type is not blank. false cannot be returned - an exception will be thrown if the response is not true.
     */
    public static boolean validateObxValueType(HL7String value) {
        enforceMandatoryValue(HL7ValueExpectation.builder()
                .withId("OBX-2.1")
                .withName("Value type")
                .withProvidedHL7Value(value)
                .build());
        return true;
    }

    /**
     * This method returns true if the status value provided is one that should be processed, and false otherwise.
     * <p>
     * A blank or unsupported status value will cause an exception to be thrown.
     *
     * @param value The OBX status value from the message
     * @return true - if the value matches one that should be processed. false - otherwise.
     */
    public static boolean validateObxStatus(HL7String value) {
        // PHR-7692: Check a value has been provided in OBX-11
        HL7ValueExpectation expectation_OBX_11_1 = HL7ValueExpectation.builder()
                .withId("OBX-11.1")
                .withName("result status")
                .withProvidedHL7Value(value)
                .build();
        enforceMandatoryValue(expectation_OBX_11_1);

        // If an acceptable value has been provided, return true
        if (OBX_STATUS_ACCEPT_VALUES.contains(value.getString())) {
            return true;
        }

        // If the value is valid but one we ignore, return false
        if (OBX_STATUS_SKIP_VALUES.contains(value.getString())) {
            return false;
        }

        // Else throw an exception explaining the value is not permitted
        throw new ApiCallMalformedException(String.format("OBX-11 value of [%s] is not valid. %s will be accepted; %s will be silently ignored; other values will be rejected.",
                value.getRawValue(), OBX_STATUS_ACCEPT_VALUES.toString(), OBX_STATUS_SKIP_VALUES.toString()));
    }

    /**
     * If the content of this OBR group meets the criteria for a textual report, then:
     * <ul>
     *     <li>The combined content will be set into the first OBX note list (not the value)</li>
     *     <li>All other OBX segments to be discarded</li>
     * </ul>
     *
     * @return true - if this group was parsed as a textual report. false - otherwise.
     */
    public boolean collapseTextReports() {
        List<HL7String> textualReportEntries = getTextualReportEntries(observationList);
        if(!textualReportEntries.isEmpty()) {
            HL7Observation obx1 = observationList.get(0);
            obx1.setValueText("");
            obx1.setValueNumeric(null);
            obx1.clearNotes();
            for (HL7String textualReportEntry : textualReportEntries) {
                obx1.addNote(textualReportEntry.getString());
            }
            observationList = observationList.subList(0, 1);
            return true;
        }
        return false;
    }

    /**
     * An OBR group is considered to be a textual report if and only if the following conditions are true:
     * <ul>
     *     <li>Condition #1: There are 1 or more OBX segments, which together contain at least 2 lines of text</li>
     *     <li>Condition #2: Each OBX segment declares a textual data type (TX, FT, ST)</li>
     *     <li>Condition #3: All the OBX segments have matching test IDs (OBX-3)</li>
     * </ul>
     * <p>
     * We publish these criteria online at https://dev.patientsknowbest.com/home/hl7-api/messages/oru-r01/laboratory.
     *
     * @return An ordered list of entries for the textual report - if this group was parsed as a textual report. An empty list - otherwise.
     */
    private static List<HL7String> getTextualReportEntries(List<HL7Observation> observations) {
        // Condition #1. Given that...
        // - Each OBX segment is automatically considered a new line
        // - The value of a single OBX can contain more than one line, either by repeating the content field or by
        // including newline escape sequences (\.br\), which is already taken care of by earlier parsing (see
        // HL7XmlDoc.joinTextContentOfTags)
        // ...it follows that condition #1 is satisfied unless there are either:
        // - No OBX segments
        // - Or a single OBX segment that does not contain a newline character
        // As such, we check for both those conditions here and return early if either fail.
        if (observations.isEmpty()) {
            return Collections.emptyList();
        }
        if (observations.size() == 1) {
            HL7String valueText = observations.get(0).getValueText();
            if (valueText == null || valueText.getString().indexOf('\n') < 0) {
                return Collections.emptyList();
            }
        }

        // Keep track of the report contents at the same time as checking the criteria for a textual report. This avoids
        // needing to iterate for a second time to extract the contents if the criteria are met.
        List<HL7String> textualReportEntryList = new ArrayList<>();

        HL7Observation obx1 = observations.get(0);
        for (HL7Observation observation : observations) {
            // Condition #2. This is not a textual report if any OBX declares a non-textual value type.
            if (!observation.isTextValueType()) {
                return Collections.emptyList();
            }
            // Condition #3. This is not a textual report if any OBX has an ID which differs from the first OBX.
            if (!observation.isTestIdMatch(obx1)) {
                return Collections.emptyList();
            }
            // Add the content and notes to the report contents.
            textualReportEntryList.add(observation.getValueText());
            textualReportEntryList.addAll(observation.getNoteList());
        }

        return textualReportEntryList;
    }

    /**
     * If OBR.25.1 is 'R' this is a deletion
     */
    public boolean isDeletionStatus() {
        return "R".equals(getObrResultStatus().toString());
    }

    /**
     * set IDs aren't always populated, but are important for ordering results and or helping with hibernate impl
     */
    public void forceObxSetIds() {
        if (observationList.isEmpty()) {
            return;
        }

        int seqNumber = 1;

        for (HL7Observation obs : observationList) {
            if (StringUtils.isNotBlank(obs.getObxSetId().getString()) && StringUtils.isNumeric(obs.getObxSetId().getString())) {
                seqNumber = Integer.parseInt(obs.getObxSetId().getString());
            } else {
                seqNumber += 1;
                obs.setObxSetId("" + seqNumber);
            }
        }
    }

    /**
     * Our spec declares that we prioritise OBR-3, but if blank we accept ORC-3.
     *
     * @return OBR-3 if not blank, else ORC-3.
     */
    public HL7String getMostSignificantFON() {
        if (obrFillerOrderNumber.isNotBlank()) {
            return obrFillerOrderNumber;
        }
        return orcFillerOrderNumber;
    }
