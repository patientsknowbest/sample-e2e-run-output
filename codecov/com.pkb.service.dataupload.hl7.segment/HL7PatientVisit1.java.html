    private HL7Date dischargeDateTime;
    
    
    public HL7PatientVisit1(HL7MessageProperties messageProperties, Element pv1) throws ApiCallMalformedException {
        this.messageProperties = messageProperties;
        this.parsePV1(pv1);
    }
    
    private void parsePV1(Element pv1) throws ApiCallMalformedException {
        // Parse PV1 Segment data
        this.patientClass = HL7XmlDoc.getHl7Text(pv1, "PV1.2.1");
        this.assignedPatientLocation.parsePl(pv1, "PV1", 3);
        this.priorPatientLocation.parsePl(pv1, "PV1", 6);
        this.attendingDoctor.parseXcn(pv1, "PV1", 7);
        this.referringDoctor.parseXcn(pv1, "PV1", 8);
        this.consultingDoctor.parseXcn(pv1, "PV1", 9);
        this.hospitalService = HL7XmlDoc.getHl7Text(pv1, "PV1.10.1");
        this.visitNumber = HL7XmlDoc.getHl7Text(pv1, "PV1.19.1");
        this.admitDateTime = HL7XmlDoc.parseDateOrTimestamp(HL7XmlDoc.getHl7Text(pv1, "PV1.44.1"), "PV1.44.1", messageProperties);  // messageProperties should contain timezone offset
        this.dischargeDateTime = HL7XmlDoc.parseDateOrTimestamp(HL7XmlDoc.getHl7Text(pv1, "PV1.45.1"), "PV1.45.1", messageProperties);  // messageProperties should contain timezone offset
    }
