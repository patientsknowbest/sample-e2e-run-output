package com.pkb.service.dataupload.hl7.segment;

import static com.pkb.service.dataupload.hl7.HL7XmlDoc.getHl7Text;
import static com.pkb.service.dataupload.hl7.HL7XmlDoc.getIdentifier;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;
import java.util.stream.Collectors;

import org.w3c.dom.Element;
import org.w3c.dom.NodeList;

import com.pkb.service.dataupload.hl7.field.HL7PersonRef;
import com.pkb.service.dataupload.hl7.value.HL7String;

/**
 * Parses an AIP segment. See HL7 v2.4 Chapter 10.
 */
public class HL7AppointmentInformationPersonnel implements Serializable {

    private List<HL7PersonRef> guests = new ArrayList<>();
    private HL7String role = new HL7String();

    public void parseAip(Element element, String segment) {
        parseAip3(element, segment);
        role = getHl7Text(element, getIdentifier(segment, 4, 2));
    }

    private void parseAip3(Element element, String segment) {
        NodeList participants = element.getElementsByTagName(segment + ".3");
        int length = participants.getLength();

        for (int index = 0; index < length; index++) {
            HL7PersonRef person = new HL7PersonRef();
            person.parseXcn((Element) participants.item(index), segment, 3);
            guests.add(person);
        }
    }

    public List<String> getParticipants() {
        return guests.stream()
                .filter(HL7PersonRef::isValid)
                .map(guest -> {
                    String displayText = guest.makeDisplayText();
                    if (role.isNotBlank()) {
                        displayText += " (" + role.getString() + ")";
                    }
                    return displayText;
                }).collect(Collectors.toList());
    }
}
