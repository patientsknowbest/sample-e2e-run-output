package com.pkb.service.dataupload.hl7.segment;

import com.pkb.exception.ApiCallMalformedException;
import com.pkb.service.dataupload.hl7.HL7MessageProperties;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.value.HL7Date;
import com.pkb.service.dataupload.hl7.value.HL7String;
import org.w3c.dom.Element;

import java.io.Serializable;

import static org.apache.commons.lang3.StringUtils.isNotEmpty;

/**
 * Parses a QRF segment. See HL7 v2.4 Chapter 5.
 *
 * @author robwhelan
 */
public class HL7QueryFilter implements Serializable {

    private static final long serialVersionUID = 1L;

    private HL7Date startTimestamp = new HL7Date();
    private HL7Date endTimestamp = new HL7Date();
    private HL7String lastSeenId = new HL7String();

    private HL7MessageProperties messageProperties;

    public HL7QueryFilter(HL7MessageProperties messageProperties) {
        this.messageProperties = messageProperties;
    }

    public void parseQrf(Element qrf) throws ApiCallMalformedException {

        lastSeenId = HL7XmlDoc.getHl7Text(qrf, "QRF.5.1");
        HL7String startTimeStr = HL7XmlDoc.getHl7Text(qrf, "QRF.2.1");

        if (lastSeenId != null && isNotEmpty(lastSeenId.getString()) && !"null".equalsIgnoreCase(lastSeenId.getString())) {
            if (startTimeStr != null  && !"null".equalsIgnoreCase(startTimeStr.getString())) {
                startTimestamp = HL7XmlDoc.parseDateOrTimestamp(startTimeStr, "QRF.2.1", messageProperties);
            }
        } else {
            if (startTimeStr != null  && isNotEmpty(startTimeStr.getString()) && !"null".equalsIgnoreCase(startTimeStr.getString())) {
                startTimestamp = HL7XmlDoc.parseDateOrTimestamp(startTimeStr, "QRF.2.1", messageProperties);
            }
        }
        HL7String endTimeStr = HL7XmlDoc.getHl7Text(qrf, "QRF.3.1");
        if (endTimeStr != null && isNotEmpty(endTimeStr.getString())
                && !"null".equalsIgnoreCase(endTimeStr.getString())) {
            endTimestamp = HL7XmlDoc.parseDateOrTimestamp(endTimeStr, "QRF.3.1", messageProperties);
        }
    }

    /** convert back to pipe/hat format! */
    public String toHl7() {

        StringBuilder sb = new StringBuilder("QRF||")
                .append(startTimestamp.getRawValue()).append("|")
                .append(endTimestamp.getRawValue()).append("|");
        if (lastSeenId.isNotBlank()) {
            sb.append("|").append(lastSeenId);
        }
        return sb.toString();
    }

    public HL7Date getStartTimestamp() {
        return startTimestamp;
    }

    public void setStartTimestamp(HL7Date startTimestamp) {
        this.startTimestamp = startTimestamp;
    }

    public HL7Date getEndTimestamp() {
        return endTimestamp;
    }

    public void setEndTimestamp(HL7Date endTimestamp) {
        this.endTimestamp = endTimestamp;
    }

    public HL7String getLastSeenId() { return lastSeenId; }

    public void setLastSeenId(HL7String lastSeenId) { this.lastSeenId = lastSeenId; }
}
