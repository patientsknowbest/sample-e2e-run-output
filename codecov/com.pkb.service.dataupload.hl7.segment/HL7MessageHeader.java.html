package com.pkb.service.dataupload.hl7.segment;

import com.pkb.exception.ApiCallMalformedException;
import com.pkb.service.dataupload.hl7.HL7MessageProperties;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.value.HL7Date;
import com.pkb.service.dataupload.hl7.value.HL7String;
import org.w3c.dom.Element;

import java.io.Serializable;

/**
 * Parses HL7 MSH Segment
 * 
 * MSH-1: Field Separator (ST)
 * MSH-2: Encoding Characters (ST)
 * MSH-3: Sending Application (HD) optional
 * MSH-4: Sending Facility (HD) optional
 * MSH-5: Receiving Application (HD) optional
 * MSH-6: Receiving Facility (HD) optional
 * MSH-7: Date/Time Of Message (TS) optional
 * MSH-8: Security (ST) optional
 * MSH-9: Message Type (MSG)
 * MSH-10: Message Control ID (ST)
 * MSH-11: Processing ID (PT)
 * MSH-12: Version ID (VID)
 * MSH-13: Sequence Number (NM) optional
 * MSH-14: Continuation Pointer (ST) optional
 * MSH-15: Accept Acknowledgment Type (ID) optional
 * MSH-16: Application Acknowledgment Type (ID) optional
 * MSH-17: Country Code (ID) optional
 * MSH-18: Character Set (ID) optional repeating
 * MSH-19: Principal Language Of Message (CE) optional
 * MSH-20: Alternate Character Set Handling Scheme (ID) optional
 *
 */
public class HL7MessageHeader implements Serializable {

    private HL7MessageProperties messageProperties;

    private HL7String messageType;
    private HL7Date dateTimeOfMessage;
    private HL7String messageControlId;
    private HL7String processingId;
    private HL7String responsibleOrganisationIdentifier;

    public HL7MessageHeader() {
    }

    public HL7MessageHeader(HL7MessageProperties messageProperties, Element msh) throws ApiCallMalformedException {
        this.messageProperties = messageProperties;
        this.parseMSH(msh);
    }

    public void parseMSH(Element msh) throws ApiCallMalformedException {
        // Parse MSH Segment
        this.messageType = new HL7String(HL7XmlDoc.getHl7Text(msh, "MSH.9.1").getString() + "^" + HL7XmlDoc.getHl7Text(msh, "MSH.9.2").getString()); // just want these guys for now, joined
        this.dateTimeOfMessage = HL7XmlDoc.parseDateOrTimestamp(HL7XmlDoc.getHl7Text(msh, "MSH.7.1"), "MSH.7.1", messageProperties);
        this.messageControlId = HL7XmlDoc.getHl7Text(msh, "MSH.10.1");
        this.processingId = HL7XmlDoc.getHl7Text(msh, "MSH.11.1");
        this.responsibleOrganisationIdentifier = HL7XmlDoc.getHl7Text(msh,"MSH.22.10");
    }

    public HL7String getMessageType() {
        return messageType;
    }

    public void setMessageType(String messageType) {
        this.messageType = new HL7String(messageType);
    }

    public HL7Date getDateTimeOfMessage() {
        return dateTimeOfMessage;
    }

    public void setDateTimeOfMessage(HL7Date dateTimeOfMessage) {
        this.dateTimeOfMessage = dateTimeOfMessage;
    }

    public HL7String getMessageControlId() {
        return messageControlId;
    }

    public void setMessageControlId(String messageControlId) {
        this.messageControlId = new HL7String(messageControlId);
    }

    public HL7String getProcessingId() {
        return processingId;
    }

    public void setProcessingId(String processingId) {
        this.processingId = new HL7String(processingId);
    }

    public HL7String getResponsibleOrganisationIdentifier() { return responsibleOrganisationIdentifier; }

    public void setResponsibleOrganisationIdentifier(HL7String responsibleOrganisationIdentifier) { this.responsibleOrganisationIdentifier = responsibleOrganisationIdentifier; }
}
