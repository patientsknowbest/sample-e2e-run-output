package com.pkb.service.dataupload.hl7.segment;

import org.w3c.dom.Element;

import com.pkb.exception.ApiCallMalformedException;
import com.pkb.service.dataupload.hl7.HL7MessageProperties;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.field.HL7Code;
import com.pkb.service.dataupload.hl7.value.HL7Date;
import com.pkb.service.dataupload.hl7.value.HL7String;

/**
 * Parses an SCH segment. See HL7 v2.4 Chapter 10.
 */
public class HL7Appointment {

    private HL7String placerId = new HL7String();
    private HL7String appointmentReasonText = new HL7String();
    private HL7Code appointmentTypeCode = new HL7Code();
    private HL7Date startTimestamp = new HL7Date();
    private HL7Date endTimestamp = new HL7Date();

    private HL7MessageProperties messageProperties;

    public HL7Appointment(HL7MessageProperties messageProperties) {
        this.messageProperties = messageProperties;
    }

    public void parseSch(Element element) throws ApiCallMalformedException {
        placerId = HL7XmlDoc.getHl7Text(element, "SCH.1.1");
        appointmentReasonText = HL7XmlDoc.getHl7Text(element, "SCH.7.2");
        appointmentTypeCode.parseCe(element, "SCH", 8);
        startTimestamp = HL7XmlDoc.parseDateOrTimestamp(HL7XmlDoc.getHl7Text(element, "SCH.11.4"), "SCH.11.4", messageProperties);
        endTimestamp = HL7XmlDoc.parseDateOrTimestamp(HL7XmlDoc.getHl7Text(element, "SCH.11.5"), "SCH.11.5", messageProperties);
    }

    public HL7String getPlacerId() {
        return placerId;
    }

    public void setPlacerId(String placerId) {
        this.placerId = new HL7String(placerId);
    }

    public HL7String getAppointmentReasonText() {
        return appointmentReasonText;
    }

    public void setAppointmentReasonText(String appointmentReasonText) {
        this.appointmentReasonText = new HL7String(appointmentReasonText);
    }

    public HL7Code getAppointmentTypeCode() {
        return appointmentTypeCode;
    }

    public void setAppointmentTypeCode(HL7Code appointmentTypeCode) {
        this.appointmentTypeCode = appointmentTypeCode;
    }

    public HL7Date getStartTimestamp() {
        return startTimestamp;
    }

    public void setStartTimestamp(HL7Date startTimestamp) {
        this.startTimestamp = startTimestamp;
    }

    public HL7Date getEndTimestamp() {
        return endTimestamp;
    }

    public void setEndTimestamp(HL7Date endTimestamp) {
        this.endTimestamp = endTimestamp;
    }
}
