package com.pkb.service.dataupload.hl7.segment;

import com.pkb.app.interfaces.DataPointWithStartDate;
import com.pkb.coding.entity.CodeableConcept;
import com.pkb.exception.ApiCallMalformedException;
import com.pkb.service.dataupload.hl7.HL7MessageProperties;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.field.HL7Code;
import com.pkb.service.dataupload.hl7.field.HL7PersonRef;
import com.pkb.service.dataupload.hl7.value.HL7Date;
import com.pkb.service.dataupload.hl7.value.HL7String;
import org.jetbrains.annotations.Nullable;
import org.w3c.dom.Element;

import java.io.Serializable;
import java.time.Instant;
import java.util.ArrayList;
import java.util.List;

/**
 * Parses an AL1 segment. See HL7 v2.4 Chapter 3.
 *
 * @author robwhelan
 */
public class HL7Allergy implements DataPointWithStartDate, Serializable {

    private static final long serialVersionUID = 1L;

    private HL7Code allergenCode = new HL7Code();
    private HL7Code allergenSeverityCode = new HL7Code();

    private List<HL7String> allergyReactionCodeList = new ArrayList<>();
    private HL7Date identificationDate = new HL7Date();

    // this is included in a NTE
    private HL7PersonRef enteredBy = new HL7PersonRef();

    private HL7MessageProperties messageProperties;

    public HL7Allergy(HL7MessageProperties messageProperties) {
        this.messageProperties = messageProperties;
    }

    @Override
    public CodeableConcept code() {
        return allergenCode.asCodeableConcept();
    }

    @Nullable
    @Override
    public Instant startDate() {
        return identificationDate.getInstant();
    }

    public void parseAl1(Element al1) throws ApiCallMalformedException {
        String setId = HL7XmlDoc.getHl7Text(al1, "AL1.1.1").getString();
        String trace = "AL1[" + setId + "]";

        allergenCode.parseCe(al1, "AL1", 3);
        allergenSeverityCode.parseCe(al1, "AL1", 4);
        allergyReactionCodeList = HL7XmlDoc.getHl7TextList(al1, "AL1.5.1");

        HL7String idenDtText = HL7XmlDoc.getHl7Text(al1, "AL1.6.1");
        identificationDate = HL7XmlDoc.parseDateOrTimestamp(idenDtText, trace + ".6.1", messageProperties);
    }

    public void parseNte(Element nte) throws ApiCallMalformedException {
        if (nte == null) {
            return;
        }

        enteredBy.parseXcn(nte, "NTE", 5);
    }

    public HL7Code getAllergenCode() {
        return allergenCode;
    }

    public void setAllergenCode(HL7Code allergenCode) {
        this.allergenCode = allergenCode;
    }

    public HL7Code getAllergenSeverityCode() {
        return allergenSeverityCode;
    }

    public void setAllergenSeverityCode(HL7Code allergenSeverityCode) {
        this.allergenSeverityCode = allergenSeverityCode;
    }

    public HL7Date getIdentificationDate() {
        return identificationDate;
    }

    public void setIdentificationDate(HL7Date identificationDate) {
        this.identificationDate = identificationDate;
    }

    public List<HL7String> getAllergyReactionCodeList() {
        return allergyReactionCodeList;
    }

    public void setAllergyReactionCodeList(List<HL7String> allergyReactionCodeList) {
        this.allergyReactionCodeList = allergyReactionCodeList;
    }

    public HL7PersonRef getEnteredBy() {
        return enteredBy;
    }

    public void setEnteredBy(HL7PersonRef enteredBy) {
        this.enteredBy = enteredBy;
    }
}
