package com.pkb.service.dataupload.hl7.segment;

import com.pkb.app.interfaces.DataPointWithStartAndEndDate;
import com.pkb.coding.entity.CodeableConcept;
import com.pkb.exception.ApiCallMalformedException;
import com.pkb.service.dataupload.hl7.HL7MessageProperties;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.field.HL7Code;
import com.pkb.service.dataupload.hl7.field.HL7PersonRef;
import com.pkb.service.dataupload.hl7.value.HL7Date;
import com.pkb.service.dataupload.hl7.value.HL7String;
import org.jetbrains.annotations.Nullable;
import org.w3c.dom.Element;

import java.io.Serializable;
import java.time.Instant;

/**
 * Parses a ZRX segment. See HL7 v2.4 Chapter 4.
 *
 * The HL7 spec for the customer ZRX segment is identical to RXE.
 *
 * @author robwhelan
 */
public class HL7Medication implements DataPointWithStartAndEndDate, Serializable {

    private static final long serialVersionUID = 1L;

    private HL7String repeatPattern = new HL7String();
    private HL7Date startTimestamp = new HL7Date();
    private HL7Date endTimestamp = new HL7Date();
    private HL7Code medicationCode = new HL7Code();
    private HL7String doseQuantity = new HL7String();
    private HL7Code unitCode = new HL7Code();
    private String instructions = "";
    private HL7PersonRef reqBy = new HL7PersonRef();

    private HL7MessageProperties messageProperties;

    public HL7Medication(HL7MessageProperties messageProperties) {
        this.messageProperties = messageProperties;
    }

    @Override
    public CodeableConcept code() {
        return medicationCode.asCodeableConcept();
    }

    @Nullable
    @Override
    public Instant startDate() {
        return startTimestamp.getInstant();
    }

    @Nullable
    @Override
    public Instant endDate() {
        return endTimestamp.getInstant();
    }

    public void parseZrx(Element zrx, int setId) throws ApiCallMalformedException {
        String trace = "ZRX[" + setId + "]";

        repeatPattern = HL7XmlDoc.getHl7Text(zrx, "ZRX.1.2");

        HL7String startTsText = HL7XmlDoc.getHl7Text(zrx, "ZRX.1.4");
        startTimestamp = HL7XmlDoc.parseDateOrTimestamp(startTsText, trace + ".1.4", messageProperties);

        HL7String endTsText = HL7XmlDoc.getHl7Text(zrx, "ZRX.1.5");
        endTimestamp = HL7XmlDoc.parseDateOrTimestamp(endTsText, trace + ".1.5", messageProperties);

        medicationCode.parseCe(zrx, "ZRX", 2);

        doseQuantity = HL7XmlDoc.getHl7Text(zrx, "ZRX.3.1");

        unitCode.parseCe(zrx, "ZRX", 5);

        // TODO: Should parse as CE, but leave for now

        instructions = HL7XmlDoc.joinTextContentOfTags(zrx, "ZRX.7.2", true);

        reqBy.parseXcn(zrx, "ZRX", 13);
    }

    public HL7String getRepeatPattern() {
        return repeatPattern;
    }

    public void setRepeatPattern(String repeatPattern) {
        this.repeatPattern = new HL7String(repeatPattern);
    }

    public HL7Date getStartTimestamp() {
        return startTimestamp;
    }

    public void setStartTimestamp(HL7Date startTimestamp) {
        this.startTimestamp = startTimestamp;
    }

    public HL7Date getEndTimestamp() {
        return endTimestamp;
    }

    public void setEndTimestamp(HL7Date endTimestamp) {
        this.endTimestamp = endTimestamp;
    }

    public HL7Code getMedicationCode() {
        return medicationCode;
    }

    public void setMedicationCode(HL7Code medicationCode) {
        this.medicationCode = medicationCode;
    }

    public HL7String getDoseQuantity() {
        return doseQuantity;
    }

    public void setDoseQuantity(String doseQuantity) {
        this.doseQuantity = new HL7String(doseQuantity);
    }

    public HL7Code getUnitCode() {
        return unitCode;
    }

    public void setUnitCode(HL7Code unitCode) {
        this.unitCode = unitCode;
    }

    public String getInstructions() {
        return instructions;
    }

    public void setInstructions(String instructions) {
        this.instructions = instructions;
    }

    public HL7PersonRef getReqBy() {
        return reqBy;
    }

    public void setReqBy(HL7PersonRef reqBy) {
        this.reqBy = reqBy;
    }
}
