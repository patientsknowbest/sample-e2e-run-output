package com.pkb.service.dataupload.hl7.segment;

import com.pkb.exception.ApiCallMalformedException;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.value.HL7String;
import org.apache.commons.lang3.StringUtils;
import org.apache.commons.validator.routines.UrlValidator;
import org.springframework.web.util.UriComponents;
import org.springframework.web.util.UriComponentsBuilder;
import org.w3c.dom.Element;

import java.io.Serializable;

public class HL7DatapointExternalIntegrationUrl implements Serializable {

    private HL7String url = new HL7String();

    public void parseZap(Element zap) {
        url = parseURL(zap);
    }

    public HL7String getUrl() {
        return url;
    }

    private boolean isInvalidUrl(String url) {
        UrlValidator validator = new UrlValidator();
        return !validator.isValid(url);
    }

    private HL7String parseURL(Element element) {
        HL7String url = HL7XmlDoc.getHl7Text(element, "ZAP.1.1");
        if (url.isBlank()) {
            throw new ApiCallMalformedException("Appointment external integration management url is blank!");
        }

        UriComponentsBuilder builder = UriComponentsBuilder.fromUriString(url.getString());
        UriComponents components = builder.build();

        if (StringUtils.isBlank(components.getScheme())) {
            builder = UriComponentsBuilder.fromUriString("https://" + url);
        }

        String parsedUrl = builder.toUriString();

        if (isInvalidUrl(parsedUrl)) {
            throw new ApiCallMalformedException("Appointment external integration management url has invalid format! URL: " + parsedUrl);
        }
        return HL7String.of(parsedUrl);
    }
}
