package com.pkb.service.dataupload.hl7.segment;

import java.io.Serializable;

import org.w3c.dom.Element;

import com.pkb.exception.ApiCallMalformedException;
import com.pkb.service.dataupload.hl7.HL7MessageProperties;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.value.HL7Date;
import com.pkb.service.dataupload.hl7.value.HL7String;

/**
 *
 * Parses an EVN segment. See HL7 v2.4 Chapter 3.
 *
 * @author johnspencepkb
 *
 */
public class HL7EventType implements Serializable {

    private static final long serialVersionUID = 1L;

    /** EVN-1 Event type code (ID) 00099 */
    private HL7String eventTypeCode = new HL7String();

    /** EVN-2 Recorded date/time (TS) 00100 */
    private HL7Date eventRecordedDate = new HL7Date();

    /** EVN-3 Date/time planned event (TS) 00101 */
    private HL7Date eventPlannedDate = new HL7Date();

    /** EVN-4 Event reason code (IS) 00102 */
    private HL7String eventReasonCode = new HL7String();

    /** EVN-5 Operator ID (XCN) 00103 */
    private HL7String operatorId = new HL7String();

    /** EVN-6 Event occurred (TS) 01278 */
    private HL7Date eventOccurredDate = new HL7Date();

    /** EVN-7 Event facility (HD) 01534 */
    private HL7String eventFacility = new HL7String();

    private HL7MessageProperties messageProperties;

    public HL7EventType(HL7MessageProperties messageProperties, Element evn) throws ApiCallMalformedException {
        this.messageProperties = messageProperties;
        this.parseEVN(evn);
    }

    /**
     * Parses the EVN HL7 Message Segment.
     * <p>
     * There will only be a single EVN segment in any given message.
     * </p>
     *
     * <p>
     * <strong>Note:</strong> In other similar Segment parsing DTO classes, the pattern is to construct
     * an instance and then call a voice method parseXXX(). This class differs in an attempt
     * to reduce mutability. If it is desired to retain the standard pattern with the HL7EventType class here,
     * simply change this method access modifier to public and switch out the above constructors.
     * </p>
     *
     * @param evn
     * @throws ApiCallMalformedException
     */
    private void parseEVN(Element evn) throws ApiCallMalformedException {
        this.eventTypeCode = HL7XmlDoc.getHl7Text(evn, "EVN.1.1");
        this.eventRecordedDate = HL7XmlDoc.parseDateOrTimestamp(HL7XmlDoc.getHl7Text(evn, "EVN.2.1"), "EVN.2.1", messageProperties);
        this.eventPlannedDate = HL7XmlDoc.parseDateOrTimestamp(HL7XmlDoc.getHl7Text(evn, "EVN.3.1"), "EVN.3.1", messageProperties);
        this.eventReasonCode = HL7XmlDoc.getHl7Text(evn, "EVN.4.1");
        this.operatorId = HL7XmlDoc.getHl7Text(evn, "EVN.5.1");
        this.eventOccurredDate = HL7XmlDoc.parseDateOrTimestamp(HL7XmlDoc.getHl7Text(evn, "EVN.6.1"), "EVN.6.1", messageProperties);
        this.eventFacility = HL7XmlDoc.getHl7Text(evn, "EVN.7.1");
    }

    /** EVN-1 Event type code (ID) 00099 */
    public HL7String getEventTypeCode() {
        return eventTypeCode;
    }

    /** EVN-2 Recorded date/time (TS) 00100 */
    public HL7Date getEventRecordedDate() {
        return eventRecordedDate;
    }

    /** EVN-3 Date/time planned event (TS) 00101 */
    public HL7Date getEventPlannedDate() {
        return eventPlannedDate;
    }

    /** EVN-4 Event reason code (IS) 00102 */
    public HL7String getEventReasonCode() {
        return eventReasonCode;
    }

    /** EVN-5 Operator ID (XCN) 00103 */
    public HL7String getOperatorId() {
        return operatorId;
    }

    /** EVN-6 Event occurred (TS) 01278 */
    public HL7Date getEventOccurredDate() {
        return eventOccurredDate;
    }

    /** EVN-7 Event facility (HD) 01534 */
    public HL7String getEventFacility() {
        return eventFacility;
    }
}
