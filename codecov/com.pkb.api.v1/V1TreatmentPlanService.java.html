/**
 * Service to upload treatmentPlan data
 *
 * @author robwhelan
 */
@Path(V1TreatmentPlanService.V1_TREATMENT_PLANS_PATH)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Component
@Api(tags = { "Treatments" }, value = "Access to the treatment service")
public class V1TreatmentPlanService {

    static final String V1_TREATMENT_PLANS_PATH = "/v1/treatmentPlans";

    private static final String GET_ATTACHMENT_PATH = "/attachment/{id}";

    private PlanManager planManager;

    private UserManager userManager;

    private PatientConsentManager patientConsentManager;

    private PlanTemplateManager planTemplateManager;

    private TeamUserManager teamUserManager;

    private RestUtil util;

    private TreatmentPlanApiHelper treatmentPlanApiHelper;

    private PhrConfig config;

    private DateTimeService dateTimeService;

    private DateUtil dateUtil;

    private LocalizedTextProvider localizedTextProvider;

    private LoincManager loincManager;

    @Autowired
    public void setUtil(RestUtil util) {
        this.util = util;
    }

    @Autowired
    public void setPlanManager(PlanManager planManager) {
        this.planManager = planManager;
    }

    @Autowired
    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    @Autowired
    public void setPatientConsentManager(PatientConsentManager patientConsentManager) {
        this.patientConsentManager = patientConsentManager;
    }

    @Autowired
    public void setPlanTemplateManager(PlanTemplateManager planTemplateManager) {
        this.planTemplateManager = planTemplateManager;
    }

    @Autowired
    public void setTeamUserManager(TeamUserManager teamUserManager) {
        this.teamUserManager = teamUserManager;
    }

    @Autowired
    public void setConfig(PhrConfig config) {
        this.config = config;
    }

    @Autowired
    public void setDateTimeService(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    @Autowired
    public void setDateUtil(DateUtil dateUtil) {
        this.dateUtil = dateUtil;
    }

    @Autowired
    public void setLocalizedTextProvider(LocalizedTextProvider localizedTextProvider) {
        this.localizedTextProvider = localizedTextProvider;
    }

    @Autowired
    public void setLoincTestService(LoincManager loincManager) {
        this.loincManager = loincManager;
    }

    private static final Logger logger = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private TreatmentPlanApiHelper getTreatmentPlanApiHelper() {

        if (treatmentPlanApiHelper == null) {
            HttpServletRequest servletRequest = util.getRequest();
            String url = servletRequest.getRequestURI();
            String baseUrl = url.substring(0, url.length() - servletRequest.getRequestURI().length())
                    + servletRequest.getContextPath()
                    + "/";
            String attachmentPath = V1_TREATMENT_PLANS_PATH.substring(1)
                    + GET_ATTACHMENT_PATH.substring(0, GET_ATTACHMENT_PATH.indexOf("{"));
            this.treatmentPlanApiHelper = new TreatmentPlanApiHelper(planManager, patientConsentManager, planTemplateManager, teamUserManager,
                    userManager, baseUrl + attachmentPath, dateTimeService, dateUtil, loincManager, localizedTextProvider);
        }

        return this.treatmentPlanApiHelper;
    }

    /**
     * <br/>
     * <a href="http://dev.patientsknowbest.com/home/rest-api/scopes"><b>Scope</b></a>: PATIENT
     *
     * @param authToken
     *            Your bearer token, in the form: Bearer myBearerToken
     * @return
     */
    @GET
    @Path("/mine")
    @Timed
    @RequiresScope(scopes = PATIENT)
    @ApiOperation(value = "Scope: PATIENT", notes = "PDF content is currently unsupported")
    public ApiReadResponse<ApiTreatmentPlan> getUsersTreatmentPlans() {
        PKBCommonApiParams params = util.getCommonParams();
        PKBPerson loggedInUser = params.getPerson();
        try {
            long patientId = loggedInUser.getId();
            return PKBApiHelper.buildReadResponse(getTreatmentPlanApiHelper().getPlanForPatient(params.getRequestContext(), patientId, loggedInUser));
        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error getting treatmentPlan list for self", e);
        }
    }

    /**
     * Create a treatmentPlan.
     * <p/>
     * <br/>
     * <a href="http://dev.patientsknowbest.com/home/rest-api/scopes"><b>Scope</b></a>: CLINICIAN
     *
     * @param treatmentPlan
     *            The treatmentPlan to create
     * @param authToken
     *            Your bearer token, in the form: Bearer myBearerToken
     */
    @POST
    @Timed
    @RequiresScope(scopes = { CLINICIAN, TEAMCOORD })
    @ApiOperation(value = "Scope: CLINICIAN, TEAMCOORD", notes = "PDF content cannot yet be accepted.")
    public ApiWriteResponse createTreatmentPlan(ApiTreatmentPlan treatmentPlan) {
        PKBCommonApiParams params = util.getCommonParams();
        PKBPerson loggedInUser = params.getPerson();
        try {

            if (config.isDisplayOfSensitiveErrorInformationEnabled()) {
                logger.info("called createTreatmentPlan() {}", treatmentPlan);
            }

            ApiWriteResponse response = new ApiWriteResponse();
            response.setOperationPerformed(OperationEnum.CREATED);
            response.setId(getTreatmentPlanApiHelper().createTreatmentPlanForPatient(params.getRequestContext(),
                    treatmentPlan, loggedInUser).toString());
            return response;

        } catch (Exception e) {
            throw PKBApiHelper.wrapError("failed saving new treatmentPlan", e);
        }

    }
}
