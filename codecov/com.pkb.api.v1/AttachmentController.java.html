package com.pkb.api.v1;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.aspect.annotation.Timed;
import com.pkb.api.error.RestError;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.api.util.v2.exception.SharingDisabledException;
import com.pkb.apiservice.AttachmentService;
import com.pkb.common.base.Error;
import com.pkb.document.AttachmentDownloadStream;
import com.pkb.document.AttachmentInformation;
import com.pkb.user.entity.PKBPerson;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import io.swagger.annotations.ApiResponse;
import io.swagger.annotations.ApiResponses;
import io.vavr.control.Either;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.PathParam;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import javax.ws.rs.core.Response.ResponseBuilder;
import java.util.Objects;

import static com.pkb.apiservice.AttachmentService.ERROR_CODE_SHARING_DISABLED;
import static com.pkb.entities.enums.api.ApiClientScope.CLINICIAN;
import static com.pkb.entities.enums.api.ApiClientScope.PATIENT;
import static com.pkb.entities.enums.api.ApiClientScope.TEAMCOORD;

@Path(AttachmentController.ATTACHMENTS_BASE_PATH)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = { "Attachments" }, value = "Manage attachments")
@Component
public class AttachmentController {

    public static final String ATTACHMENTS_BASE_PATH = "/attachments";

    @Autowired
    private RestUtil util;

    @Autowired
    private AttachmentService attachmentService;

    @GET
    @Timed(hideParams = "id")
    @Path("/{id}")
    @RequiresScope(scopes = { PATIENT, CLINICIAN, TEAMCOORD })
    @ApiOperation(value = "Scope: PATIENT, CLINICIAN, TEAMCOORD", notes = "Download attachment by id.")
    @ApiResponses(value = {
            @ApiResponse(code = 200, response = String.class, message = "Attachment downloaded successfully."),
            @ApiResponse(code = 400, response = RestError.class, message = "Invalid request!")
    })
    public Response getAttachment(@PathParam("id") Long id) {
        PKBCommonApiParams params = util.getCommonParams();
        PKBPerson loggedInUser = params.getPerson();
        Either<Error, AttachmentDownloadStream> attachment = attachmentService.getAttachment(id, params.getRequestContext(), loggedInUser);

        AttachmentDownloadStream attachmentDownloadStream = attachment
                .getOrElseThrow(e ->
                    Objects.equals(e.getErrorCode(), ERROR_CODE_SHARING_DISABLED) ?
                        new SharingDisabledException(e.getDescription()) :
                        new IllegalStateException(e.getErrorCode()));

        return createResponse(attachmentDownloadStream);
    }

    private Response createResponse(AttachmentDownloadStream attachment) {
        ResponseBuilder resultBuilder = Response.ok(attachment.getInputStream());
        AttachmentInformation information = attachment.getInformation();
        if (information.getMediaType() != null) {
            resultBuilder.header("Content-Type", information.getMediaType());
        }
        if (information.getName() != null) {
            resultBuilder.header("Content-Disposition", "attachment; filename=" + information.getName());
        }
        if (information.getSize() != null) {
            resultBuilder.header("Content-Length", information.getSize());
        }
        return resultBuilder.build();
    }
}
