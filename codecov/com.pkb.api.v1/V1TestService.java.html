package com.pkb.api.v1;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.aspect.annotation.Timed;
import com.pkb.api.error.ApiAuthError;
import com.pkb.api.util.PKBApiHelper;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.api.util.TestsApiHelper;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.api.v1.model.ApiReadResponse;
import com.pkb.api.v1.model.ApiTest;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.user.entity.PKBPerson;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import io.swagger.annotations.ApiParam;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import javax.servlet.http.HttpServletRequest;
import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.QueryParam;
import javax.ws.rs.core.Context;
import javax.ws.rs.core.MediaType;
import java.util.List;

import static com.pkb.entities.enums.api.ApiClientScope.CLINICIAN;

/**
 * Service to download test results
 */
@Path("v1/tests")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Component
@Api(tags = { "Tests" }, value = "Manage access to the test service")
public class V1TestService {

    @Autowired
    private PatientConsentManager patientConsentManager;

    @Autowired
    private TestsApiHelper testsApiHelper;

    @Autowired
    private RestUtil util;

    @Context
    private HttpServletRequest httpRequest;

    @GET
    @Path("/results/forPatient/byTestIds")
    @Timed(hideParams = "patientId")
    @RequiresScope(scopes = CLINICIAN)
    @ApiOperation(value = "Scope: CLINICIAN", notes =
                    "Allows a professional to search for the test results of a patient. Optionally, the results can be " +
                    "filtered by specific test IDs. These IDs are assigned by PKB to specific test types, which are " +
                    "always associated with a LOINC code, so you will need to contact PKB support to identify the " +
                    "correct values to supply to this parameter. If provided, search results will be filtered to tests " +
                    "that were either matched directly to the corresponding type when supplied, or which were " +
                    "subsequently mapped to that type. " +
                    "Since the filtering works on LOINC-based common test types, this endpoint cannot be used to search " +
                    "specifically for results associated with local test codes that have not been mapped to a common " +
                    "LOINC code. " +
                    "See our guidance on the <a href=\"https://deploy.patientsknowbest.com/integrations/HL7/Path/LOINC\">" +
                    "Benefits of LOINC mapping</a> for more information.",
            response = ApiTest.class,
            responseContainer = "List")
    public ApiReadResponse<ApiTest> getTestResultsForPatientByTestIds(
            @ApiParam(value = "The PKB ID of the patient whose results you wish to fetch.",
                    example = "111",
                    required = true)
                    @QueryParam("patientid") long patientId,
            @ApiParam(value = "Zero, one or more test IDs (repeat the parameter using & to provide multiple values). If " +
                    "provided, the search will only return results associated with one of the specified test types. " +
                    "Example: ?testid=1&testid=2&testid=3",
                    example = "222",
                    allowMultiple = true)
                    @QueryParam("testid") List<Long> testIds) {
        PKBCommonApiParams params = util.getCommonParams(patientId);
        PKBPerson loggedInUser = params.getPerson();
        try {
            if (!patientConsentManager.hasAccessToUser(params.getRequestContext(), patientId)) {
                throw new ApiAuthError("current user " + loggedInUser.getId() + " does not have access to patient " + patientId);
            }
            List<ApiTest> results = testsApiHelper.getTestResults(params.getRequestContext(), testIds, patientId);
            return PKBApiHelper.buildReadResponse(results);
        } catch (Exception e) {
            throw PKBApiHelper.wrapError("Error getting test results:", e);
        }
    }
}
