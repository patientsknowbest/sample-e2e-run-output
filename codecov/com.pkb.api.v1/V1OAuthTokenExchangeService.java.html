package com.pkb.api.v1;

import com.pkb.api.aspect.annotation.Timed;
import com.pkb.api.util.InternalAccessTokenIssuer;
import com.pkb.api.util.InternalAccessTokenIssuer.InternalAccessToken;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.api.v1.model.ApiAccessToken;
import com.pkb.app.entity.PersonAuthInfo;
import com.pkb.common.config.PhrConfig;
import com.pkb.entities.enums.UserType;
import io.swagger.annotations.Api;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.context.SecurityContext;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.WebApplicationException;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.NewCookie;
import javax.ws.rs.core.Response;

import static com.pkb.util.PKBWebUtil.SPRING_SECURITY_CONTEXT;
import static java.util.Optional.ofNullable;

@Path("/v1/ui")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(value = "Authorization via internal OAuth authorization server", tags = { "Internal authorization" })
public class V1OAuthTokenExchangeService {

    public static final String COOKIE_NAME = "ui-exchange";

    private final RestUtil util;
    private final InternalAccessTokenIssuer issuer;
    private final PhrConfig config;

    public V1OAuthTokenExchangeService(RestUtil util, InternalAccessTokenIssuer issuer, PhrConfig config) {
        this.util = util;
        this.issuer = issuer;
        this.config = config;
    }

    private Authentication getAuthentication() {

        Authentication authentication = ofNullable(util.getRequest().getSession(false))
                .map(session -> session.getAttribute(SPRING_SECURITY_CONTEXT))
                .map(SecurityContext.class::cast)
                .map(SecurityContext::getAuthentication)
                .orElseThrow(() -> new WebApplicationException(Response.Status.UNAUTHORIZED));
        return authentication;
    }

    private InternalAccessToken exchange(long userId, String password, UserType type) {
        try {
            return issuer.authenticate(Long.toString(userId), password, type);
        } catch (Exception e) {
            throw new WebApplicationException(e, Response.Status.UNAUTHORIZED);
        }
    }

    @GET
    @Timed
    @Path("/exchange")
    public Response exchange() {

        Authentication authentication = getAuthentication();

        String password = (String) authentication.getCredentials();
        PersonAuthInfo user = (PersonAuthInfo) authentication.getPrincipal();

        InternalAccessToken internalAccessToken = exchange(user.getId(), password, user.getUserType());

        ApiAccessToken token = new ApiAccessToken();
        token.setExp(internalAccessToken.getExpiresIn());
        token.setValue(internalAccessToken.getValue());

        NewCookie cookie = new NewCookie(COOKIE_NAME, internalAccessToken.getValue(), "/", null, null, -1, config.isEnforceSecureCookies());
        //hack: setting httpOnly does not work if cookie method is used
        return Response.ok(token).header("Set-Cookie", cookie.toString() + ";HttpOnly").build();
    }
}
