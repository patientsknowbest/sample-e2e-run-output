package com.pkb.api.v1;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.aspect.annotation.Timed;
import com.pkb.api.error.ApiValidationError;
import com.pkb.api.util.JournalApiHelper;
import com.pkb.api.util.PKBApiHelper;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.api.v1.model.ApiJournalEntry;
import com.pkb.api.v1.model.ApiWriteResponse;
import com.pkb.user.entity.PKBPerson;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import javax.servlet.http.HttpServletRequest;
import javax.ws.rs.Consumes;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.Context;
import javax.ws.rs.core.MediaType;

import static com.pkb.entities.enums.api.ApiClientScope.PATIENT;

/**
 * REST API functionality to add/remove entries into/from a patient's journal.
 *
 * @author Tom
 */
@Path("/v1/journal")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Component
@Api(tags = { "Journals" }, value = "Manage access to the journal service")
public class V1JournalService {

    private JournalApiHelper apiHelper;
    private RestUtil util;

    @Context
    private HttpServletRequest httpRequest;

    @Autowired
    protected void setUtil(RestUtil util) {
        this.util = util;
    }

    @Autowired
    public void setApiHelper(JournalApiHelper apiHelper) {
        this.apiHelper = apiHelper;
    }

    /**
     * Add a journal entry for a patient. Note, the author is always
     * taken to be the currently logged in user. It is not possible
     * to specify an author ID.
     * <p/>
     * <br/>
     * <a href="http://dev.patientsknowbest.com/home/rest-api/scopes"><b>Scope</b></a>: PATIENT
     *
     * @param journalEntry
     *            The entry to add
     * @param authToken
     *            Your bearer token, in the form: Bearer myBearerToken
     */
    @POST
    @Timed
    @RequiresScope(scopes = { PATIENT })
    @ApiOperation(value = "Scope: PATIENT")
    public ApiWriteResponse addJournalEntryForPatient(ApiJournalEntry journalEntry) {
        try {
            PKBCommonApiParams params = util.getCommonParams();
            PKBPerson loggedInUser = params.getPerson();

            if (journalEntry.getId() != null) {
                throw new ApiValidationError("updates to journal entries are not yet supported");
            }
            if ((journalEntry.getAuthorId() != null) && (!journalEntry.getAuthorId().isEmpty())) {
                throw new ApiValidationError("author taken to be logged in user, you cannot supply an author ID");
            }
            if ((journalEntry.getEntryTextHtml() != null) && (!journalEntry.getEntryTextHtml().isEmpty())) {
                throw new ApiValidationError("HTML journal entries are not yet supported");
            }
            // TODO: Mandating a date to be safe for now, but do we really need one?
            if (journalEntry.getDate() == null) {
                throw new ApiValidationError("a date must be supplied");
            }
            Long id = apiHelper.addJournalEntryForPatient(params.getRequestContext(),
                    loggedInUser, loggedInUser.getId(), journalEntry);
            ApiWriteResponse response = new ApiWriteResponse();
            response.setOperationPerformed(ApiWriteResponse.OperationEnum.CREATED);
            response.setId("" + id);
            return response;
        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error adding journal entry", e);
        }
    }
}
