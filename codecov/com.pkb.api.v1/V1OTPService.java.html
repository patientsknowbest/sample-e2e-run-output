package com.pkb.api.v1;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.aspect.annotation.Timed;
import com.pkb.api.util.PKBApiHelper;
import com.pkb.api.v1.model.ApiWebappOTP;
import com.pkb.service.user.impl.ApiAuthManager;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import javax.servlet.http.HttpServletRequest;
import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.HeaderParam;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.Context;
import javax.ws.rs.core.MediaType;

import static com.pkb.entities.enums.api.ApiClientScope.CLINICIAN;
import static com.pkb.entities.enums.api.ApiClientScope.PATIENT;

@Path("/v1/otp")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Component
@Api(tags = { "OTPs" }, value = "Manage access to the OTP service")
public class V1OTPService {

    private ApiAuthManager apiAuthManager;

    @Context
    private HttpServletRequest httpRequest;

    @Autowired
    public void setApiAuthManager(ApiAuthManager apiAuthManager) {
        this.apiAuthManager = apiAuthManager;
    }

    /**
     * This operation will return a One Time Password (OTP) that can be used
     * to authenticate against the PKB web interface.
     * <p/>
     * <br/>
     * <a href="http://dev.patientsknowbest.com/home/rest-api/scopes"><b>Scope</b></a>: PATIENT, CLINICIAN
     *
     * @param authToken
     *            Your bearer token, in the form: Bearer myBearerToken
     */
    @GET
    @Timed
    @RequiresScope(scopes = { PATIENT, CLINICIAN })
    @ApiOperation(value = "Scope: PATIENT, CLINICIAN")
    public ApiWebappOTP getOTP(
            @HeaderParam(value = "Authorization") String authToken) {
        try {
            // Encrypt user password with random key, and return the key
            ApiWebappOTP otp = new ApiWebappOTP();
            otp.setOtp_token(apiAuthManager.generateWebappOTP(authToken));
            return otp;
        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error generating OTP", e);
        }
    }
}
