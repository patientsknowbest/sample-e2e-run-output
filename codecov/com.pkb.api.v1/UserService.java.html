package com.pkb.api.v1;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.aspect.annotation.Timed;
import com.pkb.api.util.PKBApiHelper;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.api.util.UserApiHelper;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.api.v1.model.ApiUser;
import com.pkb.institute.entity.InstituteUserEntity.ContactOptions;
import com.pkb.user.entity.PKBPerson;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.QueryParam;
import java.util.List;

import static com.pkb.entities.enums.api.ApiClientScope.CLINICIAN;
import static com.pkb.entities.enums.api.ApiClientScope.PATIENT;
import static java.util.Collections.emptyList;

/**
 * Service to retrieve user info
 *
 * @author robwhelan
 */
@Component
@Path("/users")
@Api(tags = { "Users" }, value = "Manage access to the user service")
@Produces("application/json")
@Consumes("application/json")
public class UserService {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    private UserApiHelper userApiHelper;

    @Autowired
    private RestUtil util;

    /**
     * Details on the currently authenticated user (patient or clinician).
     */
    @GET
    @Path("/self")
    @Timed
    @RequiresScope(scopes = { PATIENT, CLINICIAN })
    @ApiOperation(value = "Scope: PATIENT, CLINICIAN")
    public ApiUser getUser() {
        ApiUser result = null;
        PKBCommonApiParams params = util.getCommonParams();
        PKBPerson loggedInUser = params.getPerson();

        try {

            result = userApiHelper.getUser(params.getRequestContext(),
                    loggedInUser.getId(), loggedInUser);

            LOGGER.info("got results!");
        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error getting self user", e);
        }

        return result;
    }

    /**
     * Patients of the current clinician
     */
    @GET
    @Path("/myPatients")
    @Timed
    @RequiresScope(scopes = CLINICIAN)
    @ApiOperation(value = "Scope: CLINICIAN")
    public List<ApiUser> getMyPatients(@QueryParam("registeredOnly") Boolean registeredOnly ) {
        PKBCommonApiParams params = util.getCommonParams();
        PKBPerson loggedInUser = params.getPerson();
        List<ApiUser> resultList = emptyList();
        try {
            resultList = userApiHelper.getPatients(params.getRequestContext(), loggedInUser, true,
                    registeredOnly != null ? registeredOnly : false);
        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error getting clinician's patients", e);
        }
        return resultList;
    }

    /**
     * Patients of the current clinician
     */
    @GET
    @Path("/myClinicians")
    @Timed
    @RequiresScope(scopes = PATIENT)
    @ApiOperation(value = "Scope: PATIENT")
    public List<ApiUser> getMyClinicians() {
        PKBCommonApiParams params = util.getCommonParams();
        PKBPerson loggedInUser = params.getPerson();
        // TODO review?  How to decide this?
        ContactOptions contactOptionOrNull = ContactOptions.Messages_Only; // limit to clinicians patient is allowed to message
        try {
            return userApiHelper.getClinicians(params.getRequestContext(), loggedInUser, contactOptionOrNull);
        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error getting clinician's patients", e);
        }
    }

    @GET
    @Path("/myCarers")
    @Timed
    @RequiresScope(scopes = PATIENT)
    @ApiOperation(value = "Scope: PATIENT")
    public List<ApiUser> getMyCarers() {
        List<ApiUser> resultList = emptyList();
        try {
            PKBCommonApiParams params = util.getCommonParams();
            resultList = userApiHelper.getCarers(params.getPerson());
        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error getting colleagues", e);
        }
        return resultList;
    }

}
