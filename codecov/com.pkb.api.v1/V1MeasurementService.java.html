/**
 * Service to upload measurement data
 *
 * @author robwhelan
 */
@Path("/v1/measurements")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Component
@Api(tags = { "Measurements" }, value = "Manage various measurements")
public class V1MeasurementService {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    //*** rpb: I originally implemented this as a copy of MeasurementService using the MeasurementApiHelper class to provide
    //* centralised logic. I've removed it in favour of extending MeasurementService  but have left the old code in because
    //* when measurement service needs changing, this code may be useful
    @Context
    private HttpServletRequest httpRequest;

    private RestUtil util;
    private MeasurementApiHelper measurementApiHelper;
    private PatientConsentManager patientConsentManager;

    @Autowired
    private void setUtil(RestUtil util) {
        this.util = util;
    }

    @Autowired
    public void setMeasurementApiHelper(MeasurementApiHelper measurementApiHelper) {
        this.measurementApiHelper = measurementApiHelper;
    }

    @Autowired
    public void setPatientConsentManager(PatientConsentManager patientConsentManager) {
        this.patientConsentManager = patientConsentManager;
    }

    /**
     * Create a measurement.
     * <br/>
     * <a href="http://dev.patientsknowbest.com/home/rest-api/scopes"><b>Scope</b></a>:
     * PATIENT
     *
     * @param measurement
     *            The measurement to create
     * @param authToken
     *            Your bearer token, in the form: Bearer myBearerToken
     */
    @POST
    @Timed
    @RequiresScope(scopes = PATIENT)
    @ApiOperation(value = "Scope: PATIENT")
    public ApiWriteResponse createMeasurement(
            ApiMeasurement measurement) {
        PKBCommonApiParams params = util.getCommonParams();
        PKBPerson loggedInUser = params.getPerson();
        UUID id = createMeasurementForUser(params.getRequestContext(), measurement, loggedInUser);
        ApiWriteResponse response = new ApiWriteResponse();
        response.setOperationPerformed(OperationEnum.CREATED);
        response.setId(id.toString());
        return response;
    }

    @DELETE
    @Path("{measurementId}")
    @Timed
    @RequiresScope(scopes = PATIENT)
    @ApiOperation(value = "Scope: PATIENT")
    public ApiWriteResponse deleteMeasurement(@PathParam("measurementId") UUID measurementId) {
        PKBCommonApiParams params = util.getCommonParams();
        PKBPerson loggedInUser = params.getPerson();

        if (measurementApiHelper.deleteMeasurement(params.getRequestContext(), loggedInUser.getId(), measurementId)) {
            LOGGER.info("Patient {} deleted measurement {}", loggedInUser.getId(), measurementId);
            ApiWriteResponse response = new ApiWriteResponse();
            response.setId(measurementId.toString());
            response.setOperationPerformed(OperationEnum.DELETED);
            return response;
        } else {
            throw new ApiGeneralError("Measurement not found", Response.Status.NOT_FOUND.getStatusCode());
        }
    }

    private UUID createMeasurementForUser(EHRRequestContext requestContext, ApiMeasurement measurement, PKBPerson loggedInUser) {
        try {
            // gather info, validate
            PredefinedMeasurementType type = PredefinedMeasurementType.getByApiRef(measurement.getMeasurementTypeApiRef());
            if (type == null) {
                throw new ApiValidationError("Unrecognized measurementTypeApiRef: " + measurement.getMeasurementTypeApiRef());
            }
            if (measurement.getMeasureDate() == null && StringUtils.isBlank(measurement.getMeasureDateISO())) {
                throw new ApiValidationError("Failed parsing measureDate.  Example: 2012-01-10T01:33:51.055+01:00");
            }
            if (!patientConsentManager.hasAccessToUser(requestContext, loggedInUser.getId())) {
                throw new ApiAuthError("current user " + loggedInUser.getId() + " does not have access to patient " + loggedInUser.getId());
            }

            return measurementApiHelper.createMeasurement(requestContext, measurement, loggedInUser.getId());

        } catch (Exception e) {
            throw wrapError("failed saving new measurement", e);
        }
    }
}
