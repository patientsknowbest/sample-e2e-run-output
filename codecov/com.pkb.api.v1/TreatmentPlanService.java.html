/**
 * Service to upload treatmentPlan data
 *
 * @author robwhelan
 */
@Path(TreatmentPlanService.TREATMENT_PLANS_PATH)
@Consumes(MediaType.APPLICATION_JSON)
@Produces(MediaType.APPLICATION_JSON)
@Component
@Api(tags = {"Treatment Plans"}, value = "Manage access to the treatment plan service")
public class TreatmentPlanService {

    static final String TREATMENT_PLANS_PATH = "/treatmentPlans";
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final String GET_ATTACHMENT_PATH = "/attachment/{id}";

    private PlanManager planManager;

    private PatientConsentManager userAccessManager;

    private PlanTemplateManager planTemplateManager;

    private TeamUserManager teamUserManager;

    private TestManager testManager;

    private UserManager userManager;

    private RestUtil util;

    private TreatmentPlanApiHelper treatmentPlanApiHelper;

    private PhrConfig config;

    private DateTimeService dateTimeService;

    private DateUtil dateUtil;

    private LocalizedTextProvider localizedTextProvider;

    private LoincManager loincManager;

    @Autowired
    protected void setUtil(RestUtil util) {
        this.util = util;
    }

    @Autowired
    public void setPlanManager(PlanManager planManager) {
        this.planManager = planManager;
    }

    @Autowired
    public void setUserAccessManager(PatientConsentManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }

    @Autowired
    public void setTeamUserManager(TeamUserManager teamUserManager) {
        this.teamUserManager = teamUserManager;
    }

    @Autowired
    public void setPlanTemplateManager(PlanTemplateManager planTemplateManager) {
        this.planTemplateManager = planTemplateManager;
    }

    @Autowired
    public void setTestManager(TestManager testManager) {
        this.testManager = testManager;
    }

    @Autowired
    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    @Autowired
    public void setConfig(PhrConfig config) {
        this.config = config;
    }

    @Autowired
    public void setDateTimeService(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    @Autowired
    public void setDateUtil(DateUtil dateUtil) {
        this.dateUtil = dateUtil;
    }

    @Autowired
    public void setLocalizedTextProvider(LocalizedTextProvider localizedTextProvider) {
        this.localizedTextProvider = localizedTextProvider;
    }

    @Autowired
    public void setLoincTestService(LoincManager loincManager) {
        this.loincManager = loincManager;
    }

    private TreatmentPlanApiHelper getTreatmentPlanApiHelper() {
        if (treatmentPlanApiHelper == null) {
            HttpServletRequest servletRequest = util.getRequest();
            String url = servletRequest.getRequestURI();
            String baseUrl = url.substring(0, url.length() - servletRequest.getRequestURI().length())
                    + servletRequest.getContextPath()
                    + "/";
            String attachmentPath = TREATMENT_PLANS_PATH.substring(1)
                    + GET_ATTACHMENT_PATH.substring(0, GET_ATTACHMENT_PATH.indexOf("{"));
            return new TreatmentPlanApiHelper(planManager, userAccessManager, planTemplateManager, teamUserManager, userManager,
                    baseUrl + attachmentPath, dateTimeService, dateUtil, loincManager, localizedTextProvider);
        }

        return this.treatmentPlanApiHelper;
    }

    /**
     * Get the list of treatment plans for the given patient id
     * The patient be linked to the current clinician (authenticated to the API)
     *
     * @return The list of treatment plans for the given patient id
     */
    @GET
    @Timed(hideParams = "patientId")
    @Path("/forPatient/{id}")
    @RequiresScope(scopes = {PATIENT, CLINICIAN, TEAMCOORD})
    @ApiOperation(value = "Scope: PATIENT, CLINICIAN, TEAMCOORD", notes = "PDF content is currently unsupported")
    public List<ApiTreatmentPlan> getTreatmentPlansForPatient(
            @PathParam("id") long patientId) {

        List<ApiTreatmentPlan> resultList;
        PKBCommonApiParams params = util.getCommonParams(patientId, true);
        PKBPerson loggedInUser = params.getPerson();

        try {
            resultList = getTreatmentPlanApiHelper().getPlanForPatient(params.getRequestContext(), patientId, loggedInUser);
        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error getting treatmentPlan list for patient " + patientId, e);
        }

        if (config.isDisplayOfSensitiveErrorInformationEnabled()) {
            LOGGER.info("returning resultList {}", resultList);
        }
        return resultList;
    }
}
