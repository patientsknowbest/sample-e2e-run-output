package com.pkb.api.v1;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.aspect.annotation.Timed;
import com.pkb.api.error.ApiValidationError;
import com.pkb.api.util.MedicationApiHelper;
import com.pkb.api.util.PKBApiHelper;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.api.util.v2.exception.SharingDisabledException;
import com.pkb.api.v1.model.ApiMedication;
import com.pkb.api.v1.model.ApiReadResponse;
import com.pkb.api.v1.model.ApiWriteResponse;
import com.pkb.api.v1.model.MedicationFrequency;
import com.pkb.user.entity.PKBPerson;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import io.swagger.annotations.ApiParam;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.PathParam;
import javax.ws.rs.Produces;
import javax.ws.rs.QueryParam;
import javax.ws.rs.core.MediaType;
import javax.xml.bind.DatatypeConverter;
import java.time.Instant;
import java.util.List;
import java.util.UUID;

import static com.pkb.entities.enums.api.ApiClientScope.CLINICIAN;
import static com.pkb.entities.enums.api.ApiClientScope.PATIENT;

/**
 * Service to upload medication data
 *
 * @author robwhelan
 */
@Path("/v1/medications")
@Component
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = {"Medications"}, value = "Manage access to the medication service")
public class V1MedicationService {

    @Autowired
    private RestUtil util;
    @Autowired
    private MedicationApiHelper medicationApiHelper;

    /**
     * returns active medications (currently being taken by the patient)
     * <p/>
     * <br/>
     * <a href="http://dev.patientsknowbest.com/home/rest-api/scopes"><b>Scope</b></a>: PATIENT
     *
     * @param apiRefs
     * @param updatedSinceString
     * @param includeMedicationsNotTaken
     * @return
     */
    @GET
    @Path("/active")
    @Timed
    @RequiresScope(scopes = PATIENT)
    @ApiOperation(value = "Scope: PATIENT")
    public ApiReadResponse<ApiMedication> getActiveMedications(
            @ApiParam("This parameter is not currently supported.") @QueryParam("apiRef") List<String> apiRefs,
            @QueryParam("updatedSince") String updatedSinceString,
            @QueryParam("includeMedicationsNotTaken") boolean includeMedicationsNotTaken) {

        PKBCommonApiParams params = util.getCommonParams(null);
        PKBPerson loggedInUser = params.getPerson();
        try {

            Instant updatedSinceDate = null;

            if ((updatedSinceString != null) && !updatedSinceString.isEmpty()) {
                updatedSinceDate = DatatypeConverter.parseDateTime(updatedSinceString).getTime().toInstant();
            }
            List<ApiMedication> medications = medicationApiHelper.getMedications(params.getRequestContext(),
                    loggedInUser, updatedSinceDate, true, includeMedicationsNotTaken);
            return PKBApiHelper.buildReadResponse(medications);

        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error getting active medications", e);
        }
    }

    /**
     * returns all medications
     * <p/>
     * <br/>
     * <a href="http://dev.patientsknowbest.com/home/rest-api/scopes"><b>Scope</b></a>: PATIENT
     *
     * @return
     */
    @GET
    @Path("/")
    @Timed
    @RequiresScope(scopes = PATIENT)
    @ApiOperation(value = "Scope: PATIENT")
    public ApiReadResponse<ApiMedication> getAllMedications() {

        PKBCommonApiParams params = util.getCommonParams();
        PKBPerson loggedInUser = params.getPerson();
        try {
            List<ApiMedication> medications = medicationApiHelper.getMedications(params.getRequestContext(),
                    loggedInUser, null, false, true);
            return PKBApiHelper.buildReadResponse(medications);

        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error getting active medications", e);
        }
    }

    /**
     * returns medications by apiRef
     * <p/>
     * <br/>
     * <a href="http://dev.patientsknowbest.com/home/rest-api/scopes"><b>Scope</b></a>: CLINICIAN
     *
     * @param patientId
     * @param apiRefs
     * @param updatedSinceString
     * @param includeMedicationsNotTaken
     * @return
     */
    @GET
    @Path("/forPatient/{patientId}")
    @Timed(hideParams = "patientId")
    @RequiresScope(scopes = CLINICIAN)
    @ApiOperation(value = "Scope: CLINICIAN")
    public ApiReadResponse<ApiMedication> getMedicationsForPatient(
            @PathParam("patientId") long patientId,
            @ApiParam("This parameter is not currently supported.") @QueryParam("apiRef") List<String> apiRefs,
            @QueryParam("updatedSince") String updatedSinceString,
            @QueryParam("includeMedicationsNotTaken") boolean includeMedicationsNotTaken) {

        PKBCommonApiParams params = util.getCommonParams(patientId, true);
        PKBPerson loggedInUser = params.getPerson();
        try {

            Instant updatedSinceDate = null;
            if ((updatedSinceString != null) && !updatedSinceString.isEmpty()) {
                updatedSinceDate = DatatypeConverter.parseDateTime(updatedSinceString).getTime().toInstant();
            }

            List<ApiMedication> medications = medicationApiHelper.getMedicationsForPatient(params.getRequestContext(),
                    loggedInUser, patientId, updatedSinceDate, false, includeMedicationsNotTaken);
            return PKBApiHelper.buildReadResponse(medications);

        } catch (SharingDisabledException sde) {
            throw sde; // is handled by exception mapper and returns 403
        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error getting all medications", e);
        }
    }

    /**
     * returns medications by apiRef
     * <p/>
     * <br/>
     * <a href="http://dev.patientsknowbest.com/home/rest-api/scopes"><b>Scope</b></a>: CLINICIAN
     *
     * @param patientId
     * @param apiRefs
     * @param updatedSinceString
     * @param includeMedicationsNotTaken
     * @return
     */
    @GET
    @Path("forPatient/{patientId}/active")
    @Timed(hideParams = "patientId")
    @RequiresScope(scopes = CLINICIAN)
    @ApiOperation(value = "Scope: CLINICIAN")
    public ApiReadResponse<ApiMedication> getActiveMedicationsForPatient(
            @PathParam("patientId") long patientId,
            @ApiParam("This parameter is not currently supported.") @QueryParam("apiRef") List<String> apiRefs,
            @QueryParam("updatedSince") String updatedSinceString,
            @QueryParam("includeMedicationsNotTaken") boolean includeMedicationsNotTaken) {

        PKBCommonApiParams params = util.getCommonParams(patientId, true);
        PKBPerson loggedInUser = params.getPerson();
        try {
            Instant updatedSinceDate = null;
            if ((updatedSinceString != null) && !updatedSinceString.isEmpty()) {
                updatedSinceDate = DatatypeConverter.parseDateTime(updatedSinceString).getTime().toInstant();
            }

            List<ApiMedication> medications = medicationApiHelper.getMedicationsForPatient(params.getRequestContext(),
                    loggedInUser, patientId, updatedSinceDate, true, includeMedicationsNotTaken);
            ApiReadResponse<ApiMedication> results = PKBApiHelper.buildReadResponse(medications);
            return results;

        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error getting all medications", e);
        }
    }

    /**
     * Create a medication.
     * <p/>
     * <br/>
     * <a href="http://dev.patientsknowbest.com/home/rest-api/scopes"><b>Scope</b></a>: PATIENT
     *
     * @param medication The medication to create
     */
    @POST
    @Timed
    @RequiresScope(scopes = PATIENT)
    @ApiOperation(value = "Scope: PATIENT")
    public ApiWriteResponse addOrUpdateMedication(ApiMedication medication) {

        PKBCommonApiParams params = util.getCommonParams();
        PKBPerson loggedInUser = params.getPerson();
        try {
            long patientId = loggedInUser.getId();
            ApiWriteResponse response = new ApiWriteResponse();

            if (MedicationFrequency.constructFromValue(medication.getFrequency()) == MedicationFrequency.UNRECOGNISED) {
                throw new ApiValidationError("Unrecognised frequency. The allowed values are " + MedicationFrequency.allowedValues());
            }

            //if we're updating an existing medication by ID:
            if ((medication.getId() != null) && !medication.getId().isEmpty()) {
                response.setId(medicationApiHelper.updateMedication(params.getRequestContext(),
                        patientId, medication, loggedInUser).map(UUID::toString).getOrNull());
                response.setOperationPerformed(ApiWriteResponse.OperationEnum.UPDATED);
            } else {

                response.setId(medicationApiHelper.createMedication(params.getRequestContext(),
                        medication, patientId, loggedInUser).map(UUID::toString).getOrNull());
                response.setOperationPerformed(ApiWriteResponse.OperationEnum.CREATED);
            }

            return response;

        } catch (Exception e) {
            throw PKBApiHelper.wrapError("error POSTing medications", e);
        }
    }

}
