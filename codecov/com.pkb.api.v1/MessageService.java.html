/**
 * Service to send and receive messages
 *
 * @author russell
 */

@Path("/messages")
@Consumes(MediaType.APPLICATION_JSON)
@Produces(MediaType.APPLICATION_JSON)
@Component
@Api(tags = { "Messages" }, value = "Manage access to the message service")
public class MessageService {

    @Context
    private HttpServletRequest httpRequest;

    @Autowired
    private RestUtil util;

    @Autowired
    private MessageApiHelper messageApiHelper;

    @Autowired
    private MessageSource messageSource;

    /**
     * returns messages since the given timestamp with optional limit. Returns oldest first and works forwards
     *
     * @param limit
     * @param authToken
     * @return
     */
    @GET
    @Timed(hideParams = "timestamp")
    @Path("/after/{timestamp}")
    @RequiresScope(scopes = { PATIENT, CLINICIAN })
    @ApiOperation(value = "Scope: PATIENT, CLINICIAN")
    public List<ApiConversation> getMessagesSince(
            @PathParam("timestamp") String timestamp,
            @QueryParam("limit") int limit, Locale locale) {
        PKBCommonApiParams params = util.getCommonParams();
        PKBPerson loggedInUser = params.getPerson();
        return getMessages(params.getRequestContext(), loggedInUser.getId(), timestamp, null, limit, false, locale);

    }

    private List<ApiConversation> getMessages(@NotNull LoggedInEHRRequestContext requestContext, long userId, String afterDateString,
            String beforeDateString, int limit, boolean orderLatestFirst, Locale locale) {

        Instant afterDate = null;
        Instant beforeDate = null;

        try {

            if (limit < 0) {
                throw new ApiValidationError("Can't have negative limit");
            }

            try {
                if ((afterDateString != null) && !afterDateString.isEmpty()) {
                    afterDate = DatatypeConverter.parseDateTime(afterDateString).toInstant();
                }
                if ((beforeDateString != null) && !beforeDateString.isEmpty()) {
                    beforeDate = DatatypeConverter.parseDateTime(beforeDateString).toInstant();
                }
            } catch (IllegalArgumentException ignored) {
                throw new ApiValidationError("Couldn't read timestamp. It must be formatted in ISO format");
            }

            return messageApiHelper.getConversations(requestContext, userId, afterDate, beforeDate, limit, orderLatestFirst, getText(locale));

        } catch (Exception e) {
            throw PKBApiHelper.wrapError("Error getting messages: ", e);
        }
    }

    private BiFunction<String, String[], String> getText(Locale locale) {
        return (key, args) -> messageSource.getMessage(key, args, locale);
    }
}
