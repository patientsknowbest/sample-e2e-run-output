package com.pkb.api.internal.model.response;

import com.pkb.entities.enums.UserStatus;
import com.pkb.entities.enums.UserType;
import com.pkb.datamodel.Email;

import java.io.Serializable;

/**
 * Contains non-sensitive information about a person.
 */
public class NonSensitivePersonInformationDTO implements Serializable {

    private Long id;
    private UserType userType;
    private UserStatus userStatus;
    private String humanUUID;
    private String firstName;
    private String lastName;
    private Email email;
    private boolean isRecordFrozen;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public UserType getUserType() {
        return userType;
    }

    public void setUserType(UserType userType) {
        this.userType = userType;
    }

    public UserStatus getUserStatus() {
        return userStatus;
    }

    public void setUserStatus(UserStatus userStatus) {
        this.userStatus = userStatus;
    }

    public String getHumanUUID() {
        return humanUUID;
    }

    public void setHumanUUID(String humanUUID) {
        this.humanUUID = humanUUID;
    }

    public String getFirstName() {
        return firstName;
    }

    public void setFirstName(String firstName) {
        this.firstName = firstName;
    }

    public String getLastName() {
        return lastName;
    }

    public void setLastName(String lastName) {
        this.lastName = lastName;
    }

    public Email getEmail() {
        return email;
    }

    public void setEmail(Email email) {
        this.email = email;
    }

    public boolean isRecordFrozen() {
        return isRecordFrozen;
    }

    public void setRecordFrozen(boolean isRecordFrozen) {
        this.isRecordFrozen = isRecordFrozen;
    }
}
