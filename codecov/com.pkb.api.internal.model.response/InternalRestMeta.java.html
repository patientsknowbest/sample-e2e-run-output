package com.pkb.api.internal.model.response;

import java.io.Serializable;
import java.util.HashMap;
import java.util.LinkedList;
import java.util.List;
import java.util.Map;

public class InternalRestMeta implements Serializable {

    private List<String> info = new LinkedList<>();
    private List<String> warning = new LinkedList<>();
    private List<String> error = new LinkedList<>();
    private Map<String, List<String>> fieldErrors = new HashMap<>();

    public List<String> getInfo() {
        return info;
    }

    public List<String> getWarning() {
        return warning;
    }

    public List<String> getError() {
        return error;
    }

    public Map<String, List<String>> getFieldErrors() {
        return fieldErrors;
    }

    public boolean addInfo(String message) {
        return info.add(message);
    }

    public boolean addWarning(String message) {
        return warning.add(message);
    }

    public boolean addError(String message) {
        return error.add(message);
    }

    public void addFieldError(String fieldName, String message) {
        if (fieldErrors.containsKey(fieldName)) {
            fieldErrors.get(fieldName).add(message);
        } else {
            List<String> list = new LinkedList<>();
            list.add(message);
            fieldErrors.put(fieldName, list);
        }
    }
}
