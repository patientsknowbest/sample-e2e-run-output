package com.pkb.fhir.common.decrypt;

import com.pkb.entities.enums.MenuDataType;
import com.pkb.fhir.common.EhrDataProcessingResult;
import com.pkb.fhir.common.enums.DecryptedEhrDataStatus;

/**
 * Represents a piece of decrypted xml menu_data, with a type, id and whether the decryption was successful.
 */
public class DecryptedEhrData implements EhrDataProcessingResult {
    private Long id;
    private String content;
    private DecryptedEhrDataStatus status;
    private MenuDataType dataType;

    public DecryptedEhrData(Long id, String content, DecryptedEhrDataStatus status, MenuDataType dataType) {
        this.id = id;
        this.content = content;
        this.status = status;
        this.dataType = dataType;
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getContent() {
        return content;
    }

    public void setContent(String content) {
        this.content = content;
    }

    public DecryptedEhrDataStatus getStatus() {
        return status;
    }

    public void setStatus(DecryptedEhrDataStatus status) {
        this.status = status;
    }

    public MenuDataType getDataType() {
        return dataType;
    }

    public void setDataType(MenuDataType dataType) {
        this.dataType = dataType;
    }

    @Override
    public boolean isFailure() {
        return this.status == DecryptedEhrDataStatus.FAILURE;
    }
}
