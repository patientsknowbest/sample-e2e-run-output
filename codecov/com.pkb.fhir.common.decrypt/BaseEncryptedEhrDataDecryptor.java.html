package com.pkb.fhir.common.decrypt;

import com.pkb.fhir.common.enums.DecryptedEhrDataStatus;
import com.pkb.fhir.common.exception.DecryptionKeysNotFoundException;
import com.pkb.fhir.common.exception.MigrationEntityNotFoundException;
import com.pkb.fhir.common.exception.MigrationException;
import com.pkb.fhir.common.repository.MenuDataRepository;
import com.pkb.fhir.common.retrieve.EncryptedEhrDataDataDto;

import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

public class BaseEncryptedEhrDataDecryptor {

    protected final MenuDataRepository menuDataRepository;
    private final RetryingEhrDataDecryptor ehrDataDecryptor;

    public BaseEncryptedEhrDataDecryptor(MenuDataRepository menuDataRepository, RetryingEhrDataDecryptor ehrDataDecryptor) {
        this.menuDataRepository = menuDataRepository;
        this.ehrDataDecryptor = ehrDataDecryptor;
    }

    public DecryptedEhrData process(long menuDataId) {
        List<EncryptedEhrDataDataDto> encryptedDataDtos = getEncryptedEhrDataDto(menuDataId);
        if (encryptedDataDtos.isEmpty()) {
            throw new MigrationEntityNotFoundException("Attempting to decrypt menu_data row for id " + menuDataId
                    + ", but this no longer exists or is soft deleted");
        }
        List<DecryptedEhrData> decryptionResult = decryptData(encryptedDataDtos);

        if (decryptionResult.isEmpty()) {
            throw new MigrationException("Decryption failed. Cause: decryptor returned no result");
        } else if (decryptionResult.get(0).getStatus() == DecryptedEhrDataStatus.NOKEYS) {
            throw new DecryptionKeysNotFoundException("No keys found for decryption. Info: " + decryptionResult.get(0).getContent());
        } else if (decryptionResult.get(0).isFailure()) {
            throw new MigrationException("Decryption failed. Cause: " + decryptionResult.get(0).getContent());
        } else {
            return decryptionResult.get(0);
        }
    }

    public List<EncryptedEhrDataDataDto> getEncryptedEhrDataDto(Long menuDataId) {
        return menuDataRepository.findAllEncryptedDataDtosById(List.of(menuDataId));
    }

    public List<DecryptedEhrData> decryptData(List<EncryptedEhrDataDataDto> data) {
        List<DecryptedEhrData> result = new ArrayList<>();
        Map<Boolean, List<EncryptedEhrDataDataDto>> partitionByAccountAvailability = data.stream().collect(Collectors.partitioningBy(ehrData -> ehrData.getAccountPublicId() != null));
        result.addAll(ehrDataDecryptor.decrypt(partitionByAccountAvailability.get(Boolean.TRUE)));
        result.addAll(partitionByAccountAvailability.get(Boolean.FALSE)
                .stream()
                .map(noAccountIdSourceData ->
                        new DecryptedEhrData(noAccountIdSourceData.getId(), "Missing accountPublicId or personPublicId", DecryptedEhrDataStatus.FAILURE, noAccountIdSourceData.getDataType())
                ).collect(Collectors.toList()));
        return result;
    }
}
