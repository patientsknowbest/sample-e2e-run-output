package com.pkb.fhir.common.decrypt;

import com.pkb.crypto.dto.AccountKeysDTO;
import com.pkb.kms.client.core.Kms;
import com.pkb.kms.shared.representation.KmsError;
import io.vavr.control.Either;

import java.util.List;
import java.util.UUID;

/**
 * Actually gets the decryption keys from the KMS
 */
public class AccountKeyProvider {
    private final Kms kmsClient;

    public AccountKeyProvider(Kms kmsClient) {
        this.kmsClient = kmsClient;
    }

    Either<String, List<AccountKeysDTO>> getAccountKeys(List<UUID> accountPublicIds) {
        String requestId = UUID.randomUUID().toString();
        Either<KmsError, List<AccountKeysDTO>> accountKeys = kmsClient.getAccountKeysTrustedIgnoreMissing(requestId, accountPublicIds);
        return accountKeys
                .mapLeft(KmsError::getError);
    }
}
