    @Override
    public List<PKBPerson> getSymptomReminderEmailAddressBatch(int batchSize, long minPersonId) {
        try {
            String sql = " SELECT DISTINCT iu.personId " +
                    " FROM InstituteUserEntity iu" +
                    " LEFT JOIN ReminderOptOut ro ON iu.personId = ro.personId" +
                    " LEFT JOIN iu.institute t" +
                    " WHERE ro.id is null" +
                    " AND iu.sponsorshipStatus = :sponsorshipStatus " +
                    " AND t.sendSymptomReminders = true " +
                    " AND iu.personId > :minPersonId " +
                    " ORDER BY iu.personId ";

            TypedQuery<Long> targetPersonIdsQuery = em.createQuery(sql, Long.class);

            em.flush();

            List<Long> targetPersonIds = targetPersonIdsQuery
                    .setParameter("sponsorshipStatus", SponsorshipStatus.ACTIVE)
                    .setParameter("minPersonId", minPersonId)
                    .getResultList();

            return personBean.findContactablePatients(targetPersonIds, batchSize);
        } catch (Exception e) {
            throw new PKBPluginException(
                    "Error while getting the email addresses for symtom reminder emails. ", e);

        }
    }

    @NotNull
    private List<EHRMessageDTO> getMessageDTOS(int batchSize, int offset, List<Route> routes, LocalDate dateToCheck) {
        var ehrDatas = io.vavr.collection.List.ofAll(Some(em.createQuery(
                "SELECT msg FROM EHRData msg"
                + " WHERE msg.route IN :routes"
                + "     AND msg.dataType IN :messageDateTypes"
                + "     AND msg.string04 = 'UNREAD'"
                + "     AND msg.deleted = false"
                + "     AND msg.persistedDate >= :received"
                + "     AND msg.persistedDate < :dayAfter" +
                Option.when(routes.contains(Route.HL7_API), () ->
                          " AND msg.string14 IN :documentTypes"
                ).getOrElse("")
                + "       ORDER BY msg.id", EHRData.class)
                .setParameter("routes", routes)
                .setParameter("messageDateTypes", newArrayList(MS_PATH_MESSAGE, MS_PATH_ENCOUNTER_EVENT))
                .setParameter("received", Date.from(dateToCheck.atStartOfDay().atZone(ZoneId.systemDefault()).toInstant()))
                .setParameter("dayAfter", Date.from(dateToCheck.plusDays(1).atStartOfDay().atZone(ZoneId.systemDefault()).toInstant())))
                .map(q -> Option.when(batchSize > 0, () ->
                            q.setMaxResults(batchSize)
                             .setFirstResult(offset))
                        .getOrElse(q))
                .map(q -> Option.when(routes.contains(Route.HL7_API), () -> 
                            q.setParameter("documentTypes", Arrays.stream(Message.DocumentType.values()).map(Enum::name).collect(Collectors.toList())))
                         .getOrElse(q))
                .get()
                .getResultList());

        // Find the owners of the accounts for this data, and filter them
        var accountOwnersFiltered = io.vavr.collection.HashMap.ofAll(personBean.getAccountOwners(ehrDatas.map(EHRData::getAccountId).toJavaList(),
                PKBPerson.Lazy.CONTACTS, PKBPerson.Lazy.NATIONAL_AND_LOCAL_IDS_DEEP))
                .filter((ignored, person) -> 
                        person.isPatient()
                     && person.getStatus() == UserStatus.EMAIL_CONFIRMED);
        
        // Filter the EHRDatas same as the owners.
        var ehrDatasFiltered = ehrDatas.filter(ed -> accountOwnersFiltered.containsKey(ed.getAccountId()));

        // Find the source persons
        var sourcePersonMap = io.vavr.collection.HashMap.ofAll(personBean.findPKBPersonMap(ehrDatasFiltered.map(EHRData::getSourcePersonId).asJava()));
        
        return ehrDatasFiltered.map(ehrData -> {
            var ownerPerson = accountOwnersFiltered.get(ehrData.getAccountId()).getOrElseThrow(() -> new RuntimeException(String.format("Failed to find account owner for EHRData %d", ehrData.getId())));
            var sourcePerson = sourcePersonMap.get(ehrData.getSourcePersonId());
            Org sourceOrg = ehrData.getSourceOrg();
            Team sourceTeam = ehrData.getSourceTeam();
            return (EHRMessageDTO)Some(ImmutableEHRMessageDTO.builder()
                    .messageId(ehrData.getId())
                    .receiver(ownerPerson)
                    .conversationId(ehrData.getUuid01())
                    .route(ehrData.getRoute())
                    .org(Option.of(sourceOrg))
                    .team(sourceTeam)
                    .received(ehrData.getDate01().toInstant().atZone(UTC).toLocalDateTime()))
                    .map(b -> sourcePerson.map(p -> 
                            b.sourceTitle(p.getTitle())
                             .sourceFirstName(p.getFirstName())
                             .sourceLastName(p.getLastName())
                        ).getOrElse(b))
                    .map(b -> Optional.ofNullable(sourceOrg).map(o ->
                            b.sourceOrg(o.getName()))
                             .orElse(b))
                    .get()
                    .build();
        }).asJava();
    }

    @Override
    public List<PKBPerson> claimChildBirthdayEmailRecipients(Instant birthdayFrom, Instant birthdayTo, PropertyName reminderPropertyName) {
        try {
            Set<Long> patientIds = new HashSet<>(personBean.findPKBPersonIdListByUserTypeAndStatusDobIntervalWithoutBirthdayEmailSent(
                    Collections.singletonList(UserType.PATIENT),
                    Arrays.asList(UserStatus.CREATED, UserStatus.EMAIL_CONFIRMED, UserStatus.NOCONTACT),
                    birthdayFrom, birthdayTo, reminderPropertyName));

            List<Long> activePatientIds = DbUtil.getInBatches(patientIds, batchOfIds -> em.createQuery(
                    "SELECT DISTINCT pc.patientPersonId" +
                            " FROM PatientConsent pc" +
                            " JOIN InstituteUserEntity iu ON pc.patientPersonId = iu.personId" +
                            " WHERE iu.personId IN (:patientIds)" +
                            "  AND iu.sponsorshipStatus = 'ACTIVE'" +
                            "  AND pc.accessingEntityType = 'CARER'",
                    Long.class)
                    .setParameter("patientIds", batchOfIds)
                    .getResultList());

            Map<Long, PKBPerson> patientMap = personBean.findPKBPersonMap(activePatientIds, PKBPerson.Lazy.CONTACTS);

            List<PKBPerson> result = new ArrayList<>();

            activePatientIds.forEach(patientId -> {
                PKBPerson patient = patientMap.get(patientId);
                if (patient != null) {
                    personBean.setChildBirthdayEmailProperty(patient, reminderPropertyName);
                    result.add(patient);
                }
            });
            return result;
        } catch (Exception e) {
            throw new RuntimeException("Error while getting the email addresses for child birthday reminder emails.", e);
        }
    }

    @Override
    public boolean optOut(PKBPerson person) {

        if (isOptedIn(person.getId())) {

            LOGGER.debug("writing opt out record for {}", person.getId());
            //write an opt out record for the person
            ReminderOptOut optOut = new ReminderOptOut();
            optOut.setDateOptedOut(Date.from(dateTimeService.now()));
            optOut.setPersonId(person.getId());
            try {
                em.persist(optOut);

            } catch (Exception exception) {

                throw new PKBPluginException("Exception while creating opt out record", exception);
            }
            return true;
        }
        LOGGER.debug("can't opt out person: {}. They have already opted out", person.getId());
        return false;

    }

    @Override
    public boolean optIn(PKBPerson person) {
        try {

            if (!isOptedIn(person.getId())) {
                String select =

                        "SELECT r FROM ReminderOptOut r " + "WHERE r.personId = '" + person.getId() + "'";

                LOGGER.debug("removing opt out record for {}", person.getId());

                Query query = em.createQuery(select);
                ReminderOptOut record = (ReminderOptOut) query.getSingleResult();
                em.remove(record);
                return true;
            }
        } catch (Exception e) {
            throw new PKBPluginException("Error while looking up opt out records. ", e);

        }
        LOGGER.debug("Can't opt in person: {}. They have no opt-out record to remove", person.getId());
        return false;

    }

    @Override
    public boolean isOptedIn(Long personId) {
        // TODO Auto-generated method stub
        try {
            String select =

                    "SELECT r FROM ReminderOptOut r " + "WHERE r.personId = '" + personId + "'";

            Query query = em.createQuery(select);
            return query.getResultList().isEmpty();
        } catch (Exception e) {
            throw new PKBPluginException("Error while looking up opt out records. ", e);

        }
    }
