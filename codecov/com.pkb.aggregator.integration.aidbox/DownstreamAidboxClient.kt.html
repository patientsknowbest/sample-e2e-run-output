package com.pkb.aggregator.integration.aidbox

import ca.uhn.fhir.context.FhirContext
import ca.uhn.fhir.rest.server.exceptions.UnprocessableEntityException
import com.pkb.aggregator.R4Medication
import com.pkb.aggregator.R4Organization
import com.pkb.aggregator.R4Patient
import com.pkb.aggregator.R4Person
import com.pkb.aggregator.R4Practitioner
import com.pkb.aggregator.R4PractitionerRole
import com.pkb.aggregator.extension.aggregatedResourceId
import com.pkb.aggregator.service.DownstreamClient
import com.pkb.aggregator.service.exception.DownstreamResourceExistsException
import com.pkb.aggregator.service.exception.DownstreamVersionConflictException
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import org.hl7.fhir.instance.model.api.IDomainResource
import org.hl7.fhir.instance.model.api.IIdType
import org.hl7.fhir.r4.model.OperationOutcome
import org.slf4j.LoggerFactory
import org.springframework.beans.factory.annotation.Qualifier
import org.springframework.http.HttpHeaders.CONTENT_TYPE
import org.springframework.http.MediaType.APPLICATION_JSON_VALUE
import org.springframework.stereotype.Service
import retrofit2.Response
import java.lang.invoke.MethodHandles

@Service
class DownstreamAidboxClient(
    private val hapiAidboxRestClient: HapiAidboxRestClient,
    private val downstreamDeleter: SqlHardDeleter,
    @Qualifier("fhirContextR4") private val fhirContextR4: FhirContext,
) : DownstreamClient {

    private val log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun create(resource: IDomainResource): Any? {
        val call = when (resource) {
            is R4Patient -> createPatient(resource)
            is R4Organization -> createOrganization(resource)
            is R4Practitioner -> createPractitioner(resource)
            is R4PractitionerRole -> createPractitionerRole(resource)
            is R4Person -> createPerson(resource)
            is R4Medication -> createMedication(resource)
            else -> throw UnexpectedTypeException(resource)
        }

        val response = call.execute()
        if (response.isSuccessful) {
            log.info("Created resource ${resource.aggregatedResourceId()}")
        } else {
            handleErrorResponse(resource, response)
        }
        return resource
    }

    override fun update(resource: IDomainResource): Any? {
        val call = resource.meta.versionId?.let {
            when (resource) {
                is R4Practitioner -> optimisticPractitionerUpdate(it, resource)
                is R4Organization -> optimisticOrganizationUpdate(it, resource)
                is R4Patient -> optimisticPatientUpdate(it, resource)
                is R4Person -> optimisticPersonUpdate(it, resource)
                is R4Medication -> optimisticMedicationUpdate(it, resource)
                is R4PractitionerRole -> optimisticPractitionerRoleUpdate(it, resource)
                else -> throw UnexpectedTypeException(resource)
            }
        } ?: throw RuntimeException("Cannot safely update ${resource.idElement} with no version")

        val response = call.execute()
        if (response.isSuccessful) {
            log.info("Updated resource ${resource.aggregatedResourceId()}")
        } else {
            handleErrorResponse(resource, response)
        }
        return resource
    }

    override fun softDelete(resource: IDomainResource) {
        val call = resource.meta.versionId?.let {
            when (resource) {
                is R4Patient -> optimisticPatientDelete(it, resource)
                is R4Organization -> optimisticOrganizationDelete(it, resource)
                is R4Practitioner -> optimisticPractitionerDelete(it, resource)
                is R4PractitionerRole -> optimisticPractitionerRoleDelete(it, resource)
                is R4Medication -> optimisticMedicationDelete(it, resource)
                is R4Person -> optimisticPersonDelete(it, resource)
                else -> throw UnexpectedTypeException(resource)
            }
        } ?: throw RuntimeException("Cannot safely delete ${resource.idElement} with no version")
        val response = call.execute()
        if (response.isSuccessful) {
            log.info("Soft-deleted resource ${resource.aggregatedResourceId()}")
        } else {
            handleErrorResponse(resource, response)
        }
    }

    override fun softDelete(resourceId: IIdType) {
        val call = genericResourceDelete(resourceId)
        val response = call.execute()
        if (response.isSuccessful) {
            log.info("Soft-deleted resource ${resourceId.resourceType}/${resourceId.idPart}")
        } else {
            handleErrorResponse("${resourceId.resourceType}/${resourceId.idPart}", response)
        }
    }

    override fun <T> hardDelete(id: String, klass: Class<T>): List<Pair<String, Any?>> where T : IDomainResource =
        downstreamDeleter.deleteResource(id, klass)

    private fun createPractitioner(resource: IDomainResource) =
        hapiAidboxRestClient.practitionerCreate(resource)

    private fun createPractitionerRole(resource: IDomainResource) =
        hapiAidboxRestClient.practitionerRoleCreate(resource)

    private fun createPerson(resource: IDomainResource) =
        hapiAidboxRestClient.personCreate(resource)

    private fun createOrganization(resource: IDomainResource) =
        hapiAidboxRestClient.organizationCreate(resource)

    private fun createPatient(resource: IDomainResource) =
        hapiAidboxRestClient.patientCreate(resource)

    private fun createMedication(resource: IDomainResource) =
        hapiAidboxRestClient.medicationCreate(resource)

    private fun optimisticPractitionerUpdate(versionId: String, resource: IDomainResource) =
        hapiAidboxRestClient.optimisticPractitionerUpdate(versionId, resource.idElement.idPart, resource)

    private fun optimisticPractitionerDelete(versionId: String, resource: IDomainResource) =
        hapiAidboxRestClient.optimisticPractitionerDelete(versionId, resource.idElement.idPart)

    private fun optimisticPersonUpdate(versionId: String, resource: IDomainResource) =
        hapiAidboxRestClient.optimisticPersonUpdate(versionId, resource.idElement.idPart, resource)

    private fun optimisticPersonDelete(versionId: String, resource: IDomainResource) =
        hapiAidboxRestClient.optimisticPersonDelete(versionId, resource.idElement.idPart)

    private fun optimisticOrganizationUpdate(versionId: String, resource: IDomainResource) =
        hapiAidboxRestClient.optimisticOrganizationUpdate(versionId, resource.idElement.idPart, resource)

    private fun optimisticOrganizationDelete(versionId: String, resource: IDomainResource) =
        hapiAidboxRestClient.optimisticOrganizationDelete(versionId, resource.idElement.idPart)

    private fun optimisticPatientUpdate(versionId: String, resource: IDomainResource) =
        hapiAidboxRestClient.optimisticPatientUpdate(versionId, resource.idElement.idPart, resource)

    private fun optimisticPatientDelete(versionId: String, resource: IDomainResource) =
        hapiAidboxRestClient.optimisticPatientDelete(versionId, resource.idElement.idPart)

    private fun optimisticMedicationUpdate(versionId: String, resource: IDomainResource) =
        hapiAidboxRestClient.optimisticMedicationUpdate(versionId, resource.idElement.idPart, resource)

    private fun optimisticMedicationDelete(versionId: String, resource: IDomainResource) =
        hapiAidboxRestClient.optimisticMedicationDelete(versionId, resource.idElement.idPart)

    private fun optimisticPractitionerRoleUpdate(versionId: String, resource: IDomainResource) =
        hapiAidboxRestClient.optimisticPractitionerRoleUpdate(versionId, resource.idElement.idPart, resource)

    private fun optimisticPractitionerRoleDelete(versionId: String, resource: IDomainResource) =
        hapiAidboxRestClient.optimisticPractitionerRoleDelete(versionId, resource.idElement.idPart)

    private fun genericResourceDelete(id: IIdType) =
        hapiAidboxRestClient.genericResourceDelete(
            "/fhir/${id.resourceType}",
            id.resourceType,
            id.idPart
        )

    private fun handleErrorResponse(resource: IDomainResource, response: Response<Any>): Nothing =
        handleErrorResponse(resource.aggregatedResourceId(), response)

    private fun handleErrorResponse(resourceId: String, response: Response<Any>): Nothing {

        val contentType = response.headers()[CONTENT_TYPE]
        if (contentType == APPLICATION_JSON_VALUE) {
            val errorBody = response.errorBody()?.string()
            if (errorBody != null) {
                val outcome = fhirContextR4.newJsonParser().parseResource(OperationOutcome::class.java, errorBody)
                if (outcome.hasIssue()) {
                    val issue = outcome.issueFirstRep
                    when (issue.code) {
                        OperationOutcome.IssueType.INVALID -> throw UnprocessableEntityException(issue.diagnostics, outcome)
                        OperationOutcome.IssueType.DUPLICATE -> throw DownstreamResourceExistsException(issue.diagnostics)
                        OperationOutcome.IssueType.CONFLICT -> throw DownstreamVersionConflictException(issue.diagnostics, response.code())
                        else -> throw RuntimeException("Error processing resourceId: $errorBody")
                    }
                }
            }
        }

        throw RuntimeException("Non-json content-type when processing $resourceId: $contentType")
    }
}
