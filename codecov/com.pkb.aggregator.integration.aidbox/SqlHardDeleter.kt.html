package com.pkb.aggregator.integration.aidbox

import com.fasterxml.jackson.annotation.JsonRootName
import com.fasterxml.jackson.databind.ObjectMapper
import com.fasterxml.jackson.databind.SerializationFeature
import com.pkb.aggregator.R4Medication
import com.pkb.aggregator.R4Organization
import com.pkb.aggregator.R4Patient
import com.pkb.aggregator.R4Practitioner
import com.pkb.aggregator.service.FHIR_TYPE_OBSERVATION
import com.pkb.aggregator.service.FHIR_TYPE_PATIENT
import org.hl7.fhir.instance.model.api.IDomainResource

/**
 * Issues a query to aidbox's /$sql endpoint to delete the specified resources
 *
 * We do this rather than sending a DELETE HTTP request because that would build a false modification history on the
 * downstream - i.e. a clinical data item hasn't been deleted by a clinical team, but we need to delete it prior to
 * sending a freshly-reaggregated patient account. We don't want that delete to be logged downstream because it's just
 * a side effect of our business process rather than part of the actual medical record history
 */
class SqlHardDeleter(
    private val aidboxRestClient: AidboxRestClient,
    objectMapper: ObjectMapper
) {

    private val objectMapper = objectMapper.copy().apply {
        enable((SerializationFeature.WRAP_ROOT_VALUE))
    }

    @JsonRootName(value = "subject")
    data class Subject(val resourceType: String, val id: String)

    @JsonRootName(value = "patient")
    data class Patient(val resourceType: String = FHIR_TYPE_PATIENT, val id: String)

    @JsonRootName(value = "organization")
    data class Organization(val resourceType: String = FHIR_TYPE_OBSERVATION, val id: String)

    fun <T> deleteResource(id: String, klass: Class<T>) where T: IDomainResource =
        when (klass) {
            R4Patient::class.java -> deletePatient(id)
            R4Organization::class.java -> deleteOrganization(id)
            R4Practitioner::class.java -> deletePractitioner(id)
            R4Medication::class.java -> deleteMedication(id)
            else -> throw RuntimeException("Cannot delete resources of type $klass")
        }

    private fun deletePatient(downstreamPatientId: String): List<Pair<String, Any?>> {
        val subject = objectMapper.writeValueAsString(Subject(FHIR_TYPE_PATIENT, downstreamPatientId))!!
        val patient = objectMapper.writeValueAsString(Patient(FHIR_TYPE_PATIENT, downstreamPatientId))!!
        return listOf(
            "encounter" to aidboxRestClient.postSql(
                arrayOf(DELETE_ENCOUNTER, subject)
            ).execute().body(),
            "encounter_history" to aidboxRestClient.postSql(
                arrayOf(DELETE_ENCOUNTER_HISTORY, subject)
            ).execute().body(),
            "observation" to aidboxRestClient.postSql(
                arrayOf(DELETE_OBSERVATION, subject)
            ).execute().body(),
            "observation_history" to aidboxRestClient.postSql(
                arrayOf(DELETE_OBSERVATION_HISTORY, subject)
            ).execute().body(),
            "medicationstatement" to aidboxRestClient.postSql(
                arrayOf(DELETE_MEDICATION_STATEMENT, subject)
            ).execute().body(),
            "medicationstatement_history" to aidboxRestClient.postSql(
                arrayOf(DELETE_MEDICATION_STATEMENT_HISTORY, subject)
            ).execute().body(),
            "condition" to aidboxRestClient.postSql(
                arrayOf(DELETE_CONDITION, subject)
            ).execute().body(),
            "condition_history" to aidboxRestClient.postSql(
                arrayOf(DELETE_CONDITION_HISTORY, subject)
            ).execute().body(),
            "allergyintolerance" to aidboxRestClient.postSql(
                arrayOf(DELETE_ALLERGY, patient)
            ).execute().body(),
            "allergyintolerance_history" to aidboxRestClient.postSql(
                arrayOf(DELETE_ALLERGY_HISTORY, patient)
            ).execute().body(),
            "appointment" to aidboxRestClient.postSql(
                arrayOf(DELETE_APPOINTMENT, downstreamPatientId)
            ).execute().body(),
            "appointment_history" to aidboxRestClient.postSql(
                arrayOf(DELETE_APPOINTMENT_HISTORY, downstreamPatientId)
            ).execute().body(),
            "patient" to aidboxRestClient.postSql(
                arrayOf(DELETE_PATIENT, downstreamPatientId)
            ).execute().body(),
            "patient_history" to aidboxRestClient.postSql(
                arrayOf(DELETE_PATIENT_HISTORY, downstreamPatientId)
            ).execute().body(),
            "person" to aidboxRestClient.postSql(
                arrayOf(DELETE_PERSON, downstreamPatientId)
            ).execute().body(),
            "person_history" to aidboxRestClient.postSql(
                arrayOf(DELETE_PERSON_HISTORY, downstreamPatientId)
            ).execute().body(),
        )
    }

    private fun deleteOrganization(downstreamOrganizationId: String) =
        objectMapper.writeValueAsString(Organization(id = downstreamOrganizationId)).let {
            listOf(
                "practitionerrole" to aidboxRestClient.postSql(
                    arrayOf(DELETE_PRACTITIONER_ROLE, it)
                ).execute().body(),
                "practitionerrole_history" to aidboxRestClient.postSql(
                    arrayOf(DELETE_PRACTITIONER_ROLE_HISTORY, it)
                ).execute().body(),
                "organization" to aidboxRestClient.postSql(
                    arrayOf(DELETE_ORGANIZATION, downstreamOrganizationId)
                ).execute().body(),
                "organization_history" to aidboxRestClient.postSql(
                    arrayOf(DELETE_ORGANIZATION_HISTORY, downstreamOrganizationId)
                ).execute().body(),
            )
        }

    private fun deletePractitioner(downstreamPractitionerId: String) = listOf(
        "practitioner" to aidboxRestClient.postSql(
            arrayOf(DELETE_PRACTITIONER, downstreamPractitionerId)
        ).execute().body(),
        "practitioner_history" to aidboxRestClient.postSql(
            arrayOf(DELETE_PRACTITIONER_HISTORY, downstreamPractitionerId)
        ).execute().body(),
    )

    private fun deleteMedication(downstreamMedicationId: String)= listOf(
        "medication" to aidboxRestClient.postSql(
            arrayOf(DELETE_MEDICATION, downstreamMedicationId)
        ).execute().body(),
        "medication_history" to aidboxRestClient.postSql(
            arrayOf(DELETE_MEDICATION_HISTORY, downstreamMedicationId)
        ).execute().body(),
    )

    companion object {
        const val DELETE_ENCOUNTER = "delete from encounter where resource @> cast(? as jsonb) returning id"
        const val DELETE_ENCOUNTER_HISTORY = "delete from encounter_history where resource @> cast(? as jsonb) returning id"
        const val DELETE_OBSERVATION = "delete from observation where resource @> cast(? as jsonb) returning id"
        const val DELETE_OBSERVATION_HISTORY = "delete from observation_history where resource @> cast(? as jsonb) returning id"
        const val DELETE_MEDICATION_STATEMENT = "delete from medicationstatement where resource @> cast(? as jsonb) returning id"
        const val DELETE_MEDICATION_STATEMENT_HISTORY = "delete from medicationstatement_history where resource @> cast(? as jsonb) returning id"
        const val DELETE_CONDITION = "delete from condition where resource @> cast(? as jsonb) returning id"
        const val DELETE_CONDITION_HISTORY = "delete from condition_history where resource @> cast(? as jsonb) returning id"
        const val DELETE_APPOINTMENT = """delete from appointment a1 where a1.id in (select a2.id from appointment a2, jsonb_array_elements(a2.resource->'participant') p where p->'actor'->>'id' = ?)"""
        const val DELETE_APPOINTMENT_HISTORY = """delete from appointment_history a1 where a1.id in (select a2.id from appointment a2, jsonb_array_elements(a2.resource->'participant') p where p->'actor'->>'id' = ?)"""
        const val DELETE_ALLERGY = "delete from allergyintolerance where resource @> cast(? as jsonb) returning id"
        const val DELETE_ALLERGY_HISTORY = "delete from allergyintolerance_history where resource @> cast(? as jsonb) returning id"
        const val DELETE_PATIENT = "delete from patient where id = ? returning id"
        const val DELETE_PATIENT_HISTORY = "delete from patient_history where id = ? returning id"
        const val DELETE_PERSON = "delete from person p1 where p1.id in (select p2.id from person p2, jsonb_array_elements(p2.resource->'link') p where p->'target'->>'id' = ?) returning id"
        const val DELETE_PERSON_HISTORY = "delete from person_history p1 where p1.id in (select p2.id from person p2, jsonb_array_elements(p2.resource->'link') p where p->'target'->>'id' = ?) returning id"

        const val DELETE_PRACTITIONER = "delete from practitioner where id = ? returning id"
        const val DELETE_PRACTITIONER_HISTORY = "delete from practitioner_history where id = ? returning id"
        const val DELETE_PRACTITIONER_ROLE = "delete from practitionerrole where resource @> cast(? as jsonb) returning id"
        const val DELETE_PRACTITIONER_ROLE_HISTORY = "delete from practitionerrole_history where resource @> cast(? as jsonb) returning id"
        const val DELETE_ORGANIZATION = "delete from organization where id = ? returning id"
        const val DELETE_ORGANIZATION_HISTORY = "delete from organization_history where id = ? returning id"

        const val DELETE_MEDICATION = "delete from medication where id = ? returning id"
        const val DELETE_MEDICATION_HISTORY = "delete from medication_history where id = ? returning id"

    }
}
