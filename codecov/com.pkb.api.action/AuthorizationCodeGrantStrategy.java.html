package com.pkb.api.action;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.api.error.accesstoken.AccessTokenRequestError;
import com.pkb.datamodel.ApiAuthSession;
import com.pkb.service.user.impl.ApiAuthManager;
import io.vavr.control.Either;

import javax.servlet.http.HttpServletRequest;

import static com.google.common.base.Preconditions.checkNotNull;
import static com.pkb.api.action.AccessTokenIssuer.requireMandatoryParameter;
import static com.pkb.common.api.error.accesstoken.AccessTokenRequestError.ErrorCode.INVALID_REQUEST;
import static io.vavr.control.Either.left;

public class AuthorizationCodeGrantStrategy implements AccessTokenIssuingStrategy {

    private final ApiAuthManager authManager;

    AuthorizationCodeGrantStrategy(ApiAuthManager authManager) {
        this.authManager = checkNotNull(authManager);
    }

    @Override public Either<AccessTokenRequestError, ApiAuthSession> obtainSession(HttpServletRequest request, EHRRequestContext context) {
        if (request == null) {
            return left(new AccessTokenRequestError(INVALID_REQUEST,"Request cannot be null"));
        }

        Either<AccessTokenRequestError, String> authCode = requireMandatoryParameter(request, "code");
        if (authCode.isLeft()) {
            return left(authCode.getLeft());
        }
        Either<AccessTokenRequestError, String> clientId = requireMandatoryParameter(request, "client_id");
        if (clientId.isLeft()) {
            return left(clientId.getLeft());
        }
        return authManager.useAuthCode(clientId.get(), authCode.get());
    }
}
