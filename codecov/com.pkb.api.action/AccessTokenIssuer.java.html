package com.pkb.api.action;

import com.pkb.admin.action.AdminInstitutionAction;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.api.error.accesstoken.AccessTokenRequestError;
import com.pkb.common.api.error.accesstoken.InvalidHTTPMethodError;
import com.pkb.common.api.error.accesstoken.MissingRequestParameterError;
import com.pkb.common.api.error.accesstoken.UnsupportedGrantTypeError;
import com.pkb.common.config.PhrConfig;
import com.pkb.datamodel.ApiAuthSession;
import com.pkb.datamodel.ApiClient;
import com.pkb.security.PrincipalCredentialAuthenticator;
import com.pkb.service.user.impl.ApiAuthManager;
import com.pkb.service.user.impl.UserManager;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import javax.servlet.http.HttpServletRequest;
import java.util.HashMap;
import java.util.Map;

import static com.pkb.api.action.PredicateWrapperParameterChecker.checkValueOf;
import static com.pkb.common.api.error.accesstoken.AccessTokenRequestError.ErrorCode.INVALID_REQUEST;
import static com.pkb.datamodel.ApiAuthSession.SECRET_SEPARATOR;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;
import static org.apache.commons.lang3.StringUtils.EMPTY;

public class AccessTokenIssuer {
    public static class Factory {

        private final ApiAuthManager authManager;
        private final UserManager userManager;
        private final PhrConfig config;

        public Factory(ApiAuthManager authManager,
                       UserManager userManager,
                       @NotNull PhrConfig config) {
            this.authManager = authManager;
            this.config = config;
            this.userManager = userManager;
        }

        public AccessTokenIssuer create(PrincipalCredentialAuthenticator idPasswordAuthenticator) {
            return new AccessTokenIssuer(authManager, userManager, config, idPasswordAuthenticator);
        }

    }

    static Map<String, AccessTokenIssuingStrategy> supportedStrategies(
            ApiAuthManager authManager,
            UserManager userManager,
            PrincipalCredentialAuthenticator authenticator) {
        Map<String, AccessTokenIssuingStrategy> strategies = new HashMap<>();
        strategies.put("authorization_code", new AuthorizationCodeGrantStrategy(authManager));
        strategies.put("refresh_token", new RefreshTokenGrantStrategy(authManager));
        strategies.put("password", new ResourceOwnerCredentialsGrantStrategy(authManager, userManager, authenticator));
        return strategies;
    }

    static Either<AccessTokenRequestError, String> requireMandatoryParameter(@NotNull HttpServletRequest request, @NotNull ParameterChecker parameterChecker) {
        String parameter = request.getParameter(parameterChecker.parameterName());

        if (parameter == null) {
            return left(new MissingRequestParameterError(parameterChecker.parameterName()));
        }

        if (parameterChecker.failsTest(parameter)) {
            return left(new AccessTokenRequestError(INVALID_REQUEST, parameterChecker.description()));
        }

        return right(parameter);
    }

    static Either<AccessTokenRequestError, String> requireMandatoryParameter(HttpServletRequest request, String parameterName) {
        return requireMandatoryParameter(request, checkValueOf(parameterName, $ -> true, EMPTY));
    }

    private final Map<String, AccessTokenIssuingStrategy> strategies;

    public AccessTokenIssuer(
            ApiAuthManager authManager,
            UserManager userManager,
            @NotNull PhrConfig config,
            @NotNull PrincipalCredentialAuthenticator authenticator) {
        this(supportedStrategies(authManager, userManager, authenticator));
    }

    AccessTokenIssuer(@NotNull Map<String, AccessTokenIssuingStrategy> strategies) {
        this.strategies = Map.copyOf(strategies);
    }

    public Either<AccessTokenRequestError, AccessToken> createForRequest(HttpServletRequest request, EHRRequestContext context) {
        String actualMethod = request.getMethod();
        if (!"POST".equalsIgnoreCase(actualMethod)) {
            return left(new InvalidHTTPMethodError(actualMethod));
        }

        Either<AccessTokenRequestError, String> grantType = requireMandatoryParameter(request, "grant_type");

        if (grantType.isRight()) {

            AccessTokenIssuingStrategy strategy = strategies.get(
                    grantType.get());

            if (strategy == null) {
                return left(new UnsupportedGrantTypeError(grantType.get()));
            }

            Either<AccessTokenRequestError, ApiAuthSession> sessionOrError = strategy.obtainSession(request, context);

            if (sessionOrError.isRight()) {
                ApiAuthSession session = sessionOrError.get();
                ImmutableAccessToken.Builder rval = new ImmutableAccessToken.Builder();
                populateRefreshToken(rval, session);
                return right(rval.tokenType("Bearer")
                        .scope(session.getScope().toString().toLowerCase())
                        .expiresIn(session.getApiClient().getTokenExpirySeconds())
                        .accessToken(session.getAccessTokenPart()
                                .orElseThrow(() -> new IllegalStateException("Missing access token part"))
                                + SECRET_SEPARATOR + session.getNewSecret()
                                .orElseThrow(() -> new IllegalStateException("Missing new secret")))
                        .build());
            }
            return left(sessionOrError.getLeft());
        }
        return left(grantType.getLeft());
    }

    private void populateRefreshToken(ImmutableAccessToken.Builder accessToken, ApiAuthSession session) {
        if (needsRefreshToken(session)) {
            String refreshToken = session.getRefreshTokenPart()
                    .orElseThrow(() -> new IllegalStateException("Missing refresh token part"))
                    + SECRET_SEPARATOR + session.getNewSecret()
                    .orElseThrow(() -> new IllegalStateException("Missing new secret"));
            accessToken.comments("").refreshToken(refreshToken);
        } else {
            accessToken.refreshToken("").comments(AdminInstitutionAction.NO_REFRESH);
        }
    }

    private boolean needsRefreshToken(ApiAuthSession session) {
        ApiClient apiClient = session.getApiClient();
        return apiClient.hasTokenExpiry() && (!apiClient.hasSessionExpiry() || !accessTokenExpiresSession(session));
    }

    private boolean accessTokenExpiresSession(ApiAuthSession session) {
        ApiClient apiClient = session.getApiClient();
        long accessEnd = session.getAccessTokenIssued().orElseThrow(() -> new IllegalStateException("Missing time access token issued"))
                .toEpochMilli() + (apiClient.getTokenExpirySeconds() * 1000);
        long sessionEnd = session.getAuthCodeIssued().orElseThrow(() -> new IllegalStateException("Missing time auth code issued"))
                .toEpochMilli() + (apiClient.getSessionExpirySeconds() * 1000);
        return sessionEnd < accessEnd;
    }

}
