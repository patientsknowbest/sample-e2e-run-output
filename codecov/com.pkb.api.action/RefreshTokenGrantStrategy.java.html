package com.pkb.api.action;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.api.error.accesstoken.AccessTokenRequestError;
import com.pkb.datamodel.ApiAuthSession;
import com.pkb.service.user.impl.ApiAuthManager;
import io.vavr.control.Either;
import org.apache.commons.lang3.StringUtils;

import javax.servlet.http.HttpServletRequest;

import static com.pkb.api.action.AccessTokenIssuer.requireMandatoryParameter;
import static com.pkb.api.action.PredicateWrapperParameterChecker.checkValueOf;

public class RefreshTokenGrantStrategy implements AccessTokenIssuingStrategy {
    private static final String CANNOT_BE_BLANK = "cannot be blank";
    private static final String REFRESH_TOKEN = "refresh_token";
    private static final String CLIENT_ID = "client_id";

    private final ApiAuthManager authenticationManager;

    public RefreshTokenGrantStrategy(ApiAuthManager authenticationManager) {
        this.authenticationManager = authenticationManager;
    }

    @Override
    public Either<AccessTokenRequestError, ApiAuthSession> obtainSession(HttpServletRequest request, EHRRequestContext context) {
        return requireMandatoryParameter(request, checkValueOf(REFRESH_TOKEN, StringUtils::isNotBlank, CANNOT_BE_BLANK)).flatMap(
                refreshToken -> requireMandatoryParameter(request, checkValueOf(CLIENT_ID, StringUtils::isNotBlank, CANNOT_BE_BLANK)).flatMap(
                        clientId -> attemptToRefreshToken(clientId, refreshToken, request.getParameter("scope"))));
    }

    private Either<AccessTokenRequestError, ApiAuthSession> attemptToRefreshToken(String clientId, String refreshToken, String scope) {
        return authenticationManager.useRefreshToken(clientId, refreshToken, scope);
    }
}
