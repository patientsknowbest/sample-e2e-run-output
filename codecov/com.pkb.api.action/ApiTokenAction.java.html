package com.pkb.api.action;

import com.pkb.action.BaseAction;
import com.pkb.common.api.error.accesstoken.AccessTokenRequestError;
import com.pkb.security.IdPasswordAuthenticator;
import com.pkb.security.LoginAttemptAuditor;
import io.vavr.control.Either;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import java.io.ByteArrayInputStream;
import java.io.ByteArrayOutputStream;
import java.io.IOException;
import java.io.InputStream;
import java.net.URLEncoder;

import static java.nio.charset.StandardCharsets.UTF_8;

/**
 * OAuth 2 process:
 * Docs here: https://sites.google.com/a/patientsknowbest.com/technology/Home/technologies/oauth-v20/oauth-20-the-good-bits/
 *
 * @author robwhelan
 */
public class ApiTokenAction extends BaseAction {
    private static final long serialVersionUID = 1L;

    static final String SUCCESS_STREAM = "successStream";

    static final String ERROR_STREAM = "errorStream";

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    // incoming params

    /**
     * authorization_code or refresh_token
     */
    private String grant_type;

    private String client_id;

    private String code;

    private String redirect_uri;

    private String refresh_token;

    private String scope;

    private InputStream responseInputStream;

    private int responseContentLength;

    /**
     * the full URI (including our response) that we're sending back to the end-user
     */
    private String returnRedirectUri;

    @Autowired
    private LoginAttemptAuditor.Factory loginAttemptAuditorFactory;

    @Autowired
    private IdPasswordAuthenticator.Factory idPasswordAuthenticatorFactory;

    @Autowired
    private AccessTokenIssuer.Factory accessTokenIssuerFactory;

    private void writeAsJsonToOutput(Class<?> clazz, Object obj) throws IOException {
        ByteArrayOutputStream buffer = new ByteArrayOutputStream();
        OBJECT_MAPPER.writerFor(clazz).writeValue(buffer, obj);
        byte[] rawBuffer = buffer.toByteArray();
        responseContentLength = rawBuffer.length;
        responseInputStream = new ByteArrayInputStream(rawBuffer);
    }

    /**
     * Example POSTs:
     * grant_type=authorization_code&client_id=GIVEN_CLIENT_ID&code=SplxlOBeZQQYbYS6WxSbIA&redirect_uri=https%3A%2F%2Fclient%2Eexample%2Ecom
     * %2Fcb
     * grant_type=refresh_token&client_id=GIVEN_CLIENT_ID&refresh_token=tGzv3JOkF0XG5Qx2TlKWIA&scope=patient_OR_clinician
     */
    public String getApiToken() {
        LoginAttemptAuditor loginAttemptAuditor = loginAttemptAuditorFactory.createAuditor(request.getRemoteAddr());
        IdPasswordAuthenticator authenticator = idPasswordAuthenticatorFactory.createAuthenticator(loginAttemptAuditor);
        AccessTokenIssuer accessTokenIssuer = accessTokenIssuerFactory.create(authenticator);

        try {
            Either<AccessTokenRequestError, AccessToken> accTokenOrError = accessTokenIssuer.createForRequest(request, getEHRRequestContext());
            if (accTokenOrError.isRight()) {
                writeAsJsonToOutput(AccessToken.class, accTokenOrError.get());
                return SUCCESS_STREAM;
            } else {
                writeAsJsonToOutput(AccessTokenRequestError.class, accTokenOrError.getLeft());
                return ERROR_STREAM;
            }
        } catch (IOException ignored) {
            return ERROR_STREAM;
        }
    }

    public void setClient_id(String client_id) {
        this.client_id = client_id;
    }

    public String getClient_id() {
        return client_id;
    }

    public void setScope(String scope) {
        this.scope = scope;
    }

    public void setRedirect_uri(String redirect_uri) {
        this.redirect_uri = redirect_uri;
    }

    public InputStream getResponseInputStream() {
        return responseInputStream;
    }

    public int getResponseContentLength() {
        return responseContentLength;
    }

    public String getReturnRedirectUri() {
        return returnRedirectUri;
    }

    public String getReturnRedirectUriEscaped() {

        return URLEncoder.encode(returnRedirectUri, UTF_8);

    }

    public void setGrant_type(String grant_type) {
        this.grant_type = grant_type;
    }

    public void setCode(String code) {
        this.code = code;
    }

    public void setRefresh_token(String refresh_token) {
        this.refresh_token = refresh_token;
    }

}
