package com.pkb.api.action;

import org.jetbrains.annotations.NotNull;

import java.util.function.Predicate;

import static java.lang.String.format;

public class PredicateWrapperParameterChecker implements ParameterChecker {
    public static ParameterChecker checkValueOf(@NotNull String paramterName, @NotNull Predicate<String> predicate, @NotNull String description) {
        return new PredicateWrapperParameterChecker(predicate, paramterName, description);
    }

    private final String parameterName;
    private final String description;
    private final Predicate<String> predicate;

    private PredicateWrapperParameterChecker(@NotNull Predicate<String> predicate, @NotNull String paramterName, @NotNull String description) {
        this.parameterName = paramterName;
        this.description = description;
        this.predicate = predicate;
    }

    @Override
    public @NotNull String parameterName() {
        return parameterName;
    }

    @Override
    public @NotNull String description() {
        return format("'%s': %s", parameterName, description);
    }

    @Override
    public @NotNull boolean test(String string) {
        return predicate.test(string);
    }
}
