package com.pkb.api.action;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.PersonAuthInfo;
import com.pkb.common.api.error.accesstoken.AccessTokenRequestError;
import com.pkb.common.api.error.accesstoken.BlockedRequestError;
import com.pkb.common.api.error.accesstoken.InaccessibleAccountError;
import com.pkb.common.api.error.accesstoken.InvalidClientIdError;
import com.pkb.common.api.error.accesstoken.InvalidUsernameParameterError;
import com.pkb.common.api.error.accesstoken.WrongPasswordError;
import com.pkb.datamodel.ApiAuthSession;
import com.pkb.security.CoordAwaitingActivationException;
import com.pkb.security.InvalidPrincipalException;
import com.pkb.security.PrincipalCredentialAuthenticator;
import com.pkb.security.RegIncompleteException;
import com.pkb.security.TooManyLoginAttemptsException;
import com.pkb.security.WifiInTheCemeteryException;
import com.pkb.service.user.ApiClientNotFoundException;
import com.pkb.service.user.impl.ApiAuthManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;
import org.springframework.security.authentication.BadCredentialsException;
import org.springframework.security.authentication.DisabledException;

import javax.servlet.http.HttpServletRequest;

import static com.pkb.api.action.AccessTokenIssuer.requireMandatoryParameter;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

public class ResourceOwnerCredentialsGrantStrategy implements AccessTokenIssuingStrategy {

    private final ApiAuthManager authManager;

    private final PrincipalCredentialAuthenticator authenticator;

    private final UserManager userManager;

    public ResourceOwnerCredentialsGrantStrategy(
            ApiAuthManager authManager,
            UserManager userManager,
            @NotNull PrincipalCredentialAuthenticator authenticator
    ) {
        this.authManager = authManager;
        this.authenticator = authenticator;
        this.userManager = userManager;
    }

    @Override
    public Either<AccessTokenRequestError, ApiAuthSession> obtainSession(HttpServletRequest request, EHRRequestContext context) {
        Either<AccessTokenRequestError, String> personId = requireMandatoryParameter(request, "username");
        if (personId.isLeft()) {
            return left(personId.getLeft());
        }
        Either<AccessTokenRequestError, String> password = requireMandatoryParameter(request, "password");
        if (password.isLeft()) {
            return left(password.getLeft());
        }
        Either<AccessTokenRequestError, String> scope = requireMandatoryParameter(request, "scope");
        if (scope.isLeft()) {
            return left(scope.getLeft());
        }
        Either<AccessTokenRequestError, String> clientId = requireMandatoryParameter(request, "client_id");
        if (clientId.isLeft()) {
            return left(clientId.getLeft());
        }
        try {
            PersonAuthInfo user = authenticator.authenticate(context, personId.get(), password.get());
            PKBPerson person = userManager.getPKBPerson(user.getId());
            return right(authManager.createSessionWithIssuedAccessToken(clientId.get(), person, password.get(), scope.get()));
        } catch (ApiClientNotFoundException e) {
            return left(new InvalidClientIdError("API client not found", e));
        } catch (CoordAwaitingActivationException | DisabledException | WifiInTheCemeteryException | RegIncompleteException e) {
            return left(new InaccessibleAccountError(e));
        } catch (InvalidPrincipalException e) {
            return left(new InvalidUsernameParameterError(e.getInvalidPrincipal()));
        } catch (BadCredentialsException e) {
            return left(new WrongPasswordError(e));
        } catch (TooManyLoginAttemptsException e) {
            return left(new BlockedRequestError(e));
        }
    }
}
