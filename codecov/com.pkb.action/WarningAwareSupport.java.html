package com.pkb.action;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.Collection;
import java.util.LinkedList;

import static org.apache.commons.collections.CollectionUtils.isNotEmpty;

/**
 * Provides a default implementation of {@link WarningAware}. Returns new collections for
 * warning messages (defensive copy).
 */
public class WarningAwareSupport implements WarningAware, Serializable {

    private Collection<String> actionWarnings;

    @Override
    public synchronized void setActionWarnings(Collection<String> warningMessages) {
        this.actionWarnings = warningMessages;
    }

    @Override
    public synchronized Collection<String> getActionWarnings() {
        return new LinkedList<>(internalGetActionWarnings());
    }

    @Override
    public synchronized void addActionWarning(String aWarningMessage) {
        internalGetActionWarnings().add(aWarningMessage);
    }

    @Override
    public synchronized boolean hasActionWarnings() {
        return isNotEmpty(actionWarnings);
    }

    private Collection<String> internalGetActionWarnings() {
        if (actionWarnings == null) {
            actionWarnings = new ArrayList<>();
        }

        return actionWarnings;
    }

    /**
     * Clears action warnings list.
     *
     * <p>
     * Will clear the list that contains action warnings.
     * </p>
     */
    public synchronized void clearActionWarnings() {
        internalGetActionWarnings().clear();
    }
}
