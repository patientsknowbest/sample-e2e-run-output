package com.pkb.action;

import com.pkb.action.dto.ActivityDTO;
import com.pkb.bean.IMessageWebBean;
import com.pkb.service.encounter.impl.tolven.EncounterManager;

import java.util.List;

public class PatientNotificationHistoryAction extends BaseAction {

    private static final long serialVersionUID = -2697968127946737990L;

    private static final Integer MAX_COUNT = 100;
    private static final Integer DEFAULT_COUNT = 50;
    private static final Integer DEFAULT_OFFSET = 0;

    private Integer offset;
    private Integer count;
    private List<ActivityDTO> activities;

    private boolean showActivityHistoryPageLink = false;


    private IMessageWebBean messageWebBean;
    private EncounterManager encounterManager;

    public String view() {

        long loggedInUserId = loggedInUserUtil.getLoggedInUserId(); // Never context user
        long accountId = messageWebBean.getDefaultAccountId(loggedInUserId);

        activities = encounterManager.getActivityListAsActivities(getLoggedInEHRRequestContext(), loggedInUserId, accountId, getCount(), getOffset(), this::getText);

        return SUCCESS;
    }


    public void setEncounterManager(EncounterManager encounterManager) {
        this.encounterManager = encounterManager;
    }

    public void setMessageWebBean(IMessageWebBean messageWebBean) {
        this.messageWebBean = messageWebBean;
    }



    public void setCount(Integer count) {
        this.count = count;
    }

    public void setOffset(Integer offset) {
        this.offset = offset;
    }

    public Integer getOffset() {

        if ( offset == null ){
            offset = DEFAULT_OFFSET;
        }

        return offset;
    }

    public Integer getCount() {
        if ( count == null ){
            count = DEFAULT_COUNT;
        }

        if ( count > MAX_COUNT ){
            count = MAX_COUNT;
        }

        return count;
    }

    public List<ActivityDTO> getActivities() {
        return activities;
    }

    public boolean isShowActivityHistoryPageLink() {
        return showActivityHistoryPageLink;
    }

    public String getTab(){
        return "home";
    }

    public String getSubTab(){
        return "myNotifications";
    }
}
