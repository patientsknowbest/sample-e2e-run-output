package com.pkb.action;

import org.apache.struts2.ServletActionContext;
import org.jetbrains.annotations.Nullable;

import java.time.Instant;
import java.util.Map;
import java.util.function.Supplier;

public class TimelineDisplayCache {

    private static Map<String, Object> getApplicationContext() {
        return ServletActionContext.getContext().getApplication();
    }

    private static final String CACHED_TIMELINE_HTML = "CACHED_TIMELINE_HTML";
    private static final String CACHED_TIMELINE_TIMESTAMP = "CACHED_TIMELINE_TIMESTAMP";

    private final int expiryInSeconds;

    private final Supplier<Map<String, Object>> applicationContextProvider;

    private final Supplier<Instant> timestampProvider;

    public TimelineDisplayCache(int expiryInSeconds) {
        this(expiryInSeconds, TimelineDisplayCache::getApplicationContext, Instant::now);
    }

    TimelineDisplayCache(int expiryInSeconds, Supplier<Map<String, Object>> applicationContextProvider,
            Supplier<Instant> timestampProvider) {
        this.expiryInSeconds = expiryInSeconds;
        this.applicationContextProvider = applicationContextProvider;
        this.timestampProvider = timestampProvider;
    }

    public @Nullable String getCachedHTML() {
        Instant cachedAt = (Instant) applicationContextProvider.get().get(CACHED_TIMELINE_TIMESTAMP);
        if (cachedAt == null) {
            return null;
        }
        Instant now = timestampProvider.get();
        long secondsPassed = now.getEpochSecond() - cachedAt.getEpochSecond();
        if (secondsPassed >= expiryInSeconds) {
            return null;
        }
        return (String) applicationContextProvider.get().get(CACHED_TIMELINE_HTML);
    }

    public void cache(String html) {
        applicationContextProvider.get().put(CACHED_TIMELINE_HTML, html);
        Instant now = timestampProvider.get();
        applicationContextProvider.get().put(CACHED_TIMELINE_TIMESTAMP, now);
    }

}
