package com.pkb.action;

import com.opensymphony.xwork2.Action;
import com.opensymphony.xwork2.ActionSupport;
import com.pkb.datamodel.Email;
import com.pkb.service.user.impl.UserManager;
import org.apache.struts2.interceptor.ServletResponseAware;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.io.Writer;
import java.util.Random;

import static com.pkb.datamodel.ImmutableDisclosingEmail.disclosingEmail;

/**
 * Action class to check database connection via tomcat and tolven
 *
 * @author pravinam
 *
 */
public class DBPingAction extends ActionSupport implements
        ServletResponseAware {
    private static final long serialVersionUID = 2917340653403777494L;

    private UserManager userManager;

    private HttpServletResponse servletResponse;

    private final Random randomGen = new Random();

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Override
    public String execute() throws IOException {

        // temporary hack: force userManager's cached transaction to a valid state
        try {
            userManager.ensureValidTransactionStatus();
        } catch (Exception e) {
            LOGGER.error("error ensuring valid transaction status", e);
        }

        Email userEmail = disclosingEmail("notfound" + randomGen.nextInt(Integer.MAX_VALUE) + "@pkbtest.com");
        userManager.getPKBPersonByConfirmedOrPrimaryEmail(userEmail);
        servletResponse.setContentType("text/plain");
        Writer out = servletResponse.getWriter();
        out.write("ok");
        out.flush();
        out.close();
        LOGGER.info("DB ping");
        return Action.NONE;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    @Override
    public void setServletResponse(HttpServletResponse response) {
        this.servletResponse = response;
    }
}
