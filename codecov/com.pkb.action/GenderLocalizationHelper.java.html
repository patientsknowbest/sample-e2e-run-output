package com.pkb.action;

import com.google.common.collect.ImmutableMap;
import com.google.common.collect.Maps;
import com.pkb.entities.enums.Gender;

import java.util.EnumMap;
import java.util.Locale;
import java.util.function.BiFunction;

public class GenderLocalizationHelper extends AbstractLocalizationHelper<Gender> {

    @Override
    protected ImmutableMap<Gender, String> computeMap(Locale locale, BiFunction<String, Locale, String> localizedStringSupplier) {
        EnumMap<Gender, String> result = new EnumMap<>(Gender.class);
        result.put(Gender.UNKNOWN, localizedStringSupplier.apply("global.txt.gender.unknown", locale));
        result.put(Gender.MALE, localizedStringSupplier.apply("global.txt.gender.male", locale));
        result.put(Gender.FEMALE, localizedStringSupplier.apply("global.txt.gender.female", locale));
        result.put(Gender.INDETERMINATE, localizedStringSupplier.apply("global.txt.gender.indeterminate", locale));
        return Maps.immutableEnumMap(result);
    }
}
