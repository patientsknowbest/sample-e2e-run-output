package com.pkb.action;

import org.apache.struts2.dispatcher.HttpParameters;
import org.apache.struts2.dispatcher.Parameter;

import java.util.HashMap;
import java.util.Map;

public class HttpParametersRebuilder {
    private HttpParameters parameters;

    private HttpParametersRebuilder(HttpParameters parameters) {
        this.parameters = parameters;

    }

    public static HttpParametersRebuilder modify(HttpParameters parameters) {
        return new HttpParametersRebuilder(parameters);
    }

    private Parameter newParameter(String name, Object value) {
        Map<String, Object> extraParams = new HashMap<>();
        extraParams.put(name, value);
        return HttpParameters.create().withExtraParams(extraParams).build().get(name);
    }

    public HttpParametersRebuilder put(String name, Object value) {
        parameters.remove(name);

        Map<String, Parameter> newParameter = new HashMap<>();
        newParameter.put(name, newParameter(name, value));

        parameters.appendAll(newParameter);

        return this;
    }

    public HttpParameters get() {
        return parameters;
    }
}
