package com.pkb.action;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.service.datapoint.impl.DataPointManager;
import org.jetbrains.annotations.NotNull;
import org.springframework.context.MessageSource;

import java.util.Locale;
import java.util.function.Supplier;

import static java.lang.String.format;

public class AbstractService {

    protected final DataPointManager dataPointManager;
    protected final MessageSource messageSource;

    protected static final String SERVICE_OPERATION = "operation";

    public AbstractService(DataPointManager dataPointManager, MessageSource messageSource) {
        this.dataPointManager = dataPointManager;
        this.messageSource = messageSource;
    }

    @NotNull
    protected Supplier<IllegalStateException> loggedInContextNotFoundException(String operation) {
        return () -> new IllegalStateException(format("Unable to find a logged in context to %s.", operation));
    }

    public interface BaseRequest {
        Locale getLocale();
        EHRRequestContext getEHRRequestContext();
    }

    public interface NhsStyleRequest {
        boolean nhsStyle();
    }
}
