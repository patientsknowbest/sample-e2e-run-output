package com.pkb.action;

import com.google.common.collect.ImmutableMap;
import com.google.common.collect.Maps;
import com.pkb.entities.enums.UserType;

import java.util.EnumMap;
import java.util.Locale;
import java.util.function.BiFunction;

public class UserTypeLocalizationHelper extends AbstractLocalizationHelper<UserType> {

    @Override
    protected ImmutableMap<UserType, String> computeMap(Locale locale, BiFunction<String, Locale, String> localizedStringSupplier) {
        EnumMap<UserType, String> result = new EnumMap<>(UserType.class);
        result.put(UserType.EMPLOYEE, localizedStringSupplier.apply("global.text.user_type.employee", locale));
        result.put(UserType.INSTITUTE_ADMIN, localizedStringSupplier.apply("global.text.user_type.coordinator", locale));
        result.put(UserType.ORG_COORD, localizedStringSupplier.apply("global.text.user_type.org_admin", locale));
        result.put(UserType.PATIENT, localizedStringSupplier.apply("global.text.user_type.patient", locale));
        result.put(UserType.REG_CLINICIAN, localizedStringSupplier.apply("global.text.user_type.professional", locale));
        result.put(UserType.PRIVACY_OFFICER, localizedStringSupplier.apply("global.text.user_type.privacy_officer", locale));
        result.put(UserType.SUPER_ADMIN, localizedStringSupplier.apply("global.text.user_type.super_admin", locale));
        result.put(UserType.TECH_SUPPORT, localizedStringSupplier.apply("global.text.user_type.support", locale));
        return Maps.immutableEnumMap(result);
    }
}
