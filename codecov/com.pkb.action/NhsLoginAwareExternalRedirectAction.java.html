package com.pkb.action;

import com.pkb.integration.entity.DatapointExternalIntegrationUrl;
import com.pkb.security.NhsLoginAuthenticationDetails;
import com.pkb.service.integration.DatapointExternalIntegrationUrlService;
import org.springframework.beans.factory.annotation.Autowired;

import java.net.URLEncoder;
import java.util.Optional;
import java.util.UUID;

import static java.nio.charset.StandardCharsets.UTF_8;

/**
 * Facilitates redirecting to an external DatapointExternalIntegrationUrl
 *  - will route through our NHS oauth-client if required
 *  - assumes third party service does its own auth as required
 */
public abstract class NhsLoginAwareExternalRedirectAction extends BaseAction {

    @Autowired
    private DatapointExternalIntegrationUrlService urlService;

    private String outgoingRedirectUrl; //the end result, the external url struts should redirect to

    public abstract UUID getDataPointUniqueId();

    public abstract void setDataPointUniqueId(UUID datapointUniqueId);

    /**
     * An outlet for feature-specific amendments to the url before redirect, adding additional parameters etc.
     * Not to be called directly by struts.
     * @param integrationUrl - the url found from the urlService
     */
    protected abstract String amendRedirectUrl(String integrationUrl);
    
    @Override
    public String execute() {
        try {
            Optional<String> externalIntegrationUrlForDatapoint = getExternalIntegrationUrlForDatapoint();

            String redirectUrl = amendRedirectUrl(externalIntegrationUrlForDatapoint.orElseThrow());
            outgoingRedirectUrl = buildOutgoingRedirectUrl(redirectUrl);

            return SUCCESS;
        } catch (Exception e) {
            LOGGER.error("unable to redirect for datatpoint with id {}", getDataPointUniqueId(), e);
            return ERROR;
        }
    }

    private Optional<String> getExternalIntegrationUrlForDatapoint() {
        return urlService.findByDatapointId(getDataPointUniqueId())
                .map(DatapointExternalIntegrationUrl::getUrl);
    }

    private String buildOutgoingRedirectUrl(String externalUrl) {
        if (isNhsLoginSession()) {
            return getNhsRedirectUrl(externalUrl);
        }
        return externalUrl;
    }

    private String getNhsRedirectUrl(String redirectUrl) {
        String encodedRedirect = URLEncoder.encode(redirectUrl, UTF_8);
        return config.getNhsIntegrationUrl() + "/redirect?redirectUrl=" + encodedRedirect + "&code=" +
                loggedInUserUtil.getNhsLoginAuthenticationDetails().flatMap(NhsLoginAuthenticationDetails::getJti)
                        .map(UUID::toString)
                        .orElseThrow(() -> new IllegalStateException("Unable to find Auth details despite having NHS login session, datapoint id = {}" + getDataPointUniqueId()));
    }

    public String getOutgoingRedirectUrl() {
        return outgoingRedirectUrl;
    }

    public void setOutgoingRedirectUrl(String outgoingRedirectUrl){
        this.outgoingRedirectUrl = outgoingRedirectUrl;
    }
}
