package com.pkb.action;

import com.google.common.collect.ImmutableMap;

import java.util.Locale;
import java.util.function.BiFunction;

public class GenderStringLocalizationHelper extends AbstractLocalizationHelper<String> {

    @Override
    protected ImmutableMap<String, String> computeMap(Locale locale, BiFunction<String, Locale, String> localizedStringSupplier) {
        return ImmutableMap.<String, String> builder()
                .put("Unknown", localizedStringSupplier.apply("global.txt.gender.unknown", locale))
                .put("Male", localizedStringSupplier.apply("global.txt.gender.male", locale))
                .put("Female", localizedStringSupplier.apply("global.txt.gender.female", locale))
                .put("Indeterminate", localizedStringSupplier.apply("global.txt.gender.indeterminate", locale))
                .build();
    }
}
