package com.pkb.action;

import com.pkb.exception.PKBException;
import com.pkb.service.user.impl.ApiAuthManager;
import org.apache.commons.lang3.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.security.core.context.SecurityContextImpl;

import javax.servlet.http.Cookie;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.lang.invoke.MethodHandles;
import java.util.Arrays;

import static com.pkb.api.v1.V1OAuthTokenExchangeService.COOKIE_NAME;
import static com.pkb.util.PKBWebUtil.SPRING_SECURITY_CONTEXT;


/**
 * This action log off the user and clear the user session
 *
 * @author mahendera
 */
public class LogoutAction extends BaseAction {

    private static final long serialVersionUID = 1L;
    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    @Autowired
    private ApiAuthManager apiAuthManager;

    @Override
    public String execute() {
        LOGGER.trace("Enter Execute");


        // log the user out of Tolven, if they're logged-in.
        try {
            if (loggedInUserUtil.isUserLoggedIn()) {
                long userId = loggedInUserUtil.getLoggedInUserId();
                userManager.logoutUser(userId);

                SecurityContextImpl userSession = (SecurityContextImpl) session.get(SPRING_SECURITY_CONTEXT);
                userSession.getAuthentication().setAuthenticated(false);
                userSession.setAuthentication(null);

                HttpServletRequest request = getRequest();

                Arrays.stream(request.getCookies())
                        .filter(cookie -> COOKIE_NAME.equalsIgnoreCase(cookie.getName()))
                        .peek(this::deleteCookie)
                        .findFirst()
                        .map(Cookie::getValue)
                        .map(StringUtils::trimToNull)
                        .filter(StringUtils::isNotBlank)
                        .ifPresent(this::invalidateApiSession);


            }
        } catch (Exception e) {
            addActionError(getText("logoutAction.err.error_logging_out"));
            throw new PKBException("error logging out user", e);
        }

        // Spring captures the redirect and closes the session

        LOGGER.trace("Exit Execute");
        return SUCCESS;
    }

    private void deleteCookie(Cookie cookie) {
        HttpServletResponse response = getResponse();
        cookie.setMaxAge(0);
        response.addCookie(cookie);
    }

    private void invalidateApiSession(String accessToken) {
        try {
            apiAuthManager.invalidateAuthSessionByAccessToken(accessToken);
        } catch (Exception e) {
            LOGGER.error("Cannot invalidate session", e);
        }
    }
}
