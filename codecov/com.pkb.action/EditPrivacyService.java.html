package com.pkb.action;

import com.opensymphony.xwork2.Action;
import com.pkb.action.diary.AppointmentsService;
import com.pkb.action.diary.UIPrivacyDto;
import com.pkb.app.dto.DTOBaseFields;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.calendar.entity.AppointmentDTO;

import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.personallibrary.entity.PersonalLibraryEntryDTO;
import com.pkb.service.datapoint.impl.DataPointManager;

import com.pkb.service.privacy.PrivacyService;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.context.MessageSource;

import java.util.Locale;
import java.util.Optional;
import java.util.Set;
import java.util.UUID;
import java.util.function.BiFunction;

import static com.opensymphony.xwork2.Action.ERROR;
import static com.opensymphony.xwork2.Action.INPUT;
import static com.opensymphony.xwork2.Action.SUCCESS;
import static com.pkb.action.BaseAction.RETURN_TO_APPOINTMENT;
import static com.pkb.action.BaseAction.RETURN_TO_APPOINTMENTS_LIST;
import static com.pkb.datamodel.ui.ResponseMeta.ofActionError;
import static com.pkb.user.entity.PKBPerson.Lazy.CONTACTS;
import static java.util.Collections.emptySet;
import static java.util.stream.Collectors.toSet;

public class EditPrivacyService extends AbstractService {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final String CANCEL = "cancel";

    private final AppointmentsService appointmentsService;
    private final PrivacyService privacyService;
    private final UserManager userManager;
    private final PersonalLibraryManager personalLibraryManager;

    public EditPrivacyService(AppointmentsService appointmentsService,
                              DataPointManager dataPointManager,
                              MessageSource messageSource,
                              PrivacyService privacyService,
                              UserManager userManager,
                              PersonalLibraryManager personalLibraryManager) {
        super(dataPointManager, messageSource);
        this.appointmentsService = appointmentsService;
        this.privacyService = privacyService;
        this.userManager = userManager;
        this.personalLibraryManager = personalLibraryManager;
    }

    interface PrivacyRequest<T> {
        EHRRequestContext getEHRRequestContext();
        DtoType getDtoType();
        T getDtoIdentifier();
        Locale getLocale();
        boolean getNhsStyle();
    }

    @Immutable
    public interface FormRequest<T> extends PrivacyRequest<T> {}

    @Immutable
    public interface FormRequestAppointment<T> extends FormRequest<T> {
        boolean getReturnToAppointment();
    }

    @Immutable
    public interface SaveRequest<T> extends PrivacyRequest<T> {
        String getPrivacy();
    }

    @Immutable
    public interface SaveRequestAppointment<T> extends SaveRequest<T> {
        boolean getReturnToAppointment();
    }

    public UIPrivacyDto showPrivacyForm(FormRequest request) {
        return request.getEHRRequestContext().getLoggedInContext()
                .map(loggedInContext -> {
                    UIPrivacyDto dto;
                    switch (request.getDtoType()) {
                        case APPOINTMENT:
                            PKBPerson contextOrAccessingUser = userManager.getPKBPerson(loggedInContext.getContextOrAccessingUserId());
                            Optional<AppointmentDTO> appointmentDTO = appointmentsService.retrieveAppointment(loggedInContext, (UUID) request.getDtoIdentifier(),
                                    contextOrAccessingUser.getDefaultAccountId());

                            dto = new UIPrivacyDto("diary");
                            appointmentDTO.ifPresentOrElse(appDTO -> {
                                   dto.setPrivacyFlags(getPrivacySet(appDTO.getBaseFields()));
                                   dto.setActionResult(INPUT);
                                }, () -> {
                                showError(request, dto, "editAppointment.err.notFound");
                                ImmutableFormRequestAppointment apptRequest = (ImmutableFormRequestAppointment) request;
                                dto.setActionResult(apptRequest.getReturnToAppointment() ? RETURN_TO_APPOINTMENT : RETURN_TO_APPOINTMENTS_LIST);
                            });
                            dto.setNhsStyle(request.getNhsStyle());
                            return dto;
                        case LIBRARY:
                            Optional<@Nullable PersonalLibraryEntryDTO> entry = Optional.ofNullable(personalLibraryManager.getEntry(loggedInContext,
                                    (Long) request.getDtoIdentifier(), loggedInContext.getAccessingUserId()));

                            dto = new UIPrivacyDto("treatments", "library");
                            entry.ifPresentOrElse(entryDTO -> {
                                        dto.setPrivacyFlags(getPrivacySet(entryDTO.getBaseFields()));
                                        dto.setActionResult(INPUT);
                                    }, () -> {showError(request, dto, "editLibraryEntry.err.notFound");});
                            dto.setNhsStyle(request.getNhsStyle());
                            return dto;
                        default:
                            throw new NotImplementedException("Unknown DTO type " + request.getDtoType());
                    }
                })
                .orElseThrow(() -> new IllegalStateException("Non-existent logged in context: unable to show privacy form for data point " + request.getDtoIdentifier()));
    }

    @NotNull
    private void showError(FormRequest request, UIPrivacyDto dto, String errorKey) {
        dto.setPrivacyFlags(emptySet());
        dto.setResponseMeta(ofActionError(messageSource.getMessage(errorKey, new Object[]{request.getDtoIdentifier()}, request.getLocale())));
        dto.setActionResult(ERROR);
    }

    private Set<String> getPrivacySet(DTOBaseFields baseFields) {
        return baseFields.getPrivacyFlags()
                .stream()
                .map(Enum::name)
                .collect(toSet());
    }

    public <T> UIPrivacyDto savePrivacy(SaveRequest<T> request) {
        return request.getEHRRequestContext().getLoggedInContext()
                .map(loggedInContext -> {
                    switch (request.getDtoType()) {
                        case APPOINTMENT:
                            UIPrivacyDto dto = getUIPrivacyDtoFromBaseDto((context, identifier) -> {
                                PKBPerson contextOrAccessingUser = userManager.getPKBPerson(loggedInContext.getContextOrAccessingUserId());
                                return appointmentsService.retrieveAppointment(context, (UUID) identifier, contextOrAccessingUser.getDefaultAccountId());
                            }, loggedInContext, request, "err.appointment.not.found.for.id", MenuDataType.appointment);
                            if (Action.SUCCESS.equals(dto.getActionResult())) {
                                ImmutableSaveRequestAppointment apptRequest = (ImmutableSaveRequestAppointment) request;
                                dto.setActionResult(apptRequest.getReturnToAppointment() ? RETURN_TO_APPOINTMENT : RETURN_TO_APPOINTMENTS_LIST);
                            }
                            return dto;
                        case LIBRARY:
                            return getUIPrivacyDtoFromBaseDto((context, identifier) ->
                                    Optional.ofNullable(personalLibraryManager.getEntry(loggedInContext, (Long) request.getDtoIdentifier(), loggedInContext.getAccessingUserId())
                                    ), loggedInContext, request, "err.library.entry.not.found.for.id", MenuDataType.libraryEntry);
                        default:
                            throw new NotImplementedException("Unknown DTO type " + request.getDtoType());
                    }
                })
                .orElseThrow(() -> new IllegalStateException("Non-existent logged in context: unable to save privacy for data point " + request.getDtoIdentifier()));
    }

    private <T, U extends IBaseDTO> UIPrivacyDto getUIPrivacyDtoFromBaseDto(BiFunction<LoggedInEHRRequestContext, T, Optional<U>> dtoRetriever,
                                                                            LoggedInEHRRequestContext loggedInContext,
                                                                            SaveRequest<T> request, String errorKey, MenuDataType menuDataType) {
        var privacyDto = new UIPrivacyDto();
        dtoRetriever.apply(loggedInContext, request.getDtoIdentifier()).ifPresentOrElse(dto -> {
            if (isPrivacyFlagNotEditable(dto, loggedInContext)) {
                LOGGER.error("Logged in user id [{}] can't edit the privacy flags for data point [{}]", loggedInContext.getMaybeAccessingUserId(), dto.getId());
                privacyDto.setResponseMeta(ofActionError(messageSource.getMessage("saveDatapoint.txt.noAccessPrivacyFlags", null, request.getLocale())));
                privacyDto.setActionResult(CANCEL);
                return;
            }

            var usersMap = userManager.getPKBPersonMapById(loggedInContext.getContextAndAccessingUserIds(), CONTACTS);
            PKBPerson contextOrAccessingUser = usersMap.get(loggedInContext.getContextOrAccessingUserId());

            privacyService.handlePrivacyChange(dto, menuDataType, PrivacyFlag.valueOf(request.getPrivacy()), loggedInContext,
                    contextOrAccessingUser.getDefaultAccountId(),
                    contextOrAccessingUser,
                    usersMap.get(loggedInContext.getAccessingUserId()));
            privacyDto.setNhsStyle(request.getNhsStyle());

            privacyDto.setActionResult(SUCCESS);
        }, () -> {
            privacyDto.setResponseMeta(ofActionError(
                    messageSource.getMessage(errorKey, new Object[]{ request.getDtoIdentifier() }, request.getLocale())));
            privacyDto.setActionResult(CANCEL);
        });
        return privacyDto;
    }

    private boolean isPrivacyFlagNotEditable(IBaseDTO dto, EHRRequestContext requestContext) {
        return !dataPointManager.getPermission(dto, requestContext).isPrivacyFlagEditable();
    }
}
