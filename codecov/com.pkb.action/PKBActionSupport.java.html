package com.pkb.action;

import com.opensymphony.xwork2.ActionSupport;

import java.util.Collection;

public class PKBActionSupport extends ActionSupport implements WarningAware {

    private final WarningAwareSupport warningAware = new WarningAwareSupport();

    @Override
    public void setActionWarnings(Collection<String> warningMessages) {
        warningAware.setActionWarnings(warningMessages);
    }

    @Override
    public Collection<String> getActionWarnings() {
        return warningAware.getActionWarnings();
    }

    @Override
    public void addActionWarning(String aWarningMessage) {
        warningAware.addActionWarning(aWarningMessage);
    }

    @Override
    public boolean hasActionWarnings() {
        return warningAware.hasActionWarnings();
    }

    public void clearActionWarnings() {
        warningAware.clearActionWarnings();
    }
}
