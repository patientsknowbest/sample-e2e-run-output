package com.pkb.action;


import com.pkb.exception.MailException;
import com.pkb.exception.PKBException;
import com.pkb.institute.entity.Team;
import com.pkb.service.invitation.impl.InvitationManager;
import com.pkb.service.team.TeamManager;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.user.impl.UserManager;
import org.apache.commons.lang3.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.UUID;

public class UpgradeClinicianAction extends BaseAction {

    private static final long serialVersionUID = 1L;

    private static final String RESULT_BACK_TO_LOGIN = "backToLogin";

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private TeamManager teamManager;

    private TeamUserManager teamUserManager;

    private InvitationManager invitationManager;

    private String invitationId;

    public String becomeInstituteClinician() {
        String result = SUCCESS;
        try {
            String uidString = request.getParameter("uid");
            String icode = request.getParameter("icode");
            String iId = request.getParameter("invitationId");
            if (StringUtils.isNotBlank(uidString) && StringUtils.isNotBlank(icode) && StringUtils.isNotBlank(iId)) {
                long uid = Long.parseLong(uidString);
                if (isValidUser(uid)) {
                    try {
                        invitationManager.beginTransaction();

                        Team team = teamManager.getTeamByCode(icode);
                        teamUserManager.upgradeClinician(team, uid, null);
                        // this will just update its status to complete
                        UUID invitationPublicId = invitationManager.getInvitationIdByString(invitationId);
                        invitationManager.acceptInvitation(getLoggedInEHRRequestContext().withOrg(team.getOrg()), invitationPublicId, null, null);
                        invitationId = iId;

                        invitationManager.commitTransaction();
                    } finally {
                        invitationManager.rollbackIfNotCommitted();
                    }
                } else { // NOT valid user
                    addActionError(getText("upgradeClinicianAction.err.wrong_invitee"));
                    result = RESULT_BACK_TO_LOGIN;
                }

            }
        } catch (PKBException e) {
            LOGGER.error("unexpected error", e);
            addActionError(getText("upgradeClinicianAction.err.error_upgrading"));
        } catch (MailException e) {
            LOGGER.error("Mailer exception: {}", e.getCause(), e);
            addActionError(getText("upgradeClinicianAction.err.error_upgrading"));
        } catch (Exception e) {
            LOGGER.error("unexpected error", e);
            addActionError(getText("upgradeClinicianAction.err.error_upgrading"));
        }
        return result;
    }


    private boolean isValidUser(long uid) {
       return loggedInUserUtil.getLoggedInUserId() == uid;
    }


    public TeamManager getTeamManager() {
        return teamManager;
    }

    public void setTeamManager(TeamManager teamManager) {
        this.teamManager = teamManager;
    }

    public TeamUserManager getTeamUserManager() {
        return teamUserManager;
    }

    public void setTeamUserManager(TeamUserManager teamUserManager) {
        this.teamUserManager = teamUserManager;
    }

    public InvitationManager getInvitationManager() {
        return invitationManager;
    }

    public void setInvitationManager(InvitationManager invitationManager) {
        this.invitationManager = invitationManager;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public String getInvitationId() {
        return invitationId;
    }

    public void setInvitationId(String invitationId) {
        this.invitationId = invitationId;
    }

}
