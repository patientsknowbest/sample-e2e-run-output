package com.pkb.action;

import com.pkb.entities.enums.Route;

public class NoCryptoAccessAction extends BaseAction {

    static final String SSO_NO_ACCESS = "ssoNoAccess";
    static final String PKB_WEB_NO_ACCESS = "pkbWebNoAccess";
    static final String REDIRECT_HOME = "redirectHomeResult";

    public String handleNoCryptoAccess() {
        Route route = getEHRRequestContext().getAccessRoute();
        String result;
        switch (route) {
        case WEBAPP_EMIS_PORTAL:
        case WEBAPP_SYSTMONE_PORTAL:
            addActionError(getText("global.err.no_crypto_access"));
            result = SSO_NO_ACCESS;
            break;
        default:
            if(loggedInUserUtil.isUserLoggedIn()) {
                result = REDIRECT_HOME;
            }
            else {
                addActionError(getText("global.err.no_crypto_access"));
                result = PKB_WEB_NO_ACCESS;
            }
        }

        return result;
    }


}
