package com.pkb.action;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.dto.ConsentIndividualDto;
import com.pkb.dto.ConsentTeamDto;
import com.pkb.entities.enums.AccessingEntityType;
import com.pkb.institute.entity.Team;
import com.pkb.integration.medmij.MedmijService;
import com.pkb.mappers.ConsentIndividualMapper;
import com.pkb.mappers.ConsentTeamMapper;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.patientconsent.PatientConsentManager.ConsentLinks;
import org.immutables.value.Value.Immutable;

import javax.annotation.Nullable;
import java.util.List;
import java.util.Locale;
import java.util.Set;
import java.util.UUID;

public class PatientConsentTeamService {

    private final ConsentIndividualMapper consentIndividualMapper;
    private final ConsentTeamMapper consentTeamMapper;
    private final PatientConsentManager patientConsentManager;
    private final MedmijService medmijService;


    public PatientConsentTeamService(ConsentIndividualMapper consentIndividualMapper,
                                     ConsentTeamMapper consentTeamMapper,
                                     PatientConsentManager patientConsentManager,
                                     MedmijService medmijService) {
        this.consentIndividualMapper = consentIndividualMapper;
        this.consentTeamMapper = consentTeamMapper;
        this.patientConsentManager = patientConsentManager;
        this.medmijService = medmijService;
    }

    public PatientConsentLinksDto buildPatientConsentLinksDto(LoggedInEHRRequestContext requestContext, Locale locale) {
        Long contextUserId = requestContext.getContextUserId().orElse(null);

        ConsentLinks consentLinks = patientConsentManager.getConsentLinks(requestContext,
                requestContext.getContextOrAccessingUserId(), false, Set.of(AccessingEntityType.TEAM, AccessingEntityType.INDIVIDUAL));


        return ImmutablePatientConsentLinksDto.builder()
                .consentTeamDtos(consentTeamMapper.teamsToConsentTeamDtos(consentLinks.getTeams(),
                        contextUserId,
                        requestContext.getContextOrAccessingUserId(),
                        consentLinks.getTeamConsents(),
                        locale))
                .consentIndividualDtos(consentIndividualMapper.individualsToConsentIndividualDtos(consentLinks.getIndividuals(),
                        contextUserId,
                        consentLinks.getIndividualConsents(),
                        locale))
                .teamPublicId(consentLinks.getTeams().stream().findFirst().map(Team::getPublicId).orElse(null))
                .build();
    }

    public PatientConsentTeamDto buildPatientConsentTeamDto(LoggedInEHRRequestContext requestContext) {
        ImmutablePatientConsentTeamDto.Builder builder = ImmutablePatientConsentTeamDto.builder();
        builder.updateMyRecordDto(medmijService.buildUpdateMyRecordDto(requestContext));
        return builder.build();
    }

    @Immutable
    public interface PatientConsentLinksDto {
        List<ConsentTeamDto> getConsentTeamDtos();
        List<ConsentIndividualDto> getConsentIndividualDtos();
        @Nullable UUID getTeamPublicId();
    }
}
