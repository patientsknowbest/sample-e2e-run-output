package com.pkb.action;

import com.pkb.exception.DisplayableException;
import com.pkb.exception.DisplayableException.DisplayAs;

public class RedirectHomeAction extends BaseAction {

    private static final long serialVersionUID = 1L;
    public static final String HOME_PATIENT_SUMMARY = "homePatientSummary";

    private Throwable exception; // used for DisplayableException

    private String postLoginRedirect = "/WEB-INF/view/dashboard.jsp"; // default value

    @Override
    public String execute() {

        // DisplayableException comes through here!
        // get appropriate message added to the display.
        if (exception != null) {

            // will it be wrapped, though?

            if (exception instanceof DisplayableException) {
                DisplayableException de = (DisplayableException) exception;
                if (de.getDisplayAs() == DisplayAs.error) {
                    addActionError(getText(de.getTextProperty(), de.getTextArgs()));
                } else {
                    addActionMessage(getText(de.getTextProperty(), de.getTextArgs()));
                }
            }
        }

        if (!getEHRRequestContext().isUserInOwnAccount()){
            return HOME_PATIENT_SUMMARY;
        }

        return loggedInUserUtil.getLoggedInUserHomeResult().orElse(null);
    }

    public void setException(Throwable exception) {
        this.exception = exception;
    }

    public String getPostLoginRedirect() {
        return postLoginRedirect;
    }

}
