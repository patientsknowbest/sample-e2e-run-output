package com.pkb.action;

import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.Response;
import org.jetbrains.annotations.NotNull;
import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.nodes.Document.OutputSettings;
import org.jsoup.nodes.Element;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.net.MalformedURLException;
import java.net.URL;

import static java.lang.invoke.MethodHandles.lookup;
import static org.apache.commons.lang3.StringUtils.isNotBlank;
import static org.jsoup.nodes.Document.OutputSettings.Syntax.xml;


public class TimelineDisplayGatewayImpl implements TimelineDisplayGateway {
    private static final Logger LOGGER = LoggerFactory.getLogger(lookup().lookupClass());

    private static final String RETURNED_HTML_FRAGMENT_ON_ERROR = "<!-- Failed to fetch SPA files. -->";
    private static final OutputSettings XML = new OutputSettings().syntax(xml);

    private final OkHttpClient httpClient;
    private final TimelineDisplayCache cache;
    private final String fetchURL;
    private final String normalizedBaseUrl;

    public TimelineDisplayGatewayImpl(
            @NotNull OkHttpClient httpClient,
            @NotNull TimelineDisplayCache cache,
            @NotNull String fetchURL,
            @NotNull String browserBaseURL) {
        this.httpClient = httpClient;
        this.cache = cache;
        this.fetchURL = fetchURL;
        this.normalizedBaseUrl = ensureTralingSlash(browserBaseURL);
    }

    @Override
    public String obtainHTMLFragment(String module) {
        String cachedResp = cache.getCachedHTML();

        if (cachedResp != null) {
            return cachedResp;
        }

        String url = fetchURL + "/" + module + "/";
        Request request = new Request.Builder().get().url(url).build();
        try (Response response = httpClient.newCall(request).execute()) {
            String normalizedModule = ensureNoLeadingSlash(ensureTralingSlash(module));
            String html = transform(response.body().string(), normalizedBaseUrl + normalizedModule);
            cache.cache(html);
            return html;
        } catch (Exception cause) {
            LOGGER.error("failed to obtain timeline view html", cause);
            return RETURNED_HTML_FRAGMENT_ON_ERROR;
        }
    }

    private String transform(String rawHTML, String prefix) {
        Document html = Jsoup.parse(rawHTML).outputSettings(XML);

        StringBuilder stringBuilder = new StringBuilder();

        for (Element link : html.select("link")) {
            String href = link.attr("href");
            if (isNotBlank(href) && isRelative(href)) {
                link.attr("href", prefix + ensureNoLeadingSlash(href));
            }
            stringBuilder.append(link.outerHtml());
        }

        for (Element script : html.select("script")) {
            String source = script.attr("src");
            if (isNotBlank(source) && isRelative(source)) {
                script.attr("src", prefix + ensureNoLeadingSlash(source));
            }
            stringBuilder.append(script.outerHtml());
        }

        return stringBuilder.toString();
    }

    private boolean isRelative(String url) {
        try {
            new URL(url);
            return false;
        } catch (MalformedURLException ignored) {
            // Parse error so relative ??
            return true;
        }
    }

    private String ensureNoLeadingSlash(String href) {
        return href.startsWith("/") ? href.substring(1) : href;
    }

    private String ensureTralingSlash(@NotNull String path) {
        return path.endsWith("/") ? path : path + "/";
    }
}
