package com.pkb.action;

import com.opensymphony.xwork2.Action;
import com.pkb.service.dataupload.emis.esv2.EmisDiagnosticService;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.IOException;
import java.io.Writer;
import java.util.concurrent.atomic.AtomicLong;

/**
 * Checks EMIS ES Remote SFTP System Connectivity.
 * 
 * Return values: "[status]:[outcome]"
 * 
 * Where <code>status</code> will be one of "OKAY" for success or "FAIL" for fail.
 * 
 * <p>
 * Example return values:
 * </p>
 * <ul>
 * <li>"FAIL:Unknown"</li>
 * <li>"FAIL:Ping request interval too short."</li>
 * <li>"OKAY:Connected"</li>
 * <li>"OKAY:Remote directory accessible"</li>
 * <ul>
 * 
 * <p>
 * Also optionally can specify a format parameter as either "json" or "xml" in case any external monitoring tool needs it.
 * </p>
 * 
 * Eg: /emisEsPing.action?format=json
 * { "status": "OKAY", "outcome": "Remote directory accessible" }
 *
 * Eg: /emisEsPing.action?format=xml
 * <?xml version="1.0" encoding="UTF-8"?>
 * <emis-es-ping status="OKAY">Remote directory accessible</emis-es-ping>
 * 
 * @author jspencepkb
 *
 */
public class EmisEsPingAction extends BasePingAction {
    private static final long serialVersionUID = -1L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private static AtomicLong lastPingAction = new AtomicLong(0);
    private static Long MIN_PING_INTERVAL_MILLIS = 30L * 1000;
    private EmisDiagnosticService emisDiagnosticService;

    @Override
    public String execute() throws IOException {
        PingStatus status = PingStatus.FAIL;
        String outcome = "Unknown";

        boolean connectAllowed = true;

        if ((System.currentTimeMillis() - lastPingAction.getAndSet(System.currentTimeMillis())) < MIN_PING_INTERVAL_MILLIS) {
            outcome = "Ping request interval too short.";
            connectAllowed = false;
        }

        Writer out = servletResponse.getWriter();
        if (connectAllowed) {
            // Connect to remote server and look at the list of files
            try {
                emisDiagnosticService.getRemoteFileList();
                status = PingStatus.OKAY;
            } catch (Exception e) {
                LOGGER.error("Error during EMIS Sftp file listing", e);
                outcome = "Error during file listing!";
            }

        }

        String responsePayload = "";

        LOGGER.info("Format: {}", format);

        switch (format) {
        case "json":
            servletResponse.setContentType("application/json");
            responsePayload = "{ \"status\": \"" + status + "\", \"outcome\": \"" + outcome + "\" }";
            break;
        case "xml":
            servletResponse.setContentType("application/xml");
            responsePayload = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n" +
                    "<emis-es-ping status=\"" + status + "\">" + outcome + "</emis-es-ping>";
            break;
        default:
            servletResponse.setContentType("text/plain");
            responsePayload = status + ":" + outcome;
        }
        out.write(responsePayload);
        out.flush();
        out.close();
        LOGGER.info("EMIS ES ping");

        return Action.NONE;
    }

    public void setEmisDiagnosticService(EmisDiagnosticService emisDiagnosticService) {
        this.emisDiagnosticService = emisDiagnosticService;
    }
}
