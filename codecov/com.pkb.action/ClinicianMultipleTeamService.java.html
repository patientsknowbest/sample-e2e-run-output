package com.pkb.action;

import com.google.common.collect.ComparisonChain;
import com.pkb.entities.enums.UserType;
import com.pkb.institute.entity.Team;
import com.pkb.institute.entity.TeamWithPKBPerson;
import com.pkb.util.LoggedInUserUtil;

import java.io.Serializable;
import java.util.Comparator;
import java.util.List;
import java.util.Map;

import static java.util.stream.Collectors.toList;

public class ClinicianMultipleTeamService {
    public final LoggedInUserUtil loggedInUserUtil;

    public ClinicianMultipleTeamService(LoggedInUserUtil loggedInUserUtil) {
        this.loggedInUserUtil = loggedInUserUtil;
    }

    public List<ClinicianTeamDto> buildClinicianTeamDtoCollection(
            Map<UserType, String> userTypeEnumMap, Team primaryTeam) {
        return loggedInUserUtil.getLoggedInUserTeams().stream()
                .sorted(new OrgTeamComparator())
                .map(teamWithPerson -> ImmutableClinicianTeamDto.builder()
                        .team(teamWithPerson)
                        .primaryTeam(primaryTeam.equals(teamWithPerson.getTeam()))
                        .role(createRole(teamWithPerson, userTypeEnumMap)).build())
        .collect(toList());
    }

    private String createRole(TeamWithPKBPerson teamWithPerson, Map<UserType, String> userTypeEnumMap) {
        StringBuilder roleBuilder = new StringBuilder(teamWithPerson.getTeam().getOrg().getNameAbbreviated()).append(" / ");
        UserType userType = teamWithPerson.getPerson().getUserType();
        if (userType != UserType.PRIVACY_OFFICER && userType != UserType.ORG_COORD) {
            roleBuilder.append(teamWithPerson.getTeam().getName()).append(" / ");
        }
        return roleBuilder.append(userTypeEnumMap.get(userType)).toString();
    }

    private static class OrgTeamComparator implements Comparator<TeamWithPKBPerson>, Serializable {

        public OrgTeamComparator() {
        }

        @Override
        public int compare(TeamWithPKBPerson o1, TeamWithPKBPerson o2) {
            return ComparisonChain.start()
                    .compare(o1.getTeam().getOrg().getName(), o2.getTeam().getOrg().getName())
                    .compare(o1.getTeam().getName(), o2.getTeam().getName())
                    .result();
        }
    }
}
