package com.pkb.action;

import com.pkb.service.reminders.ReminderManager;
import com.pkb.service.user.impl.UserManager;

public class ReminderOptOutAction extends BaseAction {

    private ReminderManager reminderManager;

    private boolean optedOut;
    private boolean notPatient;
    private long loggedInUserId;
    private long personId;

    public ReminderManager getReminderManager() {
        return reminderManager;
    }

    public void setReminderManager(ReminderManager reminderManager) {
        this.reminderManager = reminderManager;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public boolean isOptedOut() {
        return optedOut;
    }

    public void setOptedOut(boolean optedOut) {
        this.optedOut = optedOut;
    }

    public boolean isNotPatient() {
        return notPatient;
    }

    public long getPersonId() {
        return personId;
    }

    public void setPersonId(long personId) {
        this.personId = personId;
    }

    public void setNotPatient(boolean notPatient) {
        this.notPatient = notPatient;
    }

    public String optOut() {
        optedOut = true;
        loggedInUserId = loggedInUserUtil.getLoggedInUserId();
        if ((personId == 0) || (loggedInUserId != personId)) {
            return "invalid";
        }
        optedOut = reminderManager.optOutSymptomReminder(loggedInUserId);
        if (!optedOut) {

            return "invalid";
        }
        return SUCCESS;

    }

}
