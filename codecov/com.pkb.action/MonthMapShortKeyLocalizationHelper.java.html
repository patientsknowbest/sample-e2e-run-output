package com.pkb.action;

import com.google.common.collect.ImmutableMap;

import java.util.Locale;
import java.util.function.BiFunction;

public class MonthMapShortKeyLocalizationHelper extends AbstractLocalizationHelper<String> {

    @Override
    protected ImmutableMap<String, String> computeMap(Locale locale, BiFunction<String, Locale, String> localizedStringSupplier) {
        return ImmutableMap.<String, String> builder()
                .put("Month", localizedStringSupplier.apply("global.txt.month", locale))
                .put("Jan", localizedStringSupplier.apply("global.txt.jan_three_letter", locale))
                .put("Feb", localizedStringSupplier.apply("global.txt.feb_three_letter", locale))
                .put("Mar", localizedStringSupplier.apply("global.txt.mar_three_letter", locale))
                .put("Apr", localizedStringSupplier.apply("global.txt.apr_three_letter", locale))
                .put("May", localizedStringSupplier.apply("global.txt.may_three_letter", locale))
                .put("Jun", localizedStringSupplier.apply("global.txt.jun_three_letter", locale))
                .put("Jul", localizedStringSupplier.apply("global.txt.jul_three_letter", locale))
                .put("Aug", localizedStringSupplier.apply("global.txt.aug_three_letter", locale))
                .put("Sep", localizedStringSupplier.apply("global.txt.sep_three_letter", locale))
                .put("Oct", localizedStringSupplier.apply("global.txt.oct_three_letter", locale))
                .put("Nov", localizedStringSupplier.apply("global.txt.nov_three_letter", locale))
                .put("Dec", localizedStringSupplier.apply("global.txt.dec_three_letter", locale))
                .build();
    }
}
