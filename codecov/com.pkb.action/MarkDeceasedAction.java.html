package com.pkb.action;

import com.pkb.entities.enums.UserStatus;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.InvalidDeathTimestampException;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.InvalidUIDateException;
import com.pkb.util.UIDate;

import static com.pkb.user.entity.InvalidDeathTimestampException.Reason.FUTURE_DATE;


public class MarkDeceasedAction extends BaseAction {

    static final String MSG_FUTURE_DATE = "markdeceased.error.futuredate";

    static final String MSG_INVALID_DATE = "markdeceased.error.invaliddate";

    static final String MSG_ALREADY_DEAD = "markdeceased.error.alreadydead";

    static final String MSG_SUCCESS = "markdeceased.success";

    static final String MSG_MARKED_NOT_DECECASED = "marknotdeceased.success";

    static final String MSG_BEFORE_DOB = "markdeceased.beforedob";

    static final String ALREADY_DECEASED = "alreadyDeceased";

    private UIDate deathTimestamp;

    @Override
    public String execute() throws Exception {
        if (deathTimestamp == null) {
            return INPUT;
        }
        try {
            userManager.markAsDeceased(contextUserUtil.getContextUser(), deathTimestamp.toInstant());
        } catch (InvalidDeathTimestampException e) {
            String error;
            if (FUTURE_DATE == e.getReason()) {
                error = getText(MSG_FUTURE_DATE);
            } else {
                error = getText(MSG_BEFORE_DOB);
            }
            //noinspection StrutsI18nError,SSBasedInspection
            addActionError(error);
            return INPUT;
        } catch (IllegalStateException ignored) {
            addActionError(getText(MSG_ALREADY_DEAD));
            return ALREADY_DECEASED;
        } catch (InvalidUIDateException ignored) {
            addActionError(getText(MSG_INVALID_DATE));
            return INPUT;
        }
        addActionMessage(getText(MSG_SUCCESS));
        return SUCCESS;
    }

    @Override
    public String input() throws Exception {
        PKBPerson contextUser = contextUserUtil.getContextUser();
        if (contextUser != null && UserStatus.DEAD == contextUser.getStatus()) {
            return ALREADY_DECEASED;
        }
        return INPUT;
    }

    public String markNotDeceased() throws Exception {
        addActionMessage(getText(MSG_MARKED_NOT_DECECASED));
        userManager.markNotDeceased(contextUserUtil.getContextUser());
        return SUCCESS;
    }

    public UIDate getDeathTimestamp() {
        return deathTimestamp;
    }

    public void setDeathTimestamp(UIDate deathTimestamp) {
        this.deathTimestamp = deathTimestamp;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }
}
