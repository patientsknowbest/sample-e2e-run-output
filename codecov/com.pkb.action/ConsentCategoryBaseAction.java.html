package com.pkb.action;

import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.datamodel.consent.RequiresConsent;
import com.pkb.encounter.entity.Encounter;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.institute.entity.Team;
import com.pkb.questionnaires.action.QuestionnaireBaseAction;
import com.pkb.service.privacy.PrivacyService;
import org.apache.commons.lang3.NotImplementedException;
import org.jetbrains.annotations.NotNull;

import java.util.Map;
import java.util.Objects;
import java.util.Optional;
import java.util.function.BiConsumer;
import java.util.function.Supplier;


/**
 * Extend this class if you have an action that can add/edit entities to which
 * consent categories apply (eg teams, diagnoses, etc)
 * <p>
 * TODO: Refactor existing actions to use this where necessary
 */
public class ConsentCategoryBaseAction extends QuestionnaireBaseAction {

    private static final long serialVersionUID = -7666634605684101644L;

    protected PrivacyService privacyService;

    private boolean generalHealthConsentRequired = true;
    private boolean socialCareConsentRequired = false;
    private boolean mentalHealthConsentRequired = false;
    private boolean sexualHealthConsentRequired = false;
    protected String privacy;

    private boolean editConsentModuleEnabled = true;

    public boolean isGeneralHealthConsentRequired() {
        return generalHealthConsentRequired;
    }

    public void setGeneralHealthConsentRequired(boolean generalHealthConsentRequired) {
        this.generalHealthConsentRequired = generalHealthConsentRequired;
    }

    public boolean isSocialCareConsentRequired() {
        return socialCareConsentRequired;
    }

    public void setSocialCareConsentRequired(boolean socialCareConsentRequired) {
        this.socialCareConsentRequired = socialCareConsentRequired;
    }

    public boolean isMentalHealthConsentRequired() {
        return mentalHealthConsentRequired;
    }

    public void setMentalHealthConsentRequired(boolean mentalHealthConsentRequired) {
        this.mentalHealthConsentRequired = mentalHealthConsentRequired;
    }

    public boolean isSexualHealthConsentRequired() {
        return sexualHealthConsentRequired;
    }

    public void setSexualHealthConsentRequired(boolean sexualHealthConsentRequired) {
        this.sexualHealthConsentRequired = sexualHealthConsentRequired;
    }

    public boolean isEditConsentModuleEnabled() {
        return editConsentModuleEnabled;
    }

    public void setEditConsentModuleEnabled(boolean editConsentModuleEnabled) {
        this.editConsentModuleEnabled = editConsentModuleEnabled;
    }

    public String getPrivacy() {
        return privacy;
    }

    public void setPrivacy(String privacy) {
        this.privacy = privacy;
    }

    public void setPrivacyService(PrivacyService privacyService) {
        this.privacyService = privacyService;
    }

    protected void setConsentCategoriesOnEntity(RequiresConsent entity) {
        entity.setRequireGeneral(isGeneralHealthConsentRequired());
        entity.setRequireMentalHealth(isMentalHealthConsentRequired());
        entity.setRequireSexualHealth(isSexualHealthConsentRequired());
        entity.setRequireSocialCare(isSocialCareConsentRequired());
    }

    protected void displayConsentCategories(RequiresConsent entity) {
        setGeneralHealthConsentRequired(entity.getRequireGeneral());
        setMentalHealthConsentRequired(entity.getRequireMentalHealth());
        setSexualHealthConsentRequired(entity.getRequireSexualHealth());
        setSocialCareConsentRequired(entity.getRequireSocialCare());
    }

    protected void fillPrivacyFlagsFrom(IBaseDTO baseDTO) {
        if (baseDTO != null) {
            fillPrivacyFlagsFrom(baseDTO.getBaseFields());
        }
    }

    protected void fillPrivacyFlagsFrom(DTOBaseFields baseWithPrivacyFlags) {
        generalHealthConsentRequired = baseWithPrivacyFlags.isGeneralHealthConsentRequired();
        mentalHealthConsentRequired = baseWithPrivacyFlags.isMentalHealthConsentRequired();
        sexualHealthConsentRequired = baseWithPrivacyFlags.isSexualHealthConsentRequired();
        socialCareConsentRequired = baseWithPrivacyFlags.isSocialCareConsentRequired();
    }

    protected void fillPrivacyFlagsFrom(Team team) {
        fillPrivacyFlagsFrom(team.getDatapointPrivacyFlags());
    }

    protected void fillPrivacyFlagsFrom(String privacy) {
        fillPrivacyFlagsFrom(PrivacyFlag.valueOf(privacy));
    }

    protected void fillPrivacyFlagsFrom(@NotNull PrivacyFlag requiredFlag) {
        switch (requiredFlag) {
            case GENERAL:
                generalHealthConsentRequired = true;
                mentalHealthConsentRequired = false;
                sexualHealthConsentRequired = false;
                socialCareConsentRequired = false;
                break;
            case MENTAL_HEALTH:
                generalHealthConsentRequired = false;
                mentalHealthConsentRequired = true;
                sexualHealthConsentRequired = false;
                socialCareConsentRequired = false;
                break;
            case SEXUAL_HEALTH:
                generalHealthConsentRequired = false;
                mentalHealthConsentRequired = false;
                sexualHealthConsentRequired = true;
                socialCareConsentRequired = false;
                break;
            case SOCIAL_CARE:
                generalHealthConsentRequired = false;
                mentalHealthConsentRequired = false;
                sexualHealthConsentRequired = false;
                socialCareConsentRequired = true;
                break;
            default:
                throw new NotImplementedException("Privacy flag type " + requiredFlag + " not implemented yet!");
        }
    }

    protected void refillPrivacyFlags() {
        if (privacy != null) {
            fillPrivacyFlagsFrom(privacy);
        }
    }

    protected PrivacyFlag getNewFlag() {
        return PrivacyFlag.valueOf(privacy);
    }

    protected void fillPrivacyFlagIn(IBaseDTO baseDTO) {
        baseDTO.getBaseFields().setPrivacyFlag(getNewFlag());
    }

    protected void fillPrivacyFlagIfSetIn(IBaseDTO baseDTO) {
        if (privacy != null) {
            fillPrivacyFlagIn(baseDTO);
        }
    }

    protected void setDefaultFlag() {
        if (privacy == null) {
            setDefaultFlagToTeamDefault();
        } else {
            fillPrivacyFlagsFrom(privacy);
        }
    }

    protected void setDefaultFlagToTeamDefault() {
        if (isLoggedInUserProfessional()) {
            Team callerTeam = loggedInUserUtil.getLoggedInUserPrimaryTeam();
            if (callerTeam != null) {
                fillPrivacyFlagsFrom(callerTeam);
            }
        }
    }

    protected void disablePrivacyFlagEditForm() {
        if (privacy == null) {
            setEditConsentModuleEnabled(false);
        }
    }

    protected <T extends IBaseDTO> String checkPermissionsAndDo(BiConsumer<EHRRequestContext, T> target, Supplier<Optional<T>> fetchDto, String ret, boolean fillPrivacyFlags, BaseAction.Users users) {
        EHRRequestContext ehrRequestContext = getEHRRequestContext();
        Optional<T> optionalDto = fetchDto.get();
        if (optionalDto.isEmpty()) {
            throw new RuntimeException("File does not exist");
        }
        T dto = optionalDto.get();
        if (hasNotPermissionsToEditPrivacyLabels(dto, users.loggedInUser(), loggedInUserUtil.getLoggedInUserPrimaryTeam())) {
            return CANCEL;
        }
        if (fillPrivacyFlags) {
            fillPrivacyFlagsFrom(dto);
        }
        target.accept(ehrRequestContext, dto);
        return ret;
    }

    //Temporary hack to get correct MenuDataType...
    protected MenuDataType getType(Map<String, MenuDataType> options) {
        String requestURI = getRequest().getRequestURI();
        return options.entrySet().stream()
                .map(e -> requestURI.contains(e.getKey()) ? e.getValue() : null)
                .filter(Objects::nonNull)
                .findAny().orElseThrow(() -> new NotImplementedException("Can't get menuDataType for url!"));
    }

    // See http://dev.patientsknowbest.com/home/hl7-api/code-sets#TOC-PKBCS04
    protected MenuDataType getType(Encounter encounter) {
        switch (encounter.getEncounterClass()) {
            case VIRTUAL:
                return MenuDataType.message;
            case IN_PATIENT:
            case OUT_PATIENT:
            case AMBULATORY:
            case EMERGENCY:
            case HOME:
            case FIELD:
            case DAYTIME:
            case OTHER:
                return MenuDataType.event;
            default:
                throw new IllegalStateException("No MenuDataType associated with the Encounter");
        }
    }
}
