package com.pkb.action;

import com.pkb.datamodel.Email;
import com.pkb.user.entity.PKBPerson;

public class PromptLogoutAction extends BaseAction {

    private static final long serialVersionUID = 1L;

    @Override
    public String execute() {
        PKBPerson currentLoggedInUser = loggedInUserUtil.getLoggedInPerson();
        addActionError(getText("promptLogoutAction.err.logged_in_as_someone_else",
                new String[]{currentLoggedInUser.findEmail().map(Email::address).getOrNull()}));
        return "promptLogout";
    }
}
