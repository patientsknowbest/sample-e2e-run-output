package com.pkb.service.team;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.Affiliate;
import com.pkb.datamodel.team.AffiliateTeamWithOrg;
import com.pkb.datamodel.team.ImmutableAffiliateTeamWithOrg;
import com.pkb.domain.AffiliateService;
import com.pkb.exception.PKBException;
import com.pkb.exception.PKBPluginException;
import com.pkb.institute.entity.Team;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;

import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

public class AffiliateManager extends TransactionManager {

    private final AffiliateService affiliateService;

    public AffiliateManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, AffiliateService affiliateService) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.affiliateService = affiliateService;
    }

    public void deleteAffiliate(Long affiliateId) {
        try {
            affiliateService.deleteAffiliateById(affiliateId);
        } catch (PKBPluginException e) {
            throw new PKBException("Error while deleting affiliate-" + affiliateId, e);
        }
    }

    public List<Affiliate> getAffiliates(Long teamId) {
        try {
            return affiliateService.getAffiliatesByTeamId(teamId);
        } catch (PKBPluginException e) {
            throw new PKBException("Error while getting affiliates for team-" + teamId, e);
        }
    }

    public Long saveAffiliate(Affiliate affiliate) {
        try {
            return affiliateService.save(affiliate);
        } catch (PKBPluginException e) {
            throw new PKBException("Error while saving affiliate", e);
        }
    }

    public Affiliate getAffiliate(Long affiliateId) {
        try {
            return affiliateService.findAffiliateById(affiliateId)
                    .orElseThrow(() -> new IllegalArgumentException(String.format("There is no affiliate with id: %d", affiliateId)));
        } catch (PKBPluginException e) {
            throw new PKBException("Error while getting affiliate for id-" + affiliateId, e);
        }
    }

    public List<AffiliateTeamWithOrg> getAffiliateTeamWithOrgsForTeams(List<Team> teams, Map<Long, Long> affiliateIdMap) {
        return teams.stream()
                .map(currentTeam -> ImmutableAffiliateTeamWithOrg.builder()
                        .teamName(currentTeam.getName())
                        .orgName(currentTeam.getOrg().getName())
                        .teamCode(currentTeam.getCode())
                        .affiliateId(affiliateIdMap.get(currentTeam.getId()))
                        .build())
                .collect(Collectors.toUnmodifiableList());
    }
}
