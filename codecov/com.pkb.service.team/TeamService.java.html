package com.pkb.service.team;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.datamodel.consent.PatientConsentUpdateDTO;
import com.pkb.domain.CryptoModelSynchronisationService;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.model.RecordWithId;
import com.pkb.service.emailmessage.impl.ImmutableEmailInfo;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.user.entity.PKBPerson;
import io.vavr.Tuple2;

import java.util.Collection;
import java.util.Collections;
import java.util.HashSet;
import java.util.List;
import java.util.Map;
import java.util.Set;
import java.util.function.Function;
import java.util.stream.Stream;

import static com.pkb.service.notification.impl.tolven.AccessUpdate.ACCESS_GRANT;
import static io.micrometer.core.instrument.Metrics.counter;
import static java.util.function.Function.identity;
import static java.util.stream.Collectors.toMap;
import static java.util.stream.Collectors.toSet;

public class TeamService {

    private static final String ADDED_TO_TEAM_COUNTER_NAME = "pkb_phr_integration_addedtoteam";
    private static final String ADDED_TO_TEAM_TAG_KEY = "success";
    private static final String ADDED_TO_TEAM_ROUTE_KEY = "route";


    private final PatientConsentManager patientConsentManager;
    private final PKBEmailMessageManager emailMessageManager;

    public TeamService(PatientConsentManager patientConsentManager, PKBEmailMessageManager emailMessageManager,
                       CryptoModelSynchronisationService cryptoModelSynchronisationService) {
        this.patientConsentManager = patientConsentManager;
        this.emailMessageManager = emailMessageManager;
    }

    /**
     * It adds an existing patient to creation teams by creating consent and institute user records.
     * Patient and carers (if any) receive notifications.
     *
     * @return number of teams where the patient was added to
     */
    public long addToTeams(PKBPerson patient, Collection<Team> creationTeams, EHRRequestContext context) {
        if (patient == null) {
            return 0;
        }

        return createConsentRecords(patient, creationTeams, context);
    }

    public <ID> Stream<RecordWithId<ID, Long>> addToTeams(PKBPerson patient, List<RecordWithId<ID, Tuple2<EHRRequestContext, Set<Team>>>> teamsList) {
        if (patient == null || !patient.isSharingEnabled()) {
            return Stream.empty();
        }

        Map<Long, Team> teamMap = teamsList.stream()
                .flatMap(r -> r.record()._2.stream())
                .collect(toMap(Team::getId, identity(), (t1, t2) -> t1));

        var teamConsents = patientConsentManager.hasActiveTeamConsent(Collections.singleton(patient.getId()), teamMap.keySet());

        Set<Long> teamsWithoutConsent = teamMap.values().stream()
                .map(Team::getId)
                .filter(teamId -> {
                    Boolean activeTeamConsent = teamConsents.get(patient.getId(), teamId);
                    return activeTeamConsent == null || Boolean.FALSE.equals(activeTeamConsent);
                })
                .collect(toSet());


        //This makes sure we only add each team once, and in the order they would be added wihtout batching
        Set<Long> mutableTeamWithoutConsent = new HashSet<>(teamsWithoutConsent);
        Function<Team, Boolean> consentProvider = team -> !mutableTeamWithoutConsent.remove(team.getId());

        return teamsList.stream().map(r -> {
            var contextAndCreationTeams = r.record();
            long countCreated = createConsentRecords(patient, contextAndCreationTeams._2, contextAndCreationTeams._1, consentProvider);
            return r.switchRecord(countCreated);
        });
    }

    public boolean hasActiveConsentRecord(PKBPerson person, Org org) {
        return patientConsentManager.hasActiveConsentRecord(person, org);
    }


    private long createConsentRecords(PKBPerson patient, Collection<Team> creationTeams, EHRRequestContext context) {
        return createConsentRecords(patient, creationTeams, context, team -> patientConsentManager.hasActiveConsentRecord(patient, team));
    }

    private long createConsentRecords(PKBPerson patient, Collection<Team> creationTeams, EHRRequestContext context, Function<Team, Boolean> consentProvider) {
        counter(ADDED_TO_TEAM_COUNTER_NAME, ADDED_TO_TEAM_TAG_KEY, "true", ADDED_TO_TEAM_ROUTE_KEY, context.getAccessRoute().toString()).increment();
        if (!patient.isSharingEnabled()) {
            return 0;
        }

        return creationTeams.stream()
                .filter(team -> noConsentForTeam(team, consentProvider))
                .map(team -> createConsentRecordsAndNotify(patient, team, context))
                .count();
    }

    private boolean noConsentForTeam(Team team, Function<Team, Boolean> consentProvider) {
        return !consentProvider.apply(team);
    }

    private PatientConsentUpdateDTO createConsentRecordsAndNotify(PKBPerson patient, Team team, EHRRequestContext context) {
        var consent = patientConsentManager.grantDefaultConsentToTeam(context, patient, team);
        emailMessageManager.sendTeamAccessNotificationToPatient(ACCESS_GRANT,
                ImmutableEmailInfo.builder()
                        .recipient(patient)
                        .team(team)
                        .includeSender(false)
                        .build(),
                consent);
        return consent;
    }
}
