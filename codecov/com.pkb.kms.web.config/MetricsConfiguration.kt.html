package com.pkb.kms.web.config

import com.pkb.kms.web.KmsWebApplication
import com.pkb.kms.web.KmsWebMvcTagsContributor
import io.micrometer.core.instrument.MeterRegistry
import org.springframework.boot.actuate.autoconfigure.metrics.MeterRegistryCustomizer
import org.springframework.boot.actuate.metrics.web.servlet.WebMvcTagsContributor
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration

@Configuration
class MetricsConfiguration {

    @Bean
    fun metricsCommonTags(): MeterRegistryCustomizer<MeterRegistry> {
        val version = KmsWebApplication::class.java.`package`.implementationVersion?.toString() ?: "unknown"
        return MeterRegistryCustomizer { registry ->
            registry.config()
                    .commonTags(
                            "application", "kms",
                            "version", version,
                            "cacheManager", "cacheManager")
        }
    }

    @Bean
    fun webMvcTagsContributor() : WebMvcTagsContributor = KmsWebMvcTagsContributor()
}
