package com.pkb.kms.web.config

import com.google.common.base.MoreObjects
import com.pkb.kms.service.cache.PublishingCacheManager
import com.pkb.kms.service.routing.KeyAvailableRouteBuilder
import org.apache.camel.Produce
import org.apache.camel.ProducerTemplate
import org.redisson.Redisson
import org.redisson.api.RedissonClient
import org.redisson.config.Config
import org.redisson.spring.cache.RedissonSpringCacheManager
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.beans.factory.annotation.Value
import org.springframework.boot.context.properties.ConfigurationProperties
import org.springframework.cache.CacheManager
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import java.lang.invoke.MethodHandles
import org.redisson.spring.cache.CacheConfig as RedisCacheConfig

@Configuration
class CacheConfig(@Produce(KeyAvailableRouteBuilder.OUTBOUND_USER_KEY_AVAILABLE) private val userKeyAvailableProducer: ProducerTemplate,
                  @Produce(KeyAvailableRouteBuilder.OUTBOUND_ACCOUNT_KEY_AVAILABLE) private val accountKeyAvailableProducer: ProducerTemplate,
                  @Produce(KeyAvailableRouteBuilder.OUTBOUND_ORG_KEY_AVAILABLE) private val orgKeyAvailableProducer: ProducerTemplate
) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @Bean
    fun redisson(config: RedisConfigProperties): RedissonClient = Redisson.create(Config().apply {
        useSingleServer().address = "redis://${config.hostname}:${config.port}"
    })

    @Bean
    fun cacheManager(redissonClient: RedissonClient,
                     cacheConfig: KmsCacheProperties): CacheManager {
        log.info("Using cache configuration $cacheConfig")
        return PublishingCacheManager(RedissonSpringCacheManager(redissonClient, cacheConfig.asRedisConfigMap()),
                userKeyAvailableProducer, accountKeyAvailableProducer, orgKeyAvailableProducer)
    }
}

@Configuration
@ConfigurationProperties("redis")
class RedisConfigProperties {
    var hostname: String = "localhost"
    var port: Int = 6379
}

@Configuration
class KmsCacheProperties {

    @Value("#{\${kms.cache.config}}")
    var map: Map<String, Map<String, Long>> = mutableMapOf()

    fun asRedisConfigMap(): MutableMap<String, RedisCacheConfig> =
            map.mapValues { kv ->
                RedisCacheConfig(
                        kv.value["ttl"] ?: 0L,
                        kv.value["maxIdleTime"] ?: 0L)
            }.toMutableMap()

    override fun toString(): String {
        return MoreObjects.toStringHelper(KmsCacheProperties::class.java)
                .omitNullValues()
                .add("map", map)
                .toString()
    }
}
