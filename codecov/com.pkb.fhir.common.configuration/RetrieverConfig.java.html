package com.pkb.fhir.common.configuration;

import com.pkb.entities.enums.MenuDataType;
import com.pkb.fhir.common.decrypt.BaseEncryptedEhrDataDecryptor;
import com.pkb.fhir.common.decrypt.RetryingEhrDataDecryptor;
import com.pkb.fhir.common.repository.MenuDataRepository;
import com.pkb.fhir.common.repository.SourceEhrDataRepository;
import com.pkb.fhir.common.retrieve.AllergyEhrDataRetriever;
import com.pkb.fhir.common.retrieve.AppointmentEhrDataRetriever;
import com.pkb.fhir.common.retrieve.ConditionEhrDataRetriever;
import com.pkb.fhir.common.retrieve.DataTypeBasedEhrDataRetriever;
import com.pkb.fhir.common.retrieve.EhrDataRetriever;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.EnumMap;
import java.util.Map;

@Configuration
public class RetrieverConfig {

    @Bean(name = "RetrieverEncryptedEhrDataDecryptor")
    public BaseEncryptedEhrDataDecryptor baseEncryptedEhrDataDecryptor(MenuDataRepository menuDataRepository, RetryingEhrDataDecryptor ehrDataDecryptor) {
        return new BaseEncryptedEhrDataDecryptor(menuDataRepository, ehrDataDecryptor);
    }


    @Bean
    public AllergyEhrDataRetriever allergyEhrDataRetriever(MenuDataRepository menuDataRepository) {
        return new AllergyEhrDataRetriever(menuDataRepository);
    }

    @Bean
    public AppointmentEhrDataRetriever appointmentEhrDataRetriever(SourceEhrDataRepository sourceEhrDataRepository,
                                                                   MenuDataRepository menuDataRepository,
                                                                   @Qualifier("RetrieverEncryptedEhrDataDecryptor") BaseEncryptedEhrDataDecryptor encryptedEhrDataDecryptor) {
        return new AppointmentEhrDataRetriever(sourceEhrDataRepository, menuDataRepository, encryptedEhrDataDecryptor);
    }

    @Bean
    public ConditionEhrDataRetriever conditionEhrDataRetriever(SourceEhrDataRepository sourceEhrDataRepository) {
        return new ConditionEhrDataRetriever(sourceEhrDataRepository);
    }

    @Bean
    public Map<MenuDataType, EhrDataRetriever> ehrDataRetrieverMap(AllergyEhrDataRetriever allergyEhrDataRetriever,
                                                                   AppointmentEhrDataRetriever appointmentEhrDataRetriever,
                                                                   ConditionEhrDataRetriever conditionEhrDataRetriever) {
        EnumMap<MenuDataType, EhrDataRetriever> result = new EnumMap<>(MenuDataType.class);
        result.put(MenuDataType.allergy, allergyEhrDataRetriever);
        result.put(MenuDataType.appointment, appointmentEhrDataRetriever);
        result.put(MenuDataType.diagnosis, conditionEhrDataRetriever);
        return result;
    }

    @Bean
    public DataTypeBasedEhrDataRetriever dataTypeBasedEhrDataRetriever(Map<MenuDataType, EhrDataRetriever> ehrDataRetrieverMap) {
        return new DataTypeBasedEhrDataRetriever(ehrDataRetrieverMap);
    }
}
