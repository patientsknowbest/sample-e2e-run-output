package com.pkb.fhir.common.configuration;

import com.pkb.common.config.BaseConfig;

public interface EhrmigratorConfig extends BaseConfig {

    default boolean isSyncToFhirStoreEnabled() {
        return getConfigStorage().getBoolean("syncfhirstore.enabled", false);
    }

    default int getFhirStoreSyncBatchSize() {
        return getConfigStorage().getInt("syncfhirstore.batchsize", 1000);
    }

    default boolean getFhirSyncDebugLoggingEnabled() {
        return getConfigStorage().getBoolean("syncfhirstore.debuglogging", false);
    }

    default int getSyncRunnerThreadCount() {
        return getConfigStorage().getInt("syncfhirstore.threadcount", 20);
    }

    default long getSyncRunnerBackoffInitialInterval() {
        return getConfigStorage().getLong("syncfhirstore.backoff.initialinterval", 100);
    }

    default long getSyncRunnerBackoffMaxInterval() {
        return getConfigStorage().getLong("syncfhirstore.backoff.maxinterval", 30000);
    }

    default long getSyncRunnerBackoffMultiplier() {
        return getConfigStorage().getLong("syncfhirstore.backoff.multiplier", 2);
    }

    default String getSourceDbTimezone() {
        return getConfigStorage().getString("app.sourceDbTimezone", "UTC");
    }
}
