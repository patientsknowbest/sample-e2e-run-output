package com.pkb.fhir.common.configuration;

import ca.uhn.fhir.context.FhirContext;
import com.pkb.common.ehrdata.serialization.EHRDataDeserializer;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.fhir.common.service.OrgService;
import com.pkb.fhir.common.service.PersonService;
import com.pkb.fhir.common.service.TeamService;
import com.pkb.fhir.common.transform.AppointmentEhrDataTransformer;
import com.pkb.fhir.common.transform.BaseEhrDataTransformer;
import com.pkb.fhir.common.transform.ConditionEhrDataTransformer;
import com.pkb.fhir.migration.service.Hl7PartnerService;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.EnumMap;
import java.util.Map;

@Configuration
public class TransformationConfig {

    @Bean
    public AppointmentEhrDataTransformer appointmentEhrDataTransformer(FhirContext context, PersonService personService, TeamService teamService,
                                                                       OrgService orgService, EhrmigratorConfig config, Hl7PartnerService hl7PartnerService,
                                                                       EHRDataDeserializer ehrDataDeserializer) {
        return new AppointmentEhrDataTransformer(context, personService, teamService,
                orgService, config, hl7PartnerService, ehrDataDeserializer);
    }

    @Bean
    public ConditionEhrDataTransformer conditionEhrDataTransformer(EhrmigratorConfig config, PersonService personService, TeamService teamService,
                                                                   OrgService orgService, FhirContext context, Hl7PartnerService hl7PartnerService,
                                                                   EHRDataDeserializer ehrDataDeserializer) {
        return new ConditionEhrDataTransformer(config, personService, teamService,
                orgService, context, hl7PartnerService, ehrDataDeserializer);
    }

    @Bean
    public Map<MenuDataType, BaseEhrDataTransformer> ehrDataTransformerMap(AppointmentEhrDataTransformer appointmentEhrDataTransformer,
                                                                           ConditionEhrDataTransformer conditionEhrDataTransformer) {
        EnumMap<MenuDataType, BaseEhrDataTransformer> result = new EnumMap<>(MenuDataType.class);
        result.put(MenuDataType.appointment, appointmentEhrDataTransformer);
        result.put(MenuDataType.diagnosis, conditionEhrDataTransformer);
        return result;
    }

    @Bean
    public EHRDataDeserializer ehrDataDeserializer() {
        return new EHRDataDeserializer() {};
    }
}
