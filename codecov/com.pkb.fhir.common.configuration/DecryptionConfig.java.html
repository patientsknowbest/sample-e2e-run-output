package com.pkb.fhir.common.configuration;

import com.pkb.crypto.decryptionhelper.DecryptionHelper;
import com.pkb.fhir.common.decrypt.AccountKeyProvider;
import com.pkb.fhir.common.decrypt.EhrDataDecryptor;
import com.pkb.fhir.common.decrypt.RetryingEhrDataDecryptor;
import com.pkb.kms.client.core.Kms;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class DecryptionConfig {

    @Bean
    public AccountKeyProvider accountKeyProvider(Kms kmsClient) {
        return new AccountKeyProvider(kmsClient);
    }

    @Bean
    public DecryptionHelper decryptionHelper() {
        return new DecryptionHelper();
    }

    @Bean
    public EhrDataDecryptor defaultEhrDataDecryptor(AccountKeyProvider accountKeyProvider, DecryptionHelper decryptionHelper) {
        return new EhrDataDecryptor(accountKeyProvider, decryptionHelper);
    }

    @Bean
    public RetryingEhrDataDecryptor retryingEhrDataDecryptor(EhrDataDecryptor ehrDataDecryptor) {
        return new RetryingEhrDataDecryptor(ehrDataDecryptor);
    }
}
