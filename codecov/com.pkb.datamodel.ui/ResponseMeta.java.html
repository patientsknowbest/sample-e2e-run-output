package com.pkb.datamodel.ui;

import org.apache.commons.collections4.CollectionUtils;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.HashMap;
import java.util.LinkedList;
import java.util.List;
import java.util.Map;

import static java.util.Collections.emptyList;
import static java.util.Collections.emptyMap;
import static org.apache.commons.lang3.ObjectUtils.isNotEmpty;

public abstract class ResponseMeta {

    public abstract List<String> getInfoMessages();

    public abstract List<String> getActionWarnings();

    public abstract List<String> getActionErrors();

    public abstract Map<String, List<String>> getFieldErrors();

    private void addActionError(String message) {
        getActionErrors().add(message);
    }

    private void addActionWarning(String warning) {
        getActionWarnings().add(warning);
    }

    public void addFieldError(String field, String message) {
        getFieldErrors().computeIfAbsent(field, k -> new LinkedList<>()).add(message);
    }

    private void addInfoMessage(String infoMessage) {
        getInfoMessages().add(infoMessage);
    }

    public boolean hasErrors() {
        return isNotEmpty(getFieldErrors()) || CollectionUtils.isNotEmpty(getActionErrors());
    }

    public void merge(ResponseMeta meta) {
        meta.getActionErrors().forEach(error -> {
            if (!getActionErrors().contains(error)) {
                addActionError(error);
            }
        });

        meta.getActionWarnings().forEach(warning -> {
            if (!getActionWarnings().contains(warning)) {
                addActionWarning(warning);
            }
        });

        meta.getFieldErrors().forEach((field, fieldErrors) -> {
            getFieldErrors().merge(field, fieldErrors, (oldErrors, newErrors) -> {
                newErrors.forEach(error -> {
                    if (!oldErrors.contains(error)) {
                        oldErrors.add(error);
                    }
                });
                return oldErrors;
            });
        });

        meta.getInfoMessages().forEach(infoMessage -> {
            if (!getInfoMessages().contains(infoMessage)) {
                addInfoMessage(infoMessage);
            }
        });
    }

    public static ResponseMeta of(List<String> infoMessages,
                                  List<String> actionWarnings,
                                  List<String> actionErrors,
                                  Map<String, List<String>> fieldErrors) {
        return new ResponseMeta() {
            @Override
            public List<String> getInfoMessages() {
                return infoMessages;
            }

            @Override
            public List<String> getActionWarnings() {
                return actionWarnings;
            }

            @Override
            public List<String> getActionErrors() {
                return actionErrors;
            }

            @Override
            public Map<String, List<String>> getFieldErrors() {
                return fieldErrors;
            }
        };
    }

    public static ResponseMeta empty() {
        return of(new ArrayList<>(), new ArrayList<>(), new ArrayList<>(), new HashMap<>());
    }

    public static ResponseMeta ofInfoMessage(String infoMessage) {
        return ofInfoMessages(Arrays.asList(infoMessage));
    }

    public static ResponseMeta ofInfoMessages(List<String> infoMessages) {
        return  of(infoMessages, emptyList(), emptyList(), emptyMap());
    }
    public static ResponseMeta ofActionWarning(String actionWarning) {
        return ofActionWarnings(List.of(actionWarning));
    }

    public static ResponseMeta ofActionWarnings(List<String> actionWarnings) {
        return  of(emptyList(), actionWarnings, emptyList(), emptyMap());
    }

    public static ResponseMeta ofActionError(String actionError) {
        return ofActionErrors(Arrays.asList(actionError));
    }

    public static ResponseMeta ofActionErrors(List<String> actionErrors) {
        return of(emptyList(),emptyList(), actionErrors, emptyMap());
    }

    public static ResponseMeta ofFieldErrors(Map<String, List<String>> fieldErrors) {
        return of(emptyList(), emptyList(), emptyList(), fieldErrors);
    }
}
