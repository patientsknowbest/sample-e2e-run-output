@Repository
interface QuestionnaireResponseRepository : PagingAndSortingRepository<QuestionnaireResponseEntity, QuestionnaireResponseId>, QuestionnaireResponseRepositoryCustom,
                                            JpaSpecificationExecutor<QuestionnaireResponseEntity> {

    @Query("""
    SELECT response 
    FROM QuestionnaireResponseEntity response 
    WHERE response.questionnaireRequest.questionnaire.id = :questionnaireId
    ORDER BY response.createdDate, response.patientAccountId, response.id      
    """)
    @EntityGraph(attributePaths = ["questionnaireRequest", "questionnaireRequest.identifiers"])
    fun findByQuestionnaireId(questionnaireId: QuestionnaireId, pageable: Pageable): Page<QuestionnaireResponseEntity>

    @EntityGraph(attributePaths = ["questionnaireRequest", "questionnaireRequest.identifiers"])
    override fun findAll(spec: Specification<QuestionnaireResponseEntity>?, pageable: Pageable): Page<QuestionnaireResponseEntity>

    @Query("""
    SELECT new com.pkb.questionnaireservice.modelv2.QuestionnaireResponseCountByQuestionnaireId(response.questionnaireRequest.questionnaire.id, COUNT (response))
    FROM QuestionnaireResponseEntity response
    WHERE response.questionnaireRequest.questionnaire.id IN (:questionnaireIds)
    GROUP BY response.questionnaireRequest.questionnaire.id
    ORDER BY response.questionnaireRequest.questionnaire.id
    """)
    fun countQuestionnaireResponsesByQuestionnaireId(questionnaireIds: List<QuestionnaireId>): List<QuestionnaireResponseCountByQuestionnaireId>
}


open class QuestionnaireResponseRepositoryImpl : QuestionnaireResponseRepositoryCustom {

    @PersistenceContext
    private lateinit var entityManager: EntityManager

    @Autowired
    @org.springframework.context.annotation.Lazy //Stops spring bitching about circular references https://github.com/spring-projects/spring-data-commons/issues/2319
    private lateinit var repository: QuestionnaireResponseRepository


    override fun findResponses(orgId: OrgPublicId, searchCriteria: QuestionnaireResponsesSearchCriteria, pageable: Pageable): Page<QuestionnaireResponseEntity> =
            repository.findAll(
                    Specification.where(teamIdIn(searchCriteria.teamIds))
                            .and(patientIdIn(searchCriteria.patientIds))
                            .and(questionnaireIdIs(searchCriteria.questionnaireId))
                            .and(createdLowerThan(searchCriteria.ltDate))
                            .and(createdLowerOrEqual(searchCriteria.leDate))
                            .and(createdGreaterOrEqual(searchCriteria.geDate))
                            .and(createdGreaterThan(searchCriteria.gtDate)),
                    pageable)

    override fun save(questionnaireResponse: DecryptedQuestionnaireResponse, encryptionDetails: EncryptionDetails): Either<QuestionnaireServiceError, QuestionnaireResponse> {
        val encryptedResponse = QuestionnaireResponseEntity(
                questionnaireResponse.id,
                entityManager.getReference(QuestionnaireRequestEntity::class.java, questionnaireResponse.questionnaireRequest.id),
                questionnaireResponse.createdDate,
                questionnaireResponse.completedBy,
                questionnaireResponse.patientAccountId,
                encryptionDetails.nonce,
                encryptionDetails.dplKeyId,
                encryptionDetails.encryptedBytes)
        return kotlin.runCatching {
            repository.save(encryptedResponse)
        }
                .map { Either.right<QuestionnaireServiceError, QuestionnaireResponse>(questionnaireResponse) }
                .getOrElse { Either.left(mapRepositoryErrorOnSave(it, questionnaireResponse)) }
    }


    private fun teamIdIn(teamIds: List<TeamPublicId>) = Specification { root: Root<QuestionnaireResponseEntity>, cq, cb ->
        cq.orderBy(cb.asc(root.get(QuestionnaireResponseEntity_.createdDate)),
                cb.asc(root.get(QuestionnaireResponseEntity_.patientAccountId)),
                cb.asc(root.get(QuestionnaireResponseEntity_.id)))
        root.get(QuestionnaireResponseEntity_.questionnaireRequest)
                .get(QuestionnaireRequestEntity_.requestingTeamId)
                .`in`(teamIds)
    }

    private fun patientIdIn(patientIds: Optional<List<PersonPublicId>>) = Specification { root: Root<QuestionnaireResponseEntity>, _, cb ->
        patientIds.map {
            root.get(QuestionnaireResponseEntity_.questionnaireRequest)
                    .get(QuestionnaireRequestEntity_.patientId)
                    .`in`(it)
        }.orElseGet { cb.and() }
    }

    private fun questionnaireIdIs(questId: Optional<QuestionnaireId>) = Specification { root: Root<QuestionnaireResponseEntity>, _, cb ->
        questId.map {
            cb.equal(root.get(QuestionnaireResponseEntity_.questionnaireRequest)
                    .get(QuestionnaireRequestEntity_.questionnaire)
                    .get(QuestionnaireEntity_.id),
                    it)
        }.orElseGet { cb.and() }
    }

    private fun createdGreaterOrEqual(geDate: Optional<Instant>) = Specification { root: Root<QuestionnaireResponseEntity>, _, cb ->
        geDate.map { cb.greaterThanOrEqualTo(root.get(QuestionnaireResponseEntity_.createdDate), it) }.orElseGet { cb.and() }
    }

    private fun createdGreaterThan(gtDate: Optional<Instant>) = Specification { root: Root<QuestionnaireResponseEntity>, _, cb ->
        gtDate.map { cb.greaterThan(root.get(QuestionnaireResponseEntity_.createdDate), it) }.orElseGet { cb.and() }
    }

    private fun createdLowerThan(ltDate: Optional<Instant>) = Specification { root: Root<QuestionnaireResponseEntity>, _, cb ->
        ltDate.map { cb.lessThan(root.get(QuestionnaireResponseEntity_.createdDate), it) }.orElseGet { cb.and() }
    }

    private fun createdLowerOrEqual(leDate: Optional<Instant>) = Specification { root: Root<QuestionnaireResponseEntity>, _, cb ->
        leDate.map { cb.lessThanOrEqualTo(root.get(QuestionnaireResponseEntity_.createdDate), it) }.orElseGet { cb.and() }
    }

    override fun countQuestionnaireResponses(questionnaireIds: List<QuestionnaireId>) =
        repository.countQuestionnaireResponsesByQuestionnaireId(questionnaireIds)
}

}
