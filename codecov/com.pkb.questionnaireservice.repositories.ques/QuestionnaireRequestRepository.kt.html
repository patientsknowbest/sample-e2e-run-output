package com.pkb.questionnaireservice.repositories.ques


import com.pkb.questionnaireservice.QuestionnaireId
import com.pkb.questionnaireservice.QuestionnaireRequestId
import com.pkb.questionnaireservice.entities.QuestionnaireRequestEntity
import org.springframework.data.domain.Page
import org.springframework.data.domain.Pageable
import org.springframework.data.jpa.repository.EntityGraph
import org.springframework.data.jpa.repository.Query
import org.springframework.data.repository.CrudRepository

interface QuestionnaireRequestRepository : CrudRepository<QuestionnaireRequestEntity, QuestionnaireRequestId> {

    @EntityGraph(attributePaths = ["questionnaire", "identifiers"])
    @Query(value = """
        SELECT request FROM QuestionnaireRequestEntity request
        WHERE request.questionnaire.id IN :questionnaireIds
        ORDER BY request.createdDate, request.patientId, request.id 
    """
    )
    fun findByQuestionnaireIds(questionnaireIds: List<QuestionnaireId>, pageable: Pageable = Pageable.unpaged()): Page<QuestionnaireRequestEntity>

    @EntityGraph(attributePaths = ["questionnaire", "identifiers"])
    @Query(value = """
        SELECT request FROM QuestionnaireRequestEntity request
        LEFT OUTER JOIN QuestionnaireResponseEntity response ON response.questionnaireRequest = request
        WHERE request.questionnaire.id IN :questionnaireIds AND response is null
        ORDER BY request.createdDate DESC, request.patientId, request.id 
    """)
    fun getNonRespondersForQuestionnaires(questionnaireIds: List<QuestionnaireId>): List<QuestionnaireRequestEntity>
}
