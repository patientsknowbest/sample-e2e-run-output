package com.pkb.questionnaireservice.repositories.ques

import com.fasterxml.jackson.annotation.ObjectIdGenerator
import com.fasterxml.jackson.annotation.ObjectIdResolver
import com.fasterxml.jackson.annotation.SimpleObjectIdResolver
import com.fasterxml.jackson.databind.deser.impl.PropertyBasedObjectIdGenerator
import com.pkb.questionnaireservice.modelv2.Questionnaire
import com.pkb.questionnaireservice.modelv2.QuestionnaireComponent
import com.pkb.questionnaireservice.modelv2.QuestionnaireRequest
import javax.persistence.EntityManager
import javax.persistence.PersistenceContext

/**
 * This class allows to rehydrate complete model objects from their uuid
 */
class EntityManagerResolver: SimpleObjectIdResolver {

    constructor()

    constructor(em: EntityManager) {
        this.entityManager = em
    }


    @Suppress("JoinDeclarationAndAssignment")
    @PersistenceContext(unitName = "ques")
    lateinit var entityManager: EntityManager


    // If we need to deserialize the same Questionnaire several times, we have to check wether or not it was already created as a POJO.
    // In order to do that, we check if it was previously seen and if not, we bind it to the resolver items as well as all its components.
    override fun resolveId(id: ObjectIdGenerator.IdKey): Any? {
      return super.resolveId(id) ?: entityManager.find(id.scope, id.key)?.also{ resolved ->
          when (resolved) {
              is Questionnaire -> resolveQuestionnaire(resolved)
              is QuestionnaireRequest -> resolveQuestionnaire(resolved.questionnaire)
              else -> null
          }?.also { questionnaire ->
              this.bindItem(ObjectIdGenerator.IdKey(PropertyBasedObjectIdGenerator::class.java, Questionnaire::class.java, questionnaire.id), questionnaire)
              questionnaire.components.forEach{this.bindItem(ObjectIdGenerator.IdKey(PropertyBasedObjectIdGenerator::class.java, QuestionnaireComponent::class.java, it.id), it)}
          }
      }
    }
    
    private fun resolveQuestionnaire(questionnaire: Questionnaire): Questionnaire? {
        return if(super.resolveId(ObjectIdGenerator.IdKey(PropertyBasedObjectIdGenerator::class.java, Questionnaire::class.java, questionnaire.id)) != null) null
        else questionnaire
    }
    
    // While it may seem that this EntityManagerResolver would be persistent during the liftime of the application, it is actually scoped
    // to one request only. The below method (newForDeserialization) is meant to describe to jackson how to instantiate a new EntityManagerResolver
    // for each received request.
    override fun newForDeserialization(context: Any?): ObjectIdResolver? = EntityManagerResolver(entityManager)
    override fun canUseFor(resolverType: ObjectIdResolver?): Boolean = true
}
