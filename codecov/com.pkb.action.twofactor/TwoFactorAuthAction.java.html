package com.pkb.action.twofactor;

import com.pkb.security.PKBAuthenticationDetails;
import com.pkb.security.UserTypeToGrantedAuthorityMapper;
import com.pkb.user.entity.PKBPerson;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.security.core.GrantedAuthority;

import java.util.Collection;

import static com.pkb.security.PKBAuthenticationDetails.isTwoFactorAuthPreviouslyDone;
import static org.apache.commons.lang3.StringUtils.isNumeric;


public class TwoFactorAuthAction extends AbstractTwoFactorAction {

    private static final long serialVersionUID = 5099919234166664382L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());


    private static final String TWOFACTORAUTH_ERR_DURING_AUTH_DISPLAY = "twofactorauth.err.during.auth.display";
    private static final String TWOFACTORAUTH_ERROR_DURING_AUTH = "twofactorauth.err.duringauthentication";

    //Incoming params
    private String providedCode;

    public String displayauthorize() {
        String result = NONE;
        try {
            PKBPerson loggedInPKBPerson = getLoggedInPKBPerson();
            Object authenticationDetails = loggedInUserUtil.getAuthenticationDetails();
            if (twoFactorAuthenticationService.isTwoFactorCheckRequired(loggedInPKBPerson) && !isTwoFactorAuthPreviouslyDone(authenticationDetails)) {
                result = SUCCESS;
            }
        } catch (Exception e) {
            result = ERROR;
            addActionError(getText(TWOFACTORAUTH_ERR_DURING_AUTH_DISPLAY));
            LOGGER.error("Error during disaplying two factor authentication!", e);
        }
        return result;
    }

    public String authorize() {
        String result = NONE;
        try {
            if (isNumeric(providedCode)) {
                PKBPerson loggedInPKBPerson = getLoggedInPKBPerson();
                if (twoFactorAuthenticationService.authorize(loggedInPKBPerson, Integer.parseInt(providedCode))) {
                    updateRolesOfUser(loggedInPKBPerson);
                    result = SUCCESS;
                } else {
                    addFieldError("providedCode", getText(TWOFACTORAUTH_ERR_CODE_FAILED));
                    result = INPUT;
                }
            } else {
                addFieldError("providedCode", getText(TWOFACTORAUTH_ERR_CODE_FAILED));
                result = INPUT;
            }
        } catch (Exception e) {
            result = ERROR;
            addActionError(getText(TWOFACTORAUTH_ERROR_DURING_AUTH));
            LOGGER.error("Error during two factor authentication!", e);
        }
        return result;
    }

    public void setProvidedCode(String providedCode) {
        this.providedCode = providedCode;
    }

    private void updateRolesOfUser(PKBPerson loggedInPKBPerson) {
        PKBAuthenticationDetails newAuthDetails = PKBAuthenticationDetails.initializeIfSame(loggedInUserUtil.getAuthenticationDetails())
                .twoFactorAuthDone(true).build();

        Collection<GrantedAuthority> grantedAuthorities = UserTypeToGrantedAuthorityMapper
                .getAuthorityList(loggedInPKBPerson.getUserType());
        loggedInUserUtil.updateAuthentication(newAuthDetails, grantedAuthorities);
    }
}
