package com.pkb.action.twofactor;

import com.pkb.security.PKBAuthenticationDetails;
import com.pkb.user.entity.PKBPerson;
import org.apache.commons.lang3.NotImplementedException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Optional;

import static org.apache.commons.lang3.StringUtils.isNumeric;


/**
 * Action file to handle 2 factor authentication initialization.
 */
public class TwoFactorAuthManageAction extends AbstractTwoFactorAction {

    private static final long serialVersionUID = -5045049978867247958L;

    private static final String STATE_NOT_IMPLEMENTED_MSG = "TwoFactorAuth state not implemented!";

    private static final String ENABLE_SUCCESS_MESSAGE_CODE = "twofactorauth.msg.enabled";

    private static final String DISABLE_SUCCESS_MESSAGE_CODE = "twofactorauth.msg.disabled";

    private static final String TWOFACTORAUTH_MSG_INITIALIZATIONERROR_CODE = "twofactorauth.err.initializationerror";

    private static final String TWOFACTORAUTH_MSG_PROVIDED_KEY_ERROR_CODE = "twofactorauth.err.enable.providedcode.error";

    private static final String TWOFACTORAUTH_MSG_ENABLE_ERROR_CODE = "twofactorauth.err.enable.error";

    private static final String TWOFACTORAUTH_MSG_DISABLE_ERROR_CODE = "twofactorauth.err.disable.error";

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    //Incoming params
    private String providedCode;

    //Outgoing params
    private String qrCode;

    private String secret;

    private String account;

    private boolean mandatory = false;
    //Dependencies

    /**
     * Method to handle 2FA initialization step.
     */
    public String initialize2fa() {
        String result = NONE;
        try {
            if (twoFactorAuthenticationService.isServiceEnabled()) {
                PKBPerson person = getLoggedInPKBPerson();
                switch (person.getTwoFactorAuthState()) {
                    case ACTIVE:
                        result = "active";
                        break;
                    case DISABLED:
                        person = twoFactorAuthenticationService.initialize(getEHRRequestContext(), person);
                        fillViewFields(person);
                        result = SUCCESS;
                        break;
                    case INITIALIZED:
                        fillViewFields(person);
                        result = SUCCESS;
                        break;
                    default:
                        throw new NotImplementedException(STATE_NOT_IMPLEMENTED_MSG);
                }
                mandatory = twoFactorAuthenticationService.isTwoFactorAuthMandatory(person);
            }
        } catch (Exception e) {
            result = ERROR;
            addActionError(getText(TWOFACTORAUTH_MSG_INITIALIZATIONERROR_CODE));
            LOGGER.error("Error during two factor auth initialization!", e);
        }
        return result;
    }

    /**
     * Verifies the provided code and enables 2FA for user if the code is good.
     */
    public String enable2fa() {
        String result = NONE;
        try {
            if (twoFactorAuthenticationService.isServiceEnabled()) {
                PKBPerson person = getLoggedInPKBPerson();
                switch (person.getTwoFactorAuthState()) {
                    case INITIALIZED:
                        result = enable2fa(person);
                        break;
                    case ACTIVE:
                        result = "active";
                        break;
                    case DISABLED:
                        result = "init";
                        break;
                    default:
                        throw new NotImplementedException(STATE_NOT_IMPLEMENTED_MSG);
                }
                mandatory = twoFactorAuthenticationService.isTwoFactorAuthMandatory(person);
            }
        } catch (Exception e) {
            result = ERROR;
            addActionError(getText(TWOFACTORAUTH_MSG_ENABLE_ERROR_CODE));
            LOGGER.error("Error during enabling two factor auth!", e);
        }
        return result;
    }

    public String disable2fa() {
        String result = NONE;
        try {
            if (twoFactorAuthenticationService.isServiceEnabled()) {
                PKBPerson person = getLoggedInPKBPerson();
                switch (person.getTwoFactorAuthState()) {
                    case INITIALIZED:
                        result = NONE;
                        break;
                    case ACTIVE:
                        result = disable2fa(person);
                        break;
                    case DISABLED:
                        result = NONE;
                        break;
                    default:
                        throw new NotImplementedException(STATE_NOT_IMPLEMENTED_MSG);
                }
                mandatory = twoFactorAuthenticationService.isTwoFactorAuthMandatory(person);
            }
        } catch (Exception e) {
            result = ERROR;
            addActionError(getText(TWOFACTORAUTH_MSG_DISABLE_ERROR_CODE));
            LOGGER.error("Error during disabling two factor auth!", e);
        }
        return result;
    }

    public String getQrCode() {
        return qrCode;
    }

    public String getSecret() {
        return secret;
    }

    public void setProvidedCode(String providedCode) {
        this.providedCode = providedCode;
    }

    public String getAccount() {
        return account;
    }

    public boolean isMandatory() {
        return mandatory;
    }

    public void setMandatory(boolean mandatory) {
        this.mandatory = mandatory;
    }

    private String enable2fa(PKBPerson person) {
        try {
            if (isNumeric(providedCode)) {
                Optional<PKBPerson> maybePKBPerson = twoFactorAuthenticationService.activate(getEHRRequestContext(), person,
                        Integer.parseInt(providedCode));
                if (maybePKBPerson.isPresent()) {
                    addActionMessage(getText(ENABLE_SUCCESS_MESSAGE_CODE));
                    PKBAuthenticationDetails authDetails = setTwofactorAuthDoneForSession();
                    if (authDetails.isOauthTwoFactorAuth()) {
                        return "twoFaDoneOauthRedirect";
                    }
                    return SUCCESS;
                }
            }
            fillViewFields(person);
            return addProvidedCodeErrorAndGetResult();
        } catch (Exception e) {
            LOGGER.error("Error during enabling 2fa", e);
            fillViewFields(person);
            return addProvidedCodeErrorAndGetResult();
        }
    }

    private PKBAuthenticationDetails setTwofactorAuthDoneForSession() {
        PKBAuthenticationDetails newAuthDetails = PKBAuthenticationDetails.initializeIfSame(loggedInUserUtil.getAuthenticationDetails())
                .twoFactorAuthDone(true).build();
        loggedInUserUtil.updateAuthentication(newAuthDetails);
        return newAuthDetails;
    }

    private String disable2fa(PKBPerson person) {
        try {
            if (isNumeric(providedCode)) {
                var maybePKBPerson = twoFactorAuthenticationService.disable(getEHRRequestContext(), person, Integer.parseInt(providedCode))
                        .getOrElseThrow(s -> new IllegalArgumentException(s));
                if (maybePKBPerson.isPresent()) {
                    addActionMessage(getText(DISABLE_SUCCESS_MESSAGE_CODE));
                    return SUCCESS;
                }
            }
            return addProvidedCodeErrorAndGetResult();
        } catch (Exception e) {
            LOGGER.error("Error during disabling 2fa", e);
            return addProvidedCodeErrorAndGetResult();
        }
    }

    private String addProvidedCodeErrorAndGetResult() {
        addFieldError("providedCode", getText(TWOFACTORAUTH_MSG_PROVIDED_KEY_ERROR_CODE));
        return INPUT;
    }

    private void fillViewFields(PKBPerson person) {
        secret = person.getTwoFactorSecret();
        qrCode = twoFactorAuthenticationService.getQRCode(person);
        account = twoFactorAuthenticationService.getAccountForManualSetup(person);
    }
}
