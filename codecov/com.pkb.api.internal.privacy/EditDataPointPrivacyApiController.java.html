package com.pkb.api.internal.privacy;

import com.pkb.action.DtoType;
import com.pkb.action.EditPrivacyService;
import com.pkb.action.ImmutableFormRequest;
import com.pkb.action.ImmutableFormRequestAppointment;
import com.pkb.action.ImmutableSaveRequest;
import com.pkb.action.ImmutableSaveRequestAppointment;
import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.internal.AbstractInternalApiController;
import com.pkb.api.util.v2.RestUtil;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestBody;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.QueryParam;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import java.util.Locale;
import java.util.UUID;

import static com.pkb.entities.enums.api.ApiClientScope.CLINICIAN;
import static com.pkb.entities.enums.api.ApiClientScope.PATIENT;
import static java.text.MessageFormat.format;

@Controller
@Path(EditDataPointPrivacyApiController.EDIT_PRIVACY_PATH)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = { "Privacy" }, value = "API for edit data point privacy")
public class EditDataPointPrivacyApiController extends AbstractInternalApiController {

    static final String EDIT_PRIVACY_PATH = "/privacy";

    @Autowired
    private EditPrivacyService editPrivacyService;

    @Autowired
    private RestUtil restUtil;

    @GET
    @Path("/show")
    @ApiOperation(value = "Show data point privacy")
    @RequiresScope(scopes = { PATIENT, CLINICIAN })
    public <T> Response showPrivacyForm(
            @QueryParam("locale") String locale,
            @QueryParam("identifier") Long identifier,
            @QueryParam("type") String type,
            @QueryParam("nhsStyle") boolean nhsStyle) {
        var request = ImmutableFormRequest.builder()
                .eHRRequestContext(restUtil.getCommonParams().getRequestContext())
                .dtoType(buildType(type))
                .dtoIdentifier(identifier)
                .locale(Locale.forLanguageTag(locale))
                .nhsStyle(nhsStyle)
                .build();
        return okWithoutMessages(editPrivacyService.showPrivacyForm(request));
    }

    @GET
    @Path("/showAppointment")
    @ApiOperation(value = "Show data point privacy")
    @RequiresScope(scopes = { PATIENT, CLINICIAN })
    public <T> Response showPrivacyFormAppointment(
            @QueryParam("locale") String locale,
            @QueryParam("identifier") UUID identifier,
            @QueryParam("nhsStyle") boolean nhsStyle,
            @QueryParam("returnToAppointment") boolean returnToAppointment) {
        var request = ImmutableFormRequestAppointment.builder()
                .eHRRequestContext(restUtil.getCommonParams().getRequestContext())
                .dtoType(buildType("appointment"))
                .dtoIdentifier(identifier)
                .locale(Locale.forLanguageTag(locale))
                .nhsStyle(nhsStyle)
                .returnToAppointment(returnToAppointment)
                .build();
        return okWithoutMessages(editPrivacyService.showPrivacyForm(request));
    }

    @POST
    @Path("/edit")
    @ApiOperation(value = "Edit data point privacy")
    @RequiresScope(scopes = { PATIENT, CLINICIAN })
    public Response editPrivacy(@RequestBody EditDataPointPrivacyDto dto) {
        var request = ImmutableSaveRequest.builder()
                .eHRRequestContext(restUtil.getCommonParams().getRequestContext())
                .dtoType(buildType(dto.getType()))
                .dtoIdentifier(dto.getIdentifier())
                .locale(Locale.forLanguageTag(dto.getLocale()))
                .privacy(dto.getPrivacy())
                .nhsStyle(dto.getNhsStyle())
                .build();
        return okWithoutMessages(editPrivacyService.savePrivacy(request));
    }

    @POST
    @Path("/editAppointment")
    @ApiOperation(value = "Edit data point privacy")
    @RequiresScope(scopes = { PATIENT, CLINICIAN })
    public Response editPrivacyAppointment(@RequestBody EditAppointmentPrivacyDto dto) {
        var request = ImmutableSaveRequestAppointment.builder()
                .eHRRequestContext(restUtil.getCommonParams().getRequestContext())
                .dtoType(buildType(dto.getType()))
                .dtoIdentifier(dto.getIdentifier())
                .locale(Locale.forLanguageTag(dto.getLocale()))
                .privacy(dto.getPrivacy())
                .nhsStyle(dto.getNhsStyle())
                .returnToAppointment(dto.getReturnToAppointment())
                .build();
        return okWithoutMessages(editPrivacyService.savePrivacy(request));
    }

    private DtoType buildType(String type) {
        switch (type) {
            case "appointment":
                return DtoType.APPOINTMENT;
            case "library":
                return DtoType.LIBRARY;
            default:
                throw new RuntimeException(format("Unsupported entity type {0}", type));
        }
    }
}
