package org.tolven.security.key;

import org.tolven.core.entity.Account;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.io.Serializable;
import java.util.Base64;

@Entity
@Table(schema = "core", name = "account_symmetric_key")
public class AccountSymmetricKey implements Serializable {

    private static final long serialVersionUID = -3924492513137304698L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    @Column(name = "id")
    private Long id;

    @Column(name = "encrypted_symmetric_key_base64")
    private String encryptedSymmetricKeyBase64;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "account_id", referencedColumnName = "id", nullable = false)
    private Account account;

    @Column(name = "symmetric_key_algorithm")
    private String symmetricKeyAlgorithm;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }


    public byte[] getEncryptedSymmetricKey() {
        return Base64.getDecoder().decode(getEncryptedSymmetricKeyBase64());
    }

    public void setEncryptedSymmetricKey(byte[] encryptedSymmetricKey) {
        setEncryptedSymmetricKeyBase64(Base64.getEncoder().encodeToString(encryptedSymmetricKey));
    }

    public String getEncryptedSymmetricKeyBase64() {
        return encryptedSymmetricKeyBase64;
    }

    public void setEncryptedSymmetricKeyBase64(String encryptedSymmetricKeyBase64) {
        this.encryptedSymmetricKeyBase64 = encryptedSymmetricKeyBase64;
    }

    public String getSymmetricKeyAlgorithm() {
        return symmetricKeyAlgorithm;
    }

    public void setSymmetricKeyAlgorithm(String symmetricKeyAlgorithm) {
        this.symmetricKeyAlgorithm = symmetricKeyAlgorithm;
    }

    public Account getAccount() {
        return account;
    }

    public void setAccount(Account account) {
        this.account = account;
    }


}
