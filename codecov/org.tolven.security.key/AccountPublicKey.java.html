package org.tolven.security.key;

import javax.persistence.Column;
import javax.persistence.Embeddable;
import java.io.Serializable;
import java.security.KeyFactory;
import java.security.PublicKey;
import java.security.spec.X509EncodedKeySpec;

import static io.vavr.API.unchecked;

/**
 * This class encapsulates an x509EncodedKeySpec for a Public Key
 * 
 * @author Joseph Isaac
 * 
 */
@Embeddable
public class AccountPublicKey implements Serializable {

	private static final long serialVersionUID = 2L;

    private static final String INITIALIZED = "AccountPublicKey already initialized";
    private static final String NOT_INITIALIZED = "AccountPublicKey not initialized";

    @Column(name = "x509_encoded__key_spec_inline")
    private byte[] x509EncodedKeySpecInline;

    @Column(name = "public_key_algorithm")
    private String algorithm;

    private transient PublicKey publicKey;

    /**
     * Return an instance of UserPublicKey
     */
    public static AccountPublicKey getInstance() {
        return new AccountPublicKey();
    }

    /**
     * Initialize with aPublicKey
     */
    public void init(PublicKey aPublicKey) {
        if (x509EncodedKeySpecInline != null || algorithm != null) {
            throw new IllegalStateException(INITIALIZED);
        }
        X509EncodedKeySpec keySpec = new X509EncodedKeySpec(aPublicKey.getEncoded());
        x509EncodedKeySpecInline = keySpec.getEncoded();
        algorithm = aPublicKey.getAlgorithm();
    }

    /**
     * Decode and return the encapsulated PublicKey
     */
    public PublicKey getPublicKey() {
        if (publicKey == null) {
            if (x509EncodedKeySpecInline == null || algorithm == null) {
                throw new IllegalStateException(NOT_INITIALIZED);
            }
            X509EncodedKeySpec keySpec = new X509EncodedKeySpec(x509EncodedKeySpecInline);
            publicKey = unchecked(() -> KeyFactory.getInstance(algorithm).generatePublic(keySpec)).get();
        }
        return publicKey;
    }

    public String getAlgorithm() {
        return algorithm;
    }

    public byte[] getX509EncodedKeySpecInline() {
        return x509EncodedKeySpecInline;
    }
}
