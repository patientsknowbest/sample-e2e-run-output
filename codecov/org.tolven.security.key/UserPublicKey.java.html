package org.tolven.security.key;

import static io.vavr.API.unchecked;

import java.io.Serializable;
import java.security.KeyFactory;
import java.security.PublicKey;
import java.security.spec.X509EncodedKeySpec;

/**
 * This class encapsulates a Public Key
 * 
 * @author Joseph Isaac
 * 
 */

public class UserPublicKey implements Serializable {

    private static final long serialVersionUID = 2L;

    private transient PublicKey publicKey;

    protected UserPublicKey() {
    }

    /**
     * Return an instance of UserPublicKey
     * 
     * @return
     */
    public static UserPublicKey getInstance() {
        return new UserPublicKey();
    }

    public PublicKey getPublicKey() {
        return publicKey;
    }

    public void setPublicKey(PublicKey publicKey) {
        this.publicKey = publicKey;
    }

    public static UserPublicKey fromComPkbCrypto(com.pkb.crypto.UserPublicKey pk) {
        return unchecked(() -> {
            KeyFactory keyFactory = KeyFactory.getInstance(pk.getAlgorithm());
            PublicKey publicKey = keyFactory.generatePublic(new X509EncodedKeySpec(pk.getKeyBytes()));
            UserPublicKey userPrivateKey = new UserPublicKey();
            userPrivateKey.setPublicKey(publicKey);
            return userPrivateKey;
        }).get();
    }
}
