package org.tolven.security.key;

import static io.vavr.API.unchecked;

import java.io.Serializable;
import java.security.KeyFactory;
import java.security.PrivateKey;
import java.security.spec.PKCS8EncodedKeySpec;

/**
 * This class encapsulates a Private Key
 * 
 * @author Joseph Isaac
 * 
 */

public class UserPrivateKey implements Serializable {

    private static final long serialVersionUID = 2L;

    private transient PrivateKey privateKey;

    protected UserPrivateKey() {
    }

    /**
     * Return an instance of UserPrivateKey
     * 
     * @return
     */
    public static UserPrivateKey getInstance() {
        return new UserPrivateKey();
    }

    public PrivateKey getPrivateKey() {
        return privateKey;
    }

    public void setPrivateKey(PrivateKey privateKey) {
        this.privateKey = privateKey;
    }

    public static UserPrivateKey fromComPkbCrypto(com.pkb.crypto.UserPrivateKey pk) {
        return unchecked(() -> {
            KeyFactory keyFactory = KeyFactory.getInstance(pk.getAlgorithm());
            PrivateKey privateKey = keyFactory.generatePrivate(new PKCS8EncodedKeySpec(pk.getKeyBytes()));
            UserPrivateKey userPrivateKey = new UserPrivateKey();
            userPrivateKey.setPrivateKey(privateKey);
            return userPrivateKey;
        }).get();
    }
}
