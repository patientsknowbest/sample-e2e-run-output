package com.pkb.domain.repository.util;

import com.pkb.domain.criteria.LowerBound;
import com.pkb.domain.criteria.Restriction;
import com.pkb.domain.criteria.TemporalRangeRestriction;
import com.pkb.domain.criteria.UpperBound;
import com.pkb.ehrdata.ImmutableSearchSpec;

import java.time.LocalDate;

import static com.pkb.ehrdata.Filter.filter;
import static java.lang.String.format;

public class BoundedDateRangeRestrictions {
    public static void addUpperBoundDateRestrictions(String fieldName,
                                                     ImmutableSearchSpec.Builder builder,
                                                     Restriction<UpperBound, LocalDate> restriction) {
        switch (restriction.getConstraint()) {
            case LESS:
                builder.addFilter(
                        filter(fieldName, com.pkb.ehrdata.Filter.Operator.LESS,
                                restriction.getValue().atStartOfDay()));
                break;
            case LESS_OR_EQUAL:
                builder.addFilter(filter(fieldName, com.pkb.ehrdata.Filter.Operator.LESS,
                        restriction.getValue().atStartOfDay().plusDays(1L)));
                break;
            default:
                throw new RuntimeException(format("Unsupported constraint [%s]", restriction.getConstraint().name()));
        }
    }

    public static void addLowerBoundDateRestrictions(String fieldName,
                                                     ImmutableSearchSpec.Builder builder,
                                                     Restriction<LowerBound, LocalDate> restriction) {
        switch (restriction.getConstraint()) {
            case GREATER_OR_EQUAL:
                builder.addFilter(filter(fieldName, com.pkb.ehrdata.Filter.Operator.GREATER_OR_EQUAL,
                        restriction.getValue().atStartOfDay()));
                break;
            case GREATER:
                builder.addFilter(
                        filter(fieldName, com.pkb.ehrdata.Filter.Operator.GREATER_OR_EQUAL,
                                restriction.getValue().atStartOfDay().plusDays(1L)));
                break;
            default:
                throw new RuntimeException(format("Unsupported constraint [%s]", restriction.getConstraint().name()));
        }
    }

    public static void addDateRestrictions(String fieldName,
                                           TemporalRangeRestriction<LocalDate> restriction,
                                           ImmutableSearchSpec.Builder builder) {
        restriction.getLowerBound()
                .peek(lowerBoundDateRestriction -> addLowerBoundDateRestrictions(fieldName, builder, lowerBoundDateRestriction));
        restriction.getUpperBound()
                .peek(upperBoundDateRestriction -> addUpperBoundDateRestrictions(fieldName, builder, upperBoundDateRestriction));
    }
}
