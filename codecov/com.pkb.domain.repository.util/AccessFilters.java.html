package com.pkb.domain.repository.util;

import com.pkb.authentication.principal.team.BaseTeam;
import com.pkb.datamodel.Identifiable;
import com.pkb.datamodel.consent.PatientConsent;
import com.pkb.ehrdata.AccessFilter;
import com.pkb.ehrdata.ImmutableAccessFilter;
import com.pkb.ehrdata.ImmutablePrivacyFilter;
import com.pkb.ehrdata.PrivacyFilter;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;

import java.util.Optional;

import static com.pkb.entities.enums.PrivacyFlag.GENERAL;
import static com.pkb.entities.enums.PrivacyFlag.MENTAL_HEALTH;
import static com.pkb.entities.enums.PrivacyFlag.SEXUAL_HEALTH;
import static com.pkb.entities.enums.PrivacyFlag.SOCIAL_CARE;

public class AccessFilters {
    
    public AccessFilter createAccessFilter(long accessingUserId,
                                           BaseTeam team) {
        return createAccessFilter(accessingUserId, Option.of(team), Option.none());
    }

    
    public AccessFilter createAccessFilter(long actorPersonId,
                                           PatientConsent consent) {
        return createAccessFilter(actorPersonId, Option.none(), Option.of(consent));
    }

    public AccessFilter createAccessFilter(long actorPersonId,
                                           BaseTeam team,
                                           PatientConsent consent) {
        return createAccessFilter(actorPersonId, Option.of(team), Option.of(consent));
    }

    
    public AccessFilter createAccessFilter(long actorPersonId,
                                           Option<BaseTeam> team,
                                           Option<PatientConsent> consent) {
        return ImmutableAccessFilter.builder()
                .isTeamDataViewApplicable(team.isDefined())
                .isPiggyback(false)
                .isConsentNotRequired(false)
                .isBreakTheGlassActive(false)
                .isPatientViewSuspendedRequired(false)
                .actorPersonId(actorPersonId)
                .actorTeamId(team.map(Identifiable::getId).toJavaOptional())
                .actorOrgId(team.map(BaseTeam::getOrgId).toJavaOptional())
                .privacyFilter(consent.map(this::createPrivacyFilter).toJavaOptional())
                .build();
    }

    
    public PrivacyFilter createPrivacyFilter(@NotNull PatientConsent consent) {
        return ImmutablePrivacyFilter.builder()
                .isGeneralConsentGranted(consent.isGranted(GENERAL))
                .isMentalHealthConsentGranted(consent.isGranted(MENTAL_HEALTH))
                .isSexualHealthConsentGranted(consent.isGranted(SEXUAL_HEALTH))
                .isSocialCareConsentGranted(consent.isGranted(SOCIAL_CARE))
                .build();
    }

    
    public AccessFilter createSelfAccessFiler(long actorPersonId) {
        return ImmutableAccessFilter.builder()
                .isTeamDataViewApplicable(false)
                .isPiggyback(false)
                .isConsentNotRequired(true)
                .isBreakTheGlassActive(false)
                .isPatientViewSuspendedRequired(false)
                .actorPersonId(actorPersonId)
                .build();
    }

    
    public AccessFilter createReadReceiptAccessFilter(long accessingUserId, long orgId) {
        return ImmutableAccessFilter.builder()
                .isTeamDataViewApplicable(false)
                .isPiggyback(false)
                .isConsentNotRequired(true)
                .isBreakTheGlassActive(false)
                .isPatientViewSuspendedRequired(false)
                .actorPersonId(accessingUserId)
                .actorOrgId(orgId)
                .sourceOrgId(Optional.of(orgId))
                .isReadReceiptApplicable(true)
                .isAccountLinksJoinPermitted(false)// PHR-5032
                .build();
    }

    
    public AccessFilter createSystemUserAccessFilter(long orgId) {
        return ImmutableAccessFilter.builder()
                .isTeamDataViewApplicable(false)
                .isPiggyback(false)
                .isConsentNotRequired(true)
                .isBreakTheGlassActive(false)
                .isPatientViewSuspendedRequired(false)
                .actorOrgId(orgId)
                .sourceOrgId(Optional.of(orgId))
                .isReadReceiptApplicable(true)
                .isAccountLinksJoinPermitted(false)// PHR-5032
                .build();
    }

}
