package com.pkb.domain.repository.util;

import com.pkb.crypto.AccountPrivateKey;
import com.pkb.crypto.UserPrivateKey;
import com.pkb.datamodel.AccountKeys;
import com.pkb.entities.core.CoreAccount;
import com.pkb.entities.core.CoreAccountSymmetricKey;
import com.pkb.entities.core.UserAccountKeyLink;
import com.pkb.entities.pub.PublicPKBPerson;
import com.pkb.entities.pub.PublicViewConsent;
import io.vavr.collection.List;
import io.vavr.control.Either;

import java.util.function.Supplier;

public class TargetOwnerDataHolder {
    private final PublicPKBPerson person;
    private final CoreAccount account;
    private final PublicViewConsent consent;

    private Either<UserPrivateKey, AccountPrivateKey> userOrAccountPrivateKey;
    private List<CoreAccountSymmetricKey> accountSymmetricKeys;
    private Supplier<AccountKeys> keyReloader;
    private UserAccountKeyLink link;

    public TargetOwnerDataHolder(PublicPKBPerson person, CoreAccount account, PublicViewConsent consent) {
        this.person = person;
        this.account = account;
        this.consent = consent;
    }

    public PublicPKBPerson getPerson() {
        return person;
    }

    public CoreAccount getAccount() {
        return account;
    }

    public PublicViewConsent getConsent() {
        return consent;
    }

    public Either<UserPrivateKey, AccountPrivateKey> getUserOrAccountPrivateKey() {
        return userOrAccountPrivateKey;
    }

    public void setUserOrAccountPrivateKey(Either<UserPrivateKey, AccountPrivateKey> userOrAccountPrivateKey) {
        this.userOrAccountPrivateKey = userOrAccountPrivateKey;
    }

    public List<CoreAccountSymmetricKey> getAccountSymmetricKeys() {
        return accountSymmetricKeys;
    }

    public void setAccountSymmetricKeys(List<CoreAccountSymmetricKey> accountSymmetricKeys) {
        this.accountSymmetricKeys = accountSymmetricKeys;
    }

    public Supplier<AccountKeys> getKeyReloader() {
        return keyReloader;
    }

    public void setKeyReloader(Supplier<AccountKeys> keyReloader) {
        this.keyReloader = keyReloader;
    }

    public UserAccountKeyLink getLink() {
        return link;
    }

    public void setLink(UserAccountKeyLink link) {
        this.link = link;
    }
}
