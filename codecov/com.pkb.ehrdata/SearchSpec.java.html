package com.pkb.ehrdata;

import com.google.common.collect.ImmutableList;
import com.google.common.collect.ImmutableSet;
import com.pkb.messaging.workflow.ConversationWorkflowFilterDTO;
import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import java.util.List;
import java.util.Optional;
import java.util.function.Function;

import static java.util.function.Function.identity;
import static java.util.stream.Collectors.collectingAndThen;
import static java.util.stream.Collectors.toList;

@Value.Style(
        get = {"is*", "get*"},
        stagedBuilder = true,
        depluralize = true)
@Value.Immutable
public interface SearchSpec {

    ImmutableSet<Filter> getFilters();

    ImmutableSet<String> getDataTypes();

    ImmutableSet<Long> getAccountIds();

    Optional<LatestFilterFields> getLatestFilter();

    Optional<OrderBy> getOrderBy();

    Optional<OneResultPerDay> getOneResultPerDay();

    Optional<PageFilter> getPageFilter();

    Optional<AccessFilter> getAccessFilter();

    /**
     * All latest filters must match, i.e. all empty or all populated with the same filter
     *
     * @return true if the requirement is fulfilled, false otherwise
     */
    static boolean checkLatestFilters(@NotNull List<SearchSpec> unionSearches) {
        return unionSearches.stream().map(SearchSpec::getLatestFilter).distinct().count() < 2;
    }

    @Value.Default
    default Boolean latestByVisibleType() {
        return Boolean.FALSE;
    }

    @Value.Default
    @Value.Auxiliary
    default Optional<AccessFilter> getAccessFilter(AccessFilter accessFilter) {
        if (getAccessFilter().isPresent()) {
            return getAccessFilter();
        }
        return Optional.ofNullable(accessFilter);
    }

    @Value.Default
    @Value.Auxiliary
    default Optional<AccessFilter> getAccessFilter(Optional<AccessFilter> accessFilter) {
        if (getAccessFilter().isPresent()) {
            return getAccessFilter();
        }
        return accessFilter;
    }

    @Value.Default
    default boolean isUseParticipantFilter() {
        return false;
    }

    @Value.Default
    default boolean isIncludeDeleted() {
        return false;
    }

    @Value.Default
    @Value.Auxiliary
    default Function<String, String> getColumnMapper() {
        return identity();
    }

    @Value.Derived
    @Value.Auxiliary
    default ImmutableSet<Filter> getLatestOnlyFilters() {
        return getFilters().stream()
                .filter(Filter::isApplyWhenChoosingLatest)
                .collect(collectingAndThen(toList(), ImmutableSet::copyOf));
    }

    default ImmutableSet<Filter> getApplicableFilters(boolean choosingLatest) {
        return choosingLatest
                ? getLatestOnlyFilters()
                : getFilters();
    }

    /**
     * Every search has to have the same order-by (all-empty is OK)
     *
     * @return populated Optional with the order-by clause if all searches use the same, EMPTY otherwise
     */
    static Optional<OrderBy> tryAgreeOrderBy(@NotNull ImmutableList<SearchSpec> unionSearches) {
        return unionSearches.stream().map(SearchSpec::getOrderBy).distinct().count() < 2 ? unionSearches.stream().findFirst().flatMap(SearchSpec::getOrderBy) : Optional.empty();
    }

    Optional<ConversationWorkflowFilterDTO> getConversationWorkflowFilter();
}
