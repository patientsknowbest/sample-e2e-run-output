package com.pkb.ehrdata;

import org.apache.commons.collections4.CollectionUtils;
import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import java.util.Collection;
import java.util.Collections;
import java.util.Optional;

import static com.google.common.base.Preconditions.checkState;
import static java.lang.String.format;

/**
 * Hold the info for filtering on a single column in a query.
 */
@Value.Immutable
public abstract class Filter {

    public abstract String getField();

    public abstract Operator getOperator();

    public abstract Optional<Object> getValue();

    @Value.Default
    public boolean isApplyWhenChoosingLatest() {
        return true;
    }

    public enum Operator {
        GREATER(">", true),
        LESS("<", true),
        GREATER_OR_EQUAL(">=", true),
        LESS_OR_EQUAL("<=", true),
        EQUAL("=", true),
        NOT_EQUAL("<>", true),

        IS_NULL(" is null", false),
        IS_NOT_NULL(" is not null", false),

        LIKE(" like ", true),
        IN(" in ", true),
        NOT_IN(" not in ", true),

        // Null handling options!
        // By default any SQL comparison comes back "no match" if the column value is null.
        // This is unexpected for a "not in ()" or "<>" filter!  Use these options to fix it.
        // Null values are also tricky with true/false options where "null" should read as "false".

        // symbol text not directly inserted into query for these ones
        NOT_EQUAL_OR_NULL(" [<> or is null]", true),

        // used for left outer joins
        TRUE_AND_IS_NOT_NULL(" [= true and is not null]", false),
        FALSE_OR_IS_NULL(" [= false or is null]", false),

        NOT_IN_OR_NULL(" [not in or is null]", true);

        private final String symbol;
        private final boolean withValue;

        Operator(String symbol, boolean withValue) {
            this.symbol = symbol;
            this.withValue = withValue;
        }

        public String getSymbol() {
            return symbol;
        }

        public boolean isWithValue() {
            return withValue;
        }
    }

    /**
     * By default all filters are applied in latest by type filters
     */
    public static Filter filter(@NotNull String field, @NotNull Operator operator) {
        return filter(field, operator, null, true/*applyWhenChoosingLatest*/);
    }

    /**
     * By default all filters are applied in latest by type filters
     */
    public static Filter filter(@NotNull String field, @NotNull Operator operator, Object value) {
        return filter(field, operator, value, true/*applyWhenChoosingLatest*/);
    }

    /**
     * applyWhenChoosingLatest is true by default; here you can override to false
     */
    public static Filter filter(@NotNull String field, @NotNull Operator operator, Object value, boolean applyWhenChoosingLatest) {
        return ImmutableFilter.builder()
                .field(field)
                .operator(operator)
                .value(Optional.ofNullable(value))
                .isApplyWhenChoosingLatest(applyWhenChoosingLatest)
                .build();
    }

    public static Filter excludeDeleted() {
        return filter("deleted", Operator.EQUAL, false, false);
    }

    @Value.Check
    protected Filter normalize() {

        if (getOperator().isWithValue()) {
            // Value must exist
            Object value = getValue().orElseThrow(() -> new IllegalArgumentException(
                    format("Field %s: operator %s must not have a null value", getField(), getOperator())));

            // Cleaned value must be referentially equal to original, or we create a new filter!
            Object newValue = cleanValue(value);
            if (newValue != value) {
                return ImmutableFilter.copyOf(this).withValue(newValue);
            }

        } else {
            checkState(!getValue().isPresent(),
                    format("Field %s: operator %s should have a null value; got %s", getField(), getOperator(), getValue().getClass()));
        }

        return this;
    }

    @SuppressWarnings("UseOfObsoleteDateTimeApi")
    private Object cleanValue(@NotNull Object value) {

        // Start off with new value identical to passed-in value
        Object newValue = value;

        if (value.getClass().equals(java.util.Date.class)) {
            // JOOQ has problems with java.util.Date -- convert to java.sql.Date instead
            newValue = java.sql.Timestamp.from(java.util.Date.class.cast(value).toInstant());
        } else if (value.getClass().equals(java.time.Instant.class)) {
            newValue = java.sql.Timestamp.from(java.time.Instant.class.cast(value));
        }

        switch (getOperator()) {
        case LIKE:
            checkState(value instanceof String,
                    format("Field %s: can't use 'like' operator on non-string value; got %s", getField(), value.getClass()));
            break;
        case IN:
        case NOT_IN:
        case NOT_IN_OR_NULL:
            // if it's not a collection, wrap it
            if (!(value instanceof Collection)) {
                newValue = Collections.singletonList(value);
            }

            // and don't allow empty collections
            checkState(CollectionUtils.isNotEmpty((Collection<?>) newValue),
                    format("Field %s: empty Collection for operator %s", getField(), getOperator()));
            break;
        default:
            break;
        }

        return newValue;
    }

}
