package com.pkb.ehrdata;

import org.immutables.value.Value;

@Value.Immutable
public interface PrivacyFilter {
    boolean isGeneralConsentGranted();
    boolean isMentalHealthConsentGranted();
    boolean isSexualHealthConsentGranted();
    boolean isSocialCareConsentGranted();

    default boolean isFullConsentGranted() {
        return isGeneralConsentGranted() && isMentalHealthConsentGranted()
                && isSexualHealthConsentGranted() && isSocialCareConsentGranted();
    }
}
