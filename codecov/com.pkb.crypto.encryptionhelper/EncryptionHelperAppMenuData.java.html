package com.pkb.crypto.encryptionhelper;

import com.pkb.common.ehrdata.serialization.EHRDataSerializer;
import com.pkb.crypto.EncryptedBytesAndNonce;
import com.pkb.crypto.SymmetricKey;
import com.pkb.crypto.dto.AccountKeysDTO;
import com.pkb.crypto.dto.AccountSymmetricKeyDTO;
import com.pkb.entities.app.AppMenuData;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.util.Base64;
import java.util.Map;

import static com.pkb.crypto.CryptoConstants.AES_GCM_NOPADDING_ALGORITHM;

public class EncryptionHelperAppMenuData {

    private final AppMenuDataEncryptionKeysProvider keysProvider;
    private final EHRDataSerializer serializer;

    public EncryptionHelperAppMenuData(AppMenuDataEncryptionKeysProvider keysProvider, EHRDataSerializer serializer) {
        this.keysProvider = keysProvider;
        this.serializer = serializer;
    }

    @NotNull
    public AppMenuData encryptAppMenuData(@NotNull AppMenuData entity, @NotNull Map<String, Object> encryptedFields, @Nullable AccountKeysDTO accountKeysDTO) {
        return encryptAppMenuData(entity, serializer.serialize(encryptedFields), accountKeysDTO);
    }

    @SuppressWarnings("WeakerAccess")
    @NotNull
    public AppMenuData encryptAppMenuData(@NotNull AppMenuData entity, byte[] serializedData, @Nullable AccountKeysDTO accountKeysDTO) {
        AccountSymmetricKeyDTO symmetricKeyDTO = keysProvider.getSymmetricKey(entity, accountKeysDTO);
        EncryptedBytesAndNonce encryptedBytesAndNonce = SymmetricKey.symmetricKey(symmetricKeyDTO).encrypt(serializedData);
        entity.setXml01(encryptedBytesAndNonce.encryptedBytes());
        entity.setSecretKey(null); // we'll store the secret key in a separate table
        entity.setSecretKeyAlgorithm(AES_GCM_NOPADDING_ALGORITHM); // don't need the length to decrypt
        entity.setNonceBase64(Base64.getEncoder().encodeToString(encryptedBytesAndNonce.nonce()));
        entity.setSymmetricEncryptionKeyId(symmetricKeyDTO.getKeyId());

        return entity;
    }

}
