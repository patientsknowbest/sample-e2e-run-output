package com.pkb.service.dataupload.emis.esv2.batch;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.EHRRequestContext.AccountLinkType;
import com.pkb.app.entity.ImmutableLoggedOutEHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.entities.enums.Route;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.service.dataupload.emis.esv2.model.AdminUserInRole;
import org.jetbrains.annotations.Nullable;

import java.util.Optional;
import java.util.UUID;

import static com.pkb.consent.model.AbstractConsentStatus.noConsentRequired;

public interface EmisRequestContextProvider {
    EHRRequestContext context = ImmutableLoggedOutEHRRequestContext.of(AccountLinkType.ORG, Route.EMIS_ES, noConsentRequired(), UUID.randomUUID());

    default EHRRequestContext getEmisContext(Org org) {
        if (org != null) {
            return context.withOrg(org);
        }
        return context;
    }

    default EHRRequestContext getEmisContextWithDefaultTeam(Org org) {
        return getEmisContext(org, findDefaultPatientCreateTeamForOrg(org).orElse(null));
    }

    default EHRRequestContext getEmisContext(Org org, Team team) {
        var builder = ImmutableLoggedOutEHRRequestContext.builder()
                .from(context);
        if (org != null) {
            builder.org(org);
        }
        if (team != null) {
            builder.teamId(team.getId());
        }
        return builder.build();
    }

    default Team getDefaultPatientCreateTeamForOrg(Org org) {
        return findDefaultPatientCreateTeamForOrg(org).orElseThrow(() -> new IllegalStateException("No Default Patient Create Team set for Org: " + org.getId()));
    }

    default Optional<Team> findDefaultPatientCreateTeamForOrg(Org org) {
        if (org == null) {
            return Optional.empty();
        }
        return org.getTeams().stream().filter(t -> org.getHl7NewPatientTeamCode() != null && t.getCode() != null && t.getCode().equals(org.getHl7NewPatientTeamCode())).findFirst();
    }

    default SourceDetails initializeSourceDetails(EHRRequestContext recordContext, @Nullable AdminUserInRole enteredBy) {
        SourceDetails sourceDetails = new SourceDetails(recordContext);
        if (enteredBy != null) {
            enteredBy.displayName().ifPresent(sourceDetails::setText);
        }
        return sourceDetails;
    }
}
