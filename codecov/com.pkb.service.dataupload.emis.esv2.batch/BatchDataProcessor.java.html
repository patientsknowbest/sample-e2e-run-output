public abstract class BatchDataProcessor<T extends BaseEmisEsDomainModel> implements EmisRequestContextProvider, FileProcessor<T>, EmisEsDateTimeProcessor {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private static final String PROCESSED_ROWS_COUNTER_NAME = "pkb_phr_integration_emis_csv_processed_rows";
    private static final Counter PROCESSED_ROWS_COUNTER = Counter.builder(PROCESSED_ROWS_COUNTER_NAME)
            .tags(TAG_PROCESSOR_TYPE, "", TAG_EMIS_FILE_TYPE, "", TAG_STATUS, "") // see enum ProcessorType, EmisFileType, ProcessedRowStatus
            .description("Number of rows processed")
            .register(Metrics.globalRegistry);

    private static final String PROCESSED_CODED_DATA_COUNTER_NAME = "pkb_phr_integration_emis_csv_processed_coded_data";
    private static final String TAG_CODED_DATA_TYPE = "codingDataType";
    private static final Counter PROCESSED_CODED_DATA_COUNTER = Counter.builder(PROCESSED_CODED_DATA_COUNTER_NAME)
            .tags(TAG_PROCESSOR_TYPE, "", TAG_CODED_DATA_TYPE, "", TAG_STATUS, "") // see enum ProcessorType, CodingDataType, ProcessedRowStatus
            .description("Number of coded data processed")
            .register(Metrics.globalRegistry);


    protected final EmisEsProcessedRowService emisEsProcessedRowService;

    protected final EmisEsManager emisEsManager;

    protected final EmisProcessorAborter emisProcessAborter;

    protected final Kms kmsClient;

    protected final PhrConfig config;

    protected final EmisEsStatusService emisEsStatusService;

    protected final EmisOrgHelper emisOrgHelper;


    public BatchDataProcessor(EmisEsProcessedRowService emisEsProcessedRowService, EmisEsManager emisEsManager, EmisProcessorAborter emisProcessAborter, Kms kmsClient, PhrConfig config, EmisEsStatusService emisEsStatusService, EmisOrgHelper emisOrgHelper) {
        this.emisEsProcessedRowService = emisEsProcessedRowService;
        this.emisEsManager = emisEsManager;
        this.emisProcessAborter = emisProcessAborter;
        this.kmsClient = kmsClient;
        this.config = config;
        this.emisEsStatusService = emisEsStatusService;
        this.emisOrgHelper = emisOrgHelper;
    }

    /**
     * Process a stream of EMIS entities.
     * Implementations of this methods must assume that they
     * have no decryption access to accounts.
     *
     * @param rows The typed stream. Note: this can be a huge lazy stream.
     *             Implementations must assume that the whole stream cannot
     *             be pulled into memory in one step. Therefore never
     *             move it into a non-lazy collection or try to naively
     *             group or partition it.s
     * @return Whether processing has been successful for all rows.
     */
    public abstract Tuple2<Boolean, EmisEsProcessedRow> process(Stream<? extends ResultWithId<T, Long>> rows, EmisEsProcessedRow fileState);

    public abstract boolean isPostProcessingRequired();

    protected abstract boolean doPostProcess(EmisEsProcessedRow rowState);

    /**
     * This method as the name suggests is to be called after method process() has been called on all processors
     * <p>
     * Useful in cases like for E.g:- processing CareRecordProblems which are orphan.
     * <p>
     * EMIS sometimes sends in out-of-sync CareRecordProblems which do not have corresponding CareRecordObservations in the same set
     * Such records can be identified and stored in DB till we later gain the decryption access and apply the changes to original records
     */
    public final boolean postProcess(EmisEsProcessedRow rowState) {
        if (isPostProcessingRequired()) {
            return doPostProcess(rowState);
        }
        return true;
    }

    protected EmisEsProcessedRow updateLastProcessedRow(EmisEsProcessedRow actualFileState, RowId<Long> rowId) {
        actualFileState.setLastProcessedRow(rowId.getId());
        actualFileState.setFullyProcessed(false);
        return emisEsProcessedRowService.save(actualFileState);
    }

    protected Map<Long, AccountKeysDTO> attemptKeyLookup(Seq<Long> accountPrivateIds) {
        Map<UUID, Long> accountIds = emisEsManager.getAccountPublicIds(accountPrivateIds.distinct().toJavaList());

        Function1<List<AccountKeysDTO>, Map<Long, AccountKeysDTO>> toPkbId = Function2.of(this::mapByPrivateAccountId).apply(accountIds);

        if (accountIds.nonEmpty()) {
            return recordTime("getAccountKeys", () -> kmsClient.getAccountKeysTrustedIgnoreMissing(null, accountIds.keysIterator().toJavaList())
                    .map(toPkbId)
                    .getOrElse(HashMap.empty()));
        }

        return HashMap.empty();
    }

    private Map<Long, AccountKeysDTO> mapByPrivateAccountId(Map<UUID, Long> accountIds, List<AccountKeysDTO> keys) {
        return HashMap.ofAll(keys.stream()
                .map(key -> {
                    UUID accountPublicId = key.getAccountPublicId();
                    Long accountId = accountIds.get(accountPublicId).getOrElse((Long) null);
                    if (isNull(accountId)) {
                        LOGGER.warn("Could not find {} in {}", accountPublicId, accountIds);
                    }
                    return Tuple.of(accountId, key);
                })
                .filter(t -> nonNull(t._1))
                .collect(toMap(Tuple2::_1, Tuple2::_2)));
    }

    protected ZoneId getTimezoneOfOrg(Org o, Runnable missingZoneIdHandler) {
        return EmisEsDateUtil.getTimezoneOfOrg(o, missingZoneIdHandler);
    }

    protected ZoneId getTimezoneOfOrg(Org o) {
        return EmisEsDateUtil.getTimezoneOfOrg(o);
    }

    protected void incrementProcessedRowsCounter(ProcessedRowStatus status, double increment) {
        Metrics.counter(PROCESSED_ROWS_COUNTER_NAME, TAG_PROCESSOR_TYPE, getProcessorType().toString(), TAG_EMIS_FILE_TYPE, getSupportedType().toString(), TAG_STATUS, status.toString()).increment(increment);
    }

    protected void incrementProcessedCodedDataCounter(ProcessedRowStatus status, String dataTypeDesc, double increment) {
        Metrics.counter(PROCESSED_CODED_DATA_COUNTER_NAME, TAG_PROCESSOR_TYPE, getProcessorType().toString(), TAG_CODED_DATA_TYPE, dataTypeDesc, TAG_STATUS, status.toString()).increment(increment);
    }

    @Override
    public EmisEsManager getEmisEsManager() {
        return emisEsManager;
    }
}
