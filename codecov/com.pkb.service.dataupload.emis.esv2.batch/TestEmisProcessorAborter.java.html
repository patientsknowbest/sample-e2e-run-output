package com.pkb.service.dataupload.emis.esv2.batch;

import com.pkb.service.dataupload.emis.esv2.model.EmisFileType;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.EnumMap;
import java.util.Map;

import static java.lang.invoke.MethodHandles.lookup;

public class TestEmisProcessorAborter implements EmisProcessorAborter {

    private static final Logger LOGGER = LoggerFactory.getLogger(lookup().lookupClass());
    private final Map<EmisFileType, Long> abortPoints = new EnumMap<>(EmisFileType.class);
    private final Map<EmisFileType, Long> pausePoints = new EnumMap<>(EmisFileType.class);
    private volatile boolean paused = false;
    private volatile EmisFileType pausedAt;
    private volatile Long pausedAtRow;

    @Override
    public void setAbortPoint(EmisFileType type, Long rowId) {
        abortPoints.put(type, rowId);
    }

    @Override
    public void clearAbortPoint(EmisFileType type) {
        abortPoints.remove(type);
    }

    @Override
    public EmisProcessPauseInformation isPaused() {
        return ImmutableEmisProcessPauseInformation.builder().paused(paused).pausedAtFile(pausedAt).pausedAtRow(pausedAtRow).build();
    }

    @Override
    public void clear() {
        abortPoints.clear();
        pausePoints.clear();
        paused = false;
        pausedAt = null;
        pausedAtRow = null;
    }

    @Override
    public void executeTestControl(EmisFileType type, Long rowId) {
        waitUntilPaused(type, rowId);
        throwExceptionWhenAbortPointReached(type, rowId);
    }

    private void throwExceptionWhenAbortPointReached(EmisFileType type, Long rowId) {
        if (rowId.equals(abortPoints.get(type))) {
            throw new RuntimeException("Emis abort point reached!");
        }
    }

    @Override
    public void setPausePoint(EmisFileType type, Long rowId) {
        pausePoints.put(type, rowId);
    }

    @Override
    public void clearPausePoint(EmisFileType type) {
        pausePoints.remove(type);
    }

    private void waitUntilPaused(EmisFileType type, Long rowId) {
        paused = false;
        pausedAt = null;
        pausedAtRow = null;
        boolean wasPaused = false;
        while (shouldBePaused(type, rowId, wasPaused)) {
            try {
                pausedAt = type;
                pausedAtRow = rowId;
                paused = true;
                wasPaused = true;
                Thread.sleep(1000);
            } catch (InterruptedException e) {
                throw new RuntimeException(e);
            }
        }
        paused = false;
        pausedAt = null;
        pausedAtRow = null;
        if (wasPaused) {
            LOGGER.warn("Resumed EMIS ES processing as {} is no longer set", type);
        }
    }

    private boolean shouldBePaused(EmisFileType type, Long rowId, boolean wasPaused) {
        boolean hasPausePoint = pausePoints.containsKey(type);
        boolean reachedRow = rowId.equals(pausePoints.get(type));
        boolean result = hasPausePoint && reachedRow;
        if (result && !wasPaused) {
            LOGGER.warn("Pausing EMIS ES processing at {} row: {} as it is set!", type, rowId);
        }
        return result;
    }
}
