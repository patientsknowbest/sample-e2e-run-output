package com.pkb.service.dataupload.emis.esv2.batch;

import com.pkb.institute.OrgMergeHelper;
import com.pkb.service.dataupload.emis.esv2.EmisEsManager;
import com.pkb.service.dataupload.emis.esv2.model.BaseEmisEsDomainModel;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileType;
import com.pkb.service.dataupload.emis.esv2.sorting.ResultWithId;
import io.micrometer.core.instrument.Metrics;
import io.micrometer.core.instrument.Timer;
import io.vavr.Function1;
import io.vavr.collection.Map;
import org.jetbrains.annotations.NotNull;

import java.util.Optional;
import java.util.concurrent.Callable;

public interface FileProcessor<T extends BaseEmisEsDomainModel> {
    /**
     * The model class that this processor handles.
     * (CSV rows map to objects of this class.)
     */
    Class<T> interfaceClass();

    /**
     * Whether this processor can process the provided file type
     */
    boolean canProcess(@NotNull EmisFileType fileType);

    EmisFileType getSupportedType();

    enum ProcessorType {
        NORMAL, STATELESS, VALIDATOR
    }

    ProcessorType getProcessorType();

    enum ProcessedRowStatus {
        SKIPPED, FAILED, SAVED, DELETED, EXCLUDED_BY_CODE
    }

    String TAG_PROCESSOR_TYPE = "processorType";
    String TAG_EMIS_FILE_TYPE = "emisFileType";
    String TAG_STATUS = "status";
    String PROCESS_TIMING_NAME = "pkb_phr_integration_emis_csv_batchprocessor_timings";
    String TAG_PHASE = "phase";
    Timer PROCESS_TIMER = Timer.builder(PROCESS_TIMING_NAME)
            .tags(TAG_PROCESSOR_TYPE, "", TAG_EMIS_FILE_TYPE, "", TAG_PHASE, "")
            .description("Time needed to execute various phases of processing")
            .publishPercentiles(0.5, 0.95)
            .publishPercentileHistogram()
            .percentilePrecision(2)
            .register(Metrics.globalRegistry);

    default <T> T recordTime(String phase, Callable<T> toMeasure) {
        try {
            return Metrics.timer(PROCESS_TIMING_NAME, TAG_PROCESSOR_TYPE, getProcessorType().toString(), TAG_EMIS_FILE_TYPE, getSupportedType().toString(), TAG_PHASE, phase).recordCallable(toMeasure);
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }

    default void recordTime(String phase, Runnable toMeasure) {
        Metrics.timer(PROCESS_TIMING_NAME, TAG_PROCESSOR_TYPE, getProcessorType().toString(), TAG_EMIS_FILE_TYPE, getSupportedType().toString(), TAG_PHASE, phase).record(toMeasure);
    }

    default <U> Map<U, OrgMergeHelper> getRecordToOrgMergeHelperMapTimed(io.vavr.collection.List<ResultWithId<U, Long>> records, Function1<U, String> orgGuidExtractor) {
        return recordTime("recordToOrg", () -> getEmisEsManager().getRecordToOrgMap(records, orgGuidExtractor));
    }

    default <U> Map<U, Optional<OrgMergeHelper>> getRecordToMaybeOrgMergeHelperMapTimed(io.vavr.collection.List<ResultWithId<U, Long>> records, Function1<U, String> orgGuidExtractor) {
        return recordTime("recordToOrg", () -> getEmisEsManager().getRecordToMaybeOrgMap(records, orgGuidExtractor));
    }

    EmisEsManager getEmisEsManager();
}
