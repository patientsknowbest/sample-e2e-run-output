package com.pkb.kms.service.routing

import com.pkb.common.config.KmsConfig
import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import com.pkb.kms.service.routing.KeyAvailableRouteBuilder.Companion.INBOUND_ORG_KEY_AVAILABLE
import com.pkb.kms.service.routing.KeyAvailableRouteBuilder.Companion.INBOUND_USER_KEY_AVAILABLE
import com.pkb.kms.service.routing.KeyAvailableRouteBuilder.Companion.OUTBOUND_ACCOUNT_KEY_AVAILABLE
import com.pkb.kms.service.routing.KeyAvailableRouteBuilder.Companion.OUTBOUND_ORG_KEY_AVAILABLE
import com.pkb.kms.service.routing.KeyAvailableRouteBuilder.Companion.OUTBOUND_USER_KEY_AVAILABLE
import com.pkb.kms.shared.structures.pubsub.KeyAvailable
import com.pkb.kms.shared.structures.pubsub.KeyType
import org.apache.camel.builder.RouteBuilder
import org.apache.commons.lang3.StringUtils.isBlank
import org.springframework.beans.factory.annotation.Value
import org.springframework.context.annotation.Profile
import org.springframework.stereotype.Component

/**
 * Configure Apache Camel to publish / subscribe to user keys becoming available.
 * Use some internal endpoints (constants *_USER_KEY_AVAILABLE) in the places where
 * I actually want to send / handle messages. This class hooks up the internal
 * endpoints to google pub sub. Could swap for something else in future, or even
 * just wire them internally for integration testing, avoiding the need for pubsub
 * docker container.
 *
 * Use AbstractNamespaceAwareRouteBuilder for automatically handling namespace
 * changes between e2e tests.
 */
@Component
@Profile("!integration-test")
class KeyAvailableRouteBuilder(private val config: KmsConfig,
                               private val pubSubNamespaceService: PubSubNamespaceService,
                               @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}")
                               private val serviceAccountKey: String?) : AbstractNamespaceAwareRouteBuilder() {
    companion object {
        const val OUTBOUND_USER_KEY_AVAILABLE = "direct:userKeyAvailable"
        const val OUTBOUND_ACCOUNT_KEY_AVAILABLE = "direct:accountKeyAvailable"
        const val OUTBOUND_ORG_KEY_AVAILABLE = "direct:orgKeyAvailable"
        const val OUTBOUND_USER_KEY_AVAILABLE_STARTUP = "direct:userKeyAvailableStartup"

        const val INBOUND_ORG_KEY_AVAILABLE = "direct:handleOrgKeyAvailable"
        const val INBOUND_USER_KEY_AVAILABLE = "direct:handleUserKeyAvailable"

    }

    /**
     * Important!
     * Configure outbound routes _first_ , then inbound routes.
     * This is required for graceful shutdown and restart. Shutdown will stop the routes in _reverse_ order,
     * and we want to stop the _inbound_ routes first before the _outbound_ ones.
     * This is because inbound message process might result in a new outbound message being sent, which will fail
     * (or deadlock, or block) if the outbound route has already been shut down.
     */
    override fun configure() {
        from(OUTBOUND_USER_KEY_AVAILABLE)
                .marshal().avro(KeyAvailable.getClassSchema())
                .convertBodyTo(ByteArray::class.java)
                .to(uri(config.pubSubProjectName, config.keysAvailableTopicName))

        from(OUTBOUND_USER_KEY_AVAILABLE_STARTUP)
                .circuitBreaker()
                .faultToleranceConfiguration()
                .timeoutEnabled(config.keysAvailableHandlerTimeoutEnabled)
                .timeoutDuration(config.keysAvailableOnStartupHandlerTimeoutMillis)
                .delay(config.keysAvailableHandlerDelayMillis) // We don't really want a true circuit breaker, just something that will timeout tasks. These values effectively disable the circuit breaker in all but the most extreme failure scenarios.
                .requestVolumeThreshold(config.keysAvailableHandlerRequestVolumeThreshold)
                .failureRatio(config.keysAvailableHandlerFailureRatio) //Percentage
                .successThreshold(config.keysAvailableHandlerSuccessThreshold)
                .end()
                .to(INBOUND_USER_KEY_AVAILABLE)

        from(OUTBOUND_ACCOUNT_KEY_AVAILABLE)
                .marshal().avro(KeyAvailable.getClassSchema())
                .convertBodyTo(ByteArray::class.java)
                .to(uri(config.pubSubProjectName, config.keysAvailableTopicName))
        from(OUTBOUND_ORG_KEY_AVAILABLE)
                .marshal().avro(KeyAvailable.getClassSchema())
                .convertBodyTo(ByteArray::class.java)
                .to("google-pubsub:${config.pubSubProjectName}:${config.keysAvailableTopicName}")

        from(uri(config.pubSubProjectName, config.keysAvailableSubscriptionName))
                .unmarshal().avro(KeyAvailable.getClassSchema())
                .circuitBreaker()
                .faultToleranceConfiguration()
                .delay(config.keysAvailableHandlerDelayMillis)
                .successThreshold(config.keysAvailableHandlerSuccessThreshold)
                .failureRatio(config.keysAvailableHandlerFailureRatio)
                .requestVolumeThreshold(config.keysAvailableHandlerRequestVolumeThreshold)
                .timeoutEnabled(config.keysAvailableHandlerTimeoutEnabled)
                .timeoutDuration(config.keysAvailableHandlerTimeoutMillis)
                .end()
                .choice()
                .`when` { (it.message.body as? KeyAvailable)?.keyType == KeyType.USER_KEY }.to(INBOUND_USER_KEY_AVAILABLE)
                .`when` { (it.message.body as? KeyAvailable)?.keyType == KeyType.ORG_ACCOUNT_KEY }.to(INBOUND_ORG_KEY_AVAILABLE)
                .otherwise().stop()
    }

    override fun camelNamespaceService(): PubSubNamespaceService = pubSubNamespaceService

    override fun throwExceptionForNamespaceMismatch(): Boolean = true

    private fun uri(pubSubProjectName: String, topicName: String) = "google-pubsub:$pubSubProjectName:$topicName${if (isBlank(serviceAccountKey)) "" else "?serviceAccountKey=file://$serviceAccountKey"}"
}

@Profile("integration-test")
@Component
class IntTestKeyAvailableRouteBuilder: RouteBuilder() {
    override fun configure() {
        from(OUTBOUND_USER_KEY_AVAILABLE).to(INBOUND_USER_KEY_AVAILABLE)
        from(OUTBOUND_ACCOUNT_KEY_AVAILABLE).to("direct:trash")
        from(OUTBOUND_ORG_KEY_AVAILABLE).to(INBOUND_ORG_KEY_AVAILABLE)
        from("direct:trash").stop()
    }
}
