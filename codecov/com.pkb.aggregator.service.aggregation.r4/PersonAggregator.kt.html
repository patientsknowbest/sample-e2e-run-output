package com.pkb.aggregator.service.aggregation.r4

import com.pkb.aggregator.R4Extension
import com.pkb.aggregator.R4Person
import com.pkb.aggregator.extension.findSingleR4UpstreamTagExtension
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.extension.safeAddAll
import com.pkb.aggregator.extension.safeAddExtensions
import com.pkb.aggregator.extension.simpleUpstreamResourceId
import com.pkb.aggregator.extension.untagSource
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_PATIENT
import com.pkb.aggregator.service.PersonIdentifierService
import com.pkb.aggregator.service.aggregation.ResourceAggregator
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import com.pkb.aggregator.service.reference.ReferenceExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class PersonAggregator(
    private val personIdentifierService: PersonIdentifierService,
    private val referenceExtractor: ReferenceExtractor,
) : ResourceAggregator<R4Person> {

    private val log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun createDownstreamInstance(downstreamId: String) = R4Person().apply {
        active = true
        this.id = downstreamId
    }

    override fun mergeResource(
        downstreamResource: R4Person,
        upstreamResource: IDomainResource,
        references: ReferenceExtractor.IdentifierMap
    ) {
        val upstreamPerson = upstreamResource as? R4Person
            ?: throw UnexpectedTypeException(upstreamResource)

        val source = upstreamPerson.findSingleR4UpstreamTagExtension()

        // TODO PHR-9286
        downstreamResource.photo = upstreamPerson.photo

        downstreamResource.identifier.safeAddAll(upstreamPerson.identifier, source)
        downstreamResource.name.safeAddAll(upstreamPerson.name, source)
        downstreamResource.telecom.safeAddAll(upstreamPerson.telecom, source)
        downstreamResource.address.safeAddAll(upstreamPerson.address, source)

        val newLinks = createLinks(upstreamPerson, references)
        downstreamResource.link.safeAddAll(newLinks, source)

        // merge extensions
        downstreamResource.safeAddExtensions(upstreamPerson.extension)
    }

    override fun stripResource(downstreamResource: R4Person, sourceTag: R4Extension) {
        downstreamResource.link.untagSource(sourceTag)
        downstreamResource.extension.removeIf { it.equalsDeep(sourceTag) }
    }

    override fun mergeIdentifiers(downstreamResource: R4Person) {
        // Do nothing for now - ehrmigrated Person resources have no identifiers
    }

    override fun deriveDownstreamResourceId(currentDownstreamId: String?, resources: List<IDomainResource>) =
        currentDownstreamId ?: personIdentifierService.deriveDownstreamResourceId(resources)

    fun updateOrCreate(
        person: R4Person,
        currentDownstream: R4Person?,
        downstreamResourceId: String?
    ): R4Person {
        val resources = listOf(person)
        val references = referenceExtractor.extract(resources)

        // If we've never seen this Person before, we must use the resource id part from the target Patient
        val safeDownstreamResourceId = if (currentDownstream == null && downstreamResourceId == null) {
            val idPart = person.link.first().let {
                val source = it.getSourcedReference()
                val needle = UpstreamResource(source.upstreamSource, it.target.reference)
                references.getPatientReference(needle).referenceElement.idPart
            }
            val personId = "Person/$idPart"
            log.info("Using resource id $personId for new Person ${person.simpleUpstreamResourceId()} based on downstream resource id of target link")
            personId
        } else {
            downstreamResourceId
        }

        return merge(resources, currentDownstream, safeDownstreamResourceId, references)
    }

    private fun createLinks(person: R4Person, references: ReferenceExtractor.IdentifierMap) =
        if (person.hasLink()) {
            person.link
                .map { upstreamLink ->
                    val downstreamLink = upstreamLink.copy()
                    if (downstreamLink.hasTarget()) {
                        if (downstreamLink.target.referenceElement.resourceType == FHIR_TYPE_PATIENT) {
                            val source = upstreamLink.getSourcedReference()
                            val needle = UpstreamResource(source.upstreamSource, downstreamLink.target.reference)
                            downstreamLink.target = references.getPatientReference(needle)
                        }
                    }
                    downstreamLink
                }
                .toMutableList()
        } else {
            mutableListOf()
        }
}
