package com.pkb.aggregator.service.aggregation.r4

import com.pkb.aggregator.R4Extension
import com.pkb.aggregator.R4Medication
import com.pkb.aggregator.extension.findSingleR4UpstreamTagExtension
import com.pkb.aggregator.extension.safeAddAll
import com.pkb.aggregator.extension.safeAddExtensions
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.extension.untagSource
import com.pkb.aggregator.service.MedicationIdentifierService
import com.pkb.aggregator.service.aggregation.ResourceAggregator
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import com.pkb.aggregator.service.reference.ReferenceExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

/**
 * Aggregates multiple resources referring to the same medication into a single resource for the downstream store
 */
@Service
class MedicationAggregator(
    private val medicationIdentifierService: MedicationIdentifierService,
    private val referenceExtractor: ReferenceExtractor,
) : ResourceAggregator<R4Medication> {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun createDownstreamInstance(downstreamId: String) =
        R4Medication().apply {
            this.id = downstreamId
        }

    override fun mergeResource(
        downstreamResource: R4Medication,
        upstreamResource: IDomainResource,
        references: ReferenceExtractor.IdentifierMap
    ) {

        val upstreamMedication = upstreamResource as? R4Medication
            ?: throw UnexpectedTypeException(upstreamResource)

        val source = upstreamMedication.findSingleR4UpstreamTagExtension()

        // TODO PHR-9286
        downstreamResource.status = upstreamMedication.status
        downstreamResource.amount = upstreamMedication.amount.copy()
        downstreamResource.batch = upstreamMedication.batch.copy()
        downstreamResource.form = upstreamMedication.form.copy()

        // If the downstream does not yet have a code (e.g. first time aggregation), copy it from the upstream,
        // and we'll aggregate the actual codings separately (below)
        if (!downstreamResource.hasCode()) {
            downstreamResource.code = upstreamMedication.code.copy()
        }

        downstreamResource.identifier.safeAddAll(upstreamMedication.identifier, source)
        downstreamResource.ingredient.safeAddAll(upstreamMedication.ingredient, source)
        downstreamResource.code.coding.safeAddAll(upstreamMedication.code.coding, source)

        downstreamResource.safeAddExtensions(upstreamMedication.extension)
    }

    override fun stripResource(downstreamResource: R4Medication, sourceTag: R4Extension) {
        downstreamResource.identifier.untagSource(sourceTag)
        downstreamResource.code.coding.untagSource(sourceTag)

        downstreamResource.extension.removeIf { it.equalsDeep(sourceTag) }
    }

    override fun mergeIdentifiers(downstreamResource: R4Medication) {
        val newIdentifiers = medicationIdentifierService.mergeIdentifiers(downstreamResource)
        if (newIdentifiers > 0) {
            log.info(
                "Created {} new identifiers for downstream medication {}",
                newIdentifiers, downstreamResource.simpleResourceId()
            )
        }
    }

    override fun deriveDownstreamResourceId(currentDownstreamId: String?, resources: List<IDomainResource>): String {
        return currentDownstreamId ?: medicationIdentifierService.deriveDownstreamResourceId(resources)
    }

    fun updateOrCreate(
        medication: R4Medication,
        currentDownstream: R4Medication?,
        downstreamResourceId: String?
    ): R4Medication {
        val resources = listOf(medication)
        val references = referenceExtractor.extract(resources)
        return merge(resources, currentDownstream, downstreamResourceId, references)
    }
}
