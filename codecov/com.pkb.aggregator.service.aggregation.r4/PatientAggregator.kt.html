package com.pkb.aggregator.service.aggregation.r4

import com.pkb.aggregator.R4Extension
import com.pkb.aggregator.R4Patient
import com.pkb.aggregator.R4Reference
import com.pkb.aggregator.extension.findAllSourcedReference
import com.pkb.aggregator.extension.findSingleR4UpstreamTagExtension
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.extension.safeAddAll
import com.pkb.aggregator.extension.safeAddExtensions
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.extension.untagSource
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.PatientIdentifierService
import com.pkb.aggregator.service.aggregation.ResourceAggregator
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import com.pkb.aggregator.service.reference.PhrReferenceHelper
import com.pkb.aggregator.service.reference.ReferenceExtractor
import com.pkb.aggregator.service.reference.ReferenceExtractor.IdentifierMap
import com.pkb.aggregator.service.reference.UpstreamDownstreamIdPair
import org.hl7.fhir.instance.model.api.IDomainResource
import org.hl7.fhir.r4.model.Identifier
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

/**
 * Aggregates multiple resources referring to the same patient into a single resource for the downstream store
 */
@Service
class PatientAggregator(
    private val patientIdentifierService: PatientIdentifierService,
    private val referenceExtractor: ReferenceExtractor,
    private val phrReferenceHelper: PhrReferenceHelper,
) : ResourceAggregator<R4Patient> {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun createDownstreamInstance(downstreamId: String) = R4Patient().apply {
        this.active = true
        this.id = downstreamId
    }

    override fun mergeResource(downstreamResource: R4Patient, upstreamResource: IDomainResource, references: IdentifierMap) {

        val upstreamPatient = upstreamResource as? R4Patient
            ?: throw UnexpectedTypeException(upstreamResource)

        val sourceTag = upstreamPatient.findSingleR4UpstreamTagExtension()

        val newIdentifiers = createIdentifiersWithTranslatedReferences(upstreamPatient, references)
        downstreamResource.identifier.safeAddAll(newIdentifiers, sourceTag)

        // merge demographics
        downstreamResource.name.safeAddAll(upstreamPatient.name, sourceTag)
        downstreamResource.telecom.safeAddAll(upstreamPatient.telecom, sourceTag)
        downstreamResource.address.safeAddAll(upstreamPatient.address, sourceTag)
        downstreamResource.photo.safeAddAll(upstreamPatient.photo, sourceTag)
        downstreamResource.contact.safeAddAll(upstreamPatient.contact, sourceTag)
        downstreamResource.communication.safeAddAll(upstreamPatient.communication, sourceTag)
        val newGeneralPractitioners = createGeneralPractitionerReference(upstreamPatient, references)
        downstreamResource.generalPractitioner.safeAddAll(newGeneralPractitioners, sourceTag)

        // TODO PHR-9286
        if (upstreamPatient.hasManagingOrganization()) {
            downstreamResource.managingOrganization = createManagingOrganizationReference(upstreamPatient, references)
        }
        if (upstreamPatient.hasGender()) {
            downstreamResource.gender = upstreamPatient.gender
        }

        downstreamResource.maritalStatus = upstreamPatient.maritalStatus.copy()

        // merge extensions
        val newExtensions = createTranslatedPhrExtensionReferences(
            upstreamPatient.getSourcedReference(),
            upstreamPatient.extension,
            references,
            phrReferenceHelper
        )
        downstreamResource.safeAddExtensions(newExtensions)
    }

    override fun stripResource(downstreamResource: R4Patient, sourceTag: R4Extension) {
        downstreamResource.identifier.untagSource(sourceTag)
        downstreamResource.name.untagSource(sourceTag)
        downstreamResource.telecom.untagSource(sourceTag)
        downstreamResource.address.untagSource(sourceTag)
        downstreamResource.contact.untagSource(sourceTag)
        downstreamResource.communication.untagSource(sourceTag)
        downstreamResource.generalPractitioner.untagSource(sourceTag)

        downstreamResource.extension.removeIf { it.equalsDeep(sourceTag) }
    }

    override fun mergeIdentifiers(downstreamResource: R4Patient) {
        val newIdentifiers = patientIdentifierService.mergeIdentifiers(downstreamResource)
        if (newIdentifiers > 0) {
            log.info(
                "Created {} new identifiers for downstream patient {}",
                newIdentifiers, downstreamResource.simpleResourceId()
            )
        }
    }

    override fun deriveDownstreamResourceId(currentDownstreamId: String?, resources: List<IDomainResource>) =
        currentDownstreamId ?: patientIdentifierService.deriveDownstreamResourceId(resources)

    override fun updateReferences(downstreamResource: R4Patient, references: IdentifierMap) {
        val downstreamResourceId = downstreamResource.simpleResourceId()
        val idPairs = downstreamResource.findAllSourcedReference()
            .map {
                UpstreamDownstreamIdPair(
                    upstream = it,
                    downstreamId = downstreamResourceId
                )
            }
        references.patients.addAll(idPairs)
    }

    fun updateOrCreate(
        patient: R4Patient,
        currentDownstream: R4Patient?,
        downstreamResourceId: String?
    ): R4Patient {
        val resources = listOf(patient)
        val references = referenceExtractor.extract(resources)
        val result = merge(resources, currentDownstream, downstreamResourceId, references)
        log.debug(
            "Aggregated upstream patient {} to downstream patient {}",
            patient.simpleResourceId(),
            result.simpleResourceId()
        )
        return result
    }

    private fun createManagingOrganizationReference(patient: R4Patient, references: IdentifierMap) =
        if (patient.hasManagingOrganization()) {
            val source = patient.managingOrganization.getSourcedReference()
            val needle = UpstreamResource(source.upstreamSource, patient.managingOrganization.reference)
            references.getOrganizationReference(needle).apply {
                extension = patient.managingOrganization.extension.map { it.copy() }
            }
        } else {
            null
        }

    private fun createGeneralPractitionerReference(patient: R4Patient, references: IdentifierMap) =
        if (patient.hasGeneralPractitioner()) {
            patient.generalPractitioner.map { gp: R4Reference ->
                if (gp.referenceElement.resourceType == "Practitioner") {
                    val source = gp.getSourcedReference()
                    val needle = UpstreamResource(source.upstreamSource, gp.reference)
                    references.getPractitionerReference(needle).apply {
                        extension = gp.extension.map { it.copy() }
                    }
                } else {
                    throw RuntimeException("Unexpected resource type ${gp.referenceElement.resourceType}")
                }
            }.toMutableList()
        } else {
            mutableListOf()
        }

    private fun createIdentifiersWithTranslatedReferences(patient: R4Patient, references: IdentifierMap) =
        if (patient.hasIdentifier()) {
            patient.identifier
                .map { upstreamIdentifier: Identifier ->
                    val downstreamIdentifier = upstreamIdentifier.copy()
                    val source = upstreamIdentifier.getSourcedReference()

                    // Could be an org reference in the assigner
                    if (upstreamIdentifier.hasAssigner() && upstreamIdentifier.assigner?.reference != null) {
                        val needle = UpstreamResource(source.upstreamSource, upstreamIdentifier.assigner.reference)
                        downstreamIdentifier.assigner = references.getOrganizationReference(needle)
                    }

                    // Could be org references in custom PHR extensions for source data
                    if (upstreamIdentifier.hasExtension()) {
                        downstreamIdentifier.extension = createTranslatedPhrExtensionReferences(
                            source,
                            upstreamIdentifier.extension,
                            references,
                            phrReferenceHelper
                        )
                    }

                    downstreamIdentifier
                }
                .toMutableList()
        } else {
            mutableListOf()
        }
}
