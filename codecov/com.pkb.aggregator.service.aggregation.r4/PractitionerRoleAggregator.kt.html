package com.pkb.aggregator.service.aggregation.r4

import com.pkb.aggregator.R4Extension
import com.pkb.aggregator.R4PractitionerRole
import com.pkb.aggregator.extension.findSingleR4UpstreamTagExtension
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.extension.safeAddAll
import com.pkb.aggregator.extension.safeAddExtensions
import com.pkb.aggregator.extension.simpleDownstreamResourceId
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.extension.untagSource
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.aggregation.ResourceAggregator
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import com.pkb.aggregator.service.reference.ReferenceExtractor
import com.pkb.aggregator.service.reference.ReferenceExtractor.IdentifierMap
import org.hl7.fhir.instance.model.api.IDomainResource
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

/**
 * Aggregates multiple resources referring to the same practitioner role into a single resource for the downstream store
 */
@Service
class PractitionerRoleAggregator(
    private val referenceExtractor: ReferenceExtractor,
) : ResourceAggregator<R4PractitionerRole> {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    fun mergeAll(upstreamResources: List<IDomainResource>, references: IdentifierMap): List<R4PractitionerRole> {

        // This is a bit fiddly. Basically for each upstream role we need to translate it to a downstream practitioner
        // and downstream org, and then group by both those references (because we might have two different upstreams
        // telling us a practitioner has a role at an organization, but each of those upstreams hvae completely
        // different resource ids for both) and merge where we have two entries referring to the same org/pro
        // relationship.
        //
        // Example scenario:
        //
        // Upstream 1: up1-pro1 is a member of up1-org
        // Upstream 2: up2-pro1 is a member of up2-org
        // Upstream 2: up2-pro2 is a member of up2-org
        //
        // Downstream: up1-pro1 and up2-pro1 are the same pro: down1-pro1
        //             up2-pro2 is a different pro:            down1-pro2
        //             up1-org and up2-org are the same org:   down1-org
        //
        // Passed into this function are 3 PractitionerRoles:
        //    Role(up1-pro1, up1-org)
        //    Role(up2-pro1, up2-org)
        //    Role(up2-pro2, up2-org)
        //
        // And an identifier map:
        //    practitioners: [ (up1-pro1, down1-pro1), (up2-pro1, down1-pro1), (up2-pro2, down1-pro2) ]
        //    organizations: [ (up1-org, down1-org), (up2-org, down1-org) ]
        //
        // Expected output from this function is two roles, one for each downstream pro linking them to the downstream org.
        // For down1-pro1 the role information should be merged from the two upstreams
        //
        // [ Role(down1-pro1, down1-org), Role(down1-pro2, down1-org) ]

        val practitionerRoles = upstreamResources.mapNotNull { it as? R4PractitionerRole }

        // Group by downstream pro and org, merging cases where there's more than one
        return practitionerRoles
            .groupBy { downstreamPractitionerOrganizationKey(it, references) }
            .map { merge(it.value, null as String?, references) }
    }

    data class DownstreamPractitionerOrganizationKey(val practitionerResourceId: String, val organizationResourceId: String)

    override fun createDownstreamInstance(downstreamId: String) = R4PractitionerRole()

    override fun mergeResource(
        downstreamResource: R4PractitionerRole,
        upstreamResource: IDomainResource,
        references: IdentifierMap
    ) {
        val upstreamPractitionerRole = upstreamResource as? R4PractitionerRole
            ?: throw UnexpectedTypeException(upstreamResource)

        val source = upstreamPractitionerRole.findSingleR4UpstreamTagExtension()

        // TODO PHR-9286
        downstreamResource.active = upstreamPractitionerRole.active
        downstreamResource.organization = createOrganizationReference(upstreamPractitionerRole, references)
        downstreamResource.practitioner = createPractitionerReference(upstreamPractitionerRole, references)
        downstreamResource.period = upstreamPractitionerRole.period.copy()
        downstreamResource.availabilityExceptions = upstreamPractitionerRole.availabilityExceptions

        downstreamResource.identifier.safeAddAll(upstreamPractitionerRole.identifier, source)
        downstreamResource.code.safeAddAll(upstreamPractitionerRole.code, source)
        downstreamResource.specialty.safeAddAll(upstreamPractitionerRole.specialty, source)
        downstreamResource.location.safeAddAll(upstreamPractitionerRole.location, source)
        downstreamResource.telecom.safeAddAll(upstreamPractitionerRole.telecom, source)
        downstreamResource.availableTime.safeAddAll(upstreamPractitionerRole.availableTime, source)
        downstreamResource.notAvailable.safeAddAll(upstreamPractitionerRole.notAvailable, source)

        downstreamResource.safeAddExtensions(upstreamPractitionerRole.extension)
    }

    override fun stripResource(downstreamResource: R4PractitionerRole, sourceTag: R4Extension) {
        downstreamResource.identifier.untagSource(sourceTag)
        downstreamResource.code.untagSource(sourceTag)
        downstreamResource.specialty.untagSource(sourceTag)
        downstreamResource.telecom.untagSource(sourceTag)
        downstreamResource.availableTime.untagSource(sourceTag)
        downstreamResource.notAvailable.untagSource(sourceTag)

        downstreamResource.extension.removeIf { it.equalsDeep(sourceTag) }
    }

    override fun mergeIdentifiers(downstreamResource: R4PractitionerRole) {
        // Do nothing - we don't currently care about consistent downstream ids for this resource because they
        // are basically just a way to join orgs and pros
    }

    override fun deriveDownstreamResourceId(currentDownstreamId: String?, resources: List<IDomainResource>): String {
        // resources list contains 1..N resources representing the same relationship, so just grab the first
        val role = resources
            .mapNotNull { it as? R4PractitionerRole }
            .minByOrNull { it.practitioner.reference }!!
        return currentDownstreamId ?: role.simpleDownstreamResourceId()
    }

    fun updateOrCreate(
        practitionerRole: R4PractitionerRole,
        currentDownstream: R4PractitionerRole?,
        downstreamResourceId: String?
    ): R4PractitionerRole {
        val resources = listOf(practitionerRole)
        val references = referenceExtractor.extract(resources)
        val result = merge(resources, currentDownstream, downstreamResourceId, references)
        log.debug(
            "Aggregated upstream practitioner role {} to downstream practitioner role {}",
            practitionerRole.simpleResourceId(),
            result.simpleResourceId()
        )
        return result
    }

    private fun downstreamPractitionerOrganizationKey(
        upstreamPractitionerRole: R4PractitionerRole,
        references: IdentifierMap
    ): DownstreamPractitionerOrganizationKey {

        // Get source information
        val source = upstreamPractitionerRole.getSourcedReference()

        // Lookup the downstream practitioner
        val practitionerReference = UpstreamResource(source.upstreamSource, upstreamPractitionerRole.practitioner.reference)
        val practitionerResourceId = references.practitioners
            .find { it.upstream == practitionerReference }
            ?.downstreamId
            ?: throw RuntimeException("Could not find downstream practitioner resource id for $practitionerReference")

        // Lookup the downstream organization
        val organizationReference = UpstreamResource(source.upstreamSource, upstreamPractitionerRole.organization.reference)
        val organizationResourceId = references.organizations
            .find { it.upstream == organizationReference }
            ?.downstreamId
            ?: throw RuntimeException("Could not find downstream organization resource id for $organizationReference")

        return DownstreamPractitionerOrganizationKey(practitionerResourceId, organizationResourceId)
    }

    private fun createOrganizationReference(role: R4PractitionerRole, references: IdentifierMap) =
        if (role.hasOrganization()) {
            val source = role.organization.getSourcedReference()
            val needle = UpstreamResource(source.upstreamSource, role.organization.reference)
            references.getOrganizationReference(needle).apply {
                extension = role.organization.extension.map { it.copy() }
            }
        } else {
            null
        }

    private fun createPractitionerReference(role: R4PractitionerRole, references: IdentifierMap) =
        if (role.hasPractitioner()) {
            val source = role.practitioner.getSourcedReference()
            val needle = UpstreamResource(source.upstreamSource, role.practitioner.reference)
            references.getPractitionerReference(needle).apply {
                extension = role.practitioner.extension.map { it.copy() }
            }
        } else {
            null
        }
}
