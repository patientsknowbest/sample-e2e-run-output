/**
 * Aggregates multiple resources referring to the same organization into a single resource for the downstream store
 */
@Service
class OrganizationAggregator(
    private val organizationIdentifierService: OrganizationIdentifierService,
    private val referenceExtractor: ReferenceExtractor,
) : ResourceAggregator<R4Organization> {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun createDownstreamInstance(downstreamId: String) =
        R4Organization().apply {
            this.active = true
            this.id = downstreamId
        }

    override fun mergeResource(
        downstreamResource: R4Organization,
        upstreamResource: IDomainResource,
        references: ReferenceExtractor.IdentifierMap
    ) {

        val upstreamOrganization = upstreamResource as? R4Organization
            ?: throw UnexpectedTypeException(upstreamResource)
        val source = upstreamOrganization.findSingleR4UpstreamTagExtension()

        // TODO PHR-9286
        downstreamResource.name = upstreamOrganization.name
        if (upstreamOrganization.hasPartOf()) {
            downstreamResource.partOf = createOrganizationReference(upstreamOrganization, references)
        }

        downstreamResource.identifier.safeAddAll(upstreamOrganization.identifier, source)
        downstreamResource.type.safeAddAll(upstreamOrganization.type, source)
        downstreamResource.telecom.safeAddAll(upstreamOrganization.telecom, source)
        downstreamResource.address.safeAddAll(upstreamOrganization.address, source)
        downstreamResource.contact.safeAddAll(upstreamOrganization.contact, source)

        downstreamResource.safeAddExtensions(upstreamOrganization.extension)
    }

    override fun stripResource(downstreamResource: R4Organization, sourceTag: R4Extension) {
        downstreamResource.identifier.untagSource(sourceTag)
        downstreamResource.type.untagSource(sourceTag)
        downstreamResource.address.untagSource(sourceTag)
        downstreamResource.telecom.untagSource(sourceTag)
        downstreamResource.contact.untagSource(sourceTag)

        downstreamResource.extension.removeIf { it.equalsDeep(sourceTag) }
    }

    override fun mergeIdentifiers(downstreamResource: R4Organization) {
        val newIdentifiers = organizationIdentifierService.mergeIdentifiers(downstreamResource)
        if (newIdentifiers > 0) {
            log.info(
                "Created {} new identifiers for downstream organization {}",
                newIdentifiers, downstreamResource.simpleResourceId()
            )
        }
    }

    override fun deriveDownstreamResourceId(currentDownstreamId: String?, resources: List<IDomainResource>) =
        currentDownstreamId ?: organizationIdentifierService.deriveDownstreamResourceId(resources)

    override fun updateReferences(downstreamResource: R4Organization, references: ReferenceExtractor.IdentifierMap) {
        val downstreamResourceId = downstreamResource.simpleResourceId()
        val idPairs = downstreamResource.findAllSourcedReference()
            .map {
                UpstreamDownstreamIdPair(
                    upstream = it,
                    downstreamId = downstreamResourceId
                )
            }
        references.organizations.addAll(idPairs)
    }

    fun updateOrCreate(
        organization: R4Organization,
        currentDownstream: R4Organization?,
        downstreamResourceId: String?
    ): R4Organization {
        val resources = listOf(organization)
        val references = referenceExtractor.extract(resources)
        return merge(resources, currentDownstream, downstreamResourceId, references)
    }

    private fun createOrganizationReference(organization: R4Organization, references: ReferenceExtractor.IdentifierMap) =
        if (organization.hasPartOf()) {
            val source = organization.partOf.getSourcedReference()
            val needle = UpstreamResource(source.upstreamSource, organization.partOf.reference)
            references.getOrganizationReference(needle).apply {
                extension = organization.partOf.extension.map { it.copy() }
            }
        } else {
            null
        }
}
