package com.pkb.aggregator.service.aggregation.r4

import com.pkb.aggregator.R4Element
import com.pkb.aggregator.R4Extension
import com.pkb.aggregator.R4Practitioner
import com.pkb.aggregator.extension.findAllSourcedReference
import com.pkb.aggregator.extension.findSingleR4UpstreamTagExtension
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.extension.safeAddAll
import com.pkb.aggregator.extension.safeAddExtensions
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.extension.untagSource
import com.pkb.aggregator.service.PractitionerIdentifierService
import com.pkb.aggregator.service.aggregation.ResourceAggregator
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import com.pkb.aggregator.service.reference.PhrReferenceHelper
import com.pkb.aggregator.service.reference.ReferenceExtractor
import com.pkb.aggregator.service.reference.ReferenceExtractor.IdentifierMap
import com.pkb.aggregator.service.reference.UpstreamDownstreamIdPair
import org.hl7.fhir.instance.model.api.IDomainResource
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

/**
 * Aggregates multiple resources referring to the same practitioner into a single resource for the downstream store
 */
@Service
class PractitionerAggregator(
    private val practitionerIdentifierService: PractitionerIdentifierService,
    private val referenceExtractor: ReferenceExtractor,
    private val phrReferenceHelper: PhrReferenceHelper,
) : ResourceAggregator<R4Practitioner> {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    fun mergeAll(upstreamResources: List<IDomainResource>, references: IdentifierMap) =
        practitionerIdentifierService.groupResourcesByPractitioner(upstreamResources)
            .map { merge(it.value, it.key, references) }

    override fun createDownstreamInstance(downstreamId: String) =
        R4Practitioner().apply {
            this.active = true
            this.id = downstreamId
        }

    override fun mergeResource(
        downstreamResource: R4Practitioner,
        upstreamResource: IDomainResource,
        references: IdentifierMap
    ) {

        val upstreamPractitioner = upstreamResource as? R4Practitioner
            ?: throw UnexpectedTypeException(upstreamResource)
        val sourceTag = upstreamPractitioner.findSingleR4UpstreamTagExtension()

        downstreamResource.identifier.safeAddAll(upstreamPractitioner.identifier, sourceTag)
        downstreamResource.name.safeAddAll(upstreamPractitioner.name, sourceTag)
        val newTelecoms = createTelecomsWithTranslatedReferences(upstreamPractitioner, references)
        downstreamResource.telecom.safeAddAll(newTelecoms, sourceTag)
        downstreamResource.address.safeAddAll(upstreamPractitioner.address, sourceTag)
        downstreamResource.photo.safeAddAll(upstreamPractitioner.photo, sourceTag)
        downstreamResource.qualification.safeAddAll(upstreamPractitioner.qualification, sourceTag)
        downstreamResource.communication.safeAddAll(upstreamPractitioner.communication, sourceTag)

        // merge extensions
        val newExtensions = createTranslatedPhrExtensionReferences(
            upstreamPractitioner.getSourcedReference(),
            upstreamPractitioner.extension,
            references,
            phrReferenceHelper
        )
        downstreamResource.safeAddExtensions(newExtensions)
    }

    override fun stripResource(downstreamResource: R4Practitioner, sourceTag: R4Extension) {
        downstreamResource.identifier.untagSource(sourceTag)
        downstreamResource.name.untagSource(sourceTag)
        downstreamResource.address.untagSource(sourceTag)
        downstreamResource.telecom.untagSource(sourceTag)

        downstreamResource.extension.removeIf { it.equalsDeep(sourceTag) }
    }

    override fun mergeIdentifiers(downstreamResource: R4Practitioner) {
        val newIdentifiers = practitionerIdentifierService.mergeIdentifiers(downstreamResource)
        if (newIdentifiers > 0) {
            log.info(
                "Created {} new identifiers for downstream practitioner {}",
                newIdentifiers, downstreamResource.simpleResourceId()
            )
        }
    }

    override fun deriveDownstreamResourceId(currentDownstreamId: String?, resources: List<IDomainResource>) =
        currentDownstreamId ?: practitionerIdentifierService.deriveDownstreamResourceId(resources)

    override fun updateReferences(downstreamResource: R4Practitioner, references: IdentifierMap) {
        val downstreamResourceId = downstreamResource.simpleResourceId()
        val idPairs = downstreamResource.findAllSourcedReference()
            .map {
                UpstreamDownstreamIdPair(
                    upstream = it,
                    downstreamId = downstreamResourceId
                )
            }
        references.practitioners.addAll(idPairs)
    }

    fun updateOrCreate(
        practitioner: R4Practitioner,
        currentDownstream: R4Practitioner?,
        downstreamResourceId: String?
    ): R4Practitioner {
        val resources = listOf(practitioner)
        val references = referenceExtractor.extract(resources)
        val result = merge(resources, currentDownstream, downstreamResourceId, references)
        log.debug(
            "Aggregated upstream practitioner {} to downstream practitioner {}",
            practitioner.simpleResourceId(),
            result.simpleResourceId()
        )
        return result
    }

    private fun createTelecomsWithTranslatedReferences(practitioner: R4Practitioner, references: IdentifierMap) =
        if (practitioner.hasTelecom()) {
            tmp(practitioner.telecom, references).toMutableList()
        } else {
            mutableListOf()
        }

    private fun <T : R4Element> tmp(els: List<T>, references: IdentifierMap) =
        els.map { upstreamElement ->
            @Suppress("UNCHECKED_CAST") val downstreamElement = upstreamElement.copy() as T
            val source = upstreamElement.getSourcedReference()

            if (upstreamElement.hasExtension()) {
                downstreamElement.extension = createTranslatedPhrExtensionReferences(
                    source,
                    upstreamElement.extension,
                    references,
                    phrReferenceHelper
                )
            }

            downstreamElement
        }.toMutableList()
}
