package com.pkb.messagehasher;

import com.pkb.crypto.dto.AccountKeysDTO;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.List;

@Service
public class AccountProcessorBatch {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    MessageProcessor messageProcessor;

    @Autowired
    MetricsHelper metricsHelper;


    void processBatchOfMessages(List<Message> messageBatch, AccountKeysDTO accountKeys) {
        for (Message message : messageBatch) {
            try {
                messageProcessor.processMessage(message, accountKeys);
            } catch (Exception e) {
                LOGGER.error("Skipping message {} due to exception ", message.getId(), e);
                metricsHelper.errorCounter("AccountProcessorBatch", e.getClass().getSimpleName()).increment();
            }
        }
    }
}
