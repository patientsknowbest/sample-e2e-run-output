package com.pkb.messagehasher;

import com.pkb.entities.core.CoreAccount;
import com.pkb.messagehasher.configuration.HashConfig;
import com.pkb.messagehasher.kmsclient.KmsClientHelper;
import com.pkb.messagehasher.repository.AccountRepository;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import javax.transaction.Transactional;
import java.util.List;

import static com.google.common.collect.Iterables.getLast;

@Service
public class MessageHashCoordinator {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    HashConfig hashConfig;

    @Autowired
    AccountRepository accountRepo;

    @Autowired
    KmsClientHelper kmsClientHelper;

    @Autowired
    AccountProcessor accountProcessor;

    @Autowired
    MetricsHelper metricsHelper;

    @Autowired
    DetectDifferences detectDifferences;

    @Transactional()
    public boolean run() {

        LOGGER.info("MessageHashCoordinator instance {} starting run. There are {} instances", hashConfig.getClientInstance(), hashConfig.getCountInstances());
        if (!hashConfig.isProcessorEnabled()) {
            LOGGER.warn("MessageHashCoordinator ending as run not enabled");
            return false;
        }

        //temporary differences check for PHRZ-815
        if (hashConfig.isTempDifferencesCheckEnabled())
        {
            LOGGER.debug("DetectDifferences: starting");
            detectDifferences.checkDifferencesBetweenHardcodedMessages();
            LOGGER.debug("DetectDifferences: finished");
        }

        try {
            List<CoreAccount> accountsBatch = metricsHelper.accountRepositoryBatchQueryTimer().record(
                    () -> getBatchOfAccountsWithIdGreaterThan(Long.MIN_VALUE)
            );

            while (!accountsBatch.isEmpty()) {
                processAccounts(accountsBatch);

                Long lastIdFromBatch = getLast(accountsBatch).getId();
                accountsBatch = metricsHelper.accountRepositoryBatchQueryTimer().record(
                        () -> getBatchOfAccountsWithIdGreaterThan(lastIdFromBatch)
                );
            }
        } catch (Exception e) {
            LOGGER.error("Unexpected error processing", e);
            metricsHelper.errorCounter("MessageHashCoordinator", e.getClass().getSimpleName()).increment();
        }
        return true;
    }

    private List<CoreAccount> getBatchOfAccountsWithIdGreaterThan(long accountIdGreaterThan) {
        return accountRepo.getAccountsWithMessagesToProcess(accountIdGreaterThan,
                hashConfig.getAccountRetrieveSize(),
                hashConfig.getClientInstance(),
                hashConfig.getCountInstances());
    }

    private void processAccounts(List<CoreAccount> accounts) {
        for (CoreAccount account : accounts) {
            LOGGER.debug("processing account {}", account.getId());
            kmsClientHelper.getAccountKeys(account.getPublicId())
                    .ifPresent(accountKeys -> metricsHelper.accountTimer().record(
                            () -> accountProcessor.processMessagesForAccount(account.getId(), accountKeys)
                    ));
        }
    }
}
