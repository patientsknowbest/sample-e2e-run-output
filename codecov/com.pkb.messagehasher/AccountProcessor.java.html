package com.pkb.messagehasher;

import com.pkb.crypto.dto.AccountKeysDTO;
import com.pkb.messagehasher.configuration.HashConfig;
import com.pkb.messagehasher.repository.MessageRepository;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.List;

import static com.google.common.collect.Iterables.getLast;

@Service
public class AccountProcessor {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    HashConfig hashConfig;

    @Autowired
    MessageRepository messageRepo;

    @Autowired
    AccountProcessorBatch batchProcessor;

    @Autowired
    MetricsHelper metricsHelper;

    void processMessagesForAccount(long accountId, AccountKeysDTO accountKeys) {

        List<Message> batch = metricsHelper.messageRepositoryBatchQueryTimer().record(
                () -> messageRepo.getMessagesToProcessBatch(accountId, Long.MIN_VALUE, hashConfig.getMessageRetrieveSize())
        );

        while (!batch.isEmpty()) {

            batchProcessor.processBatchOfMessages(batch, accountKeys);

            Long lastIdFromBatch = getLast(batch).getId();
            batch = metricsHelper.messageRepositoryBatchQueryTimer().record(
                    () -> messageRepo.getMessagesToProcessBatch(accountId, lastIdFromBatch, hashConfig.getMessageRetrieveSize())
            );
        }
    }
}
