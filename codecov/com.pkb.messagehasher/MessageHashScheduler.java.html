package com.pkb.messagehasher;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
import org.springframework.context.annotation.Configuration;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Service;

@Service
@Configuration
@ConditionalOnProperty(name = "hash.schedulerEnabled", havingValue = "true")
public class MessageHashScheduler {

    @Autowired
    MessageHashCoordinator coordinator;

    @Scheduled(fixedDelayString = "${hash.runDelayMillis}")
    public void runSchedule() {
        coordinator.run();
    }
}
