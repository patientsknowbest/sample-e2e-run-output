package com.pkb.messagehasher;

import io.micrometer.core.instrument.Counter;
import io.micrometer.core.instrument.Metrics;
import io.micrometer.core.instrument.Timer;
import org.springframework.stereotype.Service;

@SuppressWarnings("WeakerAccess")
@Service
public class MetricsHelper {

    private static final String ACCOUNT_TIMER_NAME = "messagehasher.account.timer";
    private static final String HASHING_TIMER_NAME = "messagehasher.hashing.timer";
    private static final String ACCOUNT_REPOSITORY_TIMER_NAME = "messagehasher.repository.account.batchquery.timer";
    private static final String MESSAGE_REPOSITORY_TIMER_NAME = "messagehasher.repository.message.batchquery.timer";

    private static final String ERROR_COUNT = "messagehasher.error.count";
    private static final String KMS_KEY_HIT_COUNT = "messagehasher.kms.keyhit.count";

    public Timer hashingTimer() {
        return makeTimer(HASHING_TIMER_NAME);
    }

    public Timer accountTimer() {
        return makeTimer(ACCOUNT_TIMER_NAME);
    }

    public Timer accountRepositoryBatchQueryTimer() {
        return makeTimer(ACCOUNT_REPOSITORY_TIMER_NAME);
    }

    public Timer messageRepositoryBatchQueryTimer() {
        return makeTimer(MESSAGE_REPOSITORY_TIMER_NAME);
    }

    public Counter kmsKeyHitCounter(String keysReturned, String errorType) {
        return makeCounter(KMS_KEY_HIT_COUNT, "keys_returned", keysReturned, "error_type", errorType);
    }

    public Counter errorCounter(String location, String exceptionClass) {
        return makeCounter(ERROR_COUNT, "location", location, "exception", exceptionClass);
    }

    private static Timer makeTimer(String name, String... tags) {
        return Timer.builder(name)
                .publishPercentileHistogram()
                .publishPercentiles(0.5, 0.75, 0.9, 0.95, 0.99, 0.999)
                .tags(tags)
                .register(Metrics.globalRegistry);
    }

    private static Counter makeCounter(String name, String... tags) {
        return Counter.builder(name)
                .tags(tags)
                .register(Metrics.globalRegistry);
    }
}
