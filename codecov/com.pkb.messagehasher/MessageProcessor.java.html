package com.pkb.messagehasher;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.common.ehrdata.serialization.EHRDataDeserializer;
import com.pkb.crypto.dto.AccountKeysDTO;
import com.pkb.entities.pub.PublicMessageHash;
import com.pkb.messagehasher.decryptor.MessageDecryptor;
import com.pkb.messagehasher.versionmigrator.MessageMigratorHelper;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import javax.persistence.EntityManager;
import javax.persistence.PersistenceContext;
import javax.transaction.Transactional;
import java.util.Map;
import java.util.Optional;

import static javax.transaction.Transactional.TxType.REQUIRES_NEW;

@Service
public class MessageProcessor {

    public static final String NO_CONTENT_TO_HASH = "NO_CONTENT";
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @PersistenceContext
    EntityManager em;

    @Autowired
    MessageHasher hasher;

    @Autowired
    MessageDecryptor localDecryptor;

    @Autowired
    EHRDataDeserializer deserializer;

    @Autowired
    MessageMigratorHelper migratorHelper;

    @Autowired
    DateTimeService dateTimeService;

    @Autowired
    MetricsHelper metricsHelper;

    @Transactional(REQUIRES_NEW)
    public void processMessage(Message message, AccountKeysDTO accountKeys) {
        LOGGER.debug("processing message {}", message.getId());

        Optional<String> content = getContentWithMigration(message, accountKeys);

        String hash = content.map(value -> hashContentRecordingDuration(value))
                .orElse(NO_CONTENT_TO_HASH);

        PublicMessageHash mh = new PublicMessageHash();
        mh.setMessageId(message.getId());
        mh.setCreated(dateTimeService.now());
        mh.setHash(hash);
        mh.setMessageMigrationVersion(MessageMigratorHelper.MAX_MIGRATION_RESULT_VERSION);

        em.persist(mh);

        LOGGER.trace("finished processing message {}", message.getId());
    }

    @NotNull
    private String hashContentRecordingDuration(String content) {
        String hashVerboseOutput = metricsHelper.hashingTimer().record(() -> hasher.generateHash(content));
        String params = hashVerboseOutput.substring(0, hashVerboseOutput.lastIndexOf('$'));
        String paramsWithoutSalt = params.substring(0, params.lastIndexOf('$'));
        String hash = hashVerboseOutput.substring(hashVerboseOutput.lastIndexOf('$'));
        return paramsWithoutSalt + hash;
    }

    public Optional<String> getContentWithMigration(Message message, AccountKeysDTO accountKeys) {
        byte[] decrypted = localDecryptor.decryptEncryptedData(message, accountKeys);

        Map<String, Object> stringObjectMap = deserializer.deserialize(decrypted);

        stringObjectMap = migratorHelper.migrateMessageIfRequired(message, stringObjectMap);

        return Optional.ofNullable(
                (String) stringObjectMap.getOrDefault("content", "")
        );
    }
}
