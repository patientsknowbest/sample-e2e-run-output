package com.pkb.app.entity;

import com.pkb.consent.model.ConsentStatus;
import com.pkb.datamodel.Hl7Partner;
import com.pkb.entities.enums.Route;
import com.pkb.institute.entity.Org;
import org.immutables.value.Value.Check;
import org.immutables.value.Value.Default;
import org.immutables.value.Value.Derived;
import org.jetbrains.annotations.NotNull;

import java.io.Serializable;
import java.util.Optional;
import java.util.UUID;
import java.util.function.Consumer;
import java.util.function.Function;

import static com.pkb.consent.model.AbstractConsentStatus.noConsentRequired;

/**
 * Each request to EHRBean needs some extra context for logging, decryption, etc.
 * <p>
 * When EHRBean runs a query, it will need:
 * <ul>
 * <li>account private key (can derive this, or it must be provided. Deriving is a bit trickier with
 * org-based crypto keys)</li>
 * <li>the consent categories enabled (G/S/M/S) -- this may be take a few paths ("my account" vs.
 * "PatientConsent record" vs. "broken-glass status")</li>
 * <li>access reason, for adding an AccessLog entry.</li>
 * </ul>
 * This will be something like webapp/basic vs. API vs. export.
 * <p>
 * Queries can fetch data from <em>multiple</em> patient accounts, so we need to support that. Postpone for initial impl?
 * Also some queries will be fetching data from clinician accounts (messages, schedule) and thus don't require consent,
 * or even record-access logging.
 *
 * @see EHRBean
 */
@SuppressWarnings("override")
public abstract class EHRRequestContext implements Serializable, CompoundPersonRoles, PersonRoles {
    private static final long serialVersionUID = 7180638540059041727L;

    public enum AccountLinkType {
        /**
         * Used by carers, non-team professionals, current user's own record
         */
        INDIVIDUAL,
        /**
         * Used by team professionals accessing patient records, HL7 API (future)
         */
        ORG,
        /**
         * Used in special cases with no decryption -- e.g., clinician schedules
         */
        NO_LINK
    }

    public abstract Optional<Org> getOrg();

    @NotNull
    public abstract Optional<PersonAuthInfo> getMaybeAccessingUser();

    @NotNull
    public abstract Optional<Long> getMaybeRegisteringUserId();

    @Derived
    @NotNull
    public Optional<Long> getMaybeContextOrAccessingUserId() {
        return getContextUserId().or(this::getMaybeAccessingUserId);
    }

    @NotNull
    public abstract AccountLinkType getAccountLinkType();

    @NotNull
    public abstract Route getAccessRoute();

    @NotNull
    public abstract ConsentStatus getConsentStatus();

    @NotNull
    public abstract Optional<Long> getTeamId();

    @Derived
    public boolean hasTeam() {
        return getTeamId().isPresent();
    }

    @Derived
    @NotNull
    public Optional<Long> getOrgId() {
        return getOrg().map(Org::getId);
    }

    @Derived
    @NotNull
    public Optional<Long> getOrgAccountId() {
        return getOrg().map(Org::getOrgAccountId);
    }

    @NotNull
    public abstract Optional<Long> getContextUserId();

    @NotNull
    public abstract UUID getCorrelationId();

    @NotNull
    @Derived
    public Optional<Long> getMaybeAccessingUserId() {
        return getMaybeAccessingUser().map(PersonAuthInfo::getId);
    }

    @NotNull
    public abstract Optional<Hl7Partner> getMaybeHl7Partner();

    @NotNull
    @Derived
    public Optional<Long> getMaybeHl7PartnerId() {
        return getMaybeHl7Partner().flatMap(hlp -> hlp.getId().toJavaOptional());
    }

    @NotNull
    public abstract Optional<Org> getMaybeConnectingOrg();

    @Derived
    @NotNull
    public Optional<Long> getMaybeConnectingOrgId() {
        return getMaybeConnectingOrg().map(Org::getId);
    }

    @NotNull
    public abstract Optional<String> getMaybeResponsibleOrganisationIdentifier();

    @Derived
    public boolean isSharingEnabled() {
        return getConsentStatus().isPatientSharingEnabled();
    }

    /**
     * Some system components need to read data in a medical record before writing in their own data, e.g. when performing a partial update
     * on existing data. If the component does not have decryption access to the medical record (e.g. data received by the HL7 API) then the
     * data must be queued. Then, when a user who does have decryption access logs in, the queued data can be processed by piggybacking on
     * the current user's request context.
     * <p>
     * In order to ensure the data is processed successfully, piggybacked requests should not have consent filters applied to their queries.
     * This ensures that the sender of a data point always has access to their own data. It does not breach the consent model because this
     * is internal system activity only. No data is returned to a user.
     *
     * @return true - if the request context is piggybacking on the current user's request context
     */
    @Default
    public boolean isPiggyback() {
        return false;
    }

    /**
     * Use with caution - this will prevent queries that read encrypted data
     * being captured in the patient access log (a warning message will be
     * logged to papertrail). The only current example of using this is when
     * viewing the access log itself - since consent reasons are stored
     * encrypted, the act of viewing an access log containing consent changes
     * itself causes more access log entries to be created, which makes for an
     * unpleasant paging experience.
     */
    @Default
    public boolean isAccessLogSuppressed() {
        return false;
    }

    /**
     * Determines if the patient view suspended restrictions need to be applied
     *
     * @return true if the request context falls into the scope of PVS
     */
    @Derived
    public boolean isPatientViewSuspendedRequired() {
        return getMaybeAccessingUser().map(user -> user.isPatient() ||
                        (getConsentStatus().isRequired() && getAccountLinkType() != AccountLinkType.ORG))
                .orElse(false);
    }

    @Derived
    public boolean isTeamDataViewApplicable() {
        return getAccountLinkType() == AccountLinkType.ORG
                && getTeamId().isPresent();
    }

    /**
     * TeamDataView is always applicable when a team pro is viewing patient data .
     * In some cases we need to know if a team pro user is viewing data exclusively via the TeamDataView filter.
     */
    @Derived
    public boolean isTeamDataViewOnly() {
        return getConsentStatus().isPatientSharingEnabled() &&
                !getConsentStatus().isBreakTheGlassActive() &&
                getConsentStatus().getConsent().map(consent -> !consent.anyConsentGranted()).orElse(true) &&
                isTeamDataViewApplicable();
    }

    @Derived
    public boolean isUserInOwnAccount() {
        return getContextUserId().map(contextUserId -> getMaybeAccessingUserId().map(id -> id.equals(contextUserId)).orElse(false))
                .orElse(true);
    }

    public EHRRequestContext withoutAccessLog() {
        return withAccessLogSuppressed(true);
    }

    @SuppressWarnings("override")
    public abstract EHRRequestContext withAccessLogSuppressed(boolean accessLogSuppressed);

    public abstract EHRRequestContext withPiggyback(boolean piggyback);

    public abstract EHRRequestContext withOrg(Org org);

    public abstract EHRRequestContext withConsentStatus(ConsentStatus consentStatus);

    public abstract EHRRequestContext withAccessRoute(Route accessRoute);

    public abstract Optional<LoggedInEHRRequestContext> getLoggedInContext();


    public final void doIfLoggedIn(Consumer<LoggedInEHRRequestContext> action, Runnable ifLoggedOut) {
        getLoggedInContext()
                .map(context -> {
                    action.accept(context);
                    return null;
                })
                .orElseGet(() -> {
                    ifLoggedOut.run();
                    return null;
                });
    }

    public LoggedInEHRRequestContext afterUserLoggedIn(PersonAuthInfo user) {
        return ImmutableLoggedInUserEHRRequestContext.builder()
                .accountLinkType(getAccountLinkType())
                .accessRoute(getAccessRoute())
                .consentStatus(noConsentRequired())
                .accessingUser(user)
                .teamId(getTeamId())
                .org(getOrg())
                .contextUserId(getContextUserId())
                .correlationId(getCorrelationId())
                .accessLogSuppressed(isAccessLogSuppressed())
                .maybeResponsibleOrganisationIdentifier(getMaybeResponsibleOrganisationIdentifier())
                .build();
    }

    public RegisteringUserEHRRequestContext withRegisteringUser(Long maybeRegisteringUserId) {
        return ImmutableRegisteringUserEHRRequestContext.builder()
                .accountLinkType(getAccountLinkType())
                .accessRoute(getAccessRoute())
                .consentStatus(noConsentRequired())
                .teamId(getTeamId())
                .org(getOrg())
                .contextUserId(getContextUserId())
                .correlationId(getCorrelationId())
                .accessLogSuppressed(isAccessLogSuppressed())
                .maybeRegisteringUserId(maybeRegisteringUserId)
                .build();
    }

    public RegisteringUserEHRRequestContext withRegisteringUser(Optional<Long> maybeRegisteringUserId) {
        return ImmutableRegisteringUserEHRRequestContext.builder()
                .accountLinkType(getAccountLinkType())
                .accessRoute(getAccessRoute())
                .consentStatus(noConsentRequired())
                .teamId(getTeamId())
                .org(getOrg())
                .contextUserId(getContextUserId())
                .correlationId(getCorrelationId())
                .accessLogSuppressed(isAccessLogSuppressed())
                .maybeRegisteringUserId(maybeRegisteringUserId)
                .build();
    }

    public EHRRequestContext withConsentNotRequired() {
        return this.withConsentStatus(noConsentRequired());
    }

    @Check
    protected void checkOrgTeamId() {
        getTeamId().ifPresent(teamId -> getOrg().orElseThrow(() -> new RuntimeException("Request context has team id set with no org")));
    }

    @Override
    public boolean isCarer() {
        return isPatient() && !isUserInOwnAccount();
    }

    @Override
    public boolean isPro() {
        return isAccessingUser(PersonAuthInfo::isPro);
    }

    @Override
    public boolean isPatient() {
        return isAccessingUser(PersonAuthInfo::isPatient);
    }

    @Override
    public boolean isTeamCoordinator() {
        return isAccessingUser(PersonAuthInfo::isTeamCoordinator);
    }

    @Override
    public boolean isPrivacyOfficer() {
        return isAccessingUser(PersonAuthInfo::isPrivacyOfficer);
    }

    @Override
    public boolean isOrgCoordinator() {
        return isAccessingUser(PersonAuthInfo::isOrgCoordinator);
    }

    @Override
    public boolean isSuperAdmin() {
        return isAccessingUser(PersonAuthInfo::isSuperAdmin);
    }

    @Override
    public boolean isEmployee() {
        return isAccessingUser(PersonAuthInfo::isEmployee);
    }

    @Override
    public boolean isTechSupport() {
        return isAccessingUser(PersonAuthInfo::isTechSupport);
    }

    private boolean isAccessingUser(Function<? super PersonAuthInfo, Boolean> mapper) {
        return getMaybeAccessingUser()
                .map(mapper)
                .orElse(false);
    }

    public boolean isPatientAccount() {
        return isPatient() || !isUserInOwnAccount();
    }

    public boolean isUserLoggedIn() {
        return getMaybeAccessingUser().isPresent();
    }
}
