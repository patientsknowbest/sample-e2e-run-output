    public void init(UploadedDataDTO ud) {
        if (ud == null) {
            return;
        }

        if (ud.getSourceTeam() != null) {
            teamId = ud.getSourceTeam().getId();
            orgId = ud.getSourceTeam().getOrg().getId();
        } else if (ud.getSourceOrg() != null) {
            orgId = ud.getSourceOrg().getId();
        }

        if (ud.getConnectingOrg() != null) {
            connectingOrgId = ud.getConnectingOrg().getId();
        }

        if (ud.getHl7PartnerId() != null) {
            hl7PartnerId = ud.getHl7PartnerId();
        }

        if (ud.getResponsibleOrganisationIdentifier() !=null) {
            responsibleOrganisationIdentifier = ud.getResponsibleOrganisationIdentifier();
        }

        Format format = ud.getFormat();
        route = format.getRouteAssociated();
    }

    public void init(EHRData ed) {
        if (ed.getSourcePersonId() != null) {
            personId = ed.getSourcePersonId();
        }

        if (ed.getSourcePerson() != null) {
            PKBPerson sourcePerson = ed.getSourcePerson();
            sourcePersonTitle = sourcePerson.getTitle();
            sourcePersonFirstName = sourcePerson.getFirstName();
            sourcePersonLastName = sourcePerson.getLastName();
            sourcePersonPopulated = true;
        }

        if (ed.getSourceTeam() != null) {
            teamId = ed.getSourceTeam().getId();
            teamName = ed.getSourceTeam().getName();
        } else {
            teamId = ed.getSourceTeamId();
        }

        if (ed.getSourceOrg() != null) {
            orgId = ed.getSourceOrg().getId();
            orgName = ed.getSourceOrg().getName();
        } else {
            orgId = ed.getSourceOrgId();
        }

        if (ed.getConnectingOrg() != null) {
            connectingOrgId = ed.getConnectingOrg().getId();
        } else if (ed.getConnectingOrgId() != null) {
            connectingOrgId = ed.getConnectingOrgId();
        } else {
            connectingOrgId = orgId;
        }

        if (ed.getHl7PartnerId() != null) {
            hl7PartnerId = ed.getHl7PartnerId();
        }

        if (ed.getResponsibleOrganisationIdentifier() != null) {
            responsibleOrganisationIdentifier = ed.getResponsibleOrganisationIdentifier();
        }

        setText((String) ed.getEncryptedField(SOURCE_TEXT));
        route = ed.getRoute();
    }

    public void writeToEHRData(EHRData ed) {
        if ((getPersonId() != null) && (getPersonId() > 0)) {
            ed.setSourcePersonId(getPersonId());
        } else {
            ed.setSourcePersonId(null);
        }

        if ((getTeamId() != null) && (getTeamId() > 0)) {
            ed.setSourceTeamId(getTeamId());
        } else {
            ed.setSourceTeamId(null);
        }

        if ((getOrgId() != null) && (getOrgId() > 0)) {
            ed.setSourceOrgId(getOrgId());
        } else {
            ed.setSourceOrgId(null);
        }

        if (getConnectingOrgId() != null && getConnectingOrgId() > 0) {
            ed.setConnectingOrgId(getConnectingOrgId());
        } else {
            ed.setConnectingOrgId(null);
        }

        if (getHl7PartnerId() != null && getHl7PartnerId() > 0) {
            ed.setHl7PartnerId(getHl7PartnerId());
        } else {
            ed.setHl7PartnerId(null);
        }

        if (isBlank(getText())) {
            ed.setEncryptedField(SOURCE_TEXT, null);
        } else {
            ed.setEncryptedField(SOURCE_TEXT, getText());
        }

        ed.setRoute(getRoute());

        ed.setResponsibleOrganisationIdentifier(getResponsibleOrganisationIdentifier());
    }

    public String getDisplayTextFull() {
        String personName = getPersonNameText();
        return joinSkipBlanks(", ",
                text,
                personName,
                teamName,
                orgName);
    }

    public String getDisplayText() {
        // no source info
        if (personId == null && teamId == null && orgId == null) {
            return (StringUtils.isNotEmpty(text)) ? text : "";
        }
        if (!sourcePersonPopulated && teamName == null && orgName == null) {
            return "SOURCE DISPLAY NOT LOADED";
        }

        String personNameText = getPersonNameText();

        if (isBlank(personNameText)) {
            personNameText = null;
        }

        String teamNameText = joinSkipBlanks(" ", teamName, orgName);

        // only most specific source
        String mostSpecificSource = Stream.of(personNameText, teamNameText, orgName)
                .filter(Objects::nonNull)
                .findFirst()
                .orElse("");

        return joinSkipBlanks(" - ", mostSpecificSource, text);
    }

    public String getPersonNameText() {
        if (!sourcePersonPopulated) {
            return "";
        }
        return joinSkipBlanks(" ", sourcePersonTitle, sourcePersonFirstName, sourcePersonLastName);
    }

    public String getTeamAndOrgIfAvailable(String prefix) {
        return getTeamAndOrgIfAvailable(prefix, " / ");

    /**
     * Gets team name and org name joined by separator and prefixed with the
     * given prefix.
     *
     * <pre>
     * {
     *     &#64;code
     *     String s = this.getTeamAndOrgIfAvailable("Team: ", " / ");
     *     // Returns 'Team: teamName / orgName'
     * }
     * </pre>
     */
    public String getTeamAndOrgIfAvailable(String prefix, String separator) {
        StringBuilder s = new StringBuilder();
        if (isNotBlank(teamName)) {
            s.append(prefix).append(teamName);

            if (isNotBlank(orgName)) {
                s.append(separator).append(orgName);
            }
        }

        return s.toString();
    }

    /**
     * Returns the source type of the data this {@link SourceDetails} is attached to.
     * The classification leverages on the {@link Route} and the source (person, team, organization) fields of the
     * data and used for UI purposes.
     *
     * <ul>
     *     <li>Data from patient/carer/non-institutional-profs are considered to be a {@link SourceType#PERSON}</li>
     *     <li>Data from professionals or organisations/teams are considered to be an {@link SourceType#ORG}</li>
     *     <li>Data from devices are considered to be a {@link SourceType#DEVICE}</li>
     * </ul>
     *
     * @return the calculated source type of the data this {@link SourceDetails} is attached to or null if the type
     * cannot be inferred from the {@link Route}, personId, teamId, orgId fields.
     */
    @Nullable
    public SourceType getSourceType() {
        if (this.route == Route.CORE_DEVICES) {
            return SourceType.DEVICE;
        }
        if (this.personId != null) {
            if (this.teamId != null) {
                return SourceType.ORG;
            }
            return SourceType.PERSON;
        }
        if (this.orgId != null) {
            return SourceType.ORG;
        }
        return null;
    }
