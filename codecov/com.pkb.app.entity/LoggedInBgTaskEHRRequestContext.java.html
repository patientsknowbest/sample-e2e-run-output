package com.pkb.app.entity;

import org.immutables.serial.Serial;
import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

/**
 * TODO: Temporary Context: remove when uploaded data phase 2 is implemented (OMNI-63)
 * We use this context to avoid local private key decryption in KMSAccountKeyLoadingStrategy.
 */
@Value.Immutable
@Serial.Structural
@Value.Style(get = {"is*", "get*"}, passAnnotations = {NotNull.class})
public abstract class LoggedInBgTaskEHRRequestContext extends LoggedInEHRRequestContext {

    @Override
    public LoggedInBgTaskEHRRequestContext fillMissingContextUser(long userId) {
        if (getContextUserId().isPresent() && !getContextUserId().get().equals(getAccessingUserId())) {
            return this;
        }

        return ImmutableLoggedInBgTaskEHRRequestContext.builder()
                .from(this)
                .contextUserId(userId)
                .build();
    }
}
