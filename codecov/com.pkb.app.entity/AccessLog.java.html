package com.pkb.app.entity;

import com.pkb.entities.enums.MenuDataType;
import com.pkb.entities.enums.Route;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Type;
import org.jetbrains.annotations.Nullable;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.persistence.Temporal;
import javax.persistence.TemporalType;
import java.io.Serializable;
import java.util.Date;
import java.util.UUID;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

/**
 * Captured log for each access of a patient's medical record.
 * Written by EHRBean while running queries.
 *
 * @author robwhelan
 */
@Entity
@Table
public class AccessLog implements Serializable {
    private static final long serialVersionUID = 698965768708149322L;

    @Id
    @GenericGenerator(name = "seq_al", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME), // based on "pooled-lo"
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_accesslog"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(
            strategy = GenerationType.SEQUENCE,
            generator = "seq_al")
    private Long id;

    @Column(nullable = false)
    private long accountId;

    @Column // nullable
    private Long accessingPersonId;

    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
    private MenuDataType dataType;

    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
    private Route route;

    @Column(nullable = false)
    private boolean breakTheGlassActive;

    @Column // nullable
    private Long patientConsentId;

    @Column(nullable = false)
    @Temporal(TemporalType.TIMESTAMP)
    private Date accessTime;

    @Column
    private String correlationId;

    @Column(name = "privacy_flags_id") // nullable
    private Long privacyFlagsId;

    public AccessLog() {
    }

    // convenience const
    public AccessLog(long accountId, Long accessingPersonId, MenuDataType dataType, Route route,
                     boolean breakTheGlassActive, @Nullable Long patientConsentId, Date accessTime,
            UUID correlationId, Long privacyFlagsId) {
        this.accountId = accountId;
        this.accessingPersonId = accessingPersonId;
        this.dataType = dataType;
        this.route = route;
        this.breakTheGlassActive = breakTheGlassActive;
        this.patientConsentId = patientConsentId;
        this.accessTime = accessTime;
        this.correlationId = correlationId.toString();
        this.privacyFlagsId = privacyFlagsId;

    }

    @Override
    public String toString() {
        return "AccessLog{" + "id=" + id + ", accountId=" + accountId +
                ", accessingPersonId=" + accessingPersonId + ", dateType=" + dataType + ", route=" + route +
                ", breakTheGlassActive=" + breakTheGlassActive + ", patientConsentId=" + patientConsentId +
                ", accessTime=" + accessTime + ", correlationId=" + correlationId +
                ", privacyFlagsId=" + privacyFlagsId + '}';
    }

    // <editor-fold desc="Accessors">

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public long getAccountId() {
        return accountId;
    }

    public void setAccountId(long accountId) {
        this.accountId = accountId;
    }

    public Long getAccessingPersonId() {
        return accessingPersonId;
    }

    public void setAccessingPersonId(Long accessingPersonId) {
        this.accessingPersonId = accessingPersonId;
    }

    public Route getRoute() {
        return route;
    }

    public void setRoute(Route route) {
        this.route = route;
    }
    
    public boolean isBreakTheGlassActive() {
        return breakTheGlassActive;
    }

    public void setBreakTheGlassActive(boolean breakTheGlassActive) {
        this.breakTheGlassActive = breakTheGlassActive;
    }

    public Long getPatientConsentId() {
        return patientConsentId;
    }

    public void setPatientConsentId(Long patientConsentId) {
        this.patientConsentId = patientConsentId;
    }

    public Date getAccessTime() {
        return accessTime;
    }

    public void setAccessTime(Date accessTime) {
        this.accessTime = accessTime;
    }

    public String getCorrelationId() {
        return correlationId;
    }

    public void setCorrelationId(UUID correlationId) {
        this.correlationId = correlationId.toString();
    }

    public MenuDataType getDataType() {
        return dataType;
    }

    public void setDataType(MenuDataType dataType) {
        this.dataType = dataType;
    }

    public Long getPrivacyFlagsId() {
        return privacyFlagsId;
    }

    public void setPrivacyFlagsId(Long privacyFlagsId) {
        this.privacyFlagsId = privacyFlagsId;
    }

    // </editor-fold>
}
