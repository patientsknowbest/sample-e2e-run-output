package com.pkb.app.entity;

import org.apache.commons.collections4.CollectionUtils;
import org.jetbrains.annotations.NotNull;

import java.io.Serializable;
import java.util.Collection;
import java.util.Collections;

/**
 * Hold the info for filtering on a single column in a query.
 *
 * @author robwhelan
 */
public class PKBFilter implements Serializable {

    public enum Operator {
        GREATER(">", true), LESS("<", true), GREATER_OR_EQUAL(">=", true), LESS_OR_EQUAL("<=", true), EQUAL("=", true),
        NOT_EQUAL("<>", true),

        IS_NULL(" is null", false), IS_NOT_NULL(" is not null", false),

        LIKE(" like ", true), IN(" in ", true), NOT_IN(" not in ", true),

        // Null handling options!
        // By default any SQL comparison comes back "no match" if the column value is null.
        // This is unexpected for a "not in ()" or "<>" filter!  Use these options to fix it.
        // Null values are also tricky with true/false options where "null" should read as "false".

        // normal handling for avoiding 3-state booleans (T/F/null): null => false
        TRUE_AND_IS_NOT_NULL(" [= true and is not null]", false), FALSE_OR_IS_NULL(" [= false or is null]", false),

        // symbol text not directly inserted into query for these ones
        NOT_EQUAL_OR_NULL(" [<> or is null]", true), NOT_IN_OR_NULL(" [not in or is null]", true);

        private final String symbol;
        private final boolean withValue;

        Operator(String symbol, boolean withValue) {
            this.symbol = symbol;
            this.withValue = withValue;
        }

        public String getSymbol() {
            return symbol;
        }

        public boolean isWithValue() {
            return withValue;
        }
    }

    private String field;
    private Operator operator;
    private Object value;
    private boolean applyWhenChoosingLatest = true;

    /**
     * by default all filters are applied in latest by type filters
     */
    public PKBFilter(@NotNull String field, @NotNull Operator operator, Object value) {
        this(field, operator, value, true/*applyWhenChoosingLatest*/);
    }

    /**
     * applyWhenChoosingLatest is true by default; here you can override to false
     */
    public PKBFilter(@NotNull String field, @NotNull Operator operator, Object value, boolean applyWhenChoosingLatest) {
        this.field = field;
        this.operator = operator;
        this.value = value;
        this.applyWhenChoosingLatest = applyWhenChoosingLatest;

        forceValid();
    }

    private void forceValid() {
        if (operator.isWithValue() && value == null) {
            throw new IllegalArgumentException(String.format("Field %s: operator %s must not have a null value", field, operator));
        }
        if (!operator.isWithValue() && value != null) {
            throw new IllegalArgumentException(String.format("Field %s: operator %s should have a null value; got %s", field, operator, value.getClass()));
        }

        cleanCollectionValue();
        cleanDateValue();
    }

    private void cleanCollectionValue() {
        // need Collections for IN-related issues
        switch (operator) {
        case LIKE:
            if (!(value instanceof String)) {
                throw new IllegalArgumentException(String.format("Field %s: can't use 'like' operator on non-string value; got %s", field, value.getClass()));
            }
            break;
        case IN:
        case NOT_IN:
        case NOT_IN_OR_NULL:
            // input fixes/checks: if it's not a collection, wrap it
            // and don't allow empty collections
            if (!(value instanceof Collection)) {
                value = Collections.singletonList(value);
            } else if (CollectionUtils.isEmpty((Collection<?>) value)) {
                throw new IllegalArgumentException(String.format("Field %s: empty Collection for operator %s", field, operator));
            }
        default:
            break;
        }
    }

    /**
     * JOOQ has problems with java.util.Date -- convert to java.sql.Date instead
     */
    @SuppressWarnings("UseOfObsoleteDateTimeApi")
    private void cleanDateValue() {
        if (value != null) {
            if (value.getClass().equals(java.util.Date.class)) {
                value = java.sql.Timestamp.from(java.util.Date.class.cast(value).toInstant());
            } else if (value.getClass().equals(java.time.Instant.class)) {
                value = java.sql.Timestamp.from(java.time.Instant.class.cast(value));
            }
        }
    }

    public String getField() {
        return field;
    }

    public Operator getOperator() {
        return operator;
    }

    public Object getValue() {
        return value;
    }

    public void setValue(Object value) {
        this.value = value;
    }

    public boolean isApplyWhenChoosingLatest() {
        return applyWhenChoosingLatest;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (o == null || getClass() != o.getClass()) {
            return false;
        }

        PKBFilter pkbFilter = (PKBFilter) o;

        if (applyWhenChoosingLatest != pkbFilter.applyWhenChoosingLatest) {
            return false;
        }
        if (field != null ? !field.equals(pkbFilter.field) : pkbFilter.field != null) {
            return false;
        }
        if (operator != pkbFilter.operator) {
            return false;
        }
        return value != null ? value.equals(pkbFilter.value) : pkbFilter.value == null;

    }

    @Override
    public int hashCode() {
        int result = field != null ? field.hashCode() : 0;
        result = 31 * result + (operator != null ? operator.hashCode() : 0);
        result = 31 * result + (value != null ? value.hashCode() : 0);
        result = 31 * result + (applyWhenChoosingLatest ? 1 : 0);
        return result;
    }
}
