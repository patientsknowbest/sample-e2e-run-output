package com.pkb.app.entity;

import com.pkb.consent.model.ConsentStatus;
import com.pkb.entities.enums.Route;
import com.pkb.institute.entity.Org;
import org.immutables.serial.Serial;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;
import org.jetbrains.annotations.NotNull;

import java.util.Optional;
import java.util.UUID;

@Immutable
@Serial.Structural
@Style(get = { "is*", "get*" },
        passAnnotations = { NotNull.class })
public abstract class LoggedOutEHRRequestContext extends EHRRequestContext {

    @Override public @NotNull Optional<PersonAuthInfo> getMaybeAccessingUser() {
        return Optional.empty();
    }

    @Override public Optional<LoggedInEHRRequestContext> getLoggedInContext() { return Optional.empty(); }

    @Override public Optional<Long> getMaybeRegisteringUserId() { return Optional.empty(); }

    @Override public abstract ImmutableLoggedOutEHRRequestContext withOrg(Org org);

    public static ImmutableLoggedOutEHRRequestContext of(EHRRequestContext.AccountLinkType accountLinkType, Route accessRoute,
                                                         ConsentStatus consentStatus, UUID correlationId) {
        return of(accountLinkType, accessRoute, consentStatus, correlationId, Optional.empty());
    }

    public static ImmutableLoggedOutEHRRequestContext of(EHRRequestContext.AccountLinkType accountLinkType, Route accessRoute,
                                                         ConsentStatus consentStatus, UUID correlationId, String responsibleOrganisationIdentifier) {
        return of(accountLinkType, accessRoute, consentStatus, correlationId, Optional.of(responsibleOrganisationIdentifier));
    }

    private static ImmutableLoggedOutEHRRequestContext of(EHRRequestContext.AccountLinkType accountLinkType, Route accessRoute,
                                                         ConsentStatus consentStatus, UUID correlationId,
                                                         Optional<String> responsibleOrganisationIdentifier) {
        return ImmutableLoggedOutEHRRequestContext.builder()
                .accountLinkType(accountLinkType)
                .accessRoute(accessRoute)
                .consentStatus(consentStatus)
                .correlationId(correlationId)
                .maybeResponsibleOrganisationIdentifier(responsibleOrganisationIdentifier)
                .build();
    }

}
