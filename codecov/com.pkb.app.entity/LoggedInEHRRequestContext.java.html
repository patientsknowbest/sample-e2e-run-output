package com.pkb.app.entity;

import com.pkb.consent.model.ConsentStatus;
import com.pkb.institute.entity.Org;
import org.immutables.value.Value.Derived;
import org.jetbrains.annotations.NotNull;

import java.util.Optional;
import java.util.Set;
import java.util.UUID;
import java.util.stream.Stream;

import static com.pkb.consent.model.AbstractConsentStatus.noConsentRequired;
import static java.util.stream.Collectors.toUnmodifiableSet;

@SuppressWarnings("override")
public abstract class LoggedInEHRRequestContext extends EHRRequestContext {

    public abstract PersonAuthInfo getAccessingUser();


    @Derived
    public long getContextOrAccessingUserId() {
        return getContextUserId().orElseGet(this::getAccessingUserId);
    }


    @Override
    @Derived
    @NotNull
    public Optional<PersonAuthInfo> getMaybeAccessingUser() {
        return Optional.of(getAccessingUser());
    }

    @Derived
    public long getAccessingUserId() {
        return getAccessingUser().getId();
    }

    @Derived
    public UUID getAccessingUserPublicId() {
        return getAccessingUser().getPublicId();
    }

    @Override
    public Optional<LoggedInEHRRequestContext> getLoggedInContext() {
        return Optional.of(this);
    }

    @Override
    public LoggedInEHRRequestContext withoutAccessLog() {
        return (LoggedInEHRRequestContext) super.withoutAccessLog();
    }

    @Override
    public abstract LoggedInEHRRequestContext withAccessLogSuppressed(boolean accessLogSuppressed);

    @Override
    public abstract LoggedInEHRRequestContext withConsentStatus(ConsentStatus consentStatus);

    @Override
    public abstract LoggedInEHRRequestContext withOrg(Org org);

    @Override
    public abstract LoggedInEHRRequestContext withPiggyback(boolean piggyback);

    public abstract LoggedInEHRRequestContext withAccountLinkType(AccountLinkType accountLinkType);

    public LoggedInEHRRequestContext withConsentNotRequired() {
        return this.withConsentStatus(noConsentRequired());
    }

    public abstract LoggedInEHRRequestContext fillMissingContextUser(long userId);

    public abstract LoggedInEHRRequestContext withTeamId(long value);

    @Override
    public Optional<Long> getMaybeRegisteringUserId() {
        return Optional.empty();
    }

    public Set<Long> getContextAndAccessingUserIds() {
        return Stream.of(getAccessingUserId(), getContextOrAccessingUserId()).collect(toUnmodifiableSet());
    }

}
