package com.pkb.app.entity;

import com.pkb.consent.model.ConsentStatus;
import com.pkb.entities.enums.Route;
import org.immutables.serial.Serial;
import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

@Value.Immutable
@Serial.Structural
@Value.Style(get = {"is*", "get*"}, passAnnotations = {NotNull.class})
public abstract class LoggedInUserEHRRequestContext extends LoggedInEHRRequestContext {

    @Override
    public LoggedInUserEHRRequestContext fillMissingContextUser(long userId) {
        if (getContextUserId().isPresent() && !getContextUserId().get().equals(getAccessingUserId())) {
            return this;
        }

        return ImmutableLoggedInUserEHRRequestContext.builder()
                .from(this)
                .contextUserId(userId)
                .build();
    }

    public static ImmutableLoggedInUserEHRRequestContext of(EHRRequestContext.AccountLinkType accountLinkType, Route accessRoute, ConsentStatus consentStatus, UUID correlationId,
                                                            PersonAuthInfo accessingUser) {
        return ImmutableLoggedInUserEHRRequestContext.builder()
                .accountLinkType(accountLinkType)
                .accessRoute(accessRoute)
                .consentStatus(consentStatus)
                .correlationId(correlationId)
                .accessingUser(accessingUser)
                .build();
    }
}
