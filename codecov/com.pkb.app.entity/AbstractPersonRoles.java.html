package com.pkb.app.entity;

import com.pkb.entities.enums.UserType;

public abstract class AbstractPersonRoles implements PersonRoles {

    public abstract UserType getUserType();

    @Override
    public boolean isPatient() {
        return getUserType() == UserType.PATIENT;
    }

    @Override
    public boolean isPro() {
        return getUserType() == UserType.REG_CLINICIAN;
    }

    @Override
    public boolean isTeamCoordinator() {
        return getUserType() == UserType.INSTITUTE_ADMIN;
    }

    @Override
    public boolean isPrivacyOfficer() {
        return getUserType() == UserType.PRIVACY_OFFICER;
    }

    @Override
    public boolean isOrgCoordinator() {
        return getUserType() == UserType.ORG_COORD;
    }

    @Override
    public boolean isSuperAdmin() {
        return getUserType() == UserType.SUPER_ADMIN;
    }

    @Override
    public boolean isEmployee() {
        return getUserType() == UserType.EMPLOYEE;
    }

    @Override
    public boolean isTechSupport() {
        return getUserType() == UserType.TECH_SUPPORT;
    }
}
