package com.pkb.backgroundtasks.wayfinder.recordsservice.entity

import org.hibernate.annotations.Type
import javax.persistence.Column
import javax.persistence.Entity
import javax.persistence.EnumType.STRING
import javax.persistence.Enumerated
import javax.persistence.GeneratedValue
import javax.persistence.GenerationType
import javax.persistence.Id
import javax.persistence.Table

@Entity
@Table(schema = "bgtasks", name = "wayfinder_record_service_nhs_numbers_to_send")
class PendingNhsNumber(
        @Column(name = "nhs_number")
        val nhsNumber: String,

        @Column(name = "delivery_status")
        @Enumerated(STRING)
        @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
        val deliveryStatus: DeliveryStatus = DeliveryStatus.UNPROCESSED,

        @Id
        @GeneratedValue(strategy = GenerationType.AUTO)
        val id: Long = 0,
) {

    enum class DeliveryStatus {
        UNPROCESSED,
        SENT,
        FAILED
    }
}
