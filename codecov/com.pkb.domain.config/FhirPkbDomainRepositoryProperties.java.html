package com.pkb.domain.config;

import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.boot.context.properties.ConstructorBinding;
import org.springframework.boot.context.properties.bind.DefaultValue;
import org.springframework.cloud.context.config.annotation.RefreshScope;

import java.time.Duration;

@RefreshScope
@ConfigurationProperties
@ConstructorBinding
public class FhirPkbDomainRepositoryProperties {

    private final String synertecApiClientId;
    private final Duration slowDocRefQueryAlertThreshold;
    private final Duration fhirToStrutsConnectionTimeout;
    private final Duration fhirApiToStrutsWriteTimeout;
    private final Duration fhirApiToStrutsReadTimeout;
    private final String uploadedDataSyncUrl;

    public FhirPkbDomainRepositoryProperties(
            @DefaultValue("synertec-user-test") String synertecApiClientId,
            @DefaultValue("PT30S") Duration slowDocRefQueryAlertThreshold,
            @DefaultValue("PT10S") Duration fhirToStrutsConnectionTimeout,
            @DefaultValue("PT10S") Duration fhirApiToStrutsWriteTimeout,
            @DefaultValue("PT2M") Duration fhirApiToStrutsReadTimeout,
            @DefaultValue("http://phr-wf:80/synchUploadedData.action") String uploadedDataSyncUrl
    ) {

        this.synertecApiClientId = synertecApiClientId;
        this.slowDocRefQueryAlertThreshold = slowDocRefQueryAlertThreshold;
        this.fhirToStrutsConnectionTimeout = fhirToStrutsConnectionTimeout;
        this.fhirApiToStrutsWriteTimeout = fhirApiToStrutsWriteTimeout;
        this.fhirApiToStrutsReadTimeout = fhirApiToStrutsReadTimeout;
        this.uploadedDataSyncUrl = uploadedDataSyncUrl;
    }

    public String getSynertecApiClientId() {
        return synertecApiClientId;
    }

    public Duration getSlowDocRefQueryAlertThreshold() {
        return slowDocRefQueryAlertThreshold;
    }

    public Duration getFhirToStrutsConnectionTimeout() {
        return fhirToStrutsConnectionTimeout;
    }

    public Duration getFhirApiToStrutsWriteTimeout() {
        return fhirApiToStrutsWriteTimeout;
    }

    public Duration getFhirApiToStrutsReadTimeout() {
        return fhirApiToStrutsReadTimeout;
    }

    public String getUploadedDataSyncUrl() {
        return uploadedDataSyncUrl;
    }
}
