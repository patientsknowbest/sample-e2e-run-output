package com.pkb.domain.config;

import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.boot.context.properties.ConstructorBinding;
import org.springframework.boot.context.properties.bind.DefaultValue;
import org.springframework.cloud.context.config.annotation.RefreshScope;

import java.time.Duration;

@ConstructorBinding
@ConfigurationProperties("fhir.api")
@RefreshScope
public class DomainServiceProperties {


    private final String deleteRequestTopicName;

    private final String deleteResponseSubscriptionName;

    private final Duration deleteOperationCreationTimeout;
    private final Duration deleteOperationExecutionTimeout;
    private final Duration deleteOperationRollbackTimeout;

    public DomainServiceProperties(
            @DefaultValue("deleteRequest") String deleteRequestTopicName,
            @DefaultValue("deleteResponse-fhir") String deleteResponseSubscriptionName,
            @DefaultValue("PT11M") Duration deleteOperationCreationTimeout, //Default is a bit longer than the default transaction timeout, which should be the effective
            @DefaultValue("PT11M") Duration deleteOperationExecutionTimeout,// timeout in normal cases. If we hit this, something strange has happened and we need to trigger a manual
            @DefaultValue("PT11M") Duration deleteOperationRollbackTimeout  //rollback across all the participating services.
    ) {

        this.deleteRequestTopicName = deleteRequestTopicName;
        this.deleteResponseSubscriptionName = deleteResponseSubscriptionName;
        this.deleteOperationCreationTimeout = deleteOperationCreationTimeout;
        this.deleteOperationExecutionTimeout = deleteOperationExecutionTimeout;
        this.deleteOperationRollbackTimeout = deleteOperationRollbackTimeout;
    }

    public String getDeleteRequestTopicName() {
        return deleteRequestTopicName;
    }

    public String getDeleteResponseSubscriptionName() {
        return deleteResponseSubscriptionName;
    }

    public Duration getDeleteOperationCreationTimeout() {
        return deleteOperationCreationTimeout;
    }

    public Duration getDeleteOperationExecutionTimeout() {
        return deleteOperationExecutionTimeout;
    }

    public Duration getDeleteOperationRollbackTimeout() {
        return deleteOperationRollbackTimeout;
    }
}
