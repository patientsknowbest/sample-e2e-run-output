package com.pkb.domain.config;

import com.pkb.deletion.config.SpringBootDeletionServiceProperties;
import com.pkb.deletion.model.AggregateDeleteOperationResponse;
import com.pkb.deletion.model.DeleteOperationRequest;
import com.pkb.deletion.model.DeleteOperationResponse;
import com.pkb.pubsub.aggregator.models.AggregationDefinition;
import com.pkb.pubsub.aggregator.models.ImmutableAggregationDefinition;
import org.springframework.boot.context.properties.EnableConfigurationProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.List;

import static com.pkb.domain.routing.DeleteRequestRouteBuilder.DELETE_OPERATION_RESPONSE_AGGREGATED;
import static com.pkb.domain.routing.DeleteRequestRouteBuilder.HANDLE_OPERATION_REQUEST;
import static com.pkb.domain.routing.DeleteRequestRouteBuilder.OPERATION_RESPONSE;

@Configuration
@EnableConfigurationProperties(SpringBootDeletionServiceProperties.class)
public class DeletionServiceConfiguration {

    private static final List<String> DELETION_SERVICE_REQUIRED_SERVICES = List.of("phr", "questionnaireService", "bgtasks", "kms");

    @Bean
    public AggregationDefinition<DeleteOperationResponse, AggregateDeleteOperationResponse, DeleteOperationRequest> deleteOperationAggregationDefinition(DomainServiceProperties domainServiceProperties) {
        return ImmutableAggregationDefinition.<DeleteOperationResponse, AggregateDeleteOperationResponse, DeleteOperationRequest>builder()
                .inboundRequestEndpoint(HANDLE_OPERATION_REQUEST)
                .inboundResponseEndpoint(OPERATION_RESPONSE)
                .responseSerialisationClass(DeleteOperationResponse.class)
                .requestSerialisationClass(DeleteOperationRequest.class)
                .aggregateResponseEndpoint(DELETE_OPERATION_RESPONSE_AGGREGATED)
                .timeoutCalculator(req -> {
                    switch(req.getOperationType()) {
                        case CREATE:
                            return domainServiceProperties.getDeleteOperationCreationTimeout();
                        case EXECUTE:
                            return domainServiceProperties.getDeleteOperationExecutionTimeout();
                        default:
                            return domainServiceProperties.getDeleteOperationRollbackTimeout();
                    }
                })
                .addAllRequiredServiceNames(DELETION_SERVICE_REQUIRED_SERVICES)
                .build();
    }


}
