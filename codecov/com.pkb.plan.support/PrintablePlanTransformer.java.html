    public List<List<PrintableText>> transformCSVExportLatest(ExportablePatientPlans exportablePatientPlans) {

        List<List<PrintableText>> result = new ArrayList<>();
        List<PrintableText> fieldValueList;
        // For each plan
        for (ExportablePatientPlan exportablePatientPlan : exportablePatientPlans.getPlans()) {
            try {
                fieldValueList = new ArrayList<>();

                PKBPerson planCreator = exportablePatientPlan.getPlanCreator();
                fieldValueList.add(createPrintableText(planCreator.getFullName()));

                // no support for (sensitive) team-level IDs at this point
                //either national id or orgLevelId or both could be null
                fieldValueList.add(createPrintableText(exportablePatientPlan.getTargetPatientNationalId()));
                fieldValueList
                        .add(createPrintableText(exportablePatientPlan.getTargetPatientOrgLevelId() == null ? "" : exportablePatientPlan.getTargetPatientOrgLevelId().getValue()));

                if (exportablePatientPlan instanceof LatestExportablePatientPlan) {
                    fieldValueList.addAll(getLatestValueListForPlan(exportablePatientPlans, (LatestExportablePatientPlan) exportablePatientPlan));
                }

                //csvPrinter.printRecord(fieldValueList);
                result.add(fieldValueList);
            } catch (Exception e) {
                LOGGER.error("Error while exporting plan -{}", exportablePatientPlan.getPlan().getId(), e);
            }
        }
        return result;
    }

    private List<PrintableText> getLatestValueListForPlan(ExportablePatientPlans exportablePatientPlans, LatestExportablePatientPlan exportablePatientPlan) throws Exception {

        PHPlan plan = exportablePatientPlan.getPlan();
        // extract the html components names as columns
        PlanField actionField = plan.getPlanField(PHPlanTemplate.TemplateFieldLabel.ACTION_PLAN);
        List<PrintableText> fieldValueList = new ArrayList<>(createPrintableText(extractHtmlFieldsFrom(actionField)));

        PlanField greenField = plan.getPlanField(PHPlanTemplate.TemplateFieldLabel.GREEN);
        fieldValueList.addAll(createPrintableText(extractHtmlFieldsFrom(greenField)));

        PlanField amberField = plan.getPlanField(PHPlanTemplate.TemplateFieldLabel.AMBER);
        fieldValueList.addAll(createPrintableText(extractHtmlFieldsFrom(amberField)));

        PlanField redField = plan.getPlanField(PHPlanTemplate.TemplateFieldLabel.RED);
        fieldValueList.addAll(createPrintableText(extractHtmlFieldsFrom(redField)));

        Map<Long, SymptomReportDTO> latestSymptomsMap = exportablePatientPlan.getSymptomReports();
        var templateMetricsIds = templateMetricsFilter.filterTemplateMetrics(plan.getMetrics());

        for (Long key : Option.of(templateMetricsIds.get(TemplateMetric.MetricType.SYMPTOM)).getOrElse(Collections.emptyList())) {
            SymptomReportDTO reportDTO = latestSymptomsMap.get(key);
            if (reportDTO == null) {
                fieldValueList.add(createPrintableText(""));
            } else {
                fieldValueList.add(createLocalizedPrintableText(reportDTO.getSeverity().getName()));
            }
        }

        Map<MeasurementTypeId, MeasurementDTO> latestMeasurementsMap = exportablePatientPlan.getMeasurements();
        MeasurementDTO measurementDTO;
        //String measurementName;
        for (Long key : Option.of(templateMetricsIds.get(TemplateMetric.MetricType.MEASUREMENT)).getOrElse(Collections.emptyList())) {
            measurementDTO = latestMeasurementsMap.get(MeasurementTypeId.ofPredefinedTypeId(key));
            if (measurementDTO == null) {
                fieldValueList.add(createPrintableText(""));
                if (PredefinedMeasurementType.getById(key).getType().unit2().isDefined()) {
                    fieldValueList.add(createPrintableText(""));
                }
            } else {
                fieldValueList.add(createPrintableText(measurementDTO.getValue().toString()));
                if (measurementDTO.getValue2() != null) {
                    fieldValueList.add(createPrintableText(measurementDTO.getValue2().toString()));
                }
            }
        }

        Map<Long, TestResultDTO> latestTestsMap = exportablePatientPlan.getTestResults();
        for (Long testId : Option.of(templateMetricsIds.get(TemplateMetric.MetricType.LAB_TEST)).getOrElse(Collections.emptyList())) {
            TestResultDTO testResultDTO = latestTestsMap.get(testId);
            if (testResultDTO == null) {
                fieldValueList.add(createPrintableText(""));
            } else {
                LoincTest loincTest = exportablePatientPlans.getLoincTestMap().get(testId);
                String unit = getTestUnit(loincTest);
                // can't show name!  we can't localize it here... so just use the LOINC for now
                //testName = loincTest.getLoinc() + unit;
                String testValue = getTestValue(testResultDTO);
                fieldValueList.add(createPrintableText(testValue));
            }

        }
        return fieldValueList;

    }

    private String getTestUnit(LoincTest lcf) {
        String unit = "";
        if (StringUtils.isNotBlank(lcf.getUnit())) {
            unit = " (" + lcf.getUnit() + ")";
        }
        return unit;
    }

    private List<String> extractHtmlFieldsFrom(PlanField planField) throws Exception {
        if (planField != null) {
            String fieldText = planField.getText();
            if ((fieldText != null) && fieldText.startsWith("<div")) {
                return extractHtmlFields(planField).stream().map(HtmlField::getValue).collect(Collectors.toList());
            }
        }
        return Collections.emptyList();
    }

    public List<List<PrintableText>> transformCSVExportAll(ExportablePatientPlans exportablePatientPlans) {
        List<List<PrintableText>> result = new ArrayList<>();
        // These plans are all from the same template
        // First export the actionPlan contents for each plan
        for (ExportablePatientPlan exportablePatientPlan : exportablePatientPlans.getPlans()) {
            try {
                PKBPerson planCreator = exportablePatientPlan.getPlanCreator();
                String dateString = DATE_FORMAT.format(exportablePatientPlan.getPlan().getCreationDate().toInstant());
                for (List<String> row : getAllActionPlanListsForPlan(exportablePatientPlan.getPlan(), dateString)) {
                    List<PrintableText> fieldValueList = new ArrayList<>();

                    fieldValueList.add(createPrintableText(exportablePatientPlan.getTargetPatientNationalId()));
                    fieldValueList.add(
                            createPrintableText(exportablePatientPlan.getTargetPatientOrgLevelId() == null ? "" : exportablePatientPlan.getTargetPatientOrgLevelId().getValue()));

                    fieldValueList.addAll(createPrintableText(row));
                    fieldValueList.add(createPrintableText(planCreator.getFullName()));

                    result.add(fieldValueList);
                    //csvPrinter.printRecord(fieldValueList);
                }

                // Then export all the metrics being tracked in the template
                try {
                    if (exportablePatientPlan instanceof HistoricalExportablePatientPlan) {
                        HistoricalExportablePatientPlan historicalExportablePatientPlan = (HistoricalExportablePatientPlan) exportablePatientPlan;
                        for (List<PrintableText> row : getAllValueListsForPlan(historicalExportablePatientPlan)) {
                            List<PrintableText> fieldValueList = new ArrayList<>();

                            fieldValueList.add(createPrintableText(exportablePatientPlan.getTargetPatientNationalId()));
                            fieldValueList.add(createPrintableText(
                                    exportablePatientPlan.getTargetPatientOrgLevelId() == null ? "" : exportablePatientPlan.getTargetPatientOrgLevelId().getValue()));

                            fieldValueList.addAll(row);

                            result.add(fieldValueList);
                            //csvPrinter.printRecord(fieldValueList);
                        }
                    }
                } catch (Exception e) {
                    LOGGER.error("Error while exporting all tracked metrics for plan -{}", exportablePatientPlan.getPlan().getId(), e);
                }
            } catch (Exception e) {
                LOGGER.error("Error while exporting actionPlan contents for plan -{}", exportablePatientPlan.getPlan().getId(), e);
            }
        }

        return result;
    }

    private List<List<PrintableText>> getAllValueListsForPlan(HistoricalExportablePatientPlan exportablePatientPlan) throws Exception {
        List<List<PrintableText>> rowList = new ArrayList<>();
        for (SymptomReportHistoryDTO symptomHistory : exportablePatientPlan.getSymptomHistories()) {
            PrintableText symptomName = createLocalizedPrintableText(symptomHistory.getSymptom().getName());
            for (SymptomReportDTO reportDTO : symptomHistory.getSymptomReportList()) {
                List<PrintableText> fieldValueList = new ArrayList<>();
                fieldValueList.add(symptomName);
                fieldValueList.add(createPrintableText(DATE_FORMAT.format(reportDTO.getReportDate())));
                fieldValueList.add(createLocalizedPrintableText(reportDTO.getSeverity().getName()));
                fieldValueList.add(createPrintableText(""));

                PKBPerson dataEnterer = exportablePatientPlan.getSymptomIdDataEntererMapping().get(reportDTO.getId());
                if (dataEnterer == null) {
                    fieldValueList.add(createPrintableText("Not available"));
                } else {
                    fieldValueList.add(createPrintableText(dataEnterer.getFullName()));
                }
                rowList.add(fieldValueList);
            }
        }

        for (MeasurementHistoryDTO historyDTO : exportablePatientPlan.getMeasurementHistories()) {
            PrintableText measurementName = createLocalizedPrintableText(historyDTO.getMeasurementType().displayName());
            for (MeasurementDTO measurementDTO : historyDTO.getMeasurementList()) {
                measurementDTO = measurementTransformer.transformMeasurement(measurementDTO);
                List<PrintableText> fieldValueList = new ArrayList<>();
                fieldValueList.add(measurementName);
                fieldValueList.add(createPrintableText(DATE_FORMAT.format(measurementDTO.getMeasureDate())));
                fieldValueList.add(createPrintableText(measurementDTO.getValueFormatted()));
                fieldValueList.add(createPrintableText(measurementDTO.getValue2Formatted()));
                fieldValueList.add(createPrintableText(measurementDTO.getSource().getDisplayText()));

                rowList.add(fieldValueList);
            }
        }

        for (TestHistoryDTO testHistory : exportablePatientPlan.getTestHistories()) {
            PrintableText testName = createLocalizedPrintableText(testHistory.getLoincTest().getNameKey());
            for (TestResultDTO resultDTO : testHistory.getTestResultList()) {
                resultDTO = testResultTransformer.transformTestResult(resultDTO);
                String testValue = getTestValue(resultDTO);

                List<PrintableText> fieldValueList = new ArrayList<>();
                fieldValueList.add(testName);
                fieldValueList.add(createPrintableText(DATE_FORMAT.format(resultDTO.getTestDate())));
                fieldValueList.add(createPrintableText(testValue));
                fieldValueList.add(createPrintableText(""));
                fieldValueList.add(createPrintableText(resultDTO.getSource().getDisplayText()));
                rowList.add(fieldValueList);
            }
        }
        return rowList;

    }

    private String getTestValue(TestResultDTO testResultDTO) {
        String testValue = testResultDTO.getTextValue();
        if (testResultDTO.isValueWithComparator()) {
            testValue = testResultDTO.getValueComparator().toString() + testResultDTO.getValue().toString();
        } else if (testResultDTO.isValueNumeric()) {
            testValue = testResultDTO.getValue().toString();
        }
        return testValue;
    }

    @VisibleForTesting
    List<HtmlField> extractHtmlFields(PlanField planField) throws Exception {
        try {
            String fieldText = customStyledHtmlFilter.filter(planField.getText());
            DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
            DocumentBuilder db = dbf.newDocumentBuilder();
            InputSource input = new InputSource(new StringReader(fieldText));
            Document dom = db.parse(input);

            Element root = dom.getDocumentElement();
            List<HtmlField> htmlfields = new ArrayList<>();
            HtmlField htmlField;
            String value = "";
            NodeList selectNodeList = root.getElementsByTagName("select");
            List<Element> selects = new ArrayList<>();
            for (int i = 0; i < selectNodeList.getLength(); ++i) {
                selects.add((Element) selectNodeList.item(i));
            }

            for (Element select : selects) {
                htmlField = new HtmlField();
                htmlField.setName(select.getAttribute("name"));
                value = "";

                NodeList optionNodeList = select.getElementsByTagName("option");
                for (int j = 0; j < optionNodeList.getLength(); ++j) {
                    Element option = (Element) optionNodeList.item(j);
                    if ("selected".equals(option.getAttribute("selected"))) {
                        value = option.getAttribute("value");
                        break;
                    }
                }
                if ((value == null) && (optionNodeList.getLength() >= 1)) {
                    value = ((Element) optionNodeList.item(0)).getAttribute("value");
                }

                htmlField.setValue(value);
                htmlfields.add(htmlField);

            }

            NodeList inputNodeList = root.getElementsByTagName("input");
            List<Element> radios = new ArrayList<>();
            List<Element> textsDatesAndRanges = new ArrayList<>();
            List<Element> checkboxes = new ArrayList<>();
            for (int i = 0; i < inputNodeList.getLength(); ++i) {
                Element inputElement = (Element) inputNodeList.item(i);
                if ("radio".equals(inputElement.getAttribute("type"))) {
                    radios.add(inputElement);
                } else if ("text".equals(inputElement.getAttribute("type"))
                        || "date".equals(inputElement.getAttribute("type"))
                        || "range".equals(inputElement.getAttribute("type"))) {
                    textsDatesAndRanges.add((Element) inputNodeList.item(i));
                } else if ("checkbox".equals(inputElement.getAttribute("type"))) {
                    checkboxes.add(inputElement);
                }
            }

            htmlfields.addAll(
                    radios.stream()
                            .filter(r -> "checked".equals(r.getAttribute("checked")))
                            .map(this::convertSimpleInputToHtmlField)
                            .collect(Collectors.toList()));

            //NOTE: There are no reference in plan code to range elements. Is this still needed?
            htmlfields.addAll(
                    textsDatesAndRanges.stream()
                            .map(this::convertSimpleInputToHtmlField)
                            .collect(Collectors.toList()));

            htmlfields.addAll(
                    checkboxes.stream()
                            .map(this::convertCheckboxToHtmlField)
                            .collect(Collectors.toList()));

            NodeList textAreaNodeList = root.getElementsByTagName("textarea");
            List<Element> textAreas = new ArrayList<>();
            for (int i = 0; i < textAreaNodeList.getLength(); ++i) {
                Element textArea = (Element) textAreaNodeList.item(i);
                textAreas.add(textArea);
            }

            for (Element textArea : textAreas) {
                String textContent = textArea.getTextContent();
                htmlField = new HtmlField();
                htmlField.setName(textArea.getAttribute("name"));
                htmlField.setValue(textContent);
                htmlfields.add(htmlField);
            }

            return htmlfields;

        } catch (SAXParseException spe) {
            throw new RuntimeException("Error while extracting html components from template field-" + planField, spe);
        }
    }

    private HtmlField convertCheckboxToHtmlField(Element checkbox) {
        boolean checked = "checked".equals(checkbox.getAttribute("checked"));
        HtmlField htmlField = new HtmlField();
        htmlField.setName(checkbox.getAttribute("name"));
        if (checked) {
            htmlField.setValue(Boolean.TRUE.toString());
        } else {
            htmlField.setValue(Boolean.FALSE.toString());
        }
        return htmlField;
    }

    private HtmlField convertSimpleInputToHtmlField(Element text) {
        String textValue = text.getAttribute("value");
        HtmlField htmlField = new HtmlField();
        htmlField.setName(text.getAttribute("name"));
        htmlField.setValue(textValue);
        return htmlField;

    }

        return headerList;

    private List<PrintableText> extractCsvHeadersFrom(TemplateField templateField) throws Exception {
        List<PrintableText> headerList = new ArrayList<>();
        if (templateField != null) {
            String templateFieldText = templateField.getText();
            if ((templateFieldText != null) && !templateFieldText.isEmpty() && templateFieldText.startsWith("<div")) {
                List<PrintableText> csvFieldHeaders = extractCSVHeaders(templateField);
                headerList.addAll(csvFieldHeaders);
            }
        }
        return headerList;
    }

    public List<PrintableText> extractCSVHeaders(TemplateField templateField) throws Exception {
        try {
            String fieldText = customStyledHtmlFilter.filter(templateField.getText());
            DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
            DocumentBuilder db = dbf.newDocumentBuilder();
            InputSource input = new InputSource(new StringReader(fieldText));
            Document dom = db.parse(input);

            Element root = dom.getDocumentElement();
            List<PrintableText> csvHeaders = new ArrayList<>();
            //for fields like radio buttons where mutliple elements have same name
            Map<String, String> nameToFieldMap = new HashMap<>();
            String columnName;
            NodeList selectNodeList = root.getElementsByTagName("select");
            List<Element> selects = new ArrayList<>();
            for (int i = 0; i < selectNodeList.getLength(); ++i) {
                selects.add((Element) selectNodeList.item(i));
            }

            for (Element select : selects) {
                csvHeaders.add(createPrintableText(select.getAttribute("name")));
            }

            NodeList inputElementList = root.getElementsByTagName("input");
            List<Element> radios = new ArrayList<>();
            List<Element> textsDatesAndRanges = new ArrayList<>();
            List<Element> checkboxes = new ArrayList<>();

            for (int i = 0; i < inputElementList.getLength(); ++i) {
                Element radio = (Element) inputElementList.item(i);
                if ("radio".equals(radio.getAttribute("type"))) {
                    radios.add((Element) inputElementList.item(i));
                } else if ("checkbox".equals(radio.getAttribute("type"))) {
                    checkboxes.add((Element) inputElementList.item(i));
                } else if ("text".equals(radio.getAttribute("type"))
                        || "date".equals(radio.getAttribute("type"))
                        || "range".equals(radio.getAttribute("type"))) {
                    textsDatesAndRanges.add((Element) inputElementList.item(i));
                }

            }

            csvHeaders.addAll(radios.stream().map(r -> createPrintableText(r.getAttribute("name"))).distinct().collect(Collectors.toList()));
            //NOTE: There are no reference in plan code to range elements. Is this still needed?
            csvHeaders.addAll(textsDatesAndRanges.stream().map(t -> createPrintableText(t.getAttribute("name"))).collect(Collectors.toList()));

            //TODO: do we need to be doing a distinct filter for checkboxes too?
            csvHeaders.addAll(checkboxes.stream().map(c -> createPrintableText(c.getAttribute("name"))).collect(Collectors.toList()));

            NodeList textAreaNodeList = root.getElementsByTagName("textarea");
            List<Element> textAreas = new ArrayList<>();
            for (int i = 0; i < textAreaNodeList.getLength(); ++i) {
                Element textArea = (Element) textAreaNodeList.item(i);
                textAreas.add(textArea);
            }
            csvHeaders.addAll(textAreas.stream().map(t -> createPrintableText(t.getAttribute("name"))).collect(Collectors.toList()));

            return csvHeaders;

        } catch (SAXParseException spe) {
            throw new RuntimeException("Error while extracting html components from template field-" + templateField, spe);
        }
    }

    @VisibleForTesting
    static class HtmlField implements Comparable<HtmlField> {
        String name;
        String value;

        public String getName() {
            return name;
        }

        public void setName(String name) {
            this.name = name;
        }

        public String getValue() {
            return value;
        }

        public void setValue(String value) {
            this.value = value;
        }

        @Override
        public int compareTo(HtmlField field) {
            return field.getName().compareTo(this.name);
        }

        @Override
        public String toString() {
            return name + " -> " + value;
        }

    }
