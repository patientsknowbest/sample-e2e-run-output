package com.pkb.plan.support;

import com.pkb.print.domain.FormattedCompositePrintableText;
import com.pkb.print.domain.LocalizedPrintableText;
import com.pkb.print.domain.PrintableText;
import org.apache.commons.csv.CSVFormat;
import org.apache.commons.csv.CSVPrinter;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.MessageSource;
import org.springframework.stereotype.Component;

import java.io.IOException;
import java.io.StringWriter;
import java.util.List;
import java.util.Locale;

@Component
public class PrintableTextCsvPrinter {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private MessageSource messageSource;

    @Autowired
    public PrintableTextCsvPrinter(MessageSource messageSource) {
        this.messageSource = messageSource;
    }

    public StringWriter print(List<List<PrintableText>> printableRowsColumns, Locale locale) throws IOException {
        StringWriter stringWriter = new StringWriter();
        CSVPrinter csvWriter = new CSVPrinter(stringWriter, CSVFormat.DEFAULT);
        printableRowsColumns.forEach(row-> printRecord(csvWriter, row, locale));
        return stringWriter;
    }

    private void printRecord(CSVPrinter csvWriter, List<PrintableText> printableTexts, Locale locale) {
        try {
            printableTexts.forEach(t -> resolvePrintableText(t, locale));
            csvWriter.printRecord(printableTexts);
        } catch (IOException e) {
            LOGGER.error("Error while writing into csv", e);
        }
    }

    private void resolvePrintableText(PrintableText printableText, Locale locale) {
        if(printableText instanceof LocalizedPrintableText){
            LocalizedPrintableText localizedPrintableText = (LocalizedPrintableText) printableText;
            localizedPrintableText.resolve(messageSource.getMessage(localizedPrintableText.getMessageKey(), new Object[]{}, locale));
        }else if(printableText instanceof FormattedCompositePrintableText){
            FormattedCompositePrintableText compositePrintableText = (FormattedCompositePrintableText)printableText;
            compositePrintableText.getFields().forEach(t -> resolvePrintableText(t, locale));
        }
    }
}
