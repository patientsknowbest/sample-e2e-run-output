package com.pkb.plan.support;

import com.pkb.entities.enums.TemplateStatus;
import com.pkb.phplan.entity.PHPlan;
import com.pkb.service.phplan.PHPlanDTO;
import com.pkb.service.phplan.PlanManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.LocalizedTextProvider;
import org.jetbrains.annotations.Nullable;

import java.util.function.BiFunction;

public class PhPlanToPlanDtoConverter {

    private final LocalizedTextProvider localizedTextProvider;
    private final PlanManager planManager;

    public PhPlanToPlanDtoConverter(LocalizedTextProvider localizedTextProvider, PlanManager planManager) {
        this.localizedTextProvider = localizedTextProvider;
        this.planManager = planManager;
    }

    @Nullable
    public PHPlanDTO convert(PHPlan source, PKBPerson loggedInUser, boolean restrictLastEditedToAdditionalVersions) {
        return convert(source, loggedInUser, restrictLastEditedToAdditionalVersions, (p, id) -> planManager.planIsDeletableBy(p, id));
    }

    @Nullable
    public PHPlanDTO convert(PHPlan source, PKBPerson loggedInUser, boolean restrictLastEditedToAdditionalVersions, BiFunction<PHPlan, Long, Boolean> planDeletableFlagProvider) {
        if (source == null) {
            return null;
        }
        Long loggedInUserId = loggedInUser.getId();
        PHPlanDTO result = new PHPlanDTO();
        result.setDeletableByCurrentUser(planDeletableFlagProvider.apply(source, loggedInUserId));
        result.setId(source.getId());
        result.setUniqueId(source.getBaseFields().getUniqueId());
        result.setName(source.getName());
        if (source.getCreatorId() == null) {
            // result.setCreatorName("(Uploaded by institute)"); what to put here?  team/org?
            result.setCreatorName(source.getSource().getOrgName());
        } else if (source.getCreatorId().equals(loggedInUserId)) {
            result.setCreatorName(localizedTextProvider.getText("managePlansAction.txt.You"));
        } else {
            PKBPerson creator = source.getCreator();
            result.setCreatorName(creator.getFullName());
        }
        result.setCreatorId(source.getCreatorId());
        result.setCreationDate(source.getCreationDate());
        result.setStatus(TemplateStatus.valueOf(source.getStatus().toString())); // Hack!
        result.setApprovalStatus(source.getApprovalStatus());

        if (source.getReviewDate() != null) {
            result.setReviewDate(source.getReviewDate());
        }

        if (!restrictLastEditedToAdditionalVersions || source.getVersion().compareTo(0L) != 0) {
            result.setLastEditedOn(source.getLastEditedOn());
            if (source.getLastEditorId() == null) {
                result.setEditorName(source.getSource().getOrgName());
            } else if (source.getLastEditorId().equals(loggedInUserId)) {
                result.setEditorName(localizedTextProvider.getText("managePlansAction.txt.you"));
            } else {
                PKBPerson editor = source.getLastEditor();
                result.setEditorName(editor.getFullName());
            }
        }
        source.getBaseFields().getPrivacyFlag().ifPresent(result::setPrivacyFlag);
        return result;
    }
}
