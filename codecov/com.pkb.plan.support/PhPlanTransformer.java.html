package com.pkb.plan.support;

import com.pkb.app.entity.SourceDetails;
import com.pkb.phplan.entity.PHPlan;
import com.pkb.source.support.SourceDetailsViewFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.stream.Collectors;

@Component
public class PhPlanTransformer {
    private SourceDetailsViewFactory sourceDetailsViewFactory;

    @Autowired
    public PhPlanTransformer(SourceDetailsViewFactory sourceDetailsViewFactory) {
        this.sourceDetailsViewFactory = sourceDetailsViewFactory;
    }

    public PHPlan transformPlan(PHPlan plan){
        if(plan != null){
            plan.setSource(sourceDetailsViewFactory.createSourceDetailsView(plan.getSource()));
        }
        return plan;
    }

    public PHPlan transformPlanView(PHPlan plan){
        if(plan != null){
            SourceDetails sourceDetails = new SourceDetails();
            sourceDetails.init(plan.getSource());
            plan.setSource(sourceDetails);
        }
        return plan;
    }


    public List<PHPlan> transformPlan(List<PHPlan> plans){
        return plans.stream().map(this::transformPlan).collect(Collectors.toList());
    }
}
