package com.pkb.fhir.common.transform;

import ca.uhn.fhir.context.FhirContext;
import ca.uhn.fhir.parser.IParser;
import com.pkb.common.ehrdata.serialization.EHRDataDeserializer;
import com.pkb.entities.enums.UserType;
import com.pkb.entities.pub.PublicPKBPerson;
import com.pkb.fhir.common.configuration.EhrmigratorConfig;
import com.pkb.fhir.common.decrypt.DecryptedEhrData;
import com.pkb.fhir.common.retrieve.AppointmentEhrDataDto;
import com.pkb.fhir.common.service.OrgService;
import com.pkb.fhir.common.service.PersonService;
import com.pkb.fhir.common.service.TeamService;
import com.pkb.fhir.migration.service.Hl7PartnerService;
import org.apache.commons.lang3.StringUtils;
import org.hl7.fhir.r4.model.Appointment;
import org.hl7.fhir.r4.model.CodeableConcept;
import org.hl7.fhir.r4.model.Coding;
import org.hl7.fhir.r4.model.DomainResource;
import org.hl7.fhir.r4.model.Reference;
import org.hl7.fhir.r4.model.ResourceType;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.invoke.MethodHandles;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.StringJoiner;
import java.util.UUID;

@SuppressWarnings({"ClassWithTooManyMethods"})
public class AppointmentEhrDataTransformer extends BaseEhrDataTransformer {

    private static final String COMMON_ID_SYSTEM_URL = "http://fhir.patientsknowbest.com/id/common-identifier";
    private static final String EXTERNAL_ID_SYSTEM_URL = "http://fhir.patientsknowbest.com/id/external-identifier";
    private static final String DID_NOT_ATTEND_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/did-not-attend";
    private static final String EXTERNAL_EMAIL_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/external-email";
    private static final String HOSPITAL_SERVICE_CODE_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/hospital-service-code";

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private FhirContext fhirContext;

    public AppointmentEhrDataTransformer(FhirContext fhirContext,
                                         PersonService personService, TeamService teamService, OrgService orgService, EhrmigratorConfig config,
                                         Hl7PartnerService hl7PartnerService, EHRDataDeserializer ehrDataDeserializer) {
        super(config, personService, teamService, orgService, hl7PartnerService, ehrDataDeserializer);
        this.fhirContext = fhirContext;
    }

    @Override
    @SuppressWarnings("unchecked")
    public String transformToFhirJson(@NotNull TransformableDto item) {
        IParser parser = fhirContext.newJsonParser();
        AppointmentEhrDataDto dto = Optional.ofNullable(item.getTransformationObjects().get("dto"))
                .map(object -> (AppointmentEhrDataDto) object)
                .orElseThrow(() -> new IllegalArgumentException("Missing dto from TransformableDto"));

        List<DecryptedEhrData> appointmentRequests = Optional.ofNullable(item.getTransformationObjects().get("appointmentRequests"))
                .map(object -> (List<DecryptedEhrData>) object)
                .orElse(new ArrayList<>());

        Appointment appointment = new Appointment();

        setId(dto.getUniqueId(), appointment);

        setIdentifier(String.valueOf(item.getId()), MENUDATA_LEGACY_PRIMARY_KEY_SYSTEM_URL, appointment);
        setIdentifier(dto.getCommonId(), COMMON_ID_SYSTEM_URL, appointment);
        setIdentifier(dto.getExternalId(), EXTERNAL_ID_SYSTEM_URL, appointment);

        addServiceType(dto.getTypeCode(), dto.getTypeCodingSystem(), appointment);

        getDateWithZone(dto.getMenuDataSourceInfo().getCreated())
                .map(this::mapToDateTimeTypeUTC)
                .ifPresent(appointment::setCreatedElement);

        getDateWithZone(dto.getStart())
                .map(this::mapToInstantTypeUTC)
                .ifPresent(appointment::setStartElement);

        getDateWithZone(dto.getEnd())
                .map(this::mapToInstantTypeUTC)
                .ifPresent(appointment::setEndElement);

        Optional.ofNullable(dto.getStatus())
                .map(status -> {
                    switch (status) {
                        case ACTIVE:
                            return Appointment.AppointmentStatus.BOOKED;
                        case CANCELLED:
                            return Appointment.AppointmentStatus.CANCELLED;
                        case DNA:
                            return Appointment.AppointmentStatus.NOSHOW;
                        default:
                            throw new IllegalArgumentException(String.format("Unknown status: %s for appointment id: %s", status, item.getId()));
                    }
                })
                .ifPresent(appointment::setStatus);

        createMeta(appointment, dto.getPrivacyFlagsDto(), dto.getMenuDataSourceInfo());

        Optional.ofNullable(dto.getPatientId())
                .map(id -> getParticipant(String.format("Patient/%s", id.toString()), appointment.getStatus()))
                .ifPresent(appointment::addParticipant);

        processEncryptedFields(item, appointment);

        //process the appointment requests which are each separate menu_data rows
        appointmentRequests.forEach(appointmentRequest -> {
            var appointmentRequestEhrDataMap = ehrDataDeserializer.deserialize(appointmentRequest.getContent().getBytes());
            populateAppointmentRequest(appointmentRequestEhrDataMap, appointment, dto.getPatientId())
                    .ifPresent(unknownFields -> {
                        throw new IllegalArgumentException(String.format("Unknown field %s in appointment request decrypted data", unknownFields));
                    });
        });
        return parser.encodeResourceToString(appointment);
    }

    private Appointment.AppointmentParticipantComponent getParticipant(String reference, Appointment.AppointmentStatus appointmentStatus) {
        Appointment.AppointmentParticipantComponent result = new Appointment.AppointmentParticipantComponent();
        result.setActor(new Reference(reference));
        result.setStatus(getParticipantStatus(appointmentStatus));
        return result;
    }

    private Appointment.AppointmentParticipantComponent getOnlyDisplayParticipant(String display, Appointment.AppointmentStatus appointmentStatus) {
        Appointment.AppointmentParticipantComponent result = new Appointment.AppointmentParticipantComponent();
        result.setActor(new Reference().setDisplay(display));
        result.setStatus(getParticipantStatus(appointmentStatus));
        return result;
    }

    private void setIdentifier(String id, String url, Appointment appointment) {
        Optional.ofNullable(id)
                .map(legacyId -> getIdentifier(legacyId, url))
                .ifPresent(appointment::addIdentifier);
    }

    private void setLocation(String location, Appointment appointment) {
        Appointment.AppointmentParticipantComponent locationActor = new Appointment.AppointmentParticipantComponent();
        locationActor.getActor().setType(ResourceType.Location.toString());
        locationActor.getActor().setDisplay(location);
        locationActor.setStatus(getParticipantStatus(appointment.getStatus()));
        appointment.addParticipant(locationActor);
    }

    private Appointment.ParticipationStatus getParticipantStatus(Appointment.AppointmentStatus appointmentStatus) {
        if (appointmentStatus == Appointment.AppointmentStatus.CANCELLED || appointmentStatus == Appointment.AppointmentStatus.NOSHOW) {
            return Appointment.ParticipationStatus.DECLINED;
        } else if (appointmentStatus == Appointment.AppointmentStatus.BOOKED) {
            return Appointment.ParticipationStatus.ACCEPTED;
        } else {
            throw new IllegalArgumentException(String.format("Unknown status: %s for appointment", appointmentStatus));
        }
    }

    private void setSpecialtyDisplay(String specialty, Appointment appointment) {
        getSpecialtyCoding(appointment)
                .ifPresentOrElse(specialtyExtension -> specialtyExtension.setDisplay(specialty), () -> {
                    Coding coding = new Coding().setDisplay(specialty);
                    setSpecialty(appointment, coding);
                });
    }

    private void setSpecialtyCode(String code, Appointment appointment) {
        getSpecialtyCoding(appointment)
                .ifPresentOrElse(specialtyExtension -> specialtyExtension.setCode(StringUtils.normalizeSpace(code)), () -> {
                    Coding coding = new Coding().setCode(StringUtils.normalizeSpace(code));
                    setSpecialty(appointment, coding);
                });
    }

    private void setSpecialty(Appointment appointment, Coding coding) {
        appointment.addSpecialty(new CodeableConcept(coding));
    }

    private Optional<Coding> getSpecialtyCoding(Appointment appointment) {
        return Optional.ofNullable(appointment.getSpecialty())
                .filter(specialties -> !specialties.isEmpty())
                .map(specialties -> {
                    if (specialties.size() > 1) {
                        LOGGER.error(String.format("There are more than 1 specialties in appointmentResource: %s", appointment.getId()));
                    }
                    return specialties.get(0);
                })
                .map(CodeableConcept::getCoding)
                .filter(codings -> !codings.isEmpty())
                .map(codings -> {
                    if (codings.size() > 1) {
                        LOGGER.error(String.format("There are more than 1 codings in specialty for appointment: %s", appointment.getId()));
                    }
                    return codings.get(0);
                });
    }

    @Override
    Optional<String> processDeserializedEhrDataMap(@NotNull String dataType, @Nullable Object data, @NotNull DomainResource resource) {
        Appointment appointment = (Appointment)resource;
        Optional<String> unknownDatatype = Optional.empty();
        switch (dataType) {
            case "location":
                Optional.ofNullable(data)
                        .ifPresent(location -> setLocation((String)location, appointment));
                break;
            case "specialty":
                break;
            case "updatedSpeciality":
                Optional.ofNullable(data)
                        .ifPresent(specialty -> setSpecialtyDisplay((String)specialty, appointment));
                break;
            case "subject":
                Optional.ofNullable(data)
                        .ifPresent(subject -> addSubject((String)subject, appointment));
                break;
            case "description":
                Optional.ofNullable(data)
                        .ifPresent(comment -> appointment.setComment((String)comment));
                break;
            case "patientId":
                break;
            case "nonPkbParticipants":
                processNonPkbParticipants(data, appointment);
                break;
            case "hospitalServiceCode":
                Optional.ofNullable(data)
                        .ifPresent(decoded -> setSpecialtyCode((String)decoded, appointment));
                break;
            case "sourceText":
                Optional.ofNullable(data)
                    .ifPresent(sourceText -> addExtension((String)sourceText, SOURCE_TEXT_EXTENSION_URL, resource));
                break;
            default:
                unknownDatatype = Optional.of(dataType);
        }
        return unknownDatatype;
    }

    /**
     * These are a bit complicated because they can be either a String or a List<String> on the database, because
     * we've changed our typing half way through.
     *
     * So do a quick instanceof on the object we've deserialized from menu_data.
     */
    @SuppressWarnings("unchecked")
    private void processNonPkbParticipants(@Nullable Object data, Appointment appointment) {
        Optional.ofNullable(data)
                .ifPresent(nonPkbParticipants -> {
                    if (nonPkbParticipants instanceof ArrayList) {
                        ((ArrayList<String>) nonPkbParticipants)
                                .forEach(participant -> appointment.addParticipant(getOnlyDisplayParticipant(participant, appointment.getStatus())));
                    }
                    else {
                        appointment.addParticipant(getOnlyDisplayParticipant((String)nonPkbParticipants, appointment.getStatus()));
                    }
                });
    }

    private Optional<List<String>> populateAppointmentRequest(Map<String, Object> appointmentRequestEhrDataMap, Appointment appointment, UUID patientId) {
        List<String> knownFields = List.of("participantId", "externalEmail", "hospitalServiceCode", "sourceText", "didNotAttend");
        Optional.ofNullable((String)appointmentRequestEhrDataMap.get("participantId"))
                .map(Long::parseLong)
                .flatMap(personService::findPersonOptional)
                .filter(pkbPerson -> !pkbPerson.getPublicId().equals(patientId))
                .flatMap(pkbPerson -> getParticipantForPerson(pkbPerson, appointment.getStatus()))
                .ifPresent(participant -> {
                    addExtensionForParticipant(appointmentRequestEhrDataMap, participant, EXTERNAL_EMAIL_EXTENSION_URL, "externalEmail");
                    addExtensionForParticipant(appointmentRequestEhrDataMap, participant, HOSPITAL_SERVICE_CODE_EXTENSION_URL, "hospitalServiceCode");
                    addExtensionForParticipant(appointmentRequestEhrDataMap, participant, SOURCE_TEXT_EXTENSION_URL, "sourceText");
                    addBooleanExtensionForParticipant(appointmentRequestEhrDataMap, participant, DID_NOT_ATTEND_EXTENSION_URL, "didNotAttend");
                    appointment.addParticipant(participant);
                });
        appointmentRequestEhrDataMap.keySet().removeAll(knownFields);
        return appointmentRequestEhrDataMap.isEmpty() ? Optional.empty() : Optional.of(new ArrayList<>(appointmentRequestEhrDataMap.keySet()));
    }

    private void addExtensionForParticipant(Map<String, Object> appointmentRequestFields,
                                            Appointment.AppointmentParticipantComponent participant, String extensionUrl, String fieldName) {
        Optional.ofNullable(appointmentRequestFields.get(fieldName))
                .flatMap(fieldValue -> createExtension((String)fieldValue, extensionUrl))
                .ifPresent(extension -> {
                    participant.addExtension(extension);
                    appointmentRequestFields.remove(fieldName);
                });
    }

    private void addBooleanExtensionForParticipant(Map<String, Object> appointmentRequestFields,
                                            Appointment.AppointmentParticipantComponent participant, String extensionUrl, String fieldName) {
        Optional.ofNullable(appointmentRequestFields.get(fieldName))
                .flatMap(fieldValue -> createExtension(Boolean.toString((Boolean)fieldValue), extensionUrl))
                .ifPresent(extension -> {
                    participant.addExtension(extension);
                    appointmentRequestFields.remove(fieldName);
                });
    }

    private Optional<Appointment.AppointmentParticipantComponent> getParticipantForPerson(PublicPKBPerson pkbPerson,
                                                                                          Appointment.AppointmentStatus appointmentStatus) {
        if (pkbPerson.getUserType() == UserType.PATIENT) {
            return Optional.of(getOnlyDisplayParticipant(getDisplayNameForPerson(pkbPerson), appointmentStatus));
        } else {
            return getReferenceString(pkbPerson.getUserType(), pkbPerson.getPublicId())
                    .map(reference -> getParticipant(reference, appointmentStatus));
        }
    }

    private String getDisplayNameForPerson(PublicPKBPerson person) {
        StringJoiner joiner = new StringJoiner(" ");
        Optional.ofNullable(person.getTitle())
                .filter(StringUtils::isNotBlank)
                .ifPresent(joiner::add);
        Optional.ofNullable(person.getFirstName())
                .filter(StringUtils::isNotBlank)
                .ifPresent(joiner::add);
        Optional.ofNullable(person.getLastName())
                .filter(StringUtils::isNotBlank)
                .ifPresent(joiner::add);
        return joiner.toString().trim();
    }

    private void addSubject(@NotNull String subject, Appointment appointment) {
        Coding coding = new Coding();
        coding.setDisplay(subject);
        appointment.setAppointmentType(new CodeableConcept(coding));
    }

    private void addServiceType(String typeCode, String typeCodeSystem, @NotNull Appointment appointment) {
        Coding coding = new Coding();
        Optional.ofNullable(typeCode)
                .map(StringUtils::normalizeSpace)
                .ifPresent(coding::setCode);
        Optional.ofNullable(typeCodeSystem)
                .ifPresent(coding::setSystem);
        appointment.setServiceType(List.of(new CodeableConcept(coding)));
    }
}
