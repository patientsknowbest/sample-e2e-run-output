package com.pkb.fhir.common.transform;

import com.pkb.entities.enums.MenuDataType;

import java.util.Map;

/**
 * A wrapper which represents a clinical data point BEFORE it has undergone transformation to FHIR, so content is
 * still in menu_data XML format. By this point that content is, however, decrypted.
 */
public class TransformableDto {

    //This is a proper grab-bag of data objects required in some way for the transformation of this data point into
    // a FHIR resource. So for appointments, for example, this contains both the AppointmentEhrDataDto, which is the
    // non-encrypted metadata for the appointment, and 0..* DecryptedEhrData DTOs for the appointment participants.
    // These are defined in the retriever for the data type in question
    private Map<String, Object> transformationObjects;
    private MenuDataType menuDataType;
    private Long id;

    //Decrypted XML menu_data content
    private String content;

    public TransformableDto(Map<String, Object> transformationObjects, MenuDataType menuDataType, Long id, String content) {
        this.transformationObjects = transformationObjects;
        this.menuDataType = menuDataType;
        this.id = id;
        this.content = content;
    }

    public MenuDataType getMenuDataType() {
        return menuDataType;
    }

    public Long getId() {
        return id;
    }

    public String getContent() {
        return content;
    }

    Map<String, Object> getTransformationObjects() {
        return transformationObjects;
    }
}
