package com.pkb.service.diary;

import com.pkb.app.entity.EHRData;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.EHRSearch;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.PKBFilter;
import com.pkb.app.entity.SourceDetails;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.diary.entity.DiaryEntry;
import com.pkb.document.entity.Attachment;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.service.document.AttachmentManager;
import com.pkb.service.file.ChunkedDocManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import org.jetbrains.annotations.NotNull;

import java.util.LinkedList;
import java.util.List;
import java.util.UUID;

public class DiaryManager extends TransactionManager {

    private final AttachmentManager attachmentManager;
    private final ChunkedDocManager chunkedDocManager;

    public DiaryManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService,
                        UUIDProvider uuidProvider, AttachmentManager attachmentManager, ChunkedDocManager chunkedDocManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.attachmentManager = attachmentManager;
        this.chunkedDocManager = chunkedDocManager;
    }

    public long saveDiaryEntry(LoggedInEHRRequestContext requestContext, long currentUserId, long accountId, DiaryEntry diaryEntry,
            List<Attachment> attachments) {
        long[] diaryEntryId = new long[1];
        transactional(() -> {
            try {
                diaryEntry.getBaseFields().generateNewRandomUniqueId();
                diaryEntry.setId(null); // update
                EHRData ehrData = beanFactory.getEhrRemote().populateEHRData(diaryEntry, accountId, MenuDataType.diaryEntry, requestContext);
                diaryEntryId[0] = beanFactory.getEhrRemote().saveEHRData(requestContext, ehrData);

                if (attachments != null) {
                    chunkedDocManager.createAttachments(diaryEntryId[0], attachments, currentUserId, accountId);
                }
            } catch (Exception exception) {
                throw new RuntimeException("Error while adding diary entry. Accessing user "
                        + requestContext.getMaybeAccessingUser() + ", accountId " + accountId, exception);
            }
        });
        return diaryEntryId[0];
    }

    public int deleteDiaryEntry(EHRRequestContext requestContext, long accountId, long diaryId,
                                @NotNull SourceDetails deletedBy) {
        EHRData ehrData = beanFactory.getEhrRemote().findEHRData(diaryId, requestContext);
        if (ehrData != null) {
            return deleteDiaryEntryByUniqueId(requestContext, accountId, ehrData.getUniqueId(), deletedBy);
        } else {
            return 0;
        }
    }

    public int deleteDiaryEntryByUniqueId(EHRRequestContext requestContext, long accountId, UUID diaryEntryUId,
                                          @NotNull SourceDetails deletedBy) {
        EHRSearch<DiaryEntry> search = new EHRSearch<>(accountId, DiaryEntry.class, MenuDataType.diaryEntry);
        search.addFilter(new PKBFilter(EHRData.UNIQUE_ID, PKBFilter.Operator.EQUAL, diaryEntryUId));
        return beanFactory.getEhrRemote().deleteEHRDataByUniqueId(requestContext, search, deletedBy, dateTimeService.now())._1;
    }

    public DiaryEntry getDiaryEntry(@NotNull LoggedInEHRRequestContext requestContext, long diaryEntryId) {
        return beanFactory.getEhrRemote().findAndDecryptTypedData(requestContext, diaryEntryId, DiaryEntry.class);
    }

    public List<DiaryEntry> getDiaryEntries(@NotNull LoggedInEHRRequestContext requestContext, long accountId, int offset,
                                            int pageSize) {
        return beanFactory.getEhrRemote().queryAndDecryptTypedData(
                (new EHRSearch<>(accountId, DiaryEntry.class, MenuDataType.diaryEntry))
                        .setResultsMaxReturned(pageSize)
                        .setLatestPerTypeFilter(EHRData.PERSISTED_DATE, EHRData.UNIQUE_ID)
                        .setResultsStartPosition(offset)
                        .setOrderBy(EHRData.ENTERED_DATE, EHRSearch.OrderByDirection.Desc),
                requestContext);
    }

    public long getDiaryEntryCount(EHRRequestContext requestContext, long accountId) {
        return beanFactory.getEhrRemote().queryEHRData(new EHRSearch<>(accountId, DiaryEntry.class, MenuDataType.diaryEntry), requestContext).size();
    }

    public List<Attachment> getDiaryAttachments(@NotNull LoggedInEHRRequestContext requestContext, long diaryEntryId) {
        try {
            List<Attachment> attachments = chunkedDocManager.findAttachmentsAsAttachment(diaryEntryId);
            List<Attachment> decryptedAttachments = new LinkedList<>();
            for (Attachment attachment : attachments) {
                Attachment decryptedAttachment = attachmentManager.getAttachmentContent(requestContext, attachment.getId());
                decryptedAttachment.setFilename(attachment.getFilename());
                decryptedAttachment.setDocMetadataId(attachment.getDocMetadataId());
                decryptedAttachments.add(decryptedAttachment);
            }

            return decryptedAttachments;
        } catch (Exception exception) {
            throw new RuntimeException("Error while retrieving attachments for DiaryEntry "
                    + diaryEntryId + " accessing user " + requestContext.getMaybeAccessingUser(), exception);
        }
    }

    public List<Attachment> getDiaryAttachmentsWithoutContents(long diaryEntryId) {
        try {
            return chunkedDocManager.findAttachmentsAsAttachment(diaryEntryId);
        } catch (Exception exception) {
            throw new RuntimeException("Error while retrieving attachments for DiaryEntry "
                    + diaryEntryId, exception);
        }
    }

    public List<DiaryEntry> getEntryListByIds(@NotNull LoggedInEHRRequestContext requestContext, List<Long> entryIdList, long accountId) {
        return beanFactory.getEhrRemote().queryAndDecryptTypedData(
                (new EHRSearch<>(accountId, DiaryEntry.class, MenuDataType.diaryEntry))
                        .addFilter(new PKBFilter("id", PKBFilter.Operator.IN, entryIdList)),
                requestContext);
    }
}
