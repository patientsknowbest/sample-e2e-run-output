package com.pkb.calendar.entity;

import com.pkb.annotation.EHRField;
import com.pkb.annotation.EHRField.QueryField;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.user.entity.PKBPerson;

import java.io.Serializable;

/**
 * A request to participate in a calendar appointment
 *
 * @author robwhelan
 * @see AppointmentDTO
 */
public class AppointmentRequestDTO implements IBaseDTO, Serializable {
    private static final long serialVersionUID = 1L;

    public static final MenuDataType MS_PATH = MenuDataType.appointmentRequest;

    // non-encrypted
    public final static String ID = "id";
    public final static String APPOINTMENT_ID = "appointmentId";
    public final static String COMMON_ID = "commonId";
    public final static String ACCEPTANCE_STATUS = "acceptanceStatus";
    // encrypted
    public final static String PARTICIPANT_ID = "participantId";
    public final static String EXTERNAL_EMAIL = "externalEmail";

    @EHRField(queryField = QueryField.id)
    private Long id;

    /**
     * the parent appointment, in the same account as this AppointmentRequest (there's a copy in each acct)
     */
    @EHRField(queryField = QueryField.long01)
    private Long appointmentId;

    /**
     * long random id for secure email links, and updating all versions of a request when it's accepted
     */
    @EHRField(queryField = QueryField.string01)
    private String commonId;

    /**
     * Default to false; set to true only upon receipt of an SIU S26.
     */
    @EHRField
    private boolean didNotAttend = false;

    /**
     * stored in db
     */
    @EHRField
    private String participantId;
    /**
     * used for display
     */
    private PKBPerson participant;

    /**
     * if the invited participant is not a PKB person
     */
    @EHRField
    private String externalEmail;

    @EHRField(valueDtoType = EHRField.ValueDTOType.FLATTENED)
    private SourceDetails source;

    private DTOBaseFields baseFields = new DTOBaseFields();

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected AppointmentRequestDTO() {
        this.source = new SourceDetails();
    }

    public AppointmentRequestDTO(SourceDetails sourceDetails) {
        this.source = sourceDetails;
    }

    @Override
    public Long getId() {
        return id;
    }

    @Override
    public void setId(Long id) {
        this.id = id;
    }

    public String getParticipantId() {
        return participantId;
    }

    public void setParticipantId(String participantId) {
        this.participantId = participantId;
    }

    public boolean getDidNotAttend() {
        return didNotAttend;
    }

    public void setDidNotAttend(Boolean didNotAttend) {
        // TODO: Perhaps write a Migrator to turn nulls into falses?
        if (didNotAttend == null) {
            this.didNotAttend = false;
        } else {
            this.didNotAttend = didNotAttend;
        }
    }

    public String getExternalEmail() {
        return externalEmail;
    }

    public void setExternalEmail(String externalEmail) {
        this.externalEmail = externalEmail;
    }

    public Long getAppointmentId() {
        return appointmentId;
    }

    public void setAppointmentId(Long appointmentId) {
        this.appointmentId = appointmentId;
    }

    public PKBPerson getParticipant() {
        return participant;
    }

    public void setParticipant(PKBPerson participant) {
        this.participant = participant;
    }

    public String getCommonId() {
        return commonId;
    }

    public void setCommonId(String commonId) {
        this.commonId = commonId;
    }

    @Override
    public SourceDetails getSource() {
        return source;
    }

    @Override
    public DTOBaseFields getBaseFields() {
        return baseFields;
    }
}
