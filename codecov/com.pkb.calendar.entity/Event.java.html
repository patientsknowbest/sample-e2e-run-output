/**

 *
 * @author pravinam
 * <p>
 * <p>
 * Data type migration:
 * From list:
 * <column name="startDate" internal="date01" format="yyyy-MM-dd HH:mm:ss.SSS">
 * <from>#{event.startDate}</from>
 * </column>
 * <column name="endDate" internal="date02" format="yyyy-MM-dd HH:mm:ss.SSS">
 * <from>#{event.endDate}</from>
 * </column>
 * <column name="status" internal="string01">
 * <from>#{event.status}</from>
 * </column>
 * <column name="eventType" internal="long01">
 * <from>#{event.eventType}</from>
 * </column>
 * <column name="description">
 * <from>#{event.description}</from>
 * </column>
 * <p>
 * Placeholder:

 * <field name="status" internal="string01">

 * <field name="eventType" internal="long01">

 * </field>

public class Event implements IBaseDTO, Serializable {
    private static final long serialVersionUID = 1L;

    public static final MenuDataType MS_PATH = MenuDataType.event;

    // List of placeholder field names
    public static final String PH_EVENT_START_DATE = "startDate";
    public static final String PH_EVENT_END_DATE = "endDate";
    public static final String PH_EVENT_STATUS = "status";
    //    public static final String PH_EVENT_TYPE = "eventType";
    // Description was added for keeping atleast 1 encrypted field
    // As there was a problem retrieving the placeholder
    //    public static final String PH_EVENT_DESCRIPTION = "description";

    // Enumeration for event types NOT YET USED... for now they're all OoO.
    public enum EventType {
        OUT_OF_OFFICE
    }

    // This will point the corresponding menudata id
    @EHRField(queryField = QueryField.id)
    private Long id;

    @EHRField(queryField = QueryField.date01)
    private Date startDate;

    @EHRField(queryField = QueryField.date02)
    private Date endDate;

    @EHRField(queryField = QueryField.string01)
    private String status;

    // this will fail unless migrated... but we don't seem to use it anyway (all values are OUT_OF_OFFICE)
    //    @EHRField( queryField=QueryField.long01 )
    //    private EventType eventType;

    private SourceDetails source;
    private DTOBaseFields baseFields = new DTOBaseFields();

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected Event() {
        this.source = new SourceDetails();
    }

    public Event(SourceDetails sourceDetails) {
        this.source = sourceDetails;
    }

    @Override
    public Long getId() {
        return id;
    }

    @Override
    public void setId(Long id) {
        this.id = id;
    }

    public Date getStartDate() {
        return startDate;
    }

    public void setStartDate(Date startDate) {
        this.startDate = startDate;
    }

    public Date getEndDate() {
        return endDate;
    }

    public void setEndDate(Date endDate) {
        this.endDate = endDate;
    }

    public String getStatus() {
        return status;
    }

    public void setStatus(String status) {
        this.status = status;
    }

    @Override
    public SourceDetails getSource() {
        return source;
    }

    @Override
    public DTOBaseFields getBaseFields() {
        return baseFields;
    }
