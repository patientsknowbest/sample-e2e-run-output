package com.pkb.util;

import java.util.Optional;
import java.util.stream.Stream;

public enum CustomLoginSkinTypes {

    TEAM_CIE("cie"),
    TEAM_ESHT_PRT("esht_prt"),
    TEAM_UHDB_PR("uhdb_pr"),
    SHERWOOD_FOREST_HOSPITAL("sfhft_sher"),
    DARTFORD("dartfordgravesham"),
    SUSSEX("mhacr-sussex"),
    CHESHIRE_MCHFT_PR("mchft_pr"),
    CONNECTED_NOTTS("connected-notts"),
    HULL("hull_prt"),
    MID_YORKSHIRE_HOSPITALS("mid-yorkshire-hosp"),
    YORK_TEACHING_HOSP("york-teachinghosp");

    private final String teamName;
    private final String actionName;

    CustomLoginSkinTypes(String teamName) {
        this.teamName = teamName;
        this.actionName = "teamSkinSuccess";
    }

    public static Optional<String> findActionFor(String teamName) {
        return findCustomSkinTypeFor(teamName)
                .map(CustomLoginSkinTypes::getActionName);
    }

    public static Optional<CustomLoginSkinTypes> findCustomSkinTypeFor(String teamName) {
        return Stream.of(CustomLoginSkinTypes.values())
                .filter(skinType -> skinType.teamName.equalsIgnoreCase(teamName))
                .findFirst();
    }

    public String getTeamName() {
        return teamName;
    }

    public String getActionName() {
        return actionName;
    }

}
