package com.pkb.util;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;
import com.pkb.entities.enums.Route;
import com.pkb.test.entity.TestFormDTO;
import com.pkb.test.entity.TestResultDTO;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.text.DecimalFormat;
import java.time.Instant;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;
import java.util.Date;
import java.util.Optional;
import java.util.function.Function;

import static com.pkb.util.PKBConstants.DECIMAL_FORMAT;
import static org.apache.commons.lang3.StringUtils.EMPTY;

public class TestResultDTOMappers {

    static TestFormDTO.TestResultCommentDTO getTestResultCommentDTO(TestResultDTO result) {
        TestFormDTO.TestResultCommentDTO dto = new TestFormDTO.TestResultCommentDTO();
        dto.setCommented(result.hasComments());
        dto.setComments(result.getCommentsAsHtml());
        return dto;
    }

    static TestFormDTO.TestResultDateDTO getTestResultDateDTO(@Nullable Instant date, DateTimeFormatter formatter) {
        TestFormDTO.TestResultDateDTO dto = new TestFormDTO.TestResultDateDTO();
        if (date != null) {
            dto.setValue(date.toEpochMilli());
            dto.setFormatted(formatter.format(date));
        }
        return dto;
    }

    static TestFormDTO.TestResultDateDTO getTestResultDateDTO(ZonedDateTime zonedDateTime, DateTimeFormatter formatter) {
        TestFormDTO.TestResultDateDTO dto = new TestFormDTO.TestResultDateDTO();
        if (zonedDateTime != null) {
            dto.setValue(Date.from(zonedDateTime.toInstant()).getTime());
            dto.setFormatted(formatter.format(zonedDateTime));
        }
        return dto;
    }

    static TestFormDTO.TestResultPrivacyFlagsDTO getTestResultPrivacyFlagsDTO(TestResultDTO result) {
        DTOBaseFields baseFields = result.getBaseFields();
        TestFormDTO.TestResultPrivacyFlagsDTO dto = new TestFormDTO.TestResultPrivacyFlagsDTO();
        dto.setGeneralHealth(baseFields.isGeneralHealthConsentRequired());
        dto.setMentalHealth(baseFields.isMentalHealthConsentRequired());
        dto.setSexualHealth(baseFields.isSexualHealthConsentRequired());
        dto.setSocialCare(baseFields.isSocialCareConsentRequired());
        return dto;
    }

    @VisibleForTesting
    static TestFormDTO.TestResultRangeDTO getTestResultRangeDTO(TestResultDTO result, boolean delayedDisplay, String unit) {
        return getTestResultRangeDTO(result, delayedDisplay, unit, DECIMAL_FORMAT);
    }

    static TestFormDTO.TestResultRangeDTO getTestResultRangeDTO(TestResultDTO result, boolean delayedDisplay, String unit, DecimalFormat decimalFormat) {
        TestFormDTO.TestResultRangeDTO dto = new TestFormDTO.TestResultRangeDTO();
        dto.setRanged(result.isRanged());
        dto.setRangeProvided(result.isRangeProvided());
        dto.setTextRange(result.getTextRange());
        if (result.isNumericRangeProvided() && !delayedDisplay) {
            dto.setHigh(result.getRangeHigh());
            dto.setLow(result.getRangeLow());
            dto.setSign(result.getRangeSign());

            if (result.getTestDate() != null) {
                dto.setTimestamp(result.getTestDate().toEpochMilli());
            }

            dto.setDisplay(EMPTY);

            if (StringUtils.isNotBlank(dto.getSign())) {
                if (dto.getLow() != null && dto.getHigh() != null) {
                    dto.setDisplay(decimalFormat.format(dto.getLow()) + dto.getSign() + decimalFormat.format(dto.getHigh()));
                } else if (dto.getLow() != null) {
                    dto.setDisplay(dto.getSign() + " " + decimalFormat.format(dto.getLow()));
                } else {
                    dto.setDisplay(dto.getSign() + " " + decimalFormat.format(dto.getHigh()));
                }
            }

            dto.setDisplay(dto.getDisplay() + " " + unit);
        }

        return dto;
    }

    public static TestFormDTO.TestResultSourceDTO getTestResultSourceDTO(SourceDetails sourceDetails, Function<Route, String> routePrinter) {
        TestFormDTO.TestResultSourceDTO dto = new TestFormDTO.TestResultSourceDTO();
        dto.setDisplayText(sourceDetails.getDisplayTextFull());
        dto.setPersonId(sourceDetails.getPersonId());
        dto.setVia(routePrinter.apply(sourceDetails.getRoute()));
        dto.setSourceType(Optional.ofNullable(sourceDetails.getSourceType()).map(Enum::name).orElse(EMPTY));
        return dto;
    }

    @VisibleForTesting
    static TestFormDTO.TestResultValueDTO getTestResultValueDTO(TestResultDTO result, boolean delayedDisplay, String delayedDisplayText, String unit) {
        return getTestResultValueDTO(result, delayedDisplay, delayedDisplayText, unit, DECIMAL_FORMAT);
    }

    static TestFormDTO.TestResultValueDTO getTestResultValueDTO(TestResultDTO result, boolean delayedDisplay, String delayedDisplayText, String unit, DecimalFormat decimalFormat) {
        TestFormDTO.TestResultValueDTO dto = new TestFormDTO.TestResultValueDTO();

        if (delayedDisplay) {
            dto.setDisplay(delayedDisplayText);
            return dto;
        } else if (result.isValueNumeric()) {
            TestResultDTO.ComparatorOperator comparator = result.getValueComparator();
            dto.setRawNumericValue(getRawValue(result.getValue()));
            if (comparator != null) {
                dto.setComparator(comparator.toString());
                dto.setDisplay(dto.getComparator() + " " + decimalFormat.format(result.getValue()));
            } else {
                dto.setDisplay(decimalFormat.format(result.getValue()));
            }
        } else {
            dto.setDisplay(result.getTextValueAsHTML());
        }

        if (StringUtils.isNotBlank(dto.getDisplay())) {
            dto.setDisplay(dto.getDisplay() + " " + unit);
        }

        return dto;
    }

    /**
     * Convert a Double to an Integer if it is an integer. This will prevent the string representation of the value
     * having ".0" appended to integers in the web interface.
     *
     * @param rawValue The value to check.
     * @return The Double value as provided, or else an Integer representation if the input double was an integer.
     */
    private static Number getRawValue(@NotNull Double rawValue) {
        if (rawValue % 1 == 0) {
            return rawValue.intValue();
        }
        return rawValue;
    }
}
