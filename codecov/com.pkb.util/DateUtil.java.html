package com.pkb.util;

import com.google.common.annotations.VisibleForTesting;
import com.google.common.base.Function;
import com.pkb.common.datetime.DateTimeService;
import org.jetbrains.annotations.Nullable;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.context.MessageSource;

import javax.xml.datatype.DatatypeConfigurationException;
import javax.xml.datatype.DatatypeFactory;
import java.lang.invoke.MethodHandles;
import java.time.Instant;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeFormatterBuilder;
import java.time.temporal.ChronoField;
import java.time.temporal.ChronoUnit;
import java.time.temporal.TemporalAccessor;
import java.util.Date;
import java.util.Locale;
import java.util.Optional;
import java.util.function.BiFunction;

import static com.google.common.base.Preconditions.checkNotNull;
import static com.pkb.util.Constants.APPLICATION_TZ;
import static org.apache.commons.lang3.StringUtils.isBlank;

public class DateUtil {

    private static final DateTimeFormatter ISO_FORMATTER = DateTimeFormatter.ISO_OFFSET_DATE_TIME.withLocale(Locale.UK).withZone(APPLICATION_TZ);

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private final MessageSource messageSource;
    private final DateTimeService dateTimeService;

    public DateUtil(MessageSource messageSource, DateTimeService dateTimeService) {
        this.messageSource = messageSource;
        this.dateTimeService = dateTimeService;
    }

    private DateTimeFormatter getFormatter(String pattern, Locale locale, @Nullable ZoneId timeZone) {
        ZoneId actualTz = getActualTz(timeZone);
        return DateTimeFormatter.ofPattern(pattern).withLocale(locale).withZone(actualTz);
    }

    private DateTimeFormatter getBasicDateDisplay(Locale locale, @Nullable ZoneId timeZone) {
        return getFormatter("d MMM yyyy", locale, timeZone);
    }

    private DateTimeFormatter getBasicMonthDisplay(Locale locale, @Nullable ZoneId timeZone) {
        return getFormatter("MMM yyyy", locale, timeZone);
    }

    String getFormattedDateTime(Instant currentDate, Instant createdDate, Locale locale, @Nullable ZoneId timeZone) {
        DateTimeFormatter formatter = getFormatter("HH:mm:ss", locale, timeZone);
        return getFormattedDate(currentDate, createdDate, locale, timeZone) + (createdDate == null ? "" : " " + formatter.format(createdDate));
    }

    @Nullable
    public String getFormattedDate(Instant currentDate, Instant createdDate, Locale locale, @Nullable ZoneId timeZone) {
        return getFormattedDate(currentDate, createdDate, locale, timeZone, getBasicDateDisplayFunction(locale, timeZone));
    }

    @Nullable
    private String getFormattedDate(Instant currentDate, Instant createdDate, Locale locale, @Nullable ZoneId timeZone, Function<TemporalAccessor, String> standardDisplay) {
        if (createdDate == null) {
            return messageSource.getMessage("global.txt.unavailable", null, locale);
        }
        ZoneId actualTimezone = getActualTz(timeZone);

        if (isToday(createdDate, actualTimezone)) {
            return messageSource.getMessage("global.txt.today", null, locale);
        }
        return standardDisplay.apply(createdDate);
    }

    private boolean isToday(Instant aDate, ZoneId timeZone) {
        ZonedDateTime now = ZonedDateTime.ofInstant(dateTimeService.now(), timeZone);
        ZonedDateTime zonedDate = ZonedDateTime.ofInstant(aDate, timeZone);
        return now.getDayOfMonth() == zonedDate.getDayOfMonth()
                && now.getMonth() == zonedDate.getMonth()
                && now.getYear() == zonedDate.getYear();
    }

    private Function<TemporalAccessor, String> getBasicDateDisplayFunction(Locale locale, @Nullable ZoneId timeZone) {
        return ta -> this.getBasicDateDisplay(locale, timeZone).format(ta);
    }

    private Function<TemporalAccessor, String> getFormattedDisplay(String pattern, Locale locale, @Nullable ZoneId timeZone) {
        return ta -> getFormatter(pattern, locale, timeZone).format(ta);
    }

    public ZonedDateTime getStartOfDay(ZonedDateTime currentTimestamp) {
        return currentTimestamp.truncatedTo(ChronoUnit.DAYS).withEarlierOffsetAtOverlap();
    }

    String getPKBFormattedMonth(Date date, Locale locale, @Nullable ZoneId timeZone) {
        return getBasicMonthDisplay(locale, timeZone).format(date.toInstant());
    }

    /*
     *  Newer date formatter implementing 'Dates Language'
     *  from https://sites.google.com/a/patientsknowbest.com/specs/Home/patient-education/dates-language
     */
    public String getPKBFormattedDate(Instant date, Locale locale, @Nullable ZoneId timeZone) {
        return getPKBFormattedDate(date, locale, null, timeZone);
    }

    String getPKBFormattedDate(Instant date, Locale locale, String format, @Nullable ZoneId timeZone) {
        checkNotNull(locale, "locale"); // Otherwise DateTimeFormatter will fail
        Function<TemporalAccessor, String> stanardDisplay = format == null ? getBasicDateDisplayFunction(locale, timeZone) : getFormattedDisplay(format, locale, timeZone);
        return getFormattedDate(dateTimeService.now(), date, locale, timeZone, stanardDisplay);
    }

    public Optional<Instant> parseTimestampIso8601(String timestamp) {
        if (isBlank(timestamp)) {
            return Optional.empty();
        }

        try {
            return Optional.of(DatatypeFactory.newInstance().newXMLGregorianCalendar(timestamp)
                    .toGregorianCalendar().toInstant());
        } catch (DatatypeConfigurationException ignored) {
            LOGGER.info("error parsing ISO timestamp (returning null): {}", timestamp);
        }
        return Optional.empty(); // parse failed
    }

    @Nullable
    public String formatTimestampIso8601(Date timestamp) {
        if (timestamp == null) {
            return null;
        }
        return formatTimestampIso8601(timestamp.toInstant());
    }

    @Nullable
    public String formatTimestampIso8601(Instant timestamp) {
        if (timestamp == null) {
            return null;
        }
        return ISO_FORMATTER.format(timestamp);
    }

    private ZoneId getActualTz(@Nullable ZoneId timeZone) {
        return timeZone == null ? APPLICATION_TZ : timeZone;
    }

    public String formatDateToGlobalOutputFormat(Instant input, Locale locale) {
        return getGlobalDateOutputFormatter(locale).format(input);
    }

    public String formatDateToGlobalOutputFormat(LocalDate input, Locale locale) {
        return getGlobalDateOutputFormatter(locale).format(input);
    }

    private DateTimeFormatter getGlobalDateOutputFormatter(Locale locale) {
        return DateTimeFormatter.ofPattern(messageSource.getMessage("global.format.date_output", null, locale)).withLocale(locale).withZone(APPLICATION_TZ);
    }

    public DateTimeFormatter getGlobalDateInputFormatter(boolean lenient, ZoneId zone, Locale locale) {
        return getGlobalDateInputFormatter(lenient, zone, (s, l) -> messageSource.getMessage(s, null, l), locale);
    }

    public DateTimeFormatter getGlobalDateInputFormatter(boolean lenient, ZoneId zone, BiFunction<String, Locale, String> patternSupplier, Locale locale) {
        return getGlobalDateInputFormatter(getGlobalDateInputPattern(patternSupplier, locale), lenient, zone);
    }

    public String getGlobalDateInputPattern(BiFunction<String, Locale, String> patternSupplier, Locale locale) {
        return patternSupplier.apply("global.format.date_input", locale);
    }

    private DateTimeFormatter getGlobalDateInputFormatter(String pattern, boolean lenient, ZoneId zone) {
        DateTimeFormatterBuilder builder = new DateTimeFormatterBuilder()
                .parseCaseInsensitive()
                .appendPattern(pattern);
        if (lenient) {
            builder.parseLenient()
                    .parseDefaulting(ChronoField.HOUR_OF_DAY, 0)
                    .parseDefaulting(ChronoField.MINUTE_OF_HOUR, 0)
                    .parseDefaulting(ChronoField.SECOND_OF_MINUTE, 0)
                    .parseDefaulting(ChronoField.NANO_OF_SECOND, 0);
        }
        return builder.toFormatter().withZone(zone);
    }

    public DateTimeFormatter getDateWithHHmmFormatter(ZoneId zoneId, Locale locale) {
        return getDateTimeFormatter(zoneId, locale, "format.dateInputWithHHmm");
    }

    public DateTimeFormatter getDateWithHHmmssFormatter(ZoneId zoneId, Locale locale) {
        return getDateTimeFormatter(zoneId, locale, "format.dateInputWithHHmmss");
    }

    public DateTimeFormatter getDateTimeFormatter(ZoneId zoneId, Locale locale, String format) {
        DateTimeFormatterBuilder builder = new DateTimeFormatterBuilder()
                .parseCaseInsensitive()
                .appendPattern(messageSource.getMessage(format, null, locale));
        return builder.toFormatter().withZone(zoneId);
    }

    public DateTimeFormatter getDateAtTimeOutputFormatter(BiFunction<String, Locale, String> patternSupplier, Locale locale) {
        return DateTimeFormatter.ofPattern(patternSupplier.apply("global.format.date_full_at_time", locale)).withLocale(locale).withZone(APPLICATION_TZ);
    }

    public String nhsDateTime(LocalDateTime dateTime, Locale locale, String englishPattern, String standardPattern) {
        if (locale.equals(Locale.ENGLISH)) {
            DateTimeFormatter timeFormatter = DateTimeFormatter.ofPattern(englishPattern)
                    .withLocale(Locale.UK);
            return nhsFormattedTimestamp(dateTime.format(timeFormatter));

        } else {
            return dateTime.format(DateTimeFormatter.ofPattern(standardPattern).withLocale(locale));
        }
    }

    public String nhsFormattedTimestamp(String timestamp) {
        return timestamp.replace("12.00am", "midnight")
                .replace("12.00pm", "midday");
    }

    @VisibleForTesting
    DateTimeService getDateTimeService() {
        return dateTimeService;
    }
}
