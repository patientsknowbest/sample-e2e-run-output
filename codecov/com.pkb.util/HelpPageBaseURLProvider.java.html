package com.pkb.util;

import com.pkb.common.ClearableInternalState;
import com.pkb.common.config.PhrConfig;

/**
 * PHR-4222
 * This class is responsible for working out the help base base URL
 * Right now it just looks at a config value, but in future it could depend on the user's language selection,
 * locale, or organisation.
 */
public class HelpPageBaseURLProvider implements ClearableInternalState {
    private PhrConfig config;

    public HelpPageBaseURLProvider(PhrConfig config) {
        this.config = config;
    }

    public String getHelpBaseURL() {
        return config.getHelpTextURL();
    }

    boolean hideHelpIcons() {
        return getHelpBaseURL().isEmpty();
    }

    public void setOverrideHelpPageBaseUrl(String overrideHelpPageBaseUrl) {
        throw new UnsupportedOperationException("Help page base url cannot be changed.");
    }

    @Override
    public void clearState() {
        setOverrideHelpPageBaseUrl(null);
    }
}
