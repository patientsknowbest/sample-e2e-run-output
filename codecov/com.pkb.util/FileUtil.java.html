package com.pkb.util;

import java.util.Set;

import org.apache.commons.io.FilenameUtils;
import org.apache.commons.lang3.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import com.google.common.collect.Sets;

public class FileUtil {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    public static final Set<String> UPLOAD_EXTENSION_BLACKLIST = Sets.newHashSet("386", "7z", "ade", "adp", "app", "asp", "bas", "bat", "cab", "cer", "cgi", "chm", "class", "cmd",
            "com", "cpl", "crt", "csh", "csr", "dll", "dmg", "drv", "exe", "fxp", "hlp", "hta", "inf", "ins", "isp", "jar", "js", "jse", "jsp", "ksh", "lib", "lnk", "mdb", "mde",
            "mdt", "mdw", "msc", "msi", "msp", "mst", "nsh", "ocx", "ops", "pcd", "php", "pif", "pl", "prg", "rar", "reg", "scr", "sct", "sh", "shb", "shs", "sys", "torrent",
            "url", "vb", "vbe", "vbs", "vbscript", "vxd", "wsc", "wsf", "wsh");


    public static boolean isFileExtensionBlacklistedForUpload(String filename) {
        String lowercaseExt = StringUtils.lowerCase(FilenameUtils.getExtension(filename));
        return UPLOAD_EXTENSION_BLACKLIST.contains(lowercaseExt);
    }

    public static boolean isPdf(String filename, String mediaType) {
        String fileName = StringUtils.trimToEmpty(filename).toLowerCase();
        // allowing mediaType == null for backwards compatibility :(
        return ("application/pdf".equals(mediaType) || "application/x-pdf".equals(mediaType) || mediaType == null) && fileName.endsWith(".pdf");
    }

    public static boolean isConvertibleToPdf(String filename, String mediaType) {
        String fileName = StringUtils.trimToEmpty(filename).toLowerCase();
        return "application/vnd.openxmlformats-officedocument.wordprocessingml.document".equals(mediaType) || fileName.endsWith(".docx");
    }

    public static boolean isViewable(String filename) {
        String fileName = StringUtils.trimToEmpty(filename).toLowerCase();
        return fileName.endsWith(".gif") || fileName.endsWith(".png") ||
                fileName.endsWith(".jpg") || fileName.endsWith(".jpeg");
    }

    public static boolean isStreamable(String filename) {
        String fileName = StringUtils.trimToEmpty(filename).toLowerCase();
        return fileName.endsWith(".mp4") || fileName.endsWith(".mov") ||
                fileName.endsWith(".webm") || fileName.endsWith(".swf") ||
                fileName.endsWith(".ogv") || fileName.endsWith(".3gp") ||
                fileName.endsWith(".m2v");
    }

    public static String getVideoType(String filename) {
        String fileName = StringUtils.trimToEmpty(filename).toLowerCase();

        if (fileName.endsWith(".mp4") || fileName.endsWith(".mov")) {
            return "video/mp4";
        } else if (fileName.endsWith(".webm")) {
            return "video/webm";
        } else if (fileName.endsWith(".ogv")) {
            return "video/ogg";
        } else if (fileName.endsWith(".3gp")) {
            return "video/3gpp";
        } else if (filename.endsWith(".m2v")) {
            return "video/m2v";
        }

        return "unknown";
    }

}
