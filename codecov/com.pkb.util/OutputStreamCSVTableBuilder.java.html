package com.pkb.util;

import java.io.IOException;
import java.io.UncheckedIOException;
import java.io.Writer;
import java.util.ArrayList;
import java.util.List;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

import static java.util.Collections.emptyList;

public class OutputStreamCSVTableBuilder implements CSVTableBuilder {

    private static final Pattern DOUBLE_QUOTE_PATTERN = Pattern.compile("\"", Pattern.LITERAL);
    protected List<String> matchers;
    protected List<String> titles;
    protected List<String> row; // Row that we are building
    protected int position = 0; // Position in that row
    private boolean headerPrinted = false;//used to turn off header generation, .e.g when streaming large CSVs

    private final Writer writer;

    public OutputStreamCSVTableBuilder(Writer output) {
        this(output, emptyList());
    }

    public OutputStreamCSVTableBuilder(Writer output, List<String> titles) {
        this.titles = new ArrayList<>(titles);
        this.matchers = new ArrayList<>(titles);
        writer = output;
        row = new ArrayList<>(titles.size());
        initNextRow();
    }

    private void printHeader() {
        try {
            boolean doneFirstCol = false;
            for (String title : titles) {
                if (doneFirstCol) {
                    writer.write(",");
                } else {
                    doneFirstCol = true;
                }
                writer.write("\"" + escape(title) + "\"");
            }
            writer.write("\n");
        } catch (IOException e) {
            throw new UncheckedIOException(e);
        }
    }

    @Override public void addCell(String title, String value) {
        addCell(title, title, value);
    }

    @Override
    public void addCell(String matcher, String title, String value) {
        // Usually the next cell will go in the next position
        for (; position < matchers.size(); ++position) {
            if (matchers.get(position).equals(matcher)) {
                row.set(position, value);
                return;
            }
        }

        // In case the order has changed
        for (int p = 0; p < position; ++p) {
            if (matchers.get(p).equals(matcher)) {
                row.set(p, value);
                return;
            }
        }

        matchers.add(matcher);
        titles.add(title);
        row.add(value);
    }

    @Override public void endRow() {
        if (!headerPrinted) {
            printHeader();
            headerPrinted = true;
        }
        printCurrentRow();
        row.clear();
        position = 0;
        initNextRow();
    }

    private void printCurrentRow() {
        try {
            boolean doneFirstCol = false;
            for (String cell : row) {
                if (doneFirstCol) {
                    writer.write(",");
                } else {
                    doneFirstCol = true;
                }

                if (cell != null) {
                    writer.write("\"");
                    writer.write(escape(cell));
                    writer.write("\"");
                }
            }
            writer.write("\n");
        } catch (IOException e) {
            throw new UncheckedIOException(e);
        }
    }

    private void initNextRow() {
        for (int i = 0; i < matchers.size(); ++i) {
            row.add(null);
        }
    }

    /**
     * Escapes an arbitary string to be a proper CSV value. according to
     * <a href="https://tools.ietf.org/html/rfc4180">RFC 4180</a>
     * it means replacing double-quotes with double double-quotes ({@code " -> ""}).
     *
     * @param raw the unescaped string
     * @return the escaped string
     */
    private static String escape(String raw) {
        return DOUBLE_QUOTE_PATTERN.matcher(raw).replaceAll(Matcher.quoteReplacement("\"\""));
    }


}
