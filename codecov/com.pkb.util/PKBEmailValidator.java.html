package com.pkb.util;

import com.pkb.common.config.SensitiveInformationSupport;
import com.pkb.common.config.SensitiveInformationSupport.EmailSensitivityLevel;
import com.pkb.datamodel.Email;
import com.pkb.util.PKBEmailValidator.EmailValidationFailure.Blank;
import com.pkb.util.PKBEmailValidator.EmailValidationFailure.Invalid;
import io.vavr.control.Validation;
import org.apache.commons.validator.routines.EmailValidator;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.slf4j.Logger;

import java.lang.invoke.MethodHandles;

import static com.pkb.datamodel.ImmutableDisclosingEmail.disclosingEmail;
import static com.pkb.datamodel.ImmutableObfuscatingEmail.obfuscatingEmail;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.API.Try;
import static io.vavr.control.Validation.invalid;
import static java.lang.String.format;
import static org.apache.commons.lang3.StringUtils.isBlank;
import static org.slf4j.LoggerFactory.getLogger;

/**
 * Validated Email addresses.
 * {@link EmailValidator} rejects some valid emails. It is wrapped in order to give us flexibility to either replace it or guard some of the corner cases (if the need comes).
 */
public class PKBEmailValidator {
    private static final Logger LOGGER = getLogger(MethodHandles.lookup().lookupClass());
    private static final EmailValidator VALIDATOR = EmailValidator.getInstance();

    public interface EmailValidationFailure {
        String message();

        static Blank blank() {
            return new Blank();
        }

        static Invalid invalidEmail(String email) {
            return new Invalid(email);
        }

        static Invalid invalidEmail(String email, Throwable cause) {
            return new Invalid(email, cause);
        }

        final class Blank implements EmailValidationFailure {
            private Blank() {
            }

            @Override
            public String message() {
                return "blank value is not a valid email address";
            }
        }

        final class Invalid implements EmailValidationFailure {
            private final String message;
            private final @Nullable Throwable cause;

            private Invalid(String email, @Nullable Throwable cause) {
                this.message = format("%s is not a valid email address", email);
                this.cause = cause;
            }

            private Invalid(String email) {
                this(email, null);
            }

            public Throwable cause() {
                return cause;
            }

            @Override
            public String message() {
                return message;
            }
        }
    }

    private final SensitiveInformationSupport sensitiveInformation;

    public PKBEmailValidator(SensitiveInformationSupport sensitiveInformation) {
        this.sensitiveInformation = sensitiveInformation;
    }

    public boolean isValidEmail(String email) {
        return VALIDATOR.isValid(email);
    }

    public boolean isNotValid(String email) {
        return !isValidEmail(email);
    }

    /**
     * @param email
     *            Nullable value.
     * @return Either {@link EmailValidationFailure} or {@link Email} depending on if we recognize the input as a valid email address.
     */
    public @NotNull Validation<@NotNull EmailValidationFailure, @NotNull Email> validEmail(@Nullable String email) {
        if (isBlank(email)) {
            return invalid(new Blank());
        }

        if (isValidEmail(email)) {
            return Try(() -> Match(sensitiveInformation.emailSensitivityLevel()).<Email> of(
                    Case($(EmailSensitivityLevel.OBFUSCATE), () -> obfuscatingEmail(email)),
                    Case($(EmailSensitivityLevel.DISCLOSE), () -> disclosingEmail(email)),
                    Case($(), level -> {
                        LOGGER.warn("Invalid sensitivity level ({}) received, defaulting to safest option.", level.name());
                        return obfuscatingEmail(email);
                    }))).toValidation()
                            .mapError(cause -> new Invalid(email, cause));
        }

        return invalid(new Invalid(email));
    }
}
