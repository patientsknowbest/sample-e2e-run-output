package com.pkb.util;

import uk.co.e_mis.integration.portalsdk.entities.SessionRequest;

import javax.ws.rs.WebApplicationException;
import javax.ws.rs.core.Response.Status;
import javax.xml.bind.JAXBContext;
import javax.xml.bind.JAXBException;
import javax.xml.bind.Marshaller;
import javax.xml.bind.Unmarshaller;
import java.io.ByteArrayOutputStream;
import java.io.IOException;
import java.io.Reader;


public class SessionRequestUtil {

    private static JAXBContext jaxbContext;

    private synchronized static JAXBContext getJAXBContext() throws JAXBException {
        if (jaxbContext == null) {
            jaxbContext = JAXBContext.newInstance(SessionRequest.class);
        }
        return jaxbContext;
    }

    @SuppressWarnings("ThrowInsideCatchBlockWhichIgnoresCaughtException")
    public SessionRequest parseSessionRequest(Reader reader) throws IOException {
        try {
            Unmarshaller unmarshaller = getJAXBContext().createUnmarshaller();
            return (SessionRequest) unmarshaller.unmarshal(reader);
        } catch (JAXBException ignored) {
            throw new WebApplicationException("The session request data POSTed to the service is malformed and cannot be processed",
                    Status.BAD_REQUEST);
        }
    }

    public String toXml(SessionRequest sessionRequest) throws JAXBException {
        Marshaller marshaller = getJAXBContext().createMarshaller();
        marshaller.setProperty(Marshaller.JAXB_FRAGMENT, true);
        marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
        ByteArrayOutputStream baos = new ByteArrayOutputStream();
        marshaller.marshal(sessionRequest, baos);
        return baos.toString();
    }
}
