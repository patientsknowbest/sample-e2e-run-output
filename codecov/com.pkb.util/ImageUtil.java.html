package com.pkb.util;

import org.apache.commons.lang3.StringUtils;

import javax.imageio.ImageIO;
import javax.imageio.ImageReader;
import javax.imageio.stream.ImageInputStream;
import java.io.ByteArrayInputStream;
import java.io.IOException;
import java.util.Iterator;

public class ImageUtil {

    // From
    // http://today.java.net/pub/a/today/2007/04/03/perils-of-image-getscaledinstance.html
    // with mods to keep proportions and not go into an endless loop when width<maxwidth at start

    public static String getContentType(byte[] image) throws IOException {
        ByteArrayInputStream bis = new ByteArrayInputStream(image);
        ImageInputStream iis = ImageIO.createImageInputStream(bis);

        String formatName = null;

        Iterator<ImageReader> iter = ImageIO.getImageReaders(iis);
        if (iter.hasNext()) {
            ImageReader reader = iter.next();
            iis.close();
            formatName = reader.getFormatName();
        }

        return formatNameToMimeType(formatName);
    }

    /** TODO imperfect hack, but better than what we had... */
    public static String formatNameToMimeType(String formatName) {
        if (StringUtils.isBlank(formatName)) {
            return "";
        }

        String clean = StringUtils.trim(formatName).toLowerCase();
        if (clean.equals("jpg")) {
            clean = "jpeg";
        }

        return "image/" + clean;
    }

    /**
     * NOTE: early versions of radiology didn't store a mime type in the "contentType" field...
     * they stored image format name instead. That's why we need to check for the mime types
     * as well as image format name.
     *
     * Currently browsers reliably support jpeg, png, and gif.
     *
     * @param mimeType
     * @return
     */
    public static boolean isBrowserImageByMimeType(String mimeType) {
        return "image/jpeg".equals(mimeType) || "image/png".equals(mimeType) || "image/gif".equals(mimeType)
                || "jpeg".equals(mimeType) || "jpg".equals(mimeType) || "png".equals(mimeType) || "gif".equals(mimeType);

    }

    /**
     * @param fileName
     * @return true if the image ends in .jpg, .jpeg, .png, or .gif
     */
    public static boolean isBrowserImageByFilename(String filename) {
        String[] split = StringUtils.split(filename, '.');
        if ((split != null) && (split.length > 0)) {
            String ext = split[split.length - 1].trim().toLowerCase();
            return ext.equals("jpeg") || ext.equals("jpg") || ext.equals("png") || ext.equals("gif");
        }
        return false;
    }
}
