package com.pkb.util;

import com.google.common.base.Objects;
import org.apache.commons.lang3.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.invoke.MethodHandles;
import java.time.Instant;
import java.time.LocalDate;
import java.time.LocalTime;
import java.time.ZoneId;
import java.time.ZoneOffset;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeFormatterBuilder;
import java.time.format.DateTimeParseException;
import java.time.format.ResolverStyle;
import java.time.format.SignStyle;
import java.time.temporal.TemporalQueries;
import java.util.Date;
import java.util.Locale;

import static com.pkb.util.PKBWebUtil.DOB_VALIDATOR_FORMATTER;
import static java.time.temporal.ChronoField.DAY_OF_MONTH;
import static java.time.temporal.ChronoField.YEAR;


public class UIDate {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    public static final ZoneId DEFAULT_ZONEID = ZoneId.of("+0000");

    public static DateTimeFormatter BACKEND_FMT = DateTimeFormatter.ofPattern("yyyy-MM-dd", Locale.UK);

    public static DateTimeFormatter UI_FMT = new DateTimeFormatterBuilder()
            .parseCaseInsensitive()
            .appendValue(DAY_OF_MONTH, 1, 2, SignStyle.NEVER)
            .appendLiteral('/')
            .appendPattern("MMM")
            .appendLiteral('/')
            .appendValue(YEAR, 4)
            .toFormatter()
            .withResolverStyle(ResolverStyle.STRICT)
            .withLocale(Locale.UK);

    private String year;

    private String day;

    private String month;

    public String getYear() {
        return year;
    }

    public String getDay() {
        return day;
    }

    public String getMonth() {
        return month;
    }

    public void setYear(String year) {
        this.year = year;
    }

    public void setDay(String day) {
        this.day = day;
    }

    public void setMonth(String month) {
        this.month = month;
    }

    public UIDate() {
        setEmptyUIDate();
    }

    /**
     * Use new UIDate(Instant) instead
     */
    @Deprecated
    public UIDate(Date date) {
        if (date == null) {
            setEmptyUIDate();
        } else {
            initDayMonthYear(date.toInstant());
        }
    }

    public UIDate(Instant date) {
        if (date == null) {
            setEmptyUIDate();
        } else {
            initDayMonthYear(date);
        }
    }

    public UIDate(LocalDate date) {
        if (date == null) {
            setEmptyUIDate();
        } else {
            initDayMonthYear(date);
        }
    }

    public void initDayMonthYear(Instant date) {
        ZonedDateTime zdt = ZonedDateTime.ofInstant(date, DEFAULT_ZONEID);
        this.day = StringUtils.leftPad(String.valueOf(zdt.getDayOfMonth()), 2, "0");
        this.month = PKBWebUtil.getMonth(zdt.getMonthValue());
        this.year = String.valueOf(zdt.getYear());
    }

    public void initDayMonthYear(LocalDate date) {
        this.day = StringUtils.leftPad(String.valueOf(date.getDayOfMonth()), 2, "0");
        this.month = PKBWebUtil.getMonth(date.getMonthValue());
        this.year = String.valueOf(date.getYear());
    }

    public UIDate(LocalDate localDate, ZoneOffset offset) {
        initDayMonthYear(localDate.atStartOfDay(offset).toInstant());
    }

    private void setEmptyUIDate() {
        year = "";
        month = "";
        day = "";
    }

    @Override
    public String toString() {
        return this.day + "/" + this.month + "/" + this.year;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (o == null || getClass() != o.getClass()) {
            return false;
        }
        UIDate uiDate = (UIDate) o;
        return Objects.equal(year, uiDate.year) &&
                Objects.equal(day, uiDate.day) &&
                Objects.equal(month, uiDate.month);
    }

    @Override
    public int hashCode() {
        return Objects.hashCode(year, day, month);
    }

    public Date toDate() throws InvalidUIDateException {
        try {
            LocalDate localDate = UI_FMT.parse(toString(), TemporalQueries.localDate());
            return Date.from(ZonedDateTime.of(localDate, LocalTime.MIN, DEFAULT_ZONEID).toInstant());
        } catch (DateTimeParseException ignored) {
            throw new InvalidUIDateException(this);
        }
    }

    /**
     * Returns the date in yyyy-MM-dd format. It should be interpreted as as UTC date.
     */
    public String toFormattedDate() throws InvalidUIDateException {
        ZonedDateTime zdt = ZonedDateTime.ofInstant(toInstant(), DEFAULT_ZONEID);
        return BACKEND_FMT.format(zdt);
    }

    @Deprecated
    public boolean validate() {
        try {
            LocalDate.parse(year + "-" + month + "-" + day, DOB_VALIDATOR_FORMATTER);
        } catch (Exception e) {
            LOGGER.error("Error while parsing date: {}/{}/{}", day, month, year, e);
            return false;
        }
        return true;
    }

    public Instant toInstant() throws InvalidUIDateException {
        return toDate().toInstant();
    }

    public boolean isIncomplete() {
        return getDay().isEmpty() || getMonth().isEmpty() || getYear().isEmpty();
    }
}
