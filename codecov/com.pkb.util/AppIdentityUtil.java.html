package com.pkb.util;

import org.apache.commons.lang3.StringUtils;

import javax.servlet.http.HttpServletRequest;
import java.util.Arrays;
import java.util.Optional;
import java.util.function.Predicate;
import java.util.regex.Pattern;
import java.util.stream.Stream;

import static com.pkb.entities.enums.Route.WEBAPP;
import static com.pkb.entities.enums.Route.WEBAPP_EMIS_PORTAL;
import static com.pkb.entities.enums.Route.WEBAPP_SYSTMONE_PORTAL;
import static com.pkb.util.PKBConstants.ROUTE;

public class AppIdentityUtil {

    public enum AppIdentity {
        NHS_APP("nhs-app", getNhsApp()),
        NHS_APP_WAYFINDER("nhs-app-wayfinder", getWayfinderNhsApp()),
        NHS_APP_WEB_WAYFINDER("nhs-app-web-wayfinder", getWayfinderNhsAppWeb()),
        EMIS_PORTAL("emis-app", getSessionAttributeMatcher(ROUTE, WEBAPP_EMIS_PORTAL)),
        PKB_WEB("phr-web", getSessionAttributeMatcher(ROUTE, WEBAPP)),
        S1_PORTAL("s1-portal", getSessionAttributeMatcher(ROUTE, WEBAPP_SYSTMONE_PORTAL));

        private final String appIdentity;
        private Predicate<HttpServletRequest> matcher;

        AppIdentity(String appIdentity, Predicate<HttpServletRequest> matcher) {
            this.appIdentity = appIdentity;
            this.matcher = matcher;
        }

        public boolean matches(HttpServletRequest request) {
            return matcher.test(request);
        }

        @Override
        public String toString() {
            return appIdentity;
        }
    }

    // will be either nhsapp-android or nhsapp-ios. Match both (and future platforms they might support)

    private static final Predicate<HttpServletRequest> WAYFINDER_MATCHER = request -> {
        var infoPath = StringUtils.defaultIfBlank(request.getParameter("info.path"), "");
        return "true".equals(request.getParameter("nhsStyle")) || infoPath.contains("nhsStyle=true");
    };

    private static final Pattern NHS_APP_USER_AGENT = Pattern.compile("nhsapp-(.*)");
    public AppIdentity resolve(HttpServletRequest request) {
        return Arrays.stream(AppIdentity.values())
                .filter(value -> value.matches(request))
                .findFirst()
                .orElse(AppIdentity.PKB_WEB);
    }

    public boolean isEmisPortalSession(HttpServletRequest request) {
        return AppIdentity.EMIS_PORTAL.matches(request);
    }

    public boolean isSsoPortalSession(HttpServletRequest request) {
        return isSystmOnePortalSession(request) || isEmisPortalSession(request);
    }

    public boolean isSystmOnePortalSession(HttpServletRequest request) {
        return AppIdentity.S1_PORTAL.matches(request);
    }

    public boolean isNhsAppSession(HttpServletRequest request) {
        return Stream.of(AppIdentity.NHS_APP, AppIdentity.NHS_APP_WAYFINDER)
                .anyMatch(appIdentity -> appIdentity.matches(request));
    }

    private static Predicate<HttpServletRequest> getSessionAttributeMatcher(String attributeName, Object value) {
        return request -> value.equals(request.getSession().getAttribute(attributeName));
    }

    private static Predicate<HttpServletRequest> getHeaderMatcher(String headerName, Pattern matcher) {
        return request -> Optional.ofNullable(request.getHeader(headerName)).map(value -> matcher.matcher(value).find()).orElse(false);
    }

    private static Predicate<HttpServletRequest> getNhsApp() {
        return WAYFINDER_MATCHER.negate().and(getHeaderMatcher("User-Agent", NHS_APP_USER_AGENT));
    }

    private static Predicate<HttpServletRequest> getWayfinderNhsApp() {
        return WAYFINDER_MATCHER.and(getHeaderMatcher("User-Agent", NHS_APP_USER_AGENT));
    }

    private static Predicate<HttpServletRequest> getWayfinderNhsAppWeb() {
        return WAYFINDER_MATCHER.and(getHeaderMatcher("User-Agent", NHS_APP_USER_AGENT).negate());
    }
}
