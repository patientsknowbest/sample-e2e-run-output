package com.pkb.util;

import com.google.common.collect.ArrayTable;
import com.google.common.collect.ImmutableTable;
import com.google.common.collect.Table;

import java.util.Map;
import java.util.stream.Collectors;

/**
 * Remote EJB has problems with Guava table, can only return mapview, which is hard to use, this should hide it, and allow easy change to Table once off of EJB
 * "loader constraint violation: when resolving interface method"
 * The whole class can be removed once we don'T have any EJBs and remoting left...
 *
 * @param <T>
 * @param <H>
 * @param <I>
 */
public class TableRemoteDeSerializer {

    private TableRemoteDeSerializer() {
    }

    public static <J, K, L> Table<J, K, L> deSerialize(Map<J, Map<K, L>> input) {
        if (input == null || input.isEmpty()) {
            return ImmutableTable.of();
        }
        var columKeys = input.entrySet().stream().flatMap(e -> e.getValue().keySet().stream()).collect(Collectors.toSet());
        if (columKeys.isEmpty()) {
            return ImmutableTable.of();
        }
        Table<J, K, L> result = ArrayTable.create(input.keySet(), columKeys);
        input.forEach((k, v) -> {
            v.forEach((k2, v2) -> {
                result.put(k, k2, v2);
            });
        });
        return result;
    }
}
