package com.pkb.util;

import java.io.Writer;
import java.util.List;
import java.util.Map;
import java.util.function.Consumer;
import java.util.function.Function;
import java.util.stream.Stream;

public class CsvStreamService {
    private UncheckedCsvPrinter.Factory printerFactory;

    public CsvStreamService(UncheckedCsvPrinter.Factory printerFactory) {
        this.printerFactory = printerFactory;
    }

    /**
     * See {@link CsvStreamService#streamToCsvConsumer(Writer, List, Map, Consumer)}
     */
    public <T> Consumer<Stream<T>> streamToCsvConsumer(Writer outputWriter,
                                                       List<String> columnNames,
                                                       Map<String, Function<T, Object>> mapColumnNameToValueFunction,
                                                       boolean writeError)
    {
        return streamToCsvConsumer(outputWriter, columnNames, mapColumnNameToValueFunction, writeError, t -> {});
    }

    /**
     * Consumer to write a stream of records out to csv format
     * @param outputWriter - where to write the csv rows to
     * @param columnNames - the order to write the csv columns, also used as a header
     * @param mapColumnNameToValueFunction - a map providing functions to get a printable value for each column, keyed by column name
     * @param writeError - whether any errors should be fed down the stream
     * @param afterEach - A consumer that will be called after each row is printed, for example you could use to detach an entity from the persistence context.
     *
     * @return a consumer that can be fed records
     *
     */
    public <T> Consumer<Stream<T>> streamToCsvConsumer(Writer outputWriter,
                                                       List<String> columnNames,
                                                       Map<String, Function<T, Object>> mapColumnNameToValueFunction,
                                                       boolean writeError,
                                                       Consumer<T> afterEach) {
        return dataStream -> {
            try (var printer = printerFactory.getPrinter(outputWriter, columnNames)) {
                try {
                    dataStream.forEach(dataRow -> {
                        //print cells for columns matching the data stream
                        columnNames.forEach(column -> {
                            try {
                                Function<T, Object> valueFunction = mapColumnNameToValueFunction.get(column);
                                printer.print(valueFunction.apply(dataRow));
                            } catch (Exception e) {
                                throw new RuntimeException("Failed to get value for column '" + column + "' because " + e.getMessage(), e);
                            }
                        });

                        //end of row
                        printer.println();

                        afterEach.accept(dataRow);
                    });
                } catch (Exception e) {
                    if (writeError) {
                        printer.println();
                        printer.print("Error whilst streaming to CSV: " + e.getMessage());
                    }
                    throw e;
                }
            }
        };
    }
}
