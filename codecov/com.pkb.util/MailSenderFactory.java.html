package com.pkb.util;

import com.pkb.common.config.PhrConfig;
import org.springframework.mail.javamail.JavaMailSenderImpl;

import java.util.Properties;

/**
 * Created by mvarga on 2/22/16.
 */

public class MailSenderFactory {

    static public JavaMailSenderImpl getMailSender(PhrConfig cfg) {
        JavaMailSenderImpl impl = new JavaMailSenderImpl();
        impl.setHost(cfg.getSmtpHostname());
        impl.setPort(cfg.getSmtpPort());
        impl.setUsername(cfg.getSmtpUser());
        impl.setPassword(cfg.getSmtpPassword());
        Properties javaMailProps = new Properties();
        javaMailProps.setProperty("mail.smtp.auth", cfg.getSmtpAuth());
        javaMailProps.setProperty("mail.smtp.starttls.enable", cfg.getSmtpStartTlsEnable());
        javaMailProps.setProperty("mail.smtp.socketFactory.class", cfg.getSmtpSocketFactoryClass());
        javaMailProps.setProperty("mail.smtp.socketFactory.fallback", cfg.getSmtpSocketFactoryFallback());
        impl.setJavaMailProperties(javaMailProps);
        return impl;
    }
}
