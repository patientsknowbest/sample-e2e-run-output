    private static final ZoneId UTC = ZoneId.of("UTC");

    private static final List<Tuple2<String, String>> TIMEZONE_LIST = readConfig("timezonelist.xml",
            c -> Tuple.of(String.valueOf(c.getAttributes().get("zoneId")), String.valueOf(c.getValue())));
    private static final ConcurrentMap<Locale, Map<String, String>> TIMEZONES = new ConcurrentHashMap<>();

    private static final List<Tuple2<String, List<String>>> ETHNICITIES_LIST = readListConfig("ethnicitylist.xml");
    private static final ConcurrentMap<Locale, Map<String, List<String>>> ETHNICITIES = new ConcurrentHashMap<>();

    private static final List<Tuple2<String, List<String>>> RELIGION_LIST = readListConfig("religionlist.xml");
    private static final ConcurrentMap<Locale, Map<String, List<String>>> RELIGIONS = new ConcurrentHashMap<>();

    private static final List<Tuple2<String, String>> PHOTO_ID_LIST = readConfig("photoIdList.xml", c -> Tuple.of(c.getNodeName(), String.valueOf(c.getValue())));
    private static final ConcurrentMap<Locale, Map<String, String>> PHOTO_IDS = new ConcurrentHashMap<>();
    private static final List<Tuple2<String, String>> CLINICIAN_PHOTO_ID_LIST = readConfig("clinicianPhotoIdList.xml",
            c -> Tuple.of(c.getNodeName(), String.valueOf(c.getValue())));
    private static final ConcurrentMap<Locale, Map<String, String>> CLINICIAN_PHOTO_IDS = new ConcurrentHashMap<>();
    private static final Set<String> ACTIVE_PHOTO_IDS = unmodifiableSet(Arrays.stream(PhotoIdType.values())
            .filter(PhotoIdType::isActive)
            .map(PhotoIdType::toString)
            .collect(toSet()));

    private static DateTimeFormatter getDoBFormatter() {
        DateTimeFormatterBuilder builder = new DateTimeFormatterBuilder();
        return builder.parseCaseInsensitive()
                .appendPattern("uuuu-MMM-d[d]")
                .toFormatter(Locale.UK)
                .withResolverStyle(ResolverStyle.STRICT)
                .withZone(UTC);
    }

    /**
     * Sometimes we need to redirect somewhere after the team-select screen for
     * a multiple-team user. However the spring-managed savedUrl variable is
     * cleared before the team-select screen is displayed, so
     * {@link UserLoginInterceptor} makes a copy of it
     * using this key, which {@link ClinicianMultipleTeamAction}
     * can interrogate if it wants to
     */
    public static final String SAVED_URL_COPY = "savedUrlCopy";

    private static final ConcurrentMap<Tuple2<Locale, Integer>, List<String>> YEARS = new ConcurrentHashMap<>();
    private static final ConcurrentMap<Locale, List<String>> MONTHS = new ConcurrentHashMap<>();
    private static final ConcurrentMap<Locale, List<String>> DAYS = new ConcurrentHashMap<>();

    private static final Pattern UNDERSCORE_PATTERN = Pattern.compile("_", Pattern.LITERAL);

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    public PKBWebUtil(MessageSource messageSource, DateTimeService dateTimeService, TeamManager teamManager, UserManager userManager) {
        this.dateTimeService = dateTimeService;
        this.messageSource = messageSource;
        this.teamManager = teamManager;
        this.userManager = userManager;
    }

    /**
     * @param locale
     * @return map of Java-recognized timezone ID -> localized display name
     */
    public Map<String, String> getTimeZones(Locale locale) {
        return TIMEZONES.computeIfAbsent(locale, $ -> computeTimeZones($));
    }

    private Map<String, String> computeTimeZones(Locale locale) {
        return computeTranslation(TIMEZONE_LIST, s -> getTranslation(s, locale, (s2, e) -> LOGGER.error("No timezone value for '{}'", s2, e)), "Duplicate timezone!");
    }

    private Map<String, String> computePhotoIds(Locale locale) {
        return computeTranslation(PHOTO_ID_LIST, t -> ACTIVE_PHOTO_IDS.contains(t._1),
                s -> getTranslation(s, locale, (s2, e) -> LOGGER.error("No photoId value for '{}'", s2, e)),
                "Duplicate photoId!");
    }

    private Map<String, String> computeClinicianPhotoIds(Locale locale) {
        return computeTranslation(CLINICIAN_PHOTO_ID_LIST, t -> ACTIVE_PHOTO_IDS.contains(t._1),
                s -> getTranslation(s, locale, (s2, e) -> LOGGER.error("No clinician photoId value for '{}'", s2, e)),
                "Duplicate clinician photoId!");
    }

    private String getTranslation(String key, Locale locale, BiConsumer<String, Exception> errorHandler) {
        try {
            return messageSource.getMessage(key, null, locale);
        } catch (Exception e) {
            errorHandler.accept(key, e);
        }
        return null;
    }

    private Map<String, List<String>> computeEthnicities(Locale locale) {
        return computeTranslation(ETHNICITIES_LIST, l -> unmodifiableList(l
                .stream()
                .map(s -> getTranslation(s, locale, (s2, e) -> LOGGER.error("No ethnicity value for '{}'", s2, e)))
                .filter(Objects::nonNull)
                .collect(toList())), "Duplicate ethnicity!");
    }

    /**
     * Returns a map with keys of country codes and values of the different ethnicity options listed for that country. Only returns country

    public Map<String, List<String>> getEthnicities(Locale locale) {
        return ETHNICITIES.computeIfAbsent(locale, $ -> computeEthnicities($));

    public List<String> getDays(Locale locale) {
        return DAYS.computeIfAbsent(locale, $ -> computeDaysList($));
    }

    private List<String> computeDaysList(Locale locale) {
        List<String> dayList = new ArrayList<>(32);
        dayList.add(getTranslation("global.placeholder.day", locale, (s2, e) -> LOGGER.error("No day placeholder value for '{}'", s2, e)));
        for (int i = 1; i <= 31; i++) {
            if (i <= 9) {
                dayList.add("0" + i);
            } else {
                dayList.add(String.valueOf(i));
            }
        }
        return unmodifiableList(dayList);
    }

    public List<String> getMonths(Locale locale) {
        return MONTHS.computeIfAbsent(locale, $ -> computeMonthsList($));
    }

    private List<String> computeMonthsList(Locale locale) {
        List<String> monthList = new ArrayList<>(13);
        monthList.add(getTranslation("global.placeholder.month", locale, (s2, e) -> LOGGER.error("No month placeholder value for '{}'", s2, e)));
        for (int i = 1; i <= 12; i++) {
            if (i <= 9) {
                monthList.add("0" + i);
            } else {
                monthList.add(String.valueOf(i));
            }
        }
        return unmodifiableList(monthList);
    }

    public List<String> getYears(Locale locale) {
        int cal = dateTimeService.nowZonedDateTime().getYear();
        return YEARS.computeIfAbsent(Tuple.of(locale, cal), $ -> computeYears($));
    }

    private List<String> computeYears(Tuple2<Locale, Integer> input) {
        List<String> yearList = new ArrayList<>(input._2 + 1);
        yearList.add(messageSource.getMessage("global.placeholder.year", null, input._1));
        for (int i = 1900; i <= input._2; i++) {
            yearList.add(String.valueOf(i));
        }
        return unmodifiableList(yearList);
    }

    public Map<Long, String> getOrgs(boolean includeCreateOption, Locale locale) {
        Map<Long, String> map = new LinkedHashMap<>();
        try {
            List<Org> orgs = teamManager.getOrgList();

            if (includeCreateOption) {
                map.put(-1L, messageSource.getMessage("global.placeholder.create.org", null, locale));
            }

            // format into map
            for (Org org : orgs) {
                map.put(org.getId(), org.getName());
            }

        } catch (Exception e) {
            throw new RuntimeException("Unable to load orgs", e);
        }
        return map;
    }

    public Map<Long,String> getTeamsForOrg(Org org) {
        Map<Long, String> map = new LinkedHashMap<>();
        map.put(-1L,"Select a team");
        org.getTeams().forEach(t -> map.put(t.getId(),t.getName()));
        return map;
    }

    public Map<Long, String> getSpecialties() {
        return getSpecialties(null);
    }

    public Map<Long, String> getSpecialties(Locale locale) {
        Map<Long, String> map = new LinkedHashMap<>();
        try {
            List<PKBSpecialty> specialties = userManager.getSpecialties();

            // format into map
            for (PKBSpecialty spec : specialties) {
                try {
                    map.put(spec.id(), messageSource.getMessage(spec.name(), null, locale));
                } catch (Exception e) {
                    LOGGER.error("No specialty value for '{}'", spec.name(), e);
                }
            }

        } catch (Exception e) {
            throw new RuntimeException("Unable to load specialties", e);
        }

        return map;
    }

    public Map<String, String> getPatientIdSupportOptions(Locale locale) {
        Map<String, String> map = new LinkedHashMap<>();

        for (PatientIdSupport pis : PatientIdSupport.values()) {
            map.put(pis.name(), messageSource.getMessage(pis.getTranslationKey(), null, locale));
        }
        return map;
    }

    public List<PKBSecurityQuestion> getSecretQuestions(Locale locale) {
        List<PKBSecurityQuestion> questions = null;
        try {
            questions = userManager.getSecurityQuestions();
            for (PKBSecurityQuestion question : questions) {
                try {
                    question.setQuestion(messageSource.getMessage(question.getQuestion(), null, locale));
                } catch (Exception e) {
                    LOGGER.error("No security question value for '{}'", question.getQuestion(), e);
                }
            }
        } catch (Exception e) {
            LOGGER.error("Unable to load secret questions", e);
            throw new RuntimeException("Unable to load secret questions", e);
        }
        return questions;
    }

    public boolean breakTheGlassForPatient(long patientId) {
        Map<String, Object> session = ActionContext.getContext().getSession();
        Set<Long> patients = (Set<Long>) session.get(BREAK_THE_GLASS);
        if (patients == null) {
            patients = Sets.newHashSet();
        }

        patients.add(patientId);
        session.put(BREAK_THE_GLASS, patients);
        return true;
    }

    public boolean hasBreakTheGlassForPatient(Long patientId) {
        if (patientId == null) {
            return false;
        }

        Map<String, Object> session = ActionContext.getContext().getSession();
        Set<Long> patients = (Set<Long>) session.get(BREAK_THE_GLASS);
        return (patients != null) && patients.contains(patientId);
    }

    public void copyRedirectUrl(SavedRequest savedRequest) {
        ActionContext.getContext().getSession().put(SAVED_URL_COPY, savedRequest);
    }

    public DefaultSavedRequest getSavedRequest() {
        return (DefaultSavedRequest) ActionContext.getContext().getSession().get(SAVED_URL_COPY);
    }

    @Override
    public void clearState() {
        TIMEZONES.clear();
        ETHNICITIES.clear();
        RELIGIONS.clear();
        PHOTO_IDS.clear();
        CLINICIAN_PHOTO_IDS.clear();
        YEARS.clear();
        MONTHS.clear();
        DAYS.clear();
    }
