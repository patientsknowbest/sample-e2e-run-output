package com.pkb.util;

import org.apache.http.client.utils.URIBuilder;

import javax.annotation.Nullable;
import java.net.URI;
import java.net.URISyntaxException;
import java.util.Map;

import static java.lang.String.valueOf;
import static java.util.Collections.emptyMap;

public class UrlUtil {

    public static String buildHref(String baseUrl, WebNamespace namespace, String action, Map<String, String> parameters, @Nullable Long contextUserId) {
        try {
            URI uri = new URI(baseUrl);
            URIBuilder builder = new URIBuilder();
            builder.setScheme(uri.getScheme());
            builder.setHost(uri.getHost());
            builder.setPort(uri.getPort());
            return buildRelativeUrl(namespace, action, parameters, contextUserId, builder);
        } catch (URISyntaxException e) {
            throw new RuntimeException(e);
        }
    }

    public static String buildHref(WebNamespace namespace, String action, Map<String, String> parameters, @Nullable Long contextUserId) {
        return buildRelativeUrl(namespace, action, parameters, contextUserId, new URIBuilder());
    }

    public static String buildHref(WebNamespace namespace, String action, @Nullable Long contextUserId) {
        return buildHref(namespace, action, emptyMap(), contextUserId);
    }

    public static String buildHref(WebNamespace namespace, String action) {
        return buildHref(namespace, action, null);
    }

    private static String buildRelativeUrl(WebNamespace namespace, String action, Map<String, String> parameters, @Nullable Long contextUserId, URIBuilder builder) {
        builder.setPath(namespace.toString() + "/" + action + ".action");
        parameters.forEach(builder::addParameter);

        if (contextUserId != null) {
            builder.addParameter("contextUserId", valueOf(contextUserId));
        }
        return builder.toString();
    }
}
