package com.pkb.util;

import io.vavr.Tuple;
import io.vavr.Tuple2;
import org.apache.commons.configuration2.XMLConfiguration;
import org.apache.commons.configuration2.builder.FileBasedConfigurationBuilder;
import org.apache.commons.configuration2.builder.fluent.Parameters;
import org.apache.commons.configuration2.builder.fluent.XMLBuilderParameters;
import org.apache.commons.configuration2.convert.DisabledListDelimiterHandler;
import org.apache.commons.configuration2.tree.ImmutableNode;

import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;
import java.util.function.BinaryOperator;
import java.util.function.Function;
import java.util.function.Predicate;

import static java.util.Collections.unmodifiableList;
import static java.util.Collections.unmodifiableMap;
import static java.util.stream.Collectors.toList;
import static java.util.stream.Collectors.toMap;

public class XMLConfigUtil {

    private XMLConfigUtil() {
    }

    public static <R> List<R> readConfig(String fileName, Function<ImmutableNode, R> mapper) {
        try {
            XMLBuilderParameters xmlParams = new Parameters().xml()
                    .setFileName(fileName)
                    .setListDelimiterHandler(new DisabledListDelimiterHandler());
            XMLConfiguration cfg = new FileBasedConfigurationBuilder<>(XMLConfiguration.class)
                    .configure(xmlParams)
                    .getConfiguration();

            return unmodifiableList(cfg.getNodeModel().getNodeHandler().getRootNode().getChildren()
                    .stream()
                    .map(mapper)
                    .collect(toList()));
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }

    public static List<Tuple2<String, List<String>>> readListConfig(String fileName) {
        return readConfig(fileName, countryNode -> {
            List<String> keysToTranslate = unmodifiableList(countryNode.getChildren().stream()
                    .map(c -> String.valueOf(c.getValue()))
                    .collect(toList()));
            return Tuple.of(countryNode.getNodeName(), keysToTranslate);
        });
    }

    public static <T> Map<String, T> computeTranslation(List<Tuple2<String, T>> input, Function<T, T> translator, String duplicateErrorMessage) {
        return computeTranslation(input, x -> true, translator, duplicateErrorMessage);
    }

    public static <T> Map<String, T> computeTranslation(List<Tuple2<String, T>> input, Function<T, T> translator, BinaryOperator<T> mergeFunction) {
        return computeTranslation(input, x -> true, translator, mergeFunction);
    }

    public static <T> Map<String, T> computeTranslation(List<Tuple2<String, T>> input, Predicate<? super Tuple2<String, T>> predicate, Function<T, T> translator,
            String duplicateErrorMessage) {
        return computeTranslation(input, predicate, translator, (a, b) -> {
            throw new IllegalStateException(duplicateErrorMessage + " a: " + a + " b: " + b);
        });
    }

    public static <T> Map<String, T> computeTranslation(List<Tuple2<String, T>> input, Predicate<? super Tuple2<String, T>> predicate, Function<T, T> translator,
            BinaryOperator<T> mergeFunction) {
        return unmodifiableMap(input
                .stream()
                .filter(predicate)
                .map(t -> Tuple.of(t._1, translator.apply(t._2)))
                .filter(t -> t._2 != null)
                .collect(toMap(t -> t._1, t -> t._2, mergeFunction, LinkedHashMap::new)));
    }
}
