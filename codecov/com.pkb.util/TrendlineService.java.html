package com.pkb.util;

import com.pkb.entities.enums.UserType;
import com.pkb.test.entity.TestResultDTO;
import com.pkb.util.TrendlineService.TrendlineDataSet.TrendlineDataPoint;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import org.jetbrains.annotations.NotNull;

import java.util.Collection;
import java.util.List;
import java.util.stream.Collectors;

import static java.util.stream.Collectors.toList;

public class TrendlineService {
    public static class TrendlineDataSet {

        static class TrendlineDataPoint {
            // 1: x axe: timestamp,
            // 2: y axe: value
            final Tuple2<Double, Double> data;
            // 1: low value
            // 2: high value
            final Tuple2<Double, Double> range;

            TrendlineDataPoint(@NotNull Tuple2<Double, Double> data, @NotNull Tuple2<Double, Double> range) {
                this.data = data;
                this.range = range;
            }
        }

        private final List<Double> testValues;
        private final List<Double> testTimestamps;
        private final List<Double> rangeLows;
        private final List<Double> rangeHighs;

        public TrendlineDataSet(@NotNull Collection<TrendlineDataPoint> testResults) {
            this.testValues = testResults.stream().map($ -> $.data._1).collect(toList());
            this.testTimestamps = testResults.stream().map($ -> $.data._2).collect(toList());
            this.rangeLows = testResults.stream().map($ -> $.range._1).collect(toList());
            this.rangeHighs = testResults.stream().map($ -> $.range._2).collect(toList());
        }

        public List<Double> getTestValues() {
            return testValues;
        }

        public List<Double> getTestTimestamps() {
            return testTimestamps;
        }

        public List<Double> getRangeLows() {
            return rangeLows;
        }

        public List<Double> getRangeHighs() {
            return rangeHighs;
        }

        public boolean isNotEmpty() {
            return !testValues.isEmpty();
        }
    }

    private final DataDelayService delayService;

    public TrendlineService(DataDelayService delayService) {
        this.delayService = delayService;
    }

    public @NotNull TrendlineDataSet getTrendlineDataFor(@NotNull UserType usertype, @NotNull List<TestResultDTO> testResultList) {
        return new TrendlineDataSet(getAxisDataWithNumericValueForUserType(usertype, testResultList));
    }

    private List<TrendlineDataPoint> getAxisDataWithNumericValueForUserType(UserType userType, List<TestResultDTO> testResultList) {
        return testResultList.stream()
                .filter(result -> result.isValueNumeric() && !delayService.findWhenBecomesVisibleForUserType(result, userType).isPresent())
                .map(testResult -> {

                    Tuple2<Double, Double> data = Tuple.of(testResult.getValue(), (double) testResult.getTestDate().toEpochMilli());
                    Tuple2<Double, Double> ranges = Tuple.of(testResult.getRangeLow(), testResult.getRangeHigh());

                    return new TrendlineDataPoint(data, ranges);
                })
                .collect(Collectors.toList());
    }
}
