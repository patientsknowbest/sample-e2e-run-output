package com.pkb.util;

import com.pkb.common.base.Error;
import org.apache.commons.lang3.StringUtils;

import java.util.Optional;
import java.util.function.Function;
import java.util.function.Supplier;
import java.util.regex.Pattern;

public final class Validators {

    private static final String PHONE_NUMBER_REGEX = "^[+]?[\\d ()\\-]+$";
    private static final Pattern PHONE_NO_PATTERN_MANDATORY = Pattern.compile(PHONE_NUMBER_REGEX);
    private static final Pattern PHONE_NO_PATTERN_OPTIONAL = Pattern.compile("^$|" + PHONE_NUMBER_REGEX);

    public static Validator<String> required(Supplier<Error> onError) {
        return value -> StringUtils.isBlank(value) ? Optional.of(onError.get()) : Optional.empty();
    }

    public static Validator<String> minLength(int minLength, Function<String, Error> onError) {
        return value -> value != null && value.length() < minLength ? Optional.of(onError.apply(String.valueOf(minLength))) : Optional.empty();
    }

    public static Validator<String> maxLength(int maxLength, Function<String, Error> onError) {
        return value -> value != null && value.length() > maxLength ? Optional.of(onError.apply(String.valueOf(maxLength))) : Optional.empty();
    }

    public static Validator<String> regex(Pattern pattern, Supplier<Error> onError) {
        return value -> value != null && !pattern.matcher(value).matches() ? Optional.of(onError.get()) : Optional.empty();
    }

    public static Validator<String> phoneNumberMandatory(Supplier<Error> onError) {
        return regex(PHONE_NO_PATTERN_MANDATORY, onError);
    }

    public static Validator<String> phoneNumberOptional(Supplier<Error> onError) {
        return regex(PHONE_NO_PATTERN_OPTIONAL, onError);
    }

    private Validators() {

    }

}
