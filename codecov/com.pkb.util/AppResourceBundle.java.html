package com.pkb.util;

import io.vavr.Tuple;
import io.vavr.Tuple2;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.invoke.MethodHandles;
import java.util.HashMap;
import java.util.Locale;
import java.util.Map;
import java.util.MissingResourceException;
import java.util.ResourceBundle;

/**
 * Wrapper class on java ResourceBundle
 * This class can be further utilized for internationalization
 *
 * @author pravinam
 *
 */
public class AppResourceBundle {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
    public static final String ACTION_LABELS_BUNDLE_NAME = "actionLabels";
    public static final String MISCELLANEOUS_LABELS_BUNDLE_NAME = "miscellaneousLabels";

    private static final Map<Tuple2<String, Locale>, ResourceBundle> bundles = new HashMap<>();

    static { //by default
        bundles.put(Tuple.of(ACTION_LABELS_BUNDLE_NAME, Locale.getDefault()), ResourceBundle.getBundle(ACTION_LABELS_BUNDLE_NAME));
        bundles.put(Tuple.of(MISCELLANEOUS_LABELS_BUNDLE_NAME, Locale.getDefault()), ResourceBundle.getBundle(MISCELLANEOUS_LABELS_BUNDLE_NAME));
    }

    public static String getPropertyValue(String property, Locale locale, String bundleName) {
        if (!bundles.containsKey(Tuple.of(bundleName, locale))) {
            try {
                bundles.put(Tuple.of(bundleName, locale), ResourceBundle.getBundle(bundleName, locale));
            } catch (MissingResourceException e) {
                LOGGER.warn("Missing resource bundle '{}' for language {}}", bundleName, locale.getDisplayLanguage(), e);
            }
        }
        return bundles.getOrDefault(Tuple.of(bundleName, locale), bundles.get(Tuple.of(bundleName, Locale.getDefault()))).getString(property);
    }
}
