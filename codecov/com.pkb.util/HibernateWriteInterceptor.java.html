package com.pkb.util;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.HasVersionDetails;
import com.pkb.VersionDetails;
import com.pkb.entities.enums.UserStatus;
import com.pkb.user.entity.PKBPerson;
import org.hibernate.EmptyInterceptor;
import org.hibernate.type.Type;

import java.io.Serializable;
import java.util.Collection;

public class HibernateWriteInterceptor extends EmptyInterceptor {

    private static final long serialVersionUID = 1L;
    private SourceChecker sourceChecker;

    public HibernateWriteInterceptor() {
        super();
        sourceChecker = new SourceChecker();
    }

    @VisibleForTesting
    public HibernateWriteInterceptor(SourceChecker sourceChecker) {
        super();
        this.sourceChecker = sourceChecker;
    }

    @Override
    public boolean onSave(Object entity, Serializable id, Object[] state, String[] propertyNames, Type[] types) {
        if (entity instanceof PKBPerson) {
            PKBPerson person = (PKBPerson) entity;
            if (person.isPatient() && person.getStatus()== UserStatus.NEW && person.getVersionDetails().getSourcePersonId()==null) {

                // Fix sourceperson for self-reg patient now that we know the pkbperson.id
                VersionDetails patientVersionDetails = person.getVersionDetails();
                patientVersionDetails.setSourcePersonId(person.getId());

                // And pass down to any immediate children, eg pkbperson.nationalids, who will in turn be handed to onSave()
                fixImmediateChildren(patientVersionDetails,state);
            }
        }
        sourceChecker.check(entity);
        return false;
    }

    @Override
    public boolean onFlushDirty(Object entity, Serializable id, Object[] newState, Object[] previousState, String[] propertyNames, Type[] types) {
        sourceChecker.check(entity, previousState, propertyNames);
        return false;
    }

    private void fixImmediateChildren(VersionDetails parentVersionDetails, Object[] fields) {
        for (Object field: fields) {
            if (field instanceof HasVersionDetails) {
                ((HasVersionDetails<VersionDetails>) field).setVersionDetails(parentVersionDetails);
            } else if (field instanceof Collection<?>) {
                for (Object m : (Collection<?>) field) {
                    if (m instanceof HasVersionDetails) {
                        ((HasVersionDetails<VersionDetails>) m).setVersionDetails(parentVersionDetails);
                    }
                }
            }
        }
    }
}
