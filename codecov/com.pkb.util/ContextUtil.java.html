package com.pkb.util;

import com.pkb.app.entity.EHRRequestContext;
import org.jetbrains.annotations.NotNull;

import javax.servlet.http.HttpServletRequest;

public class ContextUtil {

    private static final String EHR_REQUEST_CONTEXT_KEY = "requestContextKey";


    public @NotNull EHRRequestContext getEHRRequestContext(HttpServletRequest request) {
        // Session populated by ConsentInterceptor
        if (request != null) {
            EHRRequestContext requestContext = (EHRRequestContext) request.getAttribute(EHR_REQUEST_CONTEXT_KEY);
            if (requestContext == null) {
                throw new IllegalStateException("PHRZ-584 request != null, requestContext == null");
            }
            return requestContext;
        }
        throw new IllegalStateException("PHRZ-584 request == null");
    }

    public void setEHRRequestContext(HttpServletRequest request, EHRRequestContext context) {
        request.setAttribute(EHR_REQUEST_CONTEXT_KEY, context);
    }

}
