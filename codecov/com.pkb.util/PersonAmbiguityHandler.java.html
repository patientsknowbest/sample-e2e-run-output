package com.pkb.util;

import com.pkb.user.entity.PKBPerson;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.invoke.MethodHandles;
import java.util.List;

public class PersonAmbiguityHandler {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    public static Option<PKBPerson> getPersonOrNone(Either<PKBPerson, List<PKBPerson>> personsFromId, String lookupType) {
        if (personsFromId == null) {
            return Option.none();
        }
        if (personsFromId.isLeft()) {
            return Option.of(personsFromId.getLeft());
        }

        return getFirst(personsFromId.get(), lookupType);
    }

    public static Option<PKBPerson> getMaybePersonOrNone(Either<Option<PKBPerson>, List<PKBPerson>> personsFromId, String lookupType) {
        if (personsFromId == null) {
            return Option.none();
        }
        if (personsFromId.isLeft()) {
            return personsFromId.getLeft();
        }

        return getFirst(personsFromId.get(), lookupType);
    }

    private static Option<PKBPerson> getFirst(List<PKBPerson> input, String lookupType) {
        var first = input.get(0);
        LOGGER.error("PHR-5705: Got multiple records for {}, defaulting to first one: {}!", lookupType, first.getId());
        return Option.of(first);
    }

    public static boolean personWithIdExists(Either<Option<PKBPerson>, List<PKBPerson>> personsFromId) {
        return singlePersonWithIdExists(personsFromId) || multiplePersonsWithIdExist(personsFromId);
    }

    private static boolean singlePersonWithIdExists(Either<Option<PKBPerson>, List<PKBPerson>> personsFromId) {
        return personsFromId.isLeft() && personsFromId.getLeft().isDefined();
    }

    private static boolean multiplePersonsWithIdExist(Either<Option<PKBPerson>, List<PKBPerson>> personsFromId) {
        return personsFromId.isRight();
    }

}
