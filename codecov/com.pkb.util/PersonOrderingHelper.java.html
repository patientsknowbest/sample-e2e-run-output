package com.pkb.util;

import com.pkb.user.entity.PKBPerson;

import java.util.Comparator;

public class PersonOrderingHelper {

    private OrderingEnum ordering = OrderingEnum.Ascending;

    private PersonCompareFieldEnum field = PersonCompareFieldEnum.Name;

    public OrderingEnum getOrdering() {
        return ordering;
    }

    public void setOrdering(OrderingEnum ordering) {
        this.ordering = ordering;
    }

    public PersonCompareFieldEnum getField() {
        return field;
    }

    public void setField(PersonCompareFieldEnum field) {
        this.field = field;
    }

    public Comparator<PKBPerson> getComparator() {
        Comparator<PKBPerson> byField = field.getComparator();
        if (ordering == OrderingEnum.Descending) {
            return byField.reversed();
        }
        return byField;
    }
}
