package com.pkb.util;

import java.time.Instant;
import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.util.function.Function;

public class Constants {

    public static final ZoneId APPLICATION_TZ = ZoneId.of("Europe/London");
    public static final String NHS_LOGIN_CORRELATION_ID = "nhs.login.correlation.id";
    public static final String NHS_LOGIN_IDS = "nhs.login.ids";
    public static final String NHS_LOGIN_INFO = "nhs.login.info";
    private static final long POSTGRES_DATE_NEGATIVE_INFINITY_LONG = -9223372036832400000L;
    public static final Instant POSTGRES_DATE_NEGATIVE_INFINITY_INSTANT = Instant.ofEpochMilli(POSTGRES_DATE_NEGATIVE_INFINITY_LONG);
    public static final ZonedDateTime POSTGRES_DATE_NEGATIVE_INFINITY_ZONED_UTC = POSTGRES_DATE_NEGATIVE_INFINITY_INSTANT.atZone(ZoneId.of("UTC"));

    public static final Function<String,String> PKB_PERSON_NAME_SEARCH_INDEXED_EXPRESSION = tablename ->
    String.format(
            "to_tsvector('simple',"+
              "coalesce (%s.title       , '')||'  '||"+
              "coalesce (%s.firstName   , '')||'  '||"+
              "coalesce (%s.middleName  , '')||'  '||"+
              "coalesce (%s.lastName    , ''))", tablename, tablename, tablename, tablename);

    private Constants() {
    }

    public static boolean isNegativeInfinity(Instant instant) {
        return POSTGRES_DATE_NEGATIVE_INFINITY_INSTANT.equals(instant);
    }
}
