package com.pkb.util;

import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;

import java.util.Arrays;
import java.util.Collection;
import java.util.Map;
import java.util.regex.Pattern;
import java.util.stream.Collectors;
import java.util.stream.Stream;

public final class TextParserUtil {
    private static final Pattern ALLOWED_LETTERS = Pattern.compile("^([a-zA-Z.0-9_])*$");

    public static @NotNull Map<String, String> keyValuesPairStringsToMap(@NotNull Collection<String> keyValuePairs, @NotNull String separator) {
        return keyValuePairs.stream()
                .map(StringUtils::trimToEmpty)
                .filter(StringUtils::isNoneBlank)
                .flatMap(keyValuePair -> {
                    String[] split = Arrays.stream(keyValuePair.split(separator))
                            .map(StringUtils::trimToEmpty)
                            .filter(StringUtils::isNoneBlank)
                            .filter($ -> ALLOWED_LETTERS.matcher($).find())
                            .toArray(String[]::new);

                    return split.length == 2 ? Stream.<String[]>of(split) : Stream.empty();
                })
                .collect(Collectors.toMap($ -> $[0], $1 -> $1[1]));
    }

    private TextParserUtil() {
    }
}
