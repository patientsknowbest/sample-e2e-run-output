package com.pkb.util;

import com.pkb.common.config.PhrConfig;
import org.springframework.beans.factory.annotation.Autowired;

public class ConfigBean {

    @Autowired
    private PhrConfig config;

    public String getInstituteLogoBaseDir() {
        return config.getInstituteLogoBaseDir();
    }

    public boolean isTestServer() {
        return config.isDisplayOfSensitiveErrorInformationEnabled();
    }

    public long getUploadMaxFileSize() {
        return config.getUploadMaxFileSize();
    }

    public String getUploadMaxFileTextGB() {
        return config.getUploadMaxFileTextGB();
    }

    public boolean isTrackingEnabled() {
        return config.isTrackingEnabled();
    }

    public int getTrackingSiteId() {
        return config.getMatomoSiteId();
    }

    public String getTrackingServerBaseUrl() {
        return config.getMatomoBaseUrl();
    }
}
