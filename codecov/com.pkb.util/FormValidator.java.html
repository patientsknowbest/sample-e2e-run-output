package com.pkb.util;

import com.pkb.common.base.Error;
import org.jetbrains.annotations.NotNull;

import java.util.ArrayList;
import java.util.List;
import java.util.Optional;
import java.util.function.Function;
import java.util.stream.Collectors;

import static com.google.common.collect.ImmutableList.copyOf;

public class FormValidator<FM> {

    private static class FieldValidation<FM, F> {

        private final String fieldName;

        private final Function<FM, F> fieldAccessor;

        private final List<Validator<F>> validators = new ArrayList<>(1);

        FieldValidation(String fieldName, Function<FM, F> fieldAccessor) {
            this.fieldName = fieldName;
            this.fieldAccessor = fieldAccessor;
        }

        List<Error> performValidation(FM formModel) {
            return validators.stream().map(validator -> validator.validate(fieldAccessor.apply(formModel)))
                    .filter(Optional::isPresent)
                    .map(Optional::get)
                    .collect(Collectors.toList());
        }

    }

    public static class Builder<FM> {

        private List<Validator<FM>> globalValidators = new ArrayList<>();

        private List<FieldValidation<FM, ?>> fieldValidators = new ArrayList<>();

        public Builder<FM> addGlobalValidator(Validator<FM> validator) {
            globalValidators.add(validator);
            return this;
        }

        public <F> Builder<FM> addFieldValidator(String fieldName, Function<FM, F> fieldAccessor, Validator<F> validator) {
            FieldValidation<FM, F> validation = null;
            for (FieldValidation<FM, ?> candidate: fieldValidators) {
                if (candidate.fieldName.equals(fieldName)) {
                    validation = (FieldValidation<FM, F>) candidate;
                    break;
                }
            }
            if (validation == null) {
                validation = new FieldValidation<>(fieldName, fieldAccessor);
                fieldValidators.add(validation);
            }
            validation.validators.add(validator);
            return this;
        }

        public <F> Builder<FM> addFieldValidators(String fieldName, Function<FM, F> fieldAccessor, List<Validator<F>> validators) {
            validators.forEach(v -> addFieldValidator(fieldName, fieldAccessor, v));
            return this;
        }

        public FormValidator<FM> build() {
            return new FormValidator<>(globalValidators, fieldValidators);
        }
    }

    public static <FM> Builder<FM> builder() {
        return new Builder<>();
    }

    private final List<Validator<FM>> globalValidators;

    private final List<FieldValidation<FM, ?>> fieldValidators;

    private FormValidator(@NotNull List<Validator<FM>> globalValidators, @NotNull List<FieldValidation<FM, ?>> fieldValidators) {
        this.globalValidators = copyOf(globalValidators);
        this.fieldValidators = copyOf(fieldValidators);
    }

    public FormValidationErrors validate(@NotNull FM formModel) {
        FormValidationErrors errors = new FormValidationErrors();
        FormValidationErrors globalErrors = globalValidators.stream().reduce(
                errors,
                (errs, validator) -> validator.validate(formModel).map(errs::addError).orElse(errs),
                (oldErrors, newErrors) -> newErrors
        );
        return fieldValidators.stream().reduce(globalErrors,
                (errs, fieldValidation) -> errs.addFieldErrors(fieldValidation.fieldName, fieldValidation.performValidation(formModel)),
                (oldErrors, newErrors) -> newErrors);
    }

}
