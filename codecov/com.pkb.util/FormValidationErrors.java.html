package com.pkb.util;

import com.pkb.common.base.Error;
import org.jetbrains.annotations.NotNull;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Objects;

import static java.util.Collections.emptyList;

public final class FormValidationErrors {

    public static final FormValidationErrors EMPTY = new FormValidationErrors();

    private final List<Error> globalErrors;

    private final Map<String, List<Error>> fieldErrors;

    public FormValidationErrors() {
        this(new ArrayList<>(0), new HashMap<>(0));
    }

    private FormValidationErrors(List<Error> globalErrors, Map<String, List<Error>> fieldErrors) {
        this.globalErrors = globalErrors;
        this.fieldErrors = fieldErrors;
    }

    public List<Error> getGlobalErrors() {
        return globalErrors;
    }

    public Map<String, List<Error>> getFieldErrors() {
        return fieldErrors;
    }

    public FormValidationErrors addError(@NotNull Error globalError) {
        List<Error> newGlobalErrors = new ArrayList<>(globalErrors.size() + 1);
        newGlobalErrors.addAll(globalErrors);
        newGlobalErrors.add(globalError);
        return new FormValidationErrors(newGlobalErrors, fieldErrors);
    }

    public FormValidationErrors addFieldError(@NotNull String fieldName, @NotNull Error error) {
        Map<String, List<Error>> newFieldErrors = new HashMap<>(fieldErrors);
        List<Error> oldErrors = newFieldErrors.computeIfAbsent(fieldName, k -> emptyList());
        List<Error> newErrors = new ArrayList<>(oldErrors.size() + 1);
        newErrors.addAll(oldErrors);
        newErrors.add(error);
        newFieldErrors.put(fieldName, newErrors);
        return new FormValidationErrors(globalErrors, newFieldErrors);
    }

    public FormValidationErrors addFieldErrors(@NotNull String fieldName, @NotNull List<Error> errors) {
        return errors.stream().reduce(this,
                (allErrors, err) -> allErrors.addFieldError(fieldName, err),
                (oldErr, newErr) -> newErr);
    }

    public boolean isEmpty() {
        return globalErrors.isEmpty() && fieldErrors.values().stream().allMatch(List::isEmpty);
    }

    @Override public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (!(o instanceof FormValidationErrors)) {
            return false;
        }
        FormValidationErrors that = (FormValidationErrors) o;
        return globalErrors.equals(that.globalErrors) &&
                fieldErrors.equals(that.fieldErrors);
    }

    @Override public int hashCode() {
        return Objects.hash(globalErrors, fieldErrors);
    }

    @Override
    public String toString() {
        if (isEmpty()) {
            return "FormValidationErrors: no errors";
        }
        StringBuilder sb = new StringBuilder("FormValidationErrors: global errors:\n");
        globalErrors.forEach(error -> sb.append("\t").append(error).append("\n"));
        fieldErrors.forEach((fieldName, errors) -> {
            sb.append("field [").append(fieldName).append("] errors:\n");
            errors.forEach(error -> sb.append(error).append("\n"));
        });
        return sb.toString();
    }

    public FormValidationErrors addErrors(List<Error> globalErrors) {
        if (globalErrors.isEmpty()) {
            return this;
        }
        List<Error> newGlobalErrors = new ArrayList<>(this.globalErrors.size() + globalErrors.size());
        newGlobalErrors.addAll(this.globalErrors);
        newGlobalErrors.addAll(globalErrors);
        return new FormValidationErrors(newGlobalErrors, fieldErrors);
    }

    public FormValidationErrors plus(@NotNull FormValidationErrors additionalErrors) {
        if (additionalErrors.isEmpty()) {
            return this;
        }
        FormValidationErrors errors = addErrors(additionalErrors.getGlobalErrors());
        return additionalErrors.getFieldErrors().entrySet().stream().reduce(errors,
                (oldErrors, entry) -> oldErrors.addFieldErrors(entry.getKey(), entry.getValue()),
                (oldErrors, newErrors) -> newErrors);
    }

}
