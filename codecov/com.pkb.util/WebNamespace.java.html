package com.pkb.util;

/**
 * Holds web namespaces which are mainly used for building web URLs within service classes.
 */
public enum WebNamespace {
    ALLERGIES,
    AUTH,
    DEVICES,
    DIAGNOSES,
    DIARY,
    FILE,
    LIBRARY,
    MEDIA,
    PATIENT,
    TEST;

    @Override
    public String toString() {
        return this.name().toLowerCase();
    }
}
