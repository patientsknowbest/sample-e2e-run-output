package com.pkb.util;

import com.pkb.api.internal.exception.mapper.RequestValidationExceptionMapper;
import com.pkb.api.util.RestExceptionMapper;
import com.pkb.api.util.SharingDisabledExceptionMapper;
import com.pkb.api.util.ValidationExceptionMapper;
import com.pkb.servlets.filter.ApiCorsHeadersFilter;
import io.swagger.jaxrs.config.BeanConfig;
import io.swagger.models.Info;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.core.io.ClassPathResource;
import org.springframework.core.io.Resource;
import org.springframework.core.io.support.PropertiesLoaderUtils;
import org.springframework.web.filter.RequestContextFilter;

import javax.ws.rs.core.Application;
import java.io.IOException;
import java.util.HashSet;
import java.util.Properties;
import java.util.Set;

/**
 * Created by mvarga on 1/11/2017. (c) 2017 Patients Know Best
 */


public class RestApplication extends Application {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    protected Properties properties;

    private Set<Object> singletons = new HashSet<>();

    private void loadSwaggerProperties(String path) throws IOException {
        Resource resource = new ClassPathResource(path);
        properties = PropertiesLoaderUtils.loadProperties(resource);
    }

    protected RestApplication(String propertiesPath) throws IOException {
        singletons.add(new ApiCorsHeadersFilter());
        singletons.add(new RequestContextFilter());
        singletons.add(new PKBObjectMapperProvider());
        singletons.add(new RestExceptionMapper());
        singletons.add(new ValidationExceptionMapper());
        singletons.add(new RequestValidationExceptionMapper());
        singletons.add(new SharingDisabledExceptionMapper());
        loadSwaggerProperties(propertiesPath);
        createBeanConfig();
    }

    @Override
    public Set<Class<?>> getClasses() {
        Set<Class<?>> resources = new HashSet<>();
        resources.add(io.swagger.jaxrs.listing.ApiListingResource.class);
        resources.add(io.swagger.jaxrs.listing.SwaggerSerializers.class);
        return resources;
    }

    /**
     * retrieves values from swagger.properties
     *
     * @param key property key
     * @return returns value
     */
    protected String getValue(String key) {

        String value = (String) properties.get(key);

        LOGGER.info("retrieving value: {} for property: {}", value, key);

        return value;
    }


    @Override
    public Set<Object> getSingletons() {
        return singletons;
    }

    private BeanConfig createBeanConfig() {
        BeanConfig beanConfig = new BeanConfig();
        beanConfig.setUsePathBasedConfig(true);
        beanConfig.setVersion(getValue("version"));
        beanConfig.setBasePath(getValue("basePath"));
        beanConfig.setResourcePackage(getValue("packages"));
        beanConfig.setScan(true);
        beanConfig.setPrettyPrint(true);
        Info info = new Info();
        info.setTitle(getValue("title"));
        info.setVersion(getValue("info.version"));
        beanConfig.setInfo(info);
        return beanConfig;
    }
}
