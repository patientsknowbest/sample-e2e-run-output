package com.pkb.util;

import com.pkb.crypto.document.Document;
import io.vavr.collection.HashMap;

import java.util.Optional;
import java.util.function.Function;

public class EncryptedDocumentMapperHelper {
    public final static String MEDIA_TYPE = "mediaType";
    public final static Function<Document, String> MEDIA_TYPE_PROVIDER = doc -> doc.metadata().get(MEDIA_TYPE);
    public final static String ACCOUNT_ID = "accountId";
    public final static Function<Document, Long> ACCOUNT_ID_PROVIDER = doc -> {
        var maybeAccountId = doc.metadata().get(ACCOUNT_ID);
        if (maybeAccountId == null) {
            return null;
        }
        return Long.parseLong(maybeAccountId);
    };
    public final static String AUTHOR_PERSON_ID = "authorPersonId";
    public final static Function<Document, Optional<Long>> AUTHOR_PERSON_ID_PROVIDER = doc -> Optional.ofNullable(doc.metadata().get(AUTHOR_PERSON_ID)).map(Long::parseLong);
    private final static String KIND = "kind";

    public static HashMap<String, String> genericMetadata(String mediaType, Long authorId, String kind) {
        return genericMetadata(mediaType, authorId == null ? null : Long.toString(authorId), kind);
    }

    public static HashMap<String, String> genericMetadata(String mediaType, String authorId, String kind) {
        HashMap<String, String> ret = HashMap.of(MEDIA_TYPE, mediaType,
                KIND, kind);
        return authorId == null ? ret : ret.put(AUTHOR_PERSON_ID, authorId);
    }
}
