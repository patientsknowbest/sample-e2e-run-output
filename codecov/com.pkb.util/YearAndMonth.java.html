package com.pkb.util;

import com.pkb.emis.appointment.model.EmisAvailableAppointmentSlots;

import java.time.DayOfWeek;
import java.time.LocalDate;
import java.time.YearMonth;
import java.time.format.DateTimeFormatter;
import java.time.temporal.ChronoUnit;
import java.time.temporal.TemporalAdjusters;
import java.util.List;

public class YearAndMonth {

    private static final DateTimeFormatter monthAndYearFormatter = DateTimeFormatter.ofPattern("MMM yyyy");

    private static final DateTimeFormatter dayMonthAndYearFormatter = DateTimeFormatter.ofPattern("EEE d MMMM yyyy");

    private YearMonth yearMonth;

    private List<EmisAvailableAppointmentSlots.AvailableAppointmentSlot> availableAppointmentSlots;

    public YearAndMonth(YearMonth yearMonth, List<EmisAvailableAppointmentSlots.AvailableAppointmentSlot> availableAppointmentSlots) {
        this.yearMonth = yearMonth;
        this.availableAppointmentSlots = availableAppointmentSlots;
    }

    public LocalDate getFirstDayOfTheMonth() {
        return yearMonth.atDay(1);
    }

    public String getCurrentMonthAndYear() {
        return getFirstDayOfTheMonth().format(monthAndYearFormatter);
    }

    public String getCurrentDateForADay(int day) {
        return yearMonth.atDay(day).format(dayMonthAndYearFormatter);
    }

    public List<Integer> getNumberOfWeeks() {
        LocalDate calendarStart = getFirstDayOfTheMonth().with(TemporalAdjusters.previousOrSame(DayOfWeek.MONDAY));
        LocalDate calendarStop = yearMonth.atEndOfMonth().with(TemporalAdjusters.next(DayOfWeek.MONDAY));
        int weeks = Math.toIntExact(ChronoUnit.WEEKS.between(calendarStart, calendarStop));

        return io.vavr.collection.List.rangeClosed(1, weeks)
                .toJavaList();
    }

    public List<String> getNumberOfDaysInAWeek(int week) {
        int firstDayOfWeek = getFirstDayOfTheMonth().getDayOfWeek().getValue() - 1;
        int startingDay = (week - 1) * 7 + 1;
        int endingDay = startingDay + 7;
        return io.vavr.collection.List.range(startingDay, endingDay)
                .map(day -> day - firstDayOfWeek)
                .map(day -> day < 1 || day > yearMonth.lengthOfMonth() ? "" : day.toString())
                .toJavaList();
    }

    public List<EmisAvailableAppointmentSlots.AvailableAppointmentSlot> getAvailableAppointmentSlotsForADay(String day) {
        return io.vavr.collection.List.ofAll(availableAppointmentSlots)
                .filter(aa -> !day.isEmpty() && yearMonth.atDay(Integer.parseInt(day)).isEqual(LocalDate.from(aa.getStartTime())))
                .toJavaList();
    }

    public boolean currentDayHasAnAppointment(String day) {
        return !getAvailableAppointmentSlotsForADay(day).isEmpty();
    }

    public int getCurrentMonth() {
        return getFirstDayOfTheMonth().getMonthValue();
    }

    public int getCurrentYear() {
        return getFirstDayOfTheMonth().getYear();
    }

    public int getPastMonth() {
        return getFirstDayOfTheMonth().minusMonths(1).getMonthValue();
    }

    public int getPastYear() {
        return getFirstDayOfTheMonth().minusMonths(1).getYear();
    }

    public int getFutureMonth() {
        return getFirstDayOfTheMonth().plusMonths(1).getMonthValue();
    }

    public int getFutureYear() {
        return getFirstDayOfTheMonth().plusMonths(1).getYear();
    }
}
