package com.pkb.util;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.PersonAuthInfo;
import com.pkb.common.datetime.DateTimeService;

import java.time.Instant;
import java.time.ZoneId;
import java.time.format.DateTimeFormatter;
import java.util.Locale;
import java.util.Optional;

public class SupportCodeGenerator {

    private final static DateTimeFormatter SUPPORT_CODE_FORMATTER = DateTimeFormatter.ofPattern("yyyyMMddHHmmss")
            .withLocale(Locale.UK)
            .withZone(ZoneId.of("UTC"));

    private final DateTimeService dateTimeService;

    public SupportCodeGenerator(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    public String getSupportCode(EHRRequestContext requestContext) {
        return getSupportCode(requestContext.getMaybeAccessingUserId().orElse(null), requestContext.getContextUserId().orElse(null));
    }

    public String getSupportCode(Optional<PersonAuthInfo> maybeLoggedInUser, Long maybeContextUserId) {
        return getSupportCode(maybeLoggedInUser.map(PersonAuthInfo::getId).orElse(null), maybeContextUserId);
    }

    public String getSupportCode(Long loggedInUserId, Long contextUserId) {
        return getSupportCode(loggedInUserId == null ? null : Long.toString(loggedInUserId), contextUserId == null ? null : Long.toString(contextUserId));
    }

    private String getSupportCode(String loggedInUserId, String contextUserId) {
        return getSupportCode(loggedInUserId, contextUserId, dateTimeService.now());
    }

    public static String getSupportCode(String loggedInUserId, String contextUserId, Instant now) {
        StringBuilder codeBuilder = new StringBuilder(40);
        if (loggedInUserId != null) {
            codeBuilder.append(loggedInUserId);
        }
        codeBuilder.append("n");
        if (contextUserId != null) {
            codeBuilder.append(contextUserId);
        }
        codeBuilder.append("a");
        codeBuilder.append(SUPPORT_CODE_FORMATTER.format(now));
        return codeBuilder.toString();
    }
}
