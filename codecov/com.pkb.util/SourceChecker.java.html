package com.pkb.util;

import com.google.common.annotations.VisibleForTesting;
import com.google.common.collect.Sets;
import com.pkb.BaseVersionDetails;
import com.pkb.HasVersionDetails;
import com.pkb.app.entity.EHRData;
import com.pkb.common.config.PkbPluginConfig;
import com.pkb.entities.enums.Route;
import io.prometheus.client.Summary;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Set;

import static com.pkb.entities.enums.Route.CORE_DEVICES;
import static com.pkb.entities.enums.Route.EMIS_PFS;
import static com.pkb.entities.enums.Route.SCISTORE;

public class SourceChecker {

    private final Logger logger;

    private static final Logger DEFAULT_LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private static PkbPluginConfig config;

    // Note that these are to be checked and eventually deleted
    private static final Set<Route> ROUTES_WITHOUT_SOURCE_IDS = Sets.immutableEnumSet(CORE_DEVICES,SCISTORE,EMIS_PFS);

    private static final Summary SOURCE_CHECK_DURATION = Summary.build()
            .name("pkb_phr_phrplugin_sourcechecker_duration")
            .help("Time spent inside SourceChecker")
            .register();

    public SourceChecker() {
        logger = DEFAULT_LOGGER;
    }

    @VisibleForTesting
    public SourceChecker(Logger logger, PkbPluginConfig config) {
        this.logger = logger;
        SourceChecker.config = config;
    }

    // Awkward injection from StaticContextInitializer
    // Because @Inject doesn't work anywhere inside HibernateEJBInterceptors
    public static void setConfig(PkbPluginConfig config) {
        SourceChecker.config = config;
    }

    public void check(Object newEntity){
        SOURCE_CHECK_DURATION.time(() -> {
            RouteAndSource newSave = toRouteAndSource(newEntity);
            if (newSave!=null) {
                doCheck(newSave);
            }
        });
    }

    public void check(Object updatedEntity, Object[] originalEntityFieldValues, String[] originalEntityFieldNames) {
        SOURCE_CHECK_DURATION.time(() -> {
            RouteAndSource update = toRouteAndSource(updatedEntity);
            if (update != null) {
                RouteAndSource original = toRouteAndSource(updatedEntity, originalEntityFieldValues, originalEntityFieldNames);
                if (original != null && update.isWorseThan(original)) {
                    doCheck(update);
                }
            }
        });
    }

    private void doCheck(RouteAndSource entity) {

        if (entity.isRouteNull()) {
            String message = "PHRZ-414,PNX-108 Route is null for data point!!! Fix code path to set it, and backfill DB!";
            if (config.isExceptionForNullRouteEnabled()) {
                throw new IllegalStateException(message);
            }
            LogWarningHelper.warnWithStacktrace(logger,message);
        }

        if (entity.isRouteNotCaptured()) {
            String message = "PHRZ-414,PNX-108: Route is NOT_CAPTURED for data point!!! Fix code path to set it, and backfill DB!";
            if (config.isExceptionForNotCapturedRouteEnabled()) {
                throw new IllegalStateException(message);
            }
            LogWarningHelper.warnWithStacktrace(logger,message);
        }

        if (entity.isMissingSourceIds()) {
            throw new IllegalStateException("PHRZ-105,PHRZ-104,PNX-108: Source is null for data point!!! Fix code path to set it, and backfill DB!");
        }
    }

    static class RouteAndSource {
        private Route route;
        private Long sourcePersonId;
        private Long sourceTeamId;
        private Long sourceOrgId;

        RouteAndSource(EHRData e) {
            route = e.getRoute();
            sourcePersonId = e.getSourcePersonId();
            sourceTeamId = e.getSourceTeamId();
            sourceOrgId = e.getSourceOrgId();
        }

        RouteAndSource(BaseVersionDetails bvd) {
            route = bvd.getAccessRoute();
            sourcePersonId = bvd.getSourcePersonId();
            sourceTeamId = bvd.getSourceTeamId();
            sourceOrgId = bvd.getSourceOrgId();
        }

        RouteAndSource(Object updatedEntity, Object[] fieldValues,String[] fieldNames) {
            // Made less clear to get around inspection error "Chain of 'instanceof' checks indicates abstraction failure"
            if  (updatedEntity instanceof HasVersionDetails) {
                for (int i = 0; i < fieldNames.length; ++i) {
                    if ("versionDetails".equals(fieldNames[i])) {
                        BaseVersionDetails baseVersionDetails = (BaseVersionDetails) fieldValues[i];
                        route = baseVersionDetails.getAccessRoute();
                        sourcePersonId = baseVersionDetails.getSourcePersonId();
                        sourceTeamId = baseVersionDetails.getSourceTeamId();
                        sourceOrgId = baseVersionDetails.getSourceOrgId();
                        break;
                    }
                }
            } else  {
                for (int i = 0; i < fieldNames.length; ++i) {
                    if ("route".equals(fieldNames[i]) || "accessRoute".equals(fieldNames[i])) {
                        route = (Route) fieldValues[i];
                    } else if ("sourcePersonId".equals(fieldNames[i])) {
                        sourcePersonId = (Long) fieldValues[i];
                    } else if ("sourceTeamId".equals(fieldNames[i])) {
                        sourceTeamId = (Long) fieldValues[i];
                    } else if ("sourceOrgId".equals(fieldNames[i])) {
                        sourceOrgId = (Long) fieldValues[i];
                    }
                }
            }
        }

        boolean isRouteNull() {
            return route == null;
        }

        boolean isRouteNotCaptured() {
            return route == Route.NOT_CAPTURED;
        }

        // See PHR-9841 re. discrepancy with EHRBean.isSourceDetailMissing
        boolean isMissingSourceIds() {
            return !ROUTES_WITHOUT_SOURCE_IDS.contains(route) && sourcePersonId==null && sourceTeamId==null && sourceOrgId==null;
        }

        // Dud routes are only acceptable if they are unchanged from before
        // Nothing can switch from non-null to null
        boolean isWorseThan(RouteAndSource original) {
            return !ROUTES_WITHOUT_SOURCE_IDS.contains(route) &&
                    (((route==null || route==Route.NOT_CAPTURED) && (route != original.route)) ||
                    (sourcePersonId==null && original.sourcePersonId!=null) ||
                    (sourceTeamId==null && original.sourceTeamId!=null) ||
                    (sourceOrgId==null && original.sourceOrgId!=null));
        }

        @Override
        public String toString() {
            return "{"+route+","+sourcePersonId+","+sourceTeamId+","+sourceOrgId+"}";
        }
    }

    private RouteAndSource toRouteAndSource(Object entity) {
        try {
            if (entity instanceof EHRData) {
                return new RouteAndSource((EHRData) entity);
            } else if (entity instanceof HasVersionDetails) {
                return new RouteAndSource(((HasVersionDetails) entity).getVersionDetails());
            }
        } catch (Exception e) {
            logger.error("PNX-108 error in SourceChecker",e);
        }
        return null;
    }

    private RouteAndSource toRouteAndSource(Object updatedEntity, Object[] fieldValues, String[] fieldNames) {
        try {
            return new RouteAndSource(updatedEntity,fieldValues,fieldNames);
        } catch (Exception e) {
            logger.error("PNX-108 error in SourceChecker",e);
        }
        return null;
    }


}
