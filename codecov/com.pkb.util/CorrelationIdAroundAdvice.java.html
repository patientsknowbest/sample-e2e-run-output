package com.pkb.util;

import com.pkb.app.entity.EHRRequestContext;

import javax.inject.Inject;
import javax.interceptor.AroundInvoke;
import javax.interceptor.InvocationContext;

public class CorrelationIdAroundAdvice {

    @Inject
    private CorrelationIdUtil correlationIdUtil;

    @AroundInvoke
    public Object aroundAdvice(InvocationContext invocation) throws Exception {

        // Track if we are responsible for removing the MDC value
        boolean owner = correlationIdUtil.maybeGet()
                .map((existing) -> false)
                .orElseGet(() -> extractCorrelationId(invocation.getParameters()));
        try {
            // Continue with the invocation
            return invocation.proceed();
        } finally {
            // Clear the MDC
            if (owner) {
                correlationIdUtil.remove();
            }
        }
    }

    private boolean extractCorrelationId(Object[] parameters) {
        if (parameters == null) {
            return false;
        }

        boolean found = false;

        // Loop through params, looking for non-null EHRRequestContext
        for (Object parameter : parameters) {
            if ((parameter instanceof EHRRequestContext)) {

                // If correlationId is set, use it
                EHRRequestContext requestContext = (EHRRequestContext) parameter;
                correlationIdUtil.put(requestContext.getCorrelationId());
                found = true;

                // Only one context parameter, so we can stop now
                break;
            }
        }

        return found;
    }
}
