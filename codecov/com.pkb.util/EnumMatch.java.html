package com.pkb.util;

import com.google.common.base.Strings;
import org.jetbrains.annotations.Nullable;

import java.lang.reflect.InvocationTargetException;

/**
 * This class checks whether a supplied value matches the .name or .value of an
 * enum and then returns the corresponding match or null This utility was
 * introduced to ensure enum matching compared both the .name and the .value
 * results of an enum. Much of our Enum API have no toString() method which
 * makes meaningful comparison difficult. It also means that swagger generated
 * API will always fail against out API Primarily, it helps consumers of our API
 * to not have to to find out the .value equivalents of our enum types
 *
 * @author vorekoya on 26/02/2016.
 */
public class EnumMatch {

    /**
     * @param enumValues
     *            Enum.values()
     * @param value
     *            target value. this is the value we are trying to match
     * @param <T>
     *            enum Type
     * @return matched enum value
     */
    @Nullable
    public static <T extends Enum> T match(T[] enumValues, String value) {

        for (T candidate : enumValues) {

            if (candidate.name().equalsIgnoreCase(value) // check if name
                    // matches
                    || candidate.toString().equalsIgnoreCase(value) // check if
                    // toString
                    // matches
                    || getValue(candidate, value)) { // check if getValue
                // matches

                return candidate;
            }
        }

        return null;

    }

    /**
     * @param enumValues
     *            Enum.values()
     * @param value
     *            target value
     * @param blankValue
     *            what to return if target value is an empty/null string
     * @param nullValue
     *            what to return if target value could not be matched
     * @param <T>
     *            enum type
     * @return matched enum value
     */
    public static <T extends Enum> T match(T[] enumValues, String value, T blankValue,
            T nullValue) {

        if (Strings.isNullOrEmpty(value)) {
            return blankValue;
        } else {
            T result = match(enumValues, value);
            return result == null ? nullValue : result;
        }
    }

    /**
     * This is not ideal as it uses reflection to find and call getValue this
     * method should be removed once all toStrings are properly implemented AND
     * ApiHelp code use enums consistently at the moment, some use Enum name and
     * others use value
     *
     * @param candidate
     *            target candidate
     * @param value
     *            target value
     * @param <T>
     *            enum class
     * @return returns true if a match is found and false otherwise
     */
    private static <T extends Enum> boolean getValue(T candidate, String value) {

        boolean result = false;

        try {
            String eValue = (String) candidate.getClass().getDeclaredMethod("getValue")
                    .invoke(candidate);

            if (!Strings.isNullOrEmpty(eValue) && eValue.equalsIgnoreCase(value)) {
                result = true;
            }
        } catch (NoSuchMethodException | InvocationTargetException | IllegalAccessException ignored) {
            //Its not a value.... Hide API design issues behing reflection and use exception as flow control what could go wrong...
        }
        return result;
    }
}
