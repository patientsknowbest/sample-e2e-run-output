package com.pkb.util;

import com.pkb.document.entity.Attachment;
import com.pkb.dto.ImmutableNhsAppUrlDto;
import com.pkb.dto.ImmutableUIAttachmentDto;
import com.pkb.dto.ImmutableUrlDto;
import com.pkb.dto.UIAttachmentDto;
import com.pkb.mappers.EntityDependency;

import javax.annotation.Nullable;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;

import static com.pkb.util.AttachmentUtil.Action.DOWNLOAD_FILE;
import static com.pkb.util.AttachmentUtil.Action.VIEW_IMAGE;
import static com.pkb.util.AttachmentUtil.Action.VIEW_PDF;
import static com.pkb.util.UrlUtil.buildHref;
import static com.pkb.util.WebNamespace.FILE;
import static java.lang.String.valueOf;
import static java.util.Collections.emptyMap;
import static java.util.stream.Collectors.toList;

public class AttachmentUtil {

    enum Action {
        DOWNLOAD_FILE("downloadFile"),
        VIEW_IMAGE("viewImage"),
        VIEW_PDF("viewPdf");

        private final String value;

        Action(String value) {
            this.value = value;
        }
    }

     public List<UIAttachmentDto> buildAttachments(List<Attachment> attachmentList,
                                                   Map<String, String> params,
                                                   EntityDependency<?> dependency) {
       Long contextUserId = dependency.loggedInRequestContext().getContextOrAccessingUserId();
        return attachmentList.stream()
                .map(attachment -> {
                    if (dependency.allowNHSAppAttachmentDownload()) {
                        return nhsAppAttachment(attachment, contextUserId);
                    }
                    if (attachment.isViewable()) {
                        return viewImage(attachment, params, contextUserId);
                    } else if (attachment.isPdf() || attachment.isConvertibleToPdf()) {
                        return dependency.iOSDevice() ?
                                downloadFile(attachment, contextUserId) :
                                viewPdf(attachment, params, contextUserId);
                    } else {
                        return downloadFile(attachment, contextUserId);
                    }
                })
                .collect(toList());
    }

    private UIAttachmentDto downloadFile(Attachment attachment, @Nullable Long contextUserId) {
        return attachment(attachment, DOWNLOAD_FILE, true, emptyMap(), contextUserId);
    }

    private UIAttachmentDto viewImage(Attachment attachment,
                                    Map<String, String> params,
                                    @Nullable Long contextUserId) {
        return attachment(attachment, VIEW_IMAGE, false, params, contextUserId);
    }

    private UIAttachmentDto viewPdf(Attachment attachment,
                                  Map<String, String> params,
                                  @Nullable Long contextUserId) {
        return attachment(attachment, VIEW_PDF, true, params, contextUserId);
    }

    private UIAttachmentDto attachment(Attachment attachment,
                                     Action action,
                                     boolean escapeAmp,
                                     Map<String, String> params,
                                     @Nullable Long contextUserId) {
        return attachment(attachment, action.value, escapeAmp, params, contextUserId);
    }

    private UIAttachmentDto attachment(Attachment attachment,
                                     String action,
                                     boolean escapeAmp,
                                     Map<String, String> params,
                                     @Nullable Long contextUserId) {
        Map<String, String> parameters = standardParams(attachment);
        parameters.putAll(params);
        return ImmutableUIAttachmentDto.builder()
                .pkbUrl(ImmutableUrlDto.builder()
                        .href(buildHref(FILE, action, parameters, contextUserId))
                        .content(attachment.getFilename())
                        .isEscapeAmp(escapeAmp)
                        .build())
                .build();
    }

    private UIAttachmentDto nhsAppAttachment(Attachment attachment, @Nullable Long contextUserId) {
        Map<String, String> parameters = standardParams(attachment);
        return ImmutableUIAttachmentDto.builder()
                .nhsAppUrl(ImmutableNhsAppUrlDto.builder()
                        .content(attachment.getFilename())
                        .contentMimeType(attachment.getMediaType())
                        .nhsAppHref(buildHref(FILE, DOWNLOAD_FILE.value, parameters, contextUserId))
                        .build())
                .build();
    }

    private Map<String, String> standardParams(Attachment attachment) {
        Map<String, String> parameters = new LinkedHashMap<>();
        parameters.put("parentDocId", valueOf(attachment.getDocMetadataId()));
        parameters.put("fileName", attachment.getFilename());
        return parameters;
    }
}
