package com.pkb.util;

import com.fasterxml.jackson.databind.DeserializationFeature;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.SerializationFeature;
import com.fasterxml.jackson.datatype.guava.GuavaModule;
import com.fasterxml.jackson.datatype.jdk8.Jdk8Module;
import com.fasterxml.jackson.datatype.jsr310.JavaTimeModule;

import javax.ws.rs.ext.ContextResolver;
import javax.ws.rs.ext.Provider;


@Provider
public class PKBObjectMapperProvider implements ContextResolver<ObjectMapper> {

    private final ObjectMapper defaultObjectMapper;

    public PKBObjectMapperProvider() {
        defaultObjectMapper = createDefaultMapper();
    }

    private static ObjectMapper createDefaultMapper() {
        ObjectMapper result = new ObjectMapper();
        result.disable(SerializationFeature.WRAP_ROOT_VALUE);
        result.enable(SerializationFeature.INDENT_OUTPUT);
        result.configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);

        //configure how enums are handled
        result.enable(SerializationFeature.WRITE_ENUMS_USING_TO_STRING);
        result.enable(DeserializationFeature.READ_ENUMS_USING_TO_STRING);

        //accept single-valued list as array
        result.enable(DeserializationFeature.ACCEPT_SINGLE_VALUE_AS_ARRAY);

        //Add suppoort for Optional and other J8 related stuff
        result.registerModule(new Jdk8Module());
        // Add support for Guava types
        result.registerModule(new GuavaModule());
        // Add support for java.time.*
        result.registerModule(new JavaTimeModule());
        return result;
    }

    @Override
    public ObjectMapper getContext(Class<?> aClass) {
        return defaultObjectMapper;
    }
}
