package com.pkb.util;

import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;

import com.pkb.dto.UIAttachmentDto;
import com.pkb.service.datapoint.DataPointDTO;
import com.pkb.user.entity.PKBPerson;
import org.apache.commons.lang3.StringUtils;

import java.time.Instant;
import java.time.temporal.ChronoUnit;
import java.util.Comparator;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Objects;
import java.util.stream.Collectors;
import java.util.stream.Stream;

import static java.util.Collections.singletonList;

public class MedicationDTO extends DataPointDTO {

    public static final String MEDICATION_CODE = "(#Medication#)";
    public static final String DOSE_UNIT_CODE = "(#DoseUnit#)";
    private static final String INACTIVE_STATUS = "INACTIVE";

    /**
     * Comparators for MedicationDTO
     */
    public static final Comparator<MedicationDTO> BY_REVERSE_START_DATE_THEN_ALPHABETIC_NAME = Comparator.comparing(MedicationDTO::getStartDate,
            Comparator.nullsLast(Comparator.comparing(d -> d.truncatedTo(ChronoUnit.DAYS), Comparator.reverseOrder())))
            .thenComparing(MedicationDTO::getPrimaryMedicationText, String::compareToIgnoreCase);

    public static final Comparator<MedicationDTO> BY_END_DATE = Comparator.comparing(MedicationDTO::getEndDate,
            Comparator.nullsFirst(Comparator.comparing(d -> d.truncatedTo(ChronoUnit.DAYS))))
            .thenComparing(MedicationDTO::getPrimaryMedicationText, Comparator.comparing(String::toLowerCase, Comparator.reverseOrder()));

    private Long id;
    private String substance;
    private String dose;
    private Instant startDate;
    private Instant endDate;
    private String frequency;
    private String additionalInstructions;
    private PKBPerson notedBy;
    private String status;
    private Boolean patientOnMedication;
    private String medicationText;
    private boolean past;
    private List<UIAttachmentDto> attachments;

    private CodeableConcept medication;
    private CodeableConcept doseUnit;
    private DTOBaseFields baseFields = new DTOBaseFields();

    private SourceDetails source = new SourceDetails();

    public String getPrimaryMedicationText() {
        if (StringUtils.isNotBlank(medicationText)) {
            return medicationText;
        } else {
            return null2Empty(substance);
        }
    }

    public String getSecondaryMedicationText() {
        if (StringUtils.isNotBlank(medicationText)) {
            return "";
        } else {
            return Stream.of(dose, frequency)
                    .filter(Objects::nonNull)
                    .collect(Collectors.joining(" "));
        }
    }

    public String getCompleteMedicationText() {
        return getPrimaryMedicationText() + " " + getSecondaryMedicationText();
    }

    public String getSecondaryMedicationTextAndInstructions() {
        return Stream.of(getSecondaryMedicationText(), getAdditionalInstructions())
                .filter(StringUtils::isNotBlank)
                .collect(Collectors.joining(" "));
    }

    private String null2Empty(String s) {
        if (s == null) {
            return "";
        }

        return s;
    }

    public String getMedicationText() {
        return medicationText;
    }

    public void setMedicationText(String medicationText) {
        this.medicationText = medicationText;
    }

    public CodeableConcept getMedication() {
        return medication;
    }

    public void setMedication(CodeableConcept medication) {
        this.medication = medication;
    }

    public CodeableConcept getDoseUnit() {
        return doseUnit;
    }

    public void setDoseUnit(CodeableConcept doseUnit) {
        this.doseUnit = doseUnit;
    }

    public String getClinicalCodes() {
        LinkedHashMap<String, List<CodeableConcept>> map = new LinkedHashMap<>();
        map.put(" " + MEDICATION_CODE, singletonList(getMedication()));
        map.put(" " + DOSE_UNIT_CODE, singletonList(getDoseUnit()));

        return map.entrySet().stream().flatMap(e -> e.getValue()
                .stream()
                .filter(p -> p != null && !p.getSimpleCodesForDisplay().isEmpty())
                .map(v -> v.getSimpleCodesForDisplay() + e.getKey()))
                .collect(Collectors.joining(", "));
    }
