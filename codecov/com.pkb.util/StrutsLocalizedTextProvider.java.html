package com.pkb.util;

import com.opensymphony.xwork2.TextProvider;
import com.opensymphony.xwork2.TextProviderFactory;
import com.opensymphony.xwork2.inject.Container;
import com.pkb.action.BaseAction;
import io.vavr.collection.Stream;
import org.apache.struts2.dispatcher.Dispatcher;
import org.apache.struts2.dispatcher.filter.StrutsPrepareFilter;
import org.springframework.stereotype.Component;

/**
 * 
 * As it is the bridge between struts and spring. It works only inside struts action context. Keep in mind that we use {@link StrutsPrepareFilter } configured in web.xml mapped to
 * every request.
 *
 */
@Component
public class StrutsLocalizedTextProvider implements LocalizedTextProvider {
    private com.opensymphony.xwork2.TextProvider textProvider;

    /**
     * {@inheritDoc}
     */
    @Override
    public String getText(String key) {
        checkKey(key);
        return getInitializedTextProvider().getText(key);
    }

    /**
     * {@inheritDoc}
     */
    @Override
    public String getText(String key, Object... args) {
        checkKey(key);
        return getInitializedTextProvider().getText(key, Stream.of(args).asJava());
    }

    private void checkKey(String key) {
        // TODO (utamas) GDE-773 restore check
        //        checkArgument(isNoneBlank(trimToEmpty(key)), "I18N key cannot be blank");
    }

    private TextProvider getInitializedTextProvider() {
        if (textProvider == null) {
            synchronized (this) {
                if (textProvider == null) {
                    textProvider = getTextProvider();
                }
            }
        }
        return textProvider;
    }

    private static Container getStrutsContainer() {
        return Dispatcher.getInstance().getContainer();
    }

    public static com.opensymphony.xwork2.TextProvider getTextProvider() {
        TextProviderFactory factory = getStrutsContainer().getInstance(TextProviderFactory.class);
        return factory.createInstance(BaseAction.class);
    }

}
