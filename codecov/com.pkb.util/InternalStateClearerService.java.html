package com.pkb.util;

import com.pkb.common.ClearableInternalState;
import com.pkb.service.dataupload.emis.esv2.EmisEsService;
import com.pkb.service.state.InternalStateManager;
import org.slf4j.Logger;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.lang.invoke.MethodHandles;
import java.util.List;

import static org.slf4j.LoggerFactory.getLogger;

@Service
public class InternalStateClearerService implements ClearableInternalState {

    private static final Logger LOGGER = getLogger(MethodHandles.lookup().lookupClass());
    private final InternalStateManager internalStateManager;

    @Autowired
    private List<ClearableInternalState> internalStateHolders;

    @Autowired
    private EmisEsService emisEsService;

    public InternalStateClearerService(InternalStateManager internalStateManager) {
        this.internalStateManager = internalStateManager;
    }

    @Override
    public void clearState() {
        LOGGER.info("Clearing internal state for {} classes", internalStateHolders.size() + 2);

        internalStateManager.clearState();

        ResettablePooledLoOptimizerStateClearer.clearState();

        internalStateHolders.forEach(ClearableInternalState::clearState);

        emisEsService.clearEmisDownloadDirectory();

        LOGGER.info("Finished clearing internal states");
    }
}
