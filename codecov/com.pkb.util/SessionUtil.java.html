package com.pkb.util;

import com.google.common.annotations.VisibleForTesting;
import com.opensymphony.xwork2.ActionContext;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.interceptor.DynamicFeatures;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;

import java.util.Map;
import java.util.Optional;
import java.util.function.Supplier;

import static com.google.common.base.Preconditions.checkNotNull;
import static com.opensymphony.xwork2.ActionContext.getContext;
import static java.lang.invoke.MethodHandles.lookup;
import static org.slf4j.LoggerFactory.getLogger;

public class SessionUtil {
    private static final Logger LOGGER = getLogger(lookup().lookupClass());

    public static final String CONTEXT_RECIPIENT_ID = "recipientId";

    private final Supplier<Map<String, ?>> sessionSupplier;

    public SessionUtil() {
        this(() -> getContext().getSession());
    }

    public SessionUtil(Supplier<Map<String, ?>> sessionSupplier) {
        this.sessionSupplier = sessionSupplier;
    }

    public void setDynamicFeatures(@NotNull DynamicFeatures features) {
        setDynamicFeatures(features, getSessionMap());
    }

    public void setDynamicFeatures(@NotNull DynamicFeatures features, ActionContext context) {
        setDynamicFeatures(features, context.getSession());
    }

    public void setDynamicFeatures(@NotNull DynamicFeatures features, Map<String, ?> sessionMap) {
        checkNotNull(features, "Features cannot be null");

        Object prevFeatures = sessionMap.get("dynamicFeatures");

        if (prevFeatures == null) {
            LOGGER.info("Adding dynamic features to session: {}", features);
        } else {
            if (!features.equals(prevFeatures)) {
                LOGGER.info("Replacing dynamic features in session. Old=[{}]. New=[{}].", prevFeatures, features);
            }
        }

        getActionContext().getSession().put("dynamicFeatures", features);
    }

    public DynamicFeatures getDynamicFeatures() {
        return getDynamicFeatures(getSessionMap());
    }

    public DynamicFeatures getDynamicFeatures(ActionContext context) {
        return getDynamicFeatures(context.getSession());
    }

    public DynamicFeatures getDynamicFeatures(Map<String, ?> sessionMap) {
        return DynamicFeatures.class.cast(sessionMap.get("dynamicFeatures"));
    }

    private Map<String, ?> getSessionMap() {
        return sessionSupplier.get();
    }

    @VisibleForTesting
    ActionContext getActionContext() {
        return getContext();
    }
    
    public boolean renderNhsFooter() {
        return getDynamicFeatures() != null && getDynamicFeatures().renderNhsFooter();
    }

    public boolean renderNhsStyle() {
        return getDynamicFeatures() != null && getDynamicFeatures().renderNhsStyle();
    }

    /**
     * Get message recipient {@link PKBPerson} ID as a string
     */
    public Long getContextRecipientId() {
        Map<String, Object> session =  getActionContext().getSession();
        return (Long) session.get(CONTEXT_RECIPIENT_ID);
    }

    /**
     * Remove message recipient ID from context
     */
    public void clearContextRecipientId() {
        Map<String, Object> session =  getActionContext().getSession();
        session.remove(CONTEXT_RECIPIENT_ID);
    }

    /**
     * Sets context recipient {@link PKBPerson} ID from current context user
     */
    public void setContextRecipientId(LoggedInEHRRequestContext requestContext) {
        if (!requestContext.isUserInOwnAccount()) {
            setContextRecipientId(requestContext.getContextUserId());
        }
    }

    /**
     * Sets context recipient {@link PKBPerson} ID
     */
    public void setContextRecipientId(Optional<Long> contextRecipientId) {
        if (contextRecipientId.isPresent()) {
            Map<String, Object> session =  getActionContext().getSession();
            session.put(CONTEXT_RECIPIENT_ID, contextRecipientId.get());
        }
    }
}
