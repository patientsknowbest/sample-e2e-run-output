package com.pkb.util;

import com.google.common.collect.MapDifference;
import com.google.common.collect.Maps;
import com.pkb.app.entity.EHRData;

/*
    This class has been created because DeduplicatorManager wasn't able to use this static method from EHRData.
    Because MapDifference in DeduplicatorManager and EHRData was a different class in phr-remote-client and phr-service
    due to some obscure pacgaking issues. This static method was only used in the deduplicator so moving out this
    logic into a separate class made sense. In order to achieve this getEncryptedFields had to be made public in EHRData.

    This method should be moved back to EHRData once all EJBs are dropped and phr-remote-client is deleted

    Ticket: PHR-11487
 */
public class EHRDataComparatorUtil {

    public static MapDifference<String, Object> compareEncryptedFields(EHRData ed1, EHRData ed2) {
        return Maps.difference(ed1.getEncryptedFields(), ed2.getEncryptedFields());
    }
}
