package com.pkb.util;

import org.apache.commons.csv.CSVFormat;
import org.apache.commons.csv.CSVPrinter;
import org.apache.commons.csv.QuoteMode;

import java.io.Closeable;
import java.io.IOException;
import java.io.Writer;
import java.util.List;

public class UncheckedCsvPrinter implements Closeable {

    private final CSVPrinter delegate;

    UncheckedCsvPrinter(CSVPrinter delegate) {
        this.delegate = delegate;
    }

    public void print(Object value) {
        try {
            delegate.print(value);
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
    }

    public void println() {
        try {
            delegate.println();
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
    }

    @Override
    public void close() {
        try {
            delegate.close();
        } catch (IOException ignored) {
        }
    }

    public static class Factory {
        public UncheckedCsvPrinter getPrinter(Writer outputWriter, List<String> headers) {
            try {
                var format = (headers == null) ?
                        formatWithoutHeader() : formatWithHeader(headers);
                return new UncheckedCsvPrinter(new CSVPrinter(outputWriter, format));
            } catch (IOException e) {
                try {
                    outputWriter.close();
                } catch (IOException ignored) {
                }
                throw new RuntimeException(e);
            }
        }

        private CSVFormat formatWithHeader(List<String> headers){
            return CSVFormat.DEFAULT.withHeader(headers.toArray(new String[]{}))
                    .withQuoteMode(QuoteMode.ALL)
                    .withRecordSeparator('\n');
        }

        private CSVFormat formatWithoutHeader(){
            return CSVFormat.DEFAULT.withQuoteMode(QuoteMode.MINIMAL)
                    .withRecordSeparator('\n');
        }
    }
}
