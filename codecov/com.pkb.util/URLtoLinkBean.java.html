package com.pkb.util;

import org.jetbrains.annotations.NotNull;

import java.util.regex.Matcher;
import java.util.regex.Pattern;

/**
 * JavaBean format for easier use in JSPs
 *
 * @author ronnewman
 */
public class URLtoLinkBean {

    private static final Pattern BREAK_TAG_1_PATTERN = Pattern.compile("<br>");
    private static final Pattern BREAK_TAG_2_PATTERN = Pattern.compile("<br/>");
    private static final Pattern AND_PATTERN = Pattern.compile("&", Pattern.LITERAL);
    private static final Pattern BREAK_TAG_WITH_SPACE_PATTERN = Pattern.compile(" <br>");
    private static final Pattern BREAK_TAG_WITH_SPACE_PATTERN_2 = Pattern.compile(" <br/>");

    private String text;

    final private String unquotedLink = "(?<!\\&quot;)http(s?)://[^\\s ]+(?!\\&quot;)";
    final private Pattern unquotedLinkPattern = Pattern.compile(unquotedLink, Pattern.CASE_INSENSITIVE); // TODO: consider also adding Pattern.UNICODE_CASE when internationalizing

    final private String quotedLink = "(?<=\\&quot;)http(s?)://[^\\s ]+(?=\\&quot;)";
    final private Pattern quotedLinkPattern = Pattern.compile(quotedLink, Pattern.CASE_INSENSITIVE); // TODO: consider also adding Pattern.UNICODE_CASE when internationalizing

    //final private String youtubeLink="http(s?)://www.youtube[^\\s ]+";
    //final private Pattern youtubePattern = Pattern.compile(youtubeLink, Pattern.CASE_INSENSITIVE);  // TODO: consider also adding Pattern.UNICODE_CASE when internationalizing

    public String replaceURLsWithLinks() {
        //Introducing white spaces so that hyperlinks can be properly recognized
        String textWithWSChars = BREAK_TAG_1_PATTERN.matcher(text).replaceAll(" <br>");
        textWithWSChars = BREAK_TAG_2_PATTERN.matcher(textWithWSChars).replaceAll(" <br/>");
        Matcher unquotedUrl = unquotedLinkPattern.matcher(textWithWSChars);

        StringBuilder buffer = new StringBuilder();
        String nextMatch = "";
        while (unquotedUrl.find()) {
            nextMatch = unquotedUrl.group(); // just the matched portion without surrounding text

            // see if it matches an alternate pattern.  Must be more restrictive than the "any" pattern.
            //Matcher youtube = youtubePattern.matcher(nextMatch);  // TODO: reset the matcher with a new string instead of creating a new one?
            //if (youtube.find()) {
            if (false) { // disable youtube links until <iframes> can be displayed in the JSPs
                //TODO: escape ampersands in $0
                //any.appendReplacement(buffer, "$0");  // any intermediate characters between the matches plus the new matched portion
                //bufferLength=buffer.length();
                //buffer.delete(bufferLength-nextMatch.length(),bufferLength);  // snip off the new matched portion
                //youtube.appendReplacement(buffer, "<iframe width=\"425\" height=\"349\" src=\"$0\" frameborder=\"0\" allowfullscreen></iframe>");
            } else { // it's the most general case
                unquotedUrl.appendReplacement(buffer, composeLink(nextMatch));
            }
        }
        unquotedUrl.appendTail(buffer);

        StringBuilder buffer2 = new StringBuilder();
        Matcher quotedUrl = quotedLinkPattern.matcher(buffer.toString());
        while (quotedUrl.find()) {
            nextMatch = quotedUrl.group();
            quotedUrl.appendReplacement(buffer2, composeLink(nextMatch));
        }
        quotedUrl.appendTail(buffer2);

        String returnString = buffer2.toString();
        returnString = BREAK_TAG_WITH_SPACE_PATTERN.matcher(returnString).replaceAll("<br>");
        returnString = BREAK_TAG_WITH_SPACE_PATTERN_2.matcher(returnString).replaceAll("<br/>");
        return returnString;

    }

    @NotNull
    private String composeLink(String ref) {
        return "<a rel=\"noopener noreferrer\" href=\"" + ref + "\" target=\"_blank\">" + ref + "</a>";
    }

    public String getText() {
        return text;
    }

    public void setText(String text) {
        this.text = text;
    }

}
