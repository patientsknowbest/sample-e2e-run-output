package com.pkb.util;

import com.pkb.app.entity.SourceDetails;
import com.pkb.coding.entity.CodeableConcept;
import com.pkb.dto.UIAttachmentDto;

import java.time.Instant;
import java.util.Comparator;
import java.util.List;
import java.util.function.Function;
import java.util.stream.Collectors;

import static java.util.Collections.emptyList;
import static java.util.Objects.requireNonNullElse;
import static org.apache.commons.lang3.StringUtils.EMPTY;

/***
 * A group of medication DTOs. These can be grouped in different ways depending on use case.
 */

public class MedicationGroupDTO {

    private List<MedicationDTO> medicationDTOS;

    /** Also sorts the medicationDTOS */
    public MedicationGroupDTO(List<MedicationDTO> medicationDTOS) {
        this.medicationDTOS = requireNonNullElse(medicationDTOS, emptyList());

        sortMedications();
    }

    public List<MedicationDTO> getMedicationDTOs() {
        return medicationDTOS;
    }

    public void setMedicationDTOS(List<MedicationDTO> medicationDTOS) {
        this.medicationDTOS = medicationDTOS;
    }

    private void sortMedications() {

        medicationDTOS = getMedicationDTOs().stream()
                .sorted(
                        Comparator.comparing(
                                MedicationDTO::getEndDate,
                                Comparator.nullsLast(Instant::compareTo).reversed()))
                .collect(Collectors.toList());

    }

    public SourceDetails getSource() {
        return getMedicationDTOs().stream().findFirst().map(MedicationDTO::getSource).orElse(null);
    }

    public CodeableConcept getMedicationConcept() {
        return getPropertyValueOfMedicationWithLatestEndDateOrElse(MedicationDTO::getMedication, null);
    }

    private <M> M getPropertyValueOfMedicationWithLatestEndDateOrElse(Function<? super MedicationDTO, M> propertySupplier, M other) {
        return getMedicationDTOs().stream()
                .max(Comparator.comparing(MedicationDTO::getEndDate, Comparator.nullsFirst(Instant::compareTo)))
                .map(propertySupplier)
                .orElse(other);
    }

    public boolean isCurrent() {
        return medicationDTOS.stream().anyMatch(m -> !m.isPast());
    }

    public String getPrimaryMedicationText() {

        return getPropertyValueOfMedicationWithLatestEndDateOrElse(MedicationDTO::getPrimaryMedicationText, EMPTY);

    }

    public String getSecondaryMedicationText() {
        return getPropertyValueOfMedicationWithLatestEndDateOrElse(MedicationDTO::getSecondaryMedicationText, EMPTY);

    }

    public String getCompleteMedicationText() {
        return getPrimaryMedicationText() + " " + getSecondaryMedicationText();
    }

    public int getGroupSize() {
        return medicationDTOS.size();
    }

    public Long getFirstId() {
        return getMedicationDTOs().stream().findFirst().map(MedicationDTO::getId).orElse(null);
    }

    public boolean isSexualHealthConsentRequiredInGroup() {
        return medicationDTOS.stream().anyMatch(m -> m.getBaseFields().isSexualHealthConsentRequired());
    }

    public boolean isGeneralHealthConsentRequiredInGroup() {
        return medicationDTOS.stream().anyMatch(m -> m.getBaseFields().isGeneralHealthConsentRequired());
    }

    public boolean isMentalHealthConsentRequiredInGroup() {
        return medicationDTOS.stream().anyMatch(m -> m.getBaseFields().isMentalHealthConsentRequired());
    }

    public boolean isSocialCareConsentRequiredInGroup() {
        return medicationDTOS.stream().anyMatch(m -> m.getBaseFields().isSocialCareConsentRequired());
    }

    public Instant getFirstStartDate() {
        return getMedicationDTOs().stream()
                .min(
                        Comparator.comparing(MedicationDTO::getStartDate, Comparator.nullsLast(Instant::compareTo)))
                .map(MedicationDTO::getStartDate)
                .orElse(null);
    }

    public Instant getLastStartDate() {
        return getMedicationDTOs().stream()
                .max(
                        Comparator.comparing(MedicationDTO::getStartDate, Comparator.nullsLast(Instant::compareTo)))
                .map(MedicationDTO::getStartDate)
                .orElse(null);
    }

    public Instant getLastEndDate() {
        return getPropertyValueOfMedicationWithLatestEndDateOrElse(MedicationDTO::getEndDate, null);
    }

    public String getAdditionalInstructions() {
        return getPropertyValueOfMedicationWithLatestEndDateOrElse(MedicationDTO::getAdditionalInstructions, EMPTY);
    }

    public List<UIAttachmentDto> getAttachmentUrlDtos() {
        return getPropertyValueOfMedicationWithLatestEndDateOrElse(MedicationDTO::getAttachmentUrlDtos, emptyList());
    }

    public boolean isAnyPrivacyFlagInGroupEditable() {
        return medicationDTOS.stream().anyMatch(MedicationDTO::isPrivacyFlagEditable);
    }

    public boolean isAnyInGroupEditOrDeletable() {
        return medicationDTOS.stream().anyMatch(MedicationDTO::isEditOrDeletable);
    }
}
