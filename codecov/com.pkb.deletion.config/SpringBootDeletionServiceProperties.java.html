package com.pkb.deletion.config;

import com.pkb.deletion.model.DeletionServiceProperties;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.boot.context.properties.ConstructorBinding;
import org.springframework.boot.context.properties.bind.DefaultValue;

@ConfigurationProperties(prefix = "deletion.service")
public class SpringBootDeletionServiceProperties implements DeletionServiceProperties {

    private final String backupSchema;
    private final String deleteRequestSubscriptionName;
    private final String deleteResponseTopic;


    @ConstructorBinding
    public SpringBootDeletionServiceProperties(
            String backupSchema,
            String deleteRequestSubscriptionName,
            @DefaultValue("deleteResponse") String deleteResponseTopic
    ) {

        this.backupSchema = backupSchema;
        this.deleteRequestSubscriptionName = deleteRequestSubscriptionName;
        this.deleteResponseTopic = deleteResponseTopic;
    }

    @Override
    public String getBackupSchema() {
        return backupSchema;
    }

    @Override
    public String getDeleteRequestSubscriptionName() {
        return deleteRequestSubscriptionName;
    }

    @Override
    public String getDeleteResponseTopicName() {
        return deleteResponseTopic;
    }

}
