package com.pkb.deletion.config;

import com.pkb.deletion.model.DeletionServiceProperties;
import liquibase.integration.spring.SpringLiquibase;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Profile;

import javax.sql.DataSource;
import java.util.Map;

/**
 * Only import this configuration in non-spring-boot applications (i.e. PHR) Spring boot apps will
 * automatically have the relevant LB scripts run as part of the configuration in DeletionServiceAutoConfiguration
 * instead. In our other Spring boot applications, exposing a SpringLiquibase bean will suppress the auto-configured
 * Liquibase from Spring Boot, so we have to take another approach based on piggy-backing on an existing instance
 * of SpringLiquibase.
 */
@Configuration
public class DeletionServiceLiquibaseConfiguration {

    @Bean
    @Profile("!unittest & !integration-test")
    public SpringLiquibase deletionServiceLiquibase(DataSource dataSource, DeletionServiceProperties properties) {
        SpringLiquibase lb = new SpringLiquibase();
        lb.setDataSource(dataSource);
        configureDeletionServiceLiquibase(lb, properties);
        return lb;
    }

    public static void configureDeletionServiceLiquibase(SpringLiquibase lb, DeletionServiceProperties properties) {
        lb.setLiquibaseSchema(properties.getBackupSchema());
        lb.setChangeLogParameters(Map.of("deletion-service-schema", properties.getBackupSchema()));
        lb.setChangeLog("classpath:/db/deletion-service-changelog/db.changelog-master.yaml");
    }

}
