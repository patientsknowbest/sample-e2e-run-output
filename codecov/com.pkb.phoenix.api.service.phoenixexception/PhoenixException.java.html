package com.pkb.phoenix.api.service.phoenixexception;

/**
 * By introducing our custom exception class we can distinguish between validation errors and bugs.
 * We can also use this class to handle errors in a more sophisticated way on the user interface,
 * so that we can give more information to the user about the problem and suggest useful instructions.
 */

public abstract class PhoenixException extends RuntimeException {

    private static final long serialVersionUID = 1L;
    private final String errorCode;

    public PhoenixException(String errorCode) {
        this(errorCode, null, null);
    }

    public PhoenixException(String errorCode, String message) {
        this(errorCode, message, null);
    }

    public PhoenixException(String errorCode, String message, Throwable cause) {
        super(message, cause);
        this.errorCode = errorCode;
    }

    public String getErrorCode() {
        return errorCode;
    }

}
