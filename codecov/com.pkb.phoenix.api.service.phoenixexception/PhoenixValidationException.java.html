package com.pkb.phoenix.api.service.phoenixexception;

import java.util.List;

import static java.lang.String.join;

/**
 * Meant to indicate data validation issues.
 */
public class PhoenixValidationException extends PhoenixException {
    private static final long serialVersionUID = 1L;

    private PhoenixValidationException(String errorCode, String errors) {
        super(errorCode, errors);
    }

    public static final class InvalidTaskParameters extends PhoenixValidationException {
        private static final long serialVersionUID = 1L;

        private InvalidTaskParameters(String errorCode, String errors) {
            super(errorCode, errors);
        }

        public static PhoenixValidationException invalidTaskParameters(List<String> errors) {
            return new PhoenixValidationException("invalid.task.parameters",
                    String.format("Parameters are not valid. Errors: %s", join(",", errors)));
        }
    }
}
