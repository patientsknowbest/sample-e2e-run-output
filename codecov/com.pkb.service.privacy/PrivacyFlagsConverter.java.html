package com.pkb.service.privacy;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.ImmutablePrivacyFlagsOverride;
import com.pkb.datamodel.PrivacyFlagsOverride;
import io.vavr.control.Option;

public class PrivacyFlagsConverter {

    private DateTimeService dateTimeService;

    public void setDateTimeService(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    public PrivacyFlagsOverride convert(EHRRequestContext requestContext, IBaseDTO baseDto, boolean override) {
        SourceDetails source = new SourceDetails(requestContext);

        return ImmutablePrivacyFlagsOverride.builder()
                .uniqueId(baseDto.getBaseFields().getUniqueId())
                .accessRoute(requestContext.getAccessRoute())
                .changeDate(dateTimeService.now())
                .generalRequired(baseDto.getRequireGeneral())
                .mentalHealthRequired(baseDto.getRequireMentalHealth())
                .sexualHealthRequired(baseDto.getRequireSexualHealth())
                .socialCareRequired(baseDto.getRequireSocialCare())
                .sourcePersonId(source.getPersonId())
                .sourceTeamId(Option.of(source.getTeamId()))
                .sourceOrgId(Option.of(source.getOrgId()))
                .sourceText(Option.of(source.getText()))
                .override(override)
                .build();
    }


}
