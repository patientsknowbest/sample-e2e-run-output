package com.pkb.service.privacy;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.PrivacyFlagsOverride;
import com.pkb.datamodel.consent.PatientConsentDTO;
import com.pkb.domain.PrivacyFlagsService;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.radiology.entity.RadiologyImage;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import org.jetbrains.annotations.NotNull;

import java.util.List;
import java.util.Optional;
import java.util.Set;
import java.util.stream.Collectors;

import static com.pkb.service.privacy.PrivacyService.Notify.SEND_EMAIL;
import static java.util.Collections.emptySet;

public class PrivacyService extends TransactionManager {

    public enum Notify {
        SEND_EMAIL,
        DO_NOT_SEND_EMAIL
    }

    private final PKBEmailMessageManager emailMessageManager;
    private final PrivacyFlagsService privacyFlagsService;
    private final PrivacyFlagsConverter privacyFlagsConverter;

    public PrivacyService(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider,
                          @NotNull PKBEmailMessageManager emailMessageManager, @NotNull PrivacyFlagsService privacyFlagsService,
                          @NotNull PrivacyFlagsConverter privacyFlagsConverter) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.emailMessageManager = emailMessageManager;
        this.privacyFlagsService = privacyFlagsService;
        this.privacyFlagsConverter = privacyFlagsConverter;
    }

    public void handlePrivacyChange(@NotNull IBaseDTO baseDTO, @NotNull MenuDataType type, @NotNull PrivacyFlag newFlag, @NotNull EHRRequestContext context, long ownerAccountId,
            @NotNull PKBPerson patient, @NotNull PKBPerson actor) {
        handlePrivacyChange(baseDTO, type, newFlag, context, ownerAccountId, patient, actor, SEND_EMAIL);
    }

    public void handlePrivacyChange(@NotNull IBaseDTO baseDTO, @NotNull MenuDataType type, @NotNull PrivacyFlag newFlag, @NotNull EHRRequestContext context,
            long ownerAccountId, @NotNull PKBPerson patient, @NotNull PKBPerson actor, Notify notify) {
        isPrivacyFlagChanged(baseDTO, newFlag).ifPresent(originalFlagsWhenChanged -> {
            baseDTO.getBaseFields().setPrivacyFlag(newFlag);

            if (notify == SEND_EMAIL && !actor.getId().equals(patient.getId())) {
                emailMessageManager.notifyChangeOfPrivacy(patient, actor, baseDTO, type, newFlag);
            }
            PrivacyFlagsOverride pf = privacyFlagsConverter.convert(context, baseDTO, true);
            privacyFlagsService.savePrivacyFlagsAndAccessLog(pf, ownerAccountId, type, dateTimeService.now(),
                    context.getMaybeAccessingUserId().orElse(null), context.getAccessRoute(), getConsentId(context), context.getCorrelationId().toString());
        });
    }

    public void handlePrivacyChange(@NotNull IBaseDTO mainDto, @NotNull List<RadiologyImage> additionalDtos,
                                    @NotNull MenuDataType mainType, @NotNull MenuDataType additionalType,
                                    @NotNull PrivacyFlag newFlag, @NotNull EHRRequestContext context,
                                    long ownerAccountId, @NotNull PKBPerson patient,
                                    @NotNull PKBPerson actor) {
        Optional<PrivacyFlagsOverride> mainPrivacyFlag = isPrivacyFlagChanged(mainDto, newFlag).map(originalFlagsWhenChanged -> {
            mainDto.getBaseFields().setPrivacyFlag(newFlag);
            return Optional.of(privacyFlagsConverter.convert(context, mainDto, true));
        }).orElse(Optional.empty());

        if (mainPrivacyFlag.isPresent() && !actor.getId().equals(patient.getId())) {
            emailMessageManager.notifyChangeOfPrivacy(patient, actor, mainDto, mainType, newFlag);
        }

        List<PrivacyFlagsOverride> additionalPrivacyFlags = additionalDtos.stream()
                .filter(additionalDto -> isPrivacyFlagChanged(additionalDto, newFlag).isPresent())
                .map(additionalDto -> {
                    additionalDto.getBaseFields().setPrivacyFlag(newFlag);
                    return privacyFlagsConverter.convert(context, additionalDto, true);
                })
                .collect(Collectors.toList());

        privacyFlagsService.savePrivacyFlagsAndAccessLogMultiple(mainPrivacyFlag,
                additionalPrivacyFlags, ownerAccountId, mainType, additionalType, dateTimeService.now(),
                context.getMaybeAccessingUserId().orElse(null), context.getAccessRoute(), getConsentId(context), context.getCorrelationId().toString());
    }

    private Long getConsentId(@NotNull EHRRequestContext context) {
        return context.getConsentStatus().getConsent().map(PatientConsentDTO::id).orElse(null);
    }

    /**
     * Returns the original flags in case any flag changed.
     * There are records in production which have multiple flags set, so this case must be handled.
     *
     * @param baseDTO
     *            The data transfer object to check the privacy flags of
     * @param privacy
     *            The new privacy flag
     * @return An optional set of privacy flags
     */
    Optional<Set<PrivacyFlag>> isPrivacyFlagChanged(@NotNull IBaseDTO baseDTO, @NotNull PrivacyFlag privacy) {
        Set<PrivacyFlag> originalFlags = baseDTO.getBaseFields().getPrivacyFlags();
        return arePrivacyFlagsChanged(originalFlags, privacy);
    }

    private Optional<Set<PrivacyFlag>> arePrivacyFlagsChanged(Set<PrivacyFlag> originalFlags, PrivacyFlag newFlag) {
        switch (originalFlags.size()) {
        case 0:
            return Optional.of(emptySet());
        case 1:
            if (originalFlags.contains(newFlag)) {
                return Optional.empty();
            }
            return Optional.of(originalFlags);
        default:
            return Optional.of(originalFlags);
        }
    }
}
