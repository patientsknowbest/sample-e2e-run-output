package com.pkb.service.privacy;

import com.drew.lang.annotations.NotNull;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.data.EHRRemote;
import com.pkb.dto.EhrDataMigrationRequestDTO;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.service.encounter.impl.tolven.EncounterManager;

import java.util.Map;

public class MigrationService {

    private final EHRRemote ehrBean;
    private final EncounterManager encounterManager;

    public MigrationService(EHRRemote ehrBean, EncounterManager encounterManager) {
        this.ehrBean = ehrBean;
        this.encounterManager = encounterManager;
    }

    public void runEhrMigrations(@NotNull EhrDataMigrationRequestDTO migrationRequest, @NotNull LoggedInEHRRequestContext context) {
        ehrBean.runEhrDataMigrations(migrationRequest.getAccountPrivateId(), migrationRequest.getMenuDataType(), migrationRequest.getEhrDataIds(), context);
    }

    public Map<MenuDataType, Integer> getEhrMigrationVersions() {
        return ehrBean.getEhrMigrationVersions();
    }

}
