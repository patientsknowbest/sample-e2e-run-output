package com.pkb.annotation;

import com.pkb.app.entity.EHRData;

import java.util.List;
import java.util.stream.Collectors;
import java.util.stream.Stream;

/**
 * Base class for classes that can migrate an EHRData from older versions to current.
 * If data is changed, the EHRBean will save the changes afterwards.
 *
 * @author rob
 */

public abstract class EHRMigratorBase {


    /**
     * @return the current migration version. migrate() will only be called for EHRData with lower (or missing) migration versions,
     * and afterwards will be updated with this version number.
     */
    public final int getMigrationVersion() {
        return getMigrators().getMigrationVersion();
    }

    /**
     * No exceptions allowed -- if there's an error, log it but do not fail out.
     * This method will be called iff:
     * - the EHRData has been decrypted
     * - the EHRData has an older (or missing) migration version
     * <p>
     * The EHRData migrationVersion will be updated afterwards, and the EHRData saved to the DB.
     *
     * @param ehr The EHRData to check, and migrate if needed.
     */
    public final void migrate(EHRData ehr) {
        getMigrators().getMigratorStream(ehr.getMigrationVersion())
                .forEach(migrator -> {
                    migrator.migrate(ehr);
                    ehr.setMigrationVersion(migrator.getMigrationVersion());
                });
    }

    public abstract Migrators getMigrators();

    public static final class Migrators {

        private final List<EHRVersionMigrator> migratorList;

        private Migrators(List<EHRVersionMigrator> migratorList) {
            this.migratorList = migratorList;
        }

        public List<EHRVersionMigrator> getList() {
            return migratorList;
        }

        Stream<EHRVersionMigrator> getMigratorStream(int existingVersion) {
            return migratorList.stream().filter(migrator -> migrator.getMigrationVersion() > existingVersion);
        }

        public int getMigrationVersion() {
            return migratorList.stream().reduce((a, b) -> b).map(EHRVersionMigrator::getMigrationVersion).orElse(0);
        }

        public static Migrators of(List<EHRVersionMigrator> migrators) {
            List<EHRVersionMigrator> sortedMigrators = migrators.stream().sorted().collect(Collectors.toUnmodifiableList());
            if (sortedMigrators.stream().map(EHRVersionMigrator::getMigrationVersion).distinct().count() != sortedMigrators.size()) {
                throw new IllegalStateException("Migrators list contained multiple migrators with the same migration version :" + sortedMigrators);
            }
            return new Migrators(sortedMigrators);
        }

        public static Migrators of(EHRVersionMigrator... migrators) {
            return of(List.of(migrators));
        }
    }

}
