package com.pkb.annotation;

import com.pkb.app.entity.EHRData;
import org.jetbrains.annotations.NotNull;

import java.util.function.Consumer;

public abstract class EHRVersionMigrator implements Comparable<EHRVersionMigrator> {

    private final int migrationVersion;

    protected EHRVersionMigrator(int migrationVersion) {
        this.migrationVersion = migrationVersion;
    }

    public abstract void migrate(EHRData ehr);

    public final int getMigrationVersion() {
        return migrationVersion;
    }


    @Override
    public final int compareTo(@NotNull EHRVersionMigrator o) {
        return this.getMigrationVersion() - o.getMigrationVersion();
    }

    @Override
    public String toString() {
        return getClass().getName() + " migration version: " + getMigrationVersion();
    }

    public static EHRVersionMigrator of(Consumer<EHRData> implementation, int migrationVersion) {
        return new EHRVersionMigrator(migrationVersion) {
            @Override
            public void migrate(EHRData ehr) {
                implementation.accept(ehr);
            }

        };
    }
}
