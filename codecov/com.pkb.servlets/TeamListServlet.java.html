package com.pkb.servlets;

import com.pkb.common.config.PhrConfig;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.service.team.TeamManager;
import com.pkb.util.PKBApplicationContext;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.servlet.ServletConfig;
import javax.servlet.ServletException;
import javax.servlet.ServletRequest;
import javax.servlet.ServletResponse;
import javax.servlet.http.HttpServlet;
import java.io.IOException;
import java.io.PrintWriter;

/**
 * Example URL: https://uat.patientsknowbest.com/teamList?orgCode=smoketest
 * or (to return JavaScript that will be set in the given element's innerHtml:
 * https://uat.patientsknowbest.com/teamList?orgCode=smoketest&callback=teamDiv
 */
public class TeamListServlet extends HttpServlet {

    private TeamManager teamManager;
    private PhrConfig phrConfig;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Override
    public void init(ServletConfig arg0) throws ServletException {
        teamManager = (TeamManager) PKBApplicationContext.getBean("teamManager");
        phrConfig = PKBApplicationContext.getBean(PhrConfig.class);
    }

    @Override
    public void service(ServletRequest request, ServletResponse response)
            throws ServletException, IOException {
        String orgCode = request.getParameter("orgCode");
        String callback = request.getParameter("callback");
        try {
            Org org = teamManager.getOrgByCode(orgCode, Org.Lazy.TEAMS).orElse(null);

            String baseURL = phrConfig.getBaseURL();

            PrintWriter output = response.getWriter();

            if (callback == null) {

                output.println("<div>");

                if (org == null) {
                    output.println("<p><i>Sorry, that organization was not found; please report this problem to support.</i></p>");
                } else {
                    output.println("   <p>Set up your Patients Know Best account with " + org.getName() + "</p>");
                    output.println("   <ul>");
                    for (Team team : org.getTeams()) {
                        output.println("      <li>Register for <a href=\"" + baseURL + "/go/" + team.getCode() + "\" target=\"_top\">"
                                + team.getName() + "</a></li>");
                    }
                    output.println("   </ul>");
                }
                output.println("</div>");

                response.setContentType("text/html;charset=UTF-8");
            } else {

                output.print("document.getElementById('" + callback + "').innerHTML = \"");

                if (org == null) {
                    output.println("<p><i>Sorry, that organization was not found; please report this problem to support.</i></p>");
                } else {
                    output.print("<p>Set up your Patients Know Best account with " + org.getName() + "</p><ul>");
                    for (Team team : org.getTeams()) {
                        output.print("<li>Register for <a href='" + baseURL + "/go/" + team.getCode() + "' target='_top'>" + team.getName()
                                + "</a></li>");
                    }
                    output.println("</ul>\"");
                }

                //response.setContentType("application/x-javascript;charset=UTF-8");
                response.setContentType("text/javascript;charset=UTF-8");
            }

        } catch (Exception e) {
            LOGGER.info("Error while fetching team list of orgCode={}", orgCode, e);
        }
    }

}
