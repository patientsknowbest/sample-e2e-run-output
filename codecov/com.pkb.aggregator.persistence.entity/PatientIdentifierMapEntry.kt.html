package com.pkb.aggregator.persistence.entity

import com.pkb.aggregator.persistence.entity.embeddable.MapEntryKey
import org.hibernate.annotations.Immutable
import javax.persistence.Column
import javax.persistence.EmbeddedId
import javax.persistence.Entity
import javax.persistence.Table

/**
 * Maps an upstream identifier from a specific upstream store to the corresponding downstream resource for the
 * same patient
 */
@Entity
@Table(schema = "aggregator", name = "patient_identifier_map")
@Immutable
data class PatientIdentifierMapEntry(

    @EmbeddedId
    override val mapEntryKey: MapEntryKey,

    @Column(name = "downstream_resource_id")
    override val downstreamResourceId: String
) : IdentifierMapEntry
