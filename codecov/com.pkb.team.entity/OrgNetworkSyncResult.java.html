package com.pkb.team.entity;

import com.pkb.institute.entity.Org;

import java.io.Serializable;

/**
 * This is not a database entity - it's just a convenient container object for
 * returning information about an org network sync request to the caller
 *
 * @author Russ Gray
 */
public final class OrgNetworkSyncResult implements Serializable {

    private static final long serialVersionUID = 4618813301102954966L;

    private final Org sourceOrg;
    private final Org targetOrg;
    private final int syncCount;

    public Org getSourceOrg() {
        return sourceOrg;
    }

    public Org getTargetOrg() {
        return targetOrg;
    }

    public int getSyncCount() {
        return syncCount;
    }

    @Override
    public int hashCode() {
        int hash = 3;
        hash = (37 * hash) + (this.sourceOrg != null ? this.sourceOrg.hashCode() : 0);
        hash = (37 * hash) + (this.targetOrg != null ? this.targetOrg.hashCode() : 0);
        hash = (37 * hash) + this.syncCount;
        return hash;
    }

    @Override
    public boolean equals(Object obj) {
        if (obj == null) {
            return false;
        }
        if (getClass() != obj.getClass()) {
            return false;
        }
        OrgNetworkSyncResult other = (OrgNetworkSyncResult) obj;
        if ((this.sourceOrg != other.sourceOrg) && ((this.sourceOrg == null) || !this.sourceOrg.equals(other.sourceOrg))) {
            return false;
        }
        if ((this.targetOrg != other.targetOrg) && ((this.targetOrg == null) || !this.targetOrg.equals(other.targetOrg))) {
            return false;
        }
        if (this.syncCount != other.syncCount) {
            return false;
        }
        return true;
    }

    public static final class Builder {

        private Org sourceOrg;
        private Org targetOrg;
        private int syncCount;

        private Builder() {
        }

        public Builder sourceOrg(Org value) {
            this.sourceOrg = value;
            return this;
        }

        public Builder targetOrg(Org value) {
            this.targetOrg = value;
            return this;
        }

        public Builder syncCount(int value) {
            this.syncCount = value;
            return this;
        }

        public OrgNetworkSyncResult build() {
            return new com.pkb.team.entity.OrgNetworkSyncResult(sourceOrg, targetOrg, syncCount);
        }
    }

    public static OrgNetworkSyncResult.Builder builder() {
        return new OrgNetworkSyncResult.Builder();
    }

    private OrgNetworkSyncResult(Org sourceOrg, Org targetOrg, int syncCount) {
        this.sourceOrg = sourceOrg;
        this.targetOrg = targetOrg;
        this.syncCount = syncCount;
    }

}
