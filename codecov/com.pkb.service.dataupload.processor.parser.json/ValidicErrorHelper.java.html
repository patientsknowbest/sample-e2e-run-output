package com.pkb.service.dataupload.processor.parser.json;

import io.micrometer.core.instrument.Counter;
import io.micrometer.core.instrument.Metrics;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

public class ValidicErrorHelper {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final Counter UNMAPPED_FIELD_COUNTER = Counter.builder("pkb_phr_integration_validic_unmapped_field")
            .description("Number of times we've encoutnered unmapped fields processing validic")
            .register(Metrics.globalRegistry);

    private final Map<String, String> keysAlreadySeen = new ConcurrentHashMap<>();
    private final Map<String, String> categporyAlreadySeen = new ConcurrentHashMap<>();

    public void reportUnmappedField(String validicKey) {
        if (validicKey != null) {
            UNMAPPED_FIELD_COUNTER.increment();
            if (keysAlreadySeen.putIfAbsent(validicKey, validicKey) == null) {
                LOGGER.error("Unmapped field '{}'", validicKey);
            }
        }
    }

    public void reportUnmappedField(String validicKey, String category) {
        if (validicKey != null) {
            UNMAPPED_FIELD_COUNTER.increment();
            String previousKey = keysAlreadySeen.putIfAbsent(validicKey, validicKey);
            String previousCategory = category == null ? "" : categporyAlreadySeen.putIfAbsent(category, category);
            if (previousKey == null || previousCategory == null) {
                LOGGER.error("Unmapped field '{}' in category '{}'", validicKey, category);
            }
        }
    }
}
