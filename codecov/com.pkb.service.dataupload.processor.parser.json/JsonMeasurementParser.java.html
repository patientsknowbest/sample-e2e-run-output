package com.pkb.service.dataupload.processor.parser.json;

import com.pkb.app.entity.SourceDetails;
import com.pkb.dataupload.entity.UploadedData;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.entities.enums.Route;
import com.pkb.integration.coredevices.CoreDeviceDataRecord;
import com.pkb.integration.coredevices.CoreDevicesClient;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.ParsedObservationInformation;
import com.pkb.service.dataupload.processor.parser.AbstractUploadedDataParser;
import com.pkb.test.entity.MeasurementDTO;
import com.pkb.test.entity.PredefinedMeasurementType;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Map;

@Component
public class JsonMeasurementParser extends AbstractUploadedDataParser<ParsedObservationInformation<MeasurementDTO>, LoggedInUploadedDataProcessingContext> {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private CoreDevicesClient coreDevicesClient;
    private ValidicErrorHelper validicErrorHelper;

    @Autowired
    public JsonMeasurementParser(CoreDevicesClient coreDevicesClient, ValidicErrorHelper validicErrorHelper) {
        super(UploadedData.Format.JSON_VALIDIC);
        this.coreDevicesClient = coreDevicesClient;
        this.validicErrorHelper = validicErrorHelper;
    }

    @Override
    protected List<ParsedObservationInformation<MeasurementDTO>> doParse(LoggedInUploadedDataProcessingContext context, UploadedDataDTO uploadedDataDTO) {
        List<MeasurementDTO> measurements = new ArrayList<>();
        CoreDeviceDataRecord record = null;
        try {
            record = coreDevicesClient.parseRecord(uploadedDataDTO.getData());
        } catch (Exception e) {
            throw new IllegalArgumentException("Failed to parse json record", e);
        }

        Map<String, String> fields = record.getFields();
        for (Map.Entry<String, String> stringStringEntry : fields.entrySet()) {
            SourceDetails sd = new SourceDetails();
            sd.init(context.patientId(), null, null, record.getSource(), Route.CORE_DEVICES, null);
            String fieldValue = stringStringEntry.getValue();
            if ((fieldValue != null) && !"null".equals(fieldValue) && !"0.0".equals(fieldValue)) {
                PredefinedMeasurementType predefinedMeasurementType = ("calories".equals(stringStringEntry.getKey()) || "calories_burned"
                        .equals(stringStringEntry.getKey())) ? PredefinedMeasurementType.CALORIES_BURNED_ROUTINE
                        : "distance".equals(stringStringEntry.getKey()) ? PredefinedMeasurementType.DISTANCE_ROUTINE
                        : PredefinedMeasurementType.getByApiRef(stringStringEntry.getKey());
                if (predefinedMeasurementType == null) {
                    validicErrorHelper.reportUnmappedField(stringStringEntry.getKey());
                    measurements.add(new MeasurementDTO(sd));
                } else {
                    MeasurementDTO measurement = new MeasurementDTO(sd);
                    measurement.setMeasureDate(record.getRecordTimestamp());
                    measurement.setMeasurementType(predefinedMeasurementType);
                    measurement.setValue(Double.parseDouble(fieldValue));
                    measurement.setUploadedDataId(uploadedDataDTO.getId());
                    measurements.add(measurement);
                }
            }
        }
        return Collections.singletonList(new ParsedObservationInformation<>(null, measurements));
    }
}
