package com.pkb.service.dataupload.processor.parser.json;

import static com.pkb.util.Constants.APPLICATION_TZ;

@Component
public class JsonTestResultsParser extends AbstractUploadedDataParser<ParsedObservationInformation<TestResultDTO>, LoggedInUploadedDataProcessingContext> {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final Pattern TIMESTAMP_EXTRACTOR_PATTERN = Pattern.compile(":(..)$");
    private static final DateTimeFormatter DATE_FORMATTER = DateTimeFormatter.ofPattern("yyyy-MM-dd'T'HH:mm:ssz").withLocale(Locale.UK).withZone(APPLICATION_TZ);

    private DateTimeService dateTimeService;
    private LoincManager loincManager;

    @Autowired
    public JsonTestResultsParser(DateTimeService dateTimeService, LoincManager loincManager) {
        super(UploadedData.Format.JSON_VALIDIC);
        this.dateTimeService = dateTimeService;
        this.loincManager = loincManager;
    }

    @Override
    protected List<ParsedObservationInformation<TestResultDTO>> doParse(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto) {
        String json = dto.getData();

        ObjectMapper mapper = new ObjectMapper();
        JsonNode jsonRecord = null;
        try {
            jsonRecord = mapper.readTree(json);
        } catch (IOException e) {
            throw new IllegalArgumentException("Failed to map json record", e);
        }

        Instant timestamp = dateTimeService.parseToInstantBackwardCompatibleWay(TIMESTAMP_EXTRACTOR_PATTERN.matcher(getStringField(jsonRecord, "timestamp")).replaceAll("$1"),
                DATE_FORMATTER)._1; //Kludge!


        //List<MeasurementDTO> measurements = new ArrayList<MeasurementDTO>();
        List<TestResultDTO> testResults = new LinkedList<>();

        Iterator<String> fieldNamesIterator = jsonRecord.fieldNames();
        while (fieldNamesIterator.hasNext()) {
            String fieldName = fieldNamesIterator.next();
            String fieldValue = jsonRecord.get(fieldName).asText();
            if ((fieldValue != null) && !"null".equals(fieldValue) && !"0.0".equals(fieldValue)) {
                //logger.info("SJG: read from queue name="+fieldName+" value="+fieldValue);
                //if ("weight".equals(fieldName)) {
                //	MeasurementDTO measurement = new MeasurementDTO();
                //	measurement.setMeasureDate(timestamp);
                //	measurement.setMeasurementTypeId(PredefinedMeasurementType.WEIGHT.getId());
                //	measurement.setValue(Double.parseDouble(fieldValue));
                //	measurement.setSource(sd);
                //	measurements.add(measurement);
                //} else
                if ("blood_glucose".equals(fieldName)) {
                    SourceDetails sd = new SourceDetails();
                    sd.init(context.patientId(), null, null, getStringField(jsonRecord, "source"), Route.CORE_DEVICES, null);
                    //logger.info("SJG: record timestamp="+timestamp+" source="+sd.getText());
                    TestResultDTO testResult = new TestResultDTO(sd);
                    testResult.setTestDate(timestamp);
                    testResult.setLoincTestId(getLoincTestId("2345-7", "mg/dL"));
                    testResult.setValue(Double.parseDouble(fieldValue));
                    testResults.add(testResult);
                }
            }
        }
        return Collections.singletonList(new ParsedObservationInformation<>(null, testResults));

    }

    private String getStringField(JsonNode root, String fieldName) {
        JsonNode node = root.findValue(fieldName);
        if (node == null) {
            throw new PKBException("Missing '" + fieldName + "' field");
        }
        return node.asText();
    }

    private Long getLoincTestId(String code, String units) {
        return getLoincTestByLoinc(code, units)
                .map(LoincTest::getId)
                .orElseThrow(() -> new RuntimeException("No LoincTest for code='" + code + "' units='" + units + "'"));
    }

    private Optional<LoincTest> getLoincTestByLoinc(String loinc, String unit) {
        try {
            return loincManager.findLoincTestByLoincCodeAndUnits(loinc, unit);
        } catch (Exception e) {
            throw new PKBException("failed getting test for loinc " + loinc + "/ unit " + unit, e);
        }
    }
}
