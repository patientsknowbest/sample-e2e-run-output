package com.pkb.consent.mapper;

import com.pkb.consent.entity.PatientConsent;
import com.pkb.datamodel.consent.ImmutablePatientConsentDTO;
import com.pkb.datamodel.consent.PatientConsentDTO;
import com.pkb.entities.enums.AccessingEntityType;

import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;

public interface PatientConsentMapper {

    //Used for reading
    static PatientConsentDTO map(PatientConsent pc) {
        var builder = ImmutablePatientConsentDTO.builder()
                .generalHealthConsentGranted(pc.isGeneralHealthConsentGranted())
                .sexualHealthConsentGranted(pc.isSexualHealthConsentGranted())
                .mentalHealthConsentGranted(pc.isMentalHealthConsentGranted())
                .socialCareConsentGranted(pc.isSocialCareConsentGranted())
                .proEditOnlyForChild(pc.isProEditOnlyForChild())
                .discharged(pc.isDischarged())
                .id(pc.getId())
                .uniqueId(pc.getUniqueId())
                .accessingEntityType(pc.getAccessingEntityType())
                .patientPersonId(pc.getPatientPersonId())
                .added(pc.getAdded())
                .updated(pc.getUpdated());
        return Match(pc.getAccessingEntityType()).of(
                Case($(AccessingEntityType.TEAM), () -> builder.accessingTeamId(pc.getAccessingTeamSlashPersonId())),
                Case($(AccessingEntityType.CARER), () -> builder.accessingPersonId(pc.getAccessingTeamSlashPersonId())),
                Case($(AccessingEntityType.INDIVIDUAL), () -> builder.accessingPersonId(pc.getAccessingTeamSlashPersonId()))
        ).build();
    }
}
