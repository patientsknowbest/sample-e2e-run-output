package com.pkb.api.fhir.provenance;

import com.pkb.api.fhir.CanonicalExtensionURLs;
import com.pkb.api.fhir.TemporalMapper;
import com.pkb.api.fhir.foundation.ReferenceMapper;
import com.pkb.api.fhir.person.RelatedPersonMapper;
import com.pkb.datamodel.Appointment;
import com.pkb.datamodel.Communication;
import com.pkb.datamodel.DocumentMessage;
import com.pkb.datamodel.Encounter;
import com.pkb.datamodel.Source;
import com.pkb.datamodel.user.Patient;
import com.pkb.datamodel.user.Professional;
import com.pkb.datamodel.user.TeamCoordinator;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.hl7.fhir.dstu3.model.Extension;
import org.hl7.fhir.dstu3.model.Provenance;
import org.hl7.fhir.dstu3.model.Reference;
import org.hl7.fhir.dstu3.model.RelatedPerson;
import org.hl7.fhir.dstu3.model.ResourceType;
import org.jetbrains.annotations.NotNull;

import java.time.ZonedDateTime;
import java.util.UUID;

import static com.pkb.api.fhir.DataAbsentReasonUtil.missingInstantType;
import static io.vavr.API.List;
import static org.hl7.fhir.dstu3.model.ResourceType.Appointment;
import static org.hl7.fhir.dstu3.model.ResourceType.Communication;
import static org.hl7.fhir.dstu3.model.ResourceType.DocumentReference;
import static org.hl7.fhir.dstu3.model.ResourceType.Encounter;

public class ProvenanceMapper {
    private final ReferenceMapper referenceMapper;
    private final RelatedPersonMapper relatedPersonMapper;
    private final TemporalMapper temporalMapper;

    public ProvenanceMapper(@NotNull ReferenceMapper referenceMapper,
                            @NotNull RelatedPersonMapper relatedPersonMapper,
                            @NotNull TemporalMapper temporalMapper) {
        this.referenceMapper = referenceMapper;
        this.relatedPersonMapper = relatedPersonMapper;
        this.temporalMapper = temporalMapper;
    }

    public Provenance toProvenance(Appointment appointment) {
        return toProvenance(appointment.uniqueId(),
                appointment.persisted(),
                appointment.getSource(),
                appointment.owner(),
                Appointment);
    }

    public Provenance toProvenance(Communication communication) {
        return toProvenance(communication.getId(),
                communication.getPersisted(),
                communication.getSource(),
                communication.getPatient(),
                Communication);
    }

    public Provenance toProvenance(DocumentMessage message) {
        return toProvenance(message.getUniqueId(),
                message.getPersisted(),
                message.getSource(),
                message.getPatient(),
                DocumentReference);
    }

    public List<Provenance> toProvenance(Encounter encounter) {
        return io.vavr.collection.List.ofAll(encounter.events())
                .map(encounterEvent -> toProvenance(
                        encounterEvent.getId(),
                        encounter.getId(),
                        encounterEvent.persisted(),
                        encounterEvent.getSource(),
                        encounterEvent.patient(),
                        Encounter));
    }

    @NotNull
    private Provenance toProvenance(UUID uniqueId,
            Option<ZonedDateTime> persisted,
            Source source,
            Patient owner,
            ResourceType resourceType) {
        return toProvenance(uniqueId, uniqueId, persisted, source, owner, resourceType);
    }

    @NotNull
    private Provenance toProvenance(UUID id,
            UUID targetId,
            Option<ZonedDateTime> persisted,
            Source source,
            Patient owner,
            ResourceType resourceType) {
        Provenance provenance = new Provenance();
        provenance.setId(id.toString());
        provenance.addTarget(referenceMapper.toReference(resourceType, targetId));
        provenance.setRecordedElement(persisted
                .map(temporalMapper::toInstantTypeInUTC)
                .getOrElse(missingInstantType()));

        Provenance.ProvenanceAgentComponent agent = provenance.addAgent();
        var sourcePersonExtension = source.getPerson()
                .map(sourcePerson -> {
                    Extension extension = new Extension().setUrl(CanonicalExtensionURLs.sourcePerson());
                    if (Professional.class.isInstance(sourcePerson) || TeamCoordinator.class.isInstance(sourcePerson)) {
                        extension.setValue(referenceMapper.toReferenceWithName(sourcePerson));
                    } else if (Patient.class.isInstance(sourcePerson)) {
                        if (owner.getId().equals(sourcePerson.getId())) {
                            extension.setValue(referenceMapper.toReferenceWithName(sourcePerson));
                        } else {
                            RelatedPerson relatedPerson = relatedPersonMapper.toRelatedPerson(owner, (Patient) sourcePerson);
                            relatedPerson.setId("#carer");
                            provenance.addContained(relatedPerson);

                            Reference reference = new Reference();
                            reference.setReference(relatedPerson.getId());
                            reference.setDisplay(referenceMapper.toDisplay(sourcePerson));
                            extension.setValue(reference);
                        }
                    }
                    return extension;
                });

        var sourceTeamExtension = source.getTeam()
                .map(team -> new Extension()
                        .setUrl(CanonicalExtensionURLs.sourceTeam())
                        .setValue(referenceMapper.toReferenceWithName(team)));

        var sourceOrgExtension = source.getOrg()
                .map(org -> new Extension()
                        .setUrl(CanonicalExtensionURLs.sourceOrganisation())
                        .setValue(referenceMapper.toReferenceWithName(org)));
        var extensions = List(sourcePersonExtension, sourceTeamExtension, sourceOrgExtension)
                .filter(Option::isDefined).map(Option::get);

        // TODO: GDE-841
        // message.getSource().getText()
        //         .ifPresent(text -> {
        //             Extension extension = new Extension()
        //                     .setUrl(ExtensionURL.sourceText())
        //                    .setValue(new Reference().setDisplay(text));
        //             extensions.add(extension);
        //         });

        Reference mostRelevantReference = extensions
                .headOption()
                .map(Extension::getValue)
                .map(Reference.class::cast)
                .getOrElse(() -> new Reference().setDisplay("No source information available"));

        agent.setWho(mostRelevantReference.copy());

        Reference onBehalfOf =  mostRelevantReference.copy();
        source.getResponsibleOrganisationIdentifier().map(ro -> onBehalfOf.setDisplay(ro));

        agent.setOnBehalfOf(onBehalfOf);
        extensions.forEach(onBehalfOf::addExtension);

        return provenance;
    }
}
