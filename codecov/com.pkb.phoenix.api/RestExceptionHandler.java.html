package com.pkb.phoenix.api;

import com.pkb.phoenix.api.service.phoenixexception.PhoenixAuthorizationException;
import com.pkb.phoenix.api.service.phoenixexception.PhoenixException;
import com.pkb.phoenix.api.service.phoenixexception.PhoenixIllegalStateException;
import com.pkb.phoenix.api.service.phoenixexception.PhoenixUnexpectedConflictException;
import com.pkb.phoenix.api.service.phoenixexception.PhoenixValidationException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.http.converter.HttpMessageNotReadableException;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.ResponseStatus;
import org.springframework.web.bind.annotation.RestControllerAdvice;

import java.lang.invoke.MethodHandles;

import static org.springframework.http.HttpStatus.BAD_REQUEST;
import static org.springframework.http.HttpStatus.CONFLICT;
import static org.springframework.http.HttpStatus.FORBIDDEN;
import static org.springframework.http.HttpStatus.INTERNAL_SERVER_ERROR;

@RestControllerAdvice
public class RestExceptionHandler {
    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    // https://spring.io/blog/2013/11/01/exception-handling-in-spring-mvc
    @ResponseStatus(BAD_REQUEST)
    @ExceptionHandler({
            PhoenixValidationException.class,
            PhoenixIllegalStateException.class,
    })

    BadRequest handleBadRequest(PhoenixException e) {
        LOGGER.error("Request processing failed, cause: ", e);
        return new BadRequest(e);
    }

    public static class BadRequest {
        public final int status = 400;
        public final String error = "Bad Request";
        public String phoenixErrorCode;
        public String phoenixErrorMessage;

        public BadRequest(PhoenixException e) {
            this.phoenixErrorCode = e.getErrorCode();
            this.phoenixErrorMessage = e.getMessage();
        }
    }

    @ResponseStatus(BAD_REQUEST)
    @ExceptionHandler({
            HttpMessageNotReadableException.class
    })

    MessageNotReadable handleHttpMessageNotReadableException(HttpMessageNotReadableException e) {
        LOGGER.error("Request processing failed, cause: ", e);
        return new MessageNotReadable();
    }

    public static class MessageNotReadable {
        public final int status = 400;
        public String phoenixErrorCode = "Bad Request";
        public String phoenixErrorMessage = "Parameters are not valid. Patient's Private Id is not valid!";
    }

    @ResponseStatus(FORBIDDEN)
    @ExceptionHandler({
            PhoenixAuthorizationException.class
    })

    AuthorizationFailure handleAuthorizationFailure(PhoenixException e) {
        LOGGER.error("Request processing failed, cause: ", e);
        return new AuthorizationFailure(e);
    }

    public static class AuthorizationFailure {
        public final int status = 401;
        public final String error = "Authorization failed.";
        public String phoenixErrorCode;
        public String phoenixErrorMessage;

        public AuthorizationFailure(PhoenixException e) {
            this.phoenixErrorCode = e.getErrorCode();
            this.phoenixErrorMessage = e.getMessage();
        }
    }

    @ResponseStatus(CONFLICT)
    @ExceptionHandler({
            PhoenixUnexpectedConflictException.class
    })

    Conflict handleConflict(PhoenixException e) {
        LOGGER.error("Request processing failed, cause: ", e);
        return new Conflict(e);
    }

    public static class Conflict {
        public final int status = 409;
        public final String error = "Conflict";
        public String phoenixErrorCode;
        public String phoenixErrorMessage;

        public Conflict(PhoenixException e) {
            this.phoenixErrorCode = e.getErrorCode();
            this.phoenixErrorMessage = e.getMessage();
        }
    }

    @ResponseStatus(INTERNAL_SERVER_ERROR)
    @ExceptionHandler({
            RuntimeException.class
    })

    InternalServerError internalServerError(RuntimeException e) {
        LOGGER.error("Request processing failed, cause: ", e);
        return new InternalServerError();
    }

    public static class InternalServerError {
        public final int status = 500;
        public final String error = "Internal Server Error";
    }
}
