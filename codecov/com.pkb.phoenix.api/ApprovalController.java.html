package com.pkb.phoenix.api;

import com.pkb.phoenix.api.model.Approval;
import com.pkb.phoenix.api.model.ImmutableApproval;
import com.pkb.phoenix.api.model.ImmutableIdentifiable;
import com.pkb.phoenix.api.service.task.TaskService;
import com.pkb.phoenix.api.service.task.action.ImmutableSearchTasks;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

import java.util.List;
import java.util.Optional;

import static com.pkb.phoenix.api.service.phoenixexception.PhoenixIllegalStateException.ConfigurationError.configurationError;
import static com.pkb.phoenix.api.service.task.Status.APPROVED;
import static io.vavr.API.Set;
import static java.util.stream.Collectors.toUnmodifiableList;

@RestController
@RequestMapping(value = "/phoenix/approval", consumes = "application/json", produces = "application/json")
public class ApprovalController {
    private final TaskService taskService;

    public ApprovalController(TaskService taskService) {
        this.taskService = taskService;
    }

    @GetMapping
    public List<Approval> list(@RequestParam(value = "task", required = false) Integer id) {
        if (id != null) {
            return taskService.findTaskById(id)
                    .map(task -> toModelIfApproved(task)
                            .map(List::of)
                            .orElse(List.of()))
                    .orElse(List.of());
        }

        return taskService.findTasks(ImmutableSearchTasks.builder().statuses(Set(APPROVED)).build()).stream()
                .map(task -> toModelIfApproved(task).orElseThrow(() -> configurationError("Search should have returned only approved tasks")))
                .collect(toUnmodifiableList());
    }

    private Optional<Approval> toModelIfApproved(com.pkb.phoenix.api.service.task.TaskInstance<?> task) {
        return task.events().stream()
                .filter($ -> "approved".equals($.event()))
                .map(event -> (Approval) ImmutableApproval.builder()
                        .id(event.id())
                        .task(ImmutableIdentifiable.builder().id(task.id()).build())
                        .type(task.type().code())
                        .note(event.note().orElse(null))
                        .user(event.actor().id())
                        .timestamp(event.at().toString())
                        .build())
                .findFirst();
    }
}
