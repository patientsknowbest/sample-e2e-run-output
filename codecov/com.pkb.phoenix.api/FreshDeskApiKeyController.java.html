package com.pkb.phoenix.api;

import com.pkb.phoenix.api.model.FreshDeskApiKeyStatusResponse;
import com.pkb.phoenix.api.model.ImmutableFreshDeskApiKeyStatusResponse;
import com.pkb.phoenix.api.model.StoreFreshDeskApiKeyRequest;
import com.pkb.phoenix.api.service.ImmutableKeycloakId;
import com.pkb.phoenix.api.service.KeycloakId;
import com.pkb.phoenix.api.service.Meta;
import com.pkb.phoenix.api.service.freshdesk.ApiKey;
import com.pkb.phoenix.api.service.freshdesk.FreshDeskApiKeyStatus;
import com.pkb.phoenix.api.service.freshdesk.ImmutableApiKey;
import com.pkb.phoenix.api.service.freshdesk.RedisBackedApiKeyStore;
import org.jetbrains.annotations.NotNull;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;
import reactor.core.publisher.Mono;

import java.util.List;

import static com.pkb.phoenix.api.service.phoenixexception.PhoenixValidationException.InvalidTaskParameters.invalidTaskParameters;
import static reactor.core.publisher.Mono.error;
import static reactor.core.publisher.Mono.just;

@RestController
@RequestMapping(value = "/phoenix/apikey", consumes = "application/json", produces = "application/json")
public class FreshDeskApiKeyController {
    private final RedisBackedApiKeyStore apiKeyFinder;

    public FreshDeskApiKeyController(RedisBackedApiKeyStore apiKeyFinder) {
        this.apiKeyFinder = apiKeyFinder;
    }

    @PostMapping
    Mono<FreshDeskApiKeyStatusResponse> setApiKey(@RequestBody StoreFreshDeskApiKeyRequest request, Meta meta) {
        return validate(request)
                .flatMap(apiKey -> apiKeyFinder.setApiKey(keycloakId(meta), apiKey)
                        .map(this::buildResponse));
    }


    @GetMapping("/status")
    Mono<FreshDeskApiKeyStatusResponse> getApiKeyStatus(Meta meta) {
        return apiKeyFinder.getApiKeyStatus(keycloakId(meta))
                .map(this::buildResponse);
    }

    private @NotNull Mono<ApiKey> validate(StoreFreshDeskApiKeyRequest request) {
        try {
            return just(ImmutableApiKey.of(request.apiKey()));
        } catch (IllegalArgumentException cause) {
            return error(invalidTaskParameters(List.of(cause.getMessage())));
        }
    }

    private KeycloakId keycloakId(Meta meta) {
        return ImmutableKeycloakId.of(meta.session().user().keycloakId());
    }

    private @NotNull FreshDeskApiKeyStatusResponse buildResponse(FreshDeskApiKeyStatus status) {
        return ImmutableFreshDeskApiKeyStatusResponse.of(status.name());
    }
}
