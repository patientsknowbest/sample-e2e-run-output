package com.pkb.backgroundtasks.repository

import com.pkb.backgroundtasks.OrgPrivateId
import com.pkb.backgroundtasks.entity.OrgAccessTaskComplete
import com.pkb.backgroundtasks.entity.OrgAccessTaskPending
import org.hibernate.jpa.QueryHints.HINT_FETCH_SIZE
import org.springframework.data.jpa.repository.Query
import org.springframework.data.jpa.repository.QueryHints
import org.springframework.data.repository.CrudRepository
import org.springframework.transaction.annotation.Propagation
import org.springframework.transaction.annotation.Transactional
import java.util.UUID
import java.util.stream.Stream
import javax.persistence.QueryHint

interface OrgAccessTaskPendingRepository : CrudRepository<OrgAccessTaskPending, UUID> {

    @Query(
        """
        SELECT p FROM OrgAccessTaskPending p
        LEFT OUTER JOIN OrgAccessTaskComplete c
        ON c.id = p.id
        WHERE c is null
        AND p.orgId = :orgPrivateId
    """
    )
    fun getPendingForOrgId(orgPrivateId: OrgPrivateId): List<OrgAccessTaskPending>


    @Query(
        """
            SELECT p FROM OrgAccessTaskPending p
        LEFT OUTER JOIN OrgAccessTaskComplete c
        ON c.id = p.id
        WHERE c is null
        """
    )
    @QueryHints(value = [QueryHint(name = HINT_FETCH_SIZE, value = "200")])
    fun streamAllPending(): Stream<OrgAccessTaskPending>

    @Transactional(readOnly = true)
    fun processAllPendingRows(processor: (pending: OrgAccessTaskPending) -> Unit) =
        streamAllPending().forEach(processor)

}

interface OrgAccessTaskCompleteRepository : CrudRepository<OrgAccessTaskComplete, UUID> {

    @Transactional(propagation = Propagation.REQUIRES_NEW)
    fun saveWithNewTransaction(completeTask: OrgAccessTaskComplete) = save(completeTask)
}
