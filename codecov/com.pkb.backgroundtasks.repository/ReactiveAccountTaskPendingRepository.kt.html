package com.pkb.backgroundtasks.repository

import com.pkb.backgroundtasks.entity.AccountTaskType
import com.pkb.kms.shared.representation.AccountPublicId
import org.springframework.r2dbc.core.DatabaseClient
import org.springframework.stereotype.Component
import reactor.core.publisher.Flux

@Component
class ReactiveAccountTaskPendingRepository(private val databaseClient: DatabaseClient) {


    fun getAccountsToFix(disabledTaskTypes: List<AccountTaskType>): Flux<AccountPublicId> {
        val sql = if (disabledTaskTypes.isEmpty()) { ACCOUNTS_TO_FIX_QUERY_NO_DISABLED } else { ACCOUNTS_TO_FIX_QUERY }

        return databaseClient.sql(sql)
            .let {
                if(!disabledTaskTypes.isEmpty()) {
                    it.bind("disabledTaskTypes", disabledTaskTypes.map{ it.toString() })
                }
                else {
                    it
                }
            }
            .map { r -> r.get(0, AccountPublicId::class.java)!! }
            .all()

    }


    companion object {
        private const val ACCOUNTS_TO_FIX_QUERY_START =  """
            SELECT p.account_public_id FROM bgtasks.account_task_pending_v2 p
            LEFT OUTER JOIN bgtasks.account_task_complete_v2 c ON c.id = p.id
            LEFT OUTER JOIN bgtasks.account_without_key wk ON wk.id = p.account_public_id 
            WHERE c.id IS NULL and wk.id IS NULL
            """
        private const val ACCOUNTS_TO_FIX_QUERY_DISABLED_TASK_RESTRICTION = """
            AND p.task_details->>'type' NOT IN (:disabledTaskTypes)
        """
        private const val ACCOUNT_TO_FIX_QUERY_END = """
            GROUP BY p.account_public_id
            ORDER BY max(p.created_date) DESC, p.account_public_id
        """

        private const val ACCOUNTS_TO_FIX_QUERY_NO_DISABLED = ACCOUNTS_TO_FIX_QUERY_START + ACCOUNT_TO_FIX_QUERY_END
        private const val ACCOUNTS_TO_FIX_QUERY = ACCOUNTS_TO_FIX_QUERY_START + ACCOUNTS_TO_FIX_QUERY_DISABLED_TASK_RESTRICTION + ACCOUNT_TO_FIX_QUERY_END
    }
}
