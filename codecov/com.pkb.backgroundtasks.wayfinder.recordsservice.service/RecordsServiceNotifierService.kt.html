package com.pkb.backgroundtasks.wayfinder.recordsservice.service

import com.pkb.backgroundtasks.config.BackgroundTasksProperties
import com.pkb.backgroundtasks.config.WayfinderRecordsServiceProperties
import com.pkb.backgroundtasks.routing.DownstreamChangeNotification
import com.pkb.backgroundtasks.wayfinder.recordsservice.provider.NhsNumberProvider
import com.pkb.backgroundtasks.wayfinder.recordsservice.provider.OdsCodeProvider
import com.pkb.backgroundtasks.wayfinder.recordsservice.entity.PendingNhsNumber
import com.pkb.backgroundtasks.wayfinder.recordsservice.repository.PendingNhsNumberRepository
import com.pkb.entities.enums.NationalIdType
import com.pkb.entities.enums.ValidNationalId
import org.apache.camel.Exchange
import org.apache.camel.ExchangeProperty
import org.hl7.fhir.r4.model.IdType
import org.springframework.stereotype.Service
import java.net.URI
import javax.transaction.Transactional

@Service
class RecordsServiceNotifierService(
    backgroundTasksProperties: BackgroundTasksProperties,
    private val nhsNumberProvider: NhsNumberProvider,
    private val pendingNhsNumberRepository: PendingNhsNumberRepository,
    private val odsCodeProvider: OdsCodeProvider
) {

    companion object {
        const val PROPERTY_RECORD_IDS = "recordIds"
    }

    private val properties: WayfinderRecordsServiceProperties = backgroundTasksProperties.recordsService

    fun findNhsNumber(notification: DownstreamChangeNotification, exchange: Exchange): List<ValidNationalId> {
        val resourceId = getResourceId(notification)
        exchange.properties["resourceId"] = resourceId

        return nhsNumberProvider.findNhsNumberByAppointmentResource(resourceId).orElse(null)
                ?.let { listOf(it) } ?: emptyList()
    }

    fun findOdsCode(notification: DownstreamChangeNotification, exchange: Exchange): String? {
        val resourceId = getResourceId(notification)
        exchange.properties["resourceId"] = resourceId

        return odsCodeProvider.findOdsCodeByAppointmentResource(resourceId)
    }

    fun buildHttpMessage(nhsNumbers: List<ValidNationalId>, exchange: Exchange): NotificationRequest? {
        if (nhsNumbers.isEmpty()) {
            exchange.isRouteStop = true
            return null
        }

        exchange.message.headers["Authorization"] = "Bearer ${properties.apiKey}"
        return NotificationRequest(properties.clientId, nhsNumbers.map { it.value })
    }

    fun mapSqlToValidNationalId(sqlResult: List<Map<String, Any>>, exchange: Exchange): List<ValidNationalId> {
        exchange.properties[PROPERTY_RECORD_IDS] = sqlResult.mapNotNull { it["id"] as? Long }

        return sqlResult.mapNotNull { it["nhs_number"] as? String }
                .mapNotNull { NationalIdType.NHS_NUMBER.getValidNationalIdAndType(it).orElse(null) }
    }

    @Transactional
    fun markAsSent(@ExchangeProperty(PROPERTY_RECORD_IDS) recordIds: List<Long>) {
        pendingNhsNumberRepository.updateDeliveryStatus(recordIds, PendingNhsNumber.DeliveryStatus.SENT)
    }

    @Transactional
    fun handleFailure(@ExchangeProperty(PROPERTY_RECORD_IDS) recordIds: List<Long>) {
        pendingNhsNumberRepository.updateDeliveryStatus(recordIds, PendingNhsNumber.DeliveryStatus.FAILED)
    }

    private fun getResourceId(notification: DownstreamChangeNotification): IdType {
        val serverUrl = URI(notification.data)
        return IdType(serverUrl.path.removePrefix("/fhir/"))
    }

    data class NotificationRequest(val clientId: String, val nhsNumbers: List<String>)
}
