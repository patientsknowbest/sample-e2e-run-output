package com.pkb.action.emis.pfs;

import com.google.common.collect.ImmutableMap;
import com.google.common.collect.Maps;
import com.google.common.collect.Sets;

import com.pkb.calendar.entity.AppointmentDTO;
import com.pkb.emis.appointment.service.EmisAppointmentManager;
import com.pkb.emis.appointment.service.ErrorCode;
import com.pkb.entities.enums.AppointmentStatus;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.entities.enums.Route;
import com.pkb.integration.ods.OdsService;
import com.pkb.integrations.emis.model.EmisAppointment;
import com.pkb.integrations.emis.model.datatypes.BookingMethod;
import com.pkb.repository.util.EmisContextHolder;

import java.util.stream.Collectors;

import static com.pkb.action.emis.pfs.EmisMessageHelper.AccessType.PRIMARY;
import static com.pkb.emis.appointment.service.ErrorCode.EMIS_ACCOUNT_CLOSED;
import static com.pkb.emis.appointment.service.ErrorCode.REACHED_UNVERIFIED_USER_MAX_BOOKING_COUNT;
import static com.pkb.emis.appointment.service.ErrorCode.RESTRICTED_PATIENT;
import static com.pkb.emis.appointment.service.ErrorCode.UNLINKED_PATIENT;
import static com.pkb.integrations.emis.model.datatypes.BookingMethod.PatientFacingServices;
import static com.pkb.integrations.emis.model.datatypes.BookingMethod.Practice;
import static com.pkb.integrations.emis.model.datatypes.BookingMethod.Unknown;

public class EmisMessageHelper {

    public enum AccessType {
        PRIMARY,
        VIEWING
    }

    private static final String[] EMPTY = new String[]{};
    private final ImmutableMap<BookingMethod, String> BOOKING_METHOD_I18N_MAPPER = Maps.immutableEnumMap(ImmutableMap.of(
            Unknown, "viewAppointment.gp.surgery.booking.method.unknown",
            Practice, "viewAppointment.gp.surgery.booking.method.practice",
            PatientFacingServices, "viewAppointment.gp.surgery.booking.method.pfs"));

    private EmisAppointmentManager emisAppointmentManager;
    private OdsService odsService;
    private MessageSource messageSource;

    public EmisMessageHelper(OdsService odsService, EmisAppointmentManager emisAppointmentManager, MessageSource messageSource) {
        this.odsService = odsService;
        this.emisAppointmentManager = emisAppointmentManager;
        this.messageSource = messageSource;
    }

    private Set<ErrorCode> errorCodesWithProperties = Sets.immutableEnumSet(UNLINKED_PATIENT, EMIS_ACCOUNT_CLOSED, REACHED_UNVERIFIED_USER_MAX_BOOKING_COUNT, RESTRICTED_PATIENT);

    public String getEmisActionError(ErrorCode error, Locale locale, EmisContextHolder emisContextHolder) {
        return messageSource.getMessage(error.getI18nEmisErrorKey(), getMessageParams(locale, error, emisContextHolder), locale);
    }

    private String[] getMessageParams(Locale locale, ErrorCode error, EmisContextHolder emisContextHolder) {
        if (errorCodesWithProperties.contains(error)) {
            return new String[]{ getPracticeNameForCurrentUser(locale, emisContextHolder) };
        }
        return EMPTY;
    }

    @NotNull
    private String getPracticeNameForCurrentUser(Locale locale, EmisContextHolder emisContextHolder) {
        return getPracticeNameForPersonId(locale, emisContextHolder);
    }

    @NotNull
    public String getPracticeNameForPersonId(Locale locale, EmisContextHolder emisContextHolder) {
        return emisAppointmentManager.getNationalPracticeCode(emisContextHolder)
                .map(code -> getPracticeName(locale, code))
                .getOrElse(() -> messageSource.getMessage("global.txt.gender.unknown", null, locale));
    }

    @NotNull
    private String[] getParamsPracticeNameForPersonId(Locale locale, EmisContextHolder emisContextHolder) {
        return new String[]{ getPracticeNameForPersonId(locale, emisContextHolder) };
    }

    @NotNull
    public String getPracticeName(Locale locale, String practiceCode) {
        return odsService
                .getPracticeName(practiceCode)
                .orElseGet(() -> messageSource.getMessage("viewAppointment.gp.surgery.booking.ods.code", new String[]{ practiceCode }, locale));
    }

    public String getEmisAppointmentSourceText(Locale locale, EmisContextHolder emisContextHolder, EmisAppointment emisAppointment) {
        return messageSource.getMessage(getInternationalisationKey(emisAppointment.getBookingMethod()), getParamsPracticeNameForPersonId(locale, emisContextHolder), locale);
    }

    private String getInternationalisationKey(BookingMethod bookingMethod) {
        return BOOKING_METHOD_I18N_MAPPER.get(bookingMethod);
    }

    public String retrieveAppointmentsError(ErrorCode error, Locale locale, EmisContextHolder emisContextHolder) {
        EmisPfsStatsCollector.countRetrieveAppointmentsErrors(error);
        return getEmisActionError(error, locale, emisContextHolder);
    }

    public List<AppointmentDTO> addEmisAppointmentsToAppointments(Locale locale, List<AppointmentDTO> appointments, List<EmisAppointment> emisAppointments,
                                                                  Instant currentDateTime, PrivacyFlag privacyFlag, Route route, AccessType accessType, EmisContextHolder emisContextHolder) {
        appointments.addAll(emisAppointments.stream()
                .map(emisAppointment -> convertToAppointmentDTO(emisAppointment, currentDateTime, privacyFlag, route, accessType, getEmisAppointmentSourceText(locale, emisContextHolder, emisAppointment)))
                .collect(Collectors.toList()));

        appointments.sort(Comparator.comparing(AppointmentDTO::getStartTimestamp));
        return appointments;
    }

    public AppointmentDTO convertToAppointmentDTO(EmisAppointment emisAppointment, Instant currentDateTime, PrivacyFlag privacyFlag, Route route,
                                                  AccessType accessType, String practiceName) {
        AppointmentDTO app = new AppointmentDTO(new SourceDetails(null, Route.EMIS_PFS, practiceName));

        //WARNING: This is broken. See https://pkbdev.atlassian.net/browse/PHR-7090
        app.setId(emisAppointment.getSlotId());

        app.setEmisSlotId(emisAppointment.getSlotId());
        app.setStartTimestamp(emisAppointment.getStartTime().atZone(Constants.APPLICATION_TZ).toInstant());
        app.setEndTimestamp(emisAppointment.getEndTime().atZone(Constants.APPLICATION_TZ).toInstant());
        emisAppointment.getSlotTypeName().ifPresent(app::setSubject);
        app.setLocation(emisAppointment.getSession().getLocation().getLocationName());

        app.setEmisAppointmentLive(accessType == PRIMARY && app.getStartTimestamp().isAfter(currentDateTime));

        // only ACTIVE for now
        app.setStatus(AppointmentStatus.ACTIVE);

        app.setPrivacyFlag(privacyFlag);

        app.setNonPkbParticipants(getClinicianStrings(emisAppointment));

        return app;
    }

    @NotNull
    private List<String> getClinicianStrings(EmisAppointment emisAppointment) {
        return emisAppointment.getSession().getClinicians().stream()
                .map(clinician -> clinician.getTitle() + " " + clinician.getForenames() + " " + clinician.getSurname() + " (" + clinician.getJobRole() + ")")
                .collect(Collectors.toList());
    }
}
