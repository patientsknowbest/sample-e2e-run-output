package com.pkb.action.emis.pfs;

import com.pkb.emis.appointment.service.ErrorCode;
import io.prometheus.client.Counter;

import java.util.List;

import static com.pkb.emis.appointment.service.ErrorCode.PFS_SERVICE_UNAVAILABLE;

public class EmisPfsStatsCollector {

    private static final Counter patientRegistrationTotalCounter = Counter.build()
            .name("pkb_phr_web_patients_linking_emis_total")
            .help("Total number of patients linking to their EMIS GP practice")
            .register();

    private static final Counter patientRegistrationPerPracticeCounter = Counter.build()
            .name("pkb_phr_web_patients_linking_emis_per_practice")
            .help("Number of patients linking to their GP practice per EMIS practice")
            .labelNames("ods_code")
            .register();

    private static final Counter patientRegistrationPerPKBOrgCounter = Counter.build()
            .name("pkb_phr_web_patients_linking_emis_per_pkborg")
            .help("Number of patients linking to their GP practice per PKB Org")
            .labelNames("org_code")
            .register();

    private static final Counter patientRegistrationErrorCounter = Counter.build()
            .name("pkb_phr_web_patients_linking_emis_error")
            .help("Number of EMIS registration errors")
            .labelNames("error_code")
            .register();

    private static final Counter bookingTotalCounter = Counter.build()
            .name("pkb_phr_web_patients_booking_emis_total")
            .help("Total number of patients booking appointments through EMIS")
            .register();

    private static final Counter bookingPerPracticeCounter = Counter.build()
            .name("pkb_phr_web_patients_booking_emis_per_practice")
            .help("Number of patients booking appointments per EMIS practice")
            .labelNames("ods_code")
            .register();

    private static final Counter patientBookingErrorCounter = Counter.build()
            .name("pkb_phr_web_patients_booking_emis_error")
            .help("Number of EMIS booking errors")
            .labelNames("error_code")
            .register();

    private static final Counter cancelAppointmentTotalCounter = Counter.build()
            .name("pkb_phr_web_cancel_appointment_emis_total")
            .help("Total number of EMIS appointment cancel")
            .register();

    private static final Counter cancelAppointmentPerPracticeCounter = Counter.build()
            .name("pkb_phr_web_cancel_appointment_emis_per_practice")
            .help("Number of EMIS cancel per practice")
            .labelNames("ods_code")
            .register();

    private static final Counter cancelAppointmentErrorCounter = Counter.build()
            .name("pkb_phr_web_cancel_appointment_emis_error")
            .help("Number of EMIS cancel errors")
            .labelNames("error_code")
            .register();

    private static final Counter retreiveAppointmentsErrorCounter = Counter.build()
            .name("pkb_phr_web_retrieve_appointments_emis_error")
            .help("Number of EMIS retrieve appointments errors")
            .labelNames("error_code")
            .register();

    private static final Counter unreachablePFSCounter = Counter.build()
            .name("pkb_phr_web_emis_pfs_unreachable_error")
            .help("Number of requests when EMIS Patient Facing Services are unreachable")
            .register();

    private static final Counter bookingAccountErrorCounter = Counter.build()
            .name("pkb_phr_web_emis_booking_account_errors")
            .help("Number of patient account errors when booking an appointment")
            .labelNames("error_code")
            .register();

    static void countRegistrationPerPractice(String odsCode) {
        patientRegistrationPerPracticeCounter.labels(odsCode).inc();
        patientRegistrationTotalCounter.inc();
    }

    static void countRegistrationPerPKBOrg(List<String> orgCodesForPKBPerson) {
        orgCodesForPKBPerson
                .forEach(o -> patientRegistrationPerPKBOrgCounter.labels(o).inc());
    }

    static void countRegistrationErrors(ErrorCode error) {
        incrementErrorCounterAndCheckPfsAvailable(patientRegistrationErrorCounter, error);
    }

    public static void countBookingPerPractice(String odsCode) {
        bookingPerPracticeCounter.labels(odsCode).inc();
        bookingTotalCounter.inc();
    }

    public static void countBookingErrors(ErrorCode error) {
        incrementErrorCounterAndCheckPfsAvailable(patientBookingErrorCounter, error);
    }

    public static void countCancelAppointmentPerPractice(String ods) {
        cancelAppointmentPerPracticeCounter.labels(ods).inc();
        cancelAppointmentTotalCounter.inc();
    }

    public static void countCancelAppointmentErrors(ErrorCode error) {
        incrementErrorCounterAndCheckPfsAvailable(cancelAppointmentErrorCounter, error);
    }

    public static void countRetrieveAppointmentsErrors(ErrorCode error) {
        incrementErrorCounterAndCheckPfsAvailable(retreiveAppointmentsErrorCounter, error);
    }

    public static void countBookingAccountErrors(ErrorCode error) {
        incrementErrorCounterAndCheckPfsAvailable(bookingAccountErrorCounter, error);
    }

    private static void incrementErrorCounterAndCheckPfsAvailable(Counter counter, ErrorCode error) {
        counter.labels(error.name()).inc();
        if (error == PFS_SERVICE_UNAVAILABLE) {
            unreachablePFSCounter.inc();
        }
    }
}
