package com.pkb.action.emis.pfs;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.action.diary.BaseAppointmentAction;
import com.pkb.institute.entity.Org;
import com.pkb.integrations.emis.model.datatypes.ApplicationLinkLevel;
import com.pkb.integrations.emis.model.restcalls.EmisAccountLinkRequest;
import com.pkb.integrations.emis.model.restcalls.ImmutableEmisAccountLinkRequest;
import com.pkb.integrations.emis.model.restcalls.ImmutableLinkageDetails;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;

import java.time.LocalDate;
import java.util.List;
import java.util.stream.Collectors;

public class RegisterForEmisPfsAction extends BaseAppointmentAction {

    private String linkageKey;
    private String odsCode;
    private String accountId;
    private String tab;
    private String subTab;

    public String getTab() {
        return tab;
    }

    public void setTab(String tab) {
        this.tab = tab;
    }

    public String getSubTab() {
        return subTab;
    }

    public void setSubTab(String subTab) {
        this.subTab = subTab;
    }

    @Override
    public String input() {
        return emisAppointmentManager.getSessionInfo(getEmisContextHolder())
                .map(sessionData -> sessionData.sessionType() == ApplicationLinkLevel.Restricted ? Action.SUCCESS : "redirect")
                .getOrElseGet(errorCode -> Action.SUCCESS);
    }

    @Override
    public String execute() {
        PKBPerson pkbPerson = userManager.getPKBPerson(loggedInUserUtil.getLoggedInUserId(), PKBPerson.Lazy.CONTACTS);
        LocalDate dateOfBirth = pkbPerson.findDateOfBirth().orElseThrow(() -> new RuntimeException("Patient with id " + pkbPerson.getId() + " missing DOB"));
        EmisAccountLinkRequest request = ImmutableEmisAccountLinkRequest.builder()
                .dateOfBirth(dateOfBirth)
                .linkageDetails(ImmutableLinkageDetails.builder()
                        .linkageKey(linkageKey.trim())
                        .accountId(accountId.trim())
                        .nationalPracticeCode(odsCode.trim())
                        .build())
                .surname(pkbPerson.getLastName())
                .build();

        return emisAppointmentManager.registerForUse(getEmisContextHolder(), request)
                .map(accessIdentityGuid -> {
                    addActionMessage(getText("emis.successfully.linked"));
                    EmisPfsStatsCollector.countRegistrationPerPractice(odsCode);
                    EmisPfsStatsCollector.countRegistrationPerPKBOrg(getOrgCodesForPKBPerson(pkbPerson));
                    return Action.SUCCESS;
                })
                .getOrElseGet(error -> {
                    //noinspection StrutsI18nError,SSBasedInspection
                    addActionError(emisMessageHelper.getEmisActionError(error, getLocale(), getEmisContextHolder()));
                    EmisPfsStatsCollector.countRegistrationErrors(error);
                    return Action.INPUT;
                });
    }

    public String cancel() {
        return BaseAction.CANCEL;
    }

    @Override
    public void validate() {
        if (StringUtils.isEmpty(linkageKey)) {
            addFieldError("linkageKey", getText("err.linkage.key.required"));
        }

        if (StringUtils.isEmpty(odsCode)) {
            addFieldError("odsCode", getText("err.ods.code.required"));
        }

        if (StringUtils.isEmpty(accountId)) {
            addFieldError("accountId", getText("err.account.id.required"));
        }
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public String getLinkageKey() {
        return linkageKey;
    }

    public void setLinkageKey(String linkageKey) {
        this.linkageKey = linkageKey;
    }

    public String getOdsCode() {
        return odsCode;
    }

    public void setOdsCode(String odsCode) {
        this.odsCode = odsCode;
    }

    public String getAccountId() {
        return accountId;
    }

    public void setAccountId(String accountId) {
        this.accountId = accountId;
    }

    @NotNull
    private List<String> getOrgCodesForPKBPerson(PKBPerson pkbPerson) {
        return userManager.findOrgsForPatient(pkbPerson.getId()).stream().map(Org::getCode).collect(Collectors.toList());
    }
}
