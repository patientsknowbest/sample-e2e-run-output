package com.pkb.messagehasher.decryptor;

import com.pkb.crypto.decryptionhelper.DecryptionHelper;
import com.pkb.crypto.decryptionhelper.dto.EncryptedDataWithSourceRecord;
import com.pkb.crypto.decryptionhelper.dto.ImmutableEncryptedDataWithSourceRecord;
import com.pkb.crypto.decryptionhelper.provider.accountkeys.AccountKeysProviderFactory;
import com.pkb.crypto.dto.AccountKeysDTO;
import com.pkb.messagehasher.Message;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.Optional;
import java.util.function.Function;

@Service
public class MessageDecryptor {

    final static Function<Message, EncryptedDataWithSourceRecord<Message>> MESSAGE_DTO_MAPPER = (message) -> ImmutableEncryptedDataWithSourceRecord.<Message>builder()
            .dataIdentifier("msg:" + message.getId())
            .encryptedData(message.getEncryptedData())
            .secretKeyAlgorithm(Optional.ofNullable(message.getSecretKeyAlgorithm()))
            .dESSecretKey(Optional.ofNullable(message.getSecretKeyInline()))
            .symmetricEncryptionKeyId(Optional.ofNullable(message.getSymmetricEncryptionKeyId()))
            .nonceBase64(Optional.ofNullable(message.getNonceBase64()))
            .build();

    @Autowired
    DecryptionHelper localDecryptor;

    @Autowired
    AccountKeysProviderFactory<Message> accountKeysProviderFactory;


    public byte[] decryptEncryptedData(Message message, AccountKeysDTO accountKeys) {
        return localDecryptor.decrypt(message,
                accountKeysProviderFactory.symmetricKeyFromAccountKeysProvider(accountKeys),
                accountKeysProviderFactory.accountPrivateKeyFromAccountKeysProvider(accountKeys),
                MESSAGE_DTO_MAPPER);

    }
}
