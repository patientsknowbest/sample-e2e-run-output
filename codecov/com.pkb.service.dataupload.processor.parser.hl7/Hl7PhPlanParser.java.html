package com.pkb.service.dataupload.processor.parser.hl7;

import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.externalid.entity.ExternalIdScheme;
import com.pkb.externalid.entity.ExternalIdScope;
import com.pkb.externalid.service.ExternalIdService;
import com.pkb.phplan.entity.PHPlan;
import com.pkb.service.dataupload.hl7.HL7ConnContext;
import com.pkb.service.dataupload.hl7.HL7ParsingManager;
import com.pkb.service.dataupload.hl7.HL7Wrapper;
import com.pkb.service.dataupload.processor.UploadedDataProcessingContext;
import com.pkb.service.phplan.PhPlanXhtmlValidator;
import com.pkb.service.reference.ReferenceDataManager;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.Collections;
import java.util.List;
import java.util.UUID;

import static com.pkb.service.dataupload.hl7.HL7MessageType.MDM_T02;

@Component
public class Hl7PhPlanParser extends AbstractHl7Parser<PHPlan, UploadedDataProcessingContext> {
    private final PhPlanXhtmlValidator phPlanXhtmlValidator;
    private final ReferenceDataManager referenceDataManager;
    private final ExternalIdService externalIdService;

    @Autowired
    public Hl7PhPlanParser(HL7ParsingManager hl7ParsingManager, PhPlanXhtmlValidator phPlanXhtmlValidator, ReferenceDataManager referenceDataManager, ExternalIdService externalIdService) {
        super(hl7ParsingManager);
        this.phPlanXhtmlValidator = phPlanXhtmlValidator;
        this.referenceDataManager = referenceDataManager;
        this.externalIdService = externalIdService;
    }

    @Override
    List<PHPlan> doParse(UploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper) {
        PHPlan result;
        if (MDM_T02 == wrapper.getHL7Xml().getMessageType()) {
            result = wrapper.getHL7Xml().parseMdmCarePlan(wrapper.getHl7ConnContext());

            result.setPatientId(context.patient().getId());
            result.setUploadedDataId(dto.getId());
            result.getBaseFields().setUploadedDataId(dto.getId());
            if (dto.getExternalMessageId() != null && !dto.getExternalMessageId().isEmpty()) {
                UUID uniqueId = getOrCreateUniqueIdForExternalId(wrapper.getHl7ConnContext(), dto.getExternalMessageId());
                result.getBaseFields().setUniqueId(uniqueId);
            }
            referenceDataManager.applyPrivacyLabels(result);
            validatePlan(result);
        } else {
            throw new IllegalArgumentException(String.format("Rejecting DTO %s, unknown message type", dto.getId()));
        }
        return Collections.singletonList(result);
    }

    private void validatePlan(PHPlan pHPlan) {
        phPlanXhtmlValidator.validatePlan(pHPlan);
    }

    public UUID getOrCreateUniqueIdForExternalId(@NotNull HL7ConnContext hl7ConnContext, @NotNull String externalId) {
        UUID sourceOrgId = hl7ConnContext.getSourceOrg().getPublicId();
        String scheme = hl7ConnContext.getHl7Partner()
                .map(partner -> ExternalIdScheme.getHL7PartnerScheme(partner.getCode()))
                .orElse(ExternalIdScheme.DEFAULT_SCHEME);

        return externalIdService.lookupOrCreateUniqueIdForExternalId(externalId, new ExternalIdScope(scheme, sourceOrgId));
    }

}
