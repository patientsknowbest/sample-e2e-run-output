package com.pkb.service.dataupload.processor.parser.hl7;

import com.pkb.app.entity.SourceDetails;
import com.pkb.calendar.entity.AppointmentDTO;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.entities.enums.AppointmentStatus;
import com.pkb.service.calendar.impl.AppointmentService;
import com.pkb.service.calendar.impl.CalendarManager;
import com.pkb.service.dataupload.hl7.HL7ParsingManager;
import com.pkb.service.dataupload.hl7.HL7Wrapper;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.UploadedAppointment;
import com.pkb.service.dataupload.processor.domain.UploadedAppointmentRequestType;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Component;

import java.util.Collections;
import java.util.List;

@Component
public class Hl7CalendarParser extends AbstractHl7CalendarParser {

    @Autowired
    public Hl7CalendarParser(AppointmentService appointmentService, HL7ParsingManager hl7ParsingManager, @Lazy CalendarManager calendarManager) {
        super(appointmentService, hl7ParsingManager, calendarManager);
    }

    @Override
    List<UploadedAppointment> create(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment) {
        return Collections.emptyList();
    }

    @Override
    List<UploadedAppointment> update(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment) {
        HL7XmlDoc hl7 = wrapper.getHL7Xml();
        return createUploadedAppointmentList(
                updateAppointment(dto, hl7, existingAppointment),
                UploadedAppointmentRequestType.UPDATE, dto, hl7);
    }

    @Override
    List<UploadedAppointment> cancel(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment) {
        SourceDetails sourceDetails = new SourceDetails(dto);
        return createUploadedAppointmentList(
                cancelAppointment(sourceDetails, existingAppointment),
                UploadedAppointmentRequestType.CANCEL, dto, wrapper.getHL7Xml());
    }

    @Override
    List<UploadedAppointment> dna(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment) {
        return createUploadedAppointmentList(
                dna(dto, context.patient(), existingAppointment),
                UploadedAppointmentRequestType.DNA, dto, wrapper.getHL7Xml());
    }

    private AppointmentDTO updateAppointment(UploadedDataDTO dto, HL7XmlDoc hl7, AppointmentDTO existingAppointment) {

        // Set any new data on the existing appointment
        hl7.parseUpdatedAppointment(existingAppointment);
        existingAppointment.getSource().init(dto);

        //an update message should always reset the appointment status to active (ie unset cancelled or DNA):
        existingAppointment.setStatus(AppointmentStatus.ACTIVE);

        existingAppointment.generateNewRandomUniqueId();
        return existingAppointment;

    }

    private AppointmentDTO cancelAppointment(SourceDetails sourceDetails, AppointmentDTO existingAppointment) {

        // TODO: How do we record who deleted this data point?

        // Go! This is what we came for
        existingAppointment.setStatus(AppointmentStatus.CANCELLED);
        existingAppointment.generateNewRandomUniqueId();
        populateAppointmentRequestListSource(sourceDetails, existingAppointment);
        return existingAppointment;

    }

    private AppointmentDTO dna(UploadedDataDTO dto, PKBPerson patient, AppointmentDTO existingAppointment) {
        // Set the status of the whole appointment to DNA

        existingAppointment.setStatus(AppointmentStatus.DNA);
        existingAppointment.generateNewRandomUniqueId();
        // Find the request for this specific patient, and set DNA
        existingAppointment.getAppointmentRequestList().stream().filter(req -> req.getParticipantId().equals(patient.getIdString()))
                .forEach(req -> {
                    req.setDidNotAttend(true);
                });

        existingAppointment.getSource().init(dto);
        return existingAppointment;
    }

    @NotNull
    private List<UploadedAppointment> createUploadedAppointmentList(AppointmentDTO appointment, UploadedAppointmentRequestType create, UploadedDataDTO uploadedDataDTO, HL7XmlDoc hl7) {
        SourceDetails sourceDetails = new SourceDetails(uploadedDataDTO);
        return Collections.singletonList(new UploadedAppointment(appointment, create, sourceDetails, getExternalIntegrationUrl(hl7)));
    }
}
