package com.pkb.service.dataupload.processor.parser.hl7;

import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.service.dataupload.hl7.HL7ParsingManager;
import com.pkb.service.dataupload.hl7.HL7Wrapper;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.ParsedObservationInformation;
import com.pkb.service.reference.ReferenceDataManager;
import com.pkb.test.UploadedDataDeletionMetadata;
import com.pkb.test.entity.MeasurementDTO;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.Collections;
import java.util.List;
import java.util.Optional;

import static com.pkb.service.dataupload.hl7.HL7MessageType.ORU_R01;

@Component
public class Hl7MeasurementParser extends AbstractHl7Parser<ParsedObservationInformation<MeasurementDTO>, LoggedInUploadedDataProcessingContext> {
    private ReferenceDataManager referenceDataManager;

    @Autowired
    public Hl7MeasurementParser(HL7ParsingManager hl7ParsingManager, ReferenceDataManager referenceDataManager) {
        super(hl7ParsingManager);
        this.referenceDataManager = referenceDataManager;
    }

    @Override
    List<ParsedObservationInformation<MeasurementDTO>> doParse(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper) {
        HL7XmlDoc hl7 = wrapper.getHL7Xml();
        if (ORU_R01 == hl7.getMessageType()) {
            Optional<UploadedDataDeletionMetadata> maybeDeletionInfo = hl7.parseMeasurementDeletions(wrapper.getHl7ConnContext(), dto.getId());
            List<MeasurementDTO> measurements = wrapper.getHL7Xml().parseMeasurements(wrapper.getHl7ConnContext(), referenceDataManager);
            measurements.forEach(measurement -> measurement.getBaseFields().setUploadedDataId(dto.getId()));
            return Collections.singletonList(new ParsedObservationInformation<>(maybeDeletionInfo.orElse(null), measurements));
        } else {
            throw new IllegalArgumentException(String.format("Rejecting DTO %s, unknown message type", dto.getId()));
        }
    }
}
