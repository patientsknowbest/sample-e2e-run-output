package com.pkb.service.dataupload.processor.parser.hl7;

import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.exception.ApiCallMalformedException;
import com.pkb.radiology.entity.RadiologyResult;
import com.pkb.service.dataupload.hl7.HL7ParsingManager;
import com.pkb.service.dataupload.hl7.HL7Wrapper;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;

import static com.pkb.service.dataupload.hl7.HL7MessageType.ORU_R01;

@Component
public class Hl7RadiologyParser extends AbstractHl7Parser<RadiologyResult, LoggedInUploadedDataProcessingContext> {

    @Autowired
    public Hl7RadiologyParser(HL7ParsingManager hl7ParsingManager) {
        super(hl7ParsingManager);
    }

    @Override
    List<RadiologyResult> doParse(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper) {
        List<RadiologyResult> result;
        if (ORU_R01 == wrapper.getHL7Xml().getMessageType()) {
            result = createReport(wrapper);
        } else {
            throw new IllegalArgumentException(String.format("Rejecting DTO %s, unknown message type", dto.getId()));
        }
        return result;
    }

    private List<RadiologyResult> createReport(HL7Wrapper wrapper) {
        List<RadiologyResult> resultList = wrapper.getHL7Xml().parseRadiologyResults(wrapper.getHl7ConnContext());
        if (resultList.isEmpty()) {
            throw new ApiCallMalformedException("radiology message " + wrapper.getHL7Xml().getMessageIdString() + ": no content found");
        }
        return resultList;
    }
}
