package com.pkb.service.dataupload.processor.parser.hl7;

import com.pkb.app.entity.SourceDetails;
import com.pkb.calendar.entity.AppointmentDTO;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.service.calendar.impl.AppointmentService;
import com.pkb.service.calendar.impl.CalendarManager;
import com.pkb.service.dataupload.hl7.HL7MessageType;
import com.pkb.service.dataupload.hl7.HL7ParsingManager;
import com.pkb.service.dataupload.hl7.HL7Wrapper;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.segment.HL7DatapointExternalIntegrationUrl;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.UploadedAppointment;
import com.pkb.service.dataupload.processor.domain.UploadedAppointmentRequestType;
import com.pkb.user.entity.PKBPerson;
import org.springframework.context.annotation.Lazy;

import java.util.List;
import java.util.Optional;

public abstract class AbstractHl7CalendarParser extends AbstractHl7Parser<UploadedAppointment, LoggedInUploadedDataProcessingContext> {
    private final AppointmentService appointmentService;
    private final CalendarManager calendarManager;

    AbstractHl7CalendarParser(AppointmentService appointmentService, HL7ParsingManager hl7ParsingManager, @Lazy CalendarManager calendarManager) {
        super(hl7ParsingManager);
        this.appointmentService = appointmentService;
        this.calendarManager = calendarManager;
    }

    @Override
    List<UploadedAppointment> doParse(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper) {
        HL7XmlDoc hl7 = wrapper.getHL7Xml();

        List<UploadedAppointment> result = null;
        UploadedAppointmentRequestType requestType = getAppointmentRequestType(hl7.getMessageType());
        if (requestType != null) {
            PKBPerson patient = context.patient();
            // Appointments have a multi-parser chain. If the current instance of AbstractHl7CalendarParser is an
            // Hl7CalendarPreParser, then it will gracefully create a new appointment if one does not already exist.
            // Hl7CalendarParser is then free to assume there will always be a non-null existing appointment, since the
            // pre-parser is always called first.
            AppointmentDTO existingAppointment = calendarManager.getAppointmentByExternalIdAndDecrypt(context.requestContext(), hl7.parseMandatoryAppointmentId(),
                    patient.getDefaultAccountId(), dto.getSourceOrg().getId());
            switch (requestType) {
                case CREATE:
                    result = create(context, dto, wrapper, existingAppointment);
                    break;
                case UPDATE:
                    result = update(context, dto, wrapper, existingAppointment);
                    break;
                case CANCEL:
                    result = cancel(context, dto, wrapper, existingAppointment);
                    break;
                case DNA:
                    result = dna(context, dto, wrapper, existingAppointment);
                    break;
                default:
                    rejectMessage(dto);
            }
        } else {
            rejectMessage(dto);
        }

        return result;
    }

    private void rejectMessage(UploadedDataDTO dto) {
        throw new IllegalArgumentException(String.format("Rejecting DTO %s, unknown message type", dto.getId()));
    }

    private UploadedAppointmentRequestType getAppointmentRequestType(HL7MessageType messageType) {
        UploadedAppointmentRequestType result = null;
        switch (messageType) {
            case SIU_S12:
                result = UploadedAppointmentRequestType.CREATE;
                break;
            case SIU_S13:
                result = UploadedAppointmentRequestType.UPDATE;
                break;
            case SIU_S14:
                result = UploadedAppointmentRequestType.UPDATE;
                break;
            case SIU_S15:
                result = UploadedAppointmentRequestType.CANCEL;
                break;
            case SIU_S26:
                result = UploadedAppointmentRequestType.DNA;
                break;
        }
        return result;
    }

    void populateAppointmentRequestListSource(SourceDetails sourceDetails, AppointmentDTO dto) {
        appointmentService.populateAppointmentRequestListSource(sourceDetails, dto);
    }

    abstract List<UploadedAppointment> create(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment);

    abstract List<UploadedAppointment> update(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment);

    abstract List<UploadedAppointment> cancel(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment);

    abstract List<UploadedAppointment> dna(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment);

    protected Optional<HL7DatapointExternalIntegrationUrl> getExternalIntegrationUrl(HL7XmlDoc hl7) {
        return hl7.parseHL7DatapointExternalIntegrationUrl();
    }

}
