package com.pkb.service.dataupload.processor.parser.hl7;

import com.pkb.app.entity.SourceDetails;
import com.pkb.calendar.entity.AppointmentDTO;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.entities.enums.AppointmentStatus;
import com.pkb.service.calendar.impl.AppointmentService;
import com.pkb.service.calendar.impl.CalendarManager;
import com.pkb.service.dataupload.hl7.HL7ParsingManager;
import com.pkb.service.dataupload.hl7.HL7Wrapper;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.UploadedAppointment;
import com.pkb.service.dataupload.processor.domain.UploadedAppointmentRequestType;
import com.pkb.service.reference.ReferenceDataManager;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Component;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

@Component
public class Hl7CalendarPreParser extends AbstractHl7CalendarParser {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private ReferenceDataManager referenceDataManager;

    @Autowired
    public Hl7CalendarPreParser(AppointmentService appointmentService, ReferenceDataManager referenceDataManager,
                                HL7ParsingManager hl7ParsingManager, @Lazy CalendarManager calendarManager) {
        super(appointmentService, hl7ParsingManager, calendarManager);
        this.referenceDataManager = referenceDataManager;
    }

    @Override
    List<UploadedAppointment> create(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment) {
        List<UploadedAppointment> newAppointment = prepareNewAppointment(context, dto, wrapper, null);
        List<UploadedAppointment> result = new ArrayList<>();
        if (existingAppointment != null) {
            LOGGER.info("Received S12 create appointment {} for DTO {} and appointment already exists."
                            + " We replace/overwrite the existing with the new.",
                    existingAppointment.getExternalId(), dto.getId());
            SourceDetails sourceDetails = new SourceDetails(dto);
            newAppointment.forEach(appointment -> {
                appointment.getAppointment().getBaseFields().setUniqueId(existingAppointment.getBaseFields().getUniqueId());
                appointment.getAppointment().setCommonId(existingAppointment.getCommonId());
                existingAppointment.getBaseFields().getPrivacyFlag().ifPresent(pf -> appointment.getAppointment().getBaseFields().setPrivacyFlag(pf));
                populateAppointmentRequestListSource(sourceDetails, appointment.getAppointment());
            });
            result.add(new UploadedAppointment(existingAppointment, UploadedAppointmentRequestType.DELETE, sourceDetails));
        }
        result.addAll(newAppointment);
        return result;
    }

    @Override
    List<UploadedAppointment> update(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment) {
        return prepareNewAppointment(context, dto, wrapper, existingAppointment);
    }

    @Override
    List<UploadedAppointment> cancel(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment) {
        return prepareNewAppointment(context, dto, wrapper, existingAppointment);
    }

    @Override
    List<UploadedAppointment> dna(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment) {
        return prepareNewAppointment(context, dto, wrapper, existingAppointment);
    }

    @NotNull
    private List<UploadedAppointment> prepareNewAppointment(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper, AppointmentDTO existingAppointment) {
        AppointmentDTO appointmentDTO;
        UploadedAppointmentRequestType requestType;
        SourceDetails sourceDetails = new SourceDetails(dto);
        HL7XmlDoc hl7 = wrapper.getHL7Xml();
        if (existingAppointment == null) {
            PKBPerson patient = context.patient();
            appointmentDTO = createAppointment(sourceDetails, dto, hl7, patient);
            requestType = UploadedAppointmentRequestType.CREATE;
        } else {
            appointmentDTO = existingAppointment;
            requestType = UploadedAppointmentRequestType.NO_OP;
        }
        return Collections.singletonList(new UploadedAppointment(appointmentDTO, requestType, sourceDetails, getExternalIntegrationUrl(hl7)));
    }

    private AppointmentDTO createAppointment(SourceDetails sourceDetails, UploadedDataDTO dto,
                                             HL7XmlDoc hl7, PKBPerson patient) {

        // Retrieve appointment info from the XML
        AppointmentDTO appointmentDTO = hl7.parseNewAppointment(patient, sourceDetails);
        appointmentDTO.setStatus(AppointmentStatus.ACTIVE);

        appointmentDTO.getBaseFields().generateNewRandomUniqueId();
        appointmentDTO.getBaseFields().setUploadedDataId(dto.getId());
        // Apply privacy labels
        referenceDataManager.applyPrivacyLabels(appointmentDTO);
        return appointmentDTO;
    }
}
