package com.pkb.service.dataupload.processor.parser.hl7;

import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.diagnosis.entity.Diagnosis;
import com.pkb.service.dataupload.hl7.HL7ParsingManager;
import com.pkb.service.dataupload.hl7.HL7Wrapper;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.segment.HL7Diagnosis;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;

@Component
public class Hl7DiagnosisParser extends AbstractHl7Parser<Diagnosis, LoggedInUploadedDataProcessingContext> {

    @Autowired
    public Hl7DiagnosisParser(HL7ParsingManager hl7ParsingManager) {
        super(hl7ParsingManager);
    }

    @Override
    List<Diagnosis> doParse(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper) {
        List<HL7Diagnosis> hl7ResultList = wrapper.getHL7Xml().parseHL7Diagnoses();
        return HL7XmlDoc.convertDiagnoses(hl7ResultList, dto.getId(), context.patientId(),
                context.patientAccountId(), wrapper.getHl7ConnContext());
    }
}
