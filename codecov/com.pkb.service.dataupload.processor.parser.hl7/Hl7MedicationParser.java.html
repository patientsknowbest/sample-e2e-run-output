package com.pkb.service.dataupload.processor.parser.hl7;

import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.medication.entity.Medication;
import com.pkb.service.dataupload.hl7.HL7ParsingManager;
import com.pkb.service.dataupload.hl7.HL7Wrapper;
import com.pkb.service.dataupload.hl7.HL7XmlDoc;
import com.pkb.service.dataupload.hl7.segment.HL7Medication;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;

@Component
public class Hl7MedicationParser extends AbstractHl7Parser<Medication, LoggedInUploadedDataProcessingContext> {

    @Autowired
    public Hl7MedicationParser(HL7ParsingManager hl7ParsingManager) {
        super(hl7ParsingManager);
    }

    @Override
    List<Medication> doParse(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, HL7Wrapper wrapper) {
        List<HL7Medication> hl7ResultList = wrapper.getHL7Xml().parseHL7Medications();
        return HL7XmlDoc.convertMedications(hl7ResultList,
                dto.getId(), context.patientId(), context.patientAccountId(), wrapper.getHl7ConnContext());
    }
}
