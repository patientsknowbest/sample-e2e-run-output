package com.pkb.backgroundtasks.routing

import com.pkb.backgroundtasks.config.BackgroundTasksProperties
import com.pkb.common.config.BaseConfig
import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import com.pkb.domain.pubsub.payload.GrantAccessToOrgRequest
import com.pkb.domain.pubsub.payload.GrantOrgAccessRequest
import org.apache.camel.component.google.pubsub.GooglePubsubComponent
import org.apache.commons.lang3.StringUtils
import org.springframework.beans.factory.annotation.Value
import org.springframework.context.annotation.Profile
import org.springframework.stereotype.Component

@Component
@Profile("!integration-test")
class CryptoAccessGrantRouteBuilder(
    private val baseConfig: BaseConfig,
    private val backgroundTasksProperties: BackgroundTasksProperties,
    private val pubSubNamespaceService: PubSubNamespaceService,
    @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}")
    private val serviceAccountKey: String?
) : AbstractNamespaceAwareRouteBuilder() {

    companion object {
        const val INBOUND_GRANT_ACCESS_TO_ORG_REQUEST = "direct:handleGrantAccessToOrgRequest"
        const val INBOUND_GRANT_ORG_ACCESS_REQUEST = "direct:handleGrantOrgAccessRequest"
    }

    override fun configure() {
        val component = context.getComponent("google-pubsub") as GooglePubsubComponent
        component.publisherCacheTimeout = 0
        if (this.baseConfig.pubSubEmulatorEndpoint != null) {
            component.endpoint = baseConfig.pubSubEmulatorEndpoint
            component.isAuthenticate = false
        }

        from(uri(baseConfig.pubSubProjectName, backgroundTasksProperties.grantAccessToOrgRequestSubscriptionName))
            .unmarshal().avro(GrantAccessToOrgRequest.getClassSchema())
            .to(INBOUND_GRANT_ACCESS_TO_ORG_REQUEST)


        from(uri(baseConfig.pubSubProjectName, backgroundTasksProperties.grantOrgAccessRequestSubscriptionName))
            .unmarshal().avro(GrantOrgAccessRequest.getClassSchema())
            .to(INBOUND_GRANT_ORG_ACCESS_REQUEST)

    }

    override fun camelNamespaceService(): PubSubNamespaceService = pubSubNamespaceService

    override fun throwExceptionForNamespaceMismatch(): Boolean = false

    private fun uri(pubSubProjectName: String?, topicName: String) =
        "google-pubsub:$pubSubProjectName:$topicName${if (StringUtils.isBlank(serviceAccountKey)) "" else "?serviceAccountKey=file://$serviceAccountKey"}"

}
