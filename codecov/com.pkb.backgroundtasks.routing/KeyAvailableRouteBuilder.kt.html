package com.pkb.backgroundtasks.routing

import com.pkb.backgroundtasks.config.BackgroundTasksProperties
import com.pkb.common.config.BaseConfig
import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import com.pkb.kms.shared.structures.pubsub.KeyAvailable
import com.pkb.kms.shared.structures.pubsub.KeyType
import org.apache.commons.lang3.StringUtils
import org.springframework.beans.factory.annotation.Value
import org.springframework.context.annotation.Profile
import org.springframework.stereotype.Component

/**
 * Configure Apache Camel to publish / subscribe to user keys becoming available.
 * Use some internal endpoints (constants *_KEY_AVAILABLE) in the places where
 * I actually want to send / handle messages. This class hooks up the internal
 * endpoints to google pub sub. Could swap for something else in future, or even
 * just wire them internally for integration testing, avoiding the need for pubsub
 * docker container.
 *
 * Use AbstractNamespaceAwareRouteBuilder for automatically handling namespace
 * changes between e2e tests.
 */
@Component
@Profile("!integration-test")
class KeyAvailableRouteBuilder(private val baseConfig: BaseConfig,
                               private val backgroundTasksProperties: BackgroundTasksProperties,
                               private val pubSubNamespaceService: PubSubNamespaceService,
                               @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}")
                               private val serviceAccountKey: String?) : AbstractNamespaceAwareRouteBuilder() {
    companion object {
        const val INBOUND_ACCOUNT_KEY_AVAILABLE_FOR_ACCOUNT_TASKS = "direct:handleAccountKeyAvailableForAccountMigrations"
        const val INBOUND_ORG_ACCOUNT_KEY_AVAILABLE = "direct:handleOrgAccountKeyAvailable"
    }

    override fun configure() {

        from(uri(baseConfig.pubSubProjectName, backgroundTasksProperties.keyAvailableForAccountMigrationsSubscriptionName))
            .unmarshal().avro(KeyAvailable.getClassSchema())
            .filter { setOf(KeyType.ACCOUNT_KEY, KeyType.ORG_ACCOUNT_KEY).contains((it.message.body as? KeyAvailable)?.keyType) }
            .choice()
            .`when` { (it.message.body as? KeyAvailable)?.keyType == KeyType.ACCOUNT_KEY }
            .to(INBOUND_ACCOUNT_KEY_AVAILABLE_FOR_ACCOUNT_TASKS)
            .otherwise()
            .to(INBOUND_ORG_ACCOUNT_KEY_AVAILABLE)
            .endChoice()

    }

    override fun camelNamespaceService(): PubSubNamespaceService = pubSubNamespaceService

    override fun throwExceptionForNamespaceMismatch(): Boolean = false

    private fun uri(pubSubProjectName: String?, topicName: String) = "google-pubsub:$pubSubProjectName:$topicName${if (StringUtils.isBlank(serviceAccountKey)) "" else "?serviceAccountKey=file://$serviceAccountKey"}"
}
