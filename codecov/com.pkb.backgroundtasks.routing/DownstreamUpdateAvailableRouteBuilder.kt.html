package com.pkb.backgroundtasks.routing

import com.pkb.backgroundtasks.config.BackgroundTasksProperties
import com.pkb.backgroundtasks.util.valueAsMutableMapOf
import com.pkb.backgroundtasks.wayfinder.recordsservice.routing.RecordsServiceRouteBuilder
import com.pkb.common.config.BaseConfig
import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import org.apache.camel.Message
import org.apache.camel.component.google.pubsub.GooglePubsubConstants.ATTRIBUTES
import org.apache.commons.lang3.StringUtils
import org.springframework.beans.factory.annotation.Value
import org.springframework.context.annotation.Profile
import org.springframework.stereotype.Component

@Component
@Profile("!integration-test")
class DownstreamUpdateAvailableRouteBuilder(private val baseConfig: BaseConfig,
                                            private val backgroundTasksProperties: BackgroundTasksProperties,
                                            private val pubSubNamespaceService: PubSubNamespaceService,
                                            @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}")
                                            private val serviceAccountKeyFilePath: String?) : AbstractNamespaceAwareRouteBuilder() {
    companion object {
        private const val ATTRIBUTE_ACTION = "action"
        private const val ATTRIBUTE_RESOURCE_TYPE = "resourceType"
        private val CHANGE_NOTIFICATION_ATTRIBUTES = setOf(ATTRIBUTE_RESOURCE_TYPE, ATTRIBUTE_ACTION)
    }

    override fun configure() {
        from(uri(baseConfig.pubSubProjectName, backgroundTasksProperties.aggregatedUpdateSubscriptionName))
                .routeId("incoming-downstream-change")
                .process {
                    val message = it.message
                    val attributes = getChangeNotificationAttributes(message)
                    val newBody = DownstreamChangeNotification(attributes, String(message.body as ByteArray))
                    message.body = newBody
                }
                .to(RecordsServiceRouteBuilder.INBOUND_NEW_APPOINTMENT)
    }

    private fun getChangeNotificationAttributes(message: Message) =
            message.headers.valueAsMutableMapOf<String, String>(ATTRIBUTES)
                    .filterKeys { CHANGE_NOTIFICATION_ATTRIBUTES.contains(it) }
                    .toMutableMap()

    override fun camelNamespaceService(): PubSubNamespaceService = pubSubNamespaceService

    override fun throwExceptionForNamespaceMismatch(): Boolean = false

    private fun uri(pubSubProjectName: String?, topicName: String) =
            "google-pubsub:$pubSubProjectName:$topicName${if (StringUtils.isBlank(serviceAccountKeyFilePath)) "" else "?serviceAccountKey=file://$serviceAccountKeyFilePath"}"

}
