package com.pkb.action.oauth2;

import com.google.common.base.Charsets;
import com.google.common.base.Joiner;
import com.pkb.action.BaseAction;
import com.pkb.datamodel.user.PersonIdsWithNhsLoginStatus;
import com.pkb.security.BaseAuthenticationProvider.TypedAuthentication;
import com.pkb.security.NhsLoginAuthenticationDetails;
import io.prometheus.client.Counter;
import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;
import org.apache.http.client.utils.URIBuilder;

import javax.servlet.http.HttpServletRequest;
import java.net.URISyntaxException;
import java.util.HashMap;
import java.util.List;
import java.util.Optional;

import static com.google.common.hash.Hashing.sha256;
import static com.pkb.util.Constants.NHS_LOGIN_IDS;
import static com.pkb.util.Constants.NHS_LOGIN_INFO;
import static java.util.Collections.singletonList;
import static java.util.Objects.nonNull;

public class NhsErrorAction extends BaseAction {

    private static final String PREFIX = "nhs.error.page.";
    private static final String MESSAGE = "message.";
    private static final String TITLE = "title.";
    private static final String DEFAULT = "default";
    private static final String SSO_LOGIN_REQUIRED_ERROR = "sso.login.required";

    private static final String BASE_BAD_CREDENTIALS_URL = "/nhs-login/login?forceNewMetadata=true";

    private static final Counter NHS_LOGIN_ERROR_CODE_COUNTER = Counter.build()
            .name("pkb_integrations_nhslogin_error")
            .labelNames("code")
            .help("Occurences of each error code")
            .register();

    private String error;

    public String handleError() {
        return SUCCESS;
    }

    public String getTitle() {
        return getField(TITLE);
    }

    public String getError() {
        return getField(MESSAGE);
    }

    private String getField(String field) {
        String encodedPersonInfo = getEncodedPersonInfo();

        if (error == null) {
            return getText(PREFIX + field + DEFAULT);
        }

        NHS_LOGIN_ERROR_CODE_COUNTER.labels(error).inc();

        switch (error) {
            case "age.limit":
            case "out.of.area":
            case "not.patient":
            case "person.has.access.frozen":
            case "contact.team.for.access":
                return getText(PREFIX + field + error);
            case "different.records":
            case "missing.defaultaccountid":
                return getText(PREFIX + field + error, List.of(config.getSupportNewTicketUrl(), config.getBaseURL()));
            case "bad.credentials":
                String link = BASE_BAD_CREDENTIALS_URL;
                try {
                    URIBuilder uriBuilder = new URIBuilder(BASE_BAD_CREDENTIALS_URL);
                    getRedirectPath().ifPresent(phrPath -> uriBuilder.addParameter("phrPath", phrPath));
                    link = uriBuilder.toString();
                } catch (URISyntaxException e) {
                    LOGGER.error("Invalid redirection endpoint", e);
                }

                return getText(PREFIX + field + error, singletonList(link));
            case "sso_login_required":
                return getText(PREFIX + field + SSO_LOGIN_REQUIRED_ERROR, new String[]{BASE_BAD_CREDENTIALS_URL, config.getBaseURL()});
            default:
                LOGGER.error("Failed NHS login ({}): {}", error, encodedPersonInfo);
                return getText(PREFIX + field + DEFAULT);
        }
    }

    public void setError(String error) {
        this.error = error;
    }

    Optional<String> getRedirectPath() {
        HttpServletRequest request = getRequest();

        @SuppressWarnings("unchecked")
        TypedAuthentication<NhsLoginAuthenticationDetails> info =
                (TypedAuthentication<NhsLoginAuthenticationDetails>) request.getSession().getAttribute(NHS_LOGIN_INFO);

        if (nonNull(info)) {
            return info.getDetails().getRedirectPath();
        }
        return Optional.empty();
    }

    String getEncodedPersonInfo() {
        try {
            Option<PersonIdsWithNhsLoginStatus> maybeIds = nhsLoginIds();
            Option<NhsLoginAuthenticationDetails> maybeInfo = nhsLoginInfo();

            var infoMap = new HashMap<String, String>();
            maybeIds
                    .peek(ids -> {
                        infoMap.put("userIds", ids.toString());
                    })
                    .onEmpty(() -> {
                        infoMap.put("userIds", "Missing");
                    });
            maybeInfo
                    .peek(info -> {
                        infoMap.put("loginInfo.isEmailVerified", String.valueOf(info.isEmailVerified()));
                        infoMap.put("loginInfo.nhsNumberSHA256", sha256Hash(info.getNhsNumber()));
                        infoMap.put("loginInfo.firstNameSHA256", sha256Hash(info.getFirstName()));
                        infoMap.put("loginInfo.lastNameSHA256", sha256Hash(info.getLastName()));
                    })
                    .onEmpty(() -> {
                        infoMap.put("loginInfo", "Missing");
                    });

            return Joiner.on(",").withKeyValueSeparator("=").join(infoMap);
        } catch (Exception ex) {
            LOGGER.error("Unexpected error building encoded person info", ex);
            return StringUtils.EMPTY;
        }
    }

    Option<PersonIdsWithNhsLoginStatus> nhsLoginIds() {
        return Option.of((PersonIdsWithNhsLoginStatus) getRequest().getSession().getAttribute(NHS_LOGIN_IDS));
    }

    Option<NhsLoginAuthenticationDetails> nhsLoginInfo() {
        TypedAuthentication<NhsLoginAuthenticationDetails> authc = (TypedAuthentication<NhsLoginAuthenticationDetails>)
                getRequest().getSession().getAttribute(NHS_LOGIN_INFO);

        if (authc != null) {
            return Option.of(authc.getDetails());
        } else {
            return Option.none();
        }
    }

    private String sha256Hash(String value) {
        return sha256().hashString(value, Charsets.UTF_8).toString();
    }
}
