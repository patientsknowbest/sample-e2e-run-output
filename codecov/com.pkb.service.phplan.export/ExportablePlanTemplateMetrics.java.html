package com.pkb.service.phplan.export;

import java.util.List;

public class ExportablePlanTemplateMetrics {
    private List<Long> symptomIds;
    private List<Long> measurmentTypeIds;
    private List<Long> testIds;

    public ExportablePlanTemplateMetrics(List<Long> symptomIds, List<Long> measurmentTypeIds, List<Long> testIds) {
        this.symptomIds = symptomIds;
        this.measurmentTypeIds = measurmentTypeIds;
        this.testIds = testIds;
    }

    public List<Long> getSymptomIds() {
        return symptomIds;
    }

    public List<Long> getMeasurmentTypeIds() {
        return measurmentTypeIds;
    }

    public List<Long> getTestIds() {
        return testIds;
    }
}
