package com.pkb.service.phplan.export;

import com.google.common.collect.ImmutableMap;
import com.google.common.collect.Maps;
import com.pkb.phplan.entity.TemplateMetric;
import org.springframework.stereotype.Component;

import java.util.Arrays;
import java.util.Collections;
import java.util.EnumMap;
import java.util.List;

@Component
public class TemplateMetricsFilter {
    static public ImmutableMap<TemplateMetric.MetricType, List<Long>> filterTemplateMetrics(List<TemplateMetric> templateMetrics) {
        EnumMap<TemplateMetric.MetricType, List<Long>> result = new EnumMap<>(TemplateMetric.MetricType.class);
        Arrays.stream(TemplateMetric.MetricType.values()).forEach(type -> result.put(type, Collections.emptyList()));

        if ((templateMetrics != null) && !templateMetrics.isEmpty()) {
            io.vavr.collection.Stream.ofAll(templateMetrics)
                    .groupBy(TemplateMetric::getMetricType)
                    .mapValues(metrics -> metrics.map(TemplateMetric::getMetricId).toJavaList())
                    .forEach(result::put);
        }

        return Maps.immutableEnumMap(result);
    }
}
