package com.pkb.action.registration.reset;

import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;

@Immutable(builder = false)
@Style(of = "verifiableBits", allParameters = true)
public interface VerifiableBits {
    String question();
    boolean unaffiliatedPatient();
    boolean hasDateOfBirth();
    boolean hasPostalCode();

    default boolean verifyLastName() {
        return !unaffiliatedPatient();
    }
    default boolean verifyDateOfBirth() {
        return hasDateOfBirth();
    }
    default boolean verifyPostalCode() {
        return !unaffiliatedPatient() && hasPostalCode();
    }
    default boolean verifySecurityQuestion() {
        return !unaffiliatedPatient();
    }
}
