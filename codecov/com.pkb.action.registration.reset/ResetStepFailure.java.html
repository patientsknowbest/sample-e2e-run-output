package com.pkb.action.registration.reset;

import com.pkb.datamodel.Email;
import io.vavr.control.Option;

import java.time.Instant;
import java.util.Set;
import java.util.UUID;

import static java.lang.String.format;
import static java.util.stream.Collectors.joining;

public interface ResetStepFailure {
    String message();

    static NoUserByEmail noUserByEmail(Email email) {
        return new NoUserByEmail(email);
    }

    static NotRegistered notRegistered(UUID publicId) {
        return new NotRegistered(publicId);
    }

    static TooManyInvalidDOBInputAttempts tooManyInvalidDOBInputAttempts(UUID publicId, Instant lockedUntil) {
        return new TooManyInvalidDOBInputAttempts(publicId, lockedUntil);
    }

    /**
     * Our Db is too permissive as of 2020 August.
     * It allows state to be persisted inconsistent with our business rules.
     */
    static InconsistentDB inconsistentDB(String message) {
        return new InconsistentDB(message);
    }

    static AlreadyResetByTeamMember alreadyResetByTeamMember(UUID publicId) {
        return new AlreadyResetByTeamMember(publicId);
    }

    static UnexpectedCompositeError unexpectedErrors(Set<? extends UserVerificationFailure> bits) {
        return new UnexpectedCompositeError(bits);
    }

    static UnexpectedError unexpectedError(String message, Throwable throwable) {
        return new UnexpectedError(message, throwable);
    }

    final class NoUserByEmail implements ResetStepFailure {
        private final Email email;

        NoUserByEmail(Email email) {
            this.email = email;
        }

        public Email getEmail() {
            return email;
        }

        @Override
        public String message() {
            return format("No user can be found by email=[%s]", Option.of(email).map(Email::address).getOrNull());
        }
    }

    final class NotRegistered implements UserVerificationFailure {
        private final UUID publicId;

        private NotRegistered(UUID publicId) {
            this.publicId = publicId;
        }

        @Override
        public String message() {
            return format("User with public id=[%s] is not registered", publicId);
        }
    }

    final class TooManyInvalidDOBInputAttempts implements UserVerificationFailure {
        private final UUID publicId;
        private final Instant lockedUntil;

        private TooManyInvalidDOBInputAttempts(UUID publicId, Instant lockedUntil) {
            this.publicId = publicId;
            this.lockedUntil = lockedUntil;
        }

        @Override
        public String message() {
            return format("User with public id=[%s] has too many failed date of birth input attempts. " +
                    "Attempts locked until [%s]", publicId, lockedUntil.toString());
        }

        /**
         * @return the date until the user is locked out from new date of birth input attempts
         */
        public Instant getLockedUntil() {
            return lockedUntil;
        }
    }

    final class InconsistentDB implements UserVerificationFailure {
        private final String message;

        private InconsistentDB(String message) {
            this.message = message;
        }

        @Override
        public String message() {
            return message;
        }
    }

    final class AlreadyResetByTeamMember implements UserVerificationFailure {
        private final UUID publicId;

        private AlreadyResetByTeamMember(UUID publicId) {
            this.publicId = publicId;
        }

        @Override
        public String message() {
            return format("Password was already reset by a team member for user with public id=[%s]", publicId);
        }
    }

    final class UnexpectedCompositeError implements UserVerificationFailure {
        private final Set<? extends UserVerificationFailure> bits;

        private UnexpectedCompositeError(Set<? extends UserVerificationFailure> bits) {
            this.bits = bits;
        }

        public Set<? extends UserVerificationFailure> getBits() {
            return bits;
        }

        @Override
        public String message() {
            return bits.stream()
                    .map(UserVerificationFailure::message)
                    .collect(joining(","));
        }
    }

    final class UnexpectedError implements UserVerificationFailure {
        private final String message;
        private final Throwable cause;

        private UnexpectedError(String message, Throwable cause) {
            this.message = message;
            this.cause = cause;
        }

        @Override
        public String message() {
            return message;
        }

        public Throwable getCause() {
            return cause;
        }
    }
}
