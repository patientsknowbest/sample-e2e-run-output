package com.pkb.action.registration.reset;

import com.pkb.datamodel.Email;
import org.immutables.value.Value.Check;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Redacted;
import org.immutables.value.Value.Style;

import java.time.LocalDate;
import java.util.Optional;

import static com.google.common.base.Preconditions.checkArgument;
import static io.vavr.Predicates.isNotNull;
import static org.apache.commons.lang3.StringUtils.isNoneBlank;

@Immutable(builder = false)
@Style(of = "matchingCriteria", allParameters = true)
public interface UserVerificationCriteria {
    @Redacted
    Email email();

    Optional<String> lastName();

    Optional<LocalDate> dateOfBirth();

    Optional<String> postalCode();

    String question();

    Optional<String> answer();

    @Check
    default void validate() {
        lastName().ifPresent(lastName -> checkArgument(isNoneBlank(lastName), "Lastname cannot be blank"));

        // Not really possible because of relevant parts of ForgotPasswordPersonalInfoAction.validate() and
        // ForgotPasswordPersonalInfoAction.execute() but better to double check.
        dateOfBirth()
                .ifPresent(dateOfBirth -> checkArgument(isNotNull().test(dateOfBirth), "Date of Birth cannot be null"));
        postalCode()
                .ifPresent(postalCode -> checkArgument(isNoneBlank(postalCode), "Postal code cannot be blank"));

        checkArgument(isNoneBlank(question()), "Question cannot be blank");
        answer().ifPresent(answer -> checkArgument(isNoneBlank(answer), "Answer cannot be blank"));
    }
}
