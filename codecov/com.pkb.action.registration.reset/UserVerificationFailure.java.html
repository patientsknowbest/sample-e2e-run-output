package com.pkb.action.registration.reset;

import java.util.Set;
import java.util.UUID;

import static java.lang.String.format;
import static java.lang.String.join;

public interface UserVerificationFailure extends ResetStepFailure {
    static DemographicsMismatch demographicsMismatch(UUID publicId, Set<String> mismatches) {
        return new DemographicsMismatch(publicId, mismatches);
    }

    static SecurityAnswerMismatch securityAnswerMismatch(UUID publicId) {
        return new SecurityAnswerMismatch(publicId);
    }

    final class DemographicsMismatch implements UserVerificationFailure {
        private final UUID publicId;
        private final Set<String> properties;

        private DemographicsMismatch(UUID publicId, Set<String> properties) {
            this.publicId = publicId;
            this.properties = properties;
        }

        public Set<String> getProperties() {
            return properties;
        }

        @Override
        public String message() {
            return format("User with public id=[%s] failed demographics verification. Failed properties=[%s]", publicId, join(",", properties));
        }
    }

    final class SecurityAnswerMismatch implements UserVerificationFailure {
        private final UUID publicId;

        private SecurityAnswerMismatch(UUID publicId) {
            this.publicId = publicId;
        }

        @Override
        public String message() {
            return format("User with public id=[%s] failed security question/answer verification", publicId);
        }
    }
}
