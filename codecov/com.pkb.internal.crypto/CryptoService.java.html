package com.pkb.internal.crypto;

import com.pkb.crypto.CryptoHelper;
import com.pkb.crypto.PKBKeyStore;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import javax.crypto.SecretKey;

import static com.pkb.crypto.CryptoHelper.ALGORITHM;

public class CryptoService {
    @NotNull public Either<String, PKBKeyStore> loadKeyStore(@NotNull byte[] bytes, @NotNull char[] password) {
        return CryptoHelper.loadKeyStore(bytes, password);
    }

    @NotNull public SecretKey secretKey(@NotNull String secret) {
        return CryptoHelper.secretKey(secret, ALGORITHM);
    }
    
    @NotNull public char[] decryptPassword(
            @NotNull byte[] encryptedPassword,
            @NotNull byte[] salt,
            int iterationCount,
            @NotNull SecretKey secretKey) {
        return CryptoHelper.decryptPassword(encryptedPassword, salt, iterationCount, secretKey);
    }
}
