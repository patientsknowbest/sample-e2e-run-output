package com.pkb.integrations.emis.model;

import java.time.LocalDateTime;
import java.util.Optional;

import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import com.fasterxml.jackson.annotation.JsonIdentityInfo;
import com.fasterxml.jackson.annotation.ObjectIdGenerators;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.google.common.collect.ComparisonChain;

import com.pkb.integrations.emis.model.datatypes.SlotTypeStatus;

@JsonDeserialize(as = ImmutableEmisAppointmentSlot.class)
@EmisApiStyle
@Value.Immutable
@JsonIdentityInfo(generator = ObjectIdGenerators.PropertyGenerator.class, property = "SlotId", scope = EmisAppointmentSlot.class)
@SuppressWarnings("NullableProblems")
public interface EmisAppointmentSlot extends Comparable<EmisAppointmentSlot> {

    @NotNull
    LocalDateTime getStartTime();

    @NotNull
    LocalDateTime getEndTime();

    @NotNull
    Optional<String> getSlotTypeName();

    @NotNull
    SlotTypeStatus getSlotTypeStatus();

    long getSlotId();

    @Override
    default int compareTo(@NotNull EmisAppointmentSlot o) {
        return ComparisonChain.start()
                .compare(getStartTime(), o.getStartTime())
                .compare(getEndTime(), o.getEndTime())
                .compare(getSlotId(), o.getSlotId())
                .result();
    }
}
