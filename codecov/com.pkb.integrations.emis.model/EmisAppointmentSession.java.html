package com.pkb.integrations.emis.model;

import java.time.LocalDateTime;

import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import com.fasterxml.jackson.annotation.JsonIdentityInfo;
import com.fasterxml.jackson.annotation.JsonIdentityReference;
import com.fasterxml.jackson.annotation.JsonProperty;
import com.fasterxml.jackson.annotation.JsonPropertyOrder;
import com.fasterxml.jackson.annotation.ObjectIdGenerators;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.google.common.collect.ComparisonChain;
import com.google.common.collect.ImmutableList;

import com.pkb.integrations.emis.model.datatypes.SessionType;

@JsonDeserialize(as = ImmutableEmisAppointmentSession.class)
@EmisApiStyle
@Value.Immutable
@JsonIdentityInfo(generator = ObjectIdGenerators.PropertyGenerator.class, property = "SessionId", scope = EmisAppointmentSession.class)
@JsonPropertyOrder({ "SessionName", "SessionId", "LocationId", "DefaultDuration", "SessionType", "NumberOfSlots", "ClinicianIds", "StartDate", "EndDate" })
@SuppressWarnings("NullableProblems")
public interface EmisAppointmentSession extends Comparable<EmisAppointmentSession> {

    @NotNull
    String getSessionName();

    @NotNull
    LocalDateTime getStartDate();

    @NotNull
    LocalDateTime getEndDate();

    @NotNull
    @JsonProperty("LocationId")
    @JsonIdentityReference(alwaysAsId = true)
    EmisLocation getLocation();

    int getDefaultDuration();

    @NotNull
    @JsonProperty("ClinicianIds")
    @JsonIdentityReference(alwaysAsId = true)
    ImmutableList<EmisClinician> getClinicians();

    int getNumberOfSlots();

    @NotNull
    SessionType getSessionType();

    long getSessionId();

    @Override
    default int compareTo(@NotNull EmisAppointmentSession o) {
        return ComparisonChain.start()
                .compare(getStartDate(), o.getStartDate())
                .compare(getEndDate(), o.getEndDate())
                .compare(getSessionId(), o.getSessionId())
                .result();
    }
}
