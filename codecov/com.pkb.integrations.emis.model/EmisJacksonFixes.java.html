package com.pkb.integrations.emis.model;

import java.time.LocalDate;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;
import java.util.stream.Collectors;

import javax.annotation.ParametersAreNonnullByDefault;

import org.jetbrains.annotations.NotNull;

import com.fasterxml.jackson.core.util.DefaultIndenter;
import com.fasterxml.jackson.core.util.DefaultPrettyPrinter;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.SerializationFeature;
import com.fasterxml.jackson.datatype.guava.GuavaModule;
import com.fasterxml.jackson.datatype.jdk8.Jdk8Module;
import com.fasterxml.jackson.datatype.jsr310.JavaTimeModule;
import com.fasterxml.jackson.datatype.jsr310.ser.LocalDateSerializer;
import com.fasterxml.jackson.datatype.jsr310.ser.ZonedDateTimeSerializer;

/**
 * This class contains some hacky methods that work around a bug in
 * Immutables/Jackson whereby object references don't deserialize
 * correctly, and the resulting object model ends up having intermediate
 * $Json objects in it. We apply a converter after deserialization that
 * uses the internal Immutables JsonCreator methods to complete the construction
 * of the relevant immutables objects. See
 * https://github.com/immutables/immutables/issues/666
 * https://github.com/FasterXML/jackson-databind/issues/1706
 *
 */
@ParametersAreNonnullByDefault
public class EmisJacksonFixes {

    @SuppressWarnings({ "deprecation", "ConstantConditions" })
    @NotNull
    private static EmisLocation fixLocation(EmisLocation location) {
        if (location instanceof ImmutableEmisLocation.Json) {
            return ImmutableEmisLocation.fromJson((ImmutableEmisLocation.Json) location);
        }
        return location;
    }

    @SuppressWarnings({ "deprecation", "ConstantConditions" })
    @NotNull
    private static EmisClinician fixClinician(EmisClinician clinician) {
        if (clinician instanceof ImmutableEmisClinician.Json) {
            return ImmutableEmisClinician.fromJson((ImmutableEmisClinician.Json) clinician);
        }
        return clinician;
    }

    @NotNull
    public static EmisAppointment fixAppointment(EmisAppointment appointment) {
        return ImmutableEmisAppointment.builder()
                .from(appointment)
                .session(fixSession(appointment.getSession()))
                .build();
    }

    @NotNull
    @SuppressWarnings({ "deprecation", "ConstantConditions" })
    public static EmisAppointmentSession fixSession(EmisAppointmentSession session) {
        if (session instanceof ImmutableEmisAppointmentSession.Json) {
            session = ImmutableEmisAppointmentSession.fromJson((ImmutableEmisAppointmentSession.Json) session);
        }

        return ImmutableEmisAppointmentSession.builder()
                .from(session)
                .clinicians(session.getClinicians().stream().map(EmisJacksonFixes::fixClinician).collect(Collectors.toList()))
                .location(fixLocation(session.getLocation()))
                .build();
    }

    @NotNull
    @SuppressWarnings({ "WeakerAccess", "UnusedReturnValue" })
    public static ObjectMapper configureObjectMapper(ObjectMapper mapper) {
        JavaTimeModule javaTimeModule = new JavaTimeModule();
        javaTimeModule.addSerializer(LocalDate.class,
                new LocalDateSerializer(DateTimeFormatter.ofPattern("yyyy-MM-dd'T'00:00:00")));
        javaTimeModule.addSerializer(ZonedDateTime.class,
                new ZonedDateTimeSerializer(DateTimeFormatter.ISO_INSTANT));
        DefaultIndenter defaultIndenter = DefaultIndenter.SYSTEM_LINEFEED_INSTANCE.withLinefeed("\n");
        mapper
                .registerModule(new Jdk8Module())
                .registerModule(new GuavaModule())
                .registerModule(javaTimeModule)
                .configure(SerializationFeature.INDENT_OUTPUT, true)
                .configure(SerializationFeature.WRITE_DATE_TIMESTAMPS_AS_NANOSECONDS, false)
                .configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false)
                .setDefaultPrettyPrinter(new DefaultPrettyPrinter().withObjectIndenter(defaultIndenter).withoutSpacesInObjectEntries().withArrayIndenter(defaultIndenter));
        return mapper;
    }

}
