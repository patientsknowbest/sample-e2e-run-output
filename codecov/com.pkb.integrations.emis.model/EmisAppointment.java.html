package com.pkb.integrations.emis.model;

import java.time.LocalDateTime;
import java.util.Optional;

import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import com.fasterxml.jackson.annotation.JsonIdentityReference;
import com.fasterxml.jackson.annotation.JsonProperty;
import com.fasterxml.jackson.annotation.JsonPropertyOrder;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.google.common.collect.ComparisonChain;

import com.pkb.integrations.emis.model.datatypes.BookingMethod;
import com.pkb.integrations.emis.model.datatypes.SlotTypeStatus;
import com.pkb.integrations.emis.model.datatypes.TelephoneContactType;

@EmisApiStyle
@Value.Immutable
@Value.Modifiable
@JsonDeserialize(as = ModifiableEmisAppointment.class)
@JsonPropertyOrder({ "SlotId", "StartTime", "EndTime", "BookingDate", "SlotTypeName",
        "SlotTypeStatus", "SessionId", "VidyoRoomUri", "BookingReason", "TelephoneAppointmentDetails", "BookingMethod" })
@SuppressWarnings("NullableProblems")
public interface EmisAppointment extends Comparable<EmisAppointment> {

    long getSlotId();

    @NotNull
    LocalDateTime getBookingDate();

    @NotNull
    LocalDateTime getStartTime();

    @NotNull
    LocalDateTime getEndTime();

    @NotNull
    Optional<String> getSlotTypeName();

    @NotNull
    SlotTypeStatus getSlotTypeStatus();

    @JsonProperty("SessionId")
    @JsonIdentityReference(alwaysAsId = true)
    EmisAppointmentSession getSession();

    @NotNull
    Optional<String> getVidyoRoomUri();

    @NotNull
    Optional<String> getBookingReason();

    //This is nullable rather than optional due to:
    //https://github.com/immutables/immutables/issues/1043
    @Nullable
    TelephoneDetails getTelephoneAppointmentDetails();

    @JsonDeserialize(as = ImmutableTelephoneDetails.class)
    @Value.Immutable
    @EmisApiStyle
    @SuppressWarnings("NullableProblems")
    interface TelephoneDetails {
        @NotNull
        String getTelephoneNumber();

        @NotNull
        TelephoneContactType getContactType();
    }

    @NotNull
    BookingMethod getBookingMethod();

    @Override
    default int compareTo(@NotNull EmisAppointment o) {
        return ComparisonChain.start()
                .compare(getStartTime(), o.getStartTime())
                .compare(getEndTime(), o.getEndTime())
                .compare(getSession().getSessionId(), o.getSession().getSessionId())
                .compare(getSlotId(), o.getSlotId())
                .result();
    }
}
