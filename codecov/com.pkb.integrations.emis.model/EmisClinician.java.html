package com.pkb.integrations.emis.model;

import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import com.fasterxml.jackson.annotation.JsonIdentityInfo;
import com.fasterxml.jackson.annotation.ObjectIdGenerators;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.google.common.collect.ComparisonChain;
import com.google.common.collect.ImmutableList;

import com.pkb.integrations.emis.model.datatypes.Sex;

@JsonDeserialize(as = ImmutableEmisClinician.class)
@EmisApiStyle
@Value.Immutable
@JsonIdentityInfo(generator = ObjectIdGenerators.PropertyGenerator.class, property = "ClinicianId", scope = EmisClinician.class)
@SuppressWarnings("NullableProblems")
public interface EmisClinician extends Comparable<EmisClinician> {

    long getClinicianId();

    @NotNull
    String getDisplayName();

    @NotNull
    String getForenames();

    @NotNull
    String getSurname();

    @NotNull
    String getTitle();

    @NotNull
    Sex getSex();

    @NotNull
    String getJobRole();

    @NotNull
    ImmutableList<Language> getLanguages();

    @JsonDeserialize(as = ImmutableLanguage.class)
    @Value.Immutable
    @EmisApiStyle
    interface Language {
        @NotNull
        String getLanguageCode();

        @NotNull
        String getDescription();
    }

    @Override
    default int compareTo(@NotNull EmisClinician o) {
        return ComparisonChain.start()
                .compare(getSurname(), o.getSurname())
                .compare(getForenames(), o.getForenames())
                .compare(getTitle(), o.getTitle())
                .compare(getClinicianId(), o.getClinicianId())
                .result();
    }
}
