package com.pkb.service.dataupload.emis.esv2.sftp;

import java.net.URI;
import java.nio.file.Path;
import java.util.Map;
import java.util.Optional;

import org.immutables.value.Value;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;

import com.pkb.service.dataupload.emis.esv2.ConnectionData;
import com.pkb.service.dataupload.emis.esv2.IEmisDownloadConnection;

@JsonDeserialize(as = ImmutableSftpConnectionData.class)
@Value.Immutable
public abstract class SftpConnectionData implements ConnectionData {

    @Value.Parameter
    public abstract String userName();

    @Value.Parameter
    public abstract String host();

    @Value.Parameter
    public abstract Integer port();

    @Value.Parameter
    @Value.Auxiliary
    public abstract Path privateKey();

    @Value.Auxiliary
    public abstract Optional<String> keyPassphrase();

    public abstract Map<String, String> options();

    @Value.Parameter
    public abstract String remoteDirectory();

    @Value.Parameter
    @Override
    public abstract String uniqueTag();

    @Value.Parameter
    public abstract Integer timeoutInMilliseconds();

    @Override
    public IEmisDownloadConnection connection() {
        return new SftpConnection(this);
    }

    @Override
    public URI uri() {
        return URI.create("sftp://" + host() + ":" + port() + remoteDirectory());
    }
}
