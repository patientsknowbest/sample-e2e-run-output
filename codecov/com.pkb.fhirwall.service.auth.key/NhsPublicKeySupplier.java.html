package com.pkb.fhirwall.service.auth.key;

import com.github.benmanes.caffeine.cache.Caffeine;
import com.github.benmanes.caffeine.cache.LoadingCache;
import com.pkb.common.ClearableInternalState;
import com.pkb.fhirwall.config.FhirwallConfig;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import org.slf4j.Logger;

import java.math.BigInteger;
import java.security.PublicKey;
import java.security.spec.RSAPublicKeySpec;
import java.util.function.Supplier;

import static com.google.gson.JsonParser.parseString;
import static java.lang.invoke.MethodHandles.lookup;
import static java.security.KeyFactory.getInstance;
import static java.util.concurrent.TimeUnit.HOURS;
import static org.keycloak.common.util.Base64Url.decode;
import static org.slf4j.LoggerFactory.getLogger;

public class NhsPublicKeySupplier implements Supplier<PublicKey>, ClearableInternalState {

    private static final Logger LOG = getLogger(lookup().lookupClass());
    private static final OkHttpClient HTTP_CLIENT = new OkHttpClient();

    private final LoadingCache<String, PublicKey> cache;
    private final FhirwallConfig fhirwallConfig;

    public NhsPublicKeySupplier(FhirwallConfig fhirwallConfig) {
        this.fhirwallConfig = fhirwallConfig;
        this.cache = Caffeine.newBuilder()
                .expireAfterWrite(1L, HOURS)
                .build(this::loadKey);
    }

    @Override
    public PublicKey get() {
        return cache.get("key");
    }

    @Override
    public void clearState() {
        cache.invalidateAll();
    }

    private PublicKey loadKey(String ignored) {
        var request = new Request.Builder()
                .url(fhirwallConfig.getNhsJwksUri())
                .get()
                .build();

        try (var response = HTTP_CLIENT.newCall(request).execute()) {
            if (response.isSuccessful()) {
                var key = parseString(response.body().string()).getAsJsonObject()
                        .get("keys").getAsJsonArray()
                        .get(0).getAsJsonObject();

                var modulus = new BigInteger(1, decode(key.get("n").getAsString()));
                var exponent = new BigInteger(1, decode(key.get("e").getAsString()));

                return getInstance("RSA").generatePublic(new RSAPublicKeySpec(modulus, exponent));
            } else {
                LOG.error("Can't read from NHS JWKS URI {}: {}", fhirwallConfig.getNhsJwksUri(), response.body().string());
                throw new RuntimeException("Can't read from NHS JWKS URI!");
            }
        } catch (Exception e) {
            LOG.error("Can't create public key from NHS JWKS URI {}", fhirwallConfig.getNhsJwksUri(), e);
            throw new RuntimeException("Can't create public key from NHS JWKS URI!", e);
        }
    }
}
