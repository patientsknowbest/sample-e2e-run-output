package com.pkb.action.privacyofficer;

import com.pkb.action.BaseAction;
import com.pkb.dto.PatientDTO;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.institute.entity.Org;
import com.pkb.mappers.PKBPersonMapper;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.user.entity.PKBPerson.Lazy;
import org.springframework.beans.factory.annotation.Autowired;

/**
 *
 * @author Russ Gray
 */
public class PrivacyOfficerEditPatientOptOutFormAction extends BaseAction {

    private static final long serialVersionUID = 5557073642369124471L;

    @Autowired
    private PKBPersonMapper pkbPersonMapper;

    private Long patientId;
    private PatientDTO patient;
    private String orgName;

    @Override
    public String execute() throws Exception {
        PKBPerson patientPerson = userManager.getPKBPerson(patientId, Lazy.NATIONAL_AND_LOCAL_IDS);
        Org org = loggedInUserUtil.getOptionalLoggedInUserOrg().orElseThrow(() -> new IllegalStateException("No Organization set!"));
        NationalIdType nationalIdType = NationalIdType.getPrimaryNationalIdTypeFromCountryCode(org.getCountry()).orElse(null);

        patient = pkbPersonMapper.personToPatientDTO(patientPerson, org, nationalIdType);
        orgName = org.getName();

        return SUCCESS;
    }

    public Long getPatientId() {
        return patientId;
    }

    public void setPatientId(Long patientId) {
        this.patientId = patientId;
    }

    public String getOrgName() {
        return orgName;
    }

    public void setOrgName(String orgName) {
        this.orgName = orgName;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public PatientDTO getPatient() {
        return patient;
    }

    public void setPatient(PatientDTO patient) {
        this.patient = patient;
    }

    public void setPkbPersonMapper(PKBPersonMapper pkbPersonMapper) {
        this.pkbPersonMapper = pkbPersonMapper;
    }
}
