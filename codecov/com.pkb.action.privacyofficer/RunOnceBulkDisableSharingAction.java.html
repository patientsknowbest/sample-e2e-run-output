package com.pkb.action.privacyofficer;

import com.pkb.action.BaseAction;
import com.pkb.document.entity.Attachment;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.service.team.OrgManager;
import com.pkb.service.team.TeamManager;
import com.pkb.service.team.TeamUserManager;
import com.pkb.user.entity.PKBPerson;
import org.apache.struts2.interceptor.SessionAware;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Value;

import java.util.Collections;
import java.util.List;
import java.util.Map;
import java.util.function.Function;
import java.util.stream.Collectors;
import java.util.stream.Stream;

/**
 * See https://pkbdev.atlassian.net/browse/PHR-5277
 * the original case is on hold; but this seems to be needed next for Dean Street in Chelwest (via Slack, 2019/09)
 * <p>
 * Currently merging but with functionality disabled.
 * <p>
 * We can consider this whole feature dead code if it's still unused in, say mid-2020 (being generous)
 */
public class RunOnceBulkDisableSharingAction extends BaseAction implements SessionAware {

    private static final long serialVersionUID = 1L;

    @Value("${feature.bulkdisablesharing.supportedOrgCode}")
    private String supportedOrgCode;
    private OrgManager orgManager;
    private TeamManager teamManager;
    private TeamUserManager teamUserManager;

    @Override
    public String execute() {
        PKBPerson loggedInPerson = loggedInUserUtil.getLoggedInUser();

        if (!loggedInPerson.isPrivacyOfficer()) {
            throw new IllegalStateException("Only privacy officer can call this action");
        }

        if (supportedOrgCode == null) {
            throw new IllegalStateException("Supported org code is not configured");
        }

        Org org = teamManager.getOrgByCode(supportedOrgCode).orElse(null);

        if (org == null) {
            throw new IllegalStateException("No org found for code " + supportedOrgCode + "; cannot continue");
        }

        Team officerPrimaryTeam = teamUserManager.findActivePrimaryTeam(getEHRRequestContext(), loggedInPerson.getId())
                .orElseThrow(() -> new IllegalArgumentException("no active primary team for " + loggedInPerson.getId()));

        if (!org.getId().equals(officerPrimaryTeam.getOrg().getId())) {
            throw new IllegalStateException(String.format("Signed in privacy officer %s does not belong to supported org %s", loggedInPerson.getId(), supportedOrgCode));
        }

        Map<Long, List<Attachment>> patientIdNoAttachmentsMap = mapPatientIdsWithoutAttachments();
        List<Map<String, Long>> sharingDisabledIds = orgManager.disableSharingForPatients(getLoggedInEHRRequestContext(), loggedInPerson, org, officerPrimaryTeam, patientIdNoAttachmentsMap);

        //noinspection StrutsI18nError,SSBasedInspection
        addActionMessage("Sharing disabled: confirmation IDs per patient: " + sharingDisabledIds);

        return SUCCESS;
    }

    @NotNull
    private Map<Long, List<Attachment>> mapPatientIdsWithoutAttachments() {
        return collectPatientIds().stream()
                .collect(Collectors.toMap(Function.identity(), id -> Collections.emptyList()));
    }

    private List<Long> collectPatientIds() {
        return Stream.of(getPatientIds().split(",")).map(Long::parseLong).collect(Collectors.toList());
    }

    private String getPatientIds() {
        return getConfig().getBulkDisableSharingPatientIds();
    }

    public void setOrgManager(OrgManager orgManager) {
        this.orgManager = orgManager;
    }

    public void setTeamManager(TeamManager teamManager) {
        this.teamManager = teamManager;
    }

    public void setTeamUserManager(TeamUserManager teamUserManager) {
        this.teamUserManager = teamUserManager;
    }
}
