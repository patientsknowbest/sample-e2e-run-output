public class PrivacyOfficerViewOptedOutPatientAction extends BaseAction {

    private static final long serialVersionUID = -3248795457235765612L;

    private Long patientId;
    private PatientSharingHistoryDTO sharingHistory;

    private OrgManager orgManager;

    private Long privacyOfficerOrgId;

    @Autowired
    private PatientSharingHistoryMapper patientSharingHistoryMapper;
    @Autowired
    private PatientOptOutDocumentationTransformer patientOptOutDocumentationTransformer;

    @Override
    public String execute() throws Exception {

        privacyOfficerOrgId = loggedInUserUtil.getLoggedInUserOrg().getId();

        PKBPerson patient = userManager.getPKBPerson(patientId);
        List<PatientOptOutDocumentation> optOutDocumentation = retrieveOptOutDocumentation(patientId);

        sharingHistory = patientSharingHistoryMapper.personAndDocumentationToDTO(patient, optOutDocumentation);

        return SUCCESS;
    }

    private List<PatientOptOutDocumentation> retrieveOptOutDocumentation(Long patientId) {
        return patientOptOutDocumentationTransformer.transformPatientOptOutDocumentation(orgManager.getOptOutDocumentation(getLoggedInEHRRequestContext(), patientId));
    }

    public Long getPatientId() {
        return patientId;
    }

    public void setPatientId(Long patientId) {
        this.patientId = patientId;
    }

    public PatientSharingHistoryDTO getSharingHistory() {
        return sharingHistory;
    }

    public void setOrgManager(OrgManager orgManager) {
        this.orgManager = orgManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public void setPatientSharingHistoryMapper(PatientSharingHistoryMapper patientSharingHistoryMapper) {
        this.patientSharingHistoryMapper = patientSharingHistoryMapper;
    }

    public Long getPrivacyOfficerOrgId() {
        return privacyOfficerOrgId;
    }
}
