package com.pkb.action.privacyofficer;

import com.pkb.action.BaseAction;
import com.pkb.dto.PatientDTO;
import com.pkb.dto.PatientSearchParametersDTO;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.entities.enums.ValidNationalId;
import com.pkb.institute.entity.Org;
import com.pkb.mappers.PKBPersonMapper;
import com.pkb.service.search.PatientSearchParameters;
import com.pkb.service.team.OrgManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.PatientSearchUtil;
import io.vavr.control.Either;
import org.apache.commons.lang3.StringUtils;
import org.apache.commons.validator.routines.IntegerValidator;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.List;
import java.util.function.Function;

import static java.util.stream.Collectors.toList;

public class PrivacyOfficerPatientSearchAction extends BaseAction {

    private static final long serialVersionUID = -3700231939434036377L;

    private static final PKBPerson.Lazy[] SEARCH_LAZIES = {
            PKBPerson.Lazy.NATIONAL_AND_LOCAL_IDS,
            PKBPerson.Lazy.PROPERTIES,
            PKBPerson.Lazy.CONTACTS };

    private PatientSearchParametersDTO searchParameters;
    private String orgName;
    private NationalIdType nationalIdType;
    private List<PatientDTO> patients;

    private OrgManager orgManager;
    @Autowired
    private PKBPersonMapper pkbPersonMapper;

    @Override
    public String execute() {
        Org org = loggedInUserUtil.getOptionalLoggedInUserOrg().orElseThrow(() -> new IllegalStateException("No Organization set!"));
        orgName = org.getName();
        nationalIdType = NationalIdType.getPrimaryNationalIdTypeFromCountryCode(org.getCountry()).orElse(null);

        Either<String, ValidNationalId> cleanSearchId = PatientSearchUtil.getCleanSearchId(
                StringUtils.trimToEmpty(searchParameters.getSearchId()),
                nationalIdType,
                searchParameters.getSearchIdType());
        searchParameters.setSearchId(cleanSearchId.fold(Function.identity(), vnid -> vnid.value()));

        PatientSearchParameters params = PatientSearchParameters.builder()
                .convertFrom(searchParameters)
                .build();

        patients = orgManager.searchPatients(getEHRRequestContext(), org, params, 50, SEARCH_LAZIES)
                .stream()
                .map(p -> pkbPersonMapper.personToPatientDTO(p, org, nationalIdType))
                .collect(toList());

        return SUCCESS;
    }

    @Override
    public void validate() {
        if (!validateIntegerIfPresent(searchParameters.getSearchDobYear())) {
            addActionError(getText("privacyOfficerLandingPageAction.err.invalid_year"));
        }
        if (!validateIntegerIfPresent(searchParameters.getSearchDobMonth())) {
            addActionError(getText("privacyOfficerLandingPageAction.err.invalid_month"));
        }
        if (!validateIntegerIfPresent(searchParameters.getSearchDobDay())) {
            addActionError(getText("privacyOfficerLandingPageAction.err.invalid_day"));
        }
        if (!PatientSearchParameters.SearchIdentifierType.contains(searchParameters.getSearchIdType())) {
            addActionError(getText("privacyOfficerLandingPageAction.err.invalid_id_type"));
        }
    }

    private boolean validateIntegerIfPresent(String stringValue) {
        // If no value is provided, it's OK
        if (StringUtils.isBlank(stringValue)) {
            return true;
        }

        return IntegerValidator.getInstance().isValid(stringValue);
    }

    public PatientSearchParametersDTO getSearchParameters() {
        return searchParameters;
    }

    public void setSearchParameters(PatientSearchParametersDTO searchParameters) {
        this.searchParameters = searchParameters;
    }

    public String getOrgName() {
        return orgName;
    }

    public NationalIdType getNationalIdType() {
        return nationalIdType;
    }

    public List<PatientDTO> getPatients() {
        return patients;
    }

    public void setOrgManager(OrgManager orgManager) {
        this.orgManager = orgManager;
    }

    public void setPkbPersonMapper(PKBPersonMapper pkbPersonMapper) {
        this.pkbPersonMapper = pkbPersonMapper;
    }
}
