package com.pkb.phoenix.api.service.freshdesk;

import org.immutables.value.Value.Auxiliary;
import org.immutables.value.Value.Check;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Redacted;
import org.immutables.value.Value.Style;

import java.nio.charset.StandardCharsets;
import java.util.List;

import static com.pkb.phoenix.api.service.phoenixexception.PhoenixValidationException.InvalidTaskParameters.invalidTaskParameters;
import static com.pkb.phoenix.api.service.tasktype.ValidationUtil.validateApiKey;
import static java.lang.String.format;
import static okio.ByteString.encodeString;

/**
 * An opaque string used for authentication with FreshDesk API.
 */
@Immutable
@Style(allParameters = true)
public
interface ApiKey {
    @Redacted
    String value();

    @Check
    default void validate() {
        validateApiKey(value())
                .ifPresent(validationError -> {
                    throw invalidTaskParameters(List.of("API key [" + value() + "] is not valid, because: [" + validationError + "]."));
                });
    }

    @Auxiliary
    default String encoded() {
        return encodeString(format("%s:X", value()), StandardCharsets.UTF_8).base64();
    }
}
