package com.pkb.phoenix.api.service.freshdesk;

import com.pkb.phoenix.api.service.KeycloakId;
import org.jetbrains.annotations.NotNull;
import org.springframework.data.redis.core.ReactiveValueOperations;
import reactor.core.publisher.Mono;

import java.time.Duration;
import java.util.Optional;

import static com.pkb.phoenix.api.service.freshdesk.FreshDeskApiKeyStatus.INVALID;
import static com.pkb.phoenix.api.service.freshdesk.FreshDeskApiKeyStatus.UNSET;
import static com.pkb.phoenix.api.service.freshdesk.FreshDeskApiKeyStatus.VALID;
import static com.pkb.phoenix.api.service.phoenixexception.PhoenixIllegalStateException.ConfigurationError.configurationError;
import static com.pkb.phoenix.api.service.phoenixexception.PhoenixIllegalStateException.ReadTimeoutException.readTimeoutException;
import static java.time.temporal.ChronoUnit.MILLIS;
import static reactor.core.publisher.Mono.error;
import static reactor.core.publisher.Mono.just;

public class RedisBackedApiKeyStore implements ApiKeyFinder {
    private final FreshDeskService freshDeskService;
    private final ReactiveValueOperations<String, String> valueOperations;
    private final Duration readTimeout = Duration.of(500, MILLIS);
    private final String prefix;

    public RedisBackedApiKeyStore(ReactiveValueOperations<String, String> valueOperations, FreshDeskService freshDeskService, String prefix) {
        this.valueOperations = valueOperations;
        this.freshDeskService = freshDeskService;
        this.prefix = prefix;
    }

    @Override
    public Optional<ApiKey> findApiKey(@NotNull KeycloakId keycloakId) {
        return valueOperations.get(phoenixKey(keycloakId))
                .blockOptional(readTimeout)
                .map(ImmutableApiKey::of);
    }

    public Mono<FreshDeskApiKeyStatus> setApiKey(KeycloakId keycloakId, ApiKey apiKey) {
        return freshDeskService.testApiKey(apiKey)
                .switchIfEmpty(error(new IllegalStateException("Failed to test API Key (got no conclusive result).")))
                .flatMap(status -> status == VALID ? // if valid, then we want to try to store it.
                        valueOperations.set(phoenixKey(keycloakId), apiKey.value())
                                .flatMap(isSet -> isSet ? just(VALID) : error(configurationError("Failed to store api key."))) : //
                        just(INVALID));
    }

    public Mono<FreshDeskApiKeyStatus> getApiKeyStatus(KeycloakId keycloakId) {
        return valueOperations.get(phoenixKey(keycloakId))
                .timeout(readTimeout, error(readTimeoutException("Could not read value for key=[" + keycloakId.value() + "] within [" + readTimeout + "]")))
                .flatMap(apiKey -> freshDeskService.testApiKey(ImmutableApiKey.of(apiKey))
                        .switchIfEmpty(error(new IllegalStateException("Failed to test API Key (got no conclusive result)."))))
                .defaultIfEmpty(UNSET);
    }

    private String phoenixKey(KeycloakId keycloakId) {
        return String.format("%s%s", prefix, keycloakId.value());
    }
}
