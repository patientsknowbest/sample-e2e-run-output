package com.pkb.phoenix.api.service.freshdesk;

import org.springframework.web.reactive.function.client.WebClient;
import reactor.core.publisher.Mono;

import static com.pkb.phoenix.api.service.freshdesk.FreshDeskApiKeyStatus.INVALID;
import static com.pkb.phoenix.api.service.freshdesk.FreshDeskApiKeyStatus.VALID;
import static com.pkb.phoenix.api.service.freshdesk.Status.closed;
import static java.lang.String.format;
import static org.springframework.http.HttpStatus.UNAUTHORIZED;
import static org.springframework.http.MediaType.APPLICATION_JSON;
import static reactor.core.publisher.Mono.error;
import static reactor.core.publisher.Mono.just;

public class FreshDeskService {
    private final WebClient client;

    public FreshDeskService(WebClient client) {
        this.client = client;
    }

    /**
     * Try provided API key to see if it can be used to authenticate to FD API.
     */
    Mono<FreshDeskApiKeyStatus> testApiKey(ApiKey apiKey) {
        return client.get().uri("/api/v2/agents")
                .header("Authorization", apiKey.encoded())
                .accept(APPLICATION_JSON)
                .exchangeToMono(response -> {
                    if (response.statusCode().is2xxSuccessful()) {
                        return just(VALID);
                    }

                    if (response.statusCode() == UNAUTHORIZED) {
                        return just(INVALID);
                    }

                    return response.bodyToMono(String.class)
                            .flatMap(body -> error(new IllegalStateException("FreshDesk returned an unexpected status code: [" + response.rawStatusCode() + "]. Response body: [" + body + "]")));
                });
    }

    //See api docs: https://developer.freshdesk.com/api/#reply_ticket
    public Mono<String> addNote(String ticketId, String message, ApiKey apiKey) {
        return client.post().uri(format("/api/v2/tickets/%s/reply", ticketId))
                .header("Authorization", apiKey.encoded())
                .contentType(APPLICATION_JSON)
                .body(just(new Message(message)), Message.class)
                .accept(APPLICATION_JSON)
                .exchangeToMono(response -> response.statusCode().is2xxSuccessful()  ? just("Note added") : error(new RuntimeException(format("Failed to add note to FD=[%s]. Response status code=[%s]", ticketId, response.rawStatusCode()))));
    }

    //see api docs: https://developer.freshdesk.com/api/#update_ticket
    public Mono<String> closeTicket(String ticketId, ApiKey apiKey) {
        return client.put().uri(format("/api/v2/tickets/%s", ticketId))
                .header("Authorization", apiKey.encoded())
                .contentType(APPLICATION_JSON)
                .body(just(new StatusUpdate(closed.code)), StatusUpdate.class)
                .accept(APPLICATION_JSON)
                .exchangeToMono(response -> response.statusCode().is2xxSuccessful()  ? just("Closed") : error(new RuntimeException(format("Failed to close FD=[%s]. Response status code=[%s]", ticketId, response.rawStatusCode()))));
    }
}
