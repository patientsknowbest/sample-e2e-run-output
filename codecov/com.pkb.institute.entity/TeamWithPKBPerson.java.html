package com.pkb.institute.entity;

import com.pkb.user.entity.PKBPerson;

import java.io.Serializable;

/**
 * This is a wrapper class rather than a normal hibernate entity. It's used when
 * we need to return multiple related existing entities from a single query.
 *
 * See com.pkb.institute.bean.InstituteUserBean.getAllTeamsWithUserForClinician
 * for example usage
 *
 * @author Russ Gray
 */
public class TeamWithPKBPerson implements Serializable {
    private static final long serialVersionUID = -4086386833018687972L;

    private final Team team;
    private final InstituteUser instituteUser;
    private final PKBPerson person;

    public TeamWithPKBPerson(Team team, InstituteUser instituteUser, PKBPerson person) {
        this.team = team;
        this.instituteUser = instituteUser;
        this.person = person;
    }

    public Team getTeam() {
        return team;
    }

    public InstituteUser getInstituteUser() {
        return instituteUser;
    }

    public PKBPerson getPerson() {
        return person;
    }

    @Override
    public String toString() {
        return "TeamWithPKBPerson{" + "team=" + team + ", instituteUser=" + instituteUser + ", person=" + person + '}';
    }
}
