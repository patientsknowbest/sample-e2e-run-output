package com.pkb.institute.entity;

import java.io.Serializable;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.OneToOne;
import javax.persistence.Table;

import org.hibernate.annotations.Cache;
import org.hibernate.annotations.CacheConcurrencyStrategy;

import com.pkb.symptom.entity.Symptom;

@Entity
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table(name = "teamsymptoms")
public class TeamSymptom implements Serializable {
    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @OneToOne(fetch = FetchType.LAZY, cascade = javax.persistence.CascadeType.ALL)
    @JoinColumn(name = "institute_id", referencedColumnName = "id")
    private Team team;

    @Column
    private String symptoms;

    @Column
    private Long alarmWeighting;

    @Column
    /**
     * The order in which to display the symptoms
     */
    private Long ordering;

    public Team getTeam() {
        return team;
    }

    public void setTeam(Team team) {
        this.team = team;
    }

    public Symptom getSymptom() {
        return Symptom.getByApiRef(symptoms);
    }

    public void setSymptom(Symptom symptom) {
        this.symptoms = symptom.getApiRef();
    }

    public Long getAlarmWeighting() {
        return alarmWeighting;
    }

    public void setAlarmWeighting(Long alarmWeighting) {
        this.alarmWeighting = alarmWeighting;
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    /**
     * @return the ordering
     */
    public Long getOrdering() {
        return ordering;
    }

    /**
     * @param ordering
     *            the ordering to set
     */
    public void setOrdering(Long ordering) {
        this.ordering = ordering;
    }

}
