/**
 * A medical organisation (e.g. hospital) containing at least one Team.
 *
 * @author robwhelan
 */
@Entity
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table
public class Org implements Serializable {

    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column(length = 200, nullable = false)
    private String name;

    @Column(nullable = false, unique = true)
    private String code;

    @Column(length = 200)
    private String crmLink;

    @Column
    private String dataUploadIpWhitelist;

    /**
     * csv list of the pid fields to validate on incoming HL7 messages
     */
    @Column(length = 1000)
    private String hl7UploadValidatePidList;

    /**
     * when new patients are added via the HL7 API (ADT^A28), add them to this team by default
     */
    @Column(length = 200)
    private String hl7NewPatientTeamCode;

    /**
     * csv list of aliases and authentications that this team can upload for -- e.g., CERNER:team_cerner, ALFRED:org_alfred
     */
    @Column
    private String hl7UploadSourceWhitelist;

    @OneToMany(fetch = FetchType.LAZY, mappedBy = "org", cascade = CascadeType.ALL)
    private Set<OrgExternalId> externalIds;

    @OneToMany(fetch = FetchType.EAGER, mappedBy = "org", cascade = CascadeType.ALL)
    @SortNatural
    private SortedSet<OrgLevelIdType> orgLevelIdTypes;

    @OneToMany(fetch = FetchType.LAZY, mappedBy = "org", cascade = CascadeType.ALL)
    private Set<Team> teams;

    @Column(length = 100, nullable = false)
    private String country;

    @Column(columnDefinition = "boolean default false", nullable = false)
    private boolean hl7AutoCreateRecords;

    @Column(columnDefinition = "boolean default false", nullable = false)
    private boolean hl7CheckNhsNumberStatus;

    @Column(length = 1000)
    private String emailFooterText;

    @Column(nullable = false)
    private long orgAccountId;

    @Column(length = 100, nullable = false)
    private String timeZoneId;

    @Column(length = 100)
    private String logoFileName;

    @Column(columnDefinition = "boolean default false", nullable = false)
    private boolean patientViewSuspended;

    /**
     * Image in the Org email footer.
     */
    @Column(length = 100)
    private String emailFooterImageFilename;

    @Column(name = "hl7_password_hash")
    private String hl7PasswordHash;

    // TODO: Generate in code, not DB
    @Column(name = "public_id", nullable = false, updatable = false, unique = true)
    @Type(type = "pg-uuid")
    @Generated(GenerationTime.INSERT)
    private UUID publicId;

    @Column(name = "org_type") //Unfortunately nullable due to historical data. Populate over time
    private String orgType;

    @Column(columnDefinition = "boolean default true", nullable = false)
    private boolean active = true;

    /**
     * pass these into fetchLazies() method
     */
    public enum Lazy {
        EXTERNAL_IDS(Collections.singleton("externalIds")),
        TEAMS(Collections.singleton("teams"));

        private final Set<String> subgraph;

        Lazy(Set<String> subgraph) {
            this.subgraph = subgraph;
        }

        public Set<String> getSubgraph() {
            return subgraph;
        }
    }

    // utility methods

    public List<String> getExternalIdByType(OrgExternalIdType type) {
        return externalIds.stream().filter(orgExternalId -> orgExternalId.getIdType() == type)
                .map(OrgExternalId::getIdValue).collect(Collectors.toList());
    }

    public String getNameAbbreviated() {
        return StringUtils.abbreviate(name, 12);
    }

    @Override
    public String toString() {
        return "Org: " + code + " (id: " + id + ", acct: " + orgAccountId + ")";
    }

    /**
     * Looks for an {@code OrgLevelIdType} associated with this {@code Org} which has an HL7 Assigning Authority and HL7 Type Code that
     * match the provided values. If one is not found, null is returned.
     *
     * @param aa The Assigning Authority of the HL7 ID
     * @param tc The Type Code of the HL7 ID
     * @return The matching type - if found. null - otherwise.
     */
    public @Nullable OrgLevelIdType getOrgLevelIdType(String aa, String tc) {
        for (OrgLevelIdType olidt : getOrgLevelIdTypes()) {
            // Type Code might be blank in the database if it's a default ID type, or an ID type not intended for use with HL7.
            // Don't consider this a match for an HL7 ID that did not include a Type Code.
            if (StringUtils.isNotBlank(tc) && olidt.getHl7TypeCode().equals(tc)) {
                String olidt_aa = olidt.getHl7AssigningAuthority();
                // Wildcarding is permitted by configuring a null AA for a type
                if ((olidt_aa == null) || olidt_aa.equals(aa)) {
                    return olidt;
                }
            }
        }
        return null;
    }

    public String getCode() {

        this.code = code;

        return hl7UploadValidatePidList;

    public void setHl7UploadValidatePidList(String hl7UploadValidatePidList) {

        return hl7NewPatientTeamCode;

    public String getHl7UploadSourceWhitelist() {

        this.hl7CheckNhsNumberStatus = hl7CheckNhsNumberStatus;

        return active;
    }

    @Override
    public boolean equals(Object otherOrg) {
        if (!(otherOrg instanceof Org)) {
            return false;
        }

        Long otherId = ((Org) otherOrg).getId();
        return (otherId != null) && otherId.equals(id);
    }

    @Override
    public int hashCode() {
        if (id == null) {
            return super.hashCode();
        } else {
            return (int) id.longValue();
        }
    }

    public Stream<String> getOdsCodes() {
        return getExternalIds().stream()
                .filter(code -> code.getIdType() == OrgExternalIdType.ODS)
                .map(OrgExternalId::getIdValue);
    }
