package com.pkb.institute.entity;

import com.google.common.base.MoreObjects;
import com.pkb.entities.enums.SponsorshipStatus;
import com.pkb.entities.enums.UserType;
import org.hibernate.annotations.Cache;
import org.hibernate.annotations.CacheConcurrencyStrategy;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import javax.persistence.UniqueConstraint;
import java.io.Serializable;
import java.time.Instant;

/**
 * Persistent version of institute user. Please do not use above persistence layer.
 *
 */
@Entity(name = "InstituteUserEntity")
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table(name = "instituteuser", uniqueConstraints = { @UniqueConstraint(columnNames = { "institute_id", "person_id" }) })
public class InstituteUserEntity implements Serializable {

    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "institute_id")
    private Team institute;

    @Column(name = "person_id")
    private Long personId;

    @Column(length = 30)
    @Enumerated(EnumType.STRING)
    private SponsorshipStatus sponsorshipStatus;

    @Column
    private Instant statusUpdatedOn;

    //todo: make this lazy
    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    private ContactOptions contactable = ContactOptions.All;

    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    private UserType userType;

    public enum ContactOptions {
        All, Messages_Only, Consultations_Only, None
    }

    @Override
    public String toString() {
        // don't break even if lazy-load hasn't been done -- JUST show IDs
        return MoreObjects.toStringHelper("InstituteUser")
                .omitNullValues()
                .add("personId", getPersonId())
                .add("instituteId", institute.getId())
                .add("sponsorshipStatus", sponsorshipStatus)
                .toString();
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Team getInstitute() {
        return institute;
    }

    public void setInstitute(Team institute) {
        this.institute = institute;
    }

    public SponsorshipStatus getSponsorshipStatus() {
        return sponsorshipStatus;
    }

    public void setSponsorshipStatus(SponsorshipStatus sponsorshipStatus) {
        this.sponsorshipStatus = sponsorshipStatus;
    }

    public Instant getStatusUpdatedOn() {
        return statusUpdatedOn;
    }

    public void setStatusUpdatedOn(Instant statusUpdatedOn) {
        this.statusUpdatedOn = statusUpdatedOn;
    }

    public ContactOptions getContactable() {
        return contactable;
    }

    public void setContactable(ContactOptions contactable) {
        this.contactable = contactable;
    }

    public String getContactableString() {
        if (contactable == ContactOptions.All) {
            return "true";
        }
        return "false";
    }

    public UserType getUserType() {
        return userType;
    }

    public void setUserType(UserType userType) {
        this.userType = userType;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }
}
