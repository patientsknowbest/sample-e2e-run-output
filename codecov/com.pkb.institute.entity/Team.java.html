    /**
     * Looks for a {@code TeamLevelIdType} associated with this {@code Team} which has an HL7 Assigning Authority and HL7 Type Code that
     * match the provided values. If one is not found, null is returned.
     *
     * @param aa The Assigning Authority of the HL7 ID
     * @param tc The Type Code of the HL7 ID
     * @return The matching type - if found. null - otherwise.
     */
    public @Nullable TeamLevelIdType getTeamLevelIdType(String aa, String tc) {
        for (TeamLevelIdType tlidt : getTeamLevelIdTypes()) {
            // Type Code might be blank in the database if it's a default ID type, or an ID type not intended for use with HL7.
            // Don't consider this a match for an HL7 ID that did not include a Type Code.
            if (StringUtils.isNotBlank(tc) && tlidt.getHl7TypeCode().equals(tc)) {
                String tlidt_aa = tlidt.getHl7AssigningAuthority();
                // Wildcarding is permitted by configuring a null AA for a type
                if ((tlidt_aa == null) || tlidt_aa.equals(aa)) {
                    return tlidt;
                }
            }
        }
        return null;
    }

    public PrivacyFlag getDatapointPrivacyFlags() {
        Set<PrivacyFlag> flags = privacyFlags.getPrivacyFlags();
        if (flags.size() != 1) {
            throw new IllegalStateException("Team must have exactly one default flag for data points, as data points must have exactly one!");
        }
        return flags.iterator().next();
    }
