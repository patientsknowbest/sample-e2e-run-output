package com.pkb.institute.entity;

import com.pkb.VersionDetails;
import com.pkb.entities.enums.OrgExternalIdType;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.Id;
import javax.persistence.Table;
import java.util.UUID;

@Entity
@Table(schema = "public", name = "org_surviving_state")
public class OrgSurvivingState {

    @Id
    @Type(type = "pg-uuid")
    private UUID id;

    @Column(nullable = false, name = "org_id")
    private Long orgId;

    @Column(name = "external_id_value", nullable = false)
    private String externalIdValue;

    @Column(name = "external_id_type", nullable = false)
    @Enumerated(EnumType.STRING)
    private OrgExternalIdType externalIdType;

    @Column(name = "is_survivor", nullable = false)
    private boolean isSurvivor;

    @Column(name = "description")
    private String description;

    @Embedded
    private VersionDetails versionDetails;

    public UUID getId() {
        return id;
    }

    public void setId(UUID id) {
        this.id = id;
    }

    public String getExternalIdValue() {
        return externalIdValue;
    }

    public void setExternalIdValue(String odsCode) {
        this.externalIdValue = odsCode;
    }

    public OrgExternalIdType getExternalIdType() {
        return externalIdType;
    }

    public void setExternalIdType(OrgExternalIdType idType) {
        this.externalIdType = idType;
    }

    public boolean isSurvivor() {
        return isSurvivor;
    }

    public void setSurvivor(boolean survivor) {
        isSurvivor = survivor;
    }

    public String getDescription() {
        return description;
    }

    public void setDescription(String description) {
        this.description = description;
    }

    public VersionDetails getVersionDetails() {
        return versionDetails;
    }

    public void setVersionDetails(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }

    public Long getOrgId() {
        return orgId;
    }

    public void setOrgId(Long orgId) {
        this.orgId = orgId;
    }
}
