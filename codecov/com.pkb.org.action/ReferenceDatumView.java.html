package com.pkb.org.action;

import com.google.common.collect.Maps;
import com.pkb.app.PrivacyFlagsToPrivacyFlagConverter;
import com.pkb.datamodel.ReferenceDatum;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.entities.enums.ReferenceDataType;

import java.util.EnumMap;
import java.util.Map;

/**
 * A display DTO class for rendering the privacy label in a different way for CSV output
 */
public class ReferenceDatumView {
    private final ReferenceDataType dataType;
    private final String dataId;
    private final String dataText;
    private final String privacyLabel;

    private static final Map<PrivacyFlag, String> PRIVACY_FLAG_TO_CSV_MAPPINGS;

    static {
        Map<PrivacyFlag, String> temporaryMap = new EnumMap<>(PrivacyFlag.class);
        temporaryMap.put(PrivacyFlag.GENERAL, "GENERAL_HEALTH");
        temporaryMap.put(PrivacyFlag.MENTAL_HEALTH, "MENTAL_HEALTH");
        temporaryMap.put(PrivacyFlag.SEXUAL_HEALTH, "SEXUAL_HEALTH");
        temporaryMap.put(PrivacyFlag.SOCIAL_CARE, "SOCIAL_CARE");
        PRIVACY_FLAG_TO_CSV_MAPPINGS = Maps.immutableEnumMap(temporaryMap);
    }

    public ReferenceDatumView(ReferenceDatum referenceDatum) {
        this.dataType = referenceDatum.getDataType();
        this.dataId = referenceDatum.getDataId();
        this.dataText = referenceDatum.getDataText();

        //The codes here are used in a custom PKB CSV.
        //However, in the interest of consistency, they have been chosen to match those used by the FHIR API.
        //See: http://dev.patientsknowbest.com/home/fhir-api/codesystem/privacy-label
        this.privacyLabel = PrivacyFlagsToPrivacyFlagConverter.findPrivacyFlag(
                Boolean.TRUE.equals(referenceDatum.getGeneralHealthConsentRequired()),
                Boolean.TRUE.equals(referenceDatum.getMentalHealthConsentRequired()),
                Boolean.TRUE.equals(referenceDatum.getSexualHealthConsentRequired()),
                Boolean.TRUE.equals(referenceDatum.getSocialCareConsentRequired()))
                .map(PRIVACY_FLAG_TO_CSV_MAPPINGS::get).orElse("");
    }

    public ReferenceDatumView(ReferenceDataType referenceDataType, String dataId, String dataText, String privacyLabel) {
        this.dataType = referenceDataType;
        this.dataId = dataId;
        this.dataText = dataText;
        this.privacyLabel = privacyLabel;
    }

    public ReferenceDataType getDataType() {
        return dataType;
    }

    public String getDataId() {
        return dataId;
    }

    public String getDataText() {
        return dataText;
    }

    public String getPrivacyLabel() {
        return privacyLabel;
    }
}
