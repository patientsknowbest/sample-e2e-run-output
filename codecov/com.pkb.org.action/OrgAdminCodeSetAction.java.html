package com.pkb.org.action;

import com.google.common.annotations.VisibleForTesting;
import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.datamodel.ReferenceDatum;
import com.pkb.domain.ReferenceDatumService;
import com.pkb.institute.entity.Org;
import com.pkb.util.CSVTableBuilder;
import com.pkb.util.OutputStreamCSVTableBuilder;
import org.apache.struts2.interceptor.ServletResponseAware;

import javax.servlet.http.HttpServletResponse;
import java.io.BufferedWriter;
import java.io.IOException;
import java.io.OutputStreamWriter;
import java.time.ZoneId;
import java.time.format.DateTimeFormatter;
import java.util.Collections;
import java.util.List;
import java.util.Locale;
import java.util.stream.Collectors;

public class OrgAdminCodeSetAction extends BaseAction implements ServletResponseAware {

    private static final long serialVersionUID = 1L;

    private static final DateTimeFormatter EXPORT_FILENAME_DATE_FORMAT = DateTimeFormatter.ofPattern("YYYYMMddHHmmss")
            .withLocale(Locale.UK).withZone(ZoneId.of("UTC"));

    private final String tab = "orgAdminCodeSet";

    private List<ReferenceDatumView> codeList = Collections.emptyList();

    private HttpServletResponse response;
    private ReferenceDatumService referenceDatumService;

    @Override
    public String execute() {
        Org org = getLoggedInUserUtil().getLoggedInUserOrg();
        codeList = convertToReferenceDatumView(referenceDatumService.getAllReferenceDataForOrg(org.getId()));
        return Action.SUCCESS;
    }

    public void export() throws IOException {
        Org org = getLoggedInUserUtil().getLoggedInUserOrg();
        codeList = convertToReferenceDatumView(referenceDatumService.getAllReferenceDataForOrg(org.getId()));

        response.setHeader("Content-Disposition",
                "attachment; filename=\"code_set_" + dateTimeService.nowLocalDateTime().format(EXPORT_FILENAME_DATE_FORMAT) + ".csv\"");
        response.setContentType("text/csv");

        try (OutputStreamWriter osw = new OutputStreamWriter(response.getOutputStream());
                BufferedWriter writer = new BufferedWriter(osw)) {
            OutputStreamCSVTableBuilder tableBuilder = new OutputStreamCSVTableBuilder(writer);
            addCodes(tableBuilder, codeList);
            writer.flush();
        }
    }

    public List<ReferenceDatumView> convertToReferenceDatumView(List<ReferenceDatum> referenceData) {
        return referenceData
                .stream()
                .map(ReferenceDatumView::new).collect(Collectors.toList());
    }

    @VisibleForTesting
    public void addCodes(CSVTableBuilder csvTableBuilder, List<ReferenceDatumView> referenceData) {
        for (ReferenceDatumView rd : referenceData) {
            csvTableBuilder.addCell(getText("codeset.txt.code_type"), rd.getDataType().name());
            csvTableBuilder.addCell(getText("codeset.txt.code"), rd.getDataId());
            csvTableBuilder.addCell(getText("codeset.txt.display"), rd.getDataText());
            csvTableBuilder.addCell(getText("codeset.txt.privacy_label"), rd.getPrivacyLabel());
            csvTableBuilder.endRow();
        }
    }

    public boolean isCodeListNotEmpty() {
        return !codeList.isEmpty();
    }

    public List<ReferenceDatumView> getCodes() {
        return codeList;
    }

    public void setReferenceDatumService(ReferenceDatumService referenceDatumService) {
        this.referenceDatumService = referenceDatumService;
    }

    @Override
    public void setServletResponse(HttpServletResponse response) {
        this.response = response;
    }

    public String getTab() {
        return tab;
    }
}
