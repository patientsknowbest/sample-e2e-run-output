package com.pkb.authz.condition;

import com.pkb.datamodel.usertype.UserType;
import io.vavr.collection.Set;
import org.immutables.serial.Serial;
import org.immutables.value.Value.Check;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;
import org.immutables.vavr.encodings.VavrEncodingEnabled;
import org.jetbrains.annotations.NotNull;

import static java.lang.String.format;

@Immutable(builder = false, copy = false)
@Style(passAnnotations = { NotNull.class }, of = "invalidUserType", allParameters = true)
@VavrEncodingEnabled
@Serial.Version(1)
public interface InvalidUserType<S extends UserType> extends ConditionFailure {
    @NotNull
    Class<? extends UserType> userType();

    @NotNull
    Set<Class<? extends UserType>> allowedUserTypes();

    @Override
    default @NotNull String stringify() {
        return format("User type=[%s] is not allowed.", userType().getSimpleName());
    }

    @Check
    default void check() {
        allowedUserTypes().filter(ut -> userType().isAssignableFrom(ut)).headOption().peek((ut) -> {
            throw new IllegalStateException(stringify());
        });

    }
}
