package com.pkb.api.internal;

import com.google.common.collect.Lists;
import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.internal.appointments.PersonManagementService;
import com.pkb.api.internal.converter.MapOfPKBPersonsToMapOfNonSensitivePersonInformationDTOConverter;
import com.pkb.api.internal.model.requests.PersonSearcRequestDto;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.api.util.v2.RestUtil;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import io.swagger.annotations.ApiParam;
import org.apache.commons.lang3.NotImplementedException;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.HeaderParam;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.PathParam;
import javax.ws.rs.Produces;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import java.io.UnsupportedEncodingException;
import java.net.URLDecoder;

import static com.pkb.entities.enums.api.ApiClientScope.SUPPORT;
import static java.nio.charset.StandardCharsets.UTF_8;

@Controller
@Path(PersonManagementInternalApiController.PERSON_MANAGEMENT_PATH)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = {"Person management"}, value = "Api to manage persons, find, update, delete information.")
public class PersonManagementInternalApiController extends AbstractInternalApiController {

    public static final String PERSON_MANAGEMENT_PATH = "/person";
    public static final String EMAIL_ADDRESSES_DOC = "Email addresses to search";

    @Autowired
    private PersonManagementService personManagementService;

    @Autowired
    private RestUtil util;

    @Autowired
    private MapOfPKBPersonsToMapOfNonSensitivePersonInformationDTOConverter mapOfPKBPersonsToMapOfNonSensitivePersonInformationDTOConverter;

    @GET
    @Path("/email/{email}")
    @ApiOperation(value = "Search person by one's email address.")
    @RequiresScope(scopes = SUPPORT)
    public Response findPersonByEmail(@PathParam("email") String email)
            throws UnsupportedEncodingException {
        String decodedEmail = URLDecoder.decode(email, UTF_8);
        PersonSearcRequestDto request = new PersonSearcRequestDto(Lists.newArrayList(decodedEmail));
        PKBCommonApiParams commonParams = util.getCommonParams();
        var personsByEmails = personManagementService.getPersonsByEmails(request, commonParams);
        return okWithoutMessages(mapOfPKBPersonsToMapOfNonSensitivePersonInformationDTOConverter.convert(personsByEmails));
    }

    @POST
    @Path("/emails")
    @ApiOperation(value = "Search persons by their email addresses.")
    @RequiresScope(scopes = SUPPORT)
    public Response findPersonsByEmail(
            @ApiParam(value = EMAIL_ADDRESSES_DOC) PersonSearcRequestDto request) {
        PKBCommonApiParams commonParams = util.getCommonParams();
        var personsByEmails = personManagementService.getPersonsByEmails(request, commonParams);
        return okWithoutMessages(mapOfPKBPersonsToMapOfNonSensitivePersonInformationDTOConverter.convert(personsByEmails));
    }

    @POST
    @Path("/patientIdentifiers")
    @RequiresScope(scopes = SUPPORT)
    public Response findPersonsByPatientIdentifier(
            @ApiParam(value = AUTH_TOKEN_DOC) @HeaderParam(value = AUTH_TOKEN_PARAM_NAME) String authToken) {
        throw new NotImplementedException("");
    }
}
