package com.pkb.api.internal;

import com.pkb.action.library.ImmutableBuildLibrariesRequest;
import com.pkb.action.library.LibraryService;
import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.util.v2.RestUtil;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.QueryParam;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import java.util.Locale;
import java.util.Optional;

import static com.pkb.entities.enums.api.ApiClientScope.CLINICIAN;
import static com.pkb.entities.enums.api.ApiClientScope.PATIENT;

@Controller
@Path(LibraryApiController.LIBRARY_PATH)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = { "Library" }, value = "API for library")
public class LibraryApiController extends AbstractInternalApiController{

    static final String LIBRARY_PATH = "/library";

    @Autowired
    private LibraryService libraryService;

    @Autowired
    private RestUtil util;

    @GET
    @Path("/build")
    @ApiOperation(value = "Build users personal and team libraries")
    @RequiresScope(scopes = { PATIENT, CLINICIAN })
    public Response buildLibraries(
            @QueryParam("locale") String locale,
            @QueryParam("tkid") String tkid,
            @QueryParam("nhsStyle") boolean nhsStyle) {
        var request = ImmutableBuildLibrariesRequest.builder()
                .eHRRequestContext(util.getCommonParams().getRequestContext())
                .locale(Locale.forLanguageTag(locale))
                .tkid(Optional.ofNullable(tkid).filter(s -> !s.isEmpty()))
                .nhsStyle(nhsStyle)
                .build();
        return ok(libraryService.buildLibraries(request));
    }
}
