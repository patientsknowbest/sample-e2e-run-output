package com.pkb.api.internal;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.util.v2.RestUtil;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.quartz.JobKey;
import org.quartz.Scheduler;
import org.quartz.SchedulerException;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;

import javax.ws.rs.Consumes;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;

import static com.pkb.entities.enums.api.ApiClientScope.SUPPORT;
import static config.scedule.PollForSciStoreNotificationsConfig.SCISTORE_POLL_JOB_DETAIL_KEY;
import static config.scedule.SyncNewSciStorePatientsConfig.SCISTORE_GROUP;
import static config.scedule.SyncNewSciStorePatientsConfig.SCISTORE_JOB_DETAIL_KEY;

@Controller
@Path(SciStoreSyncTriggerController.SCISTORE_PATH)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = { "SciStore" }, value = "Api to manage SciStore integration")
public class SciStoreSyncTriggerController extends AbstractInternalApiController {

    public static final String SCISTORE_PATH = "/sciStore";

    @Autowired
    private Scheduler scheduler;

    @Autowired
    private RestUtil util;

    @POST
    @Path("/startJob")
    @ApiOperation(value = "Start the integration job")
    @RequiresScope(scopes = SUPPORT)
    public Response findPersonsByEmail() throws SchedulerException {
        // Called for it's side effects, i.e. security check
        util.getCommonParams();
        scheduler.triggerJob(JobKey.jobKey(SCISTORE_JOB_DETAIL_KEY, SCISTORE_GROUP));
        return okWithoutPayloadOrMessage();
    }

    @POST
    @Path("/pollForNotifications")
    @ApiOperation(value = "Start the poll for notifications job")
    @RequiresScope(scopes = SUPPORT)
    public Response pollForNotifications() throws SchedulerException {
        // Called for it's side effects, i.e. security check
        util.getCommonParams();
        scheduler.triggerJob(JobKey.jobKey(SCISTORE_POLL_JOB_DETAIL_KEY, SCISTORE_GROUP));
        return okWithoutPayloadOrMessage();
    }
}
