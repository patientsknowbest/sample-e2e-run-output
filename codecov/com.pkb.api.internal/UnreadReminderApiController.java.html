package com.pkb.api.internal;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;

import javax.ws.rs.Consumes;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;

import static com.pkb.entities.enums.api.ApiClientScope.SUPPORT;

@Controller
@Path(UnreadReminderApiController.UNREAD_REMINDER_PATH)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = { "Unread reminder" }, value = "Api to manage unread message/document reminder")

public class UnreadReminderApiController extends AbstractInternalApiController {
    public static final String UNREAD_REMINDER_PATH = "/unreadReminder";

    @Autowired
    private PKBEmailMessageManager pKBEmailMessageManager;

    @Autowired
    private RestUtil util;

    @POST
    @Path("/startMessageJob")
    @ApiOperation(value = "Start the unread message reminder job")
    @RequiresScope(scopes = SUPPORT)
    public Response startUnreadMessageReminderJob() {
        util.getCommonParams();
        pKBEmailMessageManager.sendUnreadMessageReminder();
        return okWithoutPayloadOrMessage();
    }

    @POST
    @Path("/startDocumentJob")
    @ApiOperation(value = "Start the unread document reminder job")
    @RequiresScope(scopes = SUPPORT)
    public Response startUnreadDocumentReminderJob() {
        util.getCommonParams();
        pKBEmailMessageManager.sendUnreadDocumentReminder();
        return okWithoutPayloadOrMessage();
    }
}
