package com.pkb.api.internal;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.service.dataupload.emis.esv2.EmisDiagnosticService;
import com.pkb.service.dataupload.emis.esv2.EmisEsService;
import com.pkb.service.dataupload.emis.esv2.EmisEsStatusService;
import com.pkb.service.dataupload.emis.esv2.EmisEsWhitelistService;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.quartz.SchedulerException;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestBody;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;

import static com.pkb.entities.enums.api.ApiClientScope.PUBLIC;
import static com.pkb.entities.enums.api.ApiClientScope.SUPPORT;

@Controller
@Path(EmisEsApiController.EMIS_ES_PATH)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = {"Emis Es"}, value = "Api to manage Emis Es integration")
public class EmisEsApiController extends AbstractInternalApiController {

    static final String EMIS_ES_PATH = "/emisEs";

    @Autowired
    private EmisDiagnosticService emisDiagnosticService;

    @Autowired
    private EmisEsService emisEsService;

    @Autowired
    private EmisEsWhitelistService emisEsWhitelistService;

    @Autowired
    private RestUtil util;

    @Autowired
    private EmisEsStatusService emisEsStatusService;

    @POST
    @Path("/startJob")
    @ApiOperation(value = "Start the integration job")
    @RequiresScope(scopes = SUPPORT)
    public Response startJob() throws SchedulerException {
        // Called for side effects, i.e. authentication
        util.getCommonParams();
        emisEsService.triggerEmisJob();
        return okWithoutPayloadOrMessage();
    }

    @GET
    @Path("/getJobList")
    @ApiOperation(value = "Retrieve the job list")
    @RequiresScope(scopes = SUPPORT)
    public Response getJobList() throws SchedulerException {
        // Called for side effects, i.e. authentication
        util.getCommonParams();
        return okWithoutMessages(emisEsService.getEmisJobInformation());
    }

    @GET
    @Path("/getDiagnosticInformation")
    @ApiOperation(value = "Retrieve diagnostic information of EMIS Sftp")
    @RequiresScope(scopes = SUPPORT)
    public Response getDiagnosticInformation() throws SchedulerException {
        // Called for side effects, i.e. authentication
        util.getCommonParams();
        return okWithoutMessages(emisDiagnosticService.getInformation());
    }

    @POST
    @Path("/setWhitelist")
    @ApiOperation(value = "Set the whitelist service state")
    @RequiresScope(scopes = SUPPORT)
    public Response setWhitelist(@RequestBody String ymlConfig) throws SchedulerException {
        // Called for side effects, i.e. authentication
        util.getCommonParams();
        emisEsWhitelistService.setYamlConfig(ymlConfig);
        return okWithoutPayloadOrMessage();
    }

    @GET
    @Path("/clearDownloaded")
    @ApiOperation(value = "Clear the local copies of emis es files")
    @RequiresScope(scopes = SUPPORT)
    public Response clearDownloaded() throws SchedulerException {
        // Called for side effects, i.e. authentication
        util.getCommonParams();
        emisEsService.clearEmisDownloadDirectory();
        return okWithoutPayloadOrMessage();
    }

    @GET
    @Path("/getStatus")
    @ApiOperation(value = "Retrieve diagnostic information of EMIS ES processing")
    @RequiresScope(scopes = PUBLIC)
    public Response getEsStatus() throws SchedulerException {
        return okWithoutMessages(emisEsStatusService.getStatus());
    }


}
