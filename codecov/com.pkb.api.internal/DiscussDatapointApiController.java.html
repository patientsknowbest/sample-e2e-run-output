package com.pkb.api.internal;

import com.pkb.action.DiscussDatapointService;
import com.pkb.action.ImmutableDiscussLibraryRequest;
import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.util.v2.RestUtil;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.QueryParam;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import java.util.List;
import java.util.Locale;

import static com.pkb.entities.enums.api.ApiClientScope.CLINICIAN;
import static com.pkb.entities.enums.api.ApiClientScope.PATIENT;

@Controller
@Path(DiscussDatapointApiController.DISCUSS_PATH)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = { "Discuss" }, value = "API for discussing a data point")
public class DiscussDatapointApiController extends AbstractInternalApiController {

    static final String DISCUSS_PATH = "/discuss";

    @Autowired
    private DiscussDatapointService discussService;

    @Autowired
    private RestUtil restUtil;

    @GET
    @Path("/library")
    @ApiOperation(value = "discuss library entries")
    @RequiresScope(scopes = { PATIENT, CLINICIAN })
    public <T> Response discussLibrary(
            @QueryParam("locale") String locale,
            @QueryParam("personalLibraryIds") List<Long> personalLibraryIds,
            @QueryParam("instituteLibraryIds") List<String> instituteLibraryIds) {
        var request = ImmutableDiscussLibraryRequest.builder()
                .eHRRequestContext(restUtil.getCommonParams().getRequestContext())
                .locale(Locale.forLanguageTag(locale))
                .personalLibraryIds(personalLibraryIds)
                .instituteLibraryIds(instituteLibraryIds)
                .build();
        return okWithoutMessages(discussService.discuss(request));
    }
}
