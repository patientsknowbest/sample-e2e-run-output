package com.pkb.api.internal;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.common.config.PhrConfig;
import com.pkb.document.entity.Attachment;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.service.team.OrgManager;
import com.pkb.service.team.TeamManager;
import com.pkb.service.team.TeamUserManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.tolven.TolvenBeanFactory;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import io.swagger.annotations.ApiParam;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;

import javax.ws.rs.Consumes;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.QueryParam;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import java.util.Collections;
import java.util.List;
import java.util.Map;
import java.util.function.Function;
import java.util.stream.Collectors;

import static com.pkb.entities.enums.api.ApiClientScope.PRIVACYOFFICER;

@Controller
@Path(BulkDisableSharingApiController.DISABLE_SHARING_PATH)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = { "DisableSharing" }, value = "API to disable sharing for patients. Currently only for testing.")
public class BulkDisableSharingApiController extends AbstractInternalApiController {

    public static final String DISABLE_SHARING_PATH = "/disableSharing";
    private static final String PATIENT_ID_DOC = "PKB ID for patient(s); one to many supported";

    @Autowired
    private RestUtil restUtil;

    @Autowired
    private OrgManager orgManager;

    @Autowired
    private TeamManager teamManager;

    @Autowired
    private TeamUserManager teamUserManager;

    @Autowired
    private TolvenBeanFactory beanFactory;

    @Autowired
    private PhrConfig config;


    @POST
    @ApiOperation(value = "A privacy officer disables sharing for 1+ patients.")
    @RequiresScope(scopes = PRIVACYOFFICER)
    public Response bulkDisableSharing(
            @ApiParam(value = PATIENT_ID_DOC) @QueryParam("patientId") List<Long> patientIds) {

        if (!config.isBulkDisableSharingEnabled()) {
            throw new IllegalArgumentException("Bulk disable sharing is disabled in this environment");
        }

        PKBCommonApiParams commonParams = restUtil.getCommonParams();
        PKBPerson loggedInPerson = commonParams.getPerson();

        long privacyOfficerId = commonParams.getPerson().getId();

        Team team = teamUserManager.findActivePrimaryTeam(commonParams.getRequestContext(), privacyOfficerId)
                .orElseThrow(() -> new IllegalArgumentException("no active primary team for " + privacyOfficerId));
        Org org = teamManager.getOrg(team.getOrg().getId()).orElse(null);

        Map<Long, List<Attachment>> patientIdNoAttachmentsMap = patientIds.stream()
                .collect(Collectors.toMap(Function.identity(), id -> Collections.emptyList()));
        List<Map<String, Long>> patientAndSharingDisabledIds = orgManager.disableSharingForPatients(commonParams.getRequestContext(), loggedInPerson, org, team, patientIdNoAttachmentsMap);

        return okWithoutMessages(patientAndSharingDisabledIds);
    }
}
