package com.pkb.api.internal;

import com.pkb.action.message.ImmutableGetInboxRequest;
import com.pkb.action.message.InboxService;
import com.pkb.action.message.ShowMessageConversationService;
import com.pkb.action.message.WorkflowFilter;
import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.domain.messages.workflow.TrustedConversationWorkflowService;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.MessageSource;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestBody;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.QueryParam;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import java.time.Instant;
import java.util.Locale;
import java.util.OptionalInt;
import java.util.UUID;
import java.util.function.BiFunction;

import static com.pkb.entities.enums.api.ApiClientScope.CLINICIAN;
import static com.pkb.entities.enums.api.ApiClientScope.PATIENT;
import static com.pkb.entities.enums.api.ApiClientScope.TEAMCOORD;

@Controller
@Path(MessagingApiController.MESSAGING_PATH)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = { "Messaging" }, value = "API for messaging")
public class MessagingApiController extends AbstractInternalApiController {

    static final String MESSAGING_PATH = "/messaging";

    @Autowired
    private InboxService inboxService;

    @Autowired
    private ShowMessageConversationService showMessageConversationService;

    @Autowired
    private TrustedConversationWorkflowService trustedConversationWorkflowService;

    @Autowired
    private RestUtil util;

    @Autowired
    private MessageSource messageSource;

    @GET
    @Path("/inbox")
    @ApiOperation(value = "Messaging inbox")
    @RequiresScope(scopes = { PATIENT, CLINICIAN, TEAMCOORD })
    public Response getInbox(
            @QueryParam("locale") String locale,
            @QueryParam("offset") int offset,
            @QueryParam("filterPatientId") String filterPatientId,
            @QueryParam("filterFromDate") String filterFromDate,
            @QueryParam("filterToDate") String filterToDate,
            @QueryParam("workflowFilter") WorkflowFilter workflowFilter,
            @QueryParam("isAbleToStartConsultations") boolean isAbleToStartConsultations) {
        PKBCommonApiParams commonParams = util.getCommonParams();

        ImmutableGetInboxRequest.Builder getInboxRequestBuilder = ImmutableGetInboxRequest.builder()
                .loggedInEHRRequestContext(commonParams.getRequestContext());

                if (filterPatientId != null) {
                    getInboxRequestBuilder.filterPatientId(filterPatientId);
                }
                if (filterFromDate != null) {
                    getInboxRequestBuilder.filterFrom(parseInstant(filterFromDate));
                }
                if (filterToDate != null) {
                    getInboxRequestBuilder.filterTo(parseInstant(filterToDate));
                }
                if (workflowFilter != null) {
                    getInboxRequestBuilder.workflowFilter(workflowFilter);
                }

                getInboxRequestBuilder.offset(offset)
                        .locale(Locale.forLanguageTag(locale))
                        .ableToStartQuestionnaire(isAbleToStartConsultations);

                ImmutableGetInboxRequest getInboxRequest = getInboxRequestBuilder.build();

        return okWithoutMessages(inboxService.getInbox(getInboxRequest, getText(locale), Locale.forLanguageTag(locale)));
    }

    @GET
    @Path("/showmessage")
    @ApiOperation(value = "Show message screen")
    @RequiresScope(scopes = { PATIENT, CLINICIAN, TEAMCOORD })
    public Response showMessage(
            @QueryParam("locale") String locale,
            @QueryParam("conversationId") UUID conversationId,
            @QueryParam("message") String message) {
        // Called for it's side effects, i.e. security check
        PKBCommonApiParams commonParams = util.getCommonParams();
        return okWithoutMessages(
                showMessageConversationService.buildMessageScreenDto(commonParams.getRequestContext(), conversationId, message, Locale.forLanguageTag(locale), getText(locale)));
    }

    @POST
    @Path("/workflow/assign")
    @ApiOperation(value = "Assing message to me")
    @RequiresScope(scopes = { CLINICIAN })
    public Response assignToMe(
            @RequestBody ArchiveAssignAPIRequestDTO requestDTO) {
        PKBCommonApiParams commonParams = util.getCommonParams();
        var loggedInEHRRequestContext = commonParams.getRequestContext();
        trustedConversationWorkflowService.assignConversationToPersonId(requestDTO.getConversationId(), getTeamId(loggedInEHRRequestContext), getPersonId(loggedInEHRRequestContext),
                getOptimisticLockVersionAsOptionalInt(requestDTO.getOptimisticLockVersion().orElse(null)));
        return okWithoutPayloadOrMessage();
    }

    @NotNull
    private OptionalInt getOptimisticLockVersionAsOptionalInt(@RequestBody Integer optimisticLockVersion) {
        return optimisticLockVersion != null ? OptionalInt.of(optimisticLockVersion) : OptionalInt.empty();
    }

    @POST
    @Path("/workflow/unassign")
    @ApiOperation(value = "UnAssing message")
    @RequiresScope(scopes = { CLINICIAN })
    public Response unAssignMessage(
            @RequestBody ArchiveAssignAPIRequestDTO requestDTO) {
        PKBCommonApiParams commonParams = util.getCommonParams();
        var loggedInEHRRequestContext = commonParams.getRequestContext();
        var teamId = getTeamId(loggedInEHRRequestContext);
        trustedConversationWorkflowService.unassignConversation(requestDTO.getConversationId(), teamId, getPersonId(loggedInEHRRequestContext),
                getOptimisticLockVersionAsOptionalInt(requestDTO.getOptimisticLockVersion().orElse(null)));
        return okWithoutPayloadOrMessage();
    }

    @POST
    @Path("/workflow/archive")
    @ApiOperation(value = "Archive message")
    @RequiresScope(scopes = { CLINICIAN })
    public Response archiveMessage(
            @RequestBody ArchiveAssignAPIRequestDTO requestDTO) {
        PKBCommonApiParams commonParams = util.getCommonParams();
        var loggedInEHRRequestContext = commonParams.getRequestContext();
        var teamId = getTeamId(loggedInEHRRequestContext);
        trustedConversationWorkflowService.archiveConversation(requestDTO.getConversationId(), teamId, getPersonId(loggedInEHRRequestContext),
                getOptimisticLockVersionAsOptionalInt(requestDTO.getOptimisticLockVersion().orElse(null)));
        return okWithoutPayloadOrMessage();
    }

    @POST
    @Path("/workflow/unarchive")
    @ApiOperation(value = "UnArchive message")
    @RequiresScope(scopes = { CLINICIAN })
    public Response unArchiveMessage(
            @RequestBody ArchiveAssignAPIRequestDTO requestDTO) {
        PKBCommonApiParams commonParams = util.getCommonParams();
        var loggedInEHRRequestContext = commonParams.getRequestContext();
        var teamId = getTeamId(loggedInEHRRequestContext);
        trustedConversationWorkflowService.unarchiveConversation(requestDTO.getConversationId(), teamId, getPersonId(loggedInEHRRequestContext),
                getOptimisticLockVersionAsOptionalInt(requestDTO.getOptimisticLockVersion().orElse(null)));
        return okWithoutPayloadOrMessage();
    }

    private Instant parseInstant(String input) {
        if (input == null) {
            return null;
        }
        return Instant.ofEpochMilli(Long.parseLong(input));
    }

    private long getTeamId(LoggedInEHRRequestContext loggedInEHRRequestContext) {
        return loggedInEHRRequestContext.getTeamId()
                .orElseThrow(() -> new RuntimeException("loggedInEHRRequestContext does not have teamId"));
    }

    private long getPersonId(LoggedInEHRRequestContext loggedInEHRRequestContext) {
        return loggedInEHRRequestContext.getMaybeAccessingUserId()
                .orElseThrow(() -> new RuntimeException("loggedInEHRRequestContext does not have accessing user id"));
    }

    private BiFunction<String, String[], String> getText(String locale) {
        return (key, args) -> messageSource.getMessage(key, args, new Locale(locale));
    }
}
