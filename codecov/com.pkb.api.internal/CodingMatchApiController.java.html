package com.pkb.api.internal;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.service.coding.CodingMatchExportService;
import com.pkb.service.coding.CodingMatchImportService;
import com.pkb.service.dataupload.emis.esv2.EmisJobArtifactService;
import org.apache.struts2.dispatcher.multipart.MultiPartRequestWrapper;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import java.io.File;
import java.lang.invoke.MethodHandles;

import static com.pkb.entities.enums.api.ApiClientScope.SUPPORT;
import static org.apache.struts2.ServletActionContext.getRequest;
import static org.apache.struts2.ServletActionContext.getResponse;

@Controller
@Path(CodingMatchApiController.CODING_MATCH_PATH)
@Produces(MediaType.APPLICATION_JSON)
//No swagger docs because of https://issues.redhat.com/browse/WFLY-16322, don't fancy upgrading wf for this...
public class CodingMatchApiController extends AbstractInternalApiController {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    static final String CODING_MATCH_PATH = "/codingmatch";

    @Autowired
    private RestUtil util;

    @Autowired
    private DateTimeService dateTimeService;

    @Autowired
    private CodingMatchImportService codingMatchImportService;

    @Autowired
    private CodingMatchExportService codingMatchExportService;

    @Autowired
    private EmisJobArtifactService emisJobArtifactService;

    @GET
    @Path("/downloadExisting")
    @RequiresScope(scopes = SUPPORT)
    @Consumes(MediaType.APPLICATION_JSON)
    public Response downloadExisting() {
        return codingMatchExportService.streamAllMatches(getResponse())
                .map((unused) -> okWithoutPayloadOrMessage())
                .get();
    }

    @GET
    @Path("/downloadEmisUnmatched")
    @RequiresScope(scopes = SUPPORT)
    @Consumes(MediaType.APPLICATION_JSON)
    public Response downloadEmisUnmatched() {
        return codingMatchExportService.streamUnmatchedCodes(getResponse())
                .map((unused) -> okWithoutPayloadOrMessage())
                .get();
    }

    @POST
    @Path("/uploadNew")
    @RequiresScope(scopes = SUPPORT)
    @Consumes(MediaType.MULTIPART_FORM_DATA)
    public Response uploadNewCodingMatches(File newMatchesParamWithNoContent) {
        PKBCommonApiParams commonParams = util.getCommonParams();

        File newMatches = getActualNewCodingMatchesFile();

        try {
            String response = codingMatchImportService.saveNewMappingsCsv(newMatches, commonParams.getRequestContext())
                    .getMessage();
            return okWithoutMessages(response);
        } catch (Exception e) {
            LOGGER.error("Failed to upload new matches", e);
            return failureWithReason("error whilst saving: " + e);
        }
    }

    /**
     * Multipart file uplaads are somehow messed up. This is an awful hack to get around that.
     * Whilst the lower layers happily know about the file, resteasy is given an empty input stream so the file provided
     * to it has to be ignored....
     * Gave up figuring out why it's broken, this can obviously be thrown into the bin once we move off of struts et al.
     */
    private File getActualNewCodingMatchesFile() {
        return new File(((MultiPartRequestWrapper) getRequest()).getFiles("File")[0].getAbsolutePath());
    }
}
