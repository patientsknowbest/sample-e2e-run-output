package com.pkb.api.internal;

import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.OrgExternalId;
import com.pkb.institute.entity.Team;
import com.pkb.service.team.TeamManager;
import com.pkb.user.entity.TeamLevelIdType;
import io.swagger.annotations.Api;
import org.immutables.value.Value;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;

import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.PathParam;
import javax.ws.rs.Produces;
import javax.ws.rs.QueryParam;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import java.util.List;
import java.util.Set;
import java.util.stream.Collectors;

@Controller
@Path(TeamManagerTestingApiController.INSTITUTE_BEAN_TESTING)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = { "TeamManagerTesting" }, value = "Api to support testing institute bean sql")
public class TeamManagerTestingApiController extends AbstractInternalApiController {
    static final String INSTITUTE_BEAN_TESTING = "/instituteBeanTesting";

    @Autowired
    TeamManager teamManager;

    @GET
    @Path("/getOrg/{orgId}")
    public Response getOrg(@PathParam("orgId") Long orgId, @QueryParam("lazies") List<Org.Lazy> lazies) {
        var laziesArray = lazies.toArray(new Org.Lazy[0]);
        var org = teamManager.getOrg(orgId, laziesArray).orElse(null);
        return okWithoutMessages(mapOrg(org, lazies));
    }

    @GET
    @Path("/getOrgs/")
    public Response getOrgs(@QueryParam("orgId") List<Long> ids, @QueryParam("lazies") List<Org.Lazy> lazies) {
        var laziesArray = lazies.toArray(new Org.Lazy[0]);
        var payload = teamManager.getOrgs(ids, laziesArray).stream().map(org -> mapOrg(org, lazies)).collect(Collectors.toList());
        return okWithoutMessages(payload);
    }

    private OrgDto mapOrg(Org org, List<Org.Lazy> lazies) {
        var builder = ImmutableOrgDto.builder()
                .id(org.getId())
                .name(org.getName());
        if (lazies.contains(Org.Lazy.TEAMS)) {
            builder.teams(org.getTeams().stream().map(this::mapTeam).collect(Collectors.toSet()));
        }

        if (lazies.contains(Org.Lazy.EXTERNAL_IDS)) {
            builder.orgExternalids(org.getExternalIds().stream().map(this::mapExternalId).collect(Collectors.toSet()));
        }
        return builder.build();
    }

    private TeamDto mapTeam(Team team) {
        return ImmutableTeamDto.builder()
                .id(team.getId())
                .name(team.getName())
                .teamLevelIdTypes(team.getTeamLevelIdTypes().stream().map(this::mapTeamLevelIdType).collect(Collectors.toSet()))
                .build();
    }

    private TeamLevelIdTypeDto mapTeamLevelIdType(TeamLevelIdType type) {
        return ImmutableTeamLevelIdTypeDto.builder()
                .id(type.getId())
                .name(type.getName())
                .build();
    }

    private OrgExternalIdDto mapExternalId(OrgExternalId orgExternalId) {
        return ImmutableOrgExternalIdDto.builder()
                .id(orgExternalId.getId())
                .idType(orgExternalId.getIdType().name())
                .idValue(orgExternalId.getIdValue())
                .orgId(orgExternalId.getOrg().getId())
                .build();
    }
}

@Value.Immutable
@JsonSerialize(as = ImmutableOrgDto.class)
interface OrgDto {
    long id();

    String name();

    Set<TeamDto> teams();

    Set<OrgExternalIdDto> orgExternalids();
}

@Value.Immutable
@JsonSerialize(as = ImmutableTeamDto.class)
interface TeamDto {
    long id();

    String name();

    Set<TeamLevelIdTypeDto> teamLevelIdTypes();
}

@Value.Immutable
@JsonSerialize(as = ImmutableTeamLevelIdTypeDto.class)
interface TeamLevelIdTypeDto {
    long id();

    String name();
}

@Value.Immutable
@JsonSerialize(as = ImmutableOrgExternalIdDto.class)
interface OrgExternalIdDto {
    long id();

    String idType();

    String idValue();

    long orgId();
}
