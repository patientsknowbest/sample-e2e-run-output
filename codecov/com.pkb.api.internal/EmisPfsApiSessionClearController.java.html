package com.pkb.api.internal;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.emis.appointment.service.EmisSessionService;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestBody;

import javax.ws.rs.Consumes;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import java.util.Map;

import static com.pkb.api.internal.EmisPfsApiSessionClearController.EMIS_PFS_API_SESSION_CLEAR;
import static com.pkb.entities.enums.api.ApiClientScope.SUPPORT;

@Controller
@Path(EMIS_PFS_API_SESSION_CLEAR)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = {"EMIS PFS API"}, value = "Api to reset client session with EMIS PFS API")
public class EmisPfsApiSessionClearController extends AbstractInternalApiController {
    public static final String EMIS_PFS_API_SESSION_CLEAR = "/emisPfsApiSessionClear";

    @Autowired
    private EmisSessionService emisSessionService;

    @POST
    @ApiOperation(value = "Clears session with EMIS PFS API")
    @RequiresScope(scopes = SUPPORT)
    public Response clearEmisPfsApiSession(@RequestBody Map<String, Object> params) {
        emisSessionService.clearState();
        return okWithoutMessages("foo");
    }
}
