package com.pkb.api.internal;

import com.pkb.api.internal.model.response.InternalRestMeta;
import com.pkb.api.internal.model.response.InternalRestResponse;
import com.pkb.datamodel.ui.ResponseMeta;
import com.pkb.datamodel.ui.UIDto;

import javax.ws.rs.core.Response;

import static javax.ws.rs.core.Response.Status.BAD_REQUEST;

public abstract class AbstractInternalApiController {

    public static final String AUTH_TOKEN_PARAM_NAME = "Authorization";
    public static final String AUTH_TOKEN_DOC = "Your bearer token, in the form: Bearer <myBearerToken>";

    protected <T extends UIDto> Response ok(T payload) {
        InternalRestResponse<T> entity = new InternalRestResponse<>(payload);
        InternalRestMeta meta = entity.getMeta();
        ResponseMeta responseMeta = payload.getResponseMeta();
        if (responseMeta != null) {
            responseMeta.getActionErrors().forEach(meta::addError);
            responseMeta.getFieldErrors().forEach((field, errors) -> {
                errors.forEach(error -> meta.addFieldError(field, error));
            });
        }
        return Response.ok(entity).build();
    }

    protected <T> Response okWithoutMessages(T payload) {
        InternalRestResponse<T> entity = new InternalRestResponse<>(payload);
        return Response.ok(entity).build();
    }

    protected Response okWithoutPayloadOrMessage() {
        InternalRestResponse<Void> entity = new InternalRestResponse<>();
        return Response.ok(entity).build();
    }

    protected Response failureWithReason(String reason) {
        InternalRestResponse<String> entity = new InternalRestResponse<>(reason);
        return Response.status(BAD_REQUEST).entity(entity).build();
    }
}
