package com.pkb.api.internal;

import com.pkb.api.aspect.annotation.RequiresScope;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.api.util.v2.RestUtil;
import com.pkb.entities.enums.api.ApiClientScope;
import com.pkb.service.institute.OrgSurvivingStateService;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestBody;

import javax.ws.rs.Consumes;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;

@Controller
@Path(OrgSurvivingStateApiController.ORG_MERGE_PATH)
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Api(tags = { "OrgSurvivingState" }, value = "API for merge orgs")
public class OrgSurvivingStateApiController extends AbstractInternalApiController {

    static final String ORG_MERGE_PATH = "/survivingState";

    @Autowired
    private OrgSurvivingStateService orgSurvivingStateService;

    @Autowired
    private RestUtil util;

    @POST
    @Path("/add")
    @ApiOperation(value = "Add new Org Surviving State")
    @RequiresScope(scopes = {ApiClientScope.SUPPORT})
    public Response addOrgSurvivingState(@RequestBody OrgSurvivingStateDTO dto) {
        PKBCommonApiParams commonApiParams = util.getCommonParams();
        var context = commonApiParams.getRequestContext();

        orgSurvivingStateService.save(dto.getOrgId(), dto.getExternalIdType(), dto.getExternalIdValue(), dto.getIsSurvivor(),
                dto.getDescription(), context);
        return okWithoutPayloadOrMessage();
    }
}
