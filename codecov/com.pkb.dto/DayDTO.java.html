package com.pkb.dto;

import com.pkb.util.Constants;

import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeFormatterBuilder;
import java.time.temporal.ChronoField;
import java.util.Date;
import java.util.Locale;

/**
 * A DTO to carry information about the availability and other event related
 * information concerning a day in the user's schedule
 *
 * @author pravinam
 *
 */
public class DayDTO {

    public static final DateTimeFormatter DATE_FORMATTER = getDateFormatter();

    private static DateTimeFormatter getDateFormatter() {
        DateTimeFormatterBuilder builder = new DateTimeFormatterBuilder();
        return builder.parseCaseInsensitive()
                .appendPattern("dd/MM/yyyy")
                .parseLenient()
                .parseDefaulting(ChronoField.HOUR_OF_DAY, 0)
                .parseDefaulting(ChronoField.MINUTE_OF_HOUR, 0)
                .parseDefaulting(ChronoField.SECOND_OF_MINUTE, 0)
                .parseDefaulting(ChronoField.NANO_OF_SECOND, 0)
                .toFormatter(Locale.UK)
                .withZone(Constants.APPLICATION_TZ);
    }

    private Date date;

    private String dateStr;

    // User is available or not on the day
    private boolean available;

    private int dayOfWeek;

    // 1 - 31 are valid values
    // 0 means void day for display purpose
    private int dayOfMonth;

    public int getDayOfWeek() {
        return dayOfWeek;
    }

    public void setDayOfWeek(int dayOfWeek) {
        this.dayOfWeek = dayOfWeek;
    }

    public boolean isAvailable() {
        return available;
    }

    public void setAvailable(boolean available) {
        this.available = available;
    }

    public int getDayOfMonth() {
        return dayOfMonth;
    }

    public void setDayOfMonth(int dayOfMonth) {
        this.dayOfMonth = dayOfMonth;
    }

    public Date getDate() {
        return date;
    }

    public void setDate(Date date) {
        this.date = date;
    }

    public String getDateStr() {
        if (date != null) {
            return DATE_FORMATTER.format(date.toInstant());
        } else {
            return "";
        }
    }

    @Override
    public String toString() {
        StringBuilder buffer = new StringBuilder();
        buffer.append("[");
        buffer.append("dayStr=" + date);
        buffer.append("dayOfMonth=" + dayOfMonth);
        buffer.append("available=" + available);
        buffer.append("dayOfWeek=" + dayOfWeek);
        buffer.append("]");
        return buffer.toString();
    }
}
