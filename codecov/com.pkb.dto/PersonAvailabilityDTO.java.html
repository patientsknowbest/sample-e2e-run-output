package com.pkb.dto;

import com.pkb.calendar.entity.AvailabilityDTO;

/**
 * A wrapper DTO to combine the details of person & availability of a user
 *
 * @author pravinam
 *
 */
public class PersonAvailabilityDTO {

    PersonDto person;

    AvailabilityDTO availabilityDTO;

    boolean tomorrowAvailable;

    public PersonDto getPerson() {
        return person;
    }

    public void setPerson(PersonDto person) {
        this.person = person;
    }

    public AvailabilityDTO getAvailabilityDTO() {
        return availabilityDTO;
    }

    public void setAvailabilityDTO(AvailabilityDTO availabilityDTO) {
        this.availabilityDTO = availabilityDTO;
    }

    public boolean isTomorrowAvailable() {
        return tomorrowAvailable;
    }

    public void setTomorrowAvailable(boolean tomorrowAvailable) {
        this.tomorrowAvailable = tomorrowAvailable;
    }

}
