package com.pkb.dto;

import com.pkb.test.entity.MeasurementDisplayCategory;
import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.util.List;

@Value.Immutable
@Value.Style(get = { "is*", "get*" },
        passAnnotations = { NotNull.class },
        of = "measurementOverviewDto",
        stagedBuilder = true)
public interface MeasurementOverviewDto {

    String getTitle();

    List<MeasurementValueDto> getContent();

    MeasurementGraphsDto getGraphs();

    MeasurementDisplayCategory getDisplayCategory();

    List<String> getWarnings();

    UrlDto getDetailUrl();

    default boolean isCategory(MeasurementDisplayCategory category) {
        return getDisplayCategory() == category;
    }

    default boolean isVisible(@Nullable MeasurementDisplayCategory displayCategory) {
        // There is no displayCategory when we show measurements on the view care plan page
        return displayCategory == null || getDisplayCategory() == displayCategory;
    }

    default boolean isTimeInHours() {
        return getContent().get(0).getUnit().equals("hr");
    }
}
