package com.pkb.dto;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.pkb.entities.enums.HasNationalId;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.user.entity.NationalId;
import org.apache.commons.lang3.StringUtils;
import org.immutables.value.Value;
import org.immutables.value.Value.Immutable;

import java.util.Optional;

/**
 * Representation of {@link NationalId} entity in the service layer.
 */
@Immutable
@Value.Style(get = { "is*", "get*" })
@JsonSerialize(as = ImmutableNationalIdDTO.class)
@JsonDeserialize(as = ImmutableNationalIdDTO.class)
public abstract class NationalIdDTO implements HasNationalId {

    public abstract Optional<String> getValue();

    public abstract Optional<NationalIdType> getType();

    @Override
    public NationalIdType nationalIdType() {
        return getType().orElse(null);
    }

    @Override
    public String nationalIdValue() {
        return getValue().orElse(null);
    }

    public String getValueFormatted() {
        if (getType().isEmpty() || StringUtils.isBlank(nationalIdValue())) {
            return nationalIdValue();
        }

        return nationalIdType().getCanonicalFormat(nationalIdValue());
    }
}
