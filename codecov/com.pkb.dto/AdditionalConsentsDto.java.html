package com.pkb.dto;

import com.google.common.collect.Maps;
import com.google.common.collect.Sets;
import com.pkb.entities.enums.AdditionalConsentId;
import com.pkb.entities.enums.ResearchScope;
import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;

import java.util.Map;
import java.util.Optional;
import java.util.Set;
import java.util.stream.Collectors;

import static io.vavr.API.None;

public class AdditionalConsentsDto {
    private Map<AdditionalConsentId, Boolean> selected = Maps.newEnumMap(AdditionalConsentId.class);
    private String teamName;
    private Long teamId;
    private String orgName;
    private Map<ResearchScope, TeamResearchPartnerDto> partners = Maps.newEnumMap(ResearchScope.class);
    private boolean editPartners = false;

    public void setTeamId(Long teamId) {
        this.teamId = teamId;
    }

    public boolean isEditPartners() { return editPartners; }
    public void setEditPartners(boolean editPartners) { this.editPartners = editPartners; }

    public void setPartners(Map<ResearchScope, TeamResearchPartnerDto> partners) { this.partners = partners; }
    public Map<ResearchScope, TeamResearchPartnerDto> getPartners() {
        return this.partners;
    }

    public String getContactForResearchPartnerName() {
        return AdditionalConsentId.CONTACT_FOR_RESEARCH
                .getEnabledResearchScope()
                .flatMap(scope -> Optional.ofNullable(partners.get(scope)))
                .flatMap(partner -> partner.name().toJavaOptional())
                .orElse(null);
    }

    public boolean isContactForResearchPartnerNameDisclosed() {
        return getContactForResearchPartnerName() != null;
    }

    private void setPartnerName(String partnerName, ResearchScope scope) {
        var trimmedName = Option.of(StringUtils.trimToNull(partnerName));
        partners.compute(scope, (scop, dto) -> {
            if(dto == null) {
                return ImmutableTeamResearchPartnerDto.builder()
                        .name(trimmedName)
                        .infoUrl(None())
                        .build();
            } else {
                return ImmutableTeamResearchPartnerDto.builder()
                        .from(dto)
                        .name(trimmedName)
                        .build();
            }
        });
    }

    private void setInfoUrl(String infoUrl, ResearchScope scope) {
        var trimmedInfoUrl = Option.of(StringUtils.trimToNull(infoUrl));
        partners.compute(scope, (scop, dto) -> {
            if(dto == null) {
                return ImmutableTeamResearchPartnerDto.builder()
                        .infoUrl(trimmedInfoUrl)
                        .name(None())
                        .build();
            } else {
                return ImmutableTeamResearchPartnerDto.builder()
                        .from(dto)
                        .infoUrl(trimmedInfoUrl)
                        .build();
            }
        });
    }

    public void setContactForResearchPartnerName(String partnerName) {
        setPartnerName(partnerName, ResearchScope.RECRUITING);
    }

    public boolean isContactForResearchPartnerInfoUrlDisclosed() {
        return getContactForResearchPartnerInfoUrl() != null;
    }

    public String getContactForResearchPartnerInfoUrl() {
        return AdditionalConsentId.CONTACT_FOR_RESEARCH
                .getEnabledResearchScope()
                .flatMap(scope -> Optional.ofNullable(partners.get(scope)))
                .flatMap(partner -> partner.infoUrl().toJavaOptional())
                .orElse(null);
    }

    public void setContactForResearchPartnerInfoUrl(String infoUrl) {
        setInfoUrl(infoUrl, ResearchScope.RECRUITING);
    }

    public String getSampleStoragePartnerName() {
        return AdditionalConsentId.SAMPLE_STORAGE
                .getEnabledResearchScope()
                .flatMap(scope -> Optional.ofNullable(partners.get(scope)))
                .flatMap(partner -> partner.name().toJavaOptional())
                .orElse(null);
    }

    public boolean isSampleStoragePartnerNameDisclosed() {
        return getSampleStoragePartnerName() != null;
    }

    public void setSampleStoragePartnerName(String partnerName) {
        setPartnerName(partnerName, ResearchScope.SAMPLE_STORAGE);
    }

    public String getSampleStoragePartnerInfoUrl() {
        return AdditionalConsentId.SAMPLE_STORAGE
                .getEnabledResearchScope()
                .flatMap(scope -> Optional.ofNullable(partners.get(scope)))
                .flatMap(partner -> partner.infoUrl().toJavaOptional())
                .orElse(null);
    }

    public void setSampleStoragePartnerInfoUrl(String infoUrl) {
        setInfoUrl(infoUrl, ResearchScope.SAMPLE_STORAGE);
    }

    public boolean isSampleStoragePartnerInfoUrlDisclosed() {
        return getSampleStoragePartnerInfoUrl() != null;
    }

    public boolean getContactRelatedConsentsEnabled() {
        return selected.keySet().stream().anyMatch(AdditionalConsentId::isContactRelated);
    }

    public boolean getResearchRelatedConsentsEnabled() {
        return selected.keySet().stream().anyMatch(AdditionalConsentId::isResearchRelated);
    }

    public boolean isContactByEmailEnabled() {
        return selected.containsKey(AdditionalConsentId.CONTACT_BY_EMAIL);
    }

    public boolean isContactBySmsEnabled() {
        return selected.containsKey(AdditionalConsentId.CONTACT_BY_SMS);
    }

    public boolean isContactByPostEnabled() {
        return selected.containsKey(AdditionalConsentId.CONTACT_BY_POST);
    }

    public boolean isContactForResearchEnabled() {
        return selected.containsKey(AdditionalConsentId.CONTACT_FOR_RESEARCH);
    }

    public boolean isSampleStorageEnabled() {
        return selected.containsKey(AdditionalConsentId.SAMPLE_STORAGE);
    }

    public boolean getContactByEmailSelected() {
        return selected.getOrDefault(AdditionalConsentId.CONTACT_BY_EMAIL, false);
    }

    public boolean getContactBySmsSelected() {
        return selected.getOrDefault(AdditionalConsentId.CONTACT_BY_SMS, false);
    }

    public boolean getContactByPostSelected() {
        return selected.getOrDefault(AdditionalConsentId.CONTACT_BY_POST, false);
    }

    public boolean getContactForResearchSelected() {
        return selected.getOrDefault(AdditionalConsentId.CONTACT_FOR_RESEARCH, false);
    }

    public boolean getSampleStorageSelected() {
        return selected.getOrDefault(AdditionalConsentId.SAMPLE_STORAGE, false);
    }

    private void select(AdditionalConsentId id, boolean selected) {
        this.selected.put(id, selected);
    }

    public void setContactByEmailSelected(boolean selected) {
        select(AdditionalConsentId.CONTACT_BY_EMAIL, selected);
    }

    public void setContactBySmsSelected(boolean selected) {
        select(AdditionalConsentId.CONTACT_BY_SMS, selected);
    }

    public void setContactByPostSelected(boolean selected) {
        select(AdditionalConsentId.CONTACT_BY_POST, selected);
    }

    public void setContactForResearchSelected(boolean selected) {
        select(AdditionalConsentId.CONTACT_FOR_RESEARCH, selected);
    }

    public void setSampleStorageSelected(boolean selected) {
        select(AdditionalConsentId.SAMPLE_STORAGE, selected);
    }

    public Map<AdditionalConsentId, Boolean> getSelected() {
        return selected;
    }

    public Set<AdditionalConsentId> getSelectedIds() {
        return Sets.immutableEnumSet(getSelected().entrySet().stream().filter(Map.Entry::getValue).map(Map.Entry::getKey).collect(Collectors.toUnmodifiableSet()));
    }

    public void setSelected(Map<AdditionalConsentId, Boolean> selected) {
        this.selected = selected;
    }

    public String getTeamName() {
        return teamName;
    }

    public void setTeamName(String teamName) {
        this.teamName = teamName;
    }

    public String getOrgName() {
        return orgName;
    }

    public void setOrgName(String orgName) {
        this.orgName = orgName;
    }

}
