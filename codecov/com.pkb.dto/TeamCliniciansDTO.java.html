package com.pkb.dto;

import java.util.LinkedList;
import java.util.List;

/**
 * clinicians list for a team
 *
 * @author pravina
 *
 */
public class TeamCliniciansDTO implements Comparable<TeamDTO> {

    //if null then the clinicians are non-team
    private TeamDTO team;

    private List<PersonAvailabilityDTO> cliniciansList;

    private String tbmId;

    private String tbmName;

    public TeamDTO getTeam() {
        return team;
    }

    public void setTeam(TeamDTO team) {
        this.team = team;
    }

    public List<PersonAvailabilityDTO> getCliniciansList() {
        if (cliniciansList == null) {
            cliniciansList = new LinkedList<>();
        }
        return cliniciansList;
    }

    public void setCliniciansList(List<PersonAvailabilityDTO> cliniciansList) {
        this.cliniciansList = cliniciansList;
    }

    public String getTbmId() {
        return tbmId;
    }

    public void setTbmId(String tbmId) {
        this.tbmId = tbmId;
    }

    public String getTbmName() {
        return tbmName;
    }

    public void setTbmName(String tbmName) {
        this.tbmName = tbmName;
    }

    @Override
    public int compareTo(TeamDTO team) {
        return this.team.getName().compareTo(team.getName());
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (!(o instanceof TeamCliniciansDTO)) {
            return false;
        }

        TeamCliniciansDTO that = (TeamCliniciansDTO) o;

        return team != null ? team.equals(that.team) : that.team == null;

    }

    @Override
    public int hashCode() {
        return team != null ? team.hashCode() : 0;
    }
}
