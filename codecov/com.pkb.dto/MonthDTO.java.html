package com.pkb.dto;

import java.util.List;

/**
 * A DTO to carry information about the availability and other event related
 * information for the whole month in the schedule of an user
 *
 * @author pravinam
 *
 */
public class MonthDTO {

    private List<DayDTO> days;

    private List<String> daysAvailable;

    private int daysInMonth;

    private int firstDayOfMonth;

    private int year;

    // full or abbreviated month name
    private String name;

    public List<DayDTO> getDays() {
        return days;
    }

    public void setDays(List<DayDTO> days) {
        this.days = days;
    }

    public int getDaysInMonth() {
        return daysInMonth;
    }

    public void setDaysInMonth(int daysInMonth) {
        this.daysInMonth = daysInMonth;
    }

    public int getFirstDayOfMonth() {
        return firstDayOfMonth;
    }

    public void setFirstDayOfMonth(int firstDayOfMonth) {
        this.firstDayOfMonth = firstDayOfMonth;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getYear() {
        return year;
    }

    public void setYear(int year) {
        this.year = year;
    }

    public List<String> getDaysAvailable() {
        return daysAvailable;
    }

    public void setDaysAvailable(List<String> daysAvailable) {
        this.daysAvailable = daysAvailable;
    }

    @Override
    public String toString() {
        StringBuilder buffer = new StringBuilder();
        buffer.append("[");
        buffer.append("name=" + name);
        buffer.append("year=" + year);
        buffer.append("daysInMonth=" + daysInMonth);
        buffer.append("firstDayOfMonth=" + firstDayOfMonth);
        buffer.append("]");
        return buffer.toString();
    }

}
