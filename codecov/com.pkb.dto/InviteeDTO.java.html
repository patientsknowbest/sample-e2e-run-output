package com.pkb.dto;

import com.pkb.datamodel.Email;
import com.pkb.user.entity.NationalId;
import com.pkb.user.entity.OrgLevelId;
import com.pkb.user.entity.PersonContact;
import com.pkb.user.entity.TeamLevelId;

import java.util.List;
import java.util.SortedSet;

/**
 * DTO class for Invitee (Patient, Clinician)
 *
 * @author pravinam
 */
public class InviteeDTO {

    private Long id;
    private String title;
    private String firstName;
    private String lastName;
    private Email emailId;
    private int gender;
    private String dob;
    private List<NationalId> nationalIds;
    private SortedSet<OrgLevelId> orgLevelIds;
    private SortedSet<TeamLevelId> teamLevelIds;
    private List<PersonContact> contacts;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getTitle() {
        return title;
    }

    public void setTitle(String title) {
        this.title = title;
    }

    public String getFirstName() {
        return firstName;
    }

    public void setFirstName(String firstName) {
        this.firstName = firstName.trim();
    }

    public String getLastName() {
        return lastName;
    }

    public void setLastName(String lastName) {
        this.lastName = lastName.trim();
    }

    public Email getEmailId() {
        return emailId;
    }

    public void setEmailId(Email emailId) {
        this.emailId = emailId;
    }

    public int getGender() {
        return gender;
    }

    public void setGender(int gender) {
        this.gender = gender;
    }

    public String getDob() {
        return dob;
    }

    public void setDob(String dob) {
        this.dob = dob;
    }

    public List<NationalId> getNationalIds() {
        return nationalIds;
    }

    public void setNationalIds(List<NationalId> nationalIds) {
        this.nationalIds = nationalIds;
    }

    public SortedSet<OrgLevelId> getOrgLevelIds() {
        return orgLevelIds;
    }

    public void setOrgLevelIds(SortedSet<OrgLevelId> orgLevelIds) {
        this.orgLevelIds = orgLevelIds;
    }

    public SortedSet<TeamLevelId> getTeamLevelIds() {
        return teamLevelIds;
    }

    public void setTeamLevelIds(SortedSet<TeamLevelId> teamLevelIds) {
        this.teamLevelIds = teamLevelIds;
    }

    public List<PersonContact> getContacts() {
        return contacts;
    }

    public void setContacts(List<PersonContact> contacts) {
        this.contacts = contacts;
    }
}
