package com.pkb.dto;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.pkb.consent.entity.PrivacyFlags;
import com.pkb.institute.entity.Team;
import org.immutables.value.Value;
import org.immutables.value.Value.Immutable;

import java.util.Optional;
import java.util.UUID;

/**
 * Representation of {@link Team} entity in the service layer.
 */
@Immutable
@Value.Style(get = { "is*", "get*" })
@JsonSerialize(as = ImmutableTeamDTO.class)
@JsonDeserialize(as = ImmutableTeamDTO.class)
public abstract class TeamDTO {

    public abstract Long getId();

    public abstract UUID getPublicId();

    public abstract String getName();

    public abstract boolean isTeamBasedMessagingEnabled();

    public abstract OrgDTO getOrg();

    public abstract String getCode();

    public abstract String getCountry();

    public abstract Optional<Long> getSpecialtyId();

    public abstract Optional<String> getTeamCRMLink();

    public abstract Optional<String> getAdminCRMLink();

    public abstract String getLogoFileName();

    public abstract String getHeaderFileName();

    public abstract String getFooterFileName();

    public abstract boolean isMandatoryNationalId();

    public abstract Optional<Team.PatientIdSupport> getPatientIdSupport();

    public abstract Optional<Boolean> isSendSymptomReminders();

    public abstract Optional<Long> getSymptomAlarmThreshold();

    public abstract Optional<Long> getSymptomAlarmAmberThreshold();

    public abstract Optional<Long> getSymptomAlarmDays();

    public abstract Optional<Long> getTeamAccountId();

    public abstract Optional<String> getTimeZoneId();

    public abstract boolean isHl7AutoCreateRecords();

    public abstract Optional<Boolean> isCancerTeam();

    public abstract PrivacyFlags getPrivacyFlags();

    public abstract boolean isDefaultRequestGeneral();

    public abstract boolean isDefaultRequestSexualHealth();

    public abstract boolean isDefaultRequestMentalHealth();

    public abstract boolean isDefaultRequestSocialCare();

    public abstract String getLanguageCode();

    public abstract boolean isDefaultNewProsContactable();

}
