package com.pkb.dto;

import com.google.common.collect.Lists;
import com.pkb.allergy.entity.Allergy;
import com.pkb.app.entity.SourceDetails;
import com.pkb.coding.entity.CodeableConcept;
import com.pkb.service.datapoint.DataWithUIPermissions;
import org.jetbrains.annotations.Nullable;

import java.util.Collections;
import java.util.Comparator;
import java.util.Date;
import java.util.List;
import java.util.Objects;
import java.util.stream.Collectors;

/**
 * We group allergies by coding of disorder/allergen, source and route; this holds a group and has util methods to help display it
 */
public class AllergyGroupDto {


    private List<DataWithUIPermissions<Allergy>> allergies;

    /** also sorts */
    public AllergyGroupDto(List<DataWithUIPermissions<Allergy>> allergies) {
        if (allergies == null) {
            this.allergies = Collections.emptyList();
        } else {
            this.allergies = allergies;
        }

        sortAllergies();
    }







    public Long getFirstId() {
        return getAllergies().stream()
                .findFirst()
                .map(a -> a.getData().getId()).orElse(null);
    }

    public boolean hasDisorderData() {
        return getAllergies().stream().findFirst() // Optional
                .map(a -> a.getData().hasDisorderData())
                .orElse(false);
    }

    public @Nullable CodeableConcept getDisorder() {
        return getAllergies().stream().findFirst() // Optional
                .map(a -> a.getData().getDisorder())
                .orElse(null);
    }

    public @Nullable CodeableConcept getAllergen() {
        return getAllergies().stream().findFirst() // Optional
                .map(a -> a.getData().getAllergen())
                .orElse(null);
    }

    /**
     * @return the latest noted date OR onset date in the list (or null)
     */
    public Date getLastActivityDate() {
        List<DataWithUIPermissions<Allergy>> allergiesDescending = Lists.reverse(getAllergies());
        return allergiesDescending.stream()
                .map(a -> a.getData().getDateAsserted() != null ? a.getData().getDateAsserted() : a.getData().getOnsetOfSymptoms())
                .filter(Objects::nonNull)
                .findFirst()
                .orElse(null);
    }

    /**
     * @return the first source -- the org/team/person IDs and route will be identical across Allergies in this group, but
     *         the source _text_ may vary
     */
    public SourceDetails getSource() {
        List<DataWithUIPermissions<Allergy>> allergiesDescending = Lists.reverse(getAllergies());
        return allergiesDescending.stream()
                .findFirst()
                .map(a -> a.getData().getSource()).orElse(null);
    }

    public String getClinicalCodes() {
        List<DataWithUIPermissions<Allergy>> allergiesDescending = Lists.reverse(getAllergies());
        return allergiesDescending.stream()
                .findFirst()
                .map(a -> a.getData().getClinicalCodes()).orElse(null);
    }

    public List<DataWithUIPermissions<Allergy>> getAllergies() {
        return allergies;
    }

    /** sort by onset date / fallback to asserted date */
    private void sortAllergies() {
        Comparator<DataWithUIPermissions<Allergy>> comparing = Comparator.comparing(
                a -> a.getData().getOnsetOfSymptoms() != null ? a.getData().getOnsetOfSymptoms() : a.getData().getDateAsserted(),
                Comparator.nullsFirst(Date::compareTo));
        allergies = getAllergies().stream()
                .sorted(comparing)
                .collect(Collectors.toList());
    }

    public boolean isSexualHealthConsentRequiredInGroup() {
        return allergies.stream().anyMatch(d -> d.getData().getBaseFields().isSexualHealthConsentRequired());
    }

    public boolean isGeneralHealthConsentRequiredInGroup() {
        return allergies.stream().anyMatch(d -> d.getData().getBaseFields().isGeneralHealthConsentRequired());
    }

    public boolean isMentalHealthConsentRequiredInGroup() {
        return allergies.stream().anyMatch(d -> d.getData().getBaseFields().isMentalHealthConsentRequired());
    }

    public boolean isSocialCareConsentRequiredInGroup() {
        return allergies.stream().anyMatch(d -> d.getData().getBaseFields().isSocialCareConsentRequired());
    }

    public boolean isAnyPrivacyFlagInGroupEditable() {
        return allergies.stream().anyMatch(DataWithUIPermissions::isPrivacyFlagEditable);
    }

    public boolean isAnyInGroupEditOrDeletable() {
        return allergies.stream().anyMatch(DataWithUIPermissions::isEditOrDeletable);
    }
}
