package com.pkb.dto;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.pkb.app.entity.AbstractPersonRoles;
import org.immutables.value.Value.Immutable;

import java.util.Optional;

@Immutable
@JsonSerialize(as = ImmutableDiscussionParticipantDTO.class)
@JsonDeserialize(as = ImmutableDiscussionParticipantDTO.class)
public abstract class DiscussionParticipantDTO extends AbstractPersonRoles {

    public abstract Long getId();
    public abstract String getTitle();
    public abstract String getFirstName();
    public abstract String getLastName();
    public abstract String getFullName();
    public abstract Optional<String> getFullNameWithMeta();
    public abstract Optional<String> getTeamName();
    public abstract boolean getIsCarer();
    public abstract Optional<UrlDto> getNameAsAnchor();

    public boolean getIsPatient() {
        return isPatient();
    }

    public  boolean getIsProfessional() {
        return isPro();
    }
}
