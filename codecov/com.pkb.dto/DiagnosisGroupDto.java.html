package com.pkb.dto;

import com.pkb.app.entity.SourceDetails;
import com.pkb.coding.entity.CodeableConcept;
import com.pkb.diagnosis.entity.Diagnosis;
import com.pkb.mappers.DiagnosisMapper;
import com.pkb.service.datapoint.DataWithUIPermissions;
import com.pkb.service.diagnosis.DiagnosisDTO;

import java.util.Collections;
import java.util.Comparator;
import java.util.Date;
import java.util.List;

import static java.util.stream.Collectors.toList;

/**
 * We group diagnoses by source, code and route; this holds a group and has util methods to help display it
 */
public class DiagnosisGroupDto {

    private List<DataWithUIPermissions<Diagnosis>> diagnoses;
    private List<DiagnosisDTO> diagnosesDTOs;
    private boolean past;

    /** also sorts */
    public DiagnosisGroupDto(List<DataWithUIPermissions<Diagnosis>> diagnoses, boolean isPast) {
        if (diagnoses == null) {
            this.diagnoses = Collections.emptyList();
        } else {
            this.diagnoses = diagnoses;
        }
        this.past = isPast;

        sortDiagnoses();
    }

    public Long getFirstId() {
        return getDiagnoses().stream()
                .findFirst()
                .map(d -> d.getData().getId()).orElse(null);
    }

    public CodeableConcept getDiagnosisConcept() {
        return getDiagnoses().stream()
                .findFirst()
                .map(d -> d.getData().getDiagnosis()).orElse(null);
    }

    /**
     * @return the earliest start date in the list (or null)
     */
    public Date getFirstStartDate() {
        return getDiagnoses().stream()
                .min(
                        Comparator.comparing(d -> d.getData().getStartDate(), Comparator.nullsLast(Date::compareTo)))
                .map(d -> d.getData().getStartDate())
                .orElse(null);
    }

    /**
     * @return the latest end date in the list (or null)
     */
    public Date getLastEndDate() {
        return diagnoses.stream()
                .max(Comparator.comparing(d -> d.getData().getEndDate(), Comparator.nullsFirst(Date::compareTo)))
                .map(d -> d.getData().getEndDate()).orElse(null);
    }

    /**
     * @return the first source -- the org/team/person IDs and route will be identical across diagnoses in this group, but
     *         the source _text_ may vary
     */
    public SourceDetails getSource() {
        return getDiagnoses().stream()
                .findFirst()
                .map(d -> d.getData().getSource()).orElse(null);
    }

    public String getClinicalCodes() {
        return getDiagnoses().stream()
                .findFirst()
                .map(d -> d.getData().getClinicalCodes()).orElse(null);
    }

    public void setPast(boolean past) {
        this.past = past;
    }

    public boolean isPast() {
        return past;
    }

    public List<DataWithUIPermissions<Diagnosis>> getDiagnoses() {
        return diagnoses;
    }

    /** sort by end date */
    private void sortDiagnoses() {
        Comparator<DataWithUIPermissions<Diagnosis>> diagnosisComparator = Comparator.comparing(d -> d.getData().getEndDate(), Comparator.nullsLast(Date::compareTo));
        diagnoses = getDiagnoses().stream()
                .sorted(diagnosisComparator.reversed())
                .collect(toList());
    }

    public List<DiagnosisDTO> getDiagnosesDTOs() {
        return diagnoses.stream()
                .map(this::convertToDto)
                .collect(toList());
    }

    private DiagnosisDTO convertToDto(DataWithUIPermissions<Diagnosis> input) {
        DiagnosisDTO result = DiagnosisMapper.INSTANCE.diagnosisToDTO(input.getData());
        input.copyPermissionsInto(result);
        return result;
    }

    public boolean isSexualHealthConsentRequiredInGroup() {
        return diagnoses.stream().anyMatch(d -> d.getData().getBaseFields().isSexualHealthConsentRequired());
    }

    public boolean isGeneralHealthConsentRequiredInGroup() {
        return diagnoses.stream().anyMatch(d -> d.getData().getBaseFields().isGeneralHealthConsentRequired());
    }

    public boolean isMentalHealthConsentRequiredInGroup() {
        return diagnoses.stream().anyMatch(d -> d.getData().getBaseFields().isMentalHealthConsentRequired());
    }

    public boolean isSocialCareConsentRequiredInGroup() {
        return diagnoses.stream().anyMatch(d -> d.getData().getBaseFields().isSocialCareConsentRequired());
    }

    public boolean isAnyPrivacyFlagInGroupEditable() {
        return diagnoses.stream().anyMatch(DataWithUIPermissions::isPrivacyFlagEditable);
    }

    public boolean isAnyInGroupEditOrDeletable() {
        return diagnoses.stream().anyMatch(DataWithUIPermissions::isEditOrDeletable);
    }

}
