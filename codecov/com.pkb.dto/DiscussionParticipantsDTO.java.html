package com.pkb.dto;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import org.immutables.value.Value.Immutable;

import java.util.List;

import static org.apache.commons.collections.CollectionUtils.isNotEmpty;

@Immutable
@JsonSerialize(as = ImmutableDiscussionParticipantsDTO.class)
@JsonDeserialize(as = ImmutableDiscussionParticipantsDTO.class)
public interface DiscussionParticipantsDTO {

    List<DiscussionTbmParticipantDTO> getTbmParticipants();
    List<DiscussionParticipantDTO> getNonTbmParticipants();
    List<DiscussionNonPkbParticipantDTO> getNonPkbParticipants();

    default int size() {
        return getTbmParticipants().size() + getNonTbmParticipants().size();
    }

    default boolean hasTbmParticipants() {
        return isNotEmpty(getTbmParticipants());
    }

    default boolean hasNonTbmParticipants() {
        return isNotEmpty(getNonTbmParticipants());
    }

    default boolean hasNonPkbParticipants() {
        return isNotEmpty(getNonPkbParticipants());
    }

    default boolean hasPkbParticipants() {
        return hasTbmParticipants() || hasNonTbmParticipants();
    }

    default boolean hasParticipants() {
        return hasPkbParticipants() || hasNonPkbParticipants();
    }
}
