package com.pkb.query.domain;

import java.util.ArrayList;
import java.util.List;

public class QueryMetadata {
    private List<QuerySortOrder> orderBy;
    private QueryPagination pagination;

    protected QueryMetadata() {
        orderBy = new ArrayList<>();
    }

    public static QueryMetadata create() {
        return new QueryMetadata();
    }

    public QueryMetadata withOrder(String fieldName, QuerySortOrder.SortOrderDirection direction) {
        orderBy.add(new QuerySortOrder(direction, fieldName));
        return this;
    }

    public QueryMetadata withPagination(int pageSize, int offset) {
        if (pageSize > 0 && offset >= 0) {
            pagination = new QueryPagination(pageSize, offset);
        }
        return this;
    }

    public QueryMetadata resetPagination() {
        pagination = null;
        return this;
    }

    public QueryPagination pagination() {
        return pagination;
    }

    public List<QuerySortOrder> order() {
        return new ArrayList<>(orderBy);
    }

    public boolean isOrdered() {
        return !orderBy.isEmpty();
    }

    public boolean isPaginated() {
        return pagination != null;
    }
}
