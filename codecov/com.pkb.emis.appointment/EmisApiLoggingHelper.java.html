package com.pkb.emis.appointment;

import java.net.URI;
import java.util.regex.Pattern;

public class EmisApiLoggingHelper {

    private static final String EMIS_API_REQUEST = "EMIS-API-REQUEST ";
    private static final Pattern REDACT_USER_PATIENT_LINK_TOKEN_PATTERN = Pattern.compile("userPatientLinkToken=.*?&");
    private static final String LINK_TOKEN_REDACTED_PARAM = "userPatientLinkToken=REDACTED&";
    private static final Pattern REDACT_USER_PATIENT_LINK_TOKEN_PATTERN_EOL = Pattern.compile("userPatientLinkToken=[^&]*$");
    private static final String LINK_TOKEN_REDACTED_PARAM_EOL = "userPatientLinkToken=REDACTED";

    public static String getRedactedEmisApiRequestMessage(URI uri) {
        return EMIS_API_REQUEST
                + REDACT_USER_PATIENT_LINK_TOKEN_PATTERN_EOL.matcher(
                        REDACT_USER_PATIENT_LINK_TOKEN_PATTERN.matcher(uri.toString()).replaceAll(LINK_TOKEN_REDACTED_PARAM)
        ).replaceFirst(LINK_TOKEN_REDACTED_PARAM_EOL);
    }

    static String getEmisApiRequest() {
        return EMIS_API_REQUEST;
    }

}
