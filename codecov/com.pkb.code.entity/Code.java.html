package com.pkb.code.entity;

import org.hibernate.annotations.Cache;
import org.hibernate.annotations.CacheConcurrencyStrategy;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;

/**
 * code dictionary for looking up details for a code like code system , description , value domain etc
 *
 * @author pravina
 *
 */
@Entity
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table
public class Code implements Serializable {

    /**
     *
     */
    private static final long serialVersionUID = 4416469939725486931L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column
    private String code;

    @Column
    @Enumerated(EnumType.STRING)
    private CodeSystemName codeSystemName;

    @Column
    private String version;

    @Column
    private String displayName;

    public Code() {
    }

    public Code(String code, CodeSystemName codeSystemName, String displayName) {
        this.code = code;
        this.codeSystemName = codeSystemName;
        this.displayName = displayName;
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getCode() {
        return code;
    }

    public void setCode(String code) {
        this.code = code;
    }

    public CodeSystemName getCodeSystemName() {
        return codeSystemName;
    }

    public void setCodeSystemName(CodeSystemName codeSystemName) {
        this.codeSystemName = codeSystemName;
    }

    public String getDisplayName() {
        return displayName;
    }

    public void setDisplayName(String displayName) {
        this.displayName = displayName;
    }

    public String getVersion() {
        return version;
    }

    public void setVersion(String version) {
        this.version = version;
    }

}
