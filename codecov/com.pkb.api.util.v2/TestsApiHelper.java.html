package com.pkb.api.util.v2;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.api.util.converter.v2.TestResultConverter;
import com.pkb.api.v2.model.ApiTestResult;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.service.test.LoincManager;
import com.pkb.service.test.TestManager;
import com.pkb.test.entity.LoincTest;
import com.pkb.test.entity.TestHistoryDTO;
import com.pkb.test.entity.TestResultDTO;
import com.pkb.util.LocalizedTextProvider;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.stream.Collectors;

import static com.pkb.test.IncludeDeletedResults.EXCLUDE_DELETED;

public class TestsApiHelper {

    private final TestManager testManager;

    private final LocalizedTextProvider localizedTextProvider;

    private LoincManager loincManager;

    public TestsApiHelper(TestManager testManager, LoincManager loincManager, LocalizedTextProvider localizedTextProvider) {
        this.testManager = testManager;
        this.loincManager = loincManager;
        this.localizedTextProvider = localizedTextProvider;
    }

    /**
     * returns test results
     *
     * @param requestContext EHR request context
     * @param testLoincs     filter by test loinc codes
     * @param latestOnly     this overrides monthsOfDataOrNull
     * @param patientId      patient id
     * @return list of test results
     */
    public List<ApiTestResult> getTestResults(LoggedInEHRRequestContext requestContext, List<String> testLoincs, long patientId, boolean latestOnly) {

        List<TestHistoryDTO> testHistoryList;
        List<Long> testIds = loincManager.findLoincTestsByLoincCodesReturnAllByEmptyList(testLoincs).stream().map(LoincTest::getId)
                .collect(Collectors.toList());

        if (testIds.isEmpty()) {
            // The provided LOINC codes did not match any known to PKB. So we can skip the search and return empty now.
            return Collections.emptyList();
        }

        if (latestOnly) {

            testHistoryList = testManager.getLatestTestHistory(requestContext, patientId, testIds, EXCLUDE_DELETED);

        } else {
            testHistoryList = testManager.getTestHistoryList(requestContext, patientId, testIds, null, EXCLUDE_DELETED);
        }

        return getApiTestList(testHistoryList);
    }

    private List<ApiTestResult> getApiTestList(List<TestHistoryDTO> testHistoryList) {
        List<ApiTestResult> resultList = new ArrayList<>();

        for (TestHistoryDTO historyDto : testHistoryList) {
            String loinc = historyDto.getLoincTest().getLoinc();
            String units = historyDto.getLoincTest().getUnit();
            for (TestResultDTO resultDto : historyDto.getTestResultList()) {
                resultList.add(convert(resultDto, loinc, units));
            }
        }

        return resultList;
    }

    @VisibleForTesting
    ApiTestResult convert(TestResultDTO dto, String loincCode, String units) {
        TestResultConverter converter = new TestResultConverter(localizedTextProvider);
        ApiTestResult apiTestResult = converter.toApiModel(dto);
        apiTestResult.setLoincCode(loincCode);
        apiTestResult.setUnit(units);
        return apiTestResult;
    }
}
