package com.pkb.api.util.v2;

import com.pkb.api.error.ApiAuthError;
import com.pkb.api.util.converter.v2.DiagnosisConverter;
import com.pkb.api.v2.model.ApiDiagnosis;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.diagnosis.entity.Diagnosis;
import com.pkb.exception.PKBException;
import com.pkb.service.diagnosis.DiagnosisManager;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;

import java.time.Instant;
import java.util.ArrayList;
import java.util.List;

public class V2DiagnosisApiHelper {

    private final DiagnosisManager diagnosisManager;

    private final PatientConsentManager patientConsentManager;

    private final UserManager userManager;

    private final DateTimeService dateTimeService;

    public V2DiagnosisApiHelper(DiagnosisManager diagnosisManager,
                                PatientConsentManager patientConsentManager, UserManager userManager, @NotNull DateTimeService dateTimeService) {
        this.diagnosisManager = diagnosisManager;
        this.patientConsentManager = patientConsentManager;
        this.userManager = userManager;
        this.dateTimeService = dateTimeService;
    }

    public List<ApiDiagnosis> getDiagnoses(@NotNull LoggedInEHRRequestContext requestContext, List<String> apiRefsOrNull,
            Long patientId, PKBPerson loggedInUser, Instant updatedSince, boolean includeNegativeConditions) {

        if (loggedInUser.getId().equals(patientId)) {
            if (!patientConsentManager.hasAccessToUser(requestContext, patientId)) {
                throw new ApiAuthError("current user " + loggedInUser.getId()
                        + " does not have access to patient " + patientId);
            }
        }
        long accountId = userManager.getDefaultAccountId(patientId);
        List<ApiDiagnosis> results = new ArrayList<>();

        if ((apiRefsOrNull != null) && apiRefsOrNull.isEmpty()) {
            apiRefsOrNull = null;
        }
        if (diagnosisManager != null) {

            List<Diagnosis> diagnoses = diagnosisManager.getDiagnoses(patientId, accountId,
                    apiRefsOrNull, updatedSince, requestContext, includeNegativeConditions
            /*fetch RRs*/);
            DiagnosisConverter diagnosisConverter = new DiagnosisConverter();
            for (Diagnosis diagnosis : diagnoses) {
                results.add(diagnosisConverter.toApiModel(diagnosis));
            }

            return results;
        } else {
            throw new PKBException("how can that be null?");
        }

    }
}
