package com.pkb.api.util.v2;

import com.pkb.allergy.entity.Allergy;
import com.pkb.api.error.ApiAuthError;
import com.pkb.api.v2.model.ApiAllergy;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.coding.entity.CodeableConcept;
import com.pkb.service.allergy.impl.AllergyManager;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.stream.Collectors;

@Component
public class AllergiesApiHelper {

    private final UserManager userManager;

    private final AllergyManager allergyManager;

    private final PatientConsentManager patientConsentManager;

    @Autowired
    public AllergiesApiHelper(AllergyManager allergyManager, PatientConsentManager patientConsentManager,
                              UserManager userManager) {
        this.patientConsentManager = patientConsentManager;
        this.userManager = userManager;
        this.allergyManager = allergyManager;
    }

    public List<ApiAllergy> getAllergiesForPatient(@NotNull LoggedInEHRRequestContext requestContext, long patientId, PKBPerson user, Boolean includeNegativeAllergies) {

        if (!patientConsentManager.hasAccessToUser(requestContext, patientId)) {
            throw new ApiAuthError("current user " + user.getId() + " does not have access to patient "
                    + patientId);
        }

        Long accountId = userManager.getDefaultAccountId(patientId);

        List<Allergy> allergies = allergyManager.getAllergies(requestContext, patientId, accountId, 0, -1, includeNegativeAllergies);
        List<ApiAllergy> results = allergies.stream().map(this::convert).collect(Collectors.toList());
        return results;
    }

    private ApiAllergy convert(Allergy allergy) {

        ApiAllergy api = new ApiAllergy();
        api.setId(allergy.getBaseFields().getUniqueId());
        if (allergy.getAllergen() != null) {
            api.setAllergen(allergy.getAllergen().getTextForDisplay());
            if (!allergy.getAllergen().getCodings().isEmpty()) {
                // HACK: need richer ApiAllergy
                api.setSubstanceCode(allergy.getAllergen().getCodings().get(0).getCode());
            }
        }
        api.setOnsetOfSymptoms(allergy.getOnsetOfSymptoms());
        api.setReactions(allergy.getReactions().stream()
                .map(CodeableConcept::getDisplayText)
                .collect(Collectors.toList()));
        api.setEnteredDate(allergy.getBaseFields().getEnteredDate());
        api.setSeverity(allergy.getSeverity() != null ? allergy.getSeverity().getDisplayText() : null);
        //should present a default value of true if value hasn't been assigned (because a null value is meaningless and just indicates old data)
        if (allergy.getPatientIsAllergic() == null) {
            api.setPatientIsAllergic(true);
        } else {
            api.setPatientIsAllergic(allergy.getPatientIsAllergic());
        }
        return api;
    }
}
