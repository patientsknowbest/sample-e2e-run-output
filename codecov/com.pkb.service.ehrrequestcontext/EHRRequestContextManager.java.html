package com.pkb.service.ehrrequestcontext;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.ImmutableLoggedInUserEHRRequestContext;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.PersonAuthInfo;
import com.pkb.consent.model.ConsentStatus;
import com.pkb.datamodel.consent.PatientConsentDTO;
import com.pkb.entities.enums.AccessingEntityType;
import com.pkb.entities.enums.Route;
import com.pkb.entities.enums.UserType;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.team.TeamManager;
import com.pkb.service.user.impl.UserManager;
import io.vavr.control.Try;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Optional;
import java.util.UUID;

import static com.pkb.app.entity.EHRRequestContext.AccountLinkType.INDIVIDUAL;
import static com.pkb.app.entity.EHRRequestContext.AccountLinkType.ORG;
import static com.pkb.consent.model.AbstractConsentStatus.noConsentRequired;
import static com.pkb.entities.enums.Route.WEBAPP;
import static com.pkb.entities.enums.UserType.REG_CLINICIAN;

public class EHRRequestContextManager {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final TeamManager teamManager;
    private final PatientConsentManager patientConsentManager;
    private final UserManager userManager;

    public EHRRequestContextManager(TeamManager teamManager, PatientConsentManager patientConsentManager, UserManager userManager) {
        this.teamManager = teamManager;
        this.patientConsentManager = patientConsentManager;
        this.userManager = userManager;
    }

    public LoggedInEHRRequestContext createEHRRequestContextForOwner(PersonAuthInfo patient, UUID correlationId,
                                                                     Route route) {
        // Mark consent as not required - we're logged in as the owner
        ConsentStatus consentStatus = ConsentStatus.builder()
                .breakTheGlassActive(false)
                .required(false)
                .patientSharingEnabled(isSharingEnabled(patient.getId()))
                .build();

        LOGGER.debug("setting up request context for patient viewing their own account");

        return ImmutableLoggedInUserEHRRequestContext.builder()
                .accountLinkType(INDIVIDUAL)
                .accessRoute(route)
                .consentStatus(consentStatus)
                .accessingUser(patient)
                .correlationId(correlationId)
                .contextUserId(patient.getId())
                .build();
    }

    public LoggedInEHRRequestContext createEHRRequestContextForTeamClinician(
            PersonAuthInfo clinician, Team team, Long patientId, UUID correlationId, Route route,
            boolean breakTheGlassStatusActive) {

        // TODO: team coords currently use this context (sometimes) to fetch patient data
        // and the REST API uses it incorrectly to fetch patient data (w/o passing in a patient here)
        UserType currentUserType = clinician.getUserType();
        EHRRequestContext.AccountLinkType linkType = (patientId == null && currentUserType == REG_CLINICIAN && route == WEBAPP) ? INDIVIDUAL : ORG;

        ImmutableLoggedInUserEHRRequestContext requestContext = ImmutableLoggedInUserEHRRequestContext.builder()
                .accountLinkType(linkType)
                .accessRoute(route)
                .consentStatus(noConsentRequired())
                .accessingUser(clinician)
                .teamId(team.getId())
                .org(team.getOrg())
                .correlationId(correlationId)
                .build();

        if (patientId != null) {
            // Load consent
            ConsentStatus consentStatus = getConsentStatusForClinician(clinician, patientId, team,
                    breakTheGlassStatusActive);
            requestContext = requestContext.withContextUserId(patientId).withConsentStatus(consentStatus);
        } else {
            requestContext = requestContext.withContextUserId(clinician.getId());
        }

        LOGGER.debug("team clinician request context: {}", requestContext);
        return requestContext;
    }

    public LoggedInEHRRequestContext createEHRRequestContextForOrgCoord(PersonAuthInfo orgCoord, UUID correlationId, Route route) {

        Org org = teamManager.getOrgByActiveIUForPerson(orgCoord.getId());

        LoggedInEHRRequestContext requestContext = ImmutableLoggedInUserEHRRequestContext.of(ORG, route, ConsentStatus.builder().required(true).build(), correlationId, orgCoord)
                .withOrg(org);

        // Note: no consents are set. An org coord is not representing any
        // specific team, and will never be given any individual consent.
        // However, they do not need consent to see or process data from their
        // own organisation.

        LOGGER.debug("org coord request context: {}", requestContext);
        return requestContext;
    }

    public LoggedInEHRRequestContext createEHRRequestContextForPrivacyOfficer(
            PersonAuthInfo privacyOfficer, UUID correlationId, Route route) {

        Org org = teamManager.getOrgByActiveIUForPerson(privacyOfficer.getId());

        LoggedInEHRRequestContext requestContext = ImmutableLoggedInUserEHRRequestContext.of(ORG, route, ConsentStatus.builder().required(true).build(), correlationId, privacyOfficer)
                .withOrg(org);

        LOGGER.debug("privacy officer request context: {}", requestContext);
        return requestContext;

    }

    public LoggedInEHRRequestContext createEHRRequestContextForIndividualClinician(
            PersonAuthInfo clinician, Long patientId, UUID correlationId, Route route) {

        boolean breakTheGlassStatusActive = false; // We never support BTG for individual pros

        ImmutableLoggedInUserEHRRequestContext requestContext = ImmutableLoggedInUserEHRRequestContext.of(INDIVIDUAL, route, noConsentRequired(), correlationId, clinician);
        if (patientId != null) {
            // Load consent (can pass null for context, as EHRBean won't need
            // the org account key)
            requestContext = requestContext.withConsentStatus(
                    getConsentStatusForClinician(clinician, patientId, null, breakTheGlassStatusActive)).withContextUserId(patientId);
        }

        LOGGER.debug("non-team-affiliated clinician request context: {}", requestContext);
        return requestContext;
    }

    public LoggedInEHRRequestContext createEHRRequestContextForCarer(PersonAuthInfo carer, long patientId, UUID correlationId,
                                                                     Route route) {
        LOGGER.info("setting up request context for carer");

        LoggedInEHRRequestContext requestContext = ImmutableLoggedInUserEHRRequestContext.of(INDIVIDUAL, route, getConsentStatusForCarer(carer.getId(), patientId), correlationId, carer).withContextUserId(patientId);

        LOGGER.debug("carer request context: {}", requestContext);
        return requestContext;
    }

    public LoggedInEHRRequestContext createEHRRequestContextForPkbAdmin(
            PersonAuthInfo pkbAdmin, UUID correlationId, Route route) {
        LoggedInEHRRequestContext requestContext = ImmutableLoggedInUserEHRRequestContext.of(INDIVIDUAL, route, noConsentRequired(), correlationId, pkbAdmin);
        LOGGER.debug("PKB admin request context: {}", requestContext);
        return requestContext;
    }

    private ConsentStatus getConsentStatusForClinician(PersonAuthInfo clinician,
                                                       long patientId,
                                                       Team team,
                                                       boolean breakTheGlassStatusActive) {

        ConsentStatus consentStatus = null;

        boolean sharingEnabled = isSharingEnabled(patientId);

        if (breakTheGlassStatusActive) {
            // If we have break-the-glass, don't bother hitting the DB for consent
            // details. If caller still needs them, they can be loaded separately
            consentStatus = ConsentStatus.builder()
                    .breakTheGlassActive(true)
                    .required(true)
                    .patientSharingEnabled(sharingEnabled)
                    .build();
            return consentStatus;
        }

        PatientConsentDTO patientConsent = getTeamConsent(patientId, team).orElse(null);

        // Clinician may have access via their team (team clinician), or
        // may have individual consent (non-team clinician)

        if (patientConsent == null) {
            patientConsent = getIndividualConsent(patientId, clinician).orElse(null);
        }

        if (patientConsent != null) {
            consentStatus = ConsentStatus.builder()
                    .consent(patientConsent)
                    .breakTheGlassActive(false)
                    .required(true)
                    .patientSharingEnabled(sharingEnabled)
                    .build();
        } else {
            // no consent -- but it's still required (block access except for fallback TDV)
            consentStatus = ConsentStatus.builder()
                    .required(true)
                    .orgNetworkConsentPresent(orgNetworkConsentExists(patientId, team))
                    .patientSharingEnabled(sharingEnabled)
                    .build();
        }

        return consentStatus;
    }

    private boolean orgNetworkConsentExists(long patientId, Team team) {
        return team != null && patientConsentManager.consentExistsInOrgNetwork(patientId, team.getId());
    }

    private ConsentStatus getConsentStatusForCarer(long carerId, long patientId) {

        ConsentStatus consentStatus = null;

        boolean sharingEnabled = isSharingEnabled(patientId);

        consentStatus = getCarerConsent(patientId, carerId).map(pc -> ConsentStatus.builder()
                .consent(pc)
                .breakTheGlassActive(false)
                .required(true)
                .patientSharingEnabled(sharingEnabled)
                .build()).orElse(null);

        return consentStatus;
    }

    private Optional<PatientConsentDTO> getTeamConsent(Long patientId, Team team) {
        if (team == null) {
            // Can't have team consent without a team!
            return Optional.empty();
        }

        // We need a request context with ORG set so EHRBean knows to load the org account key.
        // So pass through our context even though we're still building it
        return patientConsentManager.findPatientConsent(patientId, team.getId(), AccessingEntityType.TEAM);
    }

    private Optional<PatientConsentDTO> getIndividualConsent(Long patientId, PersonAuthInfo accessingUser) {
        return patientConsentManager.findPatientConsent(patientId,
                accessingUser.getId(), AccessingEntityType.INDIVIDUAL);
    }

    private Optional<PatientConsentDTO> getCarerConsent(Long patientId, long accessingUserId) {
        return patientConsentManager.findPatientConsent(patientId,
                accessingUserId, AccessingEntityType.CARER);
    }

    private boolean isSharingEnabled(long personId) {
        //WARNING: this can potentially be called with an invalid personId
        //from e.g. the REST API, so this method needs to be defensive
        return Try.of(() -> userManager.findPKBPerson(personId)
                .map(person -> {
                    if (person.getOptoutDetails() != null) {
                        return person.getOptoutDetails().isSharingEnabled();
                    }
                    return true;
                }).getOrElse(true))
                .getOrElse(true);
    }

}
