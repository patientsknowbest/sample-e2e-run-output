package com.pkb.aggregator.service

import com.pkb.aggregator.R4Identifier
import com.pkb.aggregator.R4Organization
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.persistence.entity.OrganizationIdentifierMapEntry
import com.pkb.aggregator.persistence.entity.embeddable.MapEntryKey
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamIdentifier
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.persistence.repository.DatabaseStateClearer
import com.pkb.aggregator.persistence.repository.OrganizationIdentifierMapRepository
import com.pkb.aggregator.service.aggregation.toMapEntryKeys
import org.hl7.fhir.instance.model.api.IDomainResource
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles
import java.util.UUID
import javax.transaction.Transactional

const val ODS_IDENTIFIER_SYSTEM = "https://fhir.nhs.uk/Id/ods-organization-code"

@Service
class OrganizationIdentifierService(
    private val organizationIdentifierMapRepository: OrganizationIdentifierMapRepository,
    private val stateClearer: DatabaseStateClearer,
) : AbstractIdentifierService<R4Organization, OrganizationIdentifierMapEntry>() {

    private val log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun deriveDownstreamResourceId(baseResources: List<IDomainResource>): String {

        val resources = baseResources.mapNotNull { it as? R4Organization }
        return deriveDownstreamIdFromODSCode(resources)
            ?: deriveDownstreamIdFromUpstreamSource(resources)
            ?: throw RuntimeException("No ODS code or source tag in $resources")
    }

    fun findByUpstreamResourceIdIn(upstreamResourceIds: Collection<UpstreamResource>) =
        organizationIdentifierMapRepository.findByMapEntryKeyUpstreamResourceIn(
            upstreamResourceIds.toSet(),
            OrganizationIdentifierMapEntry::class.java)

    @Transactional
    override fun mergeIdentifiers(resource: R4Organization): Int {
        val downstreamResourceId = resource.simpleResourceId()
        val newIdentifiers = findNewIdentifiers(downstreamResourceId, resource)
        return newIdentifiers.map { OrganizationIdentifierMapEntry(it, downstreamResourceId) }
            .sumOf {
                val changed = organizationIdentifierMapRepository.safeInsertIdentifier(
                    it.mapEntryKey.identifier.upstreamIdentifierSystem,
                    it.mapEntryKey.identifier.upstreamIdentifierValue,
                    it.mapEntryKey.upstreamResource.upstreamSource,
                    it.mapEntryKey.upstreamResource.upstreamResourceId,
                    it.downstreamResourceId,
                )
                if (changed == 0) {
                    log.info("Skipped inserting identifier of system {} for downstream resource {}",
                        it.mapEntryKey.identifier.upstreamIdentifierSystem, it.downstreamResourceId)
                }
                changed
            }
    }

    override fun findMapEntryKeys(resource: R4Organization): List<MapEntryKey> {
        return resource.identifier.flatMap { it.toMapEntryKeys() }
    }

    override fun findMapEntryKeysDownstreamResourceId(downstreamResourceId: String) =
        organizationIdentifierMapRepository.findByDownstreamResourceId(
            downstreamResourceId,
            OrganizationIdentifierMapEntry::class.java
        ).map { it.mapEntryKey }

    fun findByUpstreamIdentifierIn(identifiers: List<R4Identifier>) =
        organizationIdentifierMapRepository.findByMapEntryKeyIdentifierIn(
            identifiers.map { UpstreamIdentifier(it.system, it.value) },
            OrganizationIdentifierMapEntry::class.java
        )

    fun findSearchIdentifiers(upstreamSystem: String, upstreamValue: String): List<OrganizationIdentifierMapEntry> {
        val upstreamIdentifier = UpstreamIdentifier(upstreamSystem, upstreamValue)
        return organizationIdentifierMapRepository.findByMapEntryKeyIdentifier(upstreamIdentifier,
            OrganizationIdentifierMapEntry::class.java)
    }

    private fun deriveDownstreamIdFromODSCode(baseResources: List<R4Organization>) =
        baseResources
            .flatMap { it.identifier }
            .find { it.system == ODS_IDENTIFIER_SYSTEM }
            ?.let { UUID.nameUUIDFromBytes("pkb-internal-${it.system}/${it.value}".toByteArray()).toString() }
            ?.let { "$FHIR_TYPE_ORGANIZATION/$it" }

    override fun clearState() = stateClearer.invoke {
        organizationIdentifierMapRepository.deleteAll()
    }
}
