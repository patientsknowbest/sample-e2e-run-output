package com.pkb.aggregator.service

import com.pkb.aggregator.R4Identifier
import com.pkb.aggregator.R4Practitioner
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.persistence.entity.PractitionerIdentifierMapEntry
import com.pkb.aggregator.persistence.entity.embeddable.MapEntryKey
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamIdentifier
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.persistence.repository.DatabaseStateClearer
import com.pkb.aggregator.persistence.repository.PractitionerIdentifierMapRepository
import com.pkb.aggregator.service.aggregation.toMapEntryKeys
import org.hl7.fhir.instance.model.api.IDomainResource
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles
import java.util.UUID
import javax.transaction.Transactional

const val DDS_IDENTIFIER_SYSTEM = "https://fhir.hl7.org.uk/Id/dds"
const val HUMANUUID_IDENTIFIER_SYSTEM = "http://fhir.patientsknowbest.com/id/human-uuid"

@Service
class PractitionerIdentifierService(
    private val practitionerIdentifierMapRepository: PractitionerIdentifierMapRepository,
    private val stateClearer: DatabaseStateClearer,
) : AbstractIdentifierService<R4Practitioner, PractitionerIdentifierMapEntry>() {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun deriveDownstreamResourceId(baseResources: List<IDomainResource>): String {

        val resources = baseResources.filterIsInstance<R4Practitioner>()
        return deriveDownstreamIdFromIdentifierWithSystem(resources, DDS_IDENTIFIER_SYSTEM)
            ?: deriveDownstreamIdFromIdentifierWithSystem(resources, HUMANUUID_IDENTIFIER_SYSTEM)
            ?: deriveDownstreamIdFromUpstreamSource(resources)
            ?: throw RuntimeException("No usable identifier or source tag in $resources")
    }

    fun findByUpstreamResourceIdIn(upstreamResourceIds: Collection<UpstreamResource>) =
        practitionerIdentifierMapRepository.findByMapEntryKeyUpstreamResourceIn(
            upstreamResourceIds.toSet(),
            PractitionerIdentifierMapEntry::class.java)

    override fun findMapEntryKeysDownstreamResourceId(downstreamResourceId: String) =
        practitionerIdentifierMapRepository.findByDownstreamResourceId(
            downstreamResourceId,
            PractitionerIdentifierMapEntry::class.java
        ).map { it.mapEntryKey }

    override fun clearState() = stateClearer.invoke {
        practitionerIdentifierMapRepository.deleteAll()
    }

    @Transactional
    override fun mergeIdentifiers(resource: R4Practitioner): Int {
        val downstreamResourceId = resource.simpleResourceId()
        val newIdentifiers = findNewIdentifiers(downstreamResourceId, resource)
        return newIdentifiers.map { PractitionerIdentifierMapEntry(it, downstreamResourceId) }
            .sumOf {
                val changed = practitionerIdentifierMapRepository.safeInsertIdentifier(
                    it.mapEntryKey.identifier.upstreamIdentifierSystem,
                    it.mapEntryKey.identifier.upstreamIdentifierValue,
                    it.mapEntryKey.upstreamResource.upstreamSource,
                    it.mapEntryKey.upstreamResource.upstreamResourceId,
                    it.downstreamResourceId,
                )
                if (changed == 0) {
                    log.info("Skipped inserting identifier of system {} for downstream resource {}",
                        it.mapEntryKey.identifier.upstreamIdentifierSystem, it.downstreamResourceId)
                }
                changed
            }
    }

    override fun findMapEntryKeys(resource: R4Practitioner): List<MapEntryKey> {
        return resource.identifier.flatMap { it.toMapEntryKeys() }
    }

    /**
     * Given a list of resources, tries to identify which ones represent the same user across different upstreams (or
     * even, for sources like DDS, the _same_ upstream) and groups them together keyed by their downstream id
     */
    fun groupResourcesByPractitioner(resources: List<IDomainResource>): Map<String, List<IDomainResource>> {

        val practitioners = resources.filterIsInstance<R4Practitioner>()

        // Grab all identifiers from all resources
        val identifiers = practitioners
            .flatMap { it.identifier }
            .map { UpstreamIdentifier(it.system, it.value) }

        // Look up all the identifiers from above
        val knownPractitioners = practitionerIdentifierMapRepository.findAllByUpstreamIdentifiers(
            identifiers,
            PractitionerIdentifierMapEntry::class.java)

        // Group by existing downstream if possible, otherwise generate new ones
        return practitioners.groupBy { getOrCreateDownstreamId(it, knownPractitioners) }
    }

    fun findByUpstreamIdentifierIn(identifiers: List<R4Identifier>) =
        practitionerIdentifierMapRepository.findByMapEntryKeyIdentifierIn(
            identifiers.map { UpstreamIdentifier(it.system, it.value) },
            PractitionerIdentifierMapEntry::class.java
        )

    fun findSearchIdentifiers(upstreamSystem: String, upstreamValue: String): List<PractitionerIdentifierMapEntry> {
        return practitionerIdentifierMapRepository.findAllByUpstreamIdentifiers(
            listOf(UpstreamIdentifier(upstreamSystem, upstreamValue)), PractitionerIdentifierMapEntry::class.java)
    }

    private fun getOrCreateDownstreamId(
        practitioner: R4Practitioner,
        knownPractitioners: List<PractitionerIdentifierMapEntry>
    ): String {

        // Look to see if the given practitioner matches any known identifiers
        val downstreamPractitioner = knownPractitioners.find {
            practitioner.identifier.any { identifier -> identifierMatch(identifier, it) }
        }

        // PHR-9768: Look for duplicates in new resources (e.g. if we get two resources for a new practitioner)

        return downstreamPractitioner?.downstreamResourceId ?: deriveDownstreamResourceId(listOf(practitioner))
    }

    private fun identifierMatch(identifier: R4Identifier, knownPractitioner: PractitionerIdentifierMapEntry) =
        identifier.system == knownPractitioner.mapEntryKey.identifier.upstreamIdentifierSystem
                && identifier.value == knownPractitioner.mapEntryKey.identifier.upstreamIdentifierValue

    private fun deriveDownstreamIdFromIdentifierWithSystem(baseResources: List<R4Practitioner>, system: String) =
        baseResources
            .flatMap { it.identifier }
            .find { it.system == system }
            ?.let { UUID.nameUUIDFromBytes("pkb-internal-${it.system}/${it.value}".toByteArray()) }
            ?.let { "$FHIR_TYPE_PRACTITIONER/$it" }
}
