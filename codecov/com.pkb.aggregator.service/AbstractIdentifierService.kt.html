package com.pkb.aggregator.service

import com.pkb.aggregator.extension.simpleDownstreamResourceId
import com.pkb.aggregator.persistence.entity.embeddable.MapEntryKey
import com.pkb.common.ClearableInternalState
import org.hl7.fhir.instance.model.api.IDomainResource

/**
 * Provides common functionality for working with resource identifiers that we track in the aggregator database
 */
abstract class AbstractIdentifierService<in T, out E> : ClearableInternalState where T : IDomainResource {

    abstract fun mergeIdentifiers(resource: T): Int

    abstract fun deriveDownstreamResourceId(baseResources: List<IDomainResource>): String

    protected fun findNewIdentifiers(downstreamResourceId: String, resource: T): Iterable<MapEntryKey> {
        val identifiersInResource = findMapEntryKeys(resource).toSet()
        val knownUpstreamIdentifiers = findMapEntryKeysDownstreamResourceId(downstreamResourceId).toSet()

        return identifiersInResource.minus(knownUpstreamIdentifiers)
    }

    protected abstract fun findMapEntryKeys(resource: T): List<MapEntryKey>

    protected abstract fun findMapEntryKeysDownstreamResourceId(downstreamResourceId: String): List<MapEntryKey>

    protected fun deriveDownstreamIdFromUpstreamSource(baseResources: List<IDomainResource>) =
        baseResources
            .minByOrNull { it.idElement.idPart }
            ?.let { it.simpleDownstreamResourceId() }
}
