package com.pkb.aggregator.service

import com.pkb.aggregator.R4Coding
import com.pkb.aggregator.R4Medication
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.persistence.entity.MedicationIdentifierMapEntry
import com.pkb.aggregator.persistence.entity.embeddable.MapEntryKey
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamIdentifier
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.persistence.repository.DatabaseStateClearer
import com.pkb.aggregator.persistence.repository.MedicationIdentifierMapRepository
import com.pkb.aggregator.service.aggregation.toMapEntryKeys
import org.hl7.fhir.instance.model.api.IDomainResource
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles
import java.util.UUID
import javax.transaction.Transactional

private const val SNOMED_IDENTIFIER_SYSTEM = "http://snomed.info/sct"

@Service
class MedicationIdentifierService(
    private val medicationIdentifierMapRepository: MedicationIdentifierMapRepository,
    private val stateClearer: DatabaseStateClearer,
) : AbstractIdentifierService<R4Medication, MedicationIdentifierMapEntry>() {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun deriveDownstreamResourceId(baseResources: List<IDomainResource>): String {

        // Look for snomed coding
        val resources = baseResources.mapNotNull { it as? R4Medication }
        return deriveDownstreamIdFromSnomedCoding(resources)
            ?: deriveDownstreamIdFromUpstreamSource(resources)
            ?: throw RuntimeException("No snomed code or source tag in $resources")
    }

    fun findByUpstreamResourceIdIn(upstreamResourceIds: Collection<UpstreamResource>) =
        medicationIdentifierMapRepository.findByMapEntryKeyUpstreamResourceIn(
            upstreamResourceIds.toSet(),
            MedicationIdentifierMapEntry::class.java
        )

    @Transactional
    override fun mergeIdentifiers(resource: R4Medication): Int {
        val downstreamResourceId = resource.simpleResourceId()
        val newIdentifiers = findNewIdentifiers(downstreamResourceId, resource)
        return newIdentifiers.map { MedicationIdentifierMapEntry(it, downstreamResourceId) }
            .sumOf {
                val changed = medicationIdentifierMapRepository.safeInsertIdentifier(
                    it.mapEntryKey.identifier.upstreamIdentifierSystem,
                    it.mapEntryKey.identifier.upstreamIdentifierValue,
                    it.mapEntryKey.upstreamResource.upstreamSource,
                    it.mapEntryKey.upstreamResource.upstreamResourceId,
                    it.downstreamResourceId,
                )
                if (changed == 0) {
                    log.info("Skipped inserting identifier of system {} for downstream resource {}",
                        it.mapEntryKey.identifier.upstreamIdentifierSystem, it.downstreamResourceId)
                }
                changed
            }
    }

    override fun findMapEntryKeys(resource: R4Medication): List<MapEntryKey> {
        return resource.code.coding.flatMap { it.toMapEntryKeys() }
    }

    override fun findMapEntryKeysDownstreamResourceId(downstreamResourceId: String) =
        medicationIdentifierMapRepository.findByDownstreamResourceId(
            downstreamResourceId,
            MedicationIdentifierMapEntry::class.java
        ).map { it.mapEntryKey }

    fun findByUpstreamIdentifierIn(identifiers: List<R4Coding>) =
        medicationIdentifierMapRepository.findByMapEntryKeyIdentifierIn(
            identifiers.map { UpstreamIdentifier(it.system, it.code) },
            MedicationIdentifierMapEntry::class.java
        )

    fun findSearchIdentifiers(upstreamSystem: String, upstreamValue: String): List<MedicationIdentifierMapEntry> {
        val upstreamIdentifier = UpstreamIdentifier(upstreamSystem, upstreamValue)
        return medicationIdentifierMapRepository.findByMapEntryKeyIdentifier(upstreamIdentifier,
            MedicationIdentifierMapEntry::class.java)
    }

    private fun deriveDownstreamIdFromSnomedCoding(baseResources: List<R4Medication>) =
        baseResources.flatMap { it.code.coding }
            .find { it.system == SNOMED_IDENTIFIER_SYSTEM }
            ?.let { UUID.nameUUIDFromBytes("pkb-internal-${it.system}/${it.code}".toByteArray()) }
            ?.let { "$FHIR_TYPE_MEDICATION/$it" }

    override fun clearState() = stateClearer.invoke {
        medicationIdentifierMapRepository.deleteAll()
    }
}
