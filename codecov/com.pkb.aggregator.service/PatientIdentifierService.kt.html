package com.pkb.aggregator.service

import com.pkb.aggregator.R4Identifier
import com.pkb.aggregator.R4Patient
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.persistence.entity.PatientIdentifierMapEntry
import com.pkb.aggregator.persistence.entity.embeddable.MapEntryKey
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamIdentifier
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.persistence.repository.DatabaseStateClearer
import com.pkb.aggregator.persistence.repository.PatientIdentifierMapRepository
import com.pkb.aggregator.service.aggregation.toMapEntryKeys
import org.hl7.fhir.instance.model.api.IDomainResource
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles
import java.util.UUID
import javax.transaction.Transactional

const val NHS_NUMBER_IDENTIFIER_SYSTEM = "https://fhir.nhs.uk/Id/nhs-number"

@Service
class PatientIdentifierService(
    private val patientIdentifierMapRepository: PatientIdentifierMapRepository,
    private val stateClearer: DatabaseStateClearer,
) : AbstractIdentifierService<R4Patient, PatientIdentifierMapEntry>() {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun deriveDownstreamResourceId(baseResources: List<IDomainResource>): String {

        val resources = baseResources.mapNotNull { it as? R4Patient }
        return deriveDownstreamIdFromNHSNumberIdentifier(resources)
            ?: deriveDownstreamIdFromUpstreamSource(resources)
            ?: throw RuntimeException("No NHS number or source tag in $resources")
    }

    fun findByUpstreamResourceIdIn(upstreamResourceIds: Collection<UpstreamResource>) =
        patientIdentifierMapRepository.findByMapEntryKeyUpstreamResourceIn(
            upstreamResourceIds.toSet(),
            PatientIdentifierMapEntry::class.java)

    override fun findMapEntryKeysDownstreamResourceId(downstreamResourceId: String) =
        patientIdentifierMapRepository.findByDownstreamResourceId(
            downstreamResourceId,
            PatientIdentifierMapEntry::class.java
        ).map { it.mapEntryKey }

    override fun clearState() = stateClearer.invoke {
        patientIdentifierMapRepository.deleteAll()
    }

    @Transactional
    override fun mergeIdentifiers(resource: R4Patient): Int {
        val downstreamResourceId = resource.simpleResourceId()
        val newIdentifiers = findNewIdentifiers(downstreamResourceId, resource)
        return newIdentifiers
            .map { PatientIdentifierMapEntry(it, downstreamResourceId) }
            .sumOf {
                val changed = patientIdentifierMapRepository.safeInsertIdentifier(
                    it.mapEntryKey.identifier.upstreamIdentifierSystem,
                    it.mapEntryKey.identifier.upstreamIdentifierValue,
                    it.mapEntryKey.upstreamResource.upstreamSource,
                    it.mapEntryKey.upstreamResource.upstreamResourceId,
                    it.downstreamResourceId,
                )
                if (changed == 0) {
                    log.info("Skipped inserting identifier of system {} for downstream resource {}",
                        it.mapEntryKey.identifier.upstreamIdentifierSystem, it.downstreamResourceId)
                }
                changed
            }
    }

    override fun findMapEntryKeys(resource: R4Patient): List<MapEntryKey> {
        return resource.identifier.flatMap { it.toMapEntryKeys() }
    }

    fun findByUpstreamIdentifierIn(identifiers: List<R4Identifier>) =
        patientIdentifierMapRepository.findByMapEntryKeyIdentifierIn(
            identifiers.map { UpstreamIdentifier(it.system, it.value) },
            PatientIdentifierMapEntry::class.java
        )

    fun findSearchIdentifiers(upstreamSystem: String, upstreamValue: String): List<PatientIdentifierMapEntry> {
        return patientIdentifierMapRepository.findAllByUpstreamIdentifiers(
            listOf(UpstreamIdentifier(upstreamSystem, upstreamValue)), PatientIdentifierMapEntry::class.java)
    }

    private fun deriveDownstreamIdFromNHSNumberIdentifier(baseResources: List<R4Patient>) =
        baseResources
            .flatMap { it.identifier }
            .find { it.system == NHS_NUMBER_IDENTIFIER_SYSTEM }
            ?.let { UUID.nameUUIDFromBytes("pkb-internal-${it.system}/${it.value}".toByteArray()) }
            ?.let { "$FHIR_TYPE_PATIENT/$it" }
}
