package org.tolven.core.entity;

import com.pkb.entities.enums.AccountUserStatus;
import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import com.pkb.institute.entity.InstituteUserEntity;
import org.hibernate.annotations.GenericGenerator;
import org.tolven.security.key.AccountPrivateKey;

import javax.persistence.Column;
import javax.persistence.Convert;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.OneToMany;
import javax.persistence.Table;
import javax.persistence.Temporal;
import javax.persistence.TemporalType;
import java.io.Serializable;
import java.util.Date;
import java.util.List;

/**
 * Link between user and account; holds this user's copy of the encrypted private key.
 */
@Entity
@Table(name = "account_user", schema = "core")
public class AccountUser implements Serializable, Cloneable {

    private static final long serialVersionUID = 1L;

    @ManyToOne
    private Account account;

    @Column(name = "person_id")
    private Long personId;

    @Id
    @GenericGenerator(name = "seq_core_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "core"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_core"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_core_gen")
    private long id;

    /**
     * @ deprecated use {@link com.pkb.user.entity.PKBPerson#defaultAccountId} instead.
     */
    private boolean defaultAccount;

    @Temporal(TemporalType.TIMESTAMP)
    @Column(name = "effective_date")
    private Date effectiveDate;

    @Temporal(TemporalType.TIMESTAMP)
    @Column(name = "expiration_date")
    private Date expirationDate;

    @Column
    @Convert(converter = AccountUserStatusConverter.class)
    private AccountUserStatus status;

    @OneToMany(fetch = FetchType.LAZY)
    @JoinColumn(name="person_id", insertable=false, updatable=false)
    private List<InstituteUserEntity> instituteUser;

    @Embedded
    private AccountPrivateKey accountPrivateKey;

    /**
     * Construct an empty AccountUser. 
     */
    public AccountUser() {
    }

    @Override
    public Object clone() throws CloneNotSupportedException {
        AccountUser clone = (AccountUser) super.clone();
        return clone;
    }

    public Account getAccount() {
        return account;
    }

    public void setAccount(Account val) {
        this.account = val;
    }

    /**
     * The meaningless, unique ID for this association. Leave Id null for a new record. The EntityManager will assign a unique Id when it is persisted. There may be 
     * more than one AccountUser association for a single combination of Account and PKBPerson. Typically only one is valid at any given point in time.
     */
    public long getId() {
        return id;
    }

    public void setId(long val) {
        this.id = val;
    }

    public Date getEffectiveDate() {
        return effectiveDate;
    }

    public void setEffectiveDate(Date val) {
        this.effectiveDate = val;
    }

    public Date getExpirationDate() {
        return expirationDate;
    }

    public void setExpirationDate(Date val) {
        this.expirationDate = val;
    }

    @Override public boolean equals(Object obj) {
        if (!(obj instanceof AccountUser)) {
            return false;
        }
        if (this.getId()==((AccountUser)obj).getId()) {
            return true;
        }
        return false;
    }

    @Override public String toString() {
        return "AccountUser: " + getId();
    }

    @Override public int hashCode() {
        if (getId()==0) {
            throw new IllegalStateException( "id not yet established in Account object");
        }
        return Long.valueOf( getId()).hashCode();
    }

    /**
     * @deprecated Use {@link com.pkb.user.entity.PKBPerson#defaultAccountId} instead. 
     */
    @Deprecated
    public boolean isDefaultAccount() {
        return defaultAccount;
    }

    public void setDefaultAccount(boolean defaultAccount) {
        this.defaultAccount = defaultAccount;
    }

    /**
     * The status of this account-user relationship.
     */
    public AccountUserStatus getStatus() {
        return status;
    }

    public void setStatus(AccountUserStatus status) {
        this.status = status;
    }

    public AccountPrivateKey getAccountPrivateKey() {
        return accountPrivateKey;
    }

    public void setAccountPrivateKey(AccountPrivateKey privateKey) {
        this.accountPrivateKey = privateKey;
    }

    public boolean hasAccountPrivateKey() {
        return accountPrivateKey != null;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }
}
