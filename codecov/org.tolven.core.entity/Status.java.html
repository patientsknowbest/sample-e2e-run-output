package org.tolven.core.entity;

/**
 * NEW_LOGIN is assigned with a LoginModule creates a valid user who previously did not exist
 * @author John Churin
 *
 */
public enum Status {
		NEW("new"),
		NULLIFIED("nullified"),    // seems unused
		OBSOLETE("obsolete"),
		ACTIVE("active"),
		OLD_ACTIVE("ACTIVE"),      // seems unused
		ACTIVATED("activated"),    // seems unused
		ACTIVATING("ACTIVATING"),  // seems unused
		COMPLETED("completed"),
		INACTIVE("inactive"),
		OLD_INACTIVE("INACTIVE"),  // seems unused
		REGISTERED("registered"),  // seems unused
		USED("USED");              // seems unused

        // We can remove statuses not referenced in code but may need to clean the DB first
        // E.g., see references in PKBInvitationBean
		
		private final String value;

	    Status(String v) {
	        value = v;
	    }

	    public String value() {
	        return value;
	    }

	    public static Status fromValue(String v) {
	        for (Status c: Status.values()) {
	            if (c.value.equals(v)) {
	                return c;
	            }
	        }
	        throw new IllegalArgumentException(v);
	    }

}
