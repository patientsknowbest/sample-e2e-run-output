package com.pkb.service.encounter.impl.tolven;

import com.pkb.encounter.entity.Message;
import com.pkb.user.entity.PKBPerson;

import java.util.List;
import java.util.Map;
import java.util.UUID;

public class MessageThreadDTO {

    private PKBPerson receiver;

    private PKBPerson composer;

    private PKBPerson referredPatient;

    private Message message;

    private Long senderMessageId;

    private Map<String, PKBPerson> personMap;

    private List<String> skipParticipants;

    private UUID existingMessageUniqueId;

    public PKBPerson getReceiver() {
        return receiver;
    }

    public void setReceiver(PKBPerson receiver) {
        this.receiver = receiver;
    }

    public PKBPerson getComposer() {
        return composer;
    }

    public void setComposer(PKBPerson composer) {
        this.composer = composer;
    }

    public PKBPerson getReferredPatient() {
        return referredPatient;
    }

    public void setReferredPatient(PKBPerson referredPatient) {
        this.referredPatient = referredPatient;
    }

    public Message getMessage() {
        return message;
    }

    /**
     * We don't want to work with the orig mutable message object on a new thread... so this method clones it
     * @param message
     */
    public void setMessageViaClone(Message message) {
        this.message = (Message)message.clone();
    }

    public Long getSenderMessageId() {
        return senderMessageId;
    }

    public void setSenderMessageId(Long senderMessageId) {
        this.senderMessageId = senderMessageId;
    }

    public Map<String, PKBPerson> getPersonMap() {
        return personMap;
    }

    public void setPersonMap(Map<String, PKBPerson> personMap) {
        this.personMap = personMap;
    }

    public List<String> getSkipParticipants() {
        return skipParticipants;
    }

    public void setSkipParticipants(List<String> skipParticipants) {
        this.skipParticipants = skipParticipants;
    }

    public UUID getExistingMessageUniqueId() {
        return existingMessageUniqueId;
    }

    public void setExistingMessageUniqueId(UUID existingMessageUniqueId) {
        this.existingMessageUniqueId = existingMessageUniqueId;
    }

}
