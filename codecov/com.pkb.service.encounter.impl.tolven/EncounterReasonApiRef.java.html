package com.pkb.service.encounter.impl.tolven;

import com.pkb.code.entity.CodeSystemName;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

/**
 * Map of apiRef string to <codeSystem, code> for reason for Encounter
 *
 * @author pravina
 *
 */
public enum EncounterReasonApiRef {

    AdmissionToIntensiveCareUnit("admissionToIntensiveCareUnit", "305351004", CodeSystemName.SNOMED.toString(), ""),
    EmergencyRoomAdmission("emergencyRoomAdmission", "50849002", CodeSystemName.SNOMED.toString(), ""),
    AntenatalAdmission("antenatalAdmission", "203221000000100", CodeSystemName.SNOMED.toString(), ""),
    AntenatalBooking("antenatalBooking", "169711001", CodeSystemName.SNOMED.toString(), ""),
    AntenatalTelephoneCall("antenatalTelephoneCall", "386479004", CodeSystemName.SNOMED.toString(), ""),
    AntenatalSecondAssessment("antenatalSecondAssessment", "antenatalSecondAssessment", CodeSystemName.NONE.toString(), ""),
    AntenatalDayAssessment("antenatalDayAssessment", "18114009", CodeSystemName.SNOMED.toString(), ""),
    AntenatalOtherContact("antenatalOtherContact", "antenatalOtherContact", CodeSystemName.NONE.toString(), "");

    private static Map<String, EncounterReasonApiRef> apiRefMap = new HashMap<>();

    // assume for now 1 to 1 mapping
    private static Map<String, String> codeToApiRefMap = new HashMap<>();

    private String apiRef;
    private String code;
    private String codeSystemName;
    private String codeSystemVersion;

    EncounterReasonApiRef(String apiRef, String code, String codeSystemName, String codeSystemVersion) {
        this.apiRef = apiRef;
        this.code = code;
        this.codeSystemName = codeSystemName;
        this.codeSystemVersion = codeSystemVersion;
    }

    static {
        // mapping of apirefs to code details
        EncounterReasonApiRef[] encounterReasonApiRefs = EncounterReasonApiRef.values();
        for (EncounterReasonApiRef enReasonApiRef : encounterReasonApiRefs) {
            EncounterReasonApiRef apiRef = apiRefMap.get(enReasonApiRef.getApiRef());
            if (apiRef == null) {
                apiRefMap.put(enReasonApiRef.getApiRef(), enReasonApiRef);
                codeToApiRefMap.put(enReasonApiRef.getCode(), enReasonApiRef.getApiRef());
            }
        }

    }

    public static Map<String, EncounterReasonApiRef> getApiRefMap() {
        return apiRefMap;
    }

    public static Map<String, String> getCodeToApiRefMap() {
        return codeToApiRefMap;
    }

    public static void setCodeToApiRefMap(Map<String, String> codeToApiRefMap) {
        EncounterReasonApiRef.codeToApiRefMap = codeToApiRefMap;
    }

    public String getApiRef() {
        return apiRef;
    }

    public String getCode() {
        return code;
    }

    public void setCode(String code) {
        this.code = code;
    }

    public String getCodeSystemName() {
        return codeSystemName;
    }

    public void setCodeSystemName(String codeSystemName) {
        this.codeSystemName = codeSystemName;
    }

    public String getCodeSystemVersion() {
        return codeSystemVersion;
    }

    public void setCodeSystemVersion(String codeSystemVersion) {
        this.codeSystemVersion = codeSystemVersion;
    }

    public static String getCodeByApiRef(String apiRefStr) {
        String apiRef = null;
        EncounterReasonApiRef encounterReasonApiRef = apiRefMap.get(apiRefStr);
        if (encounterReasonApiRef != null) {
            apiRef = encounterReasonApiRef.getCode();
        }
        return apiRef;
    }

    public static List<String> getCodesByApiRef(List<String> apiRefList) {
        List<String> codes = new ArrayList<>();
        for (String apiRefStr : apiRefList) {
            EncounterReasonApiRef apiRef = apiRefMap.get(apiRefStr);
            if (apiRef != null) {
                codes.add(apiRef.getCode());
            }
        }
        return codes;
    }

    public static String getApiRefByCode(String code) {
        return codeToApiRefMap.get(code);
    }

    public static EncounterReasonApiRef getByApiRef(String apiRef) {
        return apiRefMap.get(apiRef);
    }
}
