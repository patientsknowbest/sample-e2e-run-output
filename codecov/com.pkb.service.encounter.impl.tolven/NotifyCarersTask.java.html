package com.pkb.service.encounter.impl.tolven;

import com.google.common.collect.ImmutableMap;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.datamodel.consent.PatientConsentDTO;
import com.pkb.encounter.entity.Message;
import com.pkb.entities.enums.AccessingEntityType;
import com.pkb.exception.PKBException;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.user.entity.PKBPerson;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.EnumSet;
import java.util.Optional;

// Background job to notify carers when patient has sent a message

public class NotifyCarersTask implements Runnable {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final PKBEmailMessageManager pkbEmailMessageManager;

    private final PatientConsentManager patientConsentManager;

    private final PKBPerson patient;

    private final PKBPerson sender;
    private final EHRRequestContext context;

    private final Message message;

    NotifyCarersTask(
            PKBEmailMessageManager pkbEmailMessageManager,
            PatientConsentManager patientConsentManager,
            PKBPerson sender,
            PKBPerson patient,
            Message message,
            EHRRequestContext context) {
        this.pkbEmailMessageManager = pkbEmailMessageManager;
        this.patientConsentManager = patientConsentManager;
        this.patient = patient;
        this.message = message;
        this.sender = sender;
        this.context = context;
    }

    @Override
    public void run() {
        try {
            var consentLinks = patientConsentManager.getConsentLinks(context, patient.getId(), false, EnumSet.of(AccessingEntityType.CARER));
            consentLinks.getCarers().stream()
                    .filter(carer -> !message.getParticipants().contains(carer.getIdString()))
                    .filter(carer -> carerCanSeeMessage(consentLinks.getCarerConsents(), carer.getId(), message))
                    .forEach(carer -> pkbEmailMessageManager.sendNotificationToCarerOfSentMessage(sender, patient, carer,
                            message.getEncounterUniqueId()));
        } catch (PKBException e) {
            LOGGER.error("Exception while notifying carers of patient-{} of message on conversation-{}",
                    patient.getId(), message.getEncounterUniqueId(), e);
        }
    }

    private boolean carerCanSeeMessage(ImmutableMap<Long, PatientConsentDTO> carerConsentMap, long carerId, Message message) {
        return Optional.ofNullable(carerConsentMap.get(carerId)).map(consent -> consent.isEntityVisible(message)).orElse(false);
    }
}
