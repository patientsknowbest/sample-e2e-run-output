package com.pkb.service.encounter.impl.tolven;

import com.pkb.document.entity.DocumentsByIdAndAccount;
import com.pkb.encounter.entity.Message;

import java.util.List;
import java.util.stream.Collectors;

/**
 * A DTO to carry information for thread which will add a participant to a conversation
 *
 * @author pravinam
 *
 */
public class AddParticipantThreadDTO extends MessageThreadDTO {

    private Long inviterId;

    private Long inviterAccountId;

    private Long newUserId;

    private Long newUserAccountId;

    private List<Message> messageList;

    private DocumentsByIdAndAccount documentIdMap;

    public Long getInviterId() {
        return inviterId;
    }

    public void setInviterId(Long inviterId) {
        this.inviterId = inviterId;
    }

    public Long getInviterAccountId() {
        return inviterAccountId;
    }

    public void setInviterAccountId(Long inviterAccountId) {
        this.inviterAccountId = inviterAccountId;
    }

    public Long getNewUserId() {
        return newUserId;
    }

    public void setNewUserId(Long newUserId) {
        this.newUserId = newUserId;
    }

    public Long getNewUserAccountId() {
        return newUserAccountId;
    }

    public void setNewUserAccountId(Long newUserAccountId) {
        this.newUserAccountId = newUserAccountId;
    }

    public List<Message> getMessageList() {
        return messageList;
    }

    /**
     * We don't want to work with the orig mutable message objects on a new thread... so this method clones them
     * @param messageList
     */
    public void setMessageListViaClone(List<Message> messageList) {
        if (messageList == null) {
            this.messageList = null;
            return;
        }
        this.messageList = messageList.stream()
                .map(msg -> (Message)msg.clone())
                .collect(Collectors.toList());
    }

    public DocumentsByIdAndAccount getDocumentIdMap() {
        return documentIdMap;
    }

    public void setDocumentIdMap(DocumentsByIdAndAccount documentIdMap) {
        this.documentIdMap = documentIdMap;
    }
}
