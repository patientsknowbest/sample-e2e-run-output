package com.pkb.service.encounter.impl.tolven;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.document.entity.DocumentsByIdAndAccount;
import com.pkb.encounter.entity.Message;
import com.pkb.exception.PKBException;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.user.entity.PKBPerson;
import io.micrometer.core.instrument.Metrics;
import io.micrometer.core.instrument.Timer;
import io.prometheus.client.Counter;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.List;
import java.util.UUID;

/**
 * This class represents the task of copying the conversation and
 * saving join message to a newly added participant. It also notifies the participant
 *
 * @author pravinam
 */
public class AddParticipantTask implements Runnable {

    private static final String NOTIFICATION_TIMER_NAME = "phr_add_participant_notification_timer";

    private static final Timer NOTIFICATION_TIMER = Timer.builder(NOTIFICATION_TIMER_NAME)
            .description("Latencies of sending notifications to new recipients of a conversation")
            .publishPercentileHistogram()
            .publishPercentiles(0.5, 0.9, 0.95, 0.99)
            .register(Metrics.globalRegistry);

    private EncounterManager encounterManager;

    private PKBEmailMessageManager pkbEmailMessageManager;

    private AddParticipantThreadDTO addParticipantThreadDTO;

    private final LoggedInEHRRequestContext requestContext;

    protected static final Counter TASKS_FINISHED = Counter.build()
            .name("pkb_phr_add_participant_tasks")
            .labelNames("Success")
            .help("Total count of participants added asynchronously to conversations")
            .register();

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    AddParticipantTask(LoggedInEHRRequestContext requestContext,
                       EncounterManager encounterManager,
                       PKBEmailMessageManager pkbEmailMessageManager,
                       AddParticipantThreadDTO addParticipantThreadDTO) {
        this.encounterManager = encounterManager;
        this.addParticipantThreadDTO = addParticipantThreadDTO;
        this.requestContext = requestContext;
        this.pkbEmailMessageManager = pkbEmailMessageManager;
    }

    @Override
    public void run() {
        try {
            copyConversation();
            LOGGER.info("Adding participant-{} complete", addParticipantThreadDTO.getNewUserId());
            TASKS_FINISHED.labels("true").inc();
        } catch (Exception e) {
            LOGGER.error("PHRZ-667 caught exception when adding participant for message unique ID {}",
                    addParticipantThreadDTO.getExistingMessageUniqueId(), e);
            TASKS_FINISHED.labels("false").inc();
            throw e;
        }
    }

    private void copyConversation() {

        UUID uniqueId = addParticipantThreadDTO.getExistingMessageUniqueId();

        LOGGER.info("Entered copyConversation for existing message unique ID {} ", uniqueId);

        Long inviterId = addParticipantThreadDTO.getInviterId();
        Long newUserId = addParticipantThreadDTO.getNewUserId();
        Long newUserAccountId = addParticipantThreadDTO.getNewUserAccountId();
        List<Message> messages = addParticipantThreadDTO.getMessageList();
        DocumentsByIdAndAccount documentIdMap = addParticipantThreadDTO.getDocumentIdMap();

        encounterManager.transactional(() -> {
            //delete if any previous messages in the conversation
            //could be the case if the person is rejoining the conversation
            encounterManager.deleteConversation(requestContext, newUserId, messages.get(0).getEncounterUniqueId());
            encounterManager.copyConversationToNewParticipant(requestContext.withoutAccessLog(), inviterId, addParticipantThreadDTO.getInviterAccountId(), newUserId, newUserAccountId, messages, documentIdMap);
        });
        Metrics.timer(NOTIFICATION_TIMER_NAME).record(() -> sendNotification());

        LOGGER.info("Finished copyConversation for existing message unique ID {} ", uniqueId);
    }

    private void sendNotification() {
        PKBPerson recipient = addParticipantThreadDTO.getReceiver();
        PKBPerson patientReferredTo = addParticipantThreadDTO.getReferredPatient();
        PKBPerson composer = addParticipantThreadDTO.getComposer();
        UUID conversationId = addParticipantThreadDTO.getMessage().getEncounterUniqueId();
        try {
            if (!recipient.getId().equals(composer.getId())) {
                // if sender is not the new participant then send notification
                if (recipient.isPro()
                        && (patientReferredTo != null)
                        && (!composer.getId().equals(patientReferredTo.getId()))) {
                    pkbEmailMessageManager.notifyReceiverOnlyAboutPatient(composer, recipient,
                            patientReferredTo, conversationId);
                } else {
                    pkbEmailMessageManager.notifyReceiverOnly(composer, recipient,
                            conversationId);
                }
            }
        } catch (PKBException e) {
            LOGGER.error("Exception while sending notification to receiver-{}", recipient.getId(), e);
        }
    }
}
