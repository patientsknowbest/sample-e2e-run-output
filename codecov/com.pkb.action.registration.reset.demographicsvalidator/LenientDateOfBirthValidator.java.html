package com.pkb.action.registration.reset.demographicsvalidator;

import com.pkb.action.registration.reset.UserVerificationCriteria;
import com.pkb.action.registration.reset.UserVerificationFailure;
import com.pkb.user.entity.PKBPerson;
import io.vavr.control.Validation;

import java.time.LocalDate;

import static com.pkb.action.registration.reset.ResetStepFailure.unexpectedError;
import static io.vavr.control.Validation.invalid;
import static io.vavr.control.Validation.valid;

public final class LenientDateOfBirthValidator implements DemographicsValidator {
    public static final String PROPERTY = "dateOfBirth";
    private static final Validation<UserVerificationFailure, MatchResult> SKIPPED = valid(dateOfBirth().skipped(true).build());

    private static ImmutableMatchResult.Builder dateOfBirth() {
        return ImmutableMatchResult.builder().property(PROPERTY);
    }

    @Override
    public Validation<UserVerificationFailure, MatchResult> apply(UserVerificationCriteria criteria, PKBPerson candidate) {
        try {
            return criteria.dateOfBirth()
                    .map(dateOfBirth -> validate(candidate, dateOfBirth))
                    .orElse(SKIPPED);
        } catch (Exception cause) {
            return invalid(unexpectedError(PROPERTY, cause));
        }
    }

    private Validation<UserVerificationFailure, MatchResult> validate(PKBPerson candidate, LocalDate dateOfBirth) {
        return valid(dateOfBirth()
                .matched(candidate.findDateOfBirth()
                        .map(dob -> dob.equals(dateOfBirth))
                        .orElseThrow(() -> new IllegalArgumentException("Trying to verify a missing date of birth")))
                .build());
    }
}
