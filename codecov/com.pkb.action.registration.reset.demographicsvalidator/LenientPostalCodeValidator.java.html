package com.pkb.action.registration.reset.demographicsvalidator;

import com.pkb.action.registration.reset.UserVerificationCriteria;
import com.pkb.action.registration.reset.UserVerificationFailure;
import com.pkb.user.entity.PKBPerson;
import io.vavr.control.Validation;

import java.util.function.Function;
import java.util.regex.Pattern;

import static com.google.common.base.Preconditions.checkArgument;
import static com.pkb.action.registration.reset.ResetStepFailure.unexpectedError;
import static io.vavr.control.Validation.invalid;
import static io.vavr.control.Validation.valid;
import static java.util.regex.Matcher.quoteReplacement;
import static org.apache.commons.lang3.StringUtils.isNotBlank;



public final class LenientPostalCodeValidator implements DemographicsValidator {
    public static final String PROPERTY = "postalCode";

    private static ImmutableMatchResult.Builder postalCode() {
        return ImmutableMatchResult.builder().property(PROPERTY);
    }

    private static final Validation<UserVerificationFailure, MatchResult> SKIPPED = valid(postalCode().skipped(true).build());

    private static final Pattern SPACE_PATTERN = Pattern.compile(" ", Pattern.LITERAL);
    private static Function<String, String> POSTAL_CODE_NORMALIZER = postalCode -> SPACE_PATTERN.matcher(postalCode).replaceAll(quoteReplacement(""));

    @Override
    public Validation<UserVerificationFailure, MatchResult> apply(UserVerificationCriteria criteria, PKBPerson candidate) {
        try {
            return criteria.postalCode()
                    .map(postalCode -> validate(candidate, postalCode))
                    .orElse(SKIPPED);

        } catch (Exception cause) {
            return invalid(unexpectedError(PROPERTY, cause));
        }
    }

    private Validation<UserVerificationFailure, MatchResult> validate(PKBPerson candidate, String postalCode) {
        checkArgument(isNotBlank(candidate.getPostalCode()), "Postal code cannot be verified because it is blank");
        return valid(postalCode().matched(normalize(postalCode).equalsIgnoreCase(normalize(candidate.getPostalCode()))).build());
    }

    private String normalize(String postalCode) {
        return POSTAL_CODE_NORMALIZER.apply(postalCode);
    }
}
