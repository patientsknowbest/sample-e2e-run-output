package com.pkb.action.registration.reset.demographicsvalidator;

import com.pkb.action.registration.reset.UserVerificationCriteria;
import com.pkb.action.registration.reset.UserVerificationFailure;
import com.pkb.user.entity.PKBPerson;
import io.vavr.control.Validation;
import org.immutables.value.Value.Default;
import org.immutables.value.Value.Immutable;

import java.util.function.BiFunction;

@FunctionalInterface
public interface DemographicsValidator extends BiFunction<UserVerificationCriteria, PKBPerson, Validation<UserVerificationFailure, DemographicsValidator.MatchResult>> {
    default Validation<UserVerificationFailure, MatchResult> validate(UserVerificationCriteria criteria, PKBPerson candidate) {
        return apply(criteria, candidate);
    }

    @Immutable
    interface MatchResult {
        String property();

        /**
         * If value provided, then it'll be either {@code true} or {@code false}.
         * When value is not provided, then it'll be either the value set for skipped or default {@code false}.
         */
        @Default
        default boolean matched() {
            return skipped();
        }

        @Default
        default boolean skipped() {
            return false;
        }
    }
}
