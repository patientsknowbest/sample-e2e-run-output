package com.pkb.integration.socialcare.model;

import static org.apache.commons.collections4.CollectionUtils.isNotEmpty;

import java.util.List;

import com.fasterxml.jackson.dataformat.xml.annotation.JacksonXmlElementWrapper;

public class SocialCareRecord {

    private List<AddressDetail> addressDetails;
    private List<AssessmentDetail> assessmentDetails;
    @JacksonXmlElementWrapper(localName = "CarePackage")
    private List<CarePackage> carePackages;
    private List<CasenoteDetail> casenoteDetails;
    private List<CurrentCommunicationNeed> currentCommunicationNeeds;
    private List<ContactDetails> contactDetails;
    private List<DemographicsAndRelationship> demographicsAndRelationships;
    private List<GpDetail> gpDetails;
    private List<SafeguardingDetail> safeguardingDetails;
    private List<ServiceUserDetail> serviceUserDetails;
    private List<SocialWorkerDetail> socialWorkerDetails;

    public boolean hasData() {
        return isNotEmpty(assessmentDetails)
                || isNotEmpty(carePackages)
                || isNotEmpty(casenoteDetails)
                || isNotEmpty(currentCommunicationNeeds)
                || isNotEmpty(socialWorkerDetails);
    }

    public List<AddressDetail> getAddressDetails() {
        return addressDetails;
    }

    public void setAddressDetails(List<AddressDetail> addressDetails) {
        this.addressDetails = addressDetails;
    }

    public List<AssessmentDetail> getAssessmentDetails() {
        return assessmentDetails;
    }

    public void setAssessmentDetails(List<AssessmentDetail> assessmentDetails) {
        this.assessmentDetails = assessmentDetails;
    }

    public List<CarePackage> getCarePackages() {
        return carePackages;
    }

    public void setCarePackages(List<CarePackage> carePackages) {
        this.carePackages = carePackages;
    }

    public List<CasenoteDetail> getCasenoteDetails() {
        return casenoteDetails;
    }

    public void setCasenoteDetails(List<CasenoteDetail> casenoteDetails) {
        this.casenoteDetails = casenoteDetails;

        this.currentCommunicationNeeds = currentCommunicationNeeds;

        return contactDetails;

    public void setContactDetails(List<ContactDetails> contactDetails) {
        this.contactDetails = contactDetails;
    }

    public List<DemographicsAndRelationship> getDemographicsAndRelationships() {
        return demographicsAndRelationships;
    }

    public void setDemographicsAndRelationships(List<DemographicsAndRelationship> demographicsAndRelationships) {
        this.demographicsAndRelationships = demographicsAndRelationships;

    public List<GpDetail> getGpDetails() {
        return gpDetails;

        this.gpDetails = gpDetails;

        this.safeguardingDetails = safeguardingDetails;

        return serviceUserDetails;

        this.serviceUserDetails = serviceUserDetails;
    }

    public List<SocialWorkerDetail> getSocialWorkerDetails() {
        return socialWorkerDetails;

    public void setSocialWorkerDetails(List<SocialWorkerDetail> socialWorkerDetails) {
        this.socialWorkerDetails = socialWorkerDetails;
    }
}
