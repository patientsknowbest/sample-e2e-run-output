package com.pkb.integration.socialcare.model;

import static org.apache.commons.collections4.CollectionUtils.isNotEmpty;

public class SocialCareData {

    private EndpointInfo endpoint;
    private ServelecData servelecData;
    private ErrorDetail error;

    public boolean hasAssessmentDetails() {
        return hasSocialCareRecord() && isNotEmpty(servelecData.getSocialCareRecord().getAssessmentDetails());
    }

    public boolean hasCarePackages() {
        return hasSocialCareRecord() && isNotEmpty(servelecData.getSocialCareRecord().getCarePackages());
    }
    public boolean hasCasenoteDetails() {
        return hasSocialCareRecord() && isNotEmpty(servelecData.getSocialCareRecord().getCasenoteDetails());
    }
    public boolean hasCurrentCommunicationNeeds() {
        return hasSocialCareRecord() && isNotEmpty(servelecData.getSocialCareRecord().getCurrentCommunicationNeeds());
    }
    public boolean hasSocialWorkerDetails() {
        return hasSocialCareRecord() && isNotEmpty(servelecData.getSocialCareRecord().getSocialWorkerDetails());
    }

    public boolean hasData() {
        return !hasError() && hasSocialCareRecord() && servelecData.getSocialCareRecord().hasData();
    }

    public boolean hasError() {
        return error != null;
    }

    public EndpointInfo getEndpoint() {
        return endpoint;
    }

    public void setEndpoint(EndpointInfo endpoint) {
        this.endpoint = endpoint;
    }

    public ServelecData getServelecData() {
        return servelecData;
    }

    public void setServelecData(ServelecData servelecData) {
        this.servelecData = servelecData;
    }

    public ErrorDetail getError() {
        return error;
    }

    public void setError(ErrorDetail error) {
        this.error = error;
    }

    private boolean hasSocialCareRecord() {
        return servelecData.getSocialCareRecord() != null;
    }
}
