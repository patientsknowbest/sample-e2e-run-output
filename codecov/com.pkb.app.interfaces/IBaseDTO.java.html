package com.pkb.app.interfaces;

import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;
import com.pkb.datamodel.consent.RequiresConsent;
import com.pkb.entities.enums.PrivacyFlag;
import org.jetbrains.annotations.NotNull;

import java.io.Serializable;
import java.util.UUID;

/**
 * This interface defines methods to fetch information like source details , Entered date , persisted date etc
 *
 * @author pravina
 *
 */
public interface IBaseDTO extends Serializable, RequiresConsent {

    String ENTERED_DATE = "enteredDate";

    String PERSISTED_DATE = "persistedDate";

    String UPLOADED_DATA_ID = "uploadedDataId";
    String UNIQUE_ID = "uniqueId";

    Long getId();

    void setId(Long id);

    SourceDetails getSource();

    // don't use this -- instead never let getSource() return null
    // public void setSource(SourceDetails source);

    DTOBaseFields getBaseFields();

    // don't use this -- instead never let getBaseFields() return null
    // public void setBaseFields(DTOBaseFields baseFields);

    @Override default boolean getRequireGeneral(){ return getBaseFields().isGeneralHealthConsentRequired();}

    @Override default void setRequireGeneral(boolean val){getBaseFields().setGeneralHealthConsentRequired(val);}

    @Override default boolean getRequireSexualHealth(){return getBaseFields().isSexualHealthConsentRequired();}

    @Override default void setRequireSexualHealth(boolean val){ getBaseFields().setSexualHealthConsentRequired(val);}

    @Override default boolean getRequireMentalHealth(){return getBaseFields().isMentalHealthConsentRequired();}

    @Override default void setRequireMentalHealth(boolean val){getBaseFields().setMentalHealthConsentRequired(val);}

    @Override default boolean getRequireSocialCare(){return getBaseFields().isSocialCareConsentRequired();}

    @Override default void setRequireSocialCare(boolean val){getBaseFields().setSocialCareConsentRequired(val);}

    default void setPrivacyFlagsFrom(@NotNull IBaseDTO source){
        setRequireGeneral(source.getRequireGeneral());
        setRequireMentalHealth(source.getRequireMentalHealth());
        setRequireSexualHealth(source.getRequireSexualHealth());
        setRequireSocialCare(source.getRequireSocialCare());
    }

    default DTOBaseFields setPrivacyFlag(@NotNull PrivacyFlag requiredFlag){
        return getBaseFields().setPrivacyFlag(requiredFlag);
    }

    default UUID generateNewRandomUniqueId() {
        return getBaseFields().generateNewRandomUniqueId();
    }
}
