package com.pkb.service.coding;

import com.fasterxml.jackson.databind.util.StdConverter;
import org.apache.commons.lang3.StringUtils;

public class EnforcedNonEmptyNumericSnomedIdentityConverter extends StdConverter<String, String> {

    @Override
    public String convert(String input) {
        if (input == null) {
            throw new IllegalStateException("received null string for snomed code");
        }

        String converted = input.trim();
        if (converted.isEmpty()) {
            throw new IllegalStateException("Must provide non empty value for snomed code");
        }

        if (!StringUtils.isNumeric(converted)) {
            throw new IllegalStateException("snomed code contains unexpected characters (" + input + "), must be numeric");
        }

        return converted;
    }
}
