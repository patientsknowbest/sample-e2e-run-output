package com.pkb.service.coding;

import com.fasterxml.jackson.databind.util.StdConverter;
import com.pkb.entities.enums.PrivacyFlag;

import java.util.Map;
import java.util.Set;
import java.util.stream.Collectors;

public class EnforcedNonEmptyStringToPrivacyFlagConverter extends StdConverter<String, PrivacyFlag> {

    private static Map<Set<String>, PrivacyFlag> MAPPING_RULES = Map.of(
            Set.of("G", "GE", "GENERAL_HEALTH"), PrivacyFlag.GENERAL,
            Set.of("SEXUAL", "SE"), PrivacyFlag.SEXUAL_HEALTH,
            Set.of("MENTAL", "ME", "M"), PrivacyFlag.MENTAL_HEALTH,
            Set.of("SOCIAL", "SO"), PrivacyFlag.SOCIAL_CARE
    );

    private static Map<String, PrivacyFlag> ALLOWED_MAPPINGS_FROM_TEXT = MAPPING_RULES.entrySet().stream()
            .flatMap(entry -> entry.getKey().stream().map(text -> Map.entry(text, entry.getValue())))
            .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue));

    @Override
    public PrivacyFlag convert(String input) {
        if (input == null) {
            throw new IllegalStateException("received null string for privacy flag");
        }

        String pfString = input.toUpperCase().trim();
        if (pfString.isEmpty()) {
            throw new IllegalStateException("Must provide a non empty value for privacy flag");
        }
        //be kind to product, let them be slightly lazy before we try to map to a standard value
        PrivacyFlag privacyFlag = ALLOWED_MAPPINGS_FROM_TEXT.get(pfString);
        if (privacyFlag == null) {
            privacyFlag = PrivacyFlag.valueOf(pfString);
        }
        return privacyFlag;
    }
}
