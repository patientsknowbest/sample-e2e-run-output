package com.pkb.service.coding;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.repository.CodingMatchRepository;
import com.pkb.service.dataupload.emis.esv2.EmisJobArtifactService;
import com.pkb.util.CsvStreamService;
import io.vavr.control.Either;
import io.vavr.control.Try;
import org.springframework.transaction.annotation.Transactional;

import javax.persistence.EntityManager;
import javax.persistence.PersistenceContext;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.io.Writer;
import java.util.List;
import java.util.Map;
import java.util.function.Function;
import java.util.stream.Stream;

public class CodingMatchExportService {
    private final static Map<String, Function<CodingMatchRepository.CodingMatchAndTranslation, Object>> CODING_MATCH_COLUMN_VALUE_FUNCTIONS = Map.of(
            "id", (cmt) -> cmt.getCodingMatch().getId(),
            "coding_data_type", (cmt) -> cmt.getCodingMatch().getCodingDataType(),
            "code", (cmt) -> cmt.getCodingMatch().getCode(),
            "code_system", (cmt) -> cmt.getCodingMatch().getCodeSystem(),
            "display_text", (cmt) -> cmt.getCodingMatch().getDisplayText(),
            "privacy_flag", (cmt) -> cmt.getCodingMatch().getPrivacyFlag(),
            "created", (cmt) -> cmt.getCodingMatch().getVersionDetails().getCreated(),
            "exclude", (cmt) -> cmt.getCodingMatch().isExclude(),
            "sourcepersonid", (cmt) -> cmt.getCodingMatch().getVersionDetails().getSourcePersonId(),
            "target_code_translation", CodingMatchRepository.CodingMatchAndTranslation::getTargetCodeTranslation
            );
    private static final List<String> ORDERED_COLUMNS_TO_EXPORT = List.of("id", "coding_data_type", "code", "code_system", "display_text", "privacy_flag", "created", "exclude", "sourcepersonid", "target_code_translation");
    private static final String ALL_MATCHES_PREFIX = "AllMatches_";
    private static final String UNMATCHED_CODES_PREFIX = "UnmatchedCodes_";

    private CodingMatchRepository codingMatchRepository;
    private EmisJobArtifactService emisJobArtifactService;
    private CsvStreamService csvStreamService;
    private DateTimeService dateTimeService;

    @PersistenceContext
    EntityManager em;

    public CodingMatchExportService(CodingMatchRepository codingMatchRepository, EmisJobArtifactService emisJobArtifactService, CsvStreamService csvStreamService, DateTimeService dateTimeService) {
        this.codingMatchRepository = codingMatchRepository;
        this.csvStreamService = csvStreamService;
        this.dateTimeService = dateTimeService;
        this.emisJobArtifactService = emisJobArtifactService;
    }

    public Either<Throwable, Void> streamUnmatchedCodes(HttpServletResponse response) {
        return Try.of(() -> {
            try (Writer writer = configureCsvWriter(response, UNMATCHED_CODES_PREFIX)) {
                return emisJobArtifactService.retrieveUnmatchedEmisCodesAndWriteToOutputAsCsv(writer);
            }
        }).toEither();
    }

    @Transactional(readOnly = true)
    public Either<Throwable, Void> streamAllMatches(HttpServletResponse response) {
        return Try.of(() -> {
            try (Writer writer = configureCsvWriter(response, ALL_MATCHES_PREFIX)) {
                return retrieveAllAndStreamOutputToCsv(writer);
            }
        }).toEither();
    }

    @VisibleForTesting
    Void retrieveAllAndStreamOutputToCsv(Writer outputWriter) {
        Stream<CodingMatchRepository.CodingMatchAndTranslation> codingMatchStream = codingMatchRepository.streamAll();
        csvStreamService.streamToCsvConsumer(outputWriter,
                        ORDERED_COLUMNS_TO_EXPORT,
                        CODING_MATCH_COLUMN_VALUE_FUNCTIONS,
                        true,
                        codingMatchAndTranslation -> em.detach(codingMatchAndTranslation.getCodingMatch())) // deteach the jpa entity to remove it from the cache, to effectively stream rather than holding all records in memory.
                .accept(codingMatchStream);
        return null;
    }

    private Writer configureCsvWriter(HttpServletResponse response, String prefix) throws IOException {
        response.setContentType("text/csv");
        response.setCharacterEncoding("utf-8");
        response.addHeader("Content-Disposition", "attachment;filename=\"" + prefix + dateTimeService.today().toString() + ".csv\"");
        return response.getWriter();
    }
}
