package com.pkb.service.coding;

import com.fasterxml.jackson.databind.util.StdConverter;

import java.util.Set;

public class EnforcedNonEmptyStringToBooleanConverter extends StdConverter<String, Boolean> {

    public static Set<String> ACCEPTED_TRUTHY_VALUES = Set.of("TRUE", "T", "YES", "Y");
    public static Set<String> ACCEPTED_FALSY_VALUES = Set.of("FALSE", "F", "NO", "N");
    @Override
    public Boolean convert(String input) {
        if (input == null) {
            throw new IllegalStateException("received null string for boolean");
        }

        String outputStr = input.toUpperCase().trim();
        if (outputStr.isEmpty()) {
            throw new IllegalStateException("Must provide a value for a boolean");
        }

        //be kind to product, let them be slightly lazy
        if (ACCEPTED_TRUTHY_VALUES.contains(outputStr)) {
            return true;
        }
        if (ACCEPTED_FALSY_VALUES.contains(outputStr)) {
            return false;
        }
        throw new IllegalStateException("Unexpected value for boolean: " + input);
    }
}
