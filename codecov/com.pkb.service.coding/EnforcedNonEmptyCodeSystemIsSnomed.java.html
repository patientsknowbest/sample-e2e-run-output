package com.pkb.service.coding;

import com.fasterxml.jackson.databind.util.StdConverter;
import com.pkb.service.dataupload.emis.esv2.CodeSystems;

/**
 * For now, we're only going to be adding SNOMED mappings, if we want to support others we can add in a per-system coding
 * validation mechanism, but that's not required today.
 */
public class EnforcedNonEmptyCodeSystemIsSnomed extends StdConverter<String, String> {

    @Override
    public String convert(String input) {

        if (input == null) {
            throw new IllegalStateException("received null string for coding system. Only SNOMED CT is currently supported.");
        }

        String converted = input.trim();
        if (converted.isEmpty()) {
            throw new IllegalStateException("Must provide non empty value for coding system. Only SNOMED CT is currently supported.");
        }

        if (!converted.equals(CodeSystems.SNOMED_CT)) {
            throw new IllegalStateException("Received code system (" + input + "), only SNOMED CT is currently supported");
        }

        return converted;
    }
}
