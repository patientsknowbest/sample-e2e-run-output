package com.pkb.phoenix.api.service.task.template.datagathering.patient;

import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.phoenix.api.wiring.config.PhoenixPubSubConfig;
import com.pkb.phoenix.shared.structures.request.PatientInfoTaskRequest;
import org.apache.camel.component.jackson.JacksonDataFormat;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import static com.pkb.phoenix.api.service.task.template.AbstractPubSubTaskTemplate.OUTBOUND_REQUEST_PATIENT_INFO;
import static com.pkb.phoenix.api.service.task.template.SupportToolPubSubUtil.uri;

public class DataCollectionOutboundRouteBuilder extends AbstractNamespaceAwareRouteBuilder {
    private final PubSubNamespaceService pubSubNamespaceService;
    private final PhoenixPubSubConfig config;
    private final @Nullable String serviceAccountKey;


    public DataCollectionOutboundRouteBuilder(PubSubNamespaceService pubSubNamespaceService,
                                              PhoenixPubSubConfig config,
                                              @Nullable String serviceAccountKey) {
        this.pubSubNamespaceService = pubSubNamespaceService;
        this.config = config;
        this.serviceAccountKey = serviceAccountKey;
    }

    @Override
    public @NotNull PubSubNamespaceService camelNamespaceService() {
        return pubSubNamespaceService;
    }

    @Override
    public boolean throwExceptionForNamespaceMismatch() {
        return true;
    }

    @Override
    public void configure() {
        from(OUTBOUND_REQUEST_PATIENT_INFO)
                .noStreamCaching()
                .marshal(new JacksonDataFormat(PatientInfoTaskRequest.class))
                .routeId("patientInfo")
                .to(uri(config.getPubSubProjectName(), config.taskRequestedTopicName(), serviceAccountKey));
    }
}
