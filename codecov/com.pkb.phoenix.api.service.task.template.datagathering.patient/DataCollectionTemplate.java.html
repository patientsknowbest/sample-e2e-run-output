package com.pkb.phoenix.api.service.task.template.datagathering.patient;

import com.fasterxml.jackson.databind.ObjectReader;
import com.pkb.phoenix.api.service.task.template.AbstractPubSubTaskTemplate;
import com.pkb.phoenix.shared.structures.request.PatientInfoTaskRequest;
import com.pkb.phoenix.shared.structures.response.PatientInfoTaskResponse;
import org.apache.camel.ProducerTemplate;
import org.springframework.data.redis.core.ReactiveRedisTemplate;

import java.util.ArrayList;
import java.util.List;

import static com.pkb.phoenix.api.service.task.template.TaskTemplateCode.PATIENT_DATA;
import static com.pkb.phoenix.api.service.tasktype.ValidationUtil.validateEmail;
import static com.pkb.phoenix.api.service.tasktype.ValidationUtil.validateNationalID;
import static com.pkb.phoenix.api.service.tasktype.ValidationUtil.validatePrivateId;
import static com.pkb.phoenix.shared.structures.request.PatientInfoTaskInput.InputType.EMAIL;
import static com.pkb.phoenix.shared.structures.request.PatientInfoTaskInput.InputType.NATIONAL_ID;
import static com.pkb.phoenix.shared.structures.request.PatientInfoTaskInput.InputType.PRIVATE_ID;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;

public final class DataCollectionTemplate extends AbstractPubSubTaskTemplate<DataCollectionParameters, PatientInfoTaskRequest, PatientInfoTaskResponse, PatientDetails> {
    public DataCollectionTemplate(ReactiveRedisTemplate<String, String> redis,
            ProducerTemplate producer,
            ObjectReader json) {
        super(PATIENT_DATA, DataCollectionParameters.class, new DataCollectionTranslator(), redis, producer, OUTBOUND_REQUEST_PATIENT_INFO, json);
    }

    @Override
    public List<String> validate(DataCollectionParameters parameters) {
        var errors = new ArrayList<String>();
        Match(parameters.input().type()).of(
                Case($(EMAIL.value), () -> validateEmail(parameters.input().value()).map(errors::add)),
                Case($(PRIVATE_ID.value), () -> validatePrivateId(parameters.input().value()).map(errors::add)),
                Case($(NATIONAL_ID.value), () -> validateNationalID(parameters.input().value()).map(errors::add)));
        return errors;
    }
}
