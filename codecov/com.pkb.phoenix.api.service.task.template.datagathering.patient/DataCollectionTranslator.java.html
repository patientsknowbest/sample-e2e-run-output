package com.pkb.phoenix.api.service.task.template.datagathering.patient;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.phoenix.api.service.task.template.Translator;
import com.pkb.phoenix.repository.task.projection.Task;
import com.pkb.phoenix.shared.structures.request.ImmutablePatientInfoTaskInput;
import com.pkb.phoenix.shared.structures.request.ImmutablePatientInfoTaskOutput;
import com.pkb.phoenix.shared.structures.request.ImmutablePatientInfoTaskRequest;
import com.pkb.phoenix.shared.structures.request.SupportToolNationalIdType;
import com.pkb.phoenix.shared.structures.request.PatientInfoTaskInput.InputType;
import com.pkb.phoenix.shared.structures.request.PatientInfoTaskRequest;
import com.pkb.phoenix.shared.structures.response.PatientInfoTaskResponse;
import io.vavr.MatchError;
import reactor.util.context.ContextView;

import java.util.function.Function;
import java.util.stream.Collectors;

import static com.pkb.phoenix.api.service.task.TaskService.TASK_ID;
import static com.pkb.phoenix.shared.structures.request.PatientInfoTaskInput.InputType.NATIONAL_ID;
import static com.pkb.phoenix.shared.structures.request.PatientInfoTaskInput.InputType.EMAIL;
import static com.pkb.phoenix.shared.structures.request.PatientInfoTaskInput.InputType.PRIVATE_ID;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static java.util.Optional.ofNullable;

final class DataCollectionTranslator implements Translator<DataCollectionParameters, PatientInfoTaskRequest, PatientInfoTaskResponse, PatientDetails> {
    @Override
    public Function<Task<DataCollectionParameters>, PatientInfoTaskRequest> toRequest(ContextView context) {
        return task -> {
            var parameters = task.getParameters();
            return new ImmutablePatientInfoTaskRequest.Builder()
                    .taskId(task.getId())
                    .requestId(context.get(TASK_ID).toString())
                    .input(new ImmutablePatientInfoTaskInput.Builder()
                            .type(inputType(parameters))
                            .value(parameters.input().value())
                            .nationalIdType(mapNationalIdTypeToRequest(parameters.input().nationalIdType()))
                            .build())
                    .output(new ImmutablePatientInfoTaskOutput.Builder()
                            .email(parameters.output().email())
                            .privateId(parameters.output().privateId())
                            .orgAndTeamNames(parameters.output().orgAndTeamNames())
                            .nationalIds(parameters.output().nationalIds())
                            .userType(parameters.output().userType())
                            .userStatus(parameters.output().userStatus())
                            .userRegistration(parameters.output().userRegistration())
                            .build())
                    .build();
        };
    }

    @Override
    public Function<PatientInfoTaskResponse, PatientDetails> fromResponse(ContextView context) {

        return result -> {
            if (result.results() == null) {
                //this will be the case if no patient was found.
                return ImmutablePatientDetails.builder()
                        .addResults(new ImmutablePatientDetailsResult.Builder().maybeMessageAboutStatus(result.errorMessage()).build())
                        .build();
            }

            return ImmutablePatientDetails.builder()
                    .results(result
                            .results()
                            .stream()
                            .map(actualResult -> new ImmutablePatientDetailsResult.Builder()
                                    .email(ofNullable((actualResult).email()))
                                    .privateId(ofNullable((actualResult).patientId()))
                                    .orgAndTeamNames(ofNullable((actualResult).orgAndTeamNames()))
                                    .nationalIds(ofNullable((actualResult).nationalIds()))
                                    .userType(ofNullable((actualResult).userType()))
                                    .userStatus(ofNullable((actualResult).userStatus()))
                                    .userRegistration(ofNullable((actualResult).registrationDetails()))
                                    .status(ofNullable(actualResult.status()))
                                    .maybeMessageAboutStatus(ofNullable(actualResult.maybeMessageAboutStatus()))
                                    .build())
                            .collect(Collectors.toList()))
                    .build();
        };

    }

    @Override
    public Class<PatientInfoTaskResponse> responseType() {
        return PatientInfoTaskResponse.class;
    }

    private InputType inputType(DataCollectionParameters parameters) {
        return Match(parameters.input().type()).of(
                Case($(PRIVATE_ID.value), PRIVATE_ID),
                Case($(EMAIL.value), EMAIL),
                Case($(NATIONAL_ID.value), NATIONAL_ID));
    }

    @VisibleForTesting
    SupportToolNationalIdType mapNationalIdTypeToRequest(String maybeNationalIdType) {
        try {
            return Match(maybeNationalIdType).of(
                    Case($("NHS_NUMBER"), SupportToolNationalIdType.NHS_NUMBER),
                    Case($("CHI_NUMBER"), SupportToolNationalIdType.CHI_NUMBER),
                    Case($("H_AND_C_NUMBER"), SupportToolNationalIdType.H_AND_C_NUMBER),
                    Case($("IHI_NUMBER"), SupportToolNationalIdType.IHI_NUMBER),
                    Case($("SSN_USA"), SupportToolNationalIdType.SSN_USA),
                    Case($("SIN_CA"), SupportToolNationalIdType.SIN_CA),
                    Case($("HKID_HK"), SupportToolNationalIdType.HKID_HK),
                    Case($("BSN_NL"), SupportToolNationalIdType.BSN_NL),
                    Case($("KVN_DE"), SupportToolNationalIdType.KVN_DE),
                    Case($("CIVIL_ID_KW"), SupportToolNationalIdType.CIVIL_ID_KW));
        } catch (MatchError ignore) {
            return null;
        }
    }
}
