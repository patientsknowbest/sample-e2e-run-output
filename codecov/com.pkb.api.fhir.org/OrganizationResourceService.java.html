package com.pkb.api.fhir.org;

import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import com.pkb.api.fhir.AbstractResourceService;
import com.pkb.api.fhir.exception.ServerResponseExceptionMapper;
import com.pkb.api.fhir.organization.OrganizationMapper;
import com.pkb.authentication.AuthenticatedIdentity;
import com.pkb.authz.AuthorizationDataService;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.authz.data.AuthorizationInputs;
import com.pkb.datamodel.team.TeamWithOrg;
import com.pkb.domain.TeamService;
import io.vavr.control.Either;
import org.hl7.fhir.dstu3.model.Organization;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.pkb.authz.ImmutableTargetMeta.targetMeta;
import static com.pkb.authz.action.Action.READ;
import static com.pkb.authz.context.Path.FHIR;
import static com.pkb.authz.data.ImmutableAuthorizationInputs.inputs;
import static io.vavr.API.None;
import static io.vavr.API.Set;

public class OrganizationResourceService extends AbstractResourceService<Organization> {
    private final AuthorizationDataService authorizationDataService;
    private final OrganizationMapper organizationMapper;
    private final TeamService teamService;

    public OrganizationResourceService(
            @NotNull AuthorizationDataService authorizationDataService,
            @NotNull TeamService teamService,
            @NotNull OrganizationMapper organizationMapper,
            @NotNull ServerResponseExceptionMapper exceptionMapper) {
        super(exceptionMapper);
        this.authorizationDataService = authorizationDataService;
        this.teamService = teamService;
        this.organizationMapper = organizationMapper;
    }

    public @NotNull Either<BaseServerResponseException, Organization> findOrganization(AuthenticatedIdentity authenticatedIdentity, 
                                                                                       UUID publicId) {
        AuthorizationInputs authorizationInputs = inputs()
                .authenticatedIdentity(authenticatedIdentity)
                .action(READ)
                .addTargets(targetMeta(TeamWithOrg.class, Set(publicId), None(), false))
                .path(FHIR)
                .build();

        AuthorizationData authorizationData = authorizationDataService.fetch(authorizationInputs);
        return teamService.findTeamWithOrgByPublicId(authorizationData, publicId)
                .map(organizationMapper::toOrganization)
                .mapLeft(exception -> exceptionToFhir(exception, Organization.class));
    }
}
