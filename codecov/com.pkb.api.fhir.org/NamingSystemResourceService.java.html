package com.pkb.api.fhir.org;

import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import com.pkb.api.fhir.AbstractResourceService;
import com.pkb.api.fhir.exception.ServerResponseExceptionMapper;
import com.pkb.api.fhir.namingsystem.OrgLevelIdTypeMapper;
import com.pkb.api.fhir.namingsystem.TeamLevelIdTypeMapper;
import com.pkb.authentication.AuthenticatedIdentity;
import com.pkb.authz.AuthorizationDataService;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.datamodel.OrgLevelIdType;
import com.pkb.datamodel.TeamLevelIdType;
import com.pkb.domain.OrgLevelIdTypeService;
import com.pkb.domain.TeamLevelIdTypeService;
import com.pkb.domain.criteria.FhirCriteriaResult;
import com.pkb.domain.criteria.ImmutableFhirCriteriaResult;
import io.vavr.control.Either;
import org.hl7.fhir.dstu3.model.NamingSystem;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.pkb.authz.ImmutableTargetMeta.targetMeta;
import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.context.Path.FHIR;
import static com.pkb.authz.data.ImmutableAuthorizationInputs.inputs;
import static io.vavr.API.None;
import static io.vavr.API.Set;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

public class NamingSystemResourceService extends AbstractResourceService<NamingSystem>{

    private final AuthorizationDataService authorizationDataService;

    private final TeamLevelIdTypeMapper teamLevelIdTypeMapper;
    private final TeamLevelIdTypeService teamLevelIdTypeService;

    private final OrgLevelIdTypeMapper orgLevelIdTypeMapper;
    private final OrgLevelIdTypeService orgLevelIdTypeService;

    public NamingSystemResourceService(
            AuthorizationDataService authorizationDataService,
            TeamLevelIdTypeService teamLevelIdTypeService,
            OrgLevelIdTypeService orgLevelIdTypeService,
            OrgLevelIdTypeMapper orgLevelIdTypeMapper,
            TeamLevelIdTypeMapper teamLevelIdTypeMapper,
            ServerResponseExceptionMapper exceptionMapper) {
        super(exceptionMapper);
        this.authorizationDataService = authorizationDataService;
        this.teamLevelIdTypeService = teamLevelIdTypeService;
        this.orgLevelIdTypeService = orgLevelIdTypeService;
        this.orgLevelIdTypeMapper = orgLevelIdTypeMapper;
        this.teamLevelIdTypeMapper = teamLevelIdTypeMapper;
    }

    public @NotNull Either<BaseServerResponseException, FhirCriteriaResult<NamingSystem>> searchNamingSystem(AuthenticatedIdentity authenticatedIdentity,
                                                                                                             @NotNull UUID publicId) {
        AuthorizationData authzData = authorizationDataService.fetch(inputs()
                .authenticatedIdentity(authenticatedIdentity)
                .action(SEARCH)
                .addTargets(targetMeta(TeamLevelIdType.class, Set(publicId), None(), false),
                        targetMeta(OrgLevelIdType.class, Set(publicId), None(), false))
                .path(FHIR)
                .build());

        if (authzData.inputs().authenticatedSystemUser().isDefined()) {
            return left(toSystemUserTypeNotPermitted());
        }

        FhirCriteriaResult<NamingSystem> teamLevelIdNamingSystem = searchTeamLevelIdType(authzData, publicId);

        if (teamLevelIdNamingSystem.getTotal() > 0) {
            return right(teamLevelIdNamingSystem);
        }

        FhirCriteriaResult<NamingSystem> orgLevelIdNamingSystem = searchOrgLevelIdType(authzData, publicId);

        if (orgLevelIdNamingSystem.getTotal() > 0) {
            return right(orgLevelIdNamingSystem);
        }

        return right(teamLevelIdNamingSystem);
    }

    private FhirCriteriaResult<NamingSystem> searchTeamLevelIdType(AuthorizationData authorizationData, UUID publicId) {
        return teamLevelIdTypeService.findTeamLevelIdType(authorizationData, publicId)
                .fold($ -> FhirCriteriaResult.empty(),
                        teamLevelIdType -> ImmutableFhirCriteriaResult
                                .<NamingSystem>fhirCriteriaResult()
                                .total(1)
                                .addPage(teamLevelIdTypeMapper.toNamingSystem(teamLevelIdType))
                                .build());
    }

    private FhirCriteriaResult<NamingSystem> searchOrgLevelIdType(AuthorizationData authorizationData, UUID publicId) {
        return orgLevelIdTypeService.findOrgLevelIdType(authorizationData, publicId)
                .fold($ -> FhirCriteriaResult.empty(),
                        orgLevelIdType -> ImmutableFhirCriteriaResult
                                .<NamingSystem>fhirCriteriaResult()
                                .total(1)
                                .addPage(orgLevelIdTypeMapper.toNamingSystem(orgLevelIdType))
                                .build());
    }
}
