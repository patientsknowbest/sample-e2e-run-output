package com.pkb.action.radiology;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.document.entity.Attachment;
import com.pkb.exception.PKBException;
import com.pkb.radiology.entity.RadiologyImage;
import com.pkb.service.radiology.RadiologyManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.ImageUtil;

import java.io.ByteArrayInputStream;
import java.io.InputStream;

public class DownloadRadiologyAction extends BaseAction {

    private static final long serialVersionUID = 1L;

    // Image download
    private Long radiologyImageId;

    private RadiologyImage radiologyImage;

    private String contentType;

    private int contentLength;

    private InputStream inputStream;

    // Attachment download
    private Long radiologyAttachmentId;

    private String fileName;

    // Internal
    private Long userId;

    private PKBPerson patient;

    private Long patientAccountId;

    private RadiologyManager radiologyManager;

    private void setupUsers() {
        userId = loggedInUserUtil.getLoggedInUserId();

        patient = contextUserUtil.getContextUser();
        if (patient == null) {
            patient = userManager.getPKBPerson(loggedInUserUtil.getLoggedInUserId());
        }
        patientAccountId = userManager.getDefaultAccountId(patient.getId());
    }

    public String viewFullsizeImage() {
        if (ImageUtil.isBrowserImageByMimeType(contentType)) {
            // Just pass through RadiologyImageId
            return SUCCESS;
        }

        // download directly -- not sure the browser can display it
        return "nonImage";
    }

    public String getRadiologyAttachment() {
        try {
            setupUsers();

            Attachment att = radiologyManager.getAttachmentContent(getLoggedInEHRRequestContext(),
                    radiologyAttachmentId);
            contentType = att.getMediaType();
            contentLength = att.getContent().length;
            inputStream = new ByteArrayInputStream(att.getContent());
            // TODO: refactor so this is available...
            //fileName = att.getDescription();
        } catch (Exception exception) {
            throw new PKBException("failed getting attachment "
                    + radiologyAttachmentId, exception);
        }
        return Action.SUCCESS;
    }

    public String viewImage() {
        setupUsers();

        radiologyImage = radiologyManager.getRadiologyImage(getLoggedInEHRRequestContext(),
                patientAccountId, radiologyImageId);
        contentType = radiologyImage.getContentType();
        contentLength = radiologyImage.getContent().length;
        inputStream = new ByteArrayInputStream(radiologyImage.getContent());
        fileName = radiologyImage.getDescription();

        return Action.SUCCESS;
    }

    public Long getRadiologyImageId() {
        return radiologyImageId;
    }

    public void setRadiologyImageId(Long radiologyImageId) {
        this.radiologyImageId = radiologyImageId;
    }

    public String getContentType() {
        return contentType;
    }

    public void setContentType(String contentType) {
        this.contentType = contentType;
    }

    public int getContentLength() {
        return contentLength;
    }

    public void setContentLength(int contentLength) {
        this.contentLength = contentLength;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public RadiologyManager getRadiologyManager() {
        return radiologyManager;
    }

    public void setRadiologyManager(RadiologyManager radiologyManager) {
        this.radiologyManager = radiologyManager;
    }

    public InputStream getInputStream() {
        return inputStream;
    }

    public Long getRadiologyAttachmentId() {
        return radiologyAttachmentId;
    }

    public void setRadiologyAttachmentId(Long radiologyAttachmentId) {
        this.radiologyAttachmentId = radiologyAttachmentId;
    }

    public String getFileName() {
        return fileName;
    }

    /**
     * used for a temporary hack while we still need to refactor attachment handling to give us the
     * file name along with the content
     */
    public void setFileName(String fileName) {
        this.fileName = fileName;
    }
}
