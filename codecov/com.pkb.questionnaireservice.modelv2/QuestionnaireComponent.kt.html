package com.pkb.questionnaireservice.modelv2

import com.fasterxml.jackson.annotation.JsonIgnoreProperties
import com.fasterxml.jackson.annotation.JsonSubTypes
import com.fasterxml.jackson.annotation.JsonTypeInfo
import com.pkb.questionnaireservice.QuestionnaireComponentId
import java.io.Serializable


enum class QuestionnaireComponentType(val isQuestion: Boolean, val isMultiChoice: Boolean) {
    MultiSelectQuestion(true, true),
    SingleSelectQuestion(true, true),
    OpenQuestion(true, false),
    Title(false, false)
}

@JsonTypeInfo(
        use = JsonTypeInfo.Id.NAME,
        include = JsonTypeInfo.As.EXISTING_PROPERTY,
        property = "type")
@JsonSubTypes(
        JsonSubTypes.Type(value = MultiSelectQuestion::class, name = "MultiSelectQuestion"),
        JsonSubTypes.Type(value = SingleSelectQuestion::class, name = "SingleSelectQuestion"),
        JsonSubTypes.Type(value = OpenQuestion::class, name = "OpenQuestion"),
        JsonSubTypes.Type(value = Title::class, name = "Title")
)
@JsonIgnoreProperties(ignoreUnknown = true)
abstract class QuestionnaireComponent : Serializable {

    abstract val type: QuestionnaireComponentType

    abstract val id: QuestionnaireComponentId

    abstract val displayText: String

    abstract fun <T> accept(visitor: QuestionnaireComponentVisitor<T>): List<T>

    override fun equals(other: Any?): Boolean {
        if (this === other) return true
        if (javaClass != other?.javaClass) return false

        other as QuestionnaireComponent

        if (id != other.id) return false

        return true
    }

    override fun hashCode(): Int {
        return id.hashCode()
    }

    companion object {
        private const val serialVersionUID: Long = 1
    }

}


interface QuestionnaireComponentVisitor<out T> {

    fun visit(question: MultiSelectQuestion): List<T> = emptyList()

    fun visit(question: SingleSelectQuestion): List<T> = emptyList()

    fun visit(question: OpenQuestion): List<T> = emptyList()

    fun visit(title: Title): List<T> = emptyList()

}
