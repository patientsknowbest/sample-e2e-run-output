package com.pkb.questionnaireservice.modelv2

import com.fasterxml.jackson.annotation.JsonCreator
import com.fasterxml.jackson.annotation.JsonIdentityInfo
import com.fasterxml.jackson.annotation.JsonIdentityReference
import com.fasterxml.jackson.annotation.JsonPropertyOrder
import com.fasterxml.jackson.annotation.ObjectIdGenerators
import com.fasterxml.jackson.databind.annotation.JsonDeserialize
import com.pkb.questionnaireservice.AccountPublicId
import com.pkb.questionnaireservice.PersonPublicId
import com.pkb.questionnaireservice.QuestionnaireResponseId
import io.vavr.control.Either
import java.time.Instant
import java.util.UUID


@JsonDeserialize(`as` = QuestionnaireResponseImpl::class)
@JsonPropertyOrder("id", "questionnaireRequest")
interface QuestionnaireResponse : QuestionnaireResponseBase {
    val answers: List<Answer>

    companion object {
        @JvmStatic fun createQuestionnaireResponse(
                questionnaireRequest: QuestionnaireRequest,
                createdDate: Instant,
                completedBy: PersonPublicId?,
                patientAccountId: AccountPublicId,
                answers: List<Answer>): QuestionnaireResponse =
                QuestionnaireResponseImpl(UUID.randomUUID(), questionnaireRequest, createdDate, completedBy, patientAccountId, answers)
    }

    fun <T> accept(visitor: AnswerVisitor<T>) = answers.flatMap { it.accept(visitor) }
    fun checkMandatoryQuestionsHaveAnswers(): Either<QuestionnaireServiceError, QuestionnaireResponse> {
        val missingMandatoryAnswers = answers.filter { it.question.mandatory && it.isEmpty()}
        return if (missingMandatoryAnswers.isEmpty()) { Either.right(this) } else { Either.left(QuestionnaireServiceError.MissingAnswerToMandatoryQuestion(missingMandatoryAnswers.map { it.question })) }
    }
    fun wasCompletedByThePatient() = questionnaireRequest.patientId == completedBy
}

interface QuestionnaireResponseBase : BaseModel {
    override val id: QuestionnaireResponseId
    val questionnaireRequest: QuestionnaireRequest
    val createdDate: Instant
    val completedBy: PersonPublicId?
    val patientAccountId: AccountPublicId
}

data class QuestionnaireResponseImpl
@JsonCreator(mode = JsonCreator.Mode.PROPERTIES)
constructor(
        override val id: QuestionnaireResponseId,
        @get:JsonIdentityReference(alwaysAsId = true)
        @JsonIdentityInfo(generator = ObjectIdGenerators.PropertyGenerator::class, property = "id", resolver = QuestionnaireCachingIdResolver::class)
        override val questionnaireRequest: QuestionnaireRequest,
        override val createdDate: Instant,
        override val completedBy: PersonPublicId?,
        override val patientAccountId: AccountPublicId,
        override val answers: List<Answer>
) : BaseModelImpl(), QuestionnaireResponse
