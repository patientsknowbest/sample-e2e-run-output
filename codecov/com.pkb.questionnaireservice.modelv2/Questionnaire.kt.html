package com.pkb.questionnaireservice.modelv2

import com.fasterxml.jackson.annotation.JsonCreator
import com.fasterxml.jackson.annotation.JsonIdentityInfo
import com.fasterxml.jackson.annotation.ObjectIdGenerators
import com.fasterxml.jackson.databind.annotation.JsonDeserialize
import com.pkb.entities.enums.PrivacyFlag
import com.pkb.entities.enums.Route
import com.pkb.questionnaireservice.PersonPublicId
import com.pkb.questionnaireservice.QuestionnaireId
import com.pkb.questionnaireservice.QuestionnaireRequestId
import com.pkb.questionnaireservice.TeamPublicId
import java.time.Instant
import java.util.UUID

@JsonDeserialize(`as`=QuestionnaireImpl::class)
interface Questionnaire : BaseModel {

    override val id: QuestionnaireId

    val name: String

    val creatorPersonId: PersonPublicId

    val createdDate: Instant

    val defaultPrivacyFlag: PrivacyFlag

    val components: List<QuestionnaireComponent>

    fun buildAnswers(answerSupplier: AnswerSupplier): List<Answer> = accept(answerSupplier)


    fun <T> accept(visitor: QuestionnaireComponentVisitor<T>): List<T> = components.flatMap{ it.accept(visitor) }

    fun createRequest(id: QuestionnaireRequestId, createdDate: Instant, requestorPersonId: PersonPublicId, requestingTeamId: TeamPublicId, patientId: PersonPublicId, route: Route, identifierPairs: List<Pair<String, String>>) : QuestionnaireRequest {
        val identifiers = identifierPairs.map { QuestionnaireRequestIdentifierImpl(id, it.first, it.second) }
        return QuestionnaireRequestImpl(id, this, createdDate, requestorPersonId, requestingTeamId, patientId, route, identifiers)
    }

    fun createRequest(id: QuestionnaireRequestId, createdDate: Instant, requestorPersonId: PersonPublicId, requestingTeamId: TeamPublicId, patientId: PersonPublicId, route: Route) : QuestionnaireRequest =
            QuestionnaireRequestImpl(id, this, createdDate, requestorPersonId, requestingTeamId, patientId, route, emptyList())

    fun createRequest(createdDate: Instant, requestorPersonId: PersonPublicId, requestingTeamId: TeamPublicId, patientId: PersonPublicId, route: Route, identifierPairs: List<Pair<String, String>>) : QuestionnaireRequest =
            createRequest(UUID.randomUUID(), createdDate, requestorPersonId, requestingTeamId, patientId, route, identifierPairs)
    
    fun createRequest(createdDate: Instant, requestorPersonId: PersonPublicId, requestingTeamId: TeamPublicId, patientId: PersonPublicId, route: Route) : QuestionnaireRequest =
            createRequest(UUID.randomUUID(), createdDate, requestorPersonId, requestingTeamId, patientId, route)

    fun createTeamLink(
            teamId: TeamPublicId,
            hidden: Boolean,
            now: Instant): QuestionnaireTeam =
            QuestionnaireTeamImpl(teamId, this, hidden, now)

    companion object {

       @JvmStatic fun createQuestionnaire(
                name: String,
                creatorPersonId: PersonPublicId,
                createdDate: Instant,
                defaultPrivacyFlag: PrivacyFlag,
                vararg components: QuestionnaireComponent): QuestionnaireImpl =
                createQuestionnaire(name, creatorPersonId, createdDate, defaultPrivacyFlag, components.asList())

        @JvmStatic fun createQuestionnaire(
                name: String,
                creatorPersonId: PersonPublicId,
                createdDate: Instant,
                defaultPrivacyFlag: PrivacyFlag,
                components: List<QuestionnaireComponent>): QuestionnaireImpl =
                createQuestionnaire(UUID.randomUUID(), name, creatorPersonId, createdDate, defaultPrivacyFlag, components)


        @JvmStatic fun createQuestionnaire(
                id: QuestionnaireId,
                name: String,
                creatorPersonId: PersonPublicId,
                createdDate: Instant,
                defaultPrivacyFlag: PrivacyFlag,
                components: List<QuestionnaireComponent>): QuestionnaireImpl =
                QuestionnaireImpl(id, name, creatorPersonId, createdDate, defaultPrivacyFlag, components)

    }

}

@JsonIdentityInfo(generator = ObjectIdGenerators.PropertyGenerator::class, property = "id",resolver= QuestionnaireCachingIdResolver::class, scope=QuestionnaireImpl::class)
data class QuestionnaireImpl
@JsonCreator(mode = JsonCreator.Mode.PROPERTIES)
constructor(
        override val id: QuestionnaireId,
        override val name: String,
        override val creatorPersonId: PersonPublicId,
        override val createdDate: Instant,
        override val defaultPrivacyFlag: PrivacyFlag,
        override val components: List<QuestionnaireComponent>
) : BaseModelImpl(), Questionnaire
