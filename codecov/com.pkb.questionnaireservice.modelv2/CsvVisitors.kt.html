package com.pkb.questionnaireservice.modelv2

/**
 * Visitor for Questionnaires that produces a list of column headers as required by the legacy CSV download format
 */
object CsvHeaderVisitor : QuestionnaireComponentVisitor<String> {

    override fun visit(question: MultiSelectQuestion): List<String> = question.possibleAnswers.map { "${question.displayText}: ${it.displayText}" }

    override fun visit(question: SingleSelectQuestion): List<String> = listOf(question.displayText)

    override fun visit(question: OpenQuestion): List<String> = listOf(question.displayText)
}

/**
 * Visitor for Questionnaires that produces a list of null columns of the same length as the headers above to use for padding out
 * rows for non-responders to a survey
 */
object NonResponderCsvVisitor: QuestionnaireComponentVisitor<String?> {
    override fun visit(question: MultiSelectQuestion): List<String?> = question.possibleAnswers.map { null }

    override fun visit(question: SingleSelectQuestion): List<String?> = listOf(null)

    override fun visit(question: OpenQuestion): List<String?> = listOf(null)
}

/**
 * Visitor for QuestionnaireResponses that produces the List of answer values as required by the legacy CSV download format
 */
object CsvValueVisitor : AnswerVisitor<String?> {

    override fun visit(answer: MultiSelectAnswer): List<String?> = answer.question.possibleAnswers.map { pa -> answer.answers.contains(pa.id).toString() }

    override fun visit(answer: SingleSelectAnswer): List<String?> = listOf(answer.question.possibleAnswers.find { pa -> pa.id == answer.answerId }?.displayText)

    override fun visit(answer: OpenAnswer): List<String?> = listOf(answer.answerText)
}
