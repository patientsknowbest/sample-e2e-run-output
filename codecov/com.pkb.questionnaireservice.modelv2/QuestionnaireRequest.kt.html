package com.pkb.questionnaireservice.modelv2

import com.fasterxml.jackson.annotation.JsonIdentityReference
import com.fasterxml.jackson.databind.annotation.JsonDeserialize
import com.fasterxml.jackson.databind.annotation.JsonSerialize
import com.pkb.entities.enums.Route
import com.pkb.questionnaireservice.AccountPublicId
import com.pkb.questionnaireservice.PersonPublicId
import com.pkb.questionnaireservice.QuestionnaireRequestId
import com.pkb.questionnaireservice.TeamPublicId
import java.time.Instant

@JsonDeserialize(`as`=QuestionnaireRequestImpl::class)
@JsonSerialize(`as`=QuestionnaireRequestImpl::class)
interface QuestionnaireRequest: BaseModel {
    override val id: QuestionnaireRequestId
    val questionnaire: Questionnaire
    val createdDate: Instant
    val requestorPersonId: PersonPublicId
    val requestingTeamId: TeamPublicId
    val patientId: PersonPublicId
    val route: Route
    val identifiers: List<QuestionnaireRequestIdentifier>

    fun createResponse(createdDate: Instant, completedBy: PersonPublicId?, patientAccountId: AccountPublicId, answerSupplier: AnswerSupplier) : QuestionnaireResponse =
            QuestionnaireResponse.createQuestionnaireResponse(this, createdDate, completedBy, patientAccountId, questionnaire.buildAnswers(answerSupplier))
    
}

data class QuestionnaireRequestImpl(
        override val id: QuestionnaireRequestId,
        @get:JsonIdentityReference(alwaysAsId = true)
        override val questionnaire: Questionnaire,
        override val createdDate: Instant,
        override val requestorPersonId: PersonPublicId,
        override val requestingTeamId: TeamPublicId,
        override val patientId: PersonPublicId,
        override val route: Route,
        @get:JsonIdentityReference(alwaysAsId = true)
        override val identifiers: List<QuestionnaireRequestIdentifier>) : BaseModelImpl(), QuestionnaireRequest
