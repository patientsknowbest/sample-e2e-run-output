package com.pkb.questionnaireservice.modelv2

import com.fasterxml.jackson.annotation.JsonTypeInfo
import java.io.IOException

@JsonTypeInfo(
        use = JsonTypeInfo.Id.CLASS,
        include = JsonTypeInfo.As.PROPERTY,
        property = "type")
sealed class QuestionnaireServiceError(val message: String?) {
    class NotAuthorizedError(message: String): QuestionnaireServiceError(message)
    class ConflictError(message: String): QuestionnaireServiceError(message)
    class NotFoundError(message: String) : QuestionnaireServiceError(message)
    class BadRequestError(message: String) : QuestionnaireServiceError(message)
    class UnexpectedError(message: String) : QuestionnaireServiceError(message)
    class MissingAnswerToMandatoryQuestion(val questions:List<Question>) : QuestionnaireServiceError("Some mandatory questions were not answered")
    class NameInUseError(message: String): QuestionnaireServiceError(message)
    
    class IOError(val ioException: IOException): QuestionnaireServiceError(ioException.message)

    override fun toString(): String {
        return "${this::class.simpleName}: $message"
    }
}
