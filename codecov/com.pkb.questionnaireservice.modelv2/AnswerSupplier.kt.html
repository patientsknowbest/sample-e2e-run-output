package com.pkb.questionnaireservice.modelv2

import com.pkb.questionnaireservice.AnswerId
import com.pkb.questionnaireservice.QuestionnaireComponentId

/**
 * This interface is part of a simplified visitor pattern that allows a questionnaire to
 * build its response by querying for answers from some arbitrary source - e.g. could be implemented
 * within a struts action to extract the data from fields bound from the form submission from the user's
 * browser.
 *
 * It builds on the standard visitor support for QuestionnaireComponents to expose a simplified, more focused
 * interface for implementers to minimise that amount of boilerplate/handling of internal details required.
 */
abstract class AnswerSupplier: QuestionnaireComponentVisitor<Answer> {

    abstract fun getMultiSelectQuestionAnswers(questionId: QuestionnaireComponentId): LinkedHashSet<AnswerId>

    abstract fun getSingleSelectQuestionAnswer(questionId: QuestionnaireComponentId): AnswerId?

    abstract fun getOpenQuestionAnswer(questionId: QuestionnaireComponentId): String?

    override fun visit(question: MultiSelectQuestion): List<Answer> = visitQuestion(question)

    override fun visit(question: SingleSelectQuestion): List<Answer> = visitQuestion(question)

    override fun visit(question: OpenQuestion): List<Answer> = visitQuestion(question)

    private fun visitQuestion(question: Question): List<Answer> = listOf(question.getAnswer(this))
}
