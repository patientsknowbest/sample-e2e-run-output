package com.pkb.questionnaireservice.modelv2

import com.pkb.questionnaireservice.AnswerId

/**
 * Represents a question in questionnaire with a fixed set of possible answers.
 */
abstract class ClosedQuestion:
        Question() {

    abstract val possibleAnswers: List<PossibleAnswer>

    fun getAnswerStrings(answerIds: Set<AnswerId>) : List<String> = possibleAnswers.filter { answerIds.contains(it.id) }.map(PossibleAnswer::displayText)
}
