import com.fasterxml.jackson.databind.annotation.JsonSerialize

@JsonSerialize(`as`=QuestionnaireTeamImpl::class)
interface QuestionnaireTeam {
    val teamId: TeamPublicId
    val questionnaire: Questionnaire
    var hidden: Boolean
    var updatedDate: Instant
}

        override val teamId: TeamPublicId,
        override val questionnaire: Questionnaire,
        override var hidden: Boolean = false,
