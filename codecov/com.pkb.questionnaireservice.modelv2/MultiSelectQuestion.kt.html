package com.pkb.questionnaireservice.modelv2

import com.fasterxml.jackson.annotation.JsonCreator
import com.fasterxml.jackson.annotation.JsonIdentityInfo
import com.fasterxml.jackson.annotation.JsonIgnore
import com.fasterxml.jackson.annotation.JsonIgnoreProperties
import com.fasterxml.jackson.annotation.ObjectIdGenerators
import com.pkb.questionnaireservice.AnswerId
import com.pkb.questionnaireservice.QuestionnaireComponentId
import java.util.UUID

/**
 * ClosedQuestion that allows the user to select more than one answer
 */
@JsonIdentityInfo(generator = ObjectIdGenerators.PropertyGenerator::class, property = "id", scope=QuestionnaireComponent::class, resolver = QuestionnaireCachingIdResolver::class)
class MultiSelectQuestion
@JsonCreator(mode = JsonCreator.Mode.PROPERTIES)
@JsonIgnoreProperties(ignoreUnknown = true)
constructor(
        override var id: QuestionnaireComponentId,
        override val displayText: String,
        override val mandatory: Boolean,
        override val possibleAnswers: List<PossibleAnswer>
) : ClosedQuestion() {

    constructor(displayText: String, mandatory: Boolean, vararg answers: String) : this(displayText, mandatory, listOf(*answers))

    constructor(displayText: String, mandatory: Boolean, answers: List<String>) : this(UUID.randomUUID(), displayText, mandatory, answers.map(::PossibleAnswer))

    override val type = QuestionnaireComponentType.MultiSelectQuestion

    override fun getAnswer(answerSupplier: AnswerSupplier): Answer = MultiSelectAnswer(this, answerSupplier.getMultiSelectQuestionAnswers(id))

    override fun <T> accept(visitor: QuestionnaireComponentVisitor<T>): List<T> = visitor.visit(this)
}

data class MultiSelectAnswer
@JsonCreator(mode = JsonCreator.Mode.PROPERTIES)
constructor(
        override val question: MultiSelectQuestion,
        val answers: LinkedHashSet<AnswerId>
) : Answer {
    @get:JsonIgnore
    override val answerText: String = question.getAnswerStrings(answers).joinToString()
    override val type: QuestionnaireComponentType = QuestionnaireComponentType.MultiSelectQuestion
    override fun isEmpty() = answers.isEmpty()
    override fun <T> accept(visitor: AnswerVisitor<T>): List<T> = visitor.visit(this)

}
