package com.pkb.service.personallibrary;

import com.pkb.app.entity.EHRData;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.EHRSearch;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.data.EHRRemote;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.personallibrary.entity.PersonalLibraryEntryDTO;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.List;

public class PersonalLibraryManager extends TransactionManager {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final UserManager userManager;
    private final PatientConsentManager patientConsentManager;

    public PersonalLibraryManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, UserManager userManager,
                                  PatientConsentManager patientConsentManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.userManager = userManager;
        this.patientConsentManager = patientConsentManager;
    }

    public boolean deleteEntry(EHRRequestContext requestContext, Long deleteId, long currentUserId) {
        verifyAccess(requestContext, currentUserId);
        return beanFactory.getEhrRemote().deleteEHRData(deleteId);
    }

    /**
     * Save a new file to the user's default account. The new fileId is set in
     * the FileDTO object after it's persisted.
     *
     * @param userId
     * @return The ID of the new entry.
     */
    public Long addLibraryEntry(EHRRequestContext requestContext, PersonalLibraryEntryDTO entry, long userId) {
        Long accountId = userManager.getDefaultAccountId(userId);
        verifyAccess(requestContext, userId);
        Long entryId = saveEntry(requestContext, entry, accountId);
        entry.setId(entryId);
        return entryId;
    }

    @Nullable
    public PersonalLibraryEntryDTO getEntry(@NotNull LoggedInEHRRequestContext requestContext, Long id, long currentUserId) {
        verifyAccess(requestContext, currentUserId);
        return beanFactory.getEhrRemote().findAndDecryptTypedData(requestContext, id, PersonalLibraryEntryDTO.class);
    }

    @Nullable
    public List<PersonalLibraryEntryDTO> getEntries(@NotNull LoggedInEHRRequestContext requestContext, List<Long> ids, long currentUserId) {
        verifyAccess(requestContext, currentUserId);
        return beanFactory.getEhrRemote().findAndDecryptTypedDataList(requestContext, List.copyOf(ids), PersonalLibraryEntryDTO.class);
    }

    public Long saveEntry(EHRRequestContext requestContext, PersonalLibraryEntryDTO entry, long userId, Long accountId) {
        verifyAccess(requestContext, userId);
        return saveEntry(requestContext, entry, accountId);
    }

    /**
     * Gets the list of files in the given range (and page size) for the
     * user/account given (files are in an account; the userId determines access
     * rights). File content and mediaType are not populated.
     *
     * @param userId
     * @return
     */
    public List<PersonalLibraryEntryDTO> getLibrary(@NotNull LoggedInEHRRequestContext requestContext) {
        long userId = requestContext.getContextOrAccessingUserId();
        Long accountId = userManager.getDefaultAccountId(userId);
        verifyAccess(requestContext, userId);
        return beanFactory.getEhrRemote().queryAndDecryptTypedData(
                new EHRSearch<>(accountId, PersonalLibraryEntryDTO.class, MenuDataType.libraryEntry),
                requestContext);
    }

    private void verifyAccess(EHRRequestContext requestContext, long userId) {

        if (!patientConsentManager.hasAccessToUser(requestContext, userId)) {
            String message = String.format("User %s of Org %s with consent %s does not have access to userId %s",
                    requestContext.getMaybeAccessingUserId(), requestContext.getOrgId(),
                    requestContext.getConsentStatus().getConsent().toString(), userId);
            LOGGER.error(message);
            throw new RuntimeException(message);
        }
    }

    public Long saveEntry(EHRRequestContext requestContext, PersonalLibraryEntryDTO entry, long accountId) {
        return transactional(() ->{
            EHRRemote ehrBean = beanFactory.getEhrRemote();
            EHRData ehrData = ehrBean.populateEHRData(entry, accountId, MenuDataType.libraryEntry, requestContext);
            return ehrBean.saveEHRData(requestContext, ehrData);
        });
    }
}
