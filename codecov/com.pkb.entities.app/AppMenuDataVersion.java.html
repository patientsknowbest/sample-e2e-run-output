package com.pkb.entities.app;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.Instant;

@Entity
@Table(schema = "app", name = "menu_data_version")
public class AppMenuDataVersion {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    @Column(name = "id", nullable = false)
    private Long id;

    @Column(name = "element", nullable = true)
    private String element;

    @Column(name = "version", nullable = true)
    private Long version;

    @Column(name = "mindate", nullable = true)
    private Instant minDate;

    @Column(name = "maxdate", nullable = true)
    private Instant maxDate;

    @Column(name = "role", nullable = true)
    private String role;

    @Column(name = "account_id", nullable = true)
    private Long accountId;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getElement() {
        return element;
    }

    public void setElement(String element) {
        this.element = element;
    }

    public Long getVersion() {
        return version;
    }

    public void setVersion(Long version) {
        this.version = version;
    }

    public Instant getMinDate() {
        return minDate;
    }

    public void setMinDate(Instant minDate) {
        this.minDate = minDate;
    }

    public Instant getMaxDate() {
        return maxDate;
    }

    public void setMaxDate(Instant maxDate) {
        this.maxDate = maxDate;
    }

    public String getRole() {
        return role;
    }

    public void setRole(String role) {
        this.role = role;
    }

    public Long getAccountId() {
        return accountId;
    }

    public void setAccountId(Long accountId) {
        this.accountId = accountId;
    }
}
