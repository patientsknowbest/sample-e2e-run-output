package com.pkb.backgroundtasks.model

import com.pkb.backgroundtasks.OrgPublicId
import com.pkb.backgroundtasks.PersonPrivateId
import com.pkb.entities.enums.Gender
import com.pkb.entities.enums.NationalIdType
import com.pkb.entities.enums.ValidNationalId
import java.time.LocalDate

data class Person(val id: PersonPrivateId,
                  val nationalIds: List<ValidNationalId>,
                  val orgLevelIds: List<OrgLevelId>,
                  val gender: Gender,
                  val dateOfBirth: LocalDate?,
                  val fullName: String) {

    fun orgLevelIdsForOrg(orgId: OrgPublicId): List<OrgLevelId> =
            orgLevelIds.filter { it -> it.orgPublicId == orgId }

    fun getNationalIdAnyTypeValidForSameCountryAsType(type: NationalIdType): ValidNationalId? = nationalIds.find { nid -> nid.type() == type }
            ?: nationalIds.find { nid -> nid.type().areUsedInSameCountry(type) }
}
