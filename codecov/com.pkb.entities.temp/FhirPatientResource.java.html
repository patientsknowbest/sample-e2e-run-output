package com.pkb.entities.temp;

import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import com.vladmihalcea.hibernate.type.json.JsonBinaryType;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Parameter;
import org.hibernate.annotations.Type;
import org.hibernate.annotations.TypeDef;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.util.UUID;

/**
 * Patient resource is saved to temp_backup table until FHIR data store is up.
 */
@Entity
@Table(schema = "temp_backup", name = "phr_8757_fhir_patient_resource")
@TypeDef(name = "jsonb", typeClass = JsonBinaryType.class)
public class FhirPatientResource {

    @Id
    @GenericGenerator(name = "seq_phr_8757_fhir_patient_resource_gen", strategy = "enhanced-sequence", parameters = {
            @Parameter(name = "optimizer", value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
            @Parameter(name = "schema", value = "temp_backup"),
            @Parameter(name = "sequence_name", value = "seq_phr_8757_fhir_patient_resource"),
            @Parameter(name = "increment_size", value = "50")
    })
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_phr_8757_fhir_patient_resource_gen")
    private Long id;

    @Column(name = "org_public_id", nullable = false)
    private UUID orgPublicId;

    @Column(name = "person_id", nullable = false)
    private long personId;

    @Type(type = "jsonb")
    @Column(name = "patient_resource", columnDefinition = "jsonb", nullable = false)
    private String patientResource;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public UUID getOrgPublicId() {
        return orgPublicId;
    }

    public void setOrgPublicId(UUID orgPublicId) {
        this.orgPublicId = orgPublicId;
    }

    public long getPersonId() {
        return personId;
    }

    public void setPersonId(long personId) {
        this.personId = personId;
    }

    public String getPatientResource() {
        return patientResource;
    }

    public void setPatientResource(String patientResource) {
        this.patientResource = patientResource;
    }

}
