package com.pkb.entities.temp;

import javax.annotation.Nullable;
import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.LocalDateTime;

/**
 * In Phoenix, our users can perform the same task multiple times with the same Freshdesk ID. In order to prevent
 * problems caused by an existing backup table with the same name, we need to check if the table exists and if so,
 * inform the user that they need to add a suffix to the table name. The purpose of this class is to allow to run
 * a query to check whether the backup table exists or not.
 */

@Entity
@Table(schema = "temp_backup", name = "")
public class TempBackup {

    @Id
    @Column(name = "id")
    private Long id;

    @Column(name = "status")
    private String status;

    @Nullable
    @Column(name = "sourcetext")
    private String sourcetext;

    @Column(name = "statusdate")
    private LocalDateTime statusdate;
}
