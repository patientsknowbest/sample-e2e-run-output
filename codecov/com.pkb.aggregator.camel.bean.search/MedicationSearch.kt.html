package com.pkb.aggregator.camel.bean.search

import com.pkb.aggregator.R4Medication
import com.pkb.aggregator.camel.MEDICATION_PAGE_LINKS_PROPERTY
import com.pkb.aggregator.camel.MEDICATION_PROPERTY
import com.pkb.aggregator.extension.valueAsMutableListOf
import com.pkb.aggregator.persistence.entity.MedicationIdentifierMapEntry
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamIdentifier
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.DownstreamClient
import com.pkb.aggregator.service.MedicationIdentifierService
import com.pkb.aggregator.service.dto.ReloadRequest
import com.pkb.aggregator.service.paginate.ResourcePaginator
import org.apache.camel.Body
import org.apache.camel.ExchangeProperties
import org.hl7.fhir.instance.model.api.IAnyResource
import org.springframework.stereotype.Service

/**
 * Translates medication search and pagination requests from camel routes. Mostly exists just to keep service layer
 * decoupled from Camel-related dependencies and annotations.
 *
 * Note that in FHIR a [org.hl7.fhir.r4.model.Medication] is very different from a
 * [org.hl7.fhir.r4.model.MedicationStatement], which is why this resource is handled differently from other clinical
 * datatypes associated with a patient record.
 */
@Service
class MedicationSearch(
    private val medicationIdentifierService: MedicationIdentifierService,
    resourcePaginator: ResourcePaginator,
    downstreamClient: DownstreamClient,
) : AbstractAggregatedResourceSearch<R4Medication, MedicationIdentifierMapEntry>(
    resourcePaginator,
    downstreamClient,
) {

    override fun findIdentifiers(resource: R4Medication) =
        medicationIdentifierService.findByUpstreamIdentifierIn(resource.code.coding)

    override fun findIdentifiers(upstreamSourceId: UpstreamResource) =
        medicationIdentifierService.findByUpstreamResourceIdIn(listOf(upstreamSourceId))

    override fun findSearchIdentifiers(upstreamSystem: String, upstreamValue: String) =
        medicationIdentifierService.findSearchIdentifiers(upstreamSystem, upstreamValue)

    override fun getDownstreamResourceId(identifiers: List<MedicationIdentifierMapEntry>) =
        identifiers[0].downstreamResourceId

    override fun createSearchUrl(
        @Body request: ReloadRequest,
        @ExchangeProperties properties: MutableMap<String, Any>
    ): String {
        val searchIdentifiers = listOf(UpstreamIdentifier(request.system, request.value))
        return createAndTrackSearchInfo(request, properties, searchIdentifiers)
    }

    override fun getAccumulatedResources(properties: Map<String, Any>) =
        properties.valueAsMutableListOf<IAnyResource>(MEDICATION_PROPERTY)

    override fun getAccumulatedPageLinks(properties: Map<String, Any>) =
        properties.valueAsMutableListOf<String>(MEDICATION_PAGE_LINKS_PROPERTY)
}
