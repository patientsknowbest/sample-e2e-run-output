package com.pkb.aggregator.camel.bean.search

import com.pkb.aggregator.R4Practitioner
import com.pkb.aggregator.persistence.entity.PractitionerIdentifierMapEntry
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.DownstreamClient
import com.pkb.aggregator.service.PractitionerIdentifierService
import com.pkb.aggregator.service.paginate.ResourcePaginator
import org.springframework.stereotype.Service

@Service
class PractitionerSearch(
    private val practitionerIdentifierService: PractitionerIdentifierService,
    resourcePaginator: ResourcePaginator,
    downstreamClient: DownstreamClient,
) : AbstractAggregatedResourceSearch<R4Practitioner, PractitionerIdentifierMapEntry>(
    resourcePaginator,
    downstreamClient,
) {

    override fun findIdentifiers(resource: R4Practitioner) =
        practitionerIdentifierService.findByUpstreamIdentifierIn(resource.identifier)

    override fun findIdentifiers(upstreamSourceId: UpstreamResource) =
        practitionerIdentifierService.findByUpstreamResourceIdIn(listOf(upstreamSourceId))

    override fun getDownstreamResourceId(identifiers: List<PractitionerIdentifierMapEntry>) =
        identifiers[0].downstreamResourceId

    override fun findSearchIdentifiers(upstreamSystem: String, upstreamValue: String) =
        practitionerIdentifierService.findSearchIdentifiers(upstreamSystem, upstreamValue)
}
