package com.pkb.aggregator.camel.bean.search

import com.pkb.aggregator.R4Bundle
import com.pkb.aggregator.R4IdType
import com.pkb.aggregator.R4PractitionerRole
import com.pkb.aggregator.camel.CAMELFHIR_URL_HEADER
import com.pkb.aggregator.camel.DOWNSTREAM_RESOURCE_ID_PROPERTY
import com.pkb.aggregator.camel.PRACTITIONER_ROLE_SEARCH_IDS_PROPERTY
import com.pkb.aggregator.camel.SOURCED_RESOURCE_PROPERTY
import com.pkb.aggregator.camel.UPSTREAM_SOURCE_ID_PROPERTY
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.extension.singleEntry
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_PRACTITIONER_ROLE
import com.pkb.aggregator.service.OrganizationIdentifierService
import com.pkb.aggregator.service.PractitionerIdentifierService
import com.pkb.aggregator.service.dto.ResourceWithSourceId
import com.pkb.aggregator.service.exception.UpstreamReferenceException
import org.apache.camel.Body
import org.apache.camel.ExchangeProperties
import org.apache.camel.ExchangeProperty
import org.apache.camel.Headers
import org.springframework.stereotype.Service

@Service
class PractitionerRoleSearch(
    private val practitionerIdentifierService: PractitionerIdentifierService,
    private val organizationIdentifierService: OrganizationIdentifierService,
) {

    data class PractitionerRoleSearchResourceIds(
        val practitionerResourceId: R4IdType,
        val organizationResourceId: R4IdType
    ) {
        fun searchUrl() = "$FHIR_TYPE_PRACTITIONER_ROLE?" +
                "practitioner=${practitionerResourceId.idPart}" +
                "&organization=${organizationResourceId.idPart}"
    }

    fun findDownstreamResourceIds(
        @ExchangeProperties properties: MutableMap<String, Any>,
        @ExchangeProperty(UPSTREAM_SOURCE_ID_PROPERTY) upstreamSourceId: UpstreamResource,
        @Body resourceWithSourceId: ResourceWithSourceId<R4PractitionerRole>
    ) {

        // We want to look for the downstream PractitionerRole that corresponds to this one - so we need to figure out
        // the downstream organization and practitioner resource ids and then search for the role that links them
        val (source, resource) = resourceWithSourceId
        val upstreamPractitioner = UpstreamResource(source, resource.practitioner.reference)
        val upstreamOrganization = UpstreamResource(source, resource.organization.reference)

        val downstreamPractitionerResourceId = practitionerIdentifierService
            .findByUpstreamResourceIdIn(listOf(upstreamPractitioner))
            .firstOrNull()
            ?.downstreamResourceId
            ?.let { R4IdType(it) }
            ?: throw UpstreamReferenceException(upstreamPractitioner)

        val downstreamOrganizationResourceId = organizationIdentifierService
            .findByUpstreamResourceIdIn(listOf(upstreamOrganization))
            .firstOrNull()
            ?.downstreamResourceId
            ?.let { R4IdType(it) }
            ?: throw UpstreamReferenceException(upstreamOrganization)

        properties[SOURCED_RESOURCE_PROPERTY] = resourceWithSourceId
        properties[PRACTITIONER_ROLE_SEARCH_IDS_PROPERTY] =
            PractitionerRoleSearchResourceIds(downstreamPractitionerResourceId, downstreamOrganizationResourceId)
    }

    fun setDownstreamRoleResourceId(
        @ExchangeProperties properties: MutableMap<String, Any>,
        @Body bundle: R4Bundle
    ) {
        val role = bundle.singleEntry<R4PractitionerRole>() ?: throw RuntimeException("Expected practitioner role in bundle")
        properties[DOWNSTREAM_RESOURCE_ID_PROPERTY] = role.simpleResourceId()
    }

    fun prepareSearchUrl(
        @ExchangeProperty(PRACTITIONER_ROLE_SEARCH_IDS_PROPERTY) roleIds: PractitionerRoleSearchResourceIds,
        @Headers headers: MutableMap<String, Any>,
    ) {
        headers[CAMELFHIR_URL_HEADER] = roleIds.searchUrl()
    }
}
