package com.pkb.aggregator.camel.bean.search

import com.pkb.aggregator.R4Organization
import com.pkb.aggregator.camel.ORGANIZATION_PAGE_LINKS_PROPERTY
import com.pkb.aggregator.camel.ORGANIZATION_PROPERTY
import com.pkb.aggregator.extension.valueAsMutableListOf
import com.pkb.aggregator.persistence.entity.OrganizationIdentifierMapEntry
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.DownstreamClient
import com.pkb.aggregator.service.OrganizationIdentifierService
import com.pkb.aggregator.service.paginate.ResourcePaginator
import org.hl7.fhir.instance.model.api.IAnyResource
import org.springframework.stereotype.Service

/**
 * Translates organization search and pagination requests from camel routes. Mostly exists just to keep service layer
 * decoupled from Camel-related dependencies and annotations.
 */
@Service
class OrganizationSearch(
    private val organizationIdentifierService: OrganizationIdentifierService,
    resourcePaginator: ResourcePaginator,
    downstreamClient: DownstreamClient,
) : AbstractAggregatedResourceSearch<R4Organization, OrganizationIdentifierMapEntry>(
    resourcePaginator,
    downstreamClient,
) {
    override fun findIdentifiers(resource: R4Organization) =
        organizationIdentifierService.findByUpstreamIdentifierIn(resource.identifier)

    override fun findIdentifiers(upstreamSourceId: UpstreamResource) =
        organizationIdentifierService.findByUpstreamResourceIdIn(listOf(upstreamSourceId))

    override fun findSearchIdentifiers(upstreamSystem: String, upstreamValue: String) =
        organizationIdentifierService.findSearchIdentifiers(upstreamSystem, upstreamValue)

    override fun getDownstreamResourceId(identifiers: List<OrganizationIdentifierMapEntry>) =
        identifiers[0].downstreamResourceId

    override fun getAccumulatedResources(properties: Map<String, Any>) =
        properties.valueAsMutableListOf<IAnyResource>(ORGANIZATION_PROPERTY)

    override fun getAccumulatedPageLinks(properties: Map<String, Any>) =
        properties.valueAsMutableListOf<String>(ORGANIZATION_PAGE_LINKS_PROPERTY)
}
