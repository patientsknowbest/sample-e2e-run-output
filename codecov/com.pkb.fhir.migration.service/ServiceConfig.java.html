package com.pkb.fhir.migration.service;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.fhir.common.configuration.EhrmigratorConfig;
import com.pkb.fhir.common.repository.AccountRepository;
import com.pkb.fhir.common.repository.FhirStoreSyncCompletedRepository;
import com.pkb.fhir.common.repository.FhirStoreSyncRepository;
import com.pkb.fhir.common.repository.Hl7PartnerRepository;
import com.pkb.fhir.common.repository.InstituteUserRepository;
import com.pkb.fhir.common.repository.MenuDataRepository;
import com.pkb.fhir.common.repository.OrgExternalIdRepository;
import com.pkb.fhir.common.repository.OrgRepository;
import com.pkb.fhir.common.repository.PatientConsentRepository;
import com.pkb.fhir.common.repository.PatientOptOutRepository;
import com.pkb.fhir.common.repository.PersonRepository;
import com.pkb.fhir.common.repository.ReferenceDatumRepository;
import com.pkb.fhir.common.repository.TeamRepository;
import com.pkb.fhir.common.service.OrgService;
import com.pkb.fhir.common.service.PersonService;
import com.pkb.fhir.common.service.TeamService;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class ServiceConfig {

    @Bean
    public AccountService accountService(AccountRepository repository) {
        return new AccountService(repository);
    }

    @Bean
    public BaseMenuDataService baseMenuDataService(MenuDataRepository menuDataRepository) {
        return new BaseMenuDataService(menuDataRepository);
    }

    @Bean
    public ConsentService consentService(PatientConsentRepository patientConsentRepository) {
        return new ConsentService(patientConsentRepository);
    }

    @Bean
    public FhirStoreSyncService fhirStoreSyncService(FhirStoreSyncRepository fhirStoreSyncRepository, FhirStoreSyncCompletedRepository fhirStoreSyncCompletedRepository,
                                                     EhrmigratorConfig config, DateTimeService dateTimeService) {
        return new FhirStoreSyncService(fhirStoreSyncRepository, fhirStoreSyncCompletedRepository, config, dateTimeService);
    }

    @Bean
    public Hl7PartnerService hl7PartnerService(Hl7PartnerRepository partnerRepository) {
        return new Hl7PartnerService(partnerRepository);
    }

    @Bean
    public InstituteUserService instituteUserService(InstituteUserRepository instituteUserRepository) {
        return new InstituteUserService(instituteUserRepository);
    }

    @Bean
    public OrgExternalIdService orgExternalIdService(OrgExternalIdRepository orgExternalIdRepository) {
        return new OrgExternalIdService(orgExternalIdRepository);
    }

    @Bean
    public OrgService orgService(OrgRepository orgRepository) {
        return new OrgService(orgRepository);
    }

    @Bean
    public PatientOptOutService patientOptOutService(PatientOptOutRepository patientOptOutRepository) {
        return new PatientOptOutService(patientOptOutRepository);
    }

    @Bean
    public PersonService personService(PersonRepository personRepository) {
        return new PersonService(personRepository);
    }

    @Bean
    public TeamService teamService(TeamRepository teamRepository) {
        return new TeamService(teamRepository);
    }

    @Bean
    public ReferenceDatumService referenceDatumService(ReferenceDatumRepository referenceDatumRepository) {
        return new ReferenceDatumService(referenceDatumRepository);
    }
}
