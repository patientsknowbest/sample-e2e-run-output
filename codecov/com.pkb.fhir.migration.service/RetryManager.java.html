package com.pkb.fhir.migration.service;

import org.springframework.retry.RetryCallback;
import org.springframework.retry.support.RetrySynchronizationManager;
import org.springframework.retry.support.RetryTemplate;

/**
 * A wrapper around spring-retry functionality.
 * This class mainly exists so we can unit test this stuff effectively.
 */
public class RetryManager<T, E extends Throwable> {

    private final RetryTemplate retryTemplate;

    public RetryManager(RetryTemplate retryTemplate) {
        this.retryTemplate = retryTemplate;
    }

    public T doWithRetry(RetryCallback<T, E> callback) throws E {
        return retryTemplate.execute(callback);
    }

    public int getRetryCount() {
        return RetrySynchronizationManager.getContext().getRetryCount();
    }
}
