    private final InstituteUserRepository instituteUserRepository;

    public InstituteUserService(InstituteUserRepository instituteUserRepository) {
        this.instituteUserRepository = instituteUserRepository;
    }

    public Optional<PublicInstituteUser> findOptionalByIdFetchTeam(long id) {
        return instituteUserRepository.findOptionalByIdFetchTeam(id);
    }
