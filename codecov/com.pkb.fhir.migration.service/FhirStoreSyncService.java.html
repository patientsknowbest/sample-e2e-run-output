package com.pkb.fhir.migration.service;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.entities.enums.FhirStoreSyncStatus;
import com.pkb.entities.pub.PublicFhirStoreSync;
import com.pkb.entities.pub.PublicFhirStoreSyncCompleted;
import com.pkb.fhir.common.configuration.EhrmigratorConfig;
import com.pkb.fhir.common.repository.FhirStoreSyncCompletedRepository;
import com.pkb.fhir.common.repository.FhirStoreSyncRepository;
import org.springframework.data.domain.PageRequest;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;

public class FhirStoreSyncService {

    private final FhirStoreSyncRepository fhirStoreSyncRepository;
    private final FhirStoreSyncCompletedRepository fhirStoreSyncCompletedRepository;
    private final EhrmigratorConfig config;
    private final DateTimeService dateTimeService;

    public FhirStoreSyncService(FhirStoreSyncRepository fhirStoreSyncRepository, FhirStoreSyncCompletedRepository fhirStoreSyncCompletedRepository,
                                EhrmigratorConfig config, DateTimeService dateTimeService) {
        this.fhirStoreSyncRepository = fhirStoreSyncRepository;
        this.fhirStoreSyncCompletedRepository = fhirStoreSyncCompletedRepository;
        this.config = config;
        this.dateTimeService = dateTimeService;
    }

    public List<PublicFhirStoreSync> getUnsyncedBatch() {
        return fhirStoreSyncRepository.getPublicFhirStoreSyncs(PageRequest.of(0, config.getFhirStoreSyncBatchSize()));
    }

    public long getCountUnsyncedFhirStoreSyncs() {
        return fhirStoreSyncRepository.count();
    }

    @Transactional("sourceDbTransactionManager")
    public void completeFhirStoreSync(PublicFhirStoreSync fhirStoreSync, FhirStoreSyncStatus syncStatus, String details) {
        fhirStoreSyncRepository.delete(fhirStoreSync);

        PublicFhirStoreSyncCompleted completed = new PublicFhirStoreSyncCompleted();
        completed.setId(fhirStoreSync.getId()); //Make these always match up. No reason not to.
        completed.setSyncDetails(fhirStoreSync.getSyncDetails());
        completed.getSyncDetails().setSyncStatusTimestamp(dateTimeService.now());
        completed.setSyncStatus(syncStatus);
        completed.setDetails(details);
        fhirStoreSyncCompletedRepository.save(completed);
    }
}
