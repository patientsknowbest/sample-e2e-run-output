package com.pkb.fhir.migration.service;

import com.pkb.entities.pub.PublicOrgExternalId;
import com.pkb.fhir.common.repository.OrgExternalIdRepository;

import java.util.List;

public class OrgExternalIdService {

    private final OrgExternalIdRepository orgExternalIdRepository;

    public OrgExternalIdService(OrgExternalIdRepository orgExternalIdRepository) {
        this.orgExternalIdRepository = orgExternalIdRepository;
    }

    public List<PublicOrgExternalId> findByOrgId(long orgId) {
        return orgExternalIdRepository.findByOrgId(orgId);
    }
}
