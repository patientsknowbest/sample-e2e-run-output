package com.pkb.fhir.migration.service;

import com.pkb.entities.enums.ReferenceDataType;
import com.pkb.entities.pub.PublicReferenceDatum;
import com.pkb.fhir.common.repository.ReferenceDatumRepository;

import java.util.List;
import java.util.Optional;

public class ReferenceDatumService {

    private final ReferenceDatumRepository referenceDatumRepository;

    public ReferenceDatumService(ReferenceDatumRepository referenceDatumRepository) {
        this.referenceDatumRepository = referenceDatumRepository;
    }

    public Optional<PublicReferenceDatum> getReferenceDatum(long orgId, ReferenceDataType dataType, String dataId) {
        return referenceDatumRepository.getPublicReferenceDatumByOrgAndDataTypeAndDataId(orgId, dataType, dataId);
    }

    public List<PublicReferenceDatum> getReferenceDatumsForOrgAndType(long orgId, ReferenceDataType dataType) {
        return referenceDatumRepository.getPublicReferenceDatumByOrgAndDataType(orgId, dataType);
    }

    public Optional<PublicReferenceDatum> findByIdFetchOrg(long id) {
        return referenceDatumRepository.findByIdFetchOrg(id);
    }
}
