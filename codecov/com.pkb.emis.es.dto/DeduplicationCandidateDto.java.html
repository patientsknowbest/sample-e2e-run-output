package com.pkb.emis.es.dto;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.coding.entity.CodingReceived;
import io.vavr.Tuple;
import io.vavr.Tuple3;
import org.immutables.value.Value;
import org.jetbrains.annotations.Nullable;

import java.io.Serializable;
import java.util.UUID;

import static com.google.common.base.Preconditions.checkNotNull;

@Value.Modifiable // This holds mutable objects as fields, so is not an immutable type
public interface DeduplicationCandidateDto<T> extends Serializable {

    /**
     * The mapped PKB datatype to be deduplicated (Medication, Diagnosis, Allergy etc)
     */
    T candidateEntity();

    /**
     * Arbitrary UUID attached to EMIS DTO during construction (only used during processing to match up results,
     * never persisted in the database etc)
     */
    UUID emisDtoTransientId();

    /**
     * Clinical coding if known
     * <p>
     * Nullable because Optional isn't serializable
     */
    @Nullable
    CodingReceived coding();

    EHRRequestContext recordContext();

    default Tuple3<T, CodingReceived, EHRRequestContext> toObservationDtoWithCoding() {
        checkNotNull(coding(), "coding not set");
        return Tuple.of(candidateEntity(), coding(), recordContext());
    }

    static <T> DeduplicationCandidateDto<T> uncoded(
            UUID emisDtoTransientId,
            T candidateEntity,
            EHRRequestContext requestContext) {

        return ModifiableDeduplicationCandidateDto.<T>create()
                .setEmisDtoTransientId(emisDtoTransientId)
                .setCandidateEntity(candidateEntity)
                .setRecordContext(requestContext);
    }

    static <T> DeduplicationCandidateDto<T> coded(
            UUID emisDtoTransientId,
            T candidateEntity,
            CodingReceived coding,
            EHRRequestContext requestContext) {

        return ModifiableDeduplicationCandidateDto.<T>create()
                .setEmisDtoTransientId(emisDtoTransientId)
                .setCandidateEntity(candidateEntity)
                .setCoding(coding)
                .setRecordContext(requestContext);
    }
}
