package com.pkb.dataupload.emis.esv2;

import javax.persistence.Cacheable;
import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.persistence.Version;
import java.io.Serializable;
import java.time.Instant;

@Entity
@Cacheable(false)
@Table(schema = "emis_integration", name = "emis_es_job_state")
public class EmisEsJobState implements Serializable {

    private static final long serialVersionUID = 5588589151246384104L;

    @SuppressWarnings("SSBasedInspection")
    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Version
    @Column
    private Integer version;

    @Column(nullable = false)
    private Instant started;

    @Column(nullable = true)
    private Instant finished;

    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    private EmisJobStatus status;

    @Column(name = "retry_permitted", nullable = false)
    private boolean retryPermitted;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Integer getVersion() {
        return version;
    }

    public void setVersion(Integer version) {
        this.version = version;
    }

    public Instant getStarted() {
        return started;
    }

    public void setStarted(Instant started) {
        this.started = started;
    }

    public Instant getFinished() {
        return finished;
    }

    public void setFinished(Instant finished) {
        this.finished = finished;
    }

    public EmisJobStatus getStatus() {
        return status;
    }

    public void setStatus(EmisJobStatus status) {
        this.status = status;
    }

    public boolean isRetryPermitted() {
        return retryPermitted;
    }

    public void setRetryPermitted(boolean retryPermitted) {
        this.retryPermitted = retryPermitted;
    }
}
