package com.pkb.dataupload.emis.esv2;

import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.Instant;
import java.util.UUID;

/**
 * Sometimes we receive rubbish from emis on a given date and we need to skip it.
 * Records in this table allow us to do us.
 */
@Entity
@Table(name = "emis_es_datestamp_skip", schema = "emis_integration")
public class EmisEsDatestampSkip {

    @Id
    @Column(name = "id")
    @Type(type = "pg-uuid")
    private UUID id;

    @Column(name = "datestamp_to_skip", nullable = false)
    private String datestampToSkip;

    @Column(name = "unique_tag_to_skip_for")
    private String uniqueTagToSkipFor;

    //jira/fd ticket or equivalent that has the information behind why a given date needs to be skipped.
    @Column(name = "ticket", nullable = false)
    private String ticket;

    @Column(name = "createddate", nullable = false)
    private Instant createddate;

    public UUID getId() {
        return id;
    }

    public void setId(UUID id) {
        this.id = id;
    }

    public String getDatestampToSkip() {
        return datestampToSkip;
    }

    public void setDatestampToSkip(String datestampToSkip) {
        this.datestampToSkip = datestampToSkip;
    }

    public String getUniqueTagToSkipFor() {
        return uniqueTagToSkipFor;
    }

    public void setUniqueTagToSkipFor(String uniqueTagToSkipFor) {
        this.uniqueTagToSkipFor = uniqueTagToSkipFor;
    }

    public String getTicket() {
        return ticket;
    }

    public void setTicket(String ticket) {
        this.ticket = ticket;
    }

    public Instant getCreateddate() {
        return createddate;
    }

    public void setCreateddate(Instant createddate) {
        this.createddate = createddate;
    }
}
