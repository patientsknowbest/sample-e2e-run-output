package com.pkb.dataupload.emis.esv2;

import com.pkb.dataupload.emis.esv2.interfaces.EmisRecordWithPatientNamespacedGuid;

import javax.persistence.Cacheable;
import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.persistence.Version;
import java.io.Serializable;

@Entity
@Cacheable(false)
@Table(schema = "emis_integration", name = "emis_es_care_record_processing_state")
public class EmisEsCareRecordProcessingState implements EmisRecordWithPatientNamespacedGuid, Serializable {

    @SuppressWarnings("SSBasedInspection")
    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column(name = "batch_id", nullable = false)
    private Long batchId;

    @Version
    @Column
    private Integer version;

    @Column(name = "observation_guid", nullable = false)
    private String observationGuid;

    @Column(name = "patient_guid", nullable = false)
    private String patientGuid;

    @Override
    public String patientGuid() {
        return getPatientGuid();
    }

    @Override
    public String patientSpecificGuid() {
        return getObservationGuid();
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Long getBatchId() {
        return batchId;
    }

    public void setBatchId(Long batchId) {
        this.batchId = batchId;
    }

    public Integer getVersion() {
        return version;
    }

    public void setVersion(Integer version) {
        this.version = version;
    }

    public String getObservationGuid() {
        return observationGuid;
    }

    public void setObservationGuid(String observationGuid) {
        this.observationGuid = observationGuid;
    }

    public String getPatientGuid() {
        return patientGuid;
    }

    public void setPatientGuid(String patientGuid) {
        this.patientGuid = patientGuid;
    }
}
