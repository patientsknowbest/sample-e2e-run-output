package com.pkb.dataupload.emis.esv2;

import com.pkb.dataupload.EmisJobArtifactId;
import com.vladmihalcea.hibernate.type.json.JsonBinaryType;
import org.hibernate.annotations.Type;
import org.hibernate.annotations.TypeDef;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.Id;
import javax.persistence.IdClass;
import javax.persistence.Table;


@Entity
@IdClass(EmisJobArtifactId.class)
@Table(schema = "emis_integration", name = "emis_job_artifacts")
@TypeDef(
        name = "jsonb",
        typeClass = JsonBinaryType.class
)
public class EmisJobArtifact {

    @Id
    @Column(name = "job_id", nullable = false)
    private Long jobId;

    @Type(type = "jsonb")
    @Column(name = "report", columnDefinition = "jsonb", nullable = false)
    String report;

    @Id
    @Enumerated(EnumType.STRING)
    @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
    @Column(name = "type", nullable = false)
    EmisArtifactType type;

    protected EmisJobArtifact() {
    }

    public EmisJobArtifact(Long jobId, EmisArtifactType type, String report) {
        this.jobId = jobId;
        this.report = report;
        this.type = type;
    }

    public Long getJobId() {
        return jobId;
    }

    public void setJobId(Long jobId) {
        this.jobId = jobId;
    }

    public String getReport() {
        return report;
    }

    public void setReport(String report) {
        this.report = report;
    }

    public EmisArtifactType getType() {
        return type;
    }

    public void setType(EmisArtifactType type) {
        this.type = type;
    }
}
