package com.pkb.dataupload.emis.esv2;

import javax.persistence.Cacheable;
import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.LocalDate;

@Entity
@Cacheable(false)
@Table(schema = "emis_integration", name = "emis_es_processed_rows")
public class EmisEsProcessedRow implements Serializable {

    @SuppressWarnings("SSBasedInspection")
    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column(name = "batch_id", nullable = false)
    private Long batchId;

    @Column(name = "emis_file_type", nullable = false)
    private Integer emisFileType;

    @Column(name = "file_hash", nullable = true)
    private String fileHash;

    @Column(name = "fully_processed", nullable = false)
    private boolean fullyProcessed;

    @Column(name = "last_processed_row", nullable = true)
    private Long lastProcessedRow;

    @Column(name = "fileset_datestamp", nullable = false)
    private LocalDate filesetDatestamp;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Long getBatchId() {
        return batchId;
    }

    public void setBatchId(Long batchId) {
        this.batchId = batchId;
    }

    public Integer getEmisFileType() {
        return emisFileType;
    }

    public void setEmisFileType(Integer emisFileType) {
        this.emisFileType = emisFileType;
    }

    public String getFileHash() {
        return fileHash;
    }

    public void setFileHash(String fileHash) {
        this.fileHash = fileHash;
    }

    public boolean isFullyProcessed() {
        return fullyProcessed;
    }

    public void setFullyProcessed(boolean fullyProcessed) {
        this.fullyProcessed = fullyProcessed;
    }

    public Long getLastProcessedRow() {
        return lastProcessedRow;
    }

    public void setLastProcessedRow(Long lastProcessedRow) {
        this.lastProcessedRow = lastProcessedRow;
    }

    public LocalDate getFilesetDatestamp() {
        return filesetDatestamp;
    }

    public void setFilesetDatestamp(LocalDate filesetDatestamp) {
        this.filesetDatestamp = filesetDatestamp;
    }
}
