package com.pkb.dataupload.emis.esv2;

import javax.persistence.Cacheable;
import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.persistence.Version;
import java.io.Serializable;
import java.time.Instant;
import java.time.LocalDate;

@Entity
@Cacheable(false)
@Table(schema = "emis_integration", name = "emis_es_batch_state")
public class EmisEsBatchState implements Serializable {

    @SuppressWarnings("SSBasedInspection")
    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column(name = "job_id", nullable = false)
    private long jobId;

    @Column(name = "last_processed_batch_guid", nullable = false)
    private String lastProcessedBatchGuid;

    @Column(name = "last_processed_batch_datestamp", nullable = false)
    private LocalDate lastProcessedBatchDatestamp;

    @Column(name = "last_processed_batch_bulk", nullable = false)
    private boolean lastProcessedBatchBulk;

    @Column(name = "sftp_unique_id", nullable = false)
    private String sftpUniqueId;

    @Column(name = "recorded_date", nullable = false)
    private Instant recordedDate;

    @Version
    @Column
    private long version;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public long getJobId() {
        return jobId;
    }

    public void setJobId(long jobId) {
        this.jobId = jobId;
    }

    public String getLastProcessedBatchGuid() {
        return lastProcessedBatchGuid;
    }

    public void setLastProcessedBatchGuid(String lastProcessedBatchGuid) {
        this.lastProcessedBatchGuid = lastProcessedBatchGuid;
    }

    public LocalDate getLastProcessedBatchDatestamp() {
        return lastProcessedBatchDatestamp;
    }

    public void setLastProcessedBatchDatestamp(LocalDate lastProcessedBatchDatestamp) {
        this.lastProcessedBatchDatestamp = lastProcessedBatchDatestamp;
    }

    public boolean isLastProcessedBatchBulk() {
        return lastProcessedBatchBulk;
    }

    public void setLastProcessedBatchBulk(boolean lastProcessedWasBulk) {
        this.lastProcessedBatchBulk = lastProcessedWasBulk;
    }

    public String getSftpUniqueId() {
        return sftpUniqueId;
    }

    public void setSftpUniqueId(String sftpUniqueId) {
        this.sftpUniqueId = sftpUniqueId;
    }

    public long getVersion() {
        return version;
    }

    public void setVersion(long version) {
        this.version = version;
    }

    public Instant getRecordedDate() {
        return recordedDate;
    }

    public void setRecordedDate(Instant recordedDate) {
        this.recordedDate = recordedDate;
    }
}
