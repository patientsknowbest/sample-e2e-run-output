package com.pkb.api.fhir.namingsystem;

import org.hl7.fhir.dstu3.model.Enumerations;
import org.hl7.fhir.dstu3.model.NamingSystem;

import java.time.Instant;
import java.util.Date;

import static java.util.Collections.singletonList;

final class NamingSystemMapperUtil {

    static NamingSystem createSkeletonNamingSystem() {
        NamingSystem namingSystem = new NamingSystem();
        namingSystem.setStatus(Enumerations.PublicationStatus.ACTIVE);
        namingSystem.setKind(NamingSystem.NamingSystemType.IDENTIFIER);
        namingSystem.setDate(Date.from(Instant.now()));
        return namingSystem;
    }

    static void populateUniqueId(NamingSystem namingSystem, String publicId) {
        NamingSystem.NamingSystemUniqueIdComponent uniqueId = new NamingSystem.NamingSystemUniqueIdComponent();
        uniqueId.setType(NamingSystem.NamingSystemIdentifierType.URI);
        uniqueId.setValue("urn:uuid:" + publicId);
        namingSystem.setUniqueId(singletonList(uniqueId));
    }
}
