package com.pkb.kms.shared.representation

data class KmsError(val error: String, val description: String, val type: Type) {

    // Put factories for common/reused errors here. Otherwise, each class/module should
    // define private extension methods for creating specific errors
    companion object {
        fun personNotFound(personPublicId: PersonPublicId) = KmsError(
            "person.not.found",
            "Person $personPublicId not found",
            Type.NOT_FOUND
        )

        fun orgNotFound(orgPublicId: OrgPublicId) = KmsError(
            "org.not.found",
            "Org $orgPublicId not found",
            Type.NOT_FOUND
        )

        fun accountNotFound(accountPublicId: AccountPublicId) = KmsError(
            "account.not.found",
            "Account $accountPublicId not found",
            Type.NOT_FOUND
        )

        fun accountPrivateKeyNotCached(accountPublicId: AccountPublicId) = KmsError(
            "account.keys.unavailable",
            "Keys not currently available for account $accountPublicId",
            Type.NOT_FOUND
        )

        fun pathNotFound(requester: Requester, accountPublicId: AccountPublicId) = KmsError(
            "path.not.found",
            "No path found from requester $requester to account $accountPublicId",
            Type.NOT_AUTHORISED
        )

        fun orgAccountPrivateKeyNotCached(orgId: Long) = KmsError(
            "org.account.private.key.unavailable",
            "Private key not currently available for org with id $orgId",
            Type.NOT_FOUND
        )

        fun unexpectedRequesterType(requester: Requester) = KmsError(
            "unexpected.requester",
            "Unexpected requester $requester",
            Type.UNEXPECTED
        )

        fun symmetricKeyNotFound(accountPublicId: AccountPublicId) = KmsError(
                "symmetric.key.not.found",
                "Account $accountPublicId has no symmetric keys",
                Type.NOT_FOUND)


        fun KmsError.Companion.ssoDecryptionFailed(personPublicId: PersonPublicId) = KmsError(
            "user.private.key.decryption.failed",
            "SSO key for personPublicId $personPublicId could not be decrypted",
            KmsError.Type.NOT_AUTHORISED)

        fun KmsError.Companion.userPrivateKeyDecryptionFailed(personPublicId: PersonPublicId) = KmsError(
            "user.private.key.decryption.failed",
            "Private key for person $personPublicId could not be decrypted",
            KmsError.Type.NOT_AUTHORISED)
    }

    enum class Type {
        NOT_FOUND,
        NOT_AUTHORISED,
        BAD_REQUEST,
        CONFLICT,
        UNEXPECTED
    }
}
