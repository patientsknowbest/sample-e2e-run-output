package com.pkb.kms.shared.representation

import io.vavr.control.Either
import io.vavr.control.Either.left
import io.vavr.control.Either.right
import java.io.Serializable

sealed class Requester: Serializable {
    companion object {
        @JvmStatic
        fun derive(requesterPersonId: PersonPublicId?, requesterOrgId: OrgPublicId?): Either<KmsError, Requester> =
                if (requesterPersonId == null && requesterOrgId == null) {
                    left(KmsError("ref.not.supplied",
                            "ref_person_id or ref_org_id must be supplied",
                            KmsError.Type.BAD_REQUEST))
                } else if (requesterPersonId != null && requesterOrgId != null) {
                    left(KmsError("ref.invalid.supplied",
                            "only one of ref_person_id or ref_org_id must be supplied",
                            KmsError.Type.BAD_REQUEST))
                } else if (requesterOrgId != null) {
                    right(OrgRequester(requesterOrgId))
                } else if (requesterPersonId != null) {
                    right(PersonRequester(requesterPersonId))
                } else {
                    throw IllegalStateException("Unexpected branch")
                }
    }
} 
data class PersonRequester(val personPublicId: PersonPublicId): Requester(), Serializable
data class OrgRequester(val orgPublicId: OrgPublicId): Requester(), Serializable
