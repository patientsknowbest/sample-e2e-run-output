package com.pkb.kms.shared.representation

@Suppress("DataClassPrivateConstructor")
data class AccountKeyAvailability private constructor(val personPublicId: PersonPublicId, val available: Set<AccountPublicId>, val unavailable: Set<AccountPublicId>) {
    companion object {
        @JvmStatic
        operator fun invoke(personPublicId: PersonPublicId, available: Set<AccountPublicId>, unavailable: Set<AccountPublicId>): AccountKeyAvailability {
            val intersection = available.intersect(unavailable)
            if(intersection.isNotEmpty()) {
                throw IllegalArgumentException("AccountKeyAvailability object should not have overlap between available/unavailable sets. Intersection: $intersection")
            }
            return AccountKeyAvailability(personPublicId, available, unavailable)
        }
    }
}
