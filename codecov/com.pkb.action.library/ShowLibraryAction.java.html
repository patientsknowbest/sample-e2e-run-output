package com.pkb.action.library;

import com.pkb.action.BaseAction;

import static org.apache.logging.log4j.util.Strings.isBlank;

/**
 * this class looks at the referer to work whaich page the user came from and then redirects appropriately
 *
 * @author vorekoya on 08/12/2015.
 */
public class ShowLibraryAction extends BaseAction {

    public static final String ADMIN_REFERER = "admin_referer";
    public static final String SUPER_ADMIN_REFERER = "super_admin_referer";
    public static final String EMPLOYEE_REFERER = "employee_referer";
    public static final String PATIENT_REFERER = "patient_referer";

    private String namespace;
    private String referer;
    private String tkid;
    private String instituteCode;
    private String teamId;

    /**
     * inspects the previous page in order to work out which page to redirect the user to
     *
     * @return returns a string admin or patient
     */
    public String show() {
        if (isBlank(referer)) {
            return ADMIN_REFERER.toLowerCase();
        } else {
            if (referer.contains(",")) {
                String[] refArray = referer.split(",");
                referer = refArray[refArray.length - 1].trim();
            }
            if (getLoggedInEHRRequestContext().getAccessingUser().isEmployee()) {
                referer = EMPLOYEE_REFERER;
            }

            if (ADMIN_REFERER.equalsIgnoreCase(referer)) {
                return ADMIN_REFERER.toLowerCase();
            } else if (SUPER_ADMIN_REFERER.equalsIgnoreCase(referer)) {
                return SUPER_ADMIN_REFERER.toLowerCase();
            } else if (EMPLOYEE_REFERER.equalsIgnoreCase(referer)) {
                return EMPLOYEE_REFERER.toLowerCase();
            } else {
                return PATIENT_REFERER.toLowerCase();
            }
        }
    }

    /*
    getters and setters
     */
    public String getReferer() {
        return referer;
    }

    public void setReferer(String referer) {
        this.referer = referer;
    }

    public String getTkid() {
        return tkid;
    }

    public void setTkid(String tkid) {
        this.tkid = tkid;
    }

    public String getInstituteCode() {
        return instituteCode;
    }

    public void setInstituteCode(String instituteCode) {
        if (instituteCode != null) {

            if (instituteCode.contains(",")) {
                String[] codes = instituteCode.split(",");
                this.instituteCode = codes[codes.length - 1].trim();
            } else {
                this.instituteCode = instituteCode;

            }
        }
    }

    public String getNamespace() {
        return namespace;
    }

    public void setNamespace(String namespace) {
        this.namespace = namespace;
    }

    public String getTeamId() {
        return teamId;
    }

    public void setTeamId(String teamId) {
        this.teamId = teamId;
    }
}
