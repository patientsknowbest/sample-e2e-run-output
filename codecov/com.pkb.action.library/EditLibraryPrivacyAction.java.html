package com.pkb.action.library;

import com.pkb.action.ConsentCategoryBaseAction;
import com.pkb.action.EditPrivacyService;
import com.pkb.action.ImmutableFormRequest;
import com.pkb.action.ImmutableSaveRequest;
import com.pkb.action.diary.UIPrivacyDto;
import org.apache.struts2.interceptor.validation.SkipValidation;
import org.springframework.beans.factory.annotation.Autowired;

import static com.pkb.action.DtoType.LIBRARY;

public class EditLibraryPrivacyAction extends ConsentCategoryBaseAction {

    @Autowired
    private EditPrivacyService editPrivacyService;

    private UIPrivacyDto dto;

    private Long dataPointId;

    public String editPrivacyForm() {
        var request = ImmutableFormRequest.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .dtoType(LIBRARY)
                .dtoIdentifier(dataPointId)
                .locale(getLocale())
                .nhsStyle(sessionUtil.renderNhsStyle())
                .build();
        dto = editPrivacyService.showPrivacyForm(request);
        dto.getPrivacyFlags().forEach(this::fillPrivacyFlagsFrom);
        return getActionResult(dto);
    }

    @SkipValidation
    public String editPrivacySave() {
        var request = ImmutableSaveRequest.builder()
                .eHRRequestContext(getLoggedInEHRRequestContext())
                .dtoType(LIBRARY)
                .dtoIdentifier(dataPointId)
                .privacy(privacy)
                .locale(getLocale())
                .nhsStyle(sessionUtil.renderNhsStyle())
                .build();
        dto = editPrivacyService.savePrivacy(request);
        return getActionResult(dto);
    }

    @SkipValidation
    public String editPrivacyCancel() {
        return CANCEL;
    }

    public String getTab() {
        return dto.getTab();
    }

    public String getSubTab() {
        return dto.getSubTab();
    }

    public Long getDataPointId() {
        return dataPointId;
    }

    public void setDataPointId(Long dataPointId) {
        this.dataPointId = dataPointId;
    }
}
