package com.pkb.action.library;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.institute.entity.InstituteLibraryEntry;
import org.apache.http.client.utils.URIBuilder;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.invoke.MethodHandles;
import java.net.URISyntaxException;
import java.net.URLEncoder;
import java.util.Optional;

import static java.nio.charset.StandardCharsets.UTF_8;
import static org.apache.commons.lang3.StringUtils.defaultIfBlank;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class LibraryUtil {
    protected static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
    private static final String PATH_SHOW_LIBRARY_FOLDER = "/library/showFolder!show.action";
    private static final String PARAM_TKID = "tkid";
    private static final String PARAM_FOLDER_NAME = "folderName";
    private static final String PARAM_TEAM_ID = "teamId";
    private static final String PARAM_REFERER = "referer";
    private static final String PARAM_CONTEXT_USER_ID = "contextUserId";
    private static final String PARAM_TEAM_CODE = "instituteCode";

    public static String buildLibraryFolderURL(InstituteLibraryEntry libraryEntry,
                                               String referer) {
        return buildLibraryFolderURL(libraryEntry, referer, Optional.empty(), Optional.empty());
    }

    public static String buildLibraryFolderURL(InstituteLibraryEntry libraryEntry,
                                               String referer,
                                               Optional<String> teamCode) {
        return buildLibraryFolderURL(libraryEntry, referer, Optional.empty(), Optional.empty(), teamCode);
    }

    public static String buildLibraryFolderURL(InstituteLibraryEntry libraryEntry,
                                        String referer,
                                        Optional<Long> teamId,
                                        Optional<Long> contextUserId) {
        return buildLibraryFolderURL(libraryEntry, referer, teamId, contextUserId, Optional.empty());
    }

    @VisibleForTesting
    static String buildLibraryFolderURL(InstituteLibraryEntry libraryEntry,
                                                String referer,
                                                Optional<Long> teamId,
                                                Optional<Long> contextUserId,
                                                Optional<String> teamCode) {
        try {
            URIBuilder uriBuilder = new URIBuilder(PATH_SHOW_LIBRARY_FOLDER);
            uriBuilder.addParameter(PARAM_TKID, String.valueOf(libraryEntry.getId()));
            if (isNotBlank(referer)) {
                uriBuilder.addParameter(PARAM_REFERER, referer);
            }
            teamId.ifPresent(id -> uriBuilder.addParameter(PARAM_TEAM_ID, String.valueOf(id)));
            contextUserId.ifPresent(id -> uriBuilder.addParameter(PARAM_CONTEXT_USER_ID, String.valueOf(id)));
            teamCode.ifPresent(code -> uriBuilder.addParameter(PARAM_TEAM_CODE, code));
            return uriBuilder.toString();
        } catch (URISyntaxException e) {
            LOGGER.error("Cannot build folder URL for library entry ID=[{}]", libraryEntry.getId(), e);
        }

        return defaultIfBlank(libraryEntry.getUrl(), "#");
    }

    /**
     * Encodes the last path segment of a library file URL which contains a hash and the file name.
     *
     * Files uploaded to libraries can contain a variety of special characters that can be reserved, unsafe from an
     * URL perspective. Our file storage service (Amazon S3) stores the uploaded files regardless of the characters but
     * it causes issues when we try to access them with the original characters in the file name.
     * Accessing the files on the storage in an URL encoded format is the reliable way to do this.
     *
     * @param urlString the original library entry URL
     * @return the library entry URL with an URL encoded file path segment
     */
    public static String buildLibraryFileURL(String urlString) {
        int indexOfLastPathSeparator = urlString.lastIndexOf("/") + 1;
        String urlWithoutFileName = urlString.substring(0, indexOfLastPathSeparator);
        String fileName = urlString.substring(indexOfLastPathSeparator);
        String encodedFileName = URLEncoder.encode(fileName, UTF_8);
        return urlWithoutFileName + encodedFileName;
    }
}
