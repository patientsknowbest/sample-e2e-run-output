package com.pkb.service.dataupload.emis.esv2.batch.impl.observation;

import com.pkb.app.dto.DataPointSaveResult;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.emis.es.entity.EmisEsObservationDelta;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.institute.entity.Org;
import com.pkb.service.dataupload.emis.esv2.batch.EmisRequestContextProvider;
import com.pkb.service.dataupload.emis.esv2.batch.impl.EmisEsDateTimeProcessor;
import com.pkb.service.dataupload.emis.esv2.model.AdminPatient;
import com.pkb.service.dataupload.emis.esv2.model.AdminUserInRole;
import com.pkb.service.dataupload.emis.esv2.model.CareRecordProblem;
import com.pkb.service.emis.EmisEsObservationDeltaManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.springframework.stereotype.Component;

import java.util.Date;
import java.util.List;
import java.util.UUID;

@Component
public class ObservationDeltaHandler extends CareRecordProblemHandler<EmisEsObservationDelta> implements EmisEsDateTimeProcessor, EmisRequestContextProvider {

    private EmisEsObservationDeltaManager emisEsObservationDeltaManager;

    public ObservationDeltaHandler(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, EmisEsObservationDeltaManager emisEsObservationDeltaManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.emisEsObservationDeltaManager = emisEsObservationDeltaManager;
    }

    @Override
    protected DataPointSaveResult<EmisEsObservationDelta> saveWithExternalIds(@NotNull List<Tuple2<EmisEsObservationDelta, EHRRequestContext>> careRecordProblems) {
        return emisEsObservationDeltaManager.addToMultipleAccounts(careRecordProblems);
    }

    @Override
    protected Tuple2<EmisEsObservationDelta, EHRRequestContext> create(@NotNull CareRecordProblem careRecordProblem,
                                                                       @NotNull EHRRequestContext requestContext,
                                                                       @NotNull EmisEsObservationDelta.DeltaType deltaType,
                                                                       Org org, long patientId, @Nullable AdminPatient adminPatient, long defaultAccountId,
                                                                       @Nullable AdminUserInRole enteredBy, UUID parentUniqueId) {
        EmisEsObservationDelta emisEsObservationDelta = new EmisEsObservationDelta(deltaType, parentUniqueId, careRecordProblem.isHandleAsDeleted(adminPatient), initializeSourceDetails(requestContext, enteredBy));
        emisEsObservationDelta.setPatientAccountId(defaultAccountId);
        emisEsObservationDelta.setProblemStatusDescription(careRecordProblem.problemStatusDescription().map(StringUtils::trimToNull).orElse(null));
        // current date
        emisEsObservationDelta.getBaseFields().setEnteredDate(Date.from(dateTimeService.now()));
        //TODO: Shouldn't we set more specific flag other than general ?
        emisEsObservationDelta.getBaseFields().setPrivacyFlag(PrivacyFlag.GENERAL);
        emisEsObservationDelta.setComment(careRecordProblem.comment().map(StringUtils::trimToNull).orElse(null));
        emisEsObservationDelta.setEndDate(parseDate(careRecordProblem.endDate(), org)
                .map(Date::from)
                .orElse(null));
        return Tuple.of(emisEsObservationDelta, requestContext);
    }
}
