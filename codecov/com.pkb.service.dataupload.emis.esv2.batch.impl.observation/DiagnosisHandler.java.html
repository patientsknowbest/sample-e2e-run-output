@Component
public class DiagnosisHandler extends ObservationHandler<Diagnosis> {

    @Autowired
    private CareRecordProblemCache problemCache;

    private EHRDiagnosisManager ehrDiagnosisManager;

    public DiagnosisHandler(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, EHRDiagnosisManager ehrDiagnosisManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.ehrDiagnosisManager = ehrDiagnosisManager;
    }

    @Override
    public CodingDataType getHandledDataType() {
        return CodingDataType.DIAGNOSIS;
    }

    @Override
    protected DataPointSaveResult<Diagnosis> saveWithExternalIds(@NotNull List<Tuple3<Diagnosis, CodingReceived, EHRRequestContext>> observationsToSave) {
        return ehrDiagnosisManager.addToMultipleAccountsNoCodingLookup(observationsToSave);
    }

    @Override
    protected Set<UUID> checkForDuplicates(io.vavr.collection.List<DeduplicationCandidateDto<Diagnosis>> candidates,
                                           Map<Long, AccountKeysDTO> accountIdToAccountKeys) {
        return ehrDiagnosisManager.checkForDuplicatesWithCoding(candidates, accountIdToAccountKeys);
    }

    @Override
    protected Diagnosis create(
            CareRecordObservation observation,
            UUID uniqueId,
            Org org,
            long patientId,
            @Nullable AdminPatient adminPatient,
            long defaultAccountId,
            @Nullable AdminUserInRole enteredBy,
            CodingReceived codingReceived,
            CodeableConcept codeableConcept,
            EmisEsProcessedRow fileState,
            EHRRequestContext recordContext,
            @Nullable LoincMapping loincMapping) {
        Diagnosis diagnosis = new Diagnosis(initializeSourceDetails(recordContext, enteredBy));
        diagnosis.setPatientAccountId(defaultAccountId);

        diagnosis.getBaseFields().setUniqueId(uniqueId);
        diagnosis.getBaseFields().setPrivacyFlag(PrivacyFlag.GENERAL);
        diagnosis.setClinicalStatus(DiagnosisClinicalStatus.ACTIVE);

        Optional<Instant> maybeEffectiveDate = parseDate(observation.effectiveDate(), org);
        ResultWithId<CareRecordProblem, Long> problem = problemCache.get(fileState.getBatchId(), observation);
        if (problem != null && !problem.getValue().isHandleAsDeleted(adminPatient)) {
            diagnosis.setStartDate(maybeEffectiveDate.map(Date::from).orElse(null));
            problem.getValue().endDate().ifPresent(ld -> diagnosis.setEndDate(Date.from(ld.atStartOfDay(ZoneId.of(org.getTimeZoneId())).toInstant())));
            handleProblemStatusDescription(problem.getValue(),
                    status -> diagnosis.setClinicalStatus(DiagnosisClinicalStatus.ACTIVE),
                    status -> diagnosis.setClinicalStatus(DiagnosisClinicalStatus.INACTIVE));
            diagnosis.setCategory(ConditionCategory.PROBLEM_LIST_ITEM);
        } else {
            // no problem link -- this is likely the review or flare-up date; we can't display as "onset", just observed/noted
            diagnosis.setDateAsserted(maybeEffectiveDate.map(Date::from).orElse(null));
            // override default category -- this is NOT a problem-list-item and we don't display it
            diagnosis.setCategory(null);
        }

        // a deleted problem doesn't affect the data; but still mark it processed
        if (problem != null) {
            problemCache.markProcessed(fileState.getBatchId(), observation);
        }

        diagnosis.setDiagnosis(codeableConcept);
        diagnosis.setDiagnosisCodingReceivedId(codingReceived.getId());

        mapEnteredDate(diagnosis, observation, org);
        return diagnosis;
    }
}
