package com.pkb.service.dataupload.emis.esv2.batch.impl.observation;

@Component
public class AllergyHandler extends ObservationHandler<Allergy> {

    @Autowired
    private CareRecordProblemCache problemCache;

    private EHRAllergyManager ehrAllergyManager;

    @Override
    public CodingDataType getHandledDataType() {
        return CodingDataType.ALLERGY_DISORDER;
    }

    public AllergyHandler(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, EHRAllergyManager ehrAllergyManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.ehrAllergyManager = ehrAllergyManager;
    }

    @Override
    protected DataPointSaveResult<Allergy> saveWithExternalIds(@NotNull List<Tuple3<Allergy, CodingReceived, EHRRequestContext>> observationsToSave) {
        return ehrAllergyManager.addToMultipleAccountsNoCodingLookup(observationsToSave);
    }

    @Override
    protected Set<UUID> checkForDuplicates(
            io.vavr.collection.List<DeduplicationCandidateDto<Allergy>> candidates,
            Map<Long, AccountKeysDTO> accountIdToAccountKeys) {

        return ehrAllergyManager.checkForDuplicatesWithCoding(candidates, accountIdToAccountKeys);
    }

    @Override
    protected Allergy create(
            CareRecordObservation observation,
            UUID uniqueId,
            Org org,
            long patientId,
            @Nullable AdminPatient adminPatient,
            long defaultAccountId,
            @Nullable AdminUserInRole enteredBy,
            CodingReceived codingReceived,
            CodeableConcept codeableConcept,
            EmisEsProcessedRow fileState,
            EHRRequestContext recordContext,
            @Nullable LoincMapping loincMapping) {
        Allergy allergy = new Allergy(initializeSourceDetails(recordContext, enteredBy));
        allergy.setPatientIsAllergic(true);
        allergy.setPatientAccountId(defaultAccountId);

        allergy.getBaseFields().setUniqueId(uniqueId);

        mapEnteredDate(allergy, observation, org);

        Optional<Instant> maybeEffectiveDate = parseDate(observation.effectiveDate(), org);
        ResultWithId<CareRecordProblem, Long> problem = problemCache.get(fileState.getBatchId(), observation);
        if (problem != null && !problem.getValue().isHandleAsDeleted(adminPatient)) {
            allergy.setOnsetOfSymptoms(maybeEffectiveDate.map(Date::from).orElse(null));
            problem.getValue().endDate()
                    .ifPresent(ld -> allergy.setEndOfSymptoms(parseDate(ld, org)
                            .map(Date::from)
                            .orElse(null)));
            handleProblemStatusDescription(problem.getValue(),
                    status -> allergy.setStatus(Allergy.AllergyStatus.ACTIVE),
                    status -> allergy.setStatus(Allergy.AllergyStatus.INACTIVE));
        } else {
            allergy.setDateAsserted(maybeEffectiveDate.map(Date::from).orElse(null));
        }

        // a deleted problem doesn't affect the data; but still mark it processed
        if (problem != null) {
            problemCache.markProcessed(fileState.getBatchId(), observation);
        }

        allergy.setDisorder(codeableConcept);
        allergy.setDisorderCodingReceivedId(codingReceived.getId());

        return allergy;
    }
}
