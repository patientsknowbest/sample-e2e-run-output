package com.pkb.service.dataupload.emis.esv2.batch.impl.observation;

import java.util.List;
import java.util.UUID;

public abstract class CareRecordProblemHandler<T> extends TransactionManager implements EmisRequestContextProvider {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    CareRecordProblemHandler(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider) {
        super(config, beanFactory, dateTimeService, uuidProvider);
    }

    public DataPointSaveResult<T> handle(
            Set<CareRecordProblem> careRecordProblems,
            Map<CareRecordProblem, EmisEsObservationDelta.DeltaType> recordToDeltaType,
            Map<CareRecordProblem, Org> recordToOrg,
            Map<CareRecordProblem, PKBPersonInfo> recordToPatient,
            Map<CareRecordProblem, AdminPatient> adminPatients,
            Map<CareRecordProblem, Long> recordToDefaultAccountId,
            Map<CareRecordProblem, UUID> recordToParentUniqueId) {

        var pkbObservations = careRecordProblems.map(crp -> {
            var recordContext = getEmisContextWithDefaultTeam(recordToOrg.get(crp).getOrNull());
            return create(crp, recordContext, recordToDeltaType.apply(crp), recordToOrg.apply(crp), recordToPatient.apply(crp).id(),
                    adminPatients.get(crp).getOrNull(), recordToDefaultAccountId.apply(crp), null,
                    recordToParentUniqueId.apply(crp));
        }).toJavaList();

        LOGGER.info("Saving {} CareRecordProblems w/o CareRecordObservation", pkbObservations.size());
        return transactional(() ->
                saveWithExternalIds(pkbObservations)
        );
    }

    protected abstract DataPointSaveResult<T> saveWithExternalIds(@NotNull List<Tuple2<T, EHRRequestContext>> careRecordProblems);

    protected abstract Tuple2<T, EHRRequestContext> create(
            CareRecordProblem careRecordProblem,
            @NotNull EHRRequestContext requestContext,
            DeltaType deltaType,
            Org org,
            long patientId,
            @Nullable AdminPatient adminPatient,
            long defaultAccountId,
            @Nullable AdminUserInRole enteredBy,
            UUID parentUniqueId);
}
