@Component
public class MeasurementHandler extends ObservationHandler<MeasurementDTO> {

    @Autowired
    private CareRecordProblemCache problemCache;

    private EHRMeasurementManager ehrMeasurementManager;

    @Override
    public CodingDataType getHandledDataType() {
        return CodingDataType.MEASUREMENT;
    }

    public MeasurementHandler(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, EHRMeasurementManager ehrMeasurementManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.ehrMeasurementManager = ehrMeasurementManager;
    }

    @Override
    protected DataPointSaveResult<MeasurementDTO> saveWithExternalIds(@NotNull List<Tuple3<MeasurementDTO, CodingReceived, EHRRequestContext>> observationsToSave) {
        return ehrMeasurementManager.addToMultipleAccountsNoCodingLookup(observationsToSave);
    }

    @Override
    protected Set<UUID> checkForDuplicates(
            io.vavr.collection.List<DeduplicationCandidateDto<MeasurementDTO>> candidates,
            Map<Long, AccountKeysDTO> accountIdToAccountKeys) {

        return ehrMeasurementManager.checkForDuplicatesWithCoding(candidates, accountIdToAccountKeys);
    }

    @Override
    protected MeasurementDTO create(
            CareRecordObservation observation,
            UUID uniqueId,
            Org org,
            long patientId,
            @Nullable AdminPatient adminPatient,
            long defaultAccountId,
            @Nullable AdminUserInRole enteredBy,
            CodingReceived codingReceived,
            CodeableConcept codeableConcept,
            EmisEsProcessedRow fileState,
            EHRRequestContext recordContext,
            @Nullable LoincMapping loincMapping) {
        MeasurementDTO measurement = new MeasurementDTO(initializeSourceDetails(recordContext, enteredBy));
        measurement.getBaseFields().setUniqueId(uniqueId);
        measurement.getBaseFields().setPrivacyFlag(codingReceived.getCodingMatch().getPrivacyFlag());

        measurement.setPatientAccountId(defaultAccountId);

        var unit = observation.numericUnit().map(EmisEsUnitUtil::sanitizeUnit).orElse("");
        var pkbInternalTypeMaybe = PredefinedMeasurementType.findByConceptAndUnit(codeableConcept, unit);
        pkbInternalTypeMaybe.peek(measurement::setMeasurementType);
        measurement.setUnit(unit);
        measurement.setCodingReceivedId(codingReceived.getId());
        measurement.setCodeableConcept(codeableConcept);
        if(pkbInternalTypeMaybe.isEmpty()) {
            measurement.setCodingMatchId(codingReceived.getCodingMatch().getId());
        }

        Optional<Instant> maybeEffectiveDate = parseDate(observation.effectiveDate(), org);
        maybeEffectiveDate.ifPresent(date -> measurement.setMeasureDate(date));

        if(observation.value().isPresent()) {
            measurement.setValue(observation.value().getAsDouble());
        }

        // product decision was made to do nothing with problems for now
        ResultWithId<CareRecordProblem, Long> problem = problemCache.get(fileState.getBatchId(), observation);
        if (problem != null) {
            problemCache.markProcessed(fileState.getBatchId(), observation);
        }

        mapEnteredDate(measurement, observation, org);
        return measurement;
    }
}
