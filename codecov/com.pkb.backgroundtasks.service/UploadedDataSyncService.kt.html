package com.pkb.backgroundtasks.service

import com.pkb.backgroundtasks.config.BackgroundTasksProperties
import com.pkb.backgroundtasks.repository.UploadedDataRepository
import io.micrometer.core.annotation.Timed
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.data.domain.PageRequest
import org.springframework.data.domain.Pageable
import org.springframework.scheduling.annotation.Scheduled
import org.springframework.stereotype.Service
import org.springframework.transaction.annotation.Transactional
import java.lang.invoke.MethodHandles

//TODO Temporary Task: remove when uploaded data phase 2 is implemented (OMNI-63)
@Service
class UploadedDataSyncService(
        private val uploadedDataRepository: UploadedDataRepository,
        private val properties: BackgroundTasksProperties,
        private val uploadedDataSyncClient: UploadedDataSyncClient
) {
    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())
    @Transactional
    @Scheduled(fixedDelayString = "\${backgroundtasks.uploadeddata.fixed-delay-ms}", initialDelayString = "\${backgroundtasks.uploadeddata.initial-delay-ms}")
    @Timed(value = "backgroundtasks-uploadeddatasync", longTask = true, histogram = true)
    fun syncUploadedData() {
        if (properties.uploadedData.enabled) {
            log.info("Starting uploadeddata sync")
            var pageable: Pageable = PageRequest.of(0, properties.uploadedData.batchSize)
            while (true) {
                log.info("Starting uploadeddata sync batch ${pageable.pageNumber}")
                val slice = uploadedDataRepository.findUploadedDataSync(pageable, properties.uploadedData.destinations)
                val dataSyncs = slice.content.distinct()
                uploadedDataSyncClient.makeUploadRequests(dataSyncs)
                if (!slice.hasNext()) {
                    break;
                }
                pageable = slice.nextPageable()
            }
            log.info("Finished uploadeddata sync")
        }
    }
}
