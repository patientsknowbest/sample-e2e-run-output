package com.pkb.backgroundtasks.service

import com.pkb.backgroundtasks.OutOfAreaActivationNotificationDTO
import com.pkb.backgroundtasks.PhrClient
import com.pkb.backgroundtasks.repository.AccountRepository
import com.pkb.backgroundtasks.repository.OutOfAreaRepairJobRepository
import com.pkb.backgroundtasks.repository.PKBPersonRepository
import com.pkb.common.datetime.DateTimeService
import com.pkb.entities.enums.OutOfAreaRepairJobStatus
import com.pkb.entities.enums.OutOfAreaRepairJobStatus.FAILED
import com.pkb.entities.enums.OutOfAreaRepairJobStatus.REPAIRED
import com.pkb.entities.pub.PublicOutOfAreaRepairJob
import com.pkb.kms.client.core.Kms
import com.pkb.kms.shared.representation.GrantUserAccessToAccountRequest
import io.micrometer.core.annotation.Timed
import io.micrometer.core.instrument.Metrics.counter
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.scheduling.annotation.Scheduled
import org.springframework.stereotype.Service
import org.springframework.transaction.annotation.Transactional
import java.lang.invoke.MethodHandles
import java.util.UUID

@Service
class OwnAccessRepairService(private val personRepository: PKBPersonRepository,
                             private val accountRepository: AccountRepository,
                             private val jobRepository: OutOfAreaRepairJobRepository,
                             private val kms: Kms,
                             private val phr: PhrClient,
                             private val dateTimeService: DateTimeService) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @Transactional()
    @Scheduled(fixedDelayString = "\${backgroundtasks.own-access-repair.fixed-delay-ms}")
    @Timed(histogram = true)
    fun tryRepairAll() {
        log.trace("Repair scheduled task triggered")

        jobRepository.findNewOrFailedJobs()
                .map { tryRepair(it) }
    }

    private fun tryRepair(job: PublicOutOfAreaRepairJob) {
        Pair(personRepository.findById(job.targetPersonId), accountRepository.findById(job.accountId))
                .takeIf { it.first.isPresent && it.second.isPresent }
                ?.let {
                    GrantUserAccessToAccountRequest(
                            it.second.get().publicId,
                            it.first.get().publicId
                    )
                }?.run {
                    val requestId = UUID.randomUUID().toString()

                    kms.grantAccessToAccountOwner(requestId, this)
                            .peekLeft {
                                updateJobStatus(job, FAILED)
                                counter("access.granted", "outcome", "FAILURE").increment()
                                log.debug("Unable to grant owner access: $it")
                            }
                            .peek {
                                counter("access.granted", "outcome", "SUCCESS").increment()
                                val dto = OutOfAreaActivationNotificationDTO(this.targetPersonPublicId)
                                phr.notifyOutOfAreaPatientAccountActivation(requestId, dto).execute()
                                updateJobStatus(job, REPAIRED)
                                log.info("Access to account ${this.accountPublicId} granted to owner ${this.targetPersonPublicId}")
                            }
                }

    }

    private fun updateJobStatus(job: PublicOutOfAreaRepairJob, status: OutOfAreaRepairJobStatus): PublicOutOfAreaRepairJob {
        job.status = status
        job.statusDate = dateTimeService.nowLocalDateTime()

        return jobRepository.save(job)
    }
}
