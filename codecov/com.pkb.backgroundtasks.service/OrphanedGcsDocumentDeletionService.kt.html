package com.pkb.backgroundtasks.service

import com.pkb.backgroundtasks.PhrDocumentMetadataClient
import com.pkb.backgroundtasks.config.BackgroundTasksProperties
import io.micrometer.core.annotation.Timed
import okhttp3.MediaType.Companion.toMediaType
import okhttp3.RequestBody.Companion.toRequestBody
import org.springframework.scheduling.annotation.Scheduled
import org.springframework.stereotype.Service
import retrofit2.Call
import java.util.UUID

@Service
class OrphanedGcsDocumentDeletionService(
    private val phr: PhrDocumentMetadataClient,
    private val config: BackgroundTasksProperties
) {
    @Scheduled(fixedDelayString = "\${backgroundtasks.deleteOrphanedGcsDocuments.fixedDelay}", initialDelayString = "\${backgroundtasks.deleteOrphanedGcsDocuments.initialDelay}")
    @Timed(value = "backgroundtasks-delete-orphaned-gcs-documents", longTask = true, histogram = true)
    fun markDocumentsForDeletion() {
        if (config.deleteOrphanedGcsDocuments.enabled) {
            getPhrOrphanedGcsDocumentDeletionCall().execute()
        }
    }

    fun getPhrOrphanedGcsDocumentDeletionCall(): Call<Void> {
        return phr.deleteOrphanedGcsDocuments(
            UUID.randomUUID().toString(),
            "".toRequestBody("application/octet-stream".toMediaType())
        )
    }
}
