package com.pkb.backgroundtasks.service

import com.pkb.backgroundtasks.InvitationActivationNotificationDTO
import com.pkb.backgroundtasks.PhrClient
import com.pkb.backgroundtasks.repository.InvitationRepairJobRepository
import com.pkb.common.datetime.DateTimeService
import com.pkb.entities.enums.InvitationRepairJobStatus
import com.pkb.entities.enums.InvitationRepairJobStatus.FAILED
import com.pkb.entities.enums.InvitationRepairJobStatus.REPAIRED
import com.pkb.entities.pub.PublicInvitationRepairJob
import com.pkb.kms.client.core.Kms
import com.pkb.kms.shared.representation.GrantUserAccessToAccountRequest
import io.micrometer.core.annotation.Timed
import io.micrometer.core.instrument.Metrics.counter
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.scheduling.annotation.Scheduled
import org.springframework.stereotype.Service
import org.springframework.transaction.annotation.Transactional
import java.lang.invoke.MethodHandles
import java.util.UUID


@Service
class InvitationRepairService(private val jobRepository: InvitationRepairJobRepository,
                              private val kms: Kms,
                              private val phr: PhrClient,
                              private val dateTimeService: DateTimeService) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @Transactional()
    @Scheduled(fixedDelayString = "\${backgroundtasks.invitation-repair.fixed-delay-ms}")
    @Timed(histogram = true)
    fun tryRepairAll() {
        log.trace("Invitation repair scheduled task triggered")

        jobRepository.findNewOrFailedJobs()
                .map { tryRepair(it) }
    }

    private fun tryRepair(job: PublicInvitationRepairJob) {
        val request = GrantUserAccessToAccountRequest(job.patientAccountPublicId, job.patientPublicId)
        val requestId = UUID.randomUUID().toString()

        kms.grantAccessToAccountOwner(requestId, request)
                .peekLeft {
                    updateJobStatus(job, FAILED)
                    counter("invitation.access.granted", "outcome", "FAILURE").increment()
                    log.debug("Unable to grant owner access: $it")
                }
                .peek {
                    counter("invitation.access.granted", "outcome", "SUCCESS").increment()
                    val dto = InvitationActivationNotificationDTO(job.patientId, job.teamId)
                    phr.notifyInvitationActivation(requestId, dto).execute()
                    updateJobStatus(job, REPAIRED)
                    log.info("Access to account ${request.accountPublicId} granted to owner ${request.targetPersonPublicId}")
                }
    }

    private fun updateJobStatus(job: PublicInvitationRepairJob, status: InvitationRepairJobStatus): PublicInvitationRepairJob {
        job.status = status
        job.statusDate = dateTimeService.nowLocalDateTime()

        return jobRepository.save(job)
    }
}
