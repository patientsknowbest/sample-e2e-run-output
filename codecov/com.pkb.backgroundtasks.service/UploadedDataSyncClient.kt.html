package com.pkb.backgroundtasks.service

import com.pkb.backgroundtasks.PhrSyncUploadedDataClient
import com.pkb.backgroundtasks.projection.UploadedDataSync
import com.pkb.backgroundtasks.repository.UploadedDataRepository
import com.pkb.crypto.UserPrivateKey.userPrivateKey
import com.pkb.crypto.UserPublicKey
import com.pkb.dto.ImmutableSynchUploadedDataDTO
import okhttp3.MediaType.Companion.toMediaType
import okhttp3.RequestBody.Companion.toRequestBody
import org.apache.commons.lang3.SerializationUtils
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import retrofit2.Call
import java.lang.invoke.MethodHandles
import java.util.UUID
import java.util.concurrent.Callable
import java.util.concurrent.ExecutorService

@Service
class UploadedDataSyncClient(
        private val phr: PhrSyncUploadedDataClient,
        private val uploadedDataRepository: UploadedDataRepository,
        private val uploadedDataApiCallExecutor: ExecutorService
) {
    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    fun makeUploadRequests(uploadedataSyncs: List<UploadedDataSync>) {
        val tasks = uploadedataSyncs
                .map { sync ->
                    Callable {
                        // calling sync once per person + destination
                        try {
                            getPhrSyncCall(sync).execute()
                        } catch (t: Throwable) {
                            log.error("Failed sync for ${sync.personId}", t)
                        }
                    }
                }
        uploadedDataApiCallExecutor.invokeAll(tasks)
    }

    private fun getPhrSyncCall(uploadedData: UploadedDataSync): Call<Void> {
        // TODO Using Dummy Keys is a temporary solution. This logic should be removed when UploadedData Phase 2 is implemented
        val privKey = userPrivateKey("dummy".toByteArray(), "RSA")
        val publKey = UserPublicKey.userPublicKey("dummy".toByteArray(), "RSA")
        return phr.syncUploadedData(
                requestID = UUID.randomUUID().toString(),
                request = SerializationUtils.serialize(
                        ImmutableSynchUploadedDataDTO.builder()
                                .authenticatedUserId(uploadedData.personId)
                                .addTargetPersonIds(uploadedData.personId)
                                .userPublicKey(publKey)
                                .synchUploadType(uploadedData.destination)
                                .userPrivateKey(privKey)
                                .isBgTask(true)
                                .build()
                ).toRequestBody(contentType = "application/octet-stream".toMediaType())
        )
    }
}
