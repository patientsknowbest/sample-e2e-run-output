package com.pkb.backgroundtasks.service

import com.pkb.backgroundtasks.config.BackgroundTasksProperties
import com.pkb.backgroundtasks.repository.ApiAuthSessionRepository
import com.pkb.common.datetime.DateTimeService
import io.micrometer.core.annotation.Timed
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.scheduling.annotation.Scheduled
import org.springframework.stereotype.Service
import org.springframework.transaction.annotation.Transactional
import java.lang.invoke.MethodHandles
import java.time.temporal.ChronoUnit.DAYS

@Service
class ApiAuthSessionCleanUp(private val apiAuthSessionRepository: ApiAuthSessionRepository,
                            private val dateTimeService: DateTimeService,
                            private val properties: BackgroundTasksProperties
) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @Transactional
    @Scheduled(fixedDelayString = "\${backgroundtasks.api-auth-session-clean-up.fixed-delay-ms}")
    @Timed(value = "apiauthsession_cleanup", longTask = true, histogram = true)
    fun cleanUp() {
        log.trace("Clean up task for ApiAuthSession triggered.")

        val date = dateTimeService.now().minus(properties.apiAuthSessionCleanUp.retentionPeriodDays, DAYS)

        properties.apiAuthSessionCleanUp.clientApiIds.forEach {
            val numberOfDeletedSessions = apiAuthSessionRepository.deleteSessionByClientApiIdCreatedBefore(it, date)

            log.info("Cleaned up {} session(s) for client API {}.", numberOfDeletedSessions, it)
        }
    }

}
