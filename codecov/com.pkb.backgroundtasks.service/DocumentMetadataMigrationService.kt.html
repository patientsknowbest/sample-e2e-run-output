package com.pkb.backgroundtasks.service

import com.pkb.backgroundtasks.PhrDocumentMetadataClient
import com.pkb.backgroundtasks.config.BackgroundTasksProperties
import com.pkb.backgroundtasks.repository.DocumentMetadataRepository
import com.pkb.datamodel.DocumentMetadataIds
import io.micrometer.core.annotation.Timed
import okhttp3.MediaType.Companion.toMediaType
import okhttp3.RequestBody.Companion.toRequestBody
import org.apache.commons.lang3.SerializationUtils
import org.springframework.data.domain.PageRequest
import org.springframework.scheduling.annotation.Scheduled
import org.springframework.stereotype.Service
import org.springframework.transaction.annotation.Transactional
import retrofit2.Call
import java.util.UUID

@Service
class DocumentMetadataMigrationService(
    private val phr: PhrDocumentMetadataClient,
    private val documentMetadataRepository: DocumentMetadataRepository,
    private val config: BackgroundTasksProperties
) {
    @Transactional
    @Scheduled(fixedDelayString = "\${backgroundtasks.migrateDocuments.fixedDelay}", initialDelayString = "\${backgroundtasks.migrateDocuments.initialDelay}")
    @Timed(value = "backgroundtasks-migrate-documents", longTask = true, histogram = true)
    fun migrateDocuments() {
        if (config.migrateDocuments.enabled) {
            val uuids = documentMetadataRepository.findDocumentMetadataIds(PageRequest.of(0, config.migrateDocuments.batchSize))
                .distinct()

            if (uuids.isEmpty()) {
                return
            }

            getPhrMigrateCall(uuids).execute()
        }
    }

    fun getPhrMigrateCall(uuids: Collection<UUID>): Call<Void> {
        val content = SerializationUtils.serialize(DocumentMetadataIds(uuids))
        return phr.migrateDocuments(
            UUID.randomUUID().toString(),
            content.toRequestBody("application/octet-stream".toMediaType(), 0, content.size)
        )
    }
}
