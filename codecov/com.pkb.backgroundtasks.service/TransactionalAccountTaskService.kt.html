package com.pkb.backgroundtasks.service

import com.pkb.backgroundtasks.AccountPublicId
import com.pkb.backgroundtasks.entity.AccountTaskComplete
import com.pkb.backgroundtasks.entity.AccountTaskPending
import com.pkb.backgroundtasks.entity.AccountWithoutKey
import com.pkb.backgroundtasks.repository.AccountTaskCompleteRepository
import com.pkb.backgroundtasks.repository.AccountTaskPendingRepository
import com.pkb.backgroundtasks.repository.AccountWithoutKeyRepository
import org.springframework.stereotype.Service
import org.springframework.transaction.annotation.Transactional

@Service
class TransactionalAccountTaskService(
    private val accountWithoutKeyRepo: AccountWithoutKeyRepository,
    private val completeRepo: AccountTaskCompleteRepository,
    private val pendingRepo: AccountTaskPendingRepository
) {

    @Transactional
    fun queueTask(task: AccountTaskPending) =
        pendingRepo.save(task)

    @Transactional
    fun deleteAllTasks() {
        completeRepo.deleteAllInBatch()
        pendingRepo.deleteAllInBatch()
    }

    @Transactional
    fun deleteTasksFor(accountPublicId: AccountPublicId) {
        completeRepo.deleteByAccountPublicId(accountPublicId)
        pendingRepo.deleteByAccountPublicId(accountPublicId)
    }

    @Transactional
    fun deleteAllAccountWithoutKeyRecords() = accountWithoutKeyRepo.deleteAll()

    @Transactional
    fun recordCompletion(completed: List<Pair<AccountTaskPending, TaskResult>>, requestId: String) {
        completed
            .partition { it.second != TaskResult.KEY_NOT_AVAILABLE }
            .let { (toSave, missingKeys) ->
                if (!toSave.isEmpty()) {
                    completeRepo.saveAll(toSave.map { AccountTaskComplete(it.first, it.second, requestId) })
                    pendingRepo.deleteAllById(toSave.map { it.first.id })
                }
                if (!missingKeys.isEmpty()) {
                    val missingKeyAccounts = missingKeys.map { it.first.accountPublicId }.distinct()
                    accountWithoutKeyRepo.deleteAllByIdInBatch(missingKeyAccounts)
                    accountWithoutKeyRepo.saveAll(missingKeyAccounts.map { AccountWithoutKey(it) })
                }
            }
    }

    @Transactional
    fun deleteAccountWithoutKeyRecordForAccount(accountPublicId: AccountPublicId) {
        accountWithoutKeyRepo.deleteAllByIdInBatch(listOf(accountPublicId))
    }

}
