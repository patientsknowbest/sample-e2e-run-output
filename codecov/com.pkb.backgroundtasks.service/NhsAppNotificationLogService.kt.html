package com.pkb.backgroundtasks.service

import com.pkb.backgroundtasks.config.BackgroundTasksProperties
import com.pkb.backgroundtasks.repository.NhsAppNotificationLogRepository
import com.pkb.common.datetime.DateTimeService
import org.springframework.scheduling.annotation.Scheduled
import org.springframework.stereotype.Service
import org.springframework.transaction.annotation.Transactional

@Service
class NhsAppNotificationLogService(private val repository: NhsAppNotificationLogRepository,
                                   private val dateTimeService: DateTimeService,
                                   private val backgroundTasksProperties: BackgroundTasksProperties) {
    @Transactional()
    @Scheduled(fixedDelayString = "\${backgroundtasks.nhs-notification-log.expire-log-delay-ms}")
    fun truncateLog(): Int {
        val recordExpiryDate = dateTimeService.now()-backgroundTasksProperties.nhsNotificationRetry.recordExpiryInterval
        return repository.deleteExpiredLogEntries(recordExpiryDate)
    }
}
