package com.pkb.datamodel.crypto;

import com.pkb.crypto.AccountPrivateKey;
import com.pkb.crypto.AccountPublicKey;
import com.pkb.crypto.EncryptedAccountPrivateKey;
import com.pkb.crypto.EncryptedSymmetricKey;
import com.pkb.crypto.SymmetricKey;
import com.pkb.crypto.UserPrivateKey;
import com.pkb.datamodel.AccountKeys;
import com.pkb.entities.app.AppMenuData;
import com.pkb.entities.core.UserAccountKeyLink;
import com.pkb.entities.embeddables.EncodedEncryptedPrivateKeyInfo;
import com.pkb.entities.embeddables.EncryptedData;
import io.vavr.collection.List;
import io.vavr.collection.Map;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.Base64;

import static com.pkb.crypto.CryptoHelper.generateSymmetricKey;
import static com.pkb.crypto.EncryptedAccountPrivateKey.encryptedAccountPrivateKey;
import static com.pkb.crypto.EncryptedSymmetricKey.encryptedSymmetricKey;
import static io.vavr.API.unchecked;
import static java.util.Objects.nonNull;

abstract class AbstractCryptoService {

    public abstract byte[] decrypt(AccountKeys accountKeys, AppMenuData menuData);

    public abstract Map<Long, byte[]> decrypt(AccountKeys accountKeys, List<AppMenuData> menuData);

    public EncryptedData encrypt(AccountPublicKey accountPublicKey, byte[] plaintext) {

        // Generate secret key, and encrypt the plaintext and the key
        SymmetricKey symmetricKey = generateSymmetricKey();
        byte[] ciphertext = symmetricKey.encrypt(plaintext, null);
        byte[] encryptedSecretKey = accountPublicKey.encryptSymmetricKey(symmetricKey);

        // Build response
        EncryptedData data = new EncryptedData();
        data.setCiphertext(ciphertext);
        data.setEncryptedSecretKey(encryptedSecretKey);
        data.setSecretKeyAlgorithm(symmetricKey.getAlgorithm());

        return data;
    }

    public AccountPrivateKey getAccountPrivateKey(Either<UserPrivateKey, AccountPrivateKey> userPrivateKey, 
                                                  UserAccountKeyLink link) {
        // If there is an org key, use it. Otherwise, use the user key
        if (nonNull(link.getOrgAccountPrivateKey())) {
            AccountPrivateKey orgAccountPrivateKey = decryptAccountPrivateKey(userPrivateKey, link.getOrgAccountPrivateKey());
            return decryptAccountPrivateKey(Either.right(orgAccountPrivateKey), link.getAccountPrivateKey());
        } else {
            return decryptAccountPrivateKey(userPrivateKey, link.getAccountPrivateKey());

        }
    }

    public AccountPrivateKey decryptAccountPrivateKey(Either<UserPrivateKey, AccountPrivateKey> userOrAccountPrivateKey, 
                                                      EncodedEncryptedPrivateKeyInfo encryptedPrivateKey) {        
        EncryptedAccountPrivateKey encryptedAccountPrivateKey = getEncryptedAccountPrivateKey(encryptedPrivateKey);
        return userOrAccountPrivateKey.fold(
                userPrivateKey -> userPrivateKey.decryptAccountPrivateKey(encryptedAccountPrivateKey),
                accountPrivateKey -> accountPrivateKey.decryptAccountPrivateKey(encryptedAccountPrivateKey)
        );
    }

    protected abstract byte[] decrypt(AccountPrivateKey accountPrivateKey, AppMenuData menuData);


    @NotNull
    protected byte[] decrypt(SymmetricKey key, AppMenuData md) {
        return key.decrypt(
                md.getXml01(),
                Base64.getDecoder().decode(md.getNonceBase64()));
    }

    @NotNull
    protected SymmetricKey getSymmetricKey(AccountKeys accountKeys, long keyId) {
        return accountKeys.findSymmetricKeyById(keyId)
                .getOrElseThrow(() -> new RuntimeException("Cannot find account symmetric key " + keyId));
    }

    protected EncryptedAccountPrivateKey getEncryptedAccountPrivateKey(EncodedEncryptedPrivateKeyInfo encryptedPrivateKey) {
        return unchecked(() -> {
            EncryptedSymmetricKey encryptedSymmetricKey = encryptedSymmetricKey(
                    encryptedPrivateKey.getEncryptedSecretKey(),
                    encryptedPrivateKey.getSecretKeyAlgorithm());

            return encryptedAccountPrivateKey(
                    encryptedPrivateKey.getEncryptedPrivateKeyInfo(),
                    encryptedPrivateKey.getPrivateKeyAlgorithm(),
                    encryptedSymmetricKey,
                    null);
        }).get();
    }
}
