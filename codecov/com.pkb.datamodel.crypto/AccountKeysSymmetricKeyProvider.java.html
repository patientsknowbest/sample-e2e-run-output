package com.pkb.datamodel.crypto;

import com.pkb.crypto.SymmetricKey;
import com.pkb.crypto.decryptionhelper.dto.specialized.AESEncryptedDataWithSourceRecord;
import com.pkb.crypto.decryptionhelper.provider.SymmetricKeyProvider;
import com.pkb.datamodel.AccountKeys;

public class AccountKeysSymmetricKeyProvider<T> implements SymmetricKeyProvider<T> {

    @SuppressWarnings("WeakerAccess")
    static final String ERROR_ACCOUNT_KEYS_MISSING_SYMMETRIC_KEY = "Account Keys missing the required Symmetric key to decrypt AES for message %s";

    private AccountKeys accountKeys;

    public AccountKeysSymmetricKeyProvider(AccountKeys accountKeys) {
        this.accountKeys = accountKeys;
    }

    @Override
    public SymmetricKey getSymmetricKey(AESEncryptedDataWithSourceRecord<T> aesEncryptedData) {
        return accountKeys.findSymmetricKeyById(aesEncryptedData.getSymmetricEncryptionKeyId())
                .getOrElseThrow(() -> new IllegalStateException(String.format(ERROR_ACCOUNT_KEYS_MISSING_SYMMETRIC_KEY, aesEncryptedData.getDataIdentifier())));
    }

    @Override
    public String getProviderIdentifier() {
        return "AccountKeysSymmetricKeyProvider";
    }

    public AccountKeys getAccountKeys() {
        return accountKeys;
    }
}
