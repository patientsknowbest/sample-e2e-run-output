package com.pkb.datamodel.crypto;

import com.pkb.crypto.AccountPrivateKey;
import com.pkb.crypto.SymmetricKey;
import com.pkb.crypto.decryptionhelper.DecryptionHelperAppMenuData;
import com.pkb.datamodel.AccountKeys;
import com.pkb.entities.app.AppMenuData;
import io.vavr.collection.HashMap;
import io.vavr.collection.List;
import io.vavr.collection.Map;

public class LocalCryptoService extends AbstractCryptoService {
    private DecryptionHelperAppMenuData decryptionHelper;
    private AccountKeysProviderFactory providerFactory;

    public LocalCryptoService(DecryptionHelperAppMenuData decryptionHelper,
                              AccountKeysProviderFactory providerFactory) {
        this.decryptionHelper = decryptionHelper;
        this.providerFactory = providerFactory;
    }

    @Override
    public byte[] decrypt(AccountKeys accountKeys, AppMenuData menuData) {
        AccountKeysSymmetricKeyProvider<AppMenuData> symmetricKeyProvider = providerFactory.symmetricKeyFromAccountKeysProvider(accountKeys);
        AccountKeysAccountPrivateKeyProvider<AppMenuData> accountPrivateKeyProvider = providerFactory.accountPrivateKeyFromAccountKeysProvider(accountKeys);
        return decrypt(menuData, symmetricKeyProvider, accountPrivateKeyProvider);
    }
    
    @Override
    public Map<Long, byte[]> decrypt(AccountKeys accountKeys, List<AppMenuData> menuData) {
        AccountKeysSymmetricKeyProvider<AppMenuData> symmetricKeyProvider = providerFactory.symmetricKeyFromAccountKeysProvider(accountKeys);
        AccountKeysAccountPrivateKeyProvider<AppMenuData> accountPrivateKeyProvider = providerFactory.accountPrivateKeyFromAccountKeysProvider(accountKeys);
        return HashMap.ofAll(decryptionHelper.decryptAppMenuData(menuData.toJavaList(), symmetricKeyProvider, accountPrivateKeyProvider));
    }

    private byte[] decrypt(AppMenuData menuData,
                          AccountKeysSymmetricKeyProvider<AppMenuData> symmetricKeyProvider,
                          AccountKeysAccountPrivateKeyProvider<AppMenuData> accountPrivateKeyProvider) {
        return decryptionHelper.decryptAppMenuData(
                menuData,
                symmetricKeyProvider,
                accountPrivateKeyProvider
        );
    }

    @Override
    protected byte[] decrypt(AccountPrivateKey accountPrivateKey, AppMenuData menuData) {
        SymmetricKey symmetricKey = accountPrivateKey.decryptSymmetricKey(menuData.getSecretKey(), menuData.getSecretKeyAlgorithm());
        return symmetricKey.decrypt(menuData.getXml01(), null);
    }
}
