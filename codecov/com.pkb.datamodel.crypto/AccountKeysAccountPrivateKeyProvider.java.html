package com.pkb.datamodel.crypto;

import com.pkb.crypto.AccountPrivateKey;
import com.pkb.crypto.decryptionhelper.dto.specialized.DESEncryptedDataWithSourceRecord;
import com.pkb.crypto.decryptionhelper.provider.AccountPrivateKeyProvider;
import com.pkb.datamodel.AccountKeys;

public class AccountKeysAccountPrivateKeyProvider<T> implements AccountPrivateKeyProvider<T> {

    private AccountKeys accountKeys;

    public AccountKeysAccountPrivateKeyProvider(AccountKeys accountKeys) {
        this.accountKeys = accountKeys;
    }

    @Override
    public AccountPrivateKey getAccountPrivateKey(DESEncryptedDataWithSourceRecord<T> desEncryptedData) {
        return accountKeys.getPrivateKey();
    }

    @Override
    public String getProviderIdentifier() {
        return "AccountKeysAccountPrivateKeyProvider";
    }

    public AccountKeys getAccountKeys() {
        return accountKeys;
    }
}
