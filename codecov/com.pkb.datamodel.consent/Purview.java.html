package com.pkb.datamodel.consent;

import com.pkb.datamodel.team.Team;
import com.pkb.datamodel.user.Patient;
import com.pkb.datamodel.user.Professional;
import io.vavr.control.Option;
import org.immutables.value.Value.Immutable;
import org.immutables.vavr.encodings.VavrEncodingEnabled;

/**
 * Who am I operating on behalf of?
 * For a Patient or Individual Pro, generally I am operating by myself, so patients should
 * give their consent directly o me.
 *
 * For a Team Pro, I'm working for the Team and the patient gives consent for the whole team.
 */
@Immutable
@VavrEncodingEnabled
public interface Purview {

    Option<Patient> getPatient();

    Option<Professional> getProfessional();

    Option<Team> getTeam();

    default boolean isDefined() {
        return getPatient().isDefined() || getProfessional().isDefined() || getTeam().isDefined();
    }
}
