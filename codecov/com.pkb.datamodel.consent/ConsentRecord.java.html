package com.pkb.datamodel.consent;

import org.jetbrains.annotations.NotNull;

import java.util.stream.Stream;

import static com.pkb.datamodel.consent.AdmissionStatus.ADMITTED;
import static com.pkb.datamodel.consent.PrivacyLabel.GENERAL;
import static com.pkb.datamodel.consent.PrivacyLabel.MENTAL;
import static com.pkb.datamodel.consent.PrivacyLabel.SEXUAL;
import static com.pkb.datamodel.consent.PrivacyLabel.SOCIAL;

public final class ConsentRecord {
    private static final byte ZERO = 0x0;

    private static final int GENERAL_MASK = 0x01 << GENERAL.ordinal();
    private static final int MENTAL_MASK = 0x01 << MENTAL.ordinal();
    private static final int SEXUAL_MASK = 0x01 << SEXUAL.ordinal();
    private static final int SOCIAL_MASK = 0x01 << SOCIAL.ordinal();

    private final long id;
    private final AdmissionStatus admissionStatus;
    private final byte consents;

    public ConsentRecord(long id, @NotNull AdmissionStatus admissionStatus, @NotNull PrivacyLabel... privacyLabels) {
        this.id = id;
        this.admissionStatus = admissionStatus;
        this.consents = compactPrivacyFlags(privacyLabels);
    }

    public long getId() {
        return id;
    }

    public AdmissionStatus getAdmissionStatus() {
        return admissionStatus;
    }

    public boolean isAdmitted() {
        return ADMITTED == admissionStatus;
    }

    public boolean isDischarged() {
        return !isAdmitted();
    }

    public boolean isGeneralOn() {
        return (consents & GENERAL_MASK) > 0;
    }

    public boolean isMentalOn() {
        return (consents & MENTAL_MASK) > 0;
    }

    public boolean isSexualOn() {
        return (consents & SEXUAL_MASK) > 0;
    }

    public boolean isSocialOn() {
        return (consents & SOCIAL_MASK) > 0;
    }

    public boolean isSomethingShared() {
        return consents != 0;
    }

    public boolean isNothingShared() {
        return !isSomethingShared();
    }

    public boolean hasAnyConsent() {
        return isAdmitted() && isSomethingShared();
    }

    private byte compactPrivacyFlags(@NotNull PrivacyLabel... privacyLabels) {
        return Stream.of(privacyLabels)
                .reduce(
                        ZERO,
                        (consents, privacyLabel) -> (byte) ((int) consents | privacyLabel.mask()),
                        (consents, anotherConsents) -> (byte) (consents | anotherConsents));
    }
}
