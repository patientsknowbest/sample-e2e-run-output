package com.pkb.datamodel.consent;

import com.pkb.entities.enums.AdditionalConsentId;
import com.pkb.entities.enums.NationalIdType;

import java.util.EnumSet;

public class TeamExportDTO {
    private Long personId;
    private String title;
    private String firstName;
    private String lastName;
    private String address1;
    private String address2;
    private String city;
    private String postalCode;
    private String state;
    private String country;
    private String phone;
    private NationalIdType nationalIdType;
    private String nationalId;
    private String contactValue;
    private String olid;
    private Long teamId;
    private EnumSet<AdditionalConsentId> consentIdList = EnumSet.noneOf(AdditionalConsentId.class);


    public TeamExportDTO(Long personId, String title, String firstName, String lastName, String address1,
                         String address2, String city, String postalCode, String state, String country, String phone,
                         NationalIdType nationalIdType, String nationalId, String contactValue, String olid, Long teamId,
                         int consentEmail, int consentSms, int consentPost, int consentResearch, int consentStorage) {
        this.personId = personId;
        this.title = title;
        this.firstName = firstName;
        this.lastName = lastName;
        this.address1 = address1;
        this.address2 = address2;
        this.city = city;
        this.postalCode = postalCode;
        this.state = state;
        this.country = country;
        this.phone = phone;
        this.nationalIdType = nationalIdType;
        this.nationalId = nationalId;
        this.contactValue = contactValue;
        this.olid = olid;
        this.teamId = teamId;
        if (consentEmail == 1) {
            consentIdList.add(AdditionalConsentId.CONTACT_BY_EMAIL);
        }
        if (consentSms == 1) {
            consentIdList.add(AdditionalConsentId.CONTACT_BY_SMS);
        }
        if (consentPost == 1) {
            consentIdList.add(AdditionalConsentId.CONTACT_BY_POST);
        }
        if (consentResearch == 1) {
            consentIdList.add(AdditionalConsentId.CONTACT_FOR_RESEARCH);
        }
        if (consentStorage == 1) {
            consentIdList.add(AdditionalConsentId.SAMPLE_STORAGE);
        }
    }
