package com.pkb.datamodel.consent;

import com.pkb.entities.enums.AccessingEntityType;
import com.pkb.entities.enums.ConsentReason;
import org.immutables.serial.Serial;
import org.immutables.value.Value;

import java.util.Optional;

/**
 * Template DTO for applying patient consents in bulk
 */
@Value.Immutable
@Serial.Structural
@Value.Style(stagedBuilder = true)
public interface PatientConsentUpdateTemplate extends PatientConsentBaseDTO {
    long accessingTeamId();
    ConsentReason consentReason();
    String reasonText();
    SourceDTO source();
    Optional<Long> uploadedDataId();

    default PatientConsentUpdateDTO apply(long patientId) {
        return ImmutablePatientConsentUpdateDTO.builder()
                .generalHealthConsentGranted(generalHealthConsentGranted())
                .sexualHealthConsentGranted(sexualHealthConsentGranted())
                .mentalHealthConsentGranted(mentalHealthConsentGranted())
                .socialCareConsentGranted(socialCareConsentGranted())
                .proEditOnlyForChild(proEditOnlyForChild())
                .discharged(discharged())
                .accessingEntityType(AccessingEntityType.TEAM)
                .patientPersonId(patientId)
                .source(source())
                .generalHealthConsentGrantedReason(consentReason())
                .sexualHealthConsentGrantedReason(consentReason())
                .mentalHealthConsentGrantedReason(consentReason())
                .socialCareConsentGrantedReason(consentReason())
                .uploadedDataId(uploadedDataId())
                .accessingTeamId(accessingTeamId())
                .generalHealthConsentGrantedReasonText(reasonText())
                .mentalHealthConsentGrantedReasonText(reasonText())
                .sexualHealthConsentGrantedReasonText(reasonText())
                .socialCareConsentGrantedReasonText(reasonText())
                .build();
    }
}
