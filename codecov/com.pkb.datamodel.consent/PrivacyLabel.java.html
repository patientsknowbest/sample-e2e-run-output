package com.pkb.datamodel.consent;

/**
 * <pre>
 * PKB handles four different kind of privacy labels:
 * * General health,
 * * Mental health,
 * * Sexual care and
 * * Sexual health
 * </pre>
 */
public enum PrivacyLabel {
    GENERAL,
    MENTAL,
    SEXUAL,
    SOCIAL;

    private final byte mask = (byte) ((byte) 0x01 << ordinal());

    /** To support random passing privacy labels arguments in any order. */
    byte mask() {
        return mask;
    }
}
