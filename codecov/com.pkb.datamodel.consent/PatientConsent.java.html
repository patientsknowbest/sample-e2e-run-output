package com.pkb.datamodel.consent;

import com.pkb.datamodel.DataModelStyle;
import com.pkb.datamodel.authorization.ConsentedEntity;
import com.pkb.datamodel.user.Patient;
import com.pkb.entities.enums.PrivacyFlag;
import io.vavr.collection.List;
import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

@Value.Immutable
@Value.Style(passAnnotations = { NotNull.class }, stagedBuilder = true)
@DataModelStyle
public interface PatientConsent {

    @NotNull UUID getPublicId();

    @NotNull List<PrivacyFlag> getConsentsGranted();

    boolean isProEditOnlyForChild();

    boolean isDischarged();

    @NotNull Patient getPatient();

    @NotNull ConsentedEntity getConsentedEntity();

    default boolean isGranted(PrivacyFlag flag) {
        return getConsentsGranted().contains(flag);
    }

    @Value.Lazy
    default boolean isAtLeastOneConsentGranted() {
        return isGranted(PrivacyFlag.GENERAL)
                || isGranted(PrivacyFlag.MENTAL_HEALTH)
                || isGranted(PrivacyFlag.SEXUAL_HEALTH)
                || isGranted(PrivacyFlag.SOCIAL_CARE);
    }
}
