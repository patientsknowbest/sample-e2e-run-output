package com.pkb.datamodel.consent;

import com.google.common.base.Preconditions;
import com.pkb.entities.enums.AccessingEntityType;
import com.pkb.entities.enums.ConsentReason;
import io.vavr.Tuple2;
import org.immutables.serial.Serial;
import org.immutables.value.Value;

import java.util.Optional;

import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.API.Tuple;

/**
 * DTO for creating or updating a patient consent record.
 *
 * The combination of {@link #patientPersonId()} and _either_ {@link #accessingTeamId()} or {@link #accessingPersonId()}
 * form a natural key, and only a single record is allowed for each combination.
 */
@Value.Immutable
@Serial.Structural
@Value.Style(stagedBuilder = true)
public interface PatientConsentUpdateDTO extends PatientConsentBaseDTO {
    /**
     * Who is the consent granted to
     */
    AccessingEntityType accessingEntityType();

    /**
     * One of teamId or personId must be set, depending on accessingEntityType.
     * See {@link PatientConsentUpdateDTO#teamOrPersonIdIsSet()}
     */
    Optional<Long> accessingTeamId();

    Optional<Long> accessingPersonId();

    /**
     * Who is the consent granted from
     */
    long patientPersonId();

    /**
     * Where did the change come from?
     */
    SourceDTO source();

    /**
     * The patient consent must be created from an HL7 A28 or A31 message
     */
    Optional<Long> uploadedDataId();

    Optional<String> eventText();

    ConsentReason generalHealthConsentGrantedReason();

    Optional<String> generalHealthConsentGrantedReasonText();

    ConsentReason sexualHealthConsentGrantedReason();

    Optional<String> sexualHealthConsentGrantedReasonText();

    ConsentReason mentalHealthConsentGrantedReason();

    Optional<String> mentalHealthConsentGrantedReasonText();

    ConsentReason socialCareConsentGrantedReason();

    Optional<String> socialCareConsentGrantedReasonText();

    Optional<String> proEditOnlyForChildReasonText();

    @Value.Check
    default void teamOrPersonIdIsSet() {
        Tuple2<Boolean, String> conditionAndMessage = Match(accessingEntityType()).of(
                Case($(AccessingEntityType.TEAM), () -> Tuple(accessingTeamId().isPresent(), "accessingTeamId must be defined")),
                Case($(AccessingEntityType.CARER), () -> Tuple(accessingPersonId().isPresent(), "accessingPersonId must be defined")),
                Case($(AccessingEntityType.INDIVIDUAL), () -> Tuple(accessingPersonId().isPresent(), "accessingPersonId must be defined")));
        Preconditions.checkState(conditionAndMessage._1, conditionAndMessage._2);
    }

    default long accessingPersonOrTeamId() {
        return Match(accessingEntityType()).of(
                Case($(AccessingEntityType.TEAM), () -> accessingTeamId().orElseThrow(() -> new IllegalStateException("teamId not set"))),
                Case($(AccessingEntityType.INDIVIDUAL), () -> accessingPersonId().orElseThrow(() -> new IllegalStateException("teamId not set"))),
                Case($(AccessingEntityType.CARER), () -> accessingPersonId().orElseThrow(() -> new IllegalStateException("teamId not set"))));
    }

    default long requireAccessingPersonId() {
        return accessingPersonId().orElseThrow(() -> new IllegalStateException("accessingPersonId is not set!"));
    }

    default long requireAccessingTeamId() {
        return accessingTeamId().orElseThrow(() -> new IllegalStateException("accessingTeamId is not set!"));
    }

    static PatientConsentUpdateDTO revocationOf(PatientConsentDTO original, SourceDTO source, ConsentReason reason) {
        return ImmutablePatientConsentUpdateDTO.builder()
                .generalHealthConsentGranted(false)
                .sexualHealthConsentGranted(false)
                .mentalHealthConsentGranted(false)
                .socialCareConsentGranted(false)
                .proEditOnlyForChild(original.proEditOnlyForChild())
                .discharged(true)
                .accessingEntityType(original.accessingEntityType())
                .patientPersonId(original.patientPersonId())
                .source(source)
                .generalHealthConsentGrantedReason(reason)
                .sexualHealthConsentGrantedReason(reason)
                .mentalHealthConsentGrantedReason(reason)
                .socialCareConsentGrantedReason(reason)
                .accessingPersonId(original.accessingPersonId())
                .accessingTeamId(original.accessingTeamId())
                .build();

    }

    static PatientConsentUpdateDTO updateFor(PatientConsentDTO original, PatientConsentDTO privacyFlagUpdates, SourceDTO source, ConsentReason reason, String reasonText) {
        return ImmutablePatientConsentUpdateDTO.builder()
                .generalHealthConsentGranted(privacyFlagUpdates.generalHealthConsentGranted())
                .sexualHealthConsentGranted(privacyFlagUpdates.sexualHealthConsentGranted())
                .mentalHealthConsentGranted(privacyFlagUpdates.mentalHealthConsentGranted())
                .socialCareConsentGranted(privacyFlagUpdates.socialCareConsentGranted())
                .proEditOnlyForChild(original.proEditOnlyForChild())
                .discharged(original.discharged())
                .accessingEntityType(original.accessingEntityType())
                .patientPersonId(original.patientPersonId())
                .source(source)
                .generalHealthConsentGrantedReason(reason)
                .sexualHealthConsentGrantedReason(reason)
                .mentalHealthConsentGrantedReason(reason)
                .socialCareConsentGrantedReason(reason)
                .generalHealthConsentGrantedReasonText(reasonText)
                .sexualHealthConsentGrantedReasonText(reasonText)
                .mentalHealthConsentGrantedReasonText(reasonText)
                .socialCareConsentGrantedReasonText(reasonText)
                .accessingPersonId(original.accessingPersonId())
                .accessingTeamId(original.accessingTeamId())
                .build();
    }

}
