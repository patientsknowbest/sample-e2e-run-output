package com.pkb.datamodel.consent;

public interface PatientConsentBaseDTO {

    boolean generalHealthConsentGranted();

    boolean sexualHealthConsentGranted();

    boolean mentalHealthConsentGranted();

    boolean socialCareConsentGranted();

    /**
     * Restricts editing of consent settings for under 18s to professionals only
     * (no self-edit, no carer edits)
     */
    boolean proEditOnlyForChild();

    /**
     * Note that for _team_ consents, the patient may remove all consent flags, but the patient is still not
     * 'discharged' from care. The team are still allowed to see data they entered themselves (Team Data View)
     */
    boolean discharged();

    default boolean anyConsentGranted() {
        return generalHealthConsentGranted()
                || mentalHealthConsentGranted()
                || sexualHealthConsentGranted()
                || socialCareConsentGranted();
    }
}
