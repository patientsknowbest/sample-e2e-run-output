package com.pkb.service.dataupload.processor.support;

import java.util.List;
import java.util.stream.Collectors;

import org.springframework.stereotype.Component;

import com.pkb.encounter.entity.Message;
import com.pkb.exception.PKBException;

@Component
public class EncounterMessageFilter {
    /**
     * Convenience method to filter and return all messages of the type provided, or an empty collection.
     *
     * @param messages
     * @param messageType
     * @return
     */
    public List<Message> filterMessages(List<Message> messages, Message.MessageType messageType) {
        return filterMessages(messages, messageType, true, true);
    }

    /**
     * Get a subset of the list of messages, filtered by message type
     * and optionally define what we expect back and allow to throw an exception
     * if it doesn't comply.
     * <p>
     * Implemented here in memory rather than querying the DB directly as there will be
     * relatively few messages in a given encounter and we already get the message list.
     * <p>
     * XXX This method doesn't feel very optimal but it works and is convenient in a lot
     * of the handler methods in this class so we will keep until we come up with a better
     * approach.
     *
     * @param messages
     *            The list of messages
     * @param messageType
     *            The messageType to filter
     * @param allowEmpty
     *            Defines if it is ok that there are no matching messages of messageType
     * @param allowMultiple
     *            Defines if it is ok to have multiple instances of the same messageType
     * @return A filtered list containing only messages of the messageType provided.
     */
    public List<Message> filterMessages(List<Message> messages, Message.MessageType messageType,
            boolean allowEmpty, boolean allowMultiple) {

        List<Message> filtered = messages.stream().filter(m -> m.getMessageType() == messageType).collect(Collectors.toList());

        if (filtered.isEmpty() && !allowEmpty) {
            throw new PKBException("Expected to find at least one message of type: " + messageType + " but none found.");
        } else if ((filtered.size() > 1) && !allowMultiple) {
            throw new PKBException("Expected to find a single message of type: " + messageType + " but found multiple.");
        }
        return filtered;
    }
}
