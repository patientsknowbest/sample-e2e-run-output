package com.pkb.service.support;

import com.pkb.phoenix.shared.structures.request.MessageDeletionRequestBatch;
import com.pkb.phoenix.shared.structures.request.MessageDeletionTaskRequest;
import com.pkb.phoenix.shared.structures.response.ImmutableMessageDeletionTaskResponse;
import com.pkb.phoenix.shared.structures.response.TaskResponseStatus;
import com.pkb.repository.SupportRepository;
import org.apache.camel.Consume;
import org.apache.camel.ProducerTemplate;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;
import java.util.stream.Stream;

import static com.pkb.phoenix.shared.structures.request.MessageDeletionTaskRequest.DeletionType.CIE;
import static com.pkb.phoenix.shared.structures.response.MessageDeletionTaskResponse.MessageDeletionOutcome;
import static java.lang.invoke.MethodHandles.lookup;
import static java.util.stream.Collectors.toList;
import static org.slf4j.LoggerFactory.getLogger;

public class MessageDeletionService {
    private static final Logger LOG = getLogger(lookup().lookupClass());

    public static final String INBOUND_REQUEST_MSG_DELETION = "direct:inboundRequestMessageDeletion";
    public static final String OUTBOUND_MSG_DELETION_COMPLETE = "direct:outboundMessageDeletionComplete";
    public static final String[] CIE_TEAM_CODES = {"icht_dataq"};
    public static final String[] NO_TEAM_CODES = {};

    private final ProducerTemplate producer;
    private final SupportRepository supportRepository;

    public MessageDeletionService(ProducerTemplate producer,
                                  SupportRepository supportRepository) {
        this.producer = producer;
        this.supportRepository = supportRepository;
    }

    @Transactional
    @Consume(INBOUND_REQUEST_MSG_DELETION)
    public void processMessageDeletion(MessageDeletionTaskRequest request) {
        var teamCodes = request.type() == CIE ? CIE_TEAM_CODES : NO_TEAM_CODES;

        var builder = new ImmutableMessageDeletionTaskResponse.Builder()
                .taskId(request.taskId())
                .requestId(request.requestId());

        try {
            builder
                    .status(TaskResponseStatus.SUCCESS)
                    .errorMessage("")
                    .results(request.deletions().stream()
                            .flatMap(deletion -> getMessageDeletionOutcomeStream(teamCodes, deletion))
                            .collect(toList()));
        } catch (Exception cause) {
            //we want to report errors back via pubsub:
            builder
                    .status(TaskResponseStatus.ERROR)
                    .errorMessage(cause.getMessage())
                    .results(List.of());
            //however, we also want to rethrow the error to prevent the transaction completing
            throw cause;
        } finally {
            producer.sendBody(OUTBOUND_MSG_DELETION_COMPLETE, builder.build());
        }
    }

    @NotNull
    private Stream<MessageDeletionOutcome> getMessageDeletionOutcomeStream(String[] teamCodes, MessageDeletionRequestBatch deletion) {
        return supportRepository.messageDeletion(teamCodes, deletion);
    }
}
