package com.pkb.service.support;

import com.pkb.phoenix.shared.structures.request.EmailDeletionTaskRequest;
import com.pkb.phoenix.shared.structures.response.ImmutableEmailDeletionTaskResponse;
import com.pkb.phoenix.shared.structures.response.TaskResponseStatus;
import com.pkb.repository.SupportRepository;
import org.apache.camel.Consume;
import org.apache.camel.ProducerTemplate;
import org.springframework.transaction.annotation.Transactional;

import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;

public class EmailDeletionService {

    public static final String INBOUND_REQUEST_EMAIL_DELETION = "direct:inboundRequestEmailDeletion";
    public static final String OUTBOUND_EMAIL_DELETION_COMPLETE = "direct:outboundEmailDeletionComplete";

    private final ProducerTemplate producer;
    private final SupportRepository supportRepository;

    public EmailDeletionService(ProducerTemplate producer,
                                SupportRepository supportRepository) {
        this.producer = producer;
        this.supportRepository = supportRepository;
    }

    @Transactional
    @Consume(INBOUND_REQUEST_EMAIL_DELETION)
    public void emailDeletionRequest(EmailDeletionTaskRequest task) {
        var builder = new ImmutableEmailDeletionTaskResponse.Builder()
                .taskId(task.taskId())
                .requestId(task.requestId());

        try {
            builder
                    .status(TaskResponseStatus.SUCCESS)
                    .results(Match(emailDeletion(task)).of(
                            Case($(1), "SUCCESS"),
                            Case($(3), String.format("Table already exists with the given Freshdesk ticket ID: [%s]", task.freshDeskId())),
                            Case($(), "User does not exist or already has NOCONTACT status.")))
                    .errorMessage("");

        } catch (Exception cause) {
            builder
                    .status(TaskResponseStatus.ERROR)
                    .errorMessage(cause.getMessage());

            throw cause;
        } finally {
            producer.sendBody(OUTBOUND_EMAIL_DELETION_COMPLETE, builder.build());
        }
    }

    private Integer emailDeletion(EmailDeletionTaskRequest task) {
        return supportRepository.emailDeletion(task).orElse(null);
    }
}
