package com.pkb.service.support;

import com.pkb.VersionDetails;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.domain.duplicate.person.PersonService;
import com.pkb.entities.enums.Route;
import com.pkb.entities.enums.TwoFactorAuthState;
import com.pkb.phoenix.shared.structures.request.TwoFARemovalTaskRequest;
import com.pkb.phoenix.shared.structures.response.ImmutableTwoFARemovalTaskResponse;
import com.pkb.phoenix.shared.structures.response.TaskResponseStatus;
import com.pkb.service.twofactor.TwoFactorAuthenticationService;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import org.apache.camel.Consume;
import org.apache.camel.ProducerTemplate;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.transaction.annotation.Transactional;

import java.lang.invoke.MethodHandles;
import java.util.Date;
import java.util.Optional;

public class SupportToolTwoFARemovalService {
    public static final String INBOUND_REQUEST_TWO_FA_REMOVAL = "direct:inboundRequestTwoFARemoval";
    public static final String OUTBOUND_TWO_FA_REMOVAL_COMPLETE = "direct:outboundTwoFARemovalComplete";
    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
    private final ProducerTemplate producer;
    private final TwoFactorAuthenticationService twoFactorAuthenticationService;
    private final PersonService personService;
    private final UserManager userManager;
    private final DateTimeService dateTimeService;

    public SupportToolTwoFARemovalService(ProducerTemplate producer,
            TwoFactorAuthenticationService twoFactorAuthenticationService,
            PersonService personService,
            UserManager userManager,
            DateTimeService dateTimeService) {
        this.producer = producer;
        this.twoFactorAuthenticationService = twoFactorAuthenticationService;
        this.personService = personService;
        this.userManager = userManager;
        this.dateTimeService = dateTimeService;
    }

    @Transactional
    @Consume(INBOUND_REQUEST_TWO_FA_REMOVAL)
    public void twoFARemovalRequest(TwoFARemovalTaskRequest task) {

        var builder = new ImmutableTwoFARemovalTaskResponse.Builder()
                .taskId(task.taskId())
                .requestId(task.requestId());

        try {

            var person = personService.findPersonIdByEmail(task.email()).stream().findFirst().orElseThrow(() -> new RuntimeException("No person found for email: " + task.email()));
            var pkbPerson = userManager.findPKBPerson(person).getOrElseThrow(() -> new RuntimeException("No person found for id: " + person));

            if (pkbPerson.getTwoFactorAuthState() == TwoFactorAuthState.ACTIVE) {
                var result = removeTwoFa(pkbPerson, createVersionDetails(pkbPerson)).orElseThrow(() -> new RuntimeException("An error occurred during the process"));

                builder.results(String.format("Two-factor authentication was successfully disabled for user with ID: %s and email: %s", result.getId(), task.email()));
                LOGGER.info("Two-factor authentication was removed via Support-tool. Freshdesk Id: {} User Id: {}", task.freshDeskId(), result.getId());
            } else {
                builder.results("Two factor authentication for person is not active!");
            }

            builder.status(TaskResponseStatus.SUCCESS);
            builder.errorMessage("");
        } catch (Exception cause) {
            LOGGER.error("Error disabling 2FA via Support-Tool: ", cause);
            builder
                    .status(TaskResponseStatus.ERROR)
                    .errorMessage(cause.getMessage());

        } finally {
            producer.sendBody(OUTBOUND_TWO_FA_REMOVAL_COMPLETE, builder.build());
        }
    }

    private Optional<PKBPerson> removeTwoFa(PKBPerson person, VersionDetails versionDetails) {
        return twoFactorAuthenticationService.forceDisable(person, versionDetails);
    }

    private VersionDetails createVersionDetails(PKBPerson person) {
        VersionDetails versionDetails = new VersionDetails();

        //Person's private ID is used here as sourcePersonId since it is not possible to refer to the ID of the support-user in keycloak from PHR.
        versionDetails.setSourcePersonId(person.getId());
        versionDetails.setEnteredDate(Date.from(dateTimeService.now()));
        versionDetails.setAccessRoute(Route.SUPPORT_TOOL);
        return versionDetails;
    }

}
