package com.pkb.service.support;

import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.phoenix.shared.structures.request.EmailDeletionTaskRequest;
import com.pkb.phoenix.shared.structures.request.MessageDeletionTaskRequest;
import com.pkb.phoenix.shared.structures.request.PatientInfoTaskRequest;
import com.pkb.phoenix.shared.structures.request.TwoFARemovalTaskRequest;
import org.apache.camel.builder.ValueBuilder;
import org.apache.camel.component.jackson.JacksonDataFormat;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.slf4j.Logger;

import static com.pkb.phoenix.shared.structures.request.EmailDeletionTaskRequest.EMAIL_DELETION_TASK_TYPE;
import static com.pkb.phoenix.shared.structures.request.MessageDeletionTaskRequest.MESSAGE_DELETION_TASK_TYPE;
import static com.pkb.phoenix.shared.structures.request.PatientInfoTaskRequest.PATIENT_INFO_TASK_TYPE;
import static com.pkb.phoenix.shared.structures.request.TwoFARemovalTaskRequest.TWO_FA_REMOVAL_TASK_TYPE;
import static com.pkb.service.support.DataCollectionService.INBOUND_REQUEST_PATIENT_INFO;
import static com.pkb.service.support.EmailDeletionService.INBOUND_REQUEST_EMAIL_DELETION;
import static com.pkb.service.support.MessageDeletionService.INBOUND_REQUEST_MSG_DELETION;
import static com.pkb.service.support.PubSubUtil.uri;
import static com.pkb.service.support.SupportToolTwoFARemovalService.INBOUND_REQUEST_TWO_FA_REMOVAL;
import static java.lang.invoke.MethodHandles.lookup;
import static org.slf4j.LoggerFactory.getLogger;

public class SupportTaskInboundRequestRouteBuilder extends AbstractNamespaceAwareRouteBuilder {
    private static final Logger LOG = getLogger(lookup().lookupClass());
    private final PubSubNamespaceService pubSubNamespaceService;
    private final PhoenixPubSubConfig config;
    private final @Nullable String serviceAccountKey;

    public SupportTaskInboundRequestRouteBuilder(PubSubNamespaceService pubSubNamespaceService, PhoenixPubSubConfig config, @Nullable String serviceAccountKey) {
        this.pubSubNamespaceService = pubSubNamespaceService;
        this.config = config;
        this.serviceAccountKey = serviceAccountKey;
    }

    @Override
    public @NotNull PubSubNamespaceService camelNamespaceService() {
        return pubSubNamespaceService;
    }

    @Override
    public boolean throwExceptionForNamespaceMismatch() {
        return true;
    }

    @Override
    public void configure() {
        from(uri(config.getPubSubProjectName(), config.taskRequestedSubscriptionName(), serviceAccountKey))
                .choice()
                    .when(jsonPathTaskTypeEquals(PATIENT_INFO_TASK_TYPE))
                        .unmarshal(new JacksonDataFormat(PatientInfoTaskRequest.class))
                        .to(INBOUND_REQUEST_PATIENT_INFO)
		            .when(jsonPathTaskTypeEquals(MESSAGE_DELETION_TASK_TYPE))
                        .unmarshal(new JacksonDataFormat(MessageDeletionTaskRequest.class))
                        .to(INBOUND_REQUEST_MSG_DELETION)
                    .when(jsonPathTaskTypeEquals(EMAIL_DELETION_TASK_TYPE))
                        .unmarshal(new JacksonDataFormat(EmailDeletionTaskRequest.class))
                        .to(INBOUND_REQUEST_EMAIL_DELETION)
                    .when(jsonPathTaskTypeEquals(TWO_FA_REMOVAL_TASK_TYPE))
                        .unmarshal(new JacksonDataFormat(TwoFARemovalTaskRequest.class))
                        .to(INBOUND_REQUEST_TWO_FA_REMOVAL)
                    .endChoice()
                .otherwise()
                    .stop()
                    .end();
    }

    private ValueBuilder jsonPathTaskTypeEquals(String type){
        return jsonpath(String.format("$[?(@.taskType == '%s')]", type));
    }
}
