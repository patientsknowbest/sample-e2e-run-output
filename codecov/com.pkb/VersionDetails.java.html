package com.pkb;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.entities.enums.Route;

import javax.persistence.Embeddable;
import java.time.Instant;
import java.util.Date;

import static com.google.common.base.Preconditions.checkNotNull;

@Embeddable
public class VersionDetails extends BaseVersionDetails {

    public static VersionDetails of(EHRRequestContext requestContext, Instant enteredDate) {
        VersionDetails versionDetails = new VersionDetails();
        requestContext.getMaybeAccessingUserId().ifPresent(versionDetails::setSourcePersonId);
        requestContext.getMaybeRegisteringUserId().ifPresent(versionDetails::setSourcePersonId);
        requestContext.getTeamId().ifPresent(versionDetails::setSourceTeamId);
        requestContext.getOrgId().ifPresent(versionDetails::setSourceOrgId);
        versionDetails.setAccessRoute(requestContext.getAccessRoute());
        versionDetails.setEnteredDate(Date.from(enteredDate));
        requestContext.getMaybeConnectingOrgId().ifPresent(versionDetails::setConnectingOrgId);
        requestContext.getMaybeHl7PartnerId().ifPresent(versionDetails::setPartnerId);
        requestContext.getMaybeResponsibleOrganisationIdentifier().ifPresent(versionDetails::setResponsibleOrganisationIdentifier);
        return versionDetails;
    }

    /**
     * Can only be used for entities which use {@link com.pkb.util.VersionDetailsPersistenceListener}!!!
     *
     * @param requestContext
     * @return
     */
    public static VersionDetails of(EHRRequestContext requestContext) {
        VersionDetails versionDetails = new VersionDetails();
        requestContext.getMaybeAccessingUserId().ifPresent(versionDetails::setSourcePersonId);
        requestContext.getTeamId().ifPresent(versionDetails::setSourceTeamId);
        requestContext.getOrgId().ifPresent(versionDetails::setSourceOrgId);
        requestContext.getMaybeConnectingOrgId().ifPresent(versionDetails::setConnectingOrgId);
        requestContext.getMaybeHl7PartnerId().ifPresent(versionDetails::setPartnerId);
        versionDetails.setAccessRoute(requestContext.getAccessRoute());
        requestContext.getMaybeResponsibleOrganisationIdentifier().ifPresent(versionDetails::setResponsibleOrganisationIdentifier);
        return versionDetails;
    }

    public static VersionDetails of(SourceDetails sourceDetails, Route accessRoute, Instant enteredDate) {
        checkNotNull(sourceDetails, "sourceDetails");

        VersionDetails versionDetails = new VersionDetails();

        versionDetails.setAccessRoute(accessRoute);
        versionDetails.setSourcePersonId(sourceDetails.getPersonId());
        versionDetails.setSourceTeamId(sourceDetails.getTeamId());
        versionDetails.setSourceOrgId(sourceDetails.getOrgId());
        versionDetails.setConnectingOrgId(sourceDetails.getConnectingOrgId());
        versionDetails.setPartnerId(sourceDetails.getHl7PartnerId());
        versionDetails.setSourceText(sourceDetails.getText());
        versionDetails.setEnteredDate(Date.from(enteredDate));
        versionDetails.setResponsibleOrganisationIdentifier(sourceDetails.getResponsibleOrganisationIdentifier());

        return versionDetails;
    }

}
