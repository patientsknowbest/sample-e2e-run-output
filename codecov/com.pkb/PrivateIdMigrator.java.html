package com.pkb;

import com.pkb.annotation.EHRVersionMigrator;
import com.pkb.app.entity.DTOMapping;
import com.pkb.app.entity.EHRData;
import com.pkb.app.interfaces.IBaseDTO;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.UUID;

public class PrivateIdMigrator<T extends IBaseDTO> extends EHRVersionMigrator {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final String dtoFieldName;
    private final Class<T> dtoClass;
    private final String oldQueryFieldName;

    private PrivateIdMigrator(Class<T> dtoClass, String dtoFieldName, String oldQueryFieldName, int migrationVersion) {
        super(migrationVersion);
        this.dtoFieldName = dtoFieldName;
        this.dtoClass = dtoClass;
        this.oldQueryFieldName = oldQueryFieldName;
    }

    @Override
    public void migrate(EHRData ehr) {
        Object obj = ehr.getEncryptedField(dtoFieldName);
        DTOMapping<T> mapping = DTOMapping.get(dtoClass);
        Object oldFieldValue = ehr.getQueryField(oldQueryFieldName);
        if (obj != null) {
            String newQueryFieldName = mapping.getEHRDataField(dtoFieldName);
            ehr.setQueryField(newQueryFieldName, obj);
        }
        else if (oldFieldValue != null) {
            LOGGER.error("PHR-9677 Private id migrator attempted to migrate populated private id value for ehr data {} but encrypted blob contained null", ehr.getId());
        }
    }

    /**
     * This is pretty sketchy but is only intended for temporary testing until we can turn off the private maps migrators for good
     * It assumes that the data it is passed has already undergone the private maps migration - a  reasonable assumption for
     * e2e testing at the time of writing - and also that it is safe to "downgrade" the migrationVersion by simply subtracting 1
     * from it. This latter assumption will break as soon as we add any migrations after the private maps migrations, but TBH
     * I'm hoping that we'll bin EHRData entirely in favour of a new datastore before we start writing new migrations.
     */
    public  void unmigrate(EHRData ehrData) {
        DTOMapping<T> mapping = DTOMapping.get(dtoClass);
        String newQueryFieldName = mapping.getEHRDataField(dtoFieldName);
        Object value = ehrData.getQueryField(newQueryFieldName);
        ehrData.setEncryptedField(dtoFieldName, value);
        ehrData.setQueryField(oldQueryFieldName, value == null ? null : UUID.randomUUID().toString());
        ehrData.setMigrationVersion(ehrData.getMigrationVersion() - 1);
    }

    public static <T extends IBaseDTO> EHRVersionMigrator migratorFor(Class<T> dtoClass, String dtoFieldName, String oldQueryFieldName, int migrationVersion) {
        return new PrivateIdMigrator<>(dtoClass, dtoFieldName, oldQueryFieldName, migrationVersion);
    }

}
