package com.pkb.fhirwall.service.util;

import ca.uhn.fhir.context.FhirContext;
import ca.uhn.fhir.util.BundleUtil;
import org.hl7.fhir.instance.model.api.IBaseResource;
import org.hl7.fhir.r4.model.Bundle;

import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

/**
 * Passthrough class to allow for mocking of HAPI BundleUtil static method calls in unit tests
 */
public class BundleService {

    private FhirContext fhirContext;

    public BundleService(FhirContext fhirContext) {
        this.fhirContext = fhirContext;
    }

    public Map<String, List<IBaseResource>> getResourcesFromBundleGroupingByFhirType(Bundle bundle) {
       return BundleUtil.toListOfResources(fhirContext, bundle)
                .stream().collect(Collectors.groupingBy(IBaseResource::fhirType));
    }
}
