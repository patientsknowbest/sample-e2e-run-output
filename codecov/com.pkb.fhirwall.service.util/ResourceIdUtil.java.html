package com.pkb.fhirwall.service.util;

import ca.uhn.fhir.context.FhirContext;
import org.hl7.fhir.instance.model.api.IBaseResource;

import java.util.Comparator;
import java.util.function.Function;

import static java.lang.String.format;
import static org.apache.commons.lang3.StringUtils.isBlank;

/**
 * Utility class operating on FHIR Resource IDs
 */
public class ResourceIdUtil {

    private static final FhirContext fhirContext = FhirContext.forR4();

    public static Comparator<IBaseResource> BASE_RESOURCE_COMPARATOR_ID = (a, b) -> {
        if (a==null && b==null) {
            return 0;
        }
        if (a==null) {
            return  1;
        }
        if (b==null) {
            return -1;
        }
        return (a.getIdElement() == null || a.getIdElement().getIdPart() == null ? "" : a.getIdElement().getIdPart())
                .compareTo(b.getIdElement() == null || b.getIdElement().getIdPart() == null ? "" : b.getIdElement().getIdPart());
    };

    /**
     * @param resourceId FHIR reference in "Resource_Type/id" format
     * @return search expression for the given resource which loads Consent resources as well
     */
    public static String toSearchExpressionWithConsent(String resourceId) {
        return splitAndMap(resourceId,
                resourceParts -> {
                    if (resourceParts.length < 2) {
                        throw new RuntimeException("resourceId is not valid: " + resourceId);
                    }
                    return format("%s?id=%s&_revinclude=Consent:patient", resourceParts[0], resourceParts[1]);
                });
    }

    /**
     * @param resourceId FHIR reference in "Resource_Type/id" format
     * @return FHIR resource type extracted from resourceId
     */
    public static Class<? extends IBaseResource> getType(String resourceId) {
        return splitAndMap(resourceId,
                resourceParts -> fhirContext.getResourceDefinition(resourceParts[0]).getImplementingClass());
    }

    /**
     * @param resourceId FHIR reference in "Resource_Type/id" format which might be prefixed with the "/fhir/" context path
     * @return resource id without the context path
     */
    public static String removeContext(String resourceId) {
        return resourceId.replace("/fhir/", "");
    }

    private static <T> T splitAndMap(String resourceId, Function<String[], T> mapper) {
        if (isBlank(resourceId)) {
            throw new RuntimeException("resourceId is blank!");
        }

        return mapper.apply(resourceId.split("/"));
    }
}
