package com.pkb.api.fhir.communication;

import static com.google.common.base.Predicates.instanceOf;

import static com.pkb.authz.context.Path.FHIR;
import static com.pkb.authz.data.ImmutableAuthorizationInputs.inputs;

import static io.vavr.API.$;

import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

public class CommunicationResourceService extends AbstractResourceService<Communication> {
    private final AuthorizationDataService authorizationDataService;
    private final CommunicationService communicationService;
    private final CommunicationMapper communicationMapper;
    private final ProvenanceMapper provenanceMapper;
    private final FhirApiProperties fhirConfig;

    public CommunicationResourceService(ServerResponseExceptionMapper exceptionMapper,
                                        AuthorizationDataService authorizationDataService,
                                        CommunicationService communicationService,
                                        CommunicationMapper communicationMapper,
                                        ProvenanceMapper provenanceMapper,
                                        FhirApiProperties fhirConfig) {
        super(exceptionMapper);
        this.authorizationDataService = authorizationDataService;
        this.communicationService = communicationService;
        this.communicationMapper = communicationMapper;
        this.provenanceMapper = provenanceMapper;
        this.fhirConfig = fhirConfig;
    }

    public Either<BaseServerResponseException, FhirCriteriaResult<Communication>> search(@NotNull CommunicationSearchCriteria searchCriteria) {

        AuthorizationData authzData = authorizationDataService.fetch(inputs()
                .path(FHIR)
                .action(SEARCH)
                .addTargets(targetMeta(com.pkb.datamodel.Communication.class, Set(), Some(searchCriteria.patientId()), true))
                .authenticatedIdentity(searchCriteria.authenticatedIdentity())
                .build());
        CommunicationQuery communicationQuery = communicationQuery()
                .patientId(searchCriteria.patientId())
                .sent(searchCriteria.sent())
                .authzData(authzData)
                .maxNumberOfResources(fhirConfig.getCommunicationMaxNumberOfResources())
                .build();
        return communicationService.search(communicationQuery)
                .fold(error -> Match(error).of(
                        Case($(instanceOf(SystemUserTypeNotPermittedException.class)), nae -> left(toSystemUserTypeNotPermitted())),
                        Case($(), right(FhirCriteriaResult.empty()))),
                        communications -> right(mapCommunications(communications, searchCriteria.includes())));
    }

    private FhirCriteriaResult<Communication> mapCommunications(CriteriaResult<com.pkb.datamodel.Communication> result, 
                                                                List<IncludeResource> includes) {
        List<Communication> communications = result.getPage().map(communicationMapper::toCommunication);
        List<Provenance> provenances = includes.contains(IncludeResource.PROVENANCE) ?
                result.getPage().map(provenanceMapper::toProvenance) : List();

        return ImmutableFhirCriteriaResult.<Communication> fhirCriteriaResult()
                .total(result.getTotal())
                .page(communications)
                .includes(List.narrow(provenances))
                .build();
    }
}
