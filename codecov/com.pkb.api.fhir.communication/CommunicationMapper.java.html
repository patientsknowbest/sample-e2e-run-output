package com.pkb.api.fhir.communication;

import com.pkb.api.fhir.PrivacyFlagsMapper;
import com.pkb.api.fhir.TemporalMapper;
import com.pkb.api.fhir.foundation.ReferenceMapper;
import com.pkb.api.fhir.person.RelatedPersonMapper;
import com.pkb.datamodel.AttachmentMetadata;
import com.pkb.datamodel.Identifiable;
import com.pkb.datamodel.user.Patient;
import com.pkb.datamodel.user.Person;
import com.pkb.datamodel.user.Professional;
import com.pkb.datamodel.user.TeamCoordinator;
import io.vavr.collection.LinkedHashMap;
import io.vavr.collection.List;
import org.hl7.fhir.dstu3.model.Attachment;
import org.hl7.fhir.dstu3.model.Communication;
import org.hl7.fhir.dstu3.model.Communication.CommunicationPayloadComponent;
import org.hl7.fhir.dstu3.model.Extension;
import org.hl7.fhir.dstu3.model.Meta;
import org.hl7.fhir.dstu3.model.Reference;
import org.hl7.fhir.dstu3.model.RelatedPerson;
import org.hl7.fhir.dstu3.model.StringType;
import org.jetbrains.annotations.NotNull;

import static com.pkb.api.fhir.CanonicalExtensionURLs.communicationTitle;
import static com.pkb.api.fhir.CanonicalExtensionURLs.conversationId;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.List;
import static io.vavr.API.Match;
import static io.vavr.Predicates.instanceOf;
import static io.vavr.Predicates.is;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class CommunicationMapper {
    private final TemporalMapper timeMapper;
    private final PrivacyFlagsMapper privacyFlagsMapper;
    private final ReferenceMapper referenceMapper;
    private final RelatedPersonMapper relatedPersonMapper;

    public CommunicationMapper(TemporalMapper timeMapper,
                               PrivacyFlagsMapper privacyFlagsMapper,
                               ReferenceMapper referenceMapper,
                               RelatedPersonMapper relatedPersonMapper) {
        this.timeMapper = timeMapper;
        this.privacyFlagsMapper = privacyFlagsMapper;
        this.referenceMapper = referenceMapper;
        this.relatedPersonMapper = relatedPersonMapper;
    }

    public @NotNull Communication toCommunication(com.pkb.datamodel.Communication communication) {
        Patient patient = communication.getPatient();

        Communication fhirCommunication = new Communication();
        fhirCommunication.setId(communication.getId().toString());
        fhirCommunication.setMeta(extractMetadata(communication));
        fhirCommunication.setExtension(getExtensions(communication).toJavaList());
        fhirCommunication.setStatus(Communication.CommunicationStatus.UNKNOWN);
        fhirCommunication.setSubject(referenceMapper.toReferenceWithName(patient));

        var relatedPersonToContainedReferenceId = selectCarerToReferenceId(communication, patient);

        //map the recipients - nb this won't add any contained refs for carers
        fhirCommunication.setRecipient(communication.getParticipants()
                .map(participant -> personToReference(patient, participant, relatedPersonToContainedReferenceId))
                .toJavaList());

        //same for senders - again, won't add any contained refs
        fhirCommunication.setSender(personToReference(patient, communication.getSender(), relatedPersonToContainedReferenceId));

        //now add any contained refs we will need
        relatedPersonToContainedReferenceId.forEach( ( carer, containedRefId ) -> {
            RelatedPerson relatedPerson = relatedPersonMapper.toRelatedPerson(patient,carer);
            relatedPerson.setId(containedRefId);
            fhirCommunication.addContained(relatedPerson);
        });

        fhirCommunication.setSent(timeMapper.toDate(communication.getTimestamp()));

        fhirCommunication.setPayload(communication.getAttachmentMetadata()
                .map(this::toCommunicationPayloadComponent)
                .toJavaList());
        return fhirCommunication;
    }

    /**
     * In FHIR, we map carers to contained references. To do this we need a deduped list of carers
     * and a simple way to map them to a contained reference ID of the form: #carer-1, #carer-2
     *
     * @param communication - the communication being mapped
     * @param patient - the patient that communication is about
     * @return - A map of all carers to contained reference id, ordered by the pkb id of the carer
     */
    private LinkedHashMap<Patient, String> selectCarerToReferenceId(com.pkb.datamodel.Communication communication, Patient patient) {
        return communication.getParticipants()
                .append(communication.getSender())
                .filter(p -> p instanceof Patient && !p.equals(patient))
                //ensure stable ordering:
                .sortBy(Identifiable::getId)
                .map(p -> (Patient) p)
                .distinctBy(Identifiable::getId)
                .zipWithIndex().foldLeft(
                        LinkedHashMap.<Patient, String>empty(),
                        (map, tuple) -> map.put(tuple._1, String.format("#carer-%s", tuple._2 + 1)));
    }

    @NotNull
    private List<Extension> getExtensions(com.pkb.datamodel.Communication communication) {
        List<Extension> extensions = List(
                new Extension(conversationId(), new StringType(communication.getConversationId().toString())));
        if (isNotBlank(communication.getConversationSubject())) {
            extensions = extensions.append(
                    new Extension(communicationTitle(), new StringType(communication.getConversationSubject())));
        }
        return extensions;
    }

    private CommunicationPayloadComponent toCommunicationPayloadComponent(AttachmentMetadata attachmentMetadata) {
        CommunicationPayloadComponent payloadComponent = new CommunicationPayloadComponent();
        Attachment attachment = new Attachment();
        attachment.setTitle(attachmentMetadata.getFilename());
        attachment.setCreationElement(timeMapper.toDateTimeTypeInUTC(attachmentMetadata.getUploadTime()));
        payloadComponent.setContent(attachment);
        return payloadComponent;
    }

    private Reference personToReference(Patient patient, Person reference, LinkedHashMap<Patient, String> relatedPersonToReferenceId) {
        return Match(reference).of(
                Case($(instanceOf(Professional.class)), referenceMapper::toReferenceWithName),
                Case($(instanceOf(Patient.class).and(is(patient))), referenceMapper::toReferenceWithName),
                Case($(instanceOf(Patient.class).and(is(patient).negate())),
                        carer -> referenceMapper.toContainedReference(carer, relatedPersonToReferenceId.get(carer)
                                .getOrElseThrow(() -> new IllegalStateException("No related person id provided for carer "+carer.getId())))),
                Case($(instanceOf(TeamCoordinator.class)), referenceMapper::toReferenceWithName)
        );
    }

    private @NotNull Meta extractMetadata(com.pkb.datamodel.Communication communication) {
        Meta meta = new Meta();
        communication.getPersisted().map(timeMapper::toDate).peek(meta::setLastUpdated);
        meta.setSecurity(privacyFlagsMapper.toCodingList(communication.getPrivacyFlags()).toJavaList());
        return meta;
    }
}
