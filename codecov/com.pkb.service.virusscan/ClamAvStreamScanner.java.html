package com.pkb.service.virusscan;

import com.pkb.common.testlogging.DetailLogger;
import com.pkb.common.testlogging.DetailLoggingProvider;
import io.prometheus.client.Summary;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;
import xyz.capybara.clamav.ClamavClient;
import xyz.capybara.clamav.commands.scan.result.ScanResult;

import java.io.InputStream;
import java.util.Collection;

/**
 * Simple abstraction for a tool that can check an input stream for virus signatures and other threat vectors.
 */
public class ClamAvStreamScanner implements StreamScanner {
    private static final Summary scanDurationSumaries = Summary.build()
            .name("pkb_phr_security_clamavscans")
            .help("Latencies of ClamAV scan requests")
            .quantile(0.5, 0.05)
            .quantile(0.9, 0.01)
            .quantile(0.95, 0.005)
            .quantile(0.99, 0.001)
            .register();

    private final ClamavClient clamavClient;
    private final DetailLoggingProvider detailLoggingProvider;

    public ClamAvStreamScanner(@NotNull String host, @NotNull DetailLoggingProvider detailLoggingProvider) {
        this.clamavClient = new ClamavClient(host);
        this.detailLoggingProvider = detailLoggingProvider;
    }

    @Override
    public @NotNull Either<String, SuccessReason> scan(@NotNull InputStream inputStream) {
        DetailLogger detailedLogger = detailLoggingProvider.obtainLogger(getClass());
        ScanResult result = io.vavr.API.unchecked(() -> {
            try (Summary.Timer $ = scanDurationSumaries.startTimer()) {
                detailedLogger.info("Scan start");
                return clamavClient.scan(inputStream);
            }
        }).get();

        detailedLogger.info("Scan result: {}", result);

        if (result instanceof ScanResult.VirusFound) {
            // We're scanning a stream, not a directory so we expect a single result
            //noinspection ConstantConditions
            String virus = ((ScanResult.VirusFound) result).getFoundViruses().values().stream()
                    .flatMap(Collection::stream)
                    .findFirst()
                    .orElseThrow(() -> new IllegalStateException("Virus scan was positive, but unable to get virus name"));
            detailedLogger.info("Virus found: {}", virus);
            return Either.left(virus);
        } else {
            detailedLogger.info("Passed check");
            return Either.right(SuccessReason.CHECKED);
        }
    }
}
