package com.pkb.service.virusscan;

import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.InputStream;

import static com.pkb.service.virusscan.StreamScanner.SuccessReason.SCANNER_INOPERATIONAL;

public interface StreamScanner {

    Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    enum SuccessReason {
        CHECKED, SCANNER_DISABLED, SCANNER_INOPERATIONAL
    }

    @NotNull
    Either<String, SuccessReason> scan(@NotNull InputStream inputStream);

    @NotNull
    default Either<String, SuccessReason> safeScan(@NotNull InputStream inputStream) {
        Either<String, StreamScanner.SuccessReason> scanResult;
        try {
            scanResult = scan(inputStream);
        } catch (Exception e) {
            LOGGER.error("Error while invoking virus scanner", e);
            scanResult = Either.right(SCANNER_INOPERATIONAL);
        }
        return scanResult;
    }
}
