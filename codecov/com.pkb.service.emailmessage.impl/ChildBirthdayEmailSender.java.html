package com.pkb.service.emailmessage.impl;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.consent.model.NoConsentsRequired;
import com.pkb.entities.enums.CarerRemovalAge;
import com.pkb.entities.enums.Message;
import com.pkb.exception.MailException;
import com.pkb.service.emailmessage.IAddressInfoProvider;
import com.pkb.service.emailmessage.ITemplateParamsFactory;
import com.pkb.service.reminders.ReminderManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.PKBConstants;
import io.prometheus.client.Counter;
import io.prometheus.client.Summary;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.time.Instant;
import java.time.ZonedDateTime;
import java.time.temporal.ChronoUnit;
import java.util.List;
import java.util.Map;
import java.util.Optional;

import static java.util.Objects.requireNonNull;

public class ChildBirthdayEmailSender {

    private static final Summary METRICS = Summary.build()
            .name("pkb_phr_schedtask_childbirthdayemailsender_duration")
            .help("duration of sending notifications about carer consents expiring due to children growing up")
            .labelNames("age")
            .register();

    private static final Counter COUNTER = Counter.build()
            .name("pkb_phr_schedtask_childbirthdayemailsender_count")
            .help("count of notifications sent about carer consents expiring due to children growing up")
            .labelNames("age")
            .register();

    private final Logger logger;

    private final ReminderManager reminderManager;

    private final EmailManager emailManager;

    private final IAddressInfoProvider formatter;

    private final DateTimeService dateTimeService;

    private ITemplateParamsFactory templateParamsFactory;

    public ChildBirthdayEmailSender(ReminderManager reminderManager,
            EmailManager emailManager, IAddressInfoProvider formatter,
            ITemplateParamsFactory templateParamsFactory, DateTimeService dateTimeService) {
        this(reminderManager, emailManager, formatter,
                templateParamsFactory,
                LoggerFactory.getLogger(ChildBirthdayEmailSender.class), dateTimeService);
    }

    public ChildBirthdayEmailSender(ReminderManager reminderManager,
            EmailManager emailManager, IAddressInfoProvider formatter,
            ITemplateParamsFactory templateParamsFactory,
            Logger logger,
            DateTimeService dateTimeService) {
        this.reminderManager = requireNonNull(reminderManager, "reminderManager cannot be null");
        this.emailManager = requireNonNull(emailManager, "emailManager cannot be null");
        this.formatter = requireNonNull(formatter, "formatter cannot be null");
        this.templateParamsFactory = requireNonNull(templateParamsFactory, "templateParamsFactory cannot be null");
        this.logger = requireNonNull(logger, "logger cannot be null");
        this.dateTimeService = dateTimeService;
    }

    public void send(CarerRemovalAge age) {
        Summary.Timer timer = METRICS.labels(String.valueOf(age.getValue())).startTimer();
        int foundCount = 0;
        try {
            logger.info("Entering sendChildBirthdayEmails");
            logger.info("searching for child birthday email recipients");

            List<PKBPerson> patientsToNotify = getBirthdayEmailRecipients(age);
            foundCount = patientsToNotify.size();
            logger.info("found {} recipients", foundCount);

            patientsToNotify.forEach(patient -> sendChildBirthdayMessage(patient, age));

            if (foundCount > 0) {
                logger.info("child birthday reminders sent!  found {} turning {} to notify", foundCount, age.getValue());
            }
        } finally {
            COUNTER.labels(String.valueOf(age.getValue())).inc(foundCount);
            timer.observeDuration();
        }
    }

    private List<PKBPerson> getBirthdayEmailRecipients(CarerRemovalAge age) {
        // milestone birthdays between tomorrow and 3 months from now
        ZonedDateTime now = dateTimeService.nowZonedDateTime();
        ZonedDateTime givenYearsAgo = now.minusYears(age.getValue());
        Instant birthdayFrom = givenYearsAgo.truncatedTo(ChronoUnit.DAYS).plusDays(1).toInstant();
        Instant birthdayTo = givenYearsAgo.plusMonths(3L).truncatedTo(ChronoUnit.DAYS).toInstant();

        // no duplicates in the list
        return reminderManager.claimChildBirthdayEmailRecipients(birthdayFrom, birthdayTo, age.getReminderPropertyName());
    }

    private void sendChildBirthdayMessage(PKBPerson patient, CarerRemovalAge age) {
        Map<String, Object> properties = templateParamsFactory.createWithSiteProps();
        properties.put(PKBConstants.PATIENT_FIRST_NAME, patient.getFirstName());
        properties.put(PKBConstants.PATIENT_LAST_NAME, patient.getLastName());
        properties.put(PKBConstants.PATIENT_AGE, age.getValue());

        sendMessageToPatient(patient, properties);
        properties.remove(PKBConstants.SUBJECT_PREFIX);
        sendMessageToCarers(patient, properties);
    }

    private void sendMessageToPatient(PKBPerson patient, Map<String, Object> properties) {
        formatter.getPrimaryContactOnly(patient).ifPresentOrElse(primaryContact -> {
            try {
                emailManager.sendSystemMessage(Message.CHILD_BIRTHDAY_REMINDER_TO_PATIENT, null, primaryContact,
                        patient, properties, patient.getLocale(), null, null, null);
            } catch (MailException e) {
                logger.warn("Failed to send child birthday reminder to patient - {}", patient.getId(), e);
            }
        }, () -> logger.warn("Primary contact not found for patient {}", patient.getId()));
    }

    private void sendMessageToCarers(PKBPerson patient, Map<String, Object> properties) {
        formatter.getPrimaryAndSecondaryContacts(patient, new NoConsentsRequired()).getSecondaryAddressesAndPersons()
                .forEach((address, carer) -> Optional.ofNullable(address).ifPresentOrElse(carerContact -> {
                    properties.put(PKBConstants.CARER_FIRST_NAME, carer.getFirstName());
                    try {
                        emailManager.sendSystemMessage(Message.CHILD_BIRTHDAY_REMINDER_TO_CARER, null, new AddressInfo(carerContact),
                                carer, properties, carer.getLocale(), null, null, null);
                    } catch (MailException e) {
                        logger.warn("Failed to send child birthday reminder to carer - {}", carer.getId(), e);
                    }
                }, () -> logger.warn("Primary contact not found for carer {}", carer.getId())));
    }

}
