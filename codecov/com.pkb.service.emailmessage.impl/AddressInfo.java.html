package com.pkb.service.emailmessage.impl;

import java.io.UnsupportedEncodingException;
import java.util.Collection;
import java.util.Collections;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Map;
import java.util.Objects;
import java.util.Set;
import java.util.stream.Collectors;

import javax.mail.internet.InternetAddress;

import org.apache.commons.lang3.SerializationUtils;

import com.pkb.user.entity.PKBPerson;

/**
 * AddressInfo is a data container for InternetAddress objects
 * It contains an optional primary address and lists of (optional) secondary and cc addresses
 *
 * @author James Fox
 */
public final class AddressInfo {
    private InternetAddress primaryAddress;
    private Set<InternetAddress> ccAddresses;
    private Map<InternetAddress, PKBPerson> secondaryAddressesAndPersons;

    public AddressInfo(InternetAddress primaryAddress) {
        this(primaryAddress, Collections.emptyMap(), Collections.emptySet());
    }

    public AddressInfo(InternetAddress primaryAddress, Map<InternetAddress, PKBPerson> secondaryAddressesAndPersons) {
        this(primaryAddress, secondaryAddressesAndPersons, Collections.emptySet());
    }

    public AddressInfo(InternetAddress primaryAddress,
                       Map<InternetAddress, PKBPerson> secondaryAddressesAndPersons,
                       Collection<InternetAddress> ccAddresses) {
            this.primaryAddress = copyAddress(primaryAddress);
            this.secondaryAddressesAndPersons = copyAddessesAndPersons(secondaryAddressesAndPersons);
            this.ccAddresses = copySet(ccAddresses);
    }

    private InternetAddress copyAddress(InternetAddress toCopy) {
        if (toCopy == null) {
            return null;
        }
        try {
            return new InternetAddress(toCopy.getAddress(), toCopy.getPersonal());
        } catch (UnsupportedEncodingException e) {
          throw new RuntimeException("Error in AddressInfo", e);
        }
    }

    private Set<InternetAddress> copySet(Collection<InternetAddress> toCopy) {
        if (toCopy == null) {
            return new HashSet<>();
        }
        return toCopy.stream().map(this::copyAddress).filter(Objects::nonNull).collect(Collectors.toSet());
    }

    private Map<InternetAddress, PKBPerson> copyAddessesAndPersons(Map<InternetAddress, PKBPerson> toCopy) {
        if (toCopy == null) {
            return new HashMap<>();
        }
        return toCopy.entrySet().stream().collect(Collectors.toMap(entry -> copyAddress(entry.getKey()), entry -> SerializationUtils.clone(entry.getValue())));
    }

    public InternetAddress getPrimaryAddress() {
            return copyAddress(primaryAddress);
    }

    public Map<InternetAddress, PKBPerson> getSecondaryAddressesAndPersons() {
        return Collections.unmodifiableMap(copyAddessesAndPersons(secondaryAddressesAndPersons));
    }

    public Collection<InternetAddress> getCcAddresses() {
            return Collections.unmodifiableSet(copySet(ccAddresses));
    }

    @Override
    public String toString() {
        StringBuilder builder = new StringBuilder();
        if (getPrimaryAddress() != null) {
            builder.append("Primary: ").append(getPrimaryAddress().toString()).append(" ");
        }
        getSecondaryAddressesAndPersons().keySet().forEach(address -> builder.append("Secondary: ").append(address.toString()).append(" "));
        getCcAddresses().forEach(address -> builder.append("CC: ").append(address.toString()).append(" "));
        return builder.toString();
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (o == null || getClass() != o.getClass()) {
            return false;
        }

        AddressInfo that = (AddressInfo) o;

        if (primaryAddress == null || primaryAddress.getAddress() == null) {
            if (that.primaryAddress != null && that.primaryAddress.getAddress() != null) {
                return false;
            }
        }

        if (primaryAddress != null && !primaryAddress.equals(that.primaryAddress)) {
            return false;
        }

        if (secondaryAddressesAndPersons != null ? !secondaryAddressesAndPersons.equals(that.secondaryAddressesAndPersons) : that.secondaryAddressesAndPersons != null) {
            return false;
        }
        return ccAddresses != null ? ccAddresses.equals(that.ccAddresses) : that.ccAddresses == null;
    }

    @Override
    public int hashCode() {
        int result = primaryAddress != null ? primaryAddress.hashCode() : 0;
        result = 31 * result + (secondaryAddressesAndPersons != null ? secondaryAddressesAndPersons.hashCode() : 0);
        result = 31 * result + (ccAddresses != null ? ccAddresses.hashCode() : 0);
        return result;
    }

}
