package com.pkb.service.emailmessage.impl;

import static com.pkb.util.PKBConstants.BASE_URL;
import static com.pkb.util.PKBConstants.PKB_HELP_EMAIL;
import static com.pkb.util.PKBConstants.PKB_HELP_NAME;
import static com.pkb.util.PKBConstants.PKB_HELP_PHONE;
import static com.pkb.util.PKBConstants.PKB_HELP_WEBSITE;
import static com.pkb.util.PKBConstants.PKB_LOGO_BASE_DIR;
import static com.pkb.util.PKBConstants.PKB_LOGO_URL;
import static com.pkb.util.PKBConstants.PKB_SITE_URL;

import org.jetbrains.annotations.NotNull;

import com.pkb.common.config.PhrConfig;
import com.pkb.service.emailmessage.ITemplateParamsFactory;
import com.pkb.util.PKBConstants;

public class TemplateParamsFactory implements ITemplateParamsFactory {

    private PhrConfig config;

    public TemplateParamsFactory(@NotNull PhrConfig config) {
        this.config = config;
    }

    /**
     * Creates an {@link EmailTemplateParams} which already contains values for the following "site-related" keys:
     *
     * <ul>
     * <li>{@link PKBConstants#PKB_SITE_URL}</li>
     * <li>{@link PKBConstants#PKB_LOGO_URL}</li>
     * <li>{@link PKBConstants#PKB_LOGO_BASE_DIR}</li>
     * <li>{@link PKBConstants#PKB_HELP_NAME}</li>
     * <li>{@link PKBConstants#PKB_HELP_WEBSITE}</li>
     * <li>{@link PKBConstants#PKB_HELP_EMAIL}</li>
     * <li>{@link PKBConstants#PKB_HELP_PHONE}</li>
     * <li>{@link PKBConstants#BASE_URL}</li>
     * </ul>
     */
    @Override
    public EmailTemplateParams createWithSiteProps() {
        return new EmailTemplateParams(config.getBaseURL())
                .param(PKB_SITE_URL, config.getBaseURL())
                .urlParam(PKB_LOGO_URL, config.getPKBLogoURL())
                .urlParam(PKB_LOGO_BASE_DIR, config.getInstituteLogoBaseDir())
                .param(PKB_HELP_NAME, config.getHelpName())
                .param(PKB_HELP_WEBSITE, config.getHelpWebSite())
                .param(PKB_HELP_EMAIL, config.getHelpEmailAddress())
                .param(PKB_HELP_PHONE, config.getHelpPhone())
                .param(BASE_URL, config.getBaseURL());
    }

}
