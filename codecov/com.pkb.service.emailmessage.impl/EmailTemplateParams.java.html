package com.pkb.service.emailmessage.impl;

import java.net.URI;
import java.net.URISyntaxException;
import java.util.HashMap;

public class EmailTemplateParams extends HashMap<String, Object> {

    private static URI parse(String baseURL) {
        if (!baseURL.endsWith("/")) {
            baseURL += "/";
        }
        try {
            return new URI(baseURL);
        } catch (URISyntaxException e) {
            throw new RuntimeException(e);
        }
    }

    private final URI baseURI;

    public EmailTemplateParams(String baseURI) {
        this.baseURI = parse(baseURI);
    }

    public EmailTemplateParams param(String paramName, Object paramValue) {
        put(paramName, paramValue);
        return this;
    }

    public EmailTemplateParams urlParam(String paramName, String relativeURL) {
        URI resolved = baseURI.resolve(relativeURL);
        put(paramName, resolved.toString());
        return this;
    }

}
