package com.pkb.service.emailmessage.impl;

import java.util.Optional;
import java.util.function.BiPredicate;

import org.immutables.value.Value;

import com.google.common.base.Preconditions;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.user.entity.PKBPerson;

@Value.Immutable
public interface EmailInfo {
    PKBPerson recipient();
    Optional<PKBPerson> sender();
    Optional<Team> team();
    Optional<Org> org();
    Boolean includeSender();

    default String getOrgName() {
        return org().map(Org::getName)
                .or(() -> team().map(team -> {
                    if (team.getOrg() != null) {
                        return team.getOrg().getName();
                    } else {
                        return null;
                    }
                }).or(() -> sender().map(PKBPerson::getOrganizationName).or(Optional::empty)))
                .orElse("");
    }

    @Value.Check
    default void check() {
        BiPredicate<Boolean, PKBPerson> senderPresent = (includeSender, sender) -> !includeSender || sender != null;
        Preconditions.checkState(senderPresent.test(includeSender(), sender().orElse(null)),
                "Sender must be present if includeSender is true.");
    }
}
