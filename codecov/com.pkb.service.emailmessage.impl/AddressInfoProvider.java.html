package com.pkb.service.emailmessage.impl;

import com.pkb.datamodel.Email;
import com.pkb.datamodel.consent.RequiresConsent;
import com.pkb.dto.InviteeDTO;
import com.pkb.exception.FormattingException;
import com.pkb.exception.MailException;
import com.pkb.exception.NoEmailAddressException;
import com.pkb.service.emailmessage.IAddressInfoProvider;
import com.pkb.service.user.impl.PrimaryEmailContactManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.user.entity.PersonContact;
import com.pkb.user.entity.PrimaryEmailContactDTO;
import com.pkb.util.security.RelaxedHtmlMessageFilter;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import io.vavr.collection.Stream;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.mail.internet.InternetAddress;
import java.io.UnsupportedEncodingException;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;

import static io.vavr.API.unchecked;

public final class AddressInfoProvider implements IAddressInfoProvider {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private PrimaryEmailContactManager primaryEmailContactManager;

    private RelaxedHtmlMessageFilter htmlFilter;

    public AddressInfoProvider() {
    }

    @Override
    public Optional<AddressInfo> getPrimaryContactOnly(PKBPerson person) {
        return getPersonPrimaryEmail(person)
                .map(e -> new AddressInfo(e.toInternetAddress()));
    }

    @Override
    public AddressInfo getPrimaryAndSecondaryContacts(PKBPerson person, RequiresConsent consentRequired) {
        return getPrimaryAndSecondaryContacts(person, false, Optional.empty(), consentRequired);
    }

    @Override
    public AddressInfo getConfirmedOnlyPrimaryAndSecondaryContacts(PKBPerson person, RequiresConsent consentRequired) {
        return getPrimaryAndSecondaryContacts(person, true, Optional.empty(), consentRequired);
    }

    private InternetAddress getPrimaryContact(PKBPerson person, boolean confirmedOnly, RequiresConsent consentRequired) {
        return primaryEmailContactManager.getPrimaryEmailContactDTO(person.getId(), consentRequired)
                .findPrimaryEmailContact()
                .filter(contact -> contact.isConfirmed() || !confirmedOnly)
                .flatMap(PersonContact::getContactIfEmail)
                .map(email -> {
                    InternetAddress internetAddress = email.toInternetAddress();
                    try {
                        internetAddress.setPersonal(htmlFilter.filter(person.getFirstName() + " " + person.getLastName()));
                    } catch (UnsupportedEncodingException e) {
                        LOGGER.warn("Error while formatting the email address", e);
                    }
                    return internetAddress;
                })
                .getOrNull();
    }

    private Map<InternetAddress, PKBPerson> getAlsoContacts(PKBPerson person,
            PKBPerson personToExclude,
            RequiresConsent consent) {
        if (personToExclude != null && personToExclude.getId() != null && personToExclude.getId().equals(person.getId())) {
            LOGGER.warn("*** personToExclude property specifies the primary recipient of this email. This has been ignored");
        }
        PrimaryEmailContactDTO dto = primaryEmailContactManager.getPrimaryEmailContactDTO(person.getId(), consent);
        List<PersonContact> contacts = dto.getAlsoContactPrimaryEmailContactList().stream()
                .filter(PersonContact::isConfirmed)
                .collect(Collectors.toList());
        
        Option<Email> emailToExclude = Option.of(personToExclude).flatMap(PKBPerson::findEmail);

        return Stream.ofAll(contacts)
                .filter(personContact -> !emailToExclude.equals(personContact.getContactIfEmail()))
                .flatMap(personContact ->
                        personContact.getContactIfEmail()
                        .map(email -> fromPersonAndEmail(personContact.getPerson(), email))
                        .map(internetAddress -> Tuple.of(internetAddress, personContact.getPerson()))
                        .toStream())
                .toMap(Tuple2::_1, Tuple2::_2)
                .toJavaMap();
    }

    private static InternetAddress fromPersonAndEmail(@NotNull PKBPerson person, @NotNull Email email) {
        try {
            return new InternetAddress(email.address(), person.getFirstName() + " " + person.getLastName());
        } catch (UnsupportedEncodingException e) {
            throw new RuntimeException("Error while formatting the email address", e);
        }
    }

    @Override
    public AddressInfo getPrimaryAndSecondaryContacts(PKBPerson person, boolean confirmedOnly, Optional<PKBPerson> alsoContactToExclude,
            RequiresConsent consentRequired) {

        InternetAddress primaryAddress = getPrimaryContact(person, confirmedOnly, consentRequired);
        Map<InternetAddress, PKBPerson> alsoContacts = getAlsoContacts(person, alsoContactToExclude.orElse(null), consentRequired);
        return new AddressInfo(primaryAddress, alsoContacts);
    }

    @Override
    public AddressInfo getSenderContact(PKBPerson person) throws MailException {

        Optional<Email> maybeEmail = getPersonPrimaryEmail(person);
        if (maybeEmail.isPresent()) {
            Email email = maybeEmail.get();
            InternetAddress internetAddress = email.toInternetAddress();

            try {
                internetAddress.setPersonal(htmlFilter.filter(person.getFirstName() + " " + person.getLastName()));
                return new AddressInfo(internetAddress);
            } catch (UnsupportedEncodingException e) {
                throw new FormattingException("Error while formatting the email address", e);
            }
        } else {
            throw new NoEmailAddressException("Sender has no email address!", null);
        }
    }

    @Override
    public Option<AddressInfo> findInviteeContact(@Nullable InviteeDTO inviteeDTO) throws FormattingException {

        return Option.of(inviteeDTO)
                .flatMap(unchecked(dto -> getContact(dto.getFirstName(), dto.getLastName(), dto.getEmailId())));
    }

    @Override
    public Option<AddressInfo> getContact(@Nullable Email emailOrNull) {
        return Option.of(emailOrNull).map(email -> new AddressInfo(email.toInternetAddress()));
    }

    @Override
    public Option<AddressInfo> getContact(String firstName, String lastName, @Nullable Email nullableEmail)
            throws FormattingException {

        return Option.of(nullableEmail)
                .map(unchecked(email -> {
                    InternetAddress internetAddress = email.toInternetAddress();
                    try {
                        internetAddress.setPersonal(firstName + " " + lastName);
                        return new AddressInfo(internetAddress);
                    } catch (UnsupportedEncodingException e) {
                        throw new FormattingException("Error while formatting the email address", e);
                    }
                }));
    }

    private Optional<Email> getPersonPrimaryEmail(PKBPerson person) {
        PrimaryEmailContactDTO primaryEmailContactDTO = primaryEmailContactManager.getPrimaryEmailContactDTO(person.getId(),
                null /*consentRequired - not needed as only want the recipient not their ALSO_CONTACTs*/);
        return primaryEmailContactDTO.findPrimaryEmailContact()
                .flatMap(PersonContact::getContactIfEmail)
                .toJavaOptional();
    }

    public PrimaryEmailContactManager getPrimaryEmailContactManager() {
        return primaryEmailContactManager;
    }

    public void setPrimaryEmailContactManager(PrimaryEmailContactManager primaryEmailContactManager) {
        this.primaryEmailContactManager = primaryEmailContactManager;
    }

    public void setHtmlFilter(RelaxedHtmlMessageFilter htmlFilter) {
        this.htmlFilter = htmlFilter;
    }
}
