package com.pkb.admin.action;

import com.google.common.collect.Lists;
import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.domain.SciStoreService;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.entities.enums.ValidNationalId;
import com.pkb.integration.sciStore.SciStoreMismatch;
import com.pkb.integration.sciStore.SciStorePatient;
import com.pkb.service.integration.SciStoreManager;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.PersonAmbiguityHandler;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;
import uk.nhs.scot.show.isd.scistore.FindPatientItem;

import java.util.List;
import java.util.Optional;

public class ManageSciStoreAction extends BaseAction {
    private static final long serialVersionUID = 1L;

    private String sessionToken;
    private String CHINumber;
    private SciStorePatient sciStorePatient;
    private String sciStorePatientId;
    private String sciStoreSubscriptionStatus;
    private List<SciStoreMismatch> mismatchedPatients;

    private PKBPerson patient;
    private String pkbPatientId;

    private UserAccessManager userAccessManager;
    private SciStoreManager sciStoreManager;
    private SciStoreService sciStoreService;
    private ThreadPoolTaskExecutor taskExecutor;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    public String getTab() {
        return "sciStoreReview";
    }

    @Override
    public String execute() {
        try {
            long orgId = loggedInUserUtil.getLoggedInUserOrg().getId();
            List<Long> mismatchedPKBPatientIds = sciStoreService.getMismatchedPatientIds(orgId);
            List<PKBPerson> mismatchedPKBPatients = userManager
                    .getPKBPersonList(mismatchedPKBPatientIds, PKBPerson.Lazy.NATIONAL_AND_LOCAL_IDS);
            mismatchedPatients = sciStoreManager.getMismatchedPatients(mismatchedPKBPatients);
        } catch (Exception e) {
            LOGGER.error("Error on SciStore tab", e);
            //noinspection StrutsI18nError,SSBasedInspection
            addActionError(e.getMessage());
        }
        return Action.SUCCESS;
    }

    public String approveSync() throws Exception {
        PKBPerson patient = userManager.getPKBPerson(Long.parseLong(pkbPatientId), PKBPerson.Lazy.NATIONAL_AND_LOCAL_IDS);

        SyncSinglePatientTask syncSinglePatientTask = new SyncSinglePatientTask(sciStoreManager, patient, getEHRRequestContext());
        taskExecutor.execute(syncSinglePatientTask);
        addActionMessage(getText("scistore.sync.triggered.single.patient", Lists.newArrayList(patient.getId())));
        return Action.SUCCESS;
    }

    public String pullFullHistory() throws Exception {
        PKBPerson patient = userManager.getPKBPerson(Long.parseLong(pkbPatientId), PKBPerson.Lazy.NATIONAL_AND_LOCAL_IDS);

        PullFullHistoryTask pullTask = new PullFullHistoryTask(sciStoreManager, patient, getEHRRequestContext());
        taskExecutor.execute(pullTask);
        addActionMessage(getText("scistore.sync.triggered.history", Lists.newArrayList(patient.getId())));
        return Action.SUCCESS;
    }

    public String syncNewPatients() {
        SyncNewPatientsTask syncNewPatientsTask = new SyncNewPatientsTask(sciStoreManager);
        taskExecutor.execute(syncNewPatientsTask);
        addActionMessage(getText("scistore.sync.triggered"));
        return Action.SUCCESS;
    }

    public String pollForNotifications() {
        PollForNotificationsTask pollForNotificationsTask = new PollForNotificationsTask(sciStoreManager);
        taskExecutor.execute(pollForNotificationsTask);
        addActionMessage(getText("scistore.poll.triggered"));
        return Action.SUCCESS;
    }

    private class SyncNewPatientsTask implements Runnable {
        private SciStoreManager sciStoreManager;

        public SyncNewPatientsTask(SciStoreManager sciStoreManager) {
            this.sciStoreManager = sciStoreManager;
        }

        @Override
        public void run() {
            sciStoreManager.syncNewPatients(); // Handles it's own context
        }
    }

    private class SyncSinglePatientTask implements Runnable {
        private SciStoreManager sciStoreManager;
        private PKBPerson patient;
        private EHRRequestContext ehrRequestContext;

        public SyncSinglePatientTask(SciStoreManager sciStoreManager, PKBPerson patient, EHRRequestContext ehrRequestContext) {
            this.sciStoreManager = sciStoreManager;
            this.patient = patient;
            this.ehrRequestContext = ehrRequestContext;
        }

        @Override
        public void run() {
            sciStoreManager.syncSinglePatient(ehrRequestContext, patient, SciStoreManager.MatchType.PREAPPROVED);
        }
    }

    private class PullFullHistoryTask implements Runnable {
        private SciStoreManager sciStoreManager;
        private PKBPerson patient;
        private EHRRequestContext ehrRequestContext;

        public PullFullHistoryTask(SciStoreManager sciStoreManager, PKBPerson patient, EHRRequestContext ehrRequestContext) {
            this.sciStoreManager = sciStoreManager;
            this.patient = patient;
            this.ehrRequestContext = ehrRequestContext;
        }

        @Override
        public void run() {
            sciStoreManager.pullFullHistory(ehrRequestContext, patient);
        }
    }

    private class PollForNotificationsTask implements Runnable {
        private SciStoreManager sciStoreManager;

        public PollForNotificationsTask(SciStoreManager sciStoreManager) {
            this.sciStoreManager = sciStoreManager;
        }

        @Override
        public void run() {
            sciStoreManager.pollForNotifications(); // Handles it's own context
        }
    }

    //
    // Developer tab actions - TODO all going away soon
    //

    public String getPatientByCHInumber() throws Exception {
        Optional<ValidNationalId> maybeValidNationalId = NationalIdType.CHI_NUMBER.getValidNationalIdAndType(CHINumber);
        maybeValidNationalId.ifPresent(vnid -> {
            patient = PersonAmbiguityHandler.getMaybePersonOrNone(userManager.getPKBPersonByNationalId(vnid), "nationalId").getOrNull();

            Optional<FindPatientItem> findPatientItem = sciStoreManager.findPatient(vnid.value());

            findPatientItem.ifPresent(p -> {
                sciStorePatient = new SciStorePatient(p);
                sciStorePatientId = sciStorePatient.getSciStoreId();
                try {
                    sciStoreService
                            .findPublicSciStorePatientCredentialsByPkbPatientId(patient.getId())
                            .ifPresent(creds -> sciStoreSubscriptionStatus = creds.getStatus().toString());
                } catch (Exception e) {
                    LOGGER.error("failed getting patient credentials for sciStore", e);
                }
            });
        });

        return Action.SUCCESS;
    }

    public String getCHINumber() {
        return CHINumber;
    }

    public void setCHINumber(String cHINumber) {
        CHINumber = cHINumber;
    }

    public SciStorePatient getSciStorePatient() {
        return sciStorePatient;
    }

    public PKBPerson getPatient() {
        return patient;
    }

    public void setPkbPatientId(String pkbPatientId) {
        this.pkbPatientId = pkbPatientId;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public void setUserAccessManager(UserAccessManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }

    public void setTaskExecutor(ThreadPoolTaskExecutor taskExecutor) {
        this.taskExecutor = taskExecutor;
    }

    public void setSciStoreManager(SciStoreManager sciStoreManager) {
        this.sciStoreManager = sciStoreManager;
    }

    public void setSciStoreService(SciStoreService sciStoreService) {
        this.sciStoreService = sciStoreService;
    }

    public List<SciStoreMismatch> getMismatchedPatients() {
        return mismatchedPatients;
    }

    public String getSciStoreSubscriptionStatus() {
        return sciStoreSubscriptionStatus;
    }
}
