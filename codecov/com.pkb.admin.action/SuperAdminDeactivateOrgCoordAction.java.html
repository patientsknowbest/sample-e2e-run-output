package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;

public class SuperAdminDeactivateOrgCoordAction extends BaseAction {

    private static final long serialVersionUID = -486879433024242535L;

    private Long orgCoordId;
    private PKBPerson orgCoord;
    private Long orgId;

    private UserAccessManager userAccessManager;


    public String deactivateOrgCoordForm() {
        setOrgCoord(userManager.getPKBPerson(orgCoordId));
        return SUCCESS;
    }

    public String deactivateOrgCoord() {
        userAccessManager.deactivateUserAccount(getLoggedInEHRRequestContext(), orgCoordId);
        addActionMessage(getText("deactivate.admin.successful"));
        return SUCCESS;
    }

    @Override
    public void validate() {
        if (orgCoordId == null) {
            addActionError(getText("err.no.org.admin"));
        }
    }

    public Long getOrgCoordId() {
        return orgCoordId;
    }

    public void setOrgCoordId(Long orgCoordId) {
        this.orgCoordId = orgCoordId;
    }

    public PKBPerson getOrgCoord() {
        return orgCoord;
    }

    public void setOrgCoord(PKBPerson orgCoord) {
        this.orgCoord = orgCoord;
    }

    public Long getOrgId() {
        return orgId;
    }

    public void setOrgId(Long orgId) {
        this.orgId = orgId;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public UserAccessManager getUserAccessManager() {
        return userAccessManager;
    }

    public void setUserAccessManager(UserAccessManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }
}
