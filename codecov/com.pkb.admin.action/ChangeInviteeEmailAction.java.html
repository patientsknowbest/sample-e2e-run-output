package com.pkb.admin.action;

import com.google.common.collect.Lists;
import com.pkb.VersionDetails;
import com.pkb.action.BaseAction;
import com.pkb.datamodel.Email;
import com.pkb.entities.enums.UserStatus;
import com.pkb.exception.MailException;
import com.pkb.institute.entity.InstituteUser;
import com.pkb.institute.entity.Team;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.user.entity.PasswordAccess;
import io.vavr.control.Option;
import org.apache.struts2.interceptor.SessionAware;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * The institute coordinator can change the email stored for any patient or
 * clinician invited to the team who hasn't yet completed registration; save the change
 * then re-send the invitation (or what have you).
 * <p>
 * Not valid for:
 * Created patient accounts (no access yet to the secret password)
 * Anyone else? What about users that just haven't yet confirmed
 * their email addresses (but are otherwise registered)?
 *
 * @author robwhelan
 */
public class ChangeInviteeEmailAction extends BaseAction implements SessionAware {

    private static final long serialVersionUID = 1L;

    private static final String INPUT_PATIENTS = "inputPatients";
    private static final String INPUT_CLINICIANS = "inputClinicians";
    private static final String SUCCESS_CLINICIANS = "successClinicians";
    private static final String SUCCESS_COORDINATORS = "successCoordinators";

    private Long personId;

    private Email email;

    private PKBPerson person;

    private TeamUserManager teamUserManager;

    private PKBEmailMessageManager pkbEmailMessageManager;

    private UserAccessManager userAccessManager;

    private String tab; // patients or clinicians

    private String subTab; // nonregPatients or null

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Override
    public String execute() {
        LOGGER.info("Entering ChangeInviteeEmailAction execute");
        Email validEmail = email;

        person = userManager.getPKBPerson(personId, PKBPerson.Lazy.CONTACTS);
        if (person == null) {
            addActionError(getText("err.invite.no.user.found", new String[]{Long.toString(personId)}));
            // SHORT-CIRCUIT
            return INPUT;
        }

        // side-effect: sets instituteUser
        commonSetupAndValidation();

        // is email (lowercase, trimmed) unchanged?
        if (Option.of(validEmail).map(email -> email.address().equals(person.findEmail().map(Email::address).getOrNull())).getOrElse(false)) {
            addActionError(getText("err.invite.email.case.sensitivity"));
        } else {
            // TODO: MULTIPLE-TEAMS skip this if we're inviting a clinician?
            PKBPerson emailHolder = userManager.getSinglePersonByEmail(validEmail);
            if ((emailHolder != null) && !emailHolder.getId().equals(personId)) {
                addActionError(getText("err.invite.email.already.in.use"));
            }
        }

        if (hasActionErrors()) {
            return INPUT;
        }

        try {
            userManager.beginTransaction();

            // make change
            person.setPrimaryEmail(validEmail, VersionDetails.of(getEHRRequestContext(), dateTimeService.now()));
            userManager.updateUser(getEHRRequestContext(), person);

            userManager.commitTransaction();

            addActionMessage(getText("invite.email.change.success", Lists.newArrayList(person.getTitle(), person.getFirstName(), person.getLastName())));
        } finally {
            userManager.rollbackIfNotCommitted();
        }

        try {
            // re-send email: invitation, or request to confirm email
            if ((person.getStatus() == UserStatus.CREATED || person.getStatus() == UserStatus.NOCONTACT) && person.isPro()) {
                remindClinicianToRegister(person);
            }
        } catch (MailException e) {
            addActionError(getText("err.invite.email.send.fail"));
            LOGGER.error("Error while sending mail for changed invitee email", e);
            // return success anyway: only email failed
        }

        String successResult;
        switch (person.getUserType()) {
            case INSTITUTE_ADMIN:
                successResult = SUCCESS_COORDINATORS;
                break;
            default:
                successResult = SUCCESS_CLINICIANS;
                break;
        }

        LOGGER.info("Exiting ChangeInviteeEmailAction execute");
        return successResult;
    }

    /**
     * @param reminder
     * @param clinician
     * @throws MailException
     */
    public void remindClinicianToRegister(PKBPerson clinician) throws MailException {
        PKBPerson reminder = userManager.getPKBPerson(loggedInUserUtil.getLoggedInUserId());

        Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();

        PasswordAccess passwordAccess = userAccessManager.getPasswordAccess(clinician.getId(),
                getLoggedInEHRRequestContext());
        if (passwordAccess == null) {
            throw new NullPointerException("No passwordAccess stored for patient " + clinician.getId()); // TODO: handle this (old user) more politely
        }

        clinician.setPassword(passwordAccess.getPassword());

        pkbEmailMessageManager.notifyClinicianOfCreatedAccount(reminder, clinician, team, null);
    }

    @Override
    public String input() {

        LOGGER.info("ChangeInviteeEmailAction input() person id of user to change email of {}", personId);

        person = userManager.getPKBPerson(personId, PKBPerson.Lazy.CONTACTS);
        if (person == null) {
            addActionError(getText("err.invite.no.user.found", new String[]{Long.toString(personId)}));
            // SHORT-CIRCUIT
            return INPUT_PATIENTS; // just guess; this isn't a likely error
        }

        String inputResult = INPUT_PATIENTS;
        if (!person.isPatient()) {
            inputResult = INPUT_CLINICIANS;
        }

        commonSetupAndValidation();

        if (hasActionErrors()) {
            return inputResult;
        } else {
            // populate the form and continue!
            email = person.getEmail();
            return SUCCESS;
        }
    }

    private void commonSetupAndValidation() {
        // find the institute user link -- is the user on this coordinator's team?
        Long instituteId = loggedInUserUtil.getLoggedInUserPrimaryTeam().getId();
        // used internally (but not exposed/used in JSP)
        InstituteUser instituteUser = teamUserManager.getInstituteUser(instituteId, personId);
        // TODO: check changingIU.getSponsorshipStatus()?
        if (instituteUser == null) {
            addActionError(getText("err.invite.unexpected.not.invited"));
        } else if (person.isPatient()) {
            tab = "patients";
            subTab = "nonregPatients";

            // check user status: NEW or INVITED only are okay (to add later: CREATED)
            // not okay: EMAIL_CONFIRMED, ACTIVE, INACTIVE
            UserStatus status = person.getStatus();
            if (status != UserStatus.NEW && status != UserStatus.INVITED) {
                addActionError(getText("err.invite.patient.status.cannot.change"));
            }
        } else if (person.isPro()) {
            tab = "clinicians";
            subTab = null;

            // check user status: NEW, INVITED,
            UserStatus status = person.getStatus();
            if (status != UserStatus.NEW && status != UserStatus.INVITED && status != UserStatus.NOCONTACT
                    && status != UserStatus.CREATED) {
                addActionError(getText("err.invite.clinician.status.cannot.change"));
            }
        } else if (person.isTeamCoordinator()) {
            tab = "coordinators";
            subTab = "inviteCoordinator";

            UserStatus status = person.getStatus();
            if (status != UserStatus.CREATED) {
                addActionError(getText("err.invite.coord.status.cannot.change"));
            }
        } else {
            // neither patient nor clinician nor coordinator
            addActionError(getText("err.invite.unexpected.not.a.member"));
        }
    }

    @Override
    public void validate() {
        // only called for execute() -- check the email added is not empty
        // Pravina: Added the struts email format check through validation xml
        if (email == null) {
            addFieldError("email", getText("err.invite.field.email.not.valid"));
        }
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public String getTab() {
        return tab;
    }

    public void setTab(String tab) {
        this.tab = tab;
    }

    public Email getEmail() {
        return email;
    }

    public void setEmail(Email email) {
        this.email = email;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }

    public String getSubTab() {
        return subTab;
    }

    public void setSubTab(String subTab) {
        this.subTab = subTab;
    }

    public void setTeamUserManager(TeamUserManager teamUserManager) {
        this.teamUserManager = teamUserManager;
    }

    public PKBPerson getPerson() {
        return person;
    }

    public void setPkbEmailMessageManager(PKBEmailMessageManager pkbEmailMessageManager) {
        this.pkbEmailMessageManager = pkbEmailMessageManager;
    }

    public UserAccessManager getUserAccessManager() {
        return userAccessManager;
    }

    public void setUserAccessManager(UserAccessManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }

}
