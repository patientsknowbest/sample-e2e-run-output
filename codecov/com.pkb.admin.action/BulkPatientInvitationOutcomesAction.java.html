package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.patient.invite.BulkPatientInvitationOutcomesDto;
import com.pkb.patient.invite.bulk.BulkPatientInvitationOutcomesService;
import org.springframework.beans.factory.annotation.Autowired;

public class BulkPatientInvitationOutcomesAction extends BaseAction {

    private String tab = "patients";
    private String subTab;

    private BulkPatientInvitationOutcomesDto viewModel;

    private BulkPatientInvitationOutcomesService outcomesService;

    public String listLastBulkUploadOutcomes() {
        long teamId = loggedInUserUtil.getLoggedInUserPrimaryTeam(getEHRRequestContext()).getId();

        viewModel = outcomesService.listOutcomes(teamId);

        return SUCCESS;
    }

    // ========== Accessor methods below. ==========

    public String getTab() {
        return tab;
    }

    public void setTab(String tab) {
        this.tab = tab;
    }

    public String getSubTab() {
        return subTab;
    }

    public void setSubTab(String subTab) {
        this.subTab = subTab;
    }

    public BulkPatientInvitationOutcomesDto getViewModel() {
        return viewModel;
    }

    public BulkPatientInvitationOutcomesService getOutcomesService() {
        return outcomesService;
    }

    @Autowired
    public void setOutcomesService(BulkPatientInvitationOutcomesService outcomesService) {
        this.outcomesService = outcomesService;
    }
}
