package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.exception.PKBException;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;

/**
 * @author Russ Gray
 */
public class SuperAdminDeactivatePrivacyOfficerAction extends BaseAction {

    private static final long serialVersionUID = 5158016004790178338L;

    private Long orgId;
    private Long privacyOfficerId;
    private PKBPerson privacyOfficer;

    private boolean appointmentMessagingTeamEnabled; // Pass back to viewOrg.action

    private UserAccessManager userAccessManager;

    public String deactivateForm() throws PKBException {
        setPrivacyOfficer(userManager.getPKBPerson(privacyOfficerId));
        return SUCCESS;
    }

    public String deactivate() {
        userAccessManager.deactivateUserAccount(getLoggedInEHRRequestContext(), privacyOfficerId);
        addActionMessage(getText("deactive.privacy.officer.successful"));
        return SUCCESS;
    }

    @Override
    public void validate() {
        if (privacyOfficerId == null) {
            addActionError(getText("err.field.no.privacy.officer"));
        }
    }

    public Long getOrgId() {
        return orgId;
    }

    public void setOrgId(Long orgId) {
        this.orgId = orgId;
    }

    public Long getPrivacyOfficerId() {
        return privacyOfficerId;
    }

    public void setPrivacyOfficerId(Long privacyOfficerId) {
        this.privacyOfficerId = privacyOfficerId;
    }

    public PKBPerson getPrivacyOfficer() {
        return privacyOfficer;
    }

    public void setPrivacyOfficer(PKBPerson privacyOfficer) {
        this.privacyOfficer = privacyOfficer;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public UserAccessManager getUserAccessManager() {
        return userAccessManager;
    }

    public void setUserAccessManager(UserAccessManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }

    public boolean isAppointmentMessagingTeamEnabled() {
        return appointmentMessagingTeamEnabled;
    }

    public void setAppointmentMessagingTeamEnabled(boolean appointmentMessagingTeamEnabled) {
        this.appointmentMessagingTeamEnabled = appointmentMessagingTeamEnabled;
    }
}
