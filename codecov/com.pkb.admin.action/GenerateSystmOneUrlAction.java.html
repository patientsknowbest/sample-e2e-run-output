package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.crypto.util.RandomUtil;
import com.pkb.domain.SSOService;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.UUID;

import static com.pkb.util.SSOConstants.SYSTMONE_SSO_IV_LENGTH;
import static com.pkb.util.SSOConstants.SYSTMONE_SSO_KEY_LENGTH;

public class GenerateSystmOneUrlAction extends BaseAction {

    private String encryptionKey;
    private String initialisationVector;
    private String parameterEncryptionExternalId;
    private String baseUrl;

    private SSOService ssoService;

    @Override public String execute() throws Exception {

        // 32 length hex string = 128 bit key, so this is AES128
        // This is the max length of the field in SystmOne. Unlikely to get them to increase it to allow for AES256
        encryptionKey = RandomUtil.randomHexString(SYSTMONE_SSO_KEY_LENGTH);
        initialisationVector = RandomUtil.randomHexString(SYSTMONE_SSO_IV_LENGTH);

        parameterEncryptionExternalId = UUID.randomUUID().toString();

        baseUrl = getConfig().getBaseURL();

        ssoService.createSSOParameterEncryptionDetails(parameterEncryptionExternalId, encryptionKey, initialisationVector);
        return SUCCESS;
    }

    public String getEncryptionKey() {
        return encryptionKey;
    }

    public String getInitialisationVector() {
        return initialisationVector;
    }

    public String getParameterEncryptionExternalId() {
        return parameterEncryptionExternalId;
    }

    public String getBaseUrl() {
        return baseUrl;
    }

    public String getTab() {
        return "connectedSystems";
    }

    @Autowired
    public void setSsoService(SSOService ssoService) {
        this.ssoService = ssoService;
    }
}
