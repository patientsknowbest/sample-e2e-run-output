package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.datamodel.Affiliate;
import com.pkb.datamodel.ImmutableAffiliate;
import com.pkb.entities.enums.UserStatus;
import com.pkb.exception.PKBException;
import com.pkb.institute.entity.Team;
import com.pkb.service.team.AffiliateManager;
import com.pkb.service.team.TeamManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.tolven.TolvenBeanFactory;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.List;

/**
 * Action class to add, delete affiliate
 *
 * @author pravinam
 *
 */
public class ManageAffiliateAction extends BaseAction {
    private static final long serialVersionUID = 1L;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private TolvenBeanFactory beanFactory;

    private TeamManager teamManager;

    private AffiliateManager affiliateManager;

    private String affiliateCode;

    private Long affiliateId;

    private Affiliate affiliateToDelete;

    private String parentTeamCode;

    private Team affiliateTeamToDelete;

    /**
     * Add the affiliate team
     */
    @Override
    public String execute() {
        try {
            Team parentTeam = teamManager.getTeamByCode(parentTeamCode);
            Team affiliateTeam = teamManager.getTeamByCode(affiliateCode);
            if (affiliateTeam == null) {
                addActionError(getText("error.institute.validate.code"));
                return INPUT;
            }

            // Check of the admin exists or inactive for this team
            List<PKBPerson> affiliateAdmins = teamManager.getTeamCoord(affiliateTeam.getId());
            boolean affiliateAdminActive = false;
            for (PKBPerson affiliateAdmin : affiliateAdmins) {
                if (affiliateAdmin.getStatus() == UserStatus.EMAIL_CONFIRMED) {
                    affiliateAdminActive = true;
                    break;
                }
            }
            if (!affiliateAdminActive) {
                addActionError(getText("institute.admin.null.or.not.active",
                        new String[] { affiliateTeam.getName() }));
                return INPUT;
            }

            Affiliate affiliate = ImmutableAffiliate.builder()
                    .affiliateTeamId(affiliateTeam.getId())
                    .teamId(parentTeam.getId())
                    .build();

            try {
                affiliateManager.beginTransaction();
                affiliateManager.saveAffiliate(affiliate);
                affiliateManager.commitTransaction();
                addActionMessage(getText("affiliate.add.success"));
            } finally {
                affiliateManager.rollbackIfNotCommitted();
            }
            return SUCCESS;
        } catch (Exception e) {
            LOGGER.error("Exception while adding affiliate-{}", e.getMessage());
            return INPUT;
        }
    }

    public String deleteConfirm() {
        try {
            affiliateToDelete = affiliateManager.getAffiliate(affiliateId);
            affiliateTeamToDelete = teamManager.getTeam(affiliateToDelete.getAffiliateTeamId());
        } catch (PKBException e) {
            LOGGER.error("Exception while fetching the affiliate teamm to be deleted{}", e.getMessage());
        }
        return SUCCESS;
    }

    public String delete() {
        try {
            try {
                affiliateManager.beginTransaction();
                affiliateManager.deleteAffiliate(affiliateId);
                affiliateManager.commitTransaction();
                addActionMessage(getText("affiliate.delete.success"));
            } finally {
                affiliateManager.rollbackIfNotCommitted();
            }
        } catch (Exception e) {
            addActionError(getText("affiliate.delete.failure"));
            LOGGER.error("Exception while deleting affiliate-{}", e.getMessage());
        }
        return SUCCESS;
    }

    @Override
    public String input() {
        return SUCCESS;
    }

    public String cancel() {
        return SUCCESS;
    }

    public TolvenBeanFactory getBeanFactory() {
        return beanFactory;
    }

    public void setBeanFactory(TolvenBeanFactory beanFactory) {
        this.beanFactory = beanFactory;
    }

    public void setTeamManager(TeamManager teamManager) {
        this.teamManager = teamManager;
    }

    public void setAffiliateManager(AffiliateManager affiliateManager) {
        this.affiliateManager = affiliateManager;
    }

    public String getAffiliateCode() {
        return affiliateCode;
    }

    public void setAffiliateCode(String affiliateCode) {
        this.affiliateCode = affiliateCode;
    }

    public Long getAffiliateId() {
        return affiliateId;
    }

    public void setAffiliateId(Long affiliateId) {
        this.affiliateId = affiliateId;
    }

    public Affiliate getAffiliateToDelete() {
        return affiliateToDelete;
    }

    public void setAffiliateToDelete(Affiliate affiliateToDelete) {
        this.affiliateToDelete = affiliateToDelete;
    }

    public String getParentTeamCode() {
        return parentTeamCode;
    }

    public void setParentTeamCode(String parentTeamCode) {
        this.parentTeamCode = parentTeamCode;
    }

    public Team getAffiliateTeamToDelete() {
        return affiliateTeamToDelete;
    }

    public void setAffiliateTeamToDelete(Team affiliateTeamToDelete) {
        this.affiliateTeamToDelete = affiliateTeamToDelete;
    }
}
