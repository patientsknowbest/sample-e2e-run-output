public class RemindAction extends BaseAction implements SessionAware {
    private static final long serialVersionUID = 10012L;

    private long inviteeId;

    private InviteeDTO inviteeDTO;

    private PKBEmailMessageManager pkbEmailMessageManager;

    private TeamManager teamManager;

    private InvitationManager invitationManager;

    private UserAccessManager userAccessManager;

    private Long userId;

    private long patientId;

    private String invitationId;

    private long caregiverId;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());



    public String remindClinicianToUpgradeAccount() {
        LOGGER.info("Entering execute ");
        String result = Action.SUCCESS;
        try {
            PKBPerson admin = loggedInUserUtil.getLoggedInPerson();
            Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();

            PKBPerson clinician = userManager.getPKBPerson(inviteeId, PKBPerson.Lazy.CONTACTS);
            PKBInvitation invitation = invitationManager.getPKBInvitation(getLoggedInEHRRequestContext(), admin.getId(),
                    clinician.getId(), InvitationType.CLINICIAN_INVITATION_TO_UPGRADE_ACCOUNT);

            pkbEmailMessageManager.inviteClinicianToUpgrade(team,
                    admin, clinician, invitation.getPublicId());
        } catch (PKBException exception) {
            result = Action.ERROR;
            addActionError(getText("error.resend.invitation"));
            LOGGER.error("Error while resending invitation", exception);
        } catch (MailException exception) {
            result = Action.ERROR;
            addActionError(getText("error.resend.invitation"));
            LOGGER.error("Mailer exception: {}", exception.getCause(), exception);
        }
        LOGGER.info("Exiting execute result : {}", result);
        return result;
    }

    public String remindCreatedClinicianToRegister() {
        try {

            PKBPerson person = userManager.getPKBPerson(inviteeId, PKBPerson.Lazy.CONTACTS);

            if (person.getEmail() == null) {
                addActionError(getText("error.no.email.for.reminder",
                        new String[]{person.getTitle(), person.getFirstName(), person.getLastName()}));
                return Action.ERROR;
            }

            remindClinicianToRegister(person);

            addActionMessage(getText("registration.remind.success", Lists.newArrayList(person.getTitle(), person.getLastName())));

        } catch (PKBException e) {
            //noinspection StrutsI18nError,SSBasedInspection
            addActionError(e.getMessage());
            LOGGER.error("error", e);
            return Action.ERROR;
        } catch (MailException e) {
            //noinspection StrutsI18nError,SSBasedInspection
            addActionError(e.getMessage());
            LOGGER.error("Mailer exception: {}", e.getCause(), e);
            return Action.ERROR;
        }

        return SUCCESS;
    }

    /**
     * @param reminder
     * @param clinician
     * @throws MailException
     */
    private void remindClinicianToRegister(PKBPerson clinician) throws MailException {
        PKBPerson reminder = userManager.getPKBPerson(loggedInUserUtil.getLoggedInUserId());

        Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();

        PasswordAccess passwordAccess = userAccessManager.getPasswordAccess(clinician.getId(), getLoggedInEHRRequestContext());
        if (passwordAccess == null) {
            throw new NullPointerException("No passwordAccess stored for patient " + clinician.getId()); // TODO: handle this (old user) more politely
        }

        clinician.setPassword(passwordAccess.getPassword());

        pkbEmailMessageManager.notifyClinicianOfCreatedAccount(reminder, clinician, team, null);
    }

    public String resendConfirmationToUser() {
        LOGGER.info("Entering execute ");
        String result = Action.SUCCESS;
        try {
            Team team = teamManager.getTeam(getLoggedInEHRRequestContext().getTeamId()
                    .orElseThrow(() -> new RuntimeException("Failed to find teamId for user")));
            PKBPerson user = userManager.getPKBPerson(userId, PKBPerson.Lazy.CONTACTS);

            pkbEmailMessageManager.sendActivationMessage(user, team, null); // null: specialtyId
        } catch (PKBException exception) {
            result = Action.ERROR;
            addActionError(getText("error.resend.confirmation"));
            LOGGER.error("Error while resending confirmation message", exception);
        }
        LOGGER.info("Exiting execute result : {}", result);
        return result;
    }

    public String resendInvitation() {
        String result = Action.SUCCESS;
        try {
            UUID invitationPublicId = invitationManager.getInvitationIdByString(invitationId);
            PKBInvitation invitation = invitationManager.resendInvitation(getLoggedInEHRRequestContext(), invitationPublicId);
            inviteeId = invitation.getTargetUser().getId();
        } catch (Exception exception) {
            result = Action.ERROR;
            addActionError(getText("resendInvitationAction.err.error_resending_invitation"));
            LOGGER.error("Error while resending invitation {}, {}", invitationId, exception);
        }
        return result;
    }
