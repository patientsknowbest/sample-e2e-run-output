package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.common.config.PhrConfig;
import com.pkb.dto.EmisDiagnosticInformationDto;
import com.pkb.dto.EmisJobInformationDto;
import com.pkb.service.dataupload.emis.esv2.EmisDiagnosticService;
import com.pkb.service.dataupload.emis.esv2.EmisEsService;
import com.pkb.service.dataupload.emis.esv2.EmisEsWhitelistService;

import java.util.List;

class EmisEsAdminAction extends BaseAction {

    private EmisDiagnosticService emisDiagnosticService;
    private EmisEsService emisEsService;
    private EmisEsWhitelistService emisEsWhitelist;

    private List<EmisJobInformationDto> emisEsBatchJobs;

    private List<EmisDiagnosticInformationDto> emisSftpInfo;

    @Override
    public String execute() throws Exception {
        emisEsBatchJobs = emisEsService.getEmisJobInformation();
        return super.execute();
    }

    public String executeBatchJob() throws Exception {
        emisEsService.triggerEmisJob();
        return execute();
    }

    public String emisEsAdminSettings() throws Exception {
        return super.execute();
    }

    public String emisEsAdminDiagnostics() throws Exception {
        emisSftpInfo = emisDiagnosticService.getInformation();
        return super.execute();
    }

    public String emisEsAdminExternalSettings() throws Exception {
        return super.execute();
    }

    public List<EmisJobInformationDto> getEmisEsBatchJobs() {
        return emisEsBatchJobs;
    }

    public PhrConfig getPkbConfig() {
        return getConfig();
    }

    public EmisEsWhitelistService getEmisEsWhitelist() {
        return emisEsWhitelist;
    }

    public List<EmisDiagnosticInformationDto> getEmisSftpInfo() {
        return emisSftpInfo;
    }

    public void setEmisDiagnosticService(EmisDiagnosticService emisDiagnosticService) {
        this.emisDiagnosticService = emisDiagnosticService;
    }

    public void setEmisEsService(EmisEsService emisEsService) {
        this.emisEsService = emisEsService;
    }

    public void setEmisEsWhitelist(EmisEsWhitelistService emisEsWhitelist) {
        this.emisEsWhitelist = emisEsWhitelist;
    }
}
