package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.dto.ChangeOrgInviteeEmailDTO;
import com.pkb.mappers.PKBPersonMapper;
import com.pkb.service.team.OrgManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import org.apache.struts2.interceptor.validation.SkipValidation;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * Action for changing the email address of an invited org-level user (e.g. org
 * coord, privacy officer etc)
 */
public class ChangeOrgInviteeEmailAction extends BaseAction {

    private static final long serialVersionUID = -8396168213872727201L;
    private ChangeOrgInviteeEmailDTO invitee = new ChangeOrgInviteeEmailDTO();

    private OrgManager orgManager;
    @Autowired
    private PKBPersonMapper pkbPersonMapper;

    @Override
    public String execute() {
        PKBPerson person = orgManager.changeEmailForInvitedOrgUser(
                getLoggedInEHRRequestContext(), invitee, loggedInUserUtil.getLoggedInUser());

        addActionMessage(getText("viewOrgDetails.txt.inviteeEmailChanged", new String[]{
                person.getTitle(),
                person.getFirstName(),
                person.getLastName()
        }));
        return SUCCESS;
    }

    @Override
    @SkipValidation
    public String input() {
        pkbPersonMapper.personToChangeOrgInviteeEmailDTO(
                userManager.getPKBPerson(invitee.getPersonId(), PKBPerson.Lazy.CONTACTS),
                invitee);
        return INPUT;
    }

    @Override
    public void validate() {
        if (invitee.getPersonId() == null) {
            addActionError(getText("err.field.no.privacy.officer"));
        }
        if (invitee.getOrgId() == null) {
            addActionError(getText("viewOrgDetails.err.noOrgId"));
        }
        if (invitee.getEmailAddress() == null) {
            addActionError(getText("viewOrgDetails.err.invalidEmailAddress"));
        }
    }

    public ChangeOrgInviteeEmailDTO getInvitee() {
        return invitee;
    }

    public void setInvitee(ChangeOrgInviteeEmailDTO invitee) {
        this.invitee = invitee;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public void setOrgManager(OrgManager orgManager) {
        this.orgManager = orgManager;
    }

    public void setPkbPersonMapper(PKBPersonMapper pkbPersonMapper) {
        this.pkbPersonMapper = pkbPersonMapper;
    }
}
