package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.service.coding.CodingMatchImportService;

import java.io.File;

/**
 * Action for import coding matches to aid with the product-curated mapping process
 */
class CodingMatchImportAdminAction extends BaseAction {

    private CodingMatchImportService codingMatchImportService;

    private File upload;

    @Override
    public String execute() throws Exception {
        return super.execute();
    }

    public String validateMappings() throws Exception {
        LOGGER.info("validate with file {}",  (upload == null ? "null" : upload.getName()));
        CodingMatchImportService.CodingMatchValidationResult codingMatchValidationResult = codingMatchImportService.validateNewMappingsCsv(upload, getLoggedInEHRRequestContext());
        String message = codingMatchValidationResult.getMessage();
        if (codingMatchValidationResult.getSuccessful()) {
            addActionMessage(getText("emisAdmin.txt.coding_match_validation_success", new String[]{message}));
        } else {
            addActionError(getText("emisAdmin.txt.coding_match_validation_failure", new String[]{message}));
        }
        upload = null;
        return "success";
    }

    public String saveMappings() throws Exception {
        LOGGER.info("save with file {}",  (upload == null ? "null" : upload.getName()));
        CodingMatchImportService.CodingMatchImportResult codingMatchImportResult = codingMatchImportService.saveNewMappingsCsv(upload, getLoggedInEHRRequestContext());
        String message = codingMatchImportResult.getMessage();
        if (codingMatchImportResult.getSuccessful()) {
            addActionMessage(getText("emisAdmin.txt.coding_match_import_success", new String[]{message}));
        } else {
            addActionError(getText("emisAdmin.txt.coding_match_import_failure", new String[]{message}));
        }
        upload = null;
        return "success";
    }

    public File getUpload() {
        return upload;
    }

    public void setUpload(File upload) {
        this.upload = upload;
    }

    public CodingMatchImportService getCodingMatchImportService() {
        return codingMatchImportService;
    }

    public void setCodingMatchImportService(CodingMatchImportService codingMatchImportService) {
        this.codingMatchImportService = codingMatchImportService;
    }
}
