package com.pkb.admin.action;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.datamodel.OrgNetworkExisting;
import com.pkb.domain.OrgNetworkService;

import java.util.List;

/**
 *
 * @author robwhelan : for superadmin/employee working with org networks
 *
 */
public class OrgNetworksAction extends BaseAction {

    private static final long serialVersionUID = 7954980060061780922L;

    private final String tab = "orgNetworks";
    private final String subTab = "orgNetworkList";


    private List<OrgNetworkExisting> orgNetworkList;
    private OrgNetworkService orgNetworkService;

    /**
     * Get the list of OrgNetworks
     */
    public String list() {
        try {
            orgNetworkList = orgNetworkService.getOrgNetworkList();

            return Action.SUCCESS;
        } catch (Exception e) {
            throw new RuntimeException("failed simple fetch of org networks!", e);
        }
    }

    public String getTab() {
        return tab;
    }

    public String getSubTab() {
        return subTab;
    }

    public List<OrgNetworkExisting> getOrgNetworkList() {
        return orgNetworkList;
    }

    public void setOrgNetworkService(OrgNetworkService orgNetworkService) {
        this.orgNetworkService = orgNetworkService;
    }
}
