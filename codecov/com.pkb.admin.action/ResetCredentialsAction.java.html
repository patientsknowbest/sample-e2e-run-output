/**
 * Important: this action is used by both institute admin and PKB admin users...
 * if you make changes for one type of password reset, you must also test the other.
 * The PKB admin can reset non-institute user clinicians' passwords.

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    static final String INPUT_PATIENTS = "inputPatients";
    static final String INPUT_CLINICIANS = "inputClinicians";

    @Override
    public String execute() {
        LOGGER.info("Entering ResetCredentialsAction execute");
        String view = INPUT;
        try {

            LoggedInEHRRequestContext requestContext = getLoggedInEHRRequestContext();

            Long organizationId = requestContext.getOrgId().orElse(null);

            teamUserManager.resetUserCredentials(requestContext, this.userId, organizationId);

            addActionMessageAboutSuccessfulTrigger();

            view = selectView(requestContext.getAccessingUser().getUserType());
        } catch (MailException cause) {
            //noinspection StrutsI18nError,SSBasedInspection
            addActionError(cause.getMessage());
            LOGGER.error("Error while sending mail for resetting password", cause);
        }
        LOGGER.info("Exiting ResetCredentialsAction execute");
        return view;
    }

    /**
     * Super-admin enters an email address rather than using password reset links
     * in lists of users
     *
     */
    public String emailInput() {

        // MULTIPLE-TEAMS Superadmin only resets non-inst account, but a multi-team clinician
        // can also have one non-inst account so we might get multiple results back, in which
        // case we have to look for the correct one
        PKBPerson foundUser = null;
        List<PKBPerson> persons = userManager.getPKBPersonByConfirmedOrPrimaryEmail(email, PersonContact.Lazy.PERSON_W_CONTACTS);
        if (!persons.isEmpty()) {
            if (persons.size() == 1) {
                foundUser = persons.get(0);
            } else {
                for (PKBPerson person : persons) {
                    // Not super-efficient to call this in a loop, but OK for small numbers of users
                    if (!teamUserManager.isActiveTeamUser(person.getId())) {
                        foundUser = person;
                        break;
                    }
                }
            }
        }

        if (foundUser == null) {
            addActionError(getText("err.no.user.by.email", new String[]{ Option.of(email).map(Email::address).getOrNull() }));
            return INPUT;
        }

        // valid email - set userId and process like regular input()
        userId = foundUser.getId();
        return input();
    }

    @Override
    public String input() {

        LOGGER.info("ResetCredentialsAction input() user id of user to change password {}", userId);
        // Validate userId
        if ((userId == null)) {
            addActionError(getText("select.user.message"));
            return INPUT_PATIENTS;
        }

        PKBPerson user = userManager.getPKBPerson(userId, PKBPerson.Lazy.CONTACTS);
        if (user.getPrimaryContact() == null) {
            addActionError(getText("err.reset.password.no.identifiers"));
            return INPUT_PATIENTS;
        }

        PKBPerson admin = getContextAndLoggedInUsers().loggedInUser();

        if (StringUtils.isNotBlank(user.getHumanUUID()) && StringUtils.equalsIgnoreCase(user.getHumanUUID(), admin.getHumanUUID())) {
            addActionError(getText("err.reset.password.use.different.link"));
            if (user.isPro()) {
                return INPUT_CLINICIANS;
            } else {
                return INPUT_PATIENTS;
            }
        }

        UserType adminUserType = admin.getUserType();

        if (!validatePermissions()) {
            return  chooseInputResult(adminUserType);
        }

        // check first: does this admin have access to all of the user's linked accounts?
        var warningInfo = userAccessManager.getPasswordResetWarningInfo(user);

        if (!warningInfo.getUnavailableAccountOwners().isEmpty()) {
            if (warningInfo.defaultAccountAvailable()) {
                LOGGER.error("{} WARN: core account is shared, but some other is not", warningInfo);


                addActionError(getText("warn.reset.password.other.requests.needed", new String[]{warningInfo.getUnavailableAccountString()}));
            } else {
                LOGGER.error("{} FAIL: core account not shared", warningInfo);

                addActionError(getText("resetpassword.sorry.unlinked_accounts"));

                return chooseInputResult(adminUserType);
            }
        }
        return SUCCESS;
    }

    private String chooseInputResult(UserType adminUserType) {
        if (adminUserType == UserType.REG_CLINICIAN) {
            return "input";
        } else if (adminUserType == UserType.SUPER_ADMIN) {
            return "input";
        } else if ("patients".equals(tab)) {
            return INPUT_PATIENTS;
        } else if ("clinicians".equals(tab)) {
            return "inputClinicians";
        } else {
            return "inputCoordinators";
        }
    }

    @Override
    public void validate() {
        if (!confirmedIdAccordingToOrgProtocol) {
            addFieldError("confirmedIdAccordingToOrgProtocol", getText("invalid.fieldvalue.identity.not.confirmed"));
        }

        if (isEmpty(trimToEmpty(user.getTitle()))) {
            addFieldError("user.title", getText("user.required.title"));
        }

        if (isEmpty(trimToEmpty(user.getFirstName()))) {
            addFieldError("user.firstName", getText("user.required.firstName"));
        }

        if (isEmpty(trimToEmpty(user.getLastName()))) {
            addFieldError("user.lastName", getText("user.required.lastName"));
        }
        title = user.getTitle();
        firstName = user.getFirstName();
        lastName = user.getLastName();

        validatePermissions();
    }

    private boolean validatePermissions() {
        if (!userAccessManager.shouldHavePermissionToResetPassword(getLoggedInEHRRequestContext(), userId)) {
            addActionError(getText("resetpassword.sorry.unlinked_accounts"));
            return false;
        }

        return true;
    }


    public String cancel() {
        return "successPatients";
    }


    private void addActionMessageAboutSuccessfulTrigger() {
        PKBPerson resetUser = userManager.getPKBPerson(userId);
        addActionMessage(getText("successfully.sentemail.resetpassword",
                new String[]{resetUser.getTitle(), resetUser.getFirstName(), resetUser.getLastName()}));
    }

    private String selectView(UserType loggedInUserType) {
        String result;
        if (UserType.REG_CLINICIAN == loggedInUserType || UserType.SUPER_ADMIN == loggedInUserType) {
            result = "success";
        } else if ("patients".equals(tab)) {
            result = "successPatients";
        } else if ("coordinators".equals(tab)) {
            result = "successCoordinators";
        } else {
            result = "successClinicians";
        }
        return result;
    }
