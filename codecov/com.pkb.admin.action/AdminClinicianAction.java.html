    /**
     * Get the list of clinicians (Registered - Status[REG_CLINICIAN] and
     * invited - Status[CLINICIAN]).
     */
    public String browse() {
        if (offset < 0) {
            offset = 0;
        }
        pageSize = getConfig().getInboxPageSize() + 1;
        try {
            institute = loggedInUserUtil.getLoggedInUserPrimaryTeam();
            instituteId = institute.getId();

            // NOTE: InstituteUser.institute is lazy loaded so don't use it!
            clinicians = teamUserManager.getInstituteUsersByType(
                    instituteId, UserType.REG_CLINICIAN, Pager.ALL, offset);
            getSuccessMessage();
        } catch (PKBException e) {
            LOGGER.error("Error while getting the list of clinician", e);
        }
        return SUCCESS;
    }

    @SkipValidation
    public String previousClinicians() {

        LOGGER.info("previousClinicians-start");
        try {
            institute = loggedInUserUtil.getLoggedInUserPrimaryTeam();
            instituteId = institute.getId();

            //De Activated clinicians
            previousClinicians = teamUserManager.getInactiveTeamClinicians(getEHRRequestContext(), instituteId, config.getDeactivatedCliniciansPageSize(), offset);

            subTab = "deactivatedClinicians";

            getSuccessMessage();
        } catch (PKBException e) {
            LOGGER.info(e.getMessage());
            LOGGER.error("Error while getting the list of deactivated clinicians", e);
        }
        LOGGER.info("previousClinicians-exit");
        return SUCCESS;
    }

    public String inviteClinicianForm() {
        clinicianDTO = new PKBPersonDTO();
        if (userId != null) {
            try {
                PKBPerson clinician = userManager.getPKBPerson(userId, com.pkb.user.entity.PKBPerson.Lazy.CONTACTS);
                clinicianDTO.setTitle(clinician.getTitle());
                clinicianDTO.setFirstName(clinician.getFirstName());
                clinicianDTO.setLastName(clinician.getLastName());
                clinicianDTO.setEmailId(clinician.getEmail());
            } catch (NumberFormatException | PKBException e) {
                LOGGER.error("Error loading inviteClinicianForm {}", e.getMessage());
            }
        }
        return SUCCESS;
    }

    /**
     * Send an invitation to a clinician to register.
     */
    public String inviteClinician() {
        try {
            Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();
            String instituteCode = team.getCode();
            Option<Email> emailId = clinicianDTO.findEmailId();

            // Exit early if self-inviting
            if (emailId.isDefined() && userWebBean.isSameUser(emailId.get())) {
                addActionError(getText("error.cannot.invite.yourself", new String[]{emailId.get().address()}));
                return INPUT;
            }

            clinician = resolveEmailAddressToInvitee(team, clinicianDTO);
            if (adminInviteClinicianValidation(team, clinician)) {
                if ("resend"
                        .equals(userWebBean.adminInvitesClinician(getLoggedInEHRRequestContext(), instituteCode, clinicianDTO, clinician))) {
                    LOGGER.info("Created user {}", clinicianDTO.getId());
                    inviteeId = clinician.getIdString();
                    return "resent";
                }
            }
        } catch (PKBException e) {
            //noinspection StrutsI18nError,SSBasedInspection
            addActionError(e.getMessage());
            LOGGER.error("error inviting clinician", e);
            return Action.ERROR;
        } catch (MailException e) {
            //noinspection StrutsI18nError,SSBasedInspection
            addActionError(e.getMessage());
            LOGGER.error("Mailer exception: {}", e.getCause(), e);
            return Action.ERROR;
        }
        return SUCCESS;
    }

    private PKBPerson resolveEmailAddressToInvitee(Team team, PKBPersonDTO clinicianDTO) {
        LOGGER.info("resolving email address {} for invite to team {}", clinicianDTO.getEmailId(), team.getCode());

        List<PKBPerson> persons = userManager.getPKBPersonByConfirmedOrPrimaryEmail(clinicianDTO.getEmailId(), PersonContact.Lazy.PERSON_W_CONTACTS);
        if (persons.isEmpty()) {
            return null;
        }

        PKBPerson existingClinician = persons.get(0);
        if (StringUtils.isBlank(existingClinician.getHumanUUID())) {
            LOGGER.info("found {} unlinked persons for email {}", persons.size(), clinicianDTO.getEmailId());

            // loop through all the persons we found, looking for one in CREATED status
            // and linked to the current team
            for (PKBPerson candidateClinician : persons) {
                if (UserStatus.CREATED == candidateClinician.getStatus()) {
                    // Try to get an institute user for this team
                    InstituteUser iu = teamUserManager.getInstituteUser(team.getId(), candidateClinician.getId());
                    if (iu != null) {
                        LOGGER.info("selecting person {}", candidateClinician.toString());
                        return candidateClinician;
                    }
                }
            }
            return existingClinician;
        } else {
            // Multiple users linked by humanUUID - check all of them
            List<PKBPerson> personas = teamUserManager.getAllPersonasForHuman(getEHRRequestContext(),
                    existingClinician.getHumanUUID());

            // Gather email addresses...
            Set<Email> emailAddresses = Sets.newHashSet();
            for (PKBPerson persona : personas) {
                emailAddresses.add(persona.getEmail());
            }

            // ...then check if any of those email addresses are already on this team
            List<InstituteUser> teamClinicians = teamUserManager.getTeamCliniciansByEmailAddress(team.getId(),
                    emailAddresses);
            if ((teamClinicians != null) && !teamClinicians.isEmpty()) {
                // There should only ever be one
                if (teamClinicians.size() > 1) {
                    LOGGER.warn("multiple accounts for same humanUUID {} found on team {}", existingClinician.getHumanUUID(),
                            team.getCode());
                }

                InstituteUser instituteUser = teamClinicians.get(0);
                return instituteUser.getPerson();
            }
        }

        return null;
    }

    private boolean adminInviteClinicianValidation(Team team, PKBPerson clinicianToInvite) {

        if (clinicianToInvite == null) {
            return true;
        }

        if (!clinicianToInvite.isPro()) {
            addActionError(getText("registered.user.is.not.clinician"));
            return false;
        }

        // See if there's already an institute user linking the clinician to this team
        InstituteUser instituteUser = teamUserManager.getInstituteUserWithInstitute(team.getId(), clinicianToInvite.getId());
        if ((instituteUser != null) && instituteUser.getInstitute().getCode().equals(team.getCode())) {
            // If clinician is active on the team, check if they already have access
            if (instituteUser.getSponsorshipStatus() == SponsorshipStatus.ACTIVE) {
                addActionError(getText("user.already.active", new String[]{clinicianToInvite.getTitle(),
                        clinicianToInvite.getFirstName(), clinicianToInvite.getLastName(), clinicianDTO.findEmailId().map(Email::address).getOrNull()}));
                return false;
            }

            // If clinician is invited and their account is confirmed, don't send another
            boolean invited = instituteUser.getSponsorshipStatus() == SponsorshipStatus.INVITED;
            if (invited && clinicianToInvite.getStatus() == UserStatus.EMAIL_CONFIRMED) {
                addActionError(getText("user.activation.pending", new String[]{clinicianToInvite.getTitle(),
                        clinicianToInvite.getFirstName(), clinicianToInvite.getLastName(), clinicianDTO.findEmailId().map(Email::address).getOrNull()}));
                return false;
            }
        }

        return true;
    }

    /**
     * Get a proper success message in case of reminding an user and account
     * activation.
     */
    private void getSuccessMessage() {
        if (showActivationSuccessMessage) {
            PKBPerson person = userManager.getPKBPerson(Long.parseLong(inviteeId), PKBPerson.Lazy.CONTACTS);
            addActionMessage(getText("admin.successfully.activated.account",
                    new String[]{person.findEmail().map(Email::address).getOrNull(), person.getFirstName(),
                            person.getLastName(), person.getTitle()}));
        } else if (showResentSuccessMessage) {
            PKBPerson person = userManager.getPKBPerson(Long.parseLong(inviteeId), PKBPerson.Lazy.CONTACTS);
            addActionMessage(getText("successfully.resent.confirmation",
                    new String[]{person.getTitle(), person.getFirstName(), person.getLastName()}));
        } else if (showAutomaticResendSuccessMessage) {
            PKBPerson person = userManager.getPKBPerson(Long.parseLong(inviteeId), PKBPerson.Lazy.CONTACTS);
            addActionMessage(getText("automatically.resent.invitation",
                    new String[]{person.getTitle(), person.getFirstName(), person.getLastName()}));
        }
    }

    public String deactivationForm() {
        try {
            clinician = userManager.getPKBPerson(userId);
        } catch (PKBException e) {
            LOGGER.error("", e);
            addActionMessage(getText("deactivation.process.failure"));
            return INPUT;
        }
        return SUCCESS;
    }

    @SkipValidation
    public String deactivateClinician() {
        PKBPerson person;
        try {
            person = userManager.getPKBPerson(userId);
        } catch (Exception e1) {
            LOGGER.error("failed getPKBPerson for {}", userId, e1);
            addActionMessage(getText("deactivation.process.failure"));

            return INPUT;
        }
        try {
            institute = loggedInUserUtil.getLoggedInUserPrimaryTeam();
            instituteId = institute.getId();
            userAccessManager.deactivateUserAccount(getLoggedInEHRRequestContext(), userId);

            addActionMessage(getText("deactivation.successful",
                    new String[]{person.getTitle(), person.getFirstName(), person.getLastName()}));
            logActivity(com.pkb.notification.entity.Activity.Action.ACCOUNT_DEACTIVATED, dateTimeService.now(), person.getId(), true,
                    new NoConsentsRequired());
            return SUCCESS;
        } catch (Exception e) {
            LOGGER.error("failed deactivation for {}", userId, e);
            addActionMessage(getText("account.deactivation.failure",
                    new String[]{person.getTitle(), person.getFirstName(), person.getLastName()}));

            return SUCCESS;
        }
    }
