package com.pkb.admin.action;

import com.google.common.collect.Lists;
import com.google.common.collect.Maps;
import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.datamodel.consent.ImmutableTeamAdditionalConsent;
import com.pkb.datamodel.consent.TeamAdditionalConsent;
import com.pkb.dto.AdditionalConsentsDto;
import com.pkb.entities.enums.AdditionalConsentId;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.exception.PKBException;
import com.pkb.institute.entity.Team;
import com.pkb.service.team.TeamManager;
import com.pkb.service.team.TeamUserManager;
import com.pkb.util.Constants;
import org.apache.struts2.interceptor.ServletResponseAware;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.io.OutputStream;
import java.time.format.DateTimeFormatter;
import java.util.List;
import java.util.Locale;
import java.util.Set;
import java.util.stream.Collectors;

import static java.util.stream.Collectors.toSet;

/**
 *
 * @author Russ Gray <russ.gray@patientsknowbest.com>
 */
public class ViewAdditionalConsentOptionsAction extends BaseAction implements ServletResponseAware {

    private static final long serialVersionUID = 7748850756188802504L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final DateTimeFormatter DATE_FORMAT = DateTimeFormatter.ofPattern("YYYYMMdd_HHmmss").withLocale(Locale.UK).withZone(Constants.APPLICATION_TZ);

    private TeamUserManager teamUserManager;

    public TeamManager getTeamManager() {
        return teamManager;
    }

    public void setTeamManager(TeamManager teamManager) {
        this.teamManager = teamManager;
    }

    private TeamManager teamManager;

    private Team team;

    private AdditionalConsentsDto additionalConsentsDto = new AdditionalConsentsDto();

    //used by downloadContactDetails only
    private HttpServletResponse response;

    public String view() {
        try {
            team = loggedInUserUtil.getLoggedInUserPrimaryTeam();

            // Load complete list, and identify which consents have already been made available
            List<TeamAdditionalConsent> fullAdditionalConsentList = teamUserManager.getTeamAdditionalConsentDefaults(team.getId());
            List<TeamAdditionalConsent> teamAdditionalConsents = teamUserManager.getTeamAdditionalConsents(
                    team.getId());

            // Extract just the consent type
            Set<AdditionalConsentId> grantedConsents = teamAdditionalConsents.stream()
                    .map(TeamAdditionalConsent::getAdditionalConsent)
                    .collect(toSet());

            // Create current state based on full list, checking grantedConsents to see which are enabled
            additionalConsentsDto.setSelected(
                    Maps.immutableEnumMap(fullAdditionalConsentList.stream()
                            .collect(Collectors.toMap(TeamAdditionalConsent::getAdditionalConsent, consent ->
                                    grantedConsents.contains(consent.getAdditionalConsent())))));
            additionalConsentsDto.setTeamName(team.getName());
            additionalConsentsDto.setOrgName(team.getOrg().getName());
            additionalConsentsDto.setPartners(teamManager.getResearchPartners(team.getId()));
            additionalConsentsDto.setEditPartners(true);

            return Action.SUCCESS;

        } catch (RuntimeException ex) {
            LOGGER.error("could not load additional team consents", ex);
            return Action.ERROR;
        }
    }

    public String update() {
        try {
            team = loggedInUserUtil.getLoggedInUserPrimaryTeam();
            List<TeamAdditionalConsent> teamAdditionalConsents = Lists.newArrayList();

            if (additionalConsentsDto.getSelected() != null) {
                for (var selected : additionalConsentsDto.getSelectedIds()) {
                    teamAdditionalConsents.add(ImmutableTeamAdditionalConsent.builder().additionalConsent(selected).teamId(team.getId()).build());
                }
            }

            LOGGER.info("persisting {} records for team {}", teamAdditionalConsents.size(), team.getCode());
            teamUserManager.updateTeamAdditionalConsents(team.getId(), teamAdditionalConsents);
            teamManager.setResearchPartners(team.getId(), additionalConsentsDto.getPartners());

            return Action.SUCCESS;
        } catch (PKBException ex) {
            LOGGER.error("could not save additional team consents", ex);
            return Action.ERROR;
        }
    }

    public TeamUserManager getTeamUserManager() {
        return teamUserManager;
    }

    public void setTeamUserManager(TeamUserManager teamUserManager) {
        this.teamUserManager = teamUserManager;
    }

    public AdditionalConsentsDto getAdditionalConsentsDto() {
        return additionalConsentsDto;
    }

    public Team getTeam() {
        return team;
    }

    public void setTeam(Team team) {
        this.team = team;
    }

    public void setAdditionalConsentsDto(AdditionalConsentsDto additionalConsentsDto) {
        this.additionalConsentsDto = additionalConsentsDto;
    }

    @Override
    public void setServletResponse(HttpServletResponse response) {
        this.response = response;
    }

    public void downloadContactDetails() throws IOException {

        team = loggedInUserUtil.getLoggedInUserPrimaryTeam();

        OutputStream stream = response.getOutputStream();

        response.setHeader("Content-Disposition", "attachment; filename=\"contactDetails_" + dateTimeService.nowLocalDateTime().format(DATE_FORMAT) + ".csv\"");
        response.setContentType("text/csv;charset=utf-8");

        teamUserManager.exportPatientAdditionalConsentsForTeam(
                team.getId(),
                team.getOrg().getId(),
                NationalIdType.getPrimaryNationalIdTypeFromCountryCode(team.getCountry()),
                stream);

        stream.flush();
        stream.close();
    }
}
