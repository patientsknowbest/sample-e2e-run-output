package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.entities.enums.UserType;
import com.pkb.exception.PKBException;
import com.pkb.institute.entity.InstituteUser;
import com.pkb.institute.entity.InstituteUserEntity.ContactOptions;
import com.pkb.institute.entity.Team;
import com.pkb.service.team.TeamUserManager;
import com.pkb.util.Pager;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.ArrayList;
import java.util.List;

/**
 * This action is to list out all the clinicians associated with that particular institute.
 *
 * @author anamikac
 * @author pravinam Added action to deactivate institute clinician
 */
public class ClinicianContactOptionsAction extends BaseAction {

    private static final long serialVersionUID = 1L;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private List<InstituteUser> clinicians = new ArrayList<>(); // non-null

    private TeamUserManager teamUserManager;

    private long teamId;

    private int pageSize;

    private int offset;

    private final String tab = "clinicians";

    private List<Long> contactableCliniciansByMessages;
    private List<Long> contactableCliniciansByConsultations;

    private Team team;

    /**
     * Get the list of clinicians (Registered - Status[REG_CLINICIAN] and
     * invited - Status[CLINICIAN]).
     */
    public String viewClinicians() {
        if (offset < 0) {
            offset = 0;
        }
        pageSize = getConfig().getInboxPageSize() + 1;
        try {
            setTeam(loggedInUserUtil.getLoggedInUserPrimaryTeam());
            teamId = getTeam().getId();

            // NOTE: InstituteUser.institute is lazy loaded so don't use it!
            clinicians = teamUserManager.getInstituteUsersByType(
                    teamId, UserType.REG_CLINICIAN, Pager.ALL, offset);

        } catch (PKBException e) {
            LOGGER.error("Error while getting the list of clinicians", e);
        }
        return SUCCESS;
    }

    public String saveClinicianContactOptions() {

        // Struts gotcha: checkbox list will be null if none were checked.

        if (contactableCliniciansByMessages == null) {
            contactableCliniciansByMessages = new ArrayList<>();
        }
        if (contactableCliniciansByConsultations == null) {
            contactableCliniciansByConsultations = new ArrayList<>();
        }

        //populate clinicians:
        viewClinicians();

        for (InstituteUser clinician : clinicians) {
            long userId = clinician.getPerson().getId();

            if (clinician.getContactable() != ContactOptions.None &&
                    (!contactableCliniciansByMessages.contains(userId)
                            && !contactableCliniciansByConsultations.contains(userId))) {
                teamUserManager.updateContactOptions(getEHRRequestContext(), teamId, userId, ContactOptions.None);
            } else if (clinician.getContactable() != ContactOptions.All
                    && contactableCliniciansByMessages.contains(userId)
                    && contactableCliniciansByConsultations.contains(userId)) {
                teamUserManager.updateContactOptions(getEHRRequestContext(), teamId, userId, ContactOptions.All);
            } else if (clinician.getContactable() != ContactOptions.Messages_Only
                    && contactableCliniciansByMessages.contains(userId)
                    && !contactableCliniciansByConsultations.contains(userId)) {
                teamUserManager.updateContactOptions(getEHRRequestContext(), teamId, userId, ContactOptions.Messages_Only);
            } else if (clinician.getContactable() != ContactOptions.Consultations_Only
                    && !contactableCliniciansByMessages.contains(userId)
                    && contactableCliniciansByConsultations.contains(userId)) {
                teamUserManager.updateContactOptions(getEHRRequestContext(), teamId, userId, ContactOptions.Consultations_Only);
            }
        }

        return SUCCESS;
    }

    public String cancelClinicianContactOptions() {
        return "cancel";
    }

    public List<InstituteUser> getClinicians() {
        return clinicians;
    }

    public void setTeamUserManager(
            TeamUserManager teamUserManager) {
        this.teamUserManager = teamUserManager;
    }

    public int getPageSize() {
        return pageSize;
    }

    public void setPageSize(int pageSize) {
        this.pageSize = pageSize;
    }

    public int getOffset() {
        return offset;
    }

    public void setOffset(int offset) {
        this.offset = offset;
    }

    public String getTab() {
        return tab;
    }

    public List<Long> getContactableCliniciansByMessages() {
        return contactableCliniciansByMessages;
    }

    public void setContactableCliniciansByMessages(List<Long> contactableCliniciansByMessages) {
        this.contactableCliniciansByMessages = contactableCliniciansByMessages;
    }

    public List<Long> getContactableCliniciansByConsultations() {
        return contactableCliniciansByConsultations;
    }

    public void setContactableCliniciansByConsultations(List<Long> contactableCliniciansByConsultations) {
        this.contactableCliniciansByConsultations = contactableCliniciansByConsultations;
    }

    public Team getTeam() {
        return team;
    }

    public void setTeam(Team team) {
        this.team = team;
    }

}
