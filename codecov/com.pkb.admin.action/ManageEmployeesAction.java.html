package com.pkb.admin.action;

/**
 * Action class to list employees, add/invite employee etc
 *
 * @author pravinam
 */
public class ManageEmployeesAction extends BaseAction {

    private static final String RESEND = "resend";
    private static final String REGISTERED_ALREADY = "registeredAlready";

    /**
     *
     */
    private static final long serialVersionUID = 1L;

    private PKBPersonDTO employeeDTO;

    private PKBPerson existingEmployee;

    private String message;

    private IUserWebBean userWebBean;

    private List<Person> employees = new ArrayList<>();

    private InvitationManager invitationManager;

    private Long employeeId;

    private PersonService personService;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    // Show list of employees
    public String browse() {
        try {
            employees = personService.findEmployeesFetchContacts();
        } catch (Exception e) {
            addActionError(getText("error.loading.employee.list"));
            LOGGER.error("error", e);
        }
        return Action.SUCCESS;
    }

    @Override
    public String input() {
        return Action.SUCCESS;
    }

    /**
     * Send an invitation to an employee to register.
     */
    public String inviteEmployee() {
        try {
            if (employeeDTO != null) {
                Email emailId = employeeDTO.getEmailId();
                if (userWebBean.isSameUser(emailId)) {
                    addActionError(getText("error.cannot.invite.yourself",
                            new String[]{emailId.address()}));
                    return INPUT;
                }
            }
            String msg = "";
            String result = inviteEmployee(employeeDTO);
            if (RESEND.equals(result)) {
                msg = getText("invitation.email.resent.success", new String[]{
                        existingEmployee.getTitle(), existingEmployee.getFirstName(),
                        existingEmployee.getLastName()});

            } else if (result.equals(Action.SUCCESS)) {
                msg = getText("invitation.email.sent.success", new String[]{
                        employeeDTO.getTitle(), employeeDTO.getFirstName(),
                        employeeDTO.getLastName()});
            } else {
                msg = getText("invited.user.already.registered", new String[]{
                        existingEmployee.getTitle(), existingEmployee.getFirstName(),
                        existingEmployee.getLastName()});
            }
            //noinspection StrutsI18nError,SSBasedInspection
            addActionMessage(msg);

        } catch (PKBException e) {
            addActionError(getText("error.inviting.employee"));
            LOGGER.error("error", e);
            return Action.INPUT;
        }
        return SUCCESS;
    }

    private String inviteEmployee(PKBPersonDTO employeeDTO) {
        // Any validations ?
        long inviterId = loggedInUserUtil.getLoggedInUserId();
        UserType userType = loggedInUserUtil.getLoggedInUserType();
        Long inviterAccountId = userManager.getDefaultAccountId(inviterId);
        UUID invitationId = null;
        if (employeeId != null) {
            existingEmployee = userManager.getPKBPerson(employeeId, PKBPerson.Lazy.CONTACTS);
            Email email = existingEmployee.getPrimaryContact().getContactIfEmail().getOrNull();
            invitationId = invitationManager.getOpenInvitationId(email, inviterId).orElse(null);
        } else {
            invitationId = invitationManager.getOpenInvitationId(employeeDTO.getEmailId(), inviterId).orElse(null);
        }
        LoggedInEHRRequestContext requestContext = getLoggedInEHRRequestContext();
        if (invitationId == null) {
            PKBInvitation invitation = userWebBean.populateInvitation(message,
                    inviterAccountId, inviterId, userType, null, requestContext);
            invitation.setInvitationType(InvitationType.EMPLOYEE_INVITATION);
            employeeDTO.setTimeZoneId(PKBConstants.DEFAULT_TIMEZONE_ID);
            invitationManager.inviteEmployee(requestContext, employeeDTO, invitation, inviterId);
            return Action.SUCCESS;
        } else {
            // If there is already an open invitation, resend it
            if (existingEmployee == null) {
                existingEmployee = userManager.getSinglePersonByEmail(employeeDTO.getEmailId());
            }
            // If by mistake an already registered user is invited again
            if (existingEmployee.getStatus() == UserStatus.EMAIL_CONFIRMED) {
                return REGISTERED_ALREADY;
            }
            invitationManager.resendInvitationToEmployee(requestContext, invitationId);
            return RESEND;
        }
    }

    public PKBPersonDTO getEmployeeDTO() {
        return employeeDTO;
    }

    public void setEmployeeDTO(PKBPersonDTO employeeDTO) {
        this.employeeDTO = employeeDTO;
    }

    public String getMessage() {
        return message;
    }

    public void setMessage(String message) {
        this.message = message;
    }

    public IUserWebBean getUserWebBean() {
        return userWebBean;
    }

    public void setUserWebBean(IUserWebBean userWebBean) {
        this.userWebBean = userWebBean;
    }

    public List<Person> getEmployees() {
        return employees;
    }

    public void setEmployees(List<Person> employees) {
        this.employees = employees;
    }

    public InvitationManager getInvitationManager() {
        return invitationManager;
    }

    public void setInvitationManager(InvitationManager invitationManager) {
        this.invitationManager = invitationManager;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public Long getEmployeeId() {
        return employeeId;
    }

    public void setEmployeeId(Long employeeId) {
        this.employeeId = employeeId;
    }

    public void setPersonService(PersonService personService) {
        this.personService = personService;
    }
}
