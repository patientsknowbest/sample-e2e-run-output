package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.service.coding.CodingMatchExportService;
import com.pkb.service.dataupload.emis.esv2.EmisJobArtifactService;

import javax.annotation.Nullable;

import static java.util.Collections.singletonList;

/**
 * Action for exporting coding matches, either existing mappings or to aid with the product-curated mapping process
 */
class CodingMatchExportAdminAction extends BaseAction {

    private CodingMatchExportService codingMatchExportService;
    private EmisJobArtifactService emisJobArtifactService;

    @Override
    public String execute() throws Exception {
        return super.execute();
    }

    @Nullable
    public String downloadExistingCodingMatches() {
        codingMatchExportService.streamAllMatches(getResponse())
                .peekLeft(throwable -> {
                    LOGGER.error("Failed to stream all matches to csv", throwable);
                    addActionError(getText("emisAdmin.txt.coding_match_export_failure", singletonList(throwable.getMessage())));
                });
        return null;
    }

    @Nullable
    public String downloadLatestUnmatchedCodes() {
        codingMatchExportService.streamUnmatchedCodes(getResponse())
                .peekLeft(throwable -> {
                    addActionError(getText("emisAdmin.txt.coding_match_export_failure", singletonList(throwable.getMessage())));
                    LOGGER.error("Failed to stream unmatched codes to csv", throwable);
                });
        return null;
    }

    public CodingMatchExportService getCodingMatchExportService() {
        return codingMatchExportService;
    }

    public void setCodingMatchExportService(CodingMatchExportService codingMatchExportService) {
        this.codingMatchExportService = codingMatchExportService;
    }

    public EmisJobArtifactService getEmisJobArtifactService() {
        return emisJobArtifactService;
    }

    public void setEmisJobArtifactService(EmisJobArtifactService emisJobArtifactService) {
        this.emisJobArtifactService = emisJobArtifactService;
    }
}
