package com.pkb.admin.action;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.datamodel.Email;
import com.pkb.exception.IllegalReuseOfEmailAddressException;
import com.pkb.institute.entity.Org;
import com.pkb.service.team.OrgManager;
import com.pkb.service.team.TeamManager;
import com.pkb.user.entity.PKBPerson;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import static org.apache.commons.lang3.StringUtils.isBlank;

public class SuperAdminPrivacyOfficerAction extends BaseAction {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final long serialVersionUID = -2099439841716711861L;

    private Org org;
    private PKBPerson privacyOfficer;
    private Email privacyOfficerEmail;

    // Dependencies
    private OrgManager orgManager;
    private TeamManager teamManager;

    @Override
    public String execute() {
        try {
            orgManager.assignPrivacyOfficerToOrg(
                    getLoggedInEHRRequestContext(),
                    teamManager.getOrg(org.getId(), Org.Lazy.TEAMS).orElse(null),
                    loggedInUserUtil.getLoggedInUser(),
                    privacyOfficer,
                    privacyOfficerEmail);

        } catch (IllegalReuseOfEmailAddressException e) {
            LOGGER.error("Email in use", e);
            addActionError(getText("assignPrivacyOfficerForm.err.emailInUse", new String[]{privacyOfficerEmail.address()}));
            return Action.INPUT;
        } catch (Exception e) {
            addActionError(getText("err.privacy.officer.invite", new String[]{e.getMessage()}));
            LOGGER.error("Exception while assigning privacy officer", e);
            return Action.INPUT;
        }

        return Action.SUCCESS;
    }

    @Override
    public void validate() {
        if (privacyOfficerEmail == null) {
            addFieldError("privacyOfficerEmail", getText("global.err.field.email_missing"));
        }

        if (isBlank(privacyOfficer.getTitle())) {
            addFieldError("privacyOfficer.title", getText("global.err.field.title_missing"));
        }

        if (isBlank(privacyOfficer.getFirstName())) {
            addFieldError("privacyOfficer.firstName", getText("global.err.field.first_name_mising"));
        }

        if (isBlank(privacyOfficer.getLastName())) {
            addFieldError("privacyOfficer.lastName", getText("global.err.field.last_name_mising"));
        }
    }

    public PKBPerson getPrivacyOfficer() {
        return privacyOfficer;
    }

    public void setPrivacyOfficer(PKBPerson privacyOfficer) {
        this.privacyOfficer = privacyOfficer;
    }

    public Email getPrivacyOfficerEmail() {
        return privacyOfficerEmail;
    }

    public void setPrivacyOfficerEmail(Email privacyOfficerEmail) {
        this.privacyOfficerEmail = privacyOfficerEmail;
    }

    public Org getOrg() {
        return org;
    }

    public void setOrg(Org org) {
        this.org = org;
    }

    public void setOrgManager(OrgManager orgManager) {
        this.orgManager = orgManager;
    }

    public void setTeamManager(TeamManager teamManager) {
        this.teamManager = teamManager;
    }
}
