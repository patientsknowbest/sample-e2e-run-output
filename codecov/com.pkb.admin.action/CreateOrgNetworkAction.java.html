package com.pkb.admin.action;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.datamodel.ImmutableOrgNetworkNew;
import com.pkb.datamodel.ModifiableOrgNetworkNew;
import com.pkb.datamodel.Org;
import com.pkb.datamodel.OrgNetworkExisting;
import com.pkb.domain.OrgNetworkService;
import com.pkb.exception.MailException;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.naming.NamingException;
import java.util.List;
import java.util.Optional;

/**
 *
 * @author Russ Gray
 */
public class CreateOrgNetworkAction extends BaseAction {

    private static final long serialVersionUID = 5138646250072835176L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    // Entities
    private List<Org> orgsNotInAnyNetwork;
    private ModifiableOrgNetworkNew orgNetwork;

    // Dependencies
    private OrgNetworkService orgNetworkService;
    private PKBEmailMessageManager pkbEmailMessageManager;

    public String createOrgNetworkForm() throws NamingException {

        // This is the new entity to be populated by the jsp
        orgNetwork = ModifiableOrgNetworkNew.create();

        // Network leader may not be in any other orgs
        orgsNotInAnyNetwork = orgNetworkService.getOrgsNotInAnyNetwork();

        return Action.INPUT;
    }

    public String createOrgNetwork() throws NamingException {

        try {
            OrgNetworkExisting newOrgNetwork = orgNetworkService.createOrgNetwork(ImmutableOrgNetworkNew.builder().from(orgNetwork).createdByPersonId(loggedInUserUtil.getLoggedInUserId()).build());
            LOGGER.info("Created new org network {}", newOrgNetwork);
            Optional<Long> optionalActorId = getEHRRequestContext().getMaybeAccessingUserId();
            String actorId = optionalActorId.map(Object::toString).orElse(null);

            // Send notification
            List<PKBPerson> orgCoords = userManager.findOrgCoords(newOrgNetwork.getLeadingOrg().getId());
            for (PKBPerson orgCoord : orgCoords) {
                try {
                    pkbEmailMessageManager.notifyOrgCoordOfNewOrgNetwork(orgCoord, newOrgNetwork, actorId);
                } catch (MailException e) {
                    LOGGER.error("failed to send email notification of network creation to {}", orgCoord, e);
                }
            }
        } catch (Exception e) {
            LOGGER.error("failed to create org network", e);
            addActionError(getText("createOrgNetwork.txt.error"));
        }

        return Action.SUCCESS;
    }

    public ModifiableOrgNetworkNew getOrgNetwork() {
        return orgNetwork;
    }

    public void setOrgNetwork(ModifiableOrgNetworkNew orgNetwork) {
        this.orgNetwork = orgNetwork;
    }

    public List<Org> getOrgsNotInAnyNetwork() {
        return orgsNotInAnyNetwork;
    }

    public void setOrgNetworkService(OrgNetworkService orgNetworkService) {
        this.orgNetworkService = orgNetworkService;
    }

    public void setPkbEmailMessageManager(PKBEmailMessageManager pkbEmailMessageManager) {
        this.pkbEmailMessageManager = pkbEmailMessageManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }
}
