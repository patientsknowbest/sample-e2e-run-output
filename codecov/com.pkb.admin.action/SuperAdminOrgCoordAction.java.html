package com.pkb.admin.action;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.bean.IUserWebBean;
import com.pkb.datamodel.Email;
import com.pkb.institute.entity.Org;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.team.TeamManager;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.List;

import static org.apache.commons.lang3.StringUtils.isBlank;

public class SuperAdminOrgCoordAction extends BaseAction {

    private static final long serialVersionUID = 1L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private Org org;
    private PKBPerson orgCoord;

    private Long orgId;
    private Email orgCoordEmail;

    private TeamManager teamManager;
    private IUserWebBean userWebBean;
    private UserAccessManager userAccessManager;
    private PKBEmailMessageManager pkbEmailMessageManager;

    @Override
    public String execute() {
        try {
            List<PKBPerson> persons = userManager.getPKBPersonByConfirmedOrPrimaryEmail(orgCoordEmail);
            if (!persons.isEmpty()) {
                addFieldError("orgCoordEmail", getText("global.err.user_already_exists", new String[]{orgCoordEmail.address()}));
                return Action.INPUT;
            }

            teamManager.beginTransaction();

            org = teamManager.getOrg(org.getId(), Org.Lazy.TEAMS).orElse(null);

            PKBPerson pkbAdmin = userManager.getPKBPerson(loggedInUserUtil.getLoggedInUserId());

            PKBPerson newCoordinator = userWebBean.setupOrgCoordinator(getLoggedInEHRRequestContext(),
                    orgCoord.getTitle(),
                    orgCoord.getFirstName(),
                    orgCoord.getLastName(),
                    orgCoordEmail,
                    pkbAdmin,
                    org);

            pkbEmailMessageManager.inviteInitialOrgCoord(org, newCoordinator, pkbAdmin);
            teamManager.commitTransaction();
        } catch (Exception e) {
            addActionError(getText("err.org.coord.invite", new String[]{e.getMessage()}));
            LOGGER.error("Exception while assigning org coord", e);
            return Action.INPUT;
        } finally {
            teamManager.rollbackIfNotCommitted();
        }
        return Action.SUCCESS;
    }

    @Override
    public void validate() {
        if (orgCoordEmail == null) {
            addFieldError("orgCoordEmail", getText("global.err.field.email_missing"));
        }
        if (isBlank(orgCoord.getTitle())) {
            addFieldError("orgCoord.title", getText("global.err.field.title_missing"));
        }
        if (isBlank(orgCoord.getFirstName())) {
            addFieldError("orgCoord.firstName", getText("global.err.field.first_name_mising"));
        }
        if (isBlank(orgCoord.getLastName())) {
            addFieldError("orgCoord.lastName", getText("global.err.field.last_name_mising"));
        }
    }

    public TeamManager getTeamManager() {
        return teamManager;
    }

    public void setTeamManager(TeamManager teamManager) {
        this.teamManager = teamManager;
    }

    public Email getOrgCoordEmail() {
        return orgCoordEmail;
    }

    public void setOrgCoordEmail(Email orgCoordEmail) {
        this.orgCoordEmail = orgCoordEmail;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public IUserWebBean getUserWebBean() {
        return userWebBean;
    }

    public void setUserWebBean(IUserWebBean userWebBean) {
        this.userWebBean = userWebBean;
    }

    public PKBEmailMessageManager getPkbEmailMessageManager() {
        return pkbEmailMessageManager;
    }

    public void setPkbEmailMessageManager(
            PKBEmailMessageManager pkbEmailMessageManager) {
        this.pkbEmailMessageManager = pkbEmailMessageManager;
    }

    public Org getOrg() {
        return org;
    }

    public void setOrg(Org org) {
        this.org = org;
    }

    public Long getOrgId() {
        return orgId;
    }

    public void setOrgId(Long orgId) {
        this.orgId = orgId;
    }

    public PKBPerson getOrgCoord() {
        return orgCoord;
    }

    public void setOrgCoord(PKBPerson orgCoord) {
        this.orgCoord = orgCoord;
    }

    public UserAccessManager getUserAccessManager() {
        return userAccessManager;
    }

    public void setUserAccessManager(UserAccessManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }
}
