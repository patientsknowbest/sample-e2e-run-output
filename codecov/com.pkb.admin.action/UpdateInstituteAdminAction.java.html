package com.pkb.admin.action;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.exception.PKBException;
import com.pkb.institute.entity.InstituteUser;
import com.pkb.institute.entity.Team;
import com.pkb.service.team.TeamManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.List;

public class UpdateInstituteAdminAction extends BaseAction {

    private static final long serialVersionUID = 1L;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private TeamManager teamManager;

    private InstituteUser instituteUser;

    private List<PKBPerson> instituteAdmins;

    private final String tab = "customers";

    private final String subTab = "customerList";

    /**
     * Update institute admin and get the list of adminis to show at the
     * resultant page
     *
     */
    @Override
    public String execute() {
        try {
            updateTeamCoord();
            updateAdminCRMLinkOfInstitute();
            instituteAdmins = teamManager
                    .getTeamCoord(instituteUser.getInstitute().getId());
        } catch (PKBException e) {
            LOGGER.error("error", e);
        }
        return Action.SUCCESS;
    }

    @Override
    public void validate() {
        if (hasFieldErrors()) {
            String userId = instituteUser.getPerson().getIdString();
            String instituteCode = instituteUser.getInstitute().getCode();
            if ((userId != null) && !userId.isEmpty() && (instituteCode != null)
                    && !instituteCode.isEmpty()) {
                try {
                    loadInstituteUser(userId, instituteCode);
                } catch (PKBException e) {
                    LOGGER.error("Error while loading admin details", e);
                }
            }
        }
    }

    /**
     * Update institute details as adminCRMLink is an attribute of institute
     * object.
     *
     */
    private void updateAdminCRMLinkOfInstitute() {
        teamManager.transactional(() -> {
            Team team = teamManager.getTeamByCode(instituteUser
                    .getInstitute().getCode());
            team.setAdminCRMLink(instituteUser.getInstitute()
                    .getAdminCRMLink());
            teamManager.updateTeam(team);
            team = teamManager.getTeamByCode(instituteUser
                    .getInstitute().getCode());
            team.setLogoFileName(getConfig().getInstituteLogoBaseDir()
                    + team.getLogoFileName());
            instituteUser.setInstitute(team);
        });
    }

    /**
     * Update institute administrator details.
     *
     */
    private void updateTeamCoord() {
        PKBPerson user = instituteUser.getPerson();
        PKBPerson person = userManager.getPKBPerson(user.getId(), PKBPerson.Lazy.CONTACTS);
        person.setFirstName(user.getFirstName());
        person.setLastName(user.getLastName());
        userManager.updateUser(getEHRRequestContext(), person);
    }

    /**
     * Load update institute user form.
     *
     */
    @Override
    public String input() {
        String userId = request.getParameter("userId");
        String instituteCode = request.getParameter("instituteCode");
        if ((userId != null) && !userId.isEmpty() && (instituteCode != null)
                && !instituteCode.isEmpty()) {
            try {
                loadInstituteUser(userId, instituteCode);
            } catch (PKBException e) {
                LOGGER.error("Error while loading admin details", e);
            }
        }
        return Action.SUCCESS;
    }

    /**
     * Get institute administrator from database. Set the logo image location.
     *
     */
    private void loadInstituteUser(String userId, String instituteCode) {
        instituteUser = new InstituteUser();
        PKBPerson person = userManager.getPKBPerson(Long.parseLong(userId), PKBPerson.Lazy.CONTACTS);
        instituteUser.setPerson(person);
        instituteUser.setUserType(person.getUserType());

        Team team = teamManager.getTeamByCode(instituteCode);
        team.setLogoFileName(getConfig().getInstituteLogoBaseDir()
                + team.getLogoFileName());
        instituteUser.setInstitute(team);
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public TeamManager getTeamManager() {
        return teamManager;
    }

    public void setTeamManager(TeamManager teamManager) {
        this.teamManager = teamManager;
    }

    public InstituteUser getInstituteUser() {
        return instituteUser;
    }

    public void setInstituteUser(InstituteUser instituteUser) {
        this.instituteUser = instituteUser;
    }

    public List<PKBPerson> getInstituteAdmins() {
        return instituteAdmins;
    }

    public void setInstituteAdmins(List<PKBPerson> instituteAdmins) {
        this.instituteAdmins = instituteAdmins;
    }

    public String getTab() {
        return tab;
    }

    public String getSubTab() {
        return subTab;
    }
}
