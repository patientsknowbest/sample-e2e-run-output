package com.pkb.admin.action;

import com.opensymphony.xwork2.ActionContext;
import com.pkb.action.BaseAction;
import com.pkb.entities.enums.SponsorshipStatus;
import com.pkb.entities.enums.UserStatus;
import com.pkb.exception.PKBException;
import com.pkb.institute.entity.InstituteUser;
import com.pkb.institute.entity.Team;
import com.pkb.plan.facade.PlanExportViewFacade;
import com.pkb.service.phplan.export.ImmutableExportPlanRequestData;
import com.pkb.service.team.TeamUserManager;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.List;

/**
 * This action class triggers a csv export of all the plans created by a selected template to chosen clinicians
 *
 * @author pravina
 */
public class PlanExportAction extends BaseAction {

    private static final String NONE = "None";


    private PlanExportViewFacade planExportViewFacade;

    private Long templateId;

    private List<String> selectedClinicians;

    private String exportType = "Latest";

    private List<InstituteUser> clinicians;

    private TeamUserManager teamUserManager;
    /**
     *
     */
    private static final long serialVersionUID = 1L;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Override
    public String execute() {
        try {
            Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();
            planExportViewFacade.exportPlans(
                    ImmutableExportPlanRequestData.builder()
                            .actionContext(ActionContext.getContext())
                            .requestContext(getLoggedInEHRRequestContext())
                            .loggedInUser(loggedInUserUtil.getLoggedInPerson())
                            .templateId(templateId)
                            .teamId(team.getId())
                            .locale(getLocale())
                            .exportAll("All".equals(exportType))
                            .addAllSelectedClinicians(selectedClinicians).build());
            addActionMessage(getText("plan.export.initiated"));
        } catch (Exception e) {
            LOGGER.error("", e);
            addActionError(getText("err.plan.export.for.clinician"));
            LOGGER.error("Exception while exporting plan for template-{}", templateId, e);
            return INPUT;
        }
        return SUCCESS;
    }

    @Override
    public String input() {
        Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();
        try {
            clinicians = teamUserManager.getTeamCliniciansByStatus(team.getId(), UserStatus.EMAIL_CONFIRMED,
                    SponsorshipStatus.ACTIVE, -1, 0);
        } catch (PKBException e) {
            LOGGER.error("Exception while exporting plan for template-{}", templateId, e);
            return INPUT;
        }
        return SUCCESS;
    }

    @Override
    public void validate() {
        if ((selectedClinicians == null) || selectedClinicians.isEmpty()) {
            addActionError(getText("err.no.clinicians.for.plan.export"));
        } else {
            if (selectedClinicians.contains(NONE)) {
                addActionError(getText("err.no.clinicians.for.plan.export"));
            }
        }
    }

    public String cancel() {
        return SUCCESS;
    }

    public Long getTemplateId() {
        return templateId;
    }

    public void setTemplateId(Long templateId) {
        this.templateId = templateId;
    }

    public List<String> getSelectedClinicians() {
        return selectedClinicians;
    }

    public void setSelectedClinicians(List<String> selectedClinicians) {
        this.selectedClinicians = selectedClinicians;
    }

    public List<InstituteUser> getClinicians() {
        return clinicians;
    }

    public void setClinicians(List<InstituteUser> clinicians) {
        this.clinicians = clinicians;
    }

    public void setTeamUserManager(TeamUserManager teamUserManager) {
        this.teamUserManager = teamUserManager;
    }

    public String getExportType() {
        return exportType;
    }

    public void setExportType(String exportType) {
        this.exportType = exportType;
    }

    @Autowired
    public void setPlanExportViewFacade(PlanExportViewFacade planExportViewFacade) {
        this.planExportViewFacade = planExportViewFacade;
    }
}
