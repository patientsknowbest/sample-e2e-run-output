package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.institute.entity.Org;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.team.TeamManager;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.user.entity.PasswordAccess;

/**
 * @author Russ Gray
 */
public class RemindPrivacyOfficerToRegisterAction extends BaseAction {

    private static final long serialVersionUID = 134863512301284613L;

    private Long orgId;
    private Long privacyOfficerId;

    // Dependencies
    private TeamManager teamManager;
    private PKBEmailMessageManager pkbEmailMessageManager;
    private UserAccessManager userAccessManager;

    @Override
    public String execute() {
        PKBPerson privacyOfficer = userManager.getPKBPerson(getPrivacyOfficerId());
        Org org = teamManager.getOrg(getOrgId()).orElse(null);

        PasswordAccess passwordAccess = userAccessManager.getPasswordAccess(
                privacyOfficer.getId(),
                getLoggedInEHRRequestContext());

        privacyOfficer.setPassword(passwordAccess.getPassword());
        pkbEmailMessageManager.invitePrivacyOfficer(org, privacyOfficer, loggedInUserUtil.getLoggedInUser());

        addActionMessage(getText("registration.remind.user.sent"));
        return "privacyOfficerReminded";
    }

    @Override
    public void validate() {
        if (getOrgId() == null) {
            addActionError(getText("err.field.no.org"));
        }

        if (getPrivacyOfficerId() == null) {
            addActionError(getText("err.field.no.privacy.officer"));
        }
    }

    public Long getOrgId() {
        return orgId;
    }

    public void setOrgId(Long orgId) {
        this.orgId = orgId;
    }

    public Long getPrivacyOfficerId() {
        return privacyOfficerId;
    }

    public void setPrivacyOfficerId(Long privacyOfficerId) {
        this.privacyOfficerId = privacyOfficerId;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public TeamManager getTeamManager() {
        return teamManager;
    }

    public void setTeamManager(TeamManager teamManager) {
        this.teamManager = teamManager;
    }

    public PKBEmailMessageManager getPkbEmailMessageManager() {
        return pkbEmailMessageManager;
    }

    public void setPkbEmailMessageManager(PKBEmailMessageManager pkbEmailMessageManager) {
        this.pkbEmailMessageManager = pkbEmailMessageManager;
    }

    public UserAccessManager getUserAccessManager() {
        return userAccessManager;
    }

    public void setUserAccessManager(UserAccessManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }

}
