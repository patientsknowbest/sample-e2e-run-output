package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.institute.entity.Org;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.team.TeamManager;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.user.entity.PasswordAccess;
import org.apache.struts2.interceptor.validation.SkipValidation;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * @author Russ Gray
 */
public class RemindOrgCoordToRegisterAction extends BaseAction {

    private static final long serialVersionUID = -3959639596348321216L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private Long orgId;
    private Long orgCoordId;

    // Dependencies
    private TeamManager teamManager;
    private PKBEmailMessageManager pkbEmailMessageManager;
    private UserAccessManager userAccessManager;

    @Override
    @SkipValidation
    public String execute() {
        if (getOrgId() != null && getOrgCoordId() != null) {

            PKBPerson orgCoord = userManager.getPKBPerson(getOrgCoordId());
            Org org = teamManager.getOrg(getOrgId()).orElse(null);

            PasswordAccess passwordAccess = userAccessManager.getPasswordAccess(orgCoord.getId(), getLoggedInEHRRequestContext());
            orgCoord.setPassword(passwordAccess.getPassword());
            pkbEmailMessageManager.inviteInitialOrgCoord(org, orgCoord, loggedInUserUtil.getLoggedInUser());
            addActionMessage(getText("registration.remind.user.sent"));
        } else {
            LOGGER.error("Unable to send reminder due to missing info in request: orgId {}, orgCoordId {}", getOrgId(), getOrgCoordId());
            addActionError(getText("err.remind.failure"));
        }
        return "orgCoordReminded";
    }

    public Long getOrgId() {
        return orgId;
    }

    public void setOrgId(Long orgId) {
        this.orgId = orgId;
    }

    public Long getOrgCoordId() {
        return orgCoordId;
    }

    public void setOrgCoordId(Long orgCoordId) {
        this.orgCoordId = orgCoordId;
    }

    public void setPkbEmailMessageManager(PKBEmailMessageManager pkbEmailMessageManager) {
        this.pkbEmailMessageManager = pkbEmailMessageManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public void setTeamManager(TeamManager teamManager) {
        this.teamManager = teamManager;
    }

    public void setUserAccessManager(UserAccessManager userAccessManager) {
        this.userAccessManager = userAccessManager;
    }

}
