    @Override
    public String input() {
        if (getLoggedInEHRRequestContext().isTeamCoordinator()) {
            instituteCode = loggedInUserUtil.getLoggedInUserPrimaryTeam(getEHRRequestContext()).getCode();
        }

        if ((instituteCode == null) || instituteCode.isEmpty()) {
            instituteCode = loggedInUserUtil.getLoggedInUserPrimaryTeam().getCode();
        }
        generateSymptomList();

        return Action.INPUT;
    }

    @Override
    public void validate() {

        super.validate();

        //this is a hack to get round the inflexibility of OGNL conversion validation.
        //if there is a field error, regenerate the input so that when the framework automatically returns "input" it will
        //actually display the error!
        boolean errors = false;
        Map<String, List<String>> fieldErrors = getFieldErrors();
        errors = validateInput(fieldErrors, symptomOrder, "symptomOrder", errors);
        errors = validateInput(fieldErrors, symptomWeight, "symptomWeight", errors);
        if (!fieldErrors.isEmpty()) {
            errors = true;
        }
        if (errors) {
            try {
                input();
                editing = true;

            } catch (Exception e) {
                LOGGER.error("error regenerating input after failed validation", e);
            }

        }
    }

    private boolean validateInput(Map<String, List<String>> fieldErrors, List<Long> inputFields, String prefix, boolean errors) {
        if (inputFields != null) {
            for (int i = 0; i < inputFields.size(); i++) {
                String fieldName = prefix + "[" + i + "]";
                try {
                    Long input = inputFields.get(i);
                    if (prefix.equals("symptomOrder")) {
                        if ((input != null) && (input < 0)) {
                            addFieldError(fieldName, getText("err.validation.not.positive.number"));
                            errors = true;
                        }
                    }
                } catch (Exception ignored) {
                    fieldErrors.remove(fieldName);
                    setFieldErrors(fieldErrors);
                    addFieldError(fieldName, getText("err.validation.not.a.number"));
                    errors = true;
                }
            }
        }
        return errors;
    }

    public String save() throws MailException {

        instituteCode = loggedInUserUtil.getLoggedInUserPrimaryTeam(getEHRRequestContext()).getCode();


        LOGGER.info("Saving symptom alarms for team-{}", instituteCode);

        Long redThreshold = null;
        if (StringUtils.isBlank(symptomAlarmRedThreshold) && symptomRowId != null) {
            addFieldError("symptomAlarmRedThreshold", getText("err.validation.no.threshold"));
            editing = true;
            return RELOAD;
        } else {
            try {
                redThreshold = Long.parseLong(symptomAlarmRedThreshold);
            } catch (Exception ignored) {
                addFieldError("symptomAlarmRedThreshold", getText("err.validation.threshold.not.positive"));
                editing = true;
                return RELOAD;
            }
        }

        Long amberThreshold = null;
        if (StringUtils.isNotBlank(symptomAlarmAmberThreshold) && symptomRowId != null) {
            try {
                amberThreshold = Long.parseLong(symptomAlarmAmberThreshold);
                if (amberThreshold >= redThreshold) {
                    addFieldError("symptomAlarmAmberThreshold", getText("err.validation.threshold.amber"));
                    editing = true;
                    return RELOAD;
                }
            } catch (Exception ignored) {
                addFieldError("symptomAlarmAmberThreshold", getText("err.validation.threshold.not.positive"));
                editing = true;
                return RELOAD;
            }
        }

        Long days = null;
        if (StringUtils.isBlank(symptomAlarmDays) && symptomRowId != null) {
            addFieldError("symptomAlarmDays", getText("err.validation.number.of.days.missing"));
            editing = true;
            return RELOAD;
        } else {
            try {
                days = Long.parseLong(symptomAlarmDays);
            } catch (Exception ignored) {
                addFieldError("symptomAlarmDays", getText("err.validation.number.of.days.not.positive"));
                editing = true;
                return RELOAD;
            }
        }

        Set<TeamSymptom> symptoms = new HashSet<>();

        symptomIdsToMonitor = new ArrayList<>();
        Map<Long, Long> weightMap = new HashMap<>();
        Map<Long, Long> orderMap = new HashMap<>();
        if (symptomRowId != null) {
            for (Long rowId : symptomRowId) {
                int r = rowId.intValue();
                symptomIdsToMonitor.add(symptomTypeId.get(r));
                weightMap.put(symptomTypeId.get(r), symptomWeight.get(r));
                orderMap.put(symptomTypeId.get(r), symptomOrder.get(r));
            }

            for (Symptom symptom : Symptom.values()) {
                if (symptomIdsToMonitor.contains(symptom.getId())) {
                    TeamSymptom teamSymptom = new TeamSymptom();
                    teamSymptom.setSymptom(symptom);
                    teamSymptom.setAlarmWeighting(weightMap.get(symptom.getId()));
                    teamSymptom.setOrdering(orderMap.get(symptom.getId()));
                    symptoms.add(teamSymptom);
                }
            }
        }


        symptomReportManager.saveSymptomsForInstitute(getLoggedInEHRRequestContext(), instituteCode, symptoms, redThreshold, amberThreshold, days);


        editing = false;
        input();

        return SUCCESS;
    }

    public String stopTracking() throws MailException {
        symptomRowId = null;
        symptomWeight = null;
        symptomOrder = null;
        symptomAlarmRedThreshold = null;
        symptomAlarmAmberThreshold = null;
        symptomAlarmDays = "";

        Set<TeamSymptom> symptoms = new HashSet<>();

        symptomReportManager.saveSymptomsForInstitute(getLoggedInEHRRequestContext(), instituteCode, symptoms,
                null/*redThreshold*/, null/*amberThreshold*/, 0L/*days*/);

        return input();
    }


    private void generateSymptomList() {

        symptomTypeList = new ArrayList<>();
        Collections.addAll(symptomTypeList, Symptom.values());
        symptomTypeList.sort(new SymptomListComparable());

        symptomIdsToMonitor = new ArrayList<>();
        Map<Long, Long> weightMap = new HashMap<>();
        Map<Long, Long> orderMap = new HashMap<>();

        for (TeamSymptom teamSymptom : symptomReportManager.getTeamSymptoms(getEHRRequestContext(), instituteCode)) {
            Long symptomId = teamSymptom.getSymptom().getId();
            symptomIdsToMonitor.add(symptomId);
            weightMap.put(symptomId, teamSymptom.getAlarmWeighting());
            orderMap.put(symptomId, teamSymptom.getOrdering());
        }

        symptomWeight = new ArrayList<>();
        symptomWeight.addAll(symptomTypeList.stream().map(symptom -> weightMap.get(symptom.getId())).collect(Collectors.toList()));

        symptomOrder = new ArrayList<>();
        symptomOrder.addAll(symptomTypeList.stream().map(symptom -> orderMap.get(symptom.getId())).collect(Collectors.toList()));

        Team team = teamManager.getTeamByCode(instituteCode, Team.Lazy.SYMPTOMS);
        if (team.getSymptomAlarmRedThreshold() != null) {
            symptomAlarmRedThreshold = team.getSymptomAlarmRedThreshold().toString();
        } else {
            symptomAlarmRedThreshold = "";
        }
        if (team.getSymptomAlarmAmberThreshold() != null) {
            symptomAlarmAmberThreshold = team.getSymptomAlarmAmberThreshold().toString();
        } else {
            symptomAlarmAmberThreshold = "";
        }

        if (team.getSymptomAlarmDays() != null) {
            symptomAlarmDays = team.getSymptomAlarmDays().toString();
        } else {
            symptomAlarmDays = "";
        }
    }
