package com.pkb.admin.action;

import com.pkb.action.BaseAction;
import com.pkb.exception.AuthorizationException;
import com.pkb.institute.entity.Team;
import com.pkb.patient.invite.bulk.BulkPatientInvitationOutcomesService;
import org.apache.struts2.interceptor.ServletResponseAware;
import org.slf4j.Logger;
import org.springframework.beans.factory.annotation.Autowired;

import javax.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.io.OutputStream;
import java.lang.invoke.MethodHandles;
import java.util.OptionalLong;

import static java.lang.String.format;
import static javax.servlet.http.HttpServletResponse.SC_BAD_REQUEST;
import static javax.servlet.http.HttpServletResponse.SC_FORBIDDEN;
import static javax.servlet.http.HttpServletResponse.SC_INTERNAL_SERVER_ERROR;
import static javax.servlet.http.HttpServletResponse.SC_NOT_FOUND;
import static org.slf4j.LoggerFactory.getLogger;

public class DownloadBulkUploadOutcomeAction extends BaseAction implements ServletResponseAware {
    private static final Logger LOGGER = getLogger(MethodHandles.lookup().lookupClass());

    private static final String FILE_PATTERN = "attachment; filename=\"bulk-patient-invitation-outcomes-%s.csv\"";
    private static final String CONTENT_DISPOSITION = "Content-Disposition";
    private static final String CSV = "text/csv";

    private long reportId;

    private HttpServletResponse servletResponse;
    private BulkPatientInvitationOutcomesService outcomesService;

    public String download() {
        String action;

        Long loggedInUserId = loggedInUserUtil.getLoggedInUserId();
        OptionalLong maybeTeamId = findTeamIdOfLoggedInUser();
        if (maybeTeamId.isPresent()) {
            try {
                action = outcomesService.findBulkPatientInvitationOutcome(maybeTeamId.getAsLong(), reportId)
                        .map(this::streamOutcomes)
                        .orElseGet(() -> {
                            LOGGER.debug("Not found bulk patient invitation outcomes with id {} for User ({})", reportId, loggedInUserId);
                            return sendError(SC_NOT_FOUND, format("We did not find anything with id '%s'", reportId));
                        });
            } catch (AuthorizationException ignored) {
                action = sendError(SC_FORBIDDEN, "You are not authorized to access this outcome");
            }
        } else {
            LOGGER.debug("User ({}) attempted to download bulk patient invitation outcomes ({}) but belongs to no team.", loggedInUserId, reportId);
            action = sendError(SC_BAD_REQUEST, "You are not member of any team.");
        }

        return action;
    }

    private String sendError(int statusCode, String errorDescription) {
        try {
            servletResponse.sendError(statusCode, errorDescription);
            return ERROR;
        } catch (IOException cause) {
            LOGGER.error("Failed to send erroneous response to client: ", cause);
            throw new IllegalStateException(cause);
        }
    }

    public void setReportId(long reportId) {
        this.reportId = reportId;
    }

    @Override
    public void setServletResponse(HttpServletResponse response) {
        this.servletResponse = response;
    }

    @Autowired
    public void setOutcomesService(BulkPatientInvitationOutcomesService outcomesService) {
        this.outcomesService = outcomesService;
    }

    // ========== Helper methods below. ==========

    private OptionalLong findTeamIdOfLoggedInUser() {
        OptionalLong maybeTeamId = OptionalLong.empty();

        Team teamOfLoggedInUser = loggedInUserUtil.getLoggedInUserPrimaryTeam(getEHRRequestContext());

        if (teamOfLoggedInUser != null) {
            maybeTeamId = OptionalLong.of(teamOfLoggedInUser.getId());
        }

        return maybeTeamId;
    }

    private String streamOutcomes(String outcome) {
        try (OutputStream stream = servletResponse.getOutputStream()) {
            servletResponse.setHeader(CONTENT_DISPOSITION, format(FILE_PATTERN, reportId));
            servletResponse.setContentLength(outcome.length());
            servletResponse.setContentType(CSV);

            stream.write(outcome.getBytes());

            stream.flush();
            return NONE;
        } catch (IOException cause) {
            LOGGER.error("Failed to serve bulk patient invitation outcome {}", reportId, cause);
            return sendError(SC_INTERNAL_SERVER_ERROR, "We apologize, but we encountered with some internal difficulties while serving your response.");
        }
    }
}
