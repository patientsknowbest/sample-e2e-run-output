package com.pkb.externalid.entity;

import org.hibernate.HibernateException;
import org.hibernate.cache.spi.access.EntityDataAccess;
import org.hibernate.cache.spi.access.NaturalIdDataAccess;
import org.hibernate.engine.spi.SharedSessionContractImplementor;
import org.hibernate.mapping.PersistentClass;
import org.hibernate.persister.entity.SingleTableEntityPersister;
import org.hibernate.persister.spi.PersisterCreationContext;
import org.springframework.data.domain.Persistable;

/**
 * Hibernate by default does not play nicely with immutable entities with pre-assigned id values. For example, during the
 * persist cycle, if attempts to determine if entities referenced as properties from other entities are transient in order
 * to determine what to do with them. Since the id is populated buy they may not be part of the session, it is forced
 * to try and reload from the DB to check whether they need persisting. With Spring Data we have a way to optimise this
 * using the Persistable interface that provides and "isNew" method. Combined with hibernate onPersist and onLoad listeners,
 * this allows us to tell Spring Data whether an entity exists in the database or not without an extra DB query.  Unfortunately
 * this only helps for top-level persistence operations on Spring Data repositories, and Hibernate knows nothing about it, so
 * references to these entities as properties on other entities still cause redundant queries. By defining a customer persister
 * that overrides the isTransient method to use the isNew method where it's available, we can prevent these queries from happening.
 */
public class PersistableAwareSingleTableEntityPersister extends SingleTableEntityPersister {
    public PersistableAwareSingleTableEntityPersister(PersistentClass persistentClass, EntityDataAccess cacheAccessStrategy, NaturalIdDataAccess naturalIdRegionAccessStrategy, PersisterCreationContext creationContext) throws HibernateException {
        super(persistentClass, cacheAccessStrategy, naturalIdRegionAccessStrategy, creationContext);
    }


    @Override
    public Boolean isTransient(Object entity, SharedSessionContractImplementor session) throws HibernateException {
        if (entity instanceof Persistable) {
            return ((Persistable) entity).isNew();
        }
        return super.isTransient(entity, session);
    }
}
