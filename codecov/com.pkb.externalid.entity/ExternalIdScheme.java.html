package com.pkb.externalid.entity;

import com.google.common.annotations.VisibleForTesting;
import org.hibernate.annotations.Immutable;
import org.hibernate.annotations.Persister;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.Table;
import java.util.UUID;

@Entity
@Table(schema = "eid", name = "external_id_scheme")
@Immutable
@Persister(impl = PersistableAwareSingleTableEntityPersister.class)
public class ExternalIdScheme extends BaseImmutableEntity {

    public static final String EMIS_SCHEME = "EMIS";
    public static final String DEFAULT_SCHEME = "DEFAULT";

    public static String getHL7PartnerScheme(String partnerCode) {
        return "HL7_PARTNER_" + partnerCode;
    }

    public ExternalIdScheme(String name) {
        this.name = name;
    }

    /**
     * Really only intended for testing where you want to be able to have a predictable id value
     */
    @VisibleForTesting
    public ExternalIdScheme(UUID id, String name) {
        this.id = id;
        this.name = name;
    }

    /**
     * For JPA
     */
    protected ExternalIdScheme() {
    }

    @Column(nullable = false)
    private String name;

    public String getName() {
        return name;
    }

    @Override
    public String toString() {
        return "ExternalIdScheme{" +
                "name='" + name + '\'' +
                '}';
    }
}
