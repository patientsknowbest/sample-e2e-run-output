package com.pkb.externalid.entity;

import org.hibernate.annotations.Type;
import org.springframework.data.domain.Persistable;

import javax.persistence.Id;
import javax.persistence.MappedSuperclass;
import javax.persistence.PostLoad;
import javax.persistence.PostPersist;
import javax.persistence.Transient;
import java.util.UUID;


/**
 * Simple base class for immutable entities that have a UUID primary key.
 * Implements Spring Data's Persistable to ensure that Spring Data doesn't
 * need to do an extra query for each save operation to know if the entity
 * is already persistent or not (normally it would tell by whether the ID
 * was populated) In our case, the persistent status of an entity is set
 * when it is saved or loaded by hibernate (as tracked by the PostLoad/PostPersist
 * listeners, otherwise it is presumed to be new.
 */
@MappedSuperclass
public abstract class BaseImmutableEntity implements Persistable<UUID> {

    @Id
    @Type(type = "pg-uuid")
    protected UUID id = UUID.randomUUID();

    @Transient
    private boolean persisted = false;

    /**
     * All our entities have equality in terms of their id. Since unique ids are assigned
     * at the time of object creation rather than persistence, this is reliable no matter
     * what JPA status the entity has.
     */
    @Override
    public final boolean equals(Object other) {
        if (this == other) {
            return true;
        }
        if (other == null) {
            return false;
        }

        //Since these are UUIDs we don't have to be type-specific here, they will be globally unique
        if (!BaseImmutableEntity.class.isAssignableFrom(other.getClass())) {
            return false;
        }

        return id.equals(((BaseImmutableEntity) other).id);

    }

    @Override
    public final int hashCode() {
        return id.hashCode();
    }


    @PostLoad
    @PostPersist
    private void setPersisted() {
        persisted = true;
    }

    @Override
    public UUID getId() {
        return id;
    }

    @Override
    public boolean isNew() {
        return !persisted;
    }
}
