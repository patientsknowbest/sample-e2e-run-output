package com.pkb.externalid.entity;

import org.hibernate.annotations.Immutable;
import org.hibernate.annotations.Persister;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.util.UUID;

@Entity
@Table(schema = "eid", name = "external_id")
@Immutable
@Persister(impl = PersistableAwareSingleTableEntityPersister.class)
public class ExternalId extends BaseImmutableEntity {

    protected ExternalId() {
    }

    public ExternalId(UUID id, ExternalIdScheme scheme, UUID orgId, String externalIdValue, UUID internalIdValue) {
        this.id = id;
        this.scheme = scheme;
        this.orgId = orgId;
        this.externalIdValue = externalIdValue;
        this.internalIdValue = internalIdValue;
    }

    @ManyToOne(fetch = FetchType.EAGER)
    @JoinColumn(name = "scheme_id")
    private ExternalIdScheme scheme;

    @Column(nullable = false, name = "org_public_id")
    @Type(type = "pg-uuid")
    private UUID orgId;

    @Column(nullable = false, name = "external_id_value")
    private String externalIdValue;

    @Column(nullable = false, name = "internal_id_value")
    private UUID internalIdValue;


    public ExternalIdScheme getScheme() {
        return scheme;
    }

    public UUID getOrgId() {
        return orgId;
    }

    public String getExternalIdValue() {
        return externalIdValue;
    }

    public UUID getInternalIdValue() {
        return internalIdValue;
    }

    @Override
    public String toString() {
        return "ExternalId{" +
                "scheme=" + scheme +
                ", orgId=" + orgId +
                ", externalIdValue='" + externalIdValue + '\'' +
                ", internalIdValue=" + internalIdValue +
                '}';
    }
}
