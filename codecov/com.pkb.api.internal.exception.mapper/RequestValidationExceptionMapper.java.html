package com.pkb.api.internal.exception.mapper;

import com.pkb.api.internal.exception.RequestValidationException;
import com.pkb.api.internal.model.response.InternalRestMeta;
import com.pkb.api.internal.model.response.InternalRestResponse;
import org.springframework.validation.FieldError;
import org.springframework.validation.ObjectError;

import javax.ws.rs.core.Response;
import javax.ws.rs.ext.ExceptionMapper;
import javax.ws.rs.ext.Provider;

@Provider
public class RequestValidationExceptionMapper extends AbstractInternalApiExceptionMapper
        implements ExceptionMapper<RequestValidationException> {

    @Override
    public Response toResponse(RequestValidationException exception) {
        InternalRestResponse<Void> responseEntity = getGeneralErrorResponseEntity();
        InternalRestMeta meta = responseEntity.getMeta();
        for (FieldError fieldError : exception.getFieldErrors()) {
            meta.addFieldError(fieldError.getField(), getMessage(fieldError.getCode(), fieldError.getArguments()));
        }
        for (ObjectError generalError : exception.getGeneralErrors()) {
            meta.addError(getMessage(generalError.getCode(), generalError.getArguments()));
        }
        return buildResponse(exception.getStatusCode().value(), responseEntity);
    }
}
