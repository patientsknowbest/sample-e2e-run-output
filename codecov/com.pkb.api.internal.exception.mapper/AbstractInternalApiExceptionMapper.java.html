package com.pkb.api.internal.exception.mapper;

import com.pkb.api.internal.model.response.InternalRestResponse;

import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;

public abstract class AbstractInternalApiExceptionMapper {

    protected String getMessage(String code) {
        //TODO: impl. I18N
        return code;
    }

    protected String getMessage(String code, Object... arg) {
        //TODO: impl. I18N
        return code;
    }

    protected InternalRestResponse<Void> getGeneralErrorResponseEntity() {
        return new InternalRestResponse<>();
    }

    protected Response buildResponse(int statusCode, Object enity) {
        return Response.status(statusCode).entity(enity).type(MediaType.APPLICATION_JSON).build();
    }
}
