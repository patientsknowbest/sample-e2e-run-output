package com.pkb.kms.service.external

import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles
import java.util.Base64

@Service
@ConditionalOnProperty(name = ["gkms.enabled"], havingValue = "false", matchIfMissing = true)
class Base64DummyEncryptionService : ExternalEncryptionService {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    init {
        log.warn("Using fake external encryption service!")
    }

    override fun encrypt(plaintext: ByteArray): ByteArray = Base64.getEncoder().encode(plaintext)
            .also { log.info("Fake-encrypted plaintext using base64") }

    override fun decrypt(ciphertext: ByteArray): ByteArray = Base64.getDecoder().decode(ciphertext)
            .also { log.info("Fake-decrypted ciphertext using base64") }
}
