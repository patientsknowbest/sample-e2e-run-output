package com.pkb.kms.service.external

import com.google.cloud.kms.v1.CryptoKeyName
import com.google.cloud.kms.v1.KeyManagementServiceClient
import com.google.protobuf.ByteString
import com.pkb.kms.service.timer
import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty
import org.springframework.stereotype.Service

@Service
@ConditionalOnProperty(name = ["gkms.enabled"], havingValue = "true")
class GoogleKMSService(private val keyVersionName: CryptoKeyName,
                       private val client: KeyManagementServiceClient) : ExternalEncryptionService {

    override fun encrypt(plaintext: ByteArray): ByteArray =
            timer("gkms.crypto.seconds", "type", "encryption").recordCallable {
                client.encrypt(keyVersionName, ByteString.copyFrom(plaintext)).ciphertext.toByteArray()
            }!!

    override fun decrypt(ciphertext: ByteArray): ByteArray =
            timer("gkms.crypto.seconds", "type", "decryption").recordCallable {
                client.decrypt(keyVersionName, ByteString.copyFrom(ciphertext)).plaintext.toByteArray()
            }!!
}
