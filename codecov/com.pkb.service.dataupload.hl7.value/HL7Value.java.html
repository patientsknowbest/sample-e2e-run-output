package com.pkb.service.dataupload.hl7.value;

import org.apache.commons.lang3.StringUtils;

/**
 * This class can be used to wrap a value provided in an HL7 message. This is
 * particularly helpful when trying to keep track of whether the value was not
 * provided (i.e. the empty string) or whether the HL7 null value ("") was
 * explicitly provided.
 */
public class HL7Value {

    /** HL7 null value, as specified in HL7 v2.4 Chapter 2 Page 12. */
    protected static final String HL7_NULL = "\"\"";

    /**
     * Strictly speaking, this isn't the "raw" value, since
     * HL7XmlDoc.getHl7Text will have probably already unescaped and trimmed
     * the original value from the message.
     */
    private String rawValue;

    protected HL7Value(String rawValue) {
        this.rawValue = rawValue;
    }

    /**
     * Determines if a value was provided in the message. This can be either a
     * non-blank value, or else the HL7 null value.
     *
     * @return true - iff a value (or the HL7 null value) was provided.
     */
    public boolean isProvided() {
        return isNotBlank() || HL7_NULL.equals(rawValue);
    }

    /**
     * Determines if the value has no content. This value is considered blank
     * if the value is any of the following:
     * <ul>
     * <li>null</li>
     * <li>blank</li>
     * <li>The HL7 null value</li>
     * </ul>
     *
     * @return true - iff the value has no content.
     */
    public boolean isBlank() {
        if (rawValue == null) {
            return true;
        }
        return StringUtils.isBlank(rawValue) || rawValue.equals(HL7_NULL);
    }

    /**
     * Returns ! isBlank()
     *
     * @return ! isBlank()
     */
    public boolean isNotBlank() {
        return !isBlank();
    }

    /**
     * Determines if this value is equal to the HL7 null value.
     *
     * @return HL7_NULL.equals(getRawValue())
     */
    public boolean isHL7Null() {
        return HL7_NULL.equals(rawValue);
    }

    /**
     * Returns ! isHL7Null()
     *
     * @return ! isHL7Null()
     */
    public boolean isNotHL7Null() {
        return !isHL7Null();
    }

    /**
     * Returns the original HL7 value. If the value was the HL7 null string,
     * then this will be returned directly.
     *
     * @return The raw value.
     */
    public String getRawValue() {
        return rawValue;
    }
}
