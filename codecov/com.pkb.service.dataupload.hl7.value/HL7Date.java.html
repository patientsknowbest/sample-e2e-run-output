package com.pkb.service.dataupload.hl7.value;

import org.jetbrains.annotations.Nullable;

import java.time.Instant;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;

/**
 * An {@code HL7Value} that contains a time stamp.
 */
public class HL7Date extends HL7Value {

    private static final DateTimeFormatter OFFSET_FORMATTER = DateTimeFormatter.ofPattern("xx");

    private ZonedDateTime zonedDateTime;

    private String utcOffset = "+0000";

    public HL7Date() {
        super("");
        this.zonedDateTime = null;
        this.utcOffset = "";
    }

    /**
     * All time stamps extracted by PKB from an HL7 message are interpreted within a particular time zone.
     * <p>
     * The rawValue is helpful when we need to know the local representation of the original value, e.g. when parsing the patient's date of birth.
     *
     * @param rawValue      The literal String provided in the HL7 message
     * @param zonedDateTime The point in time, parsed from the rawValue
     * @see http://dev.patientsknowbest.com/home/hl7-api/time-stamp-values
     */
    public HL7Date(String rawValue, ZonedDateTime zonedDateTime) {
        super(rawValue);
        if (zonedDateTime != null) {
            this.zonedDateTime = zonedDateTime;
            this.utcOffset = OFFSET_FORMATTER.format(zonedDateTime);
        }
    }

    /**
     * This might return null. Callers should us isProvided to determine whether the original value was missing, or else the HL7 null value.
     *
     * @return The ZonedDateTime, as parsed from the HL7 value
     */
    @Nullable
    public ZonedDateTime getZonedDateTime() {
        return zonedDateTime;
    }

    /**
     * @return zonedDateTime.toInstant(), else null
     */
    @Nullable
    public Instant getInstant() {
        if (zonedDateTime != null) {
            return zonedDateTime.toInstant();
        }
        return null;
    }

    public String getUtcOffset() {
        return utcOffset;
    }
}
