package com.pkb.service.dataupload.hl7.value;

import org.mapstruct.BeanMapping;
import org.mapstruct.Mapper;
import org.mapstruct.ReportingPolicy;

import java.time.Instant;
import java.time.ZonedDateTime;
import java.util.Optional;

@Mapper(unmappedSourcePolicy = ReportingPolicy.ERROR)
public interface HL7DateMapper {
    @BeanMapping(ignoreUnmappedSourceProperties = { "provided", "blank", "notBlank", "HL7Null", "notHL7Null", "date" })
    HL7DateComparisonDTO map(HL7Date hl7Date);

    /** Handle ZonedDateTime objects */
    default Optional<ZonedDateTime> mapToOptional(ZonedDateTime value) {
        return Optional.ofNullable(value);
    }

    /** Handle Instant objects */
    default Optional<Instant> mapToOptional(Instant value) { return Optional.ofNullable(value); }

    /** Handle String objects */
    default Optional<String> mapToOptional(String value) {
        return Optional.ofNullable(value);
    }
}
