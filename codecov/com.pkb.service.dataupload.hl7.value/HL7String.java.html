package com.pkb.service.dataupload.hl7.value;

/**
 * An {@code HL7Value} that contains textual content.
 */
public class HL7String extends HL7Value {

    public static final HL7String EMPTY = HL7String.of("");

    public static HL7String of(String string) {
        return new HL7String(string);
    }

    /**
     * A separate copy of the string is kept here so that processing code can
     * modify the contents without destroying the original raw value.
     */
    private String string;

    public HL7String() {
        super("");
        this.string = "";
    }

    public HL7String(String value) {
        super(value);
        this.string = value;
    }

    @Override
    public String toString() {
        return getString();
    }

    /**
     * Returns the string value. If the string value is equal to the HL7 null
     * value (""), then an empty string will be returned.
     *
     * @return "" - if the string value is the HL7 null value. Else - the
     * string value is directly returned.
     */
    public String getString() {
        if (HL7_NULL.equals(string)) {
            return "";
        }
        return string;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (o == null || getClass() != o.getClass()) {
            return false;
        }
        HL7String hl7String = (HL7String) o;
        return string != null ? string.equals(hl7String.string) : hl7String.string == null;
    }

    @Override public int hashCode() {
        return string != null ? string.hashCode() : 0;
    }
}
