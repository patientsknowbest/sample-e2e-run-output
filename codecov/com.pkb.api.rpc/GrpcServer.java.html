package com.pkb.api.rpc;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.pkb.api.rpc.email.EmailService;
import com.pkb.api.rpc.message.MessageService;
import com.pkb.common.config.PhrConfig;
import com.pkb.service.emailmessage.IAddressInfoProvider;
import com.pkb.service.emailmessage.impl.EmailManager;
import com.pkb.service.emailmessage.impl.TemplateParamsFactory;
import com.pkb.service.emailmessage.sender.impl.VelocityTemplateHandler;
import com.pkb.service.encounter.impl.tolven.EncounterManager;
import com.pkb.service.user.impl.UserManager;
import io.grpc.Server;
import io.grpc.ServerBuilder;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.IOException;

public class GrpcServer {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    public GrpcServer(PhrConfig config, EncounterManager encounterManager, EmailManager emailManager, UserManager userManager, VelocityTemplateHandler velocityTemplateHandler,
                      IAddressInfoProvider formatter, TemplateParamsFactory paramsFactory, ObjectMapper objectMapper) throws IOException, InterruptedException {
        Server server =
                ServerBuilder.forPort(config.getGrpcServerPort())
                        .addService(new EmailService(emailManager, userManager, formatter, paramsFactory, objectMapper, config))
                        .addService(new MessageService(encounterManager, velocityTemplateHandler, objectMapper, config))
                        .build();
        server.start();
        LOGGER.info(String.format("Successfully started Message & Email Grpc Server on port %s", config.getGrpcServerPort()));
        Runtime.getRuntime().addShutdownHook(new Thread(() -> {
            server.shutdown();
            LOGGER.info("Successfully stopped Message & Email Grpc Server");
        }));
    }
}
