package com.pkb.aggregator.camel.routing.resource.aggregated

import com.pkb.aggregator.camel.NEXT_ROUTE_PROPERTY
import com.pkb.aggregator.camel.bean.process.PractitionerRoleProcessor
import com.pkb.aggregator.camel.bean.process.ResourcePreProcessor
import com.pkb.aggregator.camel.bean.search.PractitionerRoleSearch
import com.pkb.aggregator.camel.beanMethod
import com.pkb.aggregator.camel.routing.AbstractAggregatorRouteBuilder
import com.pkb.aggregator.camel.routing.RouteResolver
import com.pkb.aggregator.camel.routing.createFhirSearchRoute
import com.pkb.aggregator.camel.routing.resource.MissingReferenceProcessor
import com.pkb.aggregator.configuration.FhirConfiguration
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import org.springframework.stereotype.Component

@Component
class PractitionerRoleRouteBuilder(
    private val fhirConfiguration: FhirConfiguration,
    private val missingReferenceProcessor: MissingReferenceProcessor,
    pubSubNamespaceService: PubSubNamespaceService,
) : AbstractAggregatorRouteBuilder(pubSubNamespaceService) {

    companion object {
        const val PROCESS_PRACTITIONER_ROLE = "direct:aggregator/practitionerRole"
        const val PROCESS_DELETE_PRACTITIONER_ROLE = "direct:aggregator/delete/practitionerRole"

        private const val ROUTE_ID = "practitionerrole-change"
        private const val ROUTE_ID_DELETE = "$ROUTE_ID-delete"
    }

    override fun configure() {

        registerAllErrorHandlers(missingReferenceProcessor)

        from(PROCESS_PRACTITIONER_ROLE).routeId(ROUTE_ID)
            .beanMethod(PractitionerRoleSearch::findDownstreamResourceIds).id("$ROUTE_ID-downstream-id-search")
            .beanMethod(PractitionerRoleSearch::prepareSearchUrl).id("$ROUTE_ID-prepare-downstream-search")
            .to(createFhirSearchRoute(fhirConfiguration.downstream.endpoint)).id("$ROUTE_ID-perform-downstream-search")
            .choice().id("$ROUTE_ID-check-if-known")
                .`when`(simple("\${body.entry.size} > 0"))
                    .beanMethod(PractitionerRoleSearch::setDownstreamRoleResourceId).id("$ROUTE_ID-set-downstream-id")
                    .beanMethod(PractitionerRoleProcessor::processSingle).id("$ROUTE_ID-process-known")
                .otherwise()
                    .beanMethod(PractitionerRoleProcessor::processNew).id("$ROUTE_ID-process-new")

        from(PROCESS_DELETE_PRACTITIONER_ROLE).routeId(ROUTE_ID_DELETE)
            .beanMethod(RouteResolver::resolveUpstreamRouteForNotification).id("$ROUTE_ID_DELETE-resolve-upstream")
            .toD("\${exchangeProperty[$NEXT_ROUTE_PROPERTY]}").id("$ROUTE_ID_DELETE-retrieve-upstream")
            .beanMethod(ResourcePreProcessor::associateWithSource).id("${ROUTE_ID_DELETE}-associate-with-source")
            .beanMethod(PractitionerRoleSearch::findDownstreamResourceIds).id("${ROUTE_ID_DELETE}-find-downstream-id")
            .beanMethod(PractitionerRoleSearch::prepareSearchUrl).id("${ROUTE_ID_DELETE}-prepare-downstream-search")
            .to(createFhirSearchRoute(fhirConfiguration.downstream.endpoint)).id("${ROUTE_ID_DELETE}-perform-downstream-search")
            .beanMethod(PractitionerRoleProcessor::processDelete).id("${ROUTE_ID_DELETE}-process")
    }
}
