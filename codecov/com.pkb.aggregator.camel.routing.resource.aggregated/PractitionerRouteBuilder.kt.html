package com.pkb.aggregator.camel.routing.resource.aggregated

import com.pkb.aggregator.camel.CamelFhirHelper
import com.pkb.aggregator.camel.IDENTIFIERS_PROPERTY
import com.pkb.aggregator.camel.bean.process.PractitionerProcessor
import com.pkb.aggregator.camel.bean.search.PractitionerSearch
import com.pkb.aggregator.camel.beanMethod
import com.pkb.aggregator.camel.routing.AbstractAggregatorRouteBuilder
import com.pkb.aggregator.camel.routing.createFhirSearchRoute
import com.pkb.aggregator.camel.routing.resource.MissingReferenceProcessor
import com.pkb.aggregator.configuration.FhirConfiguration
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import org.springframework.stereotype.Component

@Component
class PractitionerRouteBuilder(
    private val fhirConfiguration: FhirConfiguration,
    private val missingReferenceProcessor: MissingReferenceProcessor,
    pubSubNamespaceService: PubSubNamespaceService,
) : AbstractAggregatorRouteBuilder(pubSubNamespaceService) {

    companion object {
        const val PROCESS_PRACTITIONER = "direct:aggregator/practitioner"
        const val PROCESS_KNOWN_PRACTITIONER = "direct:aggregator/update/practitioner"
        const val PROCESS_NEW_PRACTITIONER = "direct:aggregator/create/practitioner"
        const val PROCESS_DELETE_PRACTITIONER = "direct:aggregator/delete/practitioner"

        private const val ROUTE_ID = "practitioner-change"
        private const val ROUTE_ID_CREATE = "$ROUTE_ID-create"
        private const val ROUTE_ID_UPDATE = "$ROUTE_ID-update"
        private const val ROUTE_ID_DELETE = "$ROUTE_ID-delete"
    }

    override fun configure() {

        registerAllErrorHandlers(missingReferenceProcessor)

        from(PROCESS_PRACTITIONER).routeId(ROUTE_ID)
            .beanMethod(PractitionerSearch::findAllIdentifiers).id("$ROUTE_ID-identifier-search")
            .choice().id("$ROUTE_ID-check-if-known")
                .`when`(simple("\${exchangeProperty.$IDENTIFIERS_PROPERTY.size} > 0"))
                    .to(PROCESS_KNOWN_PRACTITIONER).id("$ROUTE_ID-known")
                .otherwise()
                    .to(PROCESS_NEW_PRACTITIONER).id("$ROUTE_ID-new")

        from(PROCESS_KNOWN_PRACTITIONER).routeId(ROUTE_ID_UPDATE)
            .beanMethod(CamelFhirHelper::prepareDownstreamSearchById).id("$ROUTE_ID_UPDATE-prepare-downstream-search")
            .to(createFhirSearchRoute(fhirConfiguration.downstream.endpoint)).id("$ROUTE_ID_UPDATE-perform-downstream-search")
            .beanMethod(PractitionerProcessor::processSingle).id("$ROUTE_ID_UPDATE-process")

        from(PROCESS_NEW_PRACTITIONER).routeId(ROUTE_ID_CREATE)
            .beanMethod(PractitionerProcessor::processNew).id("$ROUTE_ID_CREATE-process")

        from(PROCESS_DELETE_PRACTITIONER).routeId(ROUTE_ID_DELETE)
            .beanMethod(PractitionerSearch::findDownstreamIdByDeltaNotification).id("$ROUTE_ID_DELETE-find-downstream-id")
            .beanMethod(CamelFhirHelper::prepareDownstreamSearchById).id("$ROUTE_ID_DELETE-prepare-downstream-search")
            .to(createFhirSearchRoute(fhirConfiguration.downstream.endpoint)).id("$ROUTE_ID_DELETE-perform-downstream-search")
            .beanMethod(PractitionerProcessor::processDelete).id("$ROUTE_ID_DELETE-process")
    }
}
