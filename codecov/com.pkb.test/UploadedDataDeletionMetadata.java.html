package com.pkb.test;

import com.pkb.app.entity.SourceDetails;

import java.io.Serializable;
import java.time.Instant;

/**
 * When a deletion has been trigered by a route that creates an UploadedData entry, this class allows for some of the
 * default source tracking attributes to be overwritten.
 *
 * It also stores the Filler Order Number, where applicable.
 *
 * This is typically not needed when _adding_ data because the relevant information can be set directly onto the data
 * points being persisted, but some deletion flows use criteria-based deletion (e.g. matching unique ID or Filler Order
 * Number) so the metadata about the deletion needs to be carried separately because there might not be any instances
 * of IBaseDTO to set this information onto.
 */
public class UploadedDataDeletionMetadata implements Serializable {

    // The Filler Order Number of the data to be deleted, where applicable.
    private String maybeFillerOrderNumber;

    // The requirement for this is questionable, but for now this allows the ordering clinician's name to be included in
    // the source information about the deletion.
    private SourceDetails sourceDetails;

    // The timestamp will not always be "now" for deletions triggered by an UploadedData entry because the entry could
    // have been waiting for a long time before being processed.
    private Instant timestamp;

    // A data point can be created, uploaded and deleted by separate UploadedData entries, so we should track the ID
    // of the specific entry which triggered the deletion, rather than copying the uploadedDataId from the version being
    // deleted.
    private Long uploadedDataId;

    public UploadedDataDeletionMetadata(String maybeFillerOrderNumber, SourceDetails sourceDetails, Instant timestamp, Long uploadedDataId) {
        this.maybeFillerOrderNumber = maybeFillerOrderNumber;
        this.sourceDetails = sourceDetails;
        this.timestamp = timestamp;
        this.uploadedDataId = uploadedDataId;
    }

    public String getMaybeFillerOrderNumber() {
        return maybeFillerOrderNumber;
    }

    public SourceDetails getSourceDetails() {
        return sourceDetails;
    }

    public Instant getTimestamp() {
        return timestamp;
    }

    public Long getUploadedDataId() {
        return uploadedDataId;
    }
}
