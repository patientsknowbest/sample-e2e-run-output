package com.pkb.service.dataupload.hl7.field;

import com.pkb.service.dataupload.hl7.value.HL7String;
import org.w3c.dom.Element;

import static com.pkb.service.dataupload.hl7.HL7XmlDoc.getHl7Text;
import static com.pkb.service.dataupload.hl7.HL7XmlDoc.getIdentifier;

/**
 * Parses and holds an HD (Hierarchic Designator) field. See HL7 v2.4 Chapter 2.
 */
public class HL7HierarchicDesignator {
    private HL7String namespaceId = new HL7String();
    private HL7String universalId = new HL7String();
    private HL7String universalIdType = new HL7String();

    public HL7HierarchicDesignator() {
    }

    /**
     * For testing only
     */
    public HL7HierarchicDesignator(String namespaceId, String universalId, String universalIdType) {
        this.namespaceId = new HL7String(namespaceId);
        this.universalId = new HL7String(universalId);
        this.universalIdType = new HL7String(universalIdType);
    }

    public void parseHd(Element element, String segment, int field) {
        namespaceId = getHl7Text(element, getIdentifier(segment, field, 1));
        universalId = getHl7Text(element, getIdentifier(segment, field, 2));
        universalIdType = getHl7Text(element, getIdentifier(segment, field, 3));
    }

    /**
     * It is generally good practice in HL7 v2 syntax not to leave trailing separators, so before adding the component
     * separator we check to see if the next (or any subsequent) components are present.
     *
     * @return An HL7 v2 representation of the HD field.
     */
    public String toHl7() {
        String hl7 = namespaceId.getString();
        if (universalId.isNotBlank() || universalIdType.isNotBlank()) {
            hl7 += "^" + universalId.getString();
        }
        if (universalIdType.isNotBlank()) {
            hl7 += "^" + universalIdType.getString();
        }
        return hl7;
    }

    public HL7String getNamespaceId() {
        return namespaceId;
    }

    public HL7String getUniversalId() {
        return universalId;
    }

    public HL7String getUniversalIdType() {
        return universalIdType;
    }
}
