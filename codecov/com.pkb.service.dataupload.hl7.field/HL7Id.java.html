package com.pkb.service.dataupload.hl7.field;

import com.pkb.service.dataupload.hl7.value.HL7String;
import org.apache.commons.lang3.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.w3c.dom.Element;

import java.lang.invoke.MethodHandles;

import static com.pkb.service.dataupload.hl7.HL7XmlDoc.getHl7Text;
import static com.pkb.service.dataupload.hl7.HL7XmlDoc.getIdentifier;

/**
 * Parses and holds a CX (Extended Composite ID with Check Digit) field. See HL7 v2.4 Chapter 2.
 *
 * The 3 component values that are of interest to PKB are:
 *
 * <ul>
 * <li>Value</li>
 * <li>Assigning Authority</li>
 * <li>Type Code</li>
 * </ul>
 */
public class HL7Id {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private HL7String value = new HL7String();
    private HL7String assigningAuthority = new HL7String();
    private HL7String idTypeCode = new HL7String();

    // Note: Not of type HL7String because this is PKB specified meta-data; the HL7 null value ("") is not permitted for this value.
    private String idTypeCodeJson = "";

    public void parseCx(Element element, String segment, int field) {

        value = getHl7Text(element, getIdentifier(segment, field, 1));

        String tmpAA = StringUtils.trimToEmpty(getHl7Text(element, getIdentifier(segment, field, 4)).getRawValue());
        String tmpTC = StringUtils.trimToEmpty(getHl7Text(element, getIdentifier(segment, field, 5)).getRawValue());

        // Allow JSON metadata after type code. Initially to allow NWL to send the NHS number status, e.g. NH{status:01}
        if (tmpTC.contains("{")) {
            String orig = tmpTC;
            int idx = orig.indexOf("{");
            tmpTC = orig.substring(0, idx);
            idTypeCodeJson = orig.substring(idx);
        }

        assigningAuthority = new HL7String(tmpAA);
        idTypeCode = new HL7String(tmpTC);

        // PHR-11379: Temporary logging before using EMAIL/EMAIL as a reserved pair
        boolean emailAA = "email".equalsIgnoreCase(tmpAA);
        boolean emailTC = "email".equalsIgnoreCase(tmpTC);
        if (emailAA || emailTC) {
            LOGGER.warn("PHR-11379: email found aa={} tc={}", tmpAA, tmpTC);
        }
    }

    /**
     * An HL7 ID must contain a value and a type code to be processed by PKB. The assigning authority is optional because that portion is
     * allowed to be wildcarded.
     *
     * This method checks whether the mandatory values have been provided. This allows us to silently ignore any incomplete HL7 IDs,
     * rather than throwing a parse error up front, which makes it a little easier for integrating customers.
     *
     * @return true - iff a value and type code were provided. false - otherwise.
     */
    public boolean isValid() {
        return value.isNotBlank() && idTypeCode.isNotBlank();
    }

    public HL7String getValue() {
        return value;
    }

    public void setValue(String value) {
        this.value = new HL7String(value);
    }

    public HL7String getAssigningAuthority() {
        return assigningAuthority;
    }

    public void setAssigningAuthority(String assigningAuthority) {
        this.assigningAuthority = new HL7String(assigningAuthority);
    }

    public HL7String getIdTypeCode() {
        return idTypeCode;
    }

    public void setIdTypeCode(String idTypeCode) {
        this.idTypeCode = new HL7String(idTypeCode);
    }

    public String getIdTypeCodeJson() {
        return idTypeCodeJson;
    }

    public void setIdTypeCodeJson(String idTypeCodeJson) {
        this.idTypeCodeJson = idTypeCodeJson;
    }
}
