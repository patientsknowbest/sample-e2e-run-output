package com.pkb.service.dataupload.hl7.field;

import com.pkb.coding.entity.CodeableConcept;
import com.pkb.coding.entity.Coding;
import com.pkb.service.dataupload.hl7.value.HL7String;
import org.apache.commons.lang3.StringUtils;
import org.w3c.dom.Element;

import static com.pkb.service.dataupload.hl7.HL7XmlDoc.getHl7Text;
import static com.pkb.service.dataupload.hl7.HL7XmlDoc.getIdentifier;

/**
 * Parses and holds a CE (Coded Element) field. See HL7 v2.4 Chapter 2.
 */
public class HL7Code {

    private HL7String id = new HL7String();
    private HL7String text = new HL7String();
    private HL7String codingSystem = new HL7String();
    private HL7String altId = new HL7String();
    private HL7String altText = new HL7String();
    private HL7String altCodingSystem = new HL7String();

    public void parseCe(Element element, String segment, int field) {
        id = getHl7Text(element, getIdentifier(segment, field, 1));
        text = getHl7Text(element, getIdentifier(segment, field, 2));
        codingSystem = getHl7Text(element, getIdentifier(segment, field, 3));
        altId = getHl7Text(element, getIdentifier(segment, field, 4));
        altText = getHl7Text(element, getIdentifier(segment, field, 5));
        altCodingSystem = getHl7Text(element, getIdentifier(segment, field, 6));
    }

    public CodeableConcept asCodeableConcept() {
        Coding primary = new Coding.Builder()
                .code(StringUtils.trimToNull(id.getString()))
                .codeSystem(StringUtils.trimToNull(codingSystem.getString()))
                .displayText(StringUtils.trimToNull(text.getString())).build();
        Coding alternative = new Coding.Builder()
                .code(StringUtils.trimToNull(altId.getString()))
                .codeSystem(StringUtils.trimToNull(altCodingSystem.getString()))
                .displayText(StringUtils.trimToNull(altText.getString())).build();
        CodeableConcept.Builder ccb = new CodeableConcept.Builder()
                .coding(primary)
                .coding(alternative);
        return ccb.build();
    }

    /**
     * Determines if all mandatory values were provided.
     *
     * Returns true. PKB does not have a consistent approach to which
     * information is mandatory for a coded value.
     *
     * @return true
     */
    public boolean isValid() {
        return true;
    }

    public HL7String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = new HL7String(id);
    }

    public HL7String getText() {
        return text;
    }

    public void setText(String text) {
        this.text = new HL7String(text);
    }

    public HL7String getCodingSystem() {
        return codingSystem;
    }

    public void setCodingSystem(String codingSystem) {
        this.codingSystem = new HL7String(codingSystem);
    }

    public HL7String getAltId() {
        return altId;
    }

    public void setAltId(String altId) {
        this.altId = new HL7String(altId);
    }

    public HL7String getAltText() {
        return altText;
    }

    public void setAltText(String altText) {
        this.altText = new HL7String(altText);
    }

    public HL7String getAltCodingSystem() {
        return altCodingSystem;
    }

    public void setAltCodingSystem(String altCodingSystem) {
        this.altCodingSystem = new HL7String(altCodingSystem);
    }
}
