package com.pkb.service.dataupload.hl7.field;

import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Optional;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

import static java.util.Optional.empty;

/**
 * Despite the similar name, this is not a representation of the HL7 RFR (reference range) data type.
 *
 * Instead, this captures PKB's semantic interpretation of an ST (textual) value from OBX-7 in an ORU R01 message.
 */
@Value.Immutable
@Value.Style(strictBuilder = true)
public interface HL7ReferenceRange {

    /** PKB does not validate that the "low" value is numerically lower than (or equal to) the "high" value. */
    Optional<Double> getRangeLow();

    boolean getRangeLowInclusive();

    /** PKB does not validate that the "high" value is numerically greater than (or equal to) the "lower" value. */
    Optional<Double> getRangeHigh();

    boolean getRangeHighInclusive();

    /** This is only populated if the provided value did not successfully parse as a numeric range. */
    String getRangeText();

    String NUMBER_REGEX = "([+-]?[0-9]+(?:\\.[0-9]+)?)";
    Pattern PATTERN_LT = Pattern.compile("< *" + NUMBER_REGEX);
    Pattern PATTERN_LE = Pattern.compile("<= *" + NUMBER_REGEX);
    Pattern PATTERN_GT = Pattern.compile("> *" + NUMBER_REGEX);
    Pattern PATTERN_GE = Pattern.compile(">= *" + NUMBER_REGEX);
    Pattern PATTERN_RANGE = Pattern.compile(NUMBER_REGEX + " *- *" + NUMBER_REGEX);

    Logger LOGGER = LoggerFactory.getLogger(HL7ReferenceRange.class);

    static @NotNull HL7ReferenceRange parse(@NotNull String value) {

        // Special case #1: Treat - as blank
        if ("-".equals(value)) {
            return ImmutableHL7ReferenceRange.builder()
                    .rangeLow(empty())
                    .rangeLowInclusive(false)
                    .rangeHigh(empty())
                    .rangeHighInclusive(false)
                    .rangeText("")
                    .build();
        }

        // Special case #2: Treat 0 as 0-0
        if ("0".equals(value)) {
            return ImmutableHL7ReferenceRange.builder()
                    .rangeLow(Optional.of(Double.valueOf("0")))
                    .rangeLowInclusive(true)
                    .rangeHigh(Optional.of(Double.valueOf("0")))
                    .rangeHighInclusive(true)
                    .rangeText("")
                    .build();
        }

        Matcher m;

        try {
            m = PATTERN_LT.matcher(value);
            if (m.matches()) {
                return ImmutableHL7ReferenceRange.builder()
                        .rangeLow(empty())
                        .rangeLowInclusive(false)
                        .rangeHigh(Optional.of(Double.valueOf(m.group(1))))
                        .rangeHighInclusive(false)
                        .rangeText("")
                        .build();
            }
            m = PATTERN_LE.matcher(value);
            if (m.matches()) {
                return ImmutableHL7ReferenceRange.builder()
                        .rangeLow(empty())
                        .rangeLowInclusive(false)
                        .rangeHigh(Optional.of(Double.valueOf(m.group(1))))
                        .rangeHighInclusive(true)
                        .rangeText("")
                        .build();
            }
            m = PATTERN_GT.matcher(value);
            if (m.matches()) {
                return ImmutableHL7ReferenceRange.builder()
                        .rangeLow(Optional.of(Double.valueOf(m.group(1))))
                        .rangeLowInclusive(false)
                        .rangeHigh(empty())
                        .rangeHighInclusive(false)
                        .rangeText("")
                        .build();
            }
            m = PATTERN_GE.matcher(value);
            if (m.matches()) {
                return ImmutableHL7ReferenceRange.builder()
                        .rangeLow(Optional.of(Double.valueOf(m.group(1))))
                        .rangeLowInclusive(true)
                        .rangeHigh(empty())
                        .rangeHighInclusive(false)
                        .rangeText("")
                        .build();
            }
            m = PATTERN_RANGE.matcher(value);
            if (m.matches()) {
                return ImmutableHL7ReferenceRange.builder()
                        .rangeLow(Optional.of(Double.valueOf(m.group(1))))
                        .rangeLowInclusive(true)
                        .rangeHigh(Optional.of(Double.valueOf(m.group(2))))
                        .rangeHighInclusive(true)
                        .rangeText("")
                        .build();
            }
        } catch (Exception e) {
            // If something went wrong parsing the numerical range, we should continue to the code below which will return a textual range
            LOGGER.error("Exception parsing reference range", e);
        }

        // If we haven't returned a numerical range by this stage, then return a textual range
        return ImmutableHL7ReferenceRange.builder()
                .rangeLow(empty())
                .rangeLowInclusive(false)
                .rangeHigh(empty())
                .rangeHighInclusive(false)
                .rangeText(value)
                .build();
    }
}
