package com.pkb.service.dataupload.hl7.field;

import com.pkb.service.dataupload.hl7.value.HL7String;
import org.w3c.dom.Element;

import java.io.Serializable;

import static com.pkb.service.dataupload.hl7.HL7XmlDoc.getHl7Text;
import static com.pkb.service.dataupload.hl7.HL7XmlDoc.getIdentifier;

/**
 * Parses and holds a PL (Person Location) field. See HL7 v2.4 Chapter 2.
 * 
 * This type consists of the following components:
 * 
 * <ul>
 * <li>point of care (IS)</li>
 * <li>room (IS)</li>
 * <li>bed (IS)</li>
 * <li>facility (HD)</li>
 * <li>location status (IS)</li>
 * <li>person location type (IS)</li>
 * <li>building (IS)</li>
 * <li>floor (IS)</li>
 * <li>Location description (ST)</li>
 * </ul>
 * 
 */
public class HL7PersonLocation implements Serializable {

    private HL7String pointOfCare = new HL7String();
    private HL7String room = new HL7String();
    private HL7String bed = new HL7String();
    private HL7String facility = new HL7String();
    private HL7String locationStatus = new HL7String();
    private HL7String personLocationType = new HL7String();
    private HL7String building = new HL7String();
    private HL7String floor = new HL7String();
    private HL7String locationDescription = new HL7String();

    public void parsePl(Element element, String segment, int field) {
        this.pointOfCare = getHl7Text(element, getIdentifier(segment, field, 1));
        this.room = getHl7Text(element, getIdentifier(segment, field, 2));
        this.bed = getHl7Text(element, getIdentifier(segment, field, 3));
        this.facility = getHl7Text(element, getIdentifier(segment, field, 4));
        this.locationStatus = getHl7Text(element, getIdentifier(segment, field, 5));
        this.personLocationType = getHl7Text(element, getIdentifier(segment, field, 6));
        this.building = getHl7Text(element, getIdentifier(segment, field, 7));
        this.floor = getHl7Text(element, getIdentifier(segment, field, 8));
        this.locationDescription = getHl7Text(element, getIdentifier(segment, field, 9));
    }
    
    public boolean isValid() {
        // Currently we only require location description field
        return locationDescription.isNotBlank();
    }

    public HL7String getPointOfCare() {
        return pointOfCare;
    }

    public void setPointOfCare(HL7String pointOfCare) {
        this.pointOfCare = pointOfCare;
    }

    public HL7String getRoom() {
        return room;
    }

    public void setRoom(HL7String room) {
        this.room = room;
    }

    public HL7String getBed() {
        return bed;
    }

    public void setBed(HL7String bed) {
        this.bed = bed;
    }

    public HL7String getFacility() {
        return facility;
    }

    public void setFacility(HL7String facility) {
        this.facility = facility;
    }

    public HL7String getLocationStatus() {
        return locationStatus;
    }

    public void setLocationStatus(HL7String locationStatus) {
        this.locationStatus = locationStatus;
    }

    public HL7String getPersonLocationType() {
        return personLocationType;
    }

    public void setPersonLocationType(HL7String personLocationType) {
        this.personLocationType = personLocationType;
    }

    public HL7String getBuilding() {
        return building;
    }

    public void setBuilding(HL7String building) {
        this.building = building;
    }

    public HL7String getFloor() {
        return floor;
    }

    public void setFloor(HL7String floor) {
        this.floor = floor;
    }

    public HL7String getLocationDescription() {
        return locationDescription;
    }

    public void setLocationDescription(HL7String locationDescription) {
        this.locationDescription = locationDescription;
    }
}
