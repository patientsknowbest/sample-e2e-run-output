package com.pkb.service.dataupload.processor.domain;

import com.pkb.document.entity.Attachment;
import com.pkb.encounter.entity.EncounterEvent;

public class UploadedDocument {
    private EncounterEvent encounterEvent;
    private Attachment attachment;
    private UploadedDocumentRequestType requestType;

    public UploadedDocument(EncounterEvent encounterEvent, Attachment attachment, UploadedDocumentRequestType requestType) {
        this.encounterEvent = encounterEvent;
        this.attachment = attachment;
        this.requestType = requestType;
    }

    public EncounterEvent getEncounterEvent() {
        return encounterEvent;
    }

    public Attachment getAttachment() {
        return attachment;
    }

    public UploadedDocumentRequestType getRequestType() {
        return requestType;
    }
}
