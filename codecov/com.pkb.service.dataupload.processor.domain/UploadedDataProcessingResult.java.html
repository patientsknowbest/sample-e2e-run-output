package com.pkb.service.dataupload.processor.domain;

import java.util.List;
import java.util.Optional;

import com.pkb.dataupload.entity.UploadedData;

public class UploadedDataProcessingResult<T> {
    private UploadedDataPersistentState persistentState;
    private List<T> parsingResult;
    private Exception errorCause;

    public UploadedDataProcessingResult(UploadedDataPersistentState persistentState, List<T> parsingResult) {
        this.persistentState = persistentState;
        this.parsingResult = parsingResult;
    }

    public UploadedData.Status getStatus() {
        return persistentState.getUploadedDataStatus();
    }

    public UploadedDataPersistentState.UploadedDataPersistentStateEnum getPersistentState() {
        return persistentState.getPersistentState();
    }

    public void rollbackProcessingResult(){
        persistentState = new UploadedDataPersistentState(UploadedData.Status.ERROR, UploadedDataPersistentState.UploadedDataPersistentStateEnum.ROLLED_BACK);
    }

    public List<T> getParsingResult() {
        return parsingResult;
    }

    public Optional<T> getSingleResult() {
        return getParsingResult().stream().findFirst();

    }

    public void setErrorCause(Exception errorCause) {
        this.errorCause = errorCause;
    }

    public Exception getErrorCause() {
        return errorCause;
    }
}
