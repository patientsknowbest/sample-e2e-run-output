package com.pkb.service.dataupload.processor.domain;

import java.util.ArrayList;
import java.util.List;

import com.pkb.encounter.entity.Encounter;
import com.pkb.encounter.entity.EncounterEvent;
import com.pkb.encounter.entity.Message;
import com.pkb.service.dataupload.hl7.HL7ConnContext;
import com.pkb.service.dataupload.hl7.field.HL7Code;

public class UploadedEncounter {
    private EncounterEvent encounterEvent;
    private Encounter encounter;
    private UploadedEncounterRequestType encounterRequestType;
    private List<Message> messages;
    private HL7Code customAppointmentTypeCode;
    private HL7ConnContext hl7ConnContext;

    public UploadedEncounter(EncounterEvent encounterEvent, Encounter encounter, UploadedEncounterRequestType encounterRequestType) {
        this.encounterEvent = encounterEvent;
        this.encounter = encounter;
        this.encounterRequestType = encounterRequestType;
        this.messages = new ArrayList<>();
    }

    public UploadedEncounter(EncounterEvent encounterEvent, Encounter encounter, UploadedEncounterRequestType encounterRequestType, List<Message> messages) {
        this.encounterEvent = encounterEvent;
        this.encounter = encounter;
        this.encounterRequestType = encounterRequestType;
        this.messages = messages;
    }

    public EncounterEvent getEncounterEvent() {
        return encounterEvent;
    }

    public Encounter getEncounter() {
        return encounter;
    }

    public UploadedEncounterRequestType getEncounterRequestType() {
        return encounterRequestType;
    }

    public List<Message> getMessages() {
        return messages;
    }

    public void setCustomAppointmentTypeCode(HL7Code customAppointmentTypeCode) {
        this.customAppointmentTypeCode = customAppointmentTypeCode;
    }

    public void setHL7ConnContext(HL7ConnContext hl7ConnContext) {
        this.hl7ConnContext = hl7ConnContext;
    }

    public HL7Code getCustomAppointmentTypeCode() {
        return customAppointmentTypeCode;
    }

    public HL7ConnContext getHL7ConnContext() {
        return hl7ConnContext;
    }
}
