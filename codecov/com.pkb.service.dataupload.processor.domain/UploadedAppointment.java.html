package com.pkb.service.dataupload.processor.domain;

import com.pkb.app.entity.SourceDetails;
import com.pkb.calendar.entity.AppointmentDTO;
import com.pkb.service.dataupload.hl7.segment.HL7DatapointExternalIntegrationUrl;

import java.util.Optional;

public class UploadedAppointment {
    private AppointmentDTO appointment;
    private UploadedAppointmentRequestType requestType;
    private SourceDetails sourceDetails;
    private Optional<HL7DatapointExternalIntegrationUrl> externalIntegrationUrl;

    public UploadedAppointment(AppointmentDTO appointment, UploadedAppointmentRequestType requestType, SourceDetails sourceDetails) {
        this.appointment = appointment;
        this.requestType = requestType;
        this.sourceDetails = sourceDetails;
        externalIntegrationUrl = Optional.empty();
    }

    public UploadedAppointment(AppointmentDTO appointment, UploadedAppointmentRequestType requestType, SourceDetails sourceDetails,
                               Optional<HL7DatapointExternalIntegrationUrl> externalIntegrationUrl) {
        this.appointment = appointment;
        this.requestType = requestType;
        this.sourceDetails = sourceDetails;
        this.externalIntegrationUrl = externalIntegrationUrl;
    }

    public Optional<HL7DatapointExternalIntegrationUrl> getExternalIntegrationUrl() {
        return externalIntegrationUrl;
    }

    public AppointmentDTO getAppointment() {
        return appointment;
    }

    public UploadedAppointmentRequestType getRequestType() {
        return requestType;
    }

    public SourceDetails getSourceDetails() {
        return sourceDetails;
    }
}
