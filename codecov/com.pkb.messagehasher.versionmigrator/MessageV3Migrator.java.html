package com.pkb.messagehasher.versionmigrator;

import com.pkb.messagehasher.Message;
import org.apache.commons.text.StringEscapeUtils;
import org.springframework.stereotype.Service;

import java.util.Map;

@Service
public class MessageV3Migrator {

    void migrate(Message message, Map<String, Object> decryptedFields) {
        if (message.getMigrationVersion() >= 3) {
            return;
        }

        String content = (String) decryptedFields.get("content");
        decryptedFields.put("content", StringEscapeUtils.unescapeHtml4(content));
    }
}
