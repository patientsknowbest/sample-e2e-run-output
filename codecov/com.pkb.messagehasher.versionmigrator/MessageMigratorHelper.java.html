package com.pkb.messagehasher.versionmigrator;

import com.pkb.messagehasher.Message;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.Map;

@Service
public class MessageMigratorHelper {

    public static final int MAX_MIGRATION_RESULT_VERSION = 6;

    @Autowired
    MessageV2Migrator v2Migrator;

    @Autowired
    MessageV3Migrator v3Migrator;

    public Map<String, Object> migrateMessageIfRequired(Message message, Map<String, Object> decryptedFields) {

        if (message.getMigrationVersion() > MAX_MIGRATION_RESULT_VERSION) {
            throw new IllegalStateException("Message with unsupported migration version " + message.getMigrationVersion());
        }

        v2Migrator.migrate(message, decryptedFields);
        v3Migrator.migrate(message, decryptedFields);
        //if you add another, update MAX_MIGRATION_RESULT_VERSION

        return decryptedFields;
    }
}
