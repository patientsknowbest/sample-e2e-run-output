package com.pkb.messagehasher.versionmigrator;

import com.pkb.common.datamigrator.Html4XmlUnescaper;
import com.pkb.messagehasher.Message;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.Map;

@Service
public class MessageV2Migrator {

    @Autowired
    private final Html4XmlUnescaper migrator;

    public MessageV2Migrator(Html4XmlUnescaper migrator) {
        this.migrator = migrator;
    }

    void migrate(Message message, Map<String, Object> decryptedFields) {
        if (message.getMigrationVersion() >= 2) {
            return;
        }

        String content = (String) decryptedFields.get("content");
        decryptedFields.put("content", doubleUnescape(content));
    }

    private String doubleUnescape(String s) {
        if (s == null) {
            return "";
        }

        return migrator.unescape(s);
    }
}
