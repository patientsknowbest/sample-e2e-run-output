package com.pkb.api.fhir.practitioner;

import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import com.pkb.api.fhir.AbstractResourceService;
import com.pkb.api.fhir.exception.ServerResponseExceptionMapper;
import com.pkb.api.fhir.person.PractitionerMapper;
import com.pkb.authentication.AuthenticatedIdentity;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authz.AuthorizationDataService;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.authz.data.AuthorizationInputs;
import com.pkb.datamodel.user.Person;
import com.pkb.datamodel.user.Professional;
import com.pkb.datamodel.PubliclyIdentifiable;
import com.pkb.domain.PractitionerDemographicsService;
import io.vavr.control.Either;
import org.hl7.fhir.dstu3.model.Practitioner;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.pkb.authz.ImmutableTargetMeta.targetMeta;
import static com.pkb.authz.action.Action.READ;
import static com.pkb.authz.context.Path.FHIR;
import static com.pkb.authz.data.ImmutableAuthorizationInputs.inputs;
import static io.vavr.API.None;
import static io.vavr.API.Set;

public class PractitionerResourceService extends AbstractResourceService<Practitioner> {

    private final AuthorizationDataService authorizationDataService;
    private final PractitionerDemographicsService practitionerDemographicsService;
    private final PractitionerMapper practitionerMapper;

    public PractitionerResourceService(
            @NotNull AuthorizationDataService authorizationDataService,
            @NotNull PractitionerDemographicsService practitionerDemographicsService,
            @NotNull PractitionerMapper practitionerMapper,
            @NotNull ServerResponseExceptionMapper exceptionMapper) {
        super(exceptionMapper);
        this.authorizationDataService = authorizationDataService;
        this.practitionerDemographicsService = practitionerDemographicsService;
        this.practitionerMapper = practitionerMapper;
    }

    public @NotNull Either<BaseServerResponseException, Practitioner> findPractitionerDemographics(AuthenticatedIdentity authenticatedIdentity, 
                                                                                                   UUID practitionerId) {
        // System user can see a wider range of usertypes
        Class<? extends PubliclyIdentifiable<UUID>> targetClass = authenticatedIdentity.authenticatedUser() instanceof AuthenticatedSystemUser ? Person.class : Professional.class;

        // Actor can see full demographics for himself, otherwise title/firstName/lastName only
        boolean fullDemographics = practitionerId.equals(authenticatedIdentity.authenticatedUser().getPublicId());

        AuthorizationInputs authorizationInputs = inputs()
                .authenticatedIdentity(authenticatedIdentity)
                .action(READ)
                .addTargets(targetMeta(targetClass, Set(practitionerId), None(), true))
                .path(FHIR)
                .build();

        AuthorizationData authorizationData = authorizationDataService.fetch(authorizationInputs);
        return practitionerDemographicsService.findPractitionerDemographics(authorizationData, practitionerId)
                .map(d -> practitionerMapper.toPractitioner(d,fullDemographics))
                .mapLeft(exception -> exceptionToFhir(exception, Practitioner.class));
    }
}
