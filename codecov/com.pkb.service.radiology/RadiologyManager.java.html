package com.pkb.service.radiology;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.common.base.Error;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.dataupload.entity.UploadedData;
import com.pkb.document.entity.Attachment;
import com.pkb.exception.PKBException;
import com.pkb.filesanitizer.FileSanitizer;
import com.pkb.radiology.entity.RadiologyImage;
import com.pkb.radiology.entity.RadiologyResult;
import com.pkb.radiology.entity.RadiologyResult.MediaActCategory;
import com.pkb.service.SynchableManager;
import com.pkb.service.dataupload.processor.UploadedDataService;
import com.pkb.service.document.AttachmentManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.ImageUtil;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import io.vavr.control.Either;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.IOException;
import java.util.Collection;
import java.util.Collections;
import java.util.List;

import static com.pkb.service.dataupload.processor.ImmutableLoggedInUploadedDataProcessingContext.loggedInUploadedDataProcessingContext;

public class RadiologyManager extends TransactionManager implements SynchableManager {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final UserManager userManager;
    private final FileSanitizer fileSanitizer;
    private final UploadedDataService uploadedDataService;
    private final RadiologyPersistManager radiologyPersistManager;
    private final AttachmentManager attachmentManager;

    public RadiologyManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, UserManager userManager,
                            FileSanitizer fileSanitizer, UploadedDataService uploadedDataService, RadiologyPersistManager radiologyPersistManager,
                            AttachmentManager attachmentManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.userManager = userManager;
        this.fileSanitizer = fileSanitizer;
        this.uploadedDataService = uploadedDataService;
        this.radiologyPersistManager = radiologyPersistManager;
        this.attachmentManager = attachmentManager;
    }

    public @NotNull Either<Error, Long> saveRadiologyResult(EHRRequestContext context, long patientId, Long accountId, RadiologyResult radiologyResult) {
        try {
            long startMs = System.currentTimeMillis();

            @NotNull
            Either<Error, Long> id = radiologyPersistManager.saveRadiologyResult(patientId, accountId, radiologyResult, context);

            LOGGER.info("saveRadiologyResult() save to tolven: {} ms", (System.currentTimeMillis() - startMs));

            return id;
        } catch (Exception exception) {
            throw new PKBException("Error while saving radiologyResult",
                    exception);
        }
    }

    /**
     * Includes deletion requests
     */
    public void saveRadiologyResults(EHRRequestContext context, long patientId, Long accountId,
                                     List<RadiologyResult> resultList) {
        try {
            long startMs = System.currentTimeMillis();

            sniffAndChangeMediaType(resultList);

            LOGGER.info("saveRadiologyResults() attachment prep: {} ms", (System.currentTimeMillis() - startMs));
            startMs = System.currentTimeMillis();

            radiologyPersistManager.saveRadiologyResults(patientId, accountId, resultList, context);

            LOGGER.info("saveRadiologyResults() save to tolven: {} ms", (System.currentTimeMillis() - startMs));
        } catch (Exception exception) {
            throw new PKBException("Error while saving radiologyResults", exception);
        }
    }

    public List<RadiologyResult> getRadiologyResults(@NotNull LoggedInEHRRequestContext context, MediaActCategory category,
                                                     Long accountId, int pageSize, int offset, Collection<Long> ids) {
        try {
            synchUploadedData(context, Collections.singletonList(userManager.getAccountOwner(accountId)));
            return radiologyPersistManager.getRadiologyResults(category, accountId,
                    pageSize, offset, context, ids);
        } catch (Exception exception) {
            throw new PKBException("Error while getting radiologyResults",
                    exception);
        }
    }

    /**
     * Get a radiologyImage (including full image bytes)
     *
     * @param accountId
     * @param radiologyImageId
     * @return radiologyImage
     */
    public RadiologyImage getRadiologyImage(@NotNull LoggedInEHRRequestContext context, Long accountId,
                                            Long radiologyImageId) {
        try {
            return radiologyPersistManager.getRadiologyImage(
                    radiologyImageId, context);
        } catch (Exception exception) {
            throw new PKBException("Error while getting radiologyImage-"
                    + radiologyImageId, exception);
        }
    }

    /**
     * Fetch the contents of an attachment
     *
     * @return Attachment with content field filled in
     */
    public Attachment getAttachmentContent(@NotNull LoggedInEHRRequestContext context, Long attachmentId) {
        try {
            return attachmentManager.getAttachmentContent(context, attachmentId);
        } catch (Exception exception) {
            throw new IllegalStateException("Exception while getting contents for attachment-" + attachmentId, exception);
        }
    }

    @Nullable
    public RadiologyResult getRadiologyResult(Long radiologyResultId, @NotNull LoggedInEHRRequestContext context) {
        return radiologyPersistManager.getRadiologyResult(radiologyResultId, context);
    }

    public boolean deleteRadiologyImage(Long radiologyImageId) {
        return radiologyPersistManager.deleteRadiologyImage(radiologyImageId);
    }

    private void synchUploadedData(@NotNull LoggedInEHRRequestContext requestContext, Collection<PKBPerson> patients) {
        patients.forEach(patient -> uploadedDataService.synchUploadedData(
                loggedInUploadedDataProcessingContext()
                        .requestContext(requestContext.withPiggyback(true))
                        .patientId(patient.getId())
                        .destination(UploadedData.Destination.RADIOLOGY)
                        .build()));
    }

    private void sniffAndChangeMediaType(List<RadiologyResult> results) throws IOException {
        // fix up attachment media type if possible (content sniffing)
        for (RadiologyResult result : results) {
            List<Attachment> attList = result.getAttachments();
            if (attList != null) {
                for (Attachment att : attList) {
                    String detectedContentType = ImageUtil.getContentType(att.getContent());
                    if (StringUtils.isNotBlank(detectedContentType)) {
                        att.setMediaType(fileSanitizer.fixMediaType(att.getContent(), detectedContentType));
                    }
                }
            }
        }

    }

    @Override
    public void syncUploadedData(LoggedInEHRRequestContext requestContext, Collection<Long> targetPersonIds) {
        synchUploadedData(requestContext, userManager.getPKBPersonList(targetPersonIds));
    }
}
