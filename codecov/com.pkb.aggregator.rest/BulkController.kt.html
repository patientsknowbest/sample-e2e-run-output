package com.pkb.aggregator.rest

import com.pkb.aggregator.camel.routing.bulk.RequestReloadMedicationRouteBuilder.Companion.REQUEST_MEDICATION_RELOAD
import com.pkb.aggregator.camel.routing.bulk.RequestReloadOrganizationRouteBuilder.Companion.REQUEST_ORGANIZATION_RELOAD
import com.pkb.aggregator.camel.routing.bulk.RequestReloadPatientRecordRouteBuilder.Companion.REQUEST_PATIENT_RECORD_RELOAD
import com.pkb.aggregator.camel.routing.dto.UpstreamChangeNotification
import com.pkb.aggregator.camel.routing.resource.ResourceUpdateRouteBuilder.Companion.INCOMING_UPDATE_NOTIFICATION
import com.pkb.aggregator.service.dto.ReloadRequest
import io.micrometer.core.annotation.Timed
import org.apache.camel.ProducerTemplate
import org.springframework.http.HttpStatus
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.RequestBody
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.ResponseStatus
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("bulk")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class BulkController(private val producerTemplate: ProducerTemplate) {

    @PostMapping("patient")
    @ResponseStatus(HttpStatus.OK)
    fun bulkReloadPatient(@RequestBody request: ReloadRequest) =
        producerTemplate.sendBody(REQUEST_PATIENT_RECORD_RELOAD, request)

    @PostMapping("organization")
    @ResponseStatus(HttpStatus.OK)
    fun bulkReloadOrganization(@RequestBody request: ReloadRequest) =
        producerTemplate.sendBody(REQUEST_ORGANIZATION_RELOAD, request)

    @PostMapping("medication")
    @ResponseStatus(HttpStatus.OK)
    fun bulkReloadMedication(@RequestBody request: ReloadRequest) =
        producerTemplate.sendBody(REQUEST_MEDICATION_RELOAD, request)

    @PostMapping("one")
    @ResponseStatus(HttpStatus.OK)
    fun reloadOneResource(@RequestBody request: UpstreamChangeNotification) =
        producerTemplate.sendBody(INCOMING_UPDATE_NOTIFICATION, request)
}
