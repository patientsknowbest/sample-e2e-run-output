package com.pkb.aggregator.rest

import com.pkb.aggregator.persistence.entity.PatientIdentifierMapEntry
import com.pkb.aggregator.persistence.entity.embeddable.MapEntryKey
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamIdentifier
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.persistence.repository.PatientIdentifierMapRepository
import io.micrometer.core.annotation.Timed
import org.springframework.http.HttpStatus
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.RequestBody
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.ResponseStatus
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("patient")
@Timed(histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
class PatientController(private val patientIdentifierMapRepository: PatientIdentifierMapRepository) {

    data class NewPatientIdentifier(
        val system: String,
        val value: String,
        val upstreamSource: String,
        val upstreamResourceId: String,
        val downstreamIdentifier: String,
    )

    @PostMapping("identifier")
    @ResponseStatus(HttpStatus.OK)
    fun storeNewIdentifier(@RequestBody request: NewPatientIdentifier): PatientIdentifierMapEntry {
        val key = MapEntryKey(
            UpstreamResource(request.upstreamSource, request.upstreamResourceId),
            UpstreamIdentifier(upstreamIdentifierSystem = request.system, upstreamIdentifierValue = request.value)
        )
        return patientIdentifierMapRepository.save(PatientIdentifierMapEntry(key, request.downstreamIdentifier))
    }
}
