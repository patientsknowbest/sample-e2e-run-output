package com.pkb.aggregator.rest

import com.pkb.aggregator.integration.aidbox.AidboxRestClient
import com.pkb.common.ClearableInternalState
import org.slf4j.LoggerFactory
import org.springframework.context.annotation.Profile
import org.springframework.http.HttpStatus
import org.springframework.web.bind.annotation.DeleteMapping
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.ResponseStatus
import org.springframework.web.bind.annotation.RestController
import java.lang.invoke.MethodHandles

@RestController
@RequestMapping("test")
@Profile("test", "dev", "e2e", "integration-test")
class TestController(
    private val aidboxRestClient: AidboxRestClient,
    private val internalState: List<ClearableInternalState>
) {
    private val log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @DeleteMapping("downstream/patient/all")
    @ResponseStatus(HttpStatus.OK)
    fun clearDownstream() {
        listOf(
            "allergyintolerance",
            "appointment",
            "codesystem",
            "condition",
            "consent",
            "encounter",
            "medication",
            "medicationstatement",
            "observation",
            "organization",
            "patient",
            "person",
            "practitioner",
            "practitionerrole",
            "provenance",
            "valueset",
        ).onEach {
            aidboxRestClient.postSql(arrayOf("delete from $it")).execute()
            aidboxRestClient.postSql(arrayOf("delete from ${it}_history")).execute()
        }
    }

    @DeleteMapping("identifiers/all")
    @ResponseStatus(HttpStatus.OK)
    fun clearAllIdentifiers() {
        log.info("Clearing state from {} beans", internalState.size)
        internalState.forEach {
            it.clearState()
        }
    }
}
