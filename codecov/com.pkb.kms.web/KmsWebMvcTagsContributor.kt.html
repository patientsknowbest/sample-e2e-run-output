package com.pkb.kms.web

import com.pkb.kms.shared.KMS_CLIENT_ID_DEFAULT
import com.pkb.kms.shared.KMS_CLIENT_ID_HEADER
import io.micrometer.core.instrument.Tag
import org.springframework.boot.actuate.metrics.web.servlet.WebMvcTagsContributor
import javax.servlet.http.HttpServletRequest
import javax.servlet.http.HttpServletResponse

/**
 * Adds some extra metric tags to the default ones captured by spring for HTTP requests.
 */
class KmsWebMvcTagsContributor: WebMvcTagsContributor {
    override fun getTags(request: HttpServletRequest?, response: HttpServletResponse?, handler: Any?, exception: Throwable?): Iterable<Tag> =
            listOf(clientTag(request), paramsTag(request))

    override fun getLongRequestTags(request: HttpServletRequest?, handler: Any?): Iterable<Tag> =
            listOf(clientTag(request), paramsTag(request))

    private fun paramsTag(request: HttpServletRequest?) =
            Tag.of("parameters", request?.parameterNames?.toList()?.joinToString(",") ?: "none")

    private fun clientTag(request: HttpServletRequest?) =
            Tag.of("client", request?.getHeader(KMS_CLIENT_ID_HEADER) ?: KMS_CLIENT_ID_DEFAULT)
}
