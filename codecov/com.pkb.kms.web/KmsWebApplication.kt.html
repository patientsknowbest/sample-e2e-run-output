package com.pkb.kms.web

import org.springframework.boot.autoconfigure.SpringBootApplication
import org.springframework.boot.autoconfigure.domain.EntityScan
import org.springframework.boot.runApplication
import org.springframework.cache.annotation.EnableCaching
import org.springframework.context.annotation.ComponentScan
import org.springframework.data.jpa.repository.config.EnableJpaRepositories

@SpringBootApplication
@EntityScan(value = ["com.pkb.entities", "com.pkb.kms.service.entity"])
@EnableJpaRepositories(basePackages = ["com.pkb.kms.service.repository"])
@ComponentScan(value = ["com.pkb.kms"])
@EnableCaching
class KmsWebApplication

fun main(args: Array<String>) {
    runApplication<KmsWebApplication>(*args)
}
