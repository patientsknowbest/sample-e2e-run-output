package com.pkb.kms.service.entity

import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer
import org.hibernate.annotations.GenericGenerator
import org.hibernate.annotations.Immutable
import org.hibernate.annotations.Parameter
import java.time.Instant
import java.util.UUID
import javax.persistence.*

@Entity
@Immutable
@Table(schema = "kms", name = "nhs_login_credentials")
data class KmsNhsLoginCredentials(
        @Id
        @GenericGenerator(name = "nhs_login_credentials_id_seq_gen",
                strategy = "enhanced-sequence",
                parameters = [
                    Parameter(name = "optimizer", value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    Parameter(name = "sequence_name", value = "kms.nhs_login_credentials_id_seq"),
                    Parameter(name = "increment_size", value = "50")
                ])
        @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "nhs_login_credentials_id_seq_gen")
        var id: Long?,

        @Column(name = "public_id")
        val publicId: UUID,

        @Column(name = "person_public_id")
        val personPublicId: UUID,

        @Column(name = "encrypted_user_private_key")
        val encryptedUserPrivateKey: ByteArray,

        @Column(name = "private_key_algorithm")
        val privateKeyAlgorithm: String,

        @Column(name = "secret_key_algorithm")
        val secretKeyAlgorithm: String,

        @Column(name = "nonce")
        val nonce: ByteArray,

        @Column(name = "credentials_sha256")
        var credentialsHash: ByteArray? = null,

        @Column(name = "created")
        val created: Instant,
) {

    // equals/hashcode for entity based on https://vladmihalcea.com/the-best-way-to-implement-equals-hashcode-and-tostring-with-jpa-and-hibernate/

    override fun equals(other: Any?): Boolean {
        if (this === other) return true
        if (javaClass != other?.javaClass) return false

        other as KmsNhsLoginCredentials

        return id != null && id == other.id
    }

    override fun hashCode() = 31
}
