package com.pkb.kms.service.entity

import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer
import org.hibernate.annotations.GenericGenerator
import org.hibernate.annotations.Parameter
import java.time.Instant
import java.util.UUID
import javax.persistence.Column
import javax.persistence.Entity
import javax.persistence.FetchType
import javax.persistence.GeneratedValue
import javax.persistence.GenerationType
import javax.persistence.Id
import javax.persistence.JoinColumn
import javax.persistence.ManyToOne
import javax.persistence.Table
import javax.persistence.Version

@Entity
@Table(schema = "kms", name = "external_password_recovery_current")
data class KmsExternalPasswordRecoveryCurrent(
        @Id
        @GenericGenerator(name = "external_password_recovery_current_id_seq_gen",
                strategy = "enhanced-sequence",
                parameters = [
                    Parameter(name = "optimizer", value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    Parameter(name = "sequence_name", value = "kms.external_password_recovery_current_id_seq"),
                    Parameter(name = "increment_size", value = "50")
                ])
        @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "external_password_recovery_current_id_seq_gen")
        var id: Long? = null,

        @Column(name = "person_public_id")
        val personPublicId: UUID,

        @ManyToOne(fetch = FetchType.LAZY)
        @JoinColumn(name = "external_password_recovery_id")
        var externalPasswordRecovery: KmsExternalPasswordRecovery,

        @Column(name = "updated")
        var updated: Instant,

        @Version
        @Column(name = "version")
        val version: Int = 1
) {
    // equals/hashcode for entity based on https://vladmihalcea.com/the-best-way-to-implement-equals-hashcode-and-tostring-with-jpa-and-hibernate/

    override fun equals(other: Any?): Boolean {
        if (this === other) return true
        if (javaClass != other?.javaClass) return false

        other as KmsExternalPasswordRecoveryCurrent

        return id != null && id == other.id
    }

    override fun hashCode() = javaClass.hashCode()
}
