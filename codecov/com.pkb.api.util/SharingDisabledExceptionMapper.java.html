package com.pkb.api.util;

import com.pkb.api.error.RestError;
import com.pkb.api.util.v2.exception.SharingDisabledException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import javax.ws.rs.ext.ExceptionMapper;
import javax.ws.rs.ext.Provider;

@Provider
public class SharingDisabledExceptionMapper implements ExceptionMapper<SharingDisabledException> {

    private final static Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    public static final String SHARING_DISABLED_GENERAL_MESSAGE = "Sharing disabled";

    @Override
    public Response toResponse(SharingDisabledException exception) {

        RestError restError = new RestError();

        restError.setStatusCode(Response.Status.FORBIDDEN.getStatusCode());
        restError.setDeveloperMessage(SHARING_DISABLED_GENERAL_MESSAGE);
        restError.setErrorMessage(SHARING_DISABLED_GENERAL_MESSAGE);

        return Response.status(Response.Status.FORBIDDEN).entity(restError).type(MediaType.APPLICATION_JSON).build();
    }
}
