package com.pkb.api.util;

import com.pkb.common.base.Error;
import com.pkb.exception.ValidationException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;
import javax.ws.rs.ext.ExceptionMapper;
import javax.ws.rs.ext.Provider;
import java.util.List;

@Provider
public class ValidationExceptionMapper implements ExceptionMapper<ValidationException> {

    private final static Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    public static final String VALIDATION_GENERAL_MESSAGE = "Unable to process request due to multiple errors";

    @Override
    public Response toResponse(ValidationException exception) {

        ValidationRestError validationError = new ValidationRestError();

        List<Error> errors = exception.getErrors();

        // to match existing exception mappers and rest errors 'errorMessage' field is populated  
        if (errors.size() == 1) {
            validationError.setErrorMessage(errors.get(0).getDescription());
        } else {
            validationError.setErrorMessage(VALIDATION_GENERAL_MESSAGE);
        }

        for (Error e : errors) {
            validationError.add(e);
        }

        validationError.setStatusCode(Response.Status.BAD_REQUEST.getStatusCode());

        return Response.status(Response.Status.BAD_REQUEST).entity(validationError).type(MediaType.APPLICATION_JSON).build();
    }


}
