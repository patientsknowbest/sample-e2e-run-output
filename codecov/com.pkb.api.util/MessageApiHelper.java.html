package com.pkb.api.util;

import com.pkb.api.v1.model.ApiAttachment;
import com.pkb.api.v1.model.ApiConversation;
import com.pkb.api.v1.model.ApiMessage;
import com.pkb.api.v1.model.ApiMessageUser;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.bean.IMessageWebBean;
import com.pkb.document.entity.Attachment;
import com.pkb.encounter.entity.ApiConversationDTO;
import com.pkb.encounter.entity.Message;
import com.pkb.entities.enums.messaging.MessageStatus;
import com.pkb.service.encounter.impl.tolven.EncounterManager;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.time.Instant;
import java.util.ArrayList;
import java.util.Date;
import java.util.List;
import java.util.Optional;
import java.util.function.BiFunction;

/**
 * A helper class to centralise the logic of API calls to the message service.
 *
 *
 * @author russell
 *
 */
public class MessageApiHelper {

    private final EncounterManager encounterManager;
    private final IMessageWebBean messageWebBean;

    public MessageApiHelper(EncounterManager encounterManager, IMessageWebBean messageWebBean) {
        this.encounterManager = encounterManager;
        this.messageWebBean = messageWebBean;
    }

    public List<ApiConversation> getConversations(@NotNull LoggedInEHRRequestContext requestContext, long userId, Instant fromDateOrNull,
            Instant toDateOrNull, int limit, boolean orderLatestFirst, @Nullable BiFunction<String, String[], String> i18nMethod) {

        long accountId = Optional.ofNullable(messageWebBean.getDefaultAccountId(userId))
                .orElseThrow(() -> new IllegalStateException("Unable to get default account id."));
        List<ApiConversation> conversationResults = new ArrayList<>();
        List<ApiConversationDTO> conversations = encounterManager.getConversationListForApi(
                requestContext, userId, accountId, fromDateOrNull, toDateOrNull, limit, 0, orderLatestFirst, i18nMethod);

        for (ApiConversationDTO conversation : conversations) {
            for (Message message : conversation.getMessages()) {
                messageWebBean.expandInlineContents(requestContext, userId, accountId, message, true);
            }

            ApiConversation api = convert(conversation, requestContext);

            conversationResults.add(api);

        }

        return conversationResults;
    }

    private ApiConversation convert(ApiConversationDTO conversation, EHRRequestContext requestContext) {

        ApiConversation api = new ApiConversation();
        api.setMessages(new ArrayList<>());

        api.setId(conversation.getId().toString());
        api.setParticipants(new ArrayList<>());
        for (PKBPerson person : conversation.getParticipants()) {
            api.getParticipants().add(convert(person));
        }
        //skip draft messages
        conversation.getMessages().stream().filter(message -> message.getMessageStatus() != MessageStatus.INDRAFT)
                .forEach(message -> api.getMessages().add(convert(message, requestContext)));
        api.setLatestMessageId(conversation.getLatestMessageId());
        api.setMessageCount(conversation.getMessageCount());

        return api;
    }

    private ApiMessage convert(Message message, EHRRequestContext requestContext) {
        Instant messageTimestamp = message.getMessageTimestamp();

        ApiMessage api = new ApiMessage();
        api.setAttachmentCount(message.getAttachmentCount());
        api.setContent(message.getContent());
        api.setConversationId(message.getEncounterUniqueId().toString());
        api.setId("" + message.getId());
        api.setOnBehalfOfId(message.getOnBehalfAuthorId());
        api.setSenderId(message.getSenderId());
        api.setSubject(message.getSubject());
        api.setTimestamp(messageTimestamp == null ? null : Date.from(messageTimestamp));
        api.setRecipientIds(new ArrayList<>());
        api.setHtmlContent(message.isHtmlAllowed());
        api.setPatientId(message.getPatientId());
        api.setMessageStatus(message.getMessageStatus().toString());
        api.setRecipientIds(new ArrayList<>());
        api.getRecipientIds().addAll(message.getParticipants());

        api.setAttachments(convertAll(message.getAttachments()));

        return api;
    }

    private List<ApiAttachment> convertAll(List<Attachment> attachments) {
        if (attachments == null) {
            return null;
        }
        List<ApiAttachment> results = new ArrayList<>();
        for (Attachment attachment : attachments) {
            ApiAttachment api = new ApiAttachment();
            api.setId("" + attachment.getId());
            api.setDescription(attachment.getFilename());
            api.setName(attachment.getFilename());
            api.setMediaType(attachment.getMediaType());
            api.setUploadTime(attachment.getUploadTime());
            //todo: add URL
            results.add(api);

        }
        return results;
    }

    private ApiMessageUser convert(PKBPerson person) {

        ApiMessageUser api = new ApiMessageUser();
        api.setId(person.getIdString());
        api.setFirstName(person.getFirstName());
        api.setLastName(person.getLastName());

        if (person.isPro()) {
            api.setJobTitle(person.getJobTitle());
            api.setTeamName(person.getOrganizationUnit());

        }
        return api;

    }
}
