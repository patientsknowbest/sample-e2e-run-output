package com.pkb.api.util;

import com.pkb.api.error.RestError;
import com.pkb.common.base.Error;

import java.util.ArrayList;
import java.util.Collections;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

public class ValidationRestError extends RestError {

    private List<ErrorDetails> errorDetails;
    private Map<String, ErrorDetails> errorCodes;

    public ValidationRestError() {
        errorDetails = new ArrayList<>();
        errorCodes = new HashMap<>();
    }

    private ErrorDetails get(String errorCode) {

        ErrorDetails ed = errorCodes.get(errorCode);
        if (ed != null) {
            return ed;
        }
        ed = new ErrorDetails();
        ed.errorCode = errorCode;
        errorCodes.put(errorCode, ed);
        errorDetails.add(ed);
        return ed;
    }

    ValidationRestError add(Error error) {
        ErrorDetails ed = get(error.getErrorCode());
        ed.add(error.getDescription());
        return this;
    }

    public List<ErrorDetails> getErrorDetails() {
        return errorDetails;
    }

    public static class ErrorDetails {
        private String errorCode;

        private List<String> messages;

        public ErrorDetails() {
            messages = new ArrayList<>();
        }

        public ErrorDetails(String errorCode, String message) {
            this();
            this.errorCode = errorCode;
            add(message);
        }

        ErrorDetails add(String message) {
            messages.add(message);
            return this;
        }

        public List<String> getMessages() {
            return Collections.unmodifiableList(messages);
        }

        public String getErrorCode() {
            return errorCode;
        }
    }
}
