package com.pkb.api.util;

import com.pkb.api.v1.model.ApiSymptom;
import com.pkb.api.v1.model.ApiSymptomReport;
import com.pkb.api.v1.model.ApiTeamSymptoms;
import com.pkb.app.dto.DateFilterDTO;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.institute.entity.Team;
import com.pkb.service.symptom.SymptomReportManager;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.symptom.entity.Symptom;
import com.pkb.symptom.entity.SymptomReportDTO;
import com.pkb.symptom.entity.SymptomReportHistoryDTO;
import com.pkb.symptom.entity.SymptomSeverity;
import com.pkb.util.DateUtil;
import com.pkb.util.LocalizedTextProvider;
import org.jetbrains.annotations.NotNull;

import java.sql.Date;
import java.time.Instant;
import java.util.ArrayList;
import java.util.List;

public class SymptomApiHelper extends ApiHelper {

    private static final long serialVersionUID = 1L;

    private final SymptomReportManager symptomReportManager;
    private final TeamUserManager teamUserManager;
    private final UserManager userManager;
    private final DateUtil dateUtil;

    public SymptomApiHelper(SymptomReportManager symptomReportManager, TeamUserManager teamUserManager, DateUtil dateUtil, LocalizedTextProvider localizedTextProvider, UserManager userManager) {
        super(localizedTextProvider);
        this.symptomReportManager = symptomReportManager;
        this.teamUserManager = teamUserManager;
        this.dateUtil = dateUtil;
        this.userManager = userManager;
    }

    public List<ApiSymptomReport> getSymptomReports(@NotNull LoggedInEHRRequestContext requestContext, long userId, Instant fromDateOrNull, Instant toDateOrNull, int limit) {
        List<SymptomReportHistoryDTO> histDtoList = symptomReportManager.getSymptomReportHistoryList(requestContext, userId, new DateFilterDTO(fromDateOrNull, toDateOrNull), limit);
        return convertToApiSymptomReportsList(histDtoList);
    }

    public List<ApiSymptomReport> getSymptomReports(@NotNull LoggedInEHRRequestContext requestContext, List<Long> symptomIds,
                                                    long userId,
                                                    Instant fromDateOrNull) {

        Long patientAccountId = userManager.getDefaultAccountId(userId);
        List<SymptomReportHistoryDTO> histDtoList = symptomReportManager.getSymptomReportHistoryList(requestContext, symptomIds, userId, patientAccountId, fromDateOrNull);

        return convertToApiSymptomReportsList(histDtoList);

    }

    private List<ApiSymptomReport> convertToApiSymptomReportsList(List<SymptomReportHistoryDTO> histDtoList) {
        List<ApiSymptomReport> resultList = new ArrayList<>();

        for (SymptomReportHistoryDTO histDto : histDtoList) {
            for (SymptomReportDTO dto : histDto.getSymptomReportList()) {
                resultList.add(convert(dto));
            }
        }

        return resultList;
    }

    public List<ApiTeamSymptoms> getMonitoredSymptomsByTeam(EHRRequestContext requestContext, long patientId) {
        List<Team> teams = teamUserManager.getUserTeams(requestContext, patientId, true);

        List<ApiTeamSymptoms> results = new ArrayList<>();
        if (!teams.isEmpty()) {
            for (Team nextTeam : teams) {
                List<Symptom> forTeam = symptomReportManager
                        .getSymptomsMonitoredByTeam(requestContext, nextTeam.getCode());
                if ((forTeam != null) && !forTeam.isEmpty()) {
                    ApiTeamSymptoms teamSymptoms = new ApiTeamSymptoms();
                    teamSymptoms.setTeamName(nextTeam.getName());
                    for (Symptom symptom : forTeam) {
                        teamSymptoms.getSymptoms().add(convert(symptom));
                    }
                    results.add(teamSymptoms);

                }
            }

        }
        return results;
    }

    public List<ApiSymptom> getAllSymptoms() {
        //List<Team> teams = teamUserManager.getUserTeams(patientId, true);

        List<ApiSymptom> results = new ArrayList<>();
        for (Symptom next : Symptom.values()) {
            results.add(convert(next));
        }
        return results;
    }

    private ApiSymptom convert(Symptom symptom) {
        ApiSymptom api = new ApiSymptom();
        api.setApiRef(symptom.getApiRef());
        api.setName(getText(symptom.getName()));
        api.setSeverity0Text(symptom.getSeverityText(0));
        api.setSeverity1Text(symptom.getSeverityText(1));
        api.setSeverity2Text(symptom.getSeverityText(2));
        api.setSeverity3Text(symptom.getSeverityText(3));

        return api;
    }

    public Long createSymptomReport(@NotNull LoggedInEHRRequestContext requestContext, ApiSymptomReport symptomReport, long userId) {
        SymptomReportDTO dto = new SymptomReportDTO(new SourceDetails(requestContext));
        Symptom symptom = Symptom.getByApiRef(symptomReport.getSymptomApiRef());

        dto.setSymptomPrivateId(symptom.getId());
        dto.setReportDate(dateUtil.parseTimestampIso8601(symptomReport.getReportDateISO()).orElse(symptomReport.getReportDate().toInstant()));
        dto.setSeverity(SymptomSeverity.getBy0to3(symptomReport.getSeverity0to3()));
        dto.getBaseFields().setPrivacyFlag(PrivacyFlag.GENERAL);

        return symptomReportManager.saveSymptomReport(requestContext, dto, userId);
    }

    private ApiSymptomReport convert(SymptomReportDTO dto) {
        if (dto == null) {
            return null;
        }

        ApiSymptomReport m = new ApiSymptomReport();
        m.setId("" + dto.getId());
        m.setReportDate(Date.from(dto.getReportDate()));

        m.setReportDateISO(dateUtil.formatTimestampIso8601(dto.getReportDate()));
        m.setSymptomApiRef(Symptom.getById(dto.getSymptomPrivateId()).getApiRef());
        m.setSeverity0to3(dto.getSeverity().getNumber0to3());

        return m;
    }

}
