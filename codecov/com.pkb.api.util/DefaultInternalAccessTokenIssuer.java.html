package com.pkb.api.util;

import com.pkb.api.util.OAuthAuthorizationServerClient.OAuthAccessToken;
import com.pkb.common.config.PhrConfig;
import com.pkb.datamodel.entity.mapper.UserTypeToScopeMapper;
import com.pkb.entities.enums.UserType;
import com.pkb.entities.enums.api.ApiClientScope;
import org.apache.commons.lang3.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import retrofit2.Call;
import retrofit2.Response;

import javax.ws.rs.InternalServerErrorException;
import javax.ws.rs.WebApplicationException;
import java.util.Set;

import static java.util.Collections.unmodifiableSet;
import static java.util.EnumSet.of;

public class DefaultInternalAccessTokenIssuer implements InternalAccessTokenIssuer {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private static final String GRANT_TYPE = "password";

    //Currently we support only following user types
    private static final Set<UserType> SUPPORTED_USER_TYPES = unmodifiableSet(
            of(UserType.PATIENT, UserType.INSTITUTE_ADMIN, UserType.REG_CLINICIAN));

    private UserTypeToScopeMapper scopeMapper;

    private OAuthAuthorizationServerClient client;

    private PhrConfig config;

    public DefaultInternalAccessTokenIssuer(UserTypeToScopeMapper scopeMapper, OAuthAuthorizationServerClient client, PhrConfig config) {
        this.client = client;
        this.scopeMapper = scopeMapper;
        this.config = config;
    }

    private ApiClientScope getScope(UserType type) {
        if (!SUPPORTED_USER_TYPES.contains(type)) {
            throw new UnsupportedOperationException("Given user type is not supported in session id exhange flow");
        }
        return scopeMapper.getScopeForUserType(type);
    }

    private Response<OAuthAccessToken> callAuthentication(String username, String password, UserType type) {
        String clientId = config.getRestApiClientId();
        if (StringUtils.isBlank(clientId)) {
            throw new IllegalStateException("Missing client id");
        }
        try {
            Call<OAuthAccessToken> call = client.authorize(GRANT_TYPE, username, password, clientId, getScope(type).name());
            return call.execute();
        } catch (Exception e) {
            throw new InternalServerErrorException(e);
        }
    }

    @Override
    public InternalAccessToken authenticate(String username, String password, UserType type) {

        Response<OAuthAccessToken> response = callAuthentication(username, password, type);

        if (response.isSuccessful()) {
            OAuthAccessToken token = response.body();
            return new InternalAccessToken(token.getAccessToken(), token.getExpiresIn());
        } else {
            throw new WebApplicationException(response.code());
        }

    }

}
