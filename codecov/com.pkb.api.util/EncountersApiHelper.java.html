package com.pkb.api.util;

import com.pkb.api.error.ApiAuthError;
import com.pkb.api.util.converter.v2.EncounterConverter;
import com.pkb.api.v2.model.ApiEncounter;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.encounter.entity.Encounter;
import com.pkb.entities.enums.EncounterClass;
import com.pkb.service.encounter.impl.tolven.EncounterManager;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.time.Instant;
import java.util.List;
import java.util.Set;
import java.util.function.BiFunction;

import static java.util.Collections.emptyList;
import static java.util.stream.Collectors.toList;
import static org.apache.commons.collections4.CollectionUtils.isNotEmpty;

@Component
public class EncountersApiHelper {

    private PatientConsentManager patientConsentManager;
    private EncounterManager encounterManager;
    private EncounterConverter encounterConverter;

    @Autowired
    public void setEncounterConverter(EncounterConverter encounterConverter) {
        this.encounterConverter = encounterConverter;
    }

    @Autowired
    public void setEncounterManager(EncounterManager encounterManager) {
        this.encounterManager = encounterManager;
    }


    @Autowired
    public void setPatientConsentManager(PatientConsentManager patientConsentManager) {
        this.patientConsentManager = patientConsentManager;
    }


    public List<ApiEncounter> getEncounters(@NotNull LoggedInEHRRequestContext requestContext, long patientId,
            List<String> reasonApiRefsOrNull, PKBPerson loggedInUser, Instant updatedSince,
            Set<EncounterClass> encounterClassFilters, @Nullable BiFunction<String, String[], String> i18nMethod) {

        if (!loggedInUser.getId().equals(patientId)) {
            if (!patientConsentManager.hasAccessToUser(requestContext, patientId)) {
                throw new ApiAuthError("current user " + loggedInUser.getId()
                        + " does not have access to patient " + patientId);
            }
        }
        List<Encounter> encounters;
        //removing support for filtering by encounterClass in api. Not currently required:
        encounters = encounterManager.getAllEncounters(requestContext, Long.toString(patientId),
                updatedSince, true/* fetchMessages*/, reasonApiRefsOrNull, encounterClassFilters,
                i18nMethod);

        List<ApiEncounter> result = emptyList();
        if (isNotEmpty(encounters)) {
            result = encounters.stream().map(encounterConverter::toApiModel).collect(toList());
        }
        return result;
    }
}
