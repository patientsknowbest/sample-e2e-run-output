package com.pkb.api.util;

import com.pkb.api.error.ApiAuthError;
import com.pkb.api.util.converter.v1.DiaryEntryToApiJournalEntryConverter;
import com.pkb.api.v1.model.ApiJournalEntry;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.diary.entity.DiaryEntry;
import com.pkb.service.diary.DiaryManager;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;

import java.util.ArrayList;

/**
 * Helper class to do the heavy lifting associated with {@code V1JournalService}.
 *
 * @author Tom
 */
public class JournalApiHelper {

    private final PatientConsentManager userAccessManager;

    private final UserManager userManager;

    private final DiaryManager diaryManager;

    private final DiaryEntryToApiJournalEntryConverter diaryEntryToApiJournalEntryConverter;

    public JournalApiHelper(PatientConsentManager patientConsentManager,
                            UserManager userManager, DiaryManager diaryManager, DiaryEntryToApiJournalEntryConverter diaryEntryToApiJournalEntryConverter) {
        this.userAccessManager = patientConsentManager;
        this.userManager = userManager;
        this.diaryManager = diaryManager;
        this.diaryEntryToApiJournalEntryConverter = diaryEntryToApiJournalEntryConverter;
    }

    /**
     * Add a journal entry for a patient.
     *
     * @param loggedInUser The currently logged in user.
     * @param patientId    The ID of the patient whose journal should be updated.
     * @param journalEntry The journal entry to add.
     * @return The ID of the newly created journal entry.
     */
    public Long addJournalEntryForPatient(LoggedInEHRRequestContext requestContext, PKBPerson loggedInUser,
                                          long patientId, ApiJournalEntry journalEntry) {
        if (!userAccessManager.hasAccessToUser(requestContext, patientId)) {
            throw new ApiAuthError("current user " + loggedInUser.getId()
                    + " does not have access to patient " + patientId);
        }

        Long accountId = userManager.getDefaultAccountId(patientId);
        // The author/editor will be set to the ID of the currently logged in
        // user. This may or may not be the same user for whom the journal
        // entry is being added.
        DiaryEntry entry = convertToDTO(requestContext, journalEntry);
        return diaryManager.saveDiaryEntry(requestContext, loggedInUser.getId(), accountId, entry, new ArrayList<>());
    }

    /**
     * Convenience method for converting an {@code ApiJournalEntry} into an
     * object to send to the diary manager.
     *
     * @param apiJournalEntry The journal entry to add.
     * @param authorId        The ID of the author.
     * @return The {@code DiaryEntry} object.
     */
    private DiaryEntry convertToDTO(EHRRequestContext requestContext, ApiJournalEntry apiJournalEntry) {
        DiaryEntry dto = new DiaryEntry(new SourceDetails(requestContext));
        // Content
        dto.setAttachmentList(new ArrayList<>());
        dto.setSubjectOfEntry(apiJournalEntry.getSubjectOfEntry());
        dto.setEntryText(apiJournalEntry.getEntryTextPlain());
        dto.setDiaryEntryTimestamp(apiJournalEntry.getDate());
        return dto;
    }
}
