package com.pkb.api.util;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.api.v1.model.ApiTest;
import com.pkb.api.v1.model.ApiTestResult;
import com.pkb.api.v1.model.V1ApiTestResult;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.service.test.TestManager;
import com.pkb.test.entity.LoincTest;
import com.pkb.test.entity.TestHistoryDTO;
import com.pkb.test.entity.TestResultDTO;
import com.pkb.test.entity.TestResultType;
import com.pkb.util.LocalizedTextProvider;
import org.apache.commons.lang3.StringUtils;

import java.util.Date;
import java.util.List;

import static com.pkb.test.IncludeDeletedResults.EXCLUDE_DELETED;
import static io.vavr.collection.List.ofAll;
import static java.util.Comparator.comparing;

public class TestsApiHelper extends ApiHelper {

    private static final long serialVersionUID = 1L;

    private final TestManager testManager;

    public TestsApiHelper(TestManager testManager, LocalizedTextProvider localizedTextProvider) {
        super(localizedTextProvider);
        this.testManager = testManager;
    }

    public List<ApiTest> getTestResults(LoggedInEHRRequestContext requestContext, List<Long> testIds, long patientId) {
        List<TestHistoryDTO> testHistoryList = testManager.getTestHistoryList(requestContext, patientId, testIds, null, EXCLUDE_DELETED);
        return getApiTestList(testHistoryList);
    }

    @VisibleForTesting
    List<ApiTest> getApiTestList(List<TestHistoryDTO> testHistoryList) {
        return ofAll(testHistoryList)
                .map(historyDTO -> {
                    ApiTest apiTest = convert(historyDTO.getTestResultType());
                    apiTest.setTestResults(ofAll(historyDTO.getTestResultList())
                            .map(this::convert)
                            .sorted(comparing(ApiTestResult::getId, StringUtils::compare))
                            .toJavaList());
                    return apiTest;
                })
                .sorted(comparing(ApiTest::getId, StringUtils::compare))
                .toJavaList();
    }

    private ApiTest convert(TestResultType type) {

        ApiTest converted = new ApiTest();
        if (type != null) {
            Long id = type.getId();
            if (id != null) {
                converted.setId(String.valueOf(id));
            }
            converted.setCodeSystem(type.getTestCodingSystem());
            converted.setCode(type.getTestCode());
            converted.setUnit(type.getUnit());
        }

        // add extra info if it's a LOINC and not just a mapping...
        if (type instanceof LoincTest) {
            LoincTest loinc = (LoincTest) type;

            converted.setName(getText(loinc.getNameKey()));
        }
        // need to set empty strings for missing fields -- be kind to apps
        else {
            converted.setName((type != null) ? type.getName() : "");
        }

        return converted;
    }

    ApiTestResult convert(TestResultDTO dto) {
        V1ApiTestResult converted = new V1ApiTestResult();

        converted.setId("" + dto.getId());
        converted.setLoincTestId((dto.getLoincTestId() == null) ? null : dto.getLoincTestId());
        converted.setRangeHigh(dto.getRangeHigh());
        converted.setRangeLow(dto.getRangeLow());
        converted.setResultDate(dto.getTestDate() == null ? null : Date.from(dto.getTestDate()));

        if (dto.getValue() != null) {
            String valueNumber = dto.getValue().toString();
            if (dto.getValueComparator() != null) {
                converted.setValue(dto.getValueComparator().toString() + valueNumber);
            } else {
                converted.setValue(valueNumber);
            }
        } else {
            converted.setValue(dto.getTextValue());
        }

        converted.setComments(dto.getComments());

        return converted;
    }
}
