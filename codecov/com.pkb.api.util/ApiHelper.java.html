package com.pkb.api.util;

import com.pkb.util.LocalizedTextProvider;

public class ApiHelper {

    private final LocalizedTextProvider localizedTextProvider;

    public ApiHelper(LocalizedTextProvider localizedTextProvider) {
        this.localizedTextProvider = localizedTextProvider;
    }

    protected String getText(String name) {
        return localizedTextProvider.getText(name);
    }
}
