package com.pkb.api.util;

import com.pkb.api.error.ApiValidationError;
import com.pkb.api.v1.model.ApiMeasurement;
import com.pkb.api.v1.model.ApiMeasurementType;
import com.pkb.api.v2.model.V2ApiMeasurement;
import com.pkb.app.dto.DateFilterDTO;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.service.test.MeasurementManager;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.test.entity.MeasurementDTO;
import com.pkb.test.entity.MeasurementType;
import com.pkb.test.entity.PredefinedMeasurementType;
import com.pkb.util.DateUtil;
import com.pkb.util.LocalizedTextProvider;
import org.jetbrains.annotations.NotNull;

import java.time.ZoneId;
import java.util.ArrayList;
import java.util.Comparator;
import java.util.Date;
import java.util.List;
import java.util.UUID;
import java.util.stream.Collectors;

public class MeasurementApiHelper extends ApiHelper {

    private static final long serialVersionUID = 1L;

    private final MeasurementManager measurementManager;

    private final UserAccessManager userAccessManager;
    private final DateUtil dateUtil;

    public MeasurementApiHelper(MeasurementManager measurementManager,
                                UserAccessManager userAccessManager,
                                DateUtil dateUtil,
                                LocalizedTextProvider localizedTextProvider) {
        super(localizedTextProvider);
        this.measurementManager = measurementManager;
        this.userAccessManager = userAccessManager;
        this.dateUtil = dateUtil;
    }

    public List<V2ApiMeasurement> getAllMeasurementsV2(@NotNull LoggedInEHRRequestContext requestContext,
                                                       List<PredefinedMeasurementType> measurementTypes, long userId, DateFilterDTO dateFilterDTO) {

        List<MeasurementDTO> measurements = measurementManager.getMeasurements(requestContext, userId, measurementTypes, dateFilterDTO);
        return measurements.stream().map(this::convertV2).sorted(Comparator.comparing(V2ApiMeasurement::getApiRef)).collect(Collectors.toList());
    }

    public UUID createMeasurement(EHRRequestContext requestContext, V2ApiMeasurement measurement, long userId) {
        MeasurementDTO dto = new MeasurementDTO(new SourceDetails(requestContext));
        PredefinedMeasurementType type = PredefinedMeasurementType.getByApiRef(measurement.getApiRef());
        assert type != null;
        dto.setMeasurementType(type);
        dto.setMeasureDate(measurement.getMeasureDate().toInstant());
        dto.setValue(measurement.getValue());
        dto.setValue2(measurement.getValue2());
        dto.getBaseFields().setEnteredDate(measurement.getEnteredDate());
        return measurementManager.saveMeasurement(requestContext, dto, userId);
    }

    public UUID createMeasurement(EHRRequestContext requestContext, ApiMeasurement measurement, long userId) {
        MeasurementDTO dto = new MeasurementDTO(new SourceDetails(requestContext));
        PredefinedMeasurementType type = PredefinedMeasurementType.getByApiRef(measurement.getMeasurementTypeApiRef());
        assert type != null;
        dto.setMeasurementType(type);
        dto.setMeasureDate(dateUtil.parseTimestampIso8601(measurement.getMeasureDateISO()).orElse(measurement.getMeasureDate().atStartOfDay(ZoneId.systemDefault()).toInstant()));
        dto.setValue(measurement.getValue());
        dto.setValue2(measurement.getValue2());

        return measurementManager.saveMeasurement(requestContext, dto, userId);
    }

    public boolean deleteMeasurement(LoggedInEHRRequestContext requestContext, long patientId, UUID measurementId) {
        return measurementManager.deleteMeasurementByUniqueId(requestContext, patientId, measurementId);
    }

    public List<ApiMeasurementType> getAllMeasurementTypes() {

        List<ApiMeasurementType> results = new ArrayList<>();
        for (PredefinedMeasurementType next : PredefinedMeasurementType.values()) {
            results.add(convert(next));
        }
        return results;
    }

    private ApiMeasurementType convert(PredefinedMeasurementType mt) {
        ApiMeasurementType amt = new ApiMeasurementType();
        transferMeasurementTypeFields(mt, amt);
        return amt;
    }

    private V2ApiMeasurement convertV2(MeasurementDTO dto) {

        V2ApiMeasurement v2m = new V2ApiMeasurement();

        if (dto.getMeasurementType() == null) {
            throw new ApiValidationError("Invalid measurement type " + dto);
        }

        v2m.setApiRef(dto.getMeasurementType().apiRef().getOrNull());
        v2m.setMeasureDate(Date.from(dto.getMeasureDate()));
        v2m.setEnteredDate(dto.getBaseFields().getEnteredDate());
        v2m.setPersistedDate(dto.getBaseFields().getPersistedDate());
        v2m.setValue(dto.getValue());
        v2m.setValue2(dto.getValue2());
        v2m.setId(dto.getBaseFields().getUniqueId());

        return v2m;

    }

    private void transferMeasurementTypeFields(PredefinedMeasurementType type0,
                                               ApiMeasurementType api) {
        MeasurementType type = type0.getType();
        api.setApiRef(type0.apiRef());
        api.setCategory(type.category());
        api.setCode(type.code().getOrNull());
        api.setCode2(type.code2().getOrNull());
        api.setCodeSystem(type.codeSystem().getOrNull());
        api.setCodeSystemName(type.codeSystemName().getOrNull());
        api.setName(getText(type.displayName()));
        api.setUnit(type.unit());
        api.setUnit2(type.unit2().getOrNull());
    }

}
