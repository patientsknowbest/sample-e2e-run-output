package com.pkb.api.util;

import com.pkb.api.error.ApiGeneralError;
import com.pkb.api.v1.model.ApiReadResponse;
import com.pkb.common.util.FrameFilter;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.ws.rs.WebApplicationException;
import java.util.List;

public class PKBApiHelper {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    public static WebApplicationException wrapError(String message, Throwable t) throws WebApplicationException {
        if (t instanceof WebApplicationException) {
            LOGGER.error("Handled exception: {}", message, FrameFilter.filter(t));
            return (WebApplicationException) t;
        } else {
            // log; this isn't validation or something else expected
            LOGGER.error(message, FrameFilter.filter(t));
            return new ApiGeneralError(message, t);
        }
    }

    public static <T> ApiReadResponse<T> buildReadResponse(List<T> results) {
        ApiReadResponse<T> container = new ApiReadResponse<>();
        container.setResults(results);
        return container;
    }
}
