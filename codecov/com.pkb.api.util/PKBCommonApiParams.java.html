package com.pkb.api.util;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.user.entity.PKBPerson;

import java.io.Serializable;

/**
 * A wrapper for common params used by many different API services. These can be handled centrally by an aspect
 * and injected into each method that wants to use them.
 * <p/>
 * The intention is to add whatever common params that are required to this class
 *
 * @author russell
 * @author robwhelan
 */
public class PKBCommonApiParams implements Serializable {

    private static final long serialVersionUID = 1L;

    private PKBPerson person;

    private LoggedInEHRRequestContext requestContext;

    public PKBPerson getPerson() {
        return person;
    }

    public void setPerson(PKBPerson person) {
        this.person = person;
    }

    public void setEHRRequestContext(LoggedInEHRRequestContext requestContext) {
        this.requestContext = requestContext;
    }

    public LoggedInEHRRequestContext getRequestContext() {
        return requestContext;
    }

    /**
     * @param explicitPatientId
     *            a patientId explicitly sent in an incoming HTTP request
     * @return {@code explicitPatientId} if it is not null, otherwise falling back to {@code person.getIdString()}
     */
    public String getPatientId(Long explicitPatientId) {
        if (explicitPatientId != null) {
            return explicitPatientId.toString();
        }
        return person.getIdString();
    }
}
