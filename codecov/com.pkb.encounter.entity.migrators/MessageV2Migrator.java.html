package com.pkb.encounter.entity.migrators;

import com.pkb.annotation.EHRVersionMigrator;
import com.pkb.app.entity.EHRData;
import com.pkb.common.datamigrator.Html4XmlUnescaper;

/**
 * version 2:
 * Properly unescape message subject and body (formerly these were stored always double-escaped... we don't escape anymore)
 */
public class MessageV2Migrator extends EHRVersionMigrator {

    private final Html4XmlUnescaper migrator;

    public MessageV2Migrator(Html4XmlUnescaper migrator) {
        super(2);
        this.migrator = migrator;
    }

    @Override
    public void migrate(EHRData ehr) {
        String subject = (String) ehr.getEncryptedField("subject");
        String content = (String) ehr.getEncryptedField("content");

        ehr.setEncryptedField("subject", doubleUnescape(subject));
        ehr.setEncryptedField("content", doubleUnescape(content));
    }

    private String doubleUnescape(String s) {
        if (s == null) {
            return "";
        }

        return migrator.unescape(s);
    }
}
