package com.pkb.encounter.entity.migrators;

import com.pkb.annotation.EHRVersionMigrator;
import com.pkb.app.entity.EHRData;
import org.apache.commons.text.StringEscapeUtils;

public class MessageV3Migrator extends EHRVersionMigrator {

    public MessageV3Migrator() {
        super(3);
    }

    @Override
    public void migrate(EHRData ehr) {
        String content = (String) ehr.getEncryptedField("content");
        ehr.setEncryptedField("content", StringEscapeUtils.unescapeHtml4(content));
    }

}
