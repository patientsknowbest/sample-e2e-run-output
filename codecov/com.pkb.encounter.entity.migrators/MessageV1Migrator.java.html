package com.pkb.encounter.entity.migrators;

import com.pkb.annotation.EHRVersionMigrator;
import com.pkb.app.entity.EHRData;
import com.pkb.encounter.entity.Message;
import org.apache.commons.lang3.StringUtils;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.HashSet;
import java.util.LinkedList;
import java.util.List;
import java.util.Set;

/**
 * version 1:
 * 1. Migrate to, from and copyToList fields from old messages as participants
 * 2. Migrate encrypted field patientId to string03
 */
public class MessageV1Migrator extends EHRVersionMigrator {

    public MessageV1Migrator() {
        super(1);
    }

    @Override
    public void migrate(EHRData ehr) {
        migrateToAndFromFields(ehr);
        migrateAboutPatient(ehr);

    }

    @SuppressWarnings("unchecked")
    private void migrateToAndFromFields(EHRData ehr) {

        String to = ehr.getString01();
        String from = ehr.getString02();
        Object copyToListObj = ehr.getEncryptedField("copyToList");

        // migrate To , From, copyToList,  field in old messages as Participants
        Object participantsObject = ehr.getEncryptedField(Message.PH_PARTICIPANTS_LIST);

        // we do not know if to and from were added to the participants list if at all ; hence a set
        Set<String> participantsSet = new HashSet<>();

        if (participantsObject != null) {

            List<String> participants = new ArrayList<>();
            if (participantsObject instanceof String) {
                participants = CSVtoList((String) participantsObject);
            }

            if (participantsObject instanceof List) {
                participants = (List<String>) participantsObject;
            }

            // add existing participants to the set
            if (!participants.isEmpty()) {
                participantsSet.addAll(participants);
            }
        } else {
            // for old messages which do not have participants property
            if (StringUtils.isNotBlank(to)) {
                // add reciever
                participantsSet.add(to);
            }

            if (StringUtils.isNotBlank(from)) {
                // add sender
                participantsSet.add(from);
            }
        }

        if (copyToListObj instanceof String) {
            List<String> copyToList = CSVtoList((String) copyToListObj);
            // add copy to list participants list
            if (!copyToList.isEmpty()) {
                participantsSet.addAll(copyToList);
            }
        }

        // old messages have some of the values as "null" as a string which escapes the null and empty test
        participantsSet.remove("null");
        List<String> newParticipantsList = new ArrayList<>();
        newParticipantsList.addAll(participantsSet);

        if (!newParticipantsList.isEmpty()) {
            ehr.setEncryptedField(Message.PH_PARTICIPANTS_LIST, newParticipantsList);
        }

    }

    private void migrateAboutPatient(EHRData ehr) {
        String patientId = ehr.getString03();
        Object patientIdObj = ehr.getEncryptedField("patientId");
        // patientId is not already migrated
        if (StringUtils.isBlank(patientId)) {
            if (patientIdObj instanceof String) {
                patientId = (String) patientIdObj;
                ehr.setString03(patientId);
            }
        }
    }

    private List<String> CSVtoList(String csv) {
        if (csv == null) {
            return new LinkedList<>();
        }
        if (csv.isEmpty()) {
            return new LinkedList<>();
        }
        return new LinkedList<>(Arrays.asList(csv.split(",")));
    }
}
