package com.pkb.service.emis;

public class EmisEsObservationDeltaManager extends EhrDataPointManager<EmisEsObservationDelta> {

    private MenuDataRepository menuDataRepository;

    public EmisEsObservationDeltaManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService,
                                         UUIDProvider uuidProvider, CodingManager codingManager, DeduplicatorManager deduplicator,
                                         MenuDataRepository menuDataRepository, Class<EmisEsObservationDelta> dtoClass, MenuDataType dataType) {
        super(config, beanFactory, dateTimeService, uuidProvider, codingManager, deduplicator, dtoClass, dataType);
        this.menuDataRepository = menuDataRepository;
    }

    @Override
    protected List<Tuple2<EmisEsObservationDelta, EHRRequestContext>> adjustCodings(List<Tuple2<EmisEsObservationDelta, EHRRequestContext>> data) {
        // exclude MEASUREMENTS and TEST records
        return data.stream().filter(deltaAndContext -> isMeasurementOrTestType(deltaAndContext._1().getDeltaType())).collect(Collectors.toList());
    }

    @Override
    protected List<Tuple2<EmisEsObservationDelta, EHRRequestContext>> adjustCodingsNoCodingLookup(List<Tuple3<EmisEsObservationDelta, CodingReceived, EHRRequestContext>> data) {
        // exclude MEASUREMENTS and TEST records
        return data.stream()
                .map(deltaAndCodingAndContext -> Tuple.of(deltaAndCodingAndContext._1(), deltaAndCodingAndContext._3()))
                .filter(deltaAndContext -> isMeasurementOrTestType(deltaAndContext._1().getDeltaType()))
                .collect(Collectors.toList());
    }

    public List<EmisEsObservationDelta> getObservationDeltas(@NotNull LoggedInEHRRequestContext requestContext, Long patientAccountId,
                                                             EmisEsObservationDelta.DeltaType deltaType) {
        EHRRemote ehrBean = beanFactory.getEhrRemote();
        EHRSearch<EmisEsObservationDelta> ehrSearch = new EHRSearch<>(patientAccountId, EmisEsObservationDelta.class,
                EmisEsObservationDelta.MS_PATH);
        ehrSearch.addFilter(new PKBFilter(EmisEsObservationDelta.DELTA_TYPE, PKBFilter.Operator.EQUAL, deltaType.toString()));
        // this needs to return only latest for all uniqueIds
        List<EHRData> ehrList = ehrBean.queryAndDecryptEHRData(ehrSearch, requestContext);
        return ehrBean.populateDTOList(ehrList, EmisEsObservationDelta.class, requestContext);
    }

    public Table<UUID, Long, MenuDataType> getOriginalDataTypeForDeltas(
            @NotNull Collection<UUID> uniqueIds) {
        if (isEmpty(uniqueIds)) {
            return ImmutableTable.of();
        }
        List<javax.persistence.Tuple> rs = menuDataRepository.findDistictUniqueIdAndDataTypeAndAccountIdByUniqueIdAndNotDatatypeAndNotDeleted(uniqueIds, MenuDataType.observationDelta);
        Set<UUID> uuids = new HashSet<>(rs.size());
        Set<Long> accountIds = new HashSet<>(rs.size());
        rs.forEach(o -> {
            uuids.add((UUID) o.get("uniqueId"));
            accountIds.add((Long) o.get("accountId"));
        });
        Table<UUID, Long, MenuDataType> result = ArrayTable.create(uuids, accountIds);
        rs.forEach(o -> {
            var previous = result.put((UUID) o.get("uniqueId"), (Long) o.get("accountId"), (MenuDataType) o.get("dataType"));
            if (previous != null) {
                throw new IllegalStateException("Duplicate menu data type for uniqueId: " + o.get("uniqueId") + " accountId: " + o.get("accountId"));
            }
        });
        return result;
    }

    public void softDelete(long deltaId) {
        //we previously hard deleted these with ehrBean#removeEHRData
        // however that tried to delete associated privacy_flags for the uniqueid, which
        // unfortunately since 2020 has been shared with the parent allergy/diagnosis.
        //Quick fix for now is to soft delete instead, this means we leave wasteful rows around, but hopefully
        // we should be able to scrap observationdeltas entirely once proprietary encryption is removed anyway.
        EHRRemote ehrBean = beanFactory.getEhrRemote();
        ehrBean.deleteEHRData(deltaId);
    }

    private boolean isMeasurementOrTestType(EmisEsObservationDelta.DeltaType emisEsObservationDelta) {
        return emisEsObservationDelta == EmisEsObservationDelta.DeltaType.MEASUREMENT
                || emisEsObservationDelta == EmisEsObservationDelta.DeltaType.TEST;
    }
}
