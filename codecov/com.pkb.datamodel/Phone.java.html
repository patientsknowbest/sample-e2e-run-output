package com.pkb.datamodel;

import com.google.common.base.Preconditions;
import com.pkb.datamodel.util.ObfuscateUtil;
import org.immutables.value.Value;

import java.io.Serializable;
import java.util.function.Function;

import static org.apache.commons.lang3.StringUtils.isNotBlank;

public interface Phone extends Serializable {
    @Value.Redacted
    String number();

    abstract class AbstractPhone implements Phone {
        protected String normalize(String email) {
            return email.trim().toLowerCase();
        }

        protected <T> T doCheck(String phoneNumber, Function<String,T> factory, T current) {
            if(phoneNumber.equals(normalize(phoneNumber))) {
                Preconditions.checkState(isNotBlank(this.number()), "Phone numbers should not be empty");
                return current;
            } else {
                return factory.apply(normalize(phoneNumber));
            }
        }
    }

    @Value.Immutable(builder = false)
    @Value.Style(of = "disclosingPhone", allParameters = true)
    abstract class DisclosingPhone extends AbstractPhone {
        @Override
        public String toString() {
            return number();
        }

        @Value.Check
        DisclosingPhone check() {
            return doCheck(this.number(), ImmutableDisclosingPhone::disclosingPhone, this);
        }
    }

    @Value.Immutable(builder = false)
    @Value.Style(of = "obfuscatingPhone", allParameters = true)
    abstract class ObfuscatingPhone extends AbstractPhone {
        @Override
        public final String toString() {
            return ObfuscateUtil.mask(number());
        }

        @Value.Check
        ObfuscatingPhone check() {
            return doCheck(this.number(), ImmutableObfuscatingPhone::obfuscatingPhone, this);
        }
    }
}
