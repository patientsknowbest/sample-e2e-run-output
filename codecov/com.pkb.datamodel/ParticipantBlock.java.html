package com.pkb.datamodel;

import com.pkb.datamodel.confidential.SensitiveString;
import org.immutables.value.Value;

import static com.pkb.datamodel.ImmutableParticipantBlock.participantBlock;

@Value.Immutable
@DataModelStyle
@Value.Style(builder = "participantBlock")
public interface ParticipantBlock extends PeriodBlock {

    ExternalParticipant getParticipant();

    static ParticipantBlock fromEvents(SensitiveString participantName, EncounterEvent blockEvent, EncounterEvent nextEvent) {
        return participantBlock()
                .start(blockEvent.timestamp())
                .end(nextEvent.timestamp())
                .participant(blockEvent.requireExternalParticipantByName(participantName))
                .build();
    }

    static ParticipantBlock fromEvent(ExternalParticipant participant, EncounterEvent blockEvent) {
        return participantBlock()
                .start(blockEvent.timestamp())
                .participant(participant)
                .build();
    }
}
