package com.pkb.datamodel;

import com.pkb.datamodel.authorization.ConsentedEntity;
import com.pkb.datamodel.consent.ImmutablePatientConsent;
import com.pkb.datamodel.consent.PatientConsent;
import com.pkb.datamodel.user.Patient;
import com.pkb.entities.enums.PrivacyFlag;

import java.util.UUID;

import static com.pkb.datamodel.PatientFactory.azPatient1;
import static com.pkb.datamodel.PatientFactory.patient;
import static com.pkb.entities.enums.PrivacyFlag.GENERAL;
import static io.vavr.API.List;

public class PatientConsentFactory {

    public static PatientConsent create(Patient patient, ConsentedEntity consentedEntity, PrivacyFlag... privacyFlags) {
        return ImmutablePatientConsent.builder()
                .publicId(UUID.randomUUID())
                .isProEditOnlyForChild(false)
                .isDischarged(false)
                .patient(patient)
                .consentedEntity(consentedEntity)
                .build()
                .withConsentsGranted(List(privacyFlags));
    }

    public static PatientConsent patientConsent() {
        return create(patient(), azPatient1(), GENERAL);
    }

}
