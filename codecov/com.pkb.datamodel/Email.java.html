package com.pkb.datamodel;

import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.pkb.datamodel.util.ObfuscateUtil;
import org.immutables.value.Value;
import org.immutables.value.Value.Auxiliary;
import org.immutables.value.Value.Check;
import org.immutables.value.Value.Derived;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;

import javax.mail.internet.AddressException;
import javax.mail.internet.InternetAddress;
import java.io.Serializable;
import java.util.function.Function;

/**
 * Passing around emails as {@link String} can be convinient, but can have some potential downsides. E.g.:
 * 1. Source/Sink distance: The closer to input we turn a string into an {@link Email} instance, the faster we can detect bad input.
 * 2. Decreases potential errors, where multiple {@link String} parameters are expected in given order (swapping parameters easier).
 */
@JsonSerialize
public interface Email extends Serializable, Comparable<Email> {
    @Value.Redacted
    String address();

    @Derived
    default String asAuthPrincipal() { return address(); }

    @Derived
    InternetAddress toInternetAddress();

    @Override
    default int compareTo(Email other) {
        if (other == null) {
            return 1;
        }

        return this.address().compareTo(other.address());
    }

    abstract class AbstractEmail implements Email {
        public static final String AT = "@";

        protected String normalize(String email) {
            return email.trim().toLowerCase().replace(".@", "@");
        }

        protected <T> T doCheck(String email, Function<String,T> factory, T current) {
            if(email.equals(normalize(email))) {
                this.toInternetAddress();
                return current;
            } else {
                return factory.apply(normalize(email));
            }
        }

        @Override
        @Auxiliary
        public final InternetAddress toInternetAddress() {
            try {
                InternetAddress[] addresses = InternetAddress.parse(address());
                if(addresses.length != 1) {
                    throw new RuntimeException("Multiple or no addresses are parsed out of '" + address() + "'.");
                }
                addresses[0].validate();
                return addresses[0];
            } catch (AddressException cause) {
                throw new RuntimeException(cause);
            }
        }


        @Override
        public final boolean equals(Object other) {
            if(other instanceof AbstractEmail) {
                return this.address().equals(((AbstractEmail)other).address());
            } else {
                return false;
            }
        }

        @Override
        public final int hashCode() {
            return this.address().hashCode();
        }
    }

    @Immutable(builder = false)
    @Style(of = "disclosingEmail", allParameters = true)
    abstract class DisclosingEmail extends AbstractEmail {
        @Override
        public String toString() {
            return address();
        }

        @Check
        DisclosingEmail check() {
            return doCheck(this.address(), ImmutableDisclosingEmail::disclosingEmail, this);
        }
    }

    @Immutable(builder = false)
    @Style(of = "obfuscatingEmail", allParameters = true)
    abstract class ObfuscatingEmail extends AbstractEmail {
        @Override
        public final String toString() {
            String[] split = address().split(AT);
            return ObfuscateUtil.mask(split[0]) + AT + split[1];
        }

        @Check
        ObfuscatingEmail check() {
            return doCheck(this.address(), ImmutableObfuscatingEmail::obfuscatingEmail, this);
        }
    }
}
