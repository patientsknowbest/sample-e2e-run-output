package com.pkb.datamodel;

import com.pkb.datamodel.user.ImmutableDemographics;
import com.pkb.entities.enums.Gender;

import java.time.LocalDate;
import java.util.UUID;

import static com.pkb.datamodel.EmailContactFactory.email;
import static com.pkb.datamodel.NationalIdFactory.nationalId;
import static com.pkb.datamodel.OrgLevelIdFactory.orgLevelId;
import static com.pkb.datamodel.PhoneContactFactory.phone;
import static com.pkb.datamodel.PostalAddressFactory.address;
import static com.pkb.datamodel.TeamLevelIdFactory.teamLevelId;
import static io.vavr.API.List;
import static java.util.UUID.fromString;

public class PersonDemographicsFactory {

    public static ImmutableDemographics personDemographics(Gender gender) {
        return personDemographics(1234L, fromString("67c02c61-cc33-49e8-8ddd-18420b0b5143"), gender, false);
    }

    public static ImmutableDemographics personDemographics(long privateId, UUID publicId, Gender gender, boolean isDeceased) {
        return ImmutableDemographics.builder()
                .id(privateId)
                .publicId(publicId)
                .address(address().build())
                .gender(gender)
                .isDeceased(isDeceased)
                .isSharingDisabled(false)
                .build()
                .withNationalId(List(nationalId()))
                .withTeamLevelId(List(teamLevelId()))
                .withOrgLevelId(List(orgLevelId()))
                .withEmail(List(email()))
                .withPhone(List(phone()))
                .withDateOfBirth(LocalDate.parse("1980-01-01"))
                .withTitle("Mr")
                .withFirstName("Leon")
                .withLastName("Musk");
    }

    public static ImmutableDemographics personDemographics() {
        return personDemographics(Gender.MALE);
    }

    public static ImmutableDemographics personDemographics(long privateId, UUID publicId) {
        return personDemographics(privateId, publicId, Gender.MALE, false);
    }
}
