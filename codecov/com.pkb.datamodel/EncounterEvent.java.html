package com.pkb.datamodel;

import com.pkb.datamodel.confidential.SensitiveString;
import com.pkb.datamodel.user.Patient;
import com.pkb.entities.enums.EncounterClass;
import io.vavr.collection.List;
import io.vavr.collection.Seq;
import io.vavr.control.Option;
import org.immutables.value.Value;
import org.immutables.vavr.encodings.VavrEncodingEnabled;

import java.time.ZonedDateTime;
import java.util.UUID;

@Value.Immutable
@Value.Style(builder = "encounterEvent")
@VavrEncodingEnabled
public interface EncounterEvent extends ClinicalData, Identifiable<UUID> {

    // The time that the thing happened
    ZonedDateTime timestamp();

    // The time it was recorded
    Option<ZonedDateTime> persisted();

    Patient patient();

    EncounterClass encounterClass();

    Option<ClinicalCoding> specialty();

    Option<Location> getLocation();

    List<ExternalParticipant> getExternalParticipants();

    @Value.Lazy
    default ClinicalCoding requireSpecialty() {
        return specialty()
                .getOrElseThrow(() -> new IllegalStateException("Required specialty is not present"));
    }

    @Value.Lazy
    default SensitiveString requireLocationDescription() {
        return getLocation()
                .map(Location::getDescription)
                .getOrElseThrow(() -> new IllegalStateException("Required location is not present"));
    }

    @Value.Lazy
    default ExternalParticipant requireExternalParticipantByName(SensitiveString name) {
        return getExternalParticipants()
                .filter(p -> p.getName().isEqualCaseInsensitive(name))
                .headOption()
                .getOrElseThrow(() -> new IllegalStateException("Required participant is not present"));
    }

    default Option<ExternalParticipant> findParticipantByName(SensitiveString name) {
        return getExternalParticipants()
                .filter(p -> p.getName().equals(name))
                .headOption();
    }

    static EncounterEvent earliest(Seq<EncounterEvent> encounterEvents) {
        return encounterEvents.minBy(EncounterEvent::timestamp)
                .getOrElseThrow(() -> new IllegalStateException("Failed to find first encounterEvent"));
    }

    static EncounterEvent latest(Seq<EncounterEvent> encounterEvents) {
        return encounterEvents.maxBy(EncounterEvent::timestamp)
                .getOrElseThrow(() -> new IllegalStateException("Failed to find last encounterEvent"));
    }

    @Value.Derived
    default boolean isNotOther() {
        return encounterClass() != EncounterClass.OTHER;
    }
}
