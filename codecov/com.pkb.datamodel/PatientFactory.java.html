    public static Patient createPatient(Long privateId, Set<TeamUser> teams, NationalId nationalId, UserStatus userStatus, LocalDate dateOfBirth) {
        ImmutableDemographics demographics = ImmutableDemographics.builder()
                .id(privateId)
                .publicId(UUID.randomUUID())
                .address(ImmutablePostalAddress.builder().build())
                .gender(Gender.MALE)
                .isDeceased(userStatus == DEAD)
                .isSharingDisabled(false)
                .build()
                .withTitle("Mr")
                .withFirstName("Marc")
                .withLastName("Rebillet");
        if (nationalId != null) {
            demographics = demographics.withNationalId(List(nationalId));
        }
        if (userStatus == EMAIL_CONFIRMED) {
            demographics = demographics.withEmail(List(EmailContactFactory.email()));
        }
        if (dateOfBirth != null) {
            demographics = demographics.withDateOfBirth(dateOfBirth);
        }

        ImmutablePatient.Builder patientBuilder = ImmutablePatient.builder()
                .demographics(demographics)
                .userType(PATIENT)
                .userStatus(userStatus)
                .id(demographics.getId())
                .publicId(demographics.getPublicId())
                .defaultAccountId(2222L)
                .sharingDisabled(false)
                .languageCode("en")
                .accessFrozen(false);
        if (teams != null) {
            patientBuilder.teams(teams);
        }

        return patientBuilder.build();
    }
