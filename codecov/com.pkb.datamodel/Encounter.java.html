package com.pkb.datamodel;

import com.google.common.base.Preconditions;
import com.pkb.datamodel.confidential.SensitiveString;
import com.pkb.entities.enums.EncounterClass;
import io.vavr.Function1;
import io.vavr.Function2;
import io.vavr.Function3;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import java.time.ZonedDateTime;
import java.util.Objects;
import java.util.UUID;
import java.util.function.Predicate;

import static com.google.common.base.Preconditions.checkState;
import static io.vavr.API.List;
import static java.util.Comparator.comparing;

@Value.Immutable
@Value.Style(builder = "encounter")
@DataModelStyle
public abstract class Encounter implements Identifiable<UUID> {

    public abstract List<EncounterEvent> events();

    @Value.Check
    public void someEvents() {
        Preconditions.checkState(!events().isEmpty(), "Can't make an encounter with no events!");
    }

    @Value.Derived
    public EncounterEvent first() {
        return EncounterEvent.earliest(io.vavr.collection.List.ofAll(events()));
    }

    @Value.Derived
    public EncounterEvent last() {
        return EncounterEvent.latest(io.vavr.collection.List.ofAll(events()));
    }

    @Value.Derived
    public ZonedDateTime getStart() {
        return first().timestamp();
    }

    @Value.Derived
    public ZonedDateTime getEnd() {
        return last().timestamp();
    }

    @Value.Derived
    public EncounterClass getEncounterClass() {
        return last().encounterClass();
    }

    @Value.Derived
    public List<EncounterEvent> getEventsChronologically() {
        return events()
                .sorted(comparing(EncounterEvent::timestamp));
    }

    @Value.Derived
    public List<LocationBlock> getLocationBlocks() {

        BlockProcessor<LocationBlock, SensitiveString> proc = new BlockProcessor<>(
                (ev) -> ev.getLocation().map(Location::getDescription),
                Objects::equals,
                LocationBlock::fromEvents);

        return finaliseBlocks(startProcessing(getEventsChronologically(), proc), LocationBlock::fromEvent);
    }

    @Value.Derived
    public List<ParticipantBlock> getParticipantBlocks() {
        List<EncounterEvent> eventSeq = getEventsChronologically();
        List<ExternalParticipant> participants = eventSeq.flatMap(EncounterEvent::getExternalParticipants).distinct();
        return participants.flatMap(currentParticipant -> {
            BlockProcessor<ParticipantBlock, ExternalParticipant> proc = new BlockProcessor<>(
                    (ev) -> ev.findParticipantByName(currentParticipant.getName()),
                    ExternalParticipant::equalByName,
                    Function3.of(ParticipantBlock::fromEvents).apply(currentParticipant.getName()));
            return finaliseBlocks(
                    startProcessing(eventSeq, proc),
                    Function2.of(ParticipantBlock::fromEvent).apply(currentParticipant));
        });
    }

    @Value.Derived
    public List<SpecialtyBlock> getSpecialtyBlocks() {

        BlockProcessor<SpecialtyBlock, ClinicalCoding> proc = new BlockProcessor<>(
                EncounterEvent::specialty,
                Objects::equals,
                SpecialtyBlock::fromEvents);

        return finaliseBlocks(startProcessing(getEventsChronologically(), proc), SpecialtyBlock::fromEvent);
    }

    public List<EncounterClassBlock> getClassBlocks(Function2<EncounterClass, EncounterClass, Boolean> comparisonFn,
                                                    Predicate<EncounterEvent> eventFilterFn) {

        BlockProcessor<EncounterClassBlock, EncounterClass> proc = new BlockProcessor<>(
                (e) -> Option.of(e.encounterClass()),
                comparisonFn,
                EncounterClassBlock::fromEvents);

        List<EncounterEvent> eventSeq = getEventsChronologically().filter(eventFilterFn);
        return finaliseBlocks(startProcessing(eventSeq, proc), EncounterClassBlock::fromEvent);
    }

    private <T, V> Tuple2<Option<EncounterEvent>, List<T>> startProcessing(List<EncounterEvent> events, 
                                                                           BlockProcessor<T, V> proc) {
        return events.foldLeft(Tuple.of(Option.none(), List()), proc::collapse);
    }

    private static <T> List<T> finaliseBlocks(Tuple2<Option<EncounterEvent>, List<T>> blocks,
                                              Function1<EncounterEvent, T> factory) {
        // Deconstruct
        Option<EncounterEvent> finalEvent = blocks._1;
        List<T> list = blocks._2;

        // Invoke factory if the last event has a value
        return finalEvent.map(ev -> list.append(factory.apply(ev))).getOrElse(list);
    }

    private static class BlockProcessor<T, V> {

        // TODO: javadoc for usage of this class

        /**
         * valueFn is a user-defined function that extracts an optional value from an EncounterEvent to use when
         * building a block - examples might be location or participant name
         */
        private final Function1<EncounterEvent, Option<V>> valueFn;

        /**
         * comparisonFn is a user-defined function that compares an extracted value (see {@link BlockProcessor#valueFn})
         * for two consecutive events. If comparisonFn returns true, the current block is extended. If it returns false,
         * the current block is terminated and a new block is started
         */
        private final Function2<V, V, Boolean> comparisonFn;

        /**
         * builderFn is a user-defined function that defines how to build a block when the block is terminated
         */
        private final Function2<EncounterEvent, EncounterEvent, T> builderFn;

        private BlockProcessor(@NotNull Function1<EncounterEvent, Option<V>> valueFn,
                               @NotNull Function2<V, V, Boolean> comparisonFn,
                               @NotNull Function2<EncounterEvent, EncounterEvent, T> builderFn) {

            this.valueFn = valueFn;
            this.comparisonFn = comparisonFn;
            this.builderFn = builderFn;
        }

        public Tuple2<Option<EncounterEvent>, List<T>> collapse(
                @NotNull Tuple2<Option<EncounterEvent>, List<T>> accumulator,
                @NotNull EncounterEvent nextEvent) {

            // Deconstruct
            Option<EncounterEvent> currentEvent = accumulator._1;
            List<T> list = accumulator._2;

            return currentEvent
                    .map(ev -> handleCurrent(ev, nextEvent, list))
                    .getOrElse(() -> Tuple.of(valueFn.apply(nextEvent).flatMap($ -> Option.of(nextEvent)), list));
        }

        private Tuple2<Option<EncounterEvent>, List<T>> handleCurrent(
                EncounterEvent currentEvent,
                EncounterEvent nextEvent,
                List<T> list) {

            checkState(valueFn.apply(currentEvent).isDefined(), "Expected currentEvent to have value");

            return valueFn.apply(nextEvent)
                    .map($ -> handleCurrentWithNext(currentEvent, nextEvent, list)) // Next event has a value set
                    .getOrElse(() -> {
                        // End of block. Continue processing with no current value
                        return Tuple.of(Option.none(), list.append(builderFn.apply(currentEvent, nextEvent)));
                    });
        }

        private Tuple2<Option<EncounterEvent>, List<T>> handleCurrentWithNext(
                @NotNull EncounterEvent currentEvent,
                @NotNull EncounterEvent nextEvent,
                @NotNull List<T> list) {

            V currentEventValue = valueFn.apply(currentEvent).getOrElseThrow(() -> new IllegalStateException("Expected currentEvent to have value"));
            V nextEventValue = valueFn.apply(nextEvent).getOrElseThrow(() -> new IllegalStateException("Expected nextEvent to have value"));

            if (comparisonFn.apply(currentEventValue, nextEventValue)) {
                // Next event is same as this one - continue processing with no change
                return Tuple.of(Option.of(currentEvent), list);
            } else {
                // Next event has different value - save the block and proceed with a new one
                return Tuple.of(Option.of(nextEvent), list.append(builderFn.apply(currentEvent, nextEvent)));
            }
        }

    }
}
