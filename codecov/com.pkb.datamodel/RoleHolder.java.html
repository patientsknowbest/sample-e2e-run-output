package com.pkb.datamodel;

import com.google.common.collect.Sets;
import com.google.common.collect.Sets.SetView;
import com.pkb.entities.enums.api.ApiClientScope;

import java.util.Set;

import static java.util.Collections.emptySet;
import static java.util.Objects.requireNonNull;

/**
 * holds role information
 * this class can be proxied into a singleton bean on a per-request basis
 * hence avoiding thread-safety issues
 * <p/>
 * we are relying on Spring native functionality for this capability
 *
 * @author vorekoya on 14/12/2015.
 */
public class RoleHolder {

    private Set<ApiClientScope> allowedRoles = emptySet();

    public Set<ApiClientScope> getAllowedRoles() {
        return this.allowedRoles;
    }

    public void setAllowedRoles(Set<ApiClientScope> scopes) {
        this.allowedRoles = requireNonNull(scopes, "Scopes cannot be null!");
    }

    public void clear() {
        allowedRoles = emptySet();
    }

    public boolean contains(ApiClientScope scope) {
        return allowedRoles.contains(scope);
    }

    public boolean isEmpty() {
        return allowedRoles.isEmpty();
    }

    public boolean hasAnyOf(Set<ApiClientScope> scopes) {
        return !hasNoneOf(scopes);
    }

    public boolean hasNoneOf(Set<ApiClientScope> scopes) {
        SetView<ApiClientScope> commonScopes = Sets.intersection(allowedRoles, scopes);
        return commonScopes.isEmpty();
    }
}
