package com.pkb.datamodel;

import java.time.LocalDateTime;
import java.time.ZoneOffset;
import java.time.ZonedDateTime;
import java.util.UUID;

import static com.pkb.datamodel.UUIDFactory.constantUUID;

public class EncounterEventFactory {

    public static @NotNull EncounterEvent encounterEvent(EncounterClass klass, ZonedDateTime timestamp) {
        return ImmutableEncounterEvent.encounterEvent()
                .accountId(12345L)
                .persisted(ZonedDateTime.of(LocalDateTime.parse("1983-06-26T10:33:22"), ZoneOffset.UTC))
                .id(constantUUID("2014L"))
                .encounterClass(klass)
                .patient(PatientFactory.azPatient1())

                        .generalRequired(true)
                        .sexualHealthRequired(false)

                        .socialCareRequired(false).build())

    }
