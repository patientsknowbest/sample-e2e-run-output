package com.pkb.datamodel;

import com.google.common.base.Preconditions;
import com.google.common.collect.ImmutableSetMultimap;
import com.pkb.entities.enums.api.ApiClientScope;
import com.pkb.entities.enums.api.GrantType;
import org.apache.commons.lang3.StringUtils;
import org.immutables.value.Value;
import org.jetbrains.annotations.Nullable;

import java.util.Collections;
import java.util.List;
import java.util.Optional;
import java.util.Set;
import java.util.stream.Collectors;

import static java.lang.String.format;
import static java.util.Collections.emptySet;
import static java.util.Collections.unmodifiableSet;

@Value.Immutable
@DataModelStyle
@SuppressWarnings("ClassWithTooManyMethods")
public interface ApiClient {

    ImmutableSetMultimap<ApiClientScope, ApiClientScope> SCOPE_MAP = new ImmutableSetMultimap.Builder<ApiClientScope, ApiClientScope>()
            .put(ApiClientScope.CLINICIAN, ApiClientScope.CLINICIAN)
            .put(ApiClientScope.CLINICIAN_OR_TEAMCOORD, ApiClientScope.CLINICIAN)
            .put(ApiClientScope.CLINICIAN_OR_TEAMCOORD, ApiClientScope.TEAMCOORD)
            .put(ApiClientScope.CLINICIAN_OR_TEAMCOORD, ApiClientScope.CLINICIAN_OR_TEAMCOORD)
            .put(ApiClientScope.EMERGENCY_CARE, ApiClientScope.EMERGENCY_CARE)
            .put(ApiClientScope.PATIENT, ApiClientScope.PATIENT)
            .put(ApiClientScope.PATIENT_OR_CLINICIAN, ApiClientScope.PATIENT)
            .put(ApiClientScope.PATIENT_OR_CLINICIAN, ApiClientScope.CLINICIAN)
            .put(ApiClientScope.PATIENT_OR_CLINICIAN, ApiClientScope.PATIENT_OR_CLINICIAN)
            .put(ApiClientScope.PATIENT_OR_CLINICIAN_OR_TEAMCOORD, ApiClientScope.PATIENT)
            .put(ApiClientScope.PATIENT_OR_CLINICIAN_OR_TEAMCOORD, ApiClientScope.CLINICIAN)
            .put(ApiClientScope.PATIENT_OR_CLINICIAN_OR_TEAMCOORD, ApiClientScope.TEAMCOORD)
            .put(ApiClientScope.PATIENT_OR_CLINICIAN_OR_TEAMCOORD, ApiClientScope.PATIENT_OR_CLINICIAN_OR_TEAMCOORD)
            .put(ApiClientScope.PUBLIC, ApiClientScope.PUBLIC)
            .put(ApiClientScope.SITE, ApiClientScope.SITE)
            .put(ApiClientScope.SUPPORT, ApiClientScope.SUPPORT)
            .put(ApiClientScope.TEAM_DATA_UPLOAD, ApiClientScope.TEAM_DATA_UPLOAD)
            .put(ApiClientScope.TEAMCOORD, ApiClientScope.TEAMCOORD)
            .put(ApiClientScope.PRIVACYOFFICER, ApiClientScope.PRIVACYOFFICER)
            .build();

    Optional<Long> getId();
    boolean isEnabled();
    Optional<String> getIpWhiteList();
    long getSessionExpirySeconds();
    long getTokenExpirySeconds();
    Optional<ApiClientScope> getScopeAllowed();
    String getRedirectionEndpointList();
    Optional<String> getOwner();
    String getName();
    @Value.Redacted
    String getApiId();
    List<PermittedGrantType> getPermittedGrantTypes();

    @Value.Check
    default void checkDuplicatePermittedGrantType() {
        var duplicates = getPermittedGrantTypes()
                .stream()
                .filter(i -> Collections.frequency(getPermittedGrantTypes(), i) > 1)
                .map(pgt -> pgt.getGrantType().toString())
                .collect(Collectors.toSet());
        Preconditions.checkState(duplicates.isEmpty(),
                format("Duplicate grant %s not permitted [%s] for apiclient [apiid=%s]",
                        duplicates.size() > 1 ? "types" : "type",
                        String.join(", ", duplicates),
                        getApiId())
        );
    }

    @Value.Derived
    default boolean isGrantTypePermitted(GrantType grantType) {
        return getPermittedGrantTypes().stream().map(PermittedGrantType::getGrantType).anyMatch(type -> type == grantType);
    }

    @Value.Derived
    default boolean hasSessionExpiry() {
        return getSessionExpirySeconds() > 0;
    }

    @Value.Derived
    default boolean hasTokenExpiry() {
        return getTokenExpirySeconds() > 0;
    }

    @Value.Derived
    default Set<ApiClientScope> getScopesAllowed() {
        if (getScopeAllowed().isEmpty()) {
            return emptySet();
        }
        return unmodifiableSet(SCOPE_MAP.get(getScopeAllowed().get()));
    }

    @Value.Derived
    default Set<ApiClientScope> getScopesAllowed(ApiClientScope scope) {
        if (scope == null) {
            return emptySet();
        }
        return unmodifiableSet(SCOPE_MAP.get(scope));
    }

    /**
     * This ApiClient can have just one value, or >1 as space-separated values.
     * If there's just one, calls may send a blank value
     * If there are more as CSV, calls must send in ONE of those
     * If the call provides a value at all, it must match exactly (including case).
     */
    @SuppressWarnings("BooleanMethodIsAlwaysInverted")
    @Value.Derived
    default boolean isRedirectionEndpointValid(String checkEndpoint) {
        String[] validEPs = StringUtils.split(getRedirectionEndpointList(), " ");
        if (StringUtils.isBlank(checkEndpoint)) {
            return validEPs.length == 1; // blank value is only valid if this client has just one option
        }
        // otherwise: checkEndpoint must be in the list (can have extra params after)
        for (String validEP : validEPs) {
            if (checkEndpoint.startsWith(validEP)) {
                return true;
            }
        }
        return false;
    }

    /**
     * Determines whether the user-specified {@code scope} is permitted
     * for this particular {@code ApiClient}.
     *
     * @param scope The access requested for auth. Note, this might be a
     *              mixed type, e.g. PATIENT_OR_CLINICIAN, since this is conceptually
     *              different to a UserType.
     * @return true - if, and only if, the requested scope is permitted for
     *         this particular api client. false - otherwise.
     */
    @Value.Derived
    default boolean isScopeAllowed(ApiClientScope scope) {
        return SCOPE_MAP.containsEntry(getScopeAllowed().orElse(null), scope);
    }

    /** return the single redirectionEndpoint, or null if we have a list */
    @Value.Derived
    default @Nullable String getSingleRedirectionEndpoint() {
        var validEPs = StringUtils.split(getRedirectionEndpointList(), " ");
        if (validEPs.length == 1) {
            return getRedirectionEndpointList();
        }
        return null;
    }

    @Value.Derived
    default String getSelectedRedirectionEndpoint(String paramEndpoint) {
        if (!isRedirectionEndpointValid(paramEndpoint)) {
            throw new IllegalArgumentException(
                    "paramEndpoint " + paramEndpoint + " isn't valid for client setting " + getRedirectionEndpointList());
        }

        if (StringUtils.isBlank(paramEndpoint)) {
            return getRedirectionEndpointList();
        } else {
            return paramEndpoint;
        }
    }
}
