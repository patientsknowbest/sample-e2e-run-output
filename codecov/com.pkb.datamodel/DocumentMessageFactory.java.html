package com.pkb.datamodel;

import org.jetbrains.annotations.NotNull;

import java.time.LocalDateTime;
import java.time.ZoneOffset;
import java.time.ZonedDateTime;
import java.util.UUID;

import static com.pkb.datamodel.DocumentMessage.DocumentType.DISCHARGE_SUMMARY;
import static com.pkb.datamodel.PatientFactory.patient;
import static com.pkb.datamodel.UUIDFactory.constantUUID;

public class DocumentMessageFactory {

    public static @NotNull DocumentMessage documentMessageWithoutExternalId() {
        return documentMessageWithoutExternalId(DISCHARGE_SUMMARY, true);
    }

    public static @NotNull DocumentMessage documentMessageWithoutExternalId(DocumentMessage.DocumentType type, boolean current) {
        return ImmutableDocumentMessage.builder()
                .type(type)
                .current(current)
                .accountId(12345L)
                .created(ZonedDateTime.of(LocalDateTime.parse("1983-06-26T10:11:22"), ZoneOffset.UTC))
                .persisted(ZonedDateTime.of(LocalDateTime.parse("1983-06-26T10:33:22"), ZoneOffset.UTC))
                .encounter(constantUUID("19830626"))
                .id(2014L)
                .uniqueId(UUID.fromString("50904a87-427c-4b01-a98b-5cb0c3ccb2c6"))
                .patient(patient())
                .source(ImmutableSource.builder().build())
                .status(Message.StatusType.READ)
                .source(ImmutableSource.builder().org(OrgFactory.orgA()).build())
                .privacyFlags(ImmutablePrivacyFlags.builder().generalRequired(true).sexualHealthRequired(false)
                        .mentalHealthRequired(false)
                        .socialCareRequired(false).build())
                .build();
    }

    public static @NotNull DocumentMessage documentMessage(DocumentMessage.DocumentType type, boolean current) {
        return ImmutableDocumentMessage.copyOf(documentMessageWithoutExternalId(type, current))
                .withExternalMessageId("321BANG");
    }

    public static @NotNull DocumentMessage documentMessage() {
        return documentMessage(DISCHARGE_SUMMARY, true);
    }

    public static @NotNull DocumentMessage documentMessage(boolean current) {
        return documentMessage(DISCHARGE_SUMMARY, current);
    }

    public static @NotNull DocumentMessage documentMessage(DocumentMessage.DocumentType type) {
        return documentMessage(type, true);
    }

    public static @NotNull DocumentMessage documentMessageWithExternalMessageId(String externalMessageId) {
        return ImmutableDocumentMessage.copyOf(documentMessage())
                .withExternalMessageId(externalMessageId);
    }

    public static @NotNull DocumentMessage documentMessageWithStatus(DocumentMessage message, Message.StatusType type) {
        return ImmutableDocumentMessage.copyOf(message)
                .withStatus(type);
    }

    private DocumentMessageFactory() {
    }
}
