package com.pkb.datamodel;

import com.pkb.datamodel.user.Patient;
import com.pkb.datamodel.user.Person;
import io.vavr.collection.LinearSeq;
import io.vavr.control.Option;
import org.immutables.value.Value;

import java.time.ZonedDateTime;
import java.util.UUID;

/**
 * PKB's definition of a communication.
 * Heavily influenced by FHIR
 */
@Value.Immutable
@Value.Style(builder = "communication")
@DataModelStyle
public abstract class Communication implements Identifiable<UUID>, ClinicalData {

    public abstract Option<ZonedDateTime> getPersisted();

    public abstract ZonedDateTime getTimestamp();

    public abstract UUID getConversationId();

    public abstract String getConversationSubject();

    public abstract Patient getPatient();

    public abstract LinearSeq<Person> getParticipants();

    public abstract LinearSeq<ExternalParticipant> getNonPKBParticipants();

    public abstract Person getSender();

    public abstract LinearSeq<AttachmentMetadata> getAttachmentMetadata();

    public abstract boolean isPrivateMessage();

    public abstract String getStatus();
}
