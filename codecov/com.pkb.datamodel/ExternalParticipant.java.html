package com.pkb.datamodel;

import com.pkb.datamodel.confidential.SensitiveString;
import org.immutables.value.Value;

@Value.Immutable
@DataModelStyle
@Value.Style(builder = "externalParticipant")
public interface ExternalParticipant {

    SensitiveString getName();

    ParticipantRole getRole();

    static boolean equalByName(ExternalParticipant p1, ExternalParticipant p2) {
        return p1.getName().equals(p2.getName());
    }

    enum ParticipantRole {
        CONSULTANT,
        ADMITTER,
        ATTENDER,
        REFERRER,
        DISCHARGER,
        CARER,      // PKB role
        PATIENT     // PKB Role
    }
}
