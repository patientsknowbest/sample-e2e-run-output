package com.pkb.datamodel;

import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;

import java.time.ZonedDateTime;

/**
 * <pre>
 * Root of type hierarchy for Ids such as:
 *  - National ID (e.g.: NHS in UK, Social Care Number in US).
 *  - Org Level ID (e.g.: ID assigned to a Patient on Hospital level)
 *  - Team Level ID (e.g.: ID assigned to a Patient on a Department level).
 * </pre>
 *
 * @param <I>
 *         Sub ID type.
 * @param <T>
 *         Id Type.
 */
public interface GenericId<I extends GenericId<I, T>, T> extends Identifiable<Long>, Comparable<I> {

    Option<String> getValue();
    
    default String requireValue() {
        return getValue().getOrElseThrow(() -> new IllegalStateException("value not set on GenericId!"));
    }

    Option<ZonedDateTime> getEditDate();

    @NotNull T getType();

    @Override
    default int compareTo(@NotNull I other) {
        return getValue()
                .map(leftValue -> other.getValue()
                        .map(leftValue::compareTo)
                        .getOrElse(1))
                .getOrElse(() -> other.getValue().isDefined() ? -1 : 0);
    }
}
