package com.pkb.datamodel;

import com.pkb.entities.enums.NationalIdType;
import com.pkb.entities.enums.ValidNationalId;

import java.time.ZonedDateTime;

public class NationalIdFactory {

    public static ImmutableNationalId nationalId() {
        return ImmutableNationalId.builder()
                .id(5678L)
                .type(NationalIdType.NHS_NUMBER)
                .build()
                .withValue("9267767046")
                .withEditDate(ZonedDateTime.parse("2007-12-03T10:15:30Z"));
    }

    public static ImmutableNationalId nationalId(ValidNationalId validNationalId) {
        return nationalId(validNationalId.type(),validNationalId.value());
    }

    public static ImmutableNationalId nationalId(NationalIdType nationalIdType, String nationalIdValue) {
        return ImmutableNationalId.builder()
                .id(9876L)
                .type(nationalIdType)
                .build()
                .withValue(nationalIdValue)
                .withEditDate(ZonedDateTime.parse("2015-11-04T11:25:40Z"));
    }
}
