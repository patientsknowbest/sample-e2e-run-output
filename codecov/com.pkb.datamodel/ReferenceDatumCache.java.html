package com.pkb.datamodel;

import com.pkb.datamodel.confidential.SensitiveString;
import com.pkb.entities.pub.PublicOrg;
import io.vavr.collection.Map;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;

public class ReferenceDatumCache {

    private Map<Long, Map<String, String>> mapOrgIdToCodeToValue;

    public ReferenceDatumCache(Map<Long, Map<String, String>> mapOrgIdToCodeToValue) {
        this.mapOrgIdToCodeToValue = mapOrgIdToCodeToValue;
    }

    @NotNull
    public Option<ClinicalCoding> lookup(Option<PublicOrg> maybeOrg, Option<String> maybeCode) {
        if (maybeCode.isEmpty()) {
            return Option.none();
        }
        String code = maybeCode.get();

        if (maybeOrg.isEmpty() || mapOrgIdToCodeToValue.isEmpty()) {
            return Option.of(ImmutableClinicalCoding.of(code, Option.none()));
        }
        PublicOrg org = maybeOrg.get();
        
        Option<Map<String, String>> lookupForOrg = mapOrgIdToCodeToValue.get(org.getId());
        if (lookupForOrg.isEmpty()) {
            return Option.of(ImmutableClinicalCoding.of(code, Option.none()));
        }

        Option<SensitiveString> displayText = lookupForOrg.get().get(code).map(SensitiveString::of);
        return Option.of(ImmutableClinicalCoding.of(code, Option.none())
                .withDisplayText(displayText));
    }
}
