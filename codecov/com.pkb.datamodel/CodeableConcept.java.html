package com.pkb.datamodel;

import com.pkb.datamodel.confidential.SensitiveString;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;
import org.immutables.value.Value;
import org.immutables.vavr.encodings.VavrEncodingEnabled;

@Value.Immutable
@VavrEncodingEnabled
public abstract class CodeableConcept {

    public abstract List<ClinicalCoding> getCodings();

    public abstract Option<SensitiveString> getDisplayText();

    @Value.Default
    public SensitiveString getAnyDisplayText() {
        return getDisplayText().getOrElse(() -> 
            getCodings()
                .filter(coding -> coding.getDisplayText().isDefined())
                .map(coding -> coding.getDisplayText().get())
                .headOption()
                .getOrElse(SensitiveString.of("")));
    }

    @Value.Derived
    public Option<ClinicalCoding> getFirstCoding() {
        return getNthCoding(1);
    }

    @Value.Derived
    public Option<ClinicalCoding> getSecondCoding() {
        return getNthCoding(2);
    }

    @Value.Derived
    public boolean hasAtLeastOneCode() {
        return getCodings().map(ClinicalCoding::getCode).find(StringUtils::isNotBlank).isDefined();
    }

    /**
     * Get nth coding, if it exists. n is 1-indexed
     */
    protected Option<ClinicalCoding> getNthCoding(int n) {
        if (getCodings().size() >= n) {
            return Option.of(getCodings().get(n - 1));
        }

        return Option.none();
    }

    public static ImmutableCodeableConcept of(SensitiveString displayText) {
        return ImmutableCodeableConcept.builder()
                .displayText(displayText)
                .build();
    }

    public static ImmutableCodeableConcept of(String displayText) {
        return CodeableConcept.of(SensitiveString.of(displayText));
    }

    public static ImmutableCodeableConcept of(ClinicalCoding coding) {
        return ImmutableCodeableConcept.builder()
                .addCodings(coding)
                .build();
    }
}
