package com.pkb.datamodel;

import io.vavr.control.Option;

import java.util.UUID;

import static java.util.UUID.fromString;
import static java.util.UUID.nameUUIDFromBytes;

public class UUIDFactory {
    public static UUID constantUUID() {
        return nameUUIDFromBytes("uuid".getBytes());
    }

    public static UUID constantUUID(String from) {
        return nameUUIDFromBytes(from.getBytes());
    }

    public static UUID uuid(String from) {
        return fromString(from);
    }

    public static Option<UUID> optionUUID() {
        return Option.of(constantUUID());
    }
}
