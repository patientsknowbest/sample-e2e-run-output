package com.pkb.datamodel;

import com.pkb.common.datamigrator.Html4XmlUnescaper;
import com.pkb.crypto.decryptionhelper.DecryptionHelper;
import com.pkb.crypto.decryptionhelper.DecryptionHelperAppMenuData;
import com.pkb.datamodel.crypto.AccountKeysProviderFactory;
import com.pkb.datamodel.crypto.CryptoService;
import com.pkb.datamodel.crypto.LocalCryptoService;
import com.pkb.datamodel.entity.ClinicalDataDeserializer;
import com.pkb.datamodel.entity.mapper.migrator.CommunicationSubjectMigrator;
import com.pkb.datamodel.entity.mapper.migrator.NonPersistentCommunicationMigrator;
import com.pkb.entities.app.AppMenuData;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class DatamodelEntityMappersConfig {

    @Bean
    public AccountKeysProviderFactory accountKeysProviderFactory() {
        return new AccountKeysProviderFactory();
    }

    @Bean
    public com.pkb.crypto.decryptionhelper.provider.accountkeys.AccountKeysProviderFactory<AppMenuData> cryptoAccountKeysProviderFactory() {
        return new com.pkb.crypto.decryptionhelper.provider.accountkeys.AccountKeysProviderFactory<>();
    }

    @Bean
    public DecryptionHelperAppMenuData decryptionHelperAppMenuData(DecryptionHelper decryptionHelper,
                                                                   com.pkb.crypto.decryptionhelper.provider.accountkeys.AccountKeysProviderFactory<AppMenuData> accountKeysProviderFactory,
                                                                   ClinicalDataDeserializer deserializer) {
        return new DecryptionHelperAppMenuData(decryptionHelper, accountKeysProviderFactory, deserializer);
    }

    @Bean
    public DecryptionHelper decryptionHelper() {
        return new DecryptionHelper();
    }

    @Bean
    public LocalCryptoService localCryptoService(DecryptionHelperAppMenuData decryptionHelper,
                                          AccountKeysProviderFactory providerFactory) {
        return new LocalCryptoService(decryptionHelper, providerFactory);
    }

    @Bean
    public CryptoService cryptoService(LocalCryptoService localCryptoService) {
        return new CryptoService(localCryptoService);
    }

    @Bean
    public ClinicalDataDeserializer clinicalDataDeserializer() {
        return new ClinicalDataDeserializer();
    }

    @Bean
    public Html4XmlUnescaper doubleUnescapingMigrator() {
        return new Html4XmlUnescaper();
    }

    @Bean
    public NonPersistentCommunicationMigrator nonPersistentCommunicationMigrator(Html4XmlUnescaper html4XmlUnescaper) {
        return new CommunicationSubjectMigrator(html4XmlUnescaper);
    }
}
