package com.pkb.datamodel;

import com.pkb.datamodel.user.Demographics;
import com.pkb.datamodel.user.ImmutableDemographics;
import com.pkb.datamodel.user.ImmutablePostalAddress;

import java.util.UUID;

import static com.pkb.datamodel.PostalAddressFactory.address;
import static com.pkb.entities.enums.Gender.MALE;

public class DemographicsFactory {
    public static ImmutableDemographics demographics(long privateId, UUID publicId) {
        return ImmutableDemographics.builder()
                .id(privateId)
                .publicId(publicId)
                .address(ImmutablePostalAddress.builder().build())
                .gender(MALE)
                .isDeceased(false)
                .isSharingDisabled(false)
                .build();
    }

    public static Demographics defaultPerson(long privateId, UUID publicId, boolean sharingDisabled) {
        // See com.pkb.testutils.builders.PublicPKBPersonBuilder.getDefault
        return ImmutableDemographics.builder()
                .id(privateId)
                .publicId(publicId)
                .address(address().build())
                .gender(MALE)
                .isDeceased(false)
                .isSharingDisabled(sharingDisabled)
                .build()
                .withTitle("Mr")
                .withFirstName("Ward")
                .withLastName("Cunningham");
    }
}
