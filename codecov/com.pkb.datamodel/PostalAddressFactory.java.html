package com.pkb.datamodel;

import com.pkb.datamodel.user.ImmutablePostalAddress;

import static com.pkb.datamodel.user.ImmutablePostalAddress.builder;

public class PostalAddressFactory {

    public static ImmutablePostalAddress.Builder address() {
        return builder()
                .address1("St Margaret Street")
                .address2("Parliament Square")
                .city("London")
                .state("Westminster")
                .postCode("SW1A 0AA")
                .country("United Kingdom");
    }
    
    public static ImmutablePostalAddress.Builder utterlyNullAddress() {
        return ImmutablePostalAddress.builder();
    }
    
    public static ImmutablePostalAddress.Builder utterlyEmptyAddress() {
        return builder()
                .address1(" ")
                .address2(" ")
                .city(" ")
                .state(" ")
                .postCode(" ")
                .country(" ");
    }
}
