package com.pkb.datamodel;

import com.pkb.datamodel.user.ImmutableTeamProfessional;
import com.pkb.datamodel.user.TeamProfessional;
import com.pkb.entities.enums.UserStatus;
import com.pkb.entities.enums.UserType;

import java.util.UUID;

import static com.pkb.datamodel.OrgFactory.orgA;
import static com.pkb.datamodel.PersonDemographicsFactory.personDemographics;

public class TeamProfessionalFactory {

    public static TeamProfessional teamProfessional(long privateId, UUID publicId) {
        return ImmutableTeamProfessional.builder()
                .demographics(personDemographics(privateId, publicId))
                .userStatus(UserStatus.EMAIL_CONFIRMED)
                .id(privateId)
                .publicId(publicId)
                .userType(UserType.REG_CLINICIAN)
                .languageCode("en")
                .sourceOrg(orgA())
                .build();
    }

    public static TeamProfessional teamProfessional(long privateId, UUID publicId, UUID humanUUID) {
        return ImmutableTeamProfessional.builder()
                .demographics(personDemographics(privateId, publicId))
                .userStatus(UserStatus.EMAIL_CONFIRMED)
                .id(privateId)
                .publicId(publicId)
                .humanUUID(humanUUID)
                .userType(UserType.REG_CLINICIAN)
                .languageCode("en")
                .sourceOrg(orgA())
                .build();
    }
}
