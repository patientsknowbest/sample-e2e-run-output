package com.pkb.datamodel;

import com.pkb.datamodel.team.BaseTeam;
import com.pkb.datamodel.team.ImmutableTeam;
import com.pkb.datamodel.team.ImmutableTeamWithOrg;
import com.pkb.datamodel.team.TeamWithOrg;

import java.util.UUID;

import static com.pkb.datamodel.OrgFactory.orgA;
import static com.pkb.datamodel.OrgFactory.orgB;

public class TeamFactory {
    private static final long TEAM_AZ_ID = 42;
    private static final UUID TEAM_AZ_PUBLIC_ID = UUID.fromString("91ac8681-4337-4c4d-a635-604719a90ef7");
    static final String TEAM_AZ_NAME = "Team AZ";
    private static final String TEAM_AZ_LOGO_PATH = "/team/az.png";
    private static final String TEAM_AZ_CODE = "TeamAZCode";
    private static final long TEAM_BZ_ID = 92;
    private static final String TEAM_BZ_NAME = "Team BZ";
    private static final String TEAM_BZ_LOGO_PATH = "/team/bz.png";
    private static final String TEAM_BZ_CODE = "TeamBZCode";

    private static final BaseTeam TEAM_AZ = ImmutableTeam.builder().id(TEAM_AZ_ID).name(TEAM_AZ_NAME).teamLogoPath(TEAM_AZ_LOGO_PATH)
            .publicId(TEAM_AZ_PUBLIC_ID).code(TEAM_AZ_CODE).build();

    private static final TeamWithOrg TEAM_Z_WITH_ORG_A = ImmutableTeamWithOrg.builder()
            .org(orgA())
            .name(TEAM_AZ_NAME)
            .teamLogoPath(TEAM_AZ_LOGO_PATH)
            .id(TEAM_AZ_ID)
            .publicId(TEAM_AZ_PUBLIC_ID)
            .code(TEAM_AZ_CODE)
            .build();

    private static final TeamWithOrg TEAM_Z_WITH_ORG_B = ImmutableTeamWithOrg.builder()
            .org(orgB())
            .name(TEAM_BZ_NAME)
            .teamLogoPath(TEAM_BZ_LOGO_PATH)
            .id(TEAM_BZ_ID)
            .publicId(UUID.randomUUID())
            .code(TEAM_BZ_CODE)
            .build();

    public static BaseTeam teamAz() {
        return TEAM_AZ;
    }

    public static TeamWithOrg teamZWithOrgA() {
        return TEAM_Z_WITH_ORG_A;
    }

    public static TeamWithOrg teamZWithOrgB() {
        return TEAM_Z_WITH_ORG_B;
    }
}
