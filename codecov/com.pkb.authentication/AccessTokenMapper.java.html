package com.pkb.authentication;

import io.vavr.control.Option;

import java.util.regex.Matcher;
import java.util.regex.Pattern;

import static com.pkb.authentication.ImmutableAccessToken.accessToken;
import static org.apache.commons.lang3.StringUtils.isNotBlank;
import static org.apache.commons.lang3.StringUtils.trimToEmpty;

public class AccessTokenMapper {
    private static final Pattern TOKEN_SECRET_MATCHER = Pattern.compile("(?<token>[a-zA-Z0-9._]{16})-(?<secret>[a-zA-Z0-9._]{15})");

    public boolean isWellFormedAccessToken(String value) {
        return isNotBlank(value) && canBeSplitIntoNonBlankParts(value);
    }

    public Option<AccessToken> toAccessToken(String value) {
        if (isWellFormedAccessToken(value)) {
            Matcher matcher = TOKEN_SECRET_MATCHER.matcher(value);
            if (!matcher.matches()) {
                throw new IllegalStateException(String.format("isWellFormedAccessToken returns true but TOKEN_SECRET_MATCHER didn't match input %s", value));
            }
            return Option.of(accessToken(trimToEmpty(matcher.group("token")).getBytes(), trimToEmpty(matcher.group("secret")).getBytes()));
        } else {
            return Option.none();
        }
    }
    
    private boolean canBeSplitIntoNonBlankParts(String value) {
        Matcher matcher = TOKEN_SECRET_MATCHER.matcher(value);
        return matcher.matches() &&
                isNotBlank(matcher.group("token")) &&
                isNotBlank(matcher.group("secret"));
    }
}
