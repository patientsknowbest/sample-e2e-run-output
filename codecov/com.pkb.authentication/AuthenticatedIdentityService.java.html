package com.pkb.authentication;

import com.pkb.authentication.principal.team.Team;
import com.pkb.authentication.principal.team.TeamWithOrgTeams;
import com.pkb.authentication.principal.user.AuthenticatedPerson;
import com.pkb.authentication.principal.user.AuthenticatedUser;
import com.pkb.authentication.projection.TeamIdsAndOrgIds;
import com.pkb.authentication.repository.AuthnDataRepository;
import com.pkb.datamodel.usertype.PatientType;
import com.pkb.datamodel.usertype.PersonType;
import com.pkb.datamodel.usertype.ProfessionalType;
import com.pkb.datamodel.usertype.TeamCoordinatorType;
import io.vavr.control.Option;

import java.util.UUID;

import static com.pkb.authentication.ImmutableAuthenticatedIdentity.authenticatedIdentity;
import static com.pkb.authentication.principal.client.ImmutableApiClient.apiClient;
import static com.pkb.authentication.principal.team.ImmutableTeam.team;
import static com.pkb.authentication.principal.team.ImmutableTeamWithOrgTeams.teamWithOrgTeams;
import static com.pkb.authentication.principal.user.ImmutableAuthenticatedIndividualProfessional.authenticatedIndividualProfessional;
import static com.pkb.authentication.principal.user.ImmutableAuthenticatedPatient.authenticatedPatient;
import static com.pkb.authentication.principal.user.ImmutableAuthenticatedTeamCoordinator.authenticatedTeamCoordinator;
import static com.pkb.authentication.principal.user.ImmutableAuthenticatedTeamProfessional.authenticatedTeamProfessional;
import static java.lang.String.format;

public class AuthenticatedIdentityService {
    private final AuthnDataRepository authnDataRepository;

    public AuthenticatedIdentityService(AuthnDataRepository authnDataRepository) {
        this.authnDataRepository = authnDataRepository;
    }

    public AuthenticatedIdentity getAuthenticatedIdentity(PrimaryPrincipalProperties properties) {
        return authenticatedIdentity()
                .authenticatedUser(buildAuthenticatedUser(properties.primaryUserType(), properties.internalPkbId(), properties.publicPkbId(),
                        properties.humanUuid(), properties.isFrozen()))
                .apiClient(apiClient(properties.clientApiId(), properties.clientApiScope()))
                .userKeyPair(properties.userKeyPair())
                .build();
    }

    private AuthenticatedUser buildAuthenticatedUser(Class<? extends PersonType> primaryUserType,
                                                          long privateId,
                                                          UUID publicId,
                                                          Option<UUID> humanUuid,
                                                          boolean isFrozen) {
        if (PatientType.class.isAssignableFrom(primaryUserType)) {
            return authenticatedPatient(publicId, privateId, humanUuid, isFrozen);
        } else if (ProfessionalType.class.isAssignableFrom(primaryUserType)) {
            return professionalPrincipal(privateId, publicId, humanUuid);
        } else if (TeamCoordinatorType.class.isAssignableFrom(primaryUserType)) {
            return teamCoordinator(privateId, publicId, humanUuid);
        } else {
            throw new IllegalStateException(format("Not managed User-type=[%s]", primaryUserType.getSimpleName()));
        }
    }

    /**
     * If a Professional is both Team Professional and Individual Professional, then Team professional takes precedence.
     */
    private AuthenticatedUser professionalPrincipal(long professionalsPrivateId,
                                                    UUID professionalPublicId,
                                                    Option<UUID> professionalHumanUuid) {
        Option<TeamWithOrgTeams> maybeProfessionalsTeam = findTeamWithOrgTeams(professionalsPrivateId);
        return maybeProfessionalsTeam
                .map(team -> (AuthenticatedPerson) authenticatedTeamProfessional(professionalPublicId, professionalsPrivateId, professionalHumanUuid, team))
                .getOrElse(() -> authenticatedIndividualProfessional(professionalPublicId, professionalsPrivateId, professionalHumanUuid));
    }

    private AuthenticatedUser teamCoordinator(long coordinatorsPrivateId, UUID coordinatorsPublicId, Option<UUID> humanUuid) {
        return findTeamWithOrgTeams(coordinatorsPrivateId)
                .map(team -> authenticatedTeamCoordinator(coordinatorsPublicId, coordinatorsPrivateId, humanUuid, team))
                .getOrElseThrow(() -> new IllegalStateException(format("Could not find team for team coordinator=[%s]", coordinatorsPublicId)));
    }

    private Option<TeamWithOrgTeams> findTeamWithOrgTeams(long personPrivateId) {
        var twits = authnDataRepository.fetchTeamIdsAndOrgIds(personPrivateId);
        // Turn the lot into Teams
        var teams = twits
                .map(twit -> (Team)team(twit.getTeamId(), twit.getTeamPublicId(), twit.getOrgId(), twit.getOrgPublicId()));
        // Find my team and make that the root one.
        return twits
                .find(TeamIdsAndOrgIds::getMyTeam)
                .map(twit -> teamWithOrgTeams(twit.getTeamId(), twit.getTeamPublicId(), twit.getOrgId(), twit.getOrgPublicId(), teams));
    }
}
