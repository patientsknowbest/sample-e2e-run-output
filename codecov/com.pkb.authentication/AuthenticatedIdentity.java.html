package com.pkb.authentication;

import com.google.common.base.Preconditions;
import com.pkb.authentication.principal.client.ApiClient;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authentication.principal.user.AuthenticatedUser;
import com.pkb.crypto.AccountKeyPair;
import com.pkb.crypto.AccountPrivateKey;
import com.pkb.crypto.UserKeyPair;
import com.pkb.crypto.UserPrivateKey;
import com.pkb.crypto.UserPublicKey;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.immutables.value.Value;
import org.immutables.vavr.encodings.VavrEncodingEnabled;

@Value.Immutable
@VavrEncodingEnabled
@Value.Style(builder = "authenticatedIdentity")
public interface AuthenticatedIdentity {
    AuthenticatedUser authenticatedUser();
    Option<ApiClient> apiClient();

    /**
     * Human users will have a UserP*Key associated.
     */ 
    Option<UserKeyPair> userKeyPair();
    
    default UserPrivateKey requireUserPrivateKey() {
        return userKeyPair()
                .map(UserKeyPair::userPrivateKey)
                .getOrElseThrow(() -> new IllegalStateException("userPrivateKey is not set!"));
    }
    default UserPublicKey requireUserPublicKey() {
        return userKeyPair()
                .map(UserKeyPair::userPublicKey)
                .getOrElseThrow(() -> new IllegalStateException("userPublicKey is not set!"));
    }

    /**
     * System users will have an AccountP*Key associated.
     */
    Option<AccountKeyPair> accountKeyPair();
    
    @Value.Check
    default void keyTypes() {
        if (authenticatedUser() instanceof AuthenticatedSystemUser) {
            Preconditions.checkState(accountKeyPair().isDefined(), "Authenticated user is system user, but account keys are not set!");
        } else {
            Preconditions.checkState(userKeyPair().isDefined(), "Authenticated user is human user, but user keys are not set!");
        }
    }

    default AccountPrivateKey requireAccountPrivateKey() {
        return accountKeyPair()
                .map(AccountKeyPair::accountPrivateKey)
                .getOrElseThrow(() -> new IllegalStateException("accountPrivateKey is not set!"));
    }
    
    default Either<UserPrivateKey, AccountPrivateKey> userOrAccountPrivateKey() {
        // one or the other must be set (as per @Check keyTypes) so this is safe.
        return accountKeyPair()
                .map(AccountKeyPair::accountPrivateKey)
                .toEither(this::requireUserPrivateKey);
    }
}
