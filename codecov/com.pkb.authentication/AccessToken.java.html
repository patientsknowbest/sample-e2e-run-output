package com.pkb.authentication;

import org.immutables.value.Value.Check;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Lazy;
import org.immutables.value.Value.Style;
import org.jetbrains.annotations.NotNull;

import static com.google.common.base.Preconditions.checkState;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

@Immutable
@Style(of = "accessToken", passAnnotations = { NotNull.class }, allParameters = true)
public interface AccessToken {
    byte[] token();

    byte[] secret();

    @Lazy
    default String tokenAsString() {
        return new String(token());
    }

    @Lazy
    default String secretAsString() {
        return new String(secret());
    }

    @Check
    default void check() {
        checkState(isNotBlank(new String(token())) && isNotBlank(new String(secret())), "Neither token nor secret can be blank.");
    }
}
