package com.pkb.entities.enums;

public enum PropertyName {

    CHILD13_BIRTHDAY_REMINDER("child13thBirthdayEmailSent"),
    CHILD16_BIRTHDAY_REMINDER("child16thBirthdayEmailSent"),
    CHILD18_BIRTHDAY_REMINDER("child18thBirthdayEmailSent"),

    CHILD13_CARER_REMOVED("child13CarerRemovalEmailSent"),
    CHILD16_CARER_REMOVED("child16CarerRemovalEmailSent"),
    CHILD18_CARER_REMOVED("child18CarerRemovalEmailSent"),

    CONSENT_MIGRATION_PENDING("consentMigrationPending"),
    CONSENT_REVIEW_COMPLETED("consentReviewCompleted"),
    EMAIL_NEWS_LETTER("emailNewsLetter"),
    INVITE_CLINICIAN_TERMS("inviteClinicianTerms");

    private final String value;

    PropertyName(String strValue) {
        value = strValue;
    }

    @Override
    public String toString() {
        return value;
    }

    public static PropertyName value(String strValue) {
        for (PropertyName name : PropertyName.values()) {
            if (name.value.equals(strValue)) {
                return name;
            }
        }
        throw new IllegalArgumentException(strValue);
    }

}
