package com.pkb.entities.enums;

import java.util.Optional;

/**
 *
 * @author Russ Gray <russ.gray@patientsknowbest.com>
 */
public enum AdditionalConsentId {

    CONTACT_BY_EMAIL(Section.CONTACTS, Optional.empty()),
    CONTACT_BY_SMS(Section.CONTACTS, Optional.empty()),
    CONTACT_BY_POST(Section.CONTACTS, Optional.empty()),
    CONTACT_FOR_RESEARCH(Section.IMPROVE_HEALTHCARE, Optional.ofNullable(ResearchScope.RECRUITING)),
    SAMPLE_STORAGE(Section.IMPROVE_HEALTHCARE, Optional.ofNullable(ResearchScope.SAMPLE_STORAGE));

    final private Section section;
    final private Optional<ResearchScope> enabledResearchScope;

    AdditionalConsentId(Section section, Optional<ResearchScope> enabledResearchScope) {
        this.section = section;
        this.enabledResearchScope = enabledResearchScope;
    }

    public boolean isContactRelated() {
        return section == Section.CONTACTS;
    }

    public boolean isResearchRelated() {
        return section == Section.IMPROVE_HEALTHCARE;
    }

    public Optional<ResearchScope> getEnabledResearchScope() { return enabledResearchScope; }

    public enum Section {
        CONTACTS, IMPROVE_HEALTHCARE
    }
}
