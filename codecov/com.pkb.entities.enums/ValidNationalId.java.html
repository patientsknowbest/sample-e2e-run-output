package com.pkb.entities.enums;

import org.immutables.value.Value;

import java.io.Serializable;

/**
 * Represents a validated national identifier, ONLY VALIDATED VALUES should ever used to create it.
 * Use {@link NationalIdType#getValidNationalIdAndType(String)} to validate value and type.
 */
@Value.Style(get = {"is*", "get*"}, visibility = Value.Style.ImplementationVisibility.PACKAGE, builderVisibility = Value.Style.BuilderVisibility.PACKAGE)
@Value.Immutable
public interface ValidNationalId extends HasNationalId, Serializable {

    @Value.Parameter
    String getValue();

    @Value.Parameter
    NationalIdType getType();


    default String value() {
        return getValue();
    }

    default NationalIdType type() {
        return getType();
    }

    @Override
    default NationalIdType nationalIdType() {
        return getType();
    }

    @Override
    default String nationalIdValue() {
        return getValue();
    }
}
