package com.pkb.entities.enums;

/**
 * Which PKB entity does the PublicFhirStoreSync represent as requiring syncing to the FHIR store?
 * Some of these are mapped 1:1 to FHIR resources, but many are collapsed together or split into multiple. See
 * {@link FhirStoreSyncResourceTarget} and {@link com.pkb.entities.pub.PublicFhirStoreSync} for more info!
 */
public enum FhirStoreSyncSourceTable {

    PERSONCONTACT(false),
    PKBPERSON(true),
    TEAM(true),
    ORG(true),
    NATIONALID(false),
    ORGLEVELID(false),
    TEAMLEVELID(false),
    INSTITUTEUSER(false),
    ORGEXTERNALID(false),
    PERSON_GUID(false),
    PATIENTOPTOUT(false),
    PATIENTCONSENT(false),
    MENU_DATA(false),
    PRIVACY_FLAGS(false),
    REFERENCEDATUM(true);


    /**
     * referencedResource means "does this table become a FHIR resource which is then pointed at by other resources?"
     * This determines if we need to single thread an entire batch which contains a CREATE or DELETE for them, as there
     * could be other resources which point at them.
     *
     * Note that some tables in PKB point at these tables, but that doesn't necessarily mean that this translates
     * over to FHIR.
     */
    private boolean referencedResource = false;

    FhirStoreSyncSourceTable(boolean referencedResource) {
        this.referencedResource = referencedResource;
    }

    public boolean isReferencedResource() {
        return referencedResource;
    }
}
