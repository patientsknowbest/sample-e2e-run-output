package com.pkb.entities.enums;

import java.util.EnumSet;
import java.util.Set;
import java.util.stream.Collectors;

import static java.util.Collections.unmodifiableSet;
import static java.util.EnumSet.complementOf;
import static java.util.EnumSet.of;
import static java.util.stream.Stream.concat;

public enum UserStatus {

    // See user status and roles description under: https://pkbdev.atlassian.net/wiki/spaces/TECHNOLOGY/pages/1118911/User+states+and+registration+stories
    NEW, // patient who has begun self-registration but not confirmed their email
    CREATED, // account created and temp password sent to user in registration email.
    INVITED, // PKBPerson created and invitation email sent, but account keys etc. not yet set up
    EMAIL_CONFIRMED, // registered, active user, possible for all users
    ACTIVE,
    INACTIVE,
    NOCONTACT, // replace with hasContact(), if possible - this status is for patients only. Record created; but no email provided,
               // the lack of email could be because we deleted it, or the data about the patient we received did not contain it
    DEAD; // patients only. Set by HL7 or (newly supported) webapp

    public static final Set<UserStatus> PATIENT_INVITED_NOT_REGISTERED;

    public static final Set<UserStatus> PATIENT_NOT_INVITED_NOT_REGISTERED;

    public static final Set<UserStatus> PATIENT_NOT_REGISTERED;

    public static final Set<UserStatus> PATIENTS_REGISTRATION_NOT_ALLOWED;

    static {
        PATIENT_INVITED_NOT_REGISTERED = unmodifiableSet(of(CREATED, INVITED, NEW));
        PATIENT_NOT_INVITED_NOT_REGISTERED = unmodifiableSet(of(NOCONTACT));
        PATIENT_NOT_REGISTERED = unmodifiableSet(concat(PATIENT_INVITED_NOT_REGISTERED.stream(), PATIENT_NOT_INVITED_NOT_REGISTERED.stream())
                .collect(Collectors.toCollection(() -> EnumSet.noneOf(UserStatus.class))));
        PATIENTS_REGISTRATION_NOT_ALLOWED = unmodifiableSet(complementOf(EnumSet.copyOf(PATIENT_NOT_REGISTERED)));
    }

}
