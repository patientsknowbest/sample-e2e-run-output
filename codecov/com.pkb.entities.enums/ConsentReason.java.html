package com.pkb.entities.enums;

import java.util.Arrays;

public enum ConsentReason {
    MIGRATED,
    CREATED_ACCOUNT,
    REFERRAL,
    POA, // Power Of Atto(u)rney
    COURT_ORDER,
    EXPLICIT_CONSENT,
    DISCHARGE,
    SAFEGUARDING,
    PATIENT_OPT_OUT,
    EXPIRED(false),
    ;

    private static final ConsentReason[] USER_SELECTABLE_VALUES = Arrays.stream(ConsentReason.values()).filter(ConsentReason::isUserSelectable).toArray(ConsentReason[]::new);

    private final boolean userSelectable;

    private ConsentReason() {
        this(true);
    }

    private ConsentReason(boolean userSelectable) {
        this.userSelectable = userSelectable;
    }

    public boolean isUserSelectable() {
        return this.userSelectable;
    }

    public static ConsentReason[] userSelectableValues() {
        return USER_SELECTABLE_VALUES;
    }
}
