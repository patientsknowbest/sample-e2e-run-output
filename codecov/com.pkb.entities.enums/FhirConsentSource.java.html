package com.pkb.entities.enums;

/**
 * Used when creating Consent resources for the FHIR store.
 * Indicates what the Consent resource actually represents.
 */
public enum FhirConsentSource {

    //Coming from the patientconsent table; indicates that the linked Organization has a consented relationship
    // with the Patient which may allow it to see zero or more privacy label types.
    PATIENTCONSENT,

    //Coming from the patientoptout table; indicates that the linked Patient is either currently, or has historically,
    // opted out of sharing their data. This applies to the entire Patient, not to any specific Organization.
    OPTOUT,

    //Coming from the instituteuser table; lots of overlap with patientconsent, but the sets of each differ slightly:
    // there are some instituteusers without patientconsents, and vice versa. This  indicates that the Patient has a
    // relationship with the linked Organization, which may or may not be consented. Most Patients have both this,
    // and the patientconsent Consent.
    INSTITUTEUSER
}
