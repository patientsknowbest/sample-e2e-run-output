    SSN_USA("SSN", "US", "USA", "NI", "http://hl7.org/fhir/sid/us-ssn") {
        @Override
        public String cleanInput(String id) {
            // strip out dashes
            return trimToNull(replaceChars(id, "-", ""));
        }

        @Override
        public boolean areUsedInSameCountry(NationalIdType type) {
            return NationalIdType.areUsedInSameCountry(SSN_USA, type);
        }

        @Override
        public Optional<NationalIdType> isValid(String cleanId) {
            // see http://en.wikipedia.org/wiki/Social_Security_number#Valid_SSNs
            // and http://ssa-custhelp.ssa.gov/app/answers/detail/a_id/425
            if (isNotBlank(cleanId) && isNumeric(cleanId) && (cleanId.length() == 9)) {
                // Invalid ranges and groupings:

                // Numbers with all zeros in any digit group (000-##-####, ###-00-####, ###-##-0000)
                if (cleanId.startsWith("000") || cleanId.substring(3, 5).equals("00") || cleanId.endsWith("0000")) {
                    return Optional.empty();
                }
                // Numbers of the form 666-##-####
                if (cleanId.startsWith("666")) {
                    return Optional.empty();
                }
                // Numbers in the 900 series
                if (cleanId.startsWith("9")) {
                    return Optional.empty();
                }
                // Hilda's number 078-05-1120
                if (cleanId.equals("078051120")) {
                    return Optional.empty();
                }

                // passed tests!
                return Optional.of(SSN_USA);
            }
            return Optional.empty();
        }

        @Override
        public String getCanonicalFormat(String id) {
            // 012-34-5678
            String clean = cleanInput(id);
            if (isEmpty(clean) || (clean.length() != 9)) {
                return id;
            }
            return clean.substring(0, 3) + "-" + clean.substring(3, 5) + "-" + clean.substring(5);
        }

            // NOTE: currently accepting all initial digits, though in reality 8 and 0 aren't used (so are invalid)
            if (isNotBlank(cleanId) && isNumeric(cleanId) && (cleanId.length() == 9)) {
                // make int array
                int[] multipliers = new int[]{1, 2, 1, 2, 1, 2, 1, 2, 1};
                int luhnRootSum = 0;
                for (int i = 0; i < 9; i++) {
                    int product = multipliers[i] * Character.digit(cleanId.charAt(i), 10);
                    if (product < 10) {
                        luhnRootSum += product;
                    } else {
                        luhnRootSum += (product - 10) + 1; // add the digits.. but cheat: max product is 18
                    }
                }

                if ((luhnRootSum % 10) == 0) {
                    return Optional.of(SIN_CA);
                }
            }
            return Optional.empty();

    HKID_HK("HKID", "HK", "HK", "NI", "http://fhir.patientsknowbest.com/id/hong-kong-identity-card-number") {
        @Override
        public String cleanInput(String id) {
            // strip out parens
            return trimToNull(replaceChars(id, "()", ""));
        }

        @Override
        public boolean areUsedInSameCountry(NationalIdType type) {
            return NationalIdType.areUsedInSameCountry(HKID_HK, type);
        }

        @Override
        public Optional<NationalIdType> isValid(String cleanId) {
            // see http://en.wikipedia.org/wiki/Hong_Kong_Identity_Card
            // and http://www.xml.gov.hk/en/approved/hkid_number_v1_0.htm
            // check scheme: http://quintus.roothome.net/Projects/_test/HKID.phps
            // another check scheme impl: http://www.biswaretech.com/blog22.aspx
            // they're all different! but this claims to source wikipedia
            //   http://kopanda.creativeworks.com.hk/2010/03/08/hkid-card-check-digit-validation-with-cakephp/
            // 1 or 2 alpha chars, then 6 digits, then check digit (0-9 or A) in parens
            // luhn check variant:
            // convert the first 1 or 2 alpha chars to digits like this:
            // - if first is missing, treat like space (char=32)
            // - otherwise: (ord(strtoupper($ch))-ord('A')+10);  -- huh. There's lots of disagreement on this!
            if (isBlank(cleanId)) {
                return Optional.empty(); // catch out null or empty before starting...
            }

            // uppercase and strip parens  (No -- they need to get it right at entry, for now, since that's going into the DB...)
            // id = StringUtils.replaceChars(id.toUpperCase(), "()", "");

            int[] ints = new int[8];

            if (cleanId.length() == 8) {
                cleanId = " " + cleanId; // make it 9 chars long...
            }
            if (cleanId.length() != 9) {
                return Optional.empty();
            }

            if (!isNumeric(cleanId.substring(2, 8))) {
                return Optional.empty();
            }

            char checkChar = cleanId.charAt(8); // last char
            ints[0] = charToInt(cleanId.charAt(0));
            ints[1] = charToInt(cleanId.charAt(1));
            for (int i = 2; i < 8; i++) {
                ints[i] = Character.digit(cleanId.charAt(i), 10);
            }

            // Okay!  We have a check character, and 8 ints for the mod 11 check.
            int mod11 = (ints[0] * 9) + (ints[1] * 8) + (ints[2] * 7) + (ints[3] * 6) + (ints[4] * 5) + (ints[5] * 4) + (ints[6] * 3)
                    + (ints[7] * 2);
            mod11 = 11 - (mod11 % 11);
            if (("" + mod11).equals(Character.toString(checkChar))
                    || ((mod11 == 11) && (checkChar == '0'))
                    || ((mod11 == 10) && (checkChar == 'A'))) {
                return Optional.of(HKID_HK);
            }
            return Optional.empty();
        }

        private int charToInt(char c) {
            // http://kopanda.creativeworks.com.hk/2010/03/08/hkid-card-check-digit-validation-with-cakephp/
            //        	if( c == ' ' )
            //        		return 0;
            //        	else
            //        		return (c - 'A')%11 + 1;

            // https://groups.google.com/forum/?hl=en&fromgroups=#!topic/hk.comp.pc/6FUkKtj_2YI
            if (c == ' ') {
                return 91;
            } else {
                return c;
            }

            // http://keatonchan.blogspot.fr/2009/08/hkid-check-digit-calculation-algorithm.html
            // c-'A' + 10, ' ' is 36
            // valid IDs for testing: B123456(6), P135143(9), A182361(5), CA182361(1), AB123456(9), ZA182361(3), AZ182361(6), XZ182361(8), ZX182361(6), XX182361(2), CA182367(0), CA182362(A)
        }

        @Override
        public String getCanonicalFormat(String id) {
            String clean = cleanInput(id);
            if (isEmpty(clean) || (clean.length() < 8)) {
                return id;
            }
            int lastCharId = clean.length() - 1;
            return clean.substring(0, lastCharId) + "(" + clean.substring(lastCharId) + ")";
        }
    },

    BSN_NL("BSN", "NL", "NL", "NI", "http://fhir.nl/fhir/NamingSystem/bsn") {
        @Override
        public String cleanInput(String id) {
            // strip out spaces
            return trimToNull(replaceChars(id, " ", ""));
        }

        @Override
        public boolean areUsedInSameCountry(NationalIdType type) {
            return NationalIdType.areUsedInSameCountry(BSN_NL, type);
        }

        @Override
        public Optional<NationalIdType> isValid(String cleanId) {
            // see http://nl.wikipedia.org/wiki/Burgerservicenummer  (translation widget at top of page)
            // NOTE: we're allowing the test range (900000000 to 999999999) but could disallow in some cases
            /*
             * if number is represented by ABCDEFGHI, then:
             * (9 x A) + (8 x B) + (C x 7) + (6 x D) + (5 x E) + (F x 4) + (G x 3) + (2 x H) + (- 1 x I) is a multiple of 11.
             * Valid examples are: 111222333 and 123 456 782.
             *
             * An 8-digit BSN (a relic from the time when the social security number was used) is used for monitoring should be provided with a leading zero.
             *
             * An easy way to generate test numbers: choose 4 random numbers (eg 9413)
             *   add the reverse of the selected number (reversed: 3149, thus: 94,133,149),
             *   and finally add a 0.  Result: 941,331,490.
             */
            if (isNotBlank(cleanId) && (cleanId.length() == 8)) {
                // pad with a 0 for checking, if it's only 8 digits
                cleanId = "0" + cleanId;
            }
            // now check it
            if (isNotBlank(cleanId) && isNumeric(cleanId) && (cleanId.length() == 9)) {
                // make int array
                int[] ints = new int[cleanId.length()];
                for (int i = 0; i < cleanId.length(); i++) {
                    ints[i] = Character.digit(cleanId.charAt(i), 10);
                }
                // modified modulus 11 check
                int mod11 = (ints[0] * 9) + (ints[1] * 8) + (ints[2] * 7) + (ints[3] * 6) + (ints[4] * 5) + (ints[5] * 4) + (ints[6] * 3)
                        + (ints[7] * 2) + (ints[8] * (-1));
                mod11 = (mod11 % 11);
                if (mod11 == 0) {
                    return Optional.of(BSN_NL);
                }
            }
            return Optional.empty();
        }

        @Override
        public String getCanonicalFormat(String id) {
            String clean = cleanInput(id);
            if (isEmpty(clean) || (clean.length() != 9)) {
                return id;
            }
            return clean.substring(0, 3) + " " + clean.substring(3, 6) + " " + clean.substring(6);
        }
    },

    /**
     * Source: Maybe: http://oid-info.com/get/2.16.840.1.113883.2.6 or http://fhir.de/NamingSystem/gkv/kvnr-30
     */
    KVN_DE("Krankenversichertennummer", "DE", "DE", "NH", "http://fhir.de/NamingSystem/gkv/kvnr-30") {
        @Override
        public String cleanInput(String id) {
            return Optional.ofNullable(id)
                    .map(String::toUpperCase)
                    .map(uppercaseId -> replaceChars(uppercaseId, " ", ""))
                    .map(StringUtils::trimToNull)
                    .orElse(null);
            // strip out spaces and make sure alpha char is uppercase
        }

        @Override
        public boolean areUsedInSameCountry(NationalIdType type) {
            return NationalIdType.areUsedInSameCountry(KVN_DE, type);
        }

        @Override
        public Optional<NationalIdType> isValid(String cleanId) {
            // see JIRA PHR-857 and PDF: ftp://ftp.kbv.de/ita-update/Abrechnung/KBV_ITA_VGEX_Datensatzbeschreibung_KVDT.pdf
            // starts with alpha, then 9 numeric digits (last is the check digit)
            // convert alpha to 2-digit number (e.g., A=01, B=02, etc.
            // multiply each second digit by 2, sum any multiple digits, then sum all those results and %10: result = check digit
            if (isNotBlank(cleanId) && (cleanId.length() == 10) && StringUtils.isAlpha(cleanId.substring(0, 1))
                    && isNumeric(cleanId.substring(1, 10))) {
                // convert alpha letter into 2-digit string
                int alphaSequence = (cleanId.charAt(0) - 'A') + 1; // 1-based counting
                String alphaDigits = (alphaSequence < 10) ? "0" + alphaSequence : "" + alphaSequence;
                int checkDigit = Character.digit(cleanId.charAt(9), 10);
                String luhnInput = alphaDigits + cleanId.substring(1, 9); // skipping prefix alpha and suffix check digit

                // make int array
                int[] multipliers = new int[]{1, 2, 1, 2, 1, 2, 1, 2, 1, 2};
                int luhnRootSum = 0;
                for (int i = 0; i < 10; i++) {
                    int product = multipliers[i] * Character.digit(luhnInput.charAt(i), 10);
                    if (product < 10) {
                        luhnRootSum += product;
                    } else {
                        luhnRootSum += (product - 10) + 1; // add the digits.. but cheat: max product is 18
                    }
                }

                if ((luhnRootSum % 10) == checkDigit) {
                    return Optional.of(KVN_DE);
                }
            }
            return Optional.empty();
        }

        @Override
        public String getCanonicalFormat(String id) {
            return cleanInput(id); // no standard formatting that I know of!  Just the clean input, no spaces.
        }
    },

    CIVIL_ID_KW("Civil ID", "KW", "KW", "NI", "http://fhir.patientsknowbest.com/id/kuwait-civil-id") {
        // http://dr-mahbob.com/blog/2008/12/23/civil-id-equation-for-check-digit/
        // Sample code from Django: https://github.com/django/django-localflavor/blob/master/localflavor/kw/forms.py
        // sample numbers:
        // 273032401586  -- this seems to be a common sample number
        // 288090100173
        // 233122100058
        // 286090401364   DoB: September 4 1986
        @Override
        public String cleanInput(String id) {
            // strip out dashes or spaces
            return trimToNull(replaceChars(id, " -", ""));
        }

        @Override
        public boolean areUsedInSameCountry(NationalIdType type) {
            return NationalIdType.areUsedInSameCountry(CIVIL_ID_KW, type);
        }

        @Override
        public Optional<NationalIdType> isValid(String cleanId) {
            //   http://dr-mahbob.com/blog/2008/12/23/civil-id-equation-for-check-digit/
            //   Sample code from Django: https://github.com/django/django-localflavor/blob/master/localflavor/kw/forms.py
            // Validation:
            // - 12 digits
            // - Digit 1 ...may... be a century digit -- "2" for pre-2000 DoB years, and "3" for year 2000+ DoB.  Not confirmed, so not yet validated.
            // - Digits 2-7 are the patients DOB in YYMMDD format
            //      (so the first 2 can be anything, next 2 must be 01-12 and last 2 must be 01-31)
            // - Digits 8-11 are a sequence number (anything)
            // - Digit 12 is a check digit.
            // - Calc: 11 - Mod(((C1 * 2) + (C2 * 1)  + (C3 * 6) + (C4 * 3) + (C5 * 7) +
            //                   (C6 * 9) + (C7 * 10) + (C8 * 5) + (C9 * 8) + (C10 * 4) + (C11 * 2)),11) = C12

            if (isNotBlank(cleanId) && isNumeric(cleanId) && (cleanId.length() == 12)) {
                String monthChars = cleanId.substring(3, 5);
                if (monthChars.startsWith("0")) {
                    monthChars = monthChars.substring(1);
                }
                int monthInt = Integer.parseInt(monthChars);
                if ((monthInt < 1) || (monthInt > 12)) {
                    return Optional.empty();
                }

                String dayChars = cleanId.substring(5, 7);
                if (dayChars.startsWith("0")) {
                    dayChars = dayChars.substring(1);
                }
                int dayInt = Integer.parseInt(dayChars);
                if ((dayInt < 1) || (dayInt > 31)) {
                    return Optional.empty();
                }

                // make int array
                int[] ints = new int[cleanId.length()];
                for (int i = 0; i < cleanId.length(); i++) {
                    ints[i] = Character.digit(cleanId.charAt(i), 10);
                }
                // modulus 11 check (with randomly-chosen multipliers...?)
                // (C1 * 2) + (C2 * 1)  + (C3 * 6) + (C4 * 3) + (C5 * 7) + (C6 * 9) + (C7 * 10) + (C8 * 5) + (C9 * 8) + (C10 * 4) + (C11 * 2)
                int mod11 = (ints[0] * 2) + ints[1] + (ints[2] * 6) + (ints[3] * 3) + (ints[4] * 7) + (ints[5] * 9) + (ints[6] * 10)
                        + (ints[7] * 5) + (ints[8] * 8) + (ints[9] * 4) + (ints[10] * 2);
                mod11 = 11 - (mod11 % 11);
                if ((mod11 == ints[11])
                        || ((mod11 == 11) && (ints[11] == 0))) {
                    return Optional.of(CIVIL_ID_KW);
                }
            }
            return Optional.empty();
        }

        @Override
        public String getCanonicalFormat(String id) {
            return cleanInput(id);
        }
    };

    /**
     * Returns a validated national id type and value pair, as they only make sense as a pair.
     *
     * <b>Only validated {@link NationalIdType}s should be used for anything in the code</b> as that might differ from
     * the value used for validation! Eg. a {@link NationalIdType#NHS_NUMBER} might change into a {@link NationalIdType#CHI_NUMBER}
     * when the value provided is actually a {@link NationalIdType#CHI_NUMBER}.
     *
     * @param nationalId national id to validate
     * @return {@link Optional} {@link ValidNationalId} when the provided value passed validation, {@link Optional#empty()} otherwise.
     */
    @NotNull
    public Optional<ValidNationalId> getValidNationalIdAndType(String nationalId) {
        String cleanNid = cleanInput(nationalId);
        return isValid(cleanNid).map(nidt -> ImmutableValidNationalId.of(cleanNid, nidt));
    }

    /**
     * This method was added for backward compatibility, only ever use it for values read directly from the database.
     * At this moment we still have records which contain invalid national id type, so cannot use {@link NationalIdType#getValidNationalIdAndType} to construct the valid object.
     * <p>
     * DO NOT USE THIS TO CONSTRUCT {@link ValidNationalId} FROM ANY OTHER INPUT (USER / SERVICE / IMPORT / SYSTEM ETC)!!!!
     * DO NOT EVER ASSUME A SYSTEM CORRECTLY TELLS US THE NATIONAL ID TYPE!
     * There are customers who does not make the distinction between {@link NationalIdType#NHS_NUMBER},{@link NationalIdType#CHI_NUMBER} ,{@link NationalIdType#H_AND_C_NUMBER} as we do,
     * but simply call it {@link NationalIdType#NHS_NUMBER}, and this can cause many issues.
     *
     * @param type
     * @param value
     * @return Not validated, possibly invalid {@link ValidNationalId}.
     * @deprecated Deprecated to prevent unintended usage, and to make it very visible on PRs. Use {@link NationalIdType#getValidNationalIdAndType} instead.
     */
    @Deprecated
    public static ValidNationalId deprecated_buildFromDatabaseValues(NationalIdType type, String value) {
        return ImmutableValidNationalId.of(value, type);
    }

     * cleanInput() on the value to be checked.
     * Might return a different {@link NationalIdType} (different value from same "family") than the one used.
     * Examples {@link NationalIdType#NHS_NUMBER} / {@link NationalIdType#CHI_NUMBER}
     *
     * @param cleanId The result of calling cleanInput() on the value to be checked.
     * @return the valid {@link NationalIdType} or {@link Optional#empty()} otherwise.

    private static final ImmutableMap<String, NationalIdType> PRIMARY_NID_BY_COUNTRY_LOOKUP = logWrapped("init PRIMARY_NID_BY_COUNTRY_LOOKUP", () ->
            Arrays.stream(NationalIdType.values())
                    .flatMap(nidt -> nidt.primaryCountryCodes.stream().map(pcc -> Tuple.of(pcc, nidt)))

    /**
     * Some countries like the UK has "multiple" national identifiers in use as the NHS number is split into different ranges and different parts of UK use different ranges of it.
     * Eg. {@link NationalIdType#NHS_NUMBER} & {@link NationalIdType#CHI_NUMBER} & {@link NationalIdType#H_AND_C_NUMBER}
     * This method returns the primary type in use.
     *
     * @param countryCode
     * @return
     */
    @NotNull
    public static Optional<NationalIdType> getPrimaryNationalIdTypeFromCountryCode(String countryCode) {
        if (isBlank(countryCode)) {
            return Optional.empty();
        }
        return Optional.ofNullable(PRIMARY_NID_BY_COUNTRY_LOOKUP.get(countryCode));
    }

    private static final ImmutableMap<String, Set<NationalIdType>> NID_BY_COUNTRY_LOOKUP = logWrapped("init NID_BY_COUNTRY_LOOKUP", NationalIdType::buildNidByCountryMap);

    private static ImmutableMap<String, Set<NationalIdType>> buildNidByCountryMap() {
        Map<String, Set<NationalIdType>> groupedByCountry = new HashMap<>();
        for (NationalIdType actualNationalId : NationalIdType.values()) {
            for (String actualCountry : actualNationalId.getCountryCodes()) {
                groupedByCountry.computeIfAbsent(actualCountry, k -> EnumSet.noneOf(NationalIdType.class)).add(actualNationalId);
            }
        }
        ImmutableMap.Builder builder = ImmutableMap.builder();
        groupedByCountry.forEach((s, nids) -> builder.put(s, Collections.unmodifiableSet(nids)));
        return builder.build();
    }

    /**
     * This method returns every {@link NationalIdType} used in the given country.
     * Eg. {@link NationalIdType#NHS_NUMBER} & {@link NationalIdType#CHI_NUMBER} & {@link NationalIdType#H_AND_C_NUMBER} for UK.
     *
     * @param countryCode
     * @return
     */
    @NotNull
    public static Set<NationalIdType> getNationalIdTypesFromCountryCode(String countryCode) {
        if (isBlank(countryCode)) {
            return emptySet();
        }
        return NID_BY_COUNTRY_LOOKUP.getOrDefault(countryCode, emptySet());
    }

    private static final ImmutableMap<NationalIdType, Set<NationalIdType>> USED_TOGETHER_LOOKUP = logWrapped("init USED_TOGETHER_LOOKUP", () -> calculateUsedTogetherLookup());

    private static ImmutableMap<NationalIdType, Set<NationalIdType>> calculateUsedTogetherLookup() {
        Map<NationalIdType, Set<NationalIdType>> groupedByNationalId = new EnumMap<>(NationalIdType.class);
        for (NationalIdType actualNationalId : NationalIdType.values()) {
            for (String actualCountry : actualNationalId.primaryCountryCodes) {
                groupedByNationalId.computeIfAbsent(actualNationalId, k -> EnumSet.noneOf(NationalIdType.class)).addAll(NationalIdType.getNationalIdTypesFromCountryCode(actualCountry));
            }
        }
        EnumMap<NationalIdType, Set<NationalIdType>> builder = new EnumMap<>(NationalIdType.class);
        groupedByNationalId.forEach((nid, nids) -> builder.put(nid, Collections.unmodifiableSet(nids)));
        return Maps.immutableEnumMap(builder);
    }

    public static boolean areUsedInSameCountry(NationalIdType type1, NationalIdType type2) {
        return USED_TOGETHER_LOOKUP.get(type1).contains(type2);
    }

    @NotNull
    public static Optional<NationalIdType> getNationalIdTypeFromFhirIdentifierSystem(String fhirIdentifierSystem) {
        if (isBlank(fhirIdentifierSystem)) {
            return Optional.empty();
        }
        for (NationalIdType nit : values()) {
            if (fhirIdentifierSystem.equals(nit.fhirIdentifierSystem)) {
                return Optional.of(nit);
            }
        }
        return Optional.empty();
    }
