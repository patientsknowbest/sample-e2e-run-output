package com.pkb.entities.enums;

import java.util.Arrays;
import java.util.Collection;
import java.util.EnumSet;
import java.util.stream.Collectors;

public enum PrivacyFlag {
    GENERAL("GENERAL_HEALTH", 1),
    SEXUAL_HEALTH("SEXUAL_HEALTH",  2),
    MENTAL_HEALTH("MENTAL_HEALTH",  4),
    SOCIAL_CARE("SOCIAL_CARE", 8);

    private final String externalDesc;
    private final int bitFlag;

    PrivacyFlag(String externalDesc, int bitFlag) {
        this.externalDesc = externalDesc;
        this.bitFlag = bitFlag;
    }

    public String getExternalDesc() {
        return externalDesc;
    }

    private int getBitFlag() {
        return bitFlag;
    }

    public static int toBitFlags(Collection<PrivacyFlag> privacyFlags) {
        return privacyFlags.stream().map(PrivacyFlag::getBitFlag).reduce(0, (a,b) -> a | b);
    }

    public static EnumSet<PrivacyFlag> fromBitFlags(int bitFlags) {
        return Arrays.stream(values()).filter(e -> (e.bitFlag & bitFlags) != 0).collect(Collectors.toCollection(() -> EnumSet.noneOf(PrivacyFlag.class)));
    }
}
