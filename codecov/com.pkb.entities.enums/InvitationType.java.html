package com.pkb.entities.enums;

public enum InvitationType {
    PATIENT_INVITATION("Invitation to patient"),
    CLINICIAN_INVITATION("Invitation to clinician"),
    EMPLOYEE_INVITATION("Invitation to employee"),
    PATIENT_INVITATION_AS_CAREGIVER("Invitation to patient as caregiver"),
    CLINICIAN_INVITATION_TO_UPGRADE_ACCOUNT("Invitation to clinician to upgrade account"),
    PATIENT_INVITATION_TO_JOIN_TEAM("Invitation to patient to join team"),
    NULL(null);

    public final String description;

    InvitationType(String description) {
        this.description = description;
    }

    public static InvitationType invitationConverter(String invitationTitle) {
        if (invitationTitle.contains(CLINICIAN_INVITATION_TO_UPGRADE_ACCOUNT.description)) { return CLINICIAN_INVITATION_TO_UPGRADE_ACCOUNT; }
        if (invitationTitle.contains(PATIENT_INVITATION_TO_JOIN_TEAM.description)) { return PATIENT_INVITATION_TO_JOIN_TEAM; }
        if (invitationTitle.toLowerCase().contains(PATIENT_INVITATION_AS_CAREGIVER.description.toLowerCase()))
        {
            return PATIENT_INVITATION_AS_CAREGIVER;
            // needs to transform to lower case, becasuse in the legacy code we concatenated " as CAREGIVER" to "Invitation to patient"
            // this is for to be able to convert "Invitation to patient as CAREGIVER" to enum PATIENT_INVITATION_AS_CAREGIVER and save new records
            // with normal letter case like "Invitation to patient as caregiver as caregiver"
        }
        if (invitationTitle.contains(PATIENT_INVITATION.description)) { return PATIENT_INVITATION; }
        if (invitationTitle.contains(EMPLOYEE_INVITATION.description)) { return EMPLOYEE_INVITATION; }
        if (invitationTitle.contains(CLINICIAN_INVITATION.description)) { return CLINICIAN_INVITATION; }
        else { return NULL; }
    }


}
