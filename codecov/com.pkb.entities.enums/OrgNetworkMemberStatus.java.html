package com.pkb.entities.enums;

public enum OrgNetworkMemberStatus {

    /**
     * State for an org that has been added to a network by PKB admin,
     * and is waiting for the org coord of that org to approve
     */
    PENDING_APPROVAL,
    /**
     * State for an org who's coord has approved addition to a network,
     * and is waiting for the network leader to activate
     */
    PENDING_ACTIVATION,

    /**
     * Activation process is currently running, shouldn't be run again unless this one
     * fails
     */
    ACTIVATING,
    /**
     * State for full activated membership of a network
     */
    ACTIVE,
}
