package com.pkb.entities.enums;

public enum AccountUserStatus {
    ACTIVE("active"), INACTIVE("inactive"), EXPIRED("obsolete");

    private final String value;

    AccountUserStatus(String strValue) {
        value = strValue;
    }

    @Override
    public String toString() {
        return value;
    }

    public static AccountUserStatus value(String strValue) {
        for (AccountUserStatus status : AccountUserStatus.values()) {
            if (status.value.equals(strValue)) {
                return status;
            }
        }
        throw new IllegalArgumentException(strValue);
    }
}
