public abstract class MassUpdatableEhrDataPointManager<T extends EncryptedDataPoint & IBaseDTO & EqualityKeySupport> extends EhrDataPointManager<T> {

    public MassUpdatableEhrDataPointManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, CodingManager codingManager, DeduplicatorManager deduplicator, Class<T> dtoClass, MenuDataType dataType) {
        super(config, beanFactory, dateTimeService, uuidProvider, codingManager, deduplicator, dtoClass, dataType);
    }

    public DataPointSaveResult<T> updateDataPointsForOrg(@NotNull LoggedInEHRRequestContext requestContext,
                                                         java.util.List<T> dataPointList, long orgId, boolean deleteAbsent, SourceDetails sourceDetails) {
        if (dataPointList.isEmpty()) {
            return emptySaveResult();
        }

        EHRRemote ehrBean = beanFactory.getEhrRemote();

        Long patientAccountId = dataPointList.get(0).getPatientAccountId();

        EHRSearch<T> ehrSearch = new EHRSearch<>(patientAccountId, dtoClass, dataType);
        ehrSearch.setLatestPerTypeFilter(EHRData.ENTERED_DATE, EHRData.UNIQUE_ID);
        ehrSearch.addFilter(new PKBFilter(SourceDetails.EHRQUERY_ORG_ID, PKBFilter.Operator.EQUAL, orgId));


        List<T> existingDataPoints = List.ofAll(ehrBean.queryDecryptAndPopulateData(ehrSearch, dtoClass, requestContext));
        return diffLists(List.ofAll(dataPointList), existingDataPoints)
                .apply((updates, additions, deletions) -> {
                    updates.map(update -> update.apply((newDataPoint, existingDataPoint) -> {
                                newDataPoint.setId(null);
                                newDataPoint.getBaseFields().setUniqueId(existingDataPoint.getBaseFields().getUniqueId());
                                newDataPoint.getBaseFields().setPersistedDate(Date.from(dateTimeService.now()));
                                return newDataPoint;
                            }))
                            .forEach(dataPoint -> update(requestContext, dataPoint));

                    if (deleteAbsent) {
                        deleteByUniqueIds(requestContext, patientAccountId, deletions.map(dto -> dto.getBaseFields().getUniqueId()).collect(Collectors.toList()), sourceDetails,
                                null);
                    }
                    return save(requestContext, additions.toJavaList());
                });
    }

    private <V extends EqualityKeySupport> Tuple3<Seq<Tuple2<V, V>>, Seq<V>, Seq<V>> diffLists(List<V> newList, List<V> existingList) {
        if (existingList.isEmpty()) {
            return Tuple.of(List.empty(), newList, List.empty());
        }

        Function<V, Tuple> keyFunction = V::equalityKey;
        Map<Tuple, V> newMap = newList.toLinkedMap(keyFunction, Function.identity());
        if (newMap.size() != newList.size()) {
            throw new IllegalStateException("Datapoint list for type {" + dataType + "} contained duplicate entries");
        }

        Map<Tuple, V> existingMap = existingList.toLinkedMap(keyFunction, Function.identity());
        List<V> existingDuplicates = existingList.reject(dataPoint -> dataPoint == existingMap.get(keyFunction.apply(dataPoint)).get());

        return existingMap
                .partition(entry -> newMap.containsKey(entry._1))
                .apply((intersectionExistingValues, existingOnly) -> Tuple
                        .of(intersectionExistingValues, newMap.rejectKeys(intersectionExistingValues::containsKey).values(), existingOnly.values().appendAll(existingDuplicates)))
                .map1(intersectionExistingValues -> intersectionExistingValues.map(entry -> entry.map1(key -> newMap.get(key).get())));
    }
}
