package com.pkb.core.entity;

import com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer;
import org.hibernate.annotations.GenericGenerator;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.tolven.security.key.AccountPrivateKey;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.persistence.Temporal;
import javax.persistence.TemporalType;
import java.io.Serializable;
import java.util.Date;

/**
 * Specifies which org(s) have decryption access for a given account, and
 * holds the account private keys (encrypted to org accounts)
 *
 * Based on Tolven's AccountUser object
 *
 * @author robwhelan
 */
@Entity
@Table(schema = "core", name = "account_org")
public class AccountOrg implements Serializable, Cloneable {

    private static final long serialVersionUID = 1L;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Column(name = "account_id", nullable = false, updatable = false)
    private Long accountId;

    @Column(name = "org_id", nullable = false, updatable = false)
    private Long orgId;

    @Id
    @GenericGenerator(name = "seq_ao_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "core"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_accountorg"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "seq_ao_gen")
    private Long id;

    @Embedded
    private AccountPrivateKey accountPrivateKey;

    @Column(nullable = true)
    @Temporal(TemporalType.TIMESTAMP)
    private Date created;

    @Column(name = "sourceorg_id", nullable = false, updatable = false)
    private Long sourceOrgId;

    public AccountOrg() {}

    /**
     * The account to which this user is associated. By this means, a person can be associated with any number of Accounts and will
     * normally be
     * associated with only one at a time while logged in. A person with more than one account will be provided a selection of which
     * account they
     * want to be connected to as they login.
     */
    public long getAccountId() {
        return accountId;
    }

    public void setAccountId(long accountId) {
        this.accountId = accountId;
    }

    /**
     * The Org to be granted decryption access to the specified Account.
     */
    public long getOrgId() {
        return orgId;
    }

    public void setOrgId(long orgId) {
        this.orgId = orgId;
    }

    public long getId() {
        return id;
    }

    public void setId(long val) {
        this.id = val;
    }

    @Override
    public boolean equals(Object obj) {
        if (!(obj instanceof AccountOrg)) {
            return false;
        }
        return this.getId() == ((AccountOrg) obj).getId();
    }

    @Override
    public String toString() {
        return "AccountOrg: " + getId();
    }

    @Override
    public int hashCode() {
        if (getId() == 0) {
            throw new IllegalStateException("id not yet established in AccountOrg object");
        }
        return Long.valueOf(getId()).hashCode();
    }

    public AccountPrivateKey getAccountPrivateKey() {
        return accountPrivateKey;
    }

    public void setAccountPrivateKey(AccountPrivateKey privateKey) {
        this.accountPrivateKey = privateKey;
    }

    public boolean hasAccountPrivateKey() {
        return accountPrivateKey != null;
    }

    public Date getCreated() {
        return created;
    }

    public void setCreated(Date created) {
        this.created = created;
    }

    public long getSourceOrgId() {
        return sourceOrgId;
    }

    public void setSourceOrgId(long sourceOrgId) {
        this.sourceOrgId = sourceOrgId;
    }
}
