package com.pkb.core.entity;

import com.pkb.entities.enums.core.EncryptionAlgorithm;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.util.UUID;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

/**
 * This is a duplicate of the class of the same name from commons-hibernate-entities.
 * This is created, so both old and new service beans can use datapoint level encryption.
 */
@Entity
@Table(name = "datapoint_symmetric_key", schema = "core")
public class DatapointSymmetricKey {

    @Id
    @GenericGenerator(name = "datapoint_symmetric_key_id_seq_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "core"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "datapoint_symmetric_key_id_seq"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "datapoint_symmetric_key_id_seq_gen")
    @Column(name = "id", nullable = false)
    private long id;

    @Column(name = "key_id", nullable = false)
    private UUID keyId;

    @Column(name = "account_id", nullable = false)
    private long accountId;

    @Column(name = "encrypted_key", nullable = false)
    private byte[] encryptedKey;

    @Column(name = "encryption_algorithm", nullable = false)
    @Enumerated(EnumType.STRING)
    @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
    private EncryptionAlgorithm encryptionAlgorithm;

    @Column(name = "account_level_key_id", nullable = false)
    private long accountLevelKeyId;

    @Column(name = "nonce", nullable = false)
    private byte[] nonce;

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public UUID getKeyId() {
        return keyId;
    }

    public void setKeyId(UUID keyId) {
        this.keyId = keyId;
    }

    public long getAccountId() {
        return accountId;
    }

    public void setAccountId(long accountId) {
        this.accountId = accountId;
    }

    public byte[] getEncryptedKey() {
        return encryptedKey;
    }

    public void setEncryptedKey(byte[] encryptedKey) {
        this.encryptedKey = encryptedKey;
    }

    public EncryptionAlgorithm getEncryptionAlgorithm() {
        return encryptionAlgorithm;
    }

    public void setEncryptionAlgorithm(EncryptionAlgorithm encryptionAlgorithm) {
        this.encryptionAlgorithm = encryptionAlgorithm;
    }

    public long getAccountLevelKeyId() {
        return accountLevelKeyId;
    }

    public void setAccountLevelKeyId(long accountLevelKeyId) {
        this.accountLevelKeyId = accountLevelKeyId;
    }

    public byte[] getNonce() {
        return nonce;
    }

    public void setNonce(byte[] nonce) {
        this.nonce = nonce;
    }
}
