package com.pkb.core.entity;

/**
 * Result of granting org access to a patient account
 *
 * @author Russ Gray
 */
public enum OrgAccessResult {
    /**
     * An active link already exists (no-op)
     */
    NO_ACTION_TAKEN,

    /**
     * A new account_org record was created
     */
    LINK_CREATED,

    /**
     * An error occurred
     */
    ERROR;

    public boolean isChanged() {
        // Any result other than NO_ACTION_TAKEN or ERROR is considered a change
        return !(NO_ACTION_TAKEN == this || ERROR == this);
    }

    public boolean isSuccess() {
        // Any non-error is a success
        return ERROR != this;
    }
}
