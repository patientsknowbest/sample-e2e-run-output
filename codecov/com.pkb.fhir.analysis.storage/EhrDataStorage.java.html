package com.pkb.fhir.analysis.storage;

import com.pkb.fhir.analysis.persistence.targetdb.repository.DecryptedEhrDataRepository;
import com.pkb.fhir.common.decrypt.DecryptedEhrData;
import com.pkb.fhir.analysis.persistence.targetdb.DecryptedEhrDataEntity;

import java.util.List;
import java.util.stream.Collectors;

public class EhrDataStorage {
    private DecryptedEhrDataRepository decryptedEhrDataRepository;

    public EhrDataStorage(DecryptedEhrDataRepository decryptedEhrDataRepository) {
        this.decryptedEhrDataRepository = decryptedEhrDataRepository;
    }

    public void storeDecryptedEhrData(List<DecryptedEhrData> ehrData) {
        decryptedEhrDataRepository.saveAll(
                ehrData.stream()
                        .map(data -> new DecryptedEhrDataEntity(data.getId(), data.getContent(), data.getStatus(), data.getDataType()))
                        .collect(Collectors.toList()));


    }


}
