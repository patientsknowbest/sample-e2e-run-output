package com.pkb.aggregator.integration.aidbox.configuration

import ca.uhn.fhir.context.FhirContext
import com.fasterxml.jackson.databind.ObjectMapper
import com.pkb.aggregator.configuration.FhirConfiguration
import com.pkb.aggregator.integration.aidbox.AidboxRestClient
import com.pkb.aggregator.integration.aidbox.HapiAidboxRestClient
import com.pkb.aggregator.integration.aidbox.SqlHardDeleter
import com.pkb.aggregator.integration.retrofit.HapiConverterFactory
import okhttp3.OkHttpClient
import org.keycloak.admin.client.Keycloak
import org.springframework.beans.factory.annotation.Qualifier
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import retrofit2.Retrofit
import retrofit2.converter.jackson.JacksonConverterFactory
import retrofit2.create

@Configuration
class AidboxConfiguration {

    @Bean
    fun aidboxRestClient(
        client: OkHttpClient,
        objectMapper: ObjectMapper,
        fhirConfiguration: FhirConfiguration,
        keycloak: Keycloak?,
    ): AidboxRestClient = Retrofit.Builder()
        .client(client)
        .baseUrl(fhirConfiguration.downstream.endpoint.serverUrl)
        .addConverterFactory(JacksonConverterFactory.create(objectMapper))
        .build()
        .create()

    @Bean
    fun hapiAidboxClient(
        client: OkHttpClient,
        fhirConfiguration: FhirConfiguration,
        @Qualifier("fhirContextR4") fhirContextR4: FhirContext,
        keycloak: Keycloak?,
    ): HapiAidboxRestClient = Retrofit.Builder()
        .client(client)
        .baseUrl(fhirConfiguration.downstream.endpoint.serverUrl)
        .addConverterFactory(HapiConverterFactory.create(fhirContextR4))
        .build()
        .create()

    @Bean
    fun downstreamDeleter(
        aidboxRestClient: AidboxRestClient,
        objectMapper: ObjectMapper,
        fhirConfiguration: FhirConfiguration,
    ) = SqlHardDeleter(aidboxRestClient, objectMapper)
}
