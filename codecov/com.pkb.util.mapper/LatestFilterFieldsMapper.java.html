package com.pkb.util.mapper;

import com.pkb.app.entity.EHRSearch;
import com.pkb.ehrdata.ImmutableLatestFilterFields;
import com.pkb.ehrdata.LatestFilterFields;
import com.pkb.ehrdata.OrderBy;
import org.apache.commons.lang3.StringUtils;
import org.mapstruct.Mapper;

import java.util.Optional;

@Mapper(componentModel = "cdi")
public interface LatestFilterFieldsMapper {

    default Optional<LatestFilterFields> map(EHRSearch<?> search) {
        if (StringUtils.isBlank(search.getLatestFilterTypeField())) {
            return Optional.empty();
        }

        OrderBy.Direction orderBy = OrderBy.Direction.Desc;
        if (search.getLatestFilterOrderBy() != null) {
            switch (search.getLatestFilterOrderBy()) {
                case Asc:
                    orderBy = OrderBy.Direction.Asc;
                    break;
                default:
                    orderBy = OrderBy.Direction.Desc;
            }
        }

        return Optional.of(ImmutableLatestFilterFields.builder()
                .typeField(search.getLatestFilterTypeField())
                .dateField(search.getLatestFilterDateField())
                .order(orderBy)
                .build());
    }

}
