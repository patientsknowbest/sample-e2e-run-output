package com.pkb.util.mapper;

import com.pkb.app.entity.EHRSearch;
import com.pkb.ehrdata.ImmutablePageFilter;
import com.pkb.ehrdata.PageFilter;
import org.mapstruct.Mapper;

import java.util.Optional;

@Mapper(componentModel = "cdi")
public interface PageFilterMapper {

    default Optional<PageFilter> map(EHRSearch<?> search) {
        if ((search.getResultsMaxReturned() <= 0) && (search.getResultsStartPosition() <= 0)) {
            return Optional.empty();
        }

        return Optional.of(ImmutablePageFilter.builder()
                .startPosition(search.getResultsStartPosition())
                .count(search.getResultsMaxReturned())
                .build());
    }
}
