package com.pkb.util.mapper;

import com.google.common.collect.ImmutableList;
import com.pkb.app.entity.DTOMapping;
import com.pkb.app.entity.EHRSearch;
import com.pkb.app.entity.PKBFilter;
import com.pkb.ehrdata.ImmutableSearchSpec;
import com.pkb.ehrdata.SearchSpec;
import org.apache.commons.lang3.StringUtils;
import org.mapstruct.Mapper;

import javax.inject.Inject;
import java.util.List;
import java.util.Optional;
import java.util.function.Function;

import static java.util.stream.Collectors.toList;

@Mapper(componentModel = "cdi")
public abstract class SearchSpecMapper {

    private PKBFilterMapper pkbFilterMapper;

    private MenuDataTypeMapper menuDataTypeMapper;

    private OrderByMapper orderByMapper;

    private LatestFilterFieldsMapper latestFilterFieldsMapper;

    private PageFilterMapper pageFilterMapper;

    public List<SearchSpec> map(List<EHRSearch> searches) {
        return searches.stream().map(this::map).collect(toList());
    }

    public ImmutableSearchSpec map(EHRSearch<?> search) {

        // Pass false to retrieve all filters
        List<PKBFilter> pkbFilters = ImmutableList.copyOf(search.getDataFilters(false));

        // Create column mapping function from the DTO map, so that the query builder has a way to map DTO fields
        // to database columns if the search is specified that way
        DTOMapping<?> dtoMapping = search.map();
        Function<String, String> columnMapper = (s) -> {
            String fieldOrColumn = dtoMapping.getDbColumn(s);
            return StringUtils.defaultString(fieldOrColumn, s);
        };

        return ImmutableSearchSpec.builder()
                .useParticipantFilter(search.getUseParticipantFilter())
                .addAllAccountIds(search.getAccountIds())
                .addAllFilters(pkbFilterMapper.map(pkbFilters))
                .addAllDataTypes(menuDataTypeMapper.map(search.getDataTypes()))
                .includeDeleted(search.isIncludeDeleted())
                .orderBy(orderByMapper.map(search))
                .latestFilter(latestFilterFieldsMapper.map(search))
                .pageFilter(pageFilterMapper.map(search))
                .conversationWorkflowFilter(Optional.ofNullable(search.getConversationWorkflowFilter()))
                .columnMapper(columnMapper)
                // below the latestByVisibleType appears to be at the wrong level.
                // the problem is elsewhere in the code there is a match on LatestFilterFields
                // which fails is some searches are latest and some are not
                .latestByVisibleType(search.getLatestFilterByVisibleOnly())
                .build();
    }

    @Inject
    public void setPkbFilterMapper(PKBFilterMapper pkbFilterMapper) {
        this.pkbFilterMapper = pkbFilterMapper;
    }

    @Inject
    public void setMenuDataTypeMapper(MenuDataTypeMapper menuDataTypeMapper) {
        this.menuDataTypeMapper = menuDataTypeMapper;
    }

    @Inject
    public void setOrderByMapper(OrderByMapper orderByMapper) {
        this.orderByMapper = orderByMapper;
    }

    @Inject
    public void setLatestFilterFieldsMapper(LatestFilterFieldsMapper latestFilterFieldsMapper) {
        this.latestFilterFieldsMapper = latestFilterFieldsMapper;
    }

    @Inject
    public void setPageFilterMapper(PageFilterMapper pageFilterMapper) {
        this.pageFilterMapper = pageFilterMapper;
    }
}
