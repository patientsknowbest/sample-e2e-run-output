package com.pkb.util.mapper;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.ehrdata.ImmutablePrivacyFilter;
import com.pkb.ehrdata.PrivacyFilter;
import org.jetbrains.annotations.NotNull;
import org.mapstruct.Mapper;

import java.util.Optional;

@Mapper(componentModel = "cdi")
public interface PrivacyFilterMapper {

    default Optional<PrivacyFilter> map(@NotNull EHRRequestContext requestContext) {
        return requestContext.getConsentStatus().getConsent().map(consent -> ImmutablePrivacyFilter.builder()
                .isGeneralConsentGranted(consent.generalHealthConsentGranted())
                .isMentalHealthConsentGranted(consent.mentalHealthConsentGranted())
                .isSexualHealthConsentGranted(consent.sexualHealthConsentGranted())
                .isSocialCareConsentGranted(consent.socialCareConsentGranted())
                .build());
    }
}
