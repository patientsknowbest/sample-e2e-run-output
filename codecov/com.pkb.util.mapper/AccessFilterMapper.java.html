package com.pkb.util.mapper;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.config.PkbPluginConfig;
import com.pkb.ehrdata.AccessFilter;
import com.pkb.ehrdata.ImmutableAccessFilter;
import org.mapstruct.Mapper;

import javax.inject.Inject;

@Mapper(componentModel = "cdi")
public abstract class AccessFilterMapper {

    private PkbPluginConfig configuration;

    private PrivacyFilterMapper privacyFilterMapper;

    public AccessFilter map(EHRRequestContext requestContext,
                            boolean decryptAccessRequired) {
        return map(requestContext, decryptAccessRequired, false);
    }

    public AccessFilter map(EHRRequestContext requestContext,
                            boolean decryptAccessRequired,
                            boolean accountLinksJoinPermitted) {

        return ImmutableAccessFilter.builder()
                .isTeamDataViewApplicable(requestContext.isTeamDataViewApplicable())
                .isPiggyback(requestContext.isPiggyback())
                .isConsentNotRequired(!requestContext.getConsentStatus().isRequired())
                .isBreakTheGlassActive(requestContext.getConsentStatus().isBreakTheGlassActive())
                .isPatientViewSuspendedRequired(requestContext.isPatientViewSuspendedRequired())
                .actorPersonId(requestContext.getMaybeAccessingUserId())
                .actorTeamId(requestContext.getTeamId())
                .actorOrgId(requestContext.getOrgId())
                .privacyFilter(privacyFilterMapper.map(requestContext))
                .isDecryptAccessRequired(decryptAccessRequired)
                .isAccountLinksJoinPermitted(accountLinksJoinPermitted)
                . build();
    }

    @Inject
    public void setPrivacyFilterMapper(PrivacyFilterMapper privacyFilterMapper) {
        this.privacyFilterMapper = privacyFilterMapper;
    }

    @Inject
    public void setConfiguration(PkbPluginConfig configuration) {
        this.configuration = configuration;
    }
}
