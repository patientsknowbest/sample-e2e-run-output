package com.pkb.util.mapper;

import com.pkb.app.entity.EHRSearch;
import com.pkb.ehrdata.ImmutableOrderBy;
import com.pkb.ehrdata.OrderBy;
import org.apache.commons.lang3.StringUtils;
import org.mapstruct.Mapper;

import java.util.Optional;

@Mapper(componentModel = "cdi")
public interface OrderByMapper {

    default Optional<OrderBy> map(EHRSearch<?> search) {
        if (StringUtils.isBlank(search.getOrderByField())) {
            return Optional.empty();
        }
        return Optional.of(ImmutableOrderBy.builder()
                .field(search.getOrderByField())
                .direction(map(search.getOrderByDirection()))
                .build());

    }

    OrderBy.Direction map(EHRSearch.OrderByDirection direction);
}
