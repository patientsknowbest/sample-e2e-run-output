package com.pkb.service.document;

import com.pkb.crypto.document.Document;
import com.pkb.crypto.document.EncryptedDocument;
import com.pkb.crypto.document.PlaintextDocument;
import com.pkb.entities.enums.DocStorageBackend;
import org.jetbrains.annotations.NotNull;

import java.util.Collection;
import java.util.Map;
import java.util.Optional;

/**
 * Represents a minimalistic key-value store for encrypted data
 *
 * Note: depending on the implementation these operations
 * might or might not participate in open JPA transactions.
 */
public interface EncryptedDocumentStore {

    DocStorageBackend storageType();

    /**
     * Adds an encrypted document into the store
     * 
     * @param encryptedDocument
     *            the document
     * @return the id of the persisted document
     */
    String create(@NotNull Document encryptedDocument);

    /**
     * Retrieve an encrypted document
     * 
     * @param id
     *            the id of the encrypted document
     * @return Some document if it can be found in the store, none otherwise
     */
    Optional<byte[]> retrieve(@NotNull String id);

    /**
     * Retrieve encrypted documents mapped by provided id.
     * 
     * @param ids
     * @return
     */
    @NotNull
    Map<String, byte[]> retrieve(@NotNull Collection<String> ids);

    /**
     * Delete a document
     * 
     * @param id
     *            the id of the encrypted document
     * @return true if the document is not present in the store after this method returns, false otherwise
     */
    boolean delete(@NotNull String id);

    /**
     * Delete documents
     *
     * @param ids
     *            the ids of encrypted documents to delete
     */
    int delete(Collection<String> ids);

    static byte[] getContent(Document document) {
        if (document instanceof EncryptedDocument) {
            return ((EncryptedDocument) document).cipherText();
        }

        if (document instanceof PlaintextDocument) {
            return ((PlaintextDocument) document).plainText();
        }

        throw new IllegalStateException("Invalid document type");
    }

}
