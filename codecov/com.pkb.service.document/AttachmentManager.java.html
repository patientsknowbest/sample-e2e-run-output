package com.pkb.service.document;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.crypto.document.PlaintextDocument;
import com.pkb.document.entity.Attachment;
import com.pkb.document.entity.DocAttach;
import com.pkb.model.RecordWithId;
import com.pkb.repository.legacy.LegacyAttachmentRepository;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import io.vavr.Tuple;

import java.util.Collection;
import java.util.stream.Stream;

import static com.pkb.util.EncryptedDocumentMapperHelper.ACCOUNT_ID_PROVIDER;
import static com.pkb.util.EncryptedDocumentMapperHelper.AUTHOR_PERSON_ID_PROVIDER;
import static com.pkb.util.EncryptedDocumentMapperHelper.MEDIA_TYPE_PROVIDER;
import static java.util.function.Function.identity;
import static java.util.stream.Collectors.toList;
import static java.util.stream.Collectors.toMap;
import static java.util.stream.Collectors.toSet;
import static org.apache.commons.collections4.CollectionUtils.isEmpty;

public class AttachmentManager extends TransactionManager {

    private final LegacyAttachmentRepository repository;
    private final DocumentManager documentManager;

    public AttachmentManager(PhrConfig config,
                             TolvenBeanFactory beanFactory,
                             DateTimeService dateTimeService,
                             UUIDProvider uuidProvider,
                             LegacyAttachmentRepository repository,
                             DocumentManager documentManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.repository = repository;
        this.documentManager = documentManager;
    }

    public Attachment getAttachmentContent(LoggedInEHRRequestContext context, Long attachmentId) {
        var docAttach = repository.findById(attachmentId).orElseThrow(() -> new IllegalStateException("Could not find attachment with id=" + attachmentId));
        var document = documentManager.retrieveOrThrow(docAttach.getAttachedDocumentMetadataId());
        // decryption will fail unless we use the accountId of the document and the logged-in user id
        // user must also have access to that account
        var decrypted = beanFactory.getEhrRemote().decryptDoc(document, context);
        return getAttachment(decrypted, attachmentId, docAttach);
    }

    public Stream<Attachment> getAttachmentContent(LoggedInEHRRequestContext context, Collection<Long> attachmentIds) {
        if (isEmpty(attachmentIds)) {
            return Stream.empty();
        }
        var docAttachs = repository.findAllById(attachmentIds)
                .stream()
                .collect(toMap(DocAttach::getId, identity()));
        var docAttachmentMetadataIds = docAttachs.values()
                .stream()
                .map(DocAttach::getAttachedDocumentMetadataId)
                .collect(toSet());
        var documents = documentManager.retrieve(docAttachmentMetadataIds);
        var toDectypt = attachmentIds.stream()
                .map(attachmentId -> {
                    var docAttach = docAttachs.get(attachmentId);
                    if (docAttach == null) {
                        throw new IllegalStateException("Could not find attachment with id=" + attachmentId);
                    }

                    var document = documents.get(docAttach.getAttachedDocumentMetadataId());
                    if (document == null) {
                        throw new RuntimeException("Could not find encryptedDocument with id=" + docAttach.getAttachedDocumentMetadataId());
                    }
                    return RecordWithId.of(attachmentId, Tuple.of(document, context));
                })
                .collect(toList());
        var decrypted = beanFactory.getEhrRemote().decryptDoc(toDectypt)
                .stream()
                .collect(toMap(RecordWithId::id, RecordWithId::record));

        return attachmentIds.stream().map(attachmentId -> getAttachment(decrypted.get(attachmentId), attachmentId, docAttachs.get(attachmentId)));
    }

    private Attachment getAttachment(PlaintextDocument decrypted, Long attachmentId, DocAttach docAttach) {
        var attachment = new Attachment();
        attachment.setAccountId(ACCOUNT_ID_PROVIDER.apply(decrypted));
        AUTHOR_PERSON_ID_PROVIDER.apply(decrypted).ifPresent(attachment::setUserId);
        attachment.setId(attachmentId);
        attachment.setFilename(docAttach.getDescription());
        attachment.setMediaType(MEDIA_TYPE_PROVIDER.apply(decrypted));
        attachment.setContent(decrypted.plainText());
        return attachment;
    }
}
