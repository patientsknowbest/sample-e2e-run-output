package com.pkb.fhirwall.service.auth.oso;

import org.hl7.fhir.r4.model.Consent;
import org.hl7.fhir.r4.model.Consent.ConsentState;
import org.immutables.value.Value.Immutable;

/**
 * It represents a projection of Consent FHIR resource.
 * It aggregates all attributes which are needed for evaluating authorization policies.
 */
@Immutable
public interface ConsentAuth extends AuthBase {

    ConsentState status();

    static ConsentAuth of(Consent consent) {
        return ImmutableConsentAuth.builder()
                .id(consent.getId())
                .status(consent.getStatus())
                .build();
    }

}
