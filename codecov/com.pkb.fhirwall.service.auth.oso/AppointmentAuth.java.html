package com.pkb.fhirwall.service.auth.oso;

import org.hl7.fhir.r4.model.Appointment;
import org.hl7.fhir.r4.model.Patient;

/**
 * It represents a projection of Appointment FHIR resource.
 * It aggregates all attributes which are needed for evaluating authorization policies.
 */
public class AppointmentAuth implements AuthBase {

    public final String id;
    public final PatientAuth patient;

    public AppointmentAuth(String id, PatientAuth patient) {
        this.id = id;
        this.patient = patient;
    }

    public static AppointmentAuth of(Appointment appointment, Patient patient) {
        return new AppointmentAuth(appointment.getIdElement().getIdPart(), PatientAuth.of(patient));
    }

    public static AppointmentAuth withoutPatient(Appointment appointment) {
        return new AppointmentAuth(appointment.getIdElement().getIdPart(), PatientAuth.empty());
    }

    @Override
    public String id() {
        return id;
    }
}
