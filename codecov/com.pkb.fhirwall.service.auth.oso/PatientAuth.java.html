package com.pkb.fhirwall.service.auth.oso;

import org.hl7.fhir.r4.model.Consent;
import org.hl7.fhir.r4.model.Patient;
import org.hl7.fhir.r4.model.Reference;
import org.immutables.value.Value.Auxiliary;
import org.immutables.value.Value.Immutable;

import java.util.List;
import java.util.Optional;

import static java.util.Collections.emptyList;
import static java.util.stream.Collectors.toList;
import static org.hl7.fhir.r4.model.Consent.ConsentState.ACTIVE;

/**
 * It represents a projection of Patient FHIR resource.
 * It aggregates all attributes which are needed for evaluating authorization policies.
 */
@Immutable
public interface PatientAuth extends AuthBase {

    String FROZEN_BY_TEAM_EXTENSION_URL = "http://fhir.patientsknowbest.com/structuredefinition/frozen-by-team";

    @Auxiliary
    Optional<String> accessFrozenByTeam();
    @Auxiliary
    List<ConsentAuth> consents();
    @Auxiliary
    boolean sharingDisabled();

    static PatientAuth of(Patient patient) {
        return of(patient, emptyList(), Optional.empty());
    }

    static PatientAuth of(Patient patient, List<Consent> consentResources, Optional<Consent> optOutResource) {
        var accessFrozenByTeam = getAccessFrozenByTeamExtension(patient);
        var consents = consentResources.stream()
                .map(ConsentAuth::of)
                .collect(toList());
        var sharingDisabled = optOutResource.map(Consent::getStatus)
                .filter(ACTIVE::equals)
                .isPresent();

        return ImmutablePatientAuth.builder()
                .id(patient.getIdElement().getIdPart())
                .accessFrozenByTeam(accessFrozenByTeam)
                .consents(consents)
                .sharingDisabled(sharingDisabled)
                .build();
    }

    static PatientAuth empty() {
        return ImmutablePatientAuth.builder()
                .id("")
                .sharingDisabled(false)
                .build();
    }

    private static Optional<String> getAccessFrozenByTeamExtension(Patient resource) {
        var extensions = resource.getExtensionsByUrl(FROZEN_BY_TEAM_EXTENSION_URL);
        if (extensions.size() == 1) {
            return Optional.of(((Reference) extensions.get(0).getValue()).getResource().getIdElement().getIdPart());
        }
        if (extensions.size() > 1) {
            throw new RuntimeException("Unexpected number of frozen-by-team extensions found on Patient resource: " + resource.getId());
        }

        return Optional.empty();
    }

}
