package com.pkb.test.entity;

import io.vavr.Tuple;
import io.vavr.control.Either;

import static io.vavr.API.None;
import static io.vavr.API.Some;

public class MeasurementTypeLinearConversions {
    public static MeasurementTypeLinearConversion scaleValue1(Double scale) {
        return ImmutableMeasurementTypeLinearConversion.builder()
                .toValue1(Tuple.of(scale, 0d, 0d))
                .toValue2(None())
                .build();
    }

    public static MeasurementTypeLinearConversion hrsMinsToSeconds =
            ImmutableMeasurementTypeLinearConversion.builder()
                    .toValue1(Tuple.of(3600d, 60d, 0d))
                    .toValue2(Some(Either.left(MeasurementTypeConversionConstant.NULL)))
                    .build();
}
