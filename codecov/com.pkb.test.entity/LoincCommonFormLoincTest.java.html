package com.pkb.test.entity;

import org.hibernate.annotations.Cache;
import org.hibernate.annotations.CacheConcurrencyStrategy;

import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import javax.persistence.UniqueConstraint;

@Entity
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table(name = "loinccommonform_loinctest", uniqueConstraints = {
        @UniqueConstraint(columnNames = { "loinccommonform_id", "loinctest_id" }) })
public class LoincCommonFormLoincTest {

    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "loinccommonform_id")
    private LoincCommonForm loincCommonForm;

    @ManyToOne(fetch = FetchType.EAGER)
    @JoinColumn(name = "loinctest_id")
    private LoincTest loincTest;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public LoincCommonForm getLoincCommonForm() {
        return loincCommonForm;
    }

    public void setLoincCommonForm(LoincCommonForm loincCommonForm) {
        this.loincCommonForm = loincCommonForm;
    }

    public LoincTest getLoincTest() {
        return loincTest;
    }

    public void setLoincTest(LoincTest loincTest) {
        this.loincTest = loincTest;
    }

}
