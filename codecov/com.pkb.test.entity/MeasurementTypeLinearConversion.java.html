package com.pkb.test.entity;

import io.vavr.Tuple;
import io.vavr.Tuple2;
import io.vavr.Tuple3;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.immutables.value.Value;

import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;

// Measurements can have multiple values, eg blood pressure has a systolic and a diastolic value as/in one measurement.
// Call these two values value1 and value2.
// Linear conversions are such that there are three numbers, a, b and c such that a new value can be computed via
//
//       new value1 = a * value1 + b * value2 + c
//
// Eg for converting a measurement of value1=hours, value2=minutes to a single value of seconds, a=3600, b=60, c=0.
//
// You could
// - do the same for a "new value2" with a triple of numbers, a2, b2, c2, toValue2() being Some(Either.right(Tuple.of(a2,b2,c2)))
// - (eg) you can set value2 to a constant, toValue2() being Some(Either.right(Tuple.of(0d,0d, constant)))
// - or you can signal that the new measurement has no value2, it's a single value measurement, toValue2() being Some(Either.left(NULL))
// - finally, you can signal that the new measurement has the same value2 as the old one, toValue2() being None()
//
@Value.Immutable
@Value.Style(strictBuilder = true, stagedBuilder = true)
public interface MeasurementTypeLinearConversion {
    Tuple3<Double,Double,Double> toValue1();
    Option<Either<MeasurementTypeConversionConstant,Tuple3<Double,Double,Double>>> toValue2();

    default Tuple2<Double,Double> convert(Tuple2<Double,Double> valueIn) {
        Tuple2<Double,Double> value = Tuple.of(valueIn._1() == null ? 0d : valueIn._1(), valueIn._2() == null ? 0d : valueIn._2());
        Double value1 = value._1();
        Double value2 = value._2();
        Double a = toValue1()._1();
        Double b = toValue1()._2();
        Double c = toValue1()._3();

        Double newValue1 = a * value1 + b * value2 + c;
        Double newValue2;
        if(toValue2().isEmpty()) {
            newValue2 = valueIn._2();
        } else {
            var toV2spec = toValue2().get();
            newValue2 = toV2spec.fold(
                    /* Either.left(NULL) */ constant -> Match(constant).of(Case($(MeasurementTypeConversionConstant.NULL), () -> null)),
                    /* Either.right(triple) */ toV2 -> {
                        Double a2 = toV2._1();
                        Double b2 = toV2._2();
                        Double c2 = toV2._3();
                        return a2 * value1 + b2 * value2 + c2;
                    });
        }

        return Tuple.of(newValue1, newValue2);
    }
}
