package com.pkb.test.entity;

import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import org.hibernate.annotations.Cache;
import org.hibernate.annotations.CacheConcurrencyStrategy;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.io.Serializable;
import java.time.Instant;

/**
 * LoincMapping entries are used to capture types of test result that are sent to PKB via HL7 or from SciStore.
 *
 * Historically, they were only created if PKB could not find a match for the result type in our LoincTest table, but
 * since PHR-8243 all test results (via HL7) are linked to a LoincMapping entry. You might consider this table to be a
 * test result equivalent of CodingReceived (although it's not an exact comparison.)
 * <br/>
 * The id attribute is our internal identity. The collection of externally provided attributes which together should
 * uniquely identify an entry are:
 * <uL>
 * <li>org</li>
 * <li>testCode</li>
 * <li>testCodingSystem</li>
 * <li>unit</li>
 * </ul>
 * <br/>
 * Be aware that the name of this class is a little misleading; a LoincMapping entry can be mapped to a known LoincTest
 * (see loinctest_id) but that is optional, so the existence of a LoincMapping entity does not necessarily mean there is
 * a mapping to a known LOINC test. The LoincTest attribute can be populated in 2 ways:
 * <ul>
 * <li>Automatically - if we find a match in LoincTest when processing the data.</li>
 * <li>Manually - if the customer has asked PKB to map their local codes to LOINC codes.</li>
 * </ul>
 */
@Entity
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table
public class LoincMapping implements Serializable, TestResultType {
    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    /**
     * The Org ID forms part of the identity for LoincMapping entries.
     * E.g. Should a different Org send results for the same test types then new LoincMapping entries will be created.
     */
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "org_id", nullable = true)
    private Org org;

    /**
     * The Team ID is not included in the identity for LoincMapping entries, so this value simply captures details of who first created the entry.
     * E.g. Should a different team send results for the same test types then that team's ID will not be stored in the database.
     */
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "team_id", nullable = true)
    private Team team;

    /**
     * The HL7Partner entries stored in the hl7partner table are not managed Hibernate entities, so here we just store the ID.
     * <p>
     * The HL7 Partner ID is not included in the identity for LoincMapping entries, so this value simply captures details of who first created the entry.
     * E.g. Should a different partner send results for the same test types then that partner's ID will not be stored in the database.
     */
    @Column(name = "hl7partner_id", nullable = true)
    private Long hl7PartnerId;

    /**
     * The test code.
     */
    @Column(length = 100, nullable = false)
    private String testCode;

    /**
     * The coding system, which is the namespace from which testCode was drawn (but remember these are unique only
     * within a particular org.)
     */
    @Column(length = 100, nullable = false)
    private String testCodingSystem;

    /**
     * The unit attribute is the one which forms part of the LoincMapping identity.
     * However, since there is some fiddly business logic surrounding how that value is determined, we also store the unitCode directly.
     * This is a safety net that is not currently used.
     */
    @Column(length = 100)
    private String unitCode;

    /**
     * The unit attribute is the one which forms part of the LoincMapping identity.
     * However, since there is some fiddly business logic surrounding how that value is determined, we also store the unitCodingSystem directly.
     * This is a safety net that is not currently used.
     */
    @Column(length = 100)
    private String unitCodingSystem;

    /**
     * The unit forms part of the identity for LoincMapping entries.
     * E.g. Should an Org send a different unit value for a previously seen test code/system, then a new LoincMapping entry will be created.
     */
    @Column(length = 100, nullable = false)
    private String unit;

    /**
     * The display name for the test. This is not included in the identity for LoincMapping entries, so instead we track
     * the most recently seen name for the entry. This is lossy; we will show in the UI the name we currently know this
     * test by, which may or may not be the same name that the result was originally reported with.
     */
    @Column(length = 300)
    private String name;

    /**
     * The first service name we received for this test type.
     * Non-critical functionality - it will not be updated once the loincmapping exists.
     */
    @Column(length = 500)
    private String serviceName;

    /**
     * The first sequence number we received for this test type.
     * Non-critical functionality - it will not be updated once the loincmapping exists.
     */
    @Column
    private Integer sequenceInGroup;

    /**
     * The intention is to prevent automated type conversions in some circumstances. For example, a TEXT type should not
     * be assigned a numeric value even if the provided value does parse as a number.
     *
     * This is a flaky feature though and needs reviewing.
     */
    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    private ValueType expectedValueType = ValueType.NUMERIC;

    /**
     * If this LoincMapping has indeed been mapped to a known LoincTest, then LoincTest.id is stored here.
     */
    @ManyToOne(fetch = FetchType.EAGER)
    @JoinColumn(name = "loinctest_id", nullable = true)
    private LoincTest loincTest;

    /**
     * This captures when the entry was last updated. This is useful for debugging, but it is not read by the
     * application code because callers that need to know if they have newer information should check enteredDate;
     * enteredDate more accurately reflects the time when the information was true because it is not affected by
     * queueing.
     */
    @Column
    private Instant lastUpdated;

    /**
     * This attempts to capture when the values in this entry were believed to be true. For example, an HL7 message
     * will populate this value with the message timestamp, regardless of how long that message was queued for after
     * being received by PKB.
     *
     * Note: enteredDate only applies to attributes which get updated. You shouldn't assume that attributes (such as
     * serviceName) that do not get updated were also current/valid at the time of enteredDate.
     */
    @Column
    private Instant enteredDate;

    @Override
    public int compareTo(TestResultType type) {
        return this.name.compareTo(type.getName());
    }

    @Override
    public String getIdSource() {
        return TestResultType.ID_SOURCE_MAPPING;
    }

    @Override
    public String toString() {
        return "LoincMapping [" +
                "id " + id +
                ", code " + testCodingSystem + "/" + testCode +
                ", unit " + unit +
                ", valueType " + expectedValueType + "]";
    }

    // regular getters/setters

    @Override
    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    @Override
    public String getTestCode() {
        return testCode;
    }

    public void setTestCode(String testCode) {
        this.testCode = testCode;
    }

    @Override
    public String getTestCodingSystem() {
        return testCodingSystem;
    }

    public void setTestCodingSystem(String testCodingSystem) {
        this.testCodingSystem = testCodingSystem;
    }

    public String getUnitCode() {
        return unitCode;
    }

    public void setUnitCode(String unitCode) {
        this.unitCode = unitCode;
    }

    public String getUnitCodingSystem() {
        return unitCodingSystem;
    }

    public void setUnitCodingSystem(String unitCodingSystem) {
        this.unitCodingSystem = unitCodingSystem;
    }

    @Override
    public String getUnit() {
        return unit;
    }

    public void setUnit(String unit) {
        this.unit = unit;
    }

    @Override
    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getServiceName() {
        return serviceName;
    }

    public void setServiceName(String serviceName) {
        this.serviceName = serviceName;
    }

    public Integer getSequenceInGroup() {
        return sequenceInGroup;
    }

    public void setSequenceInGroup(Integer sequenceInGroup) {
        this.sequenceInGroup = sequenceInGroup;
    }

    public LoincTest getLoincTest() {
        return loincTest;
    }

    public void setLoincTest(LoincTest loincTest) {
        this.loincTest = loincTest;
    }

    @Override
    public ValueType getExpectedValueType() {
        return expectedValueType;
    }

    public void setExpectedValueType(ValueType expectedValueType) {
        this.expectedValueType = expectedValueType;
    }

    public void setLastUpdated(Instant lastUpdated) {
        this.lastUpdated = lastUpdated;
    }

    public Instant getEnteredDate() {
        return enteredDate;
    }

    public void setEnteredDate(Instant enteredDate) {
        this.enteredDate = enteredDate;
    }

    public Org getOrg() {
        return org;
    }

    public void setOrg(Org org) {
        this.org = org;
    }

    public Team getTeam() {
        return team;
    }

    public void setTeam(Team team) {
        this.team = team;
    }

    public Long getHl7PartnerId() {
        return hl7PartnerId;
    }

    public void setHl7PartnerId(Long hl7PartnerId) {
        this.hl7PartnerId = hl7PartnerId;
    }
}
