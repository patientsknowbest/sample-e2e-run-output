package com.pkb.test.entity;

import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.io.Serializable;
import java.util.List;
import java.util.Optional;
import java.util.Set;

import static java.util.Collections.singleton;
import static java.util.stream.Collectors.toSet;

/**
 * History of test results - LoincTest and list of test results.
 * Optional link to a source file.
 */

public class TestHistoryDTO implements Serializable {

    private static final long serialVersionUID = 1395738354626790545L;

    private LoincTest loincTest;

    /**
     * Either loincMapping or loincTest may be null; once the mapping is made, still keep this link, so we can recover
     * from an bad mapping.
     */
    private LoincMapping loincMapping;

    private List<TestResultDTO> testResultList;

    // utility methods

    /**
     * return either the LoincTest (if available) or LoincMapping
     */
    public TestResultType getTestResultType() {
        return loincTest != null ? loincTest : loincMapping;
    }

    @Nullable
    public TestResultDTO getLatestTestResult() {
        if (testResultList != null && !testResultList.isEmpty()) {
            return testResultList.get(testResultList.size() - 1);
        }
        // not available
        return null;
    }

    public @NotNull Optional<TestResultDTO> findTestResult(Long testResultId) {
        return testResultList.stream().filter(result -> result.getId().equals(testResultId)).findFirst();
    }

    /**
     * Helps the UI make display decisions for the graph.
     *
     * @return true - if there is at least one non-deleted result with a numeric value. false - otherwise.
     */
    public boolean hasNumericValues() {
        return testResultList.stream().anyMatch(r -> r.isValueNumeric() && !r.getBaseFields().isDeleted());
    }

    /**
     * This helps ManagePlansAction to determine whether or not to show the trendline for any test result metrics
     * associated with a care plan.
     *
     * @return true - if all non-deleted results have a numeric value. false - otherwise.
     */
    public boolean hasNumericValuesOnly() {
        return testResultList.stream().filter(x -> !x.getBaseFields().isDeleted()).allMatch(TestResultDTO::isValueNumeric);
    }

    public Set<String> getServiceNameIfConsistent() {
        Set<String> serviceNamesInResults = testResultList.stream()
                .map(TestResultDTO::getServiceNameFromSource)
                .filter(StringUtils::isNotBlank)
                .collect(toSet());

        if (serviceNamesInResults.size() > 1) {
            return serviceNamesInResults;
        } else {
            // return the one... or if there aren't any non-blank, use the service name in the LoincMapping
            return singleton(serviceNamesInResults.stream().findFirst()
                    .orElse(
                            Optional.ofNullable(loincMapping).map(LoincMapping::getServiceName)
                                    .orElse(null)));
        }
    }

    // getters/setters

    /**
     * sometimes null (this may be a LoincMapping instead); see getTestResultType() for the common interface
     */
    public LoincTest getLoincTest() {
        return loincTest;
    }

    public void setLoincTest(LoincTest loincTest) {
        this.loincTest = loincTest;
    }

    public LoincMapping getLoincMapping() {
        return loincMapping;
    }

    public void setLoincMapping(LoincMapping loincMapping) {
        this.loincMapping = loincMapping;
    }

    public List<TestResultDTO> getTestResultList() {
        return testResultList;
    }

    public void setTestResultList(List<TestResultDTO> testResultList) {
        this.testResultList = testResultList;
    }

    public String getLinkForTest(long contextUserIdString) {
        return "/test/testHistory.action" +
                "?testResultTypeIdSource=" + getTestResultType().getIdSource() +
                "&testResultTypeId=" + getTestResultType().getId() +
                "&contextUserId=" + contextUserIdString +
                "&tab=health";
    }
}
