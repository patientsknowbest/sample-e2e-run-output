package com.pkb.test.entity;

import com.google.common.base.MoreObjects;

import javax.annotation.Nullable;
import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

/**
 * Simple object for specifying what test history to return in a search.
 * The method signatures were getting ugly...
 * <p>
 * Main filter options:
 * - Test IDs (LOINC IDs, or LoingMapping IDs)
 * Note: these are mostly "either/or" -- if you include some LoincTest IDs, you'll get only
 * those (no mapping results), if you include some LoincMapping IDs, you'll get only those (no
 * Loinc results)... if you include both you will get an error in some cases (if getting full
 * history) and it will actually work in other cases (requesting latest results only).
 * - Range of history to return -- all, only single latest result (per-test), or X months of history
 *
 * @author robwhelan
 */
public class TestHistoryFilter implements Serializable {
    private static final long serialVersionUID = -1L;

    public enum RangeType {
        LATEST_ONLY, ALL_RESULTS, RANGE_PER_TEST
    }

    private Long patientId = null;

    List<Long> loincIdList = new ArrayList<>();
    List<Long> loincMappingIdList = new ArrayList<>();

    RangeType rangeType = RangeType.ALL_RESULTS;
    Integer rangeMonths = null;

    /*
      Filters test result by ids of loincMappings with this service name
     */
    private String loincMappingServiceName;

    public TestHistoryFilter(long patientId) {
        this.patientId = patientId;
    }

    // set other filters

    public void limitRangeXMonthsPerTest(int lastXMonths) {
        rangeType = RangeType.RANGE_PER_TEST;
        rangeMonths = lastXMonths;
    }

    public void limitRangeLatestOnly() {
        rangeType = RangeType.LATEST_ONLY;
        rangeMonths = null;
    }

    /**
     * by default all tests & mappings are returned;
     * adding LOINC test IDs or LOINC mapping IDs limits results to the added IDs.
     *
     * @param loincTestId
     */
    public void addLoincTestId(Long loincTestId) {
        loincIdList.add(loincTestId);
    }

    /**
     * by default all tests & mappings are returned;
     * adding LOINC test IDs or LOINC mapping IDs limits results to the added IDs.
     *
     * @param loincTestIdList
     */
    public void addLoincTestIdList(List<Long> loincTestIdList) {
        this.loincIdList.addAll(loincTestIdList);
    }

    /**
     * by default all tests & mappings are returned;
     * adding LOINC test IDs or LOINC mapping IDs limits results to the added IDs.
     *
     * @param loincMappingId
     */
    public void addLoincMappingId(Long loincMappingId) {
        this.loincMappingIdList.add(loincMappingId);
    }

    /**
     * by default all tests & mappings are returned;
     * adding LOINC test IDs or LOINC mapping IDs limits results to the added IDs.
     *
     * @param loincMappingIdList
     */
    public void addLoincMappingIdList(List<Long> loincMappingIdList) {
        this.loincMappingIdList.addAll(loincMappingIdList);
    }

    @Override
    public String toString() {
        return MoreObjects.toStringHelper("TestHistoryFilter")
                .omitNullValues()
                .add("patientId", patientId)
                .add("rangeType", rangeType)
                .add("rangeMonths", rangeMonths)
                .add("loincIdList", loincIdList)
                .add("loincMappingIdList", loincMappingIdList)
                .toString();
    }

    public Long getPatientId() {
        return patientId;
    }

    /**
     * @return null, or a non-empty list
     */
    @Nullable
    public List<Long> getLoincIdList() {
        if ((loincIdList != null) && loincIdList.isEmpty()) {
            return null;
        }
        return loincIdList;
    }

    /**
     * @return null, or a non-empty list
     */
    @Nullable
    public List<Long> getLoincMappingIdList() {
        if ((loincMappingIdList != null) && loincMappingIdList.isEmpty()) {
            return null;
        }
        return loincMappingIdList;
    }

    public RangeType getRangeType() {
        return rangeType;
    }

    public Integer getRangeMonths() {
        return rangeMonths;
    }

    public String getLoincMappingServiceName() {
        return loincMappingServiceName;
    }

    public void setLoincMappingServiceName(String loincMappingServiceName) {
        this.loincMappingServiceName = loincMappingServiceName;
    }
}
