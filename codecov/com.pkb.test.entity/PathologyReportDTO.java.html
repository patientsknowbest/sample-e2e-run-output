package com.pkb.test.entity;

import com.pkb.annotation.EHRField;
import com.pkb.annotation.EHRField.QueryField;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.entities.enums.MenuDataType;

import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import java.util.Date;
import java.util.Map;

public class PathologyReportDTO implements IBaseDTO {

    private static final long serialVersionUID = 1;

    public static final MenuDataType MS_PATH = MenuDataType.pathologyReport;

    // Pathology Reports are subsumed into FHIR Diagnostic Reports - see https://www.hl7.org/fhir/diagnosticreport.html

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    @EHRField(queryField = QueryField.id)
    private Long id;

    @EHRField
    private String primaryDiagnosisIcd;

    @EHRField
    private Date sampleTakenDate;

    @EHRField
    private Date sampleReceiptDate;

    @EHRField
    private Date investigationResultDate;

    @EHRField(queryField = QueryField.date01)
    private Date date;

    @EHRField
    private String histologySnomed;

    @EHRField
    private String fullReportText;

    @EHRField
    private Map<String, String> otherFields;

    private SourceDetails source;

    private DTOBaseFields baseFields = new DTOBaseFields();

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected PathologyReportDTO() {
        this.source = new SourceDetails();
    }

    public PathologyReportDTO(SourceDetails sourceDetails) {
        this.source = sourceDetails;
    }

    @Override
    public Long getId() {
        return id;
    }

    @Override
    public void setId(Long id) {
        this.id = id;
    }

    public String getFullReportText() {
        return fullReportText;
    }

    public void setFullReportText(String fullReportText) {
        this.fullReportText = fullReportText;
    }

    public Map<String, String> getOtherFields() {
        return otherFields;
    }

    public void setOtherFields(Map<String, String> otherFields) {
        this.otherFields = otherFields;
    }

    public String getOtherField(String fieldName) {
        return otherFields.get(fieldName);
    }

    @Override
    public DTOBaseFields getBaseFields() {
        return baseFields;
    }

    @Override
    public SourceDetails getSource() {
        return source;
    }

    public Date getInvestigationResultDate() {
        return investigationResultDate;
    }

    public void setInvestigationResultDate(Date investigationResultDate) {
        this.investigationResultDate = investigationResultDate;
    }

    public String getPrimaryDiagnosisIcd() {
        return primaryDiagnosisIcd;
    }

    public void setPrimaryDiagnosisIcd(String primaryDiagnosisIcd) {
        this.primaryDiagnosisIcd = primaryDiagnosisIcd;
    }

    public Date getSampleTakenDate() {
        return sampleTakenDate;
    }

    public void setSampleTakenDate(Date sampleTakenDate) {
        this.sampleTakenDate = sampleTakenDate;
    }

    public String getHistologySnomed() {
        return histologySnomed;
    }

    public void setHistologySnomed(String histologySnomed) {
        this.histologySnomed = histologySnomed;
    }

    public Date getSampleReceiptDate() {
        return sampleReceiptDate;
    }

    public void setSampleReceiptDate(Date sampleReceiptDate) {
        this.sampleReceiptDate = sampleReceiptDate;
    }

    public Date getDate() {
        return date;
    }

    public void setDate(Date date) {
        this.date = date;
    }

    public void setSource(SourceDetails source) {
        this.source = source;
    }
}
