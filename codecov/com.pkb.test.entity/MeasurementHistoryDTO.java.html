/**
 * History of measurements - basic Measurement and list of data.
 */
public class MeasurementHistoryDTO implements Serializable {

    private static final long serialVersionUID = 1395738354626790545L;
    private static final Set<PredefinedMeasurementType> DISPLAY_AS_DAILY_TOTAL = unmodifiableSet(EnumSet.of(ALCHOL_UNITS_PER_DAY, AWAKE, AWAKE_HR,
            CALORIES_BURNED_FITNESS, CALORIES_BURNED_ROUTINE, CALORIES_CONSUMED, CARBOHYDRATES, CIGARETTES_PER_DAY,
            DEEP, DEEP_HR, DISTANCE_FITNESS, DISTANCE_FITNESS_KM, DISTANCE_FITNESS_MILES, DISTANCE_ROUTINE,
            DISTANCE_ROUTINE_KM, DISTANCE_ROUTINE_MILES, DAILY_DOSES_RESCUE_MEDICATION, DURATION, DURATION_MIN,
            ELEVATION, FAT, FIBER, FLOORS, DRAIN, LIGHT, LIGHT_HR, NON_CAFFEINATED_DRINKS_PER_DAY,
            NUMBER_OF_MISSED_MEDICATIONS_TODAY, NUMBER_OF_TIMES_I_NEARLY_FELL, PROTEIN, REM, REM_HR, SEIZURE_TYPE_1,
            SEIZURE_TYPE_2, SEIZURE_TYPE_3, SEIZURE_TYPE_4, SODIUM, STEPS, STOOL_VOLUME, TIMES_WOKEN, TOTAL_SLEEP,
            TOTAL_SLEEP_HR, URINE_VOLUME, VOMIT_VOLUME, WATER, WATER_IN_L));

    private MeasurementType measurementType;

    private List<MeasurementDTO> measurementList;

    private List<MeasurementDTO> chartMeasurements;

    private List<MeasurementDTO> hiddenMeasurementList = new ArrayList<>();

    // utility methods

    public @Nullable MeasurementDTO getLatestMeasurement() {
        if ((measurementList != null) && !measurementList.isEmpty()) {
            return measurementList.get(measurementList.size() - 1);
        }
        // not available
        return null;
    }

    public @Nullable MeasurementDTO getFirstMeasurement() {
        if ((measurementList != null) && !measurementList.isEmpty()) {
            return measurementList.get(0);
        }
        // not available
        return null;
    }

    public Optional<MeasurementDTO> findMeasurement(Long measurementId) {
        if (measurementList == null || measurementId == null) {
            return Optional.empty();
        }

        return measurementList.stream()
                .filter(measurementDTO -> measurementId.equals(measurementDTO.getId()))
                .findFirst();
    }

    // for building the sparklines
    // Nb sparkline library crashes on any null input

    public List<Double> getYData() {
        List<Double> list = measurementList.stream().filter(result -> result.getValue() != null).map(MeasurementDTO::getValue).collect(toList());
        return list;
    }

    public List<Double> getXData() {
        List<Double> list = new ArrayList<>();
        // x values should start at X months ago, through today
        list.addAll(measurementList.stream().filter(result -> result.getValue() != null).map(result -> (double) result.getMeasureDate().toEpochMilli()).collect(toList()));
        return list;
    }

    public List<Double> getYData2() {
        List<Double> list = measurementList.stream().filter(result -> result.getValue2() != null).map(MeasurementDTO::getValue2).collect(toList());
        return list;
    }

    public List<Double> getXData2() {
        List<Double> list = new ArrayList<>();
        // x values should start at X months ago, through today
        list.addAll(measurementList.stream().filter(result -> result.getValue2() != null).map(result -> (double) result.getMeasureDate().toEpochMilli()).collect(toList()));
        return list;
    }

    // for building JavaScript graphs (HighCharts)

    public String getDataJson() {
        //    	[[1230771600000,2.67],
        //			[1230858000000, 0.3],
        //			[1230944400000, 5.2]
        //		]
        return chartMeasurements.stream()
                .map(measurement -> toArray(measurement, measurement.getValue()))
                .collect(joining(",", "[", "]"));
    }

    public String getDataJson2() {
        //    	[[1230771600000,2.67],
        //			[1230858000000, 0.3],
        //			[1230944400000, 5.2]
        //		]
        return chartMeasurements.stream()
                .map(measurement -> toArray(measurement, measurement.getValue2()))
                .collect(joining(",", "[", "]"));
    }

    public List<MeasurementDTO> getMeasurementListLatestFirst() {
        List<MeasurementDTO> reversedList = new ArrayList<>();
        for (MeasurementDTO m : measurementList) {
            reversedList.add(0, m);
        }
        return reversedList;
    }

    public boolean isDailyTotal() {
        return getMeasurementType().getAsPredefinedMeasurementType()
                .map(DISPLAY_AS_DAILY_TOTAL::contains).getOrElse(false);
    }

    public boolean hasSecondValue() {
        return getLatestMeasurement().getValue2() != null || getMeasurementType().isBloodPressure();
    }

    private String toArray(MeasurementDTO measurement, Double value) {
        return "[" + measurement.getMeasureDate().toEpochMilli() + "," + (value != null ? measurementType.convertForDisplay(value) : null) + "]";
    }

    // getters/setters
    public MeasurementType getMeasurementType() { return measurementType; }

    public void setMeasurementType(MeasurementType measurementType) {
        this.measurementType = measurementType;
    }

    public List<MeasurementDTO> getMeasurementList() {
        return measurementList;
    }

    public void setMeasurementList(List<MeasurementDTO> measurementList) {
        this.measurementList = measurementList;
    }

    public List<MeasurementDTO> getChartMeasurements() {
        return chartMeasurements;
    }

    public void setChartMeasurements(List<MeasurementDTO> chartMeasurements) {
        this.chartMeasurements = chartMeasurements;
    }
    public List<MeasurementDTO> getHiddenMeasurementList() {
        return hiddenMeasurementList;
    }

    public void setHiddenMeasurementList(List<MeasurementDTO> hiddenMeasurementList) {
        this.hiddenMeasurementList = hiddenMeasurementList;
    }
}
