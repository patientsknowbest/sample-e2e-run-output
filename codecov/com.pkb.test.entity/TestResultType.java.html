package com.pkb.test.entity;

import org.apache.commons.lang3.StringUtils;

/**
 * Common methods between LoincTest and LoincMapping
 *
 * @author rob
 */
public interface TestResultType extends Comparable<TestResultType> {

    // for helping w/ JSPs
    String ID_SOURCE_LOINC = "loinc";
    String ID_SOURCE_MAPPING = "loincMapping";

    /**
     * some test result values may be usually numeric but sometimes 'nil' or similar
     */
    enum ValueType {
        TEXT, NUMERIC, VARIES, POSNEG;

        public boolean isTextual() {
            return this == TEXT || this == POSNEG;
        }
    }

    Long getId();

    String getIdSource();

    String getTestCodingSystem();

    String getTestCode();

    String getName();

    /**
     * Gracefully handle an empty name for display in the web interface by defaulting to the test code.
     */
    default String getNameForDisplay() {
        return StringUtils.isBlank(getName()) ? "(" + getTestCode() + ")" : getName();
    }

    ValueType getExpectedValueType();

    String getUnit();

    @Override
    int compareTo(TestResultType type);

    default boolean expectsTextualValue() {
        return getExpectedValueType().isTextual();
    }

    default String getResultLabelledBy() {
        //get list of ids for a11y
        return "result-" + getId() + (expectsTextualValue() ? "" : " unit-" + getId());
    }
}
