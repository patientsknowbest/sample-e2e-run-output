package com.pkb.test.entity;

import com.google.common.annotations.VisibleForTesting;
import org.jetbrains.annotations.NotNull;

import static org.apache.commons.lang3.StringUtils.isBlank;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class MeasurementValidator {

    private static final int OUNCES_IN_A_POUND = 16;
    private static final int POUNDS_IN_A_STONE = 14;
    private static final int INCHES_IN_A_FOOT = 12;
    private static final int MINUTES_IN_AN_HOUR = 60;

    public static Double convertHeightToCm(String feet, String inches) {
        Double feetValue = isBlank(feet) ? 0 : Double.parseDouble(feet);
        Double inchesValue = isBlank(inches) ? 0 : Double.parseDouble(inches);
        return (feetValue*12 + inchesValue) * 2.54;
    }

    @NotNull
    @VisibleForTesting
    public static String validateBloodPressure(String systolic, String diastolic) {
        Double systolicValue = null;
        Double diastolicValue = null;

        if (isNotBlank(systolic)) {
            try {
                systolicValue = Double.parseDouble(systolic);
            } catch (NumberFormatException ignored) {
                return ".err.systolic_not_number";
            }
            if (systolicValue < 0.0) {
                return ".err.systolic_negative";
            }
        }

        if (isNotBlank(diastolic)) {
            try {
                diastolicValue = Double.parseDouble(diastolic);
            } catch (NumberFormatException ignored) {
                return ".err.diastolic_not_number";
            }
            if (diastolicValue < 0.0) {
                return ".err.diastolic_negative";
            }
        }

        if (systolicValue != null && diastolicValue != null && Double.compare(systolicValue,diastolicValue)<=0) {
            return ".err.systolic_not_higher";
        }
        return "";
    }

    @NotNull
    @VisibleForTesting
    public static String validateDuration(String hours, String minutes) {
        return validateDualUnitMeasurement(hours,minutes,"hours","minutes",MINUTES_IN_AN_HOUR);
    }

    @NotNull
    @VisibleForTesting
    public static String validateWeightPoundsOunces(String pounds, String ounces) {
        return validateDualUnitMeasurement(pounds,ounces,"pounds","ounces",OUNCES_IN_A_POUND);
    }

    @NotNull
    @VisibleForTesting
    public static String validateWeightStonePounds(String stone, String pounds) {
        return validateDualUnitMeasurement(stone,pounds,"stone","pounds",POUNDS_IN_A_STONE);
    }

    @NotNull
    @VisibleForTesting
    public static String validateLengthFeetInches(String feet, String inches) {
        return validateDualUnitMeasurement(feet,inches,"feet","inches",INCHES_IN_A_FOOT);
    }

    @NotNull
    private static String validateDualUnitMeasurement(String value1, String value2, String unit1, String unit2, int value2Rollover) {
        Double value1Number = null;
        Double value2Number = null;

        if (isNotBlank(value1)) {
            try {
                value1Number = Double.parseDouble(value1);
            } catch (NumberFormatException ignored) {
                return ".err."+unit1+"_not_number";
            }
            if (value1Number < 0.0) {
                return ".err."+unit1+"_negative";
            }
        }

        if (isNotBlank(value2)) {
            try {
                value2Number = Double.parseDouble(value2);
            } catch (NumberFormatException ignored) {
                return ".err."+unit2+"_not_number";
            }
            if (value2Number < 0.0) {
                return ".err."+unit2+"_negative";
            }
            if (value2Number >= value2Rollover) {
                return ".err."+unit2+"_over_limit";
            }
        }
        return "";
    }
}
