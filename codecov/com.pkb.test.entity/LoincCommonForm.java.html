package com.pkb.test.entity;

import org.hibernate.annotations.Cache;
import org.hibernate.annotations.CacheConcurrencyStrategy;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.persistence.Transient;
import java.io.Serializable;
import java.util.Collections;
import java.util.List;

@Entity
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table
public class LoincCommonForm implements Serializable {
    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column(length = 20)
    private String formCode;

    @Column(length = 200)
    private String name;

    @Column(length = 400)
    private String description;

    @Transient
    private List<LoincTest> loincTestList;

    public enum Lazy {
        LOINC_TEST
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getFormCode() {
        return formCode;
    }

    public void setFormCode(String formCode) {
        this.formCode = formCode;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getDescription() {
        return description;
    }

    public void setDescription(String description) {
        this.description = description;
    }

    public List<LoincTest> getLoincTestList() {
        return loincTestList;
    }

    public void setLoincTestList(List<LoincTest> loincTestList) {
        this.loincTestList = loincTestList;
    }

    public List<LoincTest> getSortedLoincTestList() {
        Collections.sort(loincTestList);
        return loincTestList;
    }
}
