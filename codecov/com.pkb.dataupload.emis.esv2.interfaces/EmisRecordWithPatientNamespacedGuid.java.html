package com.pkb.dataupload.emis.esv2.interfaces;

import com.fasterxml.jackson.annotation.JsonIgnore;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import org.apache.commons.lang3.NotImplementedException;

/**
 * For any emis record that has a patient-namespaced guid
 */
public interface EmisRecordWithPatientNamespacedGuid {

    /**
     * Quoting EMIS: > the GUID is not globally unique it is only unique to patient
     * You can't uniquely identify an EmisPatientDataPoint by its guid, it only identifies it uniquely for a given patient
     */
    @JsonIgnore
    default Tuple2<String, String> uniqueKey() {
        return Tuple.of(patientSpecificGuid(), patientGuid());
    }

    default String patientGuid() {
        throw new NotImplementedException("Must be implemented so we know a unique identifier");
    }

    default String patientSpecificGuid() {
        throw new NotImplementedException("Must be implemented so we know a unique identifier");
    }
}
