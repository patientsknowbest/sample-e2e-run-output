package com.pkb.service.document.storage;

import com.google.common.collect.Maps;
import com.pkb.entities.enums.DocStorageBackend;
import com.pkb.service.document.EncryptedDocumentStore;
import com.pkb.service.document.GCSDocumentStore;
import com.pkb.service.document.GcsFallbackDocumentStore;
import org.apache.commons.lang3.NotImplementedException;
import org.jetbrains.annotations.NotNull;

import java.util.EnumMap;
import java.util.Map;

public class DocumentStorageFactory {

    private final GCSDocumentStore gcsDocumentStore;

    private final GcsFallbackDocumentStore gcsFallbackDocumentStore;

    private final Map<DocStorageBackend, EncryptedDocumentStore> documentStoreMap;

    public DocumentStorageFactory(GCSDocumentStore gcsDocumentStore, GcsFallbackDocumentStore gcsFallbackDocumentStore) {
        this.gcsDocumentStore = gcsDocumentStore;
        this.gcsFallbackDocumentStore = gcsFallbackDocumentStore;

        Map<DocStorageBackend, EncryptedDocumentStore> storageMap = new EnumMap<>(DocStorageBackend.class);
        storageMap.put(gcsDocumentStore.storageType(), gcsDocumentStore);
        storageMap.put(gcsFallbackDocumentStore.storageType(), gcsFallbackDocumentStore);
        this.documentStoreMap = Maps.immutableEnumMap(storageMap);
    }

    @NotNull
    public EncryptedDocumentStore getStore(@NotNull DocStorageBackend storage) {
        EncryptedDocumentStore result = documentStoreMap.get(storage);
        if (result == null) {
            throw new NotImplementedException("Document storage " + storage + " not implemented yet, add it! How were you able to call this?");
        }
        return result;
    }

    @NotNull
    public GCSDocumentStore getGcsDocumentStore() {
        return gcsDocumentStore;
    }
}
