package com.pkb.service.document.storage;

import com.pkb.crypto.document.Document;
import com.pkb.document.entity.GcsFallbackDoc;
import com.pkb.entities.enums.DocStorageBackend;
import com.pkb.repository.legacy.LegacyGcsFallbackDocRepository;
import com.pkb.service.document.EncryptedDocumentStore;
import com.pkb.service.document.GcsFallbackDocumentStore;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.DbUtil;
import io.vavr.Tuple;
import org.jetbrains.annotations.NotNull;

import java.util.Collection;
import java.util.Collections;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.UUID;

import static java.util.stream.Collectors.toMap;
import static java.util.stream.Collectors.toSet;
import static org.apache.commons.collections4.CollectionUtils.isEmpty;

/**
 * This provider participates in open JPA transactions.
 */
public class GcsFallbackDocumentStoreImpl implements GcsFallbackDocumentStore {

    private final UUIDProvider uuidProvider;
    private final LegacyGcsFallbackDocRepository repository;

    public GcsFallbackDocumentStoreImpl(UUIDProvider uuidProvider, LegacyGcsFallbackDocRepository repository) {
        this.uuidProvider = uuidProvider;
        this.repository = repository;
    }

    @Override
    public DocStorageBackend storageType() {
        return DocStorageBackend.GCS_FALLBACK;
    }

    @Override
    public String create(@NotNull Document document) {
        GcsFallbackDoc doc = new GcsFallbackDoc();
        doc.setId(uuidProvider.randomUUID());
        doc.setContent(EncryptedDocumentStore.getContent(document));
        repository.save(doc);
        return String.valueOf(doc.getId());
    }

    @Override
    public Optional<byte[]> retrieve(@NotNull String id) {
        return repository.findById(UUID.fromString(id)).map(GcsFallbackDoc::getContent);
    }

    @Override
    public @NotNull Map<String, byte[]> retrieve(@NotNull Collection<String> ids) {
        if (isEmpty(ids)) {
            return Collections.emptyMap();
        }
        List<GcsFallbackDoc> documents = DbUtil.getInBatches(ids.stream().map(UUID::fromString).collect(toSet()), repository::findAllById);
        return documents.stream()
                .map(d -> Tuple.of(String.valueOf(d.getId()), d.getContent()))
                .collect(toMap(t -> t._1, t -> t._2));
    }

    @Override
    public boolean delete(@NotNull String id) {
        repository.deleteById(UUID.fromString(id));
        return true;
    }

    @Override
    public int delete(Collection<String> ids) {
        return DbUtil.executeInBatches(ids.stream().map(UUID::fromString).collect(toSet()), repository::deleteAllByIdIn);
    }
}
