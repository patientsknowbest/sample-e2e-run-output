package com.pkb.service.document.storage;

import com.google.api.gax.paging.Page;
import com.google.cloud.BatchResult;
import com.google.cloud.storage.Blob;
import com.google.cloud.storage.BlobId;
import com.google.cloud.storage.Bucket;
import com.google.cloud.storage.Storage;
import com.google.cloud.storage.StorageBatch;
import com.google.cloud.storage.StorageException;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.crypto.document.Document;
import com.pkb.document.entity.GcsDeleteQueue;
import com.pkb.entities.enums.DocStorageBackend;
import com.pkb.repository.legacy.LegacyGcsDeleteQueueRepository;
import com.pkb.service.document.EncryptedDocumentStore;
import com.pkb.service.document.GCSDocumentStore;
import com.pkb.service.document.storage.gcs.StorageProvider;
import com.pkb.service.uuid.UUIDProvider;
import io.micrometer.core.instrument.Metrics;
import io.micrometer.core.instrument.Timer;
import io.vavr.Tuple;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.time.Instant;
import java.util.Collection;
import java.util.Collections;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;

import static org.apache.commons.collections4.CollectionUtils.isEmpty;

public class GCSDocumentStoreImpl implements GCSDocumentStore {

    private static final String DELETE_TIMER_NAME = "phr_gcs_deletion_timer";
    private static final Timer STORAGE_TIMER = Timer.builder(DELETE_TIMER_NAME)
            .description("Latencies of document deletion from GCS")
            .publishPercentileHistogram()
            .publishPercentiles(0.5, 0.9, 0.95, 0.99)
            .register(Metrics.globalRegistry);

    private final Storage storage;
    private final Bucket bucket;
    private final UUIDProvider uuidProvider;
    private final DateTimeService dateTimeService;
    private final LegacyGcsDeleteQueueRepository repository;
    private final PhrConfig config;

    public GCSDocumentStoreImpl(PhrConfig config, UUIDProvider uuidProvider, StorageProvider storageProvider, DateTimeService dateTimeService, LegacyGcsDeleteQueueRepository repository) {
        this.uuidProvider = uuidProvider;
        this.dateTimeService = dateTimeService;
        this.repository = repository;
        this.storage = storageProvider.getStorage();
        this.bucket = storage.get(config.getDocumentGcsStoreBucketName());
        this.config = config;
    }

    @Override
    public DocStorageBackend storageType() {
        return DocStorageBackend.GCS;
    }

    @Override
    public String create(@NotNull Document document) {
        String id = uuidProvider.randomUUID().toString();
        Blob blob = bucket.create(id, EncryptedDocumentStore.getContent(document), "application/octet-stream");
        return blob.getBlobId().getName();
    }

    @Override
    public Optional<byte[]> retrieve(@NotNull String id) {
        Blob blob = bucket.get(id);
        return Optional.ofNullable(blob).map(Blob::getContent);
    }

    @Override
    public @NotNull Map<String, byte[]> retrieve(@NotNull Collection<String> ids) {
        if (isEmpty(ids)) {
            return Collections.emptyMap();
        }

        List<Blob> blobs = bucket.get(ids);

        return blobs.stream()
                .filter(blob -> blob != null)
                .map(blob -> Tuple.of(blob.getBlobId().getName(), blob.getContent()))
                .collect(Collectors.toMap(t -> t._1, t -> t._2));
    }

    @Override
    public boolean delete(@NotNull String id) {
        GcsDeleteQueue toDelete = new GcsDeleteQueue();
        toDelete.setIdInStore(id);
        toDelete.setDeletedDate(dateTimeService.now());
        repository.save(toDelete);
        return true;
    }

    @Override
    public int delete(Collection<String> ids) {
        Instant now = dateTimeService.now();
        ids.forEach(id -> {
            GcsDeleteQueue toDelete = new GcsDeleteQueue();
            toDelete.setIdInStore(id);
            toDelete.setDeletedDate(now);
            repository.save(toDelete);
        });
        return ids.size();
    }

    @Override
    public Map<String, Either<StorageException, Boolean>> deleteBlobs(@NotNull Collection<String> ids) {
        List<BlobId> blobIds = ids.stream().distinct().map(id -> BlobId.of(bucket.getName(), id)).collect(Collectors.toList());
        StorageBatch batch = storage.batch();
        Map<String, Either<StorageException, Boolean>> results = new HashMap<>();

        blobIds.forEach(blobId -> batch.delete(blobId, new Storage.BlobSourceOption[0]).notify(new BatchResult.Callback<>() {
            @Override
            public void success(Boolean result) {
                results.put(blobId.getName(), Either.right(result));
            }

            @Override
            public void error(StorageException e) {
                results.put(blobId.getName(), Either.left(e));
            }
        }));

        Metrics.timer(DELETE_TIMER_NAME).record(() -> batch.submit());
        return results;
    }

    @Override
    public Page<Blob> getBlobsInGcs() {
        return storage.list(bucket.getName(), Storage.BlobListOption.pageSize(config.getOrphanedGcsDocumentDeletionPageSize()),
                Storage.BlobListOption.fields(Storage.BlobField.TIME_CREATED));
    }
}
