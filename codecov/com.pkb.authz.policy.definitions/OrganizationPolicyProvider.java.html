package com.pkb.authz.policy.definitions;

import com.pkb.authentication.principal.user.AuthenticatedProfessional;
import com.pkb.authz.policy.Policy;
import com.pkb.datamodel.team.TeamWithOrg;
import io.vavr.collection.Set;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.action.Action.READ;
import static com.pkb.authz.policy.PolicyBuilder.policyBuilder;
import static com.pkb.authz.policy.condition.CommonConditions.actorIsProfessionalInOrg;
import static io.vavr.API.Set;

public class OrganizationPolicyProvider implements PolicyProvider {
    @Override
    public @NotNull Set<Policy> policies() {
        return Set(
                permitReadWhenActorIsTeamProfessionalInSameOrg()
        );
    }

    private Policy permitReadWhenActorIsTeamProfessionalInSameOrg() {
        return policyBuilder()
                .describeAs("Professional can read associated Team")
                .forObjectType(TeamWithOrg.class)
                .forUserType(AuthenticatedProfessional.class)
                .when(actorIsProfessionalInOrg())
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }
}
