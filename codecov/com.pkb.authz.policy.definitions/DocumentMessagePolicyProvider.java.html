package com.pkb.authz.policy.definitions;

import com.pkb.authentication.principal.user.AuthenticatedIndividualProfessional;
import com.pkb.authentication.principal.user.AuthenticatedPatient;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authentication.principal.user.AuthenticatedTeamProfessional;
import com.pkb.authz.policy.Policy;
import com.pkb.datamodel.DocumentMessage;
import io.vavr.collection.Set;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.action.Action.READ;
import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.policy.PolicyBuilder.policyBuilder;
import static com.pkb.authz.policy.condition.CommonConditions.actorHasDecryptionAccessToOwner;
import static com.pkb.authz.policy.condition.CommonConditions.actorIsNotAccessFrozen;
import static com.pkb.authz.policy.condition.CommonConditions.actorIsOwner;
import static com.pkb.authz.policy.condition.CommonConditions.ownerIsNotAccessFrozen;
import static com.pkb.authz.policy.condition.CommonConditions.ownerIsNotSharingDisabled;
import static io.vavr.API.Set;

public class DocumentMessagePolicyProvider implements PolicyProvider {
    @Override
    public @NotNull Set<Policy> policies() {
        return searchPolicies()
                .union(readPolicies());
    }

    /**
     * <pre>
     * Search policies cannot include the consent or team data view criteria.
     * These must be applied at query time since they are specific to individual
     * </pre>
     */
    private Set<Policy> searchPolicies() {
        return Set(
                teamProfessionalCanSearchDocuments(),
                patientCanSearchDocuments(),
                individualProCanSearchDocuments());
    }

    private Set<Policy> readPolicies() {
        return Set(
                teamProfessionalCanRead(),
                patientCanReadDocuments(),
                individualProCanReadDocuments(),
                systemUserCanReadDocuments());
    }

    private Policy teamProfessionalCanSearchDocuments() {
        return policyBuilder()
                .describeAs("Team professional can search Documents")
                .forObjectType(DocumentMessage.class)
                .forUserType(AuthenticatedTeamProfessional.class)
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy patientCanSearchDocuments() {
        return policyBuilder()
                .describeAs("Patient can search Documents")
                .forObjectType(DocumentMessage.class)
                .forUserType(AuthenticatedPatient.class)
                .when(actorIsNotAccessFrozen())
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy individualProCanSearchDocuments() {
        return policyBuilder()
                .describeAs("Individual professional can search Documents")
                .forObjectType(DocumentMessage.class)
                .forUserType(AuthenticatedIndividualProfessional.class)
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy teamProfessionalCanRead() {
        return policyBuilder()
                .describeAs("Team professional can read Documents")
                .forObjectType(DocumentMessage.class)
                .forUserType(AuthenticatedTeamProfessional.class)
                .when(actorHasDecryptionAccessToOwner())
                // TDV is manifested as a filter.
                .and(ownerIsNotAccessFrozen())
                // Consent is manifested as a filter.
                // BTG is not implemented yet.
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }

    private Policy patientCanReadDocuments() {
        return policyBuilder()
                .describeAs("Patient can read Documents")
                .forObjectType(DocumentMessage.class)
                .forUserType(AuthenticatedPatient.class)
                .when(actorIsNotAccessFrozen())
                .and(actorHasDecryptionAccessToOwner())
                .and(ownerIsNotAccessFrozen())
                .and(actorIsOwner().or(
                        ownerIsNotSharingDisabled()))
                // Consent is manifested as a filter.
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }

    private Policy individualProCanReadDocuments() {
        return policyBuilder()
                .describeAs("Individual professional can read Documents")
                .forObjectType(DocumentMessage.class)
                .forUserType(AuthenticatedIndividualProfessional.class)
                .when(actorHasDecryptionAccessToOwner())
                .and(ownerIsNotAccessFrozen())
                .and(ownerIsNotSharingDisabled())
                // Consent is manifested as a filter.
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }

    private Policy systemUserCanReadDocuments() {
        return policyBuilder()
                .describeAs("System user can read Documents")
                .forObjectType(DocumentMessage.class)
                .forUserType(AuthenticatedSystemUser.class)
                .when(actorHasDecryptionAccessToOwner())
                .and(ownerIsNotAccessFrozen())
                .and(ownerIsNotSharingDisabled())
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }
}
