package com.pkb.authz.policy.definitions;

import com.pkb.authentication.principal.user.AuthenticatedIndividualProfessional;
import com.pkb.authentication.principal.user.AuthenticatedPatient;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authentication.principal.user.AuthenticatedTeamCoordinator;
import com.pkb.authentication.principal.user.AuthenticatedTeamProfessional;
import com.pkb.authz.policy.Policy;
import com.pkb.datamodel.user.Patient;
import io.vavr.collection.Set;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.action.Action.DELETE;
import static com.pkb.authz.action.Action.READ;
import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.policy.PolicyBuilder.policyBuilder;
import static com.pkb.authz.policy.condition.CommonConditions.actorHasConsent;
import static com.pkb.authz.policy.condition.CommonConditions.actorHasDecryptionAccessToOwner;
import static com.pkb.authz.policy.condition.CommonConditions.actorIsNotAccessFrozen;
import static com.pkb.authz.policy.condition.CommonConditions.isSamePerson;
import static com.pkb.authz.policy.condition.CommonConditions.ownerIsNotAccessFrozen;
import static io.vavr.API.Set;

public class PatientPolicyProvider implements PolicyProvider {

    @Override
    public @NotNull Set<Policy> policies() {
        return Set(
                permitReadByPatient(),
                permitReadByTeamProfessional(),
                permitReadByIndividualProfessional(),
                permitReadByTeamCoordinator(),
                permitReadBySystemUser(),
                permitSearchByTeamProfessional(),
                permitSearchByTeamCoordinator(),
                permitSearchByPatient(),
                permitSearchByIndividualProfessional(),
                permitSearchBySystemUser(),
                permitDeleteByTeamCoord()
        );
    }

    private Policy permitReadByPatient() {
        return policyBuilder()
                .describeAs("Carer is linked with patient and patient has not frozen access")
                .forObjectType(Patient.class)
                .forUserType(AuthenticatedPatient.class)
                .when(actorIsNotAccessFrozen())
                .and(isSamePerson().or(actorHasConsent()))
                .and(ownerIsNotAccessFrozen())
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }

    private Policy permitReadByTeamProfessional() {
        return policyBuilder()
                .describeAs("Team professional can read patient")
                .forObjectType(Patient.class)
                .forUserType(AuthenticatedTeamProfessional.class)
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }

    private Policy permitReadByIndividualProfessional() {
        return policyBuilder()
                .describeAs("Individual professional is linked with patient")
                .forObjectType(Patient.class)
                .forUserType(AuthenticatedIndividualProfessional.class)
                .when(actorHasConsent())
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }

    private Policy permitReadByTeamCoordinator() {
        return policyBuilder()
                .describeAs("Team coordinator can read patient")
                .forObjectType(Patient.class)
                .forUserType(AuthenticatedTeamCoordinator.class)
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }

    private Policy permitSearchByTeamProfessional() {
        return policyBuilder()
                .describeAs("Team professional can search patient")
                .forObjectType(Patient.class)
                .forUserType(AuthenticatedTeamProfessional.class)
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy permitSearchByTeamCoordinator() {
        return policyBuilder()
                .describeAs("Team coordinator can search patient")
                .forObjectType(Patient.class)
                .forUserType(AuthenticatedTeamCoordinator.class)
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy permitSearchByPatient() {
        return policyBuilder()
                .describeAs("Patient can search patient")
                .forObjectType(Patient.class)
                .forUserType(AuthenticatedPatient.class)
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy permitSearchByIndividualProfessional() {
        return policyBuilder()
                .describeAs("Individual professional can search patient")
                .forObjectType(Patient.class)
                .forUserType(AuthenticatedIndividualProfessional.class)
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy permitDeleteByTeamCoord() {
        return policyBuilder()
                .describeAs("Team coordinator can delete patient")
                .forObjectType(Patient.class)
                .forUserType(AuthenticatedTeamCoordinator.class)
                .then()
                .grantedPermissionTo(DELETE)
                .definePolicy();
    }

    private Policy permitReadBySystemUser() {
        return policyBuilder()
                .describeAs("System user is linked with patient")
                .forObjectType(Patient.class)
                .forUserType(AuthenticatedSystemUser.class)
                .when(actorHasDecryptionAccessToOwner())
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }

    private Policy permitSearchBySystemUser() {
        return policyBuilder()
                .describeAs("System user can search patient")
                .forObjectType(Patient.class)
                .forUserType(AuthenticatedSystemUser.class)
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }
}
