package com.pkb.authz.policy.definitions;

import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authz.policy.Policy;
import com.pkb.questionnaireservice.modelv2.QuestionnaireRequest;
import io.vavr.collection.Set;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.action.Action.READ;
import static com.pkb.authz.policy.PolicyBuilder.policyBuilder;
import static io.vavr.API.Set;

public class QuestionnaireRequestPolicyProvider implements PolicyProvider {
    @Override
    public @NotNull Set<Policy> policies() {
        return Set(
                systemUserCanReadQuestionnaireRequests()
        );
    }

    // Rules from https://pkbdev.atlassian.net/wiki/spaces/TECHNOLOGY/pages/1107165209/FHIR+API+Roadmap+ProcedureRequest
    // Cannot implement "Actor.Team = Target.Consultation.Team", which should be "Target.Consultation.Team in Actor.Org", here - done in query
    // BTG-is-active can be assumed for OrgSystemUser in his own Org
    private Policy systemUserCanReadQuestionnaireRequests() {
        return policyBuilder()
                .describeAs("System user can read questionnaire requests")
                .forObjectType(QuestionnaireRequest.class)
                .forUserType(AuthenticatedSystemUser.class)
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }
}
