package com.pkb.authz.policy.definitions;

import com.pkb.authentication.principal.user.AuthenticatedIndividualProfessional;
import com.pkb.authentication.principal.user.AuthenticatedPatient;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authentication.principal.user.AuthenticatedTeamProfessional;
import com.pkb.authz.policy.Policy;
import com.pkb.datamodel.RadiologyImage;
import io.vavr.API;
import io.vavr.collection.Set;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.policy.PolicyBuilder.policyBuilder;
import static com.pkb.authz.policy.condition.CommonConditions.actorHasDecryptionAccessToOwner;
import static com.pkb.authz.policy.condition.CommonConditions.actorIsNotAccessFrozen;
import static com.pkb.authz.policy.condition.CommonConditions.actorIsOwner;
import static com.pkb.authz.policy.condition.CommonConditions.ownerIsNotAccessFrozen;
import static com.pkb.authz.policy.condition.CommonConditions.ownerIsNotSharingDisabled;

public class RadiologyImagePolicyProvider implements PolicyProvider {

    @Override
    public @NotNull Set<Policy> policies() {
        return API.Set(
                teamProfessionalCanSearchRadiologyImages(),
                individualProfessionalCanSearchRadiologyImages(),
                patientCanSearchRadiologyImages(),
                systemUserCanSearchRadiologyImages()
        );
    }

    private Policy teamProfessionalCanSearchRadiologyImages() {
        return policyBuilder()
                .describeAs("Team professional can search radiology images")
                .forObjectType(RadiologyImage.class)
                .forUserType(AuthenticatedTeamProfessional.class)
                .when(actorHasDecryptionAccessToOwner())
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy individualProfessionalCanSearchRadiologyImages() {
        return policyBuilder()
                .describeAs("Individual professional can search radiology images")
                .forObjectType(RadiologyImage.class)
                .forUserType(AuthenticatedIndividualProfessional.class)
                .when(actorHasDecryptionAccessToOwner())
                .and(ownerIsNotAccessFrozen())
                .and(ownerIsNotSharingDisabled())
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy patientCanSearchRadiologyImages() {
        return policyBuilder()
                .describeAs("Patient can search radiology images")
                .forObjectType(RadiologyImage.class)
                .forUserType(AuthenticatedPatient.class)
                .when(actorIsNotAccessFrozen())
                .and(actorHasDecryptionAccessToOwner())
                .and(ownerIsNotAccessFrozen())
                .and(actorIsOwner().or(
                        ownerIsNotSharingDisabled()
                ))
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy systemUserCanSearchRadiologyImages() {
        return policyBuilder()
                .describeAs("System user can search radiology images")
                .forObjectType(RadiologyImage.class)
                .forUserType(AuthenticatedSystemUser.class)
                .when(actorHasDecryptionAccessToOwner())
                .and(ownerIsNotAccessFrozen())
                .and(ownerIsNotSharingDisabled())
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }
}
