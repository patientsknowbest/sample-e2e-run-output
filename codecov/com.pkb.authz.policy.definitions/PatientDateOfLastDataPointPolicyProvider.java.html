package com.pkb.authz.policy.definitions;

import com.pkb.authentication.principal.user.AuthenticatedPatient;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authz.policy.Policy;
import com.pkb.datamodel.projections.PatientDateOfLastDataPoint;
import io.vavr.API;
import io.vavr.collection.Set;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.policy.PolicyBuilder.policyBuilder;
import static com.pkb.authz.policy.condition.CommonConditions.actorIsNotAccessFrozen;

public class PatientDateOfLastDataPointPolicyProvider implements PolicyProvider {

    @Override
    public @NotNull Set<Policy> policies() {
        return API.Set(
                permitSearchByPatient(),
                permitSearchBySystemUser()
        );
    }

    private Policy permitSearchByPatient() {
        return policyBuilder()
                .describeAs("Patient can search for the latest data point of her own account")
                .forObjectType(PatientDateOfLastDataPoint.class)
                .forUserType(AuthenticatedPatient.class)
                .when(actorIsNotAccessFrozen())
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy permitSearchBySystemUser() {
        return policyBuilder()
                .describeAs("System user can search patients' latest data points")
                .forObjectType(PatientDateOfLastDataPoint.class)
                .forUserType(AuthenticatedSystemUser.class)
                .when(actorIsNotAccessFrozen())
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }
}
