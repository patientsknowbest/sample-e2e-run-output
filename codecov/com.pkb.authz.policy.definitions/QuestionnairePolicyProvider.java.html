package com.pkb.authz.policy.definitions;

import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authz.policy.Policy;
import com.pkb.questionnaireservice.modelv2.Questionnaire;
import io.vavr.collection.Set;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.action.Action.READ;
import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.policy.PolicyBuilder.policyBuilder;
import static io.vavr.API.Set;

public class QuestionnairePolicyProvider implements PolicyProvider {
    @Override
    public @NotNull Set<Policy> policies() {
        return Set(
                systemUserCanSearchQuestionnaires(),
                systemUserCanReadQuestionnaires()
        );
    }

    private Policy systemUserCanSearchQuestionnaires() {
        return policyBuilder()
                .describeAs("System user can search questionnaires")
                .forObjectType(Questionnaire.class)
                .forUserType(AuthenticatedSystemUser.class)
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    //This does not implement the full rules as outlined in https://pkbdev.atlassian.net/wiki/spaces/TECHNOLOGY/pages/1088159935/FHIR+API+Roadmap+Questionnaire
    //we can't yet support Target.Team âŠ† Actor.Org.Teams. This is handled in the query
    private Policy systemUserCanReadQuestionnaires() {
        return policyBuilder()
                .describeAs("System user can read questionnaires")
                .forObjectType(Questionnaire.class)
                .forUserType(AuthenticatedSystemUser.class)
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }
}
