package com.pkb.authz.policy.definitions;

import com.pkb.authentication.principal.user.AuthenticatedIndividualProfessional;
import com.pkb.authentication.principal.user.AuthenticatedPatient;
import com.pkb.authentication.principal.user.AuthenticatedTeamProfessional;
import com.pkb.authz.policy.Policy;
import com.pkb.datamodel.LabResult;
import io.vavr.collection.Set;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.policy.PolicyBuilder.policyBuilder;
import static com.pkb.authz.policy.condition.CommonConditions.actorHasDecryptionAccessToOwner;
import static com.pkb.authz.policy.condition.CommonConditions.actorIsNotAccessFrozen;
import static com.pkb.authz.policy.condition.CommonConditions.actorIsOwner;
import static com.pkb.authz.policy.condition.CommonConditions.ownerIsNotAccessFrozen;
import static com.pkb.authz.policy.condition.CommonConditions.ownerIsNotSharingDisabled;
import static io.vavr.API.Set;

/**
 * Encodes policies for accessing Lab Results. See the
 * <a href="https://pkbdev.atlassian.net/wiki/spaces/TECHNOLOGY/pages/127795254/FHIR+API+Roadmap+Appointment">documentation</a>
 */
public class LabResultPolicyProvider implements PolicyProvider {

    @Override
    public @NotNull Set<Policy> policies() {
        return Set(
                teamProCanSearchLabResults(),
                patientCanSearchLabResults(),
                individualProCanSearchLabResults()
        );
    }

    /**
     * Note that SEARCH operations cannot define the whole policy upfront.
     * Some parts of it (e.g. consent status) must be applied at the query
     * level, because they relate to properties of individual data points.
     *
     * Note that the 'Sharing Disabled' criteria can be applied for Patient and
     * Individual Pro policies upfront, however it cannot be included in the Team
     * Pro policy definition due to an OR condition which permits a Team Pro to see
     * the data IF the source of the data is their own team. This information is
     * not available upfront (it applies to an individual data point)
     */
    private Policy teamProCanSearchLabResults() {
        return policyBuilder()
                .describeAs("Team pro can search lab results")
                .forObjectType(LabResult.class)
                .forUserType(AuthenticatedTeamProfessional.class)
                .when(actorHasDecryptionAccessToOwner())
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy patientCanSearchLabResults() {
        return policyBuilder()
                .describeAs("Patient can search lab results")
                .forObjectType(LabResult.class)
                .forUserType(AuthenticatedPatient.class)
                .when(actorIsNotAccessFrozen())
                .and(actorHasDecryptionAccessToOwner())
                .and(ownerIsNotAccessFrozen())
                .and(actorIsOwner().or(
                        ownerIsNotSharingDisabled()
                ))

                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy individualProCanSearchLabResults() {
        return policyBuilder()
                .describeAs("Individual professional can search lab results")
                .forObjectType(LabResult.class)
                .forUserType(AuthenticatedIndividualProfessional.class)
                .when(actorHasDecryptionAccessToOwner())
                .and(ownerIsNotAccessFrozen())
                .and(ownerIsNotSharingDisabled())
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }
}
