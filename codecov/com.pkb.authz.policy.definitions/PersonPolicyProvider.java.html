package com.pkb.authz.policy.definitions;

import com.pkb.authentication.principal.user.AuthenticatedIndividualProfessional;
import com.pkb.authentication.principal.user.AuthenticatedProfessional;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authentication.principal.user.AuthenticatedTeamProfessional;
import com.pkb.authz.policy.Policy;
import com.pkb.datamodel.user.Person;
import com.pkb.datamodel.user.Professional;
import io.vavr.collection.Set;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.action.Action.DELETE;
import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.policy.PolicyBuilder.policyBuilder;
import static com.pkb.authz.policy.condition.CommonConditions.actorHasDecryptionAccessToOwner;
import static com.pkb.authz.policy.condition.CommonConditions.isSamePerson;
import static io.vavr.API.Set;

public class PersonPolicyProvider implements PolicyProvider {
    @Override
    public @NotNull Set<Policy> policies() {
        return Set(
                permitSearchWhenActorIsSameUser(AuthenticatedTeamProfessional.class, "Team professional"),
                permitSearchWhenActorIsSameUser(AuthenticatedIndividualProfessional.class, "Individual professional"),
                permitDeleteForSystemUser()
        );
    }

    private Policy permitSearchWhenActorIsSameUser(Class<? extends AuthenticatedProfessional> userTypeClass, String userTypeName) {
        return policyBuilder()
                .describeAs(userTypeName + " can search one of their users")
                .forObjectType(Professional.class)
                .forUserType(userTypeClass)
                .when(isSamePerson())
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    //TODO: PHR-11078 implement proper authorization for deletion service
    private Policy permitDeleteForSystemUser() {
        return policyBuilder()
                .describeAs("System users can delete")
                .forObjectType(Person.class)
                .forUserType(AuthenticatedSystemUser.class)
                .when(actorHasDecryptionAccessToOwner())
                .then()
                .grantedPermissionTo(DELETE)
                .definePolicy();
    }
}
