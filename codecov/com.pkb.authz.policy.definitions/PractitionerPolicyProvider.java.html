package com.pkb.authz.policy.definitions;

import com.pkb.authentication.principal.user.AuthenticatedIndividualProfessional;
import com.pkb.authentication.principal.user.AuthenticatedProfessional;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authentication.principal.user.AuthenticatedTeamProfessional;
import com.pkb.authz.policy.Policy;
import com.pkb.datamodel.user.Person;
import com.pkb.datamodel.user.Professional;
import io.vavr.collection.Set;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.action.Action.READ;
import static com.pkb.authz.policy.PolicyBuilder.policyBuilder;
import static com.pkb.authz.policy.condition.CommonConditions.isSamePerson;
import static io.vavr.API.Set;

public class PractitionerPolicyProvider implements PolicyProvider {

    @Override
    public @NotNull Set<Policy> policies() {
        return Set(
                permitReadWhenActorIsSameUser(AuthenticatedTeamProfessional.class, "Team professional"),
                permitReadWhenActorIsSameUser(AuthenticatedIndividualProfessional.class, "Individual professional"),
                permitReadWhenActorIsSystemUser("System user")
        );
    }

    private Policy permitReadWhenActorIsSameUser(Class<? extends AuthenticatedProfessional> userTypeClass, String userTypeName) {
        return policyBuilder()
                .describeAs(userTypeName + " can read one of their users")
                .forObjectType(Professional.class)
                .forUserType(userTypeClass)
                .when(isSamePerson())
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }

    private Policy permitReadWhenActorIsSystemUser(String userTypeName) {
        return policyBuilder()
                .describeAs(userTypeName + " can read one of their users")
                .forObjectType(Person.class) // Need org_admin,team_coord,privacy_officer as well as reg_clinician
                .forUserType(AuthenticatedSystemUser.class)
                .then()
                .grantedPermissionTo(READ)
                .definePolicy();
    }

}
