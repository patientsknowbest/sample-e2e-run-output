package com.pkb.authz.policy.definitions;

import com.pkb.authentication.principal.user.AuthenticatedPatient;
import com.pkb.authentication.principal.user.AuthenticatedTeamProfessional;
import com.pkb.authz.policy.Policy;
import com.pkb.datamodel.LocalIdType;
import io.vavr.collection.Set;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.policy.PolicyBuilder.policyBuilder;
import static com.pkb.authz.policy.condition.CommonConditions.actorIsNotAccessFrozen;
import static io.vavr.API.Set;

public class LocalIdTypePolicyProvider implements PolicyProvider {

    @Override
    public @NotNull Set<Policy> policies() {
        return Set(
                teamProfessionalCanSearchLocalIdTypes(),
                patientCanSearchLocalIdTypes()
        );
    }

    private Policy teamProfessionalCanSearchLocalIdTypes() {
        return policyBuilder()
                .describeAs("Team professional can search Local Id Types")
                .forObjectType(LocalIdType.class)
                .forUserType(AuthenticatedTeamProfessional.class)
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }

    private Policy patientCanSearchLocalIdTypes() {
        return policyBuilder()
                .describeAs("Patient can search Local Id Types")
                .forObjectType(LocalIdType.class)
                .forUserType(AuthenticatedPatient.class)
                .when(actorIsNotAccessFrozen())
                .then()
                .grantedPermissionTo(SEARCH)
                .definePolicy();
    }
}
