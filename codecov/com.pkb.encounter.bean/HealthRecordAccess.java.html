package com.pkb.encounter.bean;

import com.pkb.app.bean.accountkey.AccountKeyService;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.crypto.javasecurityutil.KeyUtil;
import com.pkb.encounter.HealthRecordAccessRemote;
import com.pkb.tolven.PKBAccountDAOLocal;
import com.pkb.user.PKBPersonLocal;
import com.pkb.util.AccountPrivateKeyGenerator;
import org.tolven.core.entity.AccountUser;
import org.tolven.security.key.AccountPrivateKey;

import javax.ejb.Remote;
import javax.ejb.Stateless;
import javax.inject.Inject;
import java.io.IOException;
import java.security.GeneralSecurityException;
import java.security.PrivateKey;
import java.security.PublicKey;
import java.util.UUID;

import static com.pkb.crypto.CryptoConstants.ACCESS_TOKEN_SECRET_KEY_DATA_LENGTH;
import static com.pkb.crypto.CryptoConstants.ACCESS_TOKEN_SECRET_KEY_DERIV_ALGORITHM;
import static com.pkb.crypto.CryptoConstants.ACCESS_TOKEN_SECRET_KEY_DERIV_ITERATIONS;
import static com.pkb.crypto.CryptoConstants.ACCESS_TOKEN_SECRET_KEY_DERIV_SALT_LENGTH;
import static com.pkb.crypto.CryptoConstants.DESEDE_KEY_LENGTH;
import static com.pkb.crypto.CryptoConstants.DESEDE_SHORT_NAME;

@Stateless(mappedName = "HealthRecordAccess")
@Remote(HealthRecordAccessRemote.class)
public class HealthRecordAccess implements HealthRecordAccessRemote {

    @Inject
    private PKBPersonLocal personBean;

    @Inject
    private AccountPrivateKeyGenerator generator;

    @Inject
    private AccountKeyService accountKeyService;

    @Inject
    private PKBAccountDAOLocal pkbAccountDAOLocal;

    @Override
    public AccountPrivateKey grant(LoggedInEHRRequestContext context, String secret, long accountId)
            throws GeneralSecurityException, IOException {

        AccountUser defaultAccountUser = personBean.getOwnerAccountUser(accountId);
        int acctPrivateKeySize = defaultAccountUser.getAccountPrivateKey().getKeySize();

        AccountPrivateKey accountPrivateKey = generator.getInstance();

        UUID accountPublicId = pkbAccountDAOLocal.getAccountPublicId(accountId);
        com.pkb.crypto.AccountPrivateKey clearPrivateKey = accountKeyService.getAccountKeys(context, accountId, accountPublicId).getPrivateKey();

        accountPrivateKey.init(
                KeyUtil.convertAccountPrivateKeyToPKCS8(clearPrivateKey),
                acctPrivateKeySize,
                DESEDE_SHORT_NAME,
                secret,
                ACCESS_TOKEN_SECRET_KEY_DERIV_ALGORITHM,
                ACCESS_TOKEN_SECRET_KEY_DATA_LENGTH,
                ACCESS_TOKEN_SECRET_KEY_DERIV_SALT_LENGTH,
                ACCESS_TOKEN_SECRET_KEY_DERIV_ITERATIONS);

        return accountPrivateKey;
    }

    @Override
    public AccountPrivateKey grant(AccountPrivateKey on, String securedBy, PublicKey privilegedKey) {
        try {

            PrivateKey privateKey = on.getPrivateKey(securedBy);


            AccountPrivateKey newAccountPrivateKey = AccountPrivateKey.getInstance();
            newAccountPrivateKey.init(privateKey, privilegedKey,
                    on.getKeySize(),
                    DESEDE_SHORT_NAME,
                    DESEDE_KEY_LENGTH);

            return newAccountPrivateKey;
        } catch (IOException | GeneralSecurityException e) {
            throw new RuntimeException("Unable to grant access", e);// TODO: improve logging according guidelines
        }

    }
}
