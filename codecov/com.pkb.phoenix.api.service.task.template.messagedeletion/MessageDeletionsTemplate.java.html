public abstract class MessageDeletionsTemplate extends AbstractPostProcessingTaskTemplate<MessageDeletionsParameters, MessageDeletionsResult, MessageDeletionsResult.ExtendedMessageDeletionsResult> {
    public static final String OUTBOUND_REQUEST_MSG_DELETION = "direct:outboundRequestMessageDeletion";
    private final String serviceAccountKey;

    public static class CieMessageDeletionsTemplate extends MessageDeletionsTemplate {
        public CieMessageDeletionsTemplate(ReactiveRedisTemplate<String, String> redis, ProducerTemplate producer, ObjectReader json, FreshDeskService freshDeskService, VelocityEngine templateEngine, String serviceAccountKey) {
            super(MSGD_CIE, redis, producer, json, freshDeskService, templateEngine, serviceAccountKey);
        }
    }

    public static class GenericMessageDeletionsTemplate extends MessageDeletionsTemplate {
        public GenericMessageDeletionsTemplate(ReactiveRedisTemplate<String, String> redis, ProducerTemplate producer, ObjectReader json, FreshDeskService freshDeskService, VelocityEngine templateEngine, String serviceAccountKey) {
            super(MSGD_GEN, redis, producer, json, freshDeskService, templateEngine, serviceAccountKey);

    //We need to provide an AbstractPubSubTaskTemplate which knows how to validate and dispatch the task to redis and will be passed in as a delegate to this template:
    private static class BaseMessageDeletionTemplate extends AbstractPubSubTaskTemplate<MessageDeletionsParameters, MessageDeletionTaskRequest, MessageDeletionTaskResponse, MessageDeletionsResult> {
        private static final RawParameterParser RAW_PARAMETER_PARSER = new RawParameterParser();

        BaseMessageDeletionTemplate(TaskTemplateCode code,
                                    ReactiveRedisTemplate<String, String> redis,
                                    ProducerTemplate producer,
                                    ObjectReader json) {
            super(code, MessageDeletionsParameters.class, new MessageDeletionsTranslator(code, RAW_PARAMETER_PARSER), redis, producer, OUTBOUND_REQUEST_MSG_DELETION, json);
        }

        @Override
        public List<String> validate(MessageDeletionsParameters parameters) {
            var batches = RAW_PARAMETER_PARSER.apply(parameters.rawData());

            return io.vavr.collection.List.ofAll(batches)
                    .zipWithIndex(this::validateMessageDeletionTaskParameters)
                    .flatMap(identity())
                    .collect(toUnmodifiableList());
        }

        @NotNull
        private List<String> validateMessageDeletionTaskParameters(RawParameterParser.Batch batch, Integer lineNumber) {
            var errors = new ArrayList<String>();

            validateFreshDeskId(batch.fdId)
                    .map(message -> format("%s (in line=[%s])", message, lineNumber + 1))
                    .ifPresent(errors::add);

            if (batch.messageIds.length > 0) {
                errors.addAll(Arrays.stream(batch.messageIds)
                        .map(ValidationUtil::validateMessageId)
                        .filter(Optional::isPresent)
                        .map(Optional::get)
                        .collect(toList()));
            } else {
                errors.add(format("Error at line [%s] = At least a freshdesk ticket and a message ID is mandatory", lineNumber + 1));
            }

            return errors;
        }
    }

    private final FreshDeskService freshDeskService;
    private final VelocityEngine templateEngine;

    public MessageDeletionsTemplate(TaskTemplateCode code, ReactiveRedisTemplate<String, String> redis, ProducerTemplate producer, ObjectReader json, FreshDeskService freshDeskService, VelocityEngine templateEngine, String serviceAccountKey) {
        super(new BaseMessageDeletionTemplate(code, redis, producer, json), json);
        this.freshDeskService = freshDeskService;
        this.templateEngine = templateEngine;
        this.serviceAccountKey = serviceAccountKey;
    }


    @Override
    protected Mono<ExtendedMessageDeletionsResult> postProcess(MessageDeletionsResult result, ContextView context) {
        var meta = context.<Meta>get(TaskService.META);
        var resultsByFdId = result.results().stream().collect(groupingBy(BatchResult::fdId));

        // We add notes to all tickets
        var notesAdded = meta.session().user().fdApiKey()
                .map(ImmutableApiKey::of)
                .map(apiKey -> {
                    var notesBeingAdded = resultsByFdId.entrySet().stream()
                            .map(entry -> freshDeskService.addNote(entry.getKey(), renderMessage(entry.getValue()), apiKey))
                            .collect(toList());

                    return zip(notesBeingAdded, Arrays::asList)
                            .map($ -> "Notes added");
                }).orElseGet(Mono::empty);


        // We close all tickets
        var closedTickets = meta.session().user().fdApiKey()
                .map(ImmutableApiKey::of)
                .map(apiKey -> {
                    var ticketsBeingClosed = resultsByFdId.keySet().stream()
                            .map(ticketId -> freshDeskService.closeTicket(ticketId, apiKey))
                            .collect(toList());

                    return zip(ticketsBeingClosed, Arrays::asList)
                            .map($ -> "Closed");
                }).orElseGet(Mono::empty);

        // when all notes are added and all tickets are closed, then we yield a response.
        return notesAdded.then(closedTickets)
                .map($1 -> "Done")
                .defaultIfEmpty("Skipped")
                .onErrorResume($ -> just("Failed"))
                .map(fdStatus -> ImmutableExtendedMessageDeletionsResult.builder()
                        .fdStatus(fdStatus)
                        .addAllResults(result.results())
                        .build());
    }

    private String renderMessage(List<BatchResult> results) {
        try (var writer = new StringWriter()) {

            var singleMessageDeletion = results.size() == 1;
            var templateFile = singleMessageDeletion ? "messageDeletedTemplate" : "messageDeletedBulkTemplate";
            var template = templateEngine.getTemplate(format("%s.vm", templateFile));

            var context = new VelocityContext();
            if (singleMessageDeletion) {
                var result = results.get(0);
                context.put("uniqueId", result.uniqueId());
                context.put("readByPatient", result.patientRead());
                context.put("readByCarer", result.carersRead());
                context.put("readByPro", result.prosRead());
            } else {
                context.put("results", results);
            }

            template.merge(context, writer);
            writer.flush();

            return writer.toString();
        } catch (IOException e) {
            throw new IllegalStateException(e);
        }
    }
