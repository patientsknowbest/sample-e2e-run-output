package com.pkb.phoenix.api.service.task.template.messagedeletion;

import java.util.List;
import java.util.function.Function;
import java.util.regex.Pattern;
import java.util.stream.Stream;

import static java.util.stream.Collectors.toUnmodifiableList;

public class RawParameterParser implements Function<String, List<RawParameterParser.Batch>> {
    private static final Pattern SPACE = Pattern.compile("\\s");
    private static final Pattern NEW_LINE = Pattern.compile("\\n");

    @Override
    public List<Batch> apply(String s) {
        return Stream.of(NEW_LINE.split(s))
                .map(this::processLine)
                .collect(toUnmodifiableList());
    }

    private Batch processLine(String line) {
        var bits = io.vavr.collection.List.of(SPACE.split(line));
        return new Batch(bits.head().trim(), bits.tail().map(String::trim).toJavaArray(String[]::new));
    }

    public static class Batch {
        public final String fdId;
        public final String[] messageIds;

        Batch(String fdId, String[] messageIds) {
            this.fdId = fdId;
            this.messageIds = messageIds;
        }
    }

}
