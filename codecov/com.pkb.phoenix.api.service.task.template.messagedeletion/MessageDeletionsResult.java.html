package com.pkb.phoenix.api.service.task.template.messagedeletion;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import org.immutables.value.Value.Immutable;

import java.util.List;

public interface MessageDeletionsResult {
    @Immutable
    @JsonSerialize(as = ImmutableBatchResult.class)
    @JsonDeserialize(as = ImmutableBatchResult.class)
    interface BatchResult {
        String fdId();

        String uniqueId();

        String outcome();

        String message();

        boolean patientRead();

        int carersRead();

        int prosRead();
    }

    List<BatchResult> results();

    @Immutable
    @JsonSerialize(as = ImmutableMessageDeletionsResult.class)
    @JsonDeserialize(as = ImmutableMessageDeletionsResult.class)
    abstract class AbstractMessageDeletionsResult implements MessageDeletionsResult {
    }

    @Immutable
    @JsonSerialize(as = ImmutableExtendedMessageDeletionsResult.class)
    @JsonDeserialize(as = ImmutableExtendedMessageDeletionsResult.class)
    interface ExtendedMessageDeletionsResult extends MessageDeletionsResult {
        String fdStatus();
    }
}
