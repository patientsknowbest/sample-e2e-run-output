package com.pkb.phoenix.api.service.task.template.messagedeletion;

import com.pkb.phoenix.api.service.task.template.TaskTemplateCode;
import com.pkb.phoenix.api.service.task.template.Translator;
import com.pkb.phoenix.repository.task.projection.Task;
import com.pkb.phoenix.shared.structures.request.ImmutableMessageDeletionRequestBatch;
import com.pkb.phoenix.shared.structures.request.ImmutableMessageDeletionTaskRequest;
import com.pkb.phoenix.shared.structures.request.MessageDeletionTaskRequest;
import com.pkb.phoenix.shared.structures.request.MessageDeletionTaskRequest.DeletionType;
import com.pkb.phoenix.shared.structures.response.MessageDeletionTaskResponse;
import reactor.util.context.ContextView;

import java.util.function.Function;

import static com.pkb.phoenix.api.service.task.TaskService.TASK_ID;
import static com.pkb.phoenix.api.service.task.template.TaskTemplateCode.MSGD_CIE;
import static com.pkb.phoenix.api.service.task.template.TaskTemplateCode.MSGD_GEN;
import static com.pkb.phoenix.shared.structures.request.MessageDeletionTaskRequest.DeletionType.CIE;
import static com.pkb.phoenix.shared.structures.request.MessageDeletionTaskRequest.DeletionType.GENERIC;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static java.util.Collections.emptyList;
import static java.util.stream.Collectors.toList;

public class MessageDeletionsTranslator implements Translator<MessageDeletionsParameters, MessageDeletionTaskRequest, MessageDeletionTaskResponse, MessageDeletionsResult> {
    private final DeletionType type;
    private final RawParameterParser rawParameterParser;

    public MessageDeletionsTranslator(TaskTemplateCode code, RawParameterParser rawParameterParser) {
        this.type = Match(code).of(
                Case($(MSGD_CIE), CIE),
                Case($(MSGD_GEN), GENERIC));
        this.rawParameterParser = rawParameterParser;
    }

    @Override
    public Function<Task<MessageDeletionsParameters>, MessageDeletionTaskRequest> toRequest(ContextView context) {
        return task -> {
            var deletions = rawParameterParser.apply(task.getParameters().rawData()).stream()
                    .map(batch -> new ImmutableMessageDeletionRequestBatch.Builder()
                            .freshDeskId(batch.fdId)
                            .addMessageIds(batch.messageIds)
                            .build())
                    .collect(toList());

            return new ImmutableMessageDeletionTaskRequest.Builder()
                    .type(type)
                    .taskId(task.getId())
                    .requestId(context.get(TASK_ID).toString())
                    .addAllDeletions(deletions)
                    .build();
        };
    }

    @Override
    public Function<MessageDeletionTaskResponse, MessageDeletionsResult> fromResponse(ContextView context) {
        return response -> {
            if (response.results() == null) {
                return ImmutableMessageDeletionsResult.builder()
                        .addAllResults(emptyList())
                        .build();
            }

            var results = response.results().stream()
                    .map(outcome -> ImmutableBatchResult.builder()
                            .fdId(outcome.freshdeskId())
                            .uniqueId(outcome.uniqueId())
                            .outcome(outcome.outcome())
                            .message(outcome.message())
                            .patientRead(outcome.patientRead())
                            .carersRead(outcome.carersRead())
                            .prosRead(outcome.prosRead())
                            .build())
                    .collect(toList());
            return ImmutableMessageDeletionsResult.builder()
                    .addAllResults(results)
                    .build();
        };
    }

    @Override
    public Class<MessageDeletionTaskResponse> responseType() {
        return MessageDeletionTaskResponse.class;
    }
}
