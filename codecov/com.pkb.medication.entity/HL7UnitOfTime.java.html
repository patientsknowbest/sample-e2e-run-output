package com.pkb.medication.entity;

import java.io.Serializable;

import org.jetbrains.annotations.Nullable;

public enum HL7UnitOfTime implements Serializable {

    SECOND("s", "hl7UnitOfTime.txt.second"), MINUTE("min", "hl7UnitOfTime.txt.minute"), HOUR("h", "hl7UnitOfTime.txt.hour"),
    DAY("d", "hl7UnitOfTime.txt.day"), WEEK("wk", "hl7UnitOfTime.txt.week"), MONTH("mo", "hl7UnitOfTime.txt.month"),
    YEAR("a", "hl7UnitOfTime.txt.year");

    private String code;
    private String value;

    HL7UnitOfTime(String code, String value) {
        this.code = code;
        this.value = value;
    }

    @Nullable
    public static HL7UnitOfTime getByCode(String code) {
        if ((code == null) || code.isEmpty()) {
            return null;
        }
        for (HL7UnitOfTime value : values()) {
            if (value.getCode().equals(code)) {
                return value;
            }
        }
        return null;
    }

    public String getCode() {
        return code;
    }

    public String getValue() {
        return value;
    }

}
