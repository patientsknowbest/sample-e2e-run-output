package com.pkb.medication.entity;

import java.util.Arrays;
import java.util.List;
import java.util.Optional;
import java.util.stream.Stream;

import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

/**
 * ID column is referenced by Medication.doseUnitPkbId
 * apiRef is used by the v1 API for sending / receiving medications
 * textKey is used to retrieve the description from the translation file, medication.properties
 */
public enum MedicationUnit {
    MG(         1L,  "mg", "medicationUnit.txt.mg"),
    ML(         2L,  "ml", "medicationUnit.txt.ml"),
    UNITS(      3L,  "units", "medicationUnit.txt.units"),
    TABLETS(    4L,  "tablets", "medicationUnit.txt.tablets"),
    MCG(        5L,  "mcg", "medicationUnit.txt.mcg"),
    DAYS(       6L,  "days", "medicationUnit.txt.days"),
    DROPS(      7L,  "drops", "medicationUnit.txt.drops"),
    INJECTION(  8L,  "injection", "medicationUnit.txt.injection"),
    PATCHES(    9L,  "patches", "medicationUnit.txt.patches"),
    PUFFS(      10L, "puffs", "medicationUnit.txt.puffs"),
    MEQ(        11L, "meq", "medicationUnit.txt.mEq"),
    MEQL(       12L, "meql", "medicationUnit.txt.mEql"),
    MGML(       13L, "mqml", "medicationUnit.txt.mgml"),
    GRAMS(      14L, "grams", "medicationUnit.txt.grams"),
    LMIN(       15L, null, "medicationUnit.txt.lmin"),
    NGHR(       16L, null, "medicationUnit.txt.nghr"),
    MGHR(       17L, null, "medicationUnit.txt.mghr"),
    SACHETS(    18L, null, "medicationUnit.txt.sachets"),
    NG(         19L, null, "medicationUnit.txt.ng"),
    SPRAY(      20L, null, "medicationUnit.txt.spray"),
    MILLION_UNITS(21L, null, "medicationUnit.txt.millionUnits"),
    MCG_HR(     22L, null, "medicationUnit.txt.mcgHr"),
    CAPSULES(   23L, null, "medicationUnit.txt.capsules"),
    DOSES(      24L, null, "medicationUnit.txt.doses"),
    MMOL(       25L, null, "medicationUnit.txt.mmol"),
    APPLICATIONS(26L, null, "medicationUnit.txt.applications");

    private final Long id;
    private final @Nullable String apiRef;
    private final @NotNull String textKey;

    MedicationUnit(Long id, @Nullable String apiRef, @NotNull String textKey) {
        this.id = id;
        this.apiRef = apiRef;
        this.textKey = textKey;
    }

    public long getId() {
        return id;
    }

    @Nullable
    public String getApiRef() {
        return apiRef;
    }

    @NotNull
    public String getTextKey() {
        return textKey;
    }

    public static Stream<MedicationUnit> stream() {
        return Arrays.stream(values());
    }

    public static List<MedicationUnit> list() {
        return Arrays.asList(values());
    }

    public static Optional<MedicationUnit> getForId(Long id) {
        return stream()
                .filter(it -> it.getId() == id)
                .findFirst();
    }

    public static String getTextKeyForId(Long id) {
        return getForId(id)
                .map(MedicationUnit::getTextKey)
                .orElse("");
    }
}
