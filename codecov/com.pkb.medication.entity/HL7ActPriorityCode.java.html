package com.pkb.medication.entity;

import java.io.Serializable;

import javax.annotation.Nullable;

import com.pkb.exception.PKBPluginException;

public enum HL7ActPriorityCode implements Serializable {

    // Insert as needed
    PRN("prn", "hl7ActPriorityCode.txt.prn"), UD("ud", "hl7ActPriorityCode.txt.ud");

    private String code;
    private String value;

    HL7ActPriorityCode(String code, String value) {
        this.code = code;
        this.value = value;
    }

    @Nullable
    public static HL7ActPriorityCode getByCode(String code) {
        if ((code == null) || code.isEmpty()) {
            return null;
        }
        for (HL7ActPriorityCode value : values()) {
            if (value.getCode().equals(code)) {
                return value;
            }
        }
        throw new PKBPluginException("Invalid HL7ActPriorityCode '" + code + "'");
    }

    public String getCode() {
        return code;
    }

    public String getValue() {
        return value;
    }

}
