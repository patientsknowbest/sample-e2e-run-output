package com.pkb.phoenix.api.wiring;

import com.fasterxml.jackson.databind.ObjectReader;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.phoenix.api.service.task.template.datagathering.patient.DataCollectionOutboundRouteBuilder;
import com.pkb.phoenix.api.service.task.template.datagathering.patient.DataCollectionTemplate;
import com.pkb.phoenix.api.wiring.config.PhoenixPubSubConfig;
import org.apache.camel.ProducerTemplate;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.data.redis.core.ReactiveRedisTemplate;

@Configuration
class DataCollectionConfiguration {
    @Bean
    DataCollectionOutboundRouteBuilder dataCollectionRouteBuilder(PhoenixPubSubConfig config,
                                                                  PubSubNamespaceService namespaceService,
                                                                  @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey) {
        return new DataCollectionOutboundRouteBuilder(namespaceService, config, serviceAccountKey);
    }

    @Bean
    DataCollectionTemplate patientDataCollectionTemplate(ReactiveRedisTemplate<String, String> redis,
                                                         ProducerTemplate producer,
                                                         ObjectReader json) {
        return new DataCollectionTemplate(redis,
                producer,
                json);
    }

}
