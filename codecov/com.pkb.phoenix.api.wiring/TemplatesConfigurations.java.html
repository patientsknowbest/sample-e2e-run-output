package com.pkb.phoenix.api.wiring;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.ObjectReader;
import com.fasterxml.jackson.datatype.guava.GuavaModule;
import com.pkb.phoenix.api.service.task.template.emaildeletion.TaskResponseListener;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.data.redis.core.ReactiveRedisOperations;

import static com.fasterxml.jackson.annotation.JsonAutoDetect.Visibility.ANY;
import static com.fasterxml.jackson.annotation.PropertyAccessor.FIELD;

@Configuration
public class TemplatesConfigurations {
    @Bean
    ObjectReader json() {
        return new ObjectMapper()
                .registerModules(new GuavaModule())
                .setVisibility(FIELD, ANY)
                .reader();
    }

    @Bean
    TaskResponseListener taskResponseListener(ReactiveRedisOperations<String, String> redis, ObjectMapper objectMapper) {
        return new TaskResponseListener(redis, objectMapper);
    }


}
