package com.pkb.phoenix.api.wiring;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.phoenix.api.service.freshdesk.ApiKeyFinder;
import com.pkb.phoenix.api.service.freshdesk.FreshDeskService;
import com.pkb.phoenix.api.service.freshdesk.RedisBackedApiKeyStore;
import com.pkb.phoenix.api.service.task.TaskMapper;
import com.pkb.phoenix.api.service.task.TaskService;
import com.pkb.phoenix.api.service.task.template.TaskTemplate;
import com.pkb.phoenix.api.service.task.template.TaskTemplateLocator;
import com.pkb.phoenix.api.service.tasktype.ImmutableTaskTypeDescription;
import com.pkb.phoenix.api.service.tasktype.TaskTypeService;
import com.pkb.phoenix.api.service.user.UserMapper;
import com.pkb.phoenix.api.service.user.UserService;
import com.pkb.phoenix.repository.task.TaskJooqRepository;
import com.pkb.phoenix.repository.taskeventlog.TaskEventLogRepository;
import com.pkb.phoenix.tables.daos.PUserDao;
import com.pkb.phoenix.wiring.RepositoryConfig;
import org.apache.velocity.app.VelocityEngine;
import org.apache.velocity.runtime.RuntimeConstants;
import org.apache.velocity.runtime.resource.loader.ClasspathResourceLoader;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Import;
import org.springframework.core.env.Environment;
import org.springframework.data.redis.core.ReactiveRedisTemplate;
import org.springframework.web.reactive.function.client.WebClient;
import reactor.core.scheduler.Scheduler;
import reactor.core.scheduler.Schedulers;

import java.util.List;

import static com.pkb.phoenix.api.service.tasktype.ImmutableTaskTypeDescription.taskTypeBuilder;
import static java.lang.String.format;
import static java.util.Comparator.comparing;
import static java.util.concurrent.Executors.newFixedThreadPool;
import static java.util.function.Function.identity;
import static java.util.stream.Collectors.toMap;
import static java.util.stream.Collectors.toUnmodifiableList;
import static org.mapstruct.factory.Mappers.getMapper;

@Configuration
@Import({RepositoryConfig.class})
public class ServiceConfig {
    @Bean
    TaskTypeService taskTypeService(List<TaskTemplate<?, ?>> templates, Environment env) {
        return new TaskTypeService(templates.stream()
                .map(template -> taskTypeBuilder()
                        .code(template.code())
                        .name(env.getRequiredProperty(format("phoenix.task.template.%s.name", template.code())))
                        .description(env.getRequiredProperty(format("phoenix.task.template.%s.description", template.code())))
                        .build())
                .sorted(comparing(ImmutableTaskTypeDescription::code))
                .collect(toUnmodifiableList()));
    }

    @Bean
    Scheduler scheduler() {
        return Schedulers.fromExecutor(newFixedThreadPool(10));
    }

    @Bean
    TaskService taskService(
            TaskJooqRepository taskRepository,
            TaskTemplateLocator taskTemplateLocator,
            TaskEventLogRepository taskEventLogRepository,
            DateTimeService dateTimeService,
            Scheduler scheduler,
            ObjectMapper json) {
        return new TaskService(taskRepository, taskEventLogRepository, taskTemplateLocator, dateTimeService, getMapper(TaskMapper.class), scheduler, json);
    }

    @Bean
    UserService userService(PUserDao dao, ApiKeyFinder finder) {
        return new UserService(dao, getMapper(UserMapper.class), finder);
    }

    @Bean
    TaskTemplateLocator taskTemplateLocator(List<TaskTemplate<?, ?>> templates) {
        return new TaskTemplateLocator(templates.stream().collect(toMap(TaskTemplate::code, identity())));
    }

    @Bean
    VelocityEngine templateEngine() {
        VelocityEngine engine = new VelocityEngine();
        engine.setProperty(RuntimeConstants.RESOURCE_LOADER, "classpath");
        engine.setProperty("classpath.resource.loader.class", ClasspathResourceLoader.class.getName());
        engine.setProperty(RuntimeConstants.FILE_RESOURCE_LOADER_CACHE, "true");
        engine.init();
        return engine;
    }

    @Bean
    FreshDeskService freshDeskService(WebClient.Builder builder, @Value("${phoenix.api.freshdesk.url}") String fdUrl) {
        return new FreshDeskService(builder.baseUrl(fdUrl).build());
    }

    @Bean
    RedisBackedApiKeyStore fdApiKeyService(
            ReactiveRedisTemplate<String, String> redisTemplate,
            FreshDeskService freshDeskService,
            @Value("${phoenix.api.redis.key.prefix}") String pattern) {
        return new RedisBackedApiKeyStore(redisTemplate.opsForValue(), freshDeskService, pattern);
    }
}
