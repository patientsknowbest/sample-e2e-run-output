package com.pkb.phoenix.api.wiring;

import com.fasterxml.jackson.databind.ObjectReader;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.phoenix.api.service.freshdesk.FreshDeskService;
import com.pkb.phoenix.api.service.task.template.twofaremoval.TwoFARemovalOutboundRouteBuilder;
import com.pkb.phoenix.api.service.task.template.twofaremoval.TwoFARemovalTemplate;
import com.pkb.phoenix.api.wiring.config.PhoenixPubSubConfig;
import org.apache.camel.ProducerTemplate;
import org.apache.velocity.app.VelocityEngine;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.data.redis.core.ReactiveRedisTemplate;

@Configuration
public class TwoFARemovalConfiguration {

    @Bean
    TwoFARemovalOutboundRouteBuilder twoFARemovalOutboundRouteBuilder(PhoenixPubSubConfig config,
                                                                      PubSubNamespaceService namespaceService,
                                                                      @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey){
        return new TwoFARemovalOutboundRouteBuilder(config, namespaceService, serviceAccountKey);
    }

    @Bean
    TwoFARemovalTemplate twoFARemovalTemplate(ReactiveRedisTemplate<String, String> redis,
                                              ProducerTemplate producer,
                                              VelocityEngine templateEngine,
                                              FreshDeskService freshDeskService,
                                              DateTimeService dateTimeService,
                                              ObjectReader json) {
        return new TwoFARemovalTemplate(
                redis,
                producer,
                templateEngine,
                freshDeskService,
                dateTimeService,
                json
        );
    }
}
