package com.pkb.phoenix.api.wiring;

import com.fasterxml.jackson.databind.ObjectReader;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.phoenix.api.service.freshdesk.FreshDeskService;
import com.pkb.phoenix.api.service.task.template.messagedeletion.MessageDeletionsOutboundRouteBuilder;
import com.pkb.phoenix.api.service.task.template.messagedeletion.MessageDeletionsTemplate;
import com.pkb.phoenix.api.service.task.template.messagedeletion.MessageDeletionsTemplate.CieMessageDeletionsTemplate;
import com.pkb.phoenix.api.service.task.template.messagedeletion.MessageDeletionsTemplate.GenericMessageDeletionsTemplate;
import com.pkb.phoenix.api.wiring.config.PhoenixPubSubConfig;
import org.apache.camel.ProducerTemplate;
import org.apache.velocity.app.VelocityEngine;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.data.redis.core.ReactiveRedisTemplate;

@Configuration
class MessageDeletionConfiguration {
    @Bean
    MessageDeletionsTemplate cieMessageDeletionTemplate(ReactiveRedisTemplate<String, String> redis,
                                                        ProducerTemplate producer,
                                                        ObjectReader json,
                                                        FreshDeskService freshDeskService,
                                                        VelocityEngine templateEngine,
                                                        @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey) {
        return new CieMessageDeletionsTemplate(redis, producer, json, freshDeskService, templateEngine, serviceAccountKey);
    }

    @Bean
    MessageDeletionsTemplate genericMessageDeletionTemplate(ReactiveRedisTemplate<String, String> redis,
                                                            ProducerTemplate producer,
                                                            ObjectReader json,
                                                            FreshDeskService freshDeskService,
                                                            VelocityEngine templateEngine,
                                                            @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey) {
        return new GenericMessageDeletionsTemplate(redis, producer, json, freshDeskService, templateEngine, serviceAccountKey);
    }

    @Bean
    MessageDeletionsOutboundRouteBuilder messageDeletionRouteBuilder(PhoenixPubSubConfig config,
                                                                     PubSubNamespaceService namespaceService,
                                                                     @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") String serviceAccountKey) {
        return new MessageDeletionsOutboundRouteBuilder(namespaceService, config, serviceAccountKey);
    }
}
