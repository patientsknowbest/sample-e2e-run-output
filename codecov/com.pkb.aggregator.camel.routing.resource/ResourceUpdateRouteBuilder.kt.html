package com.pkb.aggregator.camel.routing.resource

import ca.uhn.fhir.rest.server.exceptions.ResourceGoneException
import ca.uhn.fhir.rest.server.exceptions.ResourceNotFoundException
import com.pkb.aggregator.camel.NEXT_ROUTE_PROPERTY
import com.pkb.aggregator.camel.UPSTREAM_SOURCE_ID_PROPERTY
import com.pkb.aggregator.camel.bean.process.ResourcePreProcessor
import com.pkb.aggregator.camel.beanMethod
import com.pkb.aggregator.camel.routing.AbstractAggregatorRouteBuilder
import com.pkb.aggregator.camel.routing.RouteResolver
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import org.apache.camel.LoggingLevel
import org.springframework.stereotype.Component

@Component
class ResourceUpdateRouteBuilder(pubSubNamespaceService: PubSubNamespaceService) :
    AbstractAggregatorRouteBuilder(pubSubNamespaceService) {

    companion object {
        const val INCOMING_UPDATE_NOTIFICATION = "direct:aggregator/incoming"

        const val REQUEST_CREATE = "direct:aggregator/create"
        const val REQUEST_UPDATE = "direct:aggregator/update"
        const val REQUEST_DELETE = "direct:aggregator/delete"

        private const val ROUTE_ID = "resource-change-notification"
        private const val ROUTE_ID_CREATE = "$ROUTE_ID-create"
        private const val ROUTE_ID_UPDATE = "$ROUTE_ID-update"
        private const val ROUTE_ID_DELETE = "$ROUTE_ID-delete"
    }

    override fun configure() {

        registerDeadLetter()
        registerDownstreamDuplicateHandler()

        // Take an incoming message and reroute it based on create/update/delete operation type
        from(INCOMING_UPDATE_NOTIFICATION).routeId(ROUTE_ID)
            .beanMethod(RouteResolver::resolvePubsubUpdateAction).id("$ROUTE_ID-resolve-action")
            .toD("\${exchangeProperty[$NEXT_ROUTE_PROPERTY]}").id("$ROUTE_ID-process")

        // We can treat updates and creates the same, since the downstream route knows how to handle both
        from(REQUEST_CREATE).routeId(ROUTE_ID_CREATE)
            .to(REQUEST_UPDATE).id("$ROUTE_ID_CREATE-route-to-update")

        // For an update, we need the upstream resource. So we resolve a read route, get the resource, convert to R4
        // if needed, apply a source tag, and send it off to the correct route for the resource type. If the resource
        // isn't present in the upstream we'll log it but otherwise do nothing; if it's missing in the upstream it means
        // that it's been deleted at some point subsequent to the operation we're currently processing, and since the
        // aggregated store is meant to be a snapshot of the state of the world, we can just ignore it.
        from(REQUEST_UPDATE).routeId(ROUTE_ID_UPDATE)
            .onException(ResourceGoneException::class.java, ResourceNotFoundException::class.java)
                .handled(true)
                .log(LoggingLevel.WARN, "\${exchangeProperty.$UPSTREAM_SOURCE_ID_PROPERTY} missing from upstream; skipping")
            .end()
            .beanMethod(RouteResolver::resolveUpstreamRouteForNotification).id("$ROUTE_ID_UPDATE-resolve-upstream")
            .toD("\${exchangeProperty[$NEXT_ROUTE_PROPERTY]}", -1).id("$ROUTE_ID_UPDATE-retrieve-upstream")
            .beanMethod(ResourcePreProcessor::associateWithSource).id("$ROUTE_ID_UPDATE-associate-with-source")
            .beanMethod(RouteResolver::resolveResourceProcessor).id("$ROUTE_ID_UPDATE-resolve-processor")
            .toD("\${exchangeProperty[$NEXT_ROUTE_PROPERTY]}").id("$ROUTE_ID_UPDATE-process")

        // Resolve a deleter based on resource type and reroute to it
        from(REQUEST_DELETE).routeId(ROUTE_ID_DELETE)
            .beanMethod(RouteResolver::resolveResourceDeleter).id("$ROUTE_ID_DELETE-resolve-deleter")
            .toD("\${exchangeProperty[$NEXT_ROUTE_PROPERTY]}").id("$ROUTE_ID_DELETE-process")
    }
}
