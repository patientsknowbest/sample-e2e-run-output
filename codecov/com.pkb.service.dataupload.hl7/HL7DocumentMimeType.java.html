package com.pkb.service.dataupload.hl7;

public enum HL7DocumentMimeType {
    TEXT_RTF("text/rtf"),
    TEXT_PDF("text/pdf"),
    TEXT_MSWORD("text/msword"),
    TEXT_DOCX("text/vnd.openxmlformats-officedocument.wordprocessingml.document"),
    APPLICATION_PDF("application/pdf"),
    APPLICATION_MSWORD("application/msword"),
    APPLICATION_DOCX("application/vnd.openxmlformats-officedocument.wordprocessingml.document");

    private final String mimeType;

    HL7DocumentMimeType(String name) {
        this.mimeType = name;
    }

    @Override
    public String toString() {
        return mimeType;
    }

    static HL7DocumentMimeType getConvertedMediaType(String mediaType) throws IllegalArgumentException {
        switch (mediaType) {
        case "application/pdf":
        case "text/pdf":
            return APPLICATION_PDF;
        case "application/msword":
        case "text/msword":
            return APPLICATION_MSWORD;
        case "application/vnd.openxmlformats-officedocument.wordprocessingml.document":
        case "text/vnd.openxmlformats-officedocument.wordprocessingml.document":
            return APPLICATION_DOCX;
        case "text/rtf":
            return TEXT_RTF;
        default:
            throw new IllegalArgumentException("MimeTypeNotYetSupported");
        }
    }
}
