package com.pkb.service.dataupload.hl7;

/**
 * When parsing an HL7 XML document, the output of that process is an
 * {@code HL7XmlDoc} and an {@code OrgTeam}. This class is a wrapper for both
 * these objects.
 */
public class HL7Wrapper {

    private HL7XmlDoc hl7Xml;
    private HL7ConnContext hl7ConnContext;

    public HL7Wrapper() {
    }

    public HL7Wrapper(HL7XmlDoc hl7Xml, HL7ConnContext hl7ConnContext) {
        this.hl7Xml = hl7Xml;
        this.hl7ConnContext = hl7ConnContext;
    }

    void setHL7Xml(HL7XmlDoc hl7Xml) {
        this.hl7Xml = hl7Xml;
    }

    public HL7XmlDoc getHL7Xml() {
        return hl7Xml;
    }

    void setHl7ConnContext(HL7ConnContext hl7ConnContext) {
        this.hl7ConnContext = hl7ConnContext;
    }

    public HL7ConnContext getHl7ConnContext() {
        return hl7ConnContext;
    }
}
