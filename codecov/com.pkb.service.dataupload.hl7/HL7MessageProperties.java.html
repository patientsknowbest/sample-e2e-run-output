package com.pkb.service.dataupload.hl7;

import java.util.TimeZone;

/**
 * Holds properties for a specific HL7 message.
 */
public class HL7MessageProperties {

    /**
     * The timestamp offset to be used for any timestamps within this
     * HL7message that do not include one themselves. This field is set to the
     * timestamp offset supplied in MSH-7, in accordance with the specs.
     *
     * If MSH-7 does not specify a timestamp offset, then the handling will
     * will fall back to using the timezone offset of the location
     * of the sender if present.
     *
     * Otherwise, we will finally fall back to UTC (+0000).
     */
    private String defaultTimestampOffset;

    /**
     * This is the time zone of the location of the sender of this HL7 message.
     * Used in falling back to timezone offset for dates that don't contain them
     * if there is also no tz offset in MSH-7.
     */
    private TimeZone senderTimeZone;

    public HL7MessageProperties() {
    }

    public String getDefaultTimestampOffset() {
        return defaultTimestampOffset;
    }

    public void setDefaultTimestampOffset(String defaultTimestampOffset) {
        this.defaultTimestampOffset = defaultTimestampOffset;
    }

    public TimeZone getSenderTimeZone() {
        return senderTimeZone;
    }

    public void setSenderTimeZone(TimeZone senderTimeZone) {
        this.senderTimeZone = senderTimeZone;
    }
}
