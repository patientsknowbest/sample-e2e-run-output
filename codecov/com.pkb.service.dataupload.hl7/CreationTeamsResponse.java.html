package com.pkb.service.dataupload.hl7;

import com.pkb.institute.entity.Team;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.util.HashSet;
import java.util.Set;

import static java.util.Collections.emptySet;

public class CreationTeamsResponse {

    private final Team creatingTeam;
    private final Set<Team> creationTeams;

    private CreationTeamsResponse(Team creatingTeam, Set<Team> creationTeams) {
        this.creatingTeam = creatingTeam;
        this.creationTeams = creationTeams == null ? new HashSet<>() : new HashSet<>(creationTeams);
        if (creatingTeam != null) {
            this.creationTeams.add(creatingTeam);
        }
    }

    static CreationTeamsResponse of(Team creatingTeam, Set<Team> creationTeams) {
        return new CreationTeamsResponse(creatingTeam, creationTeams);
    }

    static CreationTeamsResponse of(Team creatingTeam) {
        return new CreationTeamsResponse(creatingTeam, emptySet());
    }

    /**
     * @return the Team which created the patient.
     * See PHR-8280 for a discussion on whether this concept is still meaningful.
     */
    @Nullable
    public Team getCreatingTeam() {
        return creatingTeam;
    }

    /**
     * @return all teams into which the patient will be created.
     */
    @NotNull
    public Set<Team> getCreationTeams() {
        return creationTeams;
    }

}
