package com.pkb.service.dataupload.hl7;

import com.google.common.collect.ImmutableMap;
import org.jetbrains.annotations.Contract;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.util.Arrays;
import java.util.function.Function;

/**
 * All HL7 messages comprise:
 * <ul>
 * <li>segments: one per line, starting with the segment name (like MSH or QRD) and followed by fields</li>
 * <li>each segment has fields separated by the | character. Notation example: the third field in the “ORC” segment is
 * “ORC-3” for the MSH segment only, the first field has no field separator; it’s one character that is the field
 * separator</li>
 * <li>each field can be split into components by the ^ character. Notation is similar: the first component of MSH-9 is
 * noted as MSH-9.1</li>
 * <li>each component can be split into subcomponents with the & character</li>
 * <li>within a field that supports repetition, you can separate sets of components with the repetition character: ~</li>
 * </ul>
 *
 * @see <a href="http://dev.patientsknowbest.com/home/hl7-api/hl7-message-specifications">HL7 Message Specifications</a>
 */
public enum HL7MessageType {

    /** Query for patient demographics and registration status */
    QRY_A19("QRY^A19"),
    /** Create a new medical record */
    ADT_A28("ADT^A28"),
    /** Update an existing medical record */
    ADT_A31("ADT^A31"),
    /** Update an Encounter */
    ADT_A08("ADT^A08"),
    /** Admit a patient */
    ADT_A01("ADT^A01"),
    /** Transfer a patient */
    ADT_A02("ADT^A02"),
    /** Discharge a patient */
    ADT_A03("ADT^A03"),
    /** Cancel an admission */
    ADT_A11("ADT^A11"),
    /** Cancel a transfer */
    ADT_A12("ADT^A12"),
    /** Cancel a discharge */
    ADT_A13("ADT^A13"),
    /** Pre-Admit a Patient */
    ADT_A05("ADT^A05"),
    /** Cancel Pre-Admit */
    ADT_A38("ADT^A38"),
    /** Pending Admit */
    ADT_A14("ADT^A14"),
    /** Cancel Pending Admit */
    ADT_A27("ADT^A27"),
    /** Send test results, measurements or radiology reports to PKB */
    ORU_R01("ORU^R01"),
    /** Send a new care plan for a patient or a document*/
    MDM_T02("MDM^T02"),
    /** Delete a document */
    MDM_T11("MDM^T11"),
    /** Create a new appointment for a patient */
    SIU_S12("SIU^S12"),
    /** Update an existing appointment for a patient */
    SIU_S13("SIU^S13"),
    /** Update an existing appointment for a patient */
    SIU_S14("SIU^S14"),
    /** Cancel an appointment for a patient */
    SIU_S15("SIU^S15"),
    /** Set DNA for an appointment */
    SIU_S26("SIU^S26");

    private static ImmutableMap<String,HL7MessageType> LOOKUP = Arrays.stream(HL7MessageType.values())
            .collect(ImmutableMap.toImmutableMap(HL7MessageType::getCode, Function.identity()));

    @Nullable
    public static HL7MessageType get(@NotNull String code)
    {
        return LOOKUP.get(code);
    }

    private final String code;

    @Contract(pure = true)
    HL7MessageType(String code) {
        this.code = code;
    }

    @Contract(pure = true)
    public String getCode() {
        return this.code;
    }

    public String toString() {
        return getCode();
    }
}
