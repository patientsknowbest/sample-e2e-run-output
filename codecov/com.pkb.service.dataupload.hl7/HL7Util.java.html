package com.pkb.service.dataupload.hl7;

import com.pkb.exception.ApiCallMalformedException;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.Nullable;

import java.util.Arrays;
import java.util.Optional;
import java.util.stream.Collectors;

import static com.pkb.service.dataupload.hl7.HL7ApiHelper.VALID_MATCH_RULES;
import static org.apache.commons.lang3.StringUtils.isEmpty;

/**
 * Utility class to do some HL7 related work.
 */
public class HL7Util {

    /**
     * Returns a {@code String} representation of the date component of the provided timestamp in the form: yyyy-MM-dd
     *
     * If the provided timestamp is < 8 characters long, null will be returned. No validation is performed on the provided timestamp, and no
     * time zone processing is applied. This method simply manipulates the textual data.
     *
     * @param hl7Timestamp
     *            A timestamp in HL7 format
     * @return The date component, in yyyy-MM-dd format
     */
    @Nullable
    public static String getHL7DateAsString(String hl7Timestamp) {
        if ((hl7Timestamp != null) && (hl7Timestamp.length() >= 8)) {
            return hl7Timestamp.substring(0, 4) + "-" + hl7Timestamp.substring(4, 6) + "-" + hl7Timestamp.substring(6, 8);
        }
        return null;
    }

    /**
     * Converts the specified year, month and day values into an HL7 formatted
     * timestamp, in the form: <year><month><day>, with each value left-padded
     * as required. For example: 20150101
     *
     * This method only performs string manipulations, it does not process or
     * validate the date itself.
     *
     * @param year
     *            The year.
     * @param month
     *            The month, treating January as 1.
     * @param day
     *            The day of the month.
     * @return An HL7 formatted timestamp.
     */
    public static String getHL7Timestamp(int year, int month,
                                         int day) {
        return leftPad(String.valueOf(year), 4)
                + leftPad(String.valueOf(month), 2)
                + leftPad(String.valueOf(day), 2);
    }

    /**
     * Pad the specified value with "0" on the left hand side until the length
     * of the value is equal to the specified length.
     *
     * If the length of the value is already >= the specified length, the same
     * value will be returned.
     *
     * @param value
     *            The value to pad.
     * @param length
     *            The minimum length of the returned value.
     * @return The padded value.
     */
    private static String leftPad(String value, int length) {
        String s = value;
        while (s.length() < length) {
            s = "0" + s;
        }
        return s;
    }

    /**
     * PKB allows senders to specify a whole number of days by which a result should be delayed from being displayed to
     * the patient. This value can be set in OBX-13 of some HL7 messages (e.g. MDM T02, ORU R01) in the form
     * {patientDelay:Xdays}.
     *
     * Note: the parentheses are currently optional for backwards compatibility.
     *
     * @param metadata The JSON-like formatted metadata, usually the contents of OBX-13
     * @param obxTrace An indication of which segment the value was found in (useful for debugging)
     * @return An Optional number of delay days
     * @throws ApiCallMalformedException
     *             If "patientDelay:" was present in the metadata but a valid delay number could not be parsed
     */
    public static Optional<Long> getPatientDelayDays(String metadata, String obxTrace) throws ApiCallMalformedException {
        Long delayDays = null;

        String patientDelayIdent = "patientDelay:";

        if (StringUtils.isNotBlank(metadata) && metadata.contains(patientDelayIdent)) {
            int from = metadata.indexOf(patientDelayIdent) + patientDelayIdent.length();
            int to = metadata.indexOf("days", from);
            if (to < 0) {
                throw new ApiCallMalformedException(obxTrace
                        + ": malformed patient delay in OBX-13.1; must be in format {patientDelay:Xdays} - got "
                        + metadata);
            }

            String daysDelayString = metadata.substring(from, to);

            if (StringUtils.isNotBlank(daysDelayString) && StringUtils.isNumeric(daysDelayString)) {
                delayDays = Long.valueOf(daysDelayString);
            } else {
                throw new ApiCallMalformedException(obxTrace
                        + ": malformed patient delay in OBX-13.1; must be in format {patientDelay:Xdays} - got "
                        + metadata);
            }
        }

        return Optional.ofNullable(delayDays);
    }

    public static String cleanHL7ValidatePidList(String hl7ValidatePidList) {
        if (isEmpty(hl7ValidatePidList)) {
            return "";
        }
        return  Arrays.stream(hl7ValidatePidList.split(",")).map(StringUtils::trim)
                .map(matchRule -> {
                    if (VALID_MATCH_RULES.contains(matchRule)) {
                        return matchRule;
                    } else {
                        throw new IllegalStateException(matchRule + " is not a known match rule");
                    }
                }).collect(Collectors.joining(", "));
    }
}
