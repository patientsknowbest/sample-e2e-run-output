/**
 * Fill in the right properties from combinations of connecting org, source org, source team, and partner
 * <p>
 * There are a few entities involved in the connection and the original source/data controller of an HL7 message
 * 1. The source org or team/org: this is the data controller for this data
 * 2. The connecting org (or authorizing org): often the same as (1); but this is the org that is connecting, or who authorized a partner to connect
 * 3. (optional) The HL7 Partner: a third party authorized by (2) to submit data on behalf of (1) (the two orgs are likely a single org)
 * <p>
 * Take these all as input; this class needs to contain all of the config
 * Much of the processing of HL7 messages needs to work regardless of whether the connecting institution is an Organisation or a Team. This
 * class is intended to encapsulate this information to prevent an insane amount of duplicated null checking. Note: An Org can always be
 * retrieved from a Team, but the reverse is not true. As such, this class prioritises the Team, if present, else works on the Org. Callers
 * should use isTeam(), rather than rely on a non-null Org to determine the original connection type.
 */
public class HL7ConnContextFactory {

    private static final Pattern COMMA_PATTERN = Pattern.compile(",");

    public static Set<String> splitCsv(@Nullable String csv) {
        return ofNullable(csv)
                .map(c -> Arrays.stream(COMMA_PATTERN.split(c))
                        .map(String::trim)
                        .collect(Collectors.toSet()))
                .map(collection -> {
                    collection.removeIf(String::isEmpty);
                    return collection;
                })
                .orElse(emptySet());
    }

    /**
     * @param sourceOrg The source / data controller org for this connection
     */

    public static HL7ConnContext create(@NotNull Org connectingSourceOrg, @NotNull UUID correlationId) {
        return create(connectingSourceOrg, correlationId, Optional.empty());
    }

    public static HL7ConnContext create(@NotNull Org connectingSourceOrg, @NotNull UUID correlationId, @NotNull String responsibleOrganisationIdentifier) {
        return create(connectingSourceOrg, correlationId, Optional.of(responsibleOrganisationIdentifier));
    }

    public static HL7ConnContext create(@NotNull Org connectingSourceOrg, @NotNull UUID correlationId,
                                        @NotNull Optional<String> maybeResponsibleOrganisationIdentifier) {
        return create(connectingSourceOrg, correlationId, maybeResponsibleOrganisationIdentifier, HL7_API);
    }

    public static HL7ConnContext create(@NotNull Org connectingSourceOrg, @NotNull UUID correlationId,
                                        @NotNull Optional<String> maybeResponsibleOrganisationIdentifier, @NotNull Route route) {
        Team firstTeam = connectingSourceOrg.getTeams().iterator().next();
        return ImmutableHL7ConnContext.builder()
                .sourceOrg(connectingSourceOrg)
                .connectingOrg(connectingSourceOrg)
                .hl7AutoCreateRecords(connectingSourceOrg.getHl7AutoCreateRecords())
                .hl7CheckNhsNumberStatus(connectingSourceOrg.getHl7CheckNhsNumberStatus())
                .timeZoneId(connectingSourceOrg.getTimeZoneId())
                .correlationId(correlationId)
                .maybeResponsibleOrganisationIdentifier(maybeResponsibleOrganisationIdentifier)

                // lists
                .addAllDataUploadIpWhitelist(splitCsv(connectingSourceOrg.getDataUploadIpWhitelist()))
                .addAllHl7UploadSourceWhitelist(splitCsv(connectingSourceOrg.getHl7UploadSourceWhitelist()))
                .addAllHl7UploadValidatePidList(splitCsv(connectingSourceOrg.getHl7UploadValidatePidList()))

                // optional
                .nationalIdType(getPrimaryNationalIdTypeFromCountryCode(firstTeam.getCountry()))
                .apiRoute(route)
                .build();
    }

    public static @NotNull HL7ConnContext create(@NotNull Hl7PartnerGrant partnerOrgGrant, @NotNull Org connectingSourceOrg,
                                                 @NotNull UUID correlationId, @NotNull Optional<String> maybeResponsibleOrganisationIdentifier) {
        Hl7Partner partner = partnerOrgGrant.getHl7Partner();
        return ImmutableHL7ConnContext.builder()
                .sourceOrg(connectingSourceOrg)
                .connectingOrg(connectingSourceOrg)
                .hl7AutoCreateRecords(partnerOrgGrant.isAutoCreateRecords())
                .hl7CheckNhsNumberStatus(partnerOrgGrant.isCheckNhsNumberStatus())
                .timeZoneId(partner.getTimeZoneId())
                .correlationId(correlationId)
                .hl7Partner(partner)
                // lists
                .addAllDataUploadIpWhitelist(splitCsv(partner.getIpWhitelist()))
                .addAllHl7UploadSourceWhitelist(splitCsv(connectingSourceOrg.getHl7UploadSourceWhitelist()))
                .addAllHl7UploadValidatePidList(splitCsv(partnerOrgGrant.getValidatePidList()))

                // optional
                .nationalIdType(getPrimaryNationalIdTypeFromCountryCode(connectingSourceOrg.getCountry()))
                .partnerGrantNewPatientTeamCode(ofNullable(partnerOrgGrant.getNewPatientTeamCode().getOrNull()))
                .maybeResponsibleOrganisationIdentifier(maybeResponsibleOrganisationIdentifier)
                .build();
    }
