package com.pkb.symptom.entity;

import static java.time.temporal.ChronoUnit.DAYS;
import static java.util.stream.Collectors.toList;
import static org.apache.commons.collections4.CollectionUtils.isEmpty;

import java.io.Serializable;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.time.ZoneId;
import java.util.Calendar;
import java.util.List;

import org.jetbrains.annotations.Nullable;

/**
 * History of symptom reports - the symptom and list of data.
 */
public class SymptomReportHistoryDTO implements Serializable {

    private static final long serialVersionUID = 1395738354626790545L;

    private Symptom symptom;

    private List<SymptomReportDTO> symptomReportList;

    public @Nullable SymptomSeverity getSeverityForDateRange(Calendar startDate, int daysInRange, ZoneId localZone) {
        if (isEmpty(symptomReportList) || startDate == null) {
            return null;
        }
        LocalDate rangeStartDate = LocalDateTime.ofInstant(startDate.toInstant(), startDate.getTimeZone().toZoneId()).toLocalDate();
        // note this is a NON-inclusive end date... to be in this range, a checkDate must NOT be on the pastRangeDate
        LocalDate rangeLastDate = rangeStartDate.plus(daysInRange, DAYS);

        List<SymptomSeverity> onDateList = symptomReportList.stream()
                .filter(report -> isReportInRange(report, rangeStartDate, rangeLastDate, localZone))
                .map(SymptomReportDTO::getSeverity)
                .collect(toList());

        if (onDateList.isEmpty()) {
            return null;
        }

        // else need to average them...
        int totalSeverity = 0;
        int counted = 0;
        for (SymptomSeverity ss : onDateList) {
            if (ss.getNumber0to3() == null) {
                continue;
            }
            totalSeverity += ss.getNumber0to3();
            counted++;
        }
        int averageSeverity = Math.round((float) totalSeverity / (float) counted);
        return SymptomSeverity.getBy0to3(averageSeverity);
    }

    private boolean isReportInRange(SymptomReportDTO report, LocalDate startDate, LocalDate lastDate, ZoneId localZone) {
        return report.getReportDateUTC()
                .map(reportDate -> {
                    LocalDate symptomDate = reportDate.withZoneSameInstant(localZone).toLocalDate();

                    return !symptomDate.isBefore(startDate) && symptomDate.isBefore(lastDate);
                })
                .orElse(false);
    }

    public Symptom getSymptom() {
        return symptom;
    }

    public void setSymptom(Symptom symptom) {
        this.symptom = symptom;
    }

    public List<SymptomReportDTO> getSymptomReportList() {
        return symptomReportList;
    }

    public void setSymptomReportList(List<SymptomReportDTO> symptomReportList) {
        this.symptomReportList = symptomReportList;
    }

}
