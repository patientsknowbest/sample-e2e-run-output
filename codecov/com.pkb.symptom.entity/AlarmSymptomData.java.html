package com.pkb.symptom.entity;

import com.pkb.annotation.EHRField;

import java.io.Serializable;
import java.time.Instant;

/**
 * SymptomReportDTO is a separate data entity , saving it as a list inside SymptomAlarmDTO would require handling nested DTO annotations
 * which could
 * add a whole lot of complexity to EHRBean logic so we are introducing this private class to carry the same information as SymptomReportDTO
 * which is
 * necessary for SymptomAlarmDTO
 *
 * @author pravina
 *
 */
public class AlarmSymptomData implements Serializable {

    /**
     *
     */
    private static final long serialVersionUID = 6712239620624552010L;

    @EHRField
    Long symptomReportId;

    @EHRField
    private Long symptomId;

    @EHRField
    private Instant reportDate;

    @EHRField
    private SymptomSeverity severity;

    @EHRField
    private Long dataEntererUid;

    public AlarmSymptomData() {

    }

    public Long getSymptomReportId() {
        return symptomReportId;
    }

    public void setSymptomReportId(Long symptomReportId) {
        this.symptomReportId = symptomReportId;
    }

    public Long getSymptomId() {
        return symptomId;
    }

    public void setSymptomId(Long symptomId) {
        this.symptomId = symptomId;
    }

    public Instant getReportDate() {
        return reportDate;
    }

    public void setReportDate(Instant reportDate) {
        this.reportDate = reportDate;
    }

    public SymptomSeverity getSeverity() {
        return severity;
    }

    public void setSeverity(SymptomSeverity severity) {
        this.severity = severity;
    }

    public Long getDataEntererUid() {
        return dataEntererUid;
    }

    public void setDataEntererUid(Long dataEntererUid) {
        this.dataEntererUid = dataEntererUid;
    }

}
