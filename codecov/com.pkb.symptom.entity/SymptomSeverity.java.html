package com.pkb.symptom.entity;

import org.jetbrains.annotations.Nullable;

import java.io.Serializable;

import static com.pkb.PKBConstants.CODE_SYSTEM_SNOMED_CT_ID;
import static com.pkb.PKBConstants.CODE_SYSTEM_SNOMED_CT_NAME;

/**
 * This enum describes the symptoms that PKB supports; each enum element includes
 * name, and SNOMED-CT code details.
 *
 * @author robwhelan
 */
public enum SymptomSeverity implements Serializable {

    UNKNOWN(null, "symptom.txt.unknown", "", "", "", ""),
    NONE(0, "symptom.txt.none", "260413007", CODE_SYSTEM_SNOMED_CT_ID, CODE_SYSTEM_SNOMED_CT_NAME, "2011_07_31"),
    MILD(1, "symptom.txt.mild", "255604002", CODE_SYSTEM_SNOMED_CT_ID, CODE_SYSTEM_SNOMED_CT_NAME, "2011_07_31"),
    MODERATE(2, "symptom.txt.moderate", "6736007", CODE_SYSTEM_SNOMED_CT_ID, CODE_SYSTEM_SNOMED_CT_NAME, "2011_07_31"),
    SEVERE(3, "symptom.txt.severe", "24484000", CODE_SYSTEM_SNOMED_CT_ID, CODE_SYSTEM_SNOMED_CT_NAME, "2011_07_31");

    private Integer number0to3;
    private String displayName;
    private String code;
    private String codeSystem;
    private String codeSystemName;
    private String codeSystemVersion;

    SymptomSeverity(Integer number0to3, String displayName, String code, String codeSystem, String codeSystemName,
            String codeSystemVersion) {
        this.number0to3 = number0to3;
        this.displayName = displayName;
        this.code = code;
        this.codeSystem = codeSystem;
        this.codeSystemName = codeSystemName;
        this.codeSystemVersion = codeSystemVersion;
    }

    public String getCode() {
        return code;
    }

    public String getCodeSystem() {
        return codeSystem;
    }

    public String getCodeSystemName() {
        return codeSystemName;
    }

    public String getCodeSystemVersion() {
        return codeSystemVersion;
    }

    public Integer getNumber0to3() {
        return number0to3;
    }

    public String getName() {
        return displayName;
    }

    @Nullable
    public static SymptomSeverity getBy0to3(Integer number0to3) {
        if (number0to3 == null) {
            return UNKNOWN;
        }
        for (SymptomSeverity mt : values()) {
            if (number0to3.equals(mt.getNumber0to3())) {
                return mt;
            }
        }
        return null; // no match
    }

}
