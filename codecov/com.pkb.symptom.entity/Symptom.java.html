    // alphabetical sort
    private static final ImmutableList<Long> SORTED_SYMPTOM_IDS;

    private static final ImmutableMap<String, Symptom> API_REF_TO_SYMPTOM;

    private static final ImmutableMap<Long, Symptom> SYMPTOM_BY_ID;

    private static final ImmutableMap<String, Symptom> SYMPTOM_BY_CODE;

    private static final ImmutableMap<String, Symptom> SYMPTOM_BY_NAME;

    static {
        List<Symptom> alphaSymptoms = Arrays.asList(Symptom.values());
        alphaSymptoms.sort(Comparator.comparing(Symptom::getName));
        var apiRefMapBuilder = ImmutableMap.<String, Symptom> builder();
        var sortySymptomIdBuilder = ImmutableList.<Long> builder();
        var byIdBuilder = ImmutableMap.<Long, Symptom> builder();
        var byCodeBuilder = ImmutableMap.<String, Symptom> builder();
        var byNameBuilder = ImmutableMap.<String, Symptom> builder();
        for (Symptom s : alphaSymptoms) {
            sortySymptomIdBuilder.add(s.getId());
            apiRefMapBuilder.put(s.getApiRef(), s);
            byIdBuilder.put(s.getId(), s);
            byCodeBuilder.put(s.getCode(), s);
            byNameBuilder.put(s.getName(), s);
        }
        API_REF_TO_SYMPTOM = apiRefMapBuilder.build();
        SORTED_SYMPTOM_IDS = sortySymptomIdBuilder.build();
        SYMPTOM_BY_ID = byIdBuilder.build();
        SYMPTOM_BY_CODE = byCodeBuilder.build();
        SYMPTOM_BY_NAME = byNameBuilder.build();
    }

    Symptom(long id, String displayName, String apiRef, Map<SymptomSeverity, String> severityTextMap, String code,
            String codeSystem, String codeSystemName) {
        this.id = id;
        this.displayName = displayName;
        this.apiRef = apiRef;
        this.severityTextMap = severityTextMap == null ? null : Collections.unmodifiableMap(severityTextMap);
        this.code = code;
        this.codeSystem = codeSystem;
        this.codeSystemName = codeSystemName;
    }

    /**
     * Some symptoms have custom text for severity. This method returns this custom text if it is not null or
     * the standard name of the severity otherwise.
     *
     * @param severity0to3 the severity index; 0: NONE, 1: MILD , 2: MODERATE, 3: SEVERE
     * @return either the custom severity text for the symptom if it is not null or the default severity name otherwise
     */
    @Nullable
    public String getSeverityText(int severity0to3) {
        SymptomSeverity severity = SymptomSeverity.getBy0to3(severity0to3);
        return hasCustomSeverityText() ? getCustomSeverityText(severity) : getDefaultSeverityText(severity);
    }

    public String getDefaultSeverityText(SymptomSeverity severity) {
        return Objects.requireNonNull(severity, "Cannot get name for null SymptomSeverity").getName();
    }

    /**
     * Some symptoms have custom text for severity. This method returns this custom text if it has one or null otherwise.
     *
     * @param severity the severity to get the custom text for
     * @return the custom severity text for the symptom if found or null
     */
    @Nullable
    public String getCustomSeverityText(SymptomSeverity severity) {
        return hasCustomSeverityText() ? severityTextMap.get(severity) : null;
    }

    /** utility to simplify constructor */
    private static Map<SymptomSeverity, String> buildSeverityMap(String s0, String s1, String s2, String s3) {
        Map<SymptomSeverity, String> map = new EnumMap<>(SymptomSeverity.class);
        map.put(SymptomSeverity.NONE, s0);
        map.put(SymptomSeverity.MILD, s1);
        map.put(SymptomSeverity.MODERATE, s2);
        map.put(SymptomSeverity.SEVERE, s3);
        return map;
    }

    private boolean hasCustomSeverityText() {
        return isNotEmpty(severityTextMap);
    }

	public String getRadioGroupValue(String listName, Long teamId, int index) {
        if (teamId != null) {
            return listName + "[" + teamId + "," + index + ']';
        }
        return listName + "[" + index + "]";
    }
