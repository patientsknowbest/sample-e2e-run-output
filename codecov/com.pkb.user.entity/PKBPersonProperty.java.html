package com.pkb.user.entity;

import com.pkb.entities.enums.PropertyName;

@Entity
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table(uniqueConstraints = { @UniqueConstraint(columnNames = { "person_id", "propertyName" }) })
public class PKBPersonProperty implements Serializable, Comparable {

    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "person_id", referencedColumnName = "id")
    private PKBPerson person;

    @Column(length = 50)
    @Convert(converter = PropertyNameConverter.class)
    private PropertyName propertyName;

    @Column(length = 50)
    private String propertyValue;

    public PKBPersonProperty() {

    }

    public PKBPersonProperty(PropertyName propertyName) {
        this.propertyName = requireNonNull(propertyName, "propertyName cannot be null");
    }

    public PropertyName getPropertyName() {
        return propertyName;
    }

    public void setPropertyName(PropertyName propertyName) {
        this.propertyName = propertyName;
    }

    public String getPropertyValue() {
        return propertyValue;
    }

    public void setPropertyValue(String propertyValue) {
        this.propertyValue = propertyValue;
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    @Override
    public int compareTo(@NotNull Object o) {
        return propertyName.compareTo(((PKBPersonProperty) o).getPropertyName());
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (o == null || getClass() != o.getClass()) {
            return false;
        }
        PKBPersonProperty that = (PKBPersonProperty) o;
        return Objects.equals(getId(), that.getId());
    }

    @Override
    public int hashCode() {
        return 31; // see https://vladmihalcea.com/how-to-implement-equals-and-hashcode-using-the-jpa-entity-identifier/
    }

    public PKBPerson getPerson() {
        return person;
    }

    public void setPerson(PKBPerson person) {
        this.person = person;
    }
}
