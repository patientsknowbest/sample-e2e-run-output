package com.pkb.user.entity;

import java.io.Serializable;
import java.util.UUID;

/**
 * Represents an user account. This is not a database entity, Just a DTO to
 * carry Account information
 *
 * @author pravinam
 */
public class PKBAccount implements Serializable {
    private static final long serialVersionUID = 1L;

    private Long id;

    private UUID publicId;

    /**
     * Construct an empty Account
     */
    public PKBAccount() {
    }

    /**
     * The unique internal of the account. This ID has no meaning other than
     * uniqueness.
     */
    public Long getId() {
        return id;
    }

    public void setId(Long val) {
        this.id = val;
    }

    @Override
    public boolean equals(Object obj) {
        if (!(obj instanceof PKBAccount)) {
            return false;
        }
        if (this.getId().equals(((PKBAccount) obj).getId())) {
            return true;
        }
        return false;
    }

    @Override
    public String toString() {
        return "Account: " + getId();
    }

    @Override
    public int hashCode() {
        if (getId() == 0) {
            throw new IllegalStateException(
                    "id not yet established in Account object");
        }
        return getId().hashCode();
    }

    public UUID getPublicId() {
        return publicId;
    }

    public void setPublicId(UUID publicId) {
        this.publicId = publicId;
    }
}
