package com.pkb.user.entity;

import com.pkb.common.datetime.DateTimeService;
import org.jetbrains.annotations.Nullable;

import java.time.Instant;
import java.time.LocalDate;
import java.time.Period;
import java.util.Optional;

import static com.pkb.user.entity.InvalidDeathTimestampException.Reason.FUTURE_DATE;

public class PKBPersonHelperImpl implements PKBPersonHelper {

    private final DateTimeService dateTimeService;

    public PKBPersonHelperImpl(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    @Override
    public Optional<Boolean> isUnderAge(int age, PKBPerson person) {
        return isAtLeastYearsOld(age, person).map(b -> !b);
    }

    @Override
    public Optional<Boolean> isAtLeastYearsOld(int requiredAge, PKBPerson person) {
        return person.findDateOfBirth()
                .map(this::calculateAge)
                .map(age -> requiredAge <= age);
    }

    @Override
    public boolean isAtLeastYearsOld(int requiredAge, LocalDate dob) {
        return requiredAge <= calculateAge(dob);
    }

    @Override
    public Optional<Integer> calculateAge(PKBPerson person) {
        return person.findDateOfBirth().map(this::calculateAge);
    }

    @Override
    public int calculateAge(LocalDate dateOfBirth) {
        Period p = Period.between(dateOfBirth, dateTimeService.nowZonedDateTime().toLocalDate());
        int result = p.getYears();
        if (result < 1) {
            if (p.getDays() < 0) {
                result = -1;
            }
        }
        return result;
    }

    @Nullable
    @Override
    public Instant validateDeathTimestamp(Instant timestamp) {
        if (timestamp != null && dateTimeService.now().isBefore(timestamp)) {
            throw new InvalidDeathTimestampException(FUTURE_DATE, timestamp);
        }
        return timestamp;
    }
}
