package com.pkb.user.entity;

import com.pkb.institute.entity.Org;
import org.hibernate.annotations.Cache;
import org.hibernate.annotations.CacheConcurrencyStrategy;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import java.io.Serializable;

/**
 * Each Organisation can be configured to have multiple local ID types. Each type is represented as an instance of OrgLevelIdType.
 */
@Entity
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table
public class OrgLevelIdType implements Serializable, Comparable<OrgLevelIdType> {

    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private long id;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "org_id", referencedColumnName = "id")
    private Org org;

    @Column(length = 100)
    private String name;

    @Column(length = 50)
    private String hl7AssigningAuthority;

    @Column(length = 50)
    private String hl7TypeCode;

    @Override
    public boolean equals(Object o) {
        if (!(o instanceof OrgLevelIdType)) {
            return false;
        }

        OrgLevelIdType oidt = (OrgLevelIdType) o;
        return (id == oidt.getId());
    }

    @Override
    public int hashCode() {
        return Long.valueOf(id).hashCode();
    }

    @Override
    public int compareTo(OrgLevelIdType olidt) {
        return Long.valueOf(id).compareTo(olidt.getId());
    }

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public Org getOrg() {
        return org;
    }

    public void setOrg(Org org) {
        this.org = org;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getHl7AssigningAuthority() {
        return hl7AssigningAuthority;
    }

    public void setHl7AssigningAuthority(String hl7AssigningAuthority) {
        this.hl7AssigningAuthority = hl7AssigningAuthority;
    }

    public String getHl7TypeCode() {
        return hl7TypeCode;
    }

    public void setHl7TypeCode(String hl7TypeCode) {
        this.hl7TypeCode = hl7TypeCode;
    }
}
