package com.pkb.user.entity;

import com.pkb.annotation.EHRField;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;
import com.pkb.app.interfaces.IBaseDTO;

import java.io.Serializable;

/**
 * DTO for secure properties stored in a users profile.
 * These can include country-specific profile information (eg Race, Religion...)
 *
 * @author russell
 */
public class PersonSecurePropertyDTO implements IBaseDTO, Serializable {

    private final DTOBaseFields baseFields = new DTOBaseFields();
    private final SourceDetails sourceDetails;

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected PersonSecurePropertyDTO() {
        this.sourceDetails = new SourceDetails();
    }

    public PersonSecurePropertyDTO(SourceDetails sourceDetails) {
        this.sourceDetails = sourceDetails;
    }

    //valid names for the property
    public enum PropertyName {
        religion, race, // for politicians :)
        ethnicGroup // for determine medical implications
    }

    /**
     * The Id of the entry
     */
    @EHRField(queryField = EHRField.QueryField.id)
    private Long id;

    /**
     * The country code for the entry if appropriate. Otherwise null.
     */
    @EHRField
    private String countryOrNull;

    /**
     * The name of the property
     */
    @EHRField
    private PropertyName propertyName;

    /**
     * The value of the property
     */
    @EHRField
    private String propertyValue;

    @Override
    public Long getId() {
        return id;
    }

    @Override
    public void setId(Long id) {
        this.id = id;
    }

    public String getCountryOrNull() {
        return countryOrNull;
    }

    public void setCountryOrNull(String country) {
        this.countryOrNull = country;
    }

    public PropertyName getPropertyName() {
        return propertyName;
    }

    public void setPropertyName(PropertyName name) {
        this.propertyName = name;
    }

    public String getPropertyValue() {
        return propertyValue;
    }

    public void setPropertyValue(String value) {
        propertyValue = value;
    }

    public String getValue() {
        return propertyValue;
    }

    public void setValue(String value) {
        this.propertyValue = value;
    }

    @Override
    public SourceDetails getSource() {
        return sourceDetails;
    }

    @Override
    public DTOBaseFields getBaseFields() {
        return baseFields;
    }

}
