package com.pkb.user.entity;

import io.vavr.control.Option;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

/**
 * Represents the primary email contact info for one user. It includes:
 * <ul>
 * <li>The user's own primary email contact (if one exists)</li>
 * <li>The primary email contact (if one exists) for each ALSO_CONTACT associated with the user<li>
 * </ul>
 *
 * Note: none of the email contacts are guaranteed to be confirmed. 
 *
 * When emailing a user, the main email should go to the primaryContact email address (if non-null) and copies should go to the
 * alsoContactList email addresses, with a tweaked subject line. Something like "REGULAR SUBJECT (for Dependent Patient)".
 */
public class PrimaryEmailContactDTO implements Serializable {

    private static final long serialVersionUID = 1L;

    private PersonContact primaryEmailContact;

    private List<PersonContact> alsoContactPrimaryEmailContactList = new ArrayList<>();

    /**
     * Adds the primary email contact for an ALSO_CONTACT associated with the user to the alsoContactPrimaryList.
     *
     * @param refPersonPrimaryContact
     *            The primary email contact of the ALSO_CONTACT
     */
    public void addAlsoContactPrimary(PersonContact refPersonPrimaryContact) {
        if (refPersonPrimaryContact.getContactIfEmail().isDefined()) {
            alsoContactPrimaryEmailContactList.add(refPersonPrimaryContact);
        } else {
            throw new IllegalArgumentException("Primary email contact of ALSO_CONTACT must be of type EMAIL");
        }
    }

    public Option<PersonContact> findPrimaryEmailContact() {
        return Option.of(primaryEmailContact);
    }

    public PrimaryEmailContactDTO(PersonContact primaryContact) {
        if (Option.of(primaryContact).flatMap(PersonContact::getContactIfPrimaryEmail).isDefined()) {
            this.primaryEmailContact = primaryContact;
        } else {
            throw new IllegalArgumentException("Primary email contact must be of type EMAIL and be primary");
        }
    }

    public PrimaryEmailContactDTO() {
        this.primaryEmailContact = null;
    }

    public List<PersonContact> getAlsoContactPrimaryEmailContactList() {
        return alsoContactPrimaryEmailContactList;
    }

    public void setAlsoContactPrimaryEmailContactList(List<PersonContact> alsoContactPrimaryEmailContactList) {
        // one flow with validation
        alsoContactPrimaryEmailContactList.forEach(pc -> this.addAlsoContactPrimary(pc));
    }
}
