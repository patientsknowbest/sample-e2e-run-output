package com.pkb.user.entity;

import com.pkb.HasVersionDetails;
import com.pkb.VersionDetails;
import com.pkb.entities.enums.HasNationalId;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.entities.enums.ValidNationalId;
import com.pkb.util.VersionDetailsPersistenceListener;
import org.apache.commons.lang3.StringUtils;
import org.hibernate.annotations.Cache;
import org.hibernate.annotations.CacheConcurrencyStrategy;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.EntityListeners;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import javax.persistence.Temporal;
import javax.persistence.TemporalType;
import javax.persistence.Transient;
import javax.persistence.UniqueConstraint;
import java.io.Serializable;
import java.util.Date;

import static org.apache.commons.lang3.StringUtils.trimToNull;

/**
 * A national ID record stored for a patient (e.g. NHS number)
 *
 * @author robwhelan
 */
@Entity
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table(uniqueConstraints = {@UniqueConstraint(columnNames = {"type", "person_id"})})
@EntityListeners({VersionDetailsPersistenceListener.class})
public class NationalId implements Serializable, HasVersionDetails<VersionDetails>, HasNationalId {

    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "person_id")
    private PKBPerson person;

    @Column(length = 60)
    @Enumerated(EnumType.STRING)
    private NationalIdType type;

    @Column(length = 512)
    private String value;

    @Column
    @Temporal(TemporalType.TIMESTAMP)
    private Date editDate;

    @Embedded
    private VersionDetails versionDetails;

    // Bit of a hack to hold it here, but we need to pass it around somewhere to support checking NHS number status
    @Transient
    private String typeCodeJson = "";

    public NationalId(NationalIdType nationalIdType, String value, VersionDetails versionDetails) {
        this.type = nationalIdType;
        this.value = value;
        this.versionDetails = versionDetails;
    }

    public NationalId(ValidNationalId validNationalId, VersionDetails versionDetails) {
        setNationalId(validNationalId);
        this.versionDetails = versionDetails;
    }

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected NationalId() {
    }

    public String getValueFormatted() {
        if ((type == null) || StringUtils.isBlank(value)) {
            return value;
        }

        return type.getCanonicalFormat(value);
    }

    /**
     * custom impl: compare based only on type and value, NOT person or ID
     */
    @Override
    public boolean equals(Object o) {
        if (!(o instanceof NationalId)) {
            return false;
        }

        NationalId nid = (NationalId) o;
        return equalByValueAndType(nid.getValue(), nid.getType());
    }

    @Override
    public int hashCode() {
        String data = "type[" + type + "]" +
                "value[" + trimToNull(value) + "]";
        return data.hashCode();
    }

    public boolean equalByValueAndType(String nationalIdValue, NationalIdType nationalIdType) {
        String value1 = StringUtils.trimToEmpty(this.value);
        String value2 = StringUtils.trimToEmpty(nationalIdValue);

        return (this.type == nationalIdType && value1.equals(value2));
    }

    public boolean equalByValueAndType(ValidNationalId validNationalId) {
        String value1 = StringUtils.trimToEmpty(this.value);
        String value2 = StringUtils.trimToEmpty(validNationalId.value());

        return (this.type == validNationalId.type() && value1.equals(value2));
    }

    @Override
    public String toString() {
        return "NationalId (id " + id + ", type " + type + ", value " + value + ")";
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public PKBPerson getPerson() {
        return person;
    }

    public void setPerson(PKBPerson person) {
        this.person = person;
    }

    public NationalIdType getType() {
        return type;
    }

    public void setType(NationalIdType type) {
        this.type = type;
    }

    public String getValue() {
        return value;
    }

    public void setValue(String value) {
        this.value = value;
    }

    public Date getEditDate() {
        return editDate;
    }

    public void setEditDate(Date editDate) {
        this.editDate = editDate;
    }

    public String getTypeCodeJson() {
        return typeCodeJson;
    }

    public void setTypeCodeJson(String typeCodeJson) {
        this.typeCodeJson = typeCodeJson;
    }

    @Override
    public VersionDetails getVersionDetails() {
        return versionDetails;
    }

    @Override
    public void setVersionDetails(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }

    public void setNationalId(ValidNationalId validNationalId) {
        setValue(validNationalId.value());
        setType(validNationalId.type());
    }

    @Override
    public NationalIdType nationalIdType() {
        return getType();
    }

    @Override
    public String nationalIdValue() {
        return getValue();
    }
}
