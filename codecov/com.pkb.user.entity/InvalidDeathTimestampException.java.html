package com.pkb.user.entity;

import static com.google.common.base.Preconditions.checkNotNull;

import java.time.Instant;


public class InvalidDeathTimestampException extends RuntimeException {

    public enum Reason {
        BEFORE_DOB , FUTURE_DATE
    }

    private Reason reason;

    private Instant invalidDeathTimestamp;

    public InvalidDeathTimestampException(Reason reason, Instant invalidDeathTimestamp) {
        this.reason = checkNotNull(reason, "reason cannot be null");
        this.invalidDeathTimestamp = checkNotNull(invalidDeathTimestamp, "invalidDeathTimestamp cannot be null");
    }

    public Reason getReason() {
        return reason;
    }

    public Instant getInvalidDeathTimestamp() {
        return invalidDeathTimestamp;
    }
}
