/**
 * An org-level ID for a patient. This is typically a hospital number.
 *
 * @author robwhelan
 */
@Entity
@EntityListeners({VersionDetailsPersistenceListener.class})
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table
public class OrgLevelId implements Serializable, Comparable<OrgLevelId>, HasVersionDetails<VersionDetails> {

    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column(name="person_id")
    private Long personId;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "org_id", referencedColumnName = "id")
    private Org org;

    @ManyToOne(fetch = FetchType.EAGER)
    @JoinColumn(name = "olidtype_id", referencedColumnName = "id")
    private OrgLevelIdType type;

    @Column(length = 512)
    private String value;

    @Column
    @Temporal(TemporalType.TIMESTAMP)
    private Date editDate;

    @Embedded
    private VersionDetails versionDetails;

    /**
     * Combines the CSV values in groupedIds with the types for the org, into a
     * set of OrgLevelId instances with both value and type set.
     *
     * @return
     */
    public static SortedSet<OrgLevelId> combineOrgLevelIds(List<OrgLevelIdType> types, List<String> groupedIds) {
        SortedSet<OrgLevelId> olids = new TreeSet<>();

        if ((types == null) || (groupedIds == null)) {
            return olids;
        }

        Iterator<String> groupedIds_it = groupedIds.iterator();
        for (OrgLevelIdType olidt : types) {
            String goid = groupedIds_it.next();
            String[] ids = goid.split(",");
            for (String id : ids) {
                if (StringUtils.isNotBlank(id)) {
                    OrgLevelId olid = new OrgLevelId();
                    olid.setValue(StringUtils.trim(id));
                    olid.setType(olidt);
                    olid.setOrg(olidt.getOrg());
                    olids.add(olid);
                }
            }
        }

        return olids;
    }

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected OrgLevelId() {
    }

    public OrgLevelId(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }

    @Override
    public boolean equals(Object o) {
        if (!(o instanceof OrgLevelId)) {
            return false;
        }

        OrgLevelId olid = (OrgLevelId) o;
        long t1 = type.getId();
        long t2 = olid.getType().getId();
        String v1 = StringUtils.trimToEmpty(value);
        String v2 = StringUtils.trimToEmpty(olid.getValue());

        return ((t1 == t2) && (v1.equals(v2)));
    }

    @Override
    public int hashCode() {
        String data = "type[" + type.getId() + "]" +
                "value[" + StringUtils.trimToNull(value) + "]";
        return data.hashCode();
    }

    @Override
    public int compareTo(OrgLevelId olid) {
        if (Objects.equals(type, olid.getType())) {
            return value.compareTo(olid.getValue());
        }
        return type.compareTo(olid.getType());
    }

    @Override
    public String toString() {
        return "OrgLevellId (id " + id + ", org " + org + ", value " + value + ", type " + type + ")";
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Org getOrg() {
        return org;
    }

    public void setOrg(Org org) {
        this.org = org;
    }

    public OrgLevelIdType getType() {
        return type;
    }

    public void setType(OrgLevelIdType type) {
        this.type = type;
    }

    public String getValue() {
        return value;
    }

    public void setValue(String value) {
        this.value = value;
    }

    public Date getEditDate() {
        return editDate;
    }

    public void setEditDate(Date editDate) {
        this.editDate = editDate;
    }

    @Override
    public VersionDetails getVersionDetails() {
        return versionDetails;
    }

    @Override
    public void setVersionDetails(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }
}
