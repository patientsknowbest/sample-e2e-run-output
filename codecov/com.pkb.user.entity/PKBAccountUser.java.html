package com.pkb.user.entity;

import com.pkb.entities.enums.AccountUserStatus;

import java.io.Serializable;

/**
 * A class associating the User and Account. A account can have many users. This
 * identifies a user in a context of an account
 *
 * @author pravinam
 */
public class PKBAccountUser implements Serializable {

    private static final long serialVersionUID = 1L;

    private PKBAccount account;

    private PKBPerson user;

    private Long id;

    private AccountUserStatus status;

    /**
     * Construct an empty AccountUser.
     */
    public PKBAccountUser() {
    }

    public PKBAccount getAccount() {
        return account;
    }

    public void setAccount(PKBAccount val) {
        this.account = val;
    }

    /**
     * The User to be granted access to the specified Account.
     */
    public PKBPerson getUser() {
        return user;
    }

    public void setUser(PKBPerson val) {
        this.user = val;
    }

    /**
     * The unique ID for this association.
     */
    public Long getId() {
        return id;
    }

    public void setId(Long val) {
        this.id = val;
    }

    @Override
    public boolean equals(Object obj) {
        if (!(obj instanceof PKBAccountUser)) {
            return false;
        }
        if (this.getId().equals(((PKBAccountUser) obj).getId())) {
            return true;
        }
        return false;
    }

    @Override
    public String toString() {
        return "AccountUser: " + getId();
    }

    @Override
    public int hashCode() {
        if (getId() == 0) {
            throw new IllegalStateException("id not yet established in Account object");
        }
        return getId().hashCode();
    }

    public AccountUserStatus getStatus() {
        return status;
    }

    public void setStatus(AccountUserStatus status) {
        this.status = status;
    }


}
