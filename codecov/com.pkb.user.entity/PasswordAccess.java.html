package com.pkb.user.entity;

import java.io.Serializable;
import java.util.UUID;

/**
 * DTO to carry password access details for user with no contact (like email etc)
 *
 * @author pravinam
 *
 */
public class PasswordAccess implements Serializable {

    /**
     *
     */
    private static final long serialVersionUID = 1L;

    // Document ID
    private UUID metadataId;

    // The user with no contact
    private String userId;

    private String password;

    public UUID getMetadataId() {
        return metadataId;
    }

    public void setMetadataId(UUID metadataId) {
        this.metadataId = metadataId;
    }

    public String getUserId() {
        return userId;
    }

    public void setUserId(String userId) {
        this.userId = userId;
    }

    public String getPassword() {
        return password;
    }

    public void setPassword(String password) {
        this.password = password;
    }
}
