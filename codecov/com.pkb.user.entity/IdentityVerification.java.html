package com.pkb.user.entity;

import com.pkb.entities.enums.PhotoIdType;
import com.pkb.entities.enums.VerificationStatus;
import com.pkb.institute.entity.Team;
import org.hibernate.annotations.Cache;
import org.hibernate.annotations.CacheConcurrencyStrategy;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.Table;
import javax.persistence.Temporal;
import javax.persistence.TemporalType;
import javax.persistence.UniqueConstraint;
import java.io.Serializable;
import java.time.Instant;
import java.util.Date;

/**
 * This class represents the identity verification record for an user
 *
 * @author pravinam
 *
 */
@Entity
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table(uniqueConstraints = { @UniqueConstraint(columnNames = { "person_id" }) })
public class IdentityVerification implements Serializable {

    /**
     *
     */
    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column(name = "person_id")
    private Long personId;

    @Column(name = "verifier")
    private Long verifierId;

    @Column(length = 40)
    @Enumerated(EnumType.STRING)
    private PhotoIdType photoIdType;

    @Column(length = 512)
    private String photoIdNumber;

    @Column
    private Boolean signedConsentForm;

    @Column(length = 20)
    @Enumerated(EnumType.STRING)
    private VerificationStatus status;

    @Column(nullable = false)
    @Temporal(TemporalType.TIMESTAMP)
    private Date verificationDate;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "verifyingteam_id")
    private Team verifyingTeam;

   public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public PhotoIdType getPhotoIdType() {
        return photoIdType;
    }

    public void setPhotoIdType(PhotoIdType photoIdType) {
        this.photoIdType = photoIdType;
    }

    public String getPhotoIdNumber() {
        return photoIdNumber;
    }

    public void setPhotoIdNumber(String photoIdNumber) {
        this.photoIdNumber = photoIdNumber;
    }

    public VerificationStatus getStatus() {
        return status;
    }

    public void setStatus(VerificationStatus status) {
        this.status = status;
    }

    public Date getVerificationDate() {
        return verificationDate;
    }

    public void setVerificationDate(Date verificationDate) {
        this.verificationDate = verificationDate;
    }

    public Boolean getSignedConsentForm() {
        return signedConsentForm;
    }

    public void setSignedConsentForm(Boolean signedConsentForm) {
        this.signedConsentForm = signedConsentForm;
    }

    public Team getVerifyingTeam() {
        return verifyingTeam;
    }

    public void setVerifyingTeam(Team verifyingTeam) {
        this.verifyingTeam = verifyingTeam;
    }

    public static IdentityVerification of(PKBPerson verifiedUser, PKBPerson verifyingUser, Team team, Instant now) {
        IdentityVerification verification = new IdentityVerification();
        verification.setPersonId(verifiedUser.getId());
        verification.setVerifierId(verifyingUser.getId());
        verification.setVerificationDate(Date.from(now));
        verification.setStatus(VerificationStatus.VERIFIED);
        verification.setSignedConsentForm(false);
        verification.setPhotoIdType(PhotoIdType.VOUCHED_FOR_BY_COORDINATOR);
        if (team == null) {
            verification.setPhotoIdNumber("Vouched for by " + verifyingUser.getTitle() + " "
                    + verifyingUser.getFirstName() + " " + verifyingUser.getLastName() + "of PKB");
        } else {
            verification.setPhotoIdNumber("Vouched for by " + verifyingUser.getTitle() + " "
                    + verifyingUser.getFirstName() + " " + verifyingUser.getLastName()
                    + " of " + team.getName());
        }
        verification.setVerifyingTeam(team);
        return verification;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }

    public Long getVerifierId() {
        return verifierId;
    }

    public void setVerifierId(Long verifierId) {
        this.verifierId = verifierId;
    }

}
