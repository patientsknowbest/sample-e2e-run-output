/**
 * A team-level ID for a patient. This is typically a department-specific
 * identifier, such as a GUM number.
 *
 * @author robwhelan
 */
@Entity
@EntityListeners({VersionDetailsPersistenceListener.class})
@Cache(usage = CacheConcurrencyStrategy.NONSTRICT_READ_WRITE)
@Table
public class TeamLevelId implements Serializable, Comparable<TeamLevelId>, HasVersionDetails<VersionDetails> {

    private static final long serialVersionUID = 1L;

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column(name="person_id")
    private Long personId;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "team_id", referencedColumnName = "id")
    private Team team;

    @ManyToOne(fetch = FetchType.EAGER)
    @JoinColumn(name = "tlidtype_id", referencedColumnName = "id")
    private TeamLevelIdType type;

    @Column(length = 512)
    private String value;

    @Column
    @Temporal(TemporalType.TIMESTAMP)
    private Date editDate;

    @Embedded
    private VersionDetails versionDetails;

    /**
     * Combines the CSV values in groupedIds with the types for the team, into a
     * set of TeamLevelId instances with both value and type set.
     *
     * @return
     */
    public static SortedSet<TeamLevelId> combineTeamLevelIds(List<TeamLevelIdType> types, List<String> groupedIds) {
        SortedSet<TeamLevelId> tlids = new TreeSet<>();

        if ((types == null) || (groupedIds == null)) {
            return tlids;
        }

        Iterator<String> groupedIds_it = groupedIds.iterator();
        for (TeamLevelIdType tlidt : types) {
            String gtid = groupedIds_it.next();
            String[] ids = gtid.split(",");
            for (String id : ids) {
                if (StringUtils.isNotBlank(id)) {
                    TeamLevelId tlid = new TeamLevelId();
                    tlid.setValue(StringUtils.trim(id));
                    tlid.setType(tlidt);
                    tlid.setTeam(tlidt.getTeam());
                    tlids.add(tlid);
                }
            }
        }

        return tlids;
    }

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected TeamLevelId() {
    }

    public TeamLevelId(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }

    @Override
    public boolean equals(Object o) {
        if (!(o instanceof TeamLevelId)) {
            return false;
        }

        TeamLevelId tid = (TeamLevelId) o;
        long t1 = type.getId();
        long t2 = tid.getType().getId();
        String v1 = StringUtils.trimToEmpty(value);
        String v2 = StringUtils.trimToEmpty(tid.getValue());

        return ((t1 == t2) && (v1.equals(v2)));
    }

    @Override
    public int hashCode() {
        String data = "type[" + type.getId() + "]" +
                "value[" + StringUtils.trimToNull(value) + "]";
        return data.hashCode();
    }

    @Override
    public int compareTo(TeamLevelId tlid) {
        if (type.equals(tlid.getType())) {
            return value.compareTo(tlid.getValue());
        }
        return type.compareTo(tlid.getType());
    }

    @Override
    public String toString() {
        return "TeamLevellId (id " + id + ", team " + team + ", value " + value + ", type " + type + ")";
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Team getTeam() {
        return team;
    }

    public void setTeam(Team team) {
        this.team = team;
    }

    public TeamLevelIdType getType() {
        return type;
    }

    public void setType(TeamLevelIdType type) {
        this.type = type;
    }

    public String getValue() {
        return value;
    }

    public void setValue(String value) {
        this.value = value;
    }

    public Date getEditDate() {
        return editDate;
    }

    public void setEditDate(Date editDate) {
        this.editDate = editDate;
    }

    @Override
    public VersionDetails getVersionDetails() {
        return versionDetails;
    }

    @Override
    public void setVersionDetails(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }
}
