package com.pkb.user.entity;

import com.google.common.collect.Sets;
import com.pkb.HasVersionDetails;
import com.pkb.VersionDetails;
import com.pkb.datamodel.Email;
import com.pkb.datamodel.Phone;
import com.pkb.entities.enums.ContactType;
import com.pkb.entities.enums.UserStatus;
import com.pkb.util.VersionDetailsPersistenceListener;
import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;
import org.hibernate.annotations.DynamicInsert;
import org.hibernate.annotations.Type;
import org.jetbrains.annotations.NotNull;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.EntityListeners;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;

import static com.google.common.collect.Sets.newHashSet;
import static com.pkb.datamodel.ImmutableDisclosingEmail.disclosingEmail;
import static com.pkb.datamodel.ImmutableDisclosingPhone.disclosingPhone;
import static com.pkb.entities.enums.ContactType.ALSO_CONTACT;
import static com.pkb.entities.enums.ContactType.EMAIL;
import static com.pkb.entities.enums.ContactType.PHONE;
import static com.pkb.entities.enums.UserStatus.PATIENT_NOT_REGISTERED;
import static io.vavr.API.None;
import static io.vavr.API.Some;
import static java.util.Collections.emptySet;
import static java.util.Collections.singleton;
import static java.util.stream.Collectors.toSet;

/**
 * This class represents contact address for a PKBperson : mobile no , email etc
 */
@Entity
@DynamicInsert
@Table(uniqueConstraints = { @UniqueConstraint(columnNames = { "person_id", "contactType", "contactValue" }) })
@EntityListeners({ VersionDetailsPersistenceListener.class})
public class PersonContact implements Serializable, HasVersionDetails<VersionDetails> {

    private static final long serialVersionUID = 1L;
    private static final Set<ContactType> ACTUAL_CONTACTS = Sets.immutableEnumSet(EMAIL, ContactType.PHONE);

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column(name = "public_id", nullable = false, updatable = false, unique = true)
    @Type(type = "pg-uuid")
    private UUID publicId;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "person_id", referencedColumnName = "id", nullable = false)
    private PKBPerson person;

    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    private ContactType contactType;

    @Column()
    private String contactValue;

    @Column(nullable = false)
    private boolean confirmed;

    @Column(nullable = false)
    private boolean isPrimary;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "ref_person_id", referencedColumnName = "id")
    private PKBPerson refPerson;

    @Embedded
    private VersionDetails versionDetails = new VersionDetails();

    public PersonContact() {
    }

    public enum Lazy {

        PERSON(emptySet()), // Person will always be loaded, so don't specify a subgraph for it
        PERSON_W_CONTACTS(singleton("contacts")),
        REF_PERSON(emptySet()), // Not used
        PROPERTIES(singleton("properties")),
        NATIONAL_AND_LOCAL_IDS(newHashSet("nationalIds", "orgLevelIds", "teamLevelIds"));

        private final Set<String> subgraph;

        Lazy(Set<String> subgraph) {
            this.subgraph = subgraph;
        }

        private Set<String> getSubgraph() {
            return subgraph;
        }

        public static Set<String> getSubgraphs(Lazy... lazies) {
            return Arrays.stream(lazies)
                    .map(Lazy::getSubgraph)
                    .flatMap(Collection::stream)
                    .filter(StringUtils::isNotBlank)
                    .collect(toSet());
        }
    }

    /**
     * Compressed display. Examples:
     * email address, primary but not confirmed: myemail@contact.com(P!C)
     * mobile phone, not primary, confirmed: 029391929392(MOBILE/!PC)
     * also contact record, not primary, not confirmed: (ALSO_CONTACT/!P!C/ref:1232)
     */
    @Override
    public String toString() {
        String s = contactValue + "(";
        if (contactType != EMAIL) {
            s += contactType + "/";
        }
        s += isPrimary() ? "P" : "!P";
        s += isConfirmed() ? "C" : "!C";
        if (refPerson != null) {
            s += "/ref:" + refPerson.getId();
        }
        return s + ")";
    }

    @Override public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (!(o instanceof PersonContact)) {
            return false;
        }
        PersonContact that = (PersonContact) o;
        return confirmed == that.confirmed &&
                isPrimary == that.isPrimary &&
                Objects.equals(id, that.id) &&
                contactType == that.contactType &&
                Objects.equals(contactValue, that.contactValue);
    }

    @Override public int hashCode() {
        return Objects.hash(id, contactType, contactValue, confirmed, isPrimary);
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public UUID getPublicId() {
        return publicId;
    }

    public void setPublicId(UUID publicId) {
        this.publicId = publicId;
    }

    public PKBPerson getPerson() {
        return person;
    }

    public void setPerson(PKBPerson person) {
        this.person = person;
    }

    public boolean isConfirmed() {
        return confirmed;
    }

    public boolean isRegistered() {
        UserStatus personStatus = this.person.getStatus();
        return !PATIENT_NOT_REGISTERED.contains(personStatus);
    }

    public void setConfirmed(boolean confirmed) {
        this.confirmed = confirmed;
    }

    public Option<Email> getContactIfEmail() {
        if(contactType == EMAIL && contactValue != null) {
            return Some(disclosingEmail(contactValue));
        } else {
            return None();
        }
    }

    public Option<Email> getContactIfPrimaryEmail() {
        return getContactIfEmail().filter(ignored -> isPrimary());
    }

    public Option<Phone> getContactIfPhone() {
        if(contactType == PHONE && contactValue != null) {
            return Some(disclosingPhone(contactValue));
        } else {
            return None();
        }
    }

    public void setContactAsEmail(@NotNull Email email) {
        this.contactType = EMAIL;
        this.contactValue = email.address();
        this.refPerson = null;
    }

    public void setContactAsPhone(@NotNull Phone phone) {
        this.contactType = PHONE;
        this.contactValue = phone.number();
        this.refPerson = null;
    }

    public String getEmailOrPhoneString() {
        return getContactIfPhone().map(Phone::number)
                .getOrElse(getContactIfEmail().map(Email::address).getOrElse(""));
    }

    public void setToAlsoContact(@NotNull PKBPerson refPerson) {
        this.contactType = ALSO_CONTACT;
        this.contactValue = null;
        this.refPerson = refPerson;
    }

    public boolean contactValueEquals(PersonContact other) {
        return other != null && this.contactType == other.contactType
                && this.contactValue != null && this.contactValue.equals(other.contactValue);
    }

    public boolean isPrimary() {
        return isPrimary;
    }

    public boolean getIsPrimary() {
        return isPrimary;
    }

    public void setPrimary(boolean isPrimary) {
        this.isPrimary = isPrimary;
    }

    public Option<PKBPerson> findAlsoContactPerson() {
        return Option.of(refPerson);
    }

    @Override public VersionDetails getVersionDetails() {
        return versionDetails;
    }

    @Override public void setVersionDetails(VersionDetails versionDetails) {
        this.versionDetails = versionDetails;
    }

    public boolean isActualContact() {
        return isActualContact(contactType);
    }

    public static boolean isActualContact(ContactType type) {
        return ACTUAL_CONTACTS.contains(type);
    }

    public boolean isEmail() {
        return contactType == EMAIL;
    }
}
