package com.pkb.aggregator.integration.hapi.configuration

import ca.uhn.fhir.context.FhirContext
import ca.uhn.fhir.rest.client.impl.RestfulClientFactory
import com.pkb.aggregator.integration.hapi.BetterOkHttpRestfulClientFactory
import okhttp3.OkHttpClient
import org.springframework.beans.factory.annotation.Qualifier
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration

@Configuration
class HapiFhirConfiguration {

    @Bean(name = ["fhirContextR4"])
    fun fhirR4Context(): FhirContext = FhirContext.forR4()

    @Bean(name = ["fhirContextDstu3"])
    fun fhirDstu3Context(): FhirContext = FhirContext.forDstu3()

    @Bean(name = ["okhttpR4ClientFactory"])
    fun hapiFhirR4ClientFactory(client: OkHttpClient, @Qualifier("fhirContextR4") fhirContext: FhirContext): RestfulClientFactory =
        BetterOkHttpRestfulClientFactory(fhirContext, client)

    @Bean(name = ["okhttpDstu3ClientFactory"])
    fun hapiFhirDstu3ClientFactory(client: OkHttpClient, @Qualifier("fhirContextDstu3") fhirContext: FhirContext): RestfulClientFactory =
        BetterOkHttpRestfulClientFactory(fhirContext, client)

}
