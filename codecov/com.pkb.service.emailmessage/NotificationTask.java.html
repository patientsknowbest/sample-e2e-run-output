package com.pkb.service.emailmessage;

import com.pkb.entities.enums.Message;
import com.pkb.exception.MailException;
import com.pkb.institute.entity.Org;
import com.pkb.notification.entity.Activity;
import com.pkb.service.emailmessage.impl.AddressInfo;
import com.pkb.service.emailmessage.impl.EmailManager;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Locale;
import java.util.Map;

/**
 * This class represents the task of email notification to a user in system
 *
 * @author pravinam
 *
 */
public class NotificationTask implements Runnable {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private EmailManager emailManager;

    private Message message;
    private PKBPerson fromPerson;
    private AddressInfo toAddress;
    private PKBPerson toPerson;
    private Map<String, Object> properties;
    private Locale locale;
    private Activity.Action action;
    private String actorId;
    private Org org;

    public NotificationTask (EmailManager emailManager, Message message, PKBPerson fromPerson,
                             AddressInfo toAddress, PKBPerson toPerson, Map<String, Object> properties, Locale locale) {
        this.emailManager = emailManager;
        this.message = message;
        this.fromPerson = fromPerson;
        this.toAddress = toAddress;
        this.toPerson = toPerson;
        this.properties  = properties;
        this.locale = locale;
    }

    public NotificationTask (EmailManager emailManager, Message message, PKBPerson fromPerson,
                             AddressInfo toAddress, PKBPerson toPerson, Map<String, Object> properties, Locale locale,
                             Activity.Action action) {
        this(emailManager, message, fromPerson, toAddress, toPerson, properties, locale);
        this.action = action;
    }

    public NotificationTask (EmailManager emailManager, Message message, PKBPerson fromPerson,
                             AddressInfo toAddress, PKBPerson toPerson, Map<String, Object> properties, Locale locale,
                             Activity.Action action, String actorId, @NotNull Org org) {
        this(emailManager, message, fromPerson, toAddress, toPerson, properties, locale, action);
        this.org = org;
        this.actorId = actorId;
    }

    @Override
    public void run() {
        try {
            emailManager.sendMessage(message, fromPerson, toAddress, toPerson, properties, locale, action, actorId, org, null);
            LOGGER.info("sendMessage thread : end");
        } catch (MailException e) {
            LOGGER.error("Unable to send notification {} using template-{}", fromPerson, message, e);
        }
    }

}
