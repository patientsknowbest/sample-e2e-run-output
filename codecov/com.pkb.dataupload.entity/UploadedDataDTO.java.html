package com.pkb.dataupload.entity;

import com.pkb.dataupload.entity.UploadedData.Destination;
import com.pkb.dataupload.entity.UploadedData.Format;
import com.pkb.dataupload.entity.UploadedData.Status;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;

import java.io.Serializable;
import java.time.Instant;
import java.util.Date;
import java.util.List;
import java.util.Optional;
import java.util.UUID;

/**
 * See {@link UploadedData} entity -- this is a DTO to wrap the (decrypted) text data.
 * For parsing and getting useful data out of HL7 XML see HL7XMLDoc object in phr-service.
 *
 * @author robwhelan
 */
public class UploadedDataDTO implements Serializable, Cloneable {

    private static final long serialVersionUID = 1L;

    private Long id;

    private Long version;

    private Org sourceOrg;

    private Org connectingOrg;

    private Team sourceTeam;

    private Long accountId;

    private Destination destination;

    private Format format;

    private UploadedData.Status status;

    private Long hl7PartnerId;

    /**
     * Unique ID parsed from the uploaded data. For HL7 messages, this should
     * always be MSH-10, i.e. from getMessageId().
     */
    private String embeddedId;

    /**
     * A place to store an external ID passed to PKB from an external system.
     * This is required to e.g. subsequently update a previously received
     * message. For example, TXA-12 stores an external document id in an HL7
     * message.
     */
    private String externalMessageId;

    private String data;

    private UUID documentMetadataId;

    /** If this is an update, point to the original uploadData ID. */
    private Long updateTo;

    /**
     * Sometimes subsets of an original upload are re-queued internally to pass
     * them around, e.g. if an HL7 A28 also contains allergy information. By
     * pointing to the original upload we can track which data was received
     * from an external source, and which we generated as a result.
     *
     * This feature introduced early 2016 - any partial uploaded data entries
     * before then will not have this field set.
     */
    private Long subsetOfParentId;

    private String destinationSpecificData;

    private List<FieldMismatch> patientInfoMismatches = null;

    /** When the entry was created into the table, i.e. when we received the HL7 message. */
    private Date created = null;

    private String userFirstName;

    private String userLastName;

    private String hl7MessageType;

    private String responsibleOrganisationIdentifier;

    public UploadedDataDTO copyToNew() {
        // (shallow) clone -- the new obj will refer to the same obj references (no primitives here!)
        // but these are all things that are immutable (primitive wrappers/strings) or refs not edited here (org/team)
        try {
            UploadedDataDTO clone = (UploadedDataDTO) clone();
            // be sure ID is null; this is a new copy
            clone.setId(null);
            return clone;
        } catch (CloneNotSupportedException cnse) {
            // won't happen
            throw new RuntimeException("can't clone UploadedDataDTO!?", cnse);
        }
    }

    //
    // getters/setters
    //

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Long getAccountId() {
        return accountId;
    }

    public void setAccountId(Long accountId) {
        this.accountId = accountId;
    }

    public Destination getDestination() {
        return destination;
    }

    public void setDestination(Destination destination) {
        this.destination = destination;
    }

    public Status getStatus() {
        return status;
    }

    public void setStatus(Status status) {
        this.status = status;
    }

    public String getData() {
        return data;
    }

    public void setData(String data) {
        this.data = data;
    }

    public Format getFormat() {
        return format;
    }

    public void setFormat(Format format) {
        this.format = format;
    }

    public String getEmbeddedId() {
        return embeddedId;
    }

    public void setEmbeddedId(String embeddedId) {
        this.embeddedId = embeddedId;
    }

    public String getExternalMessageId() {
        return externalMessageId;
    }

    public void setExternalMessageId(String externalMessageId) {
        this.externalMessageId = externalMessageId;
    }

    public Org getSourceOrg() {
        return sourceOrg;
    }

    public void setSourceOrg(Org sourceOrg) {
        this.sourceOrg = sourceOrg;
    }

    public Team getSourceTeam() {
        return sourceTeam;
    }

    public void setSourceTeam(Team sourceTeam) {
        this.sourceTeam = sourceTeam;
    }

    public UUID getDocumentMetadataId() {
        return documentMetadataId;
    }

    public void setDocumentMetadataId(UUID documentMetadataId) {
        this.documentMetadataId = documentMetadataId;
    }

    public Long getUpdateTo() {
        return updateTo;
    }

    public void setUpdateTo(Long updateTo) {
        this.updateTo = updateTo;
    }

    public Long getSubsetOfParentId() {
        return subsetOfParentId;
    }

    public void setSubsetOfParentId(Long subsetOfParentId) {
        this.subsetOfParentId = subsetOfParentId;
    }

    public String getDestinationSpecificData() {
        return destinationSpecificData;
    }

    public void setDestinationSpecificData(String destinationSpecificData) {
        this.destinationSpecificData = destinationSpecificData;
    }

    public List<FieldMismatch> getPatientInfoMismatches() {
        return patientInfoMismatches;
    }

    public void setPatientInfoMismatches(List<FieldMismatch> patientInfoMismatches) {
        this.patientInfoMismatches = patientInfoMismatches;
    }

    public Date getCreated() {
        return created;
    }

    public Optional<Instant> findCreated() {
        return Optional.ofNullable(created).map(Date::toInstant);
    }

    public void setCreated(Date created) {
        this.created = created;
    }

    public String getUserFirstName() {
        return userFirstName;
    }

    public void setUserFirstName(String userFirstName) {
        this.userFirstName = userFirstName;
    }

    public String getUserLastName() {
        return userLastName;
    }

    public void setUserLastName(String userLastName) {
        this.userLastName = userLastName;
    }

    public Long getVersion() {
        return version;
    }

    public void setVersion(Long version) {
        this.version = version;
    }

    public Org getConnectingOrg() {
        return connectingOrg;
    }

    public void setConnectingOrg(Org connectingOrg) {
        this.connectingOrg = connectingOrg;
    }

    public Long getHl7PartnerId() {
        return hl7PartnerId;
    }

    public void setHl7PartnerId(Long hl7PartnerId) {
        this.hl7PartnerId = hl7PartnerId;
    }

    public String getHl7MessageType() {
        return hl7MessageType;
    }

    public void setHl7MessageType(String hl7MessageType) {
        this.hl7MessageType = hl7MessageType;
    }

    public String getResponsibleOrganisationIdentifier() {
        return responsibleOrganisationIdentifier;
    }

    public void setResponsibleOrganisationIdentifier(String responsibleOrganisationIdentifier) {
        this.responsibleOrganisationIdentifier = responsibleOrganisationIdentifier;
    }
}
