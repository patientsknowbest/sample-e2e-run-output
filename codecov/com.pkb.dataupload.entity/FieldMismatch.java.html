package com.pkb.dataupload.entity;

import java.io.Serializable;

/**
 * Initially used only in UploadedDataDTO, for patient info mismatches in HL7 messages
 *
 * @author robwhelan
 */
public class FieldMismatch implements Serializable {
    private static final long serialVersionUID = 1L;

    private String fieldName;
    private String pkbValue;
    private String receivedValue;

    public FieldMismatch(String fieldName, String pkbValue, String receivedValue) {
        this.fieldName = fieldName;
        this.pkbValue = pkbValue;
        this.receivedValue = receivedValue;
    }

    @Override
    public String toString() {
        return "FieldMismatch for " + fieldName + ": pkb:" + pkbValue + ", received:" + receivedValue + ".";
    }

    public String getFieldName() {
        return fieldName;
    }

    public String getPkbValue() {
        return pkbValue;
    }

    public String getReceivedValue() {
        return receivedValue;
    }
}
