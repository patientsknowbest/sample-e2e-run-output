package com.pkb.entities.emis_integration;

import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.Instant;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(schema = "emis_integration", name = "emis_pharmacy_details")
public class EmisPharmacyDetails {

    @Id
    @GenericGenerator(name = "emis_pharmacy_details_id_seq_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "emis_integration"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "emis_pharmacy_details_id_seq"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(
            strategy = GenerationType.SEQUENCE,
            generator = "emis_pharmacy_details_id_seq_gen")
    private Long id;

    @Column(name = "person_id", nullable = false)
    private Long personId;

    @Column(name = "pharmacy_ods_code", length = 20, nullable = false)
    private String odsCode;

    @Column(name = "pharmacy_name", nullable = false)
    private String name;

    @Column(name = "pharmacy_address", nullable = false)
    private String address;

    @Column(name = "pharmacy_postcode", nullable = false)
    private String postcode;

    @Column(name = "deleted", nullable = false)
    private boolean deleted;

    @Column(name = "created", updatable = false, nullable = false)
    private Instant created;

    @Column(name = "last_modified", nullable = false)
    private Instant lastModified;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }

    public String getOdsCode() {
        return odsCode;
    }

    public void setOdsCode(String odsCode) {
        this.odsCode = odsCode;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getAddress() {
        return address;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    public String getPostcode() {
        return postcode;
    }

    public void setPostcode(String postcode) {
        this.postcode = postcode;
    }

    public boolean isDeleted() {
        return deleted;
    }

    public void setDeleted(boolean deleted) {
        this.deleted = deleted;
    }

    public Instant getCreated() {
        return created;
    }

    public void setCreated(Instant created) {
        this.created = created;
    }

    public Instant getLastModified() {
        return lastModified;
    }

    public void setLastModified(Instant lastModified) {
        this.lastModified = lastModified;
    }

}
