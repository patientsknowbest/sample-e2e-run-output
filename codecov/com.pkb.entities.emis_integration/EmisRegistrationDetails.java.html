package com.pkb.entities.emis_integration;

import com.pkb.entities.embeddables.EncryptedWrapper;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.Instant;

@Entity
@Table(schema = "emis_integration", name = "emis_registration_details")
public class EmisRegistrationDetails {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    @Column(name = "person_id", nullable = false)
    private Long personId;

    @Column(name = "ods_code", length = 20, nullable = false)
    private String odsCode;

    @Embedded
    private EncryptedWrapper accessIdentityGuidEncryptedData;

    @Column(name = "deleted", nullable = false)
    private boolean deleted;

    @Column(name = "created", updatable = false, nullable = false)
    private Instant created;

    @Column(name = "last_modified", nullable = false)
    private Instant lastModified;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Long getPersonId() {
        return personId;
    }

    public void setPersonId(Long personId) {
        this.personId = personId;
    }

    public String getOdsCode() {
        return odsCode;
    }

    public void setOdsCode(String odsCode) {
        this.odsCode = odsCode;
    }

    public EncryptedWrapper getAccessIdentityGuidEncryptedData() {
        return accessIdentityGuidEncryptedData;
    }

    public void setAccessIdentityGuidEncryptedData(EncryptedWrapper accessIdentityGuidEncryptedData) {
        this.accessIdentityGuidEncryptedData = accessIdentityGuidEncryptedData;
    }

    public boolean isDeleted() {
        return deleted;
    }

    public void setDeleted(boolean deleted) {
        this.deleted = deleted;
    }

    public Instant getCreated() {
        return created;
    }

    public void setCreated(Instant created) {
        this.created = created;
    }

    public Instant getLastModified() {
        return lastModified;
    }

    public void setLastModified(Instant lastModified) {
        this.lastModified = lastModified;
    }

}
