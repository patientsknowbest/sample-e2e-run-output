package com.pkb.entities.emis_integration;

import com.pkb.entities.pub.PublicPKBPerson;
import org.hibernate.annotations.GenericGenerator;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.OneToOne;
import javax.persistence.Table;
import java.io.Serializable;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(name = "person_guid", schema = "emis_integration")
public class EmisEsPersonGuid implements Serializable {
    private static final long serialVersionUID = 1L;

    @Id
    @GenericGenerator(name = "person_guid_id_seq_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "emis_integration"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "person_guid_id_seq"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(
            strategy = GenerationType.SEQUENCE,
            generator = "person_guid_id_seq_gen")
    private Long id;

    @OneToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "person_id", referencedColumnName = "id")
    private PublicPKBPerson person;

    @Column(name = "value")
    private String value;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getValue() {
        return this.value;
    }

    public void setValue(String value) {
        this.value = value;
    }

    public PublicPKBPerson getPerson() {
        return person;
    }

    public void setPerson(PublicPKBPerson person) {
        this.person = person;
    }
}
