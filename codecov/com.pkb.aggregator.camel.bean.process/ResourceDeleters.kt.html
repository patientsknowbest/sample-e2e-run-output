package com.pkb.aggregator.camel.bean.process

import com.pkb.aggregator.R4Bundle
import com.pkb.aggregator.R4IdType
import com.pkb.aggregator.R4Medication
import com.pkb.aggregator.R4Organization
import com.pkb.aggregator.R4Patient
import com.pkb.aggregator.R4Practitioner
import com.pkb.aggregator.camel.DOWNSTREAM_RESOURCE_ID_PROPERTY
import com.pkb.aggregator.service.DownstreamClient
import org.apache.camel.Body
import org.apache.camel.ExchangeProperty
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class MedicationDeleter(private val downstreamClient: DownstreamClient) {

    fun hardDelete(@Body bundle: R4Bundle) = bundle.entry
        .mapNotNull { it.resource as? R4Medication }
        .forEach {
            downstreamClient.hardDelete(it.idElement.idPart, R4Medication::class.java)
        }
}

@Service
class PatientDeleter(private val downstreamClient: DownstreamClient) {

    private val log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    fun hardDelete(@ExchangeProperty(DOWNSTREAM_RESOURCE_ID_PROPERTY) downstreamPatientId: String?) {
        downstreamPatientId?.run {
            val idPart = stripResourceName(this)
            downstreamClient.hardDelete(idPart, R4Patient::class.java)
            log.info("Deleted downstream data for patient {}", downstreamPatientId)
        }
    }

    private fun stripResourceName(resourceId: String) = resourceId.replace(Regex("^Patient/"), "")
}

@Service
class OrganizationDeleter(private val downstreamClient: DownstreamClient) {
    fun hardDelete(@Body bundle: R4Bundle) {
        bundle.entry
            .mapNotNull { it.resource as? R4Organization }
            .forEach {
                downstreamClient.hardDelete(it.idElement.idPart, R4Organization::class.java)
            }
        bundle.entry
            .mapNotNull { it.resource as? R4Practitioner }
            .forEach {
                downstreamClient.hardDelete(it.idElement.idPart, R4Practitioner::class.java)
            }
    }
}

/**
 * Interface for generic resource deletions. This defines _soft_ deletes because it is intended to propagate upstream
 * deletes from a customer (hence the history is meaningful, should we ever choose to display it). Contrast with hard
 * deletes elsewhere that are used during a full reload, whereby the deletes are not part of the medical history so
 * we don't want them to show up.
 */
@Service
class ResourceDeleter(private val downstreamClient: DownstreamClient) {
    fun softDelete(@Body resourceId: R4IdType) = downstreamClient.softDelete(resourceId)
}
