package com.pkb.aggregator.camel.bean.process

import com.pkb.aggregator.R4Bundle
import com.pkb.aggregator.R4Extension
import com.pkb.aggregator.camel.DOWNSTREAM_RESOURCE_ID_PROPERTY
import com.pkb.aggregator.camel.RELOAD_REQUEST_PROPERTY
import com.pkb.aggregator.camel.SOURCED_RESOURCE_PROPERTY
import com.pkb.aggregator.camel.UPSTREAM_IDENTIFIERS_PROPERTY
import com.pkb.aggregator.camel.UPSTREAM_SOURCE_ID_PROPERTY
import com.pkb.aggregator.extension.aggregatedResourceId
import com.pkb.aggregator.extension.isOnlySourceTaggedWith
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.DownstreamClient
import com.pkb.aggregator.service.dto.ResourceWithSourceId
import com.pkb.aggregator.service.enrichment.createR4SourceExtension
import com.pkb.aggregator.service.paginate.ResourcePaginator
import org.apache.camel.Body
import org.apache.camel.Exchange
import org.apache.camel.ExchangeProperty
import org.hl7.fhir.instance.model.api.IBaseBundle
import org.hl7.fhir.instance.model.api.IDomainResource
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import java.lang.invoke.MethodHandles

abstract class AbstractResourceProcessor<T : IDomainResource>(
    private val resourcePaginator: ResourcePaginator,
    private val downstreamClient: DownstreamClient,
) {
    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    open fun processNew(
        @ExchangeProperty(SOURCED_RESOURCE_PROPERTY) resourceWithSourceId: ResourceWithSourceId<T>,
    ): Any? {
        resourcePaginator.processResource(resourceWithSourceId.resource, resourceWithSourceId.sourceId)
        val newResource = createResource(resourceWithSourceId.resource, null)
        return downstreamClient.create(newResource)
    }

    open fun processSingle(
        @Body bundle: IBaseBundle,
        @ExchangeProperty(SOURCED_RESOURCE_PROPERTY) resourceWithSourceId: ResourceWithSourceId<T>,
        @ExchangeProperty(DOWNSTREAM_RESOURCE_ID_PROPERTY) downstreamResourceId: String
    ): Any? {
        val (sourceId, resource) = resourceWithSourceId
        resourcePaginator.processResource(resource, sourceId)
        val downstreamResource = findSingleBundleEntry(bundle)

        return if (downstreamResource != null) {
            val updatedResource = updateResource(resource, downstreamResource)
            downstreamClient.update(updatedResource)
        } else {
            val newResource = createResource(resource, downstreamResourceId)
            downstreamClient.create(newResource)
        }
    }

    open fun processDelete(
        @Body bundle: IBaseBundle,
        @ExchangeProperty(UPSTREAM_SOURCE_ID_PROPERTY) upstreamSourceId: UpstreamResource,
    ) {
        val downstreamResource = findSingleBundleEntry(bundle)
            ?: throw RuntimeException("Empty bundle")

        val sourceTag = createR4SourceExtension(upstreamSourceId.upstreamSource, upstreamSourceId.upstreamResourceId)
        if (downstreamResource.isOnlySourceTaggedWith(sourceTag)) {
            downstreamClient.softDelete(downstreamResource)
            log.info("Deleted resource {} as delete notification received from single source {}",
                downstreamResource.aggregatedResourceId(), upstreamSourceId.upstreamSource)
        } else {
            // Remove all elements tagged with source, but don't remove the resource itself
            stripResource(downstreamResource, sourceTag)
            downstreamClient.update(downstreamResource)
            log.info("Stripped info from resource {} due to delete notification from source {}",
                downstreamResource.aggregatedResourceId(), upstreamSourceId.upstreamSource)
        }
    }

    fun logOutcome(exchange: Exchange) {
        val stats = exchange.message.getBody(R4Bundle::class.java)
            ?.entry
            ?.groupBy { it.resource.resourceType }
            ?.mapValues { it.value.size }

        log.info("Reload complete: route={}, request={}, bundle-stats={}, identifiers-searched={}",
            exchange.fromRouteId,
            exchange.getProperty(RELOAD_REQUEST_PROPERTY),
            stats,
            exchange.getProperty(UPSTREAM_IDENTIFIERS_PROPERTY, List::class.java)?.size,
        )
    }

    protected abstract fun createResource(resource: T, downstreamResourceId: String?): T
    protected abstract fun updateResource(upstreamResource: T, downstreamResource: T): T
    protected abstract fun findSingleBundleEntry(bundle: IBaseBundle): T?
    protected abstract fun stripResource(resource: T, sourceTag: R4Extension)
}
