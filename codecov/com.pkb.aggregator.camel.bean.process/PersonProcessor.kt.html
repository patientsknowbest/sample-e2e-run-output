package com.pkb.aggregator.camel.bean.process

import com.pkb.aggregator.R4Bundle
import com.pkb.aggregator.R4Extension
import com.pkb.aggregator.R4Person
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.extension.singleEntry
import com.pkb.aggregator.service.DownstreamClient
import com.pkb.aggregator.service.aggregation.r4.PersonAggregator
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import com.pkb.aggregator.service.paginate.ResourcePaginator
import org.hl7.fhir.instance.model.api.IBaseBundle
import org.springframework.stereotype.Service

@Service
class PersonProcessor(
    private val personAggregator: PersonAggregator,
    resourcePaginator: ResourcePaginator,
    downstreamClient: DownstreamClient,
) : AbstractResourceProcessor<R4Person>(resourcePaginator, downstreamClient) {

    override fun createResource(resource: R4Person, downstreamResourceId: String?) =
        personAggregator.updateOrCreate(resource, null, downstreamResourceId)

    override fun updateResource(upstreamResource: R4Person, downstreamResource: R4Person) =
        personAggregator.updateOrCreate(upstreamResource, downstreamResource, downstreamResource.simpleResourceId())

    override fun findSingleBundleEntry(bundle: IBaseBundle) = when(bundle) {
        is R4Bundle -> bundle.singleEntry<R4Person>()
        else -> throw UnexpectedTypeException(bundle)
    }

    override fun stripResource(resource: R4Person, sourceTag: R4Extension) =
        personAggregator.stripResource(resource, sourceTag)
}
