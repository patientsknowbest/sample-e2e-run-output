package com.pkb.aggregator.camel.bean.process

import com.pkb.aggregator.R4Bundle
import com.pkb.aggregator.R4Extension
import com.pkb.aggregator.R4Practitioner
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.extension.singleEntry
import com.pkb.aggregator.service.DownstreamClient
import com.pkb.aggregator.service.aggregation.r4.PractitionerAggregator
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import com.pkb.aggregator.service.paginate.ResourcePaginator
import org.hl7.fhir.instance.model.api.IBaseBundle
import org.springframework.stereotype.Service

@Service
class PractitionerProcessor(
    private val practitionerAggregator: PractitionerAggregator,
    resourcePaginator: ResourcePaginator,
    downstreamClient: DownstreamClient,
) : AbstractResourceProcessor<R4Practitioner>(resourcePaginator, downstreamClient) {

    override fun createResource(resource: R4Practitioner, downstreamResourceId: String?) =
        practitionerAggregator.updateOrCreate(resource, null, downstreamResourceId)

    override fun updateResource(upstreamResource: R4Practitioner, downstreamResource: R4Practitioner) =
        practitionerAggregator.updateOrCreate(
            upstreamResource,
            downstreamResource,
            downstreamResource.simpleResourceId()
        )

    override fun findSingleBundleEntry(bundle: IBaseBundle) = when(bundle) {
        is R4Bundle -> bundle.singleEntry<R4Practitioner>()
        else -> throw UnexpectedTypeException(bundle)
    }

    override fun stripResource(resource: R4Practitioner, sourceTag: R4Extension) =
        practitionerAggregator.stripResource(resource, sourceTag)
}
