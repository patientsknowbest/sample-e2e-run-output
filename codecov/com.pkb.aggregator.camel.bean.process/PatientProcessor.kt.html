package com.pkb.aggregator.camel.bean.process

import com.pkb.aggregator.camel.ENCOUNTER_PROPERTY
import com.pkb.aggregator.camel.MEDICATION_STATEMENT_PROPERTY
import com.pkb.aggregator.camel.OBSERVATION_PROPERTY
import com.pkb.aggregator.camel.PATIENT_PROPERTY
import com.pkb.aggregator.camel.PERSON_PROPERTY

import com.pkb.aggregator.service.aggregation.reload.PatientReloadAggregator.AllPatientResources
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import com.pkb.aggregator.service.paginate.ResourcePaginator
import org.apache.camel.Body
import org.apache.camel.ExchangeProperties
import org.hl7.fhir.instance.model.api.IBaseBundle
import org.springframework.stereotype.Service

@Service
class PatientProcessor(
    private val resourcePaginator: ResourcePaginator,
    private val patientAggregator: PatientAggregator,
    private val patientReloadAggregator: PatientReloadAggregator,
    downstreamClient: DownstreamClient,
) : AbstractResourceProcessor<R4Patient>(resourcePaginator, downstreamClient) {

    fun processMultiple(
        @Body bundles: Iterable<IBaseBundle>,
        @ExchangeProperties properties: MutableMap<String, Any>,
    ): String {
        properties[PATIENT_PROPERTY] = bundles.flatMap { resourcePaginator.processBundle(it).pageResources }
        return properties[UPSTREAM_IDENTIFIER_SEARCH_TERM_PROPERTY].toString()
    }

    fun aggregate(@ExchangeProperties properties: MutableMap<String, Any>): R4Bundle =
        patientReloadAggregator.aggregate(
            properties[DOWNSTREAM_RESOURCE_ID_PROPERTY]?.toString(),
            AllPatientResources(
                patient = propertyAsResourceList(properties, PATIENT_PROPERTY),
                person = propertyAsResourceList(properties, PERSON_PROPERTY),
                encounters = propertyAsResourceList(properties, ENCOUNTER_PROPERTY),
                observations = propertyAsResourceList(properties, OBSERVATION_PROPERTY),
                medicationStatements = propertyAsResourceList(properties, MEDICATION_STATEMENT_PROPERTY),
                appointments = propertyAsResourceList(properties, APPOINTMENT_PROPERTY),
                conditions = propertyAsResourceList(properties, CONDITION_PROPERTY),
                allergies = propertyAsResourceList(properties, ALLERGY_PROPERTY),
            )
        )

    override fun createResource(resource: R4Patient, downstreamResourceId: String?) =
        patientAggregator.updateOrCreate(resource, null, downstreamResourceId)

    override fun updateResource(upstreamResource: R4Patient, downstreamResource: R4Patient) =
        patientAggregator.updateOrCreate(upstreamResource, downstreamResource, downstreamResource.simpleResourceId())

    override fun findSingleBundleEntry(bundle: IBaseBundle) = when(bundle) {
        is R4Bundle -> bundle.singleEntry<R4Patient>()
        else -> throw UnexpectedTypeException(bundle)
    }

    override fun stripResource(resource: R4Patient, sourceTag: R4Extension) =
        patientAggregator.stripResource(resource, sourceTag)

    private fun propertyAsResourceList(properties: MutableMap<String, Any>, key: String) =
        when (val value = properties[key]) {
            is MutableList<*> -> value.asMutableListOfType<R4DomainResource>()
            null -> mutableListOf()
            else -> throw UnexpectedTypeException("Unexpected non-resource-list type in $key", value)
        }
}
