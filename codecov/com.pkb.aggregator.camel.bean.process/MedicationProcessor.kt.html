package com.pkb.aggregator.camel.bean.process

import com.pkb.aggregator.R4Bundle
import com.pkb.aggregator.R4Extension
import com.pkb.aggregator.R4IdType
import com.pkb.aggregator.R4Medication
import com.pkb.aggregator.camel.UPSTREAM_SOURCE_ID_PROPERTY
import com.pkb.aggregator.extension.getResources
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.extension.singleEntry
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.DownstreamClient
import com.pkb.aggregator.service.FHIR_TYPE_MEDICATION
import com.pkb.aggregator.service.aggregation.r4.MedicationAggregator
import com.pkb.aggregator.service.aggregation.reload.MedicationReloadAggregator
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import com.pkb.aggregator.service.paginate.ResourcePaginator
import org.apache.camel.Body
import org.apache.camel.ExchangeProperty
import org.hl7.fhir.instance.model.api.IBaseBundle
import org.springframework.stereotype.Service

@Service
class MedicationProcessor(
    private val medicationAggregator: MedicationAggregator,
    private val medicationReloadAggregator: MedicationReloadAggregator,
    resourcePaginator: ResourcePaginator,
    downstreamClient: DownstreamClient,
) : AbstractResourceProcessor<R4Medication>(resourcePaginator, downstreamClient) {

    fun aggregate(@Body bundles: List<IBaseBundle>): IBaseBundle {
        val resources = bundles.flatMap { it.getResources() }
        return medicationReloadAggregator.aggregate(null, resources)
    }

    override fun createResource(resource: R4Medication, downstreamResourceId: String?) =
        medicationAggregator.updateOrCreate(resource, null, downstreamResourceId)

    override fun updateResource(upstreamResource: R4Medication, downstreamResource: R4Medication) =
        medicationAggregator.updateOrCreate(upstreamResource, downstreamResource, downstreamResource.simpleResourceId())

    override fun findSingleBundleEntry(bundle: IBaseBundle) = when(bundle) {
        is R4Bundle -> bundle.singleEntry<R4Medication>()
        else -> throw UnexpectedTypeException(bundle)
    }

    override fun stripResource(resource: R4Medication, sourceTag: R4Extension) =
        medicationAggregator.stripResource(resource, sourceTag)

    fun toDownstreamResourceId(@ExchangeProperty(UPSTREAM_SOURCE_ID_PROPERTY) upstreamSourceId: UpstreamResource) =
        R4IdType(FHIR_TYPE_MEDICATION, upstreamSourceId.toDownstreamResourceId())
}
