package com.pkb.aggregator.camel.bean.process

import com.pkb.aggregator.Dstu3Resource
import com.pkb.aggregator.R4DomainResource
import com.pkb.aggregator.R4Resource
import com.pkb.aggregator.camel.UPSTREAM_SOURCE_ID_PROPERTY
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.extension.toR4
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.dto.ResourceWithSourceId
import com.pkb.aggregator.service.enrichment.createR4SourceExtension
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import org.apache.camel.Body
import org.apache.camel.ExchangeProperty
import org.hl7.fhir.instance.model.api.IAnyResource
import org.springframework.stereotype.Service

/**
 * This class provides functionality for taking resources and preparing them so that the rest of the processing
 * pipeline has everything it needs. Currently this includes parsing uploaded ndjson files into HAPI objects and
 * applying source information to them, and converting DSTU3 to R4 if required.
 */
@Service
class ResourcePreProcessor {

    /**
     * Wraps the incoming resource in a [ResourceWithSourceId] DTO, and also tags the resource with a source extension
     */
    fun associateWithSource(
        @Body resource: IAnyResource,
        @ExchangeProperty(UPSTREAM_SOURCE_ID_PROPERTY) upstreamResource: UpstreamResource,
    ) = ResourceWithSourceId(
        upstreamResource.upstreamSource,
        when (resource) {
            is R4Resource -> resource
            is Dstu3Resource -> resource.toR4()
            else -> throw UnexpectedTypeException(resource)
        }
    ).also {
        tag(it)
    }

    fun tag(resourceWithSourceId: ResourceWithSourceId<R4Resource>) {
        val (sourceId, resource) = resourceWithSourceId
        when (resource) {
            is R4DomainResource -> tagR4Resource(sourceId, resource)
            else -> throw UnexpectedTypeException(resource)
        }
    }

    private fun tagR4Resource(sourceId: String, resource: R4DomainResource) {
        val extension = createR4SourceExtension(sourceId, resource.simpleResourceId())
        resource.extension.add(extension)
    }
}
