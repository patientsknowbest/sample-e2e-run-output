package com.pkb.aggregator.camel.bean.process

import ca.uhn.fhir.rest.api.MethodOutcome
import com.pkb.aggregator.R4IdType
import com.pkb.aggregator.camel.EXTRACTED_REFERENCE_MAP
import com.pkb.aggregator.camel.UPSTREAM_SOURCE_ID_PROPERTY
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.copier.DownstreamCopier
import com.pkb.aggregator.service.dto.ResourceWithSourceId
import com.pkb.aggregator.service.paginate.ResourcePaginator
import com.pkb.aggregator.service.reference.ReferenceExtractor
import org.apache.camel.Body
import org.apache.camel.ExchangeProperties
import org.apache.camel.ExchangeProperty
import org.hl7.fhir.instance.model.api.IDomainResource
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class ClinicalResourceProcessor(
    private val copiers: List<DownstreamCopier<IDomainResource>>,
    private val resourcePaginator: ResourcePaginator,
    private val referenceExtractor: ReferenceExtractor,
) {

    private val log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    fun copy(resource: IDomainResource, references: ReferenceExtractor.IdentifierMap): IDomainResource =
        // Must be one and only one result, since only one copier should have accepted the resource
        copiers.single { it.accepts(resource) }.copy(resource, references)

    fun processSingle(
        @Body resourceWithSourceId: ResourceWithSourceId<IDomainResource>,
        @ExchangeProperties properties: MutableMap<String, Any>,
    ): IDomainResource {
        val (sourceId, resource) = resourceWithSourceId
        resourcePaginator.processResource(resource, sourceId)
        val references = referenceExtractor.extract(listOf(resource))
        properties[EXTRACTED_REFERENCE_MAP] = references
        return copy(resource, references)
    }

    fun toDownstreamResourceId(@ExchangeProperty(UPSTREAM_SOURCE_ID_PROPERTY) upstreamSourceId: UpstreamResource): R4IdType {
        val upstreamIdType = R4IdType(upstreamSourceId.upstreamResourceId)
        return R4IdType(upstreamIdType.resourceType, upstreamSourceId.toDownstreamResourceId())
    }

    fun logOutcome(
        @ExchangeProperty(UPSTREAM_SOURCE_ID_PROPERTY) upstreamResource: UpstreamResource,
        @Body outcome: MethodOutcome,
    ) {
        try {
            val result = mapOf<String, Any?>(
                "upstreamResource" to upstreamResource,
                "downstreamResourceId" to outcome.resource?.simpleResourceId(),
                "created" to outcome.created,
                "outcome" to outcome.operationOutcome,
            )
            log.info("Resource update outcome: {}", result)
        } catch (ex: Exception) {
            log.error("Resource updated but unable to record outcome", ex)
        }
    }
}
