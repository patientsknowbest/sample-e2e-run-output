package com.pkb.datamodel.plantemplate;

import com.pkb.datamodel.DataModelStyle;
import com.pkb.datamodel.Identifiable;
import org.immutables.value.Value;

import java.util.Optional;

@Value.Immutable
@DataModelStyle
public interface TemplateAttachment extends Identifiable<Long> {
    String getName();
    String getLabel();
    PHPlanTemplate getPlanTemplate();
    Optional<String> getMediaType();
    Optional<Boolean> getChecked();

    default boolean isBrowserImage() {
        String mediaType = getMediaType().orElse("NO_TYPE");
        return "image/jpeg".equals(mediaType) || "image/png".equals(mediaType) || "image/gif".equals(mediaType)
                || "jpeg".equals(mediaType) || "jpg".equals(mediaType) || "png".equals(mediaType) || "gif".equals(mediaType);
    }
}
