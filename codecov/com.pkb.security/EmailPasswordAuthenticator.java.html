package com.pkb.security;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.PersonAuthInfo;
import com.pkb.datamodel.Email;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.user.impl.PersonContactManager;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.user.entity.PersonContact;
import org.jetbrains.annotations.NotNull;
import org.springframework.security.authentication.BadCredentialsException;

import java.util.Comparator;
import java.util.List;
import java.util.Optional;

import static com.pkb.datamodel.ImmutableDisclosingEmail.disclosingEmail;
import static com.pkb.entities.enums.LoginAttemptLogResult.FAIL_BAD_USERNAME;
import static java.lang.String.format;

public class EmailPasswordAuthenticator extends AbstractPrincipalCredentialAuthenticator {

    public static class Factory {

        private final PersonContactManager personContactManager;

        private final TeamUserManager teamUserManager;

        private final UserManager userManager;

        private final UserAccessManager userAccessManager;


        public Factory(
                PersonContactManager personContactManager,
                TeamUserManager teamUserManager,
                UserManager userManager,
                UserAccessManager userAccessManager) {
            this.personContactManager = personContactManager;
            this.teamUserManager = teamUserManager;
            this.userManager = userManager;
            this.userAccessManager = userAccessManager;
        }

        EmailPasswordAuthenticator createAuthenticator(LoginAttemptAuditor loginAttemptAuditor,
                Optional<BaseAuthenticationDetails> originalLoginDetails) {
            return new EmailPasswordAuthenticator(
                    personContactManager,
                    loginAttemptAuditor,
                    teamUserManager,
                    userManager,
                    originalLoginDetails);
        }

    }

    private final PersonContactManager personContactManager;

    private final IdPasswordAuthenticator idPasswordAuthenticator;

    private Comparator<PersonContact> preferredPersonContactComparator() {
        return (pc1, pc2) -> {
            boolean pc1IsTeamUser = teamUserManager.isTeamUser(pc1.getPerson().getId());
            boolean pc2IsTeamUser = teamUserManager.isTeamUser(pc2.getPerson().getId());
            if (pc1IsTeamUser == pc2IsTeamUser) {
                return 0;
            } else if (pc1IsTeamUser) {
                return -1;
            } else {
                return 1;
            }
        };
    }

    public EmailPasswordAuthenticator(
            PersonContactManager personContactManager,
            @NotNull LoginAttemptAuditor loginAttemptAuditor,
            TeamUserManager teamUserManager,
            UserManager userManager,
            Optional<BaseAuthenticationDetails> originalLoginDetails) {
        super(teamUserManager, loginAttemptAuditor);
        this.personContactManager = personContactManager;
        this.idPasswordAuthenticator = new IdPasswordAuthenticator(teamUserManager, loginAttemptAuditor, userManager,
                originalLoginDetails);
    }

    private PKBPerson lookupMatchingPersonByEmail(Email email) {
        List<PersonContact> contacts = personContactManager.findPersonByConfirmedOrPrimaryEmail(email, PersonContact.Lazy.PERSON);
        if (contacts.isEmpty()) {
            throw unknownEmailFailure(email);
        }
        return contacts.stream()
                .filter(PersonContact::isConfirmed)
                .sorted(preferredPersonContactComparator())
                .map(PersonContact::getPerson)
                .findFirst()
                .orElseThrow(() -> unknownEmailFailure(email));
    }

    @Override
    protected PersonAuthInfo doAuthenticate(EHRRequestContext requestContext, String email, String password) {
        Email typedEmail = disclosingEmail(email);
        personWithMatchingPrincipal = lookupMatchingPersonByEmail(typedEmail);
        return idPasswordAuthenticator.authenticate(requestContext, personWithMatchingPrincipal.getIdString(), password);
    }

    private BadCredentialsException unknownEmailFailure(Email email) {
        loginAttemptAuditor.logAttempt(FAIL_BAD_USERNAME, null);
        return new BadCredentialsException(format("UserManager found no user for username >%s<; failing authentication.", email.address()));
    }
}
