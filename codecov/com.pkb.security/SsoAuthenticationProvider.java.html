package com.pkb.security;


import com.pkb.app.entity.PersonAuthInfo;
import com.pkb.entities.enums.SSOType;
import com.pkb.kms.shared.representation.KmsError;
import com.pkb.security.LoginAttemptAuditor.Factory;
import com.pkb.service.sso.impl.emis.EMISSSOAuthManager;
import com.pkb.sso.SsoMetrics;
import com.pkb.user.entity.PKBPerson;
import io.vavr.Tuple;
import io.vavr.Tuple3;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import static com.pkb.entities.enums.LoginAttemptLogResult.FAIL_BAD_PASSWORD;

public final class SsoAuthenticationProvider extends BaseAuthenticationProvider<SsoAuthenticationDetails> {

    private final EMISSSOAuthManager emisSsoAuthManager;
    private final SsoMetrics ssoMetrics;

    public SsoAuthenticationProvider(Factory loginAttemptAuditorFactory, EMISSSOAuthManager emisSsoAuthManager, SsoMetrics ssoMetrics) {
        super(SsoAuthenticationDetails.class, loginAttemptAuditorFactory);
        this.emisSsoAuthManager = emisSsoAuthManager;
        this.ssoMetrics = ssoMetrics;
    }

    @Override
    @NotNull
    protected Either<AuditableLoginException, Tuple3<PersonAuthInfo, Runnable, Runnable>> doAuthenticate(TypedAuthentication<SsoAuthenticationDetails> authentication) {
        SsoAuthenticationDetails details = authentication.getDetails();
        String username = authentication.getName();
        String ssoKey = authentication.getCredentials();
        var result = emisSsoAuthManager.authenticateSSOUser(details.getEHRRequestContext(), username, ssoKey, details.getSsoType());
        var teamCode = result.getTeamCode().orElse("");

        return result.getPerson().map(person ->
                        result.getLoginError().map(err -> getBadCredentialsResult(person, err)).orElseGet(() -> getSuccessResult(details, teamCode, person))
                )
                .orElseGet(() -> Either.left(new AuditableUsernameNotFoundException(username)))
                .peekLeft(obj -> getFailureRecorder(teamCode, details.getSsoType()).run());
    }

    @NotNull
    private Either<AuditableLoginException, Tuple3<PersonAuthInfo, Runnable, Runnable>> getBadCredentialsResult(PKBPerson person, KmsError err) {
        return Either.left(new AuditableBadCredentialsException(person.getId(), FAIL_BAD_PASSWORD, err));
    }

    @NotNull
    private Either<AuditableLoginException, Tuple3<PersonAuthInfo, Runnable, Runnable>> getSuccessResult(SsoAuthenticationDetails details, String teamCode, PKBPerson person) {
        return Either.right(Tuple.of(PersonAuthInfo.of(person), getSuccessRecorder(teamCode, details.getSsoType()), getFailureRecorder(teamCode, details.getSsoType())));
    }


    private Runnable getSuccessRecorder(String teamCode, SSOType type) {
        return () -> ssoMetrics.incrementSuccessfulAuthentication(teamCode, type);
    }

    private Runnable getFailureRecorder(String teamCode, SSOType type) {
        return () -> ssoMetrics.incrementUnSuccessfulAuthentication(teamCode, type);
    }


}
