package com.pkb.security;

import org.apache.http.client.utils.URIBuilder;
import org.slf4j.Logger;
import org.springframework.http.HttpStatus;
import org.springframework.security.core.AuthenticationException;
import org.springframework.security.web.authentication.ExceptionMappingAuthenticationFailureHandler;

import javax.servlet.ServletException;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.servlet.http.HttpSession;
import java.io.IOException;
import java.lang.invoke.MethodHandles;
import java.net.URISyntaxException;

import static com.pkb.service.oauth2.NhsLoginError.BAD_CREDENTIALS_ERROR;
import static com.pkb.service.oauth2.NhsLoginError.PERSON_DEMOGRAPHICS_CONFLICT;
import static com.pkb.service.oauth2.NhsLoginError.PERSON_DIFFERENT_RECORDS;
import static com.pkb.service.oauth2.NhsLoginError.PERSON_DOES_NOT_EXIST;
import static com.pkb.service.oauth2.NhsLoginError.PERSON_EXISTS_BUT_NO_NHS_CREDENTIALS;
import static com.pkb.service.oauth2.NhsLoginError.PERSON_HAS_ACCESS_FROZEN;
import static com.pkb.service.oauth2.NhsLoginError.PERSON_IS_DEAD;
import static com.pkb.service.oauth2.NhsLoginError.PERSON_MISSING_DEFAULT_ACCOUNT_ID;
import static com.pkb.service.oauth2.NhsLoginError.PERSON_NOT_PATIENT;
import static com.pkb.service.oauth2.NhsLoginError.PERSON_OUT_OF_AREA;
import static com.pkb.service.oauth2.NhsLoginError.PERSON_UNDER_AGE_LIMIT;
import static com.pkb.util.Constants.NHS_LOGIN_IDS;
import static com.pkb.util.Constants.NHS_LOGIN_INFO;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.Predicates.instanceOf;
import static io.vavr.Predicates.isIn;
import static org.slf4j.LoggerFactory.getLogger;

public class NhsLoginAuthenticationExceptionMappingAuthenticationFailureHandler
        extends ExceptionMappingAuthenticationFailureHandler {

    private static final Logger LOGGER = getLogger(MethodHandles.lookup().lookupClass());

    @Override
    public void onAuthenticationFailure(
            HttpServletRequest request,
            HttpServletResponse response,
            AuthenticationException exception) throws IOException, ServletException {

        String redirectUri = Match(exception).of(
                Case($(instanceOf(NhsLoginAuthenticationException.class)), nhsException -> {
                    // Poke credentials into session so that we can post them to KMS if login succeeds
                    String errorCode = nhsException.getError().getErrorCode();
                    return Match(errorCode).of(
                            Case($(isIn(PERSON_DEMOGRAPHICS_CONFLICT,
                                    PERSON_NOT_PATIENT,
                                    PERSON_UNDER_AGE_LIMIT,
                                    PERSON_DIFFERENT_RECORDS,
                                    PERSON_MISSING_DEFAULT_ACCOUNT_ID)), () -> resolveError(errorCode)),
                            Case($(PERSON_DOES_NOT_EXIST), () -> resolveException(nhsException, request.getSession(), "/activate/nhsLoginFullPatientForm.action")),
                            Case($(PERSON_EXISTS_BUT_NO_NHS_CREDENTIALS), () -> resolveException(nhsException, request.getSession(), "/nhsLoginConfirmPassword.action")),
                            Case($(PERSON_OUT_OF_AREA), () -> resolveException(nhsException, request.getSession(), "/activate/nhsLoginPatientForm.action")),
                            Case($(BAD_CREDENTIALS_ERROR), () -> resolveException(nhsException, request.getSession(), "/nhsError.action?error=bad.credentials")),
                            Case($(PERSON_IS_DEAD), () -> resolveException(nhsException, request.getSession(), "/nhsError.action?error=contact.team.for.access")),
                            Case($(PERSON_HAS_ACCESS_FROZEN), () -> resolveException(nhsException, request.getSession(), "/nhsError.action?error=person.has.access.frozen")),
                            Case($(), code -> {
                                LOGGER.error("NHS login authentication exception occurred with unknown error code: {}", code, nhsException);
                                return null;
                            }));
                }),
                Case($(), () -> {
                    LOGGER.error("Unknown exception occurred during NHS Login authentication", exception);
                    return null;
                }));

        if (redirectUri != null) {
            response.sendRedirect(redirectUri);
        } else {
            response.sendError(HttpStatus.UNAUTHORIZED.value(), exception.getMessage());
        }
    }

    private String resolveException(NhsLoginAuthenticationException nhsException, HttpSession session, String redirectionEndpoint) {
        session.setAttribute(NHS_LOGIN_INFO, nhsException.getAuthentication());
        nhsException.getError()
                .getPersonIds()
                .ifPresent(ids -> session.setAttribute(NHS_LOGIN_IDS, ids));

        return nhsException.getAuthentication().getDetails().getTeamCode().map(code -> {
            try {
                URIBuilder redirectionWithParams = new URIBuilder(redirectionEndpoint).addParameter("teamCode", code);

                nhsException.getAuthentication().getDetails().isSelfReg()
                        .ifPresent(isSelfReg -> redirectionWithParams.addParameter("isSelfReg", isSelfReg.toString()));

                return redirectionWithParams.toString();
            } catch (URISyntaxException e) {
                LOGGER.error("Invalid NHS Login redirection endpoint {}", redirectionEndpoint, e);
            }

            return redirectionEndpoint;
        }).orElse(redirectionEndpoint);
    }

    private String resolveError(String errorCode) {
        return "/nhsError.action?error=" + errorCode;
    }
}
