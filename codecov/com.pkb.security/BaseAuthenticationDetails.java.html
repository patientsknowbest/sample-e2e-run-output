package com.pkb.security;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.EHRRequestContext.AccountLinkType;
import com.pkb.app.entity.ImmutableLoggedOutEHRRequestContext;
import com.pkb.entities.enums.LoginAttemptLogResult;
import com.pkb.entities.enums.Route;
import com.pkb.util.AppIdentityUtil;
import org.immutables.value.Value.Derived;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.security.core.AuthenticationException;

import javax.servlet.http.HttpServletRequest;
import java.io.Serializable;
import java.util.Optional;
import java.util.UUID;
import java.util.function.Supplier;

import static com.pkb.consent.model.AbstractConsentStatus.noConsentRequired;

public interface BaseAuthenticationDetails extends Serializable {
    String getRemoteAddress();

    Route getRoute();

    default LoginAttemptLogResult getLoginAttemptSuccessResult() {
        return LoginAttemptLogResult.SUCCESS;
    }

    default LoginAttemptLogResult getLoginAttemptFailResult() {
        return LoginAttemptLogResult.FAIL_UNKNOWN;
    }

    default LoginAttemptLogResult getLoginAttemptBadPasswordResult() {
        return LoginAttemptLogResult.FAIL_BAD_PASSWORD;
    }

    @Derived
    default EHRRequestContext getEHRRequestContext() {
        return ImmutableLoggedOutEHRRequestContext.of(AccountLinkType.NO_LINK, getRoute(), noConsentRequired(), UUID.randomUUID());
    }

    @SuppressWarnings("override")
    interface Builder<T extends BaseAuthenticationDetails, B extends Builder<T, B>> {
        B remoteAddress(String remoteAddress);

        T build();

        default Builder<T, B> populateFrom(HttpServletRequest request, AppIdentityUtil appIdentityUtil) {
            Optional.ofNullable(request.getRemoteAddr()).ifPresent(this::remoteAddress);
            return this;
        }
    }

    final class AuthenticationDetailsSource<T extends BaseAuthenticationDetails, B extends Builder<T, B>> implements
            org.springframework.security.authentication.AuthenticationDetailsSource<HttpServletRequest, T> {

        private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

        private final Supplier<B> builderFactory;
        private AppIdentityUtil appIdentityUtil;

        private AuthenticationDetailsSource(Supplier<B> builderFactory, AppIdentityUtil appIdentityUtil) {
            this.builderFactory = builderFactory;
            this.appIdentityUtil = appIdentityUtil;
        }

        @Override
        public T buildDetails(HttpServletRequest context) throws AuthenticationException {
            try {
                return builderFactory.get().populateFrom(context, appIdentityUtil).build();
            } catch (IllegalStateException | IllegalArgumentException e) {
                LOGGER.error("Error getting authentication details from request: ", e);
                throw new MissingAuthenticationDetailsException(e);
            }
        }
    }

    static <T extends BaseAuthenticationDetails, B extends Builder<T, B>> org.springframework.security.authentication.AuthenticationDetailsSource<HttpServletRequest, T> getAuthenticationDetailsSource(
            Supplier<B> builderFactory, AppIdentityUtil appIdentityUtil) {
        return new AuthenticationDetailsSource<>(builderFactory, appIdentityUtil);
    }

    final class MissingAuthenticationDetailsException extends AuthenticationException {
        MissingAuthenticationDetailsException(Exception t) {
            super("Authentication request was missing required state: " + t.getMessage(), t);
        }
    }
}
