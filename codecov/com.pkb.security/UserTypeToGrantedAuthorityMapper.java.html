package com.pkb.security;

import com.google.common.collect.Maps;
import com.pkb.entities.enums.UserType;
import org.springframework.security.core.GrantedAuthority;
import org.springframework.security.core.authority.SimpleGrantedAuthority;

import java.util.ArrayList;
import java.util.EnumMap;
import java.util.List;
import java.util.Map;

/**
 * Utility class which helps mapping {@link UserType} to their role.
 */
public final class UserTypeToGrantedAuthorityMapper {

    private static final Map<UserType, GrantedAuthority> MAPPINGS;

    private UserTypeToGrantedAuthorityMapper() {
    }

    static {
        Map<UserType, GrantedAuthority> temporaryMap = new EnumMap<>(UserType.class);
        for (UserType actual : UserType.values()) {
            temporaryMap.put(actual, new SimpleGrantedAuthority("ROLE_" + actual.name()));
        }
        MAPPINGS = Maps.immutableEnumMap(temporaryMap);
    }

    private static GrantedAuthority getAuthority(UserType type) {
        return MAPPINGS.get(type);
    }

    public static List<GrantedAuthority> getAuthorityList(UserType type) {
        ArrayList<GrantedAuthority> result = new ArrayList<>();
        result.add(getAuthority(type));
        return result;
    }
 }
