package com.pkb.security;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.PersonAuthInfo;
import com.pkb.service.team.TeamUserManager;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;
import org.springframework.security.authentication.BadCredentialsException;

import javax.annotation.concurrent.NotThreadSafe;

import static com.pkb.entities.enums.LoginAttemptLogResult.FAIL_UNKNOWN;

@NotThreadSafe
public abstract class AbstractPrincipalCredentialAuthenticator implements PrincipalCredentialAuthenticator {

    protected PKBPerson personWithMatchingPrincipal;

    protected TeamUserManager teamUserManager;

    protected LoginAttemptAuditor loginAttemptAuditor;

    public AbstractPrincipalCredentialAuthenticator(
            TeamUserManager teamUserManager,
            @NotNull LoginAttemptAuditor loginAttemptAuditor
    ) {
        this.teamUserManager = teamUserManager;
        this.loginAttemptAuditor = loginAttemptAuditor;
    }

    @Override
    public final PersonAuthInfo authenticate(EHRRequestContext requestContext, String principal, String credential) {
        try {
            return doAuthenticate(requestContext, principal, credential);
        } catch (InvalidPrincipalException | WifiInTheCemeteryException | CoordAwaitingActivationException | FrozenByTeamException | TooManyLoginAttemptsException e) {
            throw e;
        } catch (Exception e) {
            Long loggedPersonId = personWithMatchingPrincipal == null ? null : personWithMatchingPrincipal.getId();
            loginAttemptAuditor.logAttempt(FAIL_UNKNOWN, loggedPersonId);
            if (personWithMatchingPrincipal != null && teamUserManager
                    .isUserCredentialsResetPending(personWithMatchingPrincipal.getId())) {
                throw new AdminResetInProgressException("reset in progress", e);
            } else {
                throw new BadCredentialsException("Invalid credentials. " + e.getMessage(), e);
            }
        }
    }

    protected abstract PersonAuthInfo doAuthenticate(EHRRequestContext requestContext, String principal, String credential);

}
