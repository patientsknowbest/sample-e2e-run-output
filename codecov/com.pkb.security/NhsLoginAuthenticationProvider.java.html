package com.pkb.security;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.app.entity.PersonAuthInfo;
import com.pkb.common.util.FrameFilter;
import com.pkb.datamodel.Email;
import com.pkb.security.LoginAttemptAuditor.Factory;
import com.pkb.service.nhslogin.impl.NhsLoginRegistrationManager;
import com.pkb.service.oauth2.NhsLoginService;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.team.TeamManager;
import com.pkb.service.user.impl.UserManager;
import io.vavr.Tuple;
import io.vavr.Tuple3;
import io.vavr.control.Either;
import io.vavr.control.Try;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.security.authentication.BadCredentialsException;

import java.lang.invoke.MethodHandles;

import static com.pkb.datamodel.ImmutableDisclosingEmail.disclosingEmail;

public class NhsLoginAuthenticationProvider extends BaseAuthenticationProvider<NhsLoginAuthenticationDetails> {

    private static final Runnable NOOP = () -> {};
    private final NhsLoginService nhsLoginService;
    private final UserManager userManager;
    private final TeamManager teamManager;
    private final PatientConsentManager patientConsentManager;
    private final NhsLoginRegistrationManager nhsLoginRegistrationManager;
    static Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    public NhsLoginAuthenticationProvider(NhsLoginService nhsLoginService, Factory loginAttemptAuditorFactory,
                                          UserManager userManager, TeamManager teamManager,
                                          PatientConsentManager patientConsentManager,
                                          NhsLoginRegistrationManager nhsLoginRegistrationManager) {
        super(NhsLoginAuthenticationDetails.class, loginAttemptAuditorFactory);
        this.nhsLoginService = nhsLoginService;
        this.userManager = userManager;
        this.teamManager = teamManager;
        this.patientConsentManager = patientConsentManager;
        this.nhsLoginRegistrationManager = nhsLoginRegistrationManager;
    }

    @NotNull
    @Override
    protected Either<AuditableLoginException, Tuple3<PersonAuthInfo, Runnable, Runnable>> doAuthenticate(TypedAuthentication<NhsLoginAuthenticationDetails> authentication) {
        Email email = disclosingEmail(authentication.getName());
        NhsLoginAuthenticationDetails details = authentication.getDetails();
        String secretBase64 = authentication.getCredentials();
        return nhsLoginService.findCredentialsByNhsInfo(email, secretBase64, details)
                .map(credentials ->
                        Try.of(() -> userManager.authenticateNHSUser(credentials.getPrivateId(), credentials.getUserPrivateKey()))
                                .andThenTry(() -> grantAccessToTeam(details, credentials.getPrivateId()))
                                .onFailure(e -> LOGGER.error("Unexpected login failure", FrameFilter.filter(e)))
                                .getOrElseThrow(ex -> new BadCredentialsException("NHS login failed", ex)))
                .map(authInfo -> Tuple.of(authInfo, NOOP, NOOP))
                .mapLeft(error -> new NhsLoginAuthenticationException("NHS login failed", error, authentication));
    }

    @VisibleForTesting
    void grantAccessToTeam(NhsLoginAuthenticationDetails details, long personId) {
        details.getTeamCode().ifPresent(teamCode -> {
            var team = teamManager.getTeamByCode(teamCode);
            if (team != null) {
                var person = userManager.getPKBPerson(personId);

                if (!patientConsentManager.hasActiveConsentRecord(person, team)) {
                    nhsLoginRegistrationManager.grantAccessToTeam(details.getEHRRequestContext(), details.getRoute(),
                            person, team);
                }
            } else {
                LOGGER.error("Can't grant access person {} to team. Team not found by team code: {}", personId, teamCode);
            }
        });
    }
}
