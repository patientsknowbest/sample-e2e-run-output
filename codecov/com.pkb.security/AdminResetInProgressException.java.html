package com.pkb.security;

import org.springframework.security.core.AuthenticationException;

/**
 * Thrown if an authentication request fails because the institute admin has started
 * a credentials reset (which wipes out old password info). The user must follow the
 * reset process instead of trying to sign in.
 *
 * @author robwhelan
 */
public class AdminResetInProgressException extends AuthenticationException {
    private static final long serialVersionUID = 1L;

    /**
     * Constructs an <code>AdminResetInProgressException</code> with the
     * specified message.
     *
     * @param msg
     *            the detail message
     */
    public AdminResetInProgressException(String msg) {
        super(msg);
    }

    /**
     * Constructs an <code>AdminResetInProgressException</code> with the
     * specified message and root cause.
     *
     * @param msg
     *            the detail message
     * @param t
     *            root cause
     */
    public AdminResetInProgressException(String msg, Throwable t) {
        super(msg, t);
    }
}
