package com.pkb.security;

import com.google.common.collect.ImmutableMap;
import com.pkb.entities.enums.LoginAttemptLogResult;
import com.pkb.entities.enums.Route;
import com.pkb.entities.enums.SSOType;
import com.pkb.util.AppIdentityUtil;
import com.pkb.util.PKBConstants;
import com.pkb.util.SSOConstants;
import org.immutables.value.Value.Derived;
import org.immutables.value.Value.Immutable;

import javax.servlet.http.HttpServletRequest;
import java.util.Map;
import java.util.Optional;

import static com.pkb.entities.enums.LoginAttemptLogResult.SUCCESS_EMIS_SSO;
import static com.pkb.entities.enums.LoginAttemptLogResult.SUCCESS_SYSTMONE_SSO;

@Immutable
public interface SsoAuthenticationDetails extends BaseAuthenticationDetails {

    Map<SSOType, Route> SSO_TYPE_ROUTE_MAP = ImmutableMap.of(
            SSOType.EMIS, Route.WEBAPP_EMIS_PORTAL,
            SSOType.SYSTMONE, Route.WEBAPP_SYSTMONE_PORTAL);

    Map<SSOType, LoginAttemptLogResult> SSO_TYPE_LOGIN_ATTEMPT_SUCCESS_MAP = ImmutableMap.of(
            SSOType.EMIS, SUCCESS_EMIS_SSO,
            SSOType.SYSTMONE, SUCCESS_SYSTMONE_SSO);

    SSOType getSsoType();

    long getContextUserId();

    @Derived
    @Override
    default Route getRoute() {
        return SSO_TYPE_ROUTE_MAP.get(getSsoType());
    }

    @Override
    default LoginAttemptLogResult getLoginAttemptSuccessResult() {
        return SSO_TYPE_LOGIN_ATTEMPT_SUCCESS_MAP.get(getSsoType());
    }

    @SuppressWarnings("override")
    interface Builder extends BaseAuthenticationDetails.Builder<SsoAuthenticationDetails, Builder> {

        Builder ssoType(SSOType ssoType);

        Builder contextUserId(long contextUserId);

        @Override
        SsoAuthenticationDetails build();

        @Override
        default Builder populateFrom(HttpServletRequest request, AppIdentityUtil appIdentityUtil) {
            BaseAuthenticationDetails.Builder.super.populateFrom(request, appIdentityUtil);

            Optional.ofNullable(request.getParameter(SSOConstants.SSO_TYPE))
                    .map(SSOType::valueOf)
                    .ifPresent(this::ssoType);

            Optional.ofNullable(request.getParameter(PKBConstants.CONTEXT_USER_ID))
                    .map(Long::parseLong)
                    .ifPresent(this::contextUserId);

            return this;
        }
    }
}
