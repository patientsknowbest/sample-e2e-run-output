package com.pkb.security;

import static org.apache.commons.lang3.StringUtils.defaultIfEmpty;

import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.springframework.stereotype.Component;

import io.prometheus.client.Counter;

@Component
public class LoginMetrics {

    private static final Counter COUNTER = Counter.build()
            .name("pkb_phr_security_user_login_attempts")
            .help("count login attempts with statuses")
            .labelNames("status", "subresult")
            .register();

    private void inc(@NotNull String status, @Nullable String subresult) {
        COUNTER.labels(status, defaultIfEmpty(subresult, "none")).inc();
    }

    private void inc(@NotNull String status) {
        inc(status, null);
    }

    public void successfulLogin() {
        inc("successful");
    }

    public void unsuccessfulLogin(@Nullable String subresult) {
        inc("unsuccessful", subresult);
    }

    public void blockedLogin() {
        // blocked by too many login attempts
        inc("blocked");
    }
}
