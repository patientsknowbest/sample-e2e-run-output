package com.pkb.security;


import com.google.common.annotations.VisibleForTesting;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.PersonAuthInfo;
import com.pkb.entities.enums.LoginAttemptLogResult;
import com.pkb.entities.enums.SponsorshipStatus;
import com.pkb.entities.enums.UserStatus;
import com.pkb.entities.enums.UserType;
import com.pkb.exception.PKBException;
import com.pkb.institute.entity.InstituteUser;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import io.vavr.control.Try;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.security.authentication.BadCredentialsException;
import org.springframework.security.authentication.DisabledException;

import java.lang.invoke.MethodHandles;
import java.util.EnumMap;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.function.Consumer;

import static com.pkb.entities.enums.LoginAttemptLogResult.FAIL_BAD_PASSWORD;
import static com.pkb.entities.enums.LoginAttemptLogResult.FAIL_BAD_USERNAME;
import static com.pkb.entities.enums.LoginAttemptLogResult.FAIL_FROZEN;
import static com.pkb.entities.enums.LoginAttemptLogResult.FAIL_INACTIVE;
import static com.pkb.entities.enums.LoginAttemptLogResult.FAIL_NOT_YET_REG;
import static com.pkb.entities.enums.LoginAttemptLogResult.SUCCESS;
import static java.lang.String.format;

public class IdPasswordAuthenticator extends AbstractPrincipalCredentialAuthenticator {

    public static class Factory {

        private final TeamUserManager teamUserManager;

        private final UserManager userManager;


        public Factory(
                TeamUserManager teamUserManager,
                UserManager userManager) {
            this.teamUserManager = teamUserManager;
            this.userManager = userManager;
        }

        public IdPasswordAuthenticator createAuthenticator(LoginAttemptAuditor loginAttemptAuditor) {
            return new IdPasswordAuthenticator(teamUserManager, loginAttemptAuditor, userManager,
                    Optional.empty());
        }
    }

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private final UserManager userManager;


    private final Optional<BaseAuthenticationDetails> originalLoginDetails;

    private final Map<UserStatus, Consumer<PKBPerson>> userStatusMultiplexer = new EnumMap<>(UserStatus.class);

    {
        userStatusMultiplexer.put(UserStatus.DEAD, person -> {
            throw new WifiInTheCemeteryException("user is dead");
        });
        userStatusMultiplexer.put(UserStatus.CREATED, this::failNotRegYet);
        userStatusMultiplexer.put(UserStatus.NEW, this::failInactive);
        userStatusMultiplexer.put(UserStatus.INACTIVE, this::failInactive);
        userStatusMultiplexer.put(UserStatus.INVITED, this::failInactive);
        userStatusMultiplexer.put(UserStatus.NOCONTACT, this::failInactive);
    }

    public IdPasswordAuthenticator(
            TeamUserManager teamUserManager,
            @NotNull LoginAttemptAuditor loginAttemptAuditor,
            UserManager userManager,
            Optional<BaseAuthenticationDetails> originalLoginDetails) {
        super(teamUserManager, loginAttemptAuditor);
        this.userManager = userManager;
        this.originalLoginDetails = originalLoginDetails;
    }

    @Override
    protected PersonAuthInfo doAuthenticate(EHRRequestContext previousContext, String pkbId, String password) {

        Long personId = personIdAsLong(pkbId);
        personWithMatchingPrincipal = Try.of(() -> userManager.getPKBPerson(personId)).getOrElseThrow($ -> {
            loginAttemptAuditor.logAttempt(FAIL_BAD_USERNAME, null);
            throw new BadCredentialsException(format("No user found for personId %s. Failing authentication.", personId));
        });
        loginAttemptAuditor.checkForTooManyRecentLoginAttempts(personId);
        PersonAuthInfo authUser = authenticateUser(password, personId);
        LoggedInEHRRequestContext context = previousContext.afterUserLoggedIn(authUser);
        verifyUserStatus(personWithMatchingPrincipal);
        if (personWithMatchingPrincipal.isFrozenByTeam()) {
            loginAttemptAuditor.logAttempt(FAIL_FROZEN, personWithMatchingPrincipal.getId());
            throw new FrozenByTeamException("frozen by " + personWithMatchingPrincipal.getFrozenByTeamId());
        }
        if (UserType.INSTITUTE_ADMIN == personWithMatchingPrincipal.getUserType()) {
            checkIfTeamCoordIsActivated(personWithMatchingPrincipal);
        }
        handleSuccessfulLogin(authUser);
        return authUser;
    }

    private Long personIdAsLong(String pkbId) {
        Long personId;
        try {
            personId = Long.valueOf(pkbId);
        } catch (NumberFormatException ignored) {
            throw new InvalidPrincipalException(pkbId);
        }
        return personId;
    }

    private PersonAuthInfo authenticateUser(String password, Long personId) {
        try {
            return userManager.authenticateUser(personId, password);
        } catch (PKBException e) {
            loginAttemptAuditor.logAttempt(FAIL_BAD_PASSWORD, personId);
            throw new BadCredentialsException(e.getMessage(), e);
        }
    }

    private void checkIfTeamCoordIsActivated(PKBPerson teamCoord) {
        List<InstituteUser> teamCoordRecords = teamUserManager.getTeamCoordRecords(teamCoord.getId());
        if (!teamCoordRecords.isEmpty()
                && teamCoordRecords.get(0).getSponsorshipStatus() == SponsorshipStatus.INVITED) {
            loginAttemptAuditor.logAttempt(FAIL_INACTIVE, teamCoord.getId());
            throw new CoordAwaitingActivationException("New coord awaiting activation");
        }
    }

    private void verifyUserStatus(PKBPerson user) {
        Consumer<PKBPerson> failureHandler = userStatusMultiplexer.getOrDefault(user.getStatus(), person -> {
        });
        failureHandler.accept(user);
    }

    private void failNotRegYet(PKBPerson person) {
        loginAttemptAuditor.logAttempt(FAIL_NOT_YET_REG, person.getId());
        throw new RegIncompleteException("Created patient not yet registered");
    }

    private void failInactive(PKBPerson person) {
        loginAttemptAuditor.logAttempt(FAIL_INACTIVE, person.getId());
        throw new DisabledException("User is not activated");
    }

    @VisibleForTesting
    void handleSuccessfulLogin(PersonAuthInfo authUser) {
        LoginAttemptLogResult authenticationBasedResult = originalLoginDetails
                .map(BaseAuthenticationDetails::getLoginAttemptSuccessResult)
                .orElse(SUCCESS);
        loginAttemptAuditor.logAttempt(authenticationBasedResult, authUser.getId());

    }

}
