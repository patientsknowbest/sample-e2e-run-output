package com.pkb.security;

import com.pkb.datamodel.user.PersonIdsWithNhsLoginStatus;
import com.pkb.entities.enums.LoginAttemptLogResult;
import com.pkb.security.BaseAuthenticationProvider.AuditableLoginException;
import com.pkb.security.BaseAuthenticationProvider.TypedAuthentication;
import com.pkb.service.oauth2.NhsLoginError;
import org.springframework.security.core.AuthenticationException;

public class NhsLoginAuthenticationException extends AuthenticationException implements AuditableLoginException {

    private final NhsLoginError error;
    private final TypedAuthentication<NhsLoginAuthenticationDetails> authentication;

    NhsLoginAuthenticationException(String msg, NhsLoginError error, TypedAuthentication<NhsLoginAuthenticationDetails> authentication) {
        super(msg);
        this.error = error;
        this.authentication = authentication;
    }

    public NhsLoginError getError() {
        return error;
    }

    public TypedAuthentication<NhsLoginAuthenticationDetails> getAuthentication() {
        return authentication;
    }

    @Override
    public AuthenticationException audit(LoginAttemptAuditor auditor) {
        LoginAttemptLogResult result = authentication.getDetails().getLoginAttemptFailResult();
        String subresult = error.getErrorCode();
        Long personId = error.getPersonIds().map(PersonIdsWithNhsLoginStatus::getId).orElse(error.getPersonId().orElse(null));
        auditor.logAttempt(result, personId, null, subresult);
        return this;
    }
}
