package com.pkb.security;

import com.pkb.security.BaseAuthenticationProvider.TypedAuthentication;
import io.vavr.control.Option;
import org.springframework.security.authentication.AuthenticationDetailsSource;

import javax.servlet.http.HttpServletRequest;

/**
 * Extracts the previous authentication object from the session and supplies a new {@link PKBAuthenticationDetails}
 * with that authentication object set in {@link PKBAuthenticationDetails#originalAuthentication}.
 * <p>
 * Only used for the NHS login authentication flow therefore this class only extracts NHS Login related
 * authentication objects yet.
 */
public class OriginalAuthenticationDetailsSource
        implements AuthenticationDetailsSource<HttpServletRequest, PKBAuthenticationDetails> {

    @Override
    public PKBAuthenticationDetails buildDetails(HttpServletRequest request) {
        return Option.of(request)
                .map(httpServletRequest -> httpServletRequest.getSession().getAttribute("nhs.login.info"))
                .filter(TypedAuthentication.class::isInstance)
                .map(TypedAuthentication.class::cast)
                .map(typedAuthentication -> ImmutablePKBAuthenticationDetails.builder()
                        .remoteAddress(request.getRemoteAddr())
                        .originalAuthentication(typedAuthentication)
                        .build())
                .getOrElse(() -> ImmutablePKBAuthenticationDetails.builder()
                        .remoteAddress(request.getRemoteAddr()).build());
    }
}
