package com.pkb.security;

import com.pkb.security.BaseAuthenticationProvider.TypedAuthentication;
import org.immutables.value.Value;

import java.io.Serializable;
import java.util.Optional;


/**
 * Contains additional details about a logged in user.
 */
@Value.Immutable
@Value.Style(optionalAcceptNullable = true)
public interface PKBAuthenticationDetails extends Serializable {

    Optional<Long> userId();

    Optional<String> remoteAddress();

    @Value.Default
    default boolean twoFactorAuthDone() {
        return false;
    }

    @Value.Default
    default boolean twoFactorAuthMandatory() {
        return false;
    }

    Optional<ApiAuthInfo> apiAuthInfo();

    Optional<TypedAuthentication<BaseAuthenticationDetails>> originalAuthentication();

    default boolean isOauthTwoFactorAuth() {
        return apiAuthInfo().isPresent();
    }

    static ImmutablePKBAuthenticationDetails.Builder initializeIfSame(Object input) {
        if (PKBAuthenticationDetails.class.isInstance(input)) {
            return ImmutablePKBAuthenticationDetails.builder().from((PKBAuthenticationDetails) input);
        }
        return ImmutablePKBAuthenticationDetails.builder();
    }

    static boolean isTwoFactorAuthPreviouslyDone(Object authDetails) {
        boolean result = false;
        if (PKBAuthenticationDetails.class.isInstance(authDetails)) {
            result = ((PKBAuthenticationDetails) authDetails).twoFactorAuthDone();
        }
        return result;
    }
}
