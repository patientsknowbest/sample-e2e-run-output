/**
 * Spring Security Authentication Manger
 *
 * @author mahendera
 * @author robwhelan
 * @author pravinam Changes for supporting login with any confirmed email/contact
 */
public class PKBAuthenticationManager implements AuthenticationManager {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
    private static final GrantedAuthority TWO_FACTOR_ROLE = new SimpleGrantedAuthority(ROLE_FOR_TWO_FACTOR_AUTH);

    private final UserManager userManager;

    private PersonContactManager personContactManager;
    private ApiAuthManager apiAuthManager;
    private TwoFactorAuthenticationService twoFactorAuthenticationService;

    private LoginAttemptAuditor.Factory loginAttemptLoggerFactory;

    private IdPasswordAuthenticator.Factory idPasswordAuthenticatorFactory;

    private EmailPasswordAuthenticator.Factory emailPasswordAuthenticatorFactory;

    public PKBAuthenticationManager(
            PersonContactManager personContactManager,
            ApiAuthManager apiAuthManager,
            UserManager userManager,
            TwoFactorAuthenticationService twoFactorAuthenticationService,
            LoginAttemptAuditor.Factory loginAttemptLoggerFactory,
            IdPasswordAuthenticator.Factory idPasswordAuthenticatorFactory,
            EmailPasswordAuthenticator.Factory emailPasswordAuthenticatorFactory) {
        this.personContactManager = personContactManager;
        this.apiAuthManager = apiAuthManager;
        this.userManager = userManager;
        this.twoFactorAuthenticationService = twoFactorAuthenticationService;
        this.loginAttemptLoggerFactory = loginAttemptLoggerFactory;
        this.idPasswordAuthenticatorFactory = idPasswordAuthenticatorFactory;
        this.emailPasswordAuthenticatorFactory = emailPasswordAuthenticatorFactory;
    }

    private String getUserName(Authentication authentication) {
        Object principal = authentication.getPrincipal();
        if (principal instanceof Email) {
            return ((Email) principal).address();
        } else {
            return ((String) principal).toLowerCase();
        }
    }

    @Override
    public Authentication authenticate(Authentication authentication)
            throws AuthenticationException {
        LOGGER.trace("Entering authenticate ");

        Authentication userNameAuthentication = null;
        String password = (String) authentication.getCredentials();
        String userName = getUserName(authentication);
        Long userId = null;
        Object authDetails = authentication.getDetails();
        String ipAddress = getRemoteAddressFromDetails(authDetails);
        Optional<TypedAuthentication<BaseAuthenticationDetails>> originalAuthentication = Optional.empty();


        // For now, we detect an OTP (rather than username/password
        // combination) by the absence of an @ in the username field.
        // TODO: Do something better. Spring 3.1 offers
        // authentication-details-source-ref, which allows extra
        // information to be included in the authentication details;
        // perhaps we could have a j_otp value?
        boolean isUsingOtp = !userName.contains("@");
        if (isUsingOtp) {
            String otpToken = (String) authentication.getPrincipal();

            try {
                WebAppOtp otp = apiAuthManager.retrieveAndExpireWebappOTP(otpToken);

                userName = "";
                password = otp.getPassword().orElse("");

                // Retrieve primary email to use for login
                userId = Long.parseLong(otp.getUserId());
                List<PersonContact> pcs = personContactManager.getContactListDirect(userId);
                for (PersonContact pc : pcs) {
                    userName = pc.getContactIfPrimaryEmail().map(Email::address).getOrElse(userName);
                }
            } catch (Exception ignored) {
                throw new InvalidTokenException("No ACTIVE tokens found");
            }
        }

        LoginAttemptAuditor loginAttemptAuditor = loginAttemptLoggerFactory.createAuditor(ipAddress);
        PrincipalCredentialAuthenticator authenticator;
        String principal;
        if (PKBAuthenticationDetails.class.isInstance(authDetails) && ((PKBAuthenticationDetails) authDetails).userId().isPresent()) {
            principal = Long.toString(((PKBAuthenticationDetails) authDetails).userId().get());
            authenticator = idPasswordAuthenticatorFactory.createAuthenticator(loginAttemptAuditor);
        } else if (isUsingOtp) {
            //we must ensure that we're using the same ID as generated the OTP:
            principal = Long.toString(userId);
            authenticator = idPasswordAuthenticatorFactory.createAuthenticator(loginAttemptAuditor);
        } else {
            originalAuthentication = getOriginalAuthentication(authDetails);
            principal = userName;
            authenticator = emailPasswordAuthenticatorFactory.createAuthenticator(loginAttemptAuditor,
                    originalAuthentication.map(TypedAuthentication::getDetails));
        }

        try {
            EHRRequestContext context = ImmutableLoggedOutEHRRequestContext.of(AccountLinkType.INDIVIDUAL, Route.WEBAPP, AbstractConsentStatus.noConsentRequired(), UUID.randomUUID());
            PersonAuthInfo loggedInUser = authenticator.authenticate(context, principal, password);
            PKBPerson loggedInPerson = userManager.getPKBPerson(loggedInUser.getId());
            boolean twoFactorAuthMandatoryForLoggedInPerson = twoFactorAuthenticationService.isTwoFactorAuthMandatory(loggedInPerson);
            userNameAuthentication = getAuthToken(authDetails, ipAddress, userId, twoFactorAuthMandatoryForLoggedInPerson, Optional.empty(), originalAuthentication, loggedInUser, password, loggedInPerson);
        } catch (InvalidPrincipalException e) {
            throw new BadCredentialsException("invalid credentials", e);
        }

        LOGGER.trace("Exiting authenticate ");

        return userNameAuthentication;
    }

    public UsernamePasswordAuthenticationToken getAuthToken(Object authDetails, String ipAddress, Long userId, boolean twoFactorAuthMandatory, Optional<ApiAuthInfo> maybeApiAuthInfo, Optional<TypedAuthentication<BaseAuthenticationDetails>> originalAuthentication, PersonAuthInfo loggedInUser, String password, PKBPerson loggedInPerson) {
        PKBAuthenticationDetails newAuthDetails = PKBAuthenticationDetails.initializeIfSame(authDetails)
                .remoteAddress(ipAddress)
                .userId(userId)
                .twoFactorAuthMandatory(twoFactorAuthMandatory)
                .originalAuthentication(originalAuthentication)
                .apiAuthInfo(maybeApiAuthInfo)
                .build();
        UsernamePasswordAuthenticationToken token = new UsernamePasswordAuthenticationToken(
                loggedInUser, password, getGrantedAuthorities(loggedInPerson, newAuthDetails));
        token.setDetails(newAuthDetails);
        return token;
    }

    private Optional<TypedAuthentication<BaseAuthenticationDetails>> getOriginalAuthentication(Object authDetails) {
        return Optional.ofNullable(authDetails)
                .filter(PKBAuthenticationDetails.class::isInstance)
                .map(PKBAuthenticationDetails.class::cast)
                .flatMap(PKBAuthenticationDetails::originalAuthentication);
    }

    private Collection<? extends GrantedAuthority> getGrantedAuthorities(PKBPerson loggedInPerson, Object authDetails) {
        if (twoFactorAuthenticationService.isTwoFactorCheckRequired(loggedInPerson) && !isTwoFactorAuthPreviouslyDone(authDetails)) {
            ArrayList<GrantedAuthority> result = new ArrayList<>();
            result.add(TWO_FACTOR_ROLE);
            return result;
        } else {
            return UserTypeToGrantedAuthorityMapper.getAuthorityList(loggedInPerson.getUserType());
        }
    }

    private String getRemoteAddressFromDetails(Object authDetails) {
        String result = EMPTY;
        if (WebAuthenticationDetails.class.isInstance(authDetails)) {
            result = ((WebAuthenticationDetails) authDetails).getRemoteAddress();
        } else if (PKBAuthenticationDetails.class.isInstance(authDetails)) {
            result = ((PKBAuthenticationDetails) authDetails).remoteAddress().orElse(EMPTY);
        }
        return result;
    }
}
