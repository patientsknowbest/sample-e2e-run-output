package com.pkb.security;

import static com.google.common.base.Preconditions.checkNotNull;
import static com.pkb.entities.enums.LoginAttemptLogResult.FAIL_BLOCKED;
import static com.pkb.entities.enums.LoginAttemptLogResult.SUCCESS_RESULTS;

import java.lang.invoke.MethodHandles;
import java.time.LocalDateTime;

import javax.annotation.concurrent.NotThreadSafe;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.domain.LoginAttemptLogService;
import com.pkb.entities.enums.LoginAttemptLogResult;

@NotThreadSafe
public class LoginAttemptAuditor {

    public static class Factory {

        private final LoginMetrics loginMetrics;

        private final PhrConfig config;

        private final DateTimeService dateTimeService;

        private final LoginAttemptLogService loginAttemptLogService;

        public Factory(LoginMetrics loginMetrics, PhrConfig config,
                       DateTimeService dateTimeService, LoginAttemptLogService loginAttemptLogService) {
            this.loginMetrics = loginMetrics;
            this.config = config;
            this.dateTimeService = dateTimeService;
            this.loginAttemptLogService = loginAttemptLogService;
        }

        public LoginAttemptAuditor createAuditor(String remoteIpAddress) {
            return new LoginAttemptAuditor(remoteIpAddress, loginMetrics, config, dateTimeService, loginAttemptLogService);
        }

    }

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private final LoginMetrics loginMetrics;

    private final PhrConfig config;

    private final DateTimeService dateTimeService;

    private final LoginAttemptLogService loginAttemptLogService;

    private boolean logEntrySaved = false;

    private String ipAddress;

    public LoginAttemptAuditor(String ipAddress, LoginMetrics loginMetrics, PhrConfig config,
                               DateTimeService dateTimeService, LoginAttemptLogService loginAttemptLogService) {
        this.ipAddress = checkNotNull(ipAddress, "ipAddress cannot be null");
        this.loginMetrics = loginMetrics;
        this.config = config;
        this.dateTimeService = dateTimeService;
        this.loginAttemptLogService = loginAttemptLogService;
    }

    public void logAttempt(LoginAttemptLogResult result, Long personId, String badUsername, String subresult) {
        if (!logEntrySaved) {
            logEntrySaved = true;
            loginAttemptLogService.createLoginAttemptLog(personId, result, ipAddress, badUsername, subresult);

            if (SUCCESS_RESULTS.contains(result)) {
                loginMetrics.successfulLogin();
            } else if (result == FAIL_BLOCKED) {
                loginMetrics.blockedLogin();
            } else {
                loginMetrics.unsuccessfulLogin(subresult);
            }
        }
    }

    public void logAttempt(LoginAttemptLogResult result, Long personId, String badUsername) {
        logAttempt(result, personId, badUsername, null);
    }

    public void logAttempt(LoginAttemptLogResult result, Long personId) {
        logAttempt(result, personId, null);
    }

    public void checkForTooManyRecentLoginAttempts(Long personId) {
        LocalDateTime recentTime = dateTimeService.nowLocalDateTime().minusSeconds(config.getDefinitionOfRecentLoginAttemptInSeconds());
        if (loginAttemptLogService.findCountOfRecentLoginAttempts(personId, recentTime) >= config.getNumberOfRecentLoginAttemptsAllowed()) {
            LOGGER.warn("user with person id {} has too many login attempts", personId);
            this.logAttempt(FAIL_BLOCKED, personId);
            throw new TooManyLoginAttemptsException(personId);
        }
    }
}
