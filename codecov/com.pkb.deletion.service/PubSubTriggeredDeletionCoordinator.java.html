package com.pkb.deletion.service;

import com.pkb.deletion.model.DeleteOperationRequest;
import com.pkb.deletion.model.DeletionType;
import com.pkb.deletion.model.ImmutableDeleteOperationResponse;
import com.pkb.deletion.model.handler.DeletionHandler;
import io.micrometer.core.instrument.MeterRegistry;
import io.vavr.control.Option;
import org.apache.camel.Consume;
import org.apache.camel.ProducerTemplate;
import org.slf4j.Logger;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.Map;
import java.util.function.Function;
import java.util.stream.Collectors;

import static com.pkb.deletion.routing.DeletionServiceRouting.OPERATION_REQUEST;
import static com.pkb.deletion.routing.DeletionServiceRouting.OPERATION_RESPONSE;
import static java.lang.invoke.MethodHandles.lookup;
import static org.slf4j.LoggerFactory.getLogger;

/**
 * Pub-sub/camel adapter for the deletion service. Handles the plumbing of responding to the pub-sub
 * side of things and then delegates specific deletion requests to registered DeletionHandler objects.
 */
@Service
public class PubSubTriggeredDeletionCoordinator {

    private static final Logger LOGGER = getLogger(lookup().lookupClass());


    private final Map<DeletionType, DeletionHandler> deletionHandlers;


    private final ProducerTemplate pubSubProducer;
    private final String serviceName;
    private final MeterRegistry meterRegistry;

    public PubSubTriggeredDeletionCoordinator(
            List<DeletionHandler> handlers,
            @Qualifier("producerTemplate") ProducerTemplate pubSubProducer,
            @Value("${deletion.service.serviceName}") String serviceName,
            MeterRegistry meterRegistry
    ) {
        deletionHandlers = handlers.stream().collect(Collectors.toMap(DeletionHandler::getHandledType, Function.identity()));
        this.pubSubProducer = pubSubProducer;
        this.serviceName = serviceName;
        this.meterRegistry = meterRegistry;
    }


    @Consume(OPERATION_REQUEST)
    public void runOperation(DeleteOperationRequest request) {
        LOGGER.info("Request received: {}", request);
        getHandler(request).peek(handler -> {
            var responseBuilder = ImmutableDeleteOperationResponse.builder()
                    .requestId(request.getRequestId())
                    .actor(request.getActor())
                    .targetType(request.getTargetType())
                    .aggregatableRequestId(request.getAggregatableRequestId())
                    .respondingService(serviceName)
                    .operationType(request.getOperationType());

            try {
                var result = handler.handleRequest(request);
                var response = responseBuilder.deletionStatus(result.getStatus())
                        .error(result.getError())
                        .build();



                LOGGER.info("Sending response: {}", response);


                pubSubProducer.sendBody(OPERATION_RESPONSE, response);
                meterRegistry.counter("deletion.service.operationResult", "serviceName", serviceName, "operationType", request.getOperationType().name(), "result", response.getDeletionStatus().name()).increment();
            }
            catch(Exception e) {
                LOGGER.error("Error during delete operation processing: ", e);
                meterRegistry.counter("deletion.service.operationResult", "serviceName", serviceName, "operationType", request.getOperationType().name(), "result", "EXCEPTION").increment();
                throw e;
            }
        }).onEmpty(() -> LOGGER.info("Ignoring deletion request {} because no handler is defined for it", request.getRequestId()));


    }


    private Option<DeletionHandler> getHandler(DeleteOperationRequest request) {
        return Option.of(deletionHandlers.get(request.getTargetType()));
    }

}
