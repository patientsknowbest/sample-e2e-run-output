package com.pkb.deletion.service;

import com.pkb.deletion.model.DeleteOperationRequest;
import com.pkb.deletion.model.DeleteRequestStatus;
import com.pkb.deletion.model.DeletionRules;
import com.pkb.deletion.model.DeletionType;
import com.pkb.deletion.model.EntityDeletionRequest;
import com.pkb.deletion.model.errors.DeletionServiceError;
import com.pkb.deletion.model.handler.DeleteOperationResult;
import com.pkb.deletion.model.handler.DeletionHandler;
import com.pkb.deletion.model.handler.ImmutableDeleteOperationResult;
import io.vavr.control.Either;

/**
 * Base implementation of the DeletionHandler interface that makes use of a {@link DatabaseModelBasedDeletionService}
 * to actually do the deletion work. Subclasses just need to provide a method that returns a rules object for a given
 * request, and a method to convert the original {@link CreateDeleteOperationRequest} into an {@link EntityDeletionRequest}
 * This latter indirection is important because in some cases the logical deletion target, such as a Person, translates
 * to a different root entity for the database model deletion (an Account, in the case of a Person).
 * @param <T> The primary key type of the root entity to be deleted by the DatabaseModelBasedDeletionService
 */
public abstract class AbstractDatabaseModelDeletionHandler<T> implements DeletionHandler {

    private final DatabaseModelBasedDeletionService databaseModelBasedDeletionService;
    private final DeletionType type;

    @SuppressWarnings("WeakerAccess")
    protected AbstractDatabaseModelDeletionHandler(DatabaseModelBasedDeletionService databaseModelBasedDeletionService, DeletionType type) {
        this.databaseModelBasedDeletionService = databaseModelBasedDeletionService;
        this.type = type;
    }

    @Override
    public final DeletionType getHandledType() {
        return type;
    }

    @Override
    public final DeleteOperationResult handleRequest(DeleteOperationRequest request) {
        switch(request.getOperationType()) {
            case CREATE:
                return createAndValidateDeleteOperation(request);
            case EXECUTE:
                return executeDeleteOperation(request);
            default:
                return rollbackDeleteOperation(request);
        }
    }


    private DeleteOperationResult createAndValidateDeleteOperation(DeleteOperationRequest request) {
        return getEntityDeletionRequest(request)
                .map(entityDeletionRequest -> {
                    DeletionRules rules = getDeletionRules(entityDeletionRequest);
                    return databaseModelBasedDeletionService.createAndValidateDeleteOperation(entityDeletionRequest, rules);
                }).getOrElse(ImmutableDeleteOperationResult.builder().status(DeleteRequestStatus.INVALID).build());
    }

    protected abstract DeletionRules getDeletionRules(EntityDeletionRequest<T> entityDeletionRequest);

    protected abstract Either<DeletionServiceError, EntityDeletionRequest<T>> getEntityDeletionRequest(DeleteOperationRequest originalRequest);

    private DeleteOperationResult executeDeleteOperation(DeleteOperationRequest request) {
        return databaseModelBasedDeletionService.executeDeleteOperation(request.getRequestId(), request.getActor());
    }

    private DeleteOperationResult rollbackDeleteOperation(DeleteOperationRequest request) {
        return databaseModelBasedDeletionService.rollbackDeletion(request.getRequestId(), request.getActor());
    }
}
