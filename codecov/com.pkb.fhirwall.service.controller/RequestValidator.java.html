package com.pkb.fhirwall.service.controller;

import ca.uhn.fhir.context.FhirContext;
import org.hl7.fhir.instance.model.api.IBaseResource;
import org.slf4j.Logger;

import java.util.Collections;
import java.util.Optional;
import java.util.Set;
import java.util.regex.Pattern;

import static java.lang.String.join;
import static java.lang.invoke.MethodHandles.lookup;
import static java.util.regex.Pattern.compile;
import static org.apache.commons.lang3.StringUtils.isBlank;
import static org.apache.commons.lang3.StringUtils.isNotBlank;
import static org.slf4j.LoggerFactory.getLogger;

public class RequestValidator {

    private static final Logger LOG = getLogger(lookup().lookupClass());

    private final Pattern resourcePattern;
    private final FhirContext fhirContext;
    private final Set<String> supportedResources;
    private final Set<String> unsupportedParameters;

    public RequestValidator(Set<String> supportedResources, Set<String> unsupportedParameters, FhirContext fhirContext) {
        this.supportedResources = supportedResources;
        this.unsupportedParameters = unsupportedParameters;
        this.fhirContext = fhirContext;
        this.resourcePattern = compile("^\\/fhir\\/("
                + join("|", supportedResources)
                + ")($|\\/.*)");
    }

    public boolean isResourceSupported(String requestUri) {
        if (isBlank(requestUri)) {
            return false;
        }

        return resourcePattern.matcher(requestUri).matches();
    }

    public boolean isRead(String requestUri) {
        if (isBlank(requestUri)) {
            return false;
        }

        var parts = requestUri.split("/");
        switch (parts.length) {
            case 6:
                return "fhir".equals(parts[1])
                        && supportedResources.contains(parts[2])
                        && isNotBlank(parts[3])
                        && "_history".equals(parts[4])
                        && isNotBlank(parts[5]);
            case 4:
                return "fhir".equals(parts[1])
                        && supportedResources.contains(parts[2])
                        && isNotBlank(parts[3]);
            default:
                return false;
        }
    }

    public Optional<IBaseResource> parseBody(Optional<String> maybeBody) {
        if (maybeBody.isEmpty()) {
            return Optional.empty();
        }

        try {
            var parser = fhirContext.newJsonParser();
            return Optional.of(parser.parseResource(maybeBody.get()));
        } catch (Exception exception) {
            LOG.info("Can't parse body: {}", maybeBody.get(), exception);
        }

        return Optional.empty();
    }

    // FHIR API provides methods for users to obtain linked resources in one request, e.g. 'give me patients and all
    // their medications and conditions'. We aren't implementing authorization to support this convenience feature, so
    // we filter any request that potentially does this.
    public boolean isSupportedSearchQuery(Set<String> parameters) {
        return Collections.disjoint(unsupportedParameters, parameters);
    }
}
