package com.pkb.fhirwall.service.controller;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.fhirwall.config.FhirwallConfig;
import io.jsonwebtoken.Claims;
import io.jsonwebtoken.ExpiredJwtException;
import io.jsonwebtoken.Jws;
import io.jsonwebtoken.JwtException;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.security.SignatureException;
import io.vavr.control.Either;
import org.slf4j.Logger;

import javax.servlet.http.HttpServletRequest;
import java.security.PublicKey;
import java.util.Date;
import java.util.List;
import java.util.function.Supplier;
import java.util.regex.Pattern;

import static io.jsonwebtoken.SignatureAlgorithm.RS512;
import static java.lang.invoke.MethodHandles.lookup;
import static org.apache.commons.lang3.StringUtils.isBlank;
import static org.slf4j.LoggerFactory.getLogger;
import static org.springframework.http.HttpHeaders.AUTHORIZATION;


public abstract class TokenValidator {

    private static final Logger LOG = getLogger(lookup().lookupClass());

    private static final Pattern PIPE_PATTERN = Pattern.compile("\\|");

    private final DateTimeService dateTimeService;
    private final Supplier<PublicKey> publicKeySupplier;
    private final FhirwallConfig fhirwallConfig;

    public TokenValidator(DateTimeService dateTimeService, Supplier<PublicKey> publicKeySupplier, FhirwallConfig fhirwallConfig) {
        this.dateTimeService = dateTimeService;
        this.publicKeySupplier = publicKeySupplier;
        this.fhirwallConfig = fhirwallConfig;
    }

    abstract String getIdentifierParameter(HttpServletRequest request);

    abstract List<String> getPermittedIdentifierSystems(HttpServletRequest request);

    abstract String getActorIdentifier();

    public Either<String, String> validate(HttpServletRequest request) {
        var authHeader = request.getHeader(AUTHORIZATION);
        if (isBlank(authHeader)) {
            return Either.left("Missing Authorization header");
        }
        var identifierParameter = getIdentifierParameter(request);
        if (isBlank(identifierParameter)) {
            return Either.left("Mandatory query parameter is missing: " + getActorIdentifier());
        }
        var parts = PIPE_PATTERN.split(identifierParameter);
        if (parts.length != 2) {
            return Either.left("Identifier is in wrong format! Should be: <system>|<identifier>");
        }
        if (!getPermittedIdentifierSystems(request).contains(parts[0])) {
            return Either.left("Identifier system type is not supported!");
        }
        var identifier = parts[1];
        var token = authHeader.substring(7);
        try {
            Jws<Claims> jws = Jwts.parserBuilder()
                    .setSigningKey(publicKeySupplier.get())
                    .setClock(() -> Date.from(dateTimeService.now()))
                    .build()
                    .parseClaimsJws(token);

            if (!RS512.getValue().equals(jws.getHeader().getAlgorithm())) {
                return Either.left("Token algorithm mismatch");
            }
            if (!fhirwallConfig.getNhsTokenIss().equals(jws.getBody().getIssuer())) {
                return Either.left("Token issuer mismatch");
            }
            if (!"P9".equals(jws.getBody().get("identity_proofing_level", String.class))) {
                return Either.left("Identity proofing level mismatch");
            }
            if (!identifier.equals(jws.getBody().get("nhs_number", String.class))) {
                return Either.left("NHS number mismatch");
            }
            // Temporary save of the nhs number until we refactor token parsing
            request.setAttribute("NHS_NUMBER", identifier);
        } catch (ExpiredJwtException expiredJwtException) {
            LOG.debug("Token is expired!", expiredJwtException);
            return Either.left("Token is expired");
        } catch (SignatureException signatureException) {
            LOG.debug("Token signature mismatch!", signatureException);
            return Either.left("Token signature mismatch");
        } catch (JwtException exception) {
            LOG.error("Can't parse ID token!", exception);
            return Either.left("Error while parsing token");
        }

        return Either.right(identifier);
    }

}
