package com.pkb.api.fhir.organization;

import com.pkb.api.fhir.foundation.ReferenceMapper;
import com.pkb.datamodel.team.TeamWithOrg;
import org.hl7.fhir.dstu3.model.Organization;
import org.jetbrains.annotations.NotNull;

public class OrganizationMapper {

    private ReferenceMapper referenceMapper;

    public OrganizationMapper(ReferenceMapper referenceMapper) {
        this.referenceMapper = referenceMapper;
    }

    public @NotNull Organization toOrganization(@NotNull TeamWithOrg team) {
        Organization organization = new Organization();
        organization.setId(team.getPublicId().toString());
        organization.setName(team.getName());
        organization.setPartOf(referenceMapper.toReferenceWithName(team.getOrg()));
        return organization;
    }
}
