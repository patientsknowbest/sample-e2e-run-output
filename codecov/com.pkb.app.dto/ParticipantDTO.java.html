package com.pkb.app.dto;

import com.pkb.annotation.EHRField;
import com.pkb.user.entity.PKBPerson;

import java.io.Serializable;

/**
 * Participant in an encounter or a procedure or basically in any event
 *
 * @author pravina
 *
 */
public class ParticipantDTO implements Serializable {
    private static final long serialVersionUID = 1L;

    //majority of enums have been taken from http://www.hl7.org/implement/standards/fhir/valueset-encounter-participant-type.html
    public enum ParticipantRole {
        CONSULTANT, ADMITTER, ATTENDER, REFERRER, DISCHARGER, CARER, // PKB role
        PATIENT// PKB Role 
    }

    @EHRField
    private String nonpkbParticipant;

    @EHRField
    private String personId;

    @EHRField
    private ParticipantRole role;

    // required by client layers (PHR Web , MiApp)
    private PKBPerson pkbPerson; // details of the person as it exists in PKB system

    public String getNonpkbParticipant() {
        return nonpkbParticipant;
    }

    public void setNonpkbParticipant(String nonpkbParticipant) {
        this.nonpkbParticipant = nonpkbParticipant;
    }

    public String getPersonId() {
        return personId;
    }

    public void setPersonId(String pkbpersonId) {
        this.personId = pkbpersonId;
    }

    public ParticipantRole getRole() {
        return role;
    }

    public void setRole(ParticipantRole role) {
        this.role = role;
    }

    public PKBPerson getPkbPerson() {
        return pkbPerson;
    }

    public void setPkbPerson(PKBPerson pkbPerson) {
        this.pkbPerson = pkbPerson;
    }

    public static class ParticipantDTOBuilder {
        private String nonPkbParticipant;
        private String personId;
        private PKBPerson pkbPerson;
        private ParticipantRole role;

        private ParticipantDTOBuilder() {
        }

        public static ParticipantDTOBuilder forNonPkbParticipant(String nonPkbParticipant) {
            ParticipantDTOBuilder builder = new ParticipantDTOBuilder();
            builder.nonPkbParticipant = nonPkbParticipant;
            return builder;
        }

        public static ParticipantDTOBuilder forPersonId(String personId) {
            ParticipantDTOBuilder builder = new ParticipantDTOBuilder();
            builder.personId = personId;
            return builder;
        }

        public static ParticipantDTOBuilder forPerson(PKBPerson pkbPerson) {
            ParticipantDTOBuilder builder = new ParticipantDTOBuilder();
            builder.pkbPerson = pkbPerson;
            return builder;
        }

        public ParticipantDTOBuilder withRole(ParticipantRole role) {
            this.role = role;
            return this;
        }

        public ParticipantDTOBuilder withPersonId(String personId) {
            this.personId = personId;
            return this;
        }

        public ParticipantDTOBuilder withPerson(PKBPerson pkbPerson) {
            this.pkbPerson = pkbPerson;
            return this;
        }

        public ParticipantDTO get() {
            ParticipantDTO dto = new ParticipantDTO();
            dto.setNonpkbParticipant(this.nonPkbParticipant);
            dto.setPersonId(this.personId);
            dto.setPkbPerson(this.pkbPerson);
            dto.setRole(this.role);
            return dto;
        }
    }
}
