package com.pkb.app.dto;

import javax.persistence.Column;
import javax.persistence.Embeddable;
import java.io.Serializable;
import java.util.Date;

/**
 * Wraps a startDate - endDate pair.
 *
 * @author erosb
 */
@Embeddable
public class Period implements Serializable {
    private static final long serialVersionUID = 689173363809035982L;

    private static Date copy(Date original) {
        if (original == null) {
            return null;
        }
        return new Date(original.getTime());
    }

    @Column(nullable = true)
    private Date startDate;

    @Column(nullable = true)
    private Date endDate;

    public Period(Date startDate, Date endDate) {
        this.startDate = copy(startDate);
        this.endDate = copy(endDate);
        validate();
    }

    private void validate() {
        if ((startDate != null) && (endDate != null) && (startDate.compareTo(endDate) > 0)) {
            throw new IllegalArgumentException("startDate cannot be lower than endDate");
        }
    }

    public Period() {

    }

    public Date getStartDate() {
        return startDate;
    }

    public void setStartDate(Date startDate) {
        this.startDate = copy(startDate);
        validate();
    }

    public Date getEndDate() {
        return endDate;
    }

    public void setEndDate(Date endDate) {
        this.endDate = copy(endDate);
        validate();
    }

}
