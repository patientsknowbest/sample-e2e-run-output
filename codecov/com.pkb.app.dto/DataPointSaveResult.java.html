package com.pkb.app.dto;

import com.pkb.app.interfaces.EncryptedDataPoint;
import org.immutables.value.Value;

import java.io.Serializable;
import java.util.List;
import java.util.Set;
import java.util.stream.Collectors;

import static java.util.Collections.emptySet;
import static org.apache.commons.collections4.CollectionUtils.isEmpty;

/**
 * There's other info we'll probably put in here -- but it's not clear what yet, exactly; wait to implement
 * @param <T>
 */
@Value.Immutable
public interface DataPointSaveResult<T> extends Serializable {

    List<T> getExcludedDataPoints();

    default Set<Long> persistedDataPointIds() {
        return isEmpty(persistedDataPoints()) ? emptySet() : persistedDataPoints().stream().map(t -> (EncryptedDataPoint) t).map(EncryptedDataPoint::getId).collect(Collectors.toSet());
    }

    List<T> persistedDataPoints();
}
