package com.pkb.app.dto;

import com.fasterxml.jackson.annotation.JsonIgnore;
import com.fasterxml.jackson.annotation.JsonProperty;
import com.pkb.app.PrivacyFlagsToPrivacyFlagConverter;
import com.pkb.app.entity.EHRData;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.privacy.entity.PrivacyFlags;
import org.apache.commons.lang3.NotImplementedException;
import org.jetbrains.annotations.NotNull;

import java.io.Serializable;
import java.time.Instant;
import java.util.Date;
import java.util.Optional;
import java.util.Set;
import java.util.UUID;

/**
 * Base DTO fields
 *
 * @author pravina
 *
 */
public class DTOBaseFields implements Serializable {

    private static final long serialVersionUID = 7010348832162985646L;

    public static final String ENTERED_DATE = "enteredDate";
    public static final String PERSISTED_DATE = "persistedDate";
    public static final String UPLOADED_DATA_ID = "uploadedDataId";
    public static final String GENERAL_HEALTH_CONSENT_REQUIRED = "generalHealthConsentRequired";
    public static final String MENTAL_HEALTH_CONSENT_REQUIRED = "mentalHealthConsentRequired";
    public static final String SEXUAL_HEALTH_CONSENT_REQUIRED = "sexualHealthConsentRequired";
    public static final String SOCIAL_CARE_CONSENT_REQUIRED = "socialCareConsentRequired";
    public static final String UNIQUE_ID = "uniqueId";
    public static final String HOSPITAL_SERVICE_CODE = "hospitalServiceCode";

    @JsonProperty("id")
    private UUID uniqueId;

    /** captures when the data was created/edited. E.g., when a lab result was recorded (not necessarily the test was run), or when a correction was entered */
    private Date enteredDate;

    /** set afresh in EHRBean when the data point is saved to the DB */
    private Date persistedDate;

    public DTOBaseFields() {
        deleted = false;
    }

    @JsonIgnore
    private Long uploadedDataId;

    @JsonIgnore
    private boolean generalHealthConsentRequired;

    @JsonIgnore
    private boolean mentalHealthConsentRequired;

    @JsonIgnore
    private boolean sexualHealthConsentRequired;

    @JsonIgnore
    private boolean socialCareConsentRequired;

    @JsonIgnore
    private String hospitalServiceCode;

    @JsonIgnore
    private boolean deleted;

    public void populate(EHRData ed) {
        uniqueId = ed.getUniqueId();
        enteredDate = ed.getEnteredDate();
        persistedDate = ed.getPersistedDate();
        uploadedDataId = ed.getUploadedDataId();
        deleted = ed.getDeleted();

        PrivacyFlags flags = ed.getLatestPrivacyFlags();
        if (flags != null) {
            generalHealthConsentRequired = flags.isRequireGeneral();
            mentalHealthConsentRequired = flags.isRequireMentalHealth();
            sexualHealthConsentRequired = flags.isRequireSexualHealth();
            socialCareConsentRequired = flags.isRequireSocialCare();
        } else {
            // Silly hoop-jumping required to deal with the possibility that ed.getRequire*()
            // returns null. Everyone loves tri-state booleans, right?
            generalHealthConsentRequired = Boolean.TRUE.equals(ed.getRequireGeneral());
            mentalHealthConsentRequired = Boolean.TRUE.equals(ed.getRequireMentalHealth());
            sexualHealthConsentRequired = Boolean.TRUE.equals(ed.getRequireSexualHealth());
            socialCareConsentRequired = Boolean.TRUE.equals(ed.getRequireSocialCare());
        }

        hospitalServiceCode = (String) ed.getEncryptedField(HOSPITAL_SERVICE_CODE);
    }

    /**
     * Gets consent as a privacy flag.
     */
    public Optional<PrivacyFlag> getPrivacyFlag() {
        return PrivacyFlagsToPrivacyFlagConverter.findPrivacyFlag(generalHealthConsentRequired, mentalHealthConsentRequired, sexualHealthConsentRequired,
                socialCareConsentRequired);
    }

    /**
     * Gets consent as a privacy flag set.
     * Most of the records should contain only one value, and we want only one value to be persisted, but there are records with more than one,
     * and only the users of the application who can actually read the data could correctly categorize it, so privacy flag edit flow must handle
     * multiple flags correctly.
     * Originally a data point could have had multiple flags set, and there were at least one bug around this, see PHR-4476.
     * 
     * @return
     */
    public Set<PrivacyFlag> getPrivacyFlags() {
        return PrivacyFlagsToPrivacyFlagConverter.getPrivacyFlags(generalHealthConsentRequired, mentalHealthConsentRequired, sexualHealthConsentRequired,
                socialCareConsentRequired);
    }

    public boolean isConsentEqualTo(@NotNull PrivacyFlag privacyFlag) {
        return generalHealthConsentRequired == (PrivacyFlag.GENERAL == privacyFlag) &&
                mentalHealthConsentRequired == (PrivacyFlag.MENTAL_HEALTH == privacyFlag) &&
                sexualHealthConsentRequired == (PrivacyFlag.SEXUAL_HEALTH == privacyFlag) &&
                socialCareConsentRequired == (PrivacyFlag.SOCIAL_CARE == privacyFlag);
    }

    public void writeToEHRData(EHRData ed, Instant now) {
        ed.setUniqueId(uniqueId);
        ed.setEnteredDate(enteredDate == null ? Date.from(now) : enteredDate);
        ed.setPersistedDate(persistedDate == null ? Date.from(now) : persistedDate);
        ed.setUploadedDataId(uploadedDataId);
        ed.setDeleted(deleted);

        ed.setRequireGeneral(generalHealthConsentRequired);
        ed.setRequireMentalHealth(mentalHealthConsentRequired);
        ed.setRequireSexualHealth(sexualHealthConsentRequired);
        ed.setRequireSocialCare(socialCareConsentRequired);

        ed.setEncryptedField(HOSPITAL_SERVICE_CODE, hospitalServiceCode);
    }

    // getters & setters

    public UUID getUniqueId() {
        return uniqueId;
    }

    public void setUniqueId(UUID uniqueId) {
        this.uniqueId = uniqueId;
    }

    /**
     * Sets a newly generated random UUID for this DTO and returns the generated id
     */
    public UUID generateNewRandomUniqueId() {
        if (uniqueId == null) {
            this.setUniqueId(UUID.randomUUID());
        }
        return uniqueId;
    }

    public Date getEnteredDate() {
        return enteredDate;
    }

    public Optional<Instant> findEnteredDate() {
        return Optional.ofNullable(enteredDate).map(Date::toInstant);
    }

    public void setEnteredDate(Date enteredDate) {
        this.enteredDate = enteredDate;
    }

    public Date getPersistedDate() {
        return persistedDate;
    }

    /** values set here prior to persisting a DTO are NOT respected; EHRBean sets this afresh at save time */
    public void setPersistedDate(Date persistedDate) {
        this.persistedDate = persistedDate;
    }

    public Long getUploadedDataId() {
        return uploadedDataId;
    }

    public void setUploadedDataId(Long uploadedDataId) {
        this.uploadedDataId = uploadedDataId;
    }

    public boolean isGeneralHealthConsentRequired() {
        return generalHealthConsentRequired;
    }

    public DTOBaseFields setPrivacyFlag(@NotNull PrivacyFlag requiredFlag) {
        switch (requiredFlag) {
        case GENERAL:
            generalHealthConsentRequired = true;
            mentalHealthConsentRequired = false;
            sexualHealthConsentRequired = false;
            socialCareConsentRequired = false;
            return this;
        case MENTAL_HEALTH:
            generalHealthConsentRequired = false;
            mentalHealthConsentRequired = true;
            sexualHealthConsentRequired = false;
            socialCareConsentRequired = false;
            return this;
        case SEXUAL_HEALTH:
            generalHealthConsentRequired = false;
            mentalHealthConsentRequired = false;
            sexualHealthConsentRequired = true;
            socialCareConsentRequired = false;
            return this;
        case SOCIAL_CARE:
            generalHealthConsentRequired = false;
            mentalHealthConsentRequired = false;
            sexualHealthConsentRequired = false;
            socialCareConsentRequired = true;
            return this;
        default:
            throw new NotImplementedException("Privacy flag type " + requiredFlag + " not implemented yet!");
        }
    }

    public void setDefaultGeneralConsentIfNoneSet() {
        if (!(generalHealthConsentRequired || mentalHealthConsentRequired || sexualHealthConsentRequired || socialCareConsentRequired)) {
            setPrivacyFlag(PrivacyFlag.GENERAL);
        }
    }

    public DTOBaseFields setGeneralHealthConsentRequired(boolean generalHealthConsentRequired) {
        this.generalHealthConsentRequired = generalHealthConsentRequired;
        return this;
    }

    public boolean isMentalHealthConsentRequired() {
        return mentalHealthConsentRequired;
    }

    public DTOBaseFields setMentalHealthConsentRequired(boolean mentalHealthConsentRequired) {
        this.mentalHealthConsentRequired = mentalHealthConsentRequired;
        return this;
    }

    public boolean isSexualHealthConsentRequired() {
        return sexualHealthConsentRequired;
    }

    public DTOBaseFields setSexualHealthConsentRequired(boolean sexualHealthConsentRequired) {
        this.sexualHealthConsentRequired = sexualHealthConsentRequired;
        return this;
    }

    public boolean isSocialCareConsentRequired() {
        return socialCareConsentRequired;
    }

    public DTOBaseFields setSocialCareConsentRequired(boolean socialCareConsentRequired) {
        this.socialCareConsentRequired = socialCareConsentRequired;
        return this;
    }

    public void copyConsentFlags(DTOBaseFields target) {
        target.setGeneralHealthConsentRequired(generalHealthConsentRequired)
                .setMentalHealthConsentRequired(mentalHealthConsentRequired)
                .setSexualHealthConsentRequired(sexualHealthConsentRequired)
                .setSocialCareConsentRequired(socialCareConsentRequired);
    }

    public String getHospitalServiceCode() {
        return hospitalServiceCode;
    }

    public void setHospitalServiceCode(String specialty) {
        this.hospitalServiceCode = specialty;
    }

    @Override
    public String toString() {
        return "DTOBaseFields{" + "uniqueId=" + uniqueId + ", enteredDate=" + enteredDate + ", persistedDate=" + persistedDate
                + ", uploadedDataId=" + uploadedDataId
                + ", generalHealthConsentRequired=" + generalHealthConsentRequired
                + ", mentalHealthConsentRequired=" + mentalHealthConsentRequired + ", sexualHealthConsentRequired="
                + sexualHealthConsentRequired + ", socialCareConsentRequired=" + socialCareConsentRequired + '}';
    }


    @Override
    public int hashCode() {
        int prime = 31;
        int result = 1;
        result = (prime * result) + ((enteredDate == null) ? 0 : enteredDate.hashCode());
        result = (prime * result) + (generalHealthConsentRequired ? 1231 : 1237);
        result = (prime * result) + ((hospitalServiceCode == null) ? 0 : hospitalServiceCode.hashCode());
        result = (prime * result) + (mentalHealthConsentRequired ? 1231 : 1237);
        result = (prime * result) + ((persistedDate == null) ? 0 : persistedDate.hashCode());
        result = (prime * result) + (sexualHealthConsentRequired ? 1231 : 1237);
        result = (prime * result) + (socialCareConsentRequired ? 1231 : 1237);
        result = (prime * result) + ((uniqueId == null) ? 0 : uniqueId.hashCode());
        result = (prime * result) + ((uploadedDataId == null) ? 0 : uploadedDataId.hashCode());
        return result;
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj) {
            return true;
        }
        if (obj == null) {
            return false;
        }
        if (getClass() != obj.getClass()) {
            return false;
        }
        DTOBaseFields other = (DTOBaseFields) obj;
        if (enteredDate == null) {
            if (other.enteredDate != null) {
                return false;
            }
        } else if (!enteredDate.equals(other.enteredDate)) {
            return false;
        }
        if (generalHealthConsentRequired != other.generalHealthConsentRequired) {
            return false;
        }
        if (hospitalServiceCode == null) {
            if (other.hospitalServiceCode != null) {
                return false;
            }
        } else if (!hospitalServiceCode.equals(other.hospitalServiceCode)) {
            return false;
        }
        if (mentalHealthConsentRequired != other.mentalHealthConsentRequired) {
            return false;
        }
        if (persistedDate == null) {
            if (other.persistedDate != null) {
                return false;
            }
        } else if (!persistedDate.equals(other.persistedDate)) {
            return false;
        }
        if (sexualHealthConsentRequired != other.sexualHealthConsentRequired) {
            return false;
        }
        if (socialCareConsentRequired != other.socialCareConsentRequired) {
            return false;
        }
        if (uniqueId == null) {
            if (other.uniqueId != null) {
                return false;
            }
        } else if (!uniqueId.equals(other.uniqueId)) {
            return false;
        }
        if (uploadedDataId == null) {
            return other.uploadedDataId == null;
        } else {
            return uploadedDataId.equals(other.uploadedDataId);
        }
    }

    public boolean isDeleted() {
        return deleted;
    }

    public void setDeleted(boolean deleted) {
        this.deleted = deleted;
    }
}
