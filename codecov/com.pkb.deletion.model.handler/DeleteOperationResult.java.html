package com.pkb.deletion.model.handler;

import com.pkb.deletion.model.DeleteRequestStatus;
import com.pkb.deletion.model.errors.DeletionServiceError;
import org.immutables.value.Value;

import java.util.Optional;
import java.util.function.Function;

@Value.Immutable
public interface DeleteOperationResult {
    DeleteRequestStatus getStatus();
    Optional<DeletionServiceError> getError();

    default DeleteOperationResult flatMap(Function<DeleteOperationResult, DeleteOperationResult> mapper) {
        if (getError().isPresent()) {
            return this;
        }
        return mapper.apply(this);
    }
}
