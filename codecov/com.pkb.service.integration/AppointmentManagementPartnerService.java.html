package com.pkb.service.integration;

import com.pkb.common.config.PhrConfig;
import io.vavr.control.Either;
import io.vavr.control.Try;
import org.apache.hc.core5.net.URIBuilder;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.net.URI;
import java.util.Map;
import java.util.Optional;
import java.util.Set;
import java.util.UUID;

import static java.util.function.Function.identity;
import static java.util.stream.Collectors.toMap;

/**
 * Appointments recorded in PKB may be managed by a third party (e.g. DrDr, Zesty,...)
 * <p>
 * Prefers passing URIs over URLs, as URL's equals/hashcode methods attempt dns lookups, so aren't useful for adding into hashmaps etc...
 */
public class AppointmentManagementPartnerService {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    public static final String PKB_CALLBACK_PARAM = "pkbCallback";

    private PhrConfig config;

    private DatapointExternalIntegrationUrlService datapointExternalIntegrationUrlService;

    public AppointmentManagementPartnerService(PhrConfig config, DatapointExternalIntegrationUrlService datapointExternalIntegrationUrlService) {
        this.config = config;
        this.datapointExternalIntegrationUrlService = datapointExternalIntegrationUrlService;
    }

    public Boolean hasManagementUrl(UUID appointmentUniqueId) {
        if (!config.isAppointmentPartnersEnabled()) {
            return Boolean.FALSE;
        }
        return datapointExternalIntegrationUrlService.hasIntegrationUrl(appointmentUniqueId);
    }

    /**
     * @param appointmentUniqueIds
     * @return - a map of all uniqueIds against either
     *  - False, if a url is not present or partners is disabled
     *  - True otherwise
     */
    public Map<UUID, Boolean> hasManagementUrl(Set<UUID> appointmentUniqueIds) {
        if (!config.isAppointmentPartnersEnabled()) {
            return appointmentUniqueIds.stream().collect(toMap(identity(), (uuid) -> Boolean.FALSE));
        }
        return datapointExternalIntegrationUrlService.hasIntegrationUrls(appointmentUniqueIds);
    }

    public Optional<URI> findManagementUrl(UUID appointmentUniqueId) {
        if (!config.isAppointmentPartnersEnabled()) {
            return Optional.empty();
        }

        return datapointExternalIntegrationUrlService.findIntegrationUrlAsUri(appointmentUniqueId);
    }

    /**
     * @param appointmentUniqueIds
     * @return - a map of all uniqueIds against either
     *  - empty, if a url is not present, partners is disabled, or the url is malformed (e.g. not a URL)
     *  - An Optional containing the URI otherwise
     */
    public Map<UUID, Optional<URI>> findManagementUrls(Set<UUID> appointmentUniqueIds) {
        if (!config.isAppointmentPartnersEnabled()) {
            return appointmentUniqueIds.stream().collect(toMap(identity(), (uuid) -> Optional.empty()));
        }
        return datapointExternalIntegrationUrlService.findIntegrationUrlsAsUris(appointmentUniqueIds);
    }

    public Optional<URI> validateAndAppendPkbCallback(String managementUrlString, UUID appointmentUniqueId, String callbackUrl) {
        Either<Throwable, URI> managementUri = datapointExternalIntegrationUrlService.tryConvertStringUrlToUri(managementUrlString);

        URI managementUrlWithCallback = Try.of(() -> {
                    URIBuilder builder = new URIBuilder(managementUri.get());
                    return builder.addParameter(PKB_CALLBACK_PARAM, callbackUrl).build();
                })
                .onFailure(throwable -> LOGGER.error("Unable to append callbackUrl {} for uuid {}", callbackUrl, appointmentUniqueId, throwable))
                .getOrElse((URI) null);

        return Optional.ofNullable(managementUrlWithCallback);
    }

}
