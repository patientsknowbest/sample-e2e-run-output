package com.pkb.service.integration;

import com.pkb.VersionDetails;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.integration.entity.DatapointExternalIntegrationUrl;
import com.pkb.repository.DatapointExternalIntegrationUrlRepository;
import io.vavr.control.Either;
import io.vavr.control.Try;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.net.URI;
import java.util.Collection;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.Set;
import java.util.UUID;
import java.util.function.Function;
import java.util.function.Supplier;

import static java.util.function.Function.identity;
import static java.util.stream.Collectors.toMap;

/**
 * Prefers passing URIs over URLs, as URL's equals/hashcode methods attempt dns lookups, so aren't useful for adding into hashmaps etc...
 */
public class DatapointExternalIntegrationUrlService {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final DatapointExternalIntegrationUrlRepository repository;
    private final DateTimeService dateTimeService;

    public DatapointExternalIntegrationUrlService(DatapointExternalIntegrationUrlRepository repository, DateTimeService dateTimeService) {
        this.repository = repository;
        this.dateTimeService = dateTimeService;
    }

    public DatapointExternalIntegrationUrl save(UUID datapointId, String url, EHRRequestContext requestContext) {
        DatapointExternalIntegrationUrl entity = new DatapointExternalIntegrationUrl();
        entity.setDatapointId(datapointId);
        entity.setUrl(url);
        entity.setVersionDetails(VersionDetails.of(requestContext, dateTimeService.now()));
        return repository.save(entity);
    }

    public Optional<DatapointExternalIntegrationUrl> findByDatapointId(UUID datapointId) {
        return repository.findByDatapointId(datapointId);
    }

    public List<DatapointExternalIntegrationUrl> findByDatapointIds(Collection<UUID> datapointIds) {
        return repository.findByDatapointIdIn(datapointIds);
    }

    public void delete(UUID datapointId) {
        repository.deleteByDatapointId(datapointId);
    }

    @NotNull
    public Boolean hasIntegrationUrl(UUID datapointId) {
        return hasIntegrationUrls(Set.of(datapointId)).get(datapointId);
    }

    /**
     * @param datapointIds
     * @return - a map that contains all uniqueIds against either TRUE if a record exists, or FALSE
     */
    @NotNull
    public Map<UUID, Boolean> hasIntegrationUrls(Set<UUID> datapointIds) {
        return findAndProcessIntegrationUrls(datapointIds,
                (url) -> Either.right(Boolean.TRUE),
                () -> Boolean.FALSE);
    }


    @NotNull
    public Optional<URI> findIntegrationUrlAsUri(UUID datapointId) {
        return findIntegrationUrlsAsUris(Set.of(datapointId))
                .get(datapointId);
    }

    /**
     * @param datapointIds
     * @return - a map that contains all uniqueIds against an Optional management url
     */
    @NotNull
    public Map<UUID, Optional<URI>> findIntegrationUrlsAsUris(Set<UUID> datapointIds) {
        return findAndProcessIntegrationUrls(datapointIds, (url) -> tryConvertStringUrlToUri(url).map(Optional::of), Optional::empty);
    }

    @NotNull
    private <T> Map<UUID, T> findAndProcessIntegrationUrls(Set<UUID> datapointIds, Function<String, Either<Throwable, T>> urlProcessor, Supplier<T> defaultValue) {
        var uniqueIdToRecord = findByDatapointIds(datapointIds)
                .stream()
                .collect(toMap(DatapointExternalIntegrationUrl::getDatapointId, identity()));

        return datapointIds.stream()
                .collect(toMap(identity(), uuid -> {
                    if (!uniqueIdToRecord.containsKey(uuid)) {
                        return defaultValue.get();
                    }
                    var record = uniqueIdToRecord.get(uuid);
                    var url = record.getUrl();

                    return urlProcessor.apply(url)
                            .peekLeft((throwable) -> LOGGER.error("Unable to apply processor to url for datapoint with uniqueid {}", uuid, throwable))
                            .getOrElse(defaultValue);
                }));
    }


    public Either<Throwable, URI> tryConvertStringUrlToUri(String managementUrl) {
        return Try.of(() -> {
                    URI value = URI.create(managementUrl);

                    //sanity this URI is also a valid URL (e.g. has a known handler for the protocol)
                    //noinspection ResultOfMethodCallIgnored
                    value.toURL();

                    return value;
                })
                .toEither();
    }

}
