package com.pkb.radiology;

import com.pkb.action.support.RadiologyResultTransformer;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.common.base.Error;
import com.pkb.radiology.entity.RadiologyResult;
import com.pkb.service.radiology.RadiologyManager;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;


@Component
public class RadiologyFacade {

    @Autowired
    private RadiologyManager radiologyManager;
    @Autowired
    private RadiologyResultTransformer radiologyResultTransformer;

    public List<RadiologyResult> retrieveRadiologyResults(LoggedInEHRRequestContext ehrRequestContext, RadiologyResult.MediaActCategory category, Long patientAccountId, int pageSize, int offset,
                                                          List<Long> filteredRadiologyIds) {
        return radiologyResultTransformer.transformRadiologyResult(radiologyManager.getRadiologyResults(ehrRequestContext, category,
                patientAccountId, pageSize, offset, filteredRadiologyIds));
    }

    public RadiologyResult retrieveRadiologyResult(LoggedInEHRRequestContext ehrRequestContext, Long dataPointId) {
        return radiologyResultTransformer.transformRadiologyResult(radiologyManager.getRadiologyResult(dataPointId, ehrRequestContext));
    }

    public RadiologyResult initNewRadiologyResult(EHRRequestContext ehrRequestContext) {
        RadiologyResult result = new RadiologyResult(new SourceDetails(ehrRequestContext));
        radiologyResultTransformer.transformRadiologyResult(result);
        return result;
    }

    public @NotNull Either<Error, Long> saveRadiologyResult(EHRRequestContext ehrRequestContext, long patientId, Long accountId, RadiologyResult radiologyResult) {
        return radiologyManager.saveRadiologyResult(ehrRequestContext, patientId, accountId, radiologyResultTransformer.transformRadiologyResultView(radiologyResult));
    }

}
