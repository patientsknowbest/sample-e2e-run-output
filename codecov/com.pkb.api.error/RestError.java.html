package com.pkb.api.error;

import com.pkb.common.config.PhrConfig;
import com.pkb.consent.model.ConsentStatusException;
import com.pkb.exception.RestException;
import com.pkb.util.PKBApplicationContext;

import javax.ws.rs.WebApplicationException;
import javax.ws.rs.core.Response;
import java.lang.reflect.UndeclaredThrowableException;


public class RestError {

    private int statusCode;
    private String errorMessage;
    private String developerMessage;

    public RestError() {

    }

    // TODO: Review, refactor and comment our approach to exceptions in the REST API. Our existing approach is far too confusing, and
    // potentially helpful messages are getting lost somewhere along the way.
    public RestError(Throwable t) {

        if (t instanceof RestException) {

            RestException exception = (RestException) t;

            setStatusCode(exception.getResponse().getStatus());

            this.developerMessage = exception.getAdditionalMessage();

        } else if (t instanceof WebApplicationException) {

            WebApplicationException exception = (WebApplicationException) t;

            setStatusCode(exception.getResponse().getStatus());

            this.developerMessage = exception.getMessage();

        } else {

            if (t.getCause() != null) {

                Throwable th = t.getCause();

                if (th instanceof UndeclaredThrowableException) {

                    this.developerMessage = ((UndeclaredThrowableException) th).getUndeclaredThrowable().getMessage();
                } else {

                    this.developerMessage = th.getMessage();
                }
            }

            if (t instanceof ConsentStatusException) {
                setStatusCode(Response.Status.FORBIDDEN.getStatusCode());
            } else {
                setStatusCode(Response.Status.INTERNAL_SERVER_ERROR.getStatusCode()); //defaults to internal server error 500
            }


        }

        this.errorMessage = t.getLocalizedMessage();

        // Hide the developer message if we are in a production environment
        if (!PKBApplicationContext.getBean(PhrConfig.class).isDisplayOfSensitiveErrorInformationEnabled()) {
            this.developerMessage = "An error has occured.";
        }

    }

    public int getStatusCode() {
        return statusCode;
    }

    public void setStatusCode(int statusCode) {
        this.statusCode = statusCode;
    }

    public String getErrorMessage() {
        return errorMessage;
    }

    public void setErrorMessage(String errorMessage) {
        this.errorMessage = errorMessage;
    }

    public String getDeveloperMessage() {
        return developerMessage;
    }

    public void setDeveloperMessage(String developerMessage) {
        this.developerMessage = developerMessage;
    }

}
