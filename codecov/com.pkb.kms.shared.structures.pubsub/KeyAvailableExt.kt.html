package com.pkb.kms.shared.structures.pubsub

import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.OrgPrivateId
import com.pkb.kms.shared.representation.PersonPublicId

fun accountKeyAvailable(accountPublicId: AccountPublicId) =
        KeyAvailable(KeyType.ACCOUNT_KEY, "$accountPublicId", null, null)

fun orgKeyAvailable(orgPrivateId: OrgPrivateId) =
        KeyAvailable(KeyType.ORG_ACCOUNT_KEY, null, orgPrivateId, null)

fun userKeyAvailable(personPublicId: PersonPublicId) =
        KeyAvailable(KeyType.USER_KEY, null, null, "$personPublicId")

fun KeyAvailable.orgPrivateId(): OrgPrivateId {
    assert(this.keyType == KeyType.ORG_ACCOUNT_KEY)
    return this.orgPrivateId
}

fun KeyAvailable.accountPublicId(): AccountPublicId {
    assert(this.keyType == KeyType.ACCOUNT_KEY)
    return PersonPublicId.fromString(this.accountPublicId.toString())
}

fun KeyAvailable.personPublicId(): PersonPublicId {
    assert(this.keyType == KeyType.USER_KEY)
    return PersonPublicId.fromString(this.personPublicId.toString())
}
