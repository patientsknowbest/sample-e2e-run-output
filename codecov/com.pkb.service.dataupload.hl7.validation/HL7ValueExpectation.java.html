package com.pkb.service.dataupload.hl7.validation;

import com.google.common.collect.ImmutableSet;
import com.pkb.service.dataupload.hl7.value.HL7Value;
import org.jetbrains.annotations.NotNull;

public class HL7ValueExpectation {

    private String id;
    private String name;
    private HL7Value providedHL7Value;

    private HL7ValueExpectation() {
    }

    public static Builder builder() {
        return new Builder();
    }

    public String getErrorMessageForMissingValue() {
        return String.format("%s (%s) must be provided. You provided [%s]",
                id, name, providedHL7Value.getRawValue());
    }

    public String getErrorMessageForInvalidValue(@NotNull ImmutableSet<String> permittedValues) {
        return String.format("%s (%s) value [%s] is not permitted. Permitted values are %s",
                id, name, providedHL7Value.getRawValue(), permittedValues.toString());
    }

    public String getId() {
        return id;
    }

    public String getName() {
        return name;
    }

    public HL7Value getProvidedHL7Value() {
        return providedHL7Value;
    }

    public static class Builder {

        private String id;
        private String name;
        private HL7Value providedHL7Value;

        private Builder() {
        }

        public Builder withId(String id) {
            this.id = id;
            return this;
        }

        public Builder withName(String name) {
            this.name = name;
            return this;
        }

        public Builder withProvidedHL7Value(HL7Value providedHL7Value) {
            this.providedHL7Value = providedHL7Value;
            return this;
        }

        public HL7ValueExpectation build() {
            HL7ValueExpectation ve = new HL7ValueExpectation();
            ve.id = this.id;
            ve.name = this.name;
            ve.providedHL7Value = this.providedHL7Value;
            return ve;
        }
    }
}
