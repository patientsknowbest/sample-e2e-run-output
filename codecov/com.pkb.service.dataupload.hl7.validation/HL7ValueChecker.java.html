package com.pkb.service.dataupload.hl7.validation;

import com.google.common.collect.ImmutableSet;
import com.pkb.exception.ApiCallMalformedException;
import org.jetbrains.annotations.NotNull;

import java.util.List;

public class HL7ValueChecker {

    /**
     * Calls: enforceMandatoryValue(expectation, false)
     *
     * @param expectation The HL7ValueExpectation describing the condition to check
     * @throws ApiCallMalformedException If the expectation is not met
     */
    public static void enforceMandatoryValue(@NotNull HL7ValueExpectation expectation)
            throws ApiCallMalformedException {
        enforceMandatoryValue(expectation, false);
    }

    /**
     * Provides a consistent way for the HL7 API to return an error if a mandatory value has not been provided.
     *
     * @param expectation   The HL7ValueExpectation describing the condition to check
     * @param absentAllowed If true - an absent value is permitted, but the HL7 Null Value is still forbidden
     * @throws ApiCallMalformedException If the expectation is not met
     */
    public static void enforceMandatoryValue(@NotNull HL7ValueExpectation expectation, boolean absentAllowed)
            throws ApiCallMalformedException {
        if (!checkMandatoryValue(expectation, absentAllowed)) {
            throw new ApiCallMalformedException(expectation.getErrorMessageForMissingValue());
        }
    }

    /**
     * Calls: enforceMandatoryValueAtLeastOne(expections, false)
     *
     * @param expectations The HL7ValueExpctations describing the conditions to check
     * @throws ApiCallMalformedException If none of the expectations are met
     */
    public static void enforceMandatoryValueAtLeastOne(@NotNull List<HL7ValueExpectation> expectations)
            throws ApiCallMalformedException {
        enforceMandatoryValueAtLeastOne(expectations, false);
    }

    /**
     * Checks that at least one of the provided expectations has been met. If this is not the case, an {@link ApiCallMalformedException} is
     * thrown. Note: this method only checks that at least one provided. If that is true, it is not an error for other unmet expectations to
     * have also been provided.
     *
     * @param expectations  The list of values, at least one of which is expected to be present
     * @param absentAllowed Whether or not absent values are permitted (affects HL7 null checking)
     * @throws ApiCallMalformedException If it is not true that at least one valid value was provided
     */
    public static void enforceMandatoryValueAtLeastOne(@NotNull List<HL7ValueExpectation> expectations, boolean absentAllowed)
            throws ApiCallMalformedException {
        if (expectations.isEmpty()) {
            // Nothing to check. Let's call that a win.
        } else {
            boolean atLeastOne = false;
            String idList = "";
            for (HL7ValueExpectation expectation : expectations) {
                if (checkMandatoryValue(expectation, absentAllowed)) {
                    atLeastOne = true;
                    break;
                }
                if (!idList.isEmpty()) {
                    idList += ", ";
                }
                idList += expectation.getId() + " (" + expectation.getName() + ")";
            }
            if (!atLeastOne) {
                throw new ApiCallMalformedException("You need to provide at least one of: " + idList);
            }
        }
    }

    /**
     * Checks that the provided value is one of the permitted values (case-sensitive). If this is not the case, an
     * {@link ApiCallMalformedException} is thrown.
     *
     * @param expectation The value expectation to check
     */
    public static void enforcePermittedValue(@NotNull HL7ValueExpectation expectation, @NotNull ImmutableSet<String> permittedValues)
            throws ApiCallMalformedException {
        if(!permittedValues.contains(expectation.getProvidedHL7Value().toString())) {
            throw new ApiCallMalformedException(expectation.getErrorMessageForInvalidValue(permittedValues));
        }
    }

    private static boolean checkMandatoryValue(HL7ValueExpectation expectation, boolean absentAllowed) {
        if (absentAllowed) {
            // Providing no value at all is allowed, but explicitly providing the HL7 null value is not allowed.
            if (expectation.getProvidedHL7Value().isHL7Null()) {
                return false;
            }
        } else {
            // The value must not be blank.
            return !expectation.getProvidedHL7Value().isBlank();
        }
        return true;
    }
}
