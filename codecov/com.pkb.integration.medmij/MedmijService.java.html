package com.pkb.integration.medmij;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.PersonAuthInfo;
import com.pkb.common.config.PhrConfig;
import com.pkb.service.team.TeamUserManager;
import com.pkb.user.entity.PKBPerson;
import org.apache.http.client.utils.URIBuilder;

import java.net.URISyntaxException;

public class MedmijService {

    private static final String COUNTRY_NL = "NL";
    private static final String PARAM_PKB_ID = "PKBID";

    private final TeamUserManager teamUserManager;
    private final PhrConfig config;

    public MedmijService(TeamUserManager teamUserManager, PhrConfig config) {
        this.teamUserManager = teamUserManager;
        this.config = config;
    }

    public boolean isMedMijSubject(LoggedInEHRRequestContext loggedInEHRRequestContext) {
        return isMedMijSubject(loggedInEHRRequestContext.getAccessingUserId());
    }

    public boolean isMedMijSubject(PKBPerson person) {
        return isMedMijSubject(person.getId());
    }

    public UpdateMyRecordDto buildUpdateMyRecordDto(LoggedInEHRRequestContext requestContext) {
        ImmutableUpdateMyRecordDto.Builder updateMyRecordBuilder = ImmutableUpdateMyRecordDto.builder();
        boolean buttonEnabled = false;

        if (config.isUpdateMyRecordFeatureEnabled()) {
            PersonAuthInfo accessingUser = requestContext.getAccessingUser();
            if (requestContext.isUserInOwnAccount() && accessingUser.isPatient()) {
                if (isMedMijSubject(requestContext)) {
                    String baseUrl = config.getUpdateMyRecordBaseURL();
                    String patientPublicId = accessingUser.getPublicId().toString();
                    try {
                        URIBuilder uriBuilder = new URIBuilder(baseUrl);
                        uriBuilder.addParameter(PARAM_PKB_ID, patientPublicId);
                        updateMyRecordBuilder.updateMyRecordURL(uriBuilder.toString());
                        buttonEnabled = true;
                    } catch (URISyntaxException e) {
                        throw new RuntimeException(e);
                    }
                }
            }
        }
        updateMyRecordBuilder.updateMyRecordButtonEnabled(buttonEnabled);
        return updateMyRecordBuilder.build();
    }

    private boolean isMedMijSubject(long personId) {
        return config.isMedMijIntegrationEnabled() && teamUserManager.getTeamsOfUser(personId).anyMatch(team -> team.getCountry().equals(COUNTRY_NL));
    }
}
