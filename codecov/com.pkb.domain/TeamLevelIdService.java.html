package com.pkb.domain;

import com.pkb.datamodel.TeamLevelId;
import com.pkb.datamodel.entity.mapper.TeamLevelIdMapper;
import com.pkb.repository.TeamLevelIdRepository;

import java.util.List;

public class TeamLevelIdService {

    private final TeamLevelIdRepository teamLevelIdRepository;
    private final TeamLevelIdMapper teamLevelIdMapper;

    public TeamLevelIdService (TeamLevelIdRepository teamLevelIdRepository, TeamLevelIdMapper teamLevelIdMapper) {
        this.teamLevelIdRepository = teamLevelIdRepository;
        this.teamLevelIdMapper = teamLevelIdMapper;
    }

    public List<TeamLevelId> findByPersonId(long personId) {
        return teamLevelIdRepository.findAllByPersonId(List.of(personId)).map(teamLevelIdMapper::entityToDataModel).toJavaList();
    }
}
