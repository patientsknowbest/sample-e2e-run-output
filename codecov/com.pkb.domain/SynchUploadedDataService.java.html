package com.pkb.domain;

import com.pkb.authentication.principal.user.AuthenticatedPerson;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.datamodel.Identifiable;
import com.pkb.datamodel.authorization.TargetOwnerData;
import com.pkb.datamodel.user.Person;
import com.pkb.domain.config.FhirPkbDomainRepositoryProperties;
import com.pkb.dto.FhirToStrutsDTO;
import com.pkb.dto.ImmutableSynchUploadedDataDTO;
import com.pkb.entities.enums.UploadedDataDestination;
import okhttp3.OkHttpClient;

public class SynchUploadedDataService extends FhirToStrutsService {

    public SynchUploadedDataService(FhirPkbDomainRepositoryProperties config, OkHttpClient httpClient) {
        super(config, httpClient);
    }

    public void synchUploadedData(AuthorizationData authzData, UploadedDataDestination uploadedDataDestination) {
        var authenticatedUser = authzData.inputs().authenticatedUser();
        if (authenticatedUser instanceof AuthenticatedPerson) {
            var authenticatedIdentity = authzData.inputs().authenticatedIdentity();
            ImmutableSynchUploadedDataDTO.Builder dtoBuilder = ImmutableSynchUploadedDataDTO.builder()
                    // Human users have a long ID and UserP*Key associated.
                    .authenticatedUserId(((AuthenticatedPerson) authenticatedUser).getId())
                    .userPrivateKey(authenticatedIdentity.requireUserPrivateKey())
                    .userPublicKey(authenticatedIdentity.requireUserPublicKey())
                    .addAllTargetPersonIds(authzData
                            .targetOwnerData()
                            .values()
                            .map(TargetOwnerData::person)
                            .map(Person::getId))
                    .synchUploadType(uploadedDataDestination)
                    .isBgTask(false);

            authzData.inputs()
                    .authenticatedUserTeam()
                    .map(Identifiable::getId)
                    .peek(dtoBuilder::teamId);

            FhirToStrutsDTO dto = dtoBuilder.build();
            connectToPhrAction(config.getUploadedDataSyncUrl(), "HTTP error synchronizing uploaded data ", dto);
        }
    }
}
