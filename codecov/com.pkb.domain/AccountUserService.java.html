package com.pkb.domain;

import com.pkb.common.config.PhrConfig;
import com.pkb.datamodel.AccountUser;
import com.pkb.datamodel.entity.mapper.AccountUserMapper;
import com.pkb.entities.core.CoreAccountUser;
import com.pkb.repository.AccountUserRepository;
import com.pkb.util.DbUtil;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.invoke.MethodHandles;
import java.util.List;
import java.util.Optional;
import java.util.Set;

public class AccountUserService {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
    private final AccountUserRepository accountUserRepository;
    private final AccountUserMapper mapper;
    private final PhrConfig config;

    public AccountUserService(AccountUserRepository accountUserRepository, AccountUserMapper mapper, PhrConfig config) {
        this.accountUserRepository = accountUserRepository;
        this.mapper = mapper;
        this.config = config;
    }


    @NotNull
    public List<CoreAccountUser> getOwnerAccountUsers(Set<Long> accountIds) {
        return DbUtil.getInBatches(accountIds, config.getAccountUserFetchSize(), accountUserRepository::findOwnerAccountUsers);
    }

    public Optional<CoreAccountUser> findAccountUserById(long accountUserId) {
        return accountUserRepository.findById(accountUserId);
    }

    public CoreAccountUser saveAccountUser(CoreAccountUser accountUser) {
        return accountUserRepository.save(accountUser);
    }

    public AccountUser save(AccountUser accountUser) {
        return mapper.entityToDataModel(accountUserRepository.save(mapper.dataModelToEntity(accountUser)));
    }


    public boolean hasNonDefaultAccountUser(long personId) {
        return accountUserRepository.countNonDefaultAccountUser(personId) > 0;
    }

}
