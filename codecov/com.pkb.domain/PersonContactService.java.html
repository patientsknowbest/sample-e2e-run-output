package com.pkb.domain;

import com.pkb.datamodel.contact.NewPersonContact;
import com.pkb.datamodel.contact.PersonContact;
import com.pkb.datamodel.entity.mapper.PersonContactMapper;
import com.pkb.entities.pub.PublicPersonContact;
import com.pkb.datamodel.Email;
import com.pkb.repository.PersonContactRepository;

import javax.transaction.Transactional;
import java.util.Optional;

import static java.lang.String.format;

public class PersonContactService {

    private final PersonContactRepository personContactRepository;
    private final PersonContactMapper personContactMapper;

    public PersonContactService(PersonContactRepository personContactRepository,
                                PersonContactMapper personContactMapper) {
        this.personContactRepository = personContactRepository;
        this.personContactMapper = personContactMapper;
    }

    public Optional<PersonContact> findOptionalByContactValueAndPersonId(Email email, long personId) {
        return personContactRepository.findOptionalByContactValueAndPersonId(email, personId)
                .map(contact -> personContactMapper.entityToDataModel(contact)
                        .getOrElseThrow(() -> new IllegalStateException(format("Failed to find contact for value %s", email))));
    }

    @Transactional
    public PersonContact save(NewPersonContact newPersonContact) {
        PublicPersonContact entity = personContactMapper.dataModelToEntity(newPersonContact);
        personContactRepository.save(entity);
        return personContactMapper.entityToDataModel(entity)
                .getOrElseThrow(() -> new IllegalStateException(format("Just saved a contact of type %s but I can't map it back again!", entity.getContactType())));
    }

    public int confirmContact(Email email, long personId) {
        // Access control - this method is invoked from a non-authenticated context
        return personContactRepository.confirmContact(email, personId);
    }
}
