package com.pkb.domain;

import com.pkb.datamodel.BulkInvitationOutcome;
import com.pkb.datamodel.entity.mapper.BulkInvitationOutcomeMapper;
import com.pkb.entities.enums.OperationOutcomeStatus;
import com.pkb.entities.pub.PublicBulkInvitationOutcome;
import com.pkb.repository.BulkInvitationOutcomeRepository;
import org.jetbrains.annotations.NotNull;
import org.springframework.data.domain.PageRequest;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDateTime;
import java.util.List;
import java.util.Optional;
import java.util.stream.Collectors;

import static com.pkb.entities.enums.OperationOutcomeStatus.FAILED_FOR_UNEXPECTED_REASON;
import static java.util.Comparator.comparing;

public class BulkInvitationOutcomeService {

    private final BulkInvitationOutcomeRepository bulkInvitationOutcomeRepository;
    private final BulkInvitationOutcomeMapper mapper;

    public BulkInvitationOutcomeService(BulkInvitationOutcomeRepository bulkInvitationOutcomeRepository, BulkInvitationOutcomeMapper mapper) {
        this.bulkInvitationOutcomeRepository = bulkInvitationOutcomeRepository;
        this.mapper = mapper;
    }

    public BulkInvitationOutcome updateOutcome(@NotNull BulkInvitationOutcome outcome, @NotNull String csvEnrichedWithOutcomes, @NotNull OperationOutcomeStatus status) {
        PublicBulkInvitationOutcome entity = mapper.dataModelToEntity(outcome);
        entity.setCsvEnrichedWithOutcomes(csvEnrichedWithOutcomes);
        entity.setStatus(status);
        return mapper.entityToDataModel(bulkInvitationOutcomeRepository.save(entity));
    }

    public List<BulkInvitationOutcome> findLastInvitationOutcomesForTeam(long teamId, int maxResults) {
        return bulkInvitationOutcomeRepository.getOutcomesForTeamInDescendingOrder(teamId, PageRequest.of(0, maxResults))
                .stream()
                .map(mapper::entityToDataModel)
                .sorted(comparing(BulkInvitationOutcome::getUploadedAt))
                .collect(Collectors.toList());
    }

    public Optional<String> findOutcome(long teamId, long outcomeId) {
        Optional<String> outcomes = bulkInvitationOutcomeRepository.findOutcomeInTeam(teamId, outcomeId);

        if (outcomes.isEmpty() && bulkInvitationOutcomeRepository.findById(outcomeId).isPresent()) {
            throw new IllegalArgumentException(String.format("Outcome id '%s' is not accessable for '%s' team members.", outcomeId, teamId));
        }

        return outcomes;
    }

    @Transactional
    public Optional<BulkInvitationOutcome> markAsFailed(long initiatorId, long invitingTeamId, LocalDateTime initiatedAt) {
        Optional<PublicBulkInvitationOutcome> result = bulkInvitationOutcomeRepository.findFirstByInvitingTeamIdAndUploadedAtAndUploadedById(invitingTeamId, initiatedAt, initiatorId);
        if (result.isPresent()) {
            bulkInvitationOutcomeRepository.updateOutcomeStatus(result.get().getId(), FAILED_FOR_UNEXPECTED_REASON);
            // set this on the returned object too although it won't be used
            result.get().setStatus(FAILED_FOR_UNEXPECTED_REASON);
            return Optional.of(mapper.entityToDataModel(result.get()));
        }
        return Optional.empty();
    }

    public @NotNull BulkInvitationOutcome createOutcome(
            long initiatorId,
            long invitingTeamId,
            @NotNull LocalDateTime initiatedAt,
            @NotNull OperationOutcomeStatus status) {

        PublicBulkInvitationOutcome outcome = new PublicBulkInvitationOutcome();
        outcome.setUploadedById(initiatorId);
        outcome.setInvitingTeamId(invitingTeamId);
        outcome.setUploadedAt(initiatedAt);
        outcome.setStatus(status);

        return mapper.entityToDataModel(bulkInvitationOutcomeRepository.save(outcome));
    }
}
