package com.pkb.domain;

import com.pkb.datamodel.IdentityVerification;
import com.pkb.datamodel.entity.mapper.IdentityVerificationMapper;
import com.pkb.entities.enums.VerificationStatus;
import com.pkb.entities.pub.PublicIdentityVerification;
import com.pkb.repository.IdentityVerificationRepository;
import com.pkb.util.DbUtil;

import java.util.Collection;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;

public class IdentityVerificationService {

    private final IdentityVerificationRepository identityVerificationRepository;
    private final IdentityVerificationMapper mapper;

    public IdentityVerificationService(IdentityVerificationRepository identityVerificationRepository, IdentityVerificationMapper mapper) {
        this.identityVerificationRepository = identityVerificationRepository;
        this.mapper = mapper;
    }

    public IdentityVerification save(IdentityVerification entity) {
        return mapper.entityToDatamodel(identityVerificationRepository.save(mapper.datamodelToEntity(entity)));
    }

    public Optional<IdentityVerification> getIdentityVerification(long personId) {
        return identityVerificationRepository.findByPersonId(personId)
                .map(mapper::entityToDatamodel);
    }

    public Map<Long, IdentityVerification> getIdentityVerifications(Collection<Long> personId) {
        return identityVerificationRepository.findAllByPersonIdIn(personId)
                .stream()
                .collect(Collectors.toMap(PublicIdentityVerification::getPersonId, mapper::entityToDatamodel));
    }

    public List<IdentityVerification> getIdentityVerificationRecords(Collection<Long> personId) {
        return DbUtil.getInStreamOfBatches(personId, identityVerificationRepository::findAllByPersonIdIn)
                .map(mapper::entityToDatamodel)
                .collect(Collectors.toList());
    }

    public List<Long> getAllVerifiedInstituteUserPersonIdsForTeam(long teamId) {
        return identityVerificationRepository.getAllInstituteUserPersonIdsForTeamAndStatus(teamId, VerificationStatus.VERIFIED);
    }

    public boolean isVerified(long personId) {
        return identityVerificationRepository.findByPersonIdAndStatus(personId, VerificationStatus.VERIFIED).isPresent();
    }

    public Optional<VerificationStatus> getVerificationStatusForId(long personId) {
        return identityVerificationRepository.getVerificationStatusById(personId);
    }
}
