package com.pkb.domain;

import com.pkb.domain.config.FhirPkbDomainRepositoryProperties;
import com.pkb.dto.FhirToStrutsDTO;
import okhttp3.MediaType;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.RequestBody;
import okhttp3.Response;

import java.io.IOException;

import static okhttp3.HttpUrl.get;
import static org.apache.commons.lang3.SerializationUtils.serialize;

public class FhirToStrutsService {

    protected final FhirPkbDomainRepositoryProperties config;
    private final OkHttpClient httpClient;

    public FhirToStrutsService(FhirPkbDomainRepositoryProperties config, OkHttpClient httpClient) {
        this.config = config;
        this.httpClient = httpClient;
    }

    protected void connectToPhrAction(String url, String errorMessage, FhirToStrutsDTO dto) {
        var configuredHttpClient = httpClient.newBuilder()
                .connectTimeout(config.getFhirToStrutsConnectionTimeout())
                .writeTimeout(config.getFhirApiToStrutsWriteTimeout())
                .readTimeout(config.getFhirApiToStrutsReadTimeout())
                .build();

        try (Response response = configuredHttpClient.newCall(new Request.Builder()
                .url(get(url))
                .post(RequestBody.create(serialize(dto), MediaType.parse("application/octet-stream")))
                .build())
                .execute()) {
            if (!response.isSuccessful()) {
                throw new RuntimeException(errorMessage + " " + response);
            }
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
    }

}
