package com.pkb.domain;

import com.google.common.collect.Lists;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.domain.util.SecurityQuestionCryptoUtil;
import com.pkb.entities.core.CorePasswordRecovery;
import com.pkb.entities.core.CoreSecurityQuestion;
import com.pkb.entities.enums.SecurityQuestionPurpose;
import com.pkb.entities.enums.core.PasswordRecoveryStatus;
import com.pkb.repository.PasswordRecoveryRepository;
import com.pkb.repository.SecurityQuestionRepository;
import io.vavr.control.Either;
import org.apache.commons.beanutils.BeanUtils;

import javax.transaction.Transactional;
import java.security.SecureRandom;
import java.util.List;
import java.util.Optional;

import static com.pkb.entities.enums.SecurityQuestionPurpose.LOGIN_PASSWORD_RECOVERY;
import static java.lang.String.format;
import static org.tolven.core.entity.TolvenProperties.PASSWORD_ITERATION_COUNT;
import static org.tolven.core.entity.TolvenProperties.PASSWORD_SALT_LENGTH;

public class PkbPasswordRecoveryService {
    private final SecureRandom rng = new SecureRandom();

    private final DateTimeService dateTimeService;
    private final SecurityQuestionRepository securityQuestionRepository;
    private final PasswordRecoveryRepository passwordRecoveryRepository;

    public PkbPasswordRecoveryService(DateTimeService dateTimeService, SecurityQuestionRepository securityQuestionRepository,
            PasswordRecoveryRepository passwordRecoveryRepository) {
        this.dateTimeService = dateTimeService;
        this.securityQuestionRepository = securityQuestionRepository;
        this.passwordRecoveryRepository = passwordRecoveryRepository;
    }

    @Transactional
    public void backupPassword(long userId, char[] password, SecurityQuestionPurpose passwordPurpose, String securityQuestion, String securityQuestionAnswer) {
        try {
            byte[] salt = getRandomSalt();
            byte[] encryptedPassword = SecurityQuestionCryptoUtil.encryptPassword(password, salt, PASSWORD_ITERATION_COUNT, securityQuestionAnswer);
            storeEncryptedPassword(encryptedPassword, salt, PASSWORD_ITERATION_COUNT, userId, securityQuestion, passwordPurpose);
        } catch (Exception exception) {
            throw new RuntimeException("Could not backup password for: " + userId, exception);
        }
    }

    public List<CoreSecurityQuestion> findSecurityQuestions(SecurityQuestionPurpose purpose) {
        return securityQuestionRepository.findAllByPurpose(purpose.value());
    }

    public Optional<CorePasswordRecovery> findActivePasswordRecovery(long personId, SecurityQuestionPurpose passwordPurpose) {
        return findActivePasswordRecovery(personId, passwordPurpose, null);
    }

    private Optional<CorePasswordRecovery> findActivePasswordRecovery(long pkbPersonId, SecurityQuestionPurpose passwordPurpose, String securityQuestion) {
        Optional<CorePasswordRecovery> result;
        if (securityQuestion == null) {
            result = passwordRecoveryRepository.findOptionalByPasswordPurposeAndPersonIdAndStatus(passwordPurpose.value(), pkbPersonId, PasswordRecoveryStatus.active);
        } else {
            result = passwordRecoveryRepository.findOptionalBySecurityQuestionAndPasswordPurposeAndPersonIdAndStatus(securityQuestion, passwordPurpose.value(), pkbPersonId,
                    PasswordRecoveryStatus.active);
        }
        return result;
    }

    private byte[] getRandomSalt() {
        byte[] salt = new byte[PASSWORD_SALT_LENGTH];
        rng.nextBytes(salt);
        return salt;
    }

    @Transactional
    public void backupLoginPassword(long userId, char[] password, String securityQuestion, String securityQuestionAnswer) {
        findActivePasswordRecovery(userId, LOGIN_PASSWORD_RECOVERY).ifPresent(
                passwordRecovery -> {
                    try {
                        deactivatePasswordRecovery(userId, LOGIN_PASSWORD_RECOVERY, passwordRecovery.getSecurityQuestion());
                    } catch (Exception exception) {
                        throw new RuntimeException("Failed to deactivate password recovery for: " + userId, exception);
                    }
                });

        if (securityQuestion != null) {
            backupPassword(userId, password, LOGIN_PASSWORD_RECOVERY, securityQuestion, securityQuestionAnswer);
        }
    }

    public boolean isPasswordRecoveryQuestion(long securityQuestionId) {
        Optional<CoreSecurityQuestion> question = securityQuestionRepository.findById(securityQuestionId);
        return question.isPresent() && LOGIN_PASSWORD_RECOVERY.matches(question.get().getPurpose());
    }

    public boolean verifySecurityQuestionAnswer(long userId, String securityQuestion, String securityQuestionAnswer) {
        Optional<char[]> recoveredPassword = recoverPassword(LOGIN_PASSWORD_RECOVERY, userId, securityQuestion, securityQuestionAnswer);
        return recoveredPassword.isPresent();
    }

    private void deactivatePasswordRecovery(long userId, SecurityQuestionPurpose passwordPurpose, String securityQuestion) {
        findActivePasswordRecovery(userId, passwordPurpose, securityQuestion)
                .ifPresent(this::deactivatePasswordRecovery);
    }

    private void storeEncryptedPassword(byte[] encryptedPassword,
            byte[] salt,
            int iterationCount,
            long userId,
            String securityQuestion,
            SecurityQuestionPurpose passwordPurpose) {
        deactivatePasswordRecovery(userId, passwordPurpose, null);

        CorePasswordRecovery passwordRecovery2 = new CorePasswordRecovery();
        passwordRecovery2.setEncryptedPassword(encryptedPassword);
        passwordRecovery2.setSalt(salt);
        passwordRecovery2.setIterationCount(iterationCount);
        passwordRecovery2.setPersonId(userId);
        passwordRecovery2.setSecurityQuestion(securityQuestion);
        passwordRecovery2.setPasswordPurpose(passwordPurpose.value());
        passwordRecovery2.setStatus(PasswordRecoveryStatus.active);
        passwordRecovery2.setCreation(dateTimeService.now());
        passwordRecoveryRepository.saveAndFlush(passwordRecovery2);
    }

    public Optional<char[]> recoverPassword(SecurityQuestionPurpose passwordPurpose, long userId, String securityQuestion, String securityQuestionAnswer) {
        Optional<CorePasswordRecovery> passwordRecovery = findActivePasswordRecovery(userId, passwordPurpose, securityQuestion);
        if (passwordRecovery.isEmpty()) {
            throw new RuntimeException(format("No password recovery available for '%s' using security question: '%s'",
                    passwordPurpose, securityQuestion));
        } else {
            List<char[]> answersToTry = Lists.newArrayList(securityQuestionAnswer.trim().toLowerCase().toCharArray(), securityQuestionAnswer.toCharArray(),
                    securityQuestionAnswer.trim().toCharArray(),
                    securityQuestionAnswer.toLowerCase().toCharArray());
            return answersToTry.stream()
                    .map(a -> tryPasswordRecovery(passwordRecovery.get(), a))
                    .filter(e -> e.isRight())
                    .map(Either::get)
                    .findFirst();
        }
    }

    /**
     * For multi-team users: When registering a new account, the password and
     * security must be the same across all linked accounts. This is easy for
     * the password, which is lurking around in the session, but hard for
     * security answer. We can just clone the record for the logged-in user,
     * salt and all, until we deploy the auth service to do this properly.
     * <p>
     * Sets any active PublicPasswordRecovery record(s) for the target user to
     * inactive first.
     */
    @Transactional
    public void replacePasswordBackupWithClone(long sourcePersonId, long targetPersonId) {
        // deactivate any existing active record for the target user (we're
        // replacing it!)
        deactivatePasswordRecovery(targetPersonId, LOGIN_PASSWORD_RECOVERY, null);

        CorePasswordRecovery recovery = findActivePasswordRecovery(sourcePersonId, LOGIN_PASSWORD_RECOVERY, null)
                .orElseThrow(
                        () -> new RuntimeException("could not find existing PublicPasswordRecovery to clone"));

        try {
            // Clone details field by field (hibernate doesn't seem to let me
            // detach and re-save with a new ID...), and store for other user
            CorePasswordRecovery clone = new CorePasswordRecovery();
            BeanUtils.copyProperties(clone, recovery);
            clone.setId(0L);
            clone.setPersonId(targetPersonId);
            passwordRecoveryRepository.save(clone);
        } catch (Exception e) {
            throw new RuntimeException("Exception while cloning security answer for-"
                    + targetPersonId, e);
        }

    }

    private Either<Exception, char[]> tryPasswordRecovery(CorePasswordRecovery passwordRecovery, char[] securityQuestionAnswer) {
        return SecurityQuestionCryptoUtil.decryptPassword(passwordRecovery.getEncryptedPassword(), passwordRecovery.getIterationCount(), passwordRecovery.getSalt(),
                securityQuestionAnswer);
    }

    private void deactivatePasswordRecovery(CorePasswordRecovery passwordRecovery) {
        passwordRecovery.setStatus(PasswordRecoveryStatus.inactive);
        passwordRecoveryRepository.saveAndFlush(passwordRecovery);
    }
}
