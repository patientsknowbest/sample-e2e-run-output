package com.pkb.domain;

import com.pkb.datamodel.NationalId;
import com.pkb.datamodel.NewNationalId;
import com.pkb.datamodel.entity.mapper.NationalIdMapper;
import com.pkb.entities.pub.PublicNationalId;
import com.pkb.repository.NationalIdRepository;
import org.jetbrains.annotations.NotNull;

import javax.transaction.Transactional;
import java.util.List;
import java.util.stream.Collectors;

public class NationalIdService {
    private final NationalIdRepository repository;
    private final NationalIdMapper nationalIdMapper;

    public NationalIdService(NationalIdRepository repository,
                             NationalIdMapper nationalIdMapper) {
        this.repository = repository;
        this.nationalIdMapper = nationalIdMapper;
    }

    @Transactional
    @NotNull
    public NationalId save(@NotNull NewNationalId newNationalId) {
        PublicNationalId entity = nationalIdMapper.dataModelToEntity(newNationalId);
        repository.save(entity);
        return nationalIdMapper.entityToDataModel(entity);
    }

    public List<NationalId> findByPersonId(long personId) {
        return repository.findByPersonId(personId).stream().map(nationalIdMapper::entityToDataModel).collect(Collectors.toList());
    }
}
