package com.pkb.domain;

import com.pkb.datamodel.AccessLog;
import com.pkb.datamodel.ImmutableAccessLog;
import com.pkb.datamodel.PrivacyFlagsOverride;
import com.pkb.datamodel.entity.mapper.AccessLogMapper;
import com.pkb.datamodel.entity.mapper.PrivacyFlagsMapper;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.entities.enums.Route;
import com.pkb.entities.pub.PublicPrivacyFlags;
import com.pkb.repository.AccessLogRepository;
import com.pkb.repository.PrivacyFlagsRepository;
import io.vavr.control.Option;

import javax.transaction.Transactional;
import java.time.Instant;
import java.util.Collection;
import java.util.Collections;
import java.util.List;
import java.util.Optional;
import java.util.stream.Collectors;

import static org.springframework.util.CollectionUtils.isEmpty;

public class PrivacyFlagsService {

    private final PrivacyFlagsRepository privacyFlagsRepository;
    private final PrivacyFlagsMapper privacyFlagsMapper;
    private final AccessLogRepository accessLogRepository;
    private final AccessLogMapper accessLogMapper;

    public PrivacyFlagsService(PrivacyFlagsRepository privacyFlagsRepository, PrivacyFlagsMapper privacyFlagsMapper,
                               AccessLogRepository accessLogRepository, AccessLogMapper accessLogMapper) {
        this.privacyFlagsRepository = privacyFlagsRepository;
        this.privacyFlagsMapper = privacyFlagsMapper;
        this.accessLogRepository = accessLogRepository;
        this.accessLogMapper = accessLogMapper;
    }

    public List<PrivacyFlagsOverride> getPrivacyFlagsByIds(Collection<Long> ids) {
        if (isEmpty(ids)) {
            return Collections.emptyList();
        }

        return privacyFlagsRepository.findAllById(ids)
                .stream()
                .map(privacyFlagsMapper::entityToDataModel)
                .collect(Collectors.toList());
    }

    @Transactional
    public PrivacyFlagsOverride savePrivacyFlagsAndAccessLog(PrivacyFlagsOverride privacyFlags,
                                                             long ownerAccountId, MenuDataType type, Instant now,
                                                             Long accessingUserId, Route accessRoute,
                                                             Long consentId, String correlationId) {
        PublicPrivacyFlags publicPrivacyFlags = privacyFlagsMapper.dataModelToEntity(privacyFlags);
        publicPrivacyFlags = privacyFlagsRepository.save(publicPrivacyFlags);

        AccessLog accessLog = ImmutableAccessLog.builder()
                .accountId(ownerAccountId)
                .accessingPersonId(Option.of(accessingUserId))
                .dataType(type)
                .route(accessRoute)
                .breakTheGlassActive(false)
                .patientConsentId(Option.of(consentId))
                .accessTime(now)
                .correlationId(Option.of(correlationId))
                .privacyFlagsId(Option.of(publicPrivacyFlags.getId()))
                .build();

        accessLogRepository.save(accessLogMapper.newDataModelToNewEntity(accessLog));
        return privacyFlagsMapper.entityToDataModel(publicPrivacyFlags);
    }

    @Transactional
    public void savePrivacyFlagsAndAccessLogMultiple(Optional<PrivacyFlagsOverride> maybeMainPrivacyFlag, Collection<PrivacyFlagsOverride> additionalPrivacyFlags,
                                                     long ownerAccountId, MenuDataType mainType, MenuDataType additionalType, Instant now,
                                                     Long accessingUserId, Route accessRoute,
                                                     Long consentId, String correlationId) {
        maybeMainPrivacyFlag.ifPresent(mainPrivacyFlag -> {
                    PublicPrivacyFlags publicPrivacyFlags = privacyFlagsMapper.dataModelToEntity(mainPrivacyFlag);
                    publicPrivacyFlags = privacyFlagsRepository.save(publicPrivacyFlags);

                    AccessLog accessLog = ImmutableAccessLog.builder()
                            .accountId(ownerAccountId)
                            .accessingPersonId(Option.of(accessingUserId))
                            .dataType(mainType)
                            .route(accessRoute)
                            .breakTheGlassActive(false)
                            .patientConsentId(Option.of(consentId))
                            .accessTime(now)
                            .correlationId(Option.of(correlationId))
                            .privacyFlagsId(Option.of(publicPrivacyFlags.getId()))
                            .build();

                    accessLogRepository.save(accessLogMapper.newDataModelToNewEntity(accessLog));
                });

        additionalPrivacyFlags.forEach(privacyFlag -> {
            PublicPrivacyFlags additionalPublicPrivacyFlags = privacyFlagsMapper.dataModelToEntity(privacyFlag);
            additionalPublicPrivacyFlags = privacyFlagsRepository.save(additionalPublicPrivacyFlags);

            AccessLog additionalAccessLog = ImmutableAccessLog.builder()
                    .accountId(ownerAccountId)
                    .accessingPersonId(Option.of(accessingUserId))
                    .dataType(additionalType)
                    .route(accessRoute)
                    .breakTheGlassActive(false)
                    .patientConsentId(Option.of(consentId))
                    .accessTime(now)
                    .correlationId(Option.of(correlationId))
                    .privacyFlagsId(Option.of(additionalPublicPrivacyFlags.getId()))
                    .build();

            accessLogRepository.save(accessLogMapper.newDataModelToNewEntity(additionalAccessLog));
        });
    }
}
