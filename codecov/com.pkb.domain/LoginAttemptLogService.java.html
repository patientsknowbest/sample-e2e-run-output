package com.pkb.domain;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.ImmutableLoginAttemptLog;
import com.pkb.datamodel.LoginAttemptLog;
import com.pkb.datamodel.entity.mapper.LoginAttemptLogMapper;
import com.pkb.entities.enums.LoginAttemptLogResult;
import com.pkb.entities.pub.PublicLoginAttemptLog;
import com.pkb.repository.LoginAttemptLogRepository;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import javax.transaction.Transactional;
import java.time.LocalDateTime;

import static com.pkb.entities.enums.LoginAttemptLogResult.FAIL_BAD_PASSWORD;

public class LoginAttemptLogService {

    private final LoginAttemptLogRepository loginAttemptLogRepository;
    private final DateTimeService dateTimeService;
    private final LoginAttemptLogMapper loginAttemptLogMapper;

    public LoginAttemptLogService(
            LoginAttemptLogRepository loginAttemptLogRepository,
            DateTimeService dateTimeService,
            LoginAttemptLogMapper loginAttemptLogMapper) {

        this.loginAttemptLogRepository = loginAttemptLogRepository;
        this.dateTimeService = dateTimeService;
        this.loginAttemptLogMapper = loginAttemptLogMapper;
    }

    public int findCountOfRecentLoginAttempts(Long personId, LocalDateTime fromDate) {
        return loginAttemptLogRepository.getCountOfPublicLoginAttemptLogsByPersonIdAndResultAndLoginTime(
                personId, FAIL_BAD_PASSWORD, fromDate);
    }

    public boolean isPersonNhsAppUser(Long personId) {
        return loginAttemptLogRepository.isPersonNhsAppUser(personId);
    }

    @Transactional
    public LoginAttemptLog createLoginAttemptLog(
            Long personId,
            @NotNull LoginAttemptLogResult result,
            @NotNull String ipAddress,
            @Nullable String badUsername,
            @Nullable String subresult) {

        LoginAttemptLog loginAttemptLog = ImmutableLoginAttemptLog.builder()
                .badUserName(Option.of(badUsername))
                .result(result)
                .loginTime(dateTimeService.nowLocalDateTime())
                .ipAddress(ipAddress)
                .personId(Option.of(personId))
                .subresult(Option.of(subresult))
                .build();

        PublicLoginAttemptLog publicLoginAttemptLog = loginAttemptLogMapper.newDataModelToNewEntity(loginAttemptLog);
        return loginAttemptLogMapper.entityToDataModel(loginAttemptLogRepository.save(publicLoginAttemptLog));
    }
}
