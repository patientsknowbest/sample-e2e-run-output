package com.pkb.domain;

import com.pkb.datamodel.ReferenceDatum;
import com.pkb.datamodel.entity.mapper.ReferenceDatumMapper;
import com.pkb.entities.enums.ReferenceDataType;
import com.pkb.repository.ReferenceDatumRepository;
import io.vavr.control.Option;

import java.util.Collection;
import java.util.List;
import java.util.stream.Collectors;

import static java.util.Collections.emptyList;
import static org.apache.commons.collections4.CollectionUtils.isEmpty;

public class ReferenceDatumService {

    private final ReferenceDatumRepository referenceDatumRepository;
    private final ReferenceDatumMapper referenceDatumMapper;

    public ReferenceDatumService(ReferenceDatumRepository referenceDatumRepository, ReferenceDatumMapper referenceDatumMapper) {
        this.referenceDatumRepository = referenceDatumRepository;
        this.referenceDatumMapper = referenceDatumMapper;
    }

    public Option<ReferenceDatum> getReferenceDatum(long orgId, ReferenceDataType dataType, String dataId) {
        return referenceDatumRepository.getPublicReferenceDatumByOrgAndDataTypeAndDataId(orgId, dataType, dataId)
                .map(referenceDatumMapper::entityToDataModel);
    }

    public List<ReferenceDatum> getReferenceData(Collection<Long> orgIds, Collection<ReferenceDataType> dataTypes, Collection<String> dataIds) {
        if (isEmpty(orgIds) || isEmpty(dataTypes) || isEmpty(dataIds)) {
            return emptyList();
        }

        return referenceDatumRepository.getPublicReferenceDataByOrgAndDataTypeAndDataId(orgIds, dataTypes, dataIds)
                .stream().map(referenceDatumMapper::entityToDataModel).collect(Collectors.toList());
    }

    public List<ReferenceDatum> getAllReferenceDataForOrg(long orgId) {
        return referenceDatumRepository.getPublicReferenceDataByOrg(orgId)
                .stream().map(referenceDatumMapper::entityToDataModel).collect(Collectors.toList());
    }
}
