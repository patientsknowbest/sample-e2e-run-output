package com.pkb.domain;

import com.pkb.datamodel.NewUnreadableDocument;
import com.pkb.datamodel.UnreadableDocument;
import com.pkb.datamodel.entity.mapper.UnreadableDocumentMapper;
import com.pkb.entities.doc.DocUnreadableDocument;
import com.pkb.repository.UnreadableDocumentRepository;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.transaction.Transactional;

public class UnreadableDocumentService {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final UnreadableDocumentRepository repository;
    private final UnreadableDocumentMapper mapper;

    public UnreadableDocumentService(UnreadableDocumentRepository repository, UnreadableDocumentMapper mapper) {
        this.repository = repository;
        this.mapper = mapper;
    }

    @Transactional
    public @NotNull UnreadableDocument save(NewUnreadableDocument newUnreadableDocument) {
        LOGGER.warn("Reporting unreadable document for user {} in message {}", newUnreadableDocument.getPersonPublicId(), newUnreadableDocument.getExternalMessageId());
        DocUnreadableDocument entity = mapper.dataModelToEntity(newUnreadableDocument);
        repository.save(entity);
        return mapper.entityToDataModel(entity);
    }

    public boolean isDocumentReported(long attachmentId) {
        return repository.countByAttachment_id(attachmentId) > 0;
    }

}
