package com.pkb.domain;

import com.pkb.entities.pub.PublicLetterInvitation;
import com.pkb.repository.LetterInvitationRepository;
import org.jetbrains.annotations.NotNull;

import javax.transaction.Transactional;
import java.util.Optional;

public class LetterInvitationService {

    private final LetterInvitationRepository letterInvitationRepository;

    public LetterInvitationService(LetterInvitationRepository letterInvitationRepository) {
        this.letterInvitationRepository = letterInvitationRepository;
    }

    public Optional<PublicLetterInvitation> findById(long id) {
        return letterInvitationRepository.findById(id);
    }

    public Optional<PublicLetterInvitation> findByCode(@NotNull String code) {
        return letterInvitationRepository.findPublicLetterInvitationByCode(code);
    }

    public Optional<PublicLetterInvitation> findByVerificationCode(@NotNull String code) {
        return letterInvitationRepository.findPublicLetterInvitationByVerificationCode(code);
    }

    @Transactional
    public int invalidateInvitations(long inviteeId) {
        return letterInvitationRepository.invalidateInvitations(inviteeId);
    }

    @Transactional
    public PublicLetterInvitation save(PublicLetterInvitation letterInvitation) {
        return letterInvitationRepository.save(letterInvitation);
    }
}
