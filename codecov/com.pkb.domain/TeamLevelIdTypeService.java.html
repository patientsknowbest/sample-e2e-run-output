package com.pkb.domain;

import com.pkb.authentication.principal.user.AuthenticatedPatient;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authentication.principal.user.AuthenticatedTeamProfessional;
import com.pkb.authentication.principal.user.AuthenticatedUser;
import com.pkb.authz.AuthorizationService;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.datamodel.TeamLevelIdType;
import com.pkb.datamodel.entity.mapper.TeamLevelIdTypeMapper;
import com.pkb.domain.repository.TeamLevelIdTypeRepository;
import com.pkb.entities.pub.PublicTeamLevelIdType;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.pkb.authz.NotFoundException.notFoundTarget;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.Predicates.instanceOf;
import static io.vavr.control.Either.left;

public class TeamLevelIdTypeService {
    private final AuthorizationService authorizationService;
    private final TeamLevelIdTypeRepository repository;
    private final TeamLevelIdTypeMapper teamLevelIdTypeMapper;

    public TeamLevelIdTypeService(AuthorizationService authorizationService,
                                  TeamLevelIdTypeRepository repository,
                                  TeamLevelIdTypeMapper teamLevelIdTypeMapper) {
        this.authorizationService = authorizationService;
        this.repository = repository;
        this.teamLevelIdTypeMapper = teamLevelIdTypeMapper;
    }

    public @NotNull Either<RuntimeException, TeamLevelIdType> findTeamLevelIdType(@NotNull AuthorizationData authorizationData, @NotNull UUID publicId) {
        return authorizationService.authorize(authorizationData)
                .checkFilteredSingleResult(
                        targetMeta -> targetMeta.targetType().isAssignableFrom(TeamLevelIdType.class),
                        () -> findOptionalByPublicIdAndAuthenticatedUser(authorizationData.inputs().authenticatedUser(), publicId))
                .flatMap(teamLevelIdType -> teamLevelIdType.<Either<RuntimeException, TeamLevelIdType>>map(Either::right)
                        .getOrElse(() -> left(notFoundTarget(publicId, TeamLevelIdType.class))));
    }

    @NotNull
    private Option<TeamLevelIdType> findOptionalByPublicIdAndAuthenticatedUser(AuthenticatedUser authenticatedUser, @NotNull UUID publicId) {
        return Match(authenticatedUser).of(
                Case($(instanceOf(AuthenticatedTeamProfessional.class)), teamProfessional ->
                        repository.findTeamLevelIdTypeByPublicIdForTeamProfessional(publicId, teamProfessional.getTeam().getOrgId())
                ),
                Case($(instanceOf(AuthenticatedPatient.class)), patient ->
                        repository.findTeamLevelIdTypeByPublicIdAsPatient(publicId, patient.getId())
                ),
                Case($(instanceOf(AuthenticatedSystemUser.class)), authenticatedSystemUser ->
                        repository.findTeamLevelIdTypeByPublicIdForTeamProfessional(publicId, authenticatedSystemUser.getOrgId())
                ),
                Case($(), () -> Option.<PublicTeamLevelIdType>none())
        ).map(teamLevelIdTypeMapper::entityToDataModel);
    }

}
