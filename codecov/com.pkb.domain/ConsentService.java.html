package com.pkb.domain;

import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

public class ConsentService {
    private final PersonRepository personRepository;
    private final PersonMapper personMapper;
    private final TeamRepository teamRepository;
    private final TeamMapper teamMapper;
    private final PatientConsentRepository patientConsentRepository;
    private final PatientConsentMapper patientConsentMapper;
    private final AuthorizationService authorizationService;

    public ConsentService(PersonRepository personRepository,
                          PersonMapper personMapper,
                          TeamRepository teamRepository,
                          TeamMapper teamMapper,
                          PatientConsentRepository patientConsentRepository,
                          PatientConsentMapper patientConsentMapper,
                          AuthorizationService authorizationService) {
        this.personRepository = personRepository;
        this.personMapper = personMapper;
        this.teamRepository = teamRepository;
        this.teamMapper = teamMapper;
        this.patientConsentRepository = patientConsentRepository;
        this.patientConsentMapper = patientConsentMapper;
        this.authorizationService = authorizationService;
    }

    public @NotNull Either<RuntimeException, Purview> determinePurview(@NotNull AuthorizationData authorizationData) {
        return authorizationService.authorize(authorizationData)
                .checkSingleResult(() -> {
                    long authenticatedPersonId = authorizationData.inputs().requireAuthenticatedPerson().getId();
                    ImmutablePurview.Builder purview = ImmutablePurview.builder();

                    PersonType person = personMapper.entityToDataModel(personRepository.findById(authenticatedPersonId)
                            .orElseThrow());

                    if (Patient.class.isInstance(person)) {
                        purview.patient(Patient.class.cast(person));
                    } else if (Professional.class.isInstance(person)) {
                        Professional professional = Professional.class.cast(person);
                        teamRepository.findTeamOfClinician(professional.getId())
                                .peek(t -> purview.team(teamMapper.entityToDataModel(t)))
                                .onEmpty(() -> purview.professional(professional));
                    }
                    return Tuple.of(purview.build(), person);
                }).flatMap(result -> result._1().isDefined() ?
                        right(result._1()) :
                        left(notFoundForTargetOwner(result._2().getPublicId(), Purview.class)));
    }

    public @NotNull Either<RuntimeException, CriteriaResult<PatientConsent>> findConsents(@NotNull AuthorizationData authorizationData,
                                                                                          UUID patientPublicId) {

        if (authorizationData.inputs().authenticatedSystemUser().isDefined()) {
            return left(new SystemUserTypeNotPermittedException());
        }

        return authorizationService.authorize(authorizationData)
                .checkSingleResult(() -> findAllByPatientPublicId(authorizationData, patientPublicId));
    }

    private CriteriaResult<PatientConsent> findAllByPatientPublicId(
            AuthorizationData authzData,
            @NotNull UUID patientPublicId) {
        var authenticatedUser = authzData.inputs().authenticatedUser();
        List<PatientConsent> patientConsents = Match(authenticatedUser).of(
                Case($(instanceOf(AuthenticatedTeamProfessional.class)),
                        actor -> findForTeam(authzData, actor, patientPublicId)),
                Case($(instanceOf(AuthenticatedIndividualProfessional.class)),
                        actor -> findForIndividualProfessional(actor.getId(), patientPublicId)),
                Case($(instanceOf(AuthenticatedPatient.class)), actor -> {
                    if (actor.isAccessFrozen()) {
                        return API.<PatientConsent>List();
                    } else if (Objects.equals(actor.getPublicId(), patientPublicId)) {
                        return patientConsentRepository.findOwn(patientPublicId)
                                .map(patientConsentMapper::patientConsentToDataModel);
                    } else {
                        return findForCarer(actor.getId(), patientPublicId);
                    }
                }),
                Case($(instanceOf(AuthenticatedSystemUser.class)),
                        actor -> findForTeams(actor, patientPublicId))
        );
        return ImmutableCriteriaResult.<PatientConsent>builder().addAllPage(patientConsents).total(patientConsents.size()).build();
    }

    private Option<PatientConsent> findFor(AccessingEntityType accessingEntityType, long teamOrPersonId, UUID patientId) {
        return findFor(accessingEntityType, List(teamOrPersonId), patientId);
    }

    private Option<PatientConsent> findFor(AccessingEntityType accessingEntityType, List<Long> teamOrPersonIds, UUID patientId) {
        return patientConsentRepository.findFor(accessingEntityType, teamOrPersonIds, List(patientId))
                .headOption().map(patientConsentMapper::patientConsentToDataModel);
    }

    private List<PatientConsent> findForTeam(AuthorizationData authzData,
                                             AuthenticatedTeamProfessional authenticatedTeamProfessional,
                                             UUID patientId) {
        Option<PatientConsent> maybeConsent = findFor(AccessingEntityType.TEAM, 
                authenticatedTeamProfessional.getTeam().getId(), patientId);
        if ((authzData.requireSingleTargetData().actorHasDecryptionAccess() || maybeConsent.isDefined())) {
            return maybeConsent.map(grantedConsent -> findAll(patientId, authenticatedTeamProfessional.getTeam().getId(), grantedConsent))
                    .getOrElse(() -> patientConsentRepository.findAll(patientId, authenticatedTeamProfessional.getTeam().getId(), false, false, false, false)
                        .map(patientConsentMapper::patientConsentToDataModel))
                    .toList();
        } else {
            return maybeConsent.toList();
        }
    }

    private List<PatientConsent> findForTeams(AuthenticatedSystemUser authenticatedSystemUser,
                                              UUID patientId) {
        List<Long> teamIds = List.ofAll(authenticatedSystemUser.getTeamsWithOrg()).map(Identifiable::getId);
        return findFor(AccessingEntityType.TEAM, teamIds, patientId).toList();
    }

    private List<PatientConsent> findForConsentableUser(AccessingEntityType accessingEntityType, long consentableUserId, UUID patientId) {
        Option<PatientConsent> maybeConsent = findFor(accessingEntityType, consentableUserId, patientId);
        return maybeConsent.map(grantedConsent -> {
            if (!grantedConsent.isDischarged() && grantedConsent.isAtLeastOneConsentGranted()) {
                return findAll(patientId, consentableUserId, grantedConsent);
            } else {
                return List(grantedConsent);
            }
        }).getOrElse(List());
    }

    private List<PatientConsent> findAll(UUID patientPublicId, long accessingPersonOrTeam, PatientConsent grantedConsent) {
        return patientConsentRepository.findAll(patientPublicId,
                accessingPersonOrTeam,
                grantedConsent.isGranted(PrivacyFlag.GENERAL),
                grantedConsent.isGranted(PrivacyFlag.MENTAL_HEALTH),
                grantedConsent.isGranted(PrivacyFlag.SEXUAL_HEALTH),
                grantedConsent.isGranted(PrivacyFlag.SOCIAL_CARE))
                .map(patientConsentMapper::patientConsentToDataModel);
    }

    private List<PatientConsent> findForIndividualProfessional(long individualProfessionalId, UUID patientPublicId) {
        return findForConsentableUser(AccessingEntityType.INDIVIDUAL, individualProfessionalId, patientPublicId);
    }

    private List<PatientConsent> findForCarer(long carerId, UUID patientPublicId) {
        return findForConsentableUser(AccessingEntityType.CARER, carerId, patientPublicId);
    }
}
