package com.pkb.domain;

import com.pkb.crypto.AccountPrivateKey;
import com.pkb.crypto.SymmetricKey;
import com.pkb.crypto.dto.AccountKeysDTO;
import com.pkb.crypto.dto.AccountSymmetricKeyDTO;
import com.pkb.datamodel.AccountKeys;
import com.pkb.datamodel.ImmutableAccountKeys;
import com.pkb.entities.core.CoreAccount;
import com.pkb.entities.pub.PublicPKBPerson;
import com.pkb.kms.client.core.Kms;
import com.pkb.kms.shared.representation.SymmetricKeyRequest;
import com.pkb.repository.AccountRepository;
import com.pkb.repository.PersonRepository;
import com.pkb.util.CorrelationIdUtil;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import io.vavr.collection.HashMap;
import io.vavr.collection.Map;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

public class AccountSymmetricKeyService {
    private final AccountRepository accountRepository;
    private final PersonRepository personRepository;
    private final Kms kms;

    private final CorrelationIdUtil correlationIdUtil;

    public AccountSymmetricKeyService(Kms kms, AccountRepository accountRepository, PersonRepository personRepository, CorrelationIdUtil correlationIdUtil) {
        this.kms = kms;
        this.accountRepository = accountRepository;
        this.personRepository = personRepository;
        this.correlationIdUtil = correlationIdUtil;
    }

    public AccountKeys getAccountKeys(long targetAccountId, long actorPersonId) {
        UUID personPublicId = personRepository.findById(actorPersonId).map(PublicPKBPerson::getPublicId).orElseThrow();
        UUID accountPublicId = accountRepository.findById(targetAccountId).map(CoreAccount::getPublicId).orElseThrow();
        return kms.getAccountKeysForSingleAccount(correlationIdUtil.getAsString(), accountPublicId, personPublicId, false)
                .map(dto -> ImmutableAccountKeys.builder().privateKey(AccountPrivateKey.accountPrivateKey(dto.getPrivateKey())).build().withSymmetricKeys(getSymmetricKeyMap(dto)))
                .getOrElseThrow(err -> new RuntimeException("Failed to get keys for account from kms: " + err));
    }

    private Map<Long, SymmetricKey> getSymmetricKeyMap(AccountKeysDTO dto) {
        return dto.getSymmetricKeys().stream().collect(HashMap.collector(AccountSymmetricKeyDTO::getKeyId, SymmetricKey::symmetricKey));
    }

    public Tuple2<Long, SymmetricKey> generateSymmetricKey(long accountId) {
        UUID accountPublicId = accountRepository.findById(accountId).map(CoreAccount::getPublicId).orElseThrow();

        return kms.generateSymmetricKey(correlationIdUtil.getAsString(), new SymmetricKeyRequest(accountPublicId))
                .map(AccountSymmetricKeyService::mapSymmetricKeyFromDto)
                .getOrElseThrow(err -> new RuntimeException("Failed to create new symmetric key using kms:" + err));
    }

    @NotNull
    private static Tuple2<Long, SymmetricKey> mapSymmetricKeyFromDto(AccountSymmetricKeyDTO dto) {
        return Tuple.of(dto.getKeyId(), SymmetricKey.symmetricKey(dto));
    }

    public Tuple2<Long, SymmetricKey> getSymmetricKey(long defaultAccountId) {
        UUID accountPublicId = accountRepository.findById(defaultAccountId).map(CoreAccount::getPublicId).orElseThrow();

        return kms.getSymmetricKey(correlationIdUtil.getAsString(), accountPublicId)
                .map(AccountSymmetricKeyService::mapSymmetricKeyFromDto)
                .getOrElseThrow(err -> new RuntimeException("Failed to gget symmetric key from kms: " + err));

    }



}
