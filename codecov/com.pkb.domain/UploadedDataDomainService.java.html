package com.pkb.domain;

import com.pkb.datamodel.ImmutableUploadedDataMismatches;
import com.pkb.datamodel.UploadedDataMismatches;
import com.pkb.datamodel.user.Person;
import com.pkb.domain.duplicate.person.PersonService;
import com.pkb.entities.enums.UploadedDataStatus;
import com.pkb.repository.RawUploadedDataMismatches;
import com.pkb.repository.UploadedDataRepository;

import java.util.Collection;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;

import static com.pkb.entities.enums.UploadedDataFormat.HL7;

public class UploadedDataDomainService {

    private final UploadedDataRepository uploadedDataRepository;
    private final PersonService personService;

    public UploadedDataDomainService(UploadedDataRepository uploadedDataRepository, PersonService personService) {
        this.uploadedDataRepository = uploadedDataRepository;
        this.personService = personService;
    }

    public List<UploadedDataMismatches> getMismatches(long orgId) {
        List<RawUploadedDataMismatches> mismatches = uploadedDataRepository.getMismatches(orgId, UploadedDataStatus.PATIENT_INFO_MISMATCH, HL7);
        Collection<Long> mismatchPersonIds = mismatches.stream()
                .map(RawUploadedDataMismatches::getPersonId)
                .collect(Collectors.toList());
        Map<Long, Person> personMap = personService.findPersonsByListOfIds(mismatchPersonIds);
        return mismatches.stream().map(
                m -> ImmutableUploadedDataMismatches.uploadedDataMismatches()
                        .id(m.getId())
                        .created(m.getCreated())
                        .destination(m.getDestination())
                        .patientInfoMismatches(m.getPatientInfoMismatches())
                        .personLastName(Optional.ofNullable(personMap.get(m.getPersonId()))
                                .map(person -> person.demographics().getLastName().getOrElse(""))
                                .orElse(""))
                        .personFirstName(Optional.ofNullable(personMap.get(m.getPersonId()))
                                .map(person -> person.demographics().getFirstName().getOrElse(""))
                                .orElse(""))
                        .personId(m.getPersonId())
                        .build())
                .collect(Collectors.toList());
    }
}
