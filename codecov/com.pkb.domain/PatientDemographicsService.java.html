package com.pkb.domain;


import com.pkb.authz.AuthorizationService;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.datamodel.entity.mapper.PersonDemographicsMapper;
import com.pkb.datamodel.user.Demographics;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.criteria.ImmutableCriteriaResult;
import com.pkb.domain.repository.PersonDemographicsRepository;
import com.pkb.entities.enums.ValidNationalId;
import io.vavr.collection.List;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.pkb.authz.NotFoundException.notFoundTarget;
import static com.pkb.entities.enums.UserType.PATIENT;
import static io.vavr.control.Either.left;

public class PatientDemographicsService extends BaseDemographicsService {
    public static final String NO_PATIENT_WITH_ID_EXISTS = "no.patient.with.id.exists";
    
    private final AuthorizationService authorizationService;

    public PatientDemographicsService(AuthorizationService authorizationService,
                                      PersonDemographicsRepository repository,
                                      PersonDemographicsMapper personDemographicsMapper) {
        super(repository, personDemographicsMapper);
        this.authorizationService = authorizationService;
    }

    public @NotNull Either<RuntimeException, Demographics> findPatientDemographics(@NotNull AuthorizationData authorizationData, @NotNull UUID patientId) {
        return authorizationService.authorize(authorizationData)
                .checkSingleResult(() -> findOptionalByPublicIdAndUserType(authorizationData, patientId, PATIENT))
                .flatMap(demographics -> demographics
                        .<Either<RuntimeException, Demographics>>map(Either::right)
                        .getOrElse(() -> left(notFoundTarget(patientId, Demographics.class))));
    }

    public @NotNull Either<RuntimeException, CriteriaResult<Demographics>> findPatientDemographicsByNationalId(
            @NotNull AuthorizationData authorizationData,
            @NotNull ValidNationalId nationalId) {
        return authorizationService.authorize(authorizationData)
                .checkSingleResult(() -> toCriteriaResult(findAllByNationalId(authorizationData, nationalId)));
    }
    
    private static <T> CriteriaResult<T> toCriteriaResult(List<T> l) {
        return ImmutableCriteriaResult.<T>builder()
                .total(l.size())
                .addAllPage(l)
                .build();
    }
}
