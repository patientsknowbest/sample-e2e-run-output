package com.pkb.domain;

import com.pkb.datamodel.ImmutableSciStorePatientCredentials;
import com.pkb.datamodel.SciStorePatientCredentials;
import com.pkb.datamodel.entity.mapper.SciStorePatientCredentialsMapper;
import com.pkb.entities.pub.PublicSciStorePatientCredentials.Status;
import com.pkb.repository.SciStorePatientCredentialsRepository;
import org.jetbrains.annotations.NotNull;

import javax.transaction.Transactional;
import java.util.List;
import java.util.Optional;

public class SciStoreService {

    private final SciStorePatientCredentialsRepository sciStorePatientCredentialsRepository;
    private final SciStorePatientCredentialsMapper sciStorePatientCredentialsMapper;

    public SciStoreService(SciStorePatientCredentialsRepository sciStorePatientCredentialsRepository,
                           SciStorePatientCredentialsMapper sciStorePatientCredentialsMapper) {
        this.sciStorePatientCredentialsRepository = sciStorePatientCredentialsRepository;
        this.sciStorePatientCredentialsMapper = sciStorePatientCredentialsMapper;
    }

    @Transactional
    public void createSciStorePatientCredentials(long pkbPersonId,
                                               @NotNull String sciStorePatientId,
                                               @NotNull String sciStoreSubscriptionId,
                                               @NotNull Status status) {
        SciStorePatientCredentials sciStorePatientCredentials = ImmutableSciStorePatientCredentials.builder()
                .pkbPatientId(pkbPersonId)
                .sciStorePatientId(sciStorePatientId)
                .sciStoreSubscriptionId(sciStoreSubscriptionId)
                .status(status)
                .build();
        sciStorePatientCredentialsRepository.save(sciStorePatientCredentialsMapper.newDataModelToNewEntity(sciStorePatientCredentials));
    }

    @Transactional
    public void updateSciStorePatientCredentials(long pkbPatientId,
                                                 @NotNull String sciStorePatientId,
                                                 @NotNull String sciStoreSubscriptionId,
                                                 @NotNull Status status) {
        sciStorePatientCredentialsRepository.updateSciStorePatientCredentials(pkbPatientId, sciStorePatientId, sciStoreSubscriptionId, status);
    }

    public Optional<SciStorePatientCredentials> findPublicSciStorePatientCredentialsByPkbPatientId(long pkbPersonId) {
        return sciStorePatientCredentialsRepository.findPublicSciStorePatientCredentialsByPkbPatientId(pkbPersonId)
                .map(sciStorePatientCredentialsMapper::entityToDataModel);
    }

    public List<Long> getMismatchedPatientIds(long orgId) {
        return sciStorePatientCredentialsRepository.getMismatchedPatientIds(orgId);
    }
}
