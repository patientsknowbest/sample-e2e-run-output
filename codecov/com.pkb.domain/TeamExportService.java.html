package com.pkb.domain;

import com.pkb.common.config.PhrConfig;
import com.pkb.datamodel.consent.TeamExport;
import com.pkb.datamodel.entity.mapper.TeamExportMapper;
import com.pkb.entities.enums.AccessingEntityType;
import com.pkb.entities.enums.ContactType;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.repository.TeamRepository;
import org.springframework.data.domain.PageRequest;

import java.util.List;
import java.util.Optional;
import java.util.stream.Collectors;

public class TeamExportService {

    private final TeamRepository teamRepository;
    private final TeamExportMapper mapper;
    private final PhrConfig config;

    public TeamExportService(TeamRepository teamRepository, TeamExportMapper mapper, PhrConfig config) {
        this.teamRepository = teamRepository;
        this.mapper = mapper;
        this.config = config;
    }

    public List<TeamExport> getPatientAdditionalConsentsForTeam(long teamId, long orgId, Optional<NationalIdType> teamNationalIdType, int page) {
        return teamRepository
                .getPatientAdditionalConsentsForTeam(teamId, orgId, AccessingEntityType.TEAM,
                        ContactType.EMAIL, PageRequest.of(page, config.getTeamExportBatchSize()))
                .map(t -> mapper.entityToDatamodel(t, teamNationalIdType)).stream()
                .collect(Collectors.toList());
    }
}
