package com.pkb.domain;

import com.pkb.authentication.principal.user.AuthenticatedPatient;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authentication.principal.user.AuthenticatedTeamProfessional;
import com.pkb.authentication.principal.user.AuthenticatedUser;
import com.pkb.authz.AuthorizationService;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.datamodel.OrgLevelIdType;
import com.pkb.datamodel.entity.mapper.OrgLevelIdTypeMapper;
import com.pkb.domain.repository.OrgLevelIdTypeRepository;
import com.pkb.entities.pub.PublicOrgLevelIdType;
import io.vavr.API;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.pkb.authz.NotFoundException.notFoundTarget;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.Predicates.instanceOf;
import static io.vavr.control.Either.left;

public class OrgLevelIdTypeService {
    private final AuthorizationService authorizationService;
    private final OrgLevelIdTypeRepository repository;
    private final OrgLevelIdTypeMapper orgLevelIdTypeMapper;

    public OrgLevelIdTypeService(AuthorizationService authorizationService,
                                 OrgLevelIdTypeRepository repository,
                                 OrgLevelIdTypeMapper orgLevelIdTypeMapper) {
        this.authorizationService = authorizationService;
        this.repository = repository;
        this.orgLevelIdTypeMapper = orgLevelIdTypeMapper;
    }

    public @NotNull Either<RuntimeException, OrgLevelIdType> findOrgLevelIdType(@NotNull AuthorizationData authorizationData, @NotNull UUID publicId) {
        return authorizationService.authorize(authorizationData)
                .checkFilteredSingleResult(
                        targetMeta -> targetMeta.targetType().isAssignableFrom(OrgLevelIdType.class),
                        () -> findOptionalByPublicIdAndAuthenticatedUser(authorizationData.inputs().authenticatedUser(), publicId))
                .flatMap(orgLevelIdType -> orgLevelIdType
                        .<Either<RuntimeException, OrgLevelIdType>>map(Either::right)
                        .getOrElse(() -> left(notFoundTarget(publicId, OrgLevelIdType.class))));
    }

    @NotNull
    private Option<OrgLevelIdType> findOptionalByPublicIdAndAuthenticatedUser(AuthenticatedUser authenticatedUser,
                                                                              UUID publicId) {
        return API.Match(authenticatedUser).of(
                Case($(instanceOf(AuthenticatedTeamProfessional.class)), teamProfessional ->
                        repository.findOrgLevelIdTypeByPublicIdForTeamProfessional(publicId, teamProfessional.getTeam().getOrgId())
                ),
                Case($(instanceOf(AuthenticatedPatient.class)), patient ->
                        repository.findOrgLevelIdTypeByPublicIdForPatient(publicId, patient.getId())
                ),
                Case($(instanceOf(AuthenticatedSystemUser.class)), authenticatedSystemUser ->
                        repository.findOrgLevelIdTypeByPublicIdForTeamProfessional(publicId, authenticatedSystemUser.getOrgId())
                ),
                Case($(), Option::<PublicOrgLevelIdType>none)
        ).map(orgLevelIdTypeMapper::entityToDataModel);
    }
}
