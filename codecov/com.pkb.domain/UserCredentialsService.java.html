package com.pkb.domain;

import com.pkb.crypto.PKBKeyStore;
import com.pkb.crypto.UserPrivateKey;
import com.pkb.crypto.UserPublicKey;
import com.pkb.crypto.dto.UserPrivateKeyDTO;
import com.pkb.crypto.javasecurityutil.CertificateHelper;
import com.pkb.entities.pub.PublicUserCredentials;
import com.pkb.kms.client.core.Kms;
import com.pkb.repository.UserCredentialRepository;
import io.vavr.Tuple3;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.IOException;
import java.security.GeneralSecurityException;
import java.security.KeyFactory;
import java.security.KeyStoreException;
import java.security.PrivateKey;
import java.security.PublicKey;
import java.security.cert.Certificate;
import java.security.cert.X509Certificate;
import java.security.spec.PKCS8EncodedKeySpec;
import java.util.Base64;
import java.util.Enumeration;
import java.util.Optional;
import java.util.UUID;
import java.util.function.Consumer;

import static io.vavr.API.unchecked;
import static java.lang.String.format;

public class UserCredentialsService {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final UserCredentialRepository userCredentialRepository;
    private final CertificateHelper certificateHelper = new CertificateHelper();
    private final Kms kmsClient;

    public UserCredentialsService(UserCredentialRepository userCredentialRepository, Kms kmsClient) {
        this.userCredentialRepository = userCredentialRepository;
        this.kmsClient = kmsClient;
    }

    public boolean entryExists(long userId) {
        return findUserCredential(userId).isPresent();
    }

    public PublicUserCredentials saveCredential(PublicUserCredentials credential) {
        PublicUserCredentials saved = userCredentialRepository.saveAndFlush(credential);
        LOGGER.info("Added {} to DB", credential.getUserId());
        return saved;
    }

    public void deleteCredential(long userId) {
        Optional<PublicUserCredentials> uc = userCredentialRepository.findByUserId(userId);
        if (uc.isPresent()) {
            userCredentialRepository.delete(uc.get());
            userCredentialRepository.flush();
        } else {
            LOGGER.warn("Could not find Tolven user {} credentials in the DB, cannot delete", userId);
        }
    }

    public Optional<PublicUserCredentials> findUserCredential(long userId) {
        return userCredentialRepository.findByUserId(userId);
    }

    @NotNull
    public PublicKey getPublicKey(long userId) throws IOException, GeneralSecurityException {
        return userCredentialRepository.findByUserId(userId)
                .map(uc -> uc.getCertificateBase64() != null ? uc.getCertificateBytes() : null)
                .map(CertificateHelper::getX509Certificate)
                .map(X509Certificate::getPublicKey)
                .orElseThrow(() -> new RuntimeException("Could not get public key for user: " + userId));
    }

    public PublicUserCredentials createCredentials(long userId, char[] password) {
        PKBKeyStore keyStore = certificateHelper.createPKCS12KeyStore(userId, password);
        PublicUserCredentials uc = new PublicUserCredentials();
        uc.setUserId(userId);
        uc.setCertificateBase64(Base64.getEncoder().encodeToString(certificateHelper.getX509CertificateByteArray(keyStore)));
        uc.setPkcs12Base64(Base64.getEncoder().encodeToString(keyStore.toByteArray(password)));
        return uc;
    }

    public void changeUserPassword(long userId, UUID personPublicId, char[] oldPassword, char[] newPassword)
            throws IOException, GeneralSecurityException {
        PublicUserCredentials userCredential = getUserCredential(userId);
        this.changeUserPassword(personPublicId, userCredential, oldPassword, newPassword);
    }

    public boolean authenticate(long userId, char[] password) {
        try {
            return findUserKeyRing(userId, password).isRight();
        } catch (Exception e) {
            LOGGER.warn("Could not fetch user keyring during authentication", e);
            return false;
        }
    }

    public Either<String, Tuple3<UserPrivateKey, UserPublicKey, X509Certificate>> findUserKeyRing(long userId, char[] password) {

        Consumer<Enumeration<String>> multiAliasHandler = (aliases) -> LOGGER.warn("PHR-5284 Multiple aliases for principal: {} in keystore: {}", aliases, userId);

        return Option.ofOptional(userCredentialRepository.findByUserId(userId)).toEither("Could not find credentials for user " + userId)
                .map(userCredentials -> Base64.getDecoder().decode(userCredentials.getPkcs12Base64()))
                .flatMap(userPKCS12 -> CertificateHelper.getKeyStore(userPKCS12, password))
                .flatMap(keyStore -> keyStore.getAlias(multiAliasHandler).flatMap(alias -> keyStore.getKeyRing(alias, password)));
    }

    private void changeUserPassword(UUID personPublicId, PublicUserCredentials userCredential, char[] oldPassword, char[] newPassword)
            throws IOException, GeneralSecurityException {

        // Verify that the public key x509 cert is correctly stored in usercredentials
        X509Certificate userCertificate = CertificateHelper.getX509Certificate(userCredential.getCertificateBytes());
        if (userCertificate == null) {
            throw new RuntimeException(format("No userCertificate found for user=%d", userCredential.getUserId()));
        }

        // Deserialize and load keystore
        byte[] bytes = Base64.getDecoder().decode(userCredential.getPkcs12Base64());
        PKBKeyStore keyStore = CertificateHelper.getKeyStore(bytes, oldPassword)
                .getOrElseThrow(() -> new RuntimeException(format("Could not retrieve userPKCS12 keystore for user=%d", userCredential.getUserId())));
        String alias = userCredential.getUserId().toString();

        // Verify that the public key x509 cert is correctly stored in keystore
        keyStore.findCertificate(alias)
                .orElseThrow(() -> new RuntimeException(format("No certificate found in userPKCS12 keystore for user=%d with alias: %s", userCredential.getUserId(), alias)));

        // Change password
        keyStore.changePassword(alias, oldPassword, newPassword);
        userCredential.setPkcs12Base64(Base64.getEncoder().encodeToString(keyStore.toByteArray(newPassword)));
        this.saveCredential(userCredential);

        // Remove old credentials from KMS cache
        invalidateKmsUserCache(personPublicId);
    }

    public void changeUserPasswordUsingExternalRecovery(@NotNull UUID requestId, @NotNull Long userId, UUID personPublicId, char[] password) throws KeyStoreException {
        PublicUserCredentials userCredential = getUserCredential(userId);

        // Get private key dto from KMS
        UserPrivateKeyDTO userPrivateKeyDTO = kmsClient.recoverCredentials(requestId.toString(), personPublicId)
                .getOrElseThrow(e -> new RuntimeException("Could not retrieve user private key from KMS: " + e.getDescription()));

        // Get certificate from credential
        X509Certificate certificate = CertificateHelper.getX509Certificate(userCredential.getCertificateBytes());
        Certificate[] certificateChain = new Certificate[] { certificate };

        // Convert private key dto
        PrivateKey privateKey = generatePrivateKey(userPrivateKeyDTO);

        // Create new keystore and then overwrite the private key
        PKBKeyStore newKeyStore = certificateHelper.createPKCS12KeyStore(userId, password);
        String alias = String.valueOf(userId);
        newKeyStore.setKeyEntry(alias, privateKey, password, certificateChain);

        // Update credential and save
        userCredential.setPkcs12Base64(Base64.getEncoder().encodeToString(newKeyStore.toByteArray(password)));
        saveCredential(userCredential);
    }

    public PublicUserCredentials overrideCredentials(long userId, UUID patientPublicId, char[] password) {
        PublicUserCredentials newCredentials = createCredentials(userId, password);

        return userCredentialRepository.findByUserId(userId).map(oldCredentials -> {
            oldCredentials.setCertificateBase64(newCredentials.getCertificateBase64());
            oldCredentials.setPkcs12Base64(newCredentials.getPkcs12Base64());
            userCredentialRepository.saveAndFlush(oldCredentials);
            invalidateKmsUserCache(patientPublicId);
            return oldCredentials;
        }).orElseGet(() -> userCredentialRepository.saveAndFlush(newCredentials));
    }

    private PublicUserCredentials getUserCredential(long userId) {
        return this.findUserCredential(userId).orElseThrow(() -> new RuntimeException("Missing credential for user: " + userId));
    }

    private void invalidateKmsUserCache(UUID personPublicId) {
        kmsClient.evictUserKeys(null, personPublicId);
    }

    private PrivateKey generatePrivateKey(UserPrivateKeyDTO userPrivateKeyDTO) {
        return unchecked(() -> {
            com.pkb.crypto.UserPrivateKey pk = com.pkb.crypto.UserPrivateKey.userPrivateKey(userPrivateKeyDTO.getKeyBytes(), userPrivateKeyDTO.getAlgorithm());
            return KeyFactory
                    .getInstance(pk.getAlgorithm())
                    .generatePrivate(new PKCS8EncodedKeySpec(pk.getKeyBytes()));
        }).get();
    }
}
