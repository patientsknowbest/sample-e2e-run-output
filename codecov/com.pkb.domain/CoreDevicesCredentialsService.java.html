package com.pkb.domain;

import com.pkb.datamodel.CoreDevicesCredentials;
import com.pkb.datamodel.entity.mapper.CoreDevicesCredentialsMapper;
import com.pkb.repository.CoreDevicesCredentialsRepository;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.invoke.MethodHandles;
import java.sql.Timestamp;
import java.time.Instant;
import java.util.Optional;

public class CoreDevicesCredentialsService {

    private final CoreDevicesCredentialsRepository coreDevicesCredentialsRepository;
    private final CoreDevicesCredentialsMapper coreDevicesCredentialsMapper;
    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    public CoreDevicesCredentialsService(CoreDevicesCredentialsRepository coreDevicesCredentialsRepository,
                                         CoreDevicesCredentialsMapper coreDevicesCredentialsMapper) {
        this.coreDevicesCredentialsRepository = coreDevicesCredentialsRepository;
        this.coreDevicesCredentialsMapper = coreDevicesCredentialsMapper;
    }


    public void createCoreDevicesCredentials(CoreDevicesCredentials coreDevicesCredentials) {
        coreDevicesCredentialsRepository.save(coreDevicesCredentialsMapper.newDataModelToNewEntity(coreDevicesCredentials));
    }

    public Optional<CoreDevicesCredentials> getCoreDevicesCredentials(long patientId) {
        return coreDevicesCredentialsRepository.findByPatientId(Long.toString(patientId)).stream().findFirst()
                .map(coreDevicesCredentialsMapper::entityToDataModel);
    }

    public Optional<CoreDevicesCredentials> getCoreDevicesCredentialsByCoreDevicesUserId(@NotNull String coreDevicesUserId) {
        return coreDevicesCredentialsRepository.getByCoreDevicesUserId(coreDevicesUserId)
                .map(coreDevicesCredentialsMapper::entityToDataModel);
    }

    /**
     * method should only to be invoked for Validic users which are in status:provisioned
     */
    public Optional<String> deleteCoreDevicesCredentialsIfOlderThan(@NotNull String coreDevicesUserId, @NotNull Instant olderThan) {
       return Optional.ofNullable(coreDevicesCredentialsRepository.deleteCoreDevicesUser(coreDevicesUserId, Timestamp.from(olderThan))) ;
    }


}
