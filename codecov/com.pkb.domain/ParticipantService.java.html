package com.pkb.domain;

import com.pkb.entities.pub.PublicParticipant;
import com.pkb.repository.ParticipantRepository;
import com.pkb.util.DbUtil;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.time.LocalDateTime;
import java.util.Collection;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.UUID;

import static java.util.stream.Collectors.toMap;

public class ParticipantService {

    private final ParticipantRepository participantRepository;

    public ParticipantService(ParticipantRepository participantRepository) {
        this.participantRepository = participantRepository;
    }

    public List<Long> getPublicParticipantPersonIdsByUniqueId(@NotNull UUID uniqueId) {
        return participantRepository.getPublicParticipantPersonIdsByUniqueId(uniqueId);
    }

    public Map<UUID, Long> getTotalParticipantCount(@NotNull Collection<UUID> uniqueIds) {
        return DbUtil.getInStreamOfBatches(uniqueIds, batchOdUniqueIds -> participantRepository.getTotalParticipantCount(batchOdUniqueIds))
                .collect(toMap(k -> k.getGroupField(), v -> v.getValue()));
    }

    public Optional<PublicParticipant> getParticipant(long personId, @NotNull UUID uniqueId, @Nullable Long teamId) {
        return Optional.ofNullable(teamId)
                .map(nonNullTeamId -> participantRepository.findPublicParticipantByPersonIdAndUniqueIdAndTeamId(personId, uniqueId, nonNullTeamId))
                .orElse(participantRepository.findPublicParticipantByPersonIdAndUniqueIdWithUnsetTeam(personId, uniqueId));
    }

    public void saveParticipant(long participantId, @NotNull UUID uniqueId, @NotNull LocalDateTime addedDate, @Nullable Long teamId) {
        PublicParticipant participant = new PublicParticipant();
        participant.setPersonId(participantId);
        participant.setTeamId(teamId);
        participant.setUniqueId(uniqueId);
        participant.setAddedDate(addedDate);
        participantRepository.save(participant);
    }
}
