package com.pkb.domain;

import com.pkb.authentication.principal.team.BaseTeam;
import com.pkb.authentication.principal.user.AuthenticatedIndividualProfessional;
import com.pkb.authentication.principal.user.AuthenticatedPatient;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authentication.principal.user.TeamUser;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.datamodel.entity.mapper.PersonDemographicsMapper;
import com.pkb.datamodel.user.Demographics;
import com.pkb.domain.repository.PersonDemographicsRepository;
import com.pkb.entities.enums.UserType;
import com.pkb.entities.enums.ValidNationalId;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.Predicates.instanceOf;

public class BaseDemographicsService {
    private final PersonDemographicsRepository repository;
    private final PersonDemographicsMapper personDemographicsMapper;

    public BaseDemographicsService(PersonDemographicsRepository repository, PersonDemographicsMapper personDemographicsMapper) {
        this.repository = repository;
        this.personDemographicsMapper = personDemographicsMapper;
    }

    @NotNull
    protected Option<Demographics> findOptionalByPublicIdAndUserType(@NotNull AuthorizationData authorizationData,
                                                                     @NotNull UUID publicId,
                                                                     @NotNull UserType userType) {
        return Match(authorizationData.inputs().authenticatedUser()).of(

                // Patient viewing their own demographics, includes all ids
                Case($(instanceOf(AuthenticatedPatient.class)), patient ->
                        repository.findByPublicIdAndUserTypeWithAllIds(publicId, userType)
                                .map(personDemographicsMapper::entityToDataModel)),

                // Team user can see local ids for their own team/org
                Case($(instanceOf(TeamUser.class)), teamUser ->
                        repository.findByPublicIdAndUserTypeWithAllIds(publicId, userType)
                                .map(pp -> personDemographicsMapper.entityToDataModel(pp, teamUser.getTeam().getId(), teamUser.getTeam().getOrgId()))),

                // Indiv pro can only see national ids
                Case($(instanceOf(AuthenticatedIndividualProfessional.class)), pro ->
                        repository.findByPublicIdAndUserTypeWithNationalIdsNotAccessFrozen(publicId, userType)
                                .map(personDemographicsMapper::entityToDataModel)),

                // System user can see local ids for their own org
                Case($(instanceOf(AuthenticatedSystemUser.class)), authenticatedSystemUser ->
                        repository.findByPublicIdAndUserTypeWithAllIds(publicId, userType)
                                .map(pp -> personDemographicsMapper.entityToDataModel(pp, authenticatedSystemUser.getOrgId()))),

                Case($(), o -> {
                    throw new RuntimeException("No supported principal found");
                }));
    }

    @NotNull
    protected Option<Demographics> findOptionalPractitionerByPublicId(@NotNull AuthorizationData authorizationData,
                                                                      @NotNull UUID publicId) {
        return Match(authorizationData.inputs().authenticatedUser()).of(

                // System user can see local ids for their own org
                Case($(instanceOf(AuthenticatedSystemUser.class)), authenticatedSystemUser ->
                        repository.findPractitionerByPublicIdAndOrgId(publicId,authenticatedSystemUser.getOrgId())
                        .map(pp -> personDemographicsMapper.entityToDataModel(pp, authenticatedSystemUser.getOrgId()))),

                Case($(), o -> {
                    throw new RuntimeException("No supported principal found");
                }));
    }

    @NotNull
    protected Option<Demographics> findActivePractitionerByPublicId(@NotNull AuthorizationData authorizationData,
                                                                    @NotNull UUID publicId) {
        return Match(authorizationData.inputs().authenticatedUser()).of(
                Case($(instanceOf(AuthenticatedSystemUser.class)), authenticatedSystemUser ->
                        repository.findActivePractitionerByPublicIdAndOrgId(publicId,authenticatedSystemUser.getOrgId())
                                .map(pp -> personDemographicsMapper.entityToDataModel(pp, authenticatedSystemUser.getOrgId()))),

                Case($(), o -> {
                    throw new RuntimeException("No supported principal found");
                }));
    }

    @NotNull
    protected List<Demographics> findAllByNationalId(@NotNull AuthorizationData authorizationData, ValidNationalId validNationalId) {
        return Match(authorizationData.inputs().authenticatedUser()).of(
                // Patient viewing their own demographics or carer viewing caree, includes all ids
                Case($(instanceOf(AuthenticatedPatient.class)), patient ->
                        repository.findByNationalIdAsCarerOrSelf(validNationalId, patient.getId())
                                .map(personDemographicsMapper::entityToDataModel)),

                // Team user can see local ids for their own team/org
                Case($(instanceOf(TeamUser.class)), teamUser -> {
                    BaseTeam team = teamUser.getTeam();
                    return repository.findByNationalIdWithTeamAndOrgIds(validNationalId)
                            .map(pp -> personDemographicsMapper.entityToDataModel(pp, team.getId(), team.getOrgId()));
                }),

                // Indiv pro can only see national ids
                Case($(instanceOf(AuthenticatedIndividualProfessional.class)), pro ->
                        repository.findByNationalIdAsIndividualPro(validNationalId, pro.getId())
                                .map(personDemographicsMapper::entityToDataModel)),

                // System user can see local ids for their own org
                Case($(instanceOf(AuthenticatedSystemUser.class)), authenticatedSystemUser ->
                        repository.findByNationalIdWithTeamAndOrgIds(validNationalId)
                                .map(pp -> personDemographicsMapper.entityToDataModel(pp, authenticatedSystemUser.getOrgId()))),

                Case($(), o -> {
                    throw new RuntimeException("No supported principal found");
                }));
    }
}
