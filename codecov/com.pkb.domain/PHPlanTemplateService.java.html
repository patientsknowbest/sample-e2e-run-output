package com.pkb.domain;

import com.pkb.datamodel.PrivacyFlags;
import com.pkb.datamodel.entity.mapper.PHPlanTemplateMapper;
import com.pkb.datamodel.entity.mapper.context.AvoidCyclicReferenceContext;
import com.pkb.datamodel.plantemplate.AutoSavePHPlanTemplate;
import com.pkb.datamodel.plantemplate.ImmutableNewPHPlanTemplate;
import com.pkb.datamodel.plantemplate.NewPHPlanTemplate;
import com.pkb.datamodel.plantemplate.PHPlanTemplate;
import com.pkb.datamodel.plantemplate.TemplateAttachment;
import com.pkb.datamodel.plantemplate.TemplateField;
import com.pkb.entities.enums.TemplateStatus;
import com.pkb.entities.pub.PublicPHPlanTemplate;
import com.pkb.repository.PHPlanTemplateRepository;
import org.springframework.data.jpa.repository.Modifying;
import org.springframework.transaction.annotation.Transactional;

import java.time.Instant;
import java.util.ArrayList;
import java.util.List;
import java.util.Optional;
import java.util.stream.Collectors;

public class PHPlanTemplateService {

    private final PHPlanTemplateRepository phPlanTemplateRepository;
    private final PHPlanTemplateMapper planTemplateMapper;

    public PHPlanTemplateService(PHPlanTemplateRepository phPlanTemplateRepository, PHPlanTemplateMapper planTemplateMapper) {
        this.phPlanTemplateRepository = phPlanTemplateRepository;
        this.planTemplateMapper = planTemplateMapper;
    }

    @Transactional
    public void deleteTemplate(long templateId) {
        phPlanTemplateRepository.updatePlanTemplateStatus(templateId, TemplateStatus.INACTIVE);
    }

    @Transactional
    public List<PHPlanTemplate> getTemplateList(long teamId) {
        return phPlanTemplateRepository.findAllByInstituteIdAndStatusIsNotOrderByName(teamId, TemplateStatus.INACTIVE)
                .stream()
                .map(template -> planTemplateMapper.entityToDatamodel(template, new AvoidCyclicReferenceContext()))
                .collect(Collectors.toList());
    }

    @Transactional
    public Optional<PHPlanTemplate> getPHPlanTemplate(long templateId) {
        return phPlanTemplateRepository
                .findById(templateId)
                .map(template -> planTemplateMapper.entityToDatamodel(template, new AvoidCyclicReferenceContext()));
    }

    @Transactional
    public long save(NewPHPlanTemplate planTemplate) {
        PublicPHPlanTemplate template = planTemplateMapper.datamodelToEntity(planTemplate);
        template.getAttachments().forEach(attachment -> attachment.setPlanTemplate(template));
        template.getMetrics().forEach(metric -> metric.setPlanTemplate(template));
        template.getFields().forEach(field -> field.setPlanTemplate(template));
        return phPlanTemplateRepository
                .save(template)
                .getId();
    }

    @Transactional
    public long save(AutoSavePHPlanTemplate autoSavePHPlanTemplate, long creatorId, long teamId, PrivacyFlags privacyFlags, Instant dateTime) {
        PublicPHPlanTemplate template = planTemplateMapper.autoSaveResultToDatamodel(autoSavePHPlanTemplate, creatorId, teamId, privacyFlags, dateTime);
        template.getAttachments().forEach(attachment -> attachment.setPlanTemplate(template));
        template.getMetrics().forEach(metric -> metric.setPlanTemplate(template));
        template.getFields().forEach(field -> field.setPlanTemplate(template));
        return phPlanTemplateRepository
                .save(template)
                .getId();
    }

    @Modifying
    public long completeTemplate(NewPHPlanTemplate template) {
        return save(ImmutableNewPHPlanTemplate.copyOf(template).withStatus(TemplateStatus.ACTIVE));
    }

    @Transactional
    public List<TemplateAttachment> getTemplateAttachments(long templateId) {
        return phPlanTemplateRepository.findById(templateId)
                .map(template -> planTemplateMapper.entityToDatamodel(template, new AvoidCyclicReferenceContext()))
                .map(PHPlanTemplate::getAttachments)
                .orElse(new ArrayList<>());
    }

    @Transactional
    public List<TemplateField> getTemplateFields(long templateId) {
        return phPlanTemplateRepository.findById(templateId)
                .map(template -> planTemplateMapper.entityToDatamodel(template, new AvoidCyclicReferenceContext()))
                .map(PHPlanTemplate::getFields)
                .orElse(new ArrayList<>());
    }

    @Transactional
    public List<PHPlanTemplate> findAllByInstituteIdAndStatus(long instituteId, TemplateStatus status) {
        return phPlanTemplateRepository.findAllByInstituteIdAndStatus(instituteId, status)
                .stream()
                .map(template -> planTemplateMapper.entityToDatamodel(template, new AvoidCyclicReferenceContext()))
                .collect(Collectors.toList());
    }
}
