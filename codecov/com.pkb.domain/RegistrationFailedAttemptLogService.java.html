package com.pkb.domain;

import com.pkb.datamodel.RegistrationFailedAttemptLogDto;
import com.pkb.datamodel.entity.mapper.RegistrationFailedAttemptLogMapper;
import com.pkb.entities.pub.PublicRegistrationFailedAttemptLog;
import com.pkb.repository.PublicRegistrationFailedAttemptLogRepository;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.transaction.Transactional;
import java.time.Instant;
import java.util.List;
import java.util.UUID;
import java.util.stream.Collectors;

import static java.lang.String.format;

public class RegistrationFailedAttemptLogService {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final PublicRegistrationFailedAttemptLogRepository publicRegistrationFailedAttemptLogRepository;
    private final RegistrationFailedAttemptLogMapper registrationFailedAttemptLogMapper;

    public RegistrationFailedAttemptLogService(PublicRegistrationFailedAttemptLogRepository publicRegistrationFailedAttemptLogRepository,
                                               RegistrationFailedAttemptLogMapper registrationFailedAttemptLogMapper) {
        this.publicRegistrationFailedAttemptLogRepository = publicRegistrationFailedAttemptLogRepository;
        this.registrationFailedAttemptLogMapper = registrationFailedAttemptLogMapper;
    }

    public RegistrationFailedAttemptLogDto save(RegistrationFailedAttemptLogDto dto) {
        LOGGER.warn(format("Registration attempt failed for %s unverified field %s", dto.getPersonPublicId().toString(), dto.getRegistrationField().toString()));
        PublicRegistrationFailedAttemptLog entity = registrationFailedAttemptLogMapper.toPublicRegistrationFailedAttemptLog(dto);
        entity = publicRegistrationFailedAttemptLogRepository.save(entity);
        return registrationFailedAttemptLogMapper.toRegistrationFailedAttemptLogDto(entity);
    }

    public int findCountOfRegistrationFailedAttempts(UUID publicPersonId, PublicRegistrationFailedAttemptLog.RegistrationField registrationField, Instant fromRegistrationTime) {
        return publicRegistrationFailedAttemptLogRepository.findCountOfRegistrationFailedAttempts(publicPersonId, registrationField, fromRegistrationTime);
    }

    public List<RegistrationFailedAttemptLogDto> findFailedAttempts(UUID publicPersonId, PublicRegistrationFailedAttemptLog.RegistrationField registrationField, Instant fromRegistrationTime) {
        return publicRegistrationFailedAttemptLogRepository.findFailedAttempts(publicPersonId, registrationField, fromRegistrationTime)
                .stream().map(registrationFailedAttemptLogMapper::toRegistrationFailedAttemptLogDto)
                .collect(Collectors.toList());
    }

    @Transactional
    public void deleteRegistrationFailedAttempts(UUID publicPersonId, PublicRegistrationFailedAttemptLog.RegistrationField registrationField) {
        LOGGER.info(format("Removing any failed registration attempts for %s for field %s", publicPersonId.toString(), registrationField.toString()));
        publicRegistrationFailedAttemptLogRepository.deleteRegistrationFailedAttempts(publicPersonId, registrationField);
    }
}
