package com.pkb.domain;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.Account;
import com.pkb.datamodel.Identifiable;
import com.pkb.datamodel.PubliclyIdentifiable;
import com.pkb.datamodel.entity.mapper.AccountMapper;
import com.pkb.entities.core.CoreAccount;
import com.pkb.entities.enums.AccountUserStatus;
import com.pkb.entities.pub.PublicPKBPerson;
import com.pkb.repository.AccountRepository;
import com.pkb.repository.AccountUserRepository;
import com.pkb.repository.PersonRepository;
import com.pkb.util.DbUtil;

import javax.transaction.Transactional;
import java.util.Collection;
import java.util.Map;
import java.util.Optional;
import java.util.UUID;
import java.util.stream.Collectors;
import java.util.stream.Stream;

public class AccountService {
    private final AccountRepository repository;
    private final AccountUserRepository accountUserRepository;
    private final DateTimeService dateTimeService;
    private final AccountMapper accountMapper;
    private final PersonRepository personRepository;

    public AccountService(AccountRepository repository,
                          AccountUserRepository accountUserRepository,
                          DateTimeService dateTimeService,
                          AccountMapper accountMapper,
                          PersonRepository personRepository) {
        this.repository = repository;
        this.accountUserRepository = accountUserRepository;
        this.dateTimeService = dateTimeService;
        this.accountMapper = accountMapper;
        this.personRepository = personRepository;
    }

    public Optional<Account> findPatientAccount(UUID personPublicId) {
        // Access control - this method is invoked from a non-authenticated context
        return personRepository.findOptionalByPublicId(personPublicId)
                .filter(person -> person.isPatient())
                .map(PublicPKBPerson::getDefaultAccountId)
                .flatMap(id ->
                        repository.findById(id)
                                .map(accountMapper::entityToDataModel));
    }

    public Optional<UUID> findAccountPublicId(long accountId) {
        return repository.findById(accountId).map(CoreAccount::getPublicId);
    }

    public UUID getAccountPublicId(long accountId) {
        return findAccountPublicId(accountId).orElseThrow(() ->
                new RuntimeException(String.format("Account with id %d has no public id", accountId)));
    }

    @Transactional
    public void revokeAccessToAccountForPerson(UUID personPublicId, long accountId) {
        // Access control - this method is invoked from a non-authenticated context
        personRepository.findOptionalByPublicId(personPublicId)
                .map(PublicPKBPerson::getId)
                .flatMap(personId -> accountUserRepository.findByUserPersonIdAndAccountIdAndStatus(personId, accountId, AccountUserStatus.ACTIVE))
                .ifPresent(au -> {
                    au.setStatus(AccountUserStatus.EXPIRED);
                    au.setEffectiveDate(dateTimeService.now());
                });
    }



    public Map<UUID, Long> findAccountPublicIds(Collection<Long> accountIds) {
        return findAccounts(accountIds)
                .collect(Collectors.toMap(PubliclyIdentifiable::getPublicId, Identifiable::getId));
    }

    private Stream<Account> findAccounts(Collection<Long> accountIds) {
        return DbUtil.getInStreamOfBatches(accountIds, repository::findAllById)
                .map(accountMapper::entityToDataModel);
    }

    public UUID findPersonAccountPublicId(long patientId) {
        return repository.getAccountPublicIdByPersonId(patientId);
    }
}
