package com.pkb.domain;

import com.pkb.authz.AuthorizationService;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.datamodel.entity.mapper.PersonDemographicsMapper;
import com.pkb.datamodel.user.Demographics;
import com.pkb.datamodel.user.Professional;
import com.pkb.domain.repository.PersonDemographicsRepository;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.pkb.authz.NotFoundException.notFoundTarget;
import static com.pkb.entities.enums.UserType.REG_CLINICIAN;
import static io.vavr.control.Either.left;

public class PractitionerDemographicsService extends BaseDemographicsService {
    private final AuthorizationService authorizationService;

    public PractitionerDemographicsService(AuthorizationService authorizationService, PersonDemographicsRepository repository, PersonDemographicsMapper personDemographicsMapper) {
        super(repository, personDemographicsMapper);
        this.authorizationService = authorizationService;
    }

    public @NotNull Either<RuntimeException, Demographics> findPractitionerDemographics(@NotNull AuthorizationData authorizationData,
            UUID practitionerId) {

        return authorizationService.authorize(authorizationData)
                .checkSingleResult(() -> authorizationData.inputs().authenticatedSystemUser()
                        .map($ -> findOptionalPractitionerByPublicId(authorizationData, practitionerId))
                        .getOrElse(() -> findOptionalByPublicIdAndUserType(authorizationData, practitionerId, REG_CLINICIAN)))
                .flatMap(demographics -> demographics
                        .<Either<RuntimeException, Demographics>> map(Either::right)
                        .getOrElse(() -> left(notFoundTarget(practitionerId, Professional.class))));
    }

    public @NotNull Either<RuntimeException, Demographics> findActivePractitionerDemographicsForSystemUser(@NotNull AuthorizationData authorizationData,
            UUID practitionerId) {

        return authorizationService.authorize(authorizationData)
                .checkSingleResult(() -> {
                    authorizationData.inputs().requireAuthenticatedSystemUser();
                    return findActivePractitionerByPublicId(authorizationData, practitionerId);
                })
                .flatMap(demographics -> demographics
                        .<Either<RuntimeException, Demographics>> map(Either::right)
                        .getOrElse(() -> left(notFoundTarget(practitionerId, Professional.class))));
    }
}
