    public Optional<Hl7Partner> findPartnerByCode(@NotNull String code) {
        return partnerRepository.findOptionalByCode(code).map(hl7PartnerMapper::entityToDataModel);
    }

    public Map<String, Hl7Partner> findPartnerByCodes(@NotNull Collection<String> codes) {
        return findPartnerByCodesStream(codes).collect(toMap(hp -> hp.getCode(), identity()));
    }

    public Stream<Hl7Partner> findPartnerByCodesStream(@NotNull Collection<String> codes) {
        return DbUtil.getInStreamOfBatches(codes, partnerRepository::findByCodeIn)
                .map(hl7PartnerMapper::entityToDataModel);
    }

    public Optional<Hl7Partner> findPartnerById(@NotNull Long id) {
        return partnerRepository.findOptionalById(id).map(hl7PartnerMapper::entityToDataModel);
    }
