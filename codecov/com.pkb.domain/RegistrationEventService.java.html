package com.pkb.domain;

import com.pkb.datamodel.RegistrationEventExisting;
import com.pkb.datamodel.RegistrationEventNew;
import com.pkb.datamodel.entity.mapper.RegistrationEventMapper;
import com.pkb.entities.pub.PublicRegistrationEvent;
import com.pkb.repository.RegistrationEventRepository;

import java.util.List;
import java.util.stream.Collectors;

public class RegistrationEventService {

    private final RegistrationEventRepository registrationEventRepository;
    private final RegistrationEventMapper registrationEventMapper;

    public RegistrationEventService(RegistrationEventRepository registrationEventRepository,
                                    RegistrationEventMapper registrationEventMapper) {
        this.registrationEventRepository = registrationEventRepository;
        this.registrationEventMapper = registrationEventMapper;
    }

    public List<RegistrationEventExisting> findAllByPersonId(long personId) {
        return registrationEventRepository.findEntitiesByPersonId(personId).stream().map(registrationEventMapper::entityToDataModel).collect(Collectors.toList());
    }

    public RegistrationEventExisting create(RegistrationEventNew newRegistrationEvent) {
        PublicRegistrationEvent newEntity = registrationEventMapper.dataModelToEntity(newRegistrationEvent);
        newEntity = registrationEventRepository.save(newEntity);
        return registrationEventMapper.entityToDataModel(newEntity);
    }
}
