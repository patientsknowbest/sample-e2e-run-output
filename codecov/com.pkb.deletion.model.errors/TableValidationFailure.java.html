package com.pkb.deletion.model.errors;

import com.fasterxml.jackson.annotation.JsonTypeName;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import org.immutables.value.Value;

import java.util.List;
import java.util.Objects;
import java.util.UUID;
import java.util.stream.Collectors;

@Value.Immutable
@JsonSerialize(as= ImmutableTableValidationFailure.class)
@JsonDeserialize(as=ImmutableTableValidationFailure.class)
@JsonTypeName("TableValidationFailure")
public  interface TableValidationFailure extends ValidationFailure {

    UUID getOperationId();
    String getMessagePrefix();
    String getTableName();
    List<String> getProblemIds();

    @Override
    @Value.Derived
    default String getMessage() {
        return "Validation failure for delete operation [" + getOperationId() + "] on table [" + getTableName() + "]: " + getMessagePrefix() + " Problem ids:" +
                getProblemIds().stream().map(Objects::toString).collect(Collectors.joining("\n", "\n", ""));
    }


    static <T> TableValidationFailure of(UUID requestId, UUID operationId, String messagePrefix, String tableName, List<T> problemIds) {
        return ImmutableTableValidationFailure.builder()
                .requestId(requestId)
                .operationId(operationId)
                .messagePrefix(messagePrefix)
                .tableName(tableName)
                .problemIds(problemIds.stream().map(Object::toString).collect(Collectors.toList()))
                .build();
    }

}
