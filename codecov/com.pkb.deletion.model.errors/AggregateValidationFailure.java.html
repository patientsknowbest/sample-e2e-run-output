package com.pkb.deletion.model.errors;

import com.fasterxml.jackson.annotation.JsonTypeName;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import org.immutables.value.Value;

import java.util.List;

/**
 * It makes life simpler if return types can be Either&lt;Error, Success&gt; rather
 * than the error type being a list. But in the case of validation errors it's also useful
 * to get as many of them at once rather than a drip-feed. This composite error type allows
 * us to return multiple validation errors as a single error object.
 */
@Value.Immutable
@JsonSerialize(as= ImmutableAggregateValidationFailure.class)
@JsonDeserialize(as=ImmutableAggregateValidationFailure.class)
@JsonTypeName("AggregateValidationFailure")
public interface AggregateValidationFailure extends ValidationFailure {
    List<ValidationFailure> getValidationFailures();

    @Override
    @Value.Derived
    default String getMessage() {
        return "There were validation failures.";
    }
}
