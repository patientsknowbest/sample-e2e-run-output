package com.pkb.deletion.model.errors;

import com.fasterxml.jackson.annotation.JsonTypeName;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.pkb.deletion.model.DeleteRequestStatus;
import org.immutables.value.Value;

import java.util.Map;

@Value.Immutable
@JsonSerialize(as= ImmutableIllegalOperationError.class)
@JsonDeserialize(as=ImmutableIllegalOperationError.class)
@JsonTypeName("IllegalOperationError")
public interface IllegalOperationError extends DeletionServiceError{

    String getOperationToPeform();
    DeleteRequestStatus getCurrentState();
    Map<DeleteRequestStatus, DeleteRequestStatus> getValidTransitions();

    @Override
    @Value.Derived
    default String getMessage() {
        return String.format("Illegal operation: attempted to perform operation %1$s while in state [%2$s]; but valid transitions were: %3$s", getOperationToPeform(), getCurrentState(), getValidTransitions());
    }

    static IllegalOperationError of(String operationToPerform, DeleteRequestStatus currentState, Map<DeleteRequestStatus, DeleteRequestStatus> validTransitions) {
        return ImmutableIllegalOperationError.builder().currentState(currentState).validTransitions(validTransitions).operationToPeform(operationToPerform).build();
    }
}
