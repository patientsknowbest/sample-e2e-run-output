package com.pkb.fhir.migration.transformation;

import com.pkb.entities.enums.MenuDataType;
import com.pkb.fhir.common.decrypt.DecryptedEhrData;
import com.pkb.fhir.common.enums.TransformedEhrDataStatus;
import com.pkb.fhir.common.exception.MigrationException;
import com.pkb.fhir.common.retrieve.DataTypeBasedEhrDataRetriever;
import com.pkb.fhir.common.transform.BaseEhrDataTransformer;
import com.pkb.fhir.common.transform.TransformableDto;
import com.pkb.fhir.migration.migrators.MenuDataMigrator;

import java.util.Collection;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

/**
 * For migration purposes, take a single piece of already decrypted menu_data and transform it to a TransformedEhrData
 */
public class MigrationDecryptedEhrDataTransformer {
    private final DataTypeBasedEhrDataRetriever ehrDataRetriever;
    private final Map<MenuDataType, BaseEhrDataTransformer> transformerMap;

    public MigrationDecryptedEhrDataTransformer(DataTypeBasedEhrDataRetriever ehrDataRetriever,
                                                Map<MenuDataType, BaseEhrDataTransformer> transformerMap) {
        this.ehrDataRetriever = ehrDataRetriever;
        this.transformerMap = transformerMap;
    }

    /**
     * Used for processing a single menu data point ad-hoc. {@link MenuDataMigrator#createResourceFromEncryptedEhrData}
     * We call this to avoid persisting the result which we don't need for the sync runner.
     * It is only a temporary solution until we do a proper refactoring around the migration flow.
     */
    public TransformedEhrData process(DecryptedEhrData decryptedEhrData) {
        List<? extends TransformableDto> data = ehrDataRetriever.retrieveEhrDataDto(Map.of(decryptedEhrData.getId(), decryptedEhrData));
        List<TransformedEhrData> transformationResult = transformData(data);

        if (transformationResult.isEmpty()) {
            throw new MigrationException("Transformation failed. Cause: transformer returned no result");
        } else if (transformationResult.get(0).isFailure()) {
            throw new MigrationException("Transformation failed. Cause: " + transformationResult.get(0).getError());
        } else {
            return transformationResult.get(0);
        }
    }

    private List<TransformedEhrData> transformData(Collection<? extends TransformableDto> data) {
        return data.stream().map(this::transform).collect(Collectors.toList());
    }

    public TransformedEhrData transform(TransformableDto item) {
        TransformedEhrData result = new TransformedEhrData();
        try {
            String transformedContent = transformerMap.get(item.getMenuDataType()).transformToFhirJson(item);
            result.setContent(transformedContent);
            result.setStatus(TransformedEhrDataStatus.TRANSFORMED);
        } catch (Exception e) {
            result.setStatus(TransformedEhrDataStatus.TRANSFORM_FAILED);
            result.setError(e.getMessage());
        }
        return result;
    }
}
