package com.pkb.service.encounter;

import com.pkb.encounter.entity.Message;
import com.pkb.encounter.entity.Message.MessageType;
import com.pkb.entities.enums.EncounterClass;
import com.pkb.user.entity.PKBPerson;
import org.apache.commons.lang3.StringUtils;

import java.text.MessageFormat;
import java.time.format.DateTimeFormatter;
import java.util.Locale;
import java.util.ResourceBundle;
import java.util.TimeZone;

/**
 * For generating text to populate subject and content for
 * messages that are ADT Encounter Messages.
 */
public class EncounterMessageHelper {

    private static final ResourceBundle adtMessagesBundle = ResourceBundle.getBundle("com.pkb.service.encounter.AdtMessages");
    private static final ResourceBundle encounterClassMessagesBundle = ResourceBundle
            .getBundle("com.pkb.service.encounter.EncounterClassMessages");

    private static final DateTimeFormatter ADT_DATE_FORMATTER;

    static {
        ADT_DATE_FORMATTER = DateTimeFormatter.ofPattern(adtMessagesBundle.getString("message.format.date_and_time")).withLocale(Locale.UK);
    }

    /**
     * Generates text based on property template defined in the resource bundle for the messageType
     * with the standard set of placeholders (patient title, first name, last name, location desc, message date)
     * 
     * @param message
     *            The message object for which we construct the text to describe the event
     * @param patient
     *            The patient object associated with this message
     * @param utcOffset
     *            The UTC Offset in the location of this event on the event date (messageTimestamp)
     * @return String The message text that describes this event.
     */
    public static String makeEncounterMessageContent(Message message, PKBPerson patient, String utcOffset) {

        return MessageFormat.format(
                adtMessagesBundle.getString(message.getMessageType() + ".content"),
                patient.getTitle(),
                patient.getFirstName(),
                patient.getLastName(),
                StringUtils.isBlank(message.getLocation().getDescription()) ? adtMessagesBundle.getString("location.undefined")
                        : message.getLocation().getDescription(),
                formatMessageTimestamp(message, utcOffset));
    }

    public static String makeEncounterMessageContentA01(Message message, PKBPerson patient, String utcOffset) {
        return makeEncounterMessageContent(message, patient, utcOffset);
    }

    public static String makeEncounterMessageContentA02(Message message, PKBPerson patient, String utcOffset) {
        return makeEncounterMessageContent(message, patient, utcOffset);
    }

    public static String makeEncounterMessageContentA03(Message message, PKBPerson patient, String utcOffset) {
        return makeEncounterMessageContent(message, patient, utcOffset);
    }

    public static String makeEncounterMessageContentA08(Message message, PKBPerson patient, String utcOffset) {
        // XXX What more information could be added? include which fields actually changed?
        return MessageFormat.format(
                adtMessagesBundle.getString(MessageType.PATIENT_INFORMATION_UPDATE + ".content"),
                patient.getTitle(),
                patient.getFirstName(),
                patient.getLastName());
    }

    /**
     * Message content text for A05 and A14
     */
    public static String makeEncounterMessageContentForScheduledAdmit(Message message, PKBPerson patient, String utcOffset) {
        return MessageFormat.format(
                adtMessagesBundle.getString(message.getMessageType() + ".content"),
                formatMessageTimestamp(message, utcOffset),
                StringUtils.isBlank(message.getLocation().getDescription()) ? adtMessagesBundle.getString("location.undefined")
                        : message.getLocation().getDescription());
    }

    private static String formatMessageTimestamp(Message message, String utcOffset) {
        if (message.getMessageTimestamp() == null) {
            return null;
        }
        DateTimeFormatter formatter = ADT_DATE_FORMATTER.withZone(TimeZone.getTimeZone(formatUTCOffset(utcOffset)).toZoneId());
        return formatter.format(message.getMessageTimestamp());
    }

    /*
     * All encounter messages have the same subject text construction based on Encounter Class.
     * We can implement specific subject text based on message type if it becomes a requirement. 
     */
    public static String makeEncounterMessageSubject(EncounterClass encounterClass) {
        return encounterClassMessagesBundle.getString(encounterClass + ".title");
    }

    public static String makeEncounterAppointmentSubject(EncounterClass encounterClass) {
        return encounterClassMessagesBundle.getString(encounterClass + ".title");
    }

    /**
     * Returns a UTC Offset formatted for use as a Java TimeZone ID.
     * Example: input="+0100" output="GMT+01:00"
     * 
     * @param utcOffset
     * @return
     */
    public static String formatUTCOffset(String utcOffset) {
        return "GMT" + utcOffset.subSequence(0, 3) + ":" + utcOffset.subSequence(3, 5);
    }

}
