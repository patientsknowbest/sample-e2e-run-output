package com.pkb.pubsub.aggregator.service;

import com.fasterxml.jackson.core.JsonProcessingException;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.pubsub.aggregator.entity.PublicAggregatableRequest;
import com.pkb.pubsub.aggregator.entity.PublicAggregatableResponse;
import com.pkb.pubsub.aggregator.models.AggregatableRequest;
import com.pkb.pubsub.aggregator.models.AggregatableResponse;
import com.pkb.pubsub.aggregator.models.AggregateResponse;
import com.pkb.pubsub.aggregator.models.AggregationDefinition;
import com.pkb.pubsub.aggregator.repository.AggregatableRequestRepository;
import org.apache.camel.Processor;
import org.apache.camel.ProducerTemplate;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.scheduling.concurrent.ThreadPoolTaskScheduler;
import org.springframework.stereotype.Service;
import org.springframework.transaction.PlatformTransactionManager;
import org.springframework.transaction.support.TransactionTemplate;

import java.time.Instant;
import java.util.List;
import java.util.UUID;
import java.util.function.BiConsumer;
import java.util.stream.Collectors;

import static com.pkb.pubsub.aggregator.models.AggregateResponse.Status.INCOMPLETE;
import static java.lang.invoke.MethodHandles.lookup;

/**
 * This service helps to aggregate responses to camel/pubsub messages when several services may need to respond.
 * It listens for responses on a given pubsub topic, and saves them to the database. When responses have been
 * received from all the required services, it issues an aggregate response to a specified camel endpoint (usually
 * another pubsub topic)
 *
 * To use this service, you need some sort of request message that implements the AggregatableRequest interface,
 * and then the responses that are returned from processing services should implement AggregatableResponse. Finally,
 * there should be an AggregatedResponse class that corresponds to the AggregatableResponse class. These are then
*  all wired together using the AggregationDefinition objects, which define the classes, topics and subscriptions that
 *  are used for each set of messages.
 *
 *  This module is an auto-configuring Spring Boot module. You should just include it in a Spring Boot app and define
 *  the AggregationDefinition objects as beans in your context and it will Just Work.
 *
 *  Note that this class (via PublicAggregationRequest/Response) relies on the prior creation of the relevant database
 *  tables. Eventually I might incorporate some automatic LB scripts that will run when this module is included, but for
 *  the time being these need to be created independently.
 */
@Service
public class PubSubMessageAggregatorService extends AbstractNamespaceAwareRouteBuilder  {

    private static final Logger LOGGER = LoggerFactory.getLogger(lookup().lookupClass());
    private final ProducerTemplate pubSubProducer;
    private final List<AggregationDefinition<?,?,?>> aggregationDefinitions;
    private final ObjectMapper objectMapper;

    private final AggregatableRequestRepository aggregatableRequestRepository;

    private final PubSubNamespaceService pubSubNamespaceService;

    private final DateTimeService dateTimeService;
    private final TransactionTemplate transactionTemplate;

    private final ThreadPoolTaskScheduler timeoutExecutor;

    public PubSubMessageAggregatorService(
            ProducerTemplate pubSubProducer,
            List<AggregationDefinition<?,?,?>> aggregationDefinitions,
            ObjectMapper objectMapper,
            AggregatableRequestRepository aggregatableRequestRepository,
            PubSubNamespaceService pubSubNamespaceService,
            DateTimeService dateTimeService,
            PlatformTransactionManager platformTransactionManager,
            ThreadPoolTaskScheduler timeoutExecutor

    ) {
        this.pubSubProducer = pubSubProducer;
        this.aggregationDefinitions = aggregationDefinitions;
        this.objectMapper = objectMapper;
        this.aggregatableRequestRepository = aggregatableRequestRepository;
        this.pubSubNamespaceService = pubSubNamespaceService;
        this.dateTimeService = dateTimeService;
        this.transactionTemplate = new TransactionTemplate(platformTransactionManager);
        this.timeoutExecutor = timeoutExecutor;
    }


    @SuppressWarnings("unchecked")
    @Override
    public void configure() {
        aggregationDefinitions.forEach(definition -> makeProcessor(definition).registerRoutes());
    }

    @Override
    public @NotNull PubSubNamespaceService camelNamespaceService() {
        return pubSubNamespaceService;
    }

    @Override
    public boolean throwExceptionForNamespaceMismatch() {
        return true;
    }

    /**
     * This inner class encapsulates all the processing that needs to be done for a specific AggregationDefinition.
     * It sets up the required Camel routes, and handles processing both the request and response messages.
     */
    private class AggregatableProcessorImpl<R extends AggregatableResponse<R, AR>, AR extends AggregateResponse<AR, R>, Req extends AggregatableRequest<AR, R>>   {

        private final AggregationDefinition<R, AR, Req> aggregationDefinition;
        private AggregatableProcessorImpl(AggregationDefinition<R, AR, Req> aggregationDefinition) {
            this.aggregationDefinition = aggregationDefinition;
        }

        private void registerRoutes() {
            from(aggregationDefinition.getInboundResponseEndpoint()).process(getProcessor(aggregationDefinition.getResponseSerialisationClass(), this::handleResponse));
            from(aggregationDefinition.getInboundRequestEndpoint()).process(getProcessor(aggregationDefinition.getRequestSerialisationClass(), this::handleRequest));
        }


        private <U> Processor getProcessor(Class<U> serialisationClass, BiConsumer<U, String> handlerMethod) {
            return exchange -> {
               U message =  exchange.getMessage().getBody(serialisationClass);
                LOGGER.debug("Received message: {}", message);
                String json = objectMapper.writerFor(serialisationClass).writeValueAsString(message);
                handlerMethod.accept(message, json);
            };
        }

        private void handleRequest(Req message, String json)  {
            doWithRequestLock(message.getAggregatableRequestId(), () -> {
                aggregatableRequestRepository.insertAggregatableRequest(message.getAggregatableRequestId(), json, dateTimeService.now());
                if (!checkCompletionAndSendAggregateResponse(message.getAggregatableRequestId(), false)) {
                    registerTimeout(message);
                }
            });
            LOGGER.debug("Request object written for request id: {}", message.getAggregatableRequestId());
        }

        private void handleResponse(R message, String json) {
            LOGGER.debug("Saving response for request {} responding service: {}. Message json: {} ", message.getAggregatableRequestId(), message.getRespondingService(), json);


            doWithRequestLock(message.getAggregatableRequestId(), () -> {
                aggregatableRequestRepository.insertAggregatableResponse(message.getAggregatableRequestId(), message.getRespondingService(), json, dateTimeService.now());
                LOGGER.debug("Searching for Request object for request id: {}", message.getAggregatableRequestId());
                checkCompletionAndSendAggregateResponse(message.getAggregatableRequestId(), false);
            });
            LOGGER.debug("Response save transaction committed for message: {} ", message);

        }

        private AR getAggregateResponse(PublicAggregatableRequest dbMessageRecord, boolean timesUp) {
            Req request = extractOriginalMessage(dbMessageRecord);
            return request.getAggregateResponse(dbMessageRecord.getResponses().stream().map(this::extractOriginalMessage).collect(Collectors.toList()), aggregationDefinition.getRequiredServiceNames(), timesUp);
        }


        private boolean checkCompletionAndSendAggregateResponse(UUID requestId, boolean timesUp) {
            return aggregatableRequestRepository.findById(requestId)
                    .map(dbMessageRecord -> {
                        if (dbMessageRecord.getStatus() != INCOMPLETE) {
                            LOGGER.debug("Request {} is already marked as: {}, not bothering to check further", requestId, dbMessageRecord.getStatus());
                            return true;
                        }

                        AR aggregate = getAggregateResponse(dbMessageRecord, timesUp);
                        LOGGER.debug("Aggregate response: {}", aggregate);
                        if(aggregate.getStatus() != INCOMPLETE) {
                            dbMessageRecord.setStatus(aggregate.getStatus());
                            LOGGER.info("Aggregate message complete or timed out, sending to pubsub: {}",  aggregate);
                            pubSubProducer.sendBody(aggregationDefinition.getAggregateResponseEndpoint(), aggregate);
                        }

                        return dbMessageRecord.getStatus() != INCOMPLETE;
                    }).orElseGet(() -> {
                        LOGGER.debug("Request not found {}", requestId);
                        return false;
                    });
        }

        // The advisory lock ensures that we get a consistent view of the list of services that have responded (no lost updates)
        private void doWithRequestLock(UUID requestId, Runnable action) {
            transactionTemplate.executeWithoutResult(status -> {
                aggregatableRequestRepository.acquireLock(requestId.hashCode());
                action.run();
            });
        }

        private R extractOriginalMessage(PublicAggregatableResponse dbMessageRecord) {
            try {
                return objectMapper.readValue(dbMessageRecord.getMessageJson(), aggregationDefinition.getResponseSerialisationClass());
            } catch (JsonProcessingException e) {
                throw new RuntimeException(e);
            }
        }

        private Req extractOriginalMessage(PublicAggregatableRequest dbMessageRecord) {
            try {
                return objectMapper.readValue(dbMessageRecord.getMessageJson(), aggregationDefinition.getRequestSerialisationClass());
            } catch (JsonProcessingException e) {
                throw new RuntimeException(e);
            }
        }

        private void registerTimeout(Req request) {
            var requestId = request.getAggregatableRequestId();
            timeoutExecutor.schedule(() ->
            {
                LOGGER.debug("Executing timeout handler for request {}", requestId);
                doWithRequestLock(requestId, () -> checkCompletionAndSendAggregateResponse(requestId, true));
            }, Instant.now().plus(aggregationDefinition.getTimeoutCalculator().apply(request)));
        }
    }



    private <R extends AggregatableResponse<R, AR>, AR extends AggregateResponse<AR, R>, Req extends AggregatableRequest<AR, R>> AggregatableProcessorImpl<R, AR, Req> makeProcessor(AggregationDefinition<R, AR, Req> definition) {
        return new AggregatableProcessorImpl<>(definition);
    }

}
