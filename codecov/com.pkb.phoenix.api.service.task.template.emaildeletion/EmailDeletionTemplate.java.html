package com.pkb.phoenix.api.service.task.template.emaildeletion;

import static com.pkb.phoenix.api.service.task.TaskService.TASK_INSTANCE;
import static com.pkb.phoenix.api.service.task.template.TaskTemplateCode.EMAIL_DEL;
import static com.pkb.phoenix.api.service.tasktype.ValidationUtil.validateEmail;
import static com.pkb.phoenix.api.service.tasktype.ValidationUtil.validateFreshDeskId;
import static reactor.core.publisher.Mono.just;

public final class EmailDeletionTemplate extends AbstractPostProcessingTaskTemplate<EmailDeletionParameters, EmailDeletionResult, FdEmailDeletionResult> {
    public static final String OUTBOUND_REQUEST_EMAIL_DELETION = "direct:outboundRequestEmailDeletion";

    //We need to provide an AbstractPubSubTaskTemplate which knows how to validate and dispatch the task to redis and will be passed in as a delegate to this template:
    private static final class BaseEmailDeletionTemplate extends AbstractPubSubTaskTemplate<EmailDeletionParameters, EmailDeletionTaskRequest, EmailDeletionTaskResponse, EmailDeletionResult> {
        BaseEmailDeletionTemplate(ReactiveRedisTemplate<String, String> redis,
                                  ProducerTemplate producer,
                                  DateTimeService dateTimeService, ObjectReader json) {
            super(EMAIL_DEL, EmailDeletionParameters.class, new EmailDeletionTranslator(dateTimeService), redis, producer, OUTBOUND_REQUEST_EMAIL_DELETION, json);
        }

        @Override
        public List<String> validate(EmailDeletionParameters parameters) {
            List<String> errors = new ArrayList<>();
            validateEmail(parameters.email()).ifPresent(errors::add);
            validateFreshDeskId(parameters.fdId()).ifPresent(errors::add);
            return errors;
        }
    }

    private final VelocityEngine templateEngine;
    private final FreshDeskService freshDeskService;

    public EmailDeletionTemplate(ReactiveRedisTemplate<String, String> redis,
                                 ProducerTemplate producer,
                                 VelocityEngine templateEngine,
                                 FreshDeskService freshDeskService,
                                 DateTimeService dateTimeService,
                                 ObjectReader json) {
        super(new BaseEmailDeletionTemplate(redis, producer, dateTimeService, json), json);
        this.templateEngine = templateEngine;
        this.freshDeskService = freshDeskService;
    }

    @Override
    protected Mono<FdEmailDeletionResult> postProcess(EmailDeletionResult result, ContextView context) {
        var meta = context.<Meta>get(TaskService.META);
        var parameters = context.<Task<EmailDeletionParameters>>get(TASK_INSTANCE).getParameters();

        return meta.session().user().fdApiKey()
                .map(ImmutableApiKey::of)
                .map(apiKey -> {
                    if (result.outcome().equalsIgnoreCase("SUCCESS")) {
                        return freshDeskService.addNote(parameters.fdId(), renderMessage(parameters.email()), apiKey)
                                        .then(freshDeskService.closeTicket(parameters.fdId(), apiKey))
                                .map($1 -> "Done");
                    }

                    return just("Skipped");
                })
                .orElseGet(() -> just("Skipped"))
                .onErrorResume($ -> just("Failed"))
                .map(status -> ImmutableFdEmailDeletionResult.builder()
                        .outcome(result.outcome())
                        .fdStatus(status)
                        .build());
    }

    private String renderMessage(String email) {
        var template = templateEngine.getTemplate("emailDeletedTemplate.vm");

        try (var writer = new StringWriter()) {
            var context = new VelocityContext();

            context.put("email", email);

            template.merge(context, writer);
            writer.flush();

            return writer.toString();
        } catch (IOException e) {
            throw new IllegalStateException(e);
        }
    }
}
