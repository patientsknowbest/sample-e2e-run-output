package com.pkb.phoenix.api.service.task.template.emaildeletion;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.phoenix.api.service.task.TaskService;
import com.pkb.phoenix.api.service.task.template.Translator;
import com.pkb.phoenix.repository.task.projection.Task;
import com.pkb.phoenix.shared.structures.request.EmailDeletionTaskRequest;
import com.pkb.phoenix.shared.structures.request.ImmutableEmailDeletionTaskRequest;
import com.pkb.phoenix.shared.structures.response.EmailDeletionTaskResponse;
import reactor.util.context.ContextView;

import java.util.function.Function;

import static java.util.Optional.ofNullable;

public class EmailDeletionTranslator implements Translator<EmailDeletionParameters, EmailDeletionTaskRequest, EmailDeletionTaskResponse, EmailDeletionResult> {
    private final DateTimeService dateTimeService;

    public EmailDeletionTranslator(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    @Override
    public Function<Task<EmailDeletionParameters>, EmailDeletionTaskRequest> toRequest(ContextView context) {
        return task -> {
            EmailDeletionParameters emailDeletionParameters = task.getParameters();

            return ImmutableEmailDeletionTaskRequest.builder()
                    .requestId(context.get(TaskService.TASK_ID).toString())
                    .freshDeskId(emailDeletionParameters.fdId())
                    .taskId(task.getId())
                    .privateId(emailDeletionParameters.privateId())
                    .email(emailDeletionParameters.email())
                    .date(emailDeletionParameters.date() == null ? dateTimeService.nowLocalDateTime().toString() : emailDeletionParameters.date())
                    .build();
        };
    }

    @Override
    public Function<EmailDeletionTaskResponse, EmailDeletionResult> fromResponse(ContextView context) {
        return response ->
                ImmutableEmailDeletionResult.builder()
                        .outcome(ofNullable(response.results()).orElse("Not Found"))
                        .build();
        
    }

    @Override
    public Class<EmailDeletionTaskResponse> responseType() {
        return EmailDeletionTaskResponse.class;
    }
}
