package com.pkb.sso.emis;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.entities.pub.PublicEMISSSOApplicationProvenance;
import com.pkb.entities.pub.PublicEMISSSOOrgProvenance;
import com.pkb.entities.pub.PublicEMISSSORequestLog;
import com.pkb.entities.pub.PublicEMISSSOUserProvenance;
import uk.co.e_mis.integration.portalsdk.entities.CtPersonName;
import uk.co.e_mis.integration.portalsdk.entities.SessionRequest;
import uk.co.e_mis.integration.portalsdk.entities.SessionRequest.Provenance.Application;
import uk.co.e_mis.integration.portalsdk.entities.SessionRequest.Provenance.Organisation;
import uk.co.e_mis.integration.portalsdk.entities.SessionRequest.Provenance.User;

import java.util.stream.Collectors;

public class SessionRequestConverter {

    public static final String SEPARATOR = " ";

    private SessionRequest emisSessionRequest;

    private String emisSSOUsername;

    private final DateTimeService dateTimeService;

    public SessionRequestConverter(SessionRequest sessionRequest, String emisSSOUsername, DateTimeService dateTimeService) {
        this.emisSessionRequest = sessionRequest;
        this.emisSSOUsername = emisSSOUsername;
        this.dateTimeService = dateTimeService;
    }

    public PublicEMISSSORequestLog getEMISSSORequestLog() {
        PublicEMISSSORequestLog emisssoRequestLog = new PublicEMISSSORequestLog(emisSessionRequest.getRequestId(), dateTimeService.now(), emisSSOUsername);
        emisssoRequestLog.setApplicationProvenance(getApplicationProvenance());
        emisssoRequestLog.setUserProvenance(getUserProvenance());
        emisssoRequestLog.setOrgProvenance(getOrgProvenance());
        emisssoRequestLog.setEmisSSOUsername(emisSSOUsername);
        return emisssoRequestLog;
    }

    public PublicEMISSSOOrgProvenance getOrgProvenance() {
        Organisation organisation = emisSessionRequest.getProvenance().getOrganisation();
        return new PublicEMISSSOOrgProvenance(getOrgIdentifiers(), organisation.getName());
    }

    public String getOrgIdentifiers() {
        return emisSessionRequest.getProvenance().getOrganisation().getIdentifier().stream().map(Organisation.Identifier::getValue).collect(Collectors.joining(SEPARATOR));
    }

    public PublicEMISSSOApplicationProvenance getApplicationProvenance() {
        Application application = emisSessionRequest.getProvenance().getApplication();
        return new PublicEMISSSOApplicationProvenance(application.getName(),
                application.getVersion());
    }

    public PublicEMISSSOUserProvenance getUserProvenance() {
        User user = emisSessionRequest.getProvenance().getUser();
        return new PublicEMISSSOUserProvenance(getPersonName(), (String) user.getApplicationDisplayName(),
                "" + user.getEmisUserId());
    }

    public String getPersonName() {
        CtPersonName ctPersonName = emisSessionRequest.getProvenance().getUser().getPersonName();
        return ctPersonName.getTitle() + SEPARATOR + ctPersonName.getGivenName() + SEPARATOR + ctPersonName.getFamilyName();
    }
}
