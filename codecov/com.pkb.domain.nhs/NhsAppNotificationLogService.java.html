package com.pkb.domain.nhs;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.ImmutableNhsAppNotificationLog;
import com.pkb.datamodel.NhsAppNotificationLog;
import com.pkb.datamodel.entity.mapper.NhsAppNotificationLogMapper;
import com.pkb.entities.enums.ValidNationalId;
import com.pkb.entities.nhs.PublicNhsAppNotificationLog;
import com.pkb.repository.NhsAppNotificationLogRepository;
import org.jetbrains.annotations.NotNull;

import javax.transaction.Transactional;
import java.time.Instant;
import java.time.format.DateTimeFormatter;
import java.util.Optional;

import static java.time.ZoneOffset.UTC;

public class NhsAppNotificationLogService {

    private static final DateTimeFormatter DATEFORMAT = DateTimeFormatter.ISO_INSTANT.withZone(UTC);

    private final DateTimeService dateTimeService;
    private final NhsAppNotificationLogRepository nhsNotificationLogRepository;
    private final NhsAppNotificationLogMapper nhsNotificationLogMapper;

    public NhsAppNotificationLogService(DateTimeService dateTimeService, NhsAppNotificationLogRepository nhsNotificationLogRepository, NhsAppNotificationLogMapper nhsNotificationLogMapper) {
        this.dateTimeService = dateTimeService;
        this.nhsNotificationLogRepository = nhsNotificationLogRepository;
        this.nhsNotificationLogMapper = nhsNotificationLogMapper;
    }

    @Transactional
    public void reportSuccess(@NotNull ValidNationalId targetNhsNumber, @NotNull String notificationTrigger, @NotNull String message, @NotNull String requestId, @NotNull Instant created) {
        NhsAppNotificationLog entry = ImmutableNhsAppNotificationLog.builder()
                .targetNhsNumber(targetNhsNumber)
                .notificationTrigger(notificationTrigger)
                .message(message)
                .requestId(requestId)
                .created(created)
                .processed(dateTimeService.now())
                .success(true)
                .errorMessage(Optional.empty())
                .build();

        PublicNhsAppNotificationLog notificationLog = nhsNotificationLogMapper.dataModelToEntity(entry);
        nhsNotificationLogRepository.save(notificationLog);
    }

    @Transactional
    public void reportFailure(@NotNull ValidNationalId targetNhsNumber, @NotNull String notificationTrigger, @NotNull String message, @NotNull String requestId, @NotNull Instant created, @NotNull String errorMessage) {
        NhsAppNotificationLog entry = ImmutableNhsAppNotificationLog.builder()
                .targetNhsNumber(targetNhsNumber)
                .notificationTrigger(notificationTrigger)
                .message(message)
                .requestId(requestId)
                .created(created)
                .processed(dateTimeService.now())
                .success(false)
                .errorMessage(Optional.of(errorMessage))
                .build();

        nhsNotificationLogRepository.save(nhsNotificationLogMapper.dataModelToEntity(entry));
    }
}
