package com.pkb.messagehasher.configuration;

import com.pkb.common.config.ConfigStorage;
import com.pkb.common.config.ImmutableRawConfigStorage;
import com.pkb.common.datamigrator.Html4XmlUnescaper;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.common.datetime.DefaultDateTimeService;
import com.pkb.common.ehrdata.serialization.EHRDataDeserializer;
import com.pkb.crypto.decryptionhelper.DecryptionHelper;
import com.pkb.crypto.decryptionhelper.provider.accountkeys.AccountKeysProviderFactory;
import com.pkb.kms.client.core.Kms;
import com.pkb.kms.client.core.KmsClientConfig;
import com.pkb.kms.retrofit.RetrofitKms;
import com.pkb.messagehasher.Message;
import de.mkammerer.argon2.Argon2Advanced;
import de.mkammerer.argon2.Argon2Factory;
import io.micrometer.core.instrument.binder.okhttp3.OkHttpMetricsEventListener;
import okhttp3.OkHttpClient;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import static io.micrometer.core.instrument.Metrics.globalRegistry;

@Configuration
public class AppInjectionConfig {

    @Autowired
    KmsConfig kmsConfig;

    @Bean
    public Html4XmlUnescaper html4XmlUnescaper() {
        return new Html4XmlUnescaper();
    }

    @Bean
    public Argon2Advanced argon2Advanced() {
        return Argon2Factory.createAdvanced(Argon2Factory.Argon2Types.ARGON2d);
    }

    @Bean
    public EHRDataDeserializer ehrDataDeserializer() {
        return new EHRDataDeserializer() {
        };
    }

    @Bean
    public DateTimeService dateTimeService() {
        return new DefaultDateTimeService();
    }

    @Bean
    public DecryptionHelper decryptionHelper() {
        return new DecryptionHelper();
    }

    @Bean
    public AccountKeysProviderFactory accountKeysProviderFactory() {
        return new AccountKeysProviderFactory<Message>();
    }

    @Bean
    public Kms kmsClient() {
        OkHttpClient httpClient = new OkHttpClient.Builder()
                .eventListener(OkHttpMetricsEventListener.builder(globalRegistry, "okhttp.requests").build())
                .build();

        return
                new RetrofitKms(httpClient,
                        new KmsClientConfig() {
                            @Override
                            public ConfigStorage getConfigStorage() {
                                return ImmutableRawConfigStorage.EMPTY;
                            }
                            @Override
                            public String getBaseURL() {
                                return kmsConfig.getBaseUrl();
                            }
                            @Override
                            public String getClientId() {
                                return "messagehasher";
                            }
                        });
    }
}
