package com.pkb.messagehasher.configuration;

import de.mkammerer.argon2.Argon2;
import de.mkammerer.argon2.Argon2Helper;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.PropertySource;

@Configuration
@PropertySource("classpath:hash.properties")
@ConfigurationProperties(prefix = "hash")
//TODO: make this stack specific, hashconfigdev.properties, hashconfigprod.properties? - (spring profiles)
public class HashConfig {

    /**
     * If we've got several messageHash applications running, they need to split the work up between themselves.
     * Value should start at 0 and sequentially increase to 1-countInstances
     */
    private int clientInstance;

    /**
     * The total number of message hashers that work should be divvied up between.
     */
    private int countInstances;

    /**
     * Whether hash processing is enabled. Can disable this initially during config to find out the recommended number of iterations.
     */
    private boolean processorEnabled;

    /**
     * Whether the scheduler should be enabled to trigger the processor
     */
    private boolean schedulerEnabled;

    /**
     * The query to retrieve accounts to process will limit by this number
     */
    private int accountRetrieveSize;

    /**
     * The query to retrieve messages to process will limit by this number
     */
    private int messageRetrieveSize;

    /**
     * Whether we run a test to figure out how many hashing iterations are required to match our requested config
     */
    private boolean iterationFinderEnabled;

    /**
     * Maximum number of milliseconds hashing should take. See {@link Argon2Helper#findIterations(de.mkammerer.argon2.Argon2, long, int, int)}
     */
    private int iterationFinderMaxMillis;

    /**
     * Target memory usage. See {@link Argon2#hash(int, int, int, char[])}.
     */
    private int memoryUsage;

    /**
     * Threading. See {@link Argon2#hash(int, int, int, char[])}.
     */
    private int parallelism;

    /**
     * Number of iterations. See {@link Argon2#hash(int, int, int, char[])}.
     */
    private int iterations;

    /**
     * A temporary hack to check for differences in messages across participants
     */
    private boolean tempDifferencesCheckEnabled;

    /**
     * Delay between hashing runs.
     */
    private int runDelayMillis;

    public int getClientInstance() {
        return clientInstance;
    }

    public void setClientInstance(int clientInstance) {
        this.clientInstance = clientInstance;
    }

    public int getCountInstances() {
        return countInstances;
    }

    public void setCountInstances(int countInstances) {
        this.countInstances = countInstances;
    }

    public boolean isProcessorEnabled() {
        return processorEnabled;
    }

    public void setProcessorEnabled(boolean processorEnabled) {
        this.processorEnabled = processorEnabled;
    }

    public boolean isSchedulerEnabled() {
        return schedulerEnabled;
    }

    public void setSchedulerEnabled(boolean schedulerEnabled) {
        this.schedulerEnabled = schedulerEnabled;
    }

    public int getIterationFinderMaxMillis() {
        return iterationFinderMaxMillis;
    }

    public void setIterationFinderMaxMillis(int iterationFinderMaxMillis) {
        this.iterationFinderMaxMillis = iterationFinderMaxMillis;
    }

    public int getMemoryUsage() {
        return memoryUsage;
    }

    public void setMemoryUsage(int memoryUsage) {
        this.memoryUsage = memoryUsage;
    }

    public int getParallelism() {
        return parallelism;
    }

    public void setParallelism(int parallelism) {
        this.parallelism = parallelism;
    }

    public int getIterations() {
        return iterations;
    }

    public void setIterations(int iterations) {
        this.iterations = iterations;
    }

    public int getRunDelayMillis() {
        return runDelayMillis;
    }

    public void setRunDelayMillis(int runDelayMillis) {
        this.runDelayMillis = runDelayMillis;
    }

    public boolean isIterationFinderEnabled() {
        return iterationFinderEnabled;
    }

    public void setIterationFinderEnabled(boolean iterationFinderEnabled) {
        this.iterationFinderEnabled = iterationFinderEnabled;
    }

    public int getAccountRetrieveSize() {
        return accountRetrieveSize;
    }

    public void setAccountRetrieveSize(int accountRetrieveSize) {
        this.accountRetrieveSize = accountRetrieveSize;
    }

    public int getMessageRetrieveSize() {
        return messageRetrieveSize;
    }

    public void setMessageRetrieveSize(int messageRetrieveSize) {
        this.messageRetrieveSize = messageRetrieveSize;
    }

    public boolean isTempDifferencesCheckEnabled() {
        return tempDifferencesCheckEnabled;
    }

    public void setTempDifferencesCheckEnabled(boolean tempDifferencesCheckEnabled) {
        this.tempDifferencesCheckEnabled = tempDifferencesCheckEnabled;
    }
}
