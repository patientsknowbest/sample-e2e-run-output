package com.pkb.entities.messaging;

import com.pkb.entities.embeddables.EncryptedDatapoint;
import com.pkb.entities.enums.Route;
import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import java.time.Instant;
import java.util.UUID;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

@Entity
@Table(name = "conversation", schema = "messaging")
public class Conversation {
    @Id
    @GenericGenerator(name = "conversation_id_seq_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "messaging"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "conversation_id_seq"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "conversation_id_seq_gen")
    @Column(name = "id", nullable = false)
    private long id;

    @Column(name = "patient_id", nullable = true)
    private Long patientId;

    @Column(name = "public_id", nullable = false)
    @Type(type = "pg-uuid")
    private UUID publicId;

    @Column(name = "created_date", nullable = false)
    private Instant createdDate;

    @Column(name = "updated_date", nullable = true)
    private Instant updatedDate;

    @Column(name = "route", nullable = false)
    @Enumerated(EnumType.STRING)
    @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
    private Route route;

    @Column(name = "sourcepersonid", nullable = true)
    private Long sourcePersonId;

    @Column(name = "sourceteamid", nullable = true)
    private Long sourceTeamId;

    @Column(name = "sourceorgid", nullable = true)
    private Long sourceOrgId;

    @Column(name = "connectingorgid", nullable = true)
    private Long connectingOrgId;

    @Column(name = "hl7partnerid", nullable = true)
    private Long hl7PartnerId;

    @Embedded
    private EncryptedDatapoint encryptedDatapoint;

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public Long getPatientId() {
        return patientId;
    }

    public void setPatientId(Long patientId) {
        this.patientId = patientId;
    }

    public UUID getPublicId() {
        return publicId;
    }

    public void setPublicId(UUID publicId) {
        this.publicId = publicId;
    }

    public Instant getCreatedDate() {
        return createdDate;
    }

    public void setCreatedDate(Instant createdDate) {
        this.createdDate = createdDate;
    }

    public Instant getUpdatedDate() {
        return updatedDate;
    }

    public void setUpdatedDate(Instant updatedDate) {
        this.updatedDate = updatedDate;
    }

    public Route getRoute() {
        return route;
    }

    public void setRoute(Route route) {
        this.route = route;
    }

    public Long getSourcePersonId() {
        return sourcePersonId;
    }

    public void setSourcePersonId(Long sourcepersonid) {
        this.sourcePersonId = sourcepersonid;
    }

    public Long getSourceTeamId() {
        return sourceTeamId;
    }

    public void setSourceTeamId(Long sourceteamid) {
        this.sourceTeamId = sourceteamid;
    }

    public Long getSourceOrgId() {
        return sourceOrgId;
    }

    public void setSourceOrgId(Long sourceorgid) {
        this.sourceOrgId = sourceorgid;
    }

    public Long getConnectingOrgId() {
        return connectingOrgId;
    }

    public void setConnectingOrgId(Long connectingorgid) {
        this.connectingOrgId = connectingorgid;
    }

    public Long getHl7PartnerId() {
        return hl7PartnerId;
    }

    public void setHl7PartnerId(Long hl7Partnerid) {
        this.hl7PartnerId = hl7Partnerid;
    }

    public EncryptedDatapoint getEncryptedDatapoint() {
        return encryptedDatapoint;
    }

    public void setEncryptedDatapoint(EncryptedDatapoint encryptedDatapoint) {
        this.encryptedDatapoint = encryptedDatapoint;
    }

}
