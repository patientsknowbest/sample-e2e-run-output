package config;

import com.pkb.service.notification.nhs.NhsNotificationRouteBuilder;
import org.apache.camel.Produce;
import org.apache.camel.ProducerTemplate;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class InternalMessagingConfig {

    @Produce(NhsNotificationRouteBuilder.QUEUE_NHS_NOTIFICATION)
    private ProducerTemplate nhsNotificationProducer;

    @Bean(name = "NhsNotificationProducer")
    public ProducerTemplate getNhsNotificationProducer() {
        return nhsNotificationProducer;
    }
}
