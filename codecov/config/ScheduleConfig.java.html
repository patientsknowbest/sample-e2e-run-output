package config;

import config.scedule.EmisSchedulingConfig;
import config.scedule.ExpiredConsentRemoverConfig;
import config.scedule.PollForSciStoreNotificationsConfig;
import config.scedule.SendChildBirthdayEmailConfig;
import config.scedule.SendSymptomsEmailConfig;
import config.scedule.SendUnreadEmailConfig;
import config.scedule.SyncNewSciStorePatientsConfig;
import org.quartz.Trigger;
import org.springframework.context.ApplicationContext;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Import;
import org.springframework.scheduling.annotation.EnableScheduling;
import org.springframework.scheduling.quartz.SchedulerFactoryBean;
import org.springframework.scheduling.quartz.SpringBeanJobFactory;

import java.util.Collection;

@Configuration
@Import({
        SendSymptomsEmailConfig.class,
        SendChildBirthdayEmailConfig.class,
        SendUnreadEmailConfig.class,
        ExpiredConsentRemoverConfig.class,
        PollForSciStoreNotificationsConfig.class,
        SyncNewSciStorePatientsConfig.class,
        EmisSchedulingConfig.class
})
@EnableScheduling
public class ScheduleConfig {

    @Bean
    public SpringBeanJobFactory springBeanJobFactory(ApplicationContext applicationContext) {
        AutoWiringSpringBeanJobFactory jobFactory = new AutoWiringSpringBeanJobFactory();
        jobFactory.setApplicationContext(applicationContext);
        return jobFactory;
    }

    // TODO: should we prefer an explicit list here (just to make sure that we control what gets scheduled)?
    @Bean
    public SchedulerFactoryBean schedulerFactoryBean(SpringBeanJobFactory jobFactory, Collection<Trigger> triggers) {
        SchedulerFactoryBean factoryBean = new SchedulerFactoryBean();
        factoryBean.setTriggers(triggers.toArray(new Trigger[triggers.size()]));
        factoryBean.setJobFactory(jobFactory);
        return factoryBean;
    }
}
