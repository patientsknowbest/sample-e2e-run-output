package config;

import com.pkb.api.internal.messaging.PKBPersonToMessagingPKBPersonDTOConverter;
import com.pkb.common.cdi.configuration.OkHttpClientHolder;
import com.pkb.common.config.ConfigStorage;
import com.pkb.common.config.PkbPluginConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.common.testcontrol.config.ITestControlServiceConfig;
import com.pkb.common.testcontrol.config.TestControlServiceConfig;
import com.pkb.common.testcontrol.services.DefaultPubSubNamespaceService;
import com.pkb.common.testcontrol.services.PubSubNamespaceService;
import com.pkb.common.testlogging.DetailLoggingProvider;
import com.pkb.kms.client.core.Kms;
import com.pkb.kms.client.core.KmsClientConfig;
import com.pkb.kms.retrofit.RetrofitKms;
import com.pkb.user.entity.PKBPersonHelper;
import com.pkb.user.entity.PKBPersonHelperImpl;
import com.pkb.util.CorrelationIdUtil;
import io.micrometer.core.instrument.Clock;
import io.micrometer.core.instrument.Metrics;
import io.micrometer.prometheus.PrometheusConfig;
import io.micrometer.prometheus.PrometheusMeterRegistry;
import io.prometheus.client.CollectorRegistry;
import io.prometheus.client.guava.cache.CacheMetricsCollector;
import org.apache.camel.CamelContext;
import org.apache.camel.component.google.pubsub.GooglePubsubComponent;
import org.apache.commons.lang3.StringUtils;

import javax.annotation.PostConstruct;
import javax.enterprise.context.ApplicationScoped;
import javax.enterprise.inject.Produces;
import javax.persistence.EntityManager;
import javax.persistence.PersistenceContext;
import java.util.Optional;

import static java.util.Collections.emptySet;

@ApplicationScoped
public class PluginConfiguration {

    @PersistenceContext
    private EntityManager entityManager;

    @PostConstruct
    public void addPrometheusMicrometerRegistry() {
        Metrics.addRegistry(new PrometheusMeterRegistry(PrometheusConfig.DEFAULT, CollectorRegistry.defaultRegistry, Clock.SYSTEM));
    }

    @Produces
    @ApplicationScoped
    public PKBPersonHelper pkbPersonHelper(DateTimeService dateTimeService) {
        return new PKBPersonHelperImpl(dateTimeService);
    }

    @Produces
    @ApplicationScoped
    public CacheMetricsCollector cacheMetricsCollector() {
        return new CacheMetricsCollector().register();
    }

    @Produces
    @ApplicationScoped
    public CorrelationIdUtil correlationIdUtil() {
        return new CorrelationIdUtil();
    }

    @Produces
    public Kms kmsClient(OkHttpClientHolder httpClient, KmsClientConfig kmsConfig) {
        return new RetrofitKms(httpClient.getOkHttpClient(), kmsConfig);
    }

    @Produces
    @ApplicationScoped
    PubSubNamespaceService getNamespaceService(CamelContext ctx) {
        return new DefaultPubSubNamespaceService(ctx);
    }

    @Produces
    @ApplicationScoped
    ITestControlServiceConfig getTestSupportServiceConfig(PkbPluginConfig config,
                                                          DateTimeService dateTimeService,
                                                          DetailLoggingProvider testLoggingService,
                                                          ConfigStorage configStorage,
                                                          PubSubNamespaceService namespaceService,
                                                          CamelContext camelContext) {

        GooglePubsubComponent component = (GooglePubsubComponent) camelContext.getComponent("google-pubsub");
        component.setPublisherCacheTimeout(0);
        if (StringUtils.isNotBlank(config.getPubSubEmulatorEndpoint())) {
            component.setEndpoint(config.getPubSubEmulatorEndpoint());
            component.setAuthenticate(false);
        }
        return new TestControlServiceConfig(config.isTestControlEnabled(),
                config.isTestControlRegistrationEnabled(),
                config.getTestControlAppName() + "Cdi", // To distinguish from the Spring one in the same application.
                config.getTestControlCallbackUrl(),
                config.getTestControlUrl(),
                dateTimeService,
                configStorage,
                emptySet(),
                testLoggingService,
                namespaceService,
                Optional.empty()); // PHR database clears itself 
    }
    @Produces
    @ApplicationScoped
    public PKBPersonToMessagingPKBPersonDTOConverter getPKBPersonConverter(PKBPersonHelper personHelper) {
        return new PKBPersonToMessagingPKBPersonDTOConverter(personHelper);
    }
}
