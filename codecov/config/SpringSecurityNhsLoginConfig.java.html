package config;

import com.pkb.security.BaseAuthenticationDetails;
import com.pkb.security.ImmutableNhsLoginAuthenticationDetails;
import com.pkb.security.LoginAttemptAuditor;
import com.pkb.security.NhsLoginAuthenticationExceptionMappingAuthenticationFailureHandler;
import com.pkb.security.NhsLoginAuthenticationProvider;
import com.pkb.service.nhslogin.impl.NhsLoginRegistrationManager;
import com.pkb.service.oauth2.NhsLoginService;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.team.TeamManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.servlets.filter.PreAuthenticationSuccessHandler;
import com.pkb.util.AppIdentityUtil;
import com.pkb.util.LoggedInUserUtil;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.security.authentication.AuthenticationProvider;
import org.springframework.security.authentication.ProviderManager;
import org.springframework.security.core.Authentication;
import org.springframework.security.web.authentication.AuthenticationFailureHandler;
import org.springframework.security.web.authentication.AuthenticationSuccessHandler;
import org.springframework.security.web.authentication.UsernamePasswordAuthenticationFilter;
import org.springframework.security.web.util.matcher.AntPathRequestMatcher;

import javax.servlet.ServletException;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.servlet.http.HttpServletResponseWrapper;
import java.io.IOException;
import java.net.URLDecoder;

import static java.nio.charset.StandardCharsets.UTF_8;

@Configuration
public class SpringSecurityNhsLoginConfig {

    @Bean
    public AuthenticationProvider nhsLoginAuthenticationProvider(
            NhsLoginService nhsLoginService,
            LoginAttemptAuditor.Factory factory,
            UserManager userManager,
            TeamManager teamManager,
            PatientConsentManager patientConsentManager,
            NhsLoginRegistrationManager nhsLoginRegistrationManager) {
        return new NhsLoginAuthenticationProvider(nhsLoginService, factory, userManager, teamManager,
                patientConsentManager, nhsLoginRegistrationManager);
    }

    @Bean
    public AuthenticationSuccessHandler nhsPreAuthenticationSuccessHandler(LoggedInUserUtil loggedInUserUtil) {
        PreAuthenticationSuccessHandler preAuthenticationSuccessHandler = new NhsLoginAuthenticationSuccessHandler(loggedInUserUtil);
        preAuthenticationSuccessHandler.setDefaultTargetUrl("/auth/dashboard.action");
        return preAuthenticationSuccessHandler;
    }

    @Bean
    public AuthenticationFailureHandler nhsPreAuthenticationFailureHandler() {
        return new NhsLoginAuthenticationExceptionMappingAuthenticationFailureHandler();
    }

    @Bean
    public UsernamePasswordAuthenticationFilter nhsLoginAuthenticationFilter(
            @Qualifier("nhsPreAuthenticationSuccessHandler") AuthenticationSuccessHandler nhsPreAuthenticationSuccessHandler,
            @Qualifier("nhsPreAuthenticationFailureHandler") AuthenticationFailureHandler nhsPreAuthenticationFailureHandler,
            ProviderManager providerManager,
            AppIdentityUtil appIdentityUtil) {

        UsernamePasswordAuthenticationFilter filter = new UsernamePasswordAuthenticationFilter();
        filter.setRequiresAuthenticationRequestMatcher(new AntPathRequestMatcher("/nhsLogin", "POST"));
        filter.setAuthenticationSuccessHandler(nhsPreAuthenticationSuccessHandler);
        filter.setAuthenticationFailureHandler(nhsPreAuthenticationFailureHandler);
        filter.setAuthenticationManager(providerManager);
        filter.setUsernameParameter("info.email");
        filter.setPasswordParameter("info.secretBase64");
        filter.setAuthenticationDetailsSource(BaseAuthenticationDetails.getAuthenticationDetailsSource(ImmutableNhsLoginAuthenticationDetails::builder, appIdentityUtil));
        return filter;
    }

    private class NhsLoginAuthenticationSuccessHandler extends PreAuthenticationSuccessHandler {

        NhsLoginAuthenticationSuccessHandler(LoggedInUserUtil loggedInUserUtil) {
            super(loggedInUserUtil);
        }

        @Override
        public void onAuthenticationSuccess(HttpServletRequest request, HttpServletResponse response, Authentication authentication) throws ServletException, IOException {
            String redirectionPath = request.getParameter("info.path");
            if (redirectionPath != null) {
                response.sendRedirect(URLDecoder.decode(redirectionPath, UTF_8));
            }

            HttpServletResponseWrapper wrappedResponse = new HttpServletResponseWrapper(response) {
                @Override
                public String encodeRedirectURL(String url) {
                    return url;
                }

                @Override
                public String encodeURL(String url) {
                    return url;
                }
            };

            super.onAuthenticationSuccess(request, wrappedResponse, authentication);
        }
    }

}
