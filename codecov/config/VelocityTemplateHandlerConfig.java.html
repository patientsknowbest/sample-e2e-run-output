package config;

import com.google.common.collect.ImmutableMap;
import com.pkb.common.config.PhrConfig;
import com.pkb.service.emailmessage.sender.impl.VelocityTemplateHandler;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class VelocityTemplateHandlerConfig {
    @Bean
    public VelocityTemplateHandler velocityTemplateHandler(PhrConfig config) {
        return new VelocityTemplateHandler(ImmutableMap.of(
                "resource.loader", "class",
                "class.resource.loader.class", "org.apache.velocity.runtime.resource.loader.ClasspathResourceLoader"), config);
    }
}
