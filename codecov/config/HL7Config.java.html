package config;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.domain.OrgNetworkService;
import com.pkb.reports.A19PatientStatusService;
import com.pkb.service.calendar.impl.CalendarManager;
import com.pkb.service.dataupload.hl7.HL7ApiHelper;
import com.pkb.service.dataupload.hl7.HL7PatientService;
import com.pkb.service.dataupload.hl7.HL7SoftMatchService;
import com.pkb.service.dataupload.hl7.HL7TeamService;
import com.pkb.service.dataupload.hl7.processor.MdmT02Processor;
import com.pkb.service.dataupload.hl7.processor.MdmT11Processor;
import com.pkb.service.dataupload.hl7.processor.QryA19Processor;
import com.pkb.service.dataupload.hl7.processor.SiuProcessor;
import com.pkb.service.dataupload.impl.DataUploadManager;
import com.pkb.service.dataupload.processor.UploadedDataService;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.encounter.impl.tolven.EncounterManager;
import com.pkb.service.invitation.impl.InvitationManager;
import com.pkb.service.notification.impl.tolven.INotificationManager;
import com.pkb.service.notification.nhs.NhsAppNotificationManager;
import com.pkb.service.reference.ReferenceDataManager;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.user.impl.PersonContactManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.user.entity.PKBPersonHelper;
import com.pkb.util.PKBServiceUtil;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class HL7Config {

    @Bean
    QryA19Processor qryA19Processor(TeamUserManager teamUserManager, A19PatientStatusService a19PatientStatusService, PhrConfig config, UUIDProvider uuidProvider) {
        return new QryA19Processor(teamUserManager, a19PatientStatusService, config, uuidProvider);
    }

    @Bean
    MdmT11Processor mdmT11Processor(DataUploadManager dataUploadManager, EncounterManager encounterManager, HL7SoftMatchService hl7SoftMatchService) {
        return new MdmT11Processor(dataUploadManager, encounterManager, hl7SoftMatchService);
    }

    @Bean
    MdmT02Processor mdmT02Processor(DataUploadManager dataUploadManager, EncounterManager encounterManager, UserManager userManager, UploadedDataService uploadedDataService, HL7SoftMatchService hl7SoftMatchService) {
        return new MdmT02Processor(dataUploadManager, encounterManager, userManager, uploadedDataService, hl7SoftMatchService);
    }

    @Bean
    public HL7ApiHelper hL7ApiHelper(UserManager userManager,
                                     PersonContactManager personContactManager,
                                     PKBEmailMessageManager pkbEmailMessageManager,
                                     OrgNetworkService orgNetworkService,
                                     INotificationManager notificationManager,
                                     DateTimeService dateTimeService,
                                     PKBPersonHelper pkbPersonHelper,
                                     PKBServiceUtil pkbServiceUtil,
                                     InvitationManager invitationManager,
                                     HL7TeamService hl7TeamService,
                                     HL7PatientService hl7PatientService,
                                     NhsAppNotificationManager nhsAppNotificationManager) {
        return new HL7ApiHelper(userManager, personContactManager, pkbEmailMessageManager, orgNetworkService, notificationManager, dateTimeService, pkbPersonHelper,
                pkbServiceUtil, invitationManager, hl7TeamService, hl7PatientService, nhsAppNotificationManager);
    }

    @Bean
    public SiuProcessor siuProcessor(DataUploadManager dataUploadManager, HL7SoftMatchService hl7SoftMatchService, CalendarManager calendarManager,
                                     ReferenceDataManager referenceDataManager, UserManager userManager) {
        return new SiuProcessor(dataUploadManager, hl7SoftMatchService, calendarManager, referenceDataManager, userManager);
    }

}
