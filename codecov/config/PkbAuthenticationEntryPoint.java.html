package config;

import com.pkb.util.AppIdentityUtil;
import org.apache.commons.lang3.StringUtils;
import org.apache.http.client.utils.URIBuilder;
import org.springframework.security.core.AuthenticationException;
import org.springframework.security.web.authentication.LoginUrlAuthenticationEntryPoint;
import org.springframework.util.LinkedMultiValueMap;
import org.springframework.util.MultiValueMap;
import org.springframework.web.util.UriComponentsBuilder;

import javax.servlet.ServletException;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.net.URI;
import java.net.URISyntaxException;
import java.net.URLDecoder;
import java.net.URLEncoder;
import java.util.Arrays;
import java.util.HashSet;
import java.util.Set;

import static java.nio.charset.StandardCharsets.UTF_8;

public class PkbAuthenticationEntryPoint extends LoginUrlAuthenticationEntryPoint {

    private final Set<String> oauthParameters = new HashSet<>(Arrays.asList("assertedLoginIdentity", "asserted_login_identity", "teamCode", "isSelfReg", "forceNewMetadata", "prompt"));
    private final AppIdentityUtil appIdentityUtil;
    private final String nhsLoginUrl;

    public PkbAuthenticationEntryPoint(String loginFormUrl, AppIdentityUtil appIdentityUtil, String nhsLoginUrl) {
        super(loginFormUrl);
        this.appIdentityUtil = appIdentityUtil;
        this.nhsLoginUrl = nhsLoginUrl;
    }

    @Override
    public void commence(HttpServletRequest request, HttpServletResponse response, AuthenticationException authException) throws IOException, ServletException {
        String assertedLoginIdentity = request.getParameter("assertedLoginIdentity");

        if (assertedLoginIdentity == null) {
            assertedLoginIdentity = request.getParameter("asserted_login_identity");
        }

        if (StringUtils.isNotBlank(assertedLoginIdentity)) {
            response.sendRedirect(getAssertedLoginIdentityRedirectUrl(request));
            return;
        }

        if (appIdentityUtil.isNhsAppSession(request)) {
            response.sendRedirect(nhsLoginUrl);
            return;
        }

        super.commence(request, response, authException);
    }

    private String getAssertedLoginIdentityRedirectUrl(HttpServletRequest request) {
        MultiValueMap<String, String> parameters = UriComponentsBuilder.fromHttpUrl(request.getRequestURL().toString() + "?" + request.getQueryString()).build().getQueryParams();
        MultiValueMap<String, String> oauthParams = new LinkedMultiValueMap<>();

        parameters.forEach((name, values) -> {
            if (oauthParameters.contains(name)) {
                oauthParams.addAll(name, values);
            }
        });

        oauthParams.add("phrPath", URLEncoder.encode(createPhrPathUrl(request, parameters), UTF_8));
        String formattedParameters = UriComponentsBuilder.newInstance().queryParams(oauthParams).build(false).getQuery();
        return nhsLoginUrl + "?" + formattedParameters;
    }

    private String createPhrPathUrl(HttpServletRequest request, MultiValueMap<String, String> query) {
        try {
            URI uri = new URI(request.getRequestURL().toString());
            URIBuilder ub = new URIBuilder(new URI(uri.getPath()).toString());

            query.forEach((name, values) -> {
                if (!oauthParameters.contains(name) && !name.equals("phrPath")) {
                    values.forEach(value -> ub.addParameter(name, URLDecoder.decode(value, UTF_8)));
                }
            });

            return ub.toString();
        } catch (URISyntaxException e) {
            throw new RuntimeException(e);
        }
    }
}
