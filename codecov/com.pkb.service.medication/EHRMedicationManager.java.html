package com.pkb.service.medication;

import com.pkb.app.dto.DataPointSaveResult;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.PKBFilter;
import com.pkb.coding.entity.CodingReceived;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.data.EHRRemote;
import com.pkb.entities.enums.CodingDataType;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.medication.entity.Medication;
import com.pkb.service.coding.CodingManager;
import com.pkb.service.ehr.DeduplicatorManager;
import com.pkb.service.ehr.MassUpdatableEhrDataPointManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import io.vavr.Function2;
import io.vavr.Tuple2;
import io.vavr.Tuple3;
import org.apache.commons.lang3.time.DateUtils;
import org.jetbrains.annotations.NotNull;

import java.util.Calendar;
import java.util.Date;
import java.util.List;

import static java.util.Collections.singletonList;
import static java.util.stream.Collectors.toList;

public class EHRMedicationManager extends MassUpdatableEhrDataPointManager<Medication> {
    private static final String PATIENT_ON_MEDICATION = "boolean01";

    private static final Function2<Medication, Long, Void> CODING_RECEIVED_ID_SETTER = (medication, privateId) -> {
        medication.setMedicationCodingReceivedId(privateId);
        return null;
    };
    private static final Function2<Medication, PrivacyFlag, Void> PRIVACY_FLAG_HANDLER = (medication, privacyFlag) -> {
        medication.getBaseFields().setPrivacyFlag(privacyFlag);
        return null;
    };

    public EHRMedicationManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider,
                                CodingManager codingManager, DeduplicatorManager deduplicator, Class<Medication> dtoClass, MenuDataType dataType) {
        super(config, beanFactory, dateTimeService, uuidProvider, codingManager, deduplicator, dtoClass, dataType);
    }

    public DataPointSaveResult<Medication> saveMedication(@NotNull LoggedInEHRRequestContext requestContext, Medication medication) {
        return save(requestContext, singletonList(medication));
    }

    public DataPointSaveResult<Medication> updateMedication(@NotNull LoggedInEHRRequestContext requestContext, Medication medication) {
        return update(requestContext, singletonList(medication));
    }

    public List<Medication> getMedications(@NotNull LoggedInEHRRequestContext requestContext, long patientAccountId, boolean onlyCurrent, boolean includeMedicationsNotTaken) {
        EHRRemote ehrBean = beanFactory.getEhrRemote();
        EHRSearch<Medication> ehrSearch = new EHRSearch<>(patientAccountId, Medication.class, Medication.MS_PATH);
        ehrSearch.setLatestPerTypeFilter(EHRData.ENTERED_DATE, EHRData.UNIQUE_ID);
        addMedicationNotTakenFilter(includeMedicationsNotTaken, ehrSearch);
        List<Medication> medications = ehrBean.queryDecryptAndPopulateData(ehrSearch, Medication.class, requestContext);

        Date today = DateUtils.truncate(Date.from(dateTimeService.now()), Calendar.DATE); // today @ midnight

        return medications.stream()
                .filter(medication -> !onlyCurrent || ((medication.getEndDate() == null) || medication.getEndDate().after(today)))
                .collect(toList());
    }

    private void addMedicationNotTakenFilter(boolean includeMedicationNotTaken, EHRSearch search) {
        if (!includeMedicationNotTaken) {
            search.addFilter(new PKBFilter(PATIENT_ON_MEDICATION, PKBFilter.Operator.TRUE_AND_IS_NOT_NULL, null, false/*applyWhenChoosingLatest*/));
        }
    }

    @Override
    protected List<Tuple2<Medication, EHRRequestContext>> adjustCodings(List<Tuple2<Medication, EHRRequestContext>> data) {
        return doAdjustCodings(data,
                CodingDataType.MEDICATION,
                Medication::getMedication,
                CODING_RECEIVED_ID_SETTER,
                PRIVACY_FLAG_HANDLER);
    }

    @Override
    protected List<Tuple2<Medication, EHRRequestContext>> adjustCodingsNoCodingLookup(List<Tuple3<Medication, CodingReceived, EHRRequestContext>> data) {
        return doAdjustCodingsNoCodingLookup(data, CODING_RECEIVED_ID_SETTER, PRIVACY_FLAG_HANDLER);
    }
}
