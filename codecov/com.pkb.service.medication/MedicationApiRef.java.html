package com.pkb.service.medication;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

import org.jetbrains.annotations.Nullable;

import com.pkb.code.entity.CodeSystemName;

/**
 * Map of apiRef string to <codeSystem, code> for Medication
 *
 * @author pravina
 *
 */
public enum MedicationApiRef {

    FolicAcidTablets("folicAcid", "226316008", CodeSystemName.SNOMED.toString(), ""),
    Oseltamivir("oseltamivir", "386142008", CodeSystemName.SNOMED.toString(), ""),
    Zanamivir("zanamivir", "116100000", CodeSystemName.SNOMED.toString(), ""),

    OtherAntiviral("otherAntiviral", "antiviral_medication_code_other", CodeSystemName.NONE.toString(), ""),
    UnknownAntiviral("unknownAntiviral", "antiviral_medication_code_unknown", CodeSystemName.NONE.toString(), ""),
    VitaminDPreparation("vitaminDPreparation", "11563006", CodeSystemName.SNOMED.toString(), "");

    private static Map<String, List<MedicationApiRef>> apiRefMap = new HashMap<>();

    // assume for now many to many mapping
    private static Map<String, List<String>> codeToApiRefMap = new HashMap<>();

    static {
        // mapping of apirefs to code details
        MedicationApiRef[] diagnosisApiRefs = MedicationApiRef.values();
        for (MedicationApiRef diagnosisApiRef : diagnosisApiRefs) {
            List<MedicationApiRef> apiRefs = apiRefMap.computeIfAbsent(diagnosisApiRef.getApiRef(), k -> new ArrayList<>());
            apiRefs.add(diagnosisApiRef);
        }

        //mapping of code apirefs to code details
        for (MedicationApiRef medicationApiRef : diagnosisApiRefs) {
            List<String> apiRefs = codeToApiRefMap.computeIfAbsent(medicationApiRef.getCode(), k -> new ArrayList<>());
            apiRefs.add(medicationApiRef.getApiRef());
        }
    }

    private String apiRef;
    private String code;
    private String codeSystemName;
    private String codeSystemVersion;

    MedicationApiRef(String apiRef, String code, String codeSystemName, String codeSystemVersion) {
        this.apiRef = apiRef;
        this.code = code;
        this.codeSystemName = codeSystemName;
        this.codeSystemVersion = codeSystemVersion;
    }

    public String getApiRef() {
        return apiRef;
    }

    public String getCode() {
        return code;
    }

    public String getCodeSystemName() {
        return codeSystemName;
    }

    public String getCodeSystemVersion() {
        return codeSystemVersion;
    }

    @Nullable
    public static String getCodeByApiRef(String apiRefStr) {
        List<String> codes = new ArrayList<>();
        List<MedicationApiRef> apiRefs = apiRefMap.get(apiRefStr);
        if (apiRefs != null) {
            codes.addAll(apiRefs.stream().map(MedicationApiRef::getCode).collect(Collectors.toList()));
        }
        if (codes.isEmpty()) {
            return null;
        } else {
            return codes.get(0);
        }
    }

    public static List<String> getApiRefByCode(String code) {
        return codeToApiRefMap.get(code);
    }

    public static List<String> getCodesByApiRef(List<String> apiRefList) {
        List<String> codes = new ArrayList<>();
        for (String apiRefStr : apiRefList) {
            List<MedicationApiRef> apiRefs = apiRefMap.get(apiRefStr);
            if (apiRefs != null) {
                codes.addAll(apiRefs.stream().map(MedicationApiRef::getCode).collect(Collectors.toList()));
            }
        }
        return codes;
    }

    public static List<String> getApiRefValues() {
        // mapping of apirefs to code details
        List<MedicationApiRef> medicationApiRefs = Arrays.asList(MedicationApiRef.values());
        return medicationApiRefs.stream().map(medicationApiRef -> medicationApiRef.apiRef).collect(Collectors.toList());

    }
}
