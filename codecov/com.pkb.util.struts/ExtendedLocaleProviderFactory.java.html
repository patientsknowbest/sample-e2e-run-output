package com.pkb.util.struts;

import com.opensymphony.xwork2.DefaultLocaleProvider;
import com.opensymphony.xwork2.LocaleProvider;
import com.opensymphony.xwork2.LocaleProviderFactory;
import com.opensymphony.xwork2.inject.Inject;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Locale;
import java.util.Set;
import java.util.stream.Stream;

import static java.util.stream.Collectors.toSet;
import static org.apache.commons.lang3.LocaleUtils.toLocale;
import static org.apache.commons.lang3.StringUtils.split;

/**
 * Struts by default rejects locales that aren't supported by the underlying JRE. It's annoying to extend the JRE for our other languages so we just extend Struts, here.
 *
 */
public class ExtendedLocaleProviderFactory implements LocaleProviderFactory {

    public static final String LOCALES = "com.pkb.util.struts.ExtendedLocaleProviderFactory.locales";

    private Set<Locale> customLocales;

    @Override
    public LocaleProvider createLocaleProvider() {
        return new ExtendedLocaleProvider(customLocales);
    }

    private static class ExtendedLocaleProvider extends DefaultLocaleProvider {

        private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

        private Set<Locale> customLocales;

        public ExtendedLocaleProvider(Set<Locale> customLocales) {
            this.customLocales = customLocales;
        }

        @Override
        public boolean isValidLocale(Locale locale) {
            return super.isValidLocale(locale) || isSupportedCustomLocale(locale);
        }

        private boolean isSupportedCustomLocale(Locale locale) {
            if (locale == null) {
                return false;
            }
            boolean supported = customLocales.contains(locale);
            if (!supported) {
                LOGGER.warn("Given locale ({}) is not supported.", locale);
            }
            return supported;
        }
    }

    @Inject(value = LOCALES)
    public void setCustomLocales(String customLocales) {
        this.customLocales = Stream.of(split(customLocales, ","))
                .map(locale -> toLocale(locale.trim()))
                .collect(toSet());
    }

}
