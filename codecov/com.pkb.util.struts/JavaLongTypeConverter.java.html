package com.pkb.util.struts;

import org.apache.struts2.util.StrutsTypeConverter;
import org.jetbrains.annotations.Nullable;

import java.util.Map;

//
// Bypasses struts i18n of Longs, which breaks eg <s:textfield name="measurementValue[%{#attr.measurementType.id]]/>
// if the locale is eg Arabic
//
// Configured in xwork-conversion.properties
//

public class JavaLongTypeConverter extends StrutsTypeConverter {

    @Override
    public @Nullable Object convertFromString(Map map, String[] strings, Class aClass) {
        if (strings == null || strings.length == 0 || strings[0].isBlank()) {
            return null;
        }
        return Long.parseLong(strings[0]);
    }

    @Override
    public @Nullable String convertToString(Map map, Object o) {
        return o==null ? null : ((Long) o).toString();
    }
}
