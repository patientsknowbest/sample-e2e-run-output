package com.pkb.util.struts;

import com.pkb.datamodel.Email;
import org.apache.camel.TypeConversionException;
import org.apache.struts2.util.StrutsTypeConverter;
import org.jetbrains.annotations.Nullable;

import java.util.Map;

import static com.pkb.datamodel.ImmutableDisclosingEmail.disclosingEmail;

public class EmailTypeConverter extends StrutsTypeConverter {
    @Override
    public @Nullable Object convertFromString(Map map, String[] strings, Class aClass) {
        if(strings == null || strings.length == 0 || strings[0].isBlank()) {
            return null;
        }

        try {
            return disclosingEmail(strings[0]);
        } catch (RuntimeException emailException) {
            throw new TypeConversionException("Unable to convert '" + strings[0] + "'", Email.class, emailException);
        }
    }

    @Override
    public String convertToString(Map map, Object o) {
        if(o != null && o instanceof Email) {
            return ((Email) o).address();
        }
        return "";
    }
}
