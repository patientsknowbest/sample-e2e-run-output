package com.pkb.util.struts;

import com.opensymphony.xwork2.conversion.TypeConversionException;
import org.apache.struts2.util.StrutsTypeConverter;
import org.jetbrains.annotations.Nullable;
import org.jooq.tools.StringUtils;

import java.util.Arrays;
import java.util.Map;
import java.util.Objects;
import java.util.UUID;

import static java.lang.String.format;

public class UUIDTypeConverter extends StrutsTypeConverter {

    @Override
    @Nullable
    public Object convertFromString(Map context, String[] values, Class toClass) {
        try {
            UUID[] result = Arrays.stream(values).map(this::convertFromString).filter(Objects::nonNull).toArray(UUID[]::new);
            switch (result.length) {
            case 0:
                return null;
            case 1:
                return result[0];
            default:
                return result;
            }
        } catch (IllegalArgumentException e) {
            throw new TypeConversionException(format("Failed to convert value %s as UUID", Arrays.toString(values)), e);
        }
    }

    private UUID convertFromString(String value) {
        return StringUtils.isBlank(value) ? null : UUID.fromString(value);
    }

    @Override
    public String convertToString(Map context, Object o) {
        return o.toString();
    }
}
