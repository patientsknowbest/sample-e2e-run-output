package com.pkb.util.struts;

import com.opensymphony.xwork2.conversion.ObjectTypeDeterminer;
import com.opensymphony.xwork2.conversion.TypeConverter;
import com.opensymphony.xwork2.conversion.impl.DefaultTypeConverter;
import com.opensymphony.xwork2.inject.Inject;

import java.lang.reflect.Member;
import java.util.Map;
import java.util.Optional;

import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;

public class JavaOptionalTypeConverter extends DefaultTypeConverter {

    private ObjectTypeDeterminer objectTypeDeterminer;

    @Inject
    public void setObjectTypeDeterminer(ObjectTypeDeterminer determiner) {
        this.objectTypeDeterminer = determiner;
    }

    @Override
    public Object convertValue(Map<String, Object> context, Object target, Member member, String propertyName, Object value, Class toType) {
        return Match(toType).of(
                Case($(String.class), () -> optionalToString(context, target, member, propertyName, value)),
                Case($(Optional.class), () -> stringToOptional(context, target, member, propertyName, value)));
    }

    private Optional<?> stringToOptional(Map<String, Object> context, Object target, Member member, String propertyName, Object value) {
        Class targetClass = objectTypeDeterminer.getElementClass(target.getClass(), propertyName, null);
        return Optional.ofNullable(value)
                .map(innerValue -> getTypeConverter(context).convertValue(context, target, member, propertyName, innerValue, targetClass))
                .filter(convertedValue -> !TypeConverter.NO_CONVERSION_POSSIBLE.equals(convertedValue));
    }

    private String optionalToString(Map<String, Object> context, Object target, Member member, String propertyName, Object value) {
        return asOptional(value)
                .map(innerValue -> (String) getTypeConverter(context).convertValue(context, target, member, propertyName, innerValue, String.class))
                .filter(convertedValue -> !TypeConverter.NO_CONVERSION_POSSIBLE.equals(convertedValue))
                .orElse("");
    }

    private Optional<?> asOptional(Object value) {
        if (!(value instanceof Optional)) {
            throw new IllegalStateException("Optional converter used to convert incompatible type: " +
                    Optional.ofNullable(value).map(obj -> obj.getClass().getName()).orElse("null"));
        }
        return (Optional) value;
    }
}
