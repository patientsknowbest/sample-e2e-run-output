package com.pkb.util.struts;

import com.opensymphony.xwork2.StrutsTextProviderFactory;
import com.opensymphony.xwork2.TextProvider;
import com.opensymphony.xwork2.TextProviderSupport;

import java.util.ResourceBundle;

public class CachingTextProviderFactory extends StrutsTextProviderFactory {

    @Override
    protected TextProvider getTextProvider(Class clazz) {
        TextProviderSupport delegate = (TextProviderSupport) super.getTextProvider(clazz);
        CachingTextProviderSupport support = new CachingTextProviderSupport(delegate);
        return support;
    }

    @Override
    protected TextProvider getTextProvider(ResourceBundle bundle) {
        TextProviderSupport delegate = (TextProviderSupport) super.getTextProvider(bundle);
        CachingTextProviderSupport support = new CachingTextProviderSupport(delegate);
        return support;
    }
}
