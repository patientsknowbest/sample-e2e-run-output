package com.pkb.integration.nhs.mapper;

import com.fasterxml.jackson.core.JsonProcessingException;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.pkb.common.util.FrameFilter;
import com.pkb.integration.nhs.model.Issue;
import com.pkb.integration.nhs.model.response.ErrorResponse;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import retrofit2.Response;

import java.io.IOException;
import java.lang.invoke.MethodHandles;

import static com.google.common.base.Preconditions.checkState;

public class ErrorResponseMapper {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private final ObjectMapper objectMapper;

    public ErrorResponseMapper(ObjectMapper objectMapper) {
        this.objectMapper = objectMapper;
    }

    public ErrorResponse map(Response<?> response) {
        checkState(!response.isSuccessful(), "Cannot parse ErrorResponse from successful response");

        ErrorResponse responseError = ErrorResponse.getEmptyResponseError();
        if (response.errorBody() != null) {
            String errorBodyString;
            try {
                errorBodyString = response.errorBody().string();
            } catch (IOException e) {
                LOGGER.error("response.errorBody().string()", FrameFilter.filter(e));
                errorBodyString = response.toString();
            }

            try {
                responseError = objectMapper.readValue(errorBodyString, ErrorResponse.class);
            } catch (JsonProcessingException e) {
                LOGGER.error("Could not parse json", FrameFilter.filter(e));
                responseError = ErrorResponse.getResponseError(Issue.of("unparseable.error.body", errorBodyString));
            }
        }

        return responseError;
    }
}
