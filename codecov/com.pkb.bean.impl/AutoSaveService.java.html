package com.pkb.bean.impl;

import com.pkb.action.message.MessageAutoSaveModel;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.encounter.entity.Message;
import com.pkb.service.encounter.impl.tolven.EncounterManager;
import com.pkb.util.LoggedInUserUtil;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.Objects;

import static com.pkb.entities.enums.messaging.MessageStatus.INDRAFT;

public class AutoSaveService {

    private final LoggedInUserUtil loggedInUserUtil;

    private final EncounterManager encounterManager;

    public AutoSaveService(@NotNull LoggedInUserUtil loggedInUserUtil, @NotNull EncounterManager encounterManager) {
        this.loggedInUserUtil = loggedInUserUtil;
        this.encounterManager = encounterManager;
    }

    private Either<MessageAutoSaveError, Message> validatePatch(MessageAutoSaveModel autoSaveModel, Message message) {
        if (!Objects.equals(message.getSenderId(), Long.toString(loggedInUserUtil.getLoggedInUserId()))) {
            return Either.left(new OtherSenderError(autoSaveModel.getMessageId(), loggedInUserUtil.getLoggedInUserId()));
        }
        if (message.getMessageStatus() != INDRAFT) {
            return Either.left(new NonDraftMessageError(autoSaveModel.getMessageId()));
        }
        return Either.right(message);
    }

    public Either<MessageAutoSaveError, Message> prepareMessageForAutoSave(@NotNull LoggedInEHRRequestContext requestContext, MessageAutoSaveModel autoSaveModel) {
        if (autoSaveModel.isFirstDraft()) {
            Message message = new Message(new SourceDetails(requestContext));
            if (autoSaveModel.getEncounterUniqueId() != null) {
                message.setEncounterUniqueId(autoSaveModel.getEncounterUniqueId());
            }
            message.setMessageStatus(INDRAFT);
            autoSaveModel.applyTo(message);
            return Either.right(message);
        } else {
            return encounterManager.findMessageById(requestContext, autoSaveModel.getMessageId())
                    .map(msg -> Either.<MessageAutoSaveError, Message>right(msg))
                    .orElseGet(() -> Either.left(new MessageNotFoundError(autoSaveModel.getMessageId())))
                    .flatMap(message -> validatePatch(autoSaveModel, message))
                    .map(autoSaveModel::applyTo);
        }
    }
}
