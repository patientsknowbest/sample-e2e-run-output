package com.pkb.service.invitation.impl;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.domain.AccountUserService;
import com.pkb.domain.duplicate.person.PersonService;
import com.pkb.dto.LetterInvitationDTO;
import com.pkb.entities.core.CoreAccountUser;
import com.pkb.entities.pub.PublicLetterInvitation;
import com.pkb.patient.activation.letter.AccountDetailsDTO;
import com.pkb.patient.activation.letter.InvitationTokenDTO;
import com.pkb.patient.activation.nhslogin.NhsLoginAccountDetailsDTO;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.util.Optional;
import java.util.UUID;

import static com.pkb.datamodel.ImmutableDisclosingEmail.disclosingEmail;
import static java.util.Optional.ofNullable;
import static org.apache.commons.lang3.StringUtils.trimToNull;

public class LetterInvitationService {

    private final LetterInvitationManager invitationManager;
    private final AccountUserService accountUserService;
    private final PersonService personService;

    public LetterInvitationService(LetterInvitationManager letterInvitationManager, AccountUserService accountUserService,
                                   PersonService personService) {
        this.invitationManager = letterInvitationManager;
        this.accountUserService = accountUserService;
        this.personService = personService;
    }

    private LetterInvitationDTO populate(PublicLetterInvitation invitation) {
        return populate(invitation, null);
    }

    private LetterInvitationDTO populate(PublicLetterInvitation invitation, @Nullable Long personId) {

        long inviteePersonId = Optional.ofNullable(personId).orElseGet(() -> {
            long accountId = accountUserService.findAccountUserById(invitation.getInviteeId())
                    .map(CoreAccountUser::getAccountId)
                    .orElseThrow(() -> new RuntimeException("No account found for account user " + invitation.getInviteeId()));
            return personService.findPersonIdForDefaultAccountId(accountId)
                    .orElseThrow(() -> new RuntimeException("No person found for default account id " + accountId));
        });

        LetterInvitationDTO dto = new LetterInvitationDTO();
        dto.setId(invitation.getId());
        dto.setCode(invitation.getCode());
        dto.setCreated(invitation.getCreated());
        dto.setExpiration(invitation.getExpiration());
        dto.setClaimedEmail(invitation.getClaimedEmail() == null ? null : disclosingEmail(invitation.getClaimedEmail()));
        dto.setVerificationCode(invitation.getVerificationCode());
        dto.setStatus(invitation.getInviteeStatus());
        dto.setInvalidated(invitation.getInvalidated());
        dto.setInviteePersonId(inviteePersonId);
        return dto;
    }

    public @NotNull LetterInvitationDTO generateLetterInvitation(@NotNull LoggedInEHRRequestContext requestContext, long patientId) {
        SensitiveLetterInvitation invitation = invitationManager.generateLetterInvitation(requestContext, patientId);
        LetterInvitationDTO dto = populate(invitation.getInvitation(), patientId);
        dto.setToken(invitation.getSecretToken().getValue());
        return dto;
    }

    public LetterInvitationDTO activatePatient(@NotNull InvitationTokenDTO invitationSecrets, @NotNull AccountDetailsDTO patientDetails) {
        PublicLetterInvitation invitation = invitationManager.activatePatient(invitationSecrets, patientDetails);
        return populate(invitation);
    }

    public @NotNull LetterInvitationDTO verifyEmail(@NotNull EHRRequestContext requestContext, @NotNull String verificationId, char[] password) {
        PublicLetterInvitation invitation = invitationManager.verifyEmail(requestContext, verificationId, password);
        return populate(invitation);
    }

    public @NotNull LetterInvitationDTO findByCode(@NotNull String code) {
        return populate(invitationManager.findByCode(code));
    }

    public @Nullable String getTeamCode(long letterInvitationId) {
        return invitationManager
                .getTeam(letterInvitationId)
                .flatMap(t -> ofNullable(trimToNull(t.getCode())))
                .orElse(null);
    }

    public long getTeamId(long letterInvitationId) {
        return invitationManager.findById(letterInvitationId).getTeamId();
    }

    public @NotNull LetterInvitationDTO findByVerificationCode(@NotNull String code) {
        return populate(invitationManager.findByVerificationCode(code));
    }

    public @NotNull LetterInvitationDTO findById(long id) {
        return populate(invitationManager.findById(id));
    }

    public PKBPerson activateNhsLoginPatient(long personId, UUID personPublicId, NhsLoginAccountDetailsDTO accountDetails) {
        return invitationManager.activateNhsLoginPatient(personId, personPublicId, accountDetails);
    }

    public String activatePatient(long patientId, UUID patientPublicId) {
        return invitationManager.initPersonCredential(patientId, patientPublicId);
    }

}
