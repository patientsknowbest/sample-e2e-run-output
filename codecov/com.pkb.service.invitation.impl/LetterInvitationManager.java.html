    PublicLetterInvitation setUpInviteeStatus(PublicLetterInvitation invitation, PKBPerson patient) {
        if (PATIENT_INVITED_NOT_REGISTERED.contains(patient.getStatus())) {
            invitation.setInviteeStatus(PublicLetterInvitation.Status.INVITED_NOT_REGISTERED);
        } else if (PATIENT_NOT_INVITED_NOT_REGISTERED.contains(patient.getStatus())) {
            invitation.setInviteeStatus(PublicLetterInvitation.Status.NOT_INVITED_NOT_REGISTERED);
        } else {
            throw new IllegalStateException("Patient has unsupported status[" + patient.getStatus() + "] for token registration");
        }
        return invitation;
    }

    public @NotNull SensitiveLetterInvitation generateLetterInvitation(@NotNull LoggedInEHRRequestContext context, long patientId) {
        return transactional(() -> {
            PKBPersonRemote personBean = beanFactory.getPKBPersonBean();
            PKBPerson patient = personBean.findPKBPerson(patientId).getOrElseThrow(() -> {
                fail(EC_SYSTEM_ID, "No record exists for this patient", patientId);
                throw new IllegalStateException("No record exists for this patient " + patientId);
            });

            validator.validate(patient);

            AccountUser patientAccount = Optional.ofNullable(personBean.getActiveDefaultAccountUser(patient.getId()))
                    .orElseThrow(() -> new IllegalStateException("No account user exists for this patient " + patient.getId()));
            PublicLetterInvitation invitation = new PublicLetterInvitation();
            invitation.setCode(letterInvitationUtility.generateAccessCode(config.getLetterInvitationAccessCodeSize()));
            invitation.setCreated(dateTimeService.now());
            invitation.setExpiration(letterInvitationUtility.calculateTokenExpiry(invitation.getCreated(), config.getLetterInvitationExpiry()));

            SecretToken secret = letterInvitationUtility.generateToken(config.getLetterInvitationTokenSize());
            try {
                AccountPrivateKey key = beanFactory.getHealthRecordAccessRemote().grant(
                        context,
                        new String(secret.getValue()),
                        patientAccount.getAccount().getId());
                AccountSecretKey secretKey = key.getSecretKey();
                EncodedEncryptedPrivateKeyInfoWithSecretDeriv accountPrivateKey = new EncodedEncryptedPrivateKeyInfoWithSecretDeriv(
                        key.getEncodedEncryptedPrivateKeyInfo(), key.getKeySize(), key.getAlgorithm(), secretKey.getEncryptedKey(), secretKey.getAlgorithm(),
                        secretKey.getSecretKeyDerivAlgorithm(), secretKey.getSecretKeyDerivDataLength(), secretKey.getSecretKeyDerivationSalt(), secretKey.getSecretKeyDerivIterations());
                invitation.setAccountPrivateKey(accountPrivateKey);
            } catch (GeneralSecurityException | IOException e) {
                throw new RuntimeException("Unable to generate invitation token", e);
            }

            context.getMaybeAccessingUserId().ifPresent(invitation::setAuthorId);

            context.getTeamId().ifPresent(invitation::setTeamId);

            invitation.setInviteeId(patientAccount.getId());
            letterInvitationService.save(invitation);
            return new SensitiveLetterInvitation(secret, setUpInviteeStatus(invitation, patient));
        });
    }

    private boolean isIncorrect(PublicLetterInvitation invitation, String token) {
        try {
            accountPrivateKeyMapper.getAccountPrivateKeyFromEncodedEncryptedPrivateKeyInfoWithSecretDeriv(invitation.getAccountPrivateKey(), token);
            return false;
        } catch (Exception ignored) {
            return true;
        }
    }

    private PublicLetterInvitation validatedInvitation(InvitationTokenDTO invitationSecrets) {
        PublicLetterInvitation invitation = findByCode(invitationSecrets.getInvitationCode());

        if (invitation.getInvalidated()) {
            fail(EC_INVITATION_INVALID, "Invitation invalid", config.getBaseURL(), config.getSupportNewTicketUrl());
        }
        if (invitation.getExpiration().isBefore(dateTimeService.now())) {
            fail(config.isDefaultTokenRegistrationErrorMessages() ? EC_INVITATION_CODES_EXPIRED : EC_INVITATION_EXPIRED, "Invitation expired", config.getSupportNewTicketUrl());
        }
        if (isIncorrect(invitation, invitationSecrets.getDecryptToken())) {
            fail(config.isDefaultTokenRegistrationErrorMessages() ? EC_INVITATION_CODES_INVALID : EC_INCORRECT_TOKEN, "Incorrect token");
        }
        return invitation;
    }

    boolean isEmailValid(PKBPerson person, Email email) {
        return personContactManager.findPersonByConfirmedOrPrimaryEmail(email).stream()
                .map(p -> p.getPerson().getId())
                .filter(id -> !id.equals(person.getId()))
                .findFirst()
                .map(Functions.constant(false))
                .orElse(true);
    }

    PKBPerson validatedPerson(PublicLetterInvitation invitation, AccountDetailsDTO patientDetails) {
        long accountId = accountUserService.findAccountUserById(invitation.getInviteeId())
                .map(CoreAccountUser::getAccountId)
                .orElseThrow(() -> new RuntimeException("No account found for account user " + invitation.getInviteeId()));
        PKBPerson person = Optional.ofNullable(beanFactory.getPKBPersonBean().getAccountOwner(accountId))
                .orElseThrow(() -> new RuntimeException("No account owner found for account " + accountId));

        validator.validate(person); //maybe registered in the meantime

        //after previous validation are sure that realDateOfBirth is not null, what is more claimedDateOfBirth cannot be null -- we trust AccountDetailsDTO implementation
        LocalDate claimedDateOfBirth = patientDetails.getClaimedDateOfBirth();
        LocalDate realDateOfBirth = person.findDateOfBirth()
                .orElseThrow(() -> new IllegalStateException(format("Cannot find Persons's Date Of Birth (Person ID=[%s])", person.getId())));

        new ValidationErrors()
                .error(() -> !realDateOfBirth.equals(claimedDateOfBirth),
                        EC_DOB_MISMATCH, "The date of birth you entered does not match")
                .error(() -> !passwordRecoveryService.isPasswordRecoveryQuestion(patientDetails.getQuestionId()),
                        EC_INVALID_SECURITY_QUESTION, "Invalid security question")
                .error(() -> !isEmailValid(person, patientDetails.getAccountCredentials().getEmail()),
                        EC_EMAIL_TAKEN, "Email is already taken", config.getBaseURL())
                .check();

        return person;
    }

    public @NotNull PublicLetterInvitation activatePatient(@NotNull InvitationTokenDTO invitationSecrets, @NotNull AccountDetailsDTO patientDetails) {
        return transactional(() -> {
            PublicLetterInvitation invitation = validatedInvitation(invitationSecrets);
            PKBPerson person = validatedPerson(invitation, patientDetails);

            char[] password = patientDetails.getAccountCredentials().getPassword();

            PublicUserCredentials credentials = userCredentialsService.overrideCredentials(person.getId(), person.getPublicId(), password);
            AccountPrivateKey granted = beanFactory.getHealthRecordAccessRemote().grant(
                    accountPrivateKeyMapper.getAccountPrivateKeyFromEncodedEncryptedPrivateKeyInfoWithSecretDeriv(invitation.getAccountPrivateKey(), invitationSecrets.getDecryptToken()),
                    invitationSecrets.getDecryptToken(),
                    userCredentialsService.getPublicKey(person.getId()));

            //Sets the account private key on the account
            EncodedEncryptedPrivateKeyInfo accountPrivateKey = accountPrivateKeyMapper.getEncodedEncryptedPrivateKeyInfoFromAccountPrivateKey(granted);
            CoreAccountUser accountUser = accountUserService.findAccountUserById(invitation.getInviteeId())
                    .map(au -> {
                        au.setAccountPrivateKey(accountPrivateKey);
                        return au;
                    })
                    .orElseThrow(() -> new RuntimeException("Failed to find accountUser for invitee id " + invitation.getInviteeId()));
            accountUserService.saveAccountUser(accountUser);
            invitation.setClaimedEmail(patientDetails.getAccountCredentials().getEmail().address());
            invitation.setVerificationCode(letterInvitationUtility.generateConfirmationCode());

            userAccessManager.removeUnusedPasswordAccess(person.getId());
            passwordRecoveryService.backupLoginPassword(credentials.getUserId(), password, patientDetails.getQuestion(), patientDetails.getAnswer());

            letterInvitationService.save(invitation);

            return invitation;
        });
    }

    @NotNull
    public PublicLetterInvitation verifyEmail(@NotNull EHRRequestContext context, @NotNull String verificationCode, char[] password) {
        return transactional(() -> {
            PublicLetterInvitation invitation = findByVerificationCode(verificationCode);

            PKBPersonRemote pkbPersonBean = beanFactory.getPKBPersonBean();

            long accountId = accountUserService.findAccountUserById(invitation.getInviteeId())
                    .map(CoreAccountUser::getAccountId)
                    .orElseThrow(() -> new RuntimeException(format("No account found for account user %d", invitation.getInviteeId())));
            PKBPerson person = Optional.ofNullable(pkbPersonBean.getAccountOwner(accountId))
                    .orElseThrow(() -> new RuntimeException(format("No account owner found for account id %d", accountId)));

            if (!isEmailValid(person, disclosingEmail(invitation.getClaimedEmail()))) {
                fail(EC_EMAIL_TAKEN, "Email is already taken", config.getBaseURL());
            }
            if (!userCredentialsService.authenticate(person.getId(), password)) {
                fail(EC_INVALID_CREDENTIALS, "Invalid email or password");
            }
            if (invitation.getInvalidated()) {
                fail(EC_INVITATION_INVALID, "Invitation is now expired", config.getBaseURL(), config.getSupportNewTicketUrl());
            }

            personContactManager.saveNewPrimaryEmail(context, person.getId(), disclosingEmail(invitation.getClaimedEmail()), true);

            letterInvitationService.invalidateInvitations(invitation.getInviteeId());

            person.setStatus(UserStatus.EMAIL_CONFIRMED, dateTimeService.now());
            person.setRegisteredOn(Date.from(dateTimeService.now()));
            pkbPersonBean.updatePKBPerson(context.withRegisteringUser(person.getId()), person);

            return invitation;
        });
    }

    @NotNull PublicLetterInvitation findByCode(@NotNull String code) {
        return letterInvitationService.findByCode(code)
                .orElseThrow(() -> new ValidationException(config.isDefaultTokenRegistrationErrorMessages() ? EC_INVITATION_CODES_INVALID : EC_INCORRECT_CODE, "Invitation not found"));
    }

    public Optional<PublicTeam> getTeam(long letterInvitationId) {
        return letterInvitationService.findById(letterInvitationId)
                .filter(invitation -> invitation.getTeamId() != null)
                .flatMap(invitation -> teamService.findById(invitation.getTeamId()));
    }

    @NotNull PublicLetterInvitation findByVerificationCode(@NotNull String verificationCode) {
        return letterInvitationService.findByVerificationCode(verificationCode)
                .orElseThrow(() -> new ValidationException(config.isDefaultTokenRegistrationErrorMessages() ? EC_INVITATION_CODES_INVALID : EC_INVITATION_INVALID_CODE, "Invitation not found"));
    }

    @NotNull PublicLetterInvitation findById(long letterInvitationId) {
        return letterInvitationService.findById(letterInvitationId)
                .orElseThrow(() -> new ValidationException(EC_INVITATION_INVALID, "Invitation not found", config.getBaseURL(), config.getSupportNewTicketUrl()));
    }

    PKBPerson activateNhsLoginPatient(long personId, UUID personPublicId, NhsLoginAccountDetailsDTO accountDetails) {
        return transactional(() -> {
            char[] password = accountDetails.getAccountCredentials().getPassword();
            PublicUserCredentials credentials = userCredentialsService.overrideCredentials(personId, personPublicId, password);
            userAccessManager.removeUnusedPasswordAccess(personId);
            passwordRecoveryService.backupLoginPassword(credentials.getUserId(), password, accountDetails.getQuestion(), accountDetails.getAnswer());
            return beanFactory.getPKBPersonBean().findPKBPerson(personId).getOrNull();
        });
    }

    String initPersonCredential(long personId, UUID patientPublicId) {
        return transactional(() -> {
            String password = randomPassword(INITIAL_PASSWORD_LENGTH);
            userCredentialsService.overrideCredentials(personId, patientPublicId, password.toCharArray());
            userAccessManager.removeUnusedPasswordAccess(personId);
            return password;
        });
    }
