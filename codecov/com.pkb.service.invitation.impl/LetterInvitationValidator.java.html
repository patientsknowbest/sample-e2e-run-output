package com.pkb.service.invitation.impl;

import com.pkb.domain.NationalIdService;
import com.pkb.domain.OrgLevelIdService;
import com.pkb.domain.TeamLevelIdService;
import com.pkb.entities.enums.UserStatus;
import com.pkb.exception.ValidationErrors;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;

import static com.pkb.dto.LetterInvitationDTO.EC_DOB;
import static com.pkb.dto.LetterInvitationDTO.EC_IDENTIFIER;
import static com.pkb.dto.LetterInvitationDTO.EC_STATUS;
import static com.pkb.dto.LetterInvitationDTO.EC_USER_TYPE;
import static com.pkb.entities.enums.UserStatus.PATIENTS_REGISTRATION_NOT_ALLOWED;
import static org.apache.commons.collections4.CollectionUtils.isNotEmpty;

public class LetterInvitationValidator {

    private final NationalIdService nationalIdService;
    private final TeamLevelIdService teamLevelIdService;
    private final OrgLevelIdService orgLevelIdService;

    public LetterInvitationValidator(NationalIdService nationalIdService, TeamLevelIdService teamLevelIdService, OrgLevelIdService orgLevelIdService) {
        this.nationalIdService = nationalIdService;
        this.teamLevelIdService = teamLevelIdService;
        this.orgLevelIdService = orgLevelIdService;
    }

    public boolean hasAtLeastOneNationalId(long patientId) {
        return isNotEmpty(nationalIdService.findByPersonId(patientId));
    }

    public boolean hasAtLeastOneTeamId(long patientId) {
        return isNotEmpty(teamLevelIdService.findByPersonId(patientId));
    }

    public boolean hasAtLeastOneOrgId(long patientId) {
        return isNotEmpty(orgLevelIdService.findByPersonId(patientId));
    }

    public void validate(@NotNull PKBPerson person) {
        ValidationErrors validator = new ValidationErrors();

        if (person.isPatient()) {

            if (PATIENTS_REGISTRATION_NOT_ALLOWED.contains(person.getStatus())) {
                if (person.getStatus() == UserStatus.DEAD) {
                    validator.error(EC_STATUS, "Patient died");
                } else {
                    validator.error(EC_STATUS, "Registration already completed");
                }
            } else {
                if (!hasAtLeastOneNationalId(person.getId()) && !hasAtLeastOneTeamId(person.getId())
                        && !hasAtLeastOneOrgId(person.getId())) {
                    validator.error(EC_IDENTIFIER, "Record missing a patient identifier");
                }
                if (person.findDateOfBirth().isEmpty()) {
                    validator.error(EC_DOB, "Record missing a date of birth");
                }
            }
        } else {
            validator.error(EC_USER_TYPE, "Call not valid for this type of record");
        }

        validator.check();
    }

}
