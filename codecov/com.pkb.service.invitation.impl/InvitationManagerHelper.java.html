package com.pkb.service.invitation.impl;

import com.pkb.encounter.entity.PKBInvitation;
import com.pkb.entities.doc.DocInvitation;
import com.pkb.mappers.PKBInvitationMapper;
import com.pkb.user.entity.PKBPerson;

public class InvitationManagerHelper {

    private final PKBInvitationMapper pkbInvitationMapper;

    public InvitationManagerHelper(PKBInvitationMapper pkbInvitationMapper) {
        this.pkbInvitationMapper = pkbInvitationMapper;
    }

    PKBInvitation mapToPKBInvitation(DocInvitation invitation, PKBPerson targetPerson) {
        PKBInvitation result = pkbInvitationMapper.map(invitation);
        if (targetPerson != null) {
            result.setTargetUser(targetPerson);
        }
        //not currently supporting mapping InvitationDetail
        result.setInvitationDetail(null);
        return result;
    }
}
