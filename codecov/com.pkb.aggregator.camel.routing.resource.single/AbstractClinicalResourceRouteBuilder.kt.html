package com.pkb.aggregator.camel.routing.resource.single

import com.pkb.aggregator.camel.bean.process.ClinicalResourceProcessor
import com.pkb.aggregator.camel.bean.process.ResourceDeleter
import com.pkb.aggregator.camel.beanMethod
import com.pkb.aggregator.camel.routing.AbstractAggregatorRouteBuilder
import com.pkb.aggregator.camel.routing.createFhirUpdateRoute
import com.pkb.aggregator.camel.routing.resource.MissingReferenceProcessor
import com.pkb.aggregator.configuration.FhirConfiguration
import com.pkb.common.testcontrol.services.PubSubNamespaceService

abstract class AbstractClinicalResourceRouteBuilder(
    pubSubNamespaceService: PubSubNamespaceService,
    private val fhirConfiguration: FhirConfiguration,
    private val missingReferenceProcessor: MissingReferenceProcessor,
) : AbstractAggregatorRouteBuilder(pubSubNamespaceService) {

    override fun configure() {

        registerAllErrorHandlers(missingReferenceProcessor)

        val routeId = "$name-change"

        from(processRouteName).routeId(routeId)
            .beanMethod(ClinicalResourceProcessor::processSingle).id("$routeId-process")
            .to(createFhirUpdateRoute(fhirConfiguration.downstream.endpoint)).id("$routeId-downstream-push")
            .beanMethod(ClinicalResourceProcessor::logOutcome).id("$routeId-log-outcome")

        from(deleteRouteName).routeId("$routeId-delete")
            .beanMethod(ClinicalResourceProcessor::toDownstreamResourceId).id("$routeId-derive-downstream-id")
            .beanMethod(ResourceDeleter::softDelete).id("$routeId-soft-delete")
    }

    abstract val processRouteName: String
    abstract val deleteRouteName: String
    abstract val name: String
}
