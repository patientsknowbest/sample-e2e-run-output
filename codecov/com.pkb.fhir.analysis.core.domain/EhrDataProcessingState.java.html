package com.pkb.fhir.analysis.core.domain;

import java.time.Instant;
import java.util.HashMap;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

public class EhrDataProcessingState {
    private Long latestProcessedEhrId;
    private Map<String, LocalEhrDataProcessingState> localStates = new ConcurrentHashMap<>();
    private Instant startedAt;
    private Instant finishedAt;
    private boolean isFinished;

    public EhrDataProcessingState(Instant startedAt, Long initialMaxEhrId) {
        latestProcessedEhrId = initialMaxEhrId;
        this.startedAt = startedAt;
        isFinished = false;
    }

    public Long getLatestProcessedEhrId() {
        return latestProcessedEhrId;
    }

    public void setLatestProcessedEhrId(Long value) {
        latestProcessedEhrId = value;
    }

    public void registerProcessingState(String processId) {
        localStates.put(processId, new LocalEhrDataProcessingState());
    }

    public LocalEhrDataProcessingState getLocalState(String key) {
        LocalEhrDataProcessingState result = localStates.get(key);
        if (result == null) {
            throw new IllegalStateException("Unregistered processor: " + key);
        }
        return result;
    }

    public Map<String, LocalEhrDataProcessingState> getLocalStates() {
        return new HashMap<>(localStates);
    }

    public Long getAllProcessed() {
        return localStates.values().stream().mapToLong(LocalEhrDataProcessingState::getProcessed).sum();
    }

    public Long getAllFailure() {
        return localStates.values().stream().mapToLong(LocalEhrDataProcessingState::getFailure).sum();
    }

    public Long getAllSuccess() {
        return localStates.values().stream().mapToLong(LocalEhrDataProcessingState::getSuccess).sum();
    }

    public Instant getStartedAt() {
        return startedAt;
    }

    public Instant getFinishedAt() {
        return finishedAt;
    }

    public void registerEnd(Instant end) {
        finishedAt = end;
        isFinished = true;
    }

    public Long getRuntimeSeconds() {
        return finishedAt != null ? finishedAt.getEpochSecond() - startedAt.getEpochSecond() : Instant.now().getEpochSecond() - startedAt.getEpochSecond();
    }

    public boolean isFinished() {
        return isFinished;
    }
}
