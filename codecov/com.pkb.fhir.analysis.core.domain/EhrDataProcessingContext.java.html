package com.pkb.fhir.analysis.core.domain;

import com.pkb.entities.enums.MenuDataType;
import com.pkb.fhir.analysis.core.threading.EhrDataProcessorWorker;

/**
 * The processing state is maintained/updated by the workers. {@link EhrDataProcessorWorker#updateProcessingState}
 */
public class EhrDataProcessingContext {
    private final EhrDataProcessingState state;
    private final MigrationParameters migrationParameters;

    public EhrDataProcessingContext(MigrationParameters migrationParameters, EhrDataProcessingState state) {
        this.state = state;
        this.migrationParameters = migrationParameters;
    }

    public int getDataPageSize() {
        return migrationParameters.getDataPageSize();
    }

    public EhrDataProcessingState getState() {
        return state;
    }

    public MenuDataType getMenuDataType() {
        return migrationParameters.getMenuDataType();
    }

    public Long getMaxId() {
        return migrationParameters.getEndBeforeId();
    }

    public boolean isOverrideExisting(){
        return migrationParameters.isOverrideExisting();
    }
}
