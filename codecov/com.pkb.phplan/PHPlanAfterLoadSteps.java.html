package com.pkb.phplan;

import com.pkb.app.entity.EHRData;
import com.pkb.data.RemotingBiConsumer;
import com.pkb.exception.DecryptionFailedException;
import com.pkb.phplan.entity.PHPlan;

import java.util.function.BiConsumer;

public class PHPlanAfterLoadSteps implements RemotingBiConsumer<PHPlan> {
    @Override
    public BiConsumer<EHRData, PHPlan> getConsumer() {
        return (ehrData, plan) -> {
            // Hack - old records are wrong in xml01
            plan.setStatus(PHPlan.PlanStatus.valueOf(ehrData.getString02()));
            plan.setEditableStatus(ehrData.getString05() == null
                    ? PHPlan.EditableStatus.ALL
                    : PHPlan.EditableStatus.valueOf(ehrData.getString05()));

            if (ehrData.getMigrationVersion() < 1) {
                // TODO: implement trim migrator
                throw new DecryptionFailedException("Cannot migrate plans from trim");
            }
        };
    }
}
