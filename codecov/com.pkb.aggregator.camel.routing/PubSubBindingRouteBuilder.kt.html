package com.pkb.aggregator.camel.routing

import com.pkb.aggregator.camel.GooglePubSubHelper
import com.pkb.aggregator.camel.beanMethod
import com.pkb.aggregator.camel.routing.resource.ResourceUpdateRouteBuilder.Companion.INCOMING_UPDATE_NOTIFICATION
import com.pkb.aggregator.configuration.PubSubConfiguration
import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import org.apache.camel.component.google.pubsub.GooglePubsubComponent
import org.apache.commons.lang3.StringUtils.isBlank
import org.springframework.beans.factory.annotation.Value
import org.springframework.stereotype.Component

/**
 * This component is responsible for receiving notifications from pubsub and rerouting them for processing
 */
@Component
class PubSubBindingRouteBuilder(
    private val configuration: PubSubConfiguration,
    private val pubSubNamespaceService: PubSubNamespaceService,
    @Value("#{environment.GOOGLE_APPLICATION_CREDENTIALS}") private val serviceAccountKey: String?
) : AbstractNamespaceAwareRouteBuilder() {

    override fun configure() {

        log.info("Pubsub configuration: {}", configuration)
        if (!configuration.enabled) return

        configuration.emulatorEndpoint?.let { emulatorEndpoint ->
            (context.getComponent("google-pubsub") as GooglePubsubComponent).apply {
                endpoint = emulatorEndpoint
                isAuthenticate = false
            }
        }

        from(uri(configuration.project, configuration.subscription)).routeId("incoming-pubsub-change")
            .beanMethod(GooglePubSubHelper::parseIncomingMessage).id("parse-message")
            .to(INCOMING_UPDATE_NOTIFICATION).id("process-message")
            .beanMethod(GooglePubSubHelper::checkForErrorRequiringRedelivery).id("check-for-error")
    }

    override fun camelNamespaceService(): PubSubNamespaceService = pubSubNamespaceService

    override fun throwExceptionForNamespaceMismatch(): Boolean = false

    private fun uri(project: String, subscription: String): String {
        val maybeKey = if (isBlank(serviceAccountKey)) "" else "?serviceAccountKey=file://$serviceAccountKey"
        return "google-pubsub:$project:$subscription$maybeKey".also {
            log.info("Route: {}", it)
        }
    }
}
