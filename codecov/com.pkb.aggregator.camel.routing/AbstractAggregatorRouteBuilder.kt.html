package com.pkb.aggregator.camel.routing

import ca.uhn.fhir.rest.server.exceptions.UnprocessableEntityException
import com.pkb.aggregator.camel.UPSTREAM_SOURCE_ID_PROPERTY
import com.pkb.aggregator.camel.routing.resource.MissingReferenceProcessor
import com.pkb.aggregator.service.exception.DownstreamResourceExistsException
import com.pkb.aggregator.service.exception.DownstreamVersionConflictException
import com.pkb.aggregator.service.exception.UpstreamReferenceException
import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import org.apache.camel.LoggingLevel

abstract class AbstractAggregatorRouteBuilder(
    private val pubSubNamespaceService: PubSubNamespaceService,
) : AbstractNamespaceAwareRouteBuilder() {

    override fun camelNamespaceService() = pubSubNamespaceService
    override fun throwExceptionForNamespaceMismatch() = true

    protected fun registerDeadLetter() {
        errorHandler(deadLetterChannel(DeadLetterRouteBuilder.DEAD_LETTER)
            .allowRedeliveryWhileStopping(false))
    }

    protected fun registerAllErrorHandlers(missingReferenceProcessor: MissingReferenceProcessor) {
        registerDeadLetter()
        registerMissingReferenceRetryHandler(missingReferenceProcessor)
        registerDownstreamDuplicateHandler()
        registerVersionConflictRetryHandler()
    }

    protected fun registerDownstreamDuplicateHandler() {
        /*
        This generally shouldn't happen, but is possible in niche situations e.g. a retry is scheduled to load a missing
        resource but some other thread processes the create message at the same time. We'll log a warning, but it's safe
        to continue processing
        */
        onException(DownstreamResourceExistsException::class.java)
            .continued(true)
            .log(LoggingLevel.WARN, "\${exchangeProperty.$UPSTREAM_SOURCE_ID_PROPERTY} already exists downstream; skipping")
    }

    private fun registerMissingReferenceRetryHandler(missingReferenceProcessor: MissingReferenceProcessor) {

        onException(UpstreamReferenceException::class.java, UnprocessableEntityException::class.java)
            .maximumRedeliveries(missingReferenceProcessor.maxRedeliveries)
            .redeliveryDelay(missingReferenceProcessor.redeliveryDelayMS)
            .useExponentialBackOff()
            .backOffMultiplier(missingReferenceProcessor.backoffMultiplier)
            .onExceptionOccurred(missingReferenceProcessor)
            .handled(true)
            .log(LoggingLevel.ERROR, "Retries exhausted for resource \${exchangeProperty.$UPSTREAM_SOURCE_ID_PROPERTY}")
    }

    private fun registerVersionConflictRetryHandler() {

        // When a version conflict happens, we want to have pubsub redeliver the message and process it again from
        // scratch (since the message only failed due to unfortunate timing, it should eventually succeed if we retry
        // enough). However, we can't fail the camel exchange otherwise it will end up in the dead letter handler and
        // the parent pubsub route will never know it failed. So, we continue processing and rely on
        // PubSubBindingRouteBuilder to detect the exception and nack the message
        onException(DownstreamVersionConflictException::class.java)
            .continued(true)
            .log(LoggingLevel.DEBUG, "Continuing after conflict detected for \${exchangeProperty.$UPSTREAM_SOURCE_ID_PROPERTY}")
    }
}
