package com.pkb.aggregator.camel.routing

import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import org.springframework.stereotype.Component

@Component
class DeadLetterRouteBuilder(private val pubSubNamespaceService: PubSubNamespaceService)
    : AbstractNamespaceAwareRouteBuilder() {

    companion object {
        const val DEAD_LETTER = "direct:aggregator/deadletter"
    }

    override fun configure() {

        // For now we just log an error. In future we might want to publish this on some pubsub backchannel and notify
        // customers/reporting etc
        from(DEAD_LETTER)
            .to("log:dead")
    }

    override fun camelNamespaceService() = pubSubNamespaceService

    override fun throwExceptionForNamespaceMismatch() = true
}
