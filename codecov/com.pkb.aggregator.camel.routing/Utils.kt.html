package com.pkb.aggregator.camel.routing

import com.pkb.aggregator.configuration.FhirEndpoint
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamIdentifier
import org.hl7.fhir.instance.model.api.IIdType
import java.net.URLEncoder

fun createFhirIdentifierSearch(identifiers: List<UpstreamIdentifier>) =
    identifiers.distinct().joinToString(",") {
        urlEncode("${it.upstreamIdentifierSystem}|${it.upstreamIdentifierValue}")
    }

fun urlEncode(s: String) = URLEncoder.encode(s, Charsets.UTF_8)!!

fun createFhirSearchRoute(fhirEndpoint: FhirEndpoint) =
    "fhir:search/searchByUrl?${createCommonRouteParams(fhirEndpoint)}"

fun createFhirReadRoute(fhirEndpoint: FhirEndpoint, id: IIdType) =
    "fhir:read/resourceByUrl?url=${id.resourceType}/${id.idPart}" +
            "&resourceClass=${id.resourceType}" +
            "&${createCommonRouteParams(fhirEndpoint)}"

fun createFhirTransactionRoute(fhirEndpoint: FhirEndpoint) =
    "fhir:transaction/withBundle?inBody=bundle" +
            "&${createCommonRouteParams(fhirEndpoint)}"

fun createFhirUpdateRoute(fhirEndpoint: FhirEndpoint) =
    "fhir:update/resource?inBody=resource" +
            "&${createCommonRouteParams(fhirEndpoint)}"

private fun createCommonRouteParams(fhirEndpoint: FhirEndpoint) =
    "log=${fhirEndpoint.printTrace}" +
            "&clientFactory=${resolveClientFactory(fhirEndpoint)}" +
            "&fhirContext=${resolveFhirContext(fhirEndpoint)}" +
            "&serverUrl=${fhirEndpoint.serverUrl}" +
            "&validationMode=NEVER" +
            "&prettyPrint=${fhirEndpoint.printTrace}" +
            "&fhirVersion=${fhirEndpoint.fhirVersion}"

private fun resolveClientFactory(fhirEndpoint: FhirEndpoint) = when (fhirEndpoint.fhirVersion) {
    "R4" -> "#okhttpR4ClientFactory"
    else -> "#okhttpDstu3ClientFactory"
}

private fun resolveFhirContext(fhirEndpoint: FhirEndpoint) = when (fhirEndpoint.fhirVersion) {
    "R4" -> "#fhirContextR4"
    else -> "#fhirContextDstu3"
}
