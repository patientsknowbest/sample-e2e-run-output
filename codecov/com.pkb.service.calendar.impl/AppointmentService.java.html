package com.pkb.service.calendar.impl;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.calendar.entity.AppointmentDTO;
import com.pkb.calendar.entity.AppointmentRequestDTO;
import com.pkb.exception.PKBException;
import io.vavr.Tuple2;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Component;

import java.util.UUID;

@Component
public class AppointmentService {
    private final CalendarManager calendarManager;

    @Autowired
    public AppointmentService(@Lazy CalendarManager calendarManager) {
        this.calendarManager = calendarManager;
    }

    public Tuple2<Long, UUID> createAppointment(SourceDetails sourceDetails, @NotNull EHRRequestContext requestContext, AppointmentDTO dto, Long accountId) {
        try {
            if (dto.getId() != null) {
                //noinspection ThrowCaughtLocally
                throw new IllegalArgumentException("Appointment id must be null to create a new appointment");
            }
            populateAppointmentRequestListSource(sourceDetails, dto);
            // For now, let the create method save without synching
            // We don't supported nested transactions and this is safe anyway
            return calendarManager.saveAppointment(requestContext, dto, accountId, false);
        } catch (Exception e) {
            throw new PKBException("Exception saving appointment for acct " + accountId, e);
        }
    }

    public void populateAppointmentRequestListSource(SourceDetails sourceDetails, AppointmentDTO dto) {
        for (AppointmentRequestDTO reqDto : dto.getAppointmentRequestList()) {
            if (!reqDto.getSource().isPopulated()) {
                reqDto.getSource().init(sourceDetails);
            }
        }
    }
}
