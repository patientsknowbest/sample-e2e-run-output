package com.pkb.emis.appointment.model;

import org.immutables.value.Value;

import java.time.ZoneId;
import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;
import java.util.List;
import java.util.Locale;
import java.util.Optional;

@Value.Immutable
@Value.Style(stagedBuilder = true)
public abstract class EmisAvailableAppointmentSlots {

    public abstract List<AvailableAppointmentSlot> availableAppointmentSlots();

    public AvailableAppointmentSlot getAvailableAppointmentSlotWithSlotId(long slotId) {
        return io.vavr.collection.List.ofAll(availableAppointmentSlots())
                .find(availableAppointmentSlot -> availableAppointmentSlot.getSlotId() == slotId)
                .getOrNull();
    }

    public AvailableAppointmentSlot getFirstAvailableAppointmentSlotInMonth(long month) {
        return io.vavr.collection.List.ofAll(availableAppointmentSlots())
                .find(availableAppointmentSlot -> availableAppointmentSlot.getStartTime().getMonthValue() == month)
                .getOrNull();
    }

    @Value.Immutable
    public static abstract class AvailableAppointmentSlot {
        private static final DateTimeFormatter WEEKDAY_DAY_MONTH_YEAR = DateTimeFormatter.ofPattern("EEE d MMMM yyyy")
                .withZone(ZoneId.of("Europe/London")).withLocale(
                        Locale.ENGLISH);
        private static final DateTimeFormatter HOURS_MINUTES = DateTimeFormatter.ofPattern("HH:mm");

        public abstract long getSlotId();

        public abstract long getSessionId();

        public abstract String clinician();

        public abstract String location();

        public abstract String type();

        public abstract ZonedDateTime getStartTime();

        public abstract ZonedDateTime getEndTime();

        public abstract Optional<String> getSlotTypeName();

        public abstract String getSlotTypeStatus();

        public String getFormattedDate() {
            return getStartTime().format(WEEKDAY_DAY_MONTH_YEAR);
        }

        public String getTime() {
            return getStartTime().format(HOURS_MINUTES);
        }
    }
}
