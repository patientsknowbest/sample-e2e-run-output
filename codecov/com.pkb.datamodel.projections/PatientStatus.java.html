package com.pkb.datamodel.projections;

import com.pkb.datamodel.NationalIdSupport;
import io.vavr.control.Option;
import org.immutables.value.Value;
import org.immutables.vavr.encodings.VavrEncodingEnabled;

import java.time.LocalDate;

@Value.Immutable
@Value.Style(builder = "patientStatus", stagedBuilder = true)
@VavrEncodingEnabled
public interface PatientStatus extends NationalIdSupport {

    enum DIAGNOSTICS {
        NOT_FOUND, NO_ACCESS
    }

    Option<Boolean> getRegistered();

    Option<Boolean> getDeceased();

    Option<LocalDate> getDateOfBirth();

    Option<DIAGNOSTICS> getDiagnostics();
}
