package com.pkb.phoenix.api.model;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import org.immutables.value.Value.Immutable;

/**
 * Common super type for anything, that can be identified by an integer id (E.g.: a {@link PersistedTask} or an {@link Approval}).
 */
public interface Identifiable {
    int id();

    // An abstract class is annotated with @Immutable
    // because the super interface is extended in other locations and
    // adding the annotation would cause - rightfully - warnings.
    /**
     * In {@link Approval} a task is only referred by its ID.
     * For that reason we need a serializable/deserializable model.
     */
    @Immutable
    @JsonSerialize(as = ImmutableIdentifiable.class)
    @JsonDeserialize(as = ImmutableIdentifiable.class)
    abstract class AbstractIdentifiable implements Identifiable {
    }
}
