package com.pkb.phoenix.api.model;

import com.fasterxml.jackson.annotation.JsonProperty;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.fasterxml.jackson.databind.annotation.JsonTypeIdResolver;
import com.pkb.phoenix.api.model.TaskParameterTypeIdResolver.PersistedTaskParameterTypeIdResolver;
import com.pkb.phoenix.api.service.Parameter;
import com.pkb.phoenix.api.service.task.template.TaskTemplateCode;
import org.immutables.value.Value.Immutable;
import org.jetbrains.annotations.Nullable;

/**
 * API model for an already saved task.
 * @param <PARAMETER> Data structure specific to a task (template).
 */
@JsonTypeIdResolver(PersistedTaskParameterTypeIdResolver.class)
public interface PersistedTask<PARAMETER extends Parameter> extends Identifiable, Task<PARAMETER> {
    // An abstract class is annotated with @Immutable
    // because the super interface is extended in other locations and
    // adding the annotation would cause - rightfully - warnings.
    @Immutable
    @JsonSerialize(as = ImmutablePersistedTask.class)
    @JsonDeserialize(as = ImmutablePersistedTask.class)
    abstract class AbstractPersistedTask<PARAMETER extends Parameter> implements PersistedTask<PARAMETER> {
        public static abstract class Builder<PARAMETER extends Parameter> implements TaskBuilder<PARAMETER, Builder<PARAMETER>> {
        }
    }

    String status();

    @Nullable String freshdesk();

    @JsonProperty("createdby")
    User createdby();

    @JsonProperty("createdat")
    String createdat();

    @SuppressWarnings("override")
    interface TaskBuilder<P extends Parameter, SELF extends TaskBuilder<P, SELF>> {
        SELF id(int id);

        SELF code(TaskTemplateCode code);

        SELF status(String status);

        SELF freshdesk(@Nullable String freshdesk);

        SELF parameters(P parameters);

        SELF createdby(User createdby);

        SELF createdat(String createdat);

        PersistedTask<P> build();
    }
}
