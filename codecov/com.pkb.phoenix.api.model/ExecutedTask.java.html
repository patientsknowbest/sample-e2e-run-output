package com.pkb.phoenix.api.model;

import com.fasterxml.jackson.annotation.JsonProperty;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.fasterxml.jackson.databind.annotation.JsonTypeIdResolver;
import com.google.errorprone.annotations.CanIgnoreReturnValue;
import com.pkb.phoenix.api.model.TaskParameterTypeIdResolver.ExecutedTaskParameterTypeIdResolver;
import com.pkb.phoenix.api.service.Parameter;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;

import java.util.Optional;

/**
 * API model to represent an executed Task.
 * @param <PARAMETER> Task specific set of data needed to carry out some action in the system.
 * @param <RESULT> Task specific set od data produced as a result of carrying out a given task.
 */
@JsonTypeIdResolver(ExecutedTaskParameterTypeIdResolver.class)
public interface ExecutedTask<PARAMETER extends Parameter, RESULT> extends PersistedTask<PARAMETER> {
    // An abstract class is annotated with @Immutable
    // because the super interface is extended in other locations and
    // adding the annotation would cause - rightfully - warnings.
    @Immutable
    @JsonSerialize(as = ImmutableExecutedTask.class)
    @JsonDeserialize(as = ImmutableExecutedTask.class)
    @Style(passAnnotations = JsonProperty.class)
    abstract class AbstractExecutedTask<P extends Parameter, R> implements ExecutedTask<P, R> {
        interface Builder<P extends Parameter, R> extends ExecutedTask.Builder<P, R> {
        }
    }

    Optional<RESULT> result();

    @JsonProperty("executedby")
    User executedby();

    @JsonProperty("executedat")
    String executedat();

    @SuppressWarnings("override")
    interface Builder<PARAMETER extends Parameter, RESULT> extends TaskBuilder<PARAMETER, Builder<PARAMETER, RESULT>> {
        @CanIgnoreReturnValue
        Builder<PARAMETER, RESULT> result(@SuppressWarnings("OptionalUsedAsFieldOrParameterType") Optional<? extends RESULT> result);

        @CanIgnoreReturnValue
        Builder<PARAMETER, RESULT> executedby(User user);

        @CanIgnoreReturnValue
        Builder<PARAMETER, RESULT> executedat(String executedAt);
    }
}
