package com.pkb.phoenix.api.model;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.fasterxml.jackson.databind.annotation.JsonTypeIdResolver;
import com.pkb.phoenix.api.model.TaskParameterTypeIdResolver.CreateTaskParameterTypeIdResolver;
import com.pkb.phoenix.api.service.Parameter;
import org.immutables.value.Value.Immutable;

/**
 * API model for instantiating a new task (template) with a given set of parameters.
 * @param <PARAMETER>
 */
@JsonTypeIdResolver(CreateTaskParameterTypeIdResolver.class)
public interface CreateTask<PARAMETER extends Parameter> extends Task<PARAMETER> {
    // An abstract class is annotated with @Immutable
    // because the super interface is extended in other locations and
    // adding the annotation would cause - rightfully - warnings.
    @Immutable
    @JsonSerialize(as = ImmutableCreateTask.class)
    @JsonDeserialize(as = ImmutableCreateTask.class)
    abstract class AbstractCreateTask<PARAMETER extends Parameter> implements CreateTask<PARAMETER> {
    }
}
