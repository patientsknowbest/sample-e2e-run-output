package com.pkb.phoenix.api.model;

import com.fasterxml.jackson.annotation.JsonTypeInfo;
import com.fasterxml.jackson.databind.DatabindContext;
import com.fasterxml.jackson.databind.JavaType;
import com.fasterxml.jackson.databind.jsontype.impl.TypeIdResolverBase;
import com.pkb.phoenix.api.model.CreateTask.AbstractCreateTask;
import com.pkb.phoenix.api.model.ExecutedTask.AbstractExecutedTask;
import com.pkb.phoenix.api.model.PersistedTask.AbstractPersistedTask;
import com.pkb.phoenix.api.service.task.template.TaskTemplateCode;

import static com.fasterxml.jackson.annotation.JsonTypeInfo.Id.CUSTOM;
import static java.lang.String.format;

public abstract class TaskParameterTypeIdResolver extends TypeIdResolverBase {
    public static final class CreateTaskParameterTypeIdResolver extends TaskParameterTypeIdResolver {
        public CreateTaskParameterTypeIdResolver() {
            super(CreateTask.class, AbstractCreateTask.class);
        }
    }

    public static final class PersistedTaskParameterTypeIdResolver extends TaskParameterTypeIdResolver {
        public PersistedTaskParameterTypeIdResolver() {
            super(PersistedTask.class, AbstractPersistedTask.class);
        }
    }

    public static final class ExecutedTaskParameterTypeIdResolver extends TaskParameterTypeIdResolver {
        public ExecutedTaskParameterTypeIdResolver() {
            super(ExecutedTask.class, AbstractExecutedTask.class);
        }

        @Override
        public JavaType typeFromId(DatabindContext context, String id) {
            var template = TaskTemplateCode.valueOf(id);
            var parameterType = context.getTypeFactory().constructSimpleType(template.parameterType(), null);
            var resultType = context.getTypeFactory().constructSimpleType(template.resultType(), null);
            return context.getTypeFactory().constructSimpleType(AbstractExecutedTask.class, new JavaType[]{parameterType, resultType});
        }
    }

    @SuppressWarnings("rawtypes")
    private final Class<? extends Task> minimumSuperType;
    @SuppressWarnings("rawtypes")
    private final Class<? extends Task> serializationType;

    @SuppressWarnings("rawtypes")
    TaskParameterTypeIdResolver(Class<? extends Task> minimumSuperType, Class<? extends Task> serializationType) {
        this.minimumSuperType = minimumSuperType;
        this.serializationType = serializationType;
    }

    @Override
    public String idFromValue(Object value) {
        if (!minimumSuperType.isInstance(value)) {
            throw new IllegalStateException(format("Wrong type id resolver [%s] is used for %s", value.getClass(), minimumSuperType));
        }

        return minimumSuperType.cast(value).code().name();
    }

    @Override
    public JavaType typeFromId(DatabindContext context, String id) {
        var parameterType = context.getTypeFactory().constructSimpleType(TaskTemplateCode.valueOf(id).parameterType(), null);
        return context.getTypeFactory().constructSimpleType(serializationType, new JavaType[]{parameterType});
    }

    @Override
    public String idFromValueAndType(Object value, Class<?> suggestedType) {
        return idFromValue(value);
    }

    @Override
    public JsonTypeInfo.Id getMechanism() {
        return CUSTOM;
    }
}
