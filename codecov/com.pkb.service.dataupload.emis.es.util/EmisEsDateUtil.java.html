package com.pkb.service.dataupload.emis.es.util;

import com.pkb.institute.entity.Org;
import org.jetbrains.annotations.Nullable;

import java.time.Instant;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.time.ZoneId;
import java.time.format.DateTimeFormatter;
import java.util.Optional;

import static com.pkb.service.dataupload.emis.esv2.model.BaseEmisEsDomainModel.DATE_FORMAT_PATTERN;

/**
 * Utils for date handling in EMIS ES Extract data
 */
public class EmisEsDateUtil {
    public static final ZoneId DEFAULT_ZONE_ID = ZoneId.of("Europe/London");
    // A set of sane defaults for date formats
    public static final DateTimeFormatter DEFAULT_DATE_FORMATTER = DateTimeFormatter.ofPattern(DATE_FORMAT_PATTERN);

    /**
     * Converts LocalDate to ZonedDateTime in zoneId at start of day
     * then returns instance of java.util.Date.
     * Returns empty if any of the input arguments are null.
     */
    public static Optional<Instant> parseDate(@Nullable LocalDate localDate, @Nullable ZoneId zoneId) {
        return Optional.ofNullable(localDate)
                .flatMap(d -> Optional.ofNullable(zoneId).flatMap(z -> Optional.of(d.atStartOfDay(z).toInstant())));
    }

    public static Optional<Instant> parseDate(@Nullable LocalDateTime localDateTime, @Nullable ZoneId zoneId) {
        return Optional.ofNullable(localDateTime)
                .flatMap(d -> Optional.ofNullable(zoneId).flatMap(z -> Optional.of(d.atZone(z).toInstant())));
    }

    public static LocalDate parseLocalDate(Instant date, ZoneId zoneId) {
        return date.atZone(zoneId).toLocalDate();
    }

    public static ZoneId getTimezoneOfOrg(Org o, Runnable missingZoneIdHandler) {
        return o.findZoneId().orElseGet(() -> {
            missingZoneIdHandler.run();
            return DEFAULT_ZONE_ID;
        });
    }

    public static ZoneId getTimezoneOfOrg(Org o) {
        return getTimezoneOfOrg(o, () -> {
        });
    }
}
