package com.pkb.service.dataupload.emis.es.util;

import com.pkb.VersionDetails;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.entities.enums.Gender;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.institute.entity.Org;
import com.pkb.service.dataupload.emis.esv2.model.Constants;
import com.pkb.user.entity.NationalId;
import com.pkb.user.entity.OrgLevelId;
import com.pkb.user.entity.OrgLevelIdType;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Optional;

public class EmisEsDemographicsUtil {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    /**
     * Gender throws an IllegalArgumentException if an invalid value is provided. It is up to the specific APIs to decide what
     * that means for them.
     * <p>
     * For EMIS, we will return Gender.UNKNOWN.
     *
     * @param emisGender The EMIS gender value to be matched
     * @return The matching {@link Gender}
     */
    public static Gender convertGender(String emisGender) {
        try {
            return Gender.getByWord(emisGender);
        } catch (IllegalArgumentException ignored) {
            LOGGER.warn("Received unexpected gender code: {}", emisGender);
            return Gender.UNKNOWN;
        }
    }

    public static Optional<NationalId> parseNhsNumber(Optional<String> nhsNumber, EHRRequestContext requestContext) {
        Optional<NationalId> result = nhsNumber
                .flatMap(NationalIdType.NHS_NUMBER::getValidNationalIdAndType)
                .map(t -> new NationalId(t, VersionDetails.of(requestContext)));
        if (result.isEmpty()) {
            LOGGER.warn("Invalid NHS Number provided, nhsNumber=[{}]", nhsNumber);
        }
        return result;
    }

    public static Optional<OrgLevelId> parsePatientNumber(Optional<Integer> patientNumber, Org org, EHRRequestContext requestContext) {
        return patientNumber.map(olid -> {
            OrgLevelIdType olidt = org.getOrgLevelIdType(Constants.ORG_LEVEL_ID_TYPE_AA, Constants.ORG_LEVEL_ID_TYPE_TC);
            if (olidt != null) {
                OrgLevelId id = new OrgLevelId(VersionDetails.of(requestContext));
                id.setOrg(org);
                id.setType(olidt);
                id.setValue(patientNumber.get().toString());
                return id;
            } else {
                throw new IllegalStateException("No orglevelidtype set for this GP Practice. Org=" + org.getId());
            }
        });
    }
}
