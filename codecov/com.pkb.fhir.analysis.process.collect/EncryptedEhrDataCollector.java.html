package com.pkb.fhir.analysis.process.collect;

import com.pkb.fhir.analysis.process.domain.EhrDataCollectionParameters;
import com.pkb.fhir.common.repository.MenuDataRepository;
import org.springframework.data.domain.PageRequest;
import org.springframework.data.domain.Sort;

import java.util.List;

public class EncryptedEhrDataCollector implements EhrDataCollector {
    private final MenuDataRepository menuDataRepository;

    public EncryptedEhrDataCollector(MenuDataRepository menuDataRepository) {
        this.menuDataRepository = menuDataRepository;
    }

    @Override
    public List<Long> collect(EhrDataCollectionParameters parameters) {
        return menuDataRepository.findAllByMenuDataType(parameters.getMenuDataType(), parameters.getMinId(), parameters.getMaxId(), PageRequest.ofSize(parameters.getDataPageSize()).withSort(Sort.by("id")));

    }
}
