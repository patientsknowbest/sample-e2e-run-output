package com.pkb.api.fhir.address;

import com.pkb.datamodel.user.PostalAddress;
import org.apache.commons.lang3.StringUtils;
import org.hl7.fhir.dstu3.model.Address;

public class AddressMapper {

    public Address toAddress(PostalAddress postalAddress) {
        Address address = new Address();
        postalAddress.getAddress1().filter(StringUtils::isNotBlank).peek(address::addLine);
        postalAddress.getAddress2().filter(StringUtils::isNotBlank).peek(address::addLine);
        postalAddress.getCity().filter(StringUtils::isNotBlank).peek(address::setCity);
        postalAddress.getState().filter(StringUtils::isNotBlank).peek(address::setState);
        postalAddress.getCountry().filter(StringUtils::isNotBlank).peek(address::setCountry);
        postalAddress.getPostCode().filter(StringUtils::isNotBlank).peek(address::setPostalCode);
        return address;
    }

}
