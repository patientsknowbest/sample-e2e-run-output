package com.pkb.aggregator.integration.hapi

import ca.uhn.fhir.context.FhirContext
import ca.uhn.fhir.rest.api.RequestTypeEnum
import ca.uhn.fhir.rest.client.api.Header
import ca.uhn.fhir.rest.client.api.IHttpClient
import ca.uhn.fhir.rest.client.impl.RestfulClientFactory
import io.micrometer.core.instrument.binder.okhttp3.OkHttpMetricsEventListener
import okhttp3.Call
import okhttp3.OkHttpClient
import org.apache.http.client.utils.URIBuilder
import org.slf4j.LoggerFactory
import java.lang.invoke.MethodHandles
import java.net.InetSocketAddress
import java.net.Proxy
import java.util.concurrent.TimeUnit

/**
 * This is basically a re-implementation of HAPI's OkHttpRestfulClientFactory with two specific improvements:
 * 1. It uses a client template rather than creating a new client from scratch, which means that any config set on the
 * client (such as metrics integration) is not lost when [resetHttpClient] is called
 * 2. The target FHIR resource is identified when the client is accessed and correctly recorded in the `uri` tag on the
 * outgoing metric
 *
 * [HAPI client configuration docs](https://hapifhir.io/hapi-fhir/docs/client/client_configuration.html)
 *
 * @see ca.uhn.fhir.okhttp.client.OkHttpRestfulClientFactory
 */
class BetterOkHttpRestfulClientFactory(
    fhirContext: FhirContext,
    private val clientTemplate: OkHttpClient,
    private var nativeClient: Call.Factory? = null
) : RestfulClientFactory(fhirContext) {

    private val log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun getHttpClient(theServerBase: String) = OkHttpRestfulClient(
        this.getNativeClient(),
        StringBuilder(theServerBase),
        null,
        null,
        null,
        null
    )

    override fun getHttpClient(
        theUrl: StringBuilder?,
        theIfNoneExistParams: MutableMap<String, MutableList<String>>?,
        theIfNoneExistString: String?,
        theRequestType: RequestTypeEnum?,
        theHeaders: MutableList<Header>?
    ): IHttpClient {
        identifyResource(theUrl)?.run {
            theHeaders?.add(Header(OkHttpMetricsEventListener.URI_PATTERN, this))
        }
        return OkHttpRestfulClient(
            getNativeClient(),
            theUrl,
            theIfNoneExistParams,
            theIfNoneExistString,
            theRequestType,
            theHeaders
        )
    }

    override fun <T : Any?> setHttpClient(theHttpClient: T) {
        nativeClient = theHttpClient as Call.Factory?
    }

    override fun setProxy(theHost: String?, thePort: Int?) {
        val proxy = Proxy(Proxy.Type.HTTP, InetSocketAddress(theHost, thePort!!))
        val builder = (getNativeClient() as OkHttpClient).newBuilder().proxy(proxy)
        setHttpClient(builder.build())
    }

    override fun resetHttpClient() {
        nativeClient = null
    }

    @Synchronized
    private fun getNativeClient(): Call.Factory {
        if (nativeClient == null) {
            nativeClient = clientTemplate.newBuilder()
                .connectTimeout(this.connectTimeout.toLong(), TimeUnit.MILLISECONDS)
                .readTimeout(this.socketTimeout.toLong(), TimeUnit.MILLISECONDS)
                .writeTimeout(this.socketTimeout.toLong(), TimeUnit.MILLISECONDS)
                .build()
        }

        return nativeClient!!
    }

    private fun identifyResource(theUrl: StringBuilder?): String? = try {
        /*
        Paths will be of the form /fhir/<resource> and that's the only bit we want for metric-tracking purposes.
        It's important to trim off extra path segments when the id is part of the path rather than a query param,
        e.g. we don't want to record a metric for /fhir/Patient/<uuid> as it will create unmanageable cardinality on
        the uri label of the metric
         */
        val pathSegments = URIBuilder(theUrl?.toString()).pathSegments.subList(0, 2)
        URIBuilder().setPathSegments(pathSegments).build().toString()
    } catch (ex: Exception) {
        log.warn("Unable to identify fhir resource from $theUrl, will not be recorded in metrics", ex)
        null
    }
}
