package com.pkb.util.security;

import com.pkb.filesanitizer.FileSanitizer;
import com.pkb.service.virusscan.StreamScanner;
import com.pkb.service.virusscan.StreamScanner.SuccessReason;
import com.pkb.util.FileUtil;
import io.vavr.control.Either;
import org.apache.commons.io.FilenameUtils;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.ByteArrayInputStream;

import static java.lang.String.format;

public class FileSecurityHandler {

    private final static Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final FileSanitizer fileSanitizer;
    private final StreamScanner streamScanner;

    public FileSecurityHandler(FileSanitizer fileSanitizer, StreamScanner streamScanner) {
        this.fileSanitizer = fileSanitizer;
        this.streamScanner = streamScanner;
    }

    public Either<String, String> runChecks(String fileName, @NotNull byte[] data, String originalMediaType) {
        if (FileUtil.isFileExtensionBlacklistedForUpload(fileName)) {
            return Either.left(format("Threat found: file extension [%s] blocked", FilenameUtils.getExtension(fileName)));
        }
        Either<String, SuccessReason> scanResult = streamScanner.safeScan(new ByteArrayInputStream(data));
        if (scanResult.isLeft()) {
            return Either.left("Threat found: " + scanResult.getLeft());
        } else {
            LOGGER.info("No threat found: {}", scanResult.get());
        }
        String fixedMediaType = fileSanitizer.fixMediaType(data, originalMediaType);
        return Either.right(fixedMediaType);
    }
}
