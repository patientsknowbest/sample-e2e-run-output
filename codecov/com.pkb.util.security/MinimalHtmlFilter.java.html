package com.pkb.util.security;

import org.jsoup.safety.Safelist;

public class MinimalHtmlFilter extends AbstractJsoupFilter {

    public MinimalHtmlFilter(String baseUrl) {
        super(baseUrl);
    }

    private static final Safelist MINIMAL = new Safelist()
            .addTags(
                    "a", "b", "blockquote", "br", "cite", "em",
                    "i", "p", "pre", "q", "small", "span", "strike", "strong", "sub",
                    "sup", "u")
            .addAttributes("a", "href", "target", "title")
            .addAttributes("blockquote", "cite")
            .addAttributes("q", "cite")
            .addProtocols("blockquote", "cite", "http", "https")
            .addProtocols("a", "href", "http", "https", "mailto")
            .addProtocols("cite", "cite", "http", "https");

    @Override
    public String filter(String input) {
        return filter(input, MINIMAL);
    }
}
