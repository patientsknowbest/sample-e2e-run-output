package com.pkb.util.security;

import com.fasterxml.jackson.annotation.JsonIgnore;
import com.fasterxml.jackson.core.util.DefaultIndenter;
import com.fasterxml.jackson.core.util.DefaultPrettyPrinter;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.ObjectWriter;
import com.pkb.datamodel.ImmutablePdfValidationResult;
import com.pkb.datamodel.PdfValidationResult;
import com.pkb.service.uuid.UUIDProvider;
import org.apache.commons.io.input.CountingInputStream;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.verapdf.core.EncryptedPdfException;
import org.verapdf.core.ModelParsingException;
import org.verapdf.core.ValidationException;
import org.verapdf.pdfa.Foundries;
import org.verapdf.pdfa.PDFAParser;
import org.verapdf.pdfa.PDFAValidator;
import org.verapdf.pdfa.VeraGreenfieldFoundryProvider;
import org.verapdf.pdfa.flavours.PDFAFlavour;
import org.verapdf.pdfa.results.TestAssertion;
import org.verapdf.pdfa.results.ValidationResult;

import java.io.ByteArrayInputStream;
import java.io.IOException;
import java.io.InputStream;
import java.lang.invoke.MethodHandles;
import java.util.LinkedHashMap;
import java.util.Map;

public class PdfValidator {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
    private final UUIDProvider uuidProvider;
    private static final ObjectWriter OBJECT_WRITER = new ObjectMapper().addMixIn(TestAssertion.class, OrdinalMixin.class).writer(new DefaultPrettyPrinter()
            .withObjectIndenter(new DefaultIndenter().withLinefeed("\n")));

    public PdfValidator(UUIDProvider uuidProvider) {
        VeraGreenfieldFoundryProvider.initialise();
        this.uuidProvider = uuidProvider;
    }

    public PdfValidationResult validatePDF(byte[] input) throws IOException {
        return validatePDF(new ByteArrayInputStream(input));
    }

    public PdfValidationResult validatePDF(InputStream inputStream) throws IOException {
        var resultBuilder = ImmutablePdfValidationResult.builder();
        resultBuilder.numberOfPages(0L);
        resultBuilder.numberOfBytes(0L);
        resultBuilder.isValidPDFa1b(false);
        resultBuilder.isEncrypted(false);

        PDFAFlavour flavour = PDFAFlavour.fromString("1b");
        resultBuilder.claimedFormat(flavour.name());
        try (CountingInputStream cis = new CountingInputStream(inputStream);
             PDFAParser parser = Foundries.defaultInstance().createParser(cis, flavour)) {
            PDFAValidator validator = Foundries.defaultInstance().createValidator(flavour, false);
            ValidationResult result = validator.validate(parser);
            resultBuilder.claimedFormat(result.getPDFAFlavour().name());
            if (result.isCompliant()) {
                resultBuilder.isValidPDFa1b(true);
                resultBuilder.parseReport("valid PDF/A-1b file");
            } else {
                resultBuilder.isValidPDFa1b(false);
                resultBuilder.parseReport(OBJECT_WRITER.writeValueAsString(result.getTestAssertions()));
            }

            var pdfDocument = parser.getPDFDocument();
            var infoDictionary = pdfDocument.getInfoDictionary();

            Map<String, String> metadata = new LinkedHashMap<>();
            metadata.put("Author", infoDictionary.getAuthor());
            metadata.put("CreationDate", infoDictionary.getCreationDate());
            metadata.put("Creator", infoDictionary.getCreator());
            metadata.put("ModDate", infoDictionary.getModificationDate());
            metadata.put("Producer", infoDictionary.getProducer());
            metadata.put("Title", infoDictionary.getTitle());

            resultBuilder.metadata(OBJECT_WRITER.writeValueAsString(metadata));
            resultBuilder.numberOfBytes(cis.getByteCount());
        } catch (EncryptedPdfException e) {
            var logId = uuidProvider.randomUUID();
            resultBuilder.parseReport("cannot parse encrypted PDF");
            resultBuilder.isEncrypted(true);
            LOGGER.error("PDF encryption error id: {}", logId, e);
        } catch (ModelParsingException | ValidationException e) {
            var logId = uuidProvider.randomUUID();
            resultBuilder.parseReport("error in validator - see logs id: " + logId);
            LOGGER.error("PDF Validation error id: {}", logId, e);
        }

        return resultBuilder.build();
    }

    private abstract class OrdinalMixin {

        @JsonIgnore
        public abstract int getOrdinal();
    }
}
