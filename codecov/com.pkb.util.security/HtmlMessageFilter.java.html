package com.pkb.util.security;

import org.jsoup.safety.Safelist;

public class HtmlMessageFilter extends AbstractJsoupFilter {

    public HtmlMessageFilter(String baseUrl) {
        super(baseUrl);
    }

    static final Safelist STANDARD = Safelist.relaxed()
            .addAttributes("a", "href", "target", "title")
            .addEnforcedAttribute("a", "rel", "nofollow")
            .addProtocols("a", "href", "http", "https", "mailto")
            .addTags("font", "s", "ins", "del", "strike", "tt", "big")
            .removeTags("img", "code", "cite");

    @Override
    public String filter(String input) {
        return filter(input, STANDARD);
    }
}
