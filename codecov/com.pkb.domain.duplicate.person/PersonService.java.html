package com.pkb.domain.duplicate.person;

import com.google.common.annotations.VisibleForTesting;

import static com.pkb.entities.enums.OutOfAreaRepairJobStatus.NEW;
import static com.pkb.entities.enums.UserType.INSTITUTE_ADMIN;
import static com.pkb.entities.enums.UserType.PATIENT;
import static com.pkb.entities.enums.UserType.REG_CLINICIAN;
import static io.vavr.API.$;
import static io.vavr.API.Case;

public class PersonService {

    private final DateTimeService dateTimeService;
    private final PersonRepository personRepository;
    private final OutOfAreaRepairJobRepository outOfAreaRepairJobRepo;
    private final OutOfAreaRepairJobMapper outOfAreaRepairJobMapper;
    private final InvitationRepairJobMapper invitationRepairJobMapper;
    private final InvitationRepairJobRepository invitationRepairJobRepository;
    private final InstituteUserRepository instituteUserRepository;
    private final OrgRepository orgRepository;
    private final PersonMapper personMapper;

    public PersonService(DateTimeService dateTimeService,
                         PersonRepository personRepository,
                         OutOfAreaRepairJobRepository outOfAreaRepairJobRepo,
                         OutOfAreaRepairJobMapper outOfAreaRepairJobMapper,
                         PersonMapper personMapper,
                         InvitationRepairJobMapper invitationRepairJobMapper,
                         InvitationRepairJobRepository invitationRepairJobRepository, InstituteUserRepository instituteUserRepository, OrgRepository orgRepository) {
        this.dateTimeService = dateTimeService;
        this.personRepository = personRepository;
        this.outOfAreaRepairJobRepo = outOfAreaRepairJobRepo;
        this.outOfAreaRepairJobMapper = outOfAreaRepairJobMapper;
        this.personMapper = personMapper;
        this.invitationRepairJobMapper = invitationRepairJobMapper;
        this.invitationRepairJobRepository = invitationRepairJobRepository;
        this.instituteUserRepository = instituteUserRepository;
        this.orgRepository = orgRepository;
    }

    @NotNull
    public Person findPerson(long id) {
        return Objects.requireNonNull(personRepository.findById(id).map(personMapper::entityToDataModel).orElse(null));
    }

    public List<Long> findPersonIdByEmail(@NotNull String email) {
        return personRepository.findPersonIdsByEmail(ImmutableObfuscatingEmail.obfuscatingEmail(email));
    }

    @NotNull
    public Either<RuntimeException, Patient> findPatient(long id) {
        return Try.of(() -> findPatientById(id)
                .orElseThrow(() -> new RuntimeException("Could not find patient " + id)))
                .toEither()
                .mapLeft(RuntimeException::new);
    }

    public List<Long> findPersonIdsByNationalId(ValidNationalId validNationalId) {
        return personRepository.findPersonIdsByValidNationalIdValueAndType(validNationalId);
    }

    private Optional<Patient> findPatientById(@NotNull Long id) {
        return personRepository.findAllPersonsByIdAndUserTypeQuery(id, PATIENT)
                .map(personMapper::entityToPatient);
    }

    public Optional<Person> findByPublicId(@NotNull UUID id) {
        return personRepository.findOptionalByPublicId(id)
                .map(personMapper::entityToDataModel);
    }

    public Map<UUID, Long> findPersonIdsByListOfPublicIds(@NotNull Collection<UUID> publicIds) {
        return personRepository.findPersonIdsByListOfPublicIds(publicIds)
                .stream()
                .collect(Collectors.toMap(
                        arr -> UUID.fromString(arr[0].toString()),
                        arr -> Long.valueOf(arr[1].toString())));
    }

    public Optional<Long> findPersonIdByPublicId(@NotNull UUID publicId) {
        return personRepository.findPersonIdByPublicId(publicId);
    }

    private Optional<Long> findDefaultAccountIdByPersonId(long personId){
        return personRepository.findDefaultAccountIdByPersonId(personId);
    }

    public long getDefaultAccountIdByPersonId(long personId) {
        return findDefaultAccountIdByPersonId(personId).orElseThrow(() -> new RuntimeException(String.format("No default account id was found for person %d", personId)));
    }

    public int registerPerson(UserStatus status, long id) {
        return personRepository.updatePersonStatusAndRegistrationTimestamp(id, status, dateTimeService.now());
    }

    @Transactional
    public OutOfAreaRepairJobExisting addOutOfAreaRepairJob(Patient patient) {

        OutOfAreaRepairJobNew repairJob = ImmutableOutOfAreaRepairJobNew.builder()
                .accountId(patient.requireDefaultAccountId())
                .targetPersonId(patient.getId())
                .status(NEW)
                .build();
        return createNewJob(repairJob);
    }

    @VisibleForTesting
    OutOfAreaRepairJobExisting createNewJob(OutOfAreaRepairJobNew job) {
        PublicOutOfAreaRepairJob entity = outOfAreaRepairJobMapper.dataModelToEntity(job);
        entity = outOfAreaRepairJobRepo.save(entity);
        return outOfAreaRepairJobMapper.entityToDataModel(entity);
    }

    public long getDefaultAccountId(long personId) {
        return personRepository.findById(personId).map(PublicPKBPerson::getDefaultAccountId)
                .orElseThrow(() -> new RuntimeException("Failed to find person for id " + personId));
    }

    @Transactional
    @NotNull
    public InvitationRepairJobExisting addInvitationRepairJob(InvitationRepairJobNew job) {
        PublicInvitationRepairJob entity = invitationRepairJobMapper.dataModelToEntity(job);
        invitationRepairJobRepository.save(entity);
        return invitationRepairJobMapper.entityToDataModel(entity);
    }

    @NotNull
    public List<Person> findEmployeesFetchContacts() {
        return personRepository.findEmployeesFetchContacts()
                .stream()
                .map(personMapper::entityToDataModel)
                .collect(Collectors.toList());
    }

    @NotNull
    public Map<Long, Person> findPersonsByListOfIds(Collection<Long> personIds) {
        return personRepository.findPersonsByListOfIds(personIds)
                .stream().collect(Collectors.toMap(PublicPKBPerson::getId, personMapper::entityToDataModel));
    }

    @NotNull
    public Optional<Long> findPersonIdForDefaultAccountId(long accountId) {
        return personRepository.findPersonIdForDefaultAccountId(accountId);
    }

    /**
     * Key is Account.publicId, value is Person.publicId
     */
    @NotNull
    @Transactional(readOnly = true)
    public Map<UUID, Person> findPersonsForDefaultAccountPublicIds(Collection<UUID> accountPublicIds) {
        if (accountPublicIds.isEmpty()) {
            return Collections.emptyMap();
        }
        return personRepository.findPersonsForDefaultAccountPublicIds(accountPublicIds)
                .map(o -> Tuple.of((UUID)o[0], personMapper.entityToDataModel((PublicPKBPerson) o[1])))
                .collect(Collectors.toMap(Tuple2::_1, Tuple2::_2));
    }

    public long countOrgCoordsWithAccessToOrg(long orgId) {
        return personRepository.countOrgCoordsWithAccessToOrg(orgId);
    }

    public Optional<Person> findOptionalPersonByIdWithTeamsAndAllLazies(long id) {
        var maybePerson = personRepository.findOptionalByIdFetchAllLazies(id);
        return maybePerson.map(p -> {
            var instituteUsers = instituteUserRepository.findAllByPersonIdFetchTeam(p.getId());
            //TODO PNX-190 Fix data collection task because the registering organisation returns with a false value at the moment.
            return Match(p.getUserType())
                    .of(Case($(PATIENT), () -> {
                        if (p.getVersionDetails().getSourceOrgId() != null) {
                            var sourceOrg = orgRepository.findById(p.getVersionDetails().getSourceOrgId());
                            return sourceOrg.map(org -> personMapper.entityToPatient(p, instituteUsers, org)).orElse(personMapper.entityToPatient(p, instituteUsers));
                        } else {
                            return personMapper.entityToPatient(p, instituteUsers);
                        }
                    }),
                            Case($(REG_CLINICIAN), () -> personMapper.entityToProfessional(p, instituteUsers)),
                            Case($(INSTITUTE_ADMIN), () -> personMapper.entityToTeamCoordinator(p, instituteUsers)),
                            Case($(), () -> personMapper.entityToDataModel(p)));
        });
    }


    public boolean existsUserNotContactableByMessages(Set<Long> personIds) {
        return instituteUserRepository.existsUserNotContactableByMessages(personIds);
    }
}
