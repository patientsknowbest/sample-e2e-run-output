package com.pkb.domain.internal.repository;

import io.vavr.collection.List;
import io.vavr.control.Option;
import org.jooq.DSLContext;
import org.jooq.conf.RenderNameCase;
import org.jooq.conf.RenderQuotedNames;
import org.jooq.conf.Settings;
import org.jooq.impl.DSL;

import javax.persistence.EntityManager;
import javax.persistence.PersistenceContext;
import javax.persistence.Query;
import java.util.function.Consumer;

import static org.jooq.SQLDialect.POSTGRES;

/**
 * Service to execute Jooq queries. Not a spring data JPA repository.
 */
public class JooqSupportRepository {
    @PersistenceContext
    private EntityManager em;

    public DSLContext context() {
        Settings jooqSettings = new Settings();
        jooqSettings.setRenderNameCase(RenderNameCase.AS_IS);
        jooqSettings.setRenderQuotedNames(RenderQuotedNames.EXPLICIT_DEFAULT_UNQUOTED);
        return DSL.using(POSTGRES, jooqSettings);
    }

    @SuppressWarnings("unchecked")
    public <T> List<T> nativeListQuery(org.jooq.Query query, Class<T> resultClass) {
        return List.ofAll(convertJooqQueryToJpa(query, resultClass).getResultList());
    }

    @SuppressWarnings("unchecked")
    public <T> T nativeQuery(org.jooq.Query query) {
        return (T) convertJooqQueryToJpa(query, (Class<T>) null).getSingleResult();
    }

    @SuppressWarnings("unchecked")
    public <T> List<T> nativeListQuery(org.jooq.Query query) {
        return List.ofAll(convertJooqQueryToJpa(query).getResultList());
    }

    @SuppressWarnings("unchecked")
    public <T> List<T> nativeListQuery(org.jooq.Query query, Consumer<Query> queryPostProcessor) {
        Query jpaQuery = convertJooqQueryToJpa(query);
        queryPostProcessor.accept(jpaQuery);
        return List.ofAll(jpaQuery.getResultList());
    }

    private Query convertJooqQueryToJpa(org.jooq.Query query) {
        return convertJooqQueryToJpa(query, Option.none());
    }

    private <T> Query convertJooqQueryToJpa(org.jooq.Query query, Class<T> resultClass) {
        return convertJooqQueryToJpa(query, Option.of(resultClass));
    }

    private <T> Query convertJooqQueryToJpa(org.jooq.Query query, Option<Class<T>> resultClass) {
        Query jpaQuery = resultClass
                .map(clazz -> em.createNativeQuery(query.getSQL(), clazz))
                .getOrElse(em.createNativeQuery(query.getSQL()));

        // Extract the bind values from the jOOQ query
        java.util.List<Object> values = query.getBindValues();
        for (int i = 0; i < values.size(); i++) {
            Object value = values.get(i);
            jpaQuery.setParameter(i + 1, value);
        }
        return jpaQuery;
    }
}
