package com.pkb.domain.internal.repository;

import com.google.common.collect.ImmutableList;
import com.google.common.collect.Ordering;
import com.pkb.ehrdata.AccessFilter;
import com.pkb.ehrdata.ImmutableSearchSpec;
import com.pkb.ehrdata.QueryBuilder;
import com.pkb.ehrdata.SearchSpec;
import com.pkb.entities.app.AppMenuData;
import io.vavr.API;
import io.vavr.collection.List;
import io.vavr.collection.Map;
import org.jetbrains.annotations.NotNull;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.Repository;

import java.math.BigInteger;
import java.util.Optional;
import java.util.UUID;

import static com.pkb.annotation.EHRField.QueryField.string03;
import static com.pkb.ehrdata.Filter.Operator.IN;
import static com.pkb.ehrdata.Filter.filter;
import static io.vavr.collection.List.ofAll;
import static java.util.Comparator.comparing;

class CustomAppMenuDataRepositoryImpl extends JooqSupportRepository implements CustomAppMenuDataRepository {

    @Override
    public List<Long> find(SearchSpec searchSpec, AccessFilter accessFilter) {
        return find(ImmutableList.of(searchSpec), accessFilter);
    }

    @Override
    public List<Long> find(ImmutableList<SearchSpec> searchSpecs, AccessFilter accessFilter) {
        org.jooq.Query q = QueryBuilder.newInstance()
                .withSearchSpecs(searchSpecs)
                .withAccessFilter(accessFilter)
                .build();

        // Return value is actually BigInteger, so convert to long
        List<BigInteger> ids = nativeListQuery(q);
        return (ids)
                .map(BigInteger::longValue);
    }

    @Override
    public long count(SearchSpec searchSpec, AccessFilter accessFilter) {
        SearchSpec countAllSpec = ImmutableSearchSpec.copyOf(searchSpec).withPageFilter(Optional.empty());

        org.jooq.Query q = QueryBuilder.newInstanceAsCount()
                .withSearchSpecs(ImmutableList.of(countAllSpec))
                .withAccessFilter(accessFilter)
                .build();

        return ((BigInteger) nativeQuery(q)).longValue();
    }

    // TODO querying DB for each AccessFilter should be fixed by AI-400
    @Override
    public List<Long> findBatch(SearchSpec searchSpec, @NotNull Map<Long, AccessFilter> accessFilters) {
        // Invert the map, QueryBuilder can only handle one accessFilter at a time.
        Map<AccessFilter, List<Long>> patientIdsByAccessFilter = accessFilters.toList().foldLeft(API.Map(), 
                (m, e) -> m.put(e._2, m.get(e._2).getOrElse(API.List()).append(e._1)));
        return patientIdsByAccessFilter
                .flatMap(accessFilterAndIdList -> {
                    AccessFilter accessFilter = accessFilterAndIdList._1;
                    List<Long> patientIds = accessFilterAndIdList._2;
                    org.jooq.Query q = QueryBuilder.newInstance()
                            .withSearchSpec(ImmutableSearchSpec.copyOf(searchSpec)
                                    .withFilters(ofAll(searchSpec.getFilters())
                                            .append(filter(string03.name(), IN, patientIds.map(Object::toString).asJava()))))
                            .withAccessFilter(accessFilter)
                            .build();
                    // Return value is actually BigInteger, so convert to long
                    return this.<BigInteger>nativeListQuery(q).map(BigInteger::longValue);
                }).toList();
    }
}

interface CustomAppMenuDataRepository {
    List<Long> find(SearchSpec searchSpec, AccessFilter accessFilter);

    List<Long> find(ImmutableList<SearchSpec> searchSpecs, AccessFilter accessFilter);

    long count(SearchSpec searchSpec, AccessFilter accessFilter);

    List<Long> findBatch(SearchSpec searchSpec, @NotNull Map<Long, AccessFilter> accessFilters);
}

public interface AppMenuDataRepository extends Repository<AppMenuData, Long>, CustomAppMenuDataRepository {
    /**
     * findByIdsQuery doesn't respect the order of IDs passed in.
     * Re-apply the order here.
     */
    default List<AppMenuData> findByIds(List<Long> ids) {
        return findByIdsUnordered(ids)
                .sorted(comparing(AppMenuData::getId, Ordering.explicit(ids.asJava())));
    }
    @Query("SELECT md FROM AppMenuData md " +
            "LEFT JOIN FETCH md.sourceDetails.team " +
            "LEFT JOIN FETCH md.sourceDetails.org " +
            "LEFT JOIN FETCH md.baseFields.latestPrivacyFlags " +
            "WHERE md.id IN (:ids)")
    List<AppMenuData> findByIdsUnordered(List<Long> ids);

    default List<AppMenuData> findByIdsWithAttachments(List<Long> ids) {
        return findByIdsUnorderedWithAttachments(ids)
                .sorted(comparing(AppMenuData::getId, Ordering.explicit(ids.asJava())));
    }

    /**
     * Join-fetching attachments will result in multiple rows, unless a distinct is added
     */
    @Query("SELECT DISTINCT md FROM AppMenuData md " +
            "LEFT JOIN FETCH md.sourceDetails.team " +
            "LEFT JOIN FETCH md.sourceDetails.org " +
            "LEFT JOIN FETCH md.baseFields.latestPrivacyFlags " +
            "LEFT JOIN FETCH md.attachments " +
            "WHERE md.id IN (:ids)")
    List<AppMenuData> findByIdsUnorderedWithAttachments(List<Long> ids);

    @Query("SELECT md FROM AppMenuData md WHERE md.uuid01 in (?1)")
    List<AppMenuData> findByConversationIds(List<UUID> publicIds);

    @Query(value = "SELECT md FROM AppMenuData md WHERE md.string10 IN (:externalMessageId) AND md.sourceDetails.orgId = :sourceOrgId")
    List<AppMenuData> findByExternalMessageId(List<String> externalMessageId,
                                              Long sourceOrgId);
}
