/**
 * Actually just an ordinary bean, _not_ a spring-data JPA repository.
 */
public abstract class AbstractMenuDataRepository {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private final AppMenuDataRepository menuDataRepository;
    private final ReferenceDatumRepository referenceDatumRepository;
    protected final PersonRepository personRepository;

    protected AbstractMenuDataRepository(AppMenuDataRepository menuDataRepository, ReferenceDatumRepository referenceDatumRepository,
                                         PersonRepository personRepository) {
        this.menuDataRepository = menuDataRepository;
        this.referenceDatumRepository = referenceDatumRepository;
        this.personRepository = personRepository;
    }

    @SuppressWarnings("serial")
    protected AccountKeys checkAccountKeys(TargetOwnerData targetOwnerData, List<AppMenuDataWithSourceObjects> menuData) {
        AccountKeys accountKeys = targetOwnerData.accountKeys().getOrElseThrow(() -> new RuntimeException("No account keys!"));
        return Try.of(() -> checkAccountKeys(accountKeys, menuData))
                .recover(ex -> {
                    LOGGER.warn("Symmetric keys missing from authz context, reloading", ex);
                    AccountKeys reloadedAccountKeys = targetOwnerData.accountKeysReloader().get();
                    return checkAccountKeys(reloadedAccountKeys, menuData);
                })
                .get();
    }

    private AccountKeys checkAccountKeys(AccountKeys accountKeys, List<AppMenuDataWithSourceObjects> menuData) {
        // Make sure we have all the keys we need, otherwise we'll need to go back to the db
        Set<Long> loadedKeyIds = accountKeys.getSymmetricKeys().keySet();
        Set<Long> requiredKeyIds = menuData
                .map(menuDataWrapper -> menuDataWrapper.getAppMenuData().getSymmetricEncryptionKeyId())
                .filter(Objects::nonNull)
                .toSet();
        Set<Long> missingKeys = requiredKeyIds.diff(loadedKeyIds);
        if (!missingKeys.isEmpty()) {
            throw new IllegalStateException(format("Authz context is missing symmetric keys %s", missingKeys));
        }
        return accountKeys;
    }

    protected ReferenceDatumCache getReferenceDatumCache(List<AppMenuDataWithSourceObjects> menuDataList, ReferenceDataType dataType) {
        List<PublicOrg> sourceOrgs = menuDataList.map(menuData -> menuData.getAppMenuData().getSourceDetails())
                .filter(Objects::nonNull)
                .map(SourceDetails::getOrg)
                .filter(Objects::nonNull)
                .distinctBy(PublicOrg::getId);

        if (sourceOrgs.isEmpty()) {
            return new ReferenceDatumCache(Map());
        }

        List<PublicReferenceDatum> referenceDatums = referenceDatumRepository
                .findByOrgsAndDataType(sourceOrgs, dataType);

        return new ReferenceDatumCache(referenceDatums
                .groupBy(prd -> prd.getOrg().getId())
                .mapValues(lrd -> lrd.toMap(PublicReferenceDatum::getDataId, PublicReferenceDatum::getDataText)));
    }

    @NotNull
    protected ImmutableSearchSpec.Builder getSearchSpecBuilder(@NotNull Long patientAccountId,
                                                               @NotNull String menuDataTypeName) {
        return ImmutableSearchSpec.builder()
                .addAccountId(patientAccountId)
                .addDataType(menuDataTypeName);
    }

    protected void addPageFilter(int maxNumberOfResources, ImmutableSearchSpec.Builder builder) {
        builder.pageFilter(ImmutablePageFilter.builder().startPosition(0).count(maxNumberOfResources).build());
    }


    protected void addOrderBy(String fieldName, OrderBy.Direction direction, ImmutableSearchSpec.Builder searchSpecBuilder) {
        searchSpecBuilder.orderBy(ImmutableOrderBy.builder().direction(direction).field(fieldName).build());
    }

    protected List<AppMenuDataWithSourceObjects> findMenuDatasByIds(List<Long> ids) {
        return populateSourceDetailsOnMenuDatas(menuDataRepository.findByIds(ids).sorted(comparing(AppMenuData::getId, Ordering.explicit(ids.asJava()))));
    }

    protected List<AppMenuDataWithSourceObjects> findMenuDatasByIdsWithAttachments(List<Long> ids) {
        return populateSourceDetailsOnMenuDatas(menuDataRepository.findByIdsWithAttachments(ids).sorted(comparing(AppMenuData::getId, Ordering.explicit(ids.asJava()))));
    }

    private List<AppMenuDataWithSourceObjects> populateSourceDetailsOnMenuDatas(List<AppMenuData> menuDatas) {
        HashSet<Long> sourcePersonIds = menuDatas.map(menuData -> menuData.getSourceDetails().getPersonId()).filter(Objects::nonNull)
                .collect(HashSet.collector());
        var sourcePersons = HashMap.ofAll(personRepository.findByIds(sourcePersonIds.toSet()).collect(toMap(PublicPKBPerson::getId, identity())));

        return menuDatas.map(menuData -> {
            PublicPKBPerson sourcePerson = sourcePersons.get(menuData.getSourceDetails().getPersonId())
                    .getOrNull();
            return ImmutableAppMenuDataWithSourceObjects.builder().appMenuData(menuData).sourcePerson(Option.of(sourcePerson)).build();
        });
    }

    protected List<Long> find(SearchSpec searchSpec, AccessFilter accessFilter) {
        return menuDataRepository.find(searchSpec, accessFilter);
    }

    protected long count(SearchSpec searchSpec, AccessFilter accessFilter) {
        return menuDataRepository.count(searchSpec, accessFilter);
    }

    protected List<Long> findBatch(SearchSpec searchSpec, @NotNull Map<Long, AccessFilter> accessFilters){
        return menuDataRepository.findBatch(searchSpec, accessFilters);
    }

    protected List<AppMenuData> findByConversationIds(List<UUID> conversationIds) {
        return menuDataRepository.findByConversationIds(conversationIds);
    }

    protected List<AppMenuData> findByExternalMessageId(List<String> externalMessageId,
                                                        Long sourceOrgId) {
        return menuDataRepository.findByExternalMessageId(externalMessageId, sourceOrgId);
    }
}
