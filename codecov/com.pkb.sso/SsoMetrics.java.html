package com.pkb.sso;

import com.pkb.entities.enums.SSOType;
import io.prometheus.client.Counter;
import io.prometheus.client.Summary;

public class SsoMetrics {


    // this should get us total , by team , by week stats
    private static final Counter GENERATED_CREDENTIALS_COUNTER = Counter.build()
            .name("pkb_integrations_sso_credentialsgenerated")
            .labelNames("team", "ssoType")
            .help("New SSO credentials generated by a team")
            .register();

    // this should get us total , by team , by week stats
    private static final Counter AUTHENTICATION_COUNTER = Counter.build()
            .name("pkb_integrations_sso_authentication")
            .labelNames("team", "successful", "ssoType")
            .help("SSO authentication requests/connections from a team")
            .register();

    // this should get us total , by team , by week stats
    private static  final Counter UNSUCCESSFUL_REQUESTS_COUNTER = Counter.build()
            .name("pkb_integrations_sso_unsuccessful_requests")
            .labelNames("ssoType", "errorCode")
            .help("Unsuccessful SSO authentication requests/connections by error")
            .register();

    private static final Counter EMIS_PATIENT_HOTSWAP_ATTEMPT_COUNTER = Counter.build()
            .name("pkb_integrations_emis_sso_attemptsToHotSwapPatient")
            .help("Attempts to hotswap patient in SSO mode")
            .register();

    private static final Summary SSOREQUEST_PROCESSING_STATISTICS = Summary.build()
            .name("pkb_integrations_sso_request_processing")
            .labelNames("ssoType")
            .quantile(0.5, 0.05)
            .quantile(0.95, 0.01)
            .help("SSO Request processing seconds")
            .register();


    public void incrementCredentialsGenerated(String teamCode, SSOType type) {
        GENERATED_CREDENTIALS_COUNTER.labels(teamCode, type.toString()).inc();
    }

    public void incrementSuccessfulAuthentication(String teamCode, SSOType type) {
        AUTHENTICATION_COUNTER.labels(teamCode, "true", type.toString()).inc();
    }

    public void incrementUnSuccessfulAuthentication(String teamCode, SSOType type) {
        AUTHENTICATION_COUNTER.labels(teamCode, "false", type.toString()).inc();
    }

    public void incrementUnSuccessfulSessionRequest(SSOType type, String errorCode) {
        UNSUCCESSFUL_REQUESTS_COUNTER.labels(type.toString(), errorCode).inc();
    }

    public void incrementAttemptToHotSwapPatient() {
        EMIS_PATIENT_HOTSWAP_ATTEMPT_COUNTER.labels().inc();
    }

    public Summary.Timer startTimerForProcessingSessionRequest(SSOType type) {
        return SSOREQUEST_PROCESSING_STATISTICS.labels(type.toString()).startTimer();
    }
}
