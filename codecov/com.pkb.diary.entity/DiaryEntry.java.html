package com.pkb.diary.entity;

import com.pkb.annotation.EHRField;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.document.entity.Attachment;

import java.io.Serializable;
import java.util.Date;
import java.util.List;

/**
 * This class represents an entry in the Diary of a patient to capture daily observations
 * Entries can be added by patient himself as well as the clinician participating in the care
 */
public class DiaryEntry implements IBaseDTO, Serializable {

    private static final long serialVersionUID = 1L;

    @EHRField(queryField = EHRField.QueryField.id)
    private Long id;

    private boolean htmlContentAllowed = false;

    @EHRField
    private String subjectOfEntry;

    @EHRField
    private String entryText;

    private final DTOBaseFields dtoBaseFields = new DTOBaseFields();

    private SourceDetails sourceDetails;

    private List<Attachment> attachmentList;

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected DiaryEntry() {
        this.sourceDetails = new SourceDetails();
    }

    public DiaryEntry(SourceDetails sourceDetails) {
        this.sourceDetails = sourceDetails;
    }

    public Date getDiaryEntryTimestamp() {
        return dtoBaseFields.getEnteredDate();
    }

    public void setDiaryEntryTimestamp(Date timestamp) {
        dtoBaseFields.setEnteredDate(timestamp);
    }

    public String getEntryText() {
        return entryText;
    }

    public boolean isHtmlContentAllowed() {
        return htmlContentAllowed;
    }

    public void setHtmlContentAllowed(boolean htmlContentAllowed) {
        this.htmlContentAllowed = htmlContentAllowed;
    }

    public void setEntryText(String entryText) {
        this.entryText = entryText;
    }

    @Override
    public Long getId() {
        return id;
    }

    @Override
    public void setId(Long id) {
        this.id = id;
    }

    public String getSubjectOfEntry() {
        return subjectOfEntry;
    }

    public void setSubjectOfEntry(String subjectOfEntry) {
        this.subjectOfEntry = subjectOfEntry;
    }

    public List<Attachment> getAttachmentList() {
        return attachmentList;
    }

    public void setAttachmentList(List<Attachment> attachmentList) {
        this.attachmentList = attachmentList;
    }

    @Override
    public SourceDetails getSource() {
        return sourceDetails;
    }

    @Override
    public DTOBaseFields getBaseFields() {
        return dtoBaseFields;
    }

    public boolean isGeneralHealth() {
        return id == null || getBaseFields().isGeneralHealthConsentRequired();
    }

    public boolean isMentalHealth() {
        return getBaseFields().isMentalHealthConsentRequired();
    }

    public boolean isSexualHealth() {
        return getBaseFields().isSexualHealthConsentRequired();
    }

    public boolean isSocialCare() {
        return getBaseFields().isSocialCareConsentRequired();
    }

    public void setSourceDetails(SourceDetails sourceDetails) {
        this.sourceDetails = sourceDetails;
    }
}
