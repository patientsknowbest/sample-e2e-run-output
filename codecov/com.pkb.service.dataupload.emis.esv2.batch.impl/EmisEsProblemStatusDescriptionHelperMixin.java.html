package com.pkb.service.dataupload.emis.esv2.batch.impl;

import static org.apache.commons.lang3.StringUtils.equalsIgnoreCase;
import static org.apache.commons.lang3.StringUtils.trimToNull;

import java.util.function.Consumer;

import org.apache.commons.lang3.StringUtils;

import com.pkb.emis.es.entity.EmisEsObservationDelta;
import com.pkb.service.dataupload.emis.esv2.model.CareRecordProblem;

public interface EmisEsProblemStatusDescriptionHelperMixin {

    // See PHR-2555
    String ACTIVE_PROBLEM = "Active Problem";
    String PAST_PROBLEM = "Past Problem";

    default void handleProblemStatusDescription(CareRecordProblem problem, Consumer<String> activeHandler, Consumer<String> pastHandler) {
        problem.problemStatusDescription()
                .map(StringUtils::trimToNull)
                .ifPresent(status -> handleStatus(status, activeHandler, pastHandler));
    }

    default void handleProblemStatusDescription(EmisEsObservationDelta observationDelta, Consumer<String> activeHandler, Consumer<String> pastHandler) {
        String status = trimToNull(observationDelta.getProblemStatusDescription());
        if (status != null) {
            handleStatus(status, activeHandler, pastHandler);
        }
    }

    private void handleStatus(String status, Consumer<String> activeHandler, Consumer<String> pastHandler) {
        if (equalsIgnoreCase(ACTIVE_PROBLEM, status)) {
            activeHandler.accept(status);
        } else if (equalsIgnoreCase(PAST_PROBLEM, status)) {
            pastHandler.accept(status);
        }
    }
}
