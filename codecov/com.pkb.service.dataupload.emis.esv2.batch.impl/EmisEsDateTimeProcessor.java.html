package com.pkb.service.dataupload.emis.esv2.batch.impl;

import static com.pkb.service.dataupload.emis.es.util.EmisEsDateUtil.DEFAULT_ZONE_ID;

import java.time.Instant;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.time.ZoneId;
import java.util.Date;
import java.util.Optional;

import org.jetbrains.annotations.Nullable;

import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.institute.entity.Org;
import com.pkb.service.dataupload.emis.es.util.EmisEsDateUtil;
import com.pkb.service.dataupload.emis.esv2.model.EmisEsEnteredDateTimeSupport;

public interface EmisEsDateTimeProcessor {

    default void mapEnteredDate(IBaseDTO target, EmisEsEnteredDateTimeSupport enteredDateTimeRecord, Org org) {
        target.getBaseFields().setEnteredDate(parseEnteredDateAndTime(enteredDateTimeRecord, org)
                .map(Date::from)
                .orElse(null));
    }

    default Optional<Instant> parseEnteredDateAndTime(EmisEsEnteredDateTimeSupport enteredDateTimeRecord, Org org) {
        LocalDateTime enteredDateTime = enteredDateTimeRecord.enteredTime()
                .map(t -> enteredDateTimeRecord.enteredDate().atTime(t))
                .orElse(enteredDateTimeRecord.enteredDate().atStartOfDay());
        ZoneId zoneId = Optional.ofNullable(org.getTimeZoneId()).map(ZoneId::of).orElse(DEFAULT_ZONE_ID);
        return parseDate(enteredDateTime, zoneId);
    }

    default Optional<Instant> parseDate(@Nullable LocalDateTime date, ZoneId zoneId) {
        return EmisEsDateUtil.parseDate(date, zoneId);
    }

    default Optional<Instant> parseDate(LocalDate date, Org org) {
        return EmisEsDateUtil.parseDate(date, ZoneId.of(org.getTimeZoneId()));
    }

    default Optional<Instant> parseDate(Optional<LocalDate> date, Org org) {
        return parseDate(date, ZoneId.of(org.getTimeZoneId()));
    }

    default Optional<Instant> parseDate(Optional<LocalDate> date, ZoneId zoneId) {
        return date.flatMap(ld -> EmisEsDateUtil.parseDate(ld, zoneId));
    }
}
