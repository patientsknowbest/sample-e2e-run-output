package com.pkb.service.dataupload.emis.esv2.batch.impl;

import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import com.pkb.common.ClearableInternalState;
import com.pkb.common.config.PhrConfig;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileType;

public abstract class BatchProcessingCache<T> implements ClearableInternalState {

    private static Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    protected PhrConfig configuration;

    protected Map<Long, T> cache = new ConcurrentHashMap<>();

    @Override
    public void clearState() {
        cache.clear();
        LOGGER.info("Cleared {}", this.getClass().getSimpleName());
    }

    /**
     * Clear in-memory cache, persisted data is retained.
     * 
     * @param batchId
     */
    public void clearCached(Long batchId) {
        cache.remove(batchId);
    }

    /**
     * Clear in-memory cache and persisted data.
     * 
     * @param batchId
     */
    public void clearCachedAndPersisted(Long batchId) {
        cache.remove(batchId);
    }

    public abstract EmisFileType getSupportedType();
}
