package com.pkb.service.dataupload.emis.esv2.batch.impl;

public class CareRecordObservationBatchCodeValidator extends AbstractBatchCodeValidator<CareRecordObservation> implements BatchProcessorHelperMixin<CareRecordObservation> {
    private static final EmisFileType TYPE = EmisFileType.CARE_RECORD_OBSERVATION;

    public CareRecordObservationBatchCodeValidator(@NotNull PhrConfig config, @NotNull EmisEsManager emisEsManager, EmisPersonStatHelper emisPersonStatHelper) {
        super(config, emisEsManager, emisPersonStatHelper);
    }

    @Override
    public Class<CareRecordObservation> interfaceClass() {
        return CareRecordObservation.class;
    }

    @Override
    public void processBuffer(List<ResultWithId<CareRecordObservation, Long>> rows) {
        io.vavr.collection.List<ResultWithId<CareRecordObservation, Long>> records = io.vavr.collection.List.ofAll(rows);

        // Fetch data needed for further processing
        var recordToOrgMergeHelper = getRecordToMaybeOrgMergeHelperMapTimed(records, CareRecordObservation::organisationGuid);

        var adminPatients = extractSuccesses(fetchAdminPatientsTimed(recordToOrgMergeHelper.filter(t -> t._2.isPresent()).toMap(k -> k._1, v -> v._2.orElseThrow())));

        Map<CareRecordObservation, Optional<OrgMergeHelper>> saveableRecords = recordToOrgMergeHelper.filter(t -> isActiveRecordAndPatient(adminPatients, t._1));

        recordTime("codingReceiveds", () -> ensureMappedCodingReceived(saveableRecords, CareRecordObservation::uniqueKey));
    }

    private boolean isActiveRecordAndPatient(Map<CareRecordObservation, AdminPatient> adminPatients, CareRecordObservation observation) {
        boolean handleAsDeleted = observation.isHandleAsDeleted();
        if (handleAsDeleted) {
            return false;
        }

        AdminPatient adminPatient = adminPatients.get(observation).getOrNull();
        if (adminPatient == null) {
            //may not have processed patient yet, so just assume record is relevant
            return true;
        }

        Boolean handleAsDeletedOverride = emisEsManager.getStatelessAdminPatientDeletedOverrideIfExists(adminPatient.patientGuid());
        if (handleAsDeletedOverride != null) {
            //a non-processed admin patient has a different state to where we've statefully processed
            // go with what the main processor would see if we were statefully processing
            return !handleAsDeletedOverride;
        }

        return !adminPatient.isHandleAsDeleted();
    }

    @Override
    public EmisFileType getSupportedType() {
        return TYPE;
    }

    private <K extends Serializable> void ensureMappedCodingReceived(Map<CareRecordObservation, Optional<OrgMergeHelper>> observations, Function1<CareRecordObservation, K> uniqueKeyExtractor) {
        Map<CareRecordObservation, Either<ProcessingError<CareRecordObservation>, Tuple2<CodingReceived, Function1<CodingReceived, CodeableConcept>>>> constructedCodingReceived = constructCodingReceivedOptionalOrg(
                observations, uniqueKeyExtractor);
        Map<CareRecordObservation, Tuple2<CodingReceived, Function1<CodingReceived, CodeableConcept>>> successes = extractSuccesses(
                constructedCodingReceived);
        Map<CareRecordObservation, CodingReceived> successfulCodingReceiveds = successes.mapValues(v -> v._1);
        Map<K, CareRecordObservation> keyLookup = successfulCodingReceiveds.toMap(kv -> Tuple.of(uniqueKeyExtractor.apply(kv._1), kv._1));
        Map<K, CodingReceived> uniqueKeyToCodingReceived = successfulCodingReceiveds.mapKeys(uniqueKeyExtractor);

        Map<K, CodingReceived> uniqueKeyToEnsuredCodingReceived = getEmisEsManager()
                .ensureCodingReceivedIfMatched(uniqueKeyToCodingReceived);

        Map<CareRecordObservation, CodingReceived> matchedCRs = uniqueKeyToEnsuredCodingReceived.mapKeys(keyLookup);
        Map<CareRecordObservation, Tuple2<CodingReceived, Function1<CodingReceived, CodeableConcept>>> noMatch = successes
                .removeAll(matchedCRs.keySet());
        noMatch.forEach((k, v) -> emisEsManager.trackUnmatchedCodingReceived(v._1));
    }

    @Override
    public EmisEsManager getEmisEsManager() {
        return emisEsManager;
    }

    @Override
    public UserManager getUserManager() {
        throw new NotImplementedError("unsupported in this impl");
    }

    @Override
    public PhrConfig getConfig() {
        return config;
    }

    @Override
    public EmisPersonStatHelper getEmisPersonStatHelper() {
        return emisPersonStatHelper;
    }
}
