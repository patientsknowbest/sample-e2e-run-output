package com.pkb.service.dataupload.emis.esv2.batch.impl;

import com.pkb.service.dataupload.emis.esv2.model.BaseEmisEsDomainModel;

class ProcessingError<T extends BaseEmisEsDomainModel> {

    private final T row;
    private final Severity severity;
    private final String messageKey;
    private final Object[] parameters;

    ProcessingError(T row, Severity severity, String message, Object... parameters) {
        this.row = row;
        this.severity = severity;
        this.messageKey = message;
        this.parameters = parameters;
    }

    enum Severity {
        FATAL,
        NON_FATAL
    }

    public T getRow() {
        return row;
    }

    public Severity getSeverity() {
        return severity;
    }

    public String getMessageKey() {
        return messageKey;
    }

    public String getMessage() {
        return String.format(messageKey, parameters);
    }

    @Override
    public String toString() {
        return "ProcessingError{" +
                "class=" + row.getClass().getSimpleName() +
                ", severity=" + severity +
                ", message='" + getMessage() + '\'' +
                '}';
    }
}
