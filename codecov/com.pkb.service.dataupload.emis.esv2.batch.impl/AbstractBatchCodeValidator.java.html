package com.pkb.service.dataupload.emis.esv2.batch.impl;

import com.pkb.common.base.StreamUtils;
import com.pkb.common.config.PhrConfig;
import com.pkb.service.dataupload.emis.esv2.EmisEsManager;
import com.pkb.service.dataupload.emis.esv2.EmisPersonStatHelper;
import com.pkb.service.dataupload.emis.esv2.batch.FileProcessor;
import com.pkb.service.dataupload.emis.esv2.model.BaseEmisEsDomainModel;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileType;
import com.pkb.service.dataupload.emis.esv2.sorting.ResultWithId;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Iterator;
import java.util.List;
import java.util.stream.Stream;

import static com.pkb.service.dataupload.emis.esv2.batch.FileProcessor.ProcessorType.VALIDATOR;

public abstract class AbstractBatchCodeValidator<T extends BaseEmisEsDomainModel> implements FileProcessor<T> {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    protected PhrConfig config;
    protected EmisEsManager emisEsManager;
    protected com.pkb.service.dataupload.emis.esv2.EmisPersonStatHelper emisPersonStatHelper;

    AbstractBatchCodeValidator(@NotNull PhrConfig config, @NotNull EmisEsManager emisEsManager, EmisPersonStatHelper emisPersonStatHelper) {
        this.config = config;
        this.emisEsManager = emisEsManager;
        this.emisPersonStatHelper = emisPersonStatHelper;
    }

    abstract void processBuffer(List<ResultWithId<T, Long>> rows);

    private void processBufferTimed(List<ResultWithId<T, Long>> rows) {
        recordTime("validateBatchOfCodes", () -> processBuffer(rows));
    }

    public void process(Stream<? extends ResultWithId<T, Long>> rows) {
        int total = 0;
        String entityClassName = interfaceClass().getSimpleName();
        Iterator<? extends ResultWithId<T, Long>> rowIterator = rows.iterator();
        List<ResultWithId<T, Long>> buffer;
        do {
            buffer = recordTime("pullRecords", () -> StreamUtils.pull(rowIterator, getBatchSize()));
            if (!buffer.isEmpty()) {
                processBufferTimed(buffer);
                total += buffer.size();
                LOGGER.info("Processed {} items of class {}, {} in total.", buffer.size(), entityClassName, total);
            }
        } while (!buffer.isEmpty());
    }

    public int getBatchSize() {
        return config.getEmisEsProcessingBatchSize();
    }

    /**
     * @return false This is *not* a normal processor; it's only used for scanning ahead for problems after normal processing has failed
     */
    @Override
    public boolean canProcess(@NotNull EmisFileType fileType) {
        return false;
    }

    @Override
    public ProcessorType getProcessorType() {
        return VALIDATOR;
    }


}
