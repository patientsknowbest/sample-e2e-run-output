package com.pkb.service.dataupload.emis.esv2.batch.impl;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.dataupload.emis.esv2.EmisEsProcessedRow;
import com.pkb.externalid.service.ExternalIdService;
import com.pkb.kms.client.core.Kms;
import com.pkb.service.dataupload.emis.esv2.EmisEsManager;
import com.pkb.service.dataupload.emis.esv2.EmisEsProcessedRowService;

import com.pkb.service.dataupload.emis.esv2.EmisOrgHelper;
import com.pkb.service.dataupload.emis.esv2.batch.EmisProcessorAborter;

import com.pkb.service.dataupload.emis.esv2.sorting.RowId;
import com.pkb.service.notification.impl.tolven.INotificationManager;
import org.jetbrains.annotations.NotNull;

import static com.pkb.service.dataupload.emis.esv2.batch.FileProcessor.ProcessorType.STATELESS;

/**
 * Not a processor for the normal flow - only invoke this for when we're validating failing codes
 * in upcoming batches.
 * <p>
 * This processor DOES save drug codes to the database, but doesn't track processing otherwise
 * (so we avoid side-effects on the normal delta processing flow)
 */
public class StatelessCodingDrugCodeProcessor extends CodingDrugCodeProcessor {

    public StatelessCodingDrugCodeProcessor(EmisEsProcessedRowService emisEsProcessedRowService, EmisEsManager emisEsManager, EmisProcessorAborter emisProcessAborter, Kms kmsClient, PhrConfig config, DateTimeService dateTimeService, ExternalIdService externalIdService, INotificationManager notificationManager, EmisEsStatusService emisEsStatusService, EmisOrgHelper emisOrgHelper) {
        super(emisEsProcessedRowService, emisEsManager, emisProcessAborter, kmsClient, config, dateTimeService, externalIdService, notificationManager, emisEsStatusService, emisOrgHelper);
    }

    @Override
    protected EmisEsProcessedRow updateLastProcessedRow(EmisEsProcessedRow actualFileState, RowId<Long> rowId) {
        // do not save
        return actualFileState;
    }

    /**
     * @return false This is *not* a normal processor; it's only used for scanning ahead for problems after normal processing has failed
     */
    @Override
    public boolean canProcess(@NotNull EmisFileType fileType) {
        return false;
    }

    @Override
    public ProcessorType getProcessorType() {
        return STATELESS;
    }
}
