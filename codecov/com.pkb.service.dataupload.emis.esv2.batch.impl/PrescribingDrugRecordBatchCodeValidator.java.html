package com.pkb.service.dataupload.emis.esv2.batch.impl;

import com.pkb.common.config.PhrConfig;
import com.pkb.institute.OrgMergeHelper;
import com.pkb.service.dataupload.emis.esv2.EmisEsManager;
import com.pkb.service.dataupload.emis.esv2.EmisPersonStatHelper;
import com.pkb.service.dataupload.emis.esv2.model.AdminPatient;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileType;
import com.pkb.service.dataupload.emis.esv2.model.PrescribingDrugRecord;
import com.pkb.service.dataupload.emis.esv2.sorting.ResultWithId;
import com.pkb.service.user.impl.UserManager;
import io.vavr.NotImplementedError;
import io.vavr.collection.Map;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.List;

public class PrescribingDrugRecordBatchCodeValidator extends AbstractBatchCodeValidator<PrescribingDrugRecord> implements BatchProcessorHelperMixin<PrescribingDrugRecord> {

    private static final EmisFileType TYPE = EmisFileType.PRESCRIBING_DRUG_RECORD;

    public PrescribingDrugRecordBatchCodeValidator(@NotNull PhrConfig config, @NotNull EmisEsManager emisEsManager, EmisPersonStatHelper emisPersonStatHelper) {
        super(config, emisEsManager, emisPersonStatHelper);
    }

    @Override
    public Class<PrescribingDrugRecord> interfaceClass() {
        return PrescribingDrugRecord.class;
    }

    @Override
    void processBuffer(List<ResultWithId<PrescribingDrugRecord, Long>> rows) {
        io.vavr.collection.List<ResultWithId<PrescribingDrugRecord, Long>> records = io.vavr.collection.List.ofAll(rows);

        Map<PrescribingDrugRecord, OrgMergeHelper> recordToOrgMergeHelper = getRecordToOrgMergeHelperMapTimed(records, PrescribingDrugRecord::organisationGuid);

        Map<PrescribingDrugRecord, Either<ProcessingError<PrescribingDrugRecord>, AdminPatient>> adminPatientsOrFailures = fetchAdminPatientsTimed(recordToOrgMergeHelper);
        Map<PrescribingDrugRecord, AdminPatient> adminPatients = extractSuccesses(adminPatientsOrFailures);

        io.vavr.collection.List<PrescribingDrugRecord> recordsForCodings = records.map(ResultWithId::getValue).filter(p -> !p.isHandleAsDeleted(adminPatients.get(p).getOrNull()));
        fetchCodingDrugCodesTimed(recordsForCodings);
    }

    @Override
    public EmisFileType getSupportedType() {
        return TYPE;
    }

    @Override
    public EmisEsManager getEmisEsManager() {
        return emisEsManager;
    }

    @Override
    public UserManager getUserManager() {
        throw new NotImplementedError("unsupported in this impl");
    }

    @Override
    public PhrConfig getConfig() {
        return config;
    }

    @Override
    public EmisPersonStatHelper getEmisPersonStatHelper() {
        return emisPersonStatHelper;
    }
}
