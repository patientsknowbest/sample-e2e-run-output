public class AdminOrganisationProcessor extends BatchingBatchDataProcessor<AdminOrganisation> {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private static final EmisFileType TYPE = EmisFileType.ADMIN_ORGANISATION;

    public AdminOrganisationProcessor(EmisEsProcessedRowService emisEsProcessedRowService, EmisEsManager emisEsManager, EmisProcessorAborter emisProcessAborter, Kms kmsClient, PhrConfig config, DateTimeService dateTimeService, ExternalIdService externalIdService, INotificationManager notificationManager, EmisEsStatusService emisEsStatusService, EmisOrgHelper emisOrgHelper) {
        super(emisEsProcessedRowService, emisEsManager, emisProcessAborter, kmsClient, config, dateTimeService, externalIdService, notificationManager, emisEsStatusService, emisOrgHelper);
    }

    @Override
    public Class<AdminOrganisation> interfaceClass() {
        return AdminOrganisation.class;
    }

    @Override
    public boolean canProcess(@NotNull EmisFileType fileType) {
        return fileType == TYPE;
    }

    @Override
    boolean process(List<ResultWithId<AdminOrganisation, Long>> rows, EmisEsProcessedRow fileState) {
        rows.forEach(r -> emisProcessAborter.executeTestControl(TYPE, r.getRowId().getId()));

        var zoneIdOrFailureMap = getZoneIds(rows);

        Map<AdminOrganisation, ZoneId> zoneIdMap = new HashMap<>();
        Set<ProcessingError<AdminOrganisation>> failures = new HashSet<>();
        zoneIdOrFailureMap.forEach((k, v) -> v.peekLeft(failures::add).peek(zid -> zoneIdMap.put(k, zid)));

        failures.stream().collect(Collectors.groupingBy(ProcessingError::getSeverity)).forEach((severity, errors) -> {
            switch (severity) {
                case FATAL:
                    String errorString = errors.stream().map(ProcessingError::getMessage).collect(Collectors.joining("\n"));
                    LOGGER.error("Failed to process {} admin organization records:\n{}", errors.size(), errorString);
                    incrementProcessedRowsCounter(ProcessedRowStatus.FAILED, errors.size());
                    break;
                case NON_FATAL:
                    String samples = io.vavr.collection.List.ofAll(errors).distinctBy(ProcessingError::getMessageKey).map(ProcessingError::getMessage).collect(Collectors.joining("\n"));
                    LOGGER.warn("Skipping {} admin organization records because of non-fatal failures\nmessages with samples:\n{}", errors.size(), samples);
                    incrementProcessedRowsCounter(ProcessedRowStatus.SKIPPED, errors.size());
                    break;
                default:
                    throw new NotImplementedException("Not implemented yet for " + severity);
            }
        });

        rows.stream()
                .filter(rwi -> zoneIdMap.containsKey(rwi.getValue()))
                .forEach(rwi -> {
                    emisEsManager.saveAdminOrganisation(rwi.getValue(), zoneIdMap.get(rwi.getValue()));
                    incrementProcessedRowsCounter(ProcessedRowStatus.SAVED, 1);
                });

        return failures.stream().filter(f -> f.getSeverity() == FATAL).findAny().isEmpty();
    }


    @Override
    int getBatchSize() {
        return config.getEmisEsProcessingBatchSize();
    }


    protected java.util.Map<AdminOrganisation, Either<ProcessingError<AdminOrganisation>, ZoneId>> getZoneIds(List<ResultWithId<AdminOrganisation, Long>> rows) {
        var odsCodeToOrgMap = getOdsCodeToOrgMap(rows);
        return rows.stream().map(rwi -> getZoneId(rwi, odsCodeToOrgMap)).collect(Collectors.toMap(k -> k._1, v -> v._2));
    }

    protected Tuple2<AdminOrganisation, Either<ProcessingError<AdminOrganisation>, ZoneId>> getZoneId(ResultWithId<AdminOrganisation, Long> row, java.util.Map<String, OrgMergeHelper> odsCodeToOrgMergeHelperMap) {
        var record = row.getValue();
        var maybeOdsCode = record.odsCode().map(StringUtils::trimToNull);
        ZoneId orgtz = DEFAULT_ZONE_ID;

        if (maybeOdsCode.isEmpty()) {
            return Tuple.of(record, Either.right(orgtz));
        }

        var maybeOrgMergeHelper = odsCodeToOrgMergeHelperMap.get(maybeOdsCode.get());


        if (maybeOrgMergeHelper == null) {
            if (emisEsManager.isWhitelistedOrg(maybeOdsCode.get())) {
                return Tuple.of(record, processingError(record, ProcessingError.Severity.FATAL, "Org setup is missing! Either remove Org from whitelist or create Org with ODS or add ODS code to Org! %s", record));
            } else {
                emisOrgHelper.logIfOrgWasntSeenYet(record, "EMIS Org [{}] appeared in feed and is not yet set up in PKB -- using default TZ offset", record);
            }
        } else {
            orgtz = getTimezoneOfOrg(maybeOrgMergeHelper.getOrg());
        }

        return Tuple.of(record, Either.right(orgtz));
    }

    protected java.util.Map<String, OrgMergeHelper> getOdsCodeToOrgMap(List<ResultWithId<AdminOrganisation, Long>> rows) {
        var odsCodes = rows.stream().map(rwi -> rwi.getValue().odsCode()
                .map(StringUtils::trimToNull))
                .flatMap(Optional::stream)
                .filter(StringUtils::isNotBlank)
                .collect(Collectors.toSet());
        return emisEsManager.findOrgsByODSCode(odsCodes);
    }

    protected Either<ProcessingError<AdminOrganisation>, ZoneId> processingError(AdminOrganisation t, ProcessingError.Severity severity, String message, Object... parameters) {
        return Either.left(new ProcessingError<>(t, severity, message, parameters));
    }

    @Override
    public boolean isPostProcessingRequired() {
        return false;
    }

    @Override
    protected boolean doPostProcess(EmisEsProcessedRow rowState) {
        return true;
    }

    @Override
    public ProcessorType getProcessorType() {
        return NORMAL;
    }

    @Override
    public EmisFileType getSupportedType() {
        return TYPE;
    }
}
