package com.pkb.service.dataupload.emis.esv2.batch.impl;

import com.pkb.dataupload.emis.esv2.EmisEsCareRecordProcessingState;
import com.pkb.service.dataupload.emis.esv2.EmisEsCareRecordProcessingStateService;
import com.pkb.service.dataupload.emis.esv2.model.CareRecordObservation;
import com.pkb.service.dataupload.emis.esv2.model.CareRecordProblem;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileType;
import com.pkb.service.dataupload.emis.esv2.sorting.ResultWithId;
import io.vavr.Tuple2;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.Collections;
import java.util.HashMap;
import java.util.HashSet;
import java.util.List;
import java.util.Map;
import java.util.Set;
import java.util.concurrent.ConcurrentHashMap;
import java.util.stream.Collectors;

@Component
public class CareRecordProblemCache extends BatchProcessingCache<Map<Tuple2<String, String>, ResultWithId<CareRecordProblem, Long>>> {

    @Autowired
    private EmisEsCareRecordProcessingStateService emisEsCareRecordProcessingStateService;

    private Map<Long, Set<Tuple2<String, String>>> processedObservationGuids = new ConcurrentHashMap<>();

    public void markProcessed(Long batchId, CareRecordObservation observation) {
        emisEsCareRecordProcessingStateService.save(batchId, observation.observationGuid(), observation.patientGuid());
        processedObservationGuids.computeIfAbsent(batchId, k -> new HashSet<>()).add(observation.uniqueKey());
    }

    public List<ResultWithId<CareRecordProblem, Long>> getUnProcessedCareRecordProblems(Long batchId) {
        return cache.getOrDefault(batchId, Collections.emptyMap()).entrySet().stream()
                .filter(e -> !processedObservationGuids.getOrDefault(batchId, Collections.emptySet()).contains(e.getKey()))
                .map(Map.Entry::getValue).collect(Collectors.toList());
    }

    public ResultWithId<CareRecordProblem, Long> get(Long batchId, CareRecordObservation observation) {
        return cache.getOrDefault(batchId, Collections.emptyMap()).get(observation.uniqueKey());
    }

    public void putAll(Long batchId, List<ResultWithId<CareRecordProblem, Long>> rows) {
        if (configuration.isEmisEsStateResumeEnabled()) {
            processedObservationGuids.computeIfAbsent(batchId, k ->
                emisEsCareRecordProcessingStateService.getAll(batchId)
                        .stream().map(EmisEsCareRecordProcessingState::uniqueKey).collect(Collectors.toSet())
            );
        }

        var cacheMap = cache.computeIfAbsent(batchId, k -> new HashMap<>());
        rows.forEach(rwi -> cacheMap.put(rwi.getValue().uniqueKey(), rwi));
    }

    @Override
    public void clearState() {
        processedObservationGuids.clear();
        super.clearState();
    }

    @Override
    public void clearCached(Long batchId) {
        processedObservationGuids.remove(batchId);
        super.clearCached(batchId);
    }

    @Override
    public void clearCachedAndPersisted(Long batchId) {
        processedObservationGuids.remove(batchId);
        emisEsCareRecordProcessingStateService.clearAll(batchId);
        super.clearCachedAndPersisted(batchId);
    }

    @Override
    public EmisFileType getSupportedType() {
        return EmisFileType.CARE_RECORD_PROBLEM;
    }
}
