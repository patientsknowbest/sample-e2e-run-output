package com.pkb.service.dataupload.emis.esv2.batch.impl;

import com.google.common.collect.ImmutableSortedSet;
import com.pkb.crypto.util.RandomUtil;
import com.pkb.entities.enums.Gender;
import com.pkb.entities.enums.UserStatus;
import com.pkb.entities.enums.UserType;
import com.pkb.institute.entity.Org;
import com.pkb.model.PKBPersonDTO;
import com.pkb.service.dataupload.emis.es.util.EmisEsDateUtil;
import com.pkb.service.dataupload.emis.es.util.EmisEsDemographicsUtil;
import com.pkb.service.dataupload.emis.esv2.batch.BatchItemProcessor;
import com.pkb.service.dataupload.emis.esv2.batch.ItemContext;
import com.pkb.service.dataupload.emis.esv2.model.AdminPatient;
import com.pkb.service.dataupload.emis.esv2.model.Constants;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Component;

import java.util.Date;
import java.util.Optional;
import java.util.stream.Stream;

import static java.util.Collections.singletonList;

@Component
public class AdminPatientItemProcessor implements BatchItemProcessor<AdminPatient, PKBPersonDTO>, EmisEsDateTimeProcessor {

    @Override
    public PKBPersonDTO processItem(ItemContext context, AdminPatient adminPatient) {
        Org org = context.org().orElseThrow(() -> new IllegalStateException("No org provided in item context! AdminPatient: " + adminPatient));

        PKBPersonDTO dto = new PKBPersonDTO();

        EmisEsDemographicsUtil.parseNhsNumber(adminPatient.nhsNumber(), context.requestContext())
                .ifPresent(n -> dto.setNationalIds(singletonList(n)));

        EmisEsDemographicsUtil.parsePatientNumber(adminPatient.patientNumber(), org, context.requestContext())
                .ifPresent(olid -> dto.setOrgLevelIds(ImmutableSortedSet.of(olid)));

        if (dto.getNationalIds().isEmpty() && dto.getOrgLevelIds().isEmpty()) {
            throw new IllegalStateException("Not creating new patient because missing both NHS Number and Patient Number! " + adminPatient);
        }

        if ((adminPatient.surname().map(StringUtils::trimToNull).isEmpty() || adminPatient.givenName().map(StringUtils::trimToNull).isEmpty())
                && !adminPatient.isHandleAsDischargedOrDeleted()) {
            throw new IllegalStateException("Not deleted patient don't have name fields! " + adminPatient);
        }

        adminPatient.title().map(StringUtils::trimToNull).ifPresent(dto::setTitle);
        adminPatient.surname().map(StringUtils::trimToNull).ifPresent(dto::setLastName);
        adminPatient.givenName().map(StringUtils::trimToNull).ifPresent(dto::setFirstName);
        adminPatient.middleNames()
                .map(StringUtils::trimToNull)
                .ifPresent(dto::setMiddleName);

        dto.setGender(adminPatient.sex()
                .filter(StringUtils::isNotBlank)
                .map(EmisEsDemographicsUtil::convertGender)
                .orElse(Gender.UNKNOWN)
                .getId());

        if (adminPatient.dateOfBirth().isEmpty() && !adminPatient.isHandleAsDischargedOrDeleted()) {
            throw new IllegalStateException("Not deleted patient don't have date of birth! " + adminPatient);
        }
        adminPatient.dateOfBirth()
                .map(ld -> EmisEsDateUtil.DEFAULT_DATE_FORMATTER.format(ld))
                .ifPresent(dto::setDateOfBirthString);

        Stream.of(adminPatient.houseNameFlatNumber(), adminPatient.numberAndStreet())
                .map(s -> s.map(StringUtils::trimToNull))
                .flatMap(Optional::stream)
                .reduce((t, u) -> t + ", " + u)
                .ifPresent(dto::setAddress1);
        adminPatient.village()
                .map(StringUtils::trimToNull)
                .ifPresent(dto::setAddress2);
        adminPatient.town()
                .map(StringUtils::trimToNull)
                .ifPresent(dto::setCity);
        adminPatient.county()
                .map(StringUtils::trimToNull)
                .ifPresent(dto::setState);
        adminPatient.postcode()
                .map(StringUtils::trimToNull)
                .ifPresent(dto::setPostalCode);
        dto.setCountry(org.getCountry()); // TODO See PHR-2746

        adminPatient.mobilePhone()
                .map(StringUtils::trimToNull)
                .or(adminPatient::homePhone)
                .map(StringUtils::trimToNull)
                .ifPresent(dto::setPhone);

        adminPatient.dateOfDeath().ifPresentOrElse(ld -> {
            parseDate(ld, org)
                    .map(Date::from)
                    .ifPresent(dto::setDeathTimestamp);
            dto.setStatus(UserStatus.DEAD);
        }, () -> dto.setStatus(UserStatus.NOCONTACT));

        dto.setUserType(UserType.PATIENT);
        dto.setPassword(RandomUtil.randomPassword(10));
        dto.setLanguageCode(Constants.DEFAULT_LANGUAGE_CODE);
        dto.setEmisEsPersonGuid(adminPatient.personGuid());

        return dto;
    }
}
