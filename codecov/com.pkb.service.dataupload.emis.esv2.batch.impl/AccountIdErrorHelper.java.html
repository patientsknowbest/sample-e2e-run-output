package com.pkb.service.dataupload.emis.esv2.batch.impl;

import io.micrometer.core.instrument.Counter;
import io.micrometer.core.instrument.Metrics;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.concurrent.ConcurrentHashMap;

public class AccountIdErrorHelper {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private static final Counter UNMATCHED_ACCOUND_ID_COUNTER_CARE_RECORD_PROBLEM = Counter.builder("pkb_phr_emis_care_record_problem_unmatched_account_id")
            .description("Count the unmatched account_ids at CareRecordProblems")
            .register(Metrics.globalRegistry);

    private final ConcurrentHashMap<String, Long> unmatchedCodesAlreadySeenAtCareRecordProblem = new ConcurrentHashMap<>();

    public void reportUnmatchedAccountIdsAtCareRecordProblem(String guid, Long accountId) {
        UNMATCHED_ACCOUND_ID_COUNTER_CARE_RECORD_PROBLEM.increment();
        if (unmatchedCodesAlreadySeenAtCareRecordProblem.putIfAbsent(guid, accountId) == null) {
            LOGGER.error("Can't find matching menudata at CareRecordProblemProcess with guid: '{}' and accountId: '{}'", guid, accountId);
        }
    }
}
