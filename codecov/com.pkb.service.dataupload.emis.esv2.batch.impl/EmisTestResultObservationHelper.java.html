package com.pkb.service.dataupload.emis.esv2.batch.impl;

import static com.pkb.service.dataupload.emis.esv2.batch.impl.ProcessingError.Severity.NON_FATAL;
import static java.util.function.Function.identity;
import static java.util.stream.Collectors.toMap;

public class EmisTestResultObservationHelper {

    private final EmisEsManager emisEsManager;

    private final DateTimeService dateTimeService;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private static final Counter MISSING_UNITS_OR_VALUES_OR_EFFECTIVE_DATE = Counter.builder("pkb_phr_emis_care_record_observation_missing_units_values_and_effective_date")
            .description("Count the Test type CareRecordObservation which doesn't have unit/value/effective date")
            .register(Metrics.globalRegistry);

    private final java.util.Map<String, String> croWithNoUnitOrValueOrEffectiveDateAlreadySeen = new ConcurrentHashMap<>();


    public EmisTestResultObservationHelper(EmisEsManager emisEsManager, DateTimeService dateTimeService) {
        this.emisEsManager = emisEsManager;
        this.dateTimeService = dateTimeService;
    }

    public Map<CareRecordObservation, Either<ProcessingError<CareRecordObservation>, LoincMapping>> retrieveLoincMappingsForTestRecords(Map<CareRecordObservation, Tuple2<CodeableConcept, CodingWrapper>> observations) {

        Map<CareRecordObservation, CodingWrapper> testResultObservations = observations
                .filterValues(value -> value._2.codingReceived().getCodingDataType() == CodingDataType.TEST)
                .mapValues(value -> value._2);

        Map<CareRecordObservation, CodingWrapper> filteredRecordsByUnitAndValueAndEffectiveDatePresent = filterRecordsByUnitAndValueAndEffectiveDatePresent(testResultObservations);

        java.util.Map<String, CodingTranslationCodingFields> codingTranslations = lookupCodingTranslations(filteredRecordsByUnitAndValueAndEffectiveDatePresent
                .values()
                .map(codingWrapper -> codingWrapper
                        .codingClinicalCode()
                        .snomedCTConceptId()
                        .toString())
                .collect(Collectors.toSet()));

        Map<CareRecordObservation, CodeAndUnitMatchWrapper> recordsToCodeAndUnitMatchWrapper = mapCodingWrapperToCodeAndUnitMatchWrapper(filteredRecordsByUnitAndValueAndEffectiveDatePresent, codingTranslations);

        java.util.Map<LoincCodeAndUnit, LoincTest> loincTests = lookupForLoincTests(recordsToCodeAndUnitMatchWrapper
                .values()
                .map(CodeAndUnitMatchWrapper::loincCodeAndUnit)
                .collect(Collectors.toSet()));

        Map<CareRecordObservation, CodeAndUnitMatchWrapper> filteredRecordsByHavingLoincTest = filterRecordsByLoincTest(recordsToCodeAndUnitMatchWrapper, loincTests);

        java.util.Map<OrgIdSnomedCTCodeAndUnit, LoincMapping> retrievedLoincMappings = lookupForLoincMappings(filteredRecordsByHavingLoincTest
                .values()
                .map(CodeAndUnitMatchWrapper::orgIdSnomedCTCodeAndUnit)
                .collect(Collectors.toSet()));

        List<LoincMapping> missingLoincMappings = collectMissingLoincMappings(retrievedLoincMappings, filteredRecordsByHavingLoincTest, loincTests);

        java.util.Map<OrgIdSnomedCTCodeAndUnit, LoincMapping> savedLoincMappings = saveMissingLoincMappings(missingLoincMappings);

        java.util.Map<OrgIdSnomedCTCodeAndUnit, LoincMapping> savedAndRetrievedLoincMappings = new HashMap<>();
        savedAndRetrievedLoincMappings.putAll(savedLoincMappings);
        savedAndRetrievedLoincMappings.putAll(retrievedLoincMappings);

        Map<CareRecordObservation, LoincMapping> recordsToLoincMappings = pairLoincMappingToCareRecordObservation(filteredRecordsByHavingLoincTest, savedAndRetrievedLoincMappings);

        return processResults(recordsToCodeAndUnitMatchWrapper, filteredRecordsByHavingLoincTest, recordsToLoincMappings, filteredRecordsByUnitAndValueAndEffectiveDatePresent);
    }

    private Map<CareRecordObservation, CodingWrapper> filterRecordsByUnitAndValueAndEffectiveDatePresent(Map<CareRecordObservation, CodingWrapper> testResultObservations) {
        return testResultObservations.filter((cro, cw) -> {
            if (cro.numericUnit().isPresent() && cro.effectiveDate().isPresent() && cro.value().isPresent()) {
                String numericUnit = cro.numericUnit().get();
                if (EmisEsUnitUtil.sanitizeUnit(numericUnit).isEmpty()) {
                    reportObservationMissingUnitOrValueOrEffectiveDate(cw.codingClinicalCode().snomedCTConceptId().toString(), cro.observationGuid());
                    return false;
                }
                return true;
            }
            reportObservationMissingUnitOrValueOrEffectiveDate(cw.codingClinicalCode().snomedCTConceptId().toString(), cro.observationGuid());
            return false;
        });
    }

    private java.util.Map<String, CodingTranslationCodingFields> lookupCodingTranslations(Collection<String> snomedCodes) {
        return emisEsManager.lookupCodingTranslations(snomedCodes);
    }

    private Map<CareRecordObservation, CodeAndUnitMatchWrapper> mapCodingWrapperToCodeAndUnitMatchWrapper(Map<CareRecordObservation, CodingWrapper> testResultObservations,
                                                                                                          java.util.Map<String, CodingTranslationCodingFields> codingTranslations) {
        return testResultObservations.map((cro, codingWrapper) -> {
            CodingTranslationCodingFields codingTranslation = codingTranslations.get(codingWrapper.codingClinicalCode().snomedCTConceptId().toString());
            if (codingTranslation == null) {
                throw new IllegalStateException("No CodingTranslation for SNOMED_CT: " + codingWrapper.codingClinicalCode().snomedCTConceptId().toString());
            }
            String unit = cro.numericUnit().map(EmisEsUnitUtil::sanitizeUnit) .orElseThrow(() -> new IllegalStateException("No unit for CareRecordObservation record: " + cro));
            return Tuple.of(cro, ImmutableCodeAndUnitMatchWrapper.builder()
                    .codingTranslation(codingTranslation)
                    .org(codingWrapper.org())
                    .unit(unit)
                    .build());
        });
    }

    private java.util.Map<LoincCodeAndUnit, LoincTest> lookupForLoincTests(Collection<LoincCodeAndUnit> loincCodesAndUnits) {
        return emisEsManager.lookupLoincTests(loincCodesAndUnits);
    }

    private Map<CareRecordObservation, CodeAndUnitMatchWrapper> filterRecordsByLoincTest(Map<CareRecordObservation, CodeAndUnitMatchWrapper> testResultObservations,
                                                                                         java.util.Map<LoincCodeAndUnit, LoincTest> loincTests) {
        return testResultObservations.filterValues(codeAndUnitMatchWrapper -> loincTests.containsKey(codeAndUnitMatchWrapper.loincCodeAndUnit()));
    }

    private java.util.Map<OrgIdSnomedCTCodeAndUnit, LoincMapping> lookupForLoincMappings(Set<OrgIdSnomedCTCodeAndUnit> orgIdsSnomedCodesAndUnits) {
        return emisEsManager.lookupLoincMappings(orgIdsSnomedCodesAndUnits);
    }

    private List<LoincMapping> collectMissingLoincMappings(java.util.Map<OrgIdSnomedCTCodeAndUnit, LoincMapping> loincMappings,
                                                           Map<CareRecordObservation, CodeAndUnitMatchWrapper> filteredRecordsByLoincTest,
                                                           java.util.Map<LoincCodeAndUnit, LoincTest> loincTests) {
        java.util.Map<OrgIdSnomedCTCodeAndUnit, LoincMapping> missingLoincMappings = new HashMap<>();

        filteredRecordsByLoincTest.forEach((cro, codeAndUnitMatchWrapper) -> {

            OrgIdSnomedCTCodeAndUnit orgIdSnomedCTCodeAndUnit = codeAndUnitMatchWrapper.orgIdSnomedCTCodeAndUnit();
            if (!loincMappings.containsKey(orgIdSnomedCTCodeAndUnit)) {
                missingLoincMappings.computeIfAbsent(orgIdSnomedCTCodeAndUnit, k -> createMissingLoincMapping(codeAndUnitMatchWrapper, loincTests));
                }
            });

        return new ArrayList<>(missingLoincMappings.values());
    }

    private LoincMapping createMissingLoincMapping(CodeAndUnitMatchWrapper codeAndUnitMatchWrapper,
                                                   java.util.Map<LoincCodeAndUnit, LoincTest> loincTests) {
        LoincTest loincTest = loincTests.get(codeAndUnitMatchWrapper.loincCodeAndUnit());
        if (loincTest == null) {
            throw new IllegalStateException("No LoincTest LoincCode: " + codeAndUnitMatchWrapper.loincCodeAndUnit().loincCode() + " and unit: " + codeAndUnitMatchWrapper.loincCodeAndUnit().loincCode());
        }
        LoincMapping loincMapping = new LoincMapping();
        loincMapping.setOrg(codeAndUnitMatchWrapper.org());
        loincMapping.setTestCode(codeAndUnitMatchWrapper.orgIdSnomedCTCodeAndUnit().snomedCTCode());
        loincMapping.setUnit(codeAndUnitMatchWrapper.orgIdSnomedCTCodeAndUnit().unit());
        loincMapping.setTestCodingSystem(CodeSystems.SNOMED_CT);
        loincMapping.setExpectedValueType(TestResultType.ValueType.NUMERIC);
        loincMapping.setLoincTest(loincTest);
        loincMapping.setLastUpdated(dateTimeService.now());
        loincMapping.setEnteredDate(dateTimeService.now());
        return loincMapping;
    }

    private java.util.Map<OrgIdSnomedCTCodeAndUnit, LoincMapping> saveMissingLoincMappings(List<LoincMapping> missingLoincMappings) {
        return emisEsManager.saveMissingLoincMappings(missingLoincMappings).stream()
                .collect(toMap(lm -> ImmutableOrgIdSnomedCTCodeAndUnit.builder()
                        .orgId(lm.getOrg().getId())
                        .snomedCTCode(lm.getTestCode())
                        .unit(lm.getUnit())
                        .build(), identity()));
    }

    private Map<CareRecordObservation, LoincMapping> pairLoincMappingToCareRecordObservation(Map<CareRecordObservation, CodeAndUnitMatchWrapper> testResultObservations,
                                                                                             java.util.Map<OrgIdSnomedCTCodeAndUnit, LoincMapping> savedAndRetrievedLoincMappings) {
        return testResultObservations.map((cro, codeAndUnitMatchWrapper) -> {
            OrgIdSnomedCTCodeAndUnit orgIdSnomedCTCodeAndUnit = codeAndUnitMatchWrapper.orgIdSnomedCTCodeAndUnit();
            LoincMapping loincMapping = savedAndRetrievedLoincMappings.get(orgIdSnomedCTCodeAndUnit);
            if (loincMapping == null) {
                throw new IllegalStateException("No LoincMapping record for CareRecordObservation: " + cro + " " +
                        "orgId: " + orgIdSnomedCTCodeAndUnit.orgId() + " SNOMED_CT: " + orgIdSnomedCTCodeAndUnit.snomedCTCode() + " unit: " + orgIdSnomedCTCodeAndUnit.unit());
            }
            return Tuple.of(cro, loincMapping);
        });
    }

    private Map<CareRecordObservation, Either<ProcessingError<CareRecordObservation>, LoincMapping>> processResults(
                                                                                                    Map<CareRecordObservation, CodeAndUnitMatchWrapper> recordsToCodeAndUnitMatchWrapper,
                                                                                                    Map<CareRecordObservation, CodeAndUnitMatchWrapper> filteredRecordsByHavingLoincTest,
                                                                                                    Map<CareRecordObservation, LoincMapping> recordsToLoincMappings,
                                                                                                    Map<CareRecordObservation, CodingWrapper> filteredRecordsByUnitAndValueAndEffectiveDatePresent
                                                                                                                    ) {
        return filteredRecordsByUnitAndValueAndEffectiveDatePresent.map((cro, cw) -> {
            LoincMapping loincMapping = recordsToLoincMappings.get(cro).getOrNull();
            if (loincMapping != null) {
                return Tuple.of(cro, Either.right(loincMapping));
            }
            CodeAndUnitMatchWrapper codeAndUnitMatchWrapper = recordsToCodeAndUnitMatchWrapper.get(cro).getOrNull();
            if (codeAndUnitMatchWrapper == null) {
                throw new IllegalStateException("No CodeAndUnitMatchWrapper for record: " + cro);
            }
            if (!filteredRecordsByHavingLoincTest.containsKey(cro)) {
                LoincCodeAndUnit loincCodeAndUnit = codeAndUnitMatchWrapper.loincCodeAndUnit();
                emisEsManager.trackMissingLoincTest(loincCodeAndUnit);
                return Tuple.of(cro, Either.left(new ProcessingError<>(cro, NON_FATAL, "No LoincTest for Unit: %s and LoincCode: %s", loincCodeAndUnit.unit(), loincCodeAndUnit.loincCode())));
            }
            throw new IllegalStateException("No LoincMapping, LoincTest, CodingTranslation for CareRecordObservation: " + cro.observationGuid());
        });
    }

    private void reportObservationMissingUnitOrValueOrEffectiveDate(String snomedCTCode, String observationGuid) {
        if (snomedCTCode != null && observationGuid != null) {
            MISSING_UNITS_OR_VALUES_OR_EFFECTIVE_DATE.increment();
            if (croWithNoUnitOrValueOrEffectiveDateAlreadySeen.putIfAbsent(snomedCTCode, observationGuid) == null) {
                LOGGER.warn("There is no unit/value/effectiveDate provided for CareRecordObservation Guid: '{}' with SnomedCTCode: '{}'", observationGuid, snomedCTCode);
            }
        }
    }
}
