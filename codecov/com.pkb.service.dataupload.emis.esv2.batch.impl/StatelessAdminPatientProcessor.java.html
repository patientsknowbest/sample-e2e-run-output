public class StatelessAdminPatientProcessor extends AdminPatientProcessor {

    EmisEsAdminPatientRepository emisEsAdminPatientRepository;

    public StatelessAdminPatientProcessor(EmisEsProcessedRowService emisEsProcessedRowService, EmisEsManager emisEsManager,
                                          EmisProcessorAborter emisProcessAborter, Kms kmsClient, PhrConfig config,
                                          DateTimeService dateTimeService, ExternalIdService externalIdService,
                                          INotificationManager notificationManager, TeamService teamService,
                                          TeamUserManager teamUserManager, UserManager userManager,
                                          BatchItemProcessor<AdminPatient, PKBPersonDTO> adminPatientItemProcessor,
                                          PatientConsentManager patientConsentManager,
                                          PKBPersonHelper pkbPersonHelper, EmisEsStatusService emisEsStatusService,
                                          EmisOrgHelper emisOrgHelper, EmisEsAdminPatientRepository emisEsAdminPatientRepository
            , EmisEsDummyPatientService emisEsDummyPatientService) {
        super(emisEsProcessedRowService, emisEsManager, emisProcessAborter, kmsClient, config, dateTimeService, externalIdService,
                notificationManager, teamService, teamUserManager, userManager, adminPatientItemProcessor,
                patientConsentManager, pkbPersonHelper, emisEsStatusService, emisOrgHelper, emisEsDummyPatientService);
        this.emisEsAdminPatientRepository = emisEsAdminPatientRepository;
    }

    @Override
    protected EmisEsProcessedRow updateLastProcessedRow(EmisEsProcessedRow actualFileState, RowId<Long> rowId) {
        // do not save
        return actualFileState;
    }

    @Override
    protected void warmupPersonInfoCache(List<ResultWithId<AdminPatient, Long>> rows, Map<String, Org> orgGuidToOrgMap, EHRRequestContext requestContext) {
        //we don't need to go to pkbperson as this is stateless, so warming up person cache is not needed.
    }

    @Override
    protected void saveAdminPatient(java.time.ZoneId zoneId, AdminPatient adminPatient, Optional<OrgMergeHelper> orgMergeHelper) {
        recordTime("emisSave", () -> {
            if (orgMergeHelper.isEmpty()) {
                //only possible if ignored patient type, in which case we don't care anyway
                return;
            }

            AdminPatient existingRecord = emisEsManager.lookupAdminPatient(Collections.singletonList(Tuple.of(adminPatient.patientGuid(), orgMergeHelper.get().getOrg()))).get(adminPatient.patientGuid());
            if (existingRecord == null) {
                return;
            }

            emisEsManager.clearStatelessAdminPatientDeletedOverride(adminPatient.patientGuid());
            if (adminPatient.isHandleAsDeleted() != existingRecord.isHandleAsDeleted()) {
                emisEsManager.addStatelessAdminPatientDeletedOverride(adminPatient.patientGuid(), adminPatient.isHandleAsDeleted());
            }
        });
    }

    @Override
    @NotNull
    protected Boolean processAdminPatient(Optional<OrgMergeHelper> orgMergeHelper, AdminPatient adminPatient) {
        //do nothing, this is stateless
        return TRUE;
    }

    @Override
    public boolean canProcess(@NotNull EmisFileType fileType) {
        return false;
    }

    @Override
    public ProcessorType getProcessorType() {
        return STATELESS;
    }
}
