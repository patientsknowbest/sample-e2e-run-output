public class AdminUserInRoleProcessor extends BatchingBatchDataProcessor<AdminUserInRole> {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final EmisFileType TYPE = EmisFileType.ADMIN_USER_IN_ROLE;

    public AdminUserInRoleProcessor(EmisEsProcessedRowService emisEsProcessedRowService, EmisEsManager emisEsManager, EmisProcessorAborter emisProcessAborter, Kms kmsClient, PhrConfig config, DateTimeService dateTimeService, ExternalIdService externalIdService, INotificationManager notificationManager, EmisEsStatusService emisEsStatusService, EmisOrgHelper emisOrgHelper) {
        super(emisEsProcessedRowService, emisEsManager, emisProcessAborter, kmsClient, config, dateTimeService, externalIdService, notificationManager, emisEsStatusService, emisOrgHelper);
    }

    @Override
    public Class<AdminUserInRole> interfaceClass() {
        return AdminUserInRole.class;
    }

    @Override
    public boolean canProcess(@NotNull EmisFileType fileType) {
        return fileType == TYPE;
    }

    @Override
    boolean process(List<ResultWithId<AdminUserInRole, Long>> rows, EmisEsProcessedRow fileState) {
        rows.forEach(r -> emisProcessAborter.executeTestControl(TYPE, r.getRowId().getId()));

        io.vavr.collection.List<ResultWithId<AdminUserInRole, Long>> records = io.vavr.collection.List.ofAll(rows);
        var recordToSurvivorOrg = getRecordToMaybeOrgMergeHelperMapTimed(records, AdminUserInRole::organisationGuid).mapValues(orgMergeHelper -> {
            Optional<Org> maybeOrg = Optional.empty();
            return orgMergeHelper.map(OrgMergeHelper::getOrg).or(() -> maybeOrg);
        });
        var zoneIdOrFailureMap = getZoneIds(rows, recordToSurvivorOrg);

        records.forEach(rwi -> recordTime("emisSave",
                () -> emisEsManager.saveAdminUserInRole(rwi.getValue(), zoneIdOrFailureMap.get(rwi.getValue())
                        .getOrElseThrow(() -> new IllegalStateException("No TZ info available for" + rwi.getValue())))));

        return true;
    }

    private java.util.Map<AdminUserInRole, Either<ProcessingError<AdminUserInRole>, ZoneId>> getZoneIds(List<ResultWithId<AdminUserInRole, Long>> rows, Map<AdminUserInRole, Optional<Org>> recordToOrg) {
        return rows.stream().map(rwi -> getZoneId(rwi, recordToOrg)).collect(Collectors.toMap(k -> k._1, v -> v._2));
    }

    protected Tuple2<AdminUserInRole, Either<ProcessingError<AdminUserInRole>, ZoneId>> getZoneId(ResultWithId<AdminUserInRole, Long> row, Map<AdminUserInRole, Optional<Org>> recordToOrg) {
        var record = row.getValue();
        var maybeOrg = recordToOrg.get(record).flatMap(Option::ofOptional);

        ZoneId orgtz = maybeOrg.map(this::getTimezoneOfOrg).getOrElse(() -> {
            var maybeEmisOrgRecord = emisEsManager.findEmisAdminOrganization(record.organisationGuid());
            if (maybeEmisOrgRecord.isPresent()) {
                var emisOrg = maybeEmisOrgRecord.get();
                var maybeOdsCode = emisOrg.findOdsCode();
                if (maybeOdsCode.isPresent()) {
                    var odsCode = maybeOdsCode.get();
                    if (emisEsManager.isWhitelistedOrg(odsCode)) {
                        throw new IllegalStateException("Org setup is missing! Either remove Org from whitelist or create Org with ODS or add ODS code to Org! " + emisOrg);
                    }

                    emisOrgHelper.logIfOrgWasntSeenYet(emisOrg, "EMIS Org [{}] appeared in feed and is not yet set up in PKB -- using default TZ offset", emisOrg);
                    return DEFAULT_ZONE_ID;
                } else {
                    emisOrgHelper.logIfOrgWasntSeenYet(emisOrg, "No ODS code for org {} -- using default TZ offset", emisOrg);
                    return DEFAULT_ZONE_ID;
                }
            } else {
                throw new IllegalStateException("Missing EMIS org record for org guid " + record.organisationGuid());
            }
        });
        return Tuple.of(record, Either.right(orgtz));
    }


    @Override
    int getBatchSize() {
        return config.getEmisEsProcessingBatchSize();
    }

    @Override
    public boolean isPostProcessingRequired() {
        return false;
    }

    @Override
    protected boolean doPostProcess(EmisEsProcessedRow rowState) {
        return true;
    }

    @Override
    public EmisFileType getSupportedType() {
        return TYPE;
    }

    @Override
    public ProcessorType getProcessorType() {
        return NORMAL;
    }
}
