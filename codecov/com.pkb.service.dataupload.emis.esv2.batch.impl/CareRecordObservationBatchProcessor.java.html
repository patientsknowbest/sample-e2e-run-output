    @SuppressWarnings({"resource", "ReuseOfLocalVariable"})
    @Override
    public boolean process(List<ResultWithId<CareRecordObservation, Long>> rows, EmisEsProcessedRow fileState) {
        rows.forEach(r -> emisProcessAborter.executeTestControl(TYPE, r.getRowId().getId()));
        io.vavr.collection.List<ResultWithId<CareRecordObservation, Long>> records = io.vavr.collection.List.ofAll(rows);

        // Fetch data needed for further processing
        var recordToOrgMergeHelper = getRecordToOrgMergeHelperMapTimed(records, CareRecordObservation::organisationGuid);

        Map<CareRecordObservation, Either<ProcessingError<CareRecordObservation>, AdminPatient>> adminPatientsOrFailures = fetchAdminPatientsTimed(recordToOrgMergeHelper);
        Seq<ProcessingError<CareRecordObservation>> failures = extractFailures(adminPatientsOrFailures).values();
        Map<CareRecordObservation, AdminPatient> adminPatients = extractSuccesses(adminPatientsOrFailures);
        Set<CareRecordObservation> saveableRecords = adminPatients.keySet();

        var recordToSurvivorOrg = recordToOrgMergeHelper.mapValues(OrgMergeHelper::getOrg);

        Map<CareRecordObservation, Either<ProcessingError<CareRecordObservation>, PKBPersonInfo>> pkbPersonsOrFailures = fetchPkbPersonsTimed(
                adminPatients.map((k, v) -> Tuple.of(k, Tuple.of(recordToSurvivorOrg.apply(k), v))));
        failures = failures.appendAll(extractFailures(pkbPersonsOrFailures).values());
        Map<CareRecordObservation, PKBPersonInfo> pkbPersons = extractSuccesses(pkbPersonsOrFailures);
        saveableRecords = saveableRecords.intersect(pkbPersons.keySet());

        Map<CareRecordObservation, Either<ProcessingError<CareRecordObservation>, Long>> defaultAccountIdsOrFailures = fetchDefaultAccountIdsTimed(
                restrict(pkbPersons, saveableRecords));
        Map<CareRecordObservation, Long> defaultAccountIds = extractSuccesses(defaultAccountIdsOrFailures);
        failures = failures.appendAll(extractFailures(defaultAccountIdsOrFailures).values());
        saveableRecords = saveableRecords.intersect(defaultAccountIds.keySet());
        Set<CareRecordObservation> deletableRecords = saveableRecords.filter(epdp -> epdp.isHandleAsDeleted(adminPatients.get(epdp).getOrNull()));
        saveableRecords = saveableRecords.filter(epdp -> !epdp.isHandleAsDeleted(adminPatients.get(epdp).getOrNull()));

        Map<CareRecordObservation, AdminUserInRole> adminUserInRoles = fetchAdminUserInRolesTimed(recordToSurvivorOrg, CareRecordObservation::enteredByUserInRoleGuid);

        Map<CareRecordObservation, Either<ProcessingError<CareRecordObservation>, Tuple2<CodeableConcept, CodingWrapper>>> codingReceiveds = fetchCodingReceivedTimed(restrict(recordToOrgMergeHelper, saveableRecords), CareRecordObservation::uniqueKey);
        Map<CareRecordObservation, Tuple2<CodeableConcept, CodingWrapper>> codingReceivedSuccesses = extractSuccesses(codingReceiveds);
        failures = failures.appendAll(extractFailures(codingReceiveds).values());

        Map<CareRecordObservation, Either<ProcessingError<CareRecordObservation>, LoincMapping>> recordsToLoincMappingOrFailures = emisTestResultObservationHelper.retrieveLoincMappingsForTestRecords(codingReceivedSuccesses);
        Map<CareRecordObservation, LoincMapping> recordsToLoincMapping = extractSuccesses(recordsToLoincMappingOrFailures);
        failures = failures.appendAll(extractFailures(recordsToLoincMappingOrFailures).values());
        Map<CareRecordObservation, Tuple2<CodeableConcept, CodingWrapper>> codingReceivedSuccessesAndTestResultsWithLoincMapping = filterTestResultObservationsWhichNotHaveLoincMapping(
                codingReceivedSuccesses, recordsToLoincMapping);

        saveableRecords = saveableRecords.intersect(codingReceivedSuccessesAndTestResultsWithLoincMapping.keySet());

        @SuppressWarnings("Convert2MethodRef")
        Map<CodingDataType, Map<CareRecordObservation, Tuple2<CodingReceived, CodeableConcept>>> byDataType = codingReceivedSuccessesAndTestResultsWithLoincMapping
                .groupBy(t -> t._2._2.codingReceived().getCodingDataType())
                .map((k, v) -> Tuple.of(k, v.mapValues(m -> new Tuple2<>(m._2.codingReceived(), m._1))));

        failures
                .groupBy(ProcessingError::getSeverity)
                .forEach(f -> {
                    Seq<ProcessingError<CareRecordObservation>> errors = f._2;

                    if (f._1 == FATAL) {
                        incrementProcessedRowsCounter(ProcessedRowStatus.FAILED, errors.size());
                        String errorString = errors.map(ProcessingError::getMessage).collect(Collectors.joining("\n"));
                        LOGGER.error("Failed to process {} observations:\n{}", errors.size(), errorString);
                    } else {
                        incrementProcessedRowsCounter(ProcessedRowStatus.SKIPPED, errors.size());
                        String samples = errors.distinctBy(ProcessingError::getMessageKey).map(ProcessingError::getMessage).collect(Collectors.joining("\n"));
                        LOGGER.warn("Skipping {} observations,\nmessages with samples:\n{}", errors.size(), samples);
                    }
                });

        Table<String, UUID, UUID> observationGuidToUniqueId = getOrCreateExternalIdsTimed(saveableRecords, deletableRecords, CareRecordObservation::observationGuid, recordToSurvivorOrg);
        java.util.Map<String, Long> internalIdCounts = observationGuidToUniqueId.cellSet().stream().collect(Collectors.groupingBy(Table.Cell::getRowKey, Collectors.counting()));

        Set<CareRecordObservation> finalSaveableRecords = saveableRecords;

        //noinspection deprecation
        var deletionResults = handleDeletableRecordsTimed(deletableRecords,
                CareRecordObservation::observationGuid,
                uuidAndAccountPairs -> beanFactory.getEhrRemote().deleteAllForUniqueForSystemProcessWithoutAuditRow(uuidAndAccountPairs),
                observationGuidToUniqueId,
                recordToSurvivorOrg,
                defaultAccountIds);

        notifyAboutDeletion(deletableRecords, deletionResults, CareRecordObservation::observationGuid, recordToSurvivorOrg, observationGuidToUniqueId, pkbPersons);

        Map<Long, AccountKeysDTO> accountIdToAccountKeys = getRevelantAccountKeys(defaultAccountIds, byDataType);

        recordTime("save", () -> handlers.forEach(handler -> {
            // We can only process records for which all of the necessary info is available
            java.util.Set<CareRecordObservation> saveable = byDataType
                    .getOrElse(handler.getHandledDataType(), HashMap.empty()).keySet()
                    .intersect(finalSaveableRecords)
                    .toJavaSet(); // Doing this because we want the set to be mutable

            if (saveable.isEmpty()) {
                return;
            }

            DataPointSaveResult saveResult = handler.handle(
                    saveable,
                    recordToOrgMergeHelper,
                    pkbPersons,
                    adminPatients,
                    defaultAccountIds,
                    accountIdToAccountKeys,
                    adminUserInRoles,
                    codingReceivedSuccessesAndTestResultsWithLoincMapping.mapValues(v -> v._2.codingReceived()), codingReceivedSuccessesAndTestResultsWithLoincMapping.mapValues(v -> v._1),
                    recordsToLoincMapping,
                    fileState,
                    observationGuidToUniqueId
            );

            Map<ProcessedRowStatus, Integer> saveCounts = HashMap.of(
                    ProcessedRowStatus.SAVED, saveResult.persistedDataPointIds().size(),
                    ProcessedRowStatus.EXCLUDED_BY_CODE, saveResult.getExcludedDataPoints().size());

            saveCounts
                    .filter((status, count) -> count > 0)
                    .forEach((status, count) -> {
                        incrementProcessedCodedDataCounter(status, handler.getHandledDataType().toString(), count);
                        incrementProcessedRowsCounter(status, count);
                    }); // SAVED and EXCLUDED_BY_CODE

            //noinspection deprecation
            deleteOldGpDatapointsToPreventDuplicates(saveable,
                    CareRecordObservation::observationGuid,
                    uuidsWithAccountId -> beanFactory.getEhrRemote().deleteAllForUniqueForSystemProcessWithoutAuditRow(uuidsWithAccountId),
                    observationGuidToUniqueId,
                    recordToSurvivorOrg,
                    defaultAccountIds);

            deleteOldVersionsOfSavedRecordsToPreventMultipleDatatypes(saveResult);

            io.vavr.collection.HashSet.ofAll(saveable)
                    .groupBy(o -> codingReceivedSuccessesAndTestResultsWithLoincMapping.get(o).map(t -> t._2.codingReceived().getCodingDataType()).getOrElse(CodingDataType.UNMAPPED))
                    .filterKeys(codingDataType -> codingDataType != CodingDataType.UNMAPPED)
                    .forEach(((codingDataType, updatedObservations) -> updatedObservations
                            .filter(obs -> !addedDueToGpChange(obs, CareRecordObservation::observationGuid, internalIdCounts))
                            .map(observation -> Tuple.of(pkbPersons.apply(observation), recordToSurvivorOrg.apply(observation)))
                            .forEach(t -> notify(t._1, t._2, UPDATE_ACTION_MAP.get(codingDataType)))));
        }));

        return failures.find(f -> f.getSeverity() == FATAL).isEmpty();
    }

    private Map<Long, AccountKeysDTO> getRevelantAccountKeys(Map<CareRecordObservation, Long> defaultAccountIds, Map<CodingDataType, Map<CareRecordObservation, Tuple2<CodingReceived, CodeableConcept>>> byDataType) {
        io.vavr.collection.List<Long> accountIdsWeRequireKeysFor = handlers.stream()
                .map(ObservationHandler::getHandledDataType)
                .filter(codingDataType -> codingDataType != CodingDataType.UNMAPPED) //unmapped records don't do anything, no need to get account keys
                .map(datatype -> byDataType.getOrElse(datatype, HashMap.empty()).keySet())
                .flatMap(Set::toJavaStream)
                .map(defaultAccountIds::get)
                .map(Option::get)
                .collect(io.vavr.collection.List.collector());

        return attemptKeyLookup(accountIdsWeRequireKeysFor);
    }

    /**
     * delete any previous versions of the same uniqueid.
     * this may be required if we're processing a rebulk, and e.g. we've since made a change to our codingmatches that mean
     * an observation is now processed by a different handler. Don't want to end up with multiple non-deleted records
     * for the same uniqueid if they are for different datatypes, this could cause other code to break.
     * <p>
     * No need to audit these deletions, the new record serves as the audit.
     *
     * @param saveResult
     */
    private void deleteOldVersionsOfSavedRecordsToPreventMultipleDatatypes(DataPointSaveResult saveResult) {
        List<Long> deletedIds = beanFactory.getEhrRemote().deleteAnyOtherForSameUniqueId(saveResult.persistedDataPointIds());
        if (CollectionUtils.isNotEmpty(deletedIds)) {
            Metrics.counter(OLDER_RECORD_DELETE_COUNTER_NAME).increment(deletedIds.size());
        }
    }

    private <K extends Serializable> Map<CareRecordObservation, Either<ProcessingError<CareRecordObservation>, Tuple2<CodeableConcept, CodingWrapper>>> fetchCodingReceivedTimed(
            Map<CareRecordObservation, OrgMergeHelper> observations, Function1<CareRecordObservation, K> uniqueKeyExtractor) {
        return recordTime("codingReceiveds", () -> fetchCodingReceived(observations, uniqueKeyExtractor));
    }

    private <K extends Serializable> Map<CareRecordObservation, Either<ProcessingError<CareRecordObservation>, Tuple2<CodeableConcept, CodingWrapper>>> fetchCodingReceived(
            Map<CareRecordObservation, OrgMergeHelper> observations, Function1<CareRecordObservation, K> uniqueKeyExtractor) {
        Map<CareRecordObservation, Either<ProcessingError<CareRecordObservation>, CodingWrapper>> constructedCodingReceived = constructCodingReceived(
                observations, uniqueKeyExtractor);
        Map<CareRecordObservation, CodingWrapper> successes = extractSuccesses(
                constructedCodingReceived);
        Map<CareRecordObservation, ProcessingError<CareRecordObservation>> failures = extractFailures(constructedCodingReceived);
        Map<CareRecordObservation, CodingReceived> successfulCodingReceiveds = successes.mapValues(CodingWrapper::codingReceived);
        Map<K, CareRecordObservation> keyLookup = successfulCodingReceiveds.toMap(kv -> Tuple.of(uniqueKeyExtractor.apply(kv._1), kv._1));
        Map<K, CodingReceived> uniqueKeyToCodingReceived = successfulCodingReceiveds.mapKeys(uniqueKeyExtractor);
        Map<K, CodingReceived> uniqueKeyToEnsuredCodingReceived = getEmisEsManager()
                .ensureCodingReceivedIfMatched(uniqueKeyToCodingReceived);
        Map<CareRecordObservation, CodingReceived> matchedCRs = uniqueKeyToEnsuredCodingReceived.mapKeys(keyLookup);
        Map<CareRecordObservation, CodingWrapper> noMatch = successes
                .removeAll(matchedCRs.keySet());
        Map<CareRecordObservation, ProcessingError<CareRecordObservation>> noMatchErrors = noMatch
                .map((k, v) -> Tuple.of(k, new ProcessingError<>(k, getUnmatchedCodingErrorSeverity(), "Missing codingMatch for codingReceived=[%s]", v.lazyBuilder())));
        noMatch.forEach((k, v) -> emisEsManager.trackUnmatchedCodingReceived(v.codingReceived()));

        failures = failures.merge(noMatchErrors);
        successes = successes.removeAll(failures.keySet());
        return failures
                .map((k, v) -> Tuple.of(k,
                        Either.<ProcessingError<CareRecordObservation>, Tuple2<CodeableConcept, CodingWrapper>>left(v)))
                .merge(successes.map((k, v) -> Tuple.of(k, Either.right(Tuple.of(v.lazyBuilder().apply(matchedCRs.apply(k)), ImmutableCodingWrapper.builder()
                                                                                                        .codingReceived(matchedCRs.apply(k))
                                                                                                        .codingClinicalCode(v.codingClinicalCode())
                                                                                                        .lazyBuilder(v.lazyBuilder())
                                                                                                        .org(v.org()).build())))));


    }

    private Map<CareRecordObservation, Tuple2<CodeableConcept, CodingWrapper>> filterTestResultObservationsWhichNotHaveLoincMapping(Map<CareRecordObservation, Tuple2<CodeableConcept, CodingWrapper>> careRecordObservations,
                                                                                            Map<CareRecordObservation, LoincMapping> testResultObservationsWithLoincMapping) {
        return careRecordObservations.filter((cro, coding) -> {
            if (coding._2.codingReceived().getCodingDataType() == CodingDataType.TEST) {
                return testResultObservationsWithLoincMapping.containsKey(cro);
            }
            return true;
        });
    }
