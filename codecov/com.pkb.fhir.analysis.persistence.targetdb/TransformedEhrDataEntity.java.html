package com.pkb.fhir.analysis.persistence.targetdb;


import com.pkb.entities.enums.MenuDataType;
import com.pkb.fhir.common.EhrDataProcessingResult;
import com.pkb.fhir.common.enums.TransformedEhrDataStatus;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.Id;
import javax.persistence.Table;


@Entity
@Table(schema = "tmp_protected_data", name = "transformed_ehr_data")
public class TransformedEhrDataEntity implements EhrDataProcessingResult {
    @Id
    private long id;
    @Column(name = "content")
    private String content;
    @Column(name = "status")
    @Enumerated(EnumType.STRING)
    private TransformedEhrDataStatus status;
    @Column(name = "datatype")
    @Enumerated(EnumType.STRING)
    private MenuDataType dataType;
    @Column(name = "error")
    private String error;

    public String getError() {
        return error;
    }

    public void setError(String error) {
        this.error = error;
    }

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public String getContent() {
        return content;
    }

    public void setContent(String content) {
        this.content = content;
    }

    public TransformedEhrDataStatus getStatus() {
        return status;
    }

    public void setStatus(TransformedEhrDataStatus status) {
        this.status = status;
    }

    public MenuDataType getDataType() {
        return dataType;
    }

    public void setDataType(MenuDataType dataType) {
        this.dataType = dataType;
    }

    @Override
    public boolean isFailure() {
        return status == TransformedEhrDataStatus.TRANSFORM_FAILED || status == TransformedEhrDataStatus.VALIDATION_FAILED;
    }
}
