package com.pkb.fhir.analysis.persistence.targetdb;

import com.pkb.entities.enums.MenuDataType;
import com.pkb.fhir.common.enums.DecryptedEhrDataStatus;
import org.hibernate.annotations.Type;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.Id;
import javax.persistence.Table;

@Entity
@Table(schema = "tmp_protected_data", name = "ehr_data")
public class DecryptedEhrDataEntity {
    @Id
    private long id;
    @Type(type = "com.pkb.fhir.analysis.persistence.targetdb.SqlXmlType")
    @Column(name = "content")
    private String content;
    @Column(name = "status")
    @Enumerated(EnumType.STRING)
    private DecryptedEhrDataStatus status;
    @Column(name = "datatype")
    @Enumerated(EnumType.STRING)
    private MenuDataType dataType;

    public DecryptedEhrDataEntity() {
    }

    public DecryptedEhrDataEntity(long id, String content, DecryptedEhrDataStatus status, MenuDataType dataType) {
        this.id = id;
        this.content = content;
        this.status = status;
        this.dataType = dataType;
    }

    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public String getContent() {
        return content;
    }

    public void setContent(String content) {
        this.content = content;
    }

    public DecryptedEhrDataStatus getStatus() {
        return status;
    }

    public void setStatus(DecryptedEhrDataStatus status) {
        this.status = status;
    }

    public MenuDataType getDataType() {
        return dataType;
    }

    public void setDataType(MenuDataType dataType) {
        this.dataType = dataType;
    }
}
