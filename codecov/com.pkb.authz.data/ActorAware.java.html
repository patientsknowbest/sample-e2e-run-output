package com.pkb.authz.data;

import com.google.common.base.Functions;
import com.pkb.authentication.AuthenticatedIdentity;
import com.pkb.authentication.principal.team.BaseTeam;
import com.pkb.authentication.principal.user.AuthenticatedIndividualProfessional;
import com.pkb.authentication.principal.user.AuthenticatedPatient;
import com.pkb.authentication.principal.user.AuthenticatedPerson;
import com.pkb.authentication.principal.user.AuthenticatedSystemUser;
import com.pkb.authentication.principal.user.AuthenticatedTeamCoordinator;
import com.pkb.authentication.principal.user.AuthenticatedTeamProfessional;
import com.pkb.authentication.principal.user.AuthenticatedUser;
import com.pkb.authentication.principal.user.TeamUser;
import com.pkb.datamodel.Identifiable;
import com.pkb.datamodel.PubliclyIdentifiable;
import com.pkb.entities.enums.AccessingEntityType;
import io.vavr.control.Option;

import java.util.UUID;

import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.Predicates.instanceOf;

public interface ActorAware {
    AuthenticatedIdentity authenticatedIdentity();

    default AuthenticatedUser authenticatedUser() {
        return authenticatedIdentity().authenticatedUser();
    }

    default Option<AuthenticatedSystemUser> authenticatedSystemUser() {
        return Match(authenticatedUser())
                .of(Case($(instanceOf(AuthenticatedSystemUser.class)), Option::of),
                        Case($(), Option.none()));
    }

    default Option<AuthenticatedPatient> authenticatedPatient() {
        return Match(authenticatedUser())
                .option(Case($(instanceOf(AuthenticatedPatient.class)), i -> i));
    }

    default AuthenticatedPerson requireAuthenticatedPerson() {
        return Match(authenticatedUser())
                .of(Case($(instanceOf(AuthenticatedPerson.class)), Functions.identity()),
                        Case($(), () -> {
                            throw new IllegalStateException("AuthenticatedUser isn't an AuthenticatedPerson!");
                        }));
    }

    default AuthenticatedSystemUser requireAuthenticatedSystemUser() {
        return authenticatedSystemUser()
                .getOrElseThrow(() -> new IllegalStateException("AuthenticatedUser isn't an AuthenticatedSystemUser!"));
    }

    default AuthenticatedPatient requireAuthenticatedPatient() {
        return authenticatedPatient().getOrElseThrow(() -> new IllegalStateException("AuthenticatedUser isn't an AuthenticatedPatient!"));
    }

    default Option<Long> findAuthenticatedUserWithActorId() {
        return Match(authenticatedUser())
                .of(Case($(instanceOf(Identifiable.class)), it -> Option.of((Long)it.getId())),
                    Case($(), Option.none()));
    }

    default Option<UUID> findAuthenticatedUserWithActorPublicId() {
        return Match(authenticatedUser())
                .of(Case($(instanceOf(PubliclyIdentifiable.class)), it -> Option.of(it.getPublicId())),
                        Case($(), Option.none()));
    }

    default AccessingEntityType accessingEntityType() {
        return Match(authenticatedUser()).of(
                Case($(instanceOf(AuthenticatedPatient.class)), AccessingEntityType.CARER),
                Case($(instanceOf(AuthenticatedTeamProfessional.class)), AccessingEntityType.TEAM),
                Case($(instanceOf(AuthenticatedTeamCoordinator.class)), AccessingEntityType.TEAM),
                Case($(instanceOf(AuthenticatedIndividualProfessional.class)), AccessingEntityType.INDIVIDUAL)
                // TODO: https://pkbdev.atlassian.net/browse/PHR-7728
                // System user is not catered for in this list 
        );
    }

    default Option<BaseTeam> authenticatedUserTeam() {
        return Match(authenticatedUser())
                .option(Case($(instanceOf(TeamUser.class)), TeamUser::getTeam));
    }
}
