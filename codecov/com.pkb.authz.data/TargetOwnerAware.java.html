package com.pkb.authz.data;

import com.pkb.authz.TargetMeta;
import com.pkb.datamodel.authorization.TargetOwnerData;
import io.vavr.collection.Map;
import io.vavr.control.Option;

import static com.google.common.base.Preconditions.checkState;

public interface TargetOwnerAware {
    Map<TargetMeta<?>, TargetOwnerData> targetOwnerData();

    default Option<TargetOwnerData> singleTargetData() {
        checkState(targetOwnerData().size() <= 1, "Expected up to 1 authorization result");
        return targetOwnerData().values().headOption();
    }

    default TargetOwnerData requireSingleTargetData() {
        checkState(targetOwnerData().size() == 1, "Expected exactly 1 authorization result");
        return singleTargetData().get();
    }

    default TargetMeta requireSingleTarget() {
        checkState(targetOwnerData().size() == 1, "Expected exactly 1 authorization result");
        return targetOwnerData().keySet().head();
    }
}
