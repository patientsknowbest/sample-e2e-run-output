package com.pkb.consent.entity;

import java.io.Serializable;
import java.util.Optional;
import java.util.Set;

import javax.persistence.Column;
import javax.persistence.Embeddable;

import org.apache.commons.lang3.NotImplementedException;
import org.jetbrains.annotations.NotNull;

import com.google.common.base.MoreObjects;
import com.pkb.app.PrivacyFlagsToPrivacyFlagConverter;
import com.pkb.entities.enums.PrivacyFlag;

/**
 *
 * @author Russ Gray
 */
@Embeddable
public class PrivacyFlags implements Serializable {

    private static final long serialVersionUID = -8047988055336653044L;

    @Column
    private boolean requireGeneral;

    @Column
    private boolean requireSexualHealth;

    @Column
    private boolean requireMentalHealth;

    @Column
    private boolean requireSocialCare;

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(PrivacyFlags.class)
                .omitNullValues()
                .add("requireGeneral", requireGeneral)
                .add("requireSexualHealth", requireSexualHealth)
                .add("requireMentalHealth", requireMentalHealth)
                .add("requireSocialCare", requireSocialCare)
                .toString();
    }

    public static PrivacyFlags copy(PrivacyFlags other) {
        PrivacyFlags result = new PrivacyFlags();
        result.requireGeneral = other.requireGeneral;
        result.requireSexualHealth = other.requireSexualHealth;
        result.requireMentalHealth = other.requireMentalHealth;
        result.requireSocialCare = other.requireSocialCare;
        return result;
    }

    public boolean getRequireGeneral() {
        return requireGeneral;
    }

    public void setRequireGeneral(boolean requireGeneral) {
        this.requireGeneral = requireGeneral;
    }

    public boolean getRequireSexualHealth() {
        return requireSexualHealth;
    }

    public void setRequireSexualHealth(boolean requireSexualHealth) {
        this.requireSexualHealth = requireSexualHealth;
    }

    public boolean getRequireMentalHealth() {
        return requireMentalHealth;
    }

    public void setRequireMentalHealth(boolean requireMentalHealth) {
        this.requireMentalHealth = requireMentalHealth;
    }

    public boolean getRequireSocialCare() {
        return requireSocialCare;
    }

    public void setRequireSocialCare(boolean requireSocialCare) {
        this.requireSocialCare = requireSocialCare;
    }

    /**
     * Gets consent as a privacy flag.
     */
    public Optional<PrivacyFlag> getPrivacyFlag() {
        return PrivacyFlagsToPrivacyFlagConverter.findPrivacyFlag(requireGeneral, requireMentalHealth, requireSexualHealth, requireSocialCare);
    }

    public Set<PrivacyFlag> getPrivacyFlags() {
        return PrivacyFlagsToPrivacyFlagConverter.getPrivacyFlags(requireGeneral, requireMentalHealth, requireSexualHealth, requireSocialCare);
    }

    public PrivacyFlags setPrivacyFlag(@NotNull PrivacyFlag requiredFlag) {
        switch (requiredFlag) {
        case GENERAL:
            requireGeneral = true;
            requireMentalHealth = false;
            requireSexualHealth = false;
            requireSocialCare = false;
            return this;
        case MENTAL_HEALTH:
            requireGeneral = false;
            requireMentalHealth = true;
            requireSexualHealth = false;
            requireSocialCare = false;
            return this;
        case SEXUAL_HEALTH:
            requireGeneral = false;
            requireMentalHealth = false;
            requireSexualHealth = true;
            requireSocialCare = false;
            return this;
        case SOCIAL_CARE:
            requireGeneral = false;
            requireMentalHealth = false;
            requireSexualHealth = false;
            requireSocialCare = true;
            return this;
        default:
            throw new NotImplementedException("Privacy flag type " + requiredFlag + " not implemented yet!");
        }
    }
}
