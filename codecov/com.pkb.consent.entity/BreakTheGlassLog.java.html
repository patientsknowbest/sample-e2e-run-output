package com.pkb.consent.entity;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

import java.io.Serializable;
import java.time.Instant;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;

import org.hibernate.annotations.GenericGenerator;

/**
 *
 * @author Russ Gray <russ.gray@patientsknowbest.com>
 */
@Entity
@Table
public class BreakTheGlassLog implements Serializable {

    private static final long serialVersionUID = -7241223905375747306L;

    @Id
    @GenericGenerator(name = "seq_al", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME), // based on "pooled-lo"
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_accesslog"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(
            strategy = GenerationType.SEQUENCE,
            generator = "seq_al")
    private Long id;

    @Column(nullable = false)
    private Long accessingPersonId;

    @Column(nullable = false)
    private Long teamId;

    @Column(nullable = false)
    private Long patientPersonId;

    @Column(nullable = false)
    private Instant accessTime;

    public BreakTheGlassLog() {
    }

    public BreakTheGlassLog(Long accessingPersonId, Long teamId, Long patientPersonId, Instant accessTime) {
        this.accessingPersonId = accessingPersonId;
        this.teamId = teamId;
        this.patientPersonId = patientPersonId;
        this.accessTime = accessTime;
    }

    @Override
    public String toString() {
        return "BreakTheGlassLog{" + "id=" + id + ", accessingPersonId=" + accessingPersonId + ", teamId=" + teamId + ", patientPersonId="
                + patientPersonId + ", accessTime=" + accessTime + '}';
    }

    // <editor-fold desc="Accessors">

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Long getAccessingPersonId() {
        return accessingPersonId;
    }

    public void setAccessingPersonId(Long accessingPersonId) {
        this.accessingPersonId = accessingPersonId;
    }

    public Long getTeamId() {
        return teamId;
    }

    public void setTeamId(Long teamId) {
        this.teamId = teamId;
    }

    public Long getPatientPersonId() {
        return patientPersonId;
    }

    public void setPatientPersonId(Long patientPersonId) {
        this.patientPersonId = patientPersonId;
    }

    public Instant getAccessTime() {
        return accessTime;
    }

    public void setAccessTime(Instant accessTime) {
        this.accessTime = accessTime;
    }

    // </editor-fold>
}
