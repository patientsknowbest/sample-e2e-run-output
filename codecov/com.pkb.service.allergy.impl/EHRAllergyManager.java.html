package com.pkb.service.allergy.impl;

import com.pkb.allergy.entity.Allergy;
import com.pkb.app.entity.EHRData;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.EHRSearch;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.app.entity.PKBFilter;
import com.pkb.coding.entity.CodingReceived;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.common.util.FrameFilter;
import com.pkb.data.EHRRemote;
import com.pkb.entities.enums.CodingDataType;
import com.pkb.entities.enums.MenuDataType;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.service.coding.CodingManager;
import com.pkb.service.ehr.DeduplicatorManager;
import com.pkb.service.ehr.MassUpdatableEhrDataPointManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import io.vavr.Function2;
import io.vavr.Tuple2;
import io.vavr.Tuple3;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.time.Instant;
import java.util.List;

public class EHRAllergyManager extends MassUpdatableEhrDataPointManager<Allergy> {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private static final String PATIENT_IS_ALLERGIC = "boolean01";

    private static final Function2<Allergy, Long, Void> ALLERGEN_CODING_RECEIVED_ID_SETTER = (allergy, privateId) -> {
        allergy.setAllergenCodingReceivedId(privateId);
        return null;
    };

    private static final Function2<Allergy, PrivacyFlag, Void> ALLERGEN_PRIVACY_FLAG_HANDLER = (allergy, privacyFlag) -> {
        allergy.getBaseFields().setPrivacyFlag(privacyFlag);
        return null;
    };

    private static final Function2<Allergy, Long, Void> DISORDER_CODING_RECEIVED_ID_SETTER = (allergy, privateId) -> {
        allergy.setDisorderCodingReceivedId(privateId);
        return null;
    };

    private static final Function2<Allergy, PrivacyFlag, Void> DISORDER_PRIVACY_FLAG_HANDLER = (allergy, privacyFlag) -> {
        if (allergy.getBaseFields().getPrivacyFlag().isPresent() && !allergy.getBaseFields().isConsentEqualTo(privacyFlag)) {
            // it's possible to get into this situation if a patient has manually amended the privacy flag on a datapoint,
            // and then an update occurs (e.g. EMIS ES updates same record).
            // we don't want the update to fail because of this check, but leaving a warning here in case it helps debug any future issues
            LOGGER.warn("Allergen and disorder coding matches conflict for uniqueid {}", allergy.getBaseFields().getUniqueId(), FrameFilter.filter(new Exception("privacy flag mismatch").fillInStackTrace()));
        }
        allergy.getBaseFields().setPrivacyFlag(privacyFlag);
        return null;
    };

    public EHRAllergyManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService,
                             UUIDProvider uuidProvider, CodingManager codingManager, DeduplicatorManager deduplicator,
                             Class<Allergy> dtoClass, MenuDataType dataType) {
        super(config, beanFactory, dateTimeService, uuidProvider, codingManager, deduplicator, dtoClass, dataType);
    }

    @Override
    protected List<Tuple2<Allergy, EHRRequestContext>> adjustCodings(List<Tuple2<Allergy, EHRRequestContext>> data) {
        List<Tuple2<Allergy, EHRRequestContext>> excludedDataPoints = doAdjustCodings(data, CodingDataType.ALLERGEN, Allergy::getAllergen,
                ALLERGEN_CODING_RECEIVED_ID_SETTER,
                ALLERGEN_PRIVACY_FLAG_HANDLER);

        excludedDataPoints.addAll(doAdjustCodings(data, CodingDataType.ALLERGY_DISORDER, Allergy::getDisorder,
                DISORDER_CODING_RECEIVED_ID_SETTER,
                DISORDER_PRIVACY_FLAG_HANDLER));
        return excludedDataPoints;
    }

    @Override
    protected List<Tuple2<Allergy, EHRRequestContext>> adjustCodingsNoCodingLookup(List<Tuple3<Allergy, CodingReceived, EHRRequestContext>> data) {
        List<Tuple2<Allergy, EHRRequestContext>> excludedDataPoints = doAdjustCodingsNoCodingLookup(data,
                ALLERGEN_CODING_RECEIVED_ID_SETTER,
                ALLERGEN_PRIVACY_FLAG_HANDLER);

        excludedDataPoints.addAll(doAdjustCodingsNoCodingLookup(data,
                DISORDER_CODING_RECEIVED_ID_SETTER,
                DISORDER_PRIVACY_FLAG_HANDLER));
        return excludedDataPoints;
    }

    private void addNegativeAllergiesFilter(boolean includeNegativeAllergies, EHRSearch search) {
        if (!includeNegativeAllergies) {
            search.addFilter(new PKBFilter(PATIENT_IS_ALLERGIC, PKBFilter.Operator.TRUE_AND_IS_NOT_NULL, null,
                    false));/*apply when choosing latest*/
        }
    }

    public List<Allergy> getAllergies(@NotNull LoggedInEHRRequestContext requestContext, long patientAccountId, int offset,
                                      int pageSize, boolean includeNegativeAllergies) {
        EHRRemote ehrBean = beanFactory.getEhrRemote();

        EHRSearch<Allergy> ehrSearch = new EHRSearch<>(patientAccountId, Allergy.class, Allergy.MS_PATH);

        ehrSearch.setLatestPerTypeFilter(EHRData.ENTERED_DATE, EHRData.UNIQUE_ID);
        ehrSearch.setOrderBy("enteredDate", EHRSearch.OrderByDirection.Asc);
        addNegativeAllergiesFilter(includeNegativeAllergies, ehrSearch);

        ehrSearch.setResultsMaxReturned(pageSize);
        ehrSearch.setResultsStartPosition(offset);

        // this needs to return only latest for all uniqueIds
        return ehrBean.queryDecryptAndPopulateData(ehrSearch, Allergy.class, requestContext);
    }

    public List<Allergy> getAllergies(@NotNull LoggedInEHRRequestContext requestContext, long patientAccountId,
                                      Instant updatedSinceDate, boolean includeNegativeAllergies) {

        EHRRemote ehrBean = beanFactory.getEhrRemote();

        EHRSearch<Allergy> ehrSearch = new EHRSearch<>(patientAccountId, Allergy.class, Allergy.MS_PATH);

        if (updatedSinceDate != null) {
            ehrSearch.addFilter(new PKBFilter(EHRData.ENTERED_DATE, PKBFilter.Operator.GREATER, updatedSinceDate));
        }
        ehrSearch.setLatestPerTypeFilter(EHRData.ENTERED_DATE, EHRData.UNIQUE_ID);
        addNegativeAllergiesFilter(includeNegativeAllergies, ehrSearch);

        List<EHRData> ehrList = ehrBean.queryAndDecryptEHRData(ehrSearch, requestContext);
        return ehrBean.populateDTOList(ehrList, Allergy.class, requestContext);
    }
}
