package com.pkb.service.encounter.workflow;

import com.pkb.api.internal.messaging.MessagingPKBPersonDTO;
import com.pkb.api.internal.messaging.PKBPersonToMessagingPKBPersonDTOConverter;
import com.pkb.datamodel.messages.workflow.ConversationWorkflow;
import com.pkb.datamodel.messages.workflow.ConversationWorkflowExisting;
import com.pkb.encounter.workflow.ArchivedStatus;
import com.pkb.encounter.workflow.ConversationWorkflowDTO;
import com.pkb.encounter.workflow.ImmutableConversationWorkflowDTO;
import com.pkb.user.entity.PKBPerson;
import io.vavr.Value;
import org.jetbrains.annotations.NotNull;

import java.time.Instant;
import java.util.Map;
import java.util.Set;
import java.util.UUID;
import java.util.stream.Collectors;

import static java.util.stream.Collectors.toMap;

public class ConversationWorkflowConverter {

    private PKBPersonToMessagingPKBPersonDTOConverter pkbPersonDTOConverter;

    public ConversationWorkflowConverter(PKBPersonToMessagingPKBPersonDTOConverter pkbPersonDTOConverter) {
        this.pkbPersonDTOConverter = pkbPersonDTOConverter;
    }

    private ConversationWorkflowDTO convertToDTO(ConversationWorkflowExisting conversationWorkflow, Map<Long, PKBPerson> personsMap) {
        var person = conversationWorkflow.getAssignedPersonId().map(personsMap::get).getOrElse((PKBPerson) null);

        MessagingPKBPersonDTO personDTO = pkbPersonDTOConverter.convert(person);

        return ImmutableConversationWorkflowDTO.builder()
                .archivedStatus(conversationWorkflow.isArchived() ? ArchivedStatus.ARCHIVED : ArchivedStatus.UNARCHIVED)
                .archiveChangeDate(conversationWorkflow.getArchiveChangeDate().getOrElse((Instant) null))
                .assignedPerson(personDTO)
                .assignChangeDate(conversationWorkflow.getAssignChangeDate().getOrElse((Instant) null))
                .optimisticLockVersion(conversationWorkflow.getOptimisticLockVersion())
                .build();
    }

    public Map<UUID, ConversationWorkflowDTO> convertToDTOs(@NotNull Map<UUID, ConversationWorkflowExisting> conversationWorkflowByConversationId,
            @NotNull Map<Long, PKBPerson> personsMap) {
        return conversationWorkflowByConversationId.entrySet().stream().collect(toMap(Map.Entry::getKey, e -> convertToDTO(e.getValue(), personsMap)));
    }

    public Set<Long> getAllAssignedPersonIds(@NotNull Map<UUID, ConversationWorkflowExisting> conversationWorkflowByConversationId) {
        return conversationWorkflowByConversationId.values().stream()
                .map(ConversationWorkflow::getAssignedPersonId)
                .flatMap(Value::toJavaStream)
                .collect(Collectors.toSet());
    }

}
