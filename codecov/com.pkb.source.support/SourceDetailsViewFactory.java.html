package com.pkb.source.support;

import com.pkb.app.entity.SourceDetails;
import com.pkb.source.domain.ImmutableSourceDetailsView;
import com.pkb.source.domain.SourceDetailsAdapter;
import com.pkb.source.domain.SourceDetailsView;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import static org.apache.commons.lang3.StringUtils.isNotBlank;

/**
 * Provides a single entry point for creating {@link SourceDetailsView} instances based on existing
 * {@link SourceDetails} instances.
 *
 */
@Component
public class SourceDetailsViewFactory {
    private SourceDetailsLocalizedDisplayProvider sourceDetailsLocalizedDisplayProvider;

    @Autowired
    public SourceDetailsViewFactory(SourceDetailsLocalizedDisplayProvider sourceDetailsLocalizedDisplayProvider) {
        this.sourceDetailsLocalizedDisplayProvider = sourceDetailsLocalizedDisplayProvider;
    }

    public SourceDetailsAdapter createSourceDetailsView(SourceDetails sourceDetails) {
        String displayText = createDisplayText(sourceDetails);
        SourceDetailsView sourceDetailsView = ImmutableSourceDetailsView
                .builder()
                .displayText(displayText)
                .displayTextFull(createDisplayTextFull(sourceDetails))
                .isSourceVisible(isNotBlank(displayText))
                .build();
        SourceDetailsAdapter result = new SourceDetailsAdapter(sourceDetailsView);
        result.init(sourceDetails);
        return result;
    }

    private String createDisplayTextFull(SourceDetails sourceDetails) {
        return sourceDetailsLocalizedDisplayProvider.getDisplayTextFull(sourceDetails);
    }

    private String createDisplayText(SourceDetails sourceDetails) {
        return sourceDetailsLocalizedDisplayProvider.getDisplayText(sourceDetails);
    }




}
