package com.pkb.source.support;

import com.pkb.app.entity.SourceDetails;
import com.pkb.datamodel.Hl7Partner;
import com.pkb.domain.Hl7PartnerService;
import com.pkb.service.team.TeamManager;
import com.pkb.util.LocalizedTextProvider;
import org.apache.commons.lang3.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.Arrays;
import java.util.Objects;
import java.util.Optional;
import java.util.stream.Collectors;
import java.util.stream.Stream;

import static java.util.Optional.ofNullable;
import static org.apache.commons.lang3.StringUtils.isBlank;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

@Component
public class SourceDetailsLocalizedDisplayProvider {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private LocalizedTextProvider localizedTextProvider;
    private Hl7PartnerService hl7PartnerService;
    private TeamManager teamManager;

    @Autowired
    public SourceDetailsLocalizedDisplayProvider(LocalizedTextProvider localizedTextProvider, Hl7PartnerService hl7PartnerService, TeamManager teamManager) {
        this.localizedTextProvider = localizedTextProvider;
        this.hl7PartnerService = hl7PartnerService;
        this.teamManager = teamManager;
    }

    public String getDisplayText(SourceDetails sourceDetails) {
        return forgeDisplayTextWithPartner(sourceDetails, getBaseDisplayText(sourceDetails));
    }

    public String getDisplayTextFull(SourceDetails sourceDetails) {
        return forgeDisplayTextWithPartner(sourceDetails, getBaseDisplayTextFull(sourceDetails));
    }

    private String forgeDisplayTextWithPartner(SourceDetails sourceDetails, String baseDisplayText) {
        String result = null;
        if (isNotBlank(sourceDetails.getResponsibleOrganisationIdentifier())) {
            result = formatResponsibleOrganisationDisplayText(baseDisplayText, sourceDetails);
        } else if (sourceDetails.getHl7PartnerId() != null) {
            Optional<Hl7Partner> partner = hl7PartnerService.findPartnerById(sourceDetails.getHl7PartnerId());
            if (partner.isPresent()) {
                result = formatPartnerDisplayText(partner.get().getName(), baseDisplayText);
            }
        } else if (sourceDetails.getConnectingOrgId() != null && sourceDetails.getOrgId() != null && !sourceDetails.getOrgId().equals(sourceDetails.getConnectingOrgId())) {
            LOGGER.info("PHR-11123: building display text with distinct orgId and connectingOrgId");
            result = teamManager.getOrg(sourceDetails.getConnectingOrgId()).map(t -> formatPartnerDisplayText(t.getName(), baseDisplayText)).orElse(null);
        }
        if (result == null) {
            result = baseDisplayText;
        }
        return result;
    }

    private String formatResponsibleOrganisationDisplayText(String sourceOrgName, SourceDetails sourceDetails) {
        return localizedTextProvider.getText("source.display.responsible.organisation",
                tryPrefixWithPartner(sourceDetails, sourceOrgName),
                sourceDetails.getResponsibleOrganisationIdentifier());
    }

    private String tryPrefixWithPartner(SourceDetails sourceDetails, String sourceOrgName) {
        return ofNullable(sourceDetails.getHl7PartnerId())
                .flatMap(hl7PartnerService::findPartnerById)
                .map(partner -> formatPartnerDisplayText(partner.getName(), sourceOrgName))
                .orElse(sourceOrgName);
    }

    private String formatPartnerDisplayText(String leftOrgName, String rightOrgName) {
        return String.format("%s %s %s", leftOrgName, localizedTextProvider.getText("source.display.partner.onBehalfOf"), rightOrgName);
    }

    private String getBaseDisplayText(SourceDetails sourceDetails) {
        // no source info
        if (sourceDetails.getPersonId() == null && sourceDetails.getTeamId() == null && sourceDetails.getOrgId() == null) {
            return (StringUtils.isNotEmpty(sourceDetails.getText())) ? sourceDetails.getText() : "";
        }
        if (isBlank(sourceDetails.getPersonNameText()) && sourceDetails.getTeamName() == null && sourceDetails.getOrgName() == null) {
            return "SOURCE DISPLAY NOT LOADED";
        }

        String personNameText = sourceDetails.getPersonNameText();
        if (isBlank(personNameText)) {
            personNameText = null;
        }
        String teamNameText = joinSkipBlanks(" ", sourceDetails.getTeamName(), sourceDetails.getOrgName());

        // only most specific source
        String mostSpecificSource = Stream.of(personNameText, teamNameText, sourceDetails.getOrgName())
                .filter(Objects::nonNull)
                .findFirst()
                .orElse("");

        return joinSkipBlanks(" - ", mostSpecificSource, sourceDetails.getText());
    }

    private String getBaseDisplayTextFull(SourceDetails sourceDetails) {
        String personName = sourceDetails.getPersonNameText();

        return joinSkipBlanks(", ",
                sourceDetails.getText(),
                personName,
                sourceDetails.getTeamName(),
                sourceDetails.getOrgName());
    }

    private static String joinSkipBlanks(String separator, String... parts) {
        return Arrays.stream(parts)
                .filter(StringUtils::isNotBlank)
                .collect(Collectors.joining(separator));

    }

}
