package com.pkb.domain.repository;

import com.cosium.spring.data.jpa.entity.graph.repository.EntityGraphJpaRepository;
import com.cosium.spring.data.jpa.entity.graph.repository.EntityGraphJpaSpecificationExecutor;
import com.pkb.entities.enums.ValidNationalId;
import com.pkb.entities.pub.PublicNationalId;
import io.vavr.collection.List;
import org.jetbrains.annotations.NotNull;
import org.springframework.data.jpa.domain.Specification;

import javax.persistence.criteria.CriteriaBuilder;
import javax.persistence.criteria.CriteriaQuery;
import javax.persistence.criteria.Join;
import javax.persistence.criteria.JoinType;
import javax.persistence.criteria.Predicate;
import javax.persistence.criteria.Root;

public interface NationalIdRepository extends EntityGraphJpaRepository<PublicNationalId, Long>, EntityGraphJpaSpecificationExecutor<PublicNationalId> {

    class PublicNationalIdByTypeAndValueSpecification implements Specification<PublicNationalId> {

        private static final long serialVersionUID = 1L;

        private List<ValidNationalId> nationalIds;

        public PublicNationalIdByTypeAndValueSpecification(List<ValidNationalId> nationalIds) {
            this.nationalIds = nationalIds;
        }

        @Override
        public Predicate toPredicate(Root<PublicNationalId> root, @NotNull CriteriaQuery<?> query, CriteriaBuilder builder) {
            Join person = root.join("person");
            Join optOut = person.join("optoutDetails", JoinType.LEFT);
            return builder.and(
                    builder.or(
                            builder.isNull(person.get("optoutDetails").get("sharingEnabled")),
                            builder.equal(optOut.get("sharingEnabled"), true)),
                    builder.and(
                            builder.or(
                                    nationalIds.map(
                                            nationalId -> builder.and(
                                                    builder.equal(root.get("type"), nationalId.type()),
                                                    builder.equal(root.get("value"), nationalId.value())))
                                            .toJavaArray(predicate -> new Predicate[] {}))));
        }

        public List<ValidNationalId> getNationalIdTypeAndValue() {
            return List.ofAll(nationalIds);
        }
    }
}
