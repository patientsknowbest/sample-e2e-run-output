package com.pkb.domain.repository;

import com.pkb.authz.data.AuthorizationData;
import com.pkb.domain.criteria.ImmutableTemporalRangeRestriction;
import com.pkb.domain.criteria.ResultFilter;
import com.pkb.domain.criteria.TemporalRangeRestriction;
import org.immutables.value.Value;
import org.immutables.vavr.encodings.VavrEncodingEnabled;

import java.time.LocalDate;

import static com.pkb.domain.criteria.ResultFilter.ALL;

@Value.Immutable
@VavrEncodingEnabled
public interface LabResultQuery {

    @Value.Default
    default TemporalRangeRestriction<LocalDate> getEffectiveTimestamp(){
        return ImmutableTemporalRangeRestriction.<LocalDate>builder().build();
    }
    
    AuthorizationData authData();

    @Value.Default
    default ResultFilter resultFilter() {
        return ALL;
    }
}
