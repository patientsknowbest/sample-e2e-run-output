package com.pkb.domain.repository;

import com.pkb.authentication.AccessToken;
import com.pkb.entities.pub.PublicApiAuthSession;
import com.pkb.entities.pub.PublicApiClient;
import io.vavr.collection.List;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.Repository;

import java.util.stream.Collectors;

import static java.lang.invoke.MethodHandles.lookup;
import static org.slf4j.LoggerFactory.getLogger;

public interface ApiAuthSessionRepository extends Repository<PublicApiAuthSession, Long> {
    Logger LOGGER = getLogger(lookup().lookupClass());

    default String findClientNameAndApiId(AccessToken accessToken) {
        String token = accessToken.tokenAsString();
        List<PublicApiAuthSession> apiAuthSessions = findApiAuthSessionForToken(token);

        if (apiAuthSessions.isEmpty()) {
            return "unknown";
        } else {
            if (apiAuthSessions.size() > 1) {
                LOGGER.warn("Multiple clients match token {} - {}", token,
                        apiAuthSessions
                                .map(s -> getClientApiDetails(s.getClient()))
                                .collect(Collectors.joining(",")));
            }

            PublicApiClient client = apiAuthSessions.get(0).getClient();
            return getClientApiDetails(client);
        }
    }

    @NotNull
    private String getClientApiDetails(PublicApiClient client) {
        return client.getName() + " [" + client.getApiId() + "]";
    }

    @Query("SELECT aas FROM PublicApiAuthSession aas WHERE aas.accessTokenPart = (:accessTokenPart)")
    List<PublicApiAuthSession> findApiAuthSessionForToken(String accessTokenPart);
}
