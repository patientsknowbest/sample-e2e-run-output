package com.pkb.domain.repository;

import com.pkb.domain.internal.repository.InstituteUserRepository;
import com.pkb.entities.pub.PublicInstituteUser;
import com.pkb.entities.pub.PublicTeam;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.springframework.data.jpa.repository.EntityGraph;
import org.springframework.data.repository.Repository;

import java.util.UUID;

class CustomTeamRepositoryImpl implements CustomTeamRepository {
    private final InstituteUserRepository repository;

    protected CustomTeamRepositoryImpl(InstituteUserRepository repository) {
        this.repository = repository;
    }

    @Override
    public Option<PublicTeam> findTeamOfClinician(long clinicianId) {
        return repository.findCliniciansTeamUser(clinicianId)
                .map(PublicInstituteUser::getTeam);
    }
}

interface CustomTeamRepository {
    Option<PublicTeam> findTeamOfClinician(long clinicianId);
}

public interface TeamRepository extends Repository<PublicTeam, Long>, CustomTeamRepository {
    @EntityGraph(attributePaths = {"org"})
    @NotNull
    Option<PublicTeam> findOptionalByPublicId(@NotNull UUID publicId);

    @EntityGraph(attributePaths = {"org"})
    @NotNull
    List<PublicTeam> findByOrg_publicId(@NotNull UUID orgPublicId);
}
