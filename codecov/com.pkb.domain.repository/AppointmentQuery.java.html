package com.pkb.domain.repository;

import com.google.common.base.Preconditions;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.domain.criteria.TemporalRangeRestriction;
import org.immutables.value.Value;
import org.immutables.vavr.encodings.VavrEncodingEnabled;

import java.time.LocalDate;
import java.util.Objects;
import java.util.UUID;

@Value.Immutable
@VavrEncodingEnabled
@Value.Style(builder = "appointmentQuery")
public interface AppointmentQuery {

    AuthorizationData authData();

    UUID patientId();

    TemporalRangeRestriction<LocalDate> dateRestriction();

    int maxNumberOfResources();

    @Value.Check
    default void patientIdMatchesAuthData() {
        authData()
                .singleTargetData()
                .map(it -> it.person().getPublicId())
                .peek(authDataOwnerPersonId ->
                        Preconditions.checkArgument(Objects.equals(authDataOwnerPersonId, patientId()),
                                "Requested appointments for patient ID %s, but got auth data for %s", patientId(), authDataOwnerPersonId));
    }
}
