package com.pkb.domain.repository;


import com.pkb.authz.data.AuthorizationData;
import com.pkb.datamodel.AppMenuDataWithSourceObjects;
import com.pkb.datamodel.RadiologyImage;
import com.pkb.datamodel.entity.mapper.RadiologyMapper;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.criteria.ImmutableCriteriaResult;
import com.pkb.domain.criteria.TemporalRangeRestriction;
import com.pkb.domain.internal.repository.AbstractMenuDataRepository;
import com.pkb.domain.internal.repository.AppMenuDataRepository;
import com.pkb.domain.internal.repository.ReferenceDatumRepository;
import com.pkb.domain.repository.mapper.AccessFilterMapper;
import com.pkb.domain.repository.mapper.SearchSpecMapper;
import com.pkb.ehrdata.AccessFilter;
import com.pkb.ehrdata.Filter;
import com.pkb.ehrdata.ImmutableLatestFilterFields;
import com.pkb.ehrdata.ImmutableOrderBy;
import com.pkb.ehrdata.ImmutableSearchSpec;
import com.pkb.ehrdata.OrderBy;
import com.pkb.ehrdata.SearchSpec;
import com.pkb.entities.enums.MenuDataType;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;

import java.time.LocalDate;

import static com.pkb.domain.criteria.CriteriaResult.emptyCriteriaResult;
import static com.pkb.ehrdata.Filter.filter;

public class RadiologyRepository extends AbstractMenuDataRepository {

    private final RadiologyMapper radiologyMapper;
    private final AccessFilterMapper accessFilterMapper;
    private final SearchSpecMapper searchSpecMapper;

    public RadiologyRepository(AppMenuDataRepository menuDataRepository,
                               ReferenceDatumRepository referenceDatumRepository,
                               RadiologyMapper radiologyMapper,
                               AccessFilterMapper accessFilterMapper,
                               SearchSpecMapper searchSpecMapper,
                               PersonRepository personRepository) {
        super(menuDataRepository, referenceDatumRepository, personRepository);
        this.radiologyMapper = radiologyMapper;
        this.accessFilterMapper = accessFilterMapper;
        this.searchSpecMapper = searchSpecMapper;
    }

    public CriteriaResult<RadiologyImage> findImagesBy(
            AuthorizationData authzData,
            TemporalRangeRestriction<LocalDate> effectiveTimestamp,
            Option<Integer> maybePageSize) {

        AccessFilter accessFilter = accessFilterMapper.buildAccessFilter(authzData);

        SearchSpec searchSpec = ImmutableSearchSpec.builder()
                .addAccountId(authzData.requireSingleTargetData().account().getId())
                .addDataType(MenuDataType.radiologyResult.name())
                .pageFilter(searchSpecMapper.mapPageFilter(maybePageSize).toJavaOptional())
                .addAllFilters(searchSpecMapper.mapEffectiveTimestamp(effectiveTimestamp, "date02"))
                .addFilter(filter("string01", Filter.Operator.EQUAL, "IMAGING"))
                .orderBy(ImmutableOrderBy.builder()
                        .direction(OrderBy.Direction.Asc)
                        .field("date01")
                        .build())
                .latestFilter(ImmutableLatestFilterFields.builder()
                        .dateField("enteredDate")
                        .typeField("uniqueId")
                        .build())
                .build();

        List<Long> ids = find(searchSpec, accessFilter);
        if (ids.isEmpty()) {
            return emptyCriteriaResult();
        }

        return menuDataToRadiologyImages(findMenuDatasByIds(ids));
    }

    @NotNull
    private CriteriaResult<RadiologyImage> menuDataToRadiologyImages(@NotNull List<AppMenuDataWithSourceObjects> wrappers) {
        return ImmutableCriteriaResult.<RadiologyImage>builder()
                .addAllPage(wrappers.map(radiologyMapper::menuDataToRadiologyImage))
                .total(wrappers.size())
                .build();
    }
}
