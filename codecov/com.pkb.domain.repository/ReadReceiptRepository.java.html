package com.pkb.domain.repository;

import com.pkb.datamodel.projections.ImmutableReadReceipt;
import com.pkb.datamodel.projections.ReadReceipt;
import com.pkb.domain.internal.repository.AbstractMenuDataRepository;
import com.pkb.domain.internal.repository.AppMenuDataRepository;
import com.pkb.domain.internal.repository.ReferenceDatumRepository;
import com.pkb.entities.app.AppMenuData;
import com.pkb.entities.doc.DocUnreadableDocument;
import io.vavr.collection.List;

public class ReadReceiptRepository extends AbstractMenuDataRepository {

    private final UnreadableDocumentRepository unreadableDocumentRepository;

    public ReadReceiptRepository(AppMenuDataRepository menuDataRepository,
                                 ReferenceDatumRepository referenceDatumRepository,
                                 UnreadableDocumentRepository unreadableDocumentRepository,
                                 PersonRepository personRepository) {
        super(menuDataRepository, referenceDatumRepository, personRepository);
        this.unreadableDocumentRepository = unreadableDocumentRepository;
    }

    public List<ReadReceipt> findReadReceiptByExternalMessageId(List<String> externalMessageIds, Long sourceOrgId) {
        List<AppMenuData> menuDataList = findByExternalMessageId(externalMessageIds, sourceOrgId);
        List<String> messagesWithUnreadableDocument = getUnreadableDocumentsExternalMessageIds(menuDataList.map(AppMenuData::getString10));

        return menuDataList.map(md -> ImmutableReadReceipt.builder()
                .status(md.getString04())
                .externalMessageId(md.getString10())
                .hasUnreadableDocument(messagesWithUnreadableDocument.contains(md.getString10()))
                .build()
        );
    }

    public List<String> getUnreadableDocumentsExternalMessageIds(List<String> externalMessageIds) {
        if (externalMessageIds.isEmpty()) {
            return List.empty();
        }
        return unreadableDocumentRepository.findByExternalMessageIds(externalMessageIds)
                .map(DocUnreadableDocument::getExternalMessageId);
    }
}
