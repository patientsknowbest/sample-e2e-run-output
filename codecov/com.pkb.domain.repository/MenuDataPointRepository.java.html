package com.pkb.domain.repository;

import com.pkb.datamodel.entity.mapper.DateOfLastDataPointMapper;
import com.pkb.datamodel.projections.PatientDateOfLastDataPoint;
import com.pkb.domain.internal.repository.AbstractMenuDataRepository;
import com.pkb.domain.internal.repository.AppMenuDataRepository;
import com.pkb.domain.internal.repository.ReferenceDatumRepository;
import com.pkb.entities.enums.MenuDataType;
import io.vavr.collection.List;
import io.vavr.collection.Set;
import io.vavr.control.Option;
import org.apache.commons.lang3.tuple.Pair;
import org.jetbrains.annotations.NotNull;

import javax.persistence.EntityManager;
import javax.persistence.NoResultException;
import javax.persistence.PersistenceContext;
import javax.persistence.Query;
import javax.persistence.Tuple;
import javax.persistence.TypedQuery;
import java.time.Instant;

import static com.google.common.base.Preconditions.checkArgument;

public class MenuDataPointRepository extends AbstractMenuDataRepository {

    private static final String DATA_TYPE_LIST_CANNOT_BE_EMPTY = "Exclude Data Type List cannot be empty";

    private final DateOfLastDataPointMapper dateOfLastDataPointMapper;

    @PersistenceContext
    private EntityManager em;

    public MenuDataPointRepository(AppMenuDataRepository menuDataRepository,
                                   ReferenceDatumRepository referenceDatumRepository,
                                   PersonRepository personRepository,
                                   DateOfLastDataPointMapper dateOfLastDataPointMapper) {
        super(menuDataRepository, referenceDatumRepository, personRepository);
        this.dateOfLastDataPointMapper = dateOfLastDataPointMapper;
    }

    @SuppressWarnings("unchecked")
    public List<Pair<Long, Instant>> findLastDatapointDatesByPersonId(@NotNull Set<Long> personIds,
                                                                      @NotNull Set<MenuDataType> excludeTypes) {
        checkArgument(!excludeTypes.isEmpty(), DATA_TYPE_LIST_CANNOT_BE_EMPTY);

        String queryStr = " SELECT" +
                " new org.apache.commons.lang3.tuple.ImmutablePair(au.userPersonId, max(md.baseFields.persistedDate))" +
                " FROM  AppMenuData md" +
                "    JOIN CoreAccountUser au ON au.accountId = md.accountId" +
                "    WHERE md.dataType NOT IN :excludeTypes AND md.deleted=false" +
                "    AND au.userPersonId IN :personIds" +
                "    GROUP BY au.userPersonId";
        Query query =  em.createQuery(queryStr)
                .setParameter("excludeTypes", excludeTypes.toJavaSet())
                .setParameter("personIds", personIds.toJavaSet());
        return List.ofAll((java.util.List<Pair<Long, Instant>>) query.getResultList());
    }

    public Option<PatientDateOfLastDataPoint> findSingleLastDataPointDate(long personId,
            @NotNull Set<MenuDataType> excludeTypes) {
        checkArgument(!excludeTypes.isEmpty(), DATA_TYPE_LIST_CANNOT_BE_EMPTY);
        TypedQuery<Tuple> query = em.createQuery("SELECT max(md.baseFields.persistedDate) FROM AppMenuData md" +
                "    JOIN CoreAccountUser au ON au.accountId = md.accountId" +
                "    WHERE au.userPersonId = :personId AND md.dataType NOT IN :excludeTypes AND md.deleted=false", Tuple.class);
        query.setParameter("personId", personId);
        query.setParameter("excludeTypes", excludeTypes.toJavaSet());
        try {
            Tuple tuple = query.getSingleResult();
            return Option.of(dateOfLastDataPointMapper.mapToDataModelWithDate(tuple));
        } catch (NoResultException ignored) {
            return Option.none();
        }
    }

}
