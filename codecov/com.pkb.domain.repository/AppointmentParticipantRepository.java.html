package com.pkb.domain.repository;

import com.pkb.annotation.EHRField;
import com.pkb.datamodel.AccountKeys;
import com.pkb.datamodel.AppMenuDataWithSourceObjects;
import com.pkb.datamodel.AppointmentParticipant;
import com.pkb.datamodel.entity.mapper.AppointmentParticipantMapper;
import com.pkb.domain.internal.repository.AbstractMenuDataRepository;
import com.pkb.domain.internal.repository.AppMenuDataRepository;
import com.pkb.domain.internal.repository.ReferenceDatumRepository;
import com.pkb.ehrdata.AccessFilter;
import com.pkb.ehrdata.ImmutableSearchSpec;
import com.pkb.ehrdata.SearchSpec;
import com.pkb.entities.enums.MenuDataType;
import io.vavr.collection.List;
import io.vavr.collection.Map;

import static com.pkb.ehrdata.Filter.filter;

public class AppointmentParticipantRepository extends AbstractMenuDataRepository {

    private final AppointmentParticipantMapper appointmentParticipantMapper;

    public AppointmentParticipantRepository(AppMenuDataRepository menuDataRepository,
                                            ReferenceDatumRepository referenceDatumRepository,
                                            AppointmentParticipantMapper appointmentParticipantMapper,
                                            PersonRepository personRepository) {
        super(menuDataRepository, referenceDatumRepository, personRepository);
        this.appointmentParticipantMapper = appointmentParticipantMapper;
    }

    public Map<Long, List<AppointmentParticipant>> findAppointmentParticipantsByAppointmentIds(Long patientAccountId,
            AccessFilter accessFilter,
            AccountKeys accountKeys,
            List<Long> parentAppointmentIds) {

        SearchSpec searchSpec = ImmutableSearchSpec.builder()
                .addAccountId(patientAccountId)
                .addDataType(MenuDataType.appointmentRequest.name())
                .addFilter(filter(EHRField.QueryField.long01.name(), com.pkb.ehrdata.Filter.Operator.IN, parentAppointmentIds.asJava()))
                .build();

        List<Long> ids = find(searchSpec, accessFilter);
        List<AppMenuDataWithSourceObjects> menuData = findMenuDatasByIds(ids);

        return appointmentParticipantMapper.menuDatasToAppointmentParticipants(accountKeys, menuData);
    }
}
