public class AppointmentRepository extends AbstractMenuDataRepository {
    private final AppointmentMapper appointmentMapper;
    private final AccessFilterMapper accessFilterMapper;
    private final AppointmentParticipantRepository appointmentParticipantRepository;
    private final PersonMapper personMapper;

    public AppointmentRepository(AppMenuDataRepository menuDataRepository,
                                 ReferenceDatumRepository referenceDatumRepository,
                                 AppointmentMapper appointmentMapper,
                                 AccessFilterMapper accessFilterMapper,
                                 AppointmentParticipantRepository appointmentParticipantRepository,
                                 PersonRepository personRepository, PersonMapper personMapper) {
        super(menuDataRepository, referenceDatumRepository, personRepository);
        this.appointmentMapper = appointmentMapper;
        this.accessFilterMapper = accessFilterMapper;
        this.appointmentParticipantRepository = appointmentParticipantRepository;
        this.personMapper = personMapper;
    }

    @NotNull
    public CriteriaResult<Appointment> findAppointments(AppointmentQuery appointmentQuery) {
        TargetOwnerData targetOwnerData = appointmentQuery.authData().requireSingleTargetData();

        Long patientAccountId = targetOwnerData.account().getId();
        SearchSpec searchSpec = buildSearchSpec(
                patientAccountId,
                appointmentQuery.dateRestriction(),
                appointmentQuery.maxNumberOfResources());
        AccessFilter accessFilter = accessFilterMapper.buildAccessFilter(appointmentQuery.authData());

        List<Long> appointmentIds = find(searchSpec, accessFilter);
        if (appointmentIds.isEmpty()) {
            return emptyCriteriaResult();
        }

        long total = appointmentIds.size();
        if (total == appointmentQuery.maxNumberOfResources()) {
            total = count(searchSpec, accessFilter);
        }

        // TODO in GDE-879 Read relevant rows from ReferenceDatum table into a code->value map which can be passed to appointmentWithoutParticipantsMapper.
        // Their DTOBaseFields.getHospitalServiceCode() referenced as ReferenceDatum.ReferenceDataType.HOSPITAL_SERVICE is our Specialty string
        // Within an Appointment, the encrypted String "updatedSpeciality" stores the latest specialty at the time the appointment was saved.

        List<AppMenuDataWithSourceObjects> appointmentMenuData = findMenuDatasByIds(appointmentIds);

        // Retrieve account keys, checking that we have all the keys we need to decrypt the data we have loaded
        AccountKeys accountKeys = checkAccountKeys(targetOwnerData, appointmentMenuData);

        Map<Long, List<AppointmentParticipant>> participantsByAppointmentId = appointmentParticipantRepository
                .findAppointmentParticipantsByAppointmentIds(patientAccountId, accessFilter, accountKeys, appointmentIds);

        var participantPersonIds = io.vavr.collection.HashSet.ofAll(participantsByAppointmentId.values())
                .flatMap(HashSet::ofAll)
                .map(AppointmentParticipant::participantPrivateId);

        var participantsById = personRepository.findByIds(participantPersonIds)
                .map(personMapper::entityToDataModel)
                .toMap(Person::getId, identity());

        ReferenceDatumCache specialtyCache = getReferenceDatumCache(appointmentMenuData,
                ReferenceDataType.HOSPITAL_SERVICE);

        List<Appointment> appointmentsWithParticipants = appointmentMapper
                .mapToAppointments(appointmentMenuData, accountKeys, participantsByAppointmentId, participantsById, specialtyCache);

        return ImmutableCriteriaResult.<Appointment>builder().total(total).addAllPage(appointmentsWithParticipants).build();
    }

    @NotNull
    private ImmutableSearchSpec buildSearchSpec(@NotNull Long patientAccountId,
            @NotNull TemporalRangeRestriction<LocalDate> dateRestriction,
            int maxNumberOfResources) {
        ImmutableSearchSpec.Builder searchSpecBuilder = getSearchSpecBuilder(patientAccountId, MenuDataType.appointment.name());
        BoundedDateRangeRestrictions.addDateRestrictions("date01", dateRestriction, searchSpecBuilder);
        addOrderBy("date01", Direction.Asc, searchSpecBuilder);
        searchSpecBuilder.latestFilter(defaultLatestFields());
        addPageFilter(maxNumberOfResources, searchSpecBuilder);
        return searchSpecBuilder.build();
    }
}
