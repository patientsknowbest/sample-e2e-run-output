package com.pkb.fhir.analysis.process.perform;

import com.pkb.fhir.analysis.core.threading.EhrDataProcessorWorker;
import com.pkb.fhir.analysis.persistence.targetdb.TransformedEhrDataEntity;
import com.pkb.fhir.analysis.persistence.targetdb.TransformedEhrDataRepository;
import com.pkb.fhir.analysis.validate.EhrDataValidator;
import com.pkb.fhir.common.EhrDataProcessingResult;
import com.pkb.fhir.common.enums.TransformedEhrDataStatus;

import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

/**
 * This is the implementation for the validation processor. {@link EhrDataProcessorWorker}
 * We read the successfully transformed data from the target database, and validate it using the HAPI FHIR library
 * (which is more reliable than using json schema files), then write the results back to the target database.
 */
public class TransformedEhrDataValidator implements EhrDataProcessor {
    private final TransformedEhrDataRepository transformedEhrDataRepository;
    private final EhrDataValidator ehrDataValidator;

    public TransformedEhrDataValidator(TransformedEhrDataRepository transformedEhrDataRepository, EhrDataValidator ehrDataValidator) {
        this.transformedEhrDataRepository = transformedEhrDataRepository;
        this.ehrDataValidator = ehrDataValidator;
    }

    @Override
    public List<? extends EhrDataProcessingResult> process(List<Long> idsToProcess) {
        List<TransformedEhrDataEntity> toValidate = transformedEhrDataRepository.findAllByIdWithStatusIn(idsToProcess, Arrays.asList(TransformedEhrDataStatus.TRANSFORMED, TransformedEhrDataStatus.VALIDATION_FAILED));
        List<TransformedEhrDataEntity> result = validateData(toValidate);
        transformedEhrDataRepository.saveAll(result);
        return result;
    }

    private List<TransformedEhrDataEntity> validateData(List<TransformedEhrDataEntity> data) {
        return data.stream().map(ehrDataValidator::validate).collect(Collectors.toList());
    }


}
