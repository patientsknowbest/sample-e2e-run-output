package com.pkb.fhir.analysis.process.perform;

import com.pkb.entities.enums.MenuDataType;
import com.pkb.fhir.analysis.core.threading.EhrDataProcessorWorker;
import com.pkb.fhir.analysis.persistence.targetdb.TransformedEhrDataEntity;
import com.pkb.fhir.analysis.persistence.targetdb.TransformedEhrDataRepository;
import com.pkb.fhir.analysis.persistence.targetdb.repository.DecryptedEhrDataRepository;
import com.pkb.fhir.common.decrypt.DecryptedEhrData;
import com.pkb.fhir.common.enums.DecryptedEhrDataStatus;
import com.pkb.fhir.common.enums.TransformedEhrDataStatus;
import com.pkb.fhir.common.retrieve.AllergyEhrDataRetriever;
import com.pkb.fhir.common.retrieve.DataTypeBasedEhrDataRetriever;
import com.pkb.fhir.common.transform.BaseEhrDataTransformer;
import com.pkb.fhir.common.transform.TransformableDto;
import org.jetbrains.annotations.NotNull;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.List;
import java.util.Map;
import java.util.function.Function;
import java.util.stream.Collectors;

/**
 * This is the implementation for the transformation processor. {@link EhrDataProcessorWorker}
 * We read the successfully decrypted data from the target database,
 * collect extra data (not encrypted fields from menu_data) from the source database {@link AllergyEhrDataRetriever},
 * then transform the XML + extra data to a json FHIR stucture and write it to the target database.
 */
public class AnalysisDecryptedEhrDataTransformer implements EhrDataProcessor {
    private final TransformedEhrDataRepository transformedEhrDataRepository;
    private final DataTypeBasedEhrDataRetriever ehrDataRetriever;
    private final DecryptedEhrDataRepository ehrDataRepository;
    private final Map<MenuDataType, BaseEhrDataTransformer> transformerMap;

    public AnalysisDecryptedEhrDataTransformer(TransformedEhrDataRepository transformedEhrDataRepository,
                                               DataTypeBasedEhrDataRetriever ehrDataRetriever,
                                               DecryptedEhrDataRepository ehrDataRepository,
                                               Map<MenuDataType, BaseEhrDataTransformer> transformerMap) {
        this.transformedEhrDataRepository = transformedEhrDataRepository;
        this.ehrDataRetriever = ehrDataRetriever;
        this.ehrDataRepository = ehrDataRepository;
        this.transformerMap = transformerMap;
    }

    @Override
    public List<TransformedEhrDataEntity> process(List<Long> idsToProcess) {
        List<TransformedEhrDataEntity> result = Collections.emptyList();

        Map<Long, DecryptedEhrData> successfullyDecryptedEhrData = ehrDataRepository.findAllById(idsToProcess, Collections.singletonList(DecryptedEhrDataStatus.SUCCESS))
                .stream().map(decryptedDataEntity ->
                        new DecryptedEhrData(decryptedDataEntity.getId(), decryptedDataEntity.getContent(), decryptedDataEntity.getStatus(), decryptedDataEntity.getDataType()))
                .collect(Collectors.toMap(DecryptedEhrData::getId, Function.identity()));

        Map<Long, ? extends TransformableDto> foundEhrData =
                ehrDataRetriever.retrieveEhrDataDto(successfullyDecryptedEhrData)
                        .stream()
                        .collect(Collectors.toMap(TransformableDto::getId, Function.identity()));

        if (!foundEhrData.isEmpty()) {
            MenuDataType menuDataType = foundEhrData.values().stream().findAny().get().getMenuDataType();

            result = transformData(foundEhrData.values());

            List<Long> failedToDecryptIds = collectMissingFromSubset(idsToProcess, successfullyDecryptedEhrData.keySet());
            result.addAll(createFailedTransformations(menuDataType, failedToDecryptIds, "Failed to decrypt"));

            List<Long> decryptedEhrDataDeletedFromOriginalSource = collectMissingFromSubset(successfullyDecryptedEhrData.keySet(), foundEhrData.keySet());
            result.addAll(createFailedTransformations(menuDataType, decryptedEhrDataDeletedFromOriginalSource, "Could not find in original source (menu_data)"));

            transformedEhrDataRepository.saveAll(result);
        }
        return result;
    }

    @NotNull
    private List<TransformedEhrDataEntity> createFailedTransformations(MenuDataType menuDataType, List<Long> failedToDecryptIds, String s) {
        return failedToDecryptIds.stream().map(
                id -> createFailedToTransformEhrDataEntity(menuDataType, id, s)
        ).collect(Collectors.toList());
    }

    @NotNull
    private TransformedEhrDataEntity createFailedToTransformEhrDataEntity(MenuDataType menuDataType, Long id, String errorMsg) {
        TransformedEhrDataEntity missingEntity = new TransformedEhrDataEntity();
        missingEntity.setError(errorMsg);
        missingEntity.setStatus(TransformedEhrDataStatus.TRANSFORM_FAILED);
        missingEntity.setDataType(menuDataType);
        missingEntity.setId(id);
        return missingEntity;
    }

    @NotNull
    private List<Long> collectMissingFromSubset(Collection<Long> originalDataset, Collection<Long> subset) {
        List<Long> toRemoveFrom = new ArrayList<>(originalDataset);
        toRemoveFrom.removeAll(subset);
        return toRemoveFrom;
    }

    private List<TransformedEhrDataEntity> transformData(Collection<? extends TransformableDto> data) {
        return data.stream().map(this::transform).collect(Collectors.toList());
    }

    public TransformedEhrDataEntity transform(TransformableDto item) {
        TransformedEhrDataEntity result = new TransformedEhrDataEntity();
        result.setId(item.getId());
        result.setDataType(item.getMenuDataType());
        try {
            String transformedContent = transformerMap.get(item.getMenuDataType()).transformToFhirJson(item);
            result.setContent(transformedContent);
            result.setStatus(TransformedEhrDataStatus.TRANSFORMED);
        } catch (Exception e) {
            result.setStatus(TransformedEhrDataStatus.TRANSFORM_FAILED);
            result.setError(e.getMessage());
        }
        return result;
    }
}
