package com.pkb.fhir.analysis.process.perform;

import com.pkb.fhir.analysis.core.threading.EhrDataProcessorWorker;
import com.pkb.fhir.analysis.storage.EhrDataStorage;
import com.pkb.fhir.common.decrypt.BaseEncryptedEhrDataDecryptor;
import com.pkb.fhir.common.decrypt.DecryptedEhrData;
import com.pkb.fhir.common.decrypt.RetryingEhrDataDecryptor;
import com.pkb.fhir.common.repository.MenuDataRepository;

import java.util.List;

/**
 * This is the implementation for the decryption processor. {@link EhrDataProcessorWorker}
 * We read from menu_data and call into KMS to get the account keys (if available) to perform this task.
 * If the key for the account owning the data is returned, we can decrypt the data.
 * We write the decrypted content (xml) or the error message (in case of failed processing) with status into the target database.
 */
public class AnalysisEncryptedEhrDataDecryptor extends BaseEncryptedEhrDataDecryptor implements EhrDataProcessor {
    private final EhrDataStorage ehrDataStorage;

    public AnalysisEncryptedEhrDataDecryptor(MenuDataRepository menuDataRepository, RetryingEhrDataDecryptor ehrDataDecryptor, EhrDataStorage ehrDataStorage) {
        super(menuDataRepository, ehrDataDecryptor);
        this.ehrDataStorage = ehrDataStorage;
    }

    @Override
    public List<DecryptedEhrData> process(List<Long> idsToProcess) {
        List<DecryptedEhrData> ehrData = decryptData(menuDataRepository.findAllEncryptedDataDtosById(idsToProcess));
        ehrDataStorage.storeDecryptedEhrData(ehrData);
        return ehrData;
    }
}
