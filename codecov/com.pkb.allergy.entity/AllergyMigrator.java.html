package com.pkb.allergy.entity;

import com.pkb.annotation.EHRVersionMigrator;

/**
 * version 1 moves authorLdapUID to sourcePersonId and reactions to List<String>
 * version 2 migrates to coded-data structures
 */
public class AllergyMigrator extends EHRMigratorBase {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private static final Map<String, AllergySeverity> ENGLISH_SEVERITIES = ImmutableMap.<String, AllergySeverity>builder()
            .put("Mild", AllergySeverity.MILD)
            .put("Mild to moderate", AllergySeverity.MILD_TO_MODERATE)
            .put("Moderate", AllergySeverity.MODERATE)
            .put("Moderate to severe", AllergySeverity.MODERATE_TO_SEVERE)
            .put("Severe", AllergySeverity.SEVERE)
            .put("Fatal", AllergySeverity.FATAL).build();

    private static final Map<String, AllergyReaction> ENGLISH_REACTIONS = ImmutableMap.<String, AllergyReaction>builder()
            .put("Anaphylaxis", AllergyReaction.ANAPHYLAXIS)
            .put("Diarrhea", AllergyReaction.DIARRHEA)
            .put("Difficulty in breathing", AllergyReaction.DIFFICULTY_IN_BREATHING)
            .put("Fainting", AllergyReaction.FAINTING)
            .put("Headache", AllergyReaction.HEADACHE)
            .put("Hives", AllergyReaction.HIVES)
            .put("Itching eyes", AllergyReaction.ITCHING_EYES)
            .put("Itching skin (without rash or hives)", AllergyReaction.ITCHING_SKIN)
            .put("Nausea", AllergyReaction.NAUSEA)
            .put("Pain or stuffiness in ears", AllergyReaction.PAIN_OR_STUFFINESS_IN_EARS)
            .put("Rash", AllergyReaction.RASH)
            .put("Runny nose", AllergyReaction.RUNNY_NOSE)
            .put("Shock", AllergyReaction.SHOCK)
            .put("Sneezing", AllergyReaction.SNEEZING)
            .put("Sore throat", AllergyReaction.SORE_THROAT)
            .put("Swelling of lips", AllergyReaction.SWELLING_OF_LIPS)
            .put("Swelling of ankles", AllergyReaction.SWELLING_OF_ANKLES)
            .put("Tightness in throat", AllergyReaction.TIGHTNESS_IN_THROAT)
            .put("Vomiting", AllergyReaction.VOMITING)
            .put("Wheezing", AllergyReaction.WHEEZING).build();


    private static final Migrators MIGRATORS = Migrators.of(
            EHRVersionMigrator.of(AllergyMigrator::migrateSourcePersonAndReaction, 1),
            EHRVersionMigrator.of(AllergyMigrator::migrateStringFieldsToCodeableConcepts, 2),
            PrivateIdMigrator.migratorFor(Allergy.class, "disorderCodingReceivedId",  "string05", 3),
            PrivateIdMigrator.migratorFor(Allergy.class, "allergenCodingReceivedId", "string04", 4)
    );

    @Override
    public Migrators getMigrators() {
        return MIGRATORS;
    }

    @SuppressWarnings("NewExceptionWithoutArguments")
    private static void migrateSourcePersonAndReaction(EHRData ehr) {
        if (ehr.getSourcePersonId() == null) {
            Optional.ofNullable(ehr.getEncryptedField("authorLdapUID"))
                    .map(val -> (String) val)
                    .map(Long::parseLong)
                    .ifPresentOrElse(ehr::setSourcePersonId, () -> {
                        LOGGER.error("Can't fix missing sourcePersonId for allergy {} because authorLdapUID is null ", ehr, FrameFilter.filter(new Exception().fillInStackTrace()));
                    });
        }

        Object reactionsObject = ehr.getEncryptedField("reactions");

        if (reactionsObject == null) {
            String fromReaction = (String) ehr.getEncryptedField("reaction");
            if (StringUtils.isNotBlank(fromReaction)) {
                List<String> newReactionList = new ArrayList<>();
                newReactionList.add(fromReaction);
                ehr.setEncryptedField("reactions", newReactionList);
            }
        }
    }

    private static void migrateStringFieldsToCodeableConcepts(EHRData ehr) {
        // coded-data migration
        // 3 fields changing save format: allergen, reactions, severity
        migrateStringFieldToCodeableConcept(ehr, "allergen");
        migrateStringFieldToCodeableConcept(ehr, "severity");
        migrateStringListFieldToCodeableConceptList(ehr, "reactions");
    }

    /**
     * current encrypted value for fieldName must be String or null
     * convert to CodeableConcept with string value as display text
     */
    private static void migrateStringFieldToCodeableConcept(EHRData ehr, String fieldName) {
        Object valueObj = ehr.getEncryptedField(fieldName);
        if (valueObj == null) {
            return;
        } else if (!(valueObj instanceof String)) {
            throw new IllegalStateException("Encrypted value for " + fieldName + ": expected String, found " + valueObj.getClass());
        }

        CodeableConcept cc = new CodeableConcept();
        cc.setDisplayText((String) valueObj);
        addCoding(cc, fieldName);
        ehr.setEncryptedField(fieldName, toMap(cc));
    }

    /**
     * current encrypted value for fieldName must be List<String> or null
     * convert to List<CodeableConcept> with string value as display text
     */
    private static void migrateStringListFieldToCodeableConceptList(EHRData ehr, String fieldName) {
        Object valueObj = ehr.getEncryptedField(fieldName);
        if (valueObj == null) {
            return;
        } else if (valueObj instanceof String) {
            // quick fix for real world data seen...
            valueObj = Collections.singletonList(valueObj);
        } else if (!(valueObj instanceof List)) {
            throw new IllegalStateException("Encrypted value for " + fieldName + ": expected List, found " + valueObj.getClass());
        }

        List<CodeableConcept> converted = ((List<String>) valueObj).stream()
                .map(text -> new CodeableConcept.Builder(text).build())
                .collect(Collectors.toList());

        for (CodeableConcept cc : converted) {
            addCoding(cc, fieldName);
        }
        List<Object> maps = converted.stream().map(AllergyMigrator::toMap).collect(Collectors.toList());
        ehr.setEncryptedField(fieldName, maps);
    }

    private static void addCoding(CodeableConcept cc, String fieldName) {
        Coded coded = null;
        if (fieldName.equals(Allergy.ALLERGY_SEVERITY)) {
            coded = ENGLISH_SEVERITIES.get(cc.getDisplayText());

        } else if (fieldName.equals(Allergy.ALLERGY_REACTIONS)) {
            coded = ENGLISH_REACTIONS.get(cc.getDisplayText());
        }
        if (coded != null) {
            Coding coding = new Coding.Builder(coded.getCode(), coded.getCodeSystemName())
                    .displayText(cc.getDisplayText())
                    .build();
            cc.getCodings().add(coding);
        }
    }

    private static Object toMap(CodeableConcept cc) {
        // the nested CodeableConcept mappings are all the same (once we get to one...)
        DTOMapping<Allergy> allergyMapping = DTOMapping.get(Allergy.class);
        DTOMapping<?> dtoMapping = allergyMapping.getNestedDtoMapping(Allergy.ALLERGY_SEVERITY);

        return ValueMappingUtil.prepValueForEHRData(cc, dtoMapping);
    }
}
