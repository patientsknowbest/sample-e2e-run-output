    private Long id;
    private String displayName;
    private String code;
    private String codeSystem;
    private String codeSystemName;

    AllergySeverity(long id, String displayName, String code, String codeSystem, String codeSystemName) {
        this.id = id;
        this.displayName = displayName;
        this.code = code;
        this.codeSystem = codeSystem;

    @Override public Long getId() {

    @Override public String getCodeSystemName() {
        return codeSystemName;
    }

    @Nullable
    public AllergySeverity getById(Long id) {
        for (AllergySeverity allergySeverity : values()) {
            if (allergySeverity.id.equals(id)) {
                return allergySeverity;
            }
        }
        return null;
    }

    @Nullable
    public AllergySeverity getByName(String name) {
        for (AllergySeverity allergySeverity : values()) {
            if (allergySeverity.displayName.equals(name)) {
                return allergySeverity;
            }
        }
        return null;
    }
