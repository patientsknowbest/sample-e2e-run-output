package com.pkb.allergy.entity;

import com.pkb.annotation.EHRField;
import com.pkb.annotation.EHRField.QueryField;
import com.pkb.annotation.EHRMigration;
import com.pkb.annotation.EncryptedFieldMigration;
import com.pkb.annotation.Migrate;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;
import com.pkb.app.interfaces.DataPointWithStartDate;
import com.pkb.app.interfaces.EncryptedDataPoint;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.coding.entity.CodeableConcept;
import com.pkb.entities.enums.MenuDataType;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.Nullable;

import java.time.Instant;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.Date;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.stream.Collectors;

import static com.pkb.annotation.EHRField.ValueDTOType.NESTED;

/**
 * EMIS ES provides allergies as a coded disorder -- e.g., "Adverse reaction to ___"
 * HL7 v2 provides a coded allergen and severity, plus uncoded reactions.
 * Our webapp stores an uncoded allergen, and optionally-coded severity and reaction (pick lists plus option of free text)
 */
@EncryptedFieldMigration({
        @Migrate(fromEncrypted = "authorLdapUID", toQueryField = QueryField.sourcePersonId),
        @Migrate(fromEncrypted = "reaction", toEncrypted = "reactions")
})
@EHRMigration(migratorClass = AllergyMigrator.class)
public class Allergy implements IBaseDTO, EncryptedDataPoint, DataPointWithStartDate {

    private static final long serialVersionUID = 3L;

    public static final MenuDataType MS_PATH = MenuDataType.allergy;

    public static final String ALLERGY_ID = "id";
    public static final String ALLERGY_ALLERGEN = "allergen";
    public static final String ALLERGY_ONSET_OF_SYMPTOMS = "onsetOfSymptoms";
    public static final String ALLERGY_REACTIONS = "reactions";
    public static final String ALLERGY_SEVERITY = "severity";
    public static final String ALLERGY_PATIENT_ACCOUNT_ID = "patientAccountId";
    public static final String ALLERGY_PATIENT_LDAP_ID = "patientLdapUID";
    public static final String ALLERGY_AUTHOR_LDAP_ID = "authorLdapUID";
    public static final String ALLERGY_STATUS = "status";

    public static final String UPLOADED_DATA_ID = "uploadedDataId";

    public static final String ALLERGEN_CODE = "(#Allergen#)";
    public static final String SEVERITY_CODE = "(#Severity#)";
    public static final String REACTION_CODE = "(#Reaction#)";

    public enum AllergyStatus {
        ACTIVE, INACTIVE, RESOLVED
    }

    @EHRField(queryField = QueryField.id)
    private Long id;

    @EHRField(valueDtoType = NESTED)
    private CodeableConcept disorder;

    @EHRField(queryField = QueryField.long01)
    private Long disorderCodingReceivedId;

    @EHRField(valueDtoType = NESTED)
    private CodeableConcept allergen;

    @EHRField(queryField = QueryField.long02)
    private Long allergenCodingReceivedId;

    @EHRField
    private List<CodeableConcept> reactions = new ArrayList<>();

    @EHRField
    private CodeableConcept severity;

    @EHRField(queryField = QueryField.date01)
    private Date onsetOfSymptoms;

    @EHRField(queryField = QueryField.date02)
    private Date dateAsserted;

    @EHRField(queryField = QueryField.accountId)
    private Long patientAccountId;

    @EHRField
    private String authorLdapUID; // now auto-migrated into sourcedetails

    @EHRField
    private Date endOfSymptoms;

    /**
     * This indicates whether the patient is allergic to the allergy being defined by this {@code Allergy}. Default to TRUE,
     * and it's likely we will remove this option soon (it was added for MiApp to capture a confirmation that the patient was
     * checked for an allergy and found to NOT be allergic... we'll capture this some other way when we need it).
     */
    @EHRField(queryField = QueryField.boolean01)
    private Boolean patientIsAllergic = Boolean.TRUE;

    // Audit trail

    @EHRField
    private AllergyStatus status;

    private DTOBaseFields baseFields = new DTOBaseFields();

    private SourceDetails source;

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected Allergy() {
        this.source = new SourceDetails();
    }

    public Allergy(SourceDetails source) {
        this.source = source;
    }

    @Override
    public CodeableConcept code() {
        return allergen;
    }

    @Nullable
    @Override
    public Instant startDate() {
        return onsetOfSymptoms == null ? null : onsetOfSymptoms.toInstant();
    }

    public Date getOnsetOfSymptoms() {
        return onsetOfSymptoms;
    }

    public void setOnsetOfSymptoms(Date onsetOfSymptoms) {
        this.onsetOfSymptoms = onsetOfSymptoms;
    }

    @Override
    public MenuDataType getDataType() {
        return MS_PATH;
    }

    @Override
    public Long getPatientAccountId() {
        return patientAccountId;
    }

    public void setPatientAccountId(Long patientAccountId) {
        this.patientAccountId = patientAccountId;
    }

    public String getAuthorLdapUID() {
        return authorLdapUID;
    }

    public AllergyStatus getStatus() {
        return status;
    }

    public void setStatus(AllergyStatus status) {
        this.status = status;
    }

    public void setAuthorLdapUID(String authorLdapUID) {
        this.authorLdapUID = authorLdapUID;
    }

    @Override
    public Long getId() {
        return id;
    }

    @Override
    public void setId(Long id) {
        this.id = id;
    }

    @Override
    public SourceDetails getSource() {
        return source;
    }

    @Override
    public DTOBaseFields getBaseFields() {
        return this.baseFields;
    }

    // Debug
    @Override
    public String toString() {
        return "Allergy:" + id + "/" + baseFields.getUniqueId() + ",patientAccountId=" + patientAccountId +
                ",authorLdapUID=" + authorLdapUID + ",source=" + source + ",baseFields=" + baseFields + ")";
    }

    public Boolean getPatientIsAllergic() {
        return patientIsAllergic;
    }

    public void setPatientIsAllergic(Boolean patientIsAllergic) {
        this.patientIsAllergic = patientIsAllergic;
    }

    public CodeableConcept getDisorder() {
        return disorder;
    }

    public void setDisorder(CodeableConcept disorder) {
        this.disorder = disorder;
    }

    public Long getDisorderCodingReceivedId() {
        return disorderCodingReceivedId;
    }

    public void setDisorderCodingReceivedId(Long disorderCodingReceivedId) {
        this.disorderCodingReceivedId = disorderCodingReceivedId;
    }

    public CodeableConcept getAllergen() {
        return allergen;
    }

    public void setAllergen(CodeableConcept allergen) {
        this.allergen = allergen;
    }

    public Long getAllergenCodingReceivedId() {
        return allergenCodingReceivedId;
    }

    public void setAllergenCodingReceivedId(Long allergenCodingReceivedId) {
        this.allergenCodingReceivedId = allergenCodingReceivedId;
    }

    public List<CodeableConcept> getReactions() {
        if (reactions == null) {
            reactions = new ArrayList<>();
        }
        return reactions;
    }

    public void setReactions(List<CodeableConcept> reactions) {
        this.reactions = reactions != null ? reactions : new ArrayList<>();
    }

    public CodeableConcept getSeverity() {
        return severity;
    }

    public void setSeverity(CodeableConcept severity) {
        this.severity = severity;
    }

    public Date getEndOfSymptoms() {
        return endOfSymptoms;
    }

    public void setEndOfSymptoms(Date endOfSymptoms) {
        this.endOfSymptoms = endOfSymptoms;
    }

    public String getReactionsDisplayText() {
        if (reactions != null) {
            return reactions.stream()
                    .map(CodeableConcept::getTextForDisplay)
                    .collect(Collectors.joining(","));
        }
        return "";
    }

    public String getReactionsCodes() {
        if (reactions != null) {
            return reactions.stream()
                    .map(CodeableConcept::getCodesForDisplay)
                    .collect(Collectors.joining(","));
        }
        return "";
    }

    public boolean hasDisorderData() {
        return disorder != null &&
                StringUtils.isNotBlank(disorder.getTextForDisplay());
    }

    public String getClinicalCodes() {

        LinkedHashMap<String, List<CodeableConcept>> map = new LinkedHashMap<>();
        map.put("", Arrays.asList(disorder));
        map.put(" " + ALLERGEN_CODE, Arrays.asList(allergen));
        map.put(" " + SEVERITY_CODE, Arrays.asList(severity));
        map.put(" " + REACTION_CODE, reactions);

        return map.entrySet().stream().flatMap(e -> e.getValue()
                .stream()
                .filter(p -> p != null && !p.getSimpleCodesForDisplay().isEmpty())
                .map(v -> v.getSimpleCodesForDisplay() + e.getKey()))
                .collect(Collectors.joining(", "));
    }

    public Date getDateAsserted() {
        return dateAsserted;
    }

    public void setDateAsserted(Date dateAsserted) {
        this.dateAsserted = dateAsserted;
    }

    public void setSource(SourceDetails source) {
        this.source = source;
    }
}
