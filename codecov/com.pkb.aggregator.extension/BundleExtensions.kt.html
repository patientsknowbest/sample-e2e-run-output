package com.pkb.aggregator.extension

import com.pkb.aggregator.Dstu3Bundle
import com.pkb.aggregator.R4Bundle
import com.pkb.aggregator.R4BundleEntryComponent
import com.pkb.aggregator.R4DomainResource
import com.pkb.aggregator.R4Resource
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import io.micrometer.core.instrument.Metrics.timer
import org.hl7.fhir.convertors.factory.VersionConvertorFactory_30_40
import org.hl7.fhir.instance.model.api.IBaseBundle
import org.hl7.fhir.r4.model.Bundle

fun IBaseBundle.getResources(): List<R4DomainResource> =
    when (this) {
        is R4Bundle -> entry.map { it.resource  }.asListOfType()
        else -> listOf()
    }

fun R4Bundle.addTransactionEntry(resource: R4Resource): R4BundleEntryComponent =
    this.addEntry().apply {
        this.resource = resource
        this.request.apply {
            url = resource.resourceType.name
            method = Bundle.HTTPVerb.POST
        }
    }

fun Dstu3Bundle.toR4(): R4Bundle = timer("r4.mapping", "type", "bundle").recordCallable {
    VersionConvertorFactory_30_40.convertResource(this) as R4Bundle
}!!

// Based on https://kotlinlang.org/docs/typecasts.html#unchecked-casts
inline fun <reified T> R4Bundle.singleEntry(): T? {

    if (this.entry.isEmpty()) {
        return null
    }

    if (this.entry.size > 1) {
        throw RuntimeException("Bundle contains multiple entries")
    }

    return when (val resource = this.entry.first().resource) {
        null -> null
        is T -> resource
        else -> throw UnexpectedTypeException(resource)
    }
}
