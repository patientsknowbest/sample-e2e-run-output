package com.pkb.repository;

import org.jooq.DSLContext;
import org.jooq.Query;
import org.jooq.conf.Settings;

import java.util.List;
import java.util.function.Consumer;

import static org.jooq.conf.RenderNameCase.AS_IS;
import static org.jooq.conf.RenderQuotedNames.EXPLICIT_DEFAULT_UNQUOTED;

public interface JooqSupport {

    static Settings jooqSettings() {
        Settings jooqSettings = new Settings();
        jooqSettings.setRenderNameCase(AS_IS);
        jooqSettings.setRenderQuotedNames(EXPLICIT_DEFAULT_UNQUOTED);
        return jooqSettings;
    }

    DSLContext context();

    <E> List<E> nativeListQuery(Query query, Class<E> resultClass);

    List nativeListQuery(Query query);

    List nativeListQuery(Query query, Consumer<javax.persistence.Query> queryPostProcessor);

    <E> E nativeQuery(Query query);

}
