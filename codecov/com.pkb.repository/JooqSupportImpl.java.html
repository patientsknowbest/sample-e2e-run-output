package com.pkb.repository;

import org.jooq.DSLContext;
import org.jooq.Query;
import org.jooq.impl.DSL;
import org.springframework.beans.factory.annotation.Autowired;

import javax.persistence.EntityManager;
import java.util.List;
import java.util.Optional;
import java.util.function.Consumer;

import static com.pkb.repository.JooqSupport.jooqSettings;
import static org.jooq.SQLDialect.POSTGRES;

public class JooqSupportImpl implements JooqSupport {

    @Autowired
    private EntityManager entityManager;


    @Override
    public DSLContext context() {
        return DSL.using(POSTGRES, jooqSettings());
    }

    @Override
    @SuppressWarnings("unchecked")
    public <T> List<T> nativeListQuery(Query query, Class<T> resultClass) {
        return convertJooqQueryToJpa(query, resultClass).getResultList();
    }

    @Override
    @SuppressWarnings("unchecked")
    public <T> T nativeQuery(Query query) {
        return (T) convertJooqQueryToJpa(query, (Class<T>) null).getSingleResult();
    }

    @Override
    public List nativeListQuery(Query query) {
        return convertJooqQueryToJpa(query).getResultList();
    }

    @Override
    @SuppressWarnings("unchecked")
    public List nativeListQuery(Query query, Consumer<javax.persistence.Query> queryPostProcessor) {
        javax.persistence.Query jpaQuery = convertJooqQueryToJpa(query);
        queryPostProcessor.accept(jpaQuery);
        return jpaQuery.getResultList();
    }

    private javax.persistence.Query convertJooqQueryToJpa(Query query) {
        return convertJooqQueryToJpa(query, Optional.empty());
    }

    private <T> javax.persistence.Query convertJooqQueryToJpa(Query query, Class<T> resultClass) {
        return convertJooqQueryToJpa(query, Optional.ofNullable(resultClass));
    }

    private <T> javax.persistence.Query convertJooqQueryToJpa(Query query, Optional<Class<T>> resultClass) {

        // Extract the SQL statement from the jOOQ query
        javax.persistence.Query jpaQuery = resultClass
                .map(clazz -> entityManager.createNativeQuery(query.getSQL(), clazz))
                .orElse(entityManager.createNativeQuery(query.getSQL()));

        // Extract the bind values from the jOOQ query
        List<Object> values = query.getBindValues();
        for (int i = 0; i < values.size(); i++) {
            Object value = values.get(i);
            jpaQuery.setParameter(i + 1, value);
        }

        return jpaQuery;
    }


}
