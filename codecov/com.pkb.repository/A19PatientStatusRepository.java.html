package com.pkb.repository;

import com.pkb.entities.ids.A19PK;
import com.pkb.entities.pub.PublicA19PatientStatus;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;
import org.jooq.Condition;
import org.jooq.Record;
import org.jooq.SelectQuery;
import org.jooq.SelectSeekStep2;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;

import java.sql.Timestamp;
import java.time.Instant;
import java.util.List;

import static org.jooq.impl.DSL.field;
import static org.jooq.impl.DSL.table;

public interface A19PatientStatusRepository extends JpaRepository<PublicA19PatientStatus, A19PK>, JooqSupport {

    @NotNull
    default List<PublicA19PatientStatus> getPatientStatuses(long orgId, @Nullable Instant queryStart, int pageSize, @Nullable String registeredStatus, @Nullable Long lastSeenId) {

        SelectQuery<Record> jooqQuery = null;
        Condition conditon = field("a19_patientstatus.org_id").eq(orgId);
        if (registeredStatus != null) {
            conditon = conditon.and(field("a19_patientstatus.registration_status").eq(registeredStatus));
        }

        if (queryStart != null && lastSeenId == null) {
            conditon = conditon.and(field("a19_patientstatus.status_date").ge(Timestamp.from(queryStart)));
        }
        SelectSeekStep2 jq = context().select()
                .from(table("a19_patientstatus"))
                .where(conditon)
                .orderBy(field("a19_patientstatus.status_date").asc(), field("a19_patientstatus.person_id").asc());
        if (queryStart != null && lastSeenId != null) {
            jooqQuery = jq.seek(Timestamp.from(queryStart), lastSeenId)
                    .limit(pageSize)
                    .getQuery();
        } else {
            jooqQuery = jq.limit(pageSize)
                    .getQuery();
        }

        return nativeListQuery(jooqQuery, PublicA19PatientStatus.class);
    }

    @Query("SELECT COUNT(ps) FROM PublicA19PatientStatus ps " +
            "WHERE ps.statusDate = :lastSeenTimestamp " +
            "AND ps.personId = :lastSeenId")
    long countByLastSeenValues(@Param("lastSeenTimestamp") Instant lastSeenTimestamp,
                               @Param("lastSeenId") Long lastSeenId);

    @Query("SELECT COUNT(ps) FROM PublicA19PatientStatus ps " +
            "WHERE ps.statusDate = '-infinity' " +
            "AND ps.personId = :lastSeenId")
    long countByLastSeenValuesWithNegativeInfinityTimestamp(@Param("lastSeenId") Long lastSeenId);
}
