package com.pkb.repository;

import com.pkb.entities.enums.OrgNetworkMemberStatus;
import com.pkb.entities.pub.PublicOrgNetwork;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;

import java.util.List;
import java.util.Optional;
import java.util.Set;

public interface OrgNetworkRepository extends JpaRepository<PublicOrgNetwork, Long> {

    @Query(
            "SELECT orgNetwork FROM PublicOrgNetwork orgNetwork " +
                    "INNER JOIN PublicOrgNetworkMember onm ON onm.orgNetwork = orgNetwork " +
                    "WHERE onm.memberStatus IN :statuses AND onm.orgId = :memberOrgId " +
                    "ORDER BY orgNetwork.createdDate"

    )
    List<PublicOrgNetwork> findNetworksForMemberWithStatuses(@Param("memberOrgId") long memberOrgId, @Param("statuses") Set<OrgNetworkMemberStatus> statuses);

    default Optional<PublicOrgNetwork> findFirstNetworkForMemberWithStatuses(long memberOrgId, Set<OrgNetworkMemberStatus> statuses) {
        return findNetworksForMemberWithStatuses(memberOrgId, statuses).stream().findFirst();
    }
}
