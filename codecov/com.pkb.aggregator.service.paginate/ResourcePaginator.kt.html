package com.pkb.aggregator.service.paginate

import com.google.common.annotations.VisibleForTesting
import com.pkb.aggregator.Dstu3Bundle
import com.pkb.aggregator.Dstu3DomainResource
import com.pkb.aggregator.R4Bundle
import com.pkb.aggregator.R4DomainResource
import com.pkb.aggregator.camel.routing.urlEncode
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.service.exception.UnexpectedTypeException
import com.pkb.aggregator.service.tagger.ResourceTagger
import org.apache.http.client.utils.URLEncodedUtils
import org.hl7.fhir.dstu3.model.Bundle.LINK_NEXT
import org.hl7.fhir.instance.model.api.IBaseBundle
import org.hl7.fhir.instance.model.api.IDomainResource
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import org.springframework.web.util.UriComponentsBuilder
import java.lang.invoke.MethodHandles

/**
 * Processes a bundle of resources before aggregation takes place
 *
 * Primary responsibilities are to filter out unsupported types (currently only dstu3 resources are accepted), identify
 * the upstream source of the data, and append an extension to the resource so that the aggregation process can track
 * what came from where. Also if the bundle is one of multiple pages of data, the link for the next page is detected
 * and returned.
 */
@Service
class ResourcePaginator(
    private val taggers: List<ResourceTagger>,
) {
    private val log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    fun processBundle(bundle: IBaseBundle): PaginatedBundleProcessingResult = when (bundle) {
        is Dstu3Bundle -> processFhirDstu3Bundle(bundle)
        is R4Bundle -> processFhirR4Bundle(bundle)
        else -> throw UnexpectedTypeException(bundle)
    }

    fun processResource(resource: IDomainResource, sourceId: String): PaginatedBundleProcessingResult {
        val resources = listOf(resource)
        taggers.forEach { it.tag(resources) }
        log.debug("Tagged resource {}:{}", sourceId, resource.simpleResourceId())
        return PaginatedBundleProcessingResult(resources, null)
    }

    private fun processFhirDstu3Bundle(bundle: Dstu3Bundle): PaginatedBundleProcessingResult {

        if (bundle.entry.isEmpty()) return PaginatedBundleProcessingResult.empty()

        val fullUrl = bundle.entry.first().fullUrl

        // Extract DSTU3 resources and tag them with the source extension
        val resources = bundle.entry.mapNotNull { it.resource as? Dstu3DomainResource }
        taggers.forEach { it.tag(resources) }

        // Set up next page
        val nextLink = parseNextLink(fullUrl, bundle.getLink(LINK_NEXT)?.url)

        // Done
        return PaginatedBundleProcessingResult(resources, nextLink)
    }

    private fun processFhirR4Bundle(bundle: R4Bundle): PaginatedBundleProcessingResult {

        if (bundle.entry.isEmpty()) return PaginatedBundleProcessingResult.empty()

        val fullUrl = bundle.entry.first().fullUrl

        // Extract R4 resources and tag them with the source extension
        val resources = bundle.entry.mapNotNull { it.resource as? R4DomainResource }
        taggers.forEach { it.tag(resources) }

        // Set up next page
        val nextLink = parseNextLink(fullUrl, bundle.getLink(LINK_NEXT)?.url)

        // Done
        return PaginatedBundleProcessingResult(resources, nextLink)
    }

    @VisibleForTesting
    internal fun parseNextLink(fullUrl: String, nextLink: String?): String? =
        nextLink
            ?.split("?")
            ?.let {
                val builder = UriComponentsBuilder.fromHttpUrl(fullUrl).replacePath(it[0])
                URLEncodedUtils.parse(it[1], Charsets.UTF_8).forEach {
                    builder.queryParam(it.name, urlEncode(it.value))
                }
                builder.build().toUriString()
            }
}
