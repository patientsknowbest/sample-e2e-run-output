package com.pkb.service.dataupload.emis.esv2.model;

import com.fasterxml.jackson.annotation.JsonFormat;
import com.fasterxml.jackson.annotation.JsonIgnore;
import com.fasterxml.jackson.annotation.JsonProperty;
import com.fasterxml.jackson.annotation.JsonPropertyOrder;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import org.immutables.value.Value;

import java.time.LocalDate;
import java.time.LocalTime;
import java.util.Optional;
import java.util.OptionalDouble;
import java.util.UUID;

@Value.Immutable
@Value.Modifiable
@JsonDeserialize(as = ImmutableCareRecordObservation.class)
@JsonPropertyOrder(value = {"ObservationGuid", "PatientGuid", "OrganisationGuid", "EffectiveDate", "EffectiveDatePrecision", "EnteredDate",
        "EnteredTime", "ClinicianUserInRoleGuid", "EnteredByUserInRoleGuid", "ParentObservationGuid", "CodeId", "ProblemGuid",
        "AssociatedText", "ConsultationGuid", "Value", "NumericUnit", "ObservationType", "NumericRangeLow", "NumericRangeHigh",
        "DocumentGuid", "Deleted", "IsConfidential", "ProcessingId"})
public interface CareRecordObservation extends EmisPatientDataPoint<CareRecordObservation>, EmisEsEnteredDateTimeSupport {

    @Override
    @JsonIgnore
    default boolean isHandleAsDeleted() {
        return EmisPatientDataPoint.super.isHandleAsDeleted()
                || isConfidential().orElse(false);
    }

    @Override
    @JsonIgnore
    default String patientSpecificGuid() {
        return observationGuid();
    }

    @JsonProperty("ObservationGuid")
    String observationGuid();

    @Value.Redacted
    @Override
    @JsonProperty("OrganisationGuid")
    String organisationGuid();

    @Value.Redacted
    @JsonProperty("EffectiveDate")
    @JsonFormat(pattern = DATE_FORMAT_PATTERN)
    Optional<LocalDate> effectiveDate();

    @Value.Redacted
    @JsonProperty("EffectiveDatePrecision")
    String effectiveDatePrecision();

    @Value.Redacted
    @Override
    @JsonProperty("EnteredDate")
    @JsonFormat(pattern = DATE_FORMAT_PATTERN)
    LocalDate enteredDate();

    @Value.Redacted
    @Override
    @JsonFormat(pattern = TIME_FORMAT_PATTERN)
    @JsonProperty("EnteredTime")
    Optional<LocalTime> enteredTime();

    // EMIS Spec defines this field as not nullable but they send nulls....
    @Value.Redacted
    @JsonProperty("ClinicianUserInRoleGuid")
    Optional<String> clinicianUserInRoleGuid();

    // EMIS Spec defines this field as not nullable but they send nulls....
    @Value.Redacted
    @JsonProperty("EnteredByUserInRoleGuid")
    Optional<String> enteredByUserInRoleGuid();

    // EMIS Spec defines this field as not nullable but they send nulls....
    @JsonProperty("ParentObservationGuid")
    Optional<String> parentObservationGuid();

    @JsonProperty("ProblemGuid")
    Optional<String> problemGuid();

    @Value.Redacted
    @JsonProperty("AssociatedText")
    Optional<String> associatedText();

    @Value.Redacted
    @JsonProperty("ConsultationGuid")
    Optional<String> consultationGuid();

    @Value.Redacted
    @JsonProperty("Value")
    OptionalDouble value();

    @Value.Redacted
    @JsonProperty("NumericUnit")
    Optional<String> numericUnit();

    @Value.Redacted
    @JsonProperty("ObservationType")
    Optional<String> observationType();

    @Value.Redacted
    @JsonProperty("NumericRangeLow")
    OptionalDouble numericRangeLow();

    @Value.Redacted
    @JsonProperty("NumericRangeHigh")
    OptionalDouble numericRangeHigh();

    @JsonProperty("DocumentGuid")
    Optional<String> documentGuid();

    @JsonProperty("IsConfidential")
    Optional<Boolean> isConfidential();

    @Override
    @Value.Default
    default UUID transientProcessingId() {
        return UUID.randomUUID();
    }
}
