package com.pkb.service.dataupload.emis.esv2.model;

import java.time.LocalDate;
import java.util.Optional;
import java.util.stream.Collectors;
import java.util.stream.Stream;

import org.apache.commons.lang3.StringUtils;
import org.immutables.value.Value;

import com.fasterxml.jackson.annotation.JsonFormat;
import com.fasterxml.jackson.annotation.JsonProperty;
import com.fasterxml.jackson.annotation.JsonPropertyOrder;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;

@Value.Immutable
@Value.Modifiable
@JsonDeserialize(as = ImmutableAdminUserInRole.class)
@JsonPropertyOrder(value = { "UserInRoleGuid", "OrganisationGuid", "Title", "GivenName", "Surname", "JobCategoryCode",
        "JobCategoryName", "ContractStartDate", "ContractEndDate", "ProcessingId" })
public interface AdminUserInRole extends BaseEmisEsDomainModel {

    @JsonProperty("UserInRoleGuid")
    String userInRoleGuid();

    @JsonProperty("OrganisationGuid")
    String organisationGuid();

    @Value.Redacted
    @JsonProperty("Title")
    Optional<String> title();

    @Value.Redacted
    @JsonProperty("GivenName")
    Optional<String> givenName();

    @Value.Redacted
    @JsonProperty("Surname")
    Optional<String> surname();

    @Value.Redacted
    @JsonProperty("JobCategoryCode")
    Optional<String> jobCategoryCode();

    @Value.Redacted
    @JsonProperty("JobCategoryName")
    Optional<String> jobCategoryName();

    @Value.Redacted
    @JsonProperty("ContractStartDate")
    @JsonFormat(pattern = DATE_FORMAT_PATTERN)
    LocalDate contractStartDate();

    @Value.Redacted
    @JsonProperty("ContractEndDate")
    @JsonFormat(pattern = DATE_FORMAT_PATTERN)
    Optional<LocalDate> contractEndDate();

    @Value.Redacted
    @Value.Derived
    default Optional<String> displayName() {
        if (surname().map(StringUtils::trimToNull).isEmpty()) {
            return Optional.empty();
        }
        return Optional.of(
                Stream.of(title(), givenName(), surname(),
                        jobCategoryNameFormatted())
                        .map(s -> s.map(StringUtils::trimToNull))
                        .flatMap(Optional::stream)
                        .collect(Collectors.joining(" ")));
    }

    @Value.Redacted
    @Value.Derived
    default Optional<String> jobCategoryNameFormatted() {
        return jobCategoryName()
                .map(StringUtils::trimToNull)
                .map(s -> "(" + s + ")");

    }
}
