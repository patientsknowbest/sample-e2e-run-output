package com.pkb.service.dataupload.emis.esv2.model;

import com.google.common.collect.ImmutableMap;
import org.immutables.value.Value;

import java.math.BigInteger;
import java.nio.file.Path;
import java.time.LocalDate;
import java.time.format.DateTimeFormatter;
import java.util.List;
import java.util.Map;

/**
 * Represents an EMIS delta.
 */
@Value.Immutable(builder = false)
public interface EmisFileSet {
    DateTimeFormatter DIRECTORY_DATESTAMP_FORMAT = DateTimeFormatter.ofPattern("yyyyMMdd");

    /**
     * Returns the {@link Map} of {@link List<EmisFile>}s which is supported by PKB.
     * These files should be processed by the extract job.
     *
     * @return {@link ImmutableMap} of {@link List<EmisFile>}
     */
    @Value.Parameter
    Map<EmisFileType, List<EmisFile>> supportedFiles();

    @Value.Parameter
    BigInteger supportedFilesSize();

    /**
     * returns the {@link Map} of {@link List<EmisFile>}s which is not supported by PKB.
     *
     * @return {@link ImmutableMap} of {@link List<EmisFile>}
     */
    @Value.Parameter
    Map<EmisFileType, List<EmisFile>> notSupportedFiles();

    /**
     * Returns weather the EMIS delta contains every {@link EmisFileType} which is supported and required by PKB.
     *
     * @return <code>true</code> when the delta contains every required {@link EmisFileType}, <code>false</code> otherwise.
     */
    @Value.Parameter
    boolean isComplete();

    /**
     * Returns the date stamp of the delta/bulk.
     */
    @Value.Parameter
    LocalDate dateStamp();

    @Value.Parameter
    Path path();
}
