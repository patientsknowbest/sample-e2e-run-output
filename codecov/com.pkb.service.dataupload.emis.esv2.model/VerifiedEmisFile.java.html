package com.pkb.service.dataupload.emis.esv2.model;

import org.immutables.value.Value;

import java.nio.file.Path;
import java.time.LocalDateTime;

/**
 * An {@link EmisFile} that has had its file contents verified for correctness.
 */
@Value.Immutable
public interface VerifiedEmisFile extends EmisFile {

    EmisFile getEmisFile();

    /**
     * The total number of rows to process
     */
    long rowCount();

    /**
     * Hash of the file contents
     */
    String gpgHash();

    @Override
    default Path relativePath() {
        return getEmisFile().relativePath();
    }

    @Override
    default boolean isBulk() {
        return getEmisFile().isBulk();
    }

    @Override
    default EmisFileType type() {
        return getEmisFile().type();
    }

    @Override
    default LocalDateTime timeStamp() {
        return getEmisFile().timeStamp();
    }

    @Override
    default String guid() {
        return getEmisFile().guid();
    }

    @Override
    default long size() {
        return getEmisFile().size();
    }

    static VerifiedEmisFile fromEmisFile(EmisFile ef, long rowCount, String hash) {
        return ImmutableVerifiedEmisFile.builder()
                .emisFile(ef)
                .rowCount(rowCount)
                .gpgHash(hash)
                .build();
    }
}
