package com.pkb.service.dataupload.emis.esv2.model;

import static com.google.common.base.Preconditions.checkArgument;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

/**
 * File types for EMIS ES integration.
 *
 * Based on specification v5.3.1 2017.02.03
 */
public enum EmisFileType {

    ADMIN_LOCATION("Admin_Location"),
    ADMIN_ORGANISATION("Admin_Organisation"),
    ADMIN_ORGANIZATION_LOCATION("Admin_OrganisationLocation"),
    ADMIN_PATIENT("Admin_Patient"),
    ADMIN_USER_IN_ROLE("Admin_UserInRole"),
    AGREEMENTS_SHARING_ORGANIZATION("Agreements_SharingOrganisation"),
    APPOINTMENTS_SESSION("Appointment_Session"),
    APPOINTMENTS_SESSION_USER("Appointment_SessionUser"),
    APPOINTMENTS_SLOT("Appointment_Slot"),
    CARE_RECORD_CONSULTATION("CareRecord_Consultation"),
    CARE_RECORD_DIARY("CareRecord_Diary"),
    CARE_RECORD_OBSERVATION("CareRecord_Observation"),
    CARE_RECORD_OBSERVATION_REFERRAL("CareRecord_ObservationReferral"),
    CARE_RECORD_PROBLEM("CareRecord_Problem"),
    CODING_CLINICAL_CODES("Coding_ClinicalCode"),
    CODING_DRUG_CODES("Coding_DrugCode"),
    PRESCRIBING_DRUG_RECORD("Prescribing_DrugRecord"),
    PRESCRIBING_ISSUE_RECORD("Prescribing_IssueRecord"),
    AUDIT_PATIENT_AUDIT("Audit_PatientAudit"),
    AUDIT_REGISTRATION_AUDIT("Audit_RegistrationAudit");

    private final String fileNamePart;

    EmisFileType(String fileNamePart) {
        checkArgument(isNotBlank(fileNamePart), "fileNamePart must not be null or empty!");
        this.fileNamePart = fileNamePart;
    }

    public String getFileNamePart() {
        return fileNamePart;
    }
}
