package com.pkb.service.dataupload.emis.esv2.model;

import com.fasterxml.jackson.annotation.JsonFormat;
import com.fasterxml.jackson.annotation.JsonIgnore;
import com.fasterxml.jackson.annotation.JsonProperty;
import com.fasterxml.jackson.annotation.JsonPropertyOrder;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import org.immutables.value.Value;

import java.time.LocalDate;
import java.time.LocalTime;
import java.util.Optional;
import java.util.OptionalDouble;
import java.util.OptionalInt;
import java.util.UUID;

@Value.Immutable
@Value.Modifiable
@JsonDeserialize(as = ImmutablePrescribingIssueRecord.class)
@JsonPropertyOrder(value = {"IssueRecordGuid", "PatientGuid", "OrganisationGuid", "DrugRecordGuid", "EffectiveDate",
        "EffectiveDatePrecision", "EnteredDate", "EnteredTime", "ClinicianUserInRoleGuid", "EnteredByUserInRoleGuid", "CodeId", "Dosage",
        "Quantity", "QuantityUnit", "ProblemObservationGuid", "CourseDurationInDays", "EstimatedNhsCost", "IsConfidential", "Deleted",
        "ProcessingId"})
public interface PrescribingIssueRecord extends EmisPatientDataPoint<PrescribingIssueRecord>, EmisEsEnteredDateTimeSupport {

    @Override
    @JsonIgnore
    default String patientSpecificGuid() {
        return issueRecordGuid();
    }

    @JsonProperty("IssueRecordGuid")
    String issueRecordGuid();

    @JsonProperty("DrugRecordGuid")
    String drugRecordGuid();

    @Value.Redacted
    @JsonProperty("EffectiveDate")
    @JsonFormat(pattern = DATE_FORMAT_PATTERN)
    Optional<LocalDate> effectiveDate();

    @Value.Redacted
    @JsonProperty("EffectiveDatePrecision")
    Optional<String> effectiveDatePrecision();

    @Value.Redacted
    @Override
    @JsonProperty("EnteredDate")
    @JsonFormat(pattern = DATE_FORMAT_PATTERN)
    LocalDate enteredDate();

    @Value.Redacted
    @Override
    @JsonProperty("EnteredTime")
    @JsonFormat(pattern = TIME_FORMAT_PATTERN)
    Optional<LocalTime> enteredTime();

    @Value.Redacted
    @JsonProperty("ClinicianUserInRoleGuid")
    String clinicianUserInRoleGuid();

    @Value.Redacted
    @JsonProperty("EnteredByUserInRoleGuid")
    String enteredByUserInRoleGuid();

    @Value.Redacted
    @JsonProperty("Dosage")
    String dosage();

    @Value.Redacted
    @JsonProperty("Quantity")
    OptionalDouble quantity();

    @Value.Redacted
    @JsonProperty("QuantityUnit")
    String quantityUnit();

    // EMIS Spec defines this field as not nullable but they send nulls....
    @JsonProperty("ProblemObservationGuid")
    Optional<String> problemObservationGuid();

    @Value.Redacted
    @JsonProperty("CourseDurationInDays")
    OptionalInt courseDurationInDays();

    @JsonProperty("EstimatedNhsCost")
    OptionalDouble estimatedNhsCost();

    @JsonProperty("IsConfidential")
    Optional<Boolean> isConfidential();

    @Override
    @JsonProperty("Deleted")
    Optional<Boolean> deleted();

    @Override
    @Value.Default
    default UUID transientProcessingId() {
        return UUID.randomUUID();
    }
}
