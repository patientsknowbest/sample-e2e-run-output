package com.pkb.service.dataupload.emis.esv2.model;

import com.fasterxml.jackson.annotation.JsonFormat;
import com.fasterxml.jackson.annotation.JsonIgnore;
import com.fasterxml.jackson.annotation.JsonProperty;
import com.fasterxml.jackson.annotation.JsonPropertyOrder;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import org.immutables.value.Value;

import java.time.LocalDate;
import java.util.Optional;

@Value.Immutable
@Value.Modifiable
@JsonDeserialize(as = ImmutableAdminPatient.class)
@JsonPropertyOrder(value = {"PatientGuid", "OrganisationGuid", "UsualGpUserInRoleGuid", "Sex", "DateOfBirth", "DateOfDeath", "Title",
        "GivenName", "MiddleNames", "Surname", "DateOfRegistration", "NhsNumber", "PatientNumber", "PatientTypeDescription", "DummyType",
        "HouseNameFlatNumber", "NumberAndStreet", "Village", "Town", "County", "Postcode", "ResidentialInstituteCode", "NHSNumberStatus",
        "CarerName", "CarerRelation", "PersonGuid", "DateOfDeactivation", "Deleted", "SpineSensitive", "IsConfidential", "EmailAddress",
        "HomePhone", "MobilePhone", "ExternalUsualGPGuid", "ExternalUsualGP", "ExternalUsualGPOrganisation", "ProcessingId"})
public interface AdminPatient extends BaseEmisEsDomainModel, EmisEsRecordDeleteSupport {

    @Override
    @JsonIgnore
    default boolean isHandleAsDeleted() {
        return EmisEsRecordDeleteSupport.super.isHandleAsDeleted()
                || isConfidential().orElse(false);
    }

    @JsonIgnore
    default boolean isHandleAsDischargedOrDeleted() {
        return dateOfDeactivation().isPresent() || isHandleAsDeleted();
    }

    @JsonProperty("PatientGuid")
    String patientGuid();

    @Value.Redacted
    @JsonProperty("OrganisationGuid")
    String organisationGuid();

    @Value.Redacted
    @JsonProperty("UsualGpUserInRoleGuid")
    Optional<String> usualGpUserInRoleGuid();

    @Value.Redacted
    @JsonProperty("Sex")
    Optional<String> sex();

    //EMIS nulls out fields when deletes record, which according to spec are not nullable....
    @Value.Redacted
    @JsonProperty("DateOfBirth")
    @JsonFormat(pattern = DATE_FORMAT_PATTERN)
    Optional<LocalDate> dateOfBirth();

    @Value.Redacted
    @JsonProperty("DateOfDeath")
    @JsonFormat(pattern = DATE_FORMAT_PATTERN)
    Optional<LocalDate> dateOfDeath();

    //Null in deleted record
    @Value.Redacted
    @JsonProperty("Title")
    Optional<String> title();

    //Null in deleted record
    @Value.Redacted
    @JsonProperty("GivenName")
    Optional<String> givenName();

    // emis docs say not nullable but we get null data in samples
    @Value.Redacted
    @JsonProperty("MiddleNames")
    Optional<String> middleNames();

    //Null in deleted record
    @Value.Redacted
    @JsonProperty("Surname")
    Optional<String> surname();

    @Value.Redacted
    @JsonProperty("DateOfRegistration")
    @JsonFormat(pattern = DATE_FORMAT_PATTERN)
    Optional<LocalDate> dateOfRegistration();

    @Value.Redacted
    @JsonProperty("NhsNumber")
    Optional<String> nhsNumber();

    @JsonProperty("PatientNumber")
    Optional<Integer> patientNumber();

    @JsonProperty("PatientTypeDescription")
    Optional<String> patientTypeDescription();

    @JsonProperty("DummyType")
    Optional<Boolean> dummyType();

    @Value.Redacted
    @JsonProperty("HouseNameFlatNumber")
    Optional<String> houseNameFlatNumber();

    @Value.Redacted
    @JsonProperty("NumberAndStreet")
    Optional<String> numberAndStreet();

    @Value.Redacted
    @JsonProperty("Village")
    Optional<String> village();

    @Value.Redacted
    @JsonProperty("Town")
    Optional<String> town();

    @Value.Redacted
    @JsonProperty("County")
    Optional<String> county();

    @Value.Redacted
    @JsonProperty("Postcode")
    Optional<String> postcode();

    @Value.Redacted
    @JsonProperty("ResidentialInstituteCode")
    Optional<String> residentialInstituteCode();

    @Value.Redacted
    @JsonProperty("NHSNumberStatus")
    Optional<String> nhsNumberStatus();

    @Value.Redacted
    @JsonProperty("CarerName")
    Optional<String> carerName();

    @Value.Redacted
    @JsonProperty("CarerRelation")
    Optional<String> carerRelation();

    @JsonProperty("PersonGuid")
    String personGuid();

    @Value.Redacted
    @JsonProperty("DateOfDeactivation")
    @JsonFormat(pattern = DATE_FORMAT_PATTERN)
    Optional<LocalDate> dateOfDeactivation();

    @Override
    @JsonProperty("Deleted")
    Optional<Boolean> deleted();

    @Value.Redacted
    @JsonProperty("SpineSensitive")
    Optional<Boolean> spineSensitiveOrNullIfDeleted();

    @JsonProperty("IsConfidential")
    Optional<Boolean> isConfidential();

    @Value.Redacted
    @JsonProperty("EmailAddress")
    Optional<String> emailAddress();

    @Value.Redacted
    @JsonProperty("HomePhone")
    Optional<String> homePhone();

    @Value.Redacted
    @JsonProperty("MobilePhone")
    Optional<String> mobilePhone();

    @Value.Redacted
    @JsonProperty("ExternalUsualGPGuid")
    Optional<String> externalUsualGPGuid();

    @Value.Redacted
    @JsonProperty("ExternalUsualGP")
    Optional<String> externalUsualGP();

    @Value.Redacted
    @JsonProperty("ExternalUsualGPOrganisation")
    Optional<String> externalUsualGPOrganisation();

}
