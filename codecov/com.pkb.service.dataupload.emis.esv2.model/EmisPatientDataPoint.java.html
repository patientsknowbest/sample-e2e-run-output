package com.pkb.service.dataupload.emis.esv2.model;

import com.fasterxml.jackson.annotation.JsonIgnore;
import com.fasterxml.jackson.annotation.JsonProperty;
import com.pkb.dataupload.emis.esv2.interfaces.EmisRecordWithPatientNamespacedGuid;
import org.immutables.value.Value;

import java.math.BigInteger;
import java.util.Optional;
import java.util.UUID;

/**
 * A record we receive from EMIS related to data for a patient, such as a CareRecordObservation
 */
public interface EmisPatientDataPoint<T extends EmisPatientDataPoint<T>> extends BaseEmisEsDomainModel, EmisEsRecordDeleteSupport, EmisRecordWithPatientNamespacedGuid {

    @JsonIgnore
    default boolean isHandleAsDeleted(AdminPatient adminPatient) {
        return (adminPatient != null && adminPatient.isHandleAsDeleted()) || isHandleAsDeleted();
    }

    @Value.Redacted
    @JsonProperty("PatientGuid")
    @Override
    String patientGuid();

    @JsonIgnore
    UUID transientProcessingId();

    @Override
    @JsonProperty("Deleted")
    Optional<Boolean> deleted();

    @Value.Redacted
    @JsonProperty("OrganisationGuid")
    String organisationGuid();

    @JsonProperty("CodeId")
    Optional<BigInteger> codeId();
}
