package com.pkb.service.dataupload.emis.esv2.model;

import org.immutables.value.Value;

import java.nio.file.Path;
import java.time.LocalDate;
import java.util.Map;

/*
 *
 * Wrapper interface for supplementing existing VerifiedEmisFileSet with:
 * <ul>
 * <li> sftpUniqueId : unique sftp server Id
 * <li> serverDownloadDir : directory to which data of specific sftp server is stored
 * </ul>
 * */

@Value.Immutable
public interface VerifiedEmisFileSftpSet {

    @Value.Parameter
    VerifiedEmisFileSet verifiedEmisFileSet();

    @Value.Parameter
    Path serverDownloadDir();

    @Value.Parameter
    String sftpUniqueId();

    default LocalDate dateStamp() {
        return verifiedEmisFileSet().dateStamp();
    }

    static VerifiedEmisFileSftpSet fromEmisFileSftpSet(EmisFileSftpSet efss, Map<EmisFileType, VerifiedEmisFile> files) {
        return ImmutableVerifiedEmisFileSftpSet.of(ImmutableVerifiedEmisFileSet.of(files, efss.emisFileSet().supportedFilesSize(), efss.emisFileSet().dateStamp(), efss.emisFileSet().path()), efss.serverDownloadDir(), efss.sftpUniqueId());
    }
}
