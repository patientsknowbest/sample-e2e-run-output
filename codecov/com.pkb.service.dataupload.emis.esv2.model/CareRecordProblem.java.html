package com.pkb.service.dataupload.emis.esv2.model;

import com.fasterxml.jackson.annotation.JsonFormat;
import com.fasterxml.jackson.annotation.JsonIgnore;
import com.fasterxml.jackson.annotation.JsonProperty;
import com.fasterxml.jackson.annotation.JsonPropertyOrder;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import org.immutables.value.Value;

import java.time.LocalDate;
import java.util.Optional;
import java.util.UUID;

@Value.Immutable
@Value.Modifiable
@JsonDeserialize(as = ImmutableCareRecordProblem.class)
@JsonPropertyOrder(value = {"ObservationGuid", "PatientGuid", "OrganisationGuid", "ParentProblemObservationGuid", "Deleted", "Comment",
        "EndDate", "EndDatePrecision", "ExpectedDuration", "LastReviewDate", "LastReviewDatePrecision", "LastReviewUserInRoleGuid",
        "ParentProblemRelationship", "ProblemStatusDescription", "SignificanceDescription", "ProcessingId"})
public interface CareRecordProblem extends EmisPatientDataPoint<CareRecordProblem> {

    @Override
    @JsonIgnore
    default String patientSpecificGuid() {
        return observationGuid();
    }

    @JsonProperty("ObservationGuid")
    String observationGuid();

    @JsonProperty("ParentProblemObservationGuid")
    Optional<String> parentProblemObservationGuid();

    @Value.Redacted
    @JsonProperty("Comment")
    Optional<String> comment();

    @Value.Redacted
    @JsonProperty("EndDate")
    @JsonFormat(pattern = DATE_FORMAT_PATTERN)
    Optional<LocalDate> endDate();

    @Value.Redacted
    @JsonProperty("EndDatePrecision")
    Optional<String> endDatePrecision();

    @Value.Redacted
    @JsonProperty("ExpectedDuration")
    int expectedDuration();

    @Value.Redacted
    @JsonProperty("LastReviewDate")
    @JsonFormat(pattern = DATE_FORMAT_PATTERN)
    Optional<LocalDate> lastReviewDate();

    @Value.Redacted
    @JsonProperty("LastReviewDatePrecision")
    Optional<String> lastReviewDatePrecision();

    @Value.Redacted
    @JsonProperty("LastReviewUserInRoleGuid")
    Optional<String> lastReviewUserInRoleGuid();

    @Value.Redacted
    @JsonProperty("ParentProblemRelationship")
    Optional<String> parentProblemRelationship();

    @Value.Redacted
    @JsonProperty("ProblemStatusDescription")
    Optional<String> problemStatusDescription();

    @Value.Redacted
    @JsonProperty("SignificanceDescription")
    Optional<String> significanceDescription();

    @Override
    @Value.Default
    default UUID transientProcessingId() {
        return UUID.randomUUID();
    }
}
