package com.pkb.service.dataupload.emis.esv2.model;

import com.pkb.coding.entity.CodingTranslationCodingFields;
import com.pkb.institute.entity.Org;
import org.immutables.value.Value;

@Value.Immutable
public abstract class CodeAndUnitMatchWrapper {

    public abstract Org org();

    public abstract String unit();

    public abstract CodingTranslationCodingFields codingTranslation();

    @Value.Derived
    public LoincCodeAndUnit loincCodeAndUnit() {
        return ImmutableLoincCodeAndUnit.builder()
                .loincCode(codingTranslation().getTargetCode())
                .unit(unit())
                .build();
    }

    @Value.Derived
    public OrgIdSnomedCTCodeAndUnit orgIdSnomedCTCodeAndUnit() {
        return ImmutableOrgIdSnomedCTCodeAndUnit.builder()
                .orgId(org().getId())
                .snomedCTCode(codingTranslation().getSourceCode())
                .unit(unit())
                .build();
    }
}
