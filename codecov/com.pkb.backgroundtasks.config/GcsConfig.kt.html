package com.pkb.backgroundtasks.config

import com.google.api.gax.core.CredentialsProvider
import com.google.api.gax.retrying.RetrySettings
import com.google.cloud.NoCredentials
import com.google.cloud.spring.autoconfigure.storage.GcpStorageAutoConfiguration
import com.google.cloud.spring.core.GcpProjectIdProvider
import com.google.cloud.spring.core.UserAgentHeaderProvider
import com.google.cloud.storage.Storage
import com.google.cloud.storage.StorageOptions
import com.pkb.common.config.BaseConfig
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import org.springframework.context.annotation.Profile
import org.threeten.bp.Duration

@Configuration
class GcsConfig {

    @Bean
    @Profile("e2e", "integration-test")
    fun fakeGcsStorage(gcpProjectIdProvider: GcpProjectIdProvider, properties: BackgroundTasksProperties, config: BaseConfig): Storage {
        return getBasicStorageOptions(gcpProjectIdProvider, config).setHost(properties.fakeGcsHost).setCredentials(NoCredentials.getInstance()).build().service
    }


    @Bean
    @Profile("!e2e & !integration-test")
    fun realGcsStorage(gcpProjectIdProvider: GcpProjectIdProvider, credentialsProvider: CredentialsProvider, properties: BackgroundTasksProperties, config: BaseConfig): Storage {
        return getBasicStorageOptions(gcpProjectIdProvider, config).setCredentials(credentialsProvider.credentials).build().service
    }


    private fun getBasicStorageOptions(gcpProjectIdProvider: GcpProjectIdProvider, config: BaseConfig): StorageOptions.Builder {
        val rs = RetrySettings.newBuilder()
            .setInitialRetryDelay(Duration.ofMillis(config.gcsInitialTimeout.toLong()))
            .setRetryDelayMultiplier(config.gcsTimeoutMultiplier.toDouble())
            .setMaxRetryDelay(Duration.ofMillis(config.gcsMaxTimeout.toLong()))
            .setInitialRpcTimeout(Duration.ofMillis(config.gcsInitialRpcTimeout.toLong()))
            .setRpcTimeoutMultiplier(config.gcsRpcTimeoutMultiplier.toDouble())
            .setMaxRpcTimeout(Duration.ofMillis(config.gcsMaxRpcTimeout.toLong()))
            .setMaxAttempts(config.gcsMaxAttempts)
            .setTotalTimeout(Duration.ofMillis(config.gcsMaxTotalTimeout.toLong()))
            .build()

        return StorageOptions.newBuilder()
            .setHeaderProvider(
                UserAgentHeaderProvider(GcpStorageAutoConfiguration::class.java)
            )
            .setProjectId(gcpProjectIdProvider.projectId)
            .setRetrySettings(rs)
    }

}
