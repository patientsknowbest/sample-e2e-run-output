package com.pkb.backgroundtasks.config

import io.micrometer.core.instrument.Metrics
import io.micrometer.core.instrument.binder.okhttp3.OkHttpMetricsEventListener
import okhttp3.OkHttpClient
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import java.util.concurrent.TimeUnit

@Configuration
class OkHttpConfig {

    @Bean
    fun okHttpClient(configProperties: PkbCommonConfig): OkHttpClient {
        val metrics = OkHttpMetricsEventListener.builder(Metrics.globalRegistry, "okhttp.requests").build()
        return OkHttpClient.Builder().readTimeout(5, TimeUnit.MINUTES).eventListener(metrics).build()
    }

}
