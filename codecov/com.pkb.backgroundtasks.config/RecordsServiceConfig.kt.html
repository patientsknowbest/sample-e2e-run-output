package com.pkb.backgroundtasks.config

import ca.uhn.fhir.context.FhirContext
import ca.uhn.fhir.rest.api.Constants.HEADER_AUTHORIZATION
import ca.uhn.fhir.rest.api.EncodingEnum
import ca.uhn.fhir.rest.client.api.IGenericClient
import ca.uhn.fhir.rest.client.api.IHttpRequest
import ca.uhn.fhir.rest.client.interceptor.BearerTokenAuthInterceptor
import org.keycloak.OAuth2Constants.CLIENT_CREDENTIALS
import org.keycloak.admin.client.Keycloak
import org.keycloak.admin.client.KeycloakBuilder
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration

@Configuration
class RecordsServiceConfig {

    @Bean
    fun fhirContext(): FhirContext {
        return FhirContext.forR4()
    }

    @Bean
    fun keycloakClient(properties: BackgroundTasksProperties): Keycloak {
        return KeycloakBuilder.builder()
                .serverUrl("${properties.keycloakUrl}/auth")
                .realm(properties.keycloakRealm)
                .grantType(CLIENT_CREDENTIALS)
                .clientId(properties.keycloakClientId)
                .clientSecret(properties.keycloakClientSecret)
                .build()
    }

    @Bean
    fun fhirStoreClient(fhirContext: FhirContext, keycloakClient: Keycloak, properties: BackgroundTasksProperties): IGenericClient {
        val client = fhirContext.newRestfulGenericClient("${properties.recordsService.aidboxUrl}/fhir")
        client.encoding = EncodingEnum.JSON
        client.registerInterceptor(object : BearerTokenAuthInterceptor("accessToken") {
            override fun interceptRequest(theRequest: IHttpRequest) {
                theRequest.addHeader(HEADER_AUTHORIZATION, "Bearer " + keycloakClient.tokenManager().accessTokenString)
            }
        })
        return client
    }
}
