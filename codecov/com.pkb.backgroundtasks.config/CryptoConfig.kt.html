package com.pkb.backgroundtasks.config

import com.pkb.backgroundtasks.repository.AccountRepository
import com.pkb.backgroundtasks.util.RequestIdFilter.Companion.REQUEST_ID
import com.pkb.common.ehrdata.serialization.EHRDataDeserializer
import com.pkb.common.ehrdata.serialization.EHRDataSerializer
import com.pkb.crypto.decryptionhelper.DecryptionHelper
import com.pkb.crypto.decryptionhelper.DecryptionHelperAppMenuData
import com.pkb.crypto.decryptionhelper.provider.accountkeys.AccountKeysProviderFactory
import com.pkb.crypto.encryptionhelper.AppMenuDataEncryptionKeysProvider
import com.pkb.crypto.encryptionhelper.EncryptionHelperAppMenuData
import com.pkb.kms.client.core.Kms
import org.slf4j.MDC
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import java.util.UUID

@Configuration
class CryptoConfig {

    @Bean
    fun menuDataDecryptionHelper() =
            DecryptionHelperAppMenuData(DecryptionHelper(), AccountKeysProviderFactory(), object : EHRDataDeserializer {})


    @Bean
    fun menuDataEncryptionHelper(keysProvider: AppMenuDataEncryptionKeysProvider) =
            EncryptionHelperAppMenuData(keysProvider, object : EHRDataSerializer {})

    @Bean
    fun menuDataEncryptionKeysProvider(kms: Kms, accountRepo: AccountRepository) =
            AppMenuDataEncryptionKeysProvider { accountId ->
                accountRepo.findById(accountId).map { account ->
                    kms.getSymmetricKey(MDC.get(REQUEST_ID) ?: UUID.randomUUID().toString(), account.publicId)
                            .getOrElseThrow { e -> RuntimeException(e.toString()) }
                }.orElseThrow()
            }
}
