package com.pkb.backgroundtasks.config

import com.pkb.common.datetime.DateTimeService
import com.pkb.common.datetime.DefaultDateTimeService
import com.pkb.common.datetime.FakeDateTimeService
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import org.springframework.data.auditing.DateTimeProvider
import java.util.Optional

@Configuration
class DateTimeServiceConfig {

    @Bean
    fun dateTimeService(config: PkbCommonConfig): DateTimeService {
        return if (config.isFakeDateTimeServiceEnabled) FakeDateTimeService() else DefaultDateTimeService()
    }

    @Bean
    fun dateTimeProvider(dateTimeService: DateTimeService): DateTimeProvider {
        return DateTimeProvider { Optional.of(dateTimeService.now()) }
    }
}
