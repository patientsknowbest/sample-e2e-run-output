package com.pkb.backgroundtasks.config

import com.fasterxml.jackson.databind.ObjectMapper
import com.pkb.backgroundtasks.PhrClient
import com.pkb.backgroundtasks.PhrDocumentMetadataClient
import com.pkb.backgroundtasks.PhrSyncUploadedDataClient
import okhttp3.OkHttpClient
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import java.util.concurrent.TimeUnit

@Configuration
class PhrConfig {

    @Bean
    fun phrClient(properties: BackgroundTasksProperties, okHttpClient: OkHttpClient, objectMapper: ObjectMapper): PhrClient =
            PhrClient.phrClient(properties.phrUrl, okHttpClient, objectMapper)

    @Bean
    fun phrSyncUploadedDataClient(
            okHttpClient: OkHttpClient,
            properties: BackgroundTasksProperties
    ): PhrSyncUploadedDataClient =
        PhrSyncUploadedDataClient.phrSyncUploadedDataClient(
                properties.phrUrl, okHttpClient.newBuilder()
                .connectTimeout(properties.uploadedData.requestTimeoutSeconds, TimeUnit.SECONDS)
                .writeTimeout(properties.uploadedData.requestTimeoutSeconds, TimeUnit.SECONDS)
                .readTimeout(properties.uploadedData.requestTimeoutSeconds, TimeUnit.SECONDS)
                .build()
        )

    @Bean
    fun phrDocumentMetadataClient(
            okHttpClient: OkHttpClient,
            config: BackgroundTasksProperties
    ): PhrDocumentMetadataClient =
            PhrDocumentMetadataClient.phrDocumentMetadataClient(
                    config.phrUrl, okHttpClient.newBuilder()
                    .connectTimeout(config.migrateDocuments.requestTimeoutSeconds, TimeUnit.SECONDS)
                    .writeTimeout(config.migrateDocuments.requestTimeoutSeconds, TimeUnit.SECONDS)
                    .readTimeout(config.migrateDocuments.requestTimeoutSeconds, TimeUnit.SECONDS)
                    .build()
            )
}
