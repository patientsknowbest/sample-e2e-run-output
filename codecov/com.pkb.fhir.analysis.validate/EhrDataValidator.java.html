package com.pkb.fhir.analysis.validate;

import ca.uhn.fhir.validation.FhirValidator;
import ca.uhn.fhir.validation.SingleValidationMessage;
import ca.uhn.fhir.validation.ValidationResult;
import com.pkb.fhir.analysis.persistence.targetdb.TransformedEhrDataEntity;
import com.pkb.fhir.common.enums.TransformedEhrDataStatus;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.stream.Collectors;

import static java.lang.invoke.MethodHandles.lookup;

public class EhrDataValidator {
    private static final Logger LOG = LoggerFactory.getLogger(lookup().lookupClass());
    private final FhirValidator fhirValidator;

    public EhrDataValidator(FhirValidator fhirValidator) {
        this.fhirValidator = fhirValidator;
    }

    public TransformedEhrDataEntity validate(TransformedEhrDataEntity ehrData) {
        boolean isValid;
        String errorMsg;
        try {
            ValidationResult result = fhirValidator.validateWithResult(ehrData.getContent());
            isValid = result.isSuccessful();
            errorMsg = result.getMessages()
                    .stream()
                    .map(SingleValidationMessage::getMessage)
                    .collect(Collectors.joining("\n"));
        } catch (Exception e) {
            errorMsg = String.format("Failed to validate ehr data with id %d. Cause: %s", ehrData.getId(), e.getMessage());
            LOG.error(errorMsg, e);
            isValid = false;
        }
        if (isValid) {
            ehrData.setStatus(TransformedEhrDataStatus.VALIDATED);
        } else {
            ehrData.setStatus(TransformedEhrDataStatus.VALIDATION_FAILED);
            ehrData.setError(errorMsg);
        }
        return ehrData;
    }
}
