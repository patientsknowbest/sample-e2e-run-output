package com.pkb.api.fhir.encounter;

import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import com.pkb.api.fhir.AbstractResourceService;
import com.pkb.api.fhir.config.FhirApiProperties;
import com.pkb.api.fhir.exception.ServerResponseExceptionMapper;
import com.pkb.authz.AuthorizationDataService;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.domain.SystemUserTypeNotPermittedException;
import com.pkb.domain.criteria.FhirCriteriaResult;
import com.pkb.domain.encounter.EncounterService;
import com.pkb.domain.encounter.ImmutableEncounterCriteria;
import io.vavr.control.Either;
import org.hl7.fhir.dstu3.model.Encounter;
import org.jetbrains.annotations.NotNull;

import static com.google.common.base.Predicates.instanceOf;
import static com.pkb.authz.ImmutableTargetMeta.targetMeta;
import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.context.Path.FHIR;
import static com.pkb.authz.data.ImmutableAuthorizationInputs.inputs;
import static com.pkb.domain.criteria.FhirCriteriaResult.empty;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;
import static io.vavr.API.Set;
import static io.vavr.API.Some;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

public class EncounterResourceService extends AbstractResourceService<Encounter> {

    private final EncounterService service;
    private final EncounterMapper mapper;
    private final AuthorizationDataService authorizationDataService;

    private final FhirApiProperties fhirConfig;

    public EncounterResourceService(
            @NotNull EncounterService service,
            @NotNull EncounterMapper mapper,
            @NotNull ServerResponseExceptionMapper exceptionMapper,
            @NotNull AuthorizationDataService authorizationDataService,
            FhirApiProperties fhirConfig) {
        super(exceptionMapper);
        this.service = service;
        this.mapper = mapper;
        this.authorizationDataService = authorizationDataService;
        this.fhirConfig = fhirConfig;
    }

    public @NotNull Either<BaseServerResponseException, FhirCriteriaResult<Encounter>> search(@NotNull EncounterSearchCriteria criteria) {
        AuthorizationData authzData = authorizationDataService.fetch(
                inputs()
                        .path(FHIR)
                        .authenticatedIdentity(criteria.authenticatedIdentity())
                        .action(SEARCH)
                        .addTargets(targetMeta(com.pkb.datamodel.Encounter.class,
                                Set(),
                                Some(criteria.patientId()),
                                true))
                        .build());

        return service.findEncounters(
                ImmutableEncounterCriteria
                        .builder()
                        .authzData(authzData)
                        .patientId(criteria.patientId())
                        .dateRestriction(criteria.dateRestriction())
                        .pageSize(fhirConfig.getEncounterMaxNumberOfResources())
                        .build())
                .fold(error -> Match(error).of(
                        Case($(instanceOf(SystemUserTypeNotPermittedException.class)), nae -> left(toSystemUserTypeNotPermitted())),
                        Case($(), right(empty()))),
                        appointments -> right(mapper.toCriteriaResult(appointments, criteria.includes())));
    }

}
