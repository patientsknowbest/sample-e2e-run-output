package com.pkb.datamodel.user;

import com.google.common.base.MoreObjects;
import com.pkb.datamodel.DataModelStyle;
import com.pkb.entities.enums.UserType;
import io.vavr.control.Option;
import org.immutables.value.Value;

@Value.Immutable
@DataModelStyle
public abstract class Professional extends Person {

    public abstract Option<String> getJobTitle();

    @Value.Check
    void checkUserType() {
        checkUserType(UserType.REG_CLINICIAN);
    }

    public abstract static class Builder implements PersonBuilder {
    }

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(Professional.class)
                .omitNullValues()
                .add("id", getId())
                .add("status", getUserStatus())
                .toString();
    }
}
