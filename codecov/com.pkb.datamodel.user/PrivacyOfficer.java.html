package com.pkb.datamodel.user;

import com.google.common.base.MoreObjects;
import com.pkb.datamodel.DataModelStyle;
import com.pkb.datamodel.role.PrivacyOfficerRole;
import com.pkb.entities.enums.UserType;
import org.immutables.value.Value;

@Value.Immutable
@DataModelStyle
public abstract class PrivacyOfficer extends Person implements PrivacyOfficerRole {

    @Value.Check
    void checkUserType(){ checkUserType(UserType.PRIVACY_OFFICER); }

    @Override public String toString(){
        return MoreObjects.toStringHelper(PrivacyOfficer.class)
                .omitNullValues()
                .add("id", getId())
                .add("status", getUserStatus())
                .toString();
    }

    public abstract static class Builder implements  PersonBuilder{
    }

}
