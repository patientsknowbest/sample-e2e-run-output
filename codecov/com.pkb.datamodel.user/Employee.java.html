package com.pkb.datamodel.user;

import com.google.common.base.MoreObjects;
import com.pkb.datamodel.DataModelStyle;
import com.pkb.datamodel.role.EmployeeRole;
import com.pkb.entities.enums.UserType;
import org.immutables.value.Value;

@Value.Immutable
@DataModelStyle
public abstract class Employee extends Person implements EmployeeRole {
    @Value.Check
    void checkUserType(){ checkUserType(UserType.EMPLOYEE); }

    @Override public String toString(){
        return MoreObjects.toStringHelper(Employee.class)
                .omitNullValues()
                .add("id", getId())
                .add("status", getUserStatus())
                .toString();
    }

    public abstract static class Builder implements PersonBuilder{
    }
}
