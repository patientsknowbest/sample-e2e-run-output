package com.pkb.datamodel.user;

import com.pkb.datamodel.Org;
import com.pkb.datamodel.authorization.ConsentedEntity;
import com.pkb.datamodel.team.TeamUser;
import com.pkb.datamodel.usertype.PersonType;
import com.pkb.entities.enums.UserStatus;
import com.pkb.entities.enums.UserType;
import io.vavr.collection.Set;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;

import java.time.LocalDate;
import java.time.format.DateTimeFormatter;

import static com.google.common.base.Preconditions.checkState;
import static java.text.MessageFormat.format;

public abstract class Person implements PersonType, ConsentedEntity {

    private static final DateTimeFormatter DOB_FORMATTER = DateTimeFormatter.ofPattern("yyyy-MM-dd");

    public abstract Demographics demographics();

    public abstract UserType getUserType();

    public abstract UserStatus getUserStatus();

    protected abstract Option<String> getDateOfBirthString();

    public abstract Option<Org> getSourceOrg();

    public abstract Option<Set<TeamUser>> getTeams();
    
    public abstract Option<String> getLanguageCode();

    protected Option<LocalDate> parseDateOfBirth() {
        return getDateOfBirthString().map(this::parseDateOfBirth);
    }

    protected void checkUserType(@NotNull UserType expected) {
        checkState(expected == this.getUserType(),
                format("User type {0} is not {1}", this.getUserType(), expected));
    }

    private LocalDate parseDateOfBirth(String dateOfBirthString) {
        return LocalDate.parse(dateOfBirthString, DOB_FORMATTER);
    }
}
