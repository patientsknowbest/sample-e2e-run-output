package com.pkb.datamodel.user;

import com.google.common.base.MoreObjects;
import com.pkb.datamodel.DataModelStyle;
import com.pkb.datamodel.role.PatientRole;
import com.pkb.entities.enums.UserType;
import io.vavr.control.Option;
import org.immutables.value.Value;

import java.time.LocalDate;

@Value.Immutable
@DataModelStyle
public abstract class Patient extends Person implements PatientRole {

    public abstract boolean isSharingDisabled();

    public abstract boolean isAccessFrozen();

    public abstract Option<Long> defaultAccountId();

    @Value.Lazy
    public long requireDefaultAccountId() {
        return defaultAccountId().getOrElseThrow(NullPointerException::new);
    }

    @Value.Lazy
    public Option<LocalDate> getDateOfBirthUTC() {
        return parseDateOfBirth();
    }

    @Value.Check
    void checkUserType() {
        checkUserType(UserType.PATIENT);
    }

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(Patient.class)
                .omitNullValues()
                .add("id", getId())
                .add("status", getUserStatus())
                .toString();
    }

    public abstract static class Builder implements PersonBuilder {}
}
