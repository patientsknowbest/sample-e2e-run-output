package com.pkb.datamodel.user;

import com.google.common.base.MoreObjects;
import com.pkb.datamodel.DataModelStyle;
import com.pkb.datamodel.role.TeamProfessionalRole;
import org.immutables.value.Value;

@Value.Immutable
@DataModelStyle
public abstract class TeamProfessional extends Professional implements TeamProfessionalRole {

    @Override
    public String toString() {
        return MoreObjects.toStringHelper(TeamProfessional.class)
                .omitNullValues()
                .add("id", getId())
                .add("status", getUserStatus())
                .toString();
    }

    @Value.Derived
    @Override
    public boolean isTeamProfessional() {
        return true;
    }

    public abstract static class Builder implements PersonBuilder {
    }
}
