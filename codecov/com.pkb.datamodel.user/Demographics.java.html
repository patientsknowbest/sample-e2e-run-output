package com.pkb.datamodel.user;

import com.pkb.datamodel.DataModelStyle;
import com.pkb.datamodel.Identifiable;
import com.pkb.datamodel.NationalId;
import com.pkb.datamodel.OrgLevelId;
import com.pkb.datamodel.PubliclyIdentifiable;
import com.pkb.datamodel.TeamLevelId;
import com.pkb.datamodel.contact.EmailContact;
import com.pkb.datamodel.contact.PhoneContact;
import com.pkb.entities.enums.Gender;
import com.pkb.entities.enums.NationalIdType;
import io.vavr.collection.List;
import io.vavr.control.Option;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;
import org.jetbrains.annotations.NotNull;

import java.time.LocalDate;
import java.util.UUID;

@Immutable
@Style(stagedBuilder = true, passAnnotations = NotNull.class)
@DataModelStyle
public interface Demographics extends Identifiable<Long>, PubliclyIdentifiable<UUID> {
    @NotNull Option<String> getTitle();

    @Override
    @NotNull UUID getPublicId();

    @NotNull Option<String> getFirstName();

    @NotNull Option<String> getMiddleName();

    @NotNull Option<String> getLastName();

    @NotNull PostalAddress getAddress();

    @NotNull
    List<EmailContact> getEmail();

    @NotNull
    List<PhoneContact> getPhone();

    @NotNull
    Option<LocalDate> getDateOfBirth();

    @NotNull Gender getGender();

    boolean isDeceased();

    @NotNull
    Option<LocalDate> getDeceasedDate();

    @NotNull
    List<NationalId> getNationalId(); // TODO: GDE-512 patient-specific attribute

    @NotNull
    List<TeamLevelId> getTeamLevelId(); // TODO: GDE-512 patient-specific attribute

    @NotNull
    List<OrgLevelId> getOrgLevelId(); // TODO: GDE-512 patient-specific attribute

    boolean isSharingDisabled(); // TODO: GDE-512 patient-specific attribute

    default Option<NationalId> getNationalIdByTypeInCountry(NationalIdType type) {
        if (type == null || getNationalId().isEmpty()) {
            return Option.none();
        }

        return getNationalId()
                .filter(nid -> nid.getType() == type)
                .peekOption()
                .orElse(() -> getNationalId().filter(nid -> nid.getType().areUsedInSameCountry(type)).peekOption());
    }

    default Option<NationalId> getNationalIdByExactType(NationalIdType type) {
        if (type == null || getNationalId().isEmpty()) {
            return Option.none();
        }

        for (NationalId id : getNationalId()) {
            if (id.getType() == type) {
                return Option.of(id);
            }
        }

        return Option.none();
    }

    @NotNull
    default Option<String> getPrimaryEmailValue() {
        return getEmail()
                .find(EmailContact::isPrimary)
                .map(EmailContact::getValue);
    }
}
