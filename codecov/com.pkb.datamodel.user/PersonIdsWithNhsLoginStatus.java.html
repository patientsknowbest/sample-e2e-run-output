package com.pkb.datamodel.user;

import io.vavr.control.Option;
import org.immutables.value.Value;
import org.immutables.vavr.encodings.VavrEncodingEnabled;

import java.util.UUID;

@Value.Immutable
@VavrEncodingEnabled
public interface PersonIdsWithNhsLoginStatus {

    NhsLoginStatus getNhsLoginStatus();

    Long getId();

    UUID getPublicId();

    Option<UUID> getAccountPublicId();

    enum NhsLoginStatus {
        EMAIL_MATCH,
        FULL_MATCH,
        NHS_NUMBER_MATCH_WITH_EMAIL,
        OUT_OF_AREA,
        OUT_OF_AREA_WITHOUT_NHS_NUMBER
    }
}
