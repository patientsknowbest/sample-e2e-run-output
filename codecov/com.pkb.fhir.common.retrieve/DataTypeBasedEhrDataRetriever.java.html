package com.pkb.fhir.common.retrieve;

import com.pkb.entities.enums.MenuDataType;
import com.pkb.fhir.common.decrypt.DecryptedEhrData;
import com.pkb.fhir.common.transform.TransformableDto;

import java.util.Collections;
import java.util.List;
import java.util.Map;

public class DataTypeBasedEhrDataRetriever {
    private final Map<MenuDataType, EhrDataRetriever> ehrDataRetrieverMap;

    public DataTypeBasedEhrDataRetriever(Map<MenuDataType, EhrDataRetriever> ehrDataRetrieverMap) {
        this.ehrDataRetrieverMap = ehrDataRetrieverMap;
    }

    public List<? extends TransformableDto> retrieveEhrDataDto(Map<Long, DecryptedEhrData> decryptedEhrDataMap) {
        if (decryptedEhrDataMap.isEmpty()) {
            return Collections.emptyList();
        } else {
            return ehrDataRetrieverMap.get(decryptedEhrDataMap.values().stream().findAny().get().getDataType())
                    .retrieveEhrDataDto(decryptedEhrDataMap);
        }
    }
}
