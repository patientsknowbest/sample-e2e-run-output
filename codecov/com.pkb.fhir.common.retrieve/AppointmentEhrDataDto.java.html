package com.pkb.fhir.common.retrieve;

import com.pkb.entities.enums.AppointmentStatus;

import java.time.LocalDateTime;
import java.util.UUID;

public class AppointmentEhrDataDto extends EhrDataDto {

    private AppointmentStatus status;
    private LocalDateTime start;
    private LocalDateTime end;
    private String commonId;
    private String externalId;
    private String typeCode;
    private String typeCodingSystem;

    public AppointmentEhrDataDto(long id, UUID patientId, UUID uniqueId, LocalDateTime start, LocalDateTime end, String commonId, String status,
                                 String externalId, String typeCode, String typeCodingSystem, Long menuDataSourceOrgId,
                                 Long menuDataSourceTeamId, Long menuDataSourcePersonId, LocalDateTime created, LocalDateTime persistedDate,
                                 Long uploadedDataId, Long hl7PartnerId, String route, String responsibleOrganisationIdentifier,
                                 Long pfSourcePersonId, Long pfSourceTeamId, Long pfSourceOrgId, String pfAccessRoute, LocalDateTime pfChangeDate,
                                 String requireGeneral, String requireSexualHealth, String requireMentalHealth, String requireSocialCare) {
        super(id, patientId, uniqueId, menuDataSourceOrgId,
                menuDataSourceTeamId, menuDataSourcePersonId, created, persistedDate,
                uploadedDataId, hl7PartnerId, route, responsibleOrganisationIdentifier,
                pfSourcePersonId, pfSourceTeamId, pfSourceOrgId, pfAccessRoute, pfChangeDate, requireGeneral,
                requireSexualHealth, requireMentalHealth, requireSocialCare);
        this.start = start;
        this.end = end;
        this.commonId = commonId;
        this.externalId = externalId;
        this.typeCode = typeCode;
        this.typeCodingSystem = typeCodingSystem;
        this.status = status == null ? null : AppointmentStatus.valueOf(status);
    }

    public LocalDateTime getStart() {
        return start;
    }

    public LocalDateTime getEnd() {
        return end;
    }

    public AppointmentStatus getStatus() {
        return status;
    }

    public String getCommonId() {
        return commonId;
    }

    public String getExternalId() {
        return externalId;
    }

    public String getTypeCode() {
        return typeCode;
    }

    public String getTypeCodingSystem() {
        return typeCodingSystem;
    }

}
