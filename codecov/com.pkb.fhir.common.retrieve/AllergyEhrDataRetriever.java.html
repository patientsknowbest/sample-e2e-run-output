package com.pkb.fhir.common.retrieve;

import com.pkb.entities.enums.MenuDataType;
import com.pkb.fhir.common.decrypt.DecryptedEhrData;
import com.pkb.fhir.common.repository.MenuDataRepository;
import com.pkb.fhir.common.transform.TransformableDto;

import java.time.Instant;
import java.time.ZoneId;
import java.time.format.DateTimeFormatter;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;

public class AllergyEhrDataRetriever implements EhrDataRetriever {
    private static final DateTimeFormatter dateTimeFormatter = DateTimeFormatter.ofPattern("yyyy-MM-dd'T'hh:mm:ssXXX").withZone(ZoneId.systemDefault());
    private final MenuDataRepository menuDataRepository;

    public AllergyEhrDataRetriever(MenuDataRepository menuDataRepository) {
        this.menuDataRepository = menuDataRepository;
    }

    @Override
    public List<? extends TransformableDto> retrieveEhrDataDto(Map<Long, DecryptedEhrData> decryptedEhrDataMap) {
        List<AllergyEhrDataDto> allergies = menuDataRepository.findAllergies(decryptedEhrDataMap.keySet());
        return allergies.stream()
                .map(item -> createTransformableDto(item, decryptedEhrDataMap))
                .filter(Optional::isPresent)
                .map(Optional::get)
                .collect(Collectors.toList());
    }

    private String getNullSafe(Object o) {
        return o != null ? o.toString() : "";
    }

    private Optional<TransformableDto> createTransformableDto(AllergyEhrDataDto allergyEhrDataDto,
                                                              Map<Long, DecryptedEhrData> decryptedEhrData) {
        if (decryptedEhrData.containsKey(allergyEhrDataDto.getId())) {
            Map<String, Object> asMap = new HashMap<>();
            asMap.put("allergyId", getNullSafe(allergyEhrDataDto.getUniqueId()));
            asMap.put("assertedDate", dateFormatNullSafe(allergyEhrDataDto.getAssertedDate()));
            asMap.put("onsetDateTime", dateFormatNullSafe(allergyEhrDataDto.getOnsetDateTime()));
            asMap.put("patient", getNullSafe(allergyEhrDataDto.getPatientId()));
            asMap.put("recorder", getNullSafe(allergyEhrDataDto.getRecorderId()));
            return Optional.of(new TransformableDto(
                    asMap,
                    MenuDataType.allergy,
                    allergyEhrDataDto.getId(),
                    decryptedEhrData.get(allergyEhrDataDto.getId()).getContent())
            );
        } else {
            return Optional.empty();
        }

    }

    private Object dateFormatNullSafe(Instant date) {
        return date != null ? dateTimeFormatter.format(date) : "";
    }
}
