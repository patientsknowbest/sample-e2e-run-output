package com.pkb.fhir.common.retrieve;

import com.pkb.entities.enums.MenuDataType;
import com.pkb.fhir.common.decrypt.DecryptedEhrData;
import com.pkb.fhir.common.repository.SourceEhrDataRepository;
import com.pkb.fhir.common.transform.TransformableDto;

import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;

/**
 * Condition is the FHIR term. In our current app, these are Diagnoses.
 */
public class ConditionEhrDataRetriever implements EhrDataRetriever {

    private final SourceEhrDataRepository sourceEhrDataRepository;

    public ConditionEhrDataRetriever(SourceEhrDataRepository sourceEhrDataRepository) {
        this.sourceEhrDataRepository = sourceEhrDataRepository;
    }

    @Override
    public List<? extends TransformableDto> retrieveEhrDataDto(Map<Long, DecryptedEhrData> decryptedEhrDataMap) {
        List<ConditionEhrDataDto> conditions = sourceEhrDataRepository.findConditions(decryptedEhrDataMap.keySet());

        return conditions.stream()
                .map(item -> createTransformableDto(item, decryptedEhrDataMap))
                .filter(Optional::isPresent)
                .map(Optional::get)
                .collect(Collectors.toList());
    }

    private Optional<TransformableDto> createTransformableDto(ConditionEhrDataDto conditionEhrDataDto,
                                                              Map<Long, DecryptedEhrData> decryptedEhrData) {
        if (decryptedEhrData.containsKey(conditionEhrDataDto.getId())) {
            Map<String, Object> asMap = new HashMap<>();
            asMap.put("dto", conditionEhrDataDto);
            return Optional.of(new TransformableDto(
                    asMap,
                    MenuDataType.diagnosis,
                    conditionEhrDataDto.getId(),
                    decryptedEhrData.get(conditionEhrDataDto.getId()).getContent())
            );
        } else {
            return Optional.empty();
        }
    }
}
