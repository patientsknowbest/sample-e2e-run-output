package com.pkb.fhir.common.retrieve;

import com.pkb.entities.enums.Route;

import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;

public class PrivacyFlagsDto {

    private Long sourcePersonId;
    private Long sourceTeamId;
    private Long sourceOrgId;
    private Route accessRoute;
    private List<String> privacyFlags = new ArrayList<>();
    private LocalDateTime changeDate;

    PrivacyFlagsDto(Long sourcePersonId, Long sourceTeamId, Long sourceOrgId, Route accessRoute,
                    LocalDateTime changeDate, List<String> privacyFlags) {
        this.sourcePersonId = sourcePersonId;
        this.sourceTeamId = sourceTeamId;
        this.sourceOrgId = sourceOrgId;
        this.accessRoute = accessRoute;
        this.changeDate = changeDate;
        this.privacyFlags = privacyFlags;
    }

    public Long getSourcePersonId() {
        return sourcePersonId;
    }

    public Long getSourceTeamId() {
        return sourceTeamId;
    }

    public Long getSourceOrgId() {
        return sourceOrgId;
    }

    public Route getAccessRoute() {
        return accessRoute;
    }

    public LocalDateTime getChangeDate() {
        return changeDate;
    }

    public List<String> getPrivacyFlags() {
        return privacyFlags;
    }
}
