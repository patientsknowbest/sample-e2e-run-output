package com.pkb.fhir.common.retrieve;

import com.pkb.entities.enums.ConditionCategory;

import java.time.LocalDateTime;
import java.util.UUID;

/**
 * A diagnosis in PKB
 */
public class ConditionEhrDataDto extends EhrDataDto {

    private LocalDateTime start;
    private LocalDateTime end;
    private LocalDateTime asserted;
    private Long diagnosisCodingReceivedId;
    private ConditionCategory category;
    private Boolean patientHasCondition; //This exists solely so we can identify data in this state and purge it

    public ConditionEhrDataDto(long id, UUID patientId, UUID uniqueId, LocalDateTime start, LocalDateTime end, LocalDateTime asserted,
                               Long diagnosisCodingReceivedId, String category, Boolean patientHasCondition, Long menuDataSourceOrgId,
                               Long menuDataSourceTeamId, Long menuDataSourcePersonId, LocalDateTime created, LocalDateTime persistedDate,
                               Long uploadedDataId, Long hl7PartnerId, String route, String responsibleOrganisationIdentifier,
                               Long pfSourcePersonId, Long pfSourceTeamId, Long pfSourceOrgId, String pfAccessRoute, LocalDateTime pfChangeDate,
                               String requireGeneral, String requireSexualHealth, String requireMentalHealth, String requireSocialCare) {
        super(id, patientId, uniqueId, menuDataSourceOrgId,
                menuDataSourceTeamId, menuDataSourcePersonId, created, persistedDate,
                uploadedDataId, hl7PartnerId, route, responsibleOrganisationIdentifier,
                pfSourcePersonId, pfSourceTeamId, pfSourceOrgId, pfAccessRoute, pfChangeDate, requireGeneral,
                requireSexualHealth, requireMentalHealth, requireSocialCare);
        this.start = start;
        this.end = end;
        this.asserted = asserted;
        this.diagnosisCodingReceivedId = diagnosisCodingReceivedId;
        this.category = category == null ? null : ConditionCategory.valueOf(category);
        this.patientHasCondition = patientHasCondition;
    }

    public LocalDateTime getStart() {
        return start;
    }

    public LocalDateTime getEnd() {
        return end;
    }

    public LocalDateTime getAsserted() {
        return asserted;
    }

    public Long getDiagnosisCodingReceivedId() {
        return diagnosisCodingReceivedId;
    }

    public ConditionCategory getCategory() {
        return category;
    }

    public Boolean getPatientHasCondition() {
        return patientHasCondition;
    }
}
