package com.pkb.fhir.common.retrieve;

import com.pkb.entities.enums.MenuDataType;
import com.pkb.fhir.common.decrypt.BaseEncryptedEhrDataDecryptor;
import com.pkb.fhir.common.decrypt.DecryptedEhrData;
import com.pkb.fhir.common.repository.MenuDataRepository;
import com.pkb.fhir.common.repository.SourceEhrDataRepository;
import com.pkb.fhir.common.transform.TransformableDto;

import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;

public class AppointmentEhrDataRetriever implements EhrDataRetriever {

    private final SourceEhrDataRepository sourceEhrDataRepository;
    private final MenuDataRepository menuDataRepository;
    private final BaseEncryptedEhrDataDecryptor encryptedEhrDataDecryptor;

    public AppointmentEhrDataRetriever(SourceEhrDataRepository sourceEhrDataRepository, MenuDataRepository menuDataRepository,
                                       BaseEncryptedEhrDataDecryptor encryptedEhrDataDecryptor) {
        this.sourceEhrDataRepository = sourceEhrDataRepository;
        this.menuDataRepository = menuDataRepository;
        this.encryptedEhrDataDecryptor = encryptedEhrDataDecryptor;
    }

    @Override
    public List<? extends TransformableDto> retrieveEhrDataDto(Map<Long, DecryptedEhrData> decryptedEhrDataMap) {
        List<AppointmentEhrDataDto> appointments = sourceEhrDataRepository.findAppointments(decryptedEhrDataMap.keySet());

        return appointments.stream()
                .map(item -> createTransformableDto(item, decryptedEhrDataMap))
                .filter(Optional::isPresent)
                .map(Optional::get)
                .collect(Collectors.toList());
    }

    private Optional<TransformableDto> createTransformableDto(AppointmentEhrDataDto appointmentEhrDataDto,
                                                              Map<Long, DecryptedEhrData> decryptedEhrData) {
        if (decryptedEhrData.containsKey(appointmentEhrDataDto.getId())) {
            Map<String, Object> asMap = new HashMap<>();
            List<Long> appointmentRequestIds = menuDataRepository.findAllRequestIds(appointmentEhrDataDto.getId(), MenuDataType.appointmentRequest);
            List<DecryptedEhrData> appointmentRequests = appointmentRequestIds
                    .stream()
                    .map(encryptedEhrDataDecryptor::process)
                    .collect(Collectors.toList());
            asMap.put("dto", appointmentEhrDataDto);
            asMap.put("appointmentRequests", appointmentRequests);
            return Optional.of(new TransformableDto(
                    asMap,
                    MenuDataType.appointment,
                    appointmentEhrDataDto.getId(),
                    decryptedEhrData.get(appointmentEhrDataDto.getId()).getContent())
            );
        } else {
            return Optional.empty();
        }
    }
}
