package com.pkb.fhir.common.retrieve;

import com.pkb.entities.enums.Route;

import java.time.LocalDateTime;

/**
 * An encapsulation of all the source info off a menu_data row. Going to be used by the various menu_data dtos
 * we're constructing from the database.
 */
public class MenuDataSourceInfo {

    private Long menuDataSourceOrgId;
    private Long menuDataSourceTeamId;
    private Long menuDataSourcePersonId;
    private LocalDateTime created;
    private LocalDateTime persistedDate;
    private Long uploadedDataId;
    private Long hl7PartnerId;
    private Route route;
    private String responsibleOrganisationIdentifier;

    MenuDataSourceInfo(Long menuDataSourceOrgId, Long menuDataSourceTeamId, Long menuDataSourcePersonId, LocalDateTime created,
                       LocalDateTime persistedDate, Long uploadedDataId, Long hl7PartnerId, Route route, String responsibleOrganisationIdentifier) {
        this.menuDataSourceOrgId = menuDataSourceOrgId;
        this.menuDataSourceTeamId = menuDataSourceTeamId;
        this.menuDataSourcePersonId = menuDataSourcePersonId;
        this.created = created;
        this.persistedDate = persistedDate;
        this.uploadedDataId = uploadedDataId;
        this.hl7PartnerId = hl7PartnerId;
        this.route = route;
        this.responsibleOrganisationIdentifier = responsibleOrganisationIdentifier;
    }

    public Long getMenuDataSourceOrgId() {
        return menuDataSourceOrgId;
    }

    public Long getMenuDataSourceTeamId() {
        return menuDataSourceTeamId;
    }

    public Long getMenuDataSourcePersonId() {
        return menuDataSourcePersonId;
    }

    public LocalDateTime getCreated() {
        return created;
    }

    public LocalDateTime getPersistedDate() {
        return persistedDate;
    }

    public Long getUploadedDataId() {
        return uploadedDataId;
    }

    public Long getHl7PartnerId() {
        return hl7PartnerId;
    }

    public Route getRoute() {
        return route;
    }

    public String getResponsibleOrganisationIdentifier() {
        return responsibleOrganisationIdentifier;
    }
}
