package com.pkb.phplan.entity;

import com.pkb.annotation.EHRField;
import com.pkb.annotation.EHRField.QueryField;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.user.entity.PKBPerson;

import java.io.Serializable;
import java.util.Date;
import java.util.UUID;

/**
 * A dump of all plans created from a particular templates in coordinators account
 *
 * @author pravina
 */
public class PlanExportDTO implements IBaseDTO, Serializable {

    /**
     *
     */
    private static final long serialVersionUID = -3341081047699241100L;

    @EHRField(queryField = QueryField.id)
    private Long id;

    private PKBPerson exporter;

    @EHRField(queryField = QueryField.long02)
    private Long exporterId; // Alas, is null in some early exports

    private PHPlanTemplate template;

    @EHRField(queryField = QueryField.long01)
    private Long templateId; // Alas, is null in some early exports

    @EHRField(queryField = QueryField.date01)
    private Date exportDate;

    private byte[] content;

    @EHRField
    private String mediaType;

    @EHRField
    private String name;

    @EHRField(queryField = QueryField.documentMetadataId)
    private UUID documentMetadataId;

    private final DTOBaseFields baseFields = new DTOBaseFields();
    private final SourceDetails sourceDetails;

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected PlanExportDTO() {
        this.sourceDetails = new SourceDetails();
    }

    public PlanExportDTO(SourceDetails sourceDetails) {
        this.sourceDetails = sourceDetails;
    }

    public PKBPerson getExporter() {
        return exporter;
    }

    public void setExporter(PKBPerson exporter) {
        this.exporter = exporter;
    }

    public PHPlanTemplate getTemplate() {
        return template;
    }

    public void setTemplate(PHPlanTemplate template) {
        this.template = template;
    }

    public Date getExportDate() {
        return exportDate;
    }

    public void setExportDate(Date exportDate) {
        this.exportDate = exportDate;
    }

    public byte[] getContent() {
        return content;
    }

    public void setContent(byte[] content) {
        this.content = content;
    }

    @Override
    public Long getId() {
        return id;
    }

    @Override
    public void setId(Long id) {
        this.id = id;
    }

    public String getMediaType() {
        return mediaType;
    }

    public void setMediaType(String mediaType) {
        this.mediaType = mediaType;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public Long getExporterId() {
        return exporterId;
    }

    public void setExporterId(Long exporterId) {
        this.exporterId = exporterId;
    }

    public Long getTemplateId() {
        return templateId;
    }

    public void setTemplateId(Long templateId) {
        this.templateId = templateId;
    }

    public UUID getDocumentMetadataId() {
        return documentMetadataId;
    }

    public void setDocumentMetadataId(UUID documentMetadataId) {
        this.documentMetadataId = documentMetadataId;
    }

    @Override
    public SourceDetails getSource() {
        return sourceDetails;
    }

    @Override
    public DTOBaseFields getBaseFields() {
        return baseFields;
    }
}
