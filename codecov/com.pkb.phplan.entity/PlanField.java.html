package com.pkb.phplan.entity;

import java.io.Serializable;

import com.pkb.annotation.EHRField;
import com.pkb.phplan.entity.PHPlanTemplate.TemplateFieldLabel;

/**
 * A field in a personal health plan
 *
 * @author pravinam
 *
 */
public class PlanField implements Serializable {

    /**
     *
     */
    private static final long serialVersionUID = 1L;

    public static final String PHPLAN_FIELD_LABEL = "label";

    public static final String PHPLAN_FIELD_TEXT = "text";

    public static final String PHPLAN_FIELD_SEQUENCE = "sequence";

    @EHRField
    private Long id;

    @EHRField
    private Long planTemplateId;

    @EHRField
    private TemplateFieldLabel label;

    @EHRField
    private String text;

    @EHRField
    private Long sequence;

    public PlanField() {

    }

    public PlanField(Long planTemplateId, TemplateField templateField) {
        this.planTemplateId = planTemplateId;
        this.label = templateField.getLabel();
        this.sequence = (long) templateField.getSequence();
        this.text = templateField.getText();
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public TemplateFieldLabel getLabel() {
        return label;
    }

    public void setLabel(TemplateFieldLabel label) {
        this.label = label;
    }

    public String getText() {
        return text;
    }

    public void setText(String text) {
        this.text = text;
    }

    public Long getSequence() {
        return sequence;
    }

    public void setSequence(Long sequence) {
        this.sequence = sequence;
    }

    public Long getPlanTemplateId() {
        return planTemplateId;
    }

    public void setPlanTemplateId(Long planTemplateId) {
        this.planTemplateId = planTemplateId;
    }

    @Override
    public String toString() {
        StringBuilder stringBuffer = new StringBuilder();
        stringBuffer.append("PlanField[");
        stringBuffer.append("label:" + label + ", ");
        stringBuffer.append("planTemplateId:" + planTemplateId);
        stringBuffer.append("]");
        return stringBuffer.toString();
    }

}
