package com.pkb.phplan.entity;

import com.pkb.annotation.EHRField;
import com.pkb.annotation.EHRField.QueryField;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.document.entity.Attachment;
import com.pkb.entities.enums.MenuDataType;

import javax.persistence.Transient;
import java.io.Serializable;
import java.util.Date;
import java.util.UUID;

/**
 * Class represent an attachment to a PH plan
 *
 * @author pravinam
 */
//@EHRMigration(migratorClass=)
public class PHPlanAttachment implements IBaseDTO, Serializable {

    public static final MenuDataType MS_PATH = MenuDataType.phPlanAttachment; // was "pkb:attachment";

    public static final String ATTACHMENT_ID = "attachmentId";
    public static final String ATTACHMENT_PLAN_ID = "planId";
    public static final String ATTACHMENT_LABEL = "label";
    public static final String ATTACHMENT_NAME = "name";
    //    public static final String MS_PLAN_ATTACHMENT_LIST = "pkb:attachments:all";

    private static final long serialVersionUID = 1L;

    @EHRField(queryField = QueryField.id)
    private Long id;

    @EHRField(queryField = QueryField.long01)
    private Long attachmentId;

    @EHRField(queryField = QueryField.long02)
    private Long planId;

    @EHRField(queryField = QueryField.accountId)
    private long accountId;

    @EHRField
    private String name;

    @EHRField
    private String label;

    @EHRField
    private Boolean carePlanMainContent;

    @Transient
    private Attachment attachment;

    private final DTOBaseFields baseFields = new DTOBaseFields();
    private final SourceDetails sourceDetails;

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected PHPlanAttachment() {
        this.sourceDetails = new SourceDetails();
    }

    public PHPlanAttachment(SourceDetails sourceDetails) {
        this.sourceDetails = sourceDetails;
    }

    // utility methods

    public boolean isBrowserImage() {
        return attachment.isBrowserImage();
    }

    public boolean isPdf() {
        return attachment.isPdf();
    }

    public boolean isConvertibleToPdf() {
        return attachment.isConvertibleToPdf();
    }

    public boolean isViewable() {
        return attachment.isViewable();
    }

    public boolean isStreamable() {
        return attachment.isStreamable();
    }

    @Override
    public Long getId() {
        return id;
    }

    @Override
    public void setId(Long id) {
        this.id = id;
    }

    public Long getAttachmentId() {
        return attachmentId;
    }

    public void setAttachmentId(Long attachmentId) {
        this.attachmentId = attachmentId;
    }

    public void setAccountId(long accountId) {
        this.accountId = accountId;
    }

    public Long getAccountId() {
        return this.accountId;
    }

    public Long getPlanId() {
        return planId;
    }

    public void setPlanId(Long planId) {
        this.planId = planId;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getLabel() {
        return label;
    }

    public void setLabel(String label) {
        this.label = label;
    }

    public Attachment getAttachment() {
        return attachment;
    }

    public void setAttachment(Attachment attachment) {
        this.attachment = attachment;
    }

    // for Attachment fields

    public UUID getDocMetadataId() {
        return attachment.getDocMetadataId();
    }

    public Long getUserId() {
        return attachment.getUserId();
    }

    public String getFilename() {
        return attachment.getFilename();
    }

    public String getMediaType() {
        return attachment.getMediaType();
    }

    public Date getUploadTime() {
        return attachment.getUploadTime();
    }

    public byte[] getContent() {
        return attachment.getContent();
    }

    public byte[] getThumbnailContent() {
        return attachment.getThumbnailContent();
    }

    @Override
    public SourceDetails getSource() {
        return sourceDetails;
    }

    @Override
    public DTOBaseFields getBaseFields() {
        return baseFields;
    }

    public boolean getCarePlanMainContent() {
        return carePlanMainContent != null && carePlanMainContent;
    }

    public void setCarePlanMainContent(Boolean carePlanMainContent) {
        this.carePlanMainContent = carePlanMainContent;
    }
}
