package com.pkb.phplan.entity;

import com.pkb.consent.entity.PrivacyFlags;
import com.pkb.datamodel.consent.RequiresConsent;
import com.pkb.institute.entity.Team;
import org.jetbrains.annotations.Nullable;

import javax.persistence.CascadeType;
import javax.persistence.Column;
import javax.persistence.Embedded;
import javax.persistence.Entity;
import javax.persistence.EnumType;
import javax.persistence.Enumerated;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.OneToMany;
import javax.persistence.Table;
import java.io.Serializable;
import java.util.ArrayList;
import java.util.Date;
import java.util.List;

/**
 * This class represents the template from which personal health plan will be
 * created Template itself will be created by the institute coordinator
 *
 * @author pravinam
 *
 */
@Entity
@Table
public class PHPlanTemplate implements Serializable, RequiresConsent {

    private static final long serialVersionUID = 1L;

    public enum TemplateFieldLabel {

        ACTION_PLAN, GREEN, AMBER, RED
    }

    public enum TemplateStatus {

        ACTIVE, INDRAFT, INACTIVE
    }

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private Long id;

    private String name;

    @OneToMany(fetch = FetchType.LAZY, mappedBy = "planTemplate", cascade = CascadeType.ALL)
    private List<TemplateAttachment> attachments;

    @Column(name = "creator_id")
    private Long creatorId;

    @Column
    private Date creationDate;

    @Column
    private Long reviewPeriodDays;

    @Column
    @Enumerated(EnumType.STRING)
    private TemplateStatus status;

    @OneToMany(fetch = FetchType.LAZY, mappedBy = "planTemplate", cascade = CascadeType.ALL)
    private List<TemplateField> fields;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "institute_id")
    private Team institute;

    @OneToMany(fetch = FetchType.LAZY, mappedBy = "planTemplate", cascade = CascadeType.ALL)
    private List<TemplateMetric> metrics;

    @Embedded
    private PrivacyFlags privacyFlags;

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public List<TemplateAttachment> getAttachments() {
        return attachments;
    }

    public void setAttachments(List<TemplateAttachment> attachments) {
        this.attachments = attachments;
    }

    public Date getCreationDate() {
        return creationDate;
    }

    public void setCreationDate(Date creationDate) {
        this.creationDate = creationDate;
    }

    public TemplateStatus getStatus() {
        return status;
    }

    public void setStatus(TemplateStatus status) {
        this.status = status;
    }

    public List<TemplateField> getFields() {
        return fields;
    }

    public void addField(TemplateField field) {
        if (fields == null) {
            fields = new ArrayList<>();
        }
        field.setPlanTemplate(this);
        fields.add(field);
    }

    public void setFields(List<TemplateField> fields) {
        this.fields = fields;
    }

    public void addAttachment(TemplateAttachment attachment) {
        if (attachments == null) {
            attachments = new ArrayList<>();
        }
        attachment.setPlanTemplate(this);
        attachments.add(attachment);
    }

    public @Nullable PHPlanTemplate duplicate() {
        return null;
    }

    public Team getInstitute() {
        return institute;
    }

    public void setInstitute(Team institute) {
        this.institute = institute;
    }

    public List<TemplateMetric> getMetrics() {
        return metrics;
    }

    public void setMetrics(List<TemplateMetric> metrics) {
        this.metrics = metrics;
        for (TemplateMetric metric : metrics) {
            metric.setPlanTemplate(this);
        }
    }

    public Long getReviewPeriodDays() {
        return reviewPeriodDays;
    }

    public void setReviewPeriodDays(Long reviewPeriodDays) {
        this.reviewPeriodDays = reviewPeriodDays;
    }

    public @Nullable TemplateField getTemplateField(TemplateFieldLabel label) {
        for (TemplateField field : fields) {
            if (field.getLabel() == label) {
                return field;
            }
        }
        return null;
    }

    public PrivacyFlags getPrivacyFlags() {
        return privacyFlags;
    }

    public void setPrivacyFlags(PrivacyFlags privacyFlags) {
        this.privacyFlags = privacyFlags;
    }

    @Override
    public boolean getRequireGeneral() {
        return privacyFlags.getRequireGeneral();
    }

    @Override
    public void setRequireGeneral(boolean requireGeneral) {
        privacyFlags.setRequireGeneral(requireGeneral);
    }

    @Override
    public boolean getRequireSexualHealth() {
        return privacyFlags.getRequireSexualHealth();
    }

    @Override
    public void setRequireSexualHealth(boolean requireSexualHealth) {
        privacyFlags.setRequireSexualHealth(requireSexualHealth);
    }

    @Override
    public boolean getRequireMentalHealth() {
        return privacyFlags.getRequireMentalHealth();
    }

    @Override
    public void setRequireMentalHealth(boolean requireMentalHealth) {
        privacyFlags.setRequireMentalHealth(requireMentalHealth);
    }

    @Override
    public boolean getRequireSocialCare() {
        return privacyFlags.getRequireSocialCare();
    }

    @Override
    public void setRequireSocialCare(boolean requireSocialCare) {
        privacyFlags.setRequireSocialCare(requireSocialCare);
    }

    public Long getCreatorId() {
        return creatorId;
    }

    public void setCreatorId(Long creatorId) {
        this.creatorId = creatorId;
    }
}
