package com.pkb.backgroundtasks.entity

import com.fasterxml.jackson.annotation.JsonTypeInfo
import com.fasterxml.jackson.annotation.JsonTypeInfo.Id.CUSTOM
import com.fasterxml.jackson.databind.DatabindContext
import com.fasterxml.jackson.databind.JavaType
import com.fasterxml.jackson.databind.annotation.JsonTypeIdResolver
import com.fasterxml.jackson.databind.jsontype.impl.TypeIdResolverBase
import com.pkb.backgroundtasks.AccountPrivateId
import com.pkb.backgroundtasks.service.taskexecutors.AccountTaskExecutor
import com.pkb.backgroundtasks.service.taskexecutors.TaskExecutorRegistry
import com.pkb.crypto.dto.AccountKeysDTO

/**
 *  Subclasses of this represent the details of a specific type of task to perform against an account.
 *  It is intended that there should be a 1-1-1 relationship between enum values in AccountTaskType, subclasses
 *  of AccountTaskExecutor and subclasses of this class. This is effectively enforced by having AccountTaskType values
 *  linked to subclasses of this class, and then giving AccountTaskExecutor a type parameter for subclasses of this class.
 *  (This would be a lot neater with sealed classes but there's no standard support for mapping Kotlin sealed classes to
 *  postgres enums. If I get round to writing that support this could be tidied up considerably because the "enum" values
 *  could be given a type parameter that was a subclass of this class
 *
 *  Both this and AccountTaskExecutor use self types in an effort to try and enforce as much type-safety as is possible
 *  given that these are being deserialised from JSON. It means that the subclasses of AccountTaskExecutor can define a
 *  performTask method that takes the appropriate subclass of this class as a parameter, rather than a generic
 *  superclass that then needs a cast.
 *
 */
@JsonTypeInfo(use = CUSTOM, include = JsonTypeInfo.As.PROPERTY, property = "type", visible = true)
@JsonTypeIdResolver(AccountTaskDetails.TypeIdResolver::class)
abstract class AccountTaskDetails<Self : AccountTaskDetails<Self>> constructor(val type: AccountTaskType) {
    init {
        if (type.detailsClass != this.javaClass) {
            throw IllegalStateException("AccountTaskDetails subclass ${this.javaClass} intitialised with type $type that doesn't reference it as task details class")
        }
    }


    protected abstract fun getSelf(): Self

    fun triggerTask(taskExecutors: TaskExecutorRegistry, pendingTask: AccountTaskPending, accountKeysDTO: AccountKeysDTO, accountPrivateId: AccountPrivateId): Boolean {
        val taskExecutor: AccountTaskExecutor<*, Self> = getTaskExecutor(taskExecutors)!!
        return taskExecutor.performTask(pendingTask.accountPublicId, accountKeysDTO, accountPrivateId, getSelf())
    }

    fun getTaskExecutor(taskExecutors: TaskExecutorRegistry): AccountTaskExecutor<*, Self>? =
        taskExecutors.taskExecutorFor(getSelf())

    override fun equals(other: Any?): Boolean {
        if (this === other) return true
        if (javaClass != other?.javaClass) return false

        return true
    }

    override fun hashCode(): Int = javaClass.hashCode()

    /**
     * Jackson's polymorphic deserialisation requires you to either use class names as the type parameter (fragile for persistent data)
     * or specify a mapping in an annotation on the superclass. But we're effectively already maintaining this mapping on AccountTaskType,
     * so we can use a custom resolver to look up the appropriate subclass.
     */
    class TypeIdResolver : TypeIdResolverBase() {
        override fun idFromValue(value: Any): String {
            if (value !is AccountTaskDetails<*>) {
                throw IllegalStateException("Custom type resolver for AccountTaskDetails used for wrong class ${value.javaClass}")
            }
            return value.type.toString()
        }

        override fun typeFromId(context: DatabindContext, id: String): JavaType {
            return context.typeFactory.constructSimpleType(AccountTaskType.valueOf(id).detailsClass, null)
        }

        override fun idFromValueAndType(value: Any, suggestedType: Class<*>): String = idFromValue(value)

        override fun getMechanism(): JsonTypeInfo.Id = CUSTOM

    }
}
