package com.pkb.backgroundtasks.entity

import com.pkb.backgroundtasks.OrgPrivateId
import com.pkb.backgroundtasks.PersonPublicId
import com.pkb.common.spring.data.EntityWithUUIDBase
import org.hibernate.annotations.Immutable
import org.springframework.data.annotation.CreatedDate
import org.springframework.data.jpa.domain.support.AuditingEntityListener
import java.time.Instant
import java.util.UUID
import javax.persistence.Column
import javax.persistence.Entity
import javax.persistence.EntityListeners
import javax.persistence.Id
import javax.persistence.Table


@Immutable
@Entity
@Table(schema = "bgtasks", name = "org_access_task_pending")
@EntityListeners(AuditingEntityListener::class)
data class OrgAccessTaskPending(
    @Id
    override val id: UUID,

    @Column(name = "org_id")
    val orgId: OrgPrivateId,

    @Column(name = "person_public_id")
    val personPublicId: PersonPublicId

) : EntityWithUUIDBase() {


    @CreatedDate
    @Column(name = "created_date", updatable = false)
    lateinit var createdDate: Instant
        private set

}
