package com.pkb.backgroundtasks.entity

import com.pkb.backgroundtasks.AccountPublicId
import com.pkb.backgroundtasks.projection.TaskStatus
import com.pkb.common.spring.data.EntityWithUUIDBase
import com.vladmihalcea.hibernate.type.json.JsonBinaryType
import org.hibernate.annotations.Immutable
import org.hibernate.annotations.Type
import org.hibernate.annotations.TypeDef
import org.springframework.data.annotation.CreatedDate
import org.springframework.data.jpa.domain.support.AuditingEntityListener
import java.time.Instant
import java.util.UUID
import javax.persistence.Column
import javax.persistence.EntityListeners
import javax.persistence.Id
import javax.persistence.MappedSuperclass

@Immutable
@MappedSuperclass
@EntityListeners(AuditingEntityListener::class)
@TypeDef(name = "jsonb", typeClass = JsonBinaryType::class)
abstract class AccountTaskCommon(
    @Id
    override val id: UUID,

    @Column(name = "account_public_id")
    val accountPublicId: AccountPublicId,

    @Column(nullable = false, updatable = false, name = "task_details")
    @Type(type = "jsonb")
    internal val taskDetails: AccountTaskDetails<*>
) : EntityWithUUIDBase() {

    abstract fun toTaskStatus(): TaskStatus

    @CreatedDate
    @Column(name = "created_date", nullable = false, updatable = false)
    lateinit var createdDate: Instant
        protected set

    val type: AccountTaskType
        get() = taskDetails.type

}
