package com.pkb.backgroundtasks.entity

import com.pkb.backgroundtasks.AccountPublicId
import com.pkb.backgroundtasks.projection.TaskStatus
import com.pkb.backgroundtasks.service.TaskResult
import org.hibernate.annotations.Immutable
import org.hibernate.annotations.Type
import org.springframework.data.jpa.domain.support.AuditingEntityListener
import java.time.Instant
import java.util.UUID
import javax.persistence.Column
import javax.persistence.Entity
import javax.persistence.EntityListeners
import javax.persistence.EnumType
import javax.persistence.Enumerated
import javax.persistence.Table

@Immutable
@Entity
@Table(schema = "bgtasks", name = "account_task_complete_v2")
@EntityListeners(AuditingEntityListener::class)
class AccountTaskComplete(
    id: UUID,
    accountPublicId: AccountPublicId,
    taskDetails: AccountTaskDetails<*>,

    @Column(name = "task_created_date")
    val taskCreatedDate: Instant,

    @Column(name = "task_result")
    @Enumerated(EnumType.STRING)
    @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
    val taskResult: TaskResult,

    @Column(name = "request_id")
    val requestId: String



) : AccountTaskCommon(id, accountPublicId, taskDetails) {
    constructor(pendingTask: AccountTaskPending, result: TaskResult, requestId: String) : this(pendingTask.id, pendingTask.accountPublicId, pendingTask.taskDetails, pendingTask.createdDate, result, requestId)

    internal constructor(pendingTask: AccountTaskPending, result: TaskResult, requestId: String, completionDate:Instant): this(pendingTask, result, requestId) {
        createdDate = completionDate
    }

    override fun toTaskStatus(): TaskStatus =
        TaskStatus(id, taskDetails, taskCreatedDate, taskResult, createdDate)

    override fun toString(): String {
        return "AccountTaskComplete(id=$id, accountPublicId=$accountPublicId, createdDate=$createdDate, type=$type, taskCreatedDate=$taskCreatedDate, taskResult=$taskResult)"
    }
}
