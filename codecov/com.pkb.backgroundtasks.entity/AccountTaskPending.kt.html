package com.pkb.backgroundtasks.entity

import com.pkb.backgroundtasks.AccountPrivateId
import com.pkb.backgroundtasks.AccountPublicId
import com.pkb.backgroundtasks.projection.TaskStatus
import com.pkb.backgroundtasks.service.taskexecutors.TaskExecutorRegistry
import com.pkb.crypto.dto.AccountKeysDTO
import java.time.Instant
import java.util.UUID
import javax.persistence.Entity
import javax.persistence.Table

@Entity
@Table(schema = "bgtasks", name = "account_task_pending_v2")
class AccountTaskPending(
    id: UUID,
    accountPublicId: AccountPublicId,
    taskDetails: AccountTaskDetails<*>
) : AccountTaskCommon(id, accountPublicId, taskDetails) {

    constructor(
        id: UUID,
        accountPublicId: AccountPublicId,
        taskDetails: AccountTaskDetails<*>,
        createdDate: Instant
    ) : this(id, accountPublicId, taskDetails) {
        this.createdDate = createdDate
    }

    fun getTaskExecutor(taskExecutors: TaskExecutorRegistry) = taskDetails.getTaskExecutor(taskExecutors)

    fun trigger(
        taskExecutors: TaskExecutorRegistry,
        accountKeysDTO: AccountKeysDTO,
        accountPrivateId: AccountPrivateId
    ): Boolean {
        return taskDetails.triggerTask(taskExecutors, this, accountKeysDTO, accountPrivateId)
    }

    fun duplicates(other: AccountTaskPending?): Boolean {
        return other != null && other.accountPublicId == accountPublicId
                && other.taskDetails == taskDetails
    }

    override fun toTaskStatus(): TaskStatus =
        TaskStatus(id, taskDetails, createdDate, null, null)

    override fun toString(): String {
        return "AccountTaskComplete(id=$id, accountPublicId=$accountPublicId, createdDate=$createdDate, type=$type)"
    }
}
