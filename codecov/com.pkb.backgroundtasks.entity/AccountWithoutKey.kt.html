package com.pkb.backgroundtasks.entity

import com.pkb.common.spring.data.EntityWithUUIDBase
import org.hibernate.annotations.Immutable
import java.util.UUID
import javax.persistence.Entity
import javax.persistence.Id
import javax.persistence.Table

/**
 * Represents an account for which we (temporarily, we hope)
 * do not have a key available in the KMS. Caching this in the
 * database allows us to efficiently skip these when running through the
 * backlog of account tasks to execute.
 */
@Immutable
@Entity
@Table(schema = "bgtasks", name = "account_without_key")
data class AccountWithoutKey (
    @Id
    override val id: UUID,
) : EntityWithUUIDBase()
