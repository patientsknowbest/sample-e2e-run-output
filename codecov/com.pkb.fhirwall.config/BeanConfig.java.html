import static ca.uhn.fhir.rest.api.Constants.HEADER_AUTHORIZATION;
import static ca.uhn.fhir.rest.api.EncodingEnum.JSON;
import static ca.uhn.fhir.rest.client.api.ServerValidationModeEnum.NEVER;
import static io.jsonwebtoken.SignatureAlgorithm.RS512;
import static io.jsonwebtoken.security.Keys.keyPairFor;
import static org.keycloak.OAuth2Constants.CLIENT_CREDENTIALS;

@Configuration
public class BeanConfig {

    private static final boolean LOG_ERRORS = false;

    @Bean
    public AccessRequestParser accessRequestParser() {
        return new AccessRequestParser();
    }

    @Bean
    public AuthzMapper authzMapper(PatientRepository patientRepository,ConsentRepository consentRepository) {
        return new AuthzMapper(patientRepository, consentRepository);
    }

    @Bean
    public PolicyInformationPoint authzDataService(PatientRepository patientRepository, AuthzMapper authzMapper) {
        return new FhirPolicyInformationPoint(patientRepository, authzMapper);
    }

    @Bean
    public AuthzFilterMapper authzFilterMapper() {
        return new AuthzFilterMapper();
    }

    @Bean
    public AuthorizationService authorizationService(AuthzFilterMapper authzFilterMapper, Oso oso) {
        return new AuthorizationService(authzFilterMapper, oso);
    }

    @Bean
    public RequestTransformer requestTransformer(AccessLogger accessLogger) {
        return new RequestTransformer(accessLogger);
    }

    @Bean
    public RequestForwarder requestForwarder(ResourceRepository resourceRepository) {
        return new RequestForwarder(resourceRepository);
    }

    @Bean
    public ResponseTransformer responseTransformer(FhirContext fhirContext, ResourceEnricherManager resourceEnricherManager,
                                                   ResourceRemovalManager resourceRemovalManager) {
        return new ResponseTransformer(fhirContext, resourceEnricherManager, resourceRemovalManager);
    }

    @Bean
    public AccessLogger accessLogger() {
        return new AccessLogger();
    }

    @Bean
    public DateTimeService dateTimeService(FhirwallConfig config) {
        if (config.isFakeDateTimeServiceEnabled()) {
            return new FakeDateTimeService();
        }

        return new DefaultDateTimeService();
    }

    @Bean
    public ConfigStorage configStorage(ConfigurableEnvironment environment, RefreshScope refreshScope) {
        return new SpringBootConfigStorage(environment, refreshScope);
    }

    @Bean
    public PubSubNamespaceService pubSubNamespaceService(CamelContext camelContext) {
        return new DefaultPubSubNamespaceService(camelContext);
    }

    @Bean(name = "monitoredConnectionPool")
    public ConnectionPool monitoredConnectionPool(OkHttpConfig config, MeterRegistry registry) {
        var pool = new ConnectionPool(config.getMaxIdleConnections(), 5, TimeUnit.MINUTES);
        new OkHttpConnectionPoolMetrics(pool).bindTo(registry);
        return pool;
    }

    @Bean
    public OkHttpClient httpClient(
            OkHttpConfig config,
            MeterRegistry registry,
            @Qualifier("monitoredConnectionPool") ConnectionPool connectionPool) {

        return new OkHttpClient.Builder()
                .eventListener(OkHttpMetricsEventListener.builder(registry, "okhttp.requests").build())
                .connectTimeout(config.getConnectTimeout())
                .readTimeout(config.getReadTimeout())
                .writeTimeout(config.getWriteTimeout())
                .retryOnConnectionFailure(config.isRetryOnConnectionFailure())
                .connectionPool(connectionPool)
                // PHR-11034 OKHttp has issues with http/2 and keepalives
                // https://github.com/square/okhttp/issues/6700
                .protocols(List.of(Protocol.HTTP_1_1))
                .build();
    }

    @Bean
    public FhirContext fhirContext(OkHttpClient httpClient) {
        var context = FhirContext.forR4();
        context.setParserErrorHandler(new LenientErrorHandler(LOG_ERRORS));
        MonitoredOkHttpRestfulClientFactory clientFactory = new MonitoredOkHttpRestfulClientFactory(context, httpClient);
        clientFactory.setServerValidationMode(NEVER);
        context.setRestfulClientFactory(clientFactory);
        return context;
    }

    @Bean
    public RequestValidator requestValidator(FhirwallConfig fhirwallConfig, FhirContext fhirContext) {
        return new RequestValidator(fhirwallConfig.getSupportedResources(), fhirwallConfig.getUnsupportedParameters(), fhirContext);
    }

    @Bean
    public Keycloak keycloakClient(KeycloakConfig keycloakConfig) {
        return KeycloakBuilder.builder()
                .serverUrl(keycloakConfig.getAuthServerUrl())
                .realm(keycloakConfig.getRealm())
                .grantType(CLIENT_CREDENTIALS)
                .clientId(keycloakConfig.getResource())
                .clientSecret(keycloakConfig.getCredentials().getSecret())
                .build();
    }

    @Bean
    @Profile("!integration-test")
    public IGenericClient fhirStoreClient(FhirwallConfig fhirwallConfig,
                                          FhirContext fhirContext,
                                          Keycloak keycloakClient) {
        var client = fhirContext.newRestfulGenericClient(fhirwallConfig.getFhirwallTarget() + "/fhir");
        client.setEncoding(JSON);
        client.registerInterceptor(new IClientInterceptor() {
            @Override
            public void interceptRequest(IHttpRequest theRequest) {
                theRequest.addHeader(HEADER_AUTHORIZATION, "Bearer " + keycloakClient.tokenManager().getAccessTokenString());
            }
            @Override
            public void interceptResponse(IHttpResponse theResponse) throws IOException {
            }
        });

        return client;
    }
    @Bean
    @Profile("integration-test")
    public IGenericClient fhirStoreClientBasicAuth(FhirwallConfig fhirwallConfig,
                                                   FhirContext fhirContext,
                                                   AidboxConfig aidboxConfig) {
        var client = fhirContext.newRestfulGenericClient(fhirwallConfig.getFhirwallTarget() + "/fhir");
        client.setEncoding(JSON);
        client.registerInterceptor(new BasicAuthInterceptor(aidboxConfig.getClientId(), aidboxConfig.getClientPassword()));
        return client;
    }

    @Bean
    public NhsLoginSsoTokenValidatorAppointment nhsLoginSsoTokenValidatorAppointment(DateTimeService dateTimeService,
                                                                                     Supplier<PublicKey> keyProvider,
                                                                                     FhirwallConfig fhirwallConfig) {
        return new NhsLoginSsoTokenValidatorAppointment(dateTimeService, keyProvider, fhirwallConfig);
    }

    @Bean
    public DateTimeService dateTimeService() {
        return new DefaultDateTimeService();
    }

    @Bean
    @Profile("!integration-test")
    public Supplier<PublicKey> nhsPublicKeySupplier(FhirwallConfig fhirwallConfig) {
        return new NhsPublicKeySupplier(fhirwallConfig);
    }

    @Bean
    @Profile("integration-test")
    public Supplier<PublicKey> fakeKeySupplier(KeyPair keyPair) {
        return keyPair::getPublic;
    }

    @Bean
    @Profile("integration-test")
    public KeyPair keyPair() {
        return keyPairFor(RS512);
    }

    @Bean
    public AppointmentResourceEnricher appointmentResourceEnricher(ResourceRepository resourceRepository, FhirwallConfig config) {
        return new AppointmentResourceEnricher(resourceRepository, config);
    }

    @Bean
    public ResourceEnricherManager resourceEnricherManager(AppointmentResourceEnricher appointmentResourceEnricher,
                                                           FhirContext fhirContext, BundleService bundleService) {
        return new ResourceEnricherManager(appointmentResourceEnricher, fhirContext, bundleService);
    }

    @Bean
    public ResourceRemovalManager resourceRemovalManager(AppointmentResourceRemover appointmentResourceRemover,
                                                         BundleService bundleService) {
        return new ResourceRemovalManager(appointmentResourceRemover, bundleService);
    }

    @Bean
    public AppointmentResourceRemover appointmentResourceRemover(FhirwallConfig config) {
        return new AppointmentResourceRemover(config);
    }
    @Bean
    public FhirMetrics fhirMetrics() {
        return new FhirMetrics();
    }

    @Bean
    public BundleService bundleService(FhirContext fhirContext) {
        return new BundleService(fhirContext);
    }
}
