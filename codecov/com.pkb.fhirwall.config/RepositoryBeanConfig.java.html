package com.pkb.fhirwall.config;

import ca.uhn.fhir.context.FhirContext;
import ca.uhn.fhir.rest.client.api.IGenericClient;
import com.pkb.fhirwall.domain.ConsentRepository;
import com.pkb.fhirwall.domain.PatientRepository;
import com.pkb.fhirwall.domain.ResourceRepository;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class RepositoryBeanConfig {

    @Bean
    public ResourceRepository resourceRepository(IGenericClient fhirStoreClient,
            FhirContext fhirContext) {
        return new ResourceRepository(fhirStoreClient, fhirContext);
    }

    @Bean
    public PatientRepository patientRepository(ResourceRepository resourceRepository, FhirContext context) {
        return new PatientRepository(resourceRepository, context);
    }

    @Bean
    public ConsentRepository consentRepository(ResourceRepository resourceRepository) {
        return new ConsentRepository(resourceRepository);
    }
}
