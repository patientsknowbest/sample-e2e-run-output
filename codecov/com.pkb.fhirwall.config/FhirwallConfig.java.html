package com.pkb.fhirwall.config;

import com.pkb.common.config.BaseConfig;
import org.apache.commons.lang3.StringUtils;

import java.util.List;
import java.util.Set;

import static java.util.Arrays.stream;
import static java.util.stream.Collectors.toList;

public interface FhirwallConfig extends BaseConfig {

    default Set<String> getSupportedResources() {
        return Set.of(getConfigStorage().getString("app.supportedResources", "Appointment").split(","));
    }

    default Set<String> getUnsupportedParameters() {
        return Set.of(getConfigStorage().getString("app.unsupportedParameters", "_include,_revinclude,_type").split(","));
    }

    default String getFhirwallTarget() {
        return getConfigStorage().getString("app.fhirwallTarget", "http://localhost:8888");
    }

    default String getNhsTokenIss() {
        return getConfigStorage().getString("app.nhsTokenIss", "https://auth.login.nhs.uk");
    }

    default String getNhsJwksUri() {
        return getConfigStorage().getString("app.nhsJwksUri", "https://auth.login.nhs.uk/.well-known/jwks.json");
    }

    default String getWayfinderHost() {
        return getConfigStorage().getString("app.wayfinderHost", "localhost");
    }

    default Set<String> getAllowedOrigins() {
        return Set.of(getConfigStorage().getString("app.allowedOrigins", "").split(","));
    }

    default List<String> getWayfinderOnboardedOdsCodes() {
        String wayfinderOnboardedOdsCodes = getConfigStorage().getString("app.wayfinder.onboardedodscodes", "");

        return stream(wayfinderOnboardedOdsCodes.split(","))
                .filter(StringUtils::isNotBlank)
                .collect(toList());
    }

    default String getAppointmentPagePath() {
        return getConfigStorage().getString("phr.appointmentpagepath", "/diary/viewAppointment.action");
    }

    default String getPhrBaseUrl() {
        return getConfigStorage().getString("phr.baseUrl", "http://unset.phrbaseurl.pkb.com");
    }
}
