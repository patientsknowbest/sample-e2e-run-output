package com.pkb.fhirwall.config;

import com.osohq.oso.Oso;
import com.pkb.fhirwall.service.auth.oso.AppointmentAuth;
import com.pkb.fhirwall.service.auth.oso.PatientAuth;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.io.IOException;
import java.net.URISyntaxException;

@Configuration
public class AuthzConfig {

    @Bean
    public Oso oso() throws IOException, URISyntaxException {
        var oso = new Oso();

        oso.registerClass(PatientAuth.class, "Patient");
        oso.registerClass(AppointmentAuth.class, "Appointment");
        oso.loadFilesFromResources("/oso/authorization.polar");

        return oso;
    }
}
