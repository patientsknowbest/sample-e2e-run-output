package com.pkb.service.questionnaire;

import com.google.common.collect.Iterators;
import com.pkb.app.PrivacyFlagsToPrivacyFlagConverter;
import com.pkb.entities.enums.PrivacyFlag;
import com.pkb.institute.entity.Team;
import com.pkb.questionnaireservice.modelv2.Questionnaire;
import com.pkb.questionnaireservice.modelv2.QuestionnaireComponent;
import com.pkb.questionnaireservice.modelv2.QuestionnaireComponentType;
import com.pkb.questionnaireservice.modelv2.QuestionnaireTeam;
import com.pkb.questionnaireservice.modelv2.QuestionnaireTeamImpl;
import com.pkb.user.entity.PKBPerson;
import io.vavr.API;
import io.vavr.control.Either;
import org.apache.commons.csv.CSVFormat;
import org.apache.commons.csv.CSVParser;
import org.apache.commons.csv.CSVRecord;
import org.jetbrains.annotations.NotNull;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.Reader;
import java.io.StringReader;
import java.io.UncheckedIOException;
import java.time.Instant;
import java.util.ArrayList;
import java.util.List;
import java.util.Optional;
import java.util.function.Function;
import java.util.stream.Collectors;

import static com.pkb.service.questionnaire.TooLongQuestionnaireAnswerTextException.MAX_QUESTIONNAIRE_ANSWER_LENGTH;
import static com.pkb.service.questionnaire.TooLongQuestionnaireNameException.MAX_QUESTIONNAIRE_NAME_LENGTH;
import static java.util.Objects.requireNonNull;

public class QuestionnaireCSVImporter {
    private final String questionnaireName;

    private final Team team;

    private final PKBPerson creatorPerson;

    private QuestionnaireComponentToStringMapper qsTypeToStringMapper = new QuestionnaireComponentToStringMapper();

    QuestionnaireCSVImporter(
            @NotNull String questionnaireName,
            @NotNull Team team,
            @NotNull PKBPerson creatorPerson) {
        this.questionnaireName = requireNonNull(questionnaireName, "questionnaireName cannot be null");
        this.team = requireNonNull(team, "team cannot be null");
        this.creatorPerson = requireNonNull(creatorPerson, "creatorPerson cannot be null");
    }

    /**
     * Since the apache CSVParser cannot handle \r as a line terminator, we read the CSV
     * line by line, and concatenate the lines with \n before feeding them to the CSVParser
     */
    private String readToString(Reader reader) throws IOException {
        BufferedReader br = new BufferedReader(reader);
        StringBuilder sb = new StringBuilder();
        String line;
        while ((line = br.readLine()) != null) {
            sb.append(line);
            sb.append("\n");
        }
        return sb.toString();
    }

    /**
     * Note this function tries to return _all_ the possible validation errors in one go
     */
    Either<List<QuestionnaireException>, QuestionnaireTeam> importFrom(Reader csv, Instant creationDate) {
        var importErrors = API.<QuestionnaireException>Seq();
        var components = API.<QuestionnaireComponent>Seq();
        try {
            Reader csvReader = new StringReader(readToString(csv));
            if (questionnaireName.length() > MAX_QUESTIONNAIRE_NAME_LENGTH) {
                importErrors = importErrors.append(new TooLongQuestionnaireNameException(questionnaireName));
            }
            
            CSVParser parser = new CSVParser(csvReader, CSVFormat.DEFAULT);
            var questionsOrErrors = Either.sequence(
                    parser.stream().map(this::makeQuestion).collect(Collectors.toList())
            );
            if (questionsOrErrors.isLeft()) {
                importErrors = importErrors.appendAll(questionsOrErrors.getLeft().flatMap(Function.identity()));
            } else {
                components = questionsOrErrors.get();
            }
            
            PrivacyFlag teamPrivacyFlag = PrivacyFlagsToPrivacyFlagConverter.findPrivacyFlag(
                    team.getPrivacyFlags().getRequireGeneral(),
                    team.getPrivacyFlags().getRequireMentalHealth(),
                    team.getPrivacyFlags().getRequireSexualHealth(),
                    team.getRequireSocialCare()).orElseThrow(() -> new RuntimeException("Team with public id" + team.getPublicId() + " has no privacy flag set"));

            if (importErrors.isEmpty()) {
                return Either.right(new QuestionnaireTeamImpl(team.getPublicId(),
                        Questionnaire.createQuestionnaire(questionnaireName, creatorPerson.getPublicId(), creationDate, teamPrivacyFlag, components.asJava()),
                        false,
                        creationDate));
            } else {
                return Either.left(importErrors.asJava());
            }
        } catch (IOException e) {
            throw new UncheckedIOException(e);
        } finally {
            close(csv);
        }
    }

    private void close(Reader csv) {
        try {
            csv.close();
        } catch (IOException e) {
            throw new UncheckedIOException(e);
        }
    }

    private Either<List<QuestionnaireException>, QuestionnaireComponent> makeQuestion(CSVRecord record) {
        var it = record.stream().iterator();
        // blank lines are ignored by the parser so there is always a next
        // if there is no char before the separator then the questionText is just ""
        String questionText = it.next();
        if (questionText.isBlank()){
            return Either.left(List.of(new NoQuestionTextException()));
        }

        if (!it.hasNext()) {
            return Either.left(List.of(new NoQuestionTypeException(questionText)));
        }
        String componentTypeStr = it.next();
        if (componentTypeStr.isBlank()){
            return Either.left(List.of(new NoQuestionTypeException(questionText)));
        }
        
        var errs = API.<QuestionnaireException>List();
        
        Optional<QuestionnaireComponentType> maybeQuestionnaireComponentType = qsTypeToStringMapper.map(componentTypeStr);
        var mandatory = false;
        List<String> answers = new ArrayList<>();
        
        // Make sure question type exists
        if (maybeQuestionnaireComponentType.isEmpty()) {
            errs = errs.append(new InvalidQuestionTypeException(questionText, componentTypeStr));
        }
        else {
            var componentType = maybeQuestionnaireComponentType.get();
            if (componentType.isQuestion()){
                if(it.hasNext()){
                    // if we have a 'mandatory' field, parse it (it is already set to false which is the default otherwise)
                    Either<IncorrectMandatoryFieldException, Boolean> maybeMandatoryValue = getMandatoryBooleanOrException(questionText, it.next());
                    if (maybeMandatoryValue.isRight()){
                        mandatory = maybeMandatoryValue.get();
                    } else {
                        errs = errs.append(maybeMandatoryValue.getLeft());
                    }
                }
            }
            Iterators.addAll(answers, it);
            answers = answers.stream().map(String::trim).filter(answer -> !answer.isBlank()).collect(Collectors.toList());
            
            if (componentType.isMultiChoice() && answers.isEmpty()) {
                errs = errs.append(new InvalidMultiAnswerQuestionException(questionText, componentTypeStr));
            }
            
            // Make sure answer lengths are valid
            var tooLongAnswers = answers.stream()
                    .filter(answer -> answer.length() > MAX_QUESTIONNAIRE_ANSWER_LENGTH)
                    .collect(Collectors.toList());
            if (!tooLongAnswers.isEmpty()) {
                errs = errs.appendAll(tooLongAnswers.stream().map(TooLongQuestionnaireAnswerTextException::new)
                        .collect(Collectors.toList()));
            }
        }
        
        if (errs.isEmpty()) {
            // Code inspector isn't quite smart enough to figure out this list will always be non-empty
            // if the optional isn't present.
            //noinspection OptionalGetWithoutIsPresent
            return Either.right(qsTypeToStringMapper.createQuestionnaireComponent(maybeQuestionnaireComponentType.get(), questionText, mandatory, answers));
        } else {
            return Either.left(errs.asJava());
        }
    }
    
    private Either<IncorrectMandatoryFieldException, Boolean> getMandatoryBooleanOrException(String questionText, String mandatory){
        if ("true".equalsIgnoreCase(mandatory)){
            return Either.right(true);
        }
        if ("false".equalsIgnoreCase(mandatory)|| "".equals(mandatory)){
            return Either.right(false);
        }
        return Either.left(new IncorrectMandatoryFieldException(questionText, mandatory));
    }
}
