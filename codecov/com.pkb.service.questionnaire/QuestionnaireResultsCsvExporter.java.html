/**
 * This class builds a CSV download of the results of a single Questionnaire
 * usng the new Questionnaire Service API. It uses the Spring Reactive Streams
 * WebClient to stream the data being returned from the QS API, transform it
 * into CSV rows and then write it directly to the HTTP output stream
 */
public class QuestionnaireResultsCsvExporter {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private static final String TITLE_ORG_ID = "Organisation ID";

    private static final String TITLE_AGE = "Age";

    private static final String TITLE_GENDER = "Gender";

    private static final String TITLE_DATE_COMPLETED = "Date completed";

    private static final String TITLE_DATE_REQUESTED = "Date requested";

    private static final String TITLE_REQUESTED_BY = "Requested by";

    private static final String TITLE_NATIONAL_ID = "National ID";

    private static final DateTimeFormatter FORMATTER = DateTimeFormatter.ofPattern("yyyy MMM dd HH:mm:ss").withLocale(Locale.UK).withZone(Constants.APPLICATION_TZ);

    private final QuestionnaireServiceClient qsClient;
    private final CorrelationIdUtil correlationIdUtil;
    private final UserManager userManager;
    private final PKBPersonHelper pkbPersonHelper;
    private final QuestionnaireResponseClient qsResponseClient;
    private final PhrConfig config;
    private final UncheckedCsvPrinter.Factory printerFactory;

    public QuestionnaireResultsCsvExporter(QuestionnaireServiceClient qsClient, CorrelationIdUtil correlationIdUtil, UserManager userManager, PKBPersonHelper pkbPersonHelper,
                                           QuestionnaireResponseClient qsResponseClient, PhrConfig config, UncheckedCsvPrinter.Factory printerFactory) {
        this.qsClient = qsClient;
        this.correlationIdUtil = correlationIdUtil;
        this.userManager = userManager;
        this.pkbPersonHelper = pkbPersonHelper;
        this.qsResponseClient = qsResponseClient;
        this.config = config;
        this.printerFactory = printerFactory;
    }

    Either<QuestionnaireServiceError,Void> buildQuestionnaireResultsStreaming(UUID questionnaireId, Team team, QuestionnaireReportCsvExportFilter filter, Supplier<Writer> outputWriterSupplier, UUID loggedInPersonId) {
        var requestId = correlationIdUtil.maybeGet().orElseGet(UUID::randomUUID);
        LOGGER.debug("Start building questionnaire results for questionnaire: {}", questionnaireId);
        return qsClient.getQuestionnaireById(requestId, questionnaireId)
                .map(questionnaire -> {
                    var headers = getHeadersForQuestionnaire(questionnaire, team, filter);
                    var responses = getResponsesFluxForQuestionnaire(requestId, questionnaire, team, filter, loggedInPersonId);
                    var nonResponders = getNonRespondersFluxForQuestionnaire(requestId, questionnaire, team, filter);

                    LOGGER.debug("Preparing to stream results for questionnaire: {}", questionnaireId);
                    try (var printer = printerFactory.getPrinter(outputWriterSupplier.get(), headers)) {
                        responses.concatWith(nonResponders).onErrorResume(e -> {
                            LOGGER.error("Error creating csv results for questionnaire {}", questionnaireId, e);
                            return Flux.just(List.of("Error with questionnaire " + questionnaireId));
                        }).toStream().forEach(columns -> {
                            columns.forEach(printer::print);
                            printer.println();
                        });
                        LOGGER.debug("Finished  streaming results for questionnaire: {}", questionnaireId);
                    }
                    return null;
                });
    }

    private List<String> getHeadersForQuestionnaire(Questionnaire questionnaire, Team team, QuestionnaireReportCsvExportFilter filter) {
        List<String> headers = new ArrayList<>();
        if (filter.includeNationalId()) {
            headers.add(NationalIdType.getPrimaryNationalIdTypeFromCountryCode(team.getCountry())
                    .map(NationalIdType::getName)
                    .orElse(TITLE_NATIONAL_ID));
        }
        if (filter.includeOrgIds()) {
            headers.add(TITLE_ORG_ID);
        }
        headers.addAll(asList(TITLE_AGE, TITLE_GENDER, TITLE_DATE_COMPLETED, TITLE_DATE_REQUESTED));
        if (filter.includeRequestor()) {
            headers.add(TITLE_REQUESTED_BY);
        }
        headers.addAll(questionnaire.accept(CsvHeaderVisitor.INSTANCE));
        return headers;
    }

    private Flux<List<String>> getResponsesFluxForQuestionnaire(UUID requestId, Questionnaire questionnaire, Team team, QuestionnaireReportCsvExportFilter filter, UUID loggedInUserId) {
        return getCSVRows(qsResponseClient.getResponsesForQuestionnaire(requestId, questionnaire.getId(), loggedInUserId),
                new QuestionnaireResponderAdapter(team, filter));
    }

    private Flux<List<String>> getNonRespondersFluxForQuestionnaire(UUID requestId, Questionnaire questionnaire, Team team, QuestionnaireReportCsvExportFilter filter) {
        return getCSVRows(qsResponseClient.getNonRespondersForQuestionnaire(requestId, questionnaire.getId()), new QuestionnaireNonResponderAdapter(team, filter));
    }

    private <T> Flux<List<String>> getCSVRows(Publisher<T> entityStream, CsvResultAdapter<T> adapter) {
        return JdkFlowAdapter.flowPublisherToFlux(entityStream)
                .onErrorResume(WebClientResponseException.NotFound.class, notFound -> Flux.empty())
                .window(config.getCsvDownloadGenerationBatchSize())
                .flatMap(adapter::processQuestionnaireEntityBatch);
    }

    abstract class CsvResultAdapter<T> {

        private final Team team;
        private final QuestionnaireReportCsvExportFilter filter;

        CsvResultAdapter(Team team, QuestionnaireReportCsvExportFilter filter) {
            this.team = team;
            this.filter = filter;
        }

        abstract QuestionnaireRequest getRequest(T source);

        @Nullable
        abstract Instant getResponseCreationDate(T source);

        abstract List<String> getResponseColumns(T source);

        private Flux<List<String>> processQuestionnaireEntityBatch(Flux<T> batch) {
            return batch.collectList().flatMapMany(responseBatchList -> {
                var personMap = getPersonMap(responseBatchList);
                return Flux.fromStream(responseBatchList.stream().map(source -> getCsvColumns(personMap, source)));
            });
        }

        private Map<UUID, PKBPerson> getPersonMap(List<T> sourceEntities) {
            var personIdSet = sourceEntities.stream().map(this::getRequest).flatMap(request -> Stream.of(request.getRequestorPersonId(), request.getPatientId())).collect(Collectors.toSet());
            return userManager.getPKBPersonMapByPublicId(personIdSet);
        }

        private List<String> getCsvColumns(Map<UUID, PKBPerson> personMap, T source) {
            return ListUtils.union(getFixedColumns(personMap, source), getResponseColumns(source));
        }

        private List<String> getFixedColumns(Map<UUID, PKBPerson> personMap, T source) {
            QuestionnaireRequest request = getRequest(source);
            Instant responseCreatedDate = getResponseCreationDate(source);
            PKBPerson patient = personMap.get(request.getPatientId());
            PKBPerson requestor = personMap.get(request.getRequestorPersonId());
            List<String> columns = new ArrayList<>();
            if (filter.includeNationalId()) {
                columns.add(patient.getNationalIdByCountry(team.getCountry())
                        .map(NationalId::getValueFormatted)
                        .orElse(null));
            }
            // no support for (sensitive) team-level IDs
            if (filter.includeOrgIds()) {
                columns.add(patient.getOrgLevelIdsByOrg(team.getOrg().getId()).stream().findFirst()
                        .map(OrgLevelId::getValue)
                        .orElse(null));
            }

            columns.add(patient.findDateOfBirth().map(ld -> Integer.toString(pkbPersonHelper.calculateAge(ld))).orElse("0"));
            columns.add(Gender.toLetter(patient.getGender()));

            columns.add(responseCreatedDate == null ? "" : FORMATTER.format(responseCreatedDate));
            columns.add(FORMATTER.format(request.getCreatedDate()));
            if (filter.includeRequestor()) {
                columns.add(requestor.getFullName());
            }
            return columns;
        }
    }

    class QuestionnaireResponderAdapter extends CsvResultAdapter<QuestionnaireResponse> {

        QuestionnaireResponderAdapter(Team team, QuestionnaireReportCsvExportFilter filter) {
            super(team, filter);
        }

        @Override
        QuestionnaireRequest getRequest(QuestionnaireResponse source) {
            return source.getQuestionnaireRequest();
        }

        @Override
        Instant getResponseCreationDate(QuestionnaireResponse source) {
            return source.getCreatedDate();
        }

        @Override
        List<String> getResponseColumns(QuestionnaireResponse source) {
            return source.accept(CsvValueVisitor.INSTANCE);
        }
    }

    class QuestionnaireNonResponderAdapter extends CsvResultAdapter<QuestionnaireRequest> {

        QuestionnaireNonResponderAdapter(Team team, QuestionnaireReportCsvExportFilter filter) {
            super(team, filter);
        }

        @Override
        QuestionnaireRequest getRequest(QuestionnaireRequest source) {
            return source;
        }

        @Override
        @Nullable
        Instant getResponseCreationDate(QuestionnaireRequest source) {
            return null;
        }

        @Override
        List<String> getResponseColumns(QuestionnaireRequest source) {
            return source.getQuestionnaire().accept(NonResponderCsvVisitor.INSTANCE);
        }
    }

}
