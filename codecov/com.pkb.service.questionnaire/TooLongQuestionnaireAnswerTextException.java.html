package com.pkb.service.questionnaire;

import org.jetbrains.annotations.NotNull;

public class TooLongQuestionnaireAnswerTextException extends QuestionnaireException {

    public static final int MAX_QUESTIONNAIRE_ANSWER_LENGTH = 2500;

    private final String answerText;

    public TooLongQuestionnaireAnswerTextException(@NotNull String answerText) {
        super("AdminQuestionnaireAction.txt.answer.text.too_long");
        this.answerText = answerText;
    }

    @Override
    public String[] getErrorTextParams() {
        return new String[]{Integer.toString(MAX_QUESTIONNAIRE_ANSWER_LENGTH)};
    }

    public String getAnswerText() {
        return answerText;
    }
}
