package com.pkb.service.questionnaire;

import com.pkb.questionnaireservice.modelv2.Question;

import java.util.List;

public class MissingMandatoryAnswerException extends QuestionnaireException {

    private final List<Question> mandatoryQuestionsWithoutAnswer;

    MissingMandatoryAnswerException(List<Question> mandatoryQuestionsWithoutAnswer) {
        super("patientQuestionnaireAction.err.answer_all_mandatory_questions");
        this.mandatoryQuestionsWithoutAnswer = mandatoryQuestionsWithoutAnswer;
    }

    @Override
    public String[] getErrorTextParams() {
        return new String[]{};
    }
    
    public List<Question> getMandatoryQuestionsWithoutAnswer() {
        return mandatoryQuestionsWithoutAnswer;
    }
}
