package com.pkb.service.questionnaire;

import com.google.common.collect.BiMap;
import com.google.common.collect.ImmutableBiMap;
import com.pkb.questionnaireservice.modelv2.MultiSelectQuestion;
import com.pkb.questionnaireservice.modelv2.OpenQuestion;
import com.pkb.questionnaireservice.modelv2.QuestionnaireComponent;
import com.pkb.questionnaireservice.modelv2.QuestionnaireComponentType;
import com.pkb.questionnaireservice.modelv2.SingleSelectQuestion;
import com.pkb.questionnaireservice.modelv2.Title;

import java.util.List;
import java.util.Optional;

import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;

public class QuestionnaireComponentToStringMapper {
    public static final String RADIO = "Radio";
    public static final String CHECKBOX = "Checkbox";
    public static final String TEXT = "Text";
    public static final String TITLE = "Title";

    private static final BiMap<QuestionnaireComponentType, String> TYPE_TO_STRING = new ImmutableBiMap.Builder<QuestionnaireComponentType, String>()
            .put(QuestionnaireComponentType.Title, TITLE)
            .put(QuestionnaireComponentType.SingleSelectQuestion, RADIO)
            .put(QuestionnaireComponentType.MultiSelectQuestion,CHECKBOX)
            .put(QuestionnaireComponentType.OpenQuestion, TEXT)
            .build();

    public Optional<QuestionnaireComponentType> map(String type) {
        return Optional.ofNullable(TYPE_TO_STRING.inverse().get(type));
    }
    
    public String toString(QuestionnaireComponent component) {
        return TYPE_TO_STRING.getOrDefault(component.getType(), TEXT);
    }

    public QuestionnaireComponent createQuestionnaireComponent(QuestionnaireComponentType type, String questionText, boolean mandatory, List<String> possibleAnswers) {
        return Match(type).of(
                Case($(QuestionnaireComponentType.SingleSelectQuestion), new SingleSelectQuestion(questionText.strip(), mandatory, possibleAnswers)),
                Case($(QuestionnaireComponentType.MultiSelectQuestion), new MultiSelectQuestion(questionText.strip(), mandatory, possibleAnswers)),
                Case($(QuestionnaireComponentType.Title), new Title(questionText.strip())),
                Case($(QuestionnaireComponentType.OpenQuestion), new OpenQuestion(questionText.strip(), mandatory)));
    }
}
