package com.pkb.service.questionnaire;

import com.google.common.collect.HashMultimap;
import com.google.common.collect.SetMultimap;
import com.pkb.questionnaireservice.modelv2.AnswerSupplier;
import com.pkb.questionnaireservice.modelv2.MultiSelectQuestion;
import com.pkb.questionnaireservice.modelv2.PossibleAnswer;
import com.pkb.questionnaireservice.modelv2.Questionnaire;
import com.pkb.questionnaireservice.modelv2.QuestionnaireComponentType;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.util.HashMap;
import java.util.LinkedHashSet;
import java.util.Map;
import java.util.UUID;
import java.util.stream.Collectors;

public class QuestionnaireResultAnswerSupplier extends AnswerSupplier {

    private final SetMultimap<UUID, UUID> multiSelectQuestionLookup;
    private final Map<UUID, UUID> singleSelectQuestionLookup;
    private final Map<UUID, String> openQuestionAnswerLookup;

    public QuestionnaireResultAnswerSupplier(Map<UUID, CheckboxAnswerMap> multiSelectResponses,
                                             Map<UUID, UUID> singleSelectResponses,
                                             Map<UUID, String> openQuestionResponses,
                                             Questionnaire questionnaire) {
        this.multiSelectQuestionLookup = createCheckboxAnswersMap(multiSelectResponses, questionnaire);
        this.singleSelectQuestionLookup = (singleSelectResponses == null) ? new HashMap<>() : singleSelectResponses;
        this.openQuestionAnswerLookup = openQuestionResponses;
    }

    private SetMultimap<UUID, UUID> createCheckboxAnswersMap(Map<UUID, CheckboxAnswerMap> multiSelectResponses, Questionnaire questionnaire) {
        SetMultimap<UUID, UUID> checkboxAnswersMap = HashMultimap.create();
        questionnaire.getComponents().stream().filter(q -> q.getType() == QuestionnaireComponentType.MultiSelectQuestion)
                .map(question -> checkboxAnswersMap.putAll(question.getId(), createMultiChoiceAnswer(multiSelectResponses, (MultiSelectQuestion) question)))
                .collect(Collectors.toList());
        return checkboxAnswersMap;
    }

    private LinkedHashSet<UUID> createMultiChoiceAnswer(Map<UUID, CheckboxAnswerMap> multiSelectResponses, MultiSelectQuestion question) {
        return question.getPossibleAnswers().stream().map(PossibleAnswer::getId)
                .filter(id -> multiSelectResponses.get(question.getId()).getAnswers().get(id))
                .collect(Collectors.toCollection(LinkedHashSet::new));
    }

    @Override
    public @NotNull LinkedHashSet<UUID> getMultiSelectQuestionAnswers(@NotNull UUID questionId) {
        return new LinkedHashSet<>(multiSelectQuestionLookup.get(questionId));
    }

    @Override
    public @Nullable String getOpenQuestionAnswer(@NotNull UUID questionId) {
        return openQuestionAnswerLookup.get(questionId);
    }

    @Override
    public @Nullable UUID getSingleSelectQuestionAnswer(@NotNull UUID questionId) {
        return singleSelectQuestionLookup.get(questionId);
    }

}
