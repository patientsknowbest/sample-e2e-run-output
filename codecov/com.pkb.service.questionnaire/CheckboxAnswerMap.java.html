package com.pkb.service.questionnaire;

import java.util.Map;
import java.util.UUID;

/*
 * This class allows us to fill in a Map<UUID, Map<UUID, Boolean>> from a jsp. Accessing directly the second map by
 * using map[uuid1][uuid2] does not seem to be possible, hence we are using this mapper to access it with
 * map[uuid1].answers[uuid2]. [see fillQuestionnaire.jsp for MultiSelectQuestion]
 */
public class CheckboxAnswerMap {

    private Map<UUID, Boolean> answers;

    public Map<UUID, Boolean> getAnswers() {
        return answers;
    }

    public void setAnswers(Map<UUID, Boolean> answers) {
        this.answers = answers;
    }

}
