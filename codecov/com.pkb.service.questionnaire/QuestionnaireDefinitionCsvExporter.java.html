package com.pkb.service.questionnaire;

import com.pkb.questionnaireservice.modelv2.ClosedQuestion;
import com.pkb.questionnaireservice.modelv2.PossibleAnswer;
import com.pkb.questionnaireservice.modelv2.Question;
import com.pkb.questionnaireservice.modelv2.Questionnaire;
import com.pkb.util.UncheckedCsvPrinter;

import java.io.Writer;
import java.util.function.Supplier;

public class QuestionnaireDefinitionCsvExporter {
    private QuestionnaireComponentToStringMapper qsTypeToStringMapper = new QuestionnaireComponentToStringMapper();
    private final UncheckedCsvPrinter.Factory printerFactory;

    public QuestionnaireDefinitionCsvExporter(UncheckedCsvPrinter.Factory printerFactory) {
        this.printerFactory = printerFactory;
    }

    void buildQuestionnaireStreaming(Questionnaire questionnaire, Supplier<Writer> outputWriterSupplier) {

        try (var printer = printerFactory.getPrinter(outputWriterSupplier.get(), null)) {
            questionnaire.getComponents().forEach(
                    component -> {
                        printer.print(component.getDisplayText());
                        printer.print(qsTypeToStringMapper.toString(component));
                        if (component.getType().isQuestion()){
                            printer.print(((Question) component).getMandatory());
                        }
                        if (component.getType().isMultiChoice()) {
                            for (PossibleAnswer answer : ((ClosedQuestion) component).getPossibleAnswers()) {
                                printer.print(answer.getDisplayText());
                            }
                        }
                        printer.println();
                    });
        }
    }
}
