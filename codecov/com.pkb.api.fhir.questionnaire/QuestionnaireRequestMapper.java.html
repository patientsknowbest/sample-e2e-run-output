package com.pkb.api.fhir.questionnaire;

import com.pkb.api.fhir.TemporalMapper;
import com.pkb.api.fhir.foundation.ReferenceMapper;
import com.pkb.datamodel.user.Patient;
import org.hl7.fhir.dstu3.model.CodeableConcept;
import org.hl7.fhir.dstu3.model.Extension;
import org.hl7.fhir.dstu3.model.Identifier;
import org.hl7.fhir.dstu3.model.Meta;
import org.hl7.fhir.dstu3.model.ProcedureRequest;
import org.jetbrains.annotations.NotNull;

import java.time.ZoneOffset;
import java.util.List;
import java.util.stream.Collectors;

public class QuestionnaireRequestMapper {

    private final TemporalMapper timeMapper;
    private final ReferenceMapper referenceMapper;

    public QuestionnaireRequestMapper(TemporalMapper timeMapper,
                                      ReferenceMapper referenceMapper) {
        this.timeMapper = timeMapper;
        this.referenceMapper = referenceMapper;
    }

    public @NotNull ProcedureRequest toProcedureRequest(com.pkb.questionnaireservice.modelv2.QuestionnaireRequest questionnaireRequest,
                                                        Patient patient) {
        ProcedureRequest procedureRequest = new ProcedureRequest();
        procedureRequest.setId(questionnaireRequest.getId().toString());
        procedureRequest.setMeta(extractMetadata(questionnaireRequest));
        
        procedureRequest.setIdentifier(questionnaireRequest.getIdentifiers().stream().map(
                pkbIdentifier -> new Identifier().setSystem(pkbIdentifier.getSystem()).setValue(pkbIdentifier.getValue())).collect(Collectors.toList()));
        
        Extension extension = new Extension();
        extension.setUrl("http://hl7.org/fhir/StructureDefinition/procedurerequest-questionnaireRequest");
        extension.setValue(referenceMapper.toReferenceWithName(questionnaireRequest.getQuestionnaire()));
        procedureRequest.setExtension(List.of(extension));
        
        procedureRequest.setStatus(ProcedureRequest.ProcedureRequestStatus.ACTIVE);        
        procedureRequest.setIntent(ProcedureRequest.ProcedureRequestIntent.ORDER);
        
        CodeableConcept codeableConcept = new CodeableConcept();
        codeableConcept.setText("Questionnaire");
        procedureRequest.setCode(codeableConcept);

        procedureRequest.setSubject(referenceMapper.toReferenceWithName(patient));        
        procedureRequest.setAuthoredOn(timeMapper.toDate(questionnaireRequest.getCreatedDate().atZone(ZoneOffset.UTC)));
        
        return procedureRequest;
    }

    private @NotNull Meta extractMetadata(com.pkb.questionnaireservice.modelv2.QuestionnaireRequest questionnaireRequest) {
        Meta meta = new Meta();
        meta.setLastUpdatedElement(timeMapper.toInstantTypeInUTC(questionnaireRequest.getCreatedDate().atZone(ZoneOffset.UTC)));
        return meta;
    }
}
