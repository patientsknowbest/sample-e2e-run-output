package com.pkb.api.fhir.questionnaire;

import com.pkb.api.fhir.PrivacyFlagsMapper;
import com.pkb.api.fhir.TemporalMapper;
import com.pkb.questionnaireservice.modelv2.MultiSelectQuestion;
import com.pkb.questionnaireservice.modelv2.OpenQuestion;
import com.pkb.questionnaireservice.modelv2.PossibleAnswer;
import com.pkb.questionnaireservice.modelv2.QuestionnaireComponent;
import com.pkb.questionnaireservice.modelv2.SingleSelectQuestion;
import com.pkb.questionnaireservice.modelv2.Title;
import org.hl7.fhir.dstu3.model.Enumerations;
import org.hl7.fhir.dstu3.model.Meta;
import org.hl7.fhir.dstu3.model.Questionnaire;
import org.hl7.fhir.dstu3.model.StringType;
import org.jetbrains.annotations.NotNull;

import java.time.ZoneOffset;
import java.util.List;
import java.util.stream.Collectors;

import static com.google.common.base.Predicates.instanceOf;
import static io.vavr.API.$;
import static io.vavr.API.Case;
import static io.vavr.API.Match;

public class QuestionnaireMapper {

    private final TemporalMapper timeMapper;
    private final PrivacyFlagsMapper privacyFlagsMapper;

    public QuestionnaireMapper(TemporalMapper timeMapper, PrivacyFlagsMapper privacyFlagsMapper) {
        this.timeMapper = timeMapper;
        this.privacyFlagsMapper = privacyFlagsMapper;
    }

    public @NotNull Questionnaire toQuestionnaire(com.pkb.questionnaireservice.modelv2.Questionnaire questionnaire) {
        Questionnaire fhirQuestionnaire = new Questionnaire();
        fhirQuestionnaire.setId(questionnaire.getId().toString());
        fhirQuestionnaire.setMeta(extractMetadata(questionnaire));
        fhirQuestionnaire.setTitle(questionnaire.getName());
        //At present, we always map status to active: http://dev.patientsknowbest.com/home/fhir-api/fhir-api-roadmap/questionnaire
        fhirQuestionnaire.setStatus(Enumerations.PublicationStatus.ACTIVE);
        fhirQuestionnaire.setDateElement(timeMapper.toDateTimeType(questionnaire.getCreatedDate().atZone(ZoneOffset.UTC)));

        fhirQuestionnaire.setItem(questionnaire.getComponents().stream().map(item ->

            Match(item).of(
                    Case($(instanceOf(MultiSelectQuestion.class)), question -> mapItem((MultiSelectQuestion) question)),
                    Case($(instanceOf(SingleSelectQuestion.class)), question -> mapItem((SingleSelectQuestion) question)),
                    Case($(instanceOf(OpenQuestion.class)), question -> mapItem((OpenQuestion) question)),
                    Case($(instanceOf(Title.class)), title -> mapItem((Title) title)),
                    Case($(), unknown -> { throw new InternalError("No mapping for item type: "+unknown);})
                )).collect(Collectors.toList())
        );


        return fhirQuestionnaire;
    }

    private Questionnaire.QuestionnaireItemComponent mapItem(QuestionnaireComponent questionnaireComponent) {
        Questionnaire.QuestionnaireItemComponent fhirItem = new Questionnaire.QuestionnaireItemComponent();
        fhirItem.setLinkId(questionnaireComponent.getId().toString());
        fhirItem.setText(questionnaireComponent.getDisplayText());
        return fhirItem;
    }

    private Questionnaire.QuestionnaireItemComponent mapItem(MultiSelectQuestion question) {
        Questionnaire.QuestionnaireItemComponent fhirItem = mapItem((QuestionnaireComponent) question);
        fhirItem.setType(Questionnaire.QuestionnaireItemType.CHOICE);
        fhirItem.setRepeats(true);
        fhirItem.setOption(question.getPossibleAnswers().stream().map(this::mapPossibleAnswer).collect(Collectors.toList()));
        return fhirItem;
    }

    private Questionnaire.QuestionnaireItemComponent mapItem(SingleSelectQuestion question) {
        Questionnaire.QuestionnaireItemComponent fhirItem = mapItem((QuestionnaireComponent) question);
        fhirItem.setType(Questionnaire.QuestionnaireItemType.CHOICE);
        fhirItem.setRepeats(false);
        fhirItem.setOption(question.getPossibleAnswers().stream().map(this::mapPossibleAnswer).collect(Collectors.toList()));
        return fhirItem;
    }

    private Questionnaire.QuestionnaireItemComponent mapItem(OpenQuestion question) {
        Questionnaire.QuestionnaireItemComponent fhirItem = mapItem((QuestionnaireComponent) question);
        fhirItem.setRepeats(false);
        fhirItem.setType(Questionnaire.QuestionnaireItemType.TEXT);
        return fhirItem;
    }

    private Questionnaire.QuestionnaireItemComponent mapItem(Title title) {
        Questionnaire.QuestionnaireItemComponent fhirItem = mapItem((QuestionnaireComponent) title);
        fhirItem.setType(Questionnaire.QuestionnaireItemType.DISPLAY);
        return fhirItem;
    }

    private @NotNull Meta extractMetadata(com.pkb.questionnaireservice.modelv2.Questionnaire questionnaire) {
        Meta meta = new Meta();
        meta.setLastUpdatedElement(timeMapper.toInstantTypeInUTC(questionnaire.getCreatedDate().atZone(ZoneOffset.UTC)));
        meta.setSecurity(List.of(privacyFlagsMapper.toCoding(questionnaire.getDefaultPrivacyFlag())));
        return meta;
    }

    private Questionnaire.QuestionnaireItemOptionComponent mapPossibleAnswer(PossibleAnswer possibleAnswer) {
        return new Questionnaire.QuestionnaireItemOptionComponent().setValue(new StringType(possibleAnswer.getDisplayText()));
    }
}
