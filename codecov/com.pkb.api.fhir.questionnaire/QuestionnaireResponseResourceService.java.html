package com.pkb.api.fhir.questionnaire;

import static com.pkb.authz.ImmutableTargetMeta.targetMeta;

import static io.vavr.API.Set;

import static java.util.function.Function.identity;

public class QuestionnaireResponseResourceService extends AbstractResourceService<QuestionnaireResponse> {
    private final AuthorizationDataService authorizationDataService;

    private final QuestionnaireResponseMapper questionnaireResponseMapper;
    private final QuestionnaireRequestMapper questionnaireRequestMapper;
    private final PatientService patientService;
    private final PatientMapper patientMapper;
    private final QuestionnaireResponseService questionnaireResponseService;
    private final FhirApiProperties fhirConfig;

    public QuestionnaireResponseResourceService(AuthorizationDataService authorizationDataService,
            QuestionnaireResponseMapper questionnaireResponseMapper,
            QuestionnaireRequestMapper questionnaireRequestMapper,
            PatientService patientService,
            QuestionnaireResponseService questionnaireResponseService,
            ServerResponseExceptionMapper exceptionMapper, PatientMapper patientMapper, FhirApiProperties fhirConfig) {
        super(exceptionMapper);
        this.authorizationDataService = authorizationDataService;
        this.questionnaireResponseMapper = questionnaireResponseMapper;
        this.questionnaireRequestMapper = questionnaireRequestMapper;
        this.patientService = patientService;
        this.questionnaireResponseService = questionnaireResponseService;
        this.patientMapper = patientMapper;
        this.fhirConfig = fhirConfig;
    }

    public @NotNull FhirCriteriaResult<QuestionnaireResponse> findQuestionnaireResponses(
            @NotNull QuestionnaireResponseSearchCriteria criteria) {
        //We need to handle the authorisation in two stages:
        //first fetch the questionnaires responses:
        AuthorizationInputs qrAuthInputs = inputs()
                .authenticatedIdentity(criteria.authenticatedIdentity())
                .action(Action.SEARCH)
                .addTargets(targetMeta(com.pkb.questionnaireservice.modelv2.QuestionnaireResponse.class, Set(), criteria.patientId(), true))
                .path(Path.FHIR)
                .build();
        AuthorizationData questionnaireAuthData = authorizationDataService.fetch(qrAuthInputs);

        var questionnaireResponses = questionnaireResponseService.findQuestionnaireResponses(
                questionnaireAuthData,
                ImmutableQuestionnaireResponseSearchCriteria.copyOf(criteria).withPageSize(fhirConfig.getQuestionnaireResponseMaxNumberOfResources()));
        return questionnaireResponses.map(responses -> {
            // Short circuit
            if (responses.getPage().isEmpty()) {
                return FhirCriteriaResult.<QuestionnaireResponse>empty();
            }
            // Fetch patient and carer details
                    var carersPublicIds = responses.getPage().filter(qr -> qr.getCompletedBy() != null  && !qr.wasCompletedByThePatient())
                    .map(QuestionnaireResponseBase::getCompletedBy);
            var patientsPublicIds = responses.getPage()
                    .map(qr -> qr.getQuestionnaireRequest().getPatientId());
            var patientsAndCarersPublicIds = patientsPublicIds.appendAll(carersPublicIds).distinct();
            AuthorizationInputs patientAndCarerAuthInputs = inputs()
                    .authenticatedIdentity(criteria.authenticatedIdentity())
                    .action(Action.SEARCH)
                    .targets(patientsAndCarersPublicIds.map(id ->
                            targetMeta(Patient.class, Set(id), Some(id), true)))
                    .path(Path.FHIR)
                    .build();
            AuthorizationData patientAndCompletedByAuthData = authorizationDataService.fetch(patientAndCarerAuthInputs);
            var patientsAndCarers = patientService.getPatients(patientAndCompletedByAuthData, patientsAndCarersPublicIds)
                    .filter(Either::isRight).map(Either::get)
                    .toMap(Patient::getPublicId, identity());
            return buildFhirCriteriaResult(responses, patientsAndCarers, criteria.includes());
        })
                //if an error has got to this level, it's not an auth error or not found error we can ignore -  we want to throw it
                .getOrElseThrow(e->exceptionToFhir(e, QuestionnaireResponse.class));

    }

    private FhirCriteriaResult<QuestionnaireResponse> buildFhirCriteriaResult(CriteriaResult<com.pkb.questionnaireservice.modelv2.QuestionnaireResponse> result,
                                                                              Map<UUID, Patient> patientsAndCarers,
                                                                              List<IncludeResource> includes) {
        var responses = result.getPage().flatMap(qr -> {
            // We'll simply ignore any responses where we don't have permission to see the patient
            return patientsAndCarers.get(qr.getQuestionnaireRequest().getPatientId())
                    .map(p -> questionnaireResponseMapper.toQuestionnaireResponse(qr, p, patientsAndCarers.get(qr.getCompletedBy())))
                    .toList();
        });
        var numberNotAllowedToSee = result.getPage().size() - responses.size();
        List<Resource> subjects = includes.contains(IncludeResource.SUBJECT_PATIENT)
                ? patientsAndCarers.values().map(p -> (Resource)patientMapper.toPatient(p.demographics())).toList()
                : List();
        
        
        List<Resource> basedOn = includes.contains(IncludeResource.BASED_ON_PROCEDURE_REQUEST)
                ? result.getPage().flatMap( qrep -> patientsAndCarers.get(qrep.getQuestionnaireRequest().getPatientId())
                        .map(p -> questionnaireRequestMapper.toProcedureRequest(qrep.getQuestionnaireRequest(), p))
                        .toList())
                : List();

        return ImmutableFhirCriteriaResult.<QuestionnaireResponse> fhirCriteriaResult()
                .total(result.getTotal() - numberNotAllowedToSee)
                .page(responses)
                .includes(subjects.appendAll(basedOn))
                .build();
    }
}
