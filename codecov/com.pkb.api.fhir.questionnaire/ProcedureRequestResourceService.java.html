package com.pkb.api.fhir.questionnaire;

import ca.uhn.fhir.rest.server.exceptions.BaseServerResponseException;
import com.pkb.api.fhir.AbstractResourceService;
import com.pkb.api.fhir.exception.ServerResponseExceptionMapper;
import com.pkb.authentication.AuthenticatedIdentity;
import com.pkb.authz.AuthorizationDataService;
import com.pkb.authz.action.Action;
import com.pkb.authz.context.Path;
import com.pkb.authz.data.AuthorizationData;
import com.pkb.authz.data.AuthorizationInputs;
import com.pkb.datamodel.user.Patient;
import com.pkb.domain.PatientService;
import com.pkb.domain.questionnaire.QuestionnaireService;
import io.vavr.API;
import io.vavr.collection.List;
import io.vavr.control.Either;
import org.hl7.fhir.dstu3.model.ProcedureRequest;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

import static com.pkb.authz.ImmutableTargetMeta.targetMeta;
import static com.pkb.authz.data.ImmutableAuthorizationInputs.inputs;
import static io.vavr.API.Set;
import static io.vavr.API.Some;
import static io.vavr.control.Either.left;
import static java.lang.String.format;

public class ProcedureRequestResourceService extends AbstractResourceService<ProcedureRequest> {

    private final AuthorizationDataService authorizationDataService;
    private final QuestionnaireService questionnaireService;
    private final QuestionnaireRequestMapper questionnaireRequestMapper;
    private final PatientService patientService;

    public ProcedureRequestResourceService(@NotNull AuthorizationDataService authorizationDataService,
                                           QuestionnaireService questionnaireService,
                                           PatientService patientService,
                                           @NotNull QuestionnaireRequestMapper questionnaireRequestMapper,
                                           @NotNull ServerResponseExceptionMapper exceptionMapper) {
        super(exceptionMapper);
        this.authorizationDataService = authorizationDataService;
        this.questionnaireService = questionnaireService;
        this.patientService = patientService;
        this.questionnaireRequestMapper = questionnaireRequestMapper;
    }
    
    public @NotNull Either<BaseServerResponseException, ProcedureRequest> getProcedureRequest(AuthenticatedIdentity authenticatedIdentity, @NotNull UUID publicId) {
        AuthorizationInputs authorizationInputs = inputs()
                .authenticatedIdentity(authenticatedIdentity)
                .action(Action.READ)
                .addTargets(targetMeta(com.pkb.questionnaireservice.modelv2.QuestionnaireRequest.class, Set(publicId), API.None(), false))
                .path(Path.FHIR)
                .build();
        AuthorizationData authData = authorizationDataService.fetch(authorizationInputs);
        
        return questionnaireService.getQuestionnaireRequests(authData, publicId)
                .flatMap(questRequest -> {
                    var patientId = questRequest.getPatientId();
                    AuthorizationInputs patientInputs = inputs()
                            .authenticatedIdentity(authenticatedIdentity)
                            .action(Action.READ)
                            .addTargets(targetMeta(Patient.class, Set(patientId), Some(patientId), true))
                            .path(Path.FHIR)
                            .build();
                    AuthorizationData patientAuthData = authorizationDataService.fetch(patientInputs);
                    var patient = patientService.getPatients(patientAuthData, List.of(patientId))
                            .filter(Either::isRight)
                            .map(Either::get);
                    if (patient.isEmpty()){
                        return left(new RuntimeException(format("Unauthorized access to ProcedureRequest -%s", patientId)));
                    }
                    return Either.right(questionnaireRequestMapper.toProcedureRequest(questRequest, patient.head())); })
                .mapLeft(ex -> exceptionToFhir(ex, ProcedureRequest.class));
    }
}
