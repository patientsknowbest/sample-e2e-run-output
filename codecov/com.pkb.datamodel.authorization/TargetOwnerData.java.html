package com.pkb.datamodel.authorization;

import com.pkb.datamodel.Account;
import com.pkb.datamodel.AccountKeys;
import com.pkb.datamodel.DataModelStyle;
import com.pkb.datamodel.consent.PatientConsent;
import com.pkb.datamodel.user.Person;
import io.vavr.control.Option;
import org.immutables.value.Value;

import java.util.function.Supplier;

/**
 * Additional data about the owner of a piece of data, relevant to authorization.
 */
@Value.Immutable
@Value.Style(builder = "targetOwnerData")
@DataModelStyle
public interface TargetOwnerData {

    Person person();

    Account account();

    @Value.Redacted
    Supplier<AccountKeys> accountKeysReloader();

    Option<AccountKeys> accountKeys();

    /**
     * This contains only the consent from the owner to the subject (there might be none)
     */
    Option<PatientConsent> consent();

    /**
     * True if the subject has decryption access to the owner.
     */
    @Value.Derived
    default boolean actorHasDecryptionAccess() {
        return accountKeys().isDefined();
    }
}
