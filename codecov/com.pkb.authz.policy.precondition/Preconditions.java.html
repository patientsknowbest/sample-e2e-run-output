package com.pkb.authz.policy.precondition;

import com.pkb.authentication.principal.user.AuthenticatedUser;

/**
 * Matching criteria for a policy.
 * A policy is only executed if these criteria are fulfilled.
 */
public class Preconditions {
    public static Precondition objectTypePrecondition(Class<?> objectType) {
        return facts -> objectType.isAssignableFrom(facts.targetMeta().targetType());
    }

    public static Precondition userTypePrecondition(Class<? extends AuthenticatedUser> userType) {
        return facts -> userType.isAssignableFrom(facts.authenticatedUser().getClass());
    }
}
