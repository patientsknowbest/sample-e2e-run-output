package com.pkb.authz.policy.precondition;

import com.pkb.authz.action.Action;
import com.pkb.authz.policy.Facts;

/**
 * Custom class for Action precondition. Have a custom class for it so we can identify them among other preconditions,
 * and provide a list of allowed actions in the authorization result.
 */
public class ActionMatchPrecondition implements Precondition {
    private final Action action;

    private ActionMatchPrecondition(Action action) {
        this.action = action;
    }

    public static ActionMatchPrecondition actionMatchPrecondition(Action action) {
        return new ActionMatchPrecondition(action);
    }

    @Override
    public boolean test(Facts facts) {
        return facts.action() == action;
    }

    public Action getAction() {
        return action;
    }
}
