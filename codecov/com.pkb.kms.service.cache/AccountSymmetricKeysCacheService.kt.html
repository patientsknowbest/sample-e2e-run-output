package com.pkb.kms.service.cache

import com.pkb.common.ClearableInternalState
import com.pkb.crypto.AccountPrivateKey
import com.pkb.crypto.dto.AccountKeysDTO
import com.pkb.crypto.dto.AccountSymmetricKeyDTO
import com.pkb.kms.service.MetricsHelper
import com.pkb.kms.service.extension.toEncryptedAccountSymmetricKey
import com.pkb.kms.service.repository.AccountSymmetricKeyRepository
import com.pkb.kms.shared.KMS_CACHE_PREFIX
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.KmsError
import io.vavr.control.Either
import io.vavr.kotlin.left
import io.vavr.kotlin.right
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.cache.annotation.CacheEvict
import org.springframework.cache.annotation.CachePut
import org.springframework.cache.annotation.Cacheable
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class AccountSymmetricKeysCacheService(private val accountSymmetricKeyRepository: AccountSymmetricKeyRepository,
                                       metrics: MetricsHelper) : ClearableInternalState, IsAccountKeyCache {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    init {
        metrics.registerCaches(ACCOUNT_SYMMETRIC_KEYS_CACHE_NAME)
    }

    @Cacheable(ACCOUNT_SYMMETRIC_KEYS_CACHE_NAME, key = "#accountPublicId", unless = "#result.isLeft()")
    fun getAccountSymmetricKeys(accountPublicId: AccountPublicId,
                                accountPrivateKey: AccountPrivateKey): Either<KmsError, List<AccountSymmetricKeyDTO>> =
        right(accountSymmetricKeyRepository.findByPublicAccountId(accountPublicId)
                      .map { it.toEncryptedAccountSymmetricKey() }
                      .map { it.toDTO(accountPrivateKey) })

    @CachePut(ACCOUNT_SYMMETRIC_KEYS_CACHE_NAME, key = "#accountPublicId", unless = "#result.isLeft()")
    fun getAccountSymmetricKeysNoCache(accountPublicId: AccountPublicId,
                                       accountPrivateKey: AccountPrivateKey): Either<KmsError, List<AccountSymmetricKeyDTO>> =
        right(accountSymmetricKeyRepository.findByPublicAccountId(accountPublicId)
                      .map { it.toEncryptedAccountSymmetricKey() }
                      .map { it.toDTO(accountPrivateKey) })

    @Cacheable(ACCOUNT_SYMMETRIC_KEYS_CACHE_NAME, key = "#accountPublicId", unless = "#result.isLeft()")
    fun getCachedAccountSymmetricKeys(accountPublicId: AccountPublicId): Either<KmsError, List<AccountSymmetricKeyDTO>> =
        left(KmsError.symmetricKeyNotFound(accountPublicId))

    @CacheEvict(cacheNames = [ACCOUNT_SYMMETRIC_KEYS_CACHE_NAME], key = "#accountPublicId")
    override fun evictKeysForAccount(accountPublicId: AccountPublicId) {
        log.debug("Evicted symmetric keys for account $accountPublicId from cache")
    }

    @CacheEvict(cacheNames = [ACCOUNT_SYMMETRIC_KEYS_CACHE_NAME], allEntries = true)
    override fun clearState() {
        log.warn("Cache cleared")
    }

    companion object {
        const val ACCOUNT_SYMMETRIC_KEYS_CACHE_NAME = "${KMS_CACHE_PREFIX}_accountSymmetricKeys"
    }
}
