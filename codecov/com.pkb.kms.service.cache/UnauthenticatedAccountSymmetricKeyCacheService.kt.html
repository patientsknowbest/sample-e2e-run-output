package com.pkb.kms.service.cache

import com.pkb.common.ClearableInternalState
import com.pkb.crypto.SymmetricKey
import com.pkb.crypto.dto.AccountSymmetricKeyDTO
import com.pkb.kms.service.MetricsHelper
import com.pkb.kms.service.extension.publicKey
import com.pkb.kms.service.mapper.SymmetricKeyMapper
import com.pkb.kms.service.repository.AccountSymmetricKeyRepository
import com.pkb.kms.shared.KMS_CACHE_PREFIX
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.KmsError
import io.vavr.control.Either
import io.vavr.kotlin.left
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.cache.annotation.CacheEvict
import org.springframework.cache.annotation.CachePut
import org.springframework.cache.annotation.Cacheable
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles
import javax.transaction.Transactional

/**
 * This cache holds a single symmetric key per account, which can be used for writing data into the account. It is
 * intended to be used when an unauthenticated service, e.g. hl7, needs to write data to an account for which we don't
 * have any keys already available. As a last resort, we will create a new symmetric key for the account, which will
 * then be cached here for future use.
 *
 * Invalidation events:
 *
 * - The account private key is loaded and cached
 */
@Service
class UnauthenticatedAccountSymmetricKeyCacheService(
        private val accountSymmetricKeyRepository: AccountSymmetricKeyRepository,
        private val symmetricKeyMapper: SymmetricKeyMapper,
        private val accountCache: AccountCacheService,
        metrics: MetricsHelper) : ClearableInternalState {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    init {
        metrics.registerCaches(ACCOUNT_SYMMETRIC_KEY_CACHE_NAME)
    }

    @Cacheable(ACCOUNT_SYMMETRIC_KEY_CACHE_NAME, key = "#accountPublicId", unless = "#result.isLeft()")
    fun getCachedAccountSymmetricKey(accountPublicId: AccountPublicId): Either<KmsError, AccountSymmetricKeyDTO> =
        left(KmsError.symmetricKeyNotFound(accountPublicId))

    @CachePut(ACCOUNT_SYMMETRIC_KEY_CACHE_NAME, key = "#accountPublicId", unless = "#result.isLeft()")
    @Transactional
    fun createAccountSymmetricKey(accountPublicId: AccountPublicId): Either<KmsError, AccountSymmetricKeyDTO> =
        accountCache.findAccount(accountPublicId)
                .map {
                    val newSymmetricKey = SymmetricKey.newInstance()
                    val entity = accountSymmetricKeyRepository.save(
                            symmetricKeyMapper.mapNewEntity(it.publicKey().encrypt(newSymmetricKey), it))
                    log.info("Generated $entity for account $accountPublicId")
                    symmetricKeyMapper.map(newSymmetricKey, entity)
                }

    @CacheEvict(cacheNames = [ACCOUNT_SYMMETRIC_KEY_CACHE_NAME], key = "#accountPublicId")
    fun evictKeyForAccount(accountPublicId: AccountPublicId) {
        log.debug("Evicted symmetric key for account $accountPublicId from cache")
    }

    @CacheEvict(cacheNames = [ACCOUNT_SYMMETRIC_KEY_CACHE_NAME], allEntries = true)
    override fun clearState() {
        log.warn("Cache cleared")
    }

    companion object {
        const val ACCOUNT_SYMMETRIC_KEY_CACHE_NAME = "${KMS_CACHE_PREFIX}_accountSymmetricKey"
    }
}
