package com.pkb.kms.service.cache

import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.OrgPrivateId
import com.pkb.kms.shared.representation.PersonPublicId
import com.pkb.kms.shared.structures.pubsub.accountKeyAvailable
import com.pkb.kms.shared.structures.pubsub.orgKeyAvailable
import com.pkb.kms.shared.structures.pubsub.userKeyAvailable
import org.apache.camel.ProducerTemplate
import org.springframework.cache.Cache

class PublishingCache(private val delegate: Cache,
                      private val userKeyAvailableProducer: ProducerTemplate,
                      private val accountKeyAvailableProducer: ProducerTemplate,
                      private val orgKeyAvailableProducer: ProducerTemplate) : Cache by delegate {
    override fun put(key: Any, value: Any?) {
        delegate.put(key, value)
        publishAvailability(key)
    }

    override fun putIfAbsent(key: Any, value: Any?): Cache.ValueWrapper? {
        val ret = delegate.putIfAbsent(key, value)
        publishAvailability(key)
        return ret
    }

    fun publishAvailability(key: Any) {
        when (delegate.name) {
            UserPrivateKeyCacheService.USER_PRIVATE_KEY_CACHE_NAME -> userKeyAvailableProducer.sendBody(userKeyAvailable(key as PersonPublicId))
            UserAccessibleAccountPrivateKeyCacheService.USER_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME -> accountKeyAvailableProducer.sendBody(accountKeyAvailable(key as AccountPublicId))
            OrgAccountKeyCacheService.ORG_ACCOUNT_PRIVATE_KEY_CACHE_NAME -> orgKeyAvailableProducer.sendBody(orgKeyAvailable(key as OrgPrivateId))
        }
    }
}
