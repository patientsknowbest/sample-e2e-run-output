package com.pkb.kms.service.cache

import com.pkb.common.ClearableInternalState
import com.pkb.crypto.CryptoConstants
import com.pkb.crypto.SymmetricKey
import com.pkb.crypto.dto.AccountKeysDTO
import com.pkb.crypto.dto.AccountSymmetricKeyDTO
import com.pkb.crypto.dto.DplKeyDTO
import com.pkb.entities.core.DatapointSymmetricKey
import com.pkb.entities.enums.core.EncryptionAlgorithm
import com.pkb.kms.service.MetricsHelper
import com.pkb.kms.service.repository.DplKeyRepository
import com.pkb.kms.shared.KMS_CACHE_PREFIX
import com.pkb.kms.shared.representation.AccountPrivateId
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.DplKeyId
import com.pkb.kms.shared.representation.KmsError
import io.vavr.control.Either
import io.vavr.control.Option
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.cache.annotation.CacheEvict
import org.springframework.cache.annotation.CachePut
import org.springframework.cache.annotation.Cacheable
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class DplKeyCacheService(private val dplKeyRepository: DplKeyRepository, metrics: MetricsHelper,
                         private val kmsCacheManager: KmsCacheManager) : ClearableInternalState {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    init {
        metrics.registerCaches(DPL_KEY_CACHE_NAME)
    }

    @CachePut(DPL_KEY_CACHE_NAME, key = "{#dplKeyId, #accountId}")
    fun createDplKeyForAccount(dplSymmetricKey: SymmetricKey, dplKeyId: DplKeyId, accountId: AccountPrivateId, accountSymmetricKeyDTO: AccountSymmetricKeyDTO): Either<KmsError, DplKeyDTO> {
        val accountSymmetricKey = SymmetricKey.symmetricKey(accountSymmetricKeyDTO)
        val dplKey = DatapointSymmetricKey()
        val encryptedBytesAndNonce = accountSymmetricKey.encrypt(dplSymmetricKey.keyBytes)
        dplKey.keyId = dplKeyId
        dplKey.accountId = accountId
        dplKey.accountLevelKeyId = accountSymmetricKeyDTO.keyId
        dplKey.encryptionAlgorithm = EncryptionAlgorithm.valueOf(accountSymmetricKey.algorithm)
        dplKey.encryptedKey = encryptedBytesAndNonce.encryptedBytes()
        dplKey.nonce = encryptedBytesAndNonce.nonce()
        dplKeyRepository.save(dplKey)
        return Either.right(DplKeyDTO.dplKeyDTO(dplSymmetricKey, dplKeyId))
    }

    @Cacheable(DPL_KEY_CACHE_NAME, key = "{#dplKeyId, #accountId}", unless = "#result.isLeft()")
    fun getDplKeyForAccountFromCache(accountId: AccountPrivateId, dplKeyId: DplKeyId): Either<KmsError, DplKeyDTO> = Either.left(dplKeyNotFound(dplKeyId, accountId))

    /**
     * Batch version of getDplKeyForAccountFromCache, bypassing the spring cache abstraction to 
     * fetch directly from the backing Map implementation. Results in fewer round trips to Redis.
     */
    fun getDplKeysForAccountsFromCache(keys: List<Pair<AccountPrivateId, DplKeyId>>): Pair<List<DplKeyId>, List<DplKeyDTO>> {
        // This is what Spring does with the SpEL expression on getDplKeyForAccountFromCache
        val cacheKeys = keys.map { (apk, dplki) ->  arrayListOf(dplki, apk) }
        val dplKeys = kmsCacheManager.getAll<Any, Either<KmsError, DplKeyDTO>>(DPL_KEY_CACHE_NAME, cacheKeys)
                .values
                .flatMap { it.asIterable() }
        val keyIdsNotCached = keys.map { it.second }.subtract(dplKeys.map { it.id }).toList()
        return Pair(keyIdsNotCached, dplKeys)
    }
    
    @CachePut(DPL_KEY_CACHE_NAME, key = "{#dplKey.keyId, #dplKey.accountId}", unless = "#result.isLeft()")
    fun decryptAndCacheDplKey(accountKeys: AccountKeysDTO, dplKey: DatapointSymmetricKey): Either<KmsError, DplKeyDTO> =
            doDecryptDplKey(accountKeys, dplKey)

    private fun doDecryptDplKey(accountKeys: AccountKeysDTO, dplKey: DatapointSymmetricKey): Either<KmsError, DplKeyDTO> =
            Option.ofOptional(accountKeys.getSymmetricKey(dplKey.accountLevelKeyId))
                    .map(SymmetricKey::symmetricKey)
                    .map {
                        it.decrypt(dplKey.encryptedKey, dplKey.nonce)
                    }
                    .map { SymmetricKey.symmetricKey(it, CryptoConstants.AES_SHORT_NAME) }
                    .map { DplKeyDTO.dplKeyDTO(it, dplKey.keyId) }
                    .toEither { specificSymmetricKeyNotFound(dplKey.accountLevelKeyId, accountKeys.accountPublicId) }

    @CacheEvict(cacheNames = [DPL_KEY_CACHE_NAME], allEntries = true)
    override fun clearState() {
        log.warn("Cache cleared")
    }

    companion object {
        const val DPL_KEY_CACHE_NAME = "${KMS_CACHE_PREFIX}_dplKey"
    }

    private fun specificSymmetricKeyNotFound(symmetricKeyId: Long, accountPublicId: AccountPublicId) = KmsError(
            "symmetric.key.not.found",
            "Symmetric key with id $symmetricKeyId not found for Account $accountPublicId",
            KmsError.Type.NOT_FOUND)

    private fun dplKeyNotFound(dplKeyId: DplKeyId, accountId: AccountPrivateId) = KmsError(
            "dpl.key.not.found",
            "DPL key with id $dplKeyId not found for Account $accountId",
            KmsError.Type.NOT_FOUND)
}
