package com.pkb.kms.service.cache

import com.pkb.common.ClearableInternalState
import com.pkb.crypto.AccountPrivateKey
import com.pkb.crypto.UserPrivateKey
import com.pkb.crypto.dto.AccountKeysDTO
import com.pkb.crypto.dto.AccountPrivateKeyDTO
import com.pkb.crypto.dto.AccountPrivateKeyDTO.accountPrivateKeyDTO
import com.pkb.kms.service.AccountPath
import com.pkb.kms.service.AccountPrivateKeyDecryptorService
import com.pkb.kms.service.MetricsHelper
import com.pkb.kms.service.OrgAccountKey
import com.pkb.kms.shared.KMS_CACHE_PREFIX
import com.pkb.kms.shared.representation.AccountPublicId
import com.pkb.kms.shared.representation.KmsError
import com.pkb.kms.shared.representation.OrgPrivateId
import io.vavr.control.Either
import io.vavr.kotlin.right
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.cache.annotation.CacheEvict
import org.springframework.cache.annotation.CachePut
import org.springframework.cache.annotation.Cacheable
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class OrgAccessibleAccountPrivateKeyCacheService(private val orgAccountKeyCacheService: OrgAccountKeyCacheService,
                                                 private val decryptorService: AccountPrivateKeyDecryptorService,
                                                 metricsHelper: MetricsHelper) : ClearableInternalState,
                                                                                 IsAccountKeyCache {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    init {
        metricsHelper.registerCaches(ORG_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME)
    }

    /**
     * Returns plaintext key for an account. Will use cached copy if available.
     *
     * If not cached, will retrieve the org key (either plaintext from cache, or encrypted from database and then
     * decrypted with the user key) and use that to decrypt the account key
     */
    @Cacheable(
            ORG_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME,
            key = "#accountPath.accountPublicId",
            unless = "#result.isLeft()")
    fun getAccountPrivateKeyInPath(userPrivateKey: UserPrivateKey,
                                   accountPath: AccountPath): Either<KmsError, AccountPrivateKeyDTO> =
        orgAccountKeyCacheService.getOrDecryptOrgAccountKey(userPrivateKey, accountPath)
                .map { it.accountPrivateKey.decryptAccountPrivateKey(accountPath.encryptedAccountPrivateKey) }
                .map { accountPrivateKeyDTO(it) }

    /**
     * Returns plaintext key for an account. Will use cached copy if available.
     *
     * If not cached, will try to find a plaintext copy of the org key in the path and use that to decrypt the account
     * key. If the org key is not cached return an error because we have no way to get to it without a user key
     */
    @Cacheable(
            ORG_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME,
            key = "#accountPath.accountPublicId",
            unless = "#result.isLeft()")
    fun getAccountPrivateKeyInPathNoUser(accountPath: AccountPath): Either<KmsError, AccountPrivateKeyDTO> =
        decryptorService.decryptAccountPrivateKeyNoUser(accountPath)
                .map { accountPrivateKeyDTO(it) }

    /**
     * Returns plaintext key for an account. Will use cached copy if available.
     *
     * If not cached, will look for cached org keys that can be used to reach the key and use the first one it finds.
     */
    @Cacheable(
            ORG_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME,
            key = "#accountPublicId",
            unless = "#result.isLeft()")
    fun getAccountPrivateKeyViaAnyOrgKey(accountPublicId: AccountPublicId): Either<KmsError, AccountPrivateKeyDTO> =
        decryptAccountPrivateKeyWithAnyCachedOrgKey(accountPublicId)
                .map { accountPrivateKeyDTO(it) }

    /**
     * Stores an account private key in the cache with no further processing (mainly used when migrating keys from the
     * legacy cache)
     */
    @CachePut(
            ORG_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME,
            key = "#accountKeysDTO.accountPublicId",
            unless = "#result.isLeft()")
    fun addToCache(accountKeysDTO: AccountKeysDTO): Either<KmsError, AccountPrivateKeyDTO> =
        right(accountKeysDTO.privateKey)

    @CacheEvict(cacheNames = [ORG_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME], key = "#accountPublicId")
    override fun evictKeysForAccount(accountPublicId: AccountPublicId) {
        log.debug("Evicted keys for account $accountPublicId from cache")
    }

    @CacheEvict(cacheNames = [ORG_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME], allEntries = true)
    override fun clearState() {
        log.warn("Cache cleared")
    }

    /**
     * Looks for any cached org account key that can decrypt the [accountPublicId]'s account key, and if found will
     * return a plaintext [AccountPrivateKey]
     */
    private fun decryptAccountPrivateKeyWithAnyCachedOrgKey(accountPublicId: AccountPublicId): Either<KmsError, AccountPrivateKey> =
            findAnyAvailableOrgAccountKey(accountPublicId)
                    .peek { log.debug("Accessing account key for $accountPublicId via org account key from org ${it.orgId}") }
                    .flatMap { orgAccountKeyCacheService.findAndDecryptAccountKey(it, accountPublicId) }

    fun accountsAccessibleViaCachedOrgKeys(accountPublicIds: List<AccountPublicId>): List<AccountPublicId> {
        // Find the orgs who can access each account, discard the keys just keep the IDs
        val accountIdsByOrgsWhoCanAccess: Map<AccountPublicId, List<OrgPrivateId>> =
                orgAccountKeyCacheService.findOrgIdsWithAccessToAccounts(accountPublicIds)
        // Out of all these orgs, which are cached (again, discard the keys just get the IDs)
        val cachedOrgs = orgAccountKeyCacheService.getCachedOrgAccountKeys(accountIdsByOrgsWhoCanAccess.flatMap { it.value })
                .map { it.orgId }.toSet()
        // Go back through the original map, remove any where none of the org keys are cached.
        return accountIdsByOrgsWhoCanAccess
                .filter { accountWithAccessingOrgs -> accountWithAccessingOrgs.value.any { orgId -> orgId in cachedOrgs } }
                .keys.toList()
    }

    /**
     * Looks for any org that has a copy of the private key for [accountPublicId] where the org account key is currently
     * in the cache, and returns it
     */
    private fun findAnyAvailableOrgAccountKey(accountPublicId: AccountPublicId): Either<KmsError, OrgAccountKey> {
        val orgsWithAccessToKey = orgAccountKeyCacheService.findOrgKeysWithAccessToAccountKey(accountPublicId)
                .map { it.orgId }
        return orgAccountKeyCacheService.getCachedOrgAccountKeys(orgsWithAccessToKey)
                .firstOrNull()
                ?.let { Either.right(it) }
                ?: Either.left(KmsError.accountPrivateKeyNotCached(accountPublicId))
    }

    fun getAccountPrivateKeysViaAnyOrgKey(accountPublicIds: List<AccountPublicId>): Map<AccountPublicId, AccountPrivateKeyDTO> =
            orgAccountKeyCacheService.findOrgIdsWithAccessToAccounts(accountPublicIds)
                .mapNotNull { (accountPublicId, orgIdsWithAccess) ->
                    orgAccountKeyCacheService.getCachedOrgAccountKeys(orgIdsWithAccess)
                        .firstOrNull()
                        ?.let {  accountPublicId to it }
                }
                .let { orgAccessible ->
                    orgAccessible
                        .groupBy({ it.second }, { it.first })
                        .flatMap { (orgAccountKey, accountPublicIds) ->
                            orgAccountKeyCacheService.findAndDecryptAccountKeys(orgAccountKey, accountPublicIds).entries
                        }
                        .associate { it.key to accountPrivateKeyDTO(it.value) }
                }


    companion object {
        const val ORG_ACCESSIBLE_ACCOUNT_PRIVATE_KEY_CACHE_NAME = "${KMS_CACHE_PREFIX}_orgAccessibleAccountPrivateKey"
    }
}
