package com.pkb.kms.service.cache

import org.apache.camel.ProducerTemplate
import org.springframework.cache.Cache
import org.springframework.cache.CacheManager

class PublishingCacheManager(private val delegate: CacheManager,
                             private val userKeyAvailableProducer: ProducerTemplate,
                             private val accountKeyAvailableProducer: ProducerTemplate,
                             private val orgKeyAvailableProducer: ProducerTemplate) : CacheManager by delegate {
    override fun getCache(name: String): Cache? {
        return delegate.getCache(name)?.let { PublishingCache(it, userKeyAvailableProducer, accountKeyAvailableProducer, orgKeyAvailableProducer) }
    }
}
