package com.pkb.kms.service.cache

import com.pkb.common.ClearableInternalState
import com.pkb.kms.service.MetricsHelper
import com.pkb.kms.service.projection.TeamIds
import com.pkb.kms.service.repository.TeamRepository
import com.pkb.kms.shared.KMS_CACHE_PREFIX
import com.pkb.kms.shared.representation.KmsError
import com.pkb.kms.shared.representation.TeamPublicId
import io.vavr.control.Either
import io.vavr.control.Option
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.cache.annotation.CacheEvict
import org.springframework.cache.annotation.Cacheable
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class TeamCacheService(private val teamRepository: TeamRepository,
                       metrics: MetricsHelper) : ClearableInternalState {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    init {
        metrics.registerCaches(TEAM_CACHE_NAME)
    }

    @Cacheable(TEAM_CACHE_NAME, key = "#teamPublicId", unless = "#result.isLeft()")
    fun findTeam(teamPublicId: TeamPublicId): Either<KmsError, TeamIds> =
            Option.ofOptional(teamRepository.findByPublicId(teamPublicId))
                    .toEither(teamNotFound(teamPublicId))

    private fun teamNotFound(teamPublicId: TeamPublicId): KmsError =
            KmsError("team.not.found", "Team not found $teamPublicId", KmsError.Type.NOT_FOUND)


    @CacheEvict(cacheNames = [TEAM_CACHE_NAME], allEntries = true)
    override fun clearState() {
        log.warn("Cache cleared")
    }

    companion object {
        const val TEAM_CACHE_NAME = "${KMS_CACHE_PREFIX}_team"
    }
}
