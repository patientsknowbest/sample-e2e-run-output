package com.pkb.kms.service.cache

import org.redisson.RedissonMap
import org.springframework.cache.CacheManager
import org.springframework.stereotype.Service
import kotlin.reflect.KClass
import kotlin.reflect.full.cast

@Service
class KmsCacheManager(private val cacheManager: CacheManager) {
    fun <T : Any> getAllKeysInCache(cacheName: String, klass: KClass<T>): Set<T> {
        return getCacheMap<T, Any>(cacheName).keys.map { klass.cast(it) }.toSet()
    }

    fun <T : Any> getAllKeysInCaches(cacheNames: List<String>, klass: KClass<T>): Set<T> {
        return cacheNames.map { getAllKeysInCache(it, klass) }.flatten().toSet()
    }

    fun <K : Any, V : Any> iterateAllEntries(cacheName: String, fn: (K, V) -> Unit) =
            getCacheMap<K, V>(cacheName).entrySet(null).forEach { fn(it.key, it.value) }

    fun deleteCache(cacheName: String): Boolean {
        return getCacheMap<Any, Any>(cacheName).delete()
    }

    fun <K, V> getAll(cacheName: String, keys: Iterable<K>): Map<K, V> =
            getCacheMap<K, V>(cacheName).getAll(keys.toSet())

    fun <K, V> putAll(cacheName: String, entries: Map<K, V>) {
        // We're bypassing the spring cache implementation here, so we need to go 
        // find it and publish ourselves. Yes this is tightly coupling some things unfortunately.
        getCacheMap<K, V>(cacheName).putAll(entries)
        val cache = cacheManager.getCache(cacheName) ?: throw RuntimeException("Cache $cacheName not found")
        if (cache is PublishingCache) {
            // Would be better if Camel had support for pipelining or sending multiple messages in one hit
            entries.keys.forEach { cache.publishAvailability(it as Any) }
        }
    }

    private fun <K, V> getCacheMap(cacheName: String): RedissonMap<K, V> {
        val cache = cacheManager.getCache(cacheName)
                ?: throw RuntimeException("Cache $cacheName not found")
        @Suppress("UNCHECKED_CAST")
        return (cache.nativeCache as? RedissonMap<K, V>)
                ?.let { it }
                ?: throw RuntimeException("Cache type ${cache.nativeCache.javaClass} not recognised")
    }
}
