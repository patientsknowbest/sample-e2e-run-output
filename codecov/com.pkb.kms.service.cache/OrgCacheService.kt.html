package com.pkb.kms.service.cache

import com.pkb.common.ClearableInternalState
import com.pkb.kms.service.MetricsHelper
import com.pkb.kms.service.projection.OrgIds
import com.pkb.kms.service.repository.OrgRepository
import com.pkb.kms.shared.KMS_CACHE_PREFIX
import com.pkb.kms.shared.representation.KmsError
import com.pkb.kms.shared.representation.OrgPublicId
import io.vavr.control.Either
import io.vavr.control.Option
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.cache.annotation.CacheEvict
import org.springframework.cache.annotation.Cacheable
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class OrgCacheService(private val orgRepository: OrgRepository,
                      metrics: MetricsHelper) : ClearableInternalState {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    init {
        metrics.registerCaches(ORG_CACHE_NAME)
    }

    @Cacheable(ORG_CACHE_NAME, key = "#orgPublicId", unless = "#result.isLeft()")
    fun findOrg(orgPublicId: OrgPublicId): Either<KmsError, OrgIds> =
            Option.ofOptional(orgRepository.findIdsByPublicId(orgPublicId))
                    .toEither(KmsError.orgNotFound(orgPublicId))


    @CacheEvict(cacheNames = [ORG_CACHE_NAME], allEntries = true)
    override fun clearState() {
        log.warn("Cache cleared")
    }

    companion object {
        const val ORG_CACHE_NAME = "${KMS_CACHE_PREFIX}_org"
    }
}
