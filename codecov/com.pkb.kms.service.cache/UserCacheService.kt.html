package com.pkb.kms.service.cache

import com.pkb.common.ClearableInternalState
import com.pkb.kms.service.MetricsHelper
import com.pkb.kms.service.projection.PersonIds
import com.pkb.kms.service.projection.UserIdAndPKCS12
import com.pkb.kms.service.repository.PersonRepository
import com.pkb.kms.service.repository.UserCredentialRepository
import com.pkb.kms.shared.KMS_CACHE_PREFIX
import com.pkb.kms.shared.representation.KmsError
import com.pkb.kms.shared.representation.PersonPublicId
import io.vavr.control.Either
import io.vavr.kotlin.left
import io.vavr.kotlin.right
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.cache.annotation.CacheEvict
import org.springframework.cache.annotation.CachePut
import org.springframework.cache.annotation.Cacheable
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class UserCacheService(private val userCredentialRepository: UserCredentialRepository,
                       private val personRepository: PersonRepository,
                       metrics: MetricsHelper) : ClearableInternalState, HasDeprecatedCaches {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    init {
        metrics.registerCaches(PERSON_CACHE_NAME, CREDENTIALS_CACHE_NAME)
    }

    @Cacheable(PERSON_CACHE_NAME, key = "#personPublicId", unless = "#result.isLeft()")
    fun findPerson(personPublicId: PersonPublicId): Either<KmsError, PersonIds> =
            personRepository.findByPublicId(personPublicId, PersonIds::class.java)
                    .map { right<KmsError, PersonIds>(it) }
                    .orElseGet { left(KmsError.personNotFound(personPublicId)) }

    /**
     * Returns cached credentials if they exist, otherwise attempts to load them and caches the result
     */
    @Cacheable(CREDENTIALS_CACHE_NAME, key = "#personIds.publicId", unless = "#result.isLeft()")
    fun findCredentials(personIds: PersonIds) = loadCredentials(personIds)

    /**
     * Ignores any cached credentials and loads them anew, then caches the result (useful if e.g. the keystore
     * password has changed)
     */
    @CachePut(CREDENTIALS_CACHE_NAME, key = "#personIds.publicId", unless = "#result.isLeft()")
    fun findCredentialsNoCache(personIds: PersonIds) = loadCredentials(personIds)

    @CacheEvict(cacheNames = [PERSON_CACHE_NAME, CREDENTIALS_CACHE_NAME], allEntries = true)
    override fun clearState() {
        log.warn("Cache cleared")
    }

    @CacheEvict(cacheNames = [CREDENTIALS_CACHE_NAME], key = "#personPublicId")
    fun evictUserCredentials(personPublicId: PersonPublicId) {
        log.debug("Evicted credentials for person $personPublicId from cache")
    }

    private fun loadCredentials(personIds: PersonIds): Either<KmsError, UserIdAndPKCS12> =
        userCredentialRepository.findByUserId(personIds.id)
                .map { right<KmsError, UserIdAndPKCS12>(it) }
                .orElseGet { left(KmsError.userCredentialNotFound(personIds.publicId)) }

    override val deprecatedCaches = setOf(DEPRECATED_PERSON_CACHE_NAME, DEPRECATED_CREDENTIALS_CACHE_NAME)

    private fun KmsError.Companion.userCredentialNotFound(personPublicId: PersonPublicId) = KmsError(
            "user.credential.not.found",
            "No credential found for person $personPublicId",
            KmsError.Type.NOT_FOUND)

    companion object {
        const val PERSON_CACHE_NAME = "${KMS_CACHE_PREFIX}_person2"
        const val CREDENTIALS_CACHE_NAME = "${KMS_CACHE_PREFIX}_userCredentials2"

        const val DEPRECATED_PERSON_CACHE_NAME = "${KMS_CACHE_PREFIX}_person"
        const val DEPRECATED_CREDENTIALS_CACHE_NAME = "${KMS_CACHE_PREFIX}_userCredentials"
    }
}
