package com.pkb.authz;

import io.vavr.collection.Set;
import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.util.UUID;

import static com.google.common.base.Preconditions.checkArgument;
import static io.vavr.API.Set;
import static java.lang.String.format;
import static java.util.stream.Collectors.joining;
import static org.apache.commons.lang3.StringUtils.EMPTY;
import static org.apache.commons.lang3.StringUtils.trimToEmpty;

/**
 * The result of an authorization check where the target couldn't be found.
 */
public class NotFoundException extends RuntimeException {
    private static final String INVALID_TARGET_ID_TYPE = "Supplied target id type is invalid. Accepted types are %s";
    private static final Set<Class<?>> TARGET_ID_TYPES = Set(Long.class, UUID.class);

    private static final long serialVersionUID = 1L;

    public static NotFoundException notFoundForTargetOwner(@NotNull UUID targetOwnerId, @NotNull Class<?> targetType) {
        return new NotFoundException(null, targetOwnerId, targetType);
    }

    public static NotFoundException notFoundTarget(@NotNull Object targetId, @NotNull Class<?> targetType) {
        return new NotFoundException(targetId, null, targetType);
    }

    public static NotFoundException notFoundTarget(@NotNull Object targetId, @NotNull UUID targetOwnerId, @NotNull Class<?> targetType) {
        return new NotFoundException(targetId, targetOwnerId, targetType);
    }

    // Should be a Long or a UUID, but there's no base type for these.
    private final Object targetId;
    private final UUID targetOwnerId;
    private final Class<?> targetType;

    /** @deprecated Use {@link #notFoundTarget(long, Class)}, {@link #notFoundTarget(UUID, Class)} or {@link #notFoundForTargetOwner(UUID, Class)} instead. */
    @Deprecated
    public NotFoundException(@NotNull Object targetId, @NotNull Class<?> targetType) {
        super(format("Target not found: %s, targetType=[%s]", targetId, targetType));
        this.targetId = targetId;
        this.targetOwnerId = null;
        this.targetType = targetType;
    }

    private NotFoundException(@Nullable Object targetId, @Nullable UUID targetOwnerId, @NotNull Class<?> targetType) {
        super(format("Target not found:%s targetType=[%s]", targetIds(targetId, targetOwnerId), targetType));
        checkArgument(targetId == null || TARGET_ID_TYPES.contains(targetId.getClass()), INVALID_TARGET_ID_TYPE, TARGET_ID_TYPES.map(Class::getSimpleName));
        this.targetId = targetId;
        this.targetOwnerId = targetOwnerId;
        this.targetType = targetType;
    }

    public Object getTargetId() {
        return targetId;
    }

    public Option<?> findTargetId() {
        return Option.of(targetId);
    }

    public Option<?> findTargetOwnerId() {
        return Option.of(targetOwnerId);
    }

    public Class<?> getTargetType() {
        return targetType;
    }

    private static String targetIds(@Nullable Object targetId, @Nullable UUID targetOwnerId) {
        return trimToEmpty(" " + Set(
                targetId == null ? EMPTY : format("targetId=[%s]", targetId),
                targetOwnerId == null ? EMPTY : format("targetOwnerId=[%s]", targetOwnerId))
                        .filter(StringUtils::isNotBlank)
                        .collect(joining(", ")));
    }
}
