package com.pkb.authz;

import com.pkb.authz.permission.Justification;
import com.pkb.authz.permission.Permission;
import io.vavr.collection.List;
import io.vavr.collection.Set;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;

import java.util.Comparator;

/**
 * The result of a failed authorization check where the data exists, but you are not allowed to see it.
 */
public class NotAuthorizedException extends RuntimeException {

    private static final long serialVersionUID = 1L;

    private final List<? extends Justification> justifications;

    public <T> NotAuthorizedException(@NotNull TargetMeta<T> targetMeta, @NotNull List<Permission> permissions) {
        this(targetMeta, permissions
                .filter(Permission::denied)
                .flatMap(Permission::justifications)
                .toSet());
    }

    public <T> NotAuthorizedException(@NotNull TargetMeta<T> targetMeta, @NotNull Set<? extends Justification> failures) {
        super(targetMeta.targetType().getSimpleName() + ": " +
                JustificationUtils.stringify(failures).getOrElseThrow(IllegalArgumentException::new));
        this.justifications = failures.toSortedSet(Comparator.comparing(Justification::stringify)).toList();
    }

    public @NotNull List<? extends Justification> getJustifications() {
        return justifications;
    }

    public boolean contains(Class<? extends Justification> justification) {
        return justifications.exists(justification::isInstance);
    }

    public <J extends Justification> Option<J> findFirst(Class<J> justification) {
        return justifications
                .filter(justification::isInstance)
                .map(justification::cast)
                .headOption();
    }
}
