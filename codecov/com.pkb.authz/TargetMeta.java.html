package com.pkb.authz;

import io.vavr.collection.Set;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.immutables.value.Value;
import org.immutables.value.Value.Auxiliary;
import org.immutables.value.Value.Derived;
import org.immutables.value.Value.Immutable;
import org.immutables.vavr.encodings.VavrEncodingEnabled;
import org.jetbrains.annotations.NotNull;

import java.util.UUID;

/**
 * <pre>
 * Metadata for some data in the PKB domain.
 * Used by access control code to represent the data being accessed before it's actually fetched.
 *
 * One {@link TargetMeta} is meant to represent either exactly one piece of data (identified by one or more IDs and optionally by targeted patient ID) or
 * a particular Patients data (only Patient ID is provided in this case and no data point ID).
 * </pre>
 *
 * @param <O>
 *            target type (Should be a type from the PKBDomain e.g. {@link com.pkb.datamodel.user.Patient}
 */
@Immutable
@Value.Style(of = "targetMeta", allParameters = true, strictBuilder = true, depluralize = true, get = { "is*",
        "get*" }, visibility = Value.Style.ImplementationVisibility.PUBLIC, passAnnotations = { NotNull.class })
@VavrEncodingEnabled
@SuppressWarnings("varargs")
public interface TargetMeta<O> {
    @NotNull
    Class<? extends O> targetType();

    /**
     * Should contain the ID or the UUID (or both) of the target being accessed.
     * If more than one id is present, then all have to identify the same target.
     */
    @NotNull
    Set<Object> identifiers();

    @Auxiliary
    @Derived
    default @NotNull Set<Class<?>> identifierTypes() {
        return identifiers().map(Object::getClass);
    }

    default <ID> Option<ID> findIdentifier(Class<ID> type) {
        return identifiers()
                .filter(type::isInstance).map(type::cast)
                .headOption();
    }

    default <ID1, ID2> @NotNull Option<Either<ID1, ID2>> findIdentifier(@NotNull Class<ID1> type1, @NotNull Class<ID2> type2) {
        Option<ID1> maybeIdentifier = findIdentifier(type1);
        return maybeIdentifier.isDefined() ? maybeIdentifier.map(Either::left) : findIdentifier(type2).map(Either::right);
    }

    /**
     * We don't necessarily have the target IDs we are looking for - for search operations I know the person(s)
     * ID but not the target ID.
     *
     * Searches across a bounded list of people should use multiple TargetMetas to represent that.
     */
    Option<UUID> personId();

    @Value.Default
    default boolean ownedByPerson() {
        return true;
    }

    default boolean containsAnyIdentifier() {
        return identifiers().nonEmpty() || personId().isDefined();
    }

    default UUID requireUUID() {
        return findIdentifier(UUID.class).getOrElseThrow(() -> new IllegalStateException("UUID type identifier not found"));
    }
}
