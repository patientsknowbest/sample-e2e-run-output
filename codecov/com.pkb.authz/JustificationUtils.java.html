package com.pkb.authz;

import com.pkb.authz.permission.Justification;
import io.vavr.collection.List;
import io.vavr.collection.Set;
import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;

import java.util.stream.Stream;

import static com.pkb.authz.permission.ImmutableTextualJustification.justification;
import static java.lang.String.format;
import static java.util.stream.Collectors.joining;
import static java.util.stream.Stream.concat;

public final class JustificationUtils {

    public static List<Justification> merge(List<Justification> justifications1, List<Justification> justifications2) {
        return justifications1.appendAll(justifications2);
    }

    public static Justification textual(@NotNull String justification) {
        return justification(justification);
    }

    public static Justification textual(@NotNull String justificationFormat, @NotNull Object argument,
            Object... arguments) {
        return justification(format(justificationFormat, concat(Stream.of(argument), Stream.of(arguments)).toArray()));
    }

    public static Option<String> stringify(Set<? extends Justification> justifications) {
        String message = justifications.map(Justification::stringify).toSortedSet().collect(joining(", "));
        return Option.of(message).filter(StringUtils::isNotBlank);
    }

    private JustificationUtils() {
    }
}
