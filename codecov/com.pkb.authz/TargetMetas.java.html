package com.pkb.authz;

import org.jetbrains.annotations.NotNull;

import static io.vavr.API.Set;

public class TargetMetas {
    public static <O> TargetMeta<O> targetMeta(@NotNull Class<? extends O> objectType, @NotNull Class<?>... idTypes) {
        return ImmutableTargetMeta.<O>builder().targetType(objectType).build();
    }

    public static <O> TargetMeta<O> targetMeta(@NotNull Class<? extends O> objectType, @NotNull Object... ids) {
        return ImmutableTargetMeta.<O>builder().targetType(objectType).identifiers(Set(ids)).build();
    }
    public static <O> TargetMeta<O> targetMeta(@NotNull Class<? extends O> objectType, boolean isOwnedByPerson, @NotNull Object... ids) {
        return ImmutableTargetMeta.<O>builder().targetType(objectType).ownedByPerson(isOwnedByPerson).identifiers(Set(ids)).build();
    }

    public static <O> TargetMeta<O> targetMeta(@NotNull Class<? extends O> objectType, boolean isOwnedByPerson, @NotNull Class<?>... idTypes) {
        return ImmutableTargetMeta.<O>builder().targetType(objectType).ownedByPerson(isOwnedByPerson).build();
    }
}
