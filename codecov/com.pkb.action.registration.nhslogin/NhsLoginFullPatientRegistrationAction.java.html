package com.pkb.action.registration.nhslogin;

import com.pkb.entities.enums.Route;
import com.pkb.institute.entity.Team;
import com.pkb.service.nhslogin.impl.ImmutableNhsLoginPatientMetadata;
import com.pkb.service.nhslogin.impl.NhsLoginRegistrationManager;
import com.pkb.service.team.TeamManager;
import com.pkb.util.TermsAndConditionsConfigDTO;
import org.apache.struts2.interceptor.validation.SkipValidation;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.List;
import java.util.Map;
import java.util.Optional;

import static com.pkb.util.AppIdentityUtil.AppIdentity.PKB_WEB;

/**
 * Registers a patient <em>without</em> an existing PKB record after successful NHS Login autentication.
 * Implemented in NHS Login Phase 3.
 */
public class NhsLoginFullPatientRegistrationAction extends NhsLoginBasePatientRegistrationAction {

    private String teamCode;

    private Team team;

    private NhsLoginFullPatientRegistrationDTO fullRegistrationDetails;

    private TermsAndConditionsConfigDTO termsAndConditionsConfig;

    private boolean isSelfReg;

    @Autowired
    private NhsLoginRegistrationManager nhsLoginRegistrationManager;

    @Autowired
    private TeamManager teamManager;

    @Override
    @SkipValidation
    public String displayForm() {
        setTeamBy(teamCode);
        activationDetails = (NhsLoginFullPatientRegistrationDTO) request.getSession().getAttribute("activationDetails");
        termsAndConditionsConfig = buildTermsAndConditionsConfig();
        return INPUT;
    }

    @Override
    void updateFormValidation() {
        super.updateFormValidation();
        fullRegistrationDetails = (NhsLoginFullPatientRegistrationDTO) activationDetails;
        setTeamBy(isSelfReg ? teamCode : fullRegistrationDetails.getTeamMembership());

        if (getTeamBy(teamCode) == null || isSelfReg) {
            Map<String, List<String>> fieldErrors = getFieldErrors();
            fieldErrors.remove("activationDetails.teamMembership");
            setFieldErrors(fieldErrors);
        }
    }

    @Override
    String doPatientRegistration() {
        String action = SUCCESS;
        Route route = nhsLoginInfo().getDetails().getAppIdentity() == PKB_WEB ? Route.WEBAPP : Route.WEBAPP_NHS_APP;

        try {
            fullRegistrationDetails.setLanguageCode(getLocale().getLanguage());
            nhsLoginRegistrationManager.registerPatient(getEHRRequestContext().withAccessRoute(route), ImmutableNhsLoginPatientMetadata.builder()
                    .patient(activationDetailsConverter.createPersonDtoFrom(fullRegistrationDetails, nhsLoginInfo().getDetails()))
                    .password(j_password)
                    .securityQuestion(fullRegistrationDetails.getSecurityQuestion())
                    .securityQuestionAnswer(fullRegistrationDetails.getAnswerForSecurityQuestion())
                    .nhsLoginCredentials(nhsLoginInfo().getCredentials())
                    .route(nhsLoginInfo().getDetails().getAppIdentity() == PKB_WEB ? Route.WEBAPP : Route.WEBAPP_NHS_APP)
                    .team(Optional.ofNullable(team))
                    .build());
        } catch (Exception e) {
            LOGGER.error("Unable to register new user through NHS Login", e);
            action = ERROR;
        } finally {
            request.getSession().removeAttribute("activationDetails");
        }

        return action;
    }

    @Override
    public boolean isFullNhsLoginRegistration() {
        return true;
    }

    @Override
    public NhsLoginPatientActivationDTO getActivationDetails() {
        if (activationDetails == null) {
            activationDetails = new NhsLoginFullPatientRegistrationDTO();
        }
        return activationDetails;
    }

    public String getTeamCode() {
        return teamCode;
    }

    public void setTeamCode(String teamCode) {
        this.teamCode = teamCode;
    }

    public boolean isOrgSelectorVisible() {
        if (config.isNhsLoginWebUiButtonEnabledForSelfRegistration()) {
            return team != null && !isSelfReg;
        }

        return team != null;
    }

    private Team getTeamBy(String code) {
        return teamManager.getTeamByCode(code);
    }

    private void setTeamBy(String code) {
        this.team = getTeamBy(code);
    }

    public String getOrgName() {
        return team.getOrg().getName();
    }

    public String getAction() {
        return "nhsLoginFullPatient";
    }

    public boolean getSelfReg() {
        return isSelfReg;
    }

    public void setIsSelfReg(boolean isSelfReg) {
        this.isSelfReg = isSelfReg;
    }

    public TermsAndConditionsConfigDTO getTermsAndConditionsConfig() {
        return termsAndConditionsConfig;
    }
}
