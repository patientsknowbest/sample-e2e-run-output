package com.pkb.action.registration.nhslogin;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.action.BaseAction;
import com.pkb.action.registration.PatientActivationDetailsDTOConverter;
import com.pkb.datamodel.Email;
import com.pkb.datamodel.ImmutableDisclosingEmail;
import com.pkb.model.PKBSecurityQuestion;
import com.pkb.security.BaseAuthenticationProvider.TypedAuthentication;
import com.pkb.security.NhsLoginAuthenticationDetails;
import com.pkb.util.CorrelationIdUtil;
import com.pkb.util.validation.stereotypes.PKBPassword;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Autowired;

import javax.inject.Inject;
import javax.validation.Valid;
import javax.validation.constraints.NotBlank;
import java.util.List;
import java.util.Map;

import static com.pkb.action.registration.AbstractPatientActivationDTO.PASSWORD_MUST_MATCH_WITH_PASSWORD_CONFIRMATION;
import static com.pkb.action.registration.letter.StartLetterInvitedPatientActivationAction.setQuestionText;
import static com.pkb.util.Constants.NHS_LOGIN_INFO;

public abstract class NhsLoginBasePatientRegistrationAction extends BaseAction {

    String nhsLoginError;

    Email j_username;

    @PKBPassword
    @NotBlank(message = "password.cannot.be.blank")
    String j_password;

    @Valid
    NhsLoginPatientActivationDTO activationDetails;

    @Inject
    PatientActivationDetailsDTOConverter activationDetailsConverter;

    @Autowired
    CorrelationIdUtil correlationIdUtil;

    abstract String displayForm();

    abstract String doPatientRegistration();

    public String activatePatient() {
        updateFormValidation();
        if (isRequestNotValid()) {
            cleanUpAndSetDetailsInSession();
            return displayForm();
        }

        setQuestionText(activationDetails, getQuestions());
        setSpringCredentials(activationDetails);
        return doPatientRegistration();
    }

    void updateFormValidation() {
        if (!j_password.equals(activationDetails.getPasswordConfirmation())) {
            addFieldError("j_password", getText(PASSWORD_MUST_MATCH_WITH_PASSWORD_CONFIRMATION));
        }
        Map<String, List<String>> fieldErrors = getFieldErrors();
        fieldErrors.remove("activationDetails.email");
        fieldErrors.remove("activationDetails.password");
        setFieldErrors(fieldErrors);
    }

    void setSpringCredentials(@NotNull NhsLoginPatientActivationDTO activationDetails) {
        activationDetails.setEmail(j_username);
        activationDetails.setPassword(j_password);
    }

    private void cleanUpAndSetDetailsInSession() {
        activationDetails.setPassword(null);
        activationDetails.setPasswordConfirmation(null);
        request.getSession().setAttribute("activationDetails", activationDetails);
    }

    public Email getEmailValue() {
        return Option.of(nhsLoginInfo().getName()).map(ImmutableDisclosingEmail::disclosingEmail).getOrNull();
    }

    TypedAuthentication<NhsLoginAuthenticationDetails> nhsLoginInfo() {
        return (TypedAuthentication<NhsLoginAuthenticationDetails>) request.getSession().getAttribute(NHS_LOGIN_INFO);
    }

    public List<PKBSecurityQuestion> getQuestions() {
        return pkbWebUtil.getSecretQuestions(getLocale());
    }

    String unexpectedError() {
        return error("unexpected-error");
    }

    String error(String error) {
        setNhsLoginError(error);
        return ERROR;
    }

    public String getNhsLoginError() {
        return nhsLoginError;
    }

    @VisibleForTesting
    void setNhsLoginError(String error) {
        this.nhsLoginError = error;
    }

    public void setJ_username(Email j_username) {
        this.j_username = j_username;
    }

    public void setJ_password(String j_password) {
        this.j_password = j_password;
    }

    public NhsLoginPatientActivationDTO getActivationDetails() {
        return activationDetails;
    }

    public void setActivationDetails(NhsLoginPatientActivationDTO activationDetails) {
        this.activationDetails = activationDetails;
    }

    public boolean isFullNhsLoginRegistration() {
        return false;
    }
}
