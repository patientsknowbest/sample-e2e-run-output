package com.pkb.action.registration.nhslogin;

import com.pkb.datamodel.user.Patient;
import com.pkb.datamodel.user.PersonIdsWithNhsLoginStatus;
import com.pkb.patient.PatientActivationManager;
import com.pkb.patient.activation.nhslogin.NhsLoginAccountDetailsDTO;
import com.pkb.util.TermsAndConditionsConfigDTO;
import org.apache.struts2.interceptor.validation.SkipValidation;

import javax.inject.Inject;
import java.util.Optional;

import static com.pkb.util.Constants.NHS_LOGIN_IDS;

/**
 * Registers a patient <em>with</em> an existing PKB record after successful NHS Login autentication.
 * Implemented in NHS Login Phase 2.
 */
public class NhsLoginPatientRegistrationAction extends NhsLoginBasePatientRegistrationAction {

    @Inject
    private PatientActivationManager patientActivationManager;

    private TermsAndConditionsConfigDTO termsAndConditionsConfig;

    @Override
    @SkipValidation
    public String displayForm() {
        activationDetails = (NhsLoginPatientActivationDTO) request.getSession().getAttribute("activationDetails");
        termsAndConditionsConfig = buildTermsAndConditionsConfig();
        return nhsLoginStatus().map(personIdsWithNhsLoginStatus -> {
            if (personIdsWithNhsLoginStatus.getAccountPublicId().isEmpty()) {
                return unexpectedError();
            } else {
                /* First page load otherwise activationDetails is populated from an invalid form submission */
                if (activationDetails == null) {
                    activationDetails = new NhsLoginPatientActivationDTO();
                    Optional.ofNullable(userManager.getPKBPerson(personIdsWithNhsLoginStatus.getId()))
                            .ifPresent(pkbPerson -> activationDetailsConverter.copyGenderAndAddressFields(pkbPerson, activationDetails));
                }
                return INPUT;
            }
        }).orElseGet(this::unexpectedError);
    }

    @Override
    public String doPatientRegistration() {
        termsAndConditionsConfig = buildTermsAndConditionsConfig();
        return nhsLoginStatus().map(status -> {
            NhsLoginAccountDetailsDTO accountDetails = activationDetailsConverter.createNhsLoginAccountDetailsFrom(activationDetails);
            Patient patient = patientActivationManager.activateNhsLoginPatient(status.getId(), status.getPublicId(), getEmailValue(), accountDetails);
            Optional.ofNullable(userManager.getPKBPerson(status.getId()))
                    .filter(pkbPerson -> activationDetailsConverter.copyGenderAndAddressFields(activationDetails, pkbPerson))
                    .ifPresent(pkbPerson -> userManager.updateUser(getEHRRequestContext().withRegisteringUser(status.getId()), pkbPerson));
            return status.getAccountPublicId().map(accountPublicId -> {
                if (!userManager.recreateOwnerAccessToAccount(correlationIdUtil.getAsString(), accountPublicId, status.getPublicId()).succesfullyRecreated()) {
                    userManager.recordOutOfAreaRepairJob(patient);
                    return error("out.of.area");
                }
                patientActivationManager.notifyNhsLoginPatientOfClaimedRecord(status.getId());
                return SUCCESS;
            }).getOrElse(this::unexpectedError);
        }).orElseGet(this::unexpectedError);
    }

    @Override
    public NhsLoginPatientActivationDTO getActivationDetails() {
        if (activationDetails == null) {
            activationDetails = new NhsLoginPatientActivationDTO();
        }
        return activationDetails;
    }

    Optional<PersonIdsWithNhsLoginStatus> nhsLoginStatus() {
        return Optional.ofNullable((PersonIdsWithNhsLoginStatus) request.getSession().getAttribute(NHS_LOGIN_IDS));
    }

    public String getAction() {
        return "nhsLoginPatient";
    }

    public TermsAndConditionsConfigDTO getTermsAndConditionsConfig() {
        return termsAndConditionsConfig;
    }

    public void setTermsAndConditionsConfig(TermsAndConditionsConfigDTO termsAndConditionsConfig) {
        this.termsAndConditionsConfig = termsAndConditionsConfig;
    }
}
