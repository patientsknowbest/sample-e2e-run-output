package com.pkb.action.support;

import com.pkb.app.entity.SourceDetails;
import com.pkb.source.support.SourceDetailsViewFactory;
import com.pkb.test.entity.TestResultDTO;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.stream.Collectors;

/**
 * Please see AppointmentTransformer.
 */
@Component
public class TestResultTransformer {

    private SourceDetailsViewFactory sourceDetailsViewFactory;

    @Autowired
    public TestResultTransformer(SourceDetailsViewFactory sourceDetailsViewFactory) {
        this.sourceDetailsViewFactory = sourceDetailsViewFactory;
    }

    public TestResultDTO transformTestResult(TestResultDTO testResultDto){
        if(testResultDto != null){
            testResultDto.setSource(sourceDetailsViewFactory.createSourceDetailsView(testResultDto.getSource()));
        }
        return testResultDto;
    }


    public List<TestResultDTO> transformTestResult(List<TestResultDTO> testResults){
        return testResults.stream().map(this::transformTestResult).collect(Collectors.toList());
    }

    public TestResultDTO transformTestResultView(TestResultDTO testResultDTO) {
        if(testResultDTO != null){
            SourceDetails sourceDetails = new SourceDetails();
            sourceDetails.init(testResultDTO.getSource());
            testResultDTO.setSource(sourceDetails);
        }
        return testResultDTO;
    }
}
