package com.pkb.action.support;

import com.pkb.coding.entity.CodeableConcept;
import com.pkb.diagnosis.entity.Diagnosis;
import com.pkb.dto.DiagnosisGroupDto;
import com.pkb.entities.enums.Route;
import com.pkb.service.datapoint.DataWithUIPermissions;
import com.pkb.source.support.SourceDetailsLocalizedDisplayProvider;
import com.pkb.source.support.SourceDetailsViewFactory;
import io.vavr.Tuple;
import io.vavr.Tuple5;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.time.temporal.ChronoUnit;
import java.util.Comparator;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;

import static java.util.Collections.singletonList;
import static java.util.stream.Collectors.toList;

/**
 * Please see AppointmentTransformer.
 */
@Component
public class DiagnosisGroupTransformer {

    public final Comparator<DiagnosisGroupDto> BY_REVERSE_START_DATE_THEN_ALPHABETIC_NAME = Comparator.comparing(DiagnosisGroupDto::getFirstStartDate,
            Comparator.nullsFirst(Comparator.comparing(d -> d.toInstant().truncatedTo(ChronoUnit.DAYS))))
            .reversed()
            .thenComparing(this::getTextForSorting, String::compareToIgnoreCase);
    public final Comparator<DiagnosisGroupDto> BY_REVERSE_END_DATE_THEN_ALPHABETIC_NAME = Comparator
            .nullsFirst(Comparator.comparing(DiagnosisGroupDto::getLastEndDate,
                    Comparator.nullsFirst(Comparator.comparing(d -> d.toInstant().truncatedTo(ChronoUnit.DAYS)))))
            .reversed()
            .thenComparing(this::getTextForSorting, String::compareToIgnoreCase);

    private SourceDetailsLocalizedDisplayProvider sourceDetailsLocalizedDisplayProvider;
    private SourceDetailsViewFactory sourceDetailsViewFactory;

    @Autowired
    public DiagnosisGroupTransformer(SourceDetailsLocalizedDisplayProvider sourceDetailsLocalizedDisplayProvider, SourceDetailsViewFactory sourceDetailsViewFactory) {
        this.sourceDetailsLocalizedDisplayProvider = sourceDetailsLocalizedDisplayProvider;
        this.sourceDetailsViewFactory = sourceDetailsViewFactory;
    }

    /**
     * Utility; group by Diagnosis coding and SourceDetails (including org/team/person and route; NOT sourceText)
     * sort by reversed end date then display name and source org name
     *
     * This should be used on past diagnoses
     */
    public List<DiagnosisGroupDto> groupByReverseEndDateAndAlphabetical(List<DataWithUIPermissions<Diagnosis>> listOfDiagnoses, boolean isPast) {
        return createGroupsUsingComparator(listOfDiagnoses, isPast, BY_REVERSE_END_DATE_THEN_ALPHABETIC_NAME);
    }

    private List<DiagnosisGroupDto> createGroupsUsingComparator(List<DataWithUIPermissions<Diagnosis>> listOfDiagnoses, boolean isPast,
            Comparator<DiagnosisGroupDto> comparator) {
        Map<Tuple5<CodeableConcept, Long, Long, Long, Route>, List<DataWithUIPermissions<Diagnosis>>> grouped = listOfDiagnoses.stream()
                .collect(Collectors.groupingBy(d -> {
                    Diagnosis diagnosis = d.getData();
                    return Tuple.of(diagnosis.getDiagnosis(), diagnosis.getSource().getOrgId(), diagnosis.getSource().getTeamId(),
                            diagnosis.getSource().getPersonId(), diagnosis.getSource().getRoute());
                }));

        return grouped.values().stream()
                .map(d -> new DiagnosisGroupDto(d, isPast))
                .sorted(comparator)
                .collect(toList());
    }

    /**
     * Utility; return diagnoses each in their own "groups", sorted by reversed end date then display name and source org name
     *
     * This should be used on current diagnoses
     */
    public List<DiagnosisGroupDto> transformDiagnosisDTOsWithoutGrouping(List<DataWithUIPermissions<Diagnosis>> listOfDiagnoses, boolean isPast) {
        return listOfDiagnoses.stream().map(diagnosis -> new DiagnosisGroupDto(singletonList(diagnosis), isPast)).sorted(BY_REVERSE_START_DATE_THEN_ALPHABETIC_NAME)
                .collect(toList());
    }

    public Diagnosis transformDiagnosis(Diagnosis diagnosis) {
        if(diagnosis != null){
            diagnosis.setSource(sourceDetailsViewFactory.createSourceDetailsView(diagnosis.getSource()));
        }
        return diagnosis;
    }

    public List<Diagnosis> transformDiagnosis(List<Diagnosis> diagnoses) {
        return diagnoses.stream().map(this::transformDiagnosis).collect(Collectors.toList());
    }

    String getTextForSorting(DiagnosisGroupDto diagnosisGroupDto) {
        return Optional.ofNullable(diagnosisGroupDto.getDiagnosisConcept())
                .map(CodeableConcept::getTextForDisplay)
                .orElse("")
                + "|" + sourceDetailsLocalizedDisplayProvider.getDisplayTextFull(diagnosisGroupDto.getSource());
    }
}
