package com.pkb.action.support;

import com.pkb.app.entity.SourceDetails;
import com.pkb.calendar.entity.AppointmentDTO;
import com.pkb.source.support.SourceDetailsViewFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.stream.Collectors;

/**
 * Transformer that is responsible for creating AppointmentView instances based on {@link AppointmentDTO}.
 *
 * Currently, this transformer is not in its final form, as no AppointmentView is existing yet.
 * By following the approach of SourceDetailsAdapter that implements SourceDetailsView, it is possible to migrate fields
 * from the lower-level domain to the higher one (view).
 * As such, AppointmentView should be implmented by an immutable and the temporal AppointmentDtoAdapter that also extends
 * {@link AppointmentDTO} so that not-yet-migrated fields can be accessed where it is still referenced,
 * by interacting with the adapter.
 * When the view class is created, transformAppointment() method can return AppointmentView type with the actual object
 * AppointmentDtoAdapter.
 *
 * Whenever all fields are migrated, AppointmentDtoAdapter can be deleted without high impact on the dependent code.
 */
@Component
public class AppointmentTransformer {
    private SourceDetailsViewFactory sourceDetailsViewFactory;

    @Autowired
    public AppointmentTransformer(SourceDetailsViewFactory sourceDetailsViewFactory) {
        this.sourceDetailsViewFactory = sourceDetailsViewFactory;
    }

    public AppointmentDTO transformAppointment(AppointmentDTO appointmentDto){
        if(appointmentDto != null){
            appointmentDto.setSource(sourceDetailsViewFactory.createSourceDetailsView(appointmentDto.getSource()));
        }
        return appointmentDto;
    }

    public List<AppointmentDTO> transformAppointment(List<AppointmentDTO> appointmens){
        return appointmens.stream().map(this::transformAppointment).collect(Collectors.toList());
    }

    public AppointmentDTO transformAppointmentView(AppointmentDTO appointment) {
        if(appointment != null){
            SourceDetails sourceDetails = new SourceDetails();
            sourceDetails.init(appointment.getSource());
            appointment.setSource(sourceDetails);
        }
        return appointment;

    }
}
