package com.pkb.action.support;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.test.entity.MeasurementDTO;
import com.pkb.test.entity.MeasurementHistoryDTO;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.time.LocalDate;
import java.time.ZoneId;
import java.util.ArrayList;
import java.util.List;
import java.util.Objects;
import java.util.TreeMap;
import java.util.stream.Stream;

import static java.time.LocalDate.ofInstant;
import static java.util.stream.Collectors.toMap;

/**
 * Please see AppointmentTransformer.
 */
@Component
public class MeasurementHistoryTransformer {
    private MeasurementTransformer measurementTransformer;

    @Autowired
    public MeasurementHistoryTransformer(MeasurementTransformer measurementTransformer) {
        this.measurementTransformer = measurementTransformer;
    }

    public MeasurementHistoryDTO transformMeasurementHistory(MeasurementHistoryDTO measurementHistoryDto, ZoneId zoneId) {
        measurementHistoryDto.setMeasurementList(measurementTransformer.transformMeasurement(measurementHistoryDto.getMeasurementList()));
        measurementHistoryDto.setChartMeasurements(getChartMeasurements(measurementHistoryDto, zoneId));

        return measurementHistoryDto;
    }

    @VisibleForTesting
    List<MeasurementDTO> getChartMeasurements(MeasurementHistoryDTO measurementHistoryDto, ZoneId zoneId) {
        if (!measurementHistoryDto.isDailyTotal()) {
            return measurementHistoryDto.getMeasurementList();
        }

        return new ArrayList<>(measurementHistoryDto.getMeasurementList().stream()
                .collect(toMap(dto -> toDate(dto, zoneId),
                        dto -> atStartOfDay(dto, zoneId),
                        (a, b) -> aggregateDailyValue(a, b, zoneId),
                        () -> new TreeMap<>(LocalDate::compareTo)))
                .values());
    }

    private MeasurementDTO atStartOfDay(MeasurementDTO dto, ZoneId zoneId) {
        var measurement = new MeasurementDTO(dto.getSource());
        measurement.setMeasureDate(toDate(dto, zoneId).atStartOfDay(zoneId).toInstant());
        measurement.setValue(dto.getValue());
        measurement.setValue2(dto.getValue2());
        return measurement;
    }

    private MeasurementDTO aggregateDailyValue(MeasurementDTO a, MeasurementDTO b, ZoneId zoneId) {
        var dailyTotal = new MeasurementDTO(a.getSource());
        dailyTotal.setMeasureDate(toDate(a, zoneId).atStartOfDay(zoneId).toInstant());
        dailyTotal.setValue(sum(a.getValue(), b.getValue()));
        dailyTotal.setValue2(sum(a.getValue2(), b.getValue2()));
        return dailyTotal;
    }

    private Double sum(Double a, Double b) {
        if (a == null && b == null) {
            return null;
        }

        return Stream.of(a, b)
                .filter(Objects::nonNull)
                .reduce(0d, Double::sum);
    }

    private LocalDate toDate(MeasurementDTO dto, ZoneId zoneId) {
        return ofInstant(dto.getMeasureDate(), zoneId);
    }
}
