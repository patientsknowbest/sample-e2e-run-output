package com.pkb.action.support;

import com.pkb.diary.entity.DiaryEntry;
import com.pkb.source.support.SourceDetailsViewFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.stream.Collectors;

/**
 * Please see AppointmentTransformer.
 */
@Component
public class DiaryEntryTransformer {
    private SourceDetailsViewFactory sourceDetailsViewFactory;

    @Autowired
    public DiaryEntryTransformer(SourceDetailsViewFactory sourceDetailsViewFactory) {
        this.sourceDetailsViewFactory = sourceDetailsViewFactory;
    }

    public DiaryEntry transformDiaryEntry(DiaryEntry diaryEntry){
        if(diaryEntry != null){
            diaryEntry.setSourceDetails(sourceDetailsViewFactory.createSourceDetailsView(diaryEntry.getSource()));
        }
        return diaryEntry;
    }


    public List<DiaryEntry> transformDiaryEntry(List<DiaryEntry> diaryEntries){
        return diaryEntries.stream().map(this::transformDiaryEntry).collect(Collectors.toList());
    }
}
