package com.pkb.action.support;

import com.pkb.allergy.entity.Allergy;
import com.pkb.coding.entity.CodeableConcept;
import com.pkb.dto.AllergyGroupDto;
import com.pkb.entities.enums.Route;
import com.pkb.service.datapoint.DataWithUIPermissions;
import com.pkb.source.support.SourceDetailsLocalizedDisplayProvider;
import com.pkb.source.support.SourceDetailsViewFactory;
import io.vavr.Tuple;
import io.vavr.Tuple6;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.time.Instant;
import java.time.temporal.ChronoUnit;
import java.util.Comparator;
import java.util.Date;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;


/**
 * Please see AppointmentTransformer.
 */
@Component
public class AllergyGroupTransformer {
    private final Comparator<AllergyGroupDto> BY_REVERSE_LAST_ACTIVE_DATE_THEN_ALPHABETIC_NAME = this::compareReverseLastActiveDateThenAlphabeticName;

    private SourceDetailsLocalizedDisplayProvider sourceDetailsLocalizedDisplayProvider;
    private SourceDetailsViewFactory sourceDetailsViewFactory;

    @Autowired
    public AllergyGroupTransformer(SourceDetailsLocalizedDisplayProvider sourceDetailsLocalizedDisplayProvider, SourceDetailsViewFactory sourceDetailsViewFactory) {
        this.sourceDetailsLocalizedDisplayProvider = sourceDetailsLocalizedDisplayProvider;
        this.sourceDetailsViewFactory = sourceDetailsViewFactory;
    }

    private int compareReverseLastActiveDateThenAlphabeticName(AllergyGroupDto a, AllergyGroupDto b) {
        int instantComparison = getInstantDateWithoutTime(b.getLastActivityDate())
                .compareTo(getInstantDateWithoutTime(a.getLastActivityDate()));

        if (instantComparison == 0) {
            return getTextForSorting(a).compareToIgnoreCase(getTextForSorting(b));
        }

        return instantComparison;
    }

    private static Instant getInstantDateWithoutTime(Date date) {
        if (date != null) {
            Instant instant = Instant.ofEpochMilli(date.getTime());
            return instant.truncatedTo(ChronoUnit.DAYS);
        }
        return Instant.MIN;
    }

    public List<AllergyGroupDto> groupByReverseLastActiveDateAndAlphabetical(List<DataWithUIPermissions<Allergy>> listOfAllergies) {
        Map<Tuple6<CodeableConcept, CodeableConcept, Long, Long, Long, Route>, List<DataWithUIPermissions<Allergy>>> grouped = groupAllergies(listOfAllergies);

        return grouped.values().stream()
                .map(AllergyGroupDto::new)
                .sorted(BY_REVERSE_LAST_ACTIVE_DATE_THEN_ALPHABETIC_NAME)
                .collect(Collectors.toList());
    }

    private static Map<Tuple6<CodeableConcept, CodeableConcept, Long, Long, Long, Route>, List<DataWithUIPermissions<Allergy>>> groupAllergies(
            List<DataWithUIPermissions<Allergy>> listOfAllergies) {
        return listOfAllergies.stream()
                .collect(Collectors.groupingBy(a -> {
                    Allergy allergy = a.getData();
                    return Tuple.of(allergy.getDisorder(), allergy.getAllergen(), allergy.getSource().getOrgId(),
                            allergy.getSource().getTeamId(), allergy.getSource().getPersonId(), allergy.getSource().getRoute());
                }));
    }

    /**
     * must cover all of the fields we group on; otherwise we'll get flaky tests
     */
    private String getTextForSorting(AllergyGroupDto allergyGroupDto) {
        if (allergyGroupDto.hasDisorderData()) {
            return String.join("|",
                    allergyGroupDto.getDisorder().getTextForDisplay(),
                    allergyGroupDto.getDisorder().getSimpleCodesForDisplay(),
                    sourceDetailsLocalizedDisplayProvider.getDisplayTextFull(allergyGroupDto.getSource()));
        }
        // ...if we can safely assume all allergies have either disorder or allergen, this can skip the null dodge
        return Optional.ofNullable(allergyGroupDto.getAllergen())
                .map(cc -> cc.getTextForDisplay() + "|" + cc.getSimpleCodesForDisplay())
                .orElse("")
                + "|" + sourceDetailsLocalizedDisplayProvider.getDisplayTextFull(allergyGroupDto.getSource());
    }

    public Allergy transformAllergy(Allergy allergy) {
        if (allergy != null) {
            allergy.setSource(sourceDetailsViewFactory.createSourceDetailsView(allergy.getSource()));
        }
        return allergy;
    }

    public List<Allergy> transformAllergy(List<Allergy> allergies) {
        return allergies.stream().map(this::transformAllergy).collect(Collectors.toList());
    }

}
