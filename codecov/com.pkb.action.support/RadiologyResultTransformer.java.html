package com.pkb.action.support;

import com.pkb.app.entity.SourceDetails;
import com.pkb.radiology.entity.RadiologyResult;
import com.pkb.source.support.SourceDetailsViewFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.stream.Collectors;

/**
 * Please see AppointmentTransformer.
 */
@Component
public class RadiologyResultTransformer {
    private SourceDetailsViewFactory sourceDetailsViewFactory;

    @Autowired
    public RadiologyResultTransformer(SourceDetailsViewFactory sourceDetailsViewFactory) {
        this.sourceDetailsViewFactory = sourceDetailsViewFactory;
    }

    /**
     * Place to create RadiologyResultView as soon as it is created.
     * @param radiologyResult dto to be transformed to radiology result view
     * @return radiology result view
     */
    public RadiologyResult transformRadiologyResult(RadiologyResult radiologyResult){
        if(radiologyResult != null){
            radiologyResult.setSource(sourceDetailsViewFactory.createSourceDetailsView(radiologyResult.getSource()));
        }
        return radiologyResult;
    }

    /**
     * Place to create RadiologyResult from view as soon as it is created.
     * @param radiologyResult view to be transformed to radiology result dto
     * @return radiology result dto
     */
    public RadiologyResult transformRadiologyResultView(RadiologyResult radiologyResult){
        if(radiologyResult != null){
            SourceDetails sourceDetails = new SourceDetails();
            sourceDetails.init(radiologyResult.getSource());
            radiologyResult.setSource(sourceDetails);
        }
        return radiologyResult;
    }

    public List<RadiologyResult> transformRadiologyResult(List<RadiologyResult> radiologyResult){
        return radiologyResult.stream().map(this::transformRadiologyResult).collect(Collectors.toList());
    }

}
