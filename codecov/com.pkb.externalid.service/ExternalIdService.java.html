package com.pkb.externalid.service;

import com.google.common.collect.HashBasedTable;
import com.google.common.collect.ImmutableTable;
import com.google.common.collect.Table;
import com.pkb.externalid.entity.ExternalId;
import com.pkb.externalid.entity.ExternalIdScheme;
import com.pkb.externalid.entity.ExternalIdScope;
import com.pkb.externalid.repository.ExternalIdRepository;
import com.pkb.externalid.repository.ExternalIdSchemeRepository;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import org.springframework.transaction.annotation.Transactional;

import java.util.Collection;
import java.util.Collections;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.Set;
import java.util.UUID;
import java.util.stream.Collector;
import java.util.stream.Collectors;
import java.util.stream.Stream;

public class ExternalIdService {

    private static final Collector<ExternalId, ?, Map<String, UUID>> EXTERNAL_ID_MAP_COLLECTOR = Collectors.toMap(ExternalId::getExternalIdValue, ExternalId::getInternalIdValue);
    private final ExternalIdRepository externalIdRepository;
    private final ExternalIdSchemeRepository externalIdSchemeRepository;

    public ExternalIdService(ExternalIdRepository externalIdRepository, ExternalIdSchemeRepository externalIdSchemeRepository) {
        this.externalIdRepository = externalIdRepository;
        this.externalIdSchemeRepository = externalIdSchemeRepository;
    }

    @Transactional
    public UUID lookupOrCreateUniqueIdForExternalId(String externalId, ExternalIdScope scope) {
        return lookupOrCreateUniqueIdsForExternalIds(Set.of(externalId), scope).get(externalId);
    }

    @Transactional
    public Map<String, UUID> lookupOrCreateUniqueIdsForExternalIds(Set<String> externalIds, ExternalIdScope scope) {
        if (externalIds.isEmpty()) {
            return Collections.emptyMap();
        }

        var scheme = getOrCreateScheme(scope.getSchemeName());
        var existingEntities = externalIdRepository.getExternalIds(scope.getOrgId(), scheme.getId(), externalIds);
        var existingExternalIds = existingEntities.stream().map(ExternalId::getExternalIdValue).collect(Collectors.toSet());
        var newIdsToCreate = externalIds.stream().filter(it -> !existingExternalIds.contains(it))
                .map(it -> createNewExternalId(scope.getOrgId(), scheme, it)).collect(Collectors.toList());
        if (!newIdsToCreate.isEmpty()) {
            externalIdRepository.saveAll(newIdsToCreate);
        }

        return Stream.of(existingEntities, newIdsToCreate).flatMap(Collection::stream).collect(EXTERNAL_ID_MAP_COLLECTOR);
    }

    /**
     * Get or create internal UUID for every external id provided.
     *
     * @param schemeName
     * @param externalIds Tuple of Org public id (UUID) & external id.
     * @return {@link Table} of external id, Org public id (UUID), internal id (UUID)
     */
    @Transactional
    public Table<String, UUID, UUID> lookupOrCreateUniqueIdsForExternalIdsAcrossOrgs(String schemeName, io.vavr.collection.Set<Tuple2<UUID, String>> externalIds) {
        if (externalIds.isEmpty()) {
            return ImmutableTable.<String, UUID, UUID>builder().build();
        }

        Table<String, UUID, UUID> result = HashBasedTable.create();
        var scheme = getOrCreateScheme(schemeName);
        externalIdRepository.getExternalIdsAcrossOrgs(scheme.getId(), externalIds.map(Tuple2::_2)).forEach(eid -> result.put(eid.getExternalIdValue(), eid.getOrgId(), eid.getInternalIdValue()));

        var externalIdsToPersist = externalIds.filter(id -> !result.contains(id._2, id._1)).map(toCreate -> createNewExternalId(toCreate._1, scheme, toCreate._2));
        externalIdRepository.saveAll(externalIdsToPersist).forEach(eid -> result.put(eid.getExternalIdValue(), eid.getOrgId(), eid.getInternalIdValue()));

        return result;
    }


    private ExternalId createNewExternalId(UUID orgId, ExternalIdScheme scheme, String externalId) {
        return new ExternalId(UUID.randomUUID(), scheme, orgId, externalId, UUID.randomUUID());
    }

    public Map<String, UUID> lookupUniqueIdsForExternalIds(List<String> externalIds, ExternalIdScope scope) {
        if (externalIds.isEmpty()) {
            return Collections.emptyMap();
        }
        return getScheme(scope.getSchemeName()).map(scheme ->
                externalIdRepository.getExternalIds(scope.getOrgId(), scheme.getId(), externalIds).stream().collect(EXTERNAL_ID_MAP_COLLECTOR))
                .orElseGet(Collections::emptyMap);
    }

    /**
     * Like it's lookupOrCreate twin above, this method is designed for schemes that have ids that are globally unique across orgs to save doing
     * multiple queries, one per org.
     */
    @Transactional(readOnly = true) //Required for streaming support
    public Table<String, UUID, UUID> lookupUniqueIdsForExternalIdsAcrossOrgs(String schemeName, io.vavr.collection.Set<String> externalIds) {
        if (externalIds.isEmpty()) {
            return ImmutableTable.<String, UUID, UUID>builder().build();
        }

        Table<String, UUID, UUID> result = HashBasedTable.create();
        var scheme = getOrCreateScheme(schemeName);
        externalIdRepository.getExternalIdsAcrossOrgs(scheme.getId(), externalIds).forEach(eid -> result.put(eid.getExternalIdValue(), eid.getOrgId(), eid.getInternalIdValue()));
        return result;
    }

    private Optional<ExternalIdScheme> getScheme(String schemeName) {
        return externalIdSchemeRepository.findSchemeByName(schemeName);
    }

    private ExternalIdScheme getOrCreateScheme(String schemeName) {
        return getScheme(schemeName).orElseGet(() -> createNewScheme(schemeName));
    }

    private ExternalIdScheme createNewScheme(String schemeName) {
        return externalIdSchemeRepository.save(new ExternalIdScheme(schemeName));
    }

    private io.vavr.collection.Map<String, UUID> mapExternalIdStream(Stream<ExternalId> externalIdStream) {
        return externalIdStream.map(eid -> Tuple.of(eid.getExternalIdValue(), eid.getInternalIdValue()))
                .collect(io.vavr.collection.HashMap.collector());
    }

}
