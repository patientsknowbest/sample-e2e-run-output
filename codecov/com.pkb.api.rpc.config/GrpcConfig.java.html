package com.pkb.api.rpc.config;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.pkb.api.rpc.GrpcServer;
import com.pkb.common.config.PhrConfig;
import com.pkb.service.emailmessage.IAddressInfoProvider;
import com.pkb.service.emailmessage.impl.EmailManager;
import com.pkb.service.emailmessage.impl.TemplateParamsFactory;
import com.pkb.service.emailmessage.sender.impl.VelocityTemplateHandler;
import com.pkb.service.encounter.impl.tolven.EncounterManager;
import com.pkb.service.user.impl.UserManager;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Profile;

import java.io.IOException;

@Configuration
public class GrpcConfig {

    @Bean
    @Profile("!unittest") // Otherwise breaks everything that extends BaseStrutsSpringTest
    public GrpcServer grpcServer(PhrConfig phrConfig, EncounterManager encounterManager, EmailManager emailManager, UserManager userManager,
                                 VelocityTemplateHandler velocityTemplateHandler, IAddressInfoProvider formatter,
                                 TemplateParamsFactory paramsFactory, ObjectMapper objectMapper) throws IOException, InterruptedException {
        return new GrpcServer(phrConfig, encounterManager, emailManager, userManager, velocityTemplateHandler, formatter, paramsFactory, objectMapper);
    }
}
