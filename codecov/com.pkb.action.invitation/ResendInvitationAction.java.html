package com.pkb.action.invitation;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.exception.PKBException;
import com.pkb.service.invitation.impl.InvitationManager;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.UUID;

/**
 * Resends clinician invitation email
 *
 * @author mahendera
 */

public class ResendInvitationAction extends BaseAction {

    private static final long serialVersionUID = -1L;
    private InvitationManager invitationManager;

    private String invitationId;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Override
    public String execute() {
        String result = Action.SUCCESS;
        try {
            UUID invitationPublicId = invitationManager.getInvitationIdByString(invitationId);
            invitationManager.resendInvitation(getLoggedInEHRRequestContext(), invitationPublicId);
        } catch (PKBException exception) {
            result = Action.ERROR;
            addActionError(getText("resendInvitationAction.err.error_resending_invitation"));
            LOGGER.error("Error while resending invitation {}, {}", invitationId, exception);
        }
        return result;
    }

    public String getInvitationId() {
        return invitationId;
    }

    public void setInvitationId(String invitationId) {
        this.invitationId = invitationId;
    }

    public void setInvitationManager(InvitationManager invitationManager) {
        this.invitationManager = invitationManager;
    }

}
