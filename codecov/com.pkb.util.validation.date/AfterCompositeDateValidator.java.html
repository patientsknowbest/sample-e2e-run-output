package com.pkb.util.validation.date;

import org.jetbrains.annotations.NotNull;

import java.time.LocalDate;

import static com.pkb.util.validation.date.InclusionMode.INCLUSIVE;

public class AfterCompositeDateValidator extends AbstractCompositeDateValidator<AfterCompositeDate> {
    private LocalDate referenceDate;

    public AfterCompositeDateValidator() {
        super(LocalDate::isAfter);
    }

    @Override
    public void initialize(AfterCompositeDate annotation) {
        LocalDate referenceDate = LocalDate.of(annotation.year(), annotation.month(), annotation.day());
        this.referenceDate = (annotation.mode() == INCLUSIVE) ? referenceDate.minusDays(1) : referenceDate;
    }

    @Override
    protected @NotNull LocalDate getReferenceDate() {
        return referenceDate;
    }
}
