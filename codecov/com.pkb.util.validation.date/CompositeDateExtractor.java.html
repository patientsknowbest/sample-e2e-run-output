package com.pkb.util.validation.date;

import com.google.common.collect.ImmutableMap;
import com.pkb.util.validation.date.CompositeDate.DayPart;
import com.pkb.util.validation.date.CompositeDate.MonthPart;
import com.pkb.util.validation.date.CompositeDate.YearPart;
import io.vavr.Tuple;
import io.vavr.collection.List;
import io.vavr.collection.Seq;
import io.vavr.collection.Traversable;
import io.vavr.control.Try;
import io.vavr.control.Validation;
import org.apache.commons.beanutils.PropertyUtils;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;

import java.lang.annotation.Annotation;
import java.lang.reflect.Field;
import java.math.BigDecimal;
import java.time.LocalDate;
import java.util.Map;
import java.util.function.ToIntFunction;

import static io.vavr.control.Validation.combine;
import static io.vavr.control.Validation.invalid;
import static io.vavr.control.Validation.valid;
import static java.lang.String.format;
import static java.lang.invoke.MethodHandles.lookup;
import static org.apache.commons.lang3.reflect.FieldUtils.getFieldsWithAnnotation;
import static org.slf4j.LoggerFactory.getLogger;

public class CompositeDateExtractor {
    private static final Logger LOGGER = getLogger(lookup().lookupClass());

    private static final Map<Class<?>, ToIntFunction<Object>> MAPPINGS = defineMappings();

    private static ImmutableMap<Class<?>, ToIntFunction<Object>> defineMappings() {
        return ImmutableMap.<Class<?>, ToIntFunction<Object>>builder()
                .put(String.class, value -> Integer.parseInt(String.class.cast(value)))
                .put(short.class, short.class::cast)
                .put(int.class, int.class::cast)
                .put(long.class, value -> long.class.cast(value).intValue())
                .put(Short.class, Short.class::cast)
                .put(Number.class, value -> Number.class.cast(value).intValue())
                .put(Integer.class, Integer.class::cast)
                .put(Long.class, value -> Long.class.cast(value).intValue())
                .put(BigDecimal.class, value -> BigDecimal.class.cast(value).intValue())
                .build();
    }

    public @NotNull Validation<Seq<String>, LocalDate> extract(@NotNull Object model) {
        try {
            Class<?> modelType = model.getClass();

            Validation<Seq<String>, Integer> validYear = getValidValue(model, modelType, YearPart.class);
            Validation<Seq<String>, Integer> validMonth = getValidValue(model, modelType, MonthPart.class);
            Validation<Seq<String>, Integer> validDay = getValidValue(model, modelType, DayPart.class);

            return combine(validYear, validMonth, validDay)
                    .ap(Tuple::of)
                    // flatten errors
                    .mapError(errors -> errors.flatMap(Traversable::iterator))
                    .flatMap(parts -> toValidDate(parts._1(), parts._2(), parts._3()));
        } catch (Exception cause) {
            LOGGER.error("Failed to extract a composite date from model (of {})", model.getClass(), cause);
            return invalid(List.of(cause.getMessage()));
        }
    }

    private Validation<Seq<String>, LocalDate> toValidDate(int year, int month, int day) {
        return Try.of(() -> LocalDate.of(year, month, day))
                .toValid(() -> List.of(format("year=[%s], month=[%s], day=[%s] does not represent a valid date", year, month, day)));
    }

    private Validation<Seq<String>, Integer> getValidValue(Object model, Class<?> modelType, Class<? extends Annotation> annotation) {
        return findExactlyOneValidFieldWithAnnotation(modelType, annotation)
                .flatMap(field -> MAPPINGS.containsKey(field.getType()) ?
                        accessValidValue(model, field) :
                        invalid(List.of(format("There is no mapping defined from %s to int", field.getType()))));
    }

    private Validation<Seq<String>, Integer> accessValidValue(Object model, Field field) {
        try {
            Object value = PropertyUtils.getProperty(model, field.getName());

            return value != null ? valid(MAPPINGS.get(value.getClass()).applyAsInt(value)) : invalid(List.of("Value cannot be null"));
        } catch (Exception cause) {
            return invalid(List.of(format("Failed to access value of field=[%s], because: %s", field.getName(), cause.getMessage())));
        }
    }

    private Validation<Seq<String>, Field> findExactlyOneValidFieldWithAnnotation(Class<?> modelType, Class<? extends Annotation> annotation) {
        Field[] fields = getFieldsWithAnnotation(modelType, annotation);

        if (fields.length == 1) {
            return valid(fields[0]);
        } else if (fields.length > 1) {
            return invalid(List.of(format("There is more than one field annotated with %s", annotation.getSimpleName())));
        } else {
            return invalid(List.of(format("There is no field annotated with %s", annotation.getSimpleName())));
        }
    }
}
