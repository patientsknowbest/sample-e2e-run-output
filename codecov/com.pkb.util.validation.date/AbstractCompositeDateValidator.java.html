package com.pkb.util.validation.date;

import org.jetbrains.annotations.NotNull;

import javax.validation.ConstraintValidator;
import javax.validation.ConstraintValidatorContext;
import java.lang.annotation.Annotation;
import java.time.LocalDate;

public abstract class AbstractCompositeDateValidator<T extends Annotation> implements ConstraintValidator<T, Object> {
    @FunctionalInterface
    public interface LocalDatePredicate {
        boolean apply(LocalDate date, LocalDate referenceDate);
    }

    private final CompositeDateExtractor dateExtractor = new CompositeDateExtractor();
    private final LocalDatePredicate predicate;

    AbstractCompositeDateValidator(@NotNull LocalDatePredicate predicate) {
        this.predicate = predicate;
    }

    @Override
    public void initialize(T constraintAnnotation) {
    }

    @Override
    public boolean isValid(Object value, ConstraintValidatorContext context) {
        return dateExtractor.extract(value)
                .map(date -> predicate.apply(date, getReferenceDate()))
                .getOrElse(false);
    }

    protected abstract @NotNull LocalDate getReferenceDate();
}
