package com.pkb.service.dataupload.processor.persister;

import com.pkb.app.dto.DataPointSaveResult;
import com.pkb.app.entity.SourceDetails;
import com.pkb.dataupload.entity.UploadedData;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.medication.entity.Medication;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.UploadedDataPersistentState;
import com.pkb.service.medication.EHRMedicationManager;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;

@Component
public class MedicationPersister extends AbstractUploadedDataPersister<Medication, LoggedInUploadedDataProcessingContext> {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private EHRMedicationManager ehrMedicationManager;

    @Autowired
    public MedicationPersister(EHRMedicationManager ehrMedicationManager) {
        this.ehrMedicationManager = ehrMedicationManager;
    }

    @Override
    UploadedDataPersistentState doPersist(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, List<Medication> data) {
        long oid = (dto.getSourceOrg() == null) ? dto.getSourceTeam().getOrg().getId() : dto.getSourceOrg().getId();
        SourceDetails sourceDetails = new SourceDetails(dto);
        DataPointSaveResult<Medication> saveResult = ehrMedicationManager.updateDataPointsForOrg(context.requestContext(), data,
                oid, true /*deleteAbsent*/, sourceDetails);
        saveResult.getExcludedDataPoints();
        LOGGER.info("Updated {} HL7 medications for {}; {}", data.size(), context.patientId(), dto.getId());
        return new UploadedDataPersistentState(UploadedData.Status.COMPLETED, UploadedDataPersistentState.UploadedDataPersistentStateEnum.CREATED_OR_UPDATED);
    }
}
