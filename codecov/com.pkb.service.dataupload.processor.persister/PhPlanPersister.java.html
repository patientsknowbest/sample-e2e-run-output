package com.pkb.service.dataupload.processor.persister;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.datamodel.plantemplate.TemplateAttachment;
import com.pkb.dataupload.entity.UploadedData;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.phplan.entity.PHPlan;
import com.pkb.service.dataupload.processor.UploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.UploadedDataPersistentState;
import com.pkb.service.dataupload.processor.domain.UploadedDataPersistentState.UploadedDataPersistentStateEnum;
import com.pkb.service.phplan.PhPlanSaveResult;
import com.pkb.service.phplan.PhPlanSaver;
import com.pkb.service.phplan.PlanTemplateManager;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.Optional;

@Component
public class PhPlanPersister extends AbstractUploadedDataPersister<PHPlan, UploadedDataProcessingContext> {
    private PlanTemplateManager planTemplateManager;
    private PhPlanSaver phPlanSaver;

    @Autowired
    public PhPlanPersister(PlanTemplateManager planTemplateManager, PhPlanSaver phPlanSaver) {
        this.planTemplateManager = planTemplateManager;
        this.phPlanSaver = phPlanSaver;
    }

    @Override
    UploadedDataPersistentState doPersist(UploadedDataProcessingContext context, UploadedDataDTO dto, List<PHPlan> data) {
        if (data.size() > 1) {
            throw new IllegalStateException("Only one care plan can be persisted at once currently");
        }
        UploadedDataPersistentState result;
        Optional<PHPlan> uploadedCarePlan = data.stream().findFirst();

        if (uploadedCarePlan.isPresent()) {
            SourceDetails sourceDetails = new SourceDetails(dto);
            result = doPersist(sourceDetails, context, uploadedCarePlan.get());
        } else {
            result = new UploadedDataPersistentState(UploadedData.Status.COMPLETED, UploadedDataPersistentStateEnum.NO_OP);
        }
        return result;
    }

    UploadedDataPersistentState doPersist(SourceDetails sourceDetails, UploadedDataProcessingContext context, PHPlan phPlan) {
        List<TemplateAttachment> planTemplateAttachments = planTemplateManager.getTemplateAttachments(phPlan.getTemplateId());

        PhPlanSaveResult phPlanSaveResult = saveUserPlan(sourceDetails, context.requestContext(),context.patientId(), phPlan, planTemplateAttachments);
        return new UploadedDataPersistentState(UploadedData.Status.COMPLETED,
                phPlanSaveResult.isUpdated() ? UploadedDataPersistentStateEnum.UPDATED : UploadedDataPersistentStateEnum.CREATED);
    }

    private PhPlanSaveResult saveUserPlan(SourceDetails sourceDetails, @NotNull EHRRequestContext requestContext, long userId, PHPlan pHPlan,
            List<TemplateAttachment> planTemplateAttachments) {

        return phPlanSaver.insertOrOverwriteUserPlan(sourceDetails, requestContext, userId, pHPlan, planTemplateAttachments);
    }

}
