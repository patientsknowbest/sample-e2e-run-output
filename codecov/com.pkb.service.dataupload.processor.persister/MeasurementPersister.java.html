package com.pkb.service.dataupload.processor.persister;

import com.pkb.app.entity.SourceDetails;
import com.pkb.common.testlogging.DetailLoggingProvider;
import com.pkb.dataupload.entity.UploadedData;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.ParsedObservationInformation;
import com.pkb.service.dataupload.processor.domain.UploadedDataPersistentState;
import com.pkb.service.measurement.EHRMeasurementManager;
import com.pkb.test.UploadedDataDeletionMetadata;
import com.pkb.test.entity.MeasurementDTO;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.ArrayList;
import java.util.List;
import java.util.Set;

import static com.pkb.service.dataupload.processor.domain.UploadedDataPersistentState.UploadedDataPersistentStateEnum.CREATED_OR_UPDATED;
import static java.util.stream.Collectors.toList;
import static java.util.stream.Collectors.toSet;

@Component
public class MeasurementPersister extends AbstractUploadedDataPersister<ParsedObservationInformation<MeasurementDTO>, LoggedInUploadedDataProcessingContext> {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private DetailLoggingProvider testLoggingService;
    private EHRMeasurementManager ehrMeasurementManager;

    @Autowired
    public MeasurementPersister(DetailLoggingProvider testLoggingService, EHRMeasurementManager ehrMeasurementManager) {
        this.testLoggingService = testLoggingService;
        this.ehrMeasurementManager = ehrMeasurementManager;
    }

    @Override
    public UploadedDataPersistentState doPersist(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, List<ParsedObservationInformation<MeasurementDTO>> parsedMeasurementInformationList) {
        if (parsedMeasurementInformationList.size() > 1) {
            throw new IllegalStateException("Multiple measurements must be bundled into a single ParsedObservationInformation instance");
        }
        List<Boolean> allMappedOutcomeList = new ArrayList<>();
        parsedMeasurementInformationList.forEach(parsedMeasurementInformation -> {
            allMappedOutcomeList.add(doPersist(context, dto, parsedMeasurementInformation));
        });
        boolean allMapped = allMappedOutcomeList.stream().allMatch(x -> Boolean.TRUE.equals(x));
        return new UploadedDataPersistentState(allMapped ? UploadedData.Status.COMPLETED : UploadedData.Status.COMPLETED_UNMAPPED, CREATED_OR_UPDATED);
    }

    private boolean doPersist(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, ParsedObservationInformation<MeasurementDTO> data) {
        List<MeasurementDTO> toPersist = data.getObservations().stream()
                .filter(m -> m.getMeasurementType() != null)
                .collect(toList());
        boolean allMapped = toPersist.size() == data.getObservations().size();

        deletePreviousVersions(context, dto, toPersist);

        if (data.getMaybeDeletionInfo() != null) {
            ehrMeasurementManager.deleteMeasurementsByFillerOrderNumber(context.requestContext(), data.getMaybeDeletionInfo(), dto.getSourceOrg().getId(), context.patientId());
        }
        ehrMeasurementManager.saveMeasurements(context.requestContext(), toPersist, context.patientId());
        testLoggingService.obtainLogger(LOGGER).info("just saved measurements {}", toPersist);
        return allMapped;
    }

    // This filtering should be reviewed, see PHR-5486. We are not consistent across the different API/routes about how
    // a data point is versioned or when it is deleted. E.g. Validic and SciStore seem to use UploadedDataDTO.updateTo
    // as part of their versioning strategy, and SciStore additionally makes use of UplodedData.Status.OBSOLETE.
    // Ideally, we would have a consistent way of versioning data across the system.
    private void deletePreviousVersions(LoggedInUploadedDataProcessingContext context, UploadedDataDTO uploadedDataDTO, List<MeasurementDTO> measurements) {
        if (uploadedDataDTO.getUpdateTo() != null) {
            var measurementTypeIds = measurements.stream()
                    .map(measurementDTO -> measurementDTO.getMeasurementType().id())
                    .collect(toSet());
            Set<Long> existingMeasurementIds = ehrMeasurementManager.getMeasurementIds(context.requestContext(), context.patientAccountId(), uploadedDataDTO.getUpdateTo(), measurementTypeIds);
            // See PHR-2790 for a discussion on whether enteredDate should be overwritten.
            // For now, pass in null to maintain current behaviour of defaulting to "now" when persisting.
            // Note: there is no FON in this workflow.
            UploadedDataDeletionMetadata deletionMetadata = new UploadedDataDeletionMetadata("", new SourceDetails(context.requestContext()), null, uploadedDataDTO.getId());
            ehrMeasurementManager.deleteMeasurementsById(context.requestContext(), existingMeasurementIds, deletionMetadata);
        }
    }
}
