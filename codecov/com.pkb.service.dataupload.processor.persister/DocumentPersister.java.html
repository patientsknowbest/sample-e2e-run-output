package com.pkb.service.dataupload.processor.persister;

import com.pkb.app.entity.SourceDetails;
import com.pkb.dataupload.entity.UploadedData;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.document.entity.Attachment;
import com.pkb.document.entity.ChunkedDocDTO;
import com.pkb.encounter.entity.EncounterEvent;
import com.pkb.service.dataupload.processor.UploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.UploadedDataPersistentState;
import com.pkb.service.dataupload.processor.domain.UploadedDocument;
import com.pkb.service.encounter.impl.tolven.EncounterManager;
import com.pkb.service.file.ChunkedDocManager;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.Optional;
import java.util.UUID;

@Component
public class DocumentPersister extends AbstractUploadedDataPersister<UploadedDocument, UploadedDataProcessingContext> {
    private final ChunkedDocManager chunkedDocManager;
    private final EncounterManager encounterManager;

    @Autowired
    public DocumentPersister(ChunkedDocManager chunkedDocManager, @Lazy EncounterManager encounterManager) {
        this.chunkedDocManager = chunkedDocManager;
        this.encounterManager = encounterManager;
    }

    @Override
    public UploadedDataPersistentState doPersist(UploadedDataProcessingContext context, UploadedDataDTO dto, List<UploadedDocument> data) {
        if (data.size() > 1) {
            throw new IllegalStateException("Only one document can be persisted at once currently");
        }
        UploadedDataPersistentState result;
        Optional<UploadedDocument> uploadedDocument = data.stream().findFirst();

        if (uploadedDocument.isPresent()) {
            result = doPersist(context, dto, uploadedDocument.get());
        } else {
            result = new UploadedDataPersistentState(UploadedData.Status.COMPLETED, UploadedDataPersistentState.UploadedDataPersistentStateEnum.NO_OP);
        }
        return result;
    }

    private UploadedDataPersistentState doPersist(UploadedDataProcessingContext context, UploadedDataDTO dto, UploadedDocument uploadedDocument) {
        UploadedDataPersistentState result;
        switch (uploadedDocument.getRequestType()) {
            case SAVE:
                saveDocument(context, uploadedDocument);
                result = new UploadedDataPersistentState(UploadedData.Status.COMPLETED, UploadedDataPersistentState.UploadedDataPersistentStateEnum.CREATED_OR_UPDATED);
                break;
            case DELETE:
                deleteDocument(context, dto);
                result = new UploadedDataPersistentState(UploadedData.Status.COMPLETED, UploadedDataPersistentState.UploadedDataPersistentStateEnum.DELETED);
                break;
            default:
                throw new IllegalArgumentException(String.format("Failed to save processed uploaded data. Unknown action: %s", uploadedDocument.getRequestType()));
        }
        return result;
    }

    private void deleteDocument(UploadedDataProcessingContext context, UploadedDataDTO dto) {
        String externalDocumentId = dto.getExternalMessageId();
        SourceDetails sourceDetails = new SourceDetails(dto);

        UUID encounterUniqueId = encounterManager.deleteEncountersByExternalDocumentId(context.requestContext(), context.patientAccountId(), sourceDetails, externalDocumentId,
                dto.getSourceOrg())._1;
        if (encounterUniqueId == null) {
            throw new IllegalArgumentException(String.format("Could not delete encounter by document id %s, as it does not exist.", externalDocumentId));
        }
    }

    private void saveDocument(UploadedDataProcessingContext context, UploadedDocument uploadedDocument) {
        EncounterEvent encounterEvent = uploadedDocument.getEncounterEvent();
        Attachment attachment = uploadedDocument.getAttachment();

        ChunkedDocDTO chunkDTO = new ChunkedDocDTO();
        chunkDTO.setMediaType(attachment.getMediaType());
        chunkDTO.setContent(attachment.getContent());
        chunkDTO.setName(attachment.getFilename());

        // Probably no point in actually chunking, the entire message is
        // already in memory by this point.
        UUID savedChunkId = chunkedDocManager.addChunk(context.requestContext(), chunkDTO, context.patientId(), context.patientAccountId());
        attachment.setDocMetadataId(savedChunkId);

        // Go! This is what we came for.
        Long messageId = encounterManager.saveMessage(context.requestContext(), context.patientAccountId(), encounterEvent);
        encounterManager.addAttachment(context.requestContext(), messageId, attachment);
    }
}
