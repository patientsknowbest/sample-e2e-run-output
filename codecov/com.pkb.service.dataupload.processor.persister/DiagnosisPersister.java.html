package com.pkb.service.dataupload.processor.persister;

import com.pkb.app.dto.DataPointSaveResult;
import com.pkb.app.entity.SourceDetails;
import com.pkb.dataupload.entity.UploadedData;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.diagnosis.entity.Diagnosis;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.UploadedDataPersistentState;
import com.pkb.service.diagnosis.EHRDiagnosisManager;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;

@Component
public class DiagnosisPersister extends AbstractUploadedDataPersister<Diagnosis, LoggedInUploadedDataProcessingContext> {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private EHRDiagnosisManager ehrDiagnosisManager;

    @Autowired
    public DiagnosisPersister(EHRDiagnosisManager ehrDiagnosisManager) {
        this.ehrDiagnosisManager = ehrDiagnosisManager;
    }

    @Override
    public UploadedDataPersistentState doPersist(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, List<Diagnosis> data) {
        long oid = (dto.getSourceOrg() == null) ? dto.getSourceTeam().getOrg().getId() : dto.getSourceOrg().getId();
        SourceDetails sourceDetails = new SourceDetails(dto);
        DataPointSaveResult<Diagnosis> saveResult = ehrDiagnosisManager.updateDataPointsForOrg(context.requestContext(),
                data, oid, true/*deleteAbsent*/, sourceDetails);
        saveResult.getExcludedDataPoints();
        LOGGER.info("Saved {} diagnoses for {}; {}", data.size(), context.patientId(), dto.getId());
        return new UploadedDataPersistentState(UploadedData.Status.COMPLETED, UploadedDataPersistentState.UploadedDataPersistentStateEnum.UPDATED);
    }
}
