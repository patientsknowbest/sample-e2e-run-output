package com.pkb.service.dataupload.processor.persister;

import com.pkb.common.config.PhrConfig;
import com.pkb.dataupload.entity.UploadedData;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.document.entity.Attachment;
import com.pkb.document.entity.ChunkedDocDTO;
import com.pkb.filesanitizer.FileSanitizer;
import com.pkb.radiology.entity.RadiologyResult;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.UploadedDataPersistentState;
import com.pkb.service.file.ChunkedDocManager;
import com.pkb.service.radiology.RadiologyPersistManager;
import com.pkb.service.reference.ReferenceDataManager;
import com.pkb.util.ImageUtil;
import org.apache.commons.lang3.StringUtils;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.io.IOException;
import java.util.Arrays;
import java.util.List;
import java.util.UUID;

@Component
public class RadiologyPersister extends AbstractUploadedDataPersister<RadiologyResult, LoggedInUploadedDataProcessingContext> {
    private ReferenceDataManager referenceDataManager;
    private ChunkedDocManager chunkedDocManager;
    private FileSanitizer fileSanitizer;
    private PhrConfig config;
    private RadiologyPersistManager radiologyPersistManager;

    @Autowired
    public RadiologyPersister(ReferenceDataManager referenceDataManager, ChunkedDocManager chunkedDocManager,
                              FileSanitizer fileSanitizer, PhrConfig config, RadiologyPersistManager radiologyPersistManager) {
        this.referenceDataManager = referenceDataManager;
        this.chunkedDocManager = chunkedDocManager;
        this.fileSanitizer = fileSanitizer;
        this.config = config;
        this.radiologyPersistManager = radiologyPersistManager;
    }

    @Override
    public UploadedDataPersistentState doPersist(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, List<RadiologyResult> resultList) {

        for (RadiologyResult result : resultList) {
            result.setPatientId(context.patientId());

            result.getBaseFields().setUploadedDataId(dto.getId());

            // Apply privacy labels
            referenceDataManager.applyPrivacyLabels(result, result.getObrServiceCode());
        }

        // need to upload any attachment in chunks first
        int maxChunkSize = config.getFileChunkSizeInBytes();

        for (RadiologyResult result : resultList) {
            for (Attachment att : result.getAttachments()) {
                byte[] content = att.getContent();
                int index = 0;
                int chunkCount = 0;
                UUID firstChunkId = null;

                while (index < content.length) {

                    int to = Math.min(index + maxChunkSize, content.length);
                    byte[] chunkBytes = Arrays.copyOfRange(content, index, to);
                    index = to;

                    ChunkedDocDTO chunkDTO = new ChunkedDocDTO();
                    chunkDTO.setMediaType(att.getMediaType());
                    chunkDTO.setContent(chunkBytes);
                    chunkDTO.setName(att.getFilename());
                    if (firstChunkId != null) {
                        chunkDTO.setParentMetadataId(firstChunkId);
                        chunkDTO.setName(chunkDTO.getName() + ".part" + chunkCount);
                    }

                    UUID savedChunkId = chunkedDocManager.addChunk(context.requestContext(), chunkDTO,
                            context.patientId(), context.patientAccountId());

                    if (firstChunkId == null) {
                        firstChunkId = savedChunkId;
                    }

                    chunkCount++;
                }

                // content uploaded: remove from attachment (and add ref to docId)
                att.setDocMetadataId(firstChunkId);
                // Do we need to save this, or not? We should be consistent.
                // See discharge summaries, for example.
                //att.setContent( null );
            }
        }

        // Go! This is what we came for
        sniffAndChangeMediaType(resultList);

        radiologyPersistManager.saveRadiologyResults(context.patientId(), context.patientAccountId(), resultList, context.requestContext());
        return new UploadedDataPersistentState(UploadedData.Status.COMPLETED, UploadedDataPersistentState.UploadedDataPersistentStateEnum.CREATED_OR_UPDATED);
    }

    private void sniffAndChangeMediaType(List<RadiologyResult> results) {
        // fix up attachment media type if possible (content sniffing)
        for (RadiologyResult result : results) {
            List<Attachment> attList = result.getAttachments();
            if (attList != null) {
                for (Attachment att : attList) {
                    String detectedContentType = getImageContentType(att);
                    if (StringUtils.isNotBlank(detectedContentType)) {
                        att.setMediaType(fileSanitizer.fixMediaType(att.getContent(), detectedContentType));
                    }
                }
            }
        }

    }

    private String getImageContentType(Attachment att) {
        String result;
        try {
            result = ImageUtil.getContentType(att.getContent());
        } catch (IOException exception) {
            throw new IllegalArgumentException("Failed to detect image content type", exception);
        }

        return result;
    }
}
