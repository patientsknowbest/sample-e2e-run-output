package com.pkb.service.dataupload.processor.persister;

import com.pkb.app.dto.SavedIdsDTO;
import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.dataupload.entity.UploadedData;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.exception.PKBException;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.ParsedObservationInformation;
import com.pkb.service.dataupload.processor.domain.UploadedDataPersistentState;
import com.pkb.service.test.LoincManager;
import com.pkb.test.UploadedDataDeletionMetadata;
import com.pkb.test.entity.TestResultDTO;
import com.pkb.util.tolven.TolvenBeanFactory;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.List;

import static java.lang.String.format;

@Component
public class TestResultPersister extends AbstractUploadedDataPersister<ParsedObservationInformation<TestResultDTO>, LoggedInUploadedDataProcessingContext> {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private TolvenBeanFactory beanFactory;
    private LoincManager loincManager;

    @Autowired
    public TestResultPersister(TolvenBeanFactory beanFactory, LoincManager loincManager) {
        this.beanFactory = beanFactory;
        this.loincManager = loincManager;
    }

    @Override
    UploadedDataPersistentState doPersist(LoggedInUploadedDataProcessingContext context, UploadedDataDTO uploadedTestResult, List<ParsedObservationInformation<TestResultDTO>> parsedTestResultInformationList) {
        if (parsedTestResultInformationList.size() > 1) {
            throw new IllegalStateException("Multiple test results must be bundled into a single ParsedObservationInformation instance");
        }
        parsedTestResultInformationList.forEach(parsedTestResultInformation ->
                doPersist(context, uploadedTestResult, parsedTestResultInformation)
        );
        return new UploadedDataPersistentState(UploadedData.Status.COMPLETED, UploadedDataPersistentState.UploadedDataPersistentStateEnum.CREATED_OR_UPDATED);
    }

    private void doPersist(LoggedInUploadedDataProcessingContext context, UploadedDataDTO uploadedTestResult, ParsedObservationInformation<TestResultDTO> testResultInfo) {
        deleteResultsByFillerOrderNumber(context.requestContext(), testResultInfo.getMaybeDeletionInfo(), uploadedTestResult.getSourceOrg().getId(), context.patientId());

        saveTestResults(context, testResultInfo.getObservations(), context.patientId());
        LOGGER.info("Saved {} HL7 lab results for {}; {}", testResultInfo.getObservations().size(), context.patientId(), uploadedTestResult.getId());
    }

    /**
     * Bulk deletion based on Filler Order Number. Currently for a single patient only.
     *
     * @param requestContext
     * @param maybeDeletionInfo
     * @param orgId
     * @param patientId
     */
    private void deleteResultsByFillerOrderNumber(LoggedInEHRRequestContext requestContext, UploadedDataDeletionMetadata maybeDeletionInfo, long orgId, long patientId) {
        if (maybeDeletionInfo == null) {
            return;
        }
        loincManager.deleteResultsByFillerOrderNumber(requestContext, maybeDeletionInfo, orgId, patientId);
    }

    private List<SavedIdsDTO> saveTestResults(LoggedInUploadedDataProcessingContext context, List<TestResultDTO> testResults, long patientUserId) {
        Long accountId = context.patientAccountId();
        try {
            List<SavedIdsDTO> savedTestResults = loincManager.saveTestResults(context.requestContext(), testResults, patientUserId, accountId);

            // update the uniqueId and ID values in the DTOs
            for (int i = 0; i < savedTestResults.size(); i++) {
                testResults.get(i).setId(savedTestResults.get(i).getId());
                testResults.get(i).getBaseFields().setUniqueId(savedTestResults.get(i).getUniqueId());
            }

            return savedTestResults;
        } catch (Exception e) {
            throw new PKBException(format("failed saving test results for %s", patientUserId), e);
        }
    }
}
