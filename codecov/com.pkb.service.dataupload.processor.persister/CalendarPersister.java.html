package com.pkb.service.dataupload.processor.persister;

import com.pkb.dataupload.entity.UploadedData;
import com.pkb.dataupload.entity.UploadedDataDTO;
import com.pkb.exception.CalendarException;
import com.pkb.service.calendar.impl.CalendarManager;
import com.pkb.service.dataupload.processor.LoggedInUploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.UploadedDataProcessingContext;
import com.pkb.service.dataupload.processor.domain.UploadedAppointment;
import com.pkb.service.dataupload.processor.domain.UploadedDataPersistentState;
import com.pkb.service.dataupload.processor.domain.UploadedDataPersistentState.UploadedDataPersistentStateEnum;
import com.pkb.service.integration.DatapointExternalIntegrationUrlService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Component;

import java.util.List;

@Component
public class CalendarPersister extends AbstractUploadedDataPersister<UploadedAppointment, LoggedInUploadedDataProcessingContext> {
    private CalendarManager calendarManager;
    private DatapointExternalIntegrationUrlService datapointExternalIntegrationUrlService;

    @Autowired
    public CalendarPersister(@Lazy CalendarManager calendarManager, DatapointExternalIntegrationUrlService datapointExternalIntegrationUrlService) {
        this.calendarManager = calendarManager;
        this.datapointExternalIntegrationUrlService = datapointExternalIntegrationUrlService;
    }

    @Override
    public UploadedDataPersistentState doPersist(LoggedInUploadedDataProcessingContext context, UploadedDataDTO dto, List<UploadedAppointment> data) {
        UploadedDataPersistentState result;
        if (data.isEmpty()) {
            result = new UploadedDataPersistentState(UploadedData.Status.COMPLETED, UploadedDataPersistentStateEnum.NO_OP);
        } else if (data.size() == 1) {
            result = doPersist(context, data.stream().findFirst().get());
        } else {
            data.forEach(uploadedAppointment -> doPersist(context, uploadedAppointment));
            result = new UploadedDataPersistentState(UploadedData.Status.COMPLETED, UploadedDataPersistentStateEnum.CREATED_OR_UPDATED);
        }
        return result;
    }

    private UploadedDataPersistentState doPersist(LoggedInUploadedDataProcessingContext context, UploadedAppointment uploadedAppointment) {
        UploadedDataPersistentState result;
        switch (uploadedAppointment.getRequestType()) {
        case CREATE:
            createAppointment(context, uploadedAppointment);
            result = createUploadedDataPersistentState(UploadedDataPersistentStateEnum.CREATED);
            break;
        case UPDATE:
            updateAppointment(context, uploadedAppointment);
            result = createUploadedDataPersistentState(UploadedDataPersistentStateEnum.UPDATED);
            break;
        case DELETE:
            deleteAppointment(context, uploadedAppointment);
            result = createUploadedDataPersistentState(UploadedDataPersistentStateEnum.DELETED);
            break;
        case CANCEL:
            cancelAppointment(context, uploadedAppointment);
            result = createUploadedDataPersistentState(UploadedDataPersistentStateEnum.UPDATED);
            break;
        case DNA:
            dnaAppointment(context, uploadedAppointment);
            result = createUploadedDataPersistentState(UploadedDataPersistentStateEnum.UPDATED);
            break;
        case NO_OP:
            result = createUploadedDataPersistentState(UploadedDataPersistentStateEnum.NO_OP);
            break;
        default:
            throw new IllegalArgumentException(String.format("Could not save processed uploaded data. Unknown action: %s", uploadedAppointment.getRequestType()));
        }
        return result;
    }

    private UploadedDataPersistentState createUploadedDataPersistentState(UploadedDataPersistentStateEnum persistentState) {
        return new UploadedDataPersistentState(UploadedData.Status.COMPLETED, persistentState);
    }

    private void dnaAppointment(LoggedInUploadedDataProcessingContext context, UploadedAppointment uploadedAppointment) {
        saveAppointment(context, uploadedAppointment);
    }

    private void cancelAppointment(LoggedInUploadedDataProcessingContext context, UploadedAppointment uploadedAppointment) {
        saveAppointment(context, uploadedAppointment);
    }

    private void deleteAppointment(LoggedInUploadedDataProcessingContext context, UploadedAppointment uploadedAppointment) {
        try {
            calendarManager.deleteAppointment(context.requestContext(), uploadedAppointment.getAppointment().getId(), context.patientAccountId());
        } catch (Exception e) {
            throw new CalendarException("Exception while deleting appointment " + uploadedAppointment.getAppointment().getId() + " for acct "
                    + context.patientAccountId(), e);
        }
    }

    private void updateAppointment(LoggedInUploadedDataProcessingContext context, UploadedAppointment uploadedAppointment) {
        saveAppointment(context, uploadedAppointment);
    }

    private void saveAppointment(UploadedDataProcessingContext context, UploadedAppointment uploadedAppointment) {
        calendarManager.saveAppointment(context.requestContext(), uploadedAppointment.getAppointment(), context.patientAccountId(), false);
        uploadedAppointment.getExternalIntegrationUrl().ifPresent(url -> datapointExternalIntegrationUrlService.save(
                uploadedAppointment.getAppointment().getBaseFields().getUniqueId(),
                url.getUrl().getString(),
                context.requestContext()));
    }

    private void createAppointment(LoggedInUploadedDataProcessingContext context, UploadedAppointment uploadedAppointment) {
        saveAppointment(context, uploadedAppointment);
    }
}
