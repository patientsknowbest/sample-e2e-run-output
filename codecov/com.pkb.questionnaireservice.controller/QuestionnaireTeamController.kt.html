package com.pkb.questionnaireservice.controller

import com.pkb.questionnaireservice.ErrorException
import com.pkb.questionnaireservice.QuestionnaireId
import com.pkb.questionnaireservice.QuestionnaireRequestId
import com.pkb.questionnaireservice.TeamPublicId
import com.pkb.questionnaireservice.entities.QuestionnaireTeamEntity
import com.pkb.questionnaireservice.modelv2.QuestionnaireResponseCount
import com.pkb.questionnaireservice.service.QuestionnaireService
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.PutMapping
import org.springframework.web.bind.annotation.RequestBody
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RequestParam
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("/questionnaireteam")
class QuestionnaireTeamController(private val questionnaireService: QuestionnaireService) {

    @GetMapping("/{id}")
    fun getQuestionnaireTeamByQuestionnaireIdAndTeamId(@PathVariable("id") id: QuestionnaireId,
                                                       @RequestParam("team_ids") teamIds: List<TeamPublicId>): ResponseEntity<List<QuestionnaireTeamEntity>> =
            questionnaireService.getQuestionnaireTeamByQuestionnaireIdAndTeamId(id, teamIds)
                    .map { ResponseEntity(it, HttpStatus.OK) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @GetMapping("/questionnairerequest/{id}")
    fun getQuestionnaireTeamByQuestionnaireRequestId(@PathVariable("id") id: QuestionnaireRequestId): ResponseEntity<QuestionnaireTeamEntity> =
            questionnaireService.getQuestionnaireTeamByQuestionnaireRequestId(id)
                    .map { ResponseEntity(it, HttpStatus.OK) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @GetMapping("/questionnaireresponsecount/{team_id}")
    fun getQuestionnaireResponseCountsByTeamId(@PathVariable("team_id") teamId: TeamPublicId): ResponseEntity<List<QuestionnaireResponseCount>> =
            questionnaireService.getQuestionnaireResponseCountsByTeamId(teamId)
                    .map { ResponseEntity(it, HttpStatus.OK) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @GetMapping(params = ["team_id"])
    fun getQuestionnaireTeamsInTeam(@RequestParam("team_id") id: TeamPublicId,
                                    @RequestParam("includeHidden") includeHidden: Boolean): ResponseEntity<List<QuestionnaireTeamEntity>> =
            questionnaireService.getQuestionnaireTeamsInTeams(listOf(id), includeHidden)
                    .map { ResponseEntity(it, HttpStatus.OK) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @GetMapping(params = ["team_ids"])
    fun getQuestionnaireTeamsInTeams(@RequestParam("team_ids") ids: List<TeamPublicId>,
                                     @RequestParam("includeHidden") includeHidden: Boolean): ResponseEntity<List<QuestionnaireTeamEntity>> =
            questionnaireService.getQuestionnaireTeamsInTeams(ids, includeHidden)
                    .map { ResponseEntity(it, HttpStatus.OK) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @PostMapping
    fun postQuestionnaireAndLinkToTeam(@RequestBody questAndTeamId: QuestionnaireTeamEntity): ResponseEntity<QuestionnaireTeamEntity> =
            questionnaireService.postQuestionnaireAndLinkToTeam(questAndTeamId)
                    .map { ResponseEntity(it, HttpStatus.OK) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @PutMapping("update")
    fun updateTeamLink(@RequestBody questAndTeamId: QuestionnaireTeamEntity): ResponseEntity<QuestionnaireTeamEntity> =
            questionnaireService.updateTeamLink(questAndTeamId)
                    .map { ResponseEntity(it, HttpStatus.OK) }
                    .getOrElseThrow { err -> ErrorException(err) }
}
