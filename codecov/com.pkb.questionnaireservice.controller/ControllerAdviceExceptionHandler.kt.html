package com.pkb.questionnaireservice.controller

import com.pkb.questionnaireservice.ErrorException
import com.pkb.questionnaireservice.modelv2.QuestionnaireServiceError
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.beans.TypeMismatchException
import org.springframework.http.HttpHeaders
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.http.converter.HttpMessageNotReadableException
import org.springframework.web.HttpRequestMethodNotSupportedException
import org.springframework.web.bind.MethodArgumentNotValidException
import org.springframework.web.bind.UnsatisfiedServletRequestParameterException
import org.springframework.web.bind.annotation.ControllerAdvice
import org.springframework.web.bind.annotation.ExceptionHandler
import org.springframework.web.context.request.WebRequest
import org.springframework.web.servlet.NoHandlerFoundException
import org.springframework.web.servlet.mvc.method.annotation.ResponseEntityExceptionHandler
import java.lang.invoke.MethodHandles

/**
 * This class maps any exception thrown from the controllers to a QuestionnaireServiceError
 */
@ControllerAdvice
class ControllerAdviceExceptionHandler : ResponseEntityExceptionHandler() {
    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @ExceptionHandler(Exception::class)
    fun handle(ex: Exception, request: WebRequest): ResponseEntity<Any> {
        logException(ex)
        return when (ex) {
            is ErrorException -> handle(ex.error, request)
            else -> handle(QuestionnaireServiceError.UnexpectedError(ex.message!!), request)
        }
    }

    @ExceptionHandler(UnsatisfiedServletRequestParameterException::class)
    fun handleMissingParams(ex: Exception, request: WebRequest): ResponseEntity<Any> = logAndHandleBadRequestException(ex, request)

    override fun handleNoHandlerFoundException(ex: NoHandlerFoundException, headers: HttpHeaders,
                                               status: HttpStatus, request: WebRequest) : ResponseEntity<Any> {
        logException(ex)
        return handle(QuestionnaireServiceError.NotFoundError(ex.message!!), request)
    }

    override fun handleHttpMessageNotReadable(ex: HttpMessageNotReadableException, headers: HttpHeaders,
                                              status: HttpStatus, request: WebRequest) = logAndHandleBadRequestException(ex, request)

    override fun handleHttpRequestMethodNotSupported(ex: HttpRequestMethodNotSupportedException, headers: HttpHeaders,
                                                     status: HttpStatus, request: WebRequest) = logAndHandleBadRequestException(ex, request)

    override fun handleMethodArgumentNotValid(ex: MethodArgumentNotValidException, headers: HttpHeaders,
                                              status: HttpStatus, request: WebRequest) = logAndHandleBadRequestException(ex, request)

    override fun handleTypeMismatch(ex: TypeMismatchException, headers: HttpHeaders,
                                    status: HttpStatus, request: WebRequest) = logAndHandleBadRequestException(ex, request)

    private fun logAndHandleBadRequestException(ex: Exception,
                                                request: WebRequest): ResponseEntity<Any> {
        logException(ex)
        return handle(QuestionnaireServiceError.BadRequestError(ex.message!!), request)
    }

    private fun logException(ex: Exception) = when (ex) {
        is ErrorException -> log.error(ex.error.message, ex)
        else -> log.error("Unexpected exception", ex)
    }

    private fun handle(err: QuestionnaireServiceError, request: WebRequest) =
            handleExceptionInternal(ErrorException(err), err, HttpHeaders(), err.httpStatus(), request)
    
    private fun QuestionnaireServiceError.httpStatus(): HttpStatus =
            when (this) {
                is QuestionnaireServiceError.NotFoundError -> HttpStatus.NOT_FOUND
                is QuestionnaireServiceError.NotAuthorizedError -> HttpStatus.UNAUTHORIZED
                is QuestionnaireServiceError.NameInUseError -> HttpStatus.CONFLICT
                is QuestionnaireServiceError.ConflictError -> HttpStatus.CONFLICT
                is QuestionnaireServiceError.BadRequestError, is QuestionnaireServiceError.MissingAnswerToMandatoryQuestion -> HttpStatus.BAD_REQUEST
                is QuestionnaireServiceError.UnexpectedError -> HttpStatus.INTERNAL_SERVER_ERROR
                is QuestionnaireServiceError.IOError -> HttpStatus.INTERNAL_SERVER_ERROR
            }
}
