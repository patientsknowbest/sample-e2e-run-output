package com.pkb.questionnaireservice.controller

import com.pkb.questionnaireservice.ErrorException
import com.pkb.questionnaireservice.QuestionnaireId
import com.pkb.questionnaireservice.QuestionnaireRequestId
import com.pkb.questionnaireservice.entities.QuestionnaireRequestEntity
import com.pkb.questionnaireservice.modelv2.SearchResults
import com.pkb.questionnaireservice.service.QuestionnaireRequestService
import org.springframework.data.domain.Pageable
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.RequestBody
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RequestParam
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("/questionnaire-request")
class QuestionnaireRequestController(private val questionnaireRequestService: QuestionnaireRequestService) {

    @PostMapping
    fun postQuestionnaireRequest(@RequestBody questionnaireRequestEntity: QuestionnaireRequestEntity): ResponseEntity<QuestionnaireRequestEntity> =
            questionnaireRequestService.post(questionnaireRequestEntity)
                    .map { ResponseEntity(it, HttpStatus.CREATED) }
                    .getOrElseThrow { err -> ErrorException(err) }
    
    @PostMapping("batch")
    fun postQuestionnaireRequestsInBatch(@RequestBody questionnaireRequests: List<QuestionnaireRequestEntity>): ResponseEntity<List<QuestionnaireRequestEntity>> =
            questionnaireRequestService.postBatch(questionnaireRequests)
                .map { ResponseEntity(it, HttpStatus.CREATED) }
                .getOrElseThrow { err -> ErrorException(err) }

    @GetMapping("{id}")
    fun getQuestionnaireRequest(@PathVariable("id") id: QuestionnaireRequestId) : ResponseEntity<QuestionnaireRequestEntity> =
            questionnaireRequestService.get(id)
                    .map { ResponseEntity(it, HttpStatus.OK) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @GetMapping(params = ["questionnaire_ids"])
    fun getQuestionnaireRequestsForQuestionnaires(@RequestParam("questionnaire_ids") questionnaireIds: List<QuestionnaireId>, pageable: Pageable)
            : ResponseEntity<SearchResults<QuestionnaireRequestEntity>> =
            questionnaireRequestService.getByQuestionnaireIds(questionnaireIds, pageable)
                    .map { ResponseEntity(it, HttpStatus.OK) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @GetMapping("/non-responders/all/{questionnaire_id}")
    fun getAllNonRespondersForQuestionnaire(@PathVariable("questionnaire_id") id: QuestionnaireId)
            : ResponseEntity<List<QuestionnaireRequestEntity>> =
            questionnaireRequestService.getNonResponders(id, Pageable.unpaged())
                    .map { ResponseEntity(it, HttpStatus.OK) }
            .getOrElseThrow { err -> ErrorException(err) }
}
