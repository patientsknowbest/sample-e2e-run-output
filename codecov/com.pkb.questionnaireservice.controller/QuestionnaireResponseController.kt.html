package com.pkb.questionnaireservice.controller

import com.pkb.kms.shared.representation.PersonPublicId
import com.pkb.questionnaireservice.ErrorException
import com.pkb.questionnaireservice.OrgPublicId
import com.pkb.questionnaireservice.QuestionnaireId
import com.pkb.questionnaireservice.entities.DecryptedQuestionnaireResponse
import com.pkb.questionnaireservice.messages.QuestionnaireResponsesSearchCriteria
import com.pkb.questionnaireservice.modelv2.QuestionnaireResponse
import com.pkb.questionnaireservice.modelv2.SearchResults
import com.pkb.questionnaireservice.service.QuestionnaireResponseService
import org.springframework.data.domain.Pageable
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.RequestBody
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RequestParam
import org.springframework.web.bind.annotation.RestController
import reactor.core.publisher.Flux
import javax.validation.Valid

@RestController
@RequestMapping("/questionnaire-response")
class QuestionnaireResponseController(private val questionnaireResponseService: QuestionnaireResponseService) {

    @PostMapping
    fun postQuestionnaireResponse(@RequestBody questionnaireResponse: DecryptedQuestionnaireResponse): ResponseEntity<QuestionnaireResponse> =
            questionnaireResponseService.save(questionnaireResponse)
                    .map { ResponseEntity(it, HttpStatus.CREATED) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @GetMapping("{questionnaire_id}")
    fun getQuestionnaireResponsesByQuestionnaireId(@PathVariable("questionnaire_id") id: QuestionnaireId,
                                                   @RequestParam("user_id") loggedInUserId: PersonPublicId,
                                                   pageable:Pageable): ResponseEntity<SearchResults<QuestionnaireResponse>> =
            questionnaireResponseService.getByQuestionnaireId(id, loggedInUserId, pageable)
                    .map { ResponseEntity(it, HttpStatus.OK) }
                    .getOrElseThrow { err -> ErrorException(err) }

    @GetMapping("/all/{questionnaire_id}")
    fun getAllQuestionnaireResponsesByQuestionnaireId(@PathVariable("questionnaire_id") id: QuestionnaireId,
                                                      @RequestParam("user_id") loggedInUserId: PersonPublicId): Flux<QuestionnaireResponse> =
            questionnaireResponseService.streamByQuestionnaireId(id, loggedInUserId)

    @GetMapping
    fun getQuestionnaireResponsesWithinOrg(@RequestParam("org_id") orgId: OrgPublicId,
                                           @Valid questionnaireResponsesSearchCriteria: QuestionnaireResponsesSearchCriteria,
                                           pageable: Pageable): ResponseEntity<SearchResults<QuestionnaireResponse>>? =
            questionnaireResponseService.getWithinOrg(orgId, questionnaireResponsesSearchCriteria, pageable)
                    .map { ResponseEntity(it, HttpStatus.OK) }
                    .getOrElseThrow { err -> ErrorException(err) }
}
