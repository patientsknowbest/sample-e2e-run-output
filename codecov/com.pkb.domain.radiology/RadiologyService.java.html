package com.pkb.domain.radiology;

import com.pkb.authz.AuthorizationService;
import com.pkb.datamodel.RadiologyImage;
import com.pkb.domain.SynchUploadedDataService;
import com.pkb.domain.SystemUserTypeNotPermittedException;
import com.pkb.domain.criteria.CriteriaResult;
import com.pkb.domain.repository.RadiologyRepository;
import io.vavr.control.Either;
import io.vavr.control.Option;
import org.jetbrains.annotations.NotNull;

import static com.pkb.entities.enums.UploadedDataDestination.RADIOLOGY;
import static io.vavr.control.Either.left;

public class RadiologyService {
    private final RadiologyRepository radiologyRepository;
    private final AuthorizationService authorizationService;
    private final SynchUploadedDataService synchUploadedDataService;

    public RadiologyService(RadiologyRepository radiologyRepository,
                            AuthorizationService authorizationService,
                            SynchUploadedDataService synchUploadedDataService) {
        this.radiologyRepository = radiologyRepository;
        this.authorizationService = authorizationService;
        this.synchUploadedDataService = synchUploadedDataService;
    }

    @NotNull
    public Either<RuntimeException, CriteriaResult<RadiologyImage>> search(@NotNull RadiologyCriteria criteria) {
        if (criteria.authzData().inputs().authenticatedSystemUser().isDefined()) {
            return left(new SystemUserTypeNotPermittedException());
        }
        return authorizationService.authorize(criteria.authzData())
                .checkSingleResult(() -> doSearch(criteria));
    }

    private CriteriaResult<RadiologyImage> doSearch(@NotNull RadiologyCriteria criteria) {
        synchUploadedDataService.synchUploadedData(criteria.authzData(), RADIOLOGY);
        return radiologyRepository.findImagesBy(
                criteria.authzData(),
                criteria.getEffectiveTimestamp(),
                Option.of(criteria.getPageSize()));
    }
}
