package com.pkb.file.entity;

import com.pkb.annotation.EHRField;
import com.pkb.annotation.EHRField.QueryField;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.SourceDetails;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.util.FileUtil;
import org.apache.commons.text.StringEscapeUtils;

import java.io.Serializable;
import java.util.Date;
import java.util.UUID;
import java.util.regex.Pattern;

import static com.pkb.annotation.EHRField.QueryField.date01;
import static com.pkb.annotation.EHRField.QueryField.long04;

/**
 * A file uploaded by a patient into their account.
 * Data transfer object for passing through to the web layer.
 * <p>
 * Content is NOT necessarily populated -- left blank for most cases
 *
 * @author pravina : Added chunked upload/download functionality
 */

public class FileDTO implements IBaseDTO, Serializable {

    private static final long serialVersionUID = 5663264063098923700L;
    private static final Pattern WINDOWS_NEWLINE_PATTERN = Pattern.compile("\r\n");
    private static final Pattern LINUX_NEWLINE_PATTERN = Pattern.compile("\n");
    private static final Pattern WINDOWS_CARRIAGE_RETURN_PATTERN = Pattern.compile("\r");

    @EHRField(queryField = QueryField.id)
    private Long id;

    @EHRField(queryField = QueryField.documentMetadataId)
    private UUID documentMetadataId;

    @EHRField
    private String fileName;

    private byte[] content;

    private boolean sanitized;

    @EHRField
    private String mediaType;

    @EHRField(queryField = date01)
    private Date date;

    @EHRField
    private String notes;

    @EHRField
    private Long totalBytes;

    @EHRField(queryField = long04)
    private Long uploadedDataId;

    private final DTOBaseFields dtoBaseFields = new DTOBaseFields();

    private SourceDetails sourceDetails;

    /**
     * DELIBERATELY PROTECTED DO NOT CHANGE! Use the available constructors!
     * <p>
     * Would have even less visibility but JPA spec & Hibernate requires a no arg const. which is public or protected.
     * Other than those, nothing should be able to construct without parameters, as changelog wouldn't be filled correctly etc.
     */
    protected FileDTO() {
        this.sourceDetails = new SourceDetails();
    }

    public FileDTO(SourceDetails sourceDetails) {
        this.sourceDetails = sourceDetails;
    }

    public String getNotesHtml() {
        String htmlNotes = StringEscapeUtils.escapeHtml4(getNotes());
        // first escape unwanted HTML in the message (script tags, etc...)
        // because escaping will be disabled in the JSP
        if (htmlNotes != null) {
            htmlNotes = WINDOWS_NEWLINE_PATTERN.matcher(htmlNotes).replaceAll("<br/>");
            htmlNotes = LINUX_NEWLINE_PATTERN.matcher(htmlNotes).replaceAll("<br/>");
            htmlNotes = WINDOWS_CARRIAGE_RETURN_PATTERN.matcher(htmlNotes).replaceAll("<br/>");
        }
        return htmlNotes;
    }

    public boolean isPdf() {
        return FileUtil.isPdf(fileName, mediaType);
    }

    public boolean isConvertibleToPdf() {
        return FileUtil.isConvertibleToPdf(fileName, mediaType);
    }

    public boolean isViewable() {
        return FileUtil.isViewable(fileName);
    }

    public boolean isStreamable() {
        return FileUtil.isStreamable(fileName);
    }

    public String getVideoType() {
        return FileUtil.getVideoType(fileName);
    }

    @Override
    public Long getId() {
        return id;
    }

    @Override
    public void setId(Long id) {
        this.id = id;
    }

    public String getFileName() {
        return fileName;
    }

    public void setFileName(String fileName) {
        this.fileName = fileName;
    }

    public byte[] getContent() {
        return content;
    }

    public void setContent(byte[] content) {
        this.content = content;
    }

    public String getMediaType() {
        return mediaType;
    }

    public void setMediaType(String mediaType) {
        this.mediaType = mediaType;
    }

    public Date getDate() {
        return date;
    }

    public void setDate(Date date) {
        this.date = date;
    }

    public String getNotes() {
        return notes;
    }

    public void setNotes(String notes) {
        this.notes = notes;
    }

    public UUID getDocumentMetadataId() {
        return documentMetadataId;
    }

    public void setDocumentMetadataId(UUID documentMetadataId) {
        this.documentMetadataId = documentMetadataId;
    }

    public Long getTotalBytes() {
        return totalBytes;
    }

    public void setTotalBytes(Long totalBytes) {
        this.totalBytes = totalBytes;
    }

    public Long getUploadedDataId() {
        return uploadedDataId;
    }

    public void setUploadedDataId(Long uploadedDataId) {
        this.uploadedDataId = uploadedDataId;
    }

    @Override
    public SourceDetails getSource() {
        return sourceDetails;
    }

    @Override
    public DTOBaseFields getBaseFields() {
        return dtoBaseFields;
    }

    public boolean isSanitized() {
        return sanitized;
    }

    public void setSanitized(boolean sanitized) {
        this.sanitized = sanitized;
    }

    public void setSourceDetails(SourceDetails sourceDetails) {
        this.sourceDetails = sourceDetails;
    }
}
