package com.pkb.api.util.converter.v2;

import com.pkb.api.util.converter.AbstractConverter;
import com.pkb.api.util.converter.ConversionException;
import com.pkb.api.v2.model.ApiCarePlan;
import com.pkb.api.v2.model.ApiParticipant;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.phplan.entity.PHPlan;
import com.pkb.phplan.entity.PHPlanTemplate.TemplateFieldLabel;
import com.pkb.phplan.entity.PlanField;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.DateUtil;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.Nullable;

import java.util.Date;

/**
 * https://www.hl7.org/fhir/careplan.html
 *
 * @author pravina
 *
 */
public class CarePlanConverter extends AbstractConverter<ApiCarePlan, PHPlan> {

    private final DateUtil dateUtil;

    public CarePlanConverter(DateUtil dateUtil) {
        this.dateUtil = dateUtil;
    }

    @Override
    @Nullable
    public ApiCarePlan toApiModel(@Nullable PHPlan dto) throws ConversionException {
        if (dto == null) {
            return null;
        }

        ApiCarePlan api = new ApiCarePlan(dto.getPatientId().toString(), dto.getStatus());
        api.setId(dto.getBaseFields().getUniqueId());
        api.setVersion(dto.getVersion());
        api.setCreationDate(dto.getCreationDate());
        api.setPersistedDate(dto.getBaseFields().getPersistedDate());
        api.setEnteredDate(dto.getBaseFields().getEnteredDate());
        api.setDescription(dto.getDescription());
        api.setNote(dto.getNote());
        api.setName(dto.getName());
        api.setModified(dto.getLastEditedOn());
        api.setLastEditedOnISO(dateUtil.formatTimestampIso8601(dto.getLastEditedOn()));
        api.setApprovalStatus(dto.getApprovalStatus());
        api.setApproverId(dto.getApproverId());
        api.setEditableStatus(dto.getEditableStatus());
        api.setReviewDate(dto.getReviewDate());

        ParticipantConverter participantConverter = new ParticipantConverter();
        PKBPerson creator = dto.getCreator();
        if (creator != null) {
            api.setCreator(participantConverter.toApiModel(creator));
        }
        PKBPerson lastEditor = dto.getLastEditor();
        if (lastEditor != null) {
            api.setLastEditor(participantConverter.toApiModel(lastEditor));
        }

        return api;
    }

    @Override
    @Nullable
    public PHPlan toDTO(@Nullable ApiCarePlan api, EHRRequestContext requestContext) throws ConversionException {
        if (api == null) {
            return null;
        }

        if (api.getCreator() == null) {
            throw new ConversionException("creator cannot be null");
        }

        if (api.getPersistedDate() != null) {
            throw new ConversionException(
                    "Please do not set a persisted date - this will be done on the server");
        }

        String patientId = api.getSubject();
        PHPlan dto = new PHPlan(new SourceDetails(requestContext));
        dto.setName(api.getName());
        dto.setDescription(api.getDescription());
        dto.setCreationDate(api.getCreationDate());
        dto.setReviewDate(api.getReviewDate());
        dto.setNote(api.getNote());
        dto.setPatientId(Long.parseLong(patientId));
        dto.getBaseFields().setEnteredDate(api.getEnteredDate());
        dto.getBaseFields().setPersistedDate(api.getPersistedDate());
        dto.getBaseFields().setUniqueId(api.getId());

        //insert empty fields for now into care plans
        for (TemplateFieldLabel label : TemplateFieldLabel.values()) {
            PlanField newField = new PlanField();
            newField.setLabel(label);
            newField.setText(""); //empty text
            dto.addField(newField);
        }

        if (!StringUtils.isEmpty(api.getCreator().getPkbParticipantId())) {
            dto.setCreatorId(Long.valueOf(api.getCreator().getPkbParticipantId()));
        }

        ApiParticipant lastEditor = api.getLastEditor();
        if ((lastEditor != null) && !StringUtils.isEmpty(lastEditor.getPkbParticipantId())) {
            dto.setLastEditorId(Long.valueOf(api.getLastEditor().getPkbParticipantId()));
        }

        dto.setEditableStatus(api.getEditableStatus());
        dto.setLastEditedOn(dateUtil.parseTimestampIso8601(api.getLastEditedOnISO()).map(Date::from).orElse(api.getModified()));
        dto.setApprovalStatus(api.getApprovalStatus());
        dto.setApproverId(api.getApproverId());
        dto.setStatus(api.getStatus());
        return dto;
    }

}
