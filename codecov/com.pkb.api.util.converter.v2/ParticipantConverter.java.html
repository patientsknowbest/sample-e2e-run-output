package com.pkb.api.util.converter.v2;

import com.pkb.api.util.converter.AbstractConverter;
import com.pkb.api.util.converter.ConversionException;
import com.pkb.api.v2.model.ApiParticipant;
import com.pkb.api.v2.model.ApiParticipant.ApiParticipantRole;
import com.pkb.app.dto.ParticipantDTO;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.Nullable;
import org.springframework.stereotype.Component;


@Component
public class ParticipantConverter extends AbstractConverter<ApiParticipant, ParticipantDTO> {

    @Override
    public ApiParticipant toApiModel(ParticipantDTO dto) throws ConversionException {

        ApiParticipant api = new ApiParticipant();
        api.setNonPkbParticipant(dto.getNonpkbParticipant());
        api.setPkbParticipantId(dto.getPersonId());
        if (dto.getRole() != null) {
            for (ApiParticipantRole candidate : ApiParticipantRole.values()) {
                if (candidate.toString().equals(dto.getRole().toString())) {
                    api.setParticipantRole(candidate);
                }
            }
            if (api.getParticipantRole() == null) {
                throw new ConversionException("Couldn't map participant role "
                        + dto.getRole().toString());
            }
        }
        PKBPerson participantPerson = dto.getPkbPerson();

        if (participantPerson != null) {
            api.setFirstName(participantPerson.getFirstName());
            api.setLastName(participantPerson.getLastName());
            api.setTitle(participantPerson.getTitle());
            api.setJobTitle(participantPerson.getJobTitle());
        }
        return api;

    }

    public ApiParticipant toApiModel(PKBPerson person) {
        ApiParticipant api = new ApiParticipant();
        api.setFirstName(person.getFirstName());
        api.setLastName(person.getLastName());
        api.setJobTitle(person.getJobTitle());
        api.setTeamName(person.getOrganizationName());
        api.setTitle(person.getTitle());
        api.setPkbParticipantId(person.getIdString());
        api.setParticipantRole(ApiParticipantRole.BLANK);
        return api;
    }

    @Override
    @Nullable
    public ParticipantDTO toDTO(@Nullable ApiParticipant api, EHRRequestContext requestContext) throws ConversionException {

        if (api == null) {
            return null;
        }
        ParticipantDTO dto = new ParticipantDTO();

        dto.setNonpkbParticipant(api.getNonPkbParticipant());
        dto.setPersonId(api.getPkbParticipantId());
        if (api.getParticipantRole() != null) {
            for (com.pkb.app.dto.ParticipantDTO.ParticipantRole candidate : com.pkb.app.dto.ParticipantDTO.ParticipantRole
                    .values()) {
                if (candidate.toString().equals(api.getParticipantRole().toString())) {
                    dto.setRole(candidate);
                }
            }
            if (dto.getRole() == null) {
                throw new ConversionException("Couldn't map participant role "
                        + api.getParticipantRole());
            }
        }

        return dto;

    }

}
