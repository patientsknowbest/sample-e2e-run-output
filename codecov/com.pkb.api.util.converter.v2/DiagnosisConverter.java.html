package com.pkb.api.util.converter.v2;

import com.pkb.api.util.converter.AbstractConverter;
import com.pkb.api.util.converter.ConversionException;
import com.pkb.api.util.converter.IMergingConverter;
import com.pkb.api.v2.model.ApiDiagnosis;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.coding.entity.CodeableConcept;
import com.pkb.diagnosis.entity.Diagnosis;
import org.apache.commons.lang3.StringUtils;

/**
 * @author pravina
 */
public class DiagnosisConverter extends AbstractConverter<ApiDiagnosis, Diagnosis> implements IMergingConverter<ApiDiagnosis, Diagnosis> {


    @Override
    public ApiDiagnosis toApiModel(Diagnosis dto) throws ConversionException {
        DTOBaseFields baseFields = dto.getBaseFields();
        ApiDiagnosis api = new ApiDiagnosis(baseFields.getUniqueId(), baseFields.getEnteredDate(), baseFields.getPersistedDate());
        api.setDiagnosis(dto.getDiagnosis().getTextForDisplay());
        api.setDetails(dto.getDetails());
        api.setStartDate(dto.getStartDate());
        api.setEndDate(dto.getEndDate());

        return api;
    }

    @Override
    public Diagnosis toDTO(ApiDiagnosis api, EHRRequestContext requestContext) throws ConversionException {
        Diagnosis dto = new Diagnosis(new SourceDetails(requestContext));
        mergeToDTO(api, dto, requestContext);
        return dto;
    }

    @Override
    public void mergeToDTO(ApiDiagnosis api, Diagnosis dto, EHRRequestContext requestContext) throws ConversionException {

        if (StringUtils.isBlank(api.getDiagnosis())) {
            throw new ConversionException("Please specify a diagnosis property");
        }
        if (api.getPersistedDate() != null) {
            throw new ConversionException(
                    "Please do not set a persisted date - this will be done on the server");
        }

        BaseFieldPopulator baseFieldPopulator = new BaseFieldPopulator();
        baseFieldPopulator.populateDTO(api, dto);
        dto.setDiagnosis(new CodeableConcept.Builder(api.getDiagnosis()).build());
        dto.setDetails(api.getDetails());
        dto.setStartDate(api.getStartDate());
        dto.setEndDate(api.getEndDate());
    }

}
