package com.pkb.api.util.converter.v2;

import com.google.common.primitives.Longs;
import com.pkb.api.util.converter.AbstractConverter;
import com.pkb.api.util.converter.ConversionException;
import com.pkb.api.v1.AttachmentController;
import com.pkb.api.v1.model.ApiAttachment;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.config.PhrConfig;
import com.pkb.document.entity.Attachment;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import javax.annotation.PostConstruct;


@Component
public class AttachmentConverter extends AbstractConverter<ApiAttachment, Attachment> {

    @Autowired
    private PhrConfig config;

    private String attchmentUrl;

    @PostConstruct
    protected void setAttachmentUrl() {
        attchmentUrl = config.getBaseURL() + "/json" + AttachmentController.ATTACHMENTS_BASE_PATH + "/";
    }

    @Override
    public ApiAttachment toApiModel(Attachment attachment) throws ConversionException {
        ApiAttachment ap = new ApiAttachment();
        ap.setId(attachment.getId().toString());
        ap.setDescription(attachment.getFilename());
        ap.setMediaType(attachment.getMediaType());
        ap.setLabel(attachment.getFilename());
        ap.setUploadTime(attachment.getUploadTime());
        ap.setContentUrl(attchmentUrl + attachment.getId());
        ap.setName(attachment.getFilename());
        return ap;
    }

    @Override
    public Attachment toDTO(ApiAttachment apiAttachment, EHRRequestContext requestContext) throws ConversionException {
        Attachment at = new Attachment();
        at.setId(Longs.tryParse(apiAttachment.getId()));
        at.setFilename(apiAttachment.getDescription());
        at.setMediaType(apiAttachment.getMediaType());
        at.setUploadTime(apiAttachment.getUploadTime());
        return at;
    }
}
