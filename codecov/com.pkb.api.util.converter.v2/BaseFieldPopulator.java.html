package com.pkb.api.util.converter.v2;

import com.pkb.api.util.converter.ConversionException;
import com.pkb.api.v2.model.BaseApiModel;
import com.pkb.api.v2.model.BaseApiModelBuilder;
import com.pkb.app.dto.DTOBaseFields;
import com.pkb.app.interfaces.IBaseDTO;

public class BaseFieldPopulator {


    /**
     *
     * @param builder
     * @param baseFields
     * @return {@code builder} (just to make it chainable)
     * @throws ConversionException
     */
    public <B extends BaseApiModelBuilder<?, B>> B populateBuilder(B builder, DTOBaseFields baseFields)
            throws ConversionException {
        return builder
                .id(baseFields.getUniqueId())
                .enteredDate(baseFields.getEnteredDate())
                .persistedDate(baseFields.getPersistedDate());
    }

    public void populateDTO(BaseApiModel apiModel, DTOBaseFields baseFields) throws ConversionException {
        baseFields.setUniqueId(apiModel.getId());
        baseFields.setEnteredDate(apiModel.getEnteredDate());
        baseFields.setPersistedDate(apiModel.getPersistedDate());
    }

    public void populateDTO(BaseApiModel apiModel, IBaseDTO dto) throws ConversionException {
        populateDTO(apiModel, dto.getBaseFields());
    }

    public <B extends BaseApiModelBuilder<?, B>> B populateBuilder(B builder, IBaseDTO dto) throws ConversionException {
        return populateBuilder(builder, dto.getBaseFields());
    }

}
