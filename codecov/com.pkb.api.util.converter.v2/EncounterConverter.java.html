package com.pkb.api.util.converter.v2;

import com.pkb.api.error.ApiValidationError;
import com.pkb.api.util.converter.AbstractConverter;
import com.pkb.api.util.converter.ConversionException;
import com.pkb.api.util.mapper.ApiEncounterToEncounterMapper;
import com.pkb.api.util.mapper.EncounterStatusMapper;
import com.pkb.api.util.mapper.ParticipantRoleMapper;
import com.pkb.api.v2.model.ApiEncounter;
import com.pkb.api.v2.model.ApiEncounterMessage;
import com.pkb.api.v2.model.ApiEncounterStatus;
import com.pkb.api.v2.model.ApiParticipant;
import com.pkb.app.dto.ParticipantDTO;
import com.pkb.app.dto.ParticipantDTO.ParticipantRole;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.encounter.entity.Encounter;
import com.pkb.encounter.entity.Encounter.EncounterStatus;
import com.pkb.encounter.entity.Message;
import com.pkb.entities.enums.EncounterClass;
import com.pkb.service.encounter.impl.tolven.EncounterReasonApiRef;
import com.pkb.user.entity.PKBPerson;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Optional;

@Component
public class EncounterConverter extends AbstractConverter<ApiEncounter, Encounter> {

    @Autowired
    private PkbPersonToApiParticipantConverter pkbPersonToApiParticipantConverter;

    @Autowired
    private ParticipantConverter participantConverter;

    @Autowired
    private MessageToApiEncounterMessageConverter messageToApiEncounterMessageConverter;

    @Autowired
    private ApiEncounterToEncounterMapper apiEncounterToEncounterMapper;

    @Autowired
    private EncounterStatusMapper encounterStatusMapper;

    @Autowired
    private ParticipantRoleMapper participantRoleMapper;

    @Override
    public ApiEncounter toApiModel(Encounter encounter) throws ConversionException {

        ApiEncounter api = new ApiEncounter();

        if (encounter.getEncounterClass() != null) {
            api.setEncounterClass(apiEncounterToEncounterMapper.getValue(encounter.getEncounterClass()));
        }

        api.setEndDate(encounter.getEndDate());
        api.setEnteredDate(encounter.getBaseFields().getEnteredDate());
        api.setId(encounter.getEncounterUniqueId());
        api.setIndication(encounter.getIndication());
        api.setMessages(new ArrayList<>());
        for (Message message : encounter.getMessages()) {
            api.getMessages().add(messageToApiEncounterMessageConverter.toApiModel(message));
        }

        api.setParticipants(new ArrayList<>());
        if (encounter.getParticipantsPKB() != null) {
            for (PKBPerson person : encounter.getParticipantsPKB()) {
                api.getParticipants().add(pkbPersonToApiParticipantConverter.convert(person));
            }
        }
        if (encounter.getParticipantsNonPKB() != null) {
            for (com.pkb.app.dto.ParticipantDTO dto : encounter.getParticipantsNonPKB()) {
                api.getParticipants().add(participantConverter.toApiModel(dto));
            }
        }

        api.setPatientId(encounter.getAboutPatientId());
        api.setPersistedDate(encounter.getBaseFields().getPersistedDate());
        api.setStartDate(encounter.getStartDate());

        if (encounter.getEncounterStatus() != null) {
            api.setStatus(encounterStatusMapper.getValue(encounter.getEncounterStatus()));
        }
        if (encounter.getReasonCode() != null) {
            api.setReason(EncounterReasonApiRef.getApiRefByCode(encounter.getReasonCode()));
        }
        return api;
    }

    @Override
    public Encounter toDTO(ApiEncounter api, EHRRequestContext requestContext) throws ConversionException {
        Encounter dto = new Encounter(new SourceDetails(requestContext));

        if (api.getEncounterClass() != null) {
            Optional<EncounterClass> maybeEncounter = apiEncounterToEncounterMapper.getPossibleValue(api.getEncounterClass());
            dto.setEncounterClass(
                    maybeEncounter.orElseThrow(() -> new ConversionException("Couldn't map encounter class: " + api.getEncounterClass())));
        }
        dto.setEndDate(api.getEndDate());
        dto.getBaseFields().setEnteredDate(api.getEnteredDate());
        dto.setEncounterUniqueId(api.getId());
        dto.setIndication(api.getIndication());

        if (api.getMessages() != null) {
            for (ApiEncounterMessage message : api.getMessages()) {
                dto.getMessages().add(messageToApiEncounterMessageConverter.toDTO(message, requestContext));
            }
        }

        dto.setParticipants(new ArrayList<>());
        for (ApiParticipant participant : api.getParticipants()) {
            if (participant.getPkbParticipantId() != null) {
                dto.getParticipants().add(participant.getPkbParticipantId());
            } else {
                ParticipantDTO participantDto = new ParticipantDTO();
                participantDto.setNonpkbParticipant(participant.getNonPkbParticipant());
                if (participant.getParticipantRole() != null) {
                    Optional<ParticipantRole> maybeParticipantRole = participantRoleMapper
                            .getPossibleValue(participant.getParticipantRole());
                    participantDto.setRole(maybeParticipantRole.orElseThrow(
                            () -> new ConversionException("Couldn't map ParticipantRole: " + participant.getParticipantRole())));
                }
            }

        }

        dto.setAboutPatientId(api.getPatientId());
        dto.setStartDate(api.getStartDate());
        if ((api.getStatus() != null) && api.getStatus() != ApiEncounterStatus.BLANK) {
            Optional<EncounterStatus> maybeEncounterStatus = encounterStatusMapper.getPossibleValue(api.getStatus());
            dto.setEncounterStatus(maybeEncounterStatus
                    .orElseThrow(() -> new ConversionException("Couldn't map encounter status: " + api.getStatus())));
        }
        if ((api.getReason() != null) && !api.getReason().isEmpty()) {
            if (EncounterReasonApiRef.getCodeByApiRef(api.getReason()) == null) {
                throw new ApiValidationError(String.format("Unrecognised reason: %s. Valid values include: %s", api.getReason(),
                        Arrays.toString(EncounterReasonApiRef.values())));

            }
            dto.setApiRef(api.getReason());
        }

        return dto;
    }

}
