package com.pkb.api.util.converter.v2;

import com.pkb.api.v2.model.ApiParticipant;
import com.pkb.api.v2.model.ApiParticipant.ApiParticipantRole;
import com.pkb.user.entity.PKBPerson;
import org.springframework.core.convert.converter.Converter;
import org.springframework.stereotype.Component;

@Component
public class PkbPersonToApiParticipantConverter implements Converter<PKBPerson, ApiParticipant> {

    @Override
    public ApiParticipant convert(PKBPerson person) {
        ApiParticipant api = new ApiParticipant();
        api.setFirstName(person.getFirstName());
        api.setLastName(person.getLastName());
        api.setJobTitle(person.getJobTitle());
        api.setTeamName(person.getOrganizationName());
        api.setTitle(person.getTitle());
        api.setPkbParticipantId(person.getIdString());
        api.setParticipantRole(ApiParticipantRole.BLANK);
        return api;
    }

}
