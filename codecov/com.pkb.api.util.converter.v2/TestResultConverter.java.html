package com.pkb.api.util.converter.v2;

import com.pkb.api.error.ApiValidationError;
import com.pkb.api.util.converter.AbstractConverter;
import com.pkb.api.util.converter.ConversionException;
import com.pkb.api.util.converter.IMergingConverter;
import com.pkb.api.v2.model.ApiTestResult;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.entities.enums.Route;
import com.pkb.test.entity.TestFormDTO;
import com.pkb.test.entity.TestResultDTO;
import com.pkb.util.LocalizedTextProvider;

import java.util.Date;

import static com.pkb.util.TestResultDTOMappers.getTestResultSourceDTO;
import static org.apache.commons.lang3.StringUtils.EMPTY;

/**
 * @author pravina
 */
public class TestResultConverter extends AbstractConverter<ApiTestResult, TestResultDTO>
        implements IMergingConverter<ApiTestResult, TestResultDTO> {

    private final LocalizedTextProvider localizedTextProvider;

    public TestResultConverter(LocalizedTextProvider localizedTextProvider) {
        this.localizedTextProvider = localizedTextProvider;
    }

    @Override
    public ApiTestResult toApiModel(TestResultDTO dto) throws ConversionException {
        ApiTestResult api = new ApiTestResult();
        api.setId(dto.getBaseFields().getUniqueId());
        api.setLoincTestId((dto.getLoincTestId() == null) ? null : dto.getLoincTestId());
        api.setRangeHigh(dto.getRangeHigh());
        api.setRangeHighInclusive(dto.isRangeHighInclusive());
        api.setRangeLow(dto.getRangeLow());
        api.setRangeLowInclusive(dto.isRangeLowInclusive());
        api.setResultDate(dto.getTestDate() == null ? null : Date.from(dto.getTestDate()));
        api.setTextValue(dto.getTextValue());
        api.setComments(dto.getComments());
        api.setEnteredDate(dto.getBaseFields().getEnteredDate());
        api.setPersistedDate(dto.getBaseFields().getPersistedDate());
        if (dto.getValue() != null) {
            api.setValue(dto.getValue().toString());
        }
        api.setDelayDisplayDays(dto.getDelayDisplayDays());
        api.setTextRange(dto.getTextRange());

        if (dto.getLabOrderId() != null) {
            api.setLabOrderId(dto.getLabOrderId());
        }

        TestFormDTO.TestResultSourceDTO testResultSourceDTO = getTestResultSourceDTO(dto.getSource(), this::routePrinter);

        api.setSourceType(testResultSourceDTO.getSourceType());
        api.setSourceEnteredBy(testResultSourceDTO.getDisplayText());
        api.setSourcePersonId(testResultSourceDTO.getPersonId());
        api.setSourceRoute(testResultSourceDTO.getVia());

        return api;
    }

    private String routePrinter(Route route) {
        if (route != null && route.isCaptured()) {
            String valueOfRoute = localizedTextProvider.getText("com.pkb.app.entity.Route." + route);
            return localizedTextProvider.getText("source.via", valueOfRoute);
        }
        return EMPTY;
    }

    @Override
    public TestResultDTO toDTO(ApiTestResult apiModel, EHRRequestContext requestContext) throws ConversionException {
        TestResultDTO dto = new TestResultDTO(new SourceDetails(requestContext));
        mergeToDTO(apiModel, dto, requestContext);
        return dto;
    }

    @Override
    public void mergeToDTO(ApiTestResult apiModel, TestResultDTO targetDTO, EHRRequestContext requestContext) throws ConversionException {
        if (apiModel.getResultDate() == null) {
            throw new ApiValidationError("Data error: null test date in existing test result record");
        }

        if (apiModel.getPersistedDate() != null) {
            throw new ApiValidationError("Please don't set persistedDate. This will be done on the server.");
        }

        if (apiModel.getValue() == null && apiModel.getTextValue() == null) {
            throw new ApiValidationError("A value must be provided for test results");
        }

        if (apiModel.getValue() != null) {
            if (apiModel.getTextValue() != null) {
                throw new ApiValidationError("You cannot specify both value and textValue for a result");
            }
            try {
                targetDTO.setValue(Double.parseDouble(apiModel.getValue()));
            } catch (NumberFormatException ignored) {
                throw new ApiValidationError(
                        "Couldn't read value for this result. The value field must be a numeric. If you want to use a text value, please use the textValue field");
            }
        } else {
            targetDTO.setTextValue(apiModel.getTextValue());
        }

        targetDTO.setTestDate(apiModel.getResultDate().toInstant());
        targetDTO.setComments(apiModel.getComments());
        if (apiModel.getLoincTestId() != null) {
            targetDTO.setLoincTestId(apiModel.getLoincTestId());
        }

        if (apiModel.getRangeLow() != null) {
            targetDTO.setRangeLow(apiModel.getRangeLow());
            targetDTO.setRangeLowInclusive(true);
        }
        if (apiModel.getRangeHigh() != null) {
            targetDTO.setRangeHigh(apiModel.getRangeHigh());
            targetDTO.setRangeHighInclusive(true);
        }
        if (apiModel.getEnteredDate() != null) {
            targetDTO.getBaseFields().setEnteredDate(apiModel.getEnteredDate());
        }

        if (apiModel.getLabOrderId() != null) {
            targetDTO.setLabOrderId(apiModel.getLabOrderId());
        }
    }
}
