package com.pkb.api.util.converter.v2;

import com.pkb.api.error.ApiValidationError;
import com.pkb.api.util.converter.AbstractConverter;
import com.pkb.api.util.converter.ConversionException;
import com.pkb.api.v2.model.ApiEncounterMessage;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.encounter.entity.Message;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import javax.ws.rs.core.MediaType;
import java.util.Date;

@Component
public class MessageToApiEncounterMessageConverter extends AbstractConverter<ApiEncounterMessage, Message> {

    @Autowired
    private AttachmentConverter attachmentConverter;

    @Autowired
    private DateTimeService dateTimeService;

    @Override
    public Message toDTO(ApiEncounterMessage message, EHRRequestContext requestContext) throws ConversionException {
        Message dto = new Message(new SourceDetails(requestContext));
        dto.setContent(message.getBody());
        if (MediaType.TEXT_HTML.equals(message.getContentType())) {
            dto.setHtmlAllowed(true);
        }
        dto.setSenderId(message.getSenderId());

        if (message.getEnteredDate() == null) {
            message.setEnteredDate(Date.from(dateTimeService.now()));
        }

        dto.getBaseFields().setEnteredDate(message.getEnteredDate());
        dto.setMessageTimestamp(message.getEnteredDate().toInstant());
        if (message.getPersistedDate() != null) {
            throw new ApiValidationError("Please do not specify a persisted date. This will be done on the server");
        }
        return dto;
    }

    @Override
    public ApiEncounterMessage toApiModel(Message message) throws ConversionException {
        ApiEncounterMessage api = new ApiEncounterMessage();
        api.setId(message.getId().toString());
        api.setSubject(message.getSubject());
        // we support only text/html and text/plain
        if (message.isHtmlAllowed()) {
            api.setContentType(MediaType.TEXT_HTML);
        } else {
            api.setContentType(MediaType.TEXT_PLAIN);
        }
        api.setBody(message.getContent());
        api.setSenderId(message.getSenderId());
        api.setEnteredDate(message.getBaseFields().getEnteredDate());
        api.setPersistedDate(message.getBaseFields().getPersistedDate());
        api.setAttachments(attachmentConverter.toApiModel(message.getAttachments()));
        return api;
    }
}
