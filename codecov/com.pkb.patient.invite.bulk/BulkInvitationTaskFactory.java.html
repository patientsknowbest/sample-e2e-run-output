package com.pkb.patient.invite.bulk;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.common.csv.CsvRecord;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.consent.PatientConsentUpdateTemplate;
import com.pkb.institute.entity.Team;
import com.pkb.patient.InvitationNationalIdValidator;
import com.pkb.patient.invite.bulk.csv.BulkInvitationCsvProcessorFactory;
import com.pkb.patient.invite.bulk.csv.BulkInvitationFacade;
import com.pkb.patient.invite.bulk.csv.BulkInvitationResultProcessor;
import com.pkb.patient.invite.bulk.csv.CsvContentConverter;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.user.BulkInvitationOutcomesManager;
import com.pkb.user.entity.PKBPerson;
import io.vavr.collection.List;
import org.jetbrains.annotations.NotNull;

import static com.pkb.patient.invite.bulk.csv.BulkInvitationFacade.parallelStrategy;

public class BulkInvitationTaskFactory {
    private final PKBEmailMessageManager pkbEmailMessageManager;
    private final BulkInvitationResultProcessor resultProcessor;
    private final CsvContentConverter csvContentConverter;
    private final BulkInvitationOutcomesManager outcomesManager;
    private final DateTimeService dateTimeService;
    private final BulkInvitationCsvProcessorFactory csvProcessorFactory;

    public BulkInvitationTaskFactory(
            PKBEmailMessageManager pkbEmailMessageManager,
            @NotNull BulkInvitationResultProcessor resultProcessor,
            @NotNull CsvContentConverter csvContentConverter,
            @NotNull BulkInvitationOutcomesManager outcomesManager,
            @NotNull DateTimeService dateTimeService,
            @NotNull BulkInvitationCsvProcessorFactory csvProcessorFactory) {
        this.pkbEmailMessageManager = pkbEmailMessageManager;
        this.resultProcessor = resultProcessor;
        this.csvContentConverter = csvContentConverter;
        this.outcomesManager = outcomesManager;
        this.dateTimeService = dateTimeService;
        this.csvProcessorFactory = csvProcessorFactory;
    }

    public Runnable create(
            @NotNull LoggedInEHRRequestContext requestContext,
            @NotNull PKBPerson inviter,
            @NotNull List<CsvRecord> csvContent,
            @NotNull Team invitingTeam,
            @NotNull InvitationNationalIdValidator invitationNationalIdValidator,
            @NotNull PatientConsentUpdateTemplate patientConsent) {

        BulkInvitationFacade bulkInvitationTask = new BulkInvitationFacade(
                pkbEmailMessageManager,
                resultProcessor,
                csvContentConverter,
                outcomesManager,
                dateTimeService,
                csvProcessorFactory.create(requestContext, inviter, invitationNationalIdValidator, invitingTeam));

        return () -> bulkInvitationTask.invitePatientsInBulk(inviter, invitingTeam, csvContent, parallelStrategy(), patientConsent, requestContext.getAccessingUserId());
    }
}
