package com.pkb.patient.invite.bulk;

import org.jetbrains.annotations.NotNull;

import java.util.Comparator;

public abstract class AbstractLineNumberedResult<R> {
    private static final Comparator<AbstractLineNumberedResult<?>> COMPARATOR = (a, b) -> (a.lineNumber > b.lineNumber) ? 1 : (a.lineNumber < b.lineNumber ? -1 : 0);

    @SuppressWarnings("unchecked")
    public static @NotNull <R extends AbstractLineNumberedResult> Comparator<R> byLineNumber() {
        return (Comparator<R>) COMPARATOR;
    }

    private final long lineNumber;

    public AbstractLineNumberedResult(long lineNumber) {
        this.lineNumber = lineNumber;
    }

    public long getLineNumber() {
        return lineNumber;
    }

    public abstract @NotNull R getResult();
}
