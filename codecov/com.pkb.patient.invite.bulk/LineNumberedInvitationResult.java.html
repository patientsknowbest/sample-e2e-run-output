package com.pkb.patient.invite.bulk;

import com.pkb.patient.invite.PatientRegister.FailedInvitation;
import com.pkb.patient.invite.PatientRegister.SuccessfulInvitation;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

public final class LineNumberedInvitationResult extends AbstractLineNumberedResult<Either<FailedInvitation, SuccessfulInvitation>> {
    private final Either<FailedInvitation, SuccessfulInvitation> result;

    public LineNumberedInvitationResult(long lineNumber, @NotNull SuccessfulInvitation successfulInvitation) {
        this(lineNumber, Either.right(successfulInvitation));
    }

    public LineNumberedInvitationResult(long lineNumber, @NotNull FailedInvitation failedInvitation) {
        this(lineNumber, Either.left(failedInvitation));
    }

    private LineNumberedInvitationResult(long lineNumber, Either<FailedInvitation, SuccessfulInvitation> result) {
        super(lineNumber);
        this.result = result;
    }

    @Override
    public @NotNull Either<FailedInvitation, SuccessfulInvitation> getResult() {
        return result;
    }
}
