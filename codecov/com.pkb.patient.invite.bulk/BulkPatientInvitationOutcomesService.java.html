package com.pkb.patient.invite.bulk;

import com.pkb.datamodel.BulkInvitationOutcome;
import com.pkb.datamodel.Email;
import com.pkb.patient.invite.BulkPatientInvitationOutcomesDto;
import com.pkb.patient.invite.BulkPatientInvitationOutcomesDto.BulkPatientCreationOutcomeDto;
import com.pkb.service.user.BulkInvitationOutcomesManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;

import java.time.format.DateTimeFormatter;
import java.util.List;
import java.util.Locale;
import java.util.Map;
import java.util.Optional;
import java.util.function.Function;
import java.util.stream.Collectors;

import static com.pkb.util.Constants.APPLICATION_TZ;
import static java.lang.String.format;
import static java.util.stream.Collectors.toList;

public class BulkPatientInvitationOutcomesService {
    private static final DateTimeFormatter dateTimeFormatter = DateTimeFormatter.ofPattern("dd MMM yyyy HH:mm:ss").withLocale(Locale.UK).withZone(APPLICATION_TZ);

    private final BulkInvitationOutcomesManager outcomesManager;
    private final UserManager userManager;

    public BulkPatientInvitationOutcomesService(BulkInvitationOutcomesManager outcomesManager, UserManager userManager) {
        this.outcomesManager = outcomesManager;
        this.userManager = userManager;
    }

    public BulkPatientInvitationOutcomesDto listOutcomes(long teamId) {
        List<BulkInvitationOutcome> outcomes = outcomesManager.findLastInvitationOutcomesForTeam(teamId);
        Map<Long, PKBPerson> inviterMap = userManager.getPKBPersonList(
                outcomes.stream().map(BulkInvitationOutcome::getUploadedById)
                        .collect(toList()),
                PKBPerson.Lazy.CONTACTS).stream().collect(
                        Collectors.toMap(PKBPerson::getId, Function.identity()));
        return new BulkPatientInvitationOutcomesDto(outcomes.stream()
                .map(outcome -> {
                    PKBPerson inviter = inviterMap.get(outcome.getUploadedById());

                    String uploadedAt = dateTimeFormatter.format(outcome.getUploadedAt());
                    String initiatorName = nameOf(inviter);
                    String link = Long.toString(outcome.getId());
                    Email userName = inviter.getUserName();
                    String status = outcome.getStatus().name();
                    return new BulkPatientCreationOutcomeDto(uploadedAt, initiatorName, link, userName, status);
                })
                .collect(toList()));
    }

    public Optional<String> findBulkPatientInvitationOutcome(long teamId, long reportId) {
        return outcomesManager.findOutcomeById(teamId, reportId);
    }

    private String nameOf(PKBPerson uploadedBy) {
        return format("%s %s %s", uploadedBy.getTitle(), uploadedBy.getFirstName(), uploadedBy.getLastName());
    }
}
