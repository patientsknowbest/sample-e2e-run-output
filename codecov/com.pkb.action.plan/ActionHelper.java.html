package com.pkb.action.plan;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.test.entity.MeasurementDTO;
import com.pkb.test.entity.MeasurementTypeId;
import com.pkb.util.StringUtil;

import java.time.Instant;
import java.util.Optional;

import static com.pkb.entities.enums.PrivacyFlag.GENERAL;
import static java.lang.Double.valueOf;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class ActionHelper {

    public static Optional<MeasurementDTO> convertToMeasurement(String value1, String value2, MeasurementTypeId typeId, Instant measureDate, EHRRequestContext context) {
        Optional<MeasurementDTO> result = Optional.empty();
        if (isNotBlank(value1) || isNotBlank(value2)) {
            MeasurementDTO dto = new MeasurementDTO(new SourceDetails(context));
            dto.setCompositeMeasurementTypeId(typeId);
            dto.setMeasureDate(measureDate);
            dto.getBaseFields().setPrivacyFlag(GENERAL); //See slack conversation https://patientsknowbest.slack.com/archives/CDJSGK7QX/p1547741069036600
            if (StringUtil.isDecimalNumber(value1)) {
                dto.setValue(valueOf(value1));
            }
            if (StringUtil.isDecimalNumber(value2)) {
                dto.setValue2(valueOf(value2));
            }
            result = Optional.of(dto);
        }
        return result;
    }
}
