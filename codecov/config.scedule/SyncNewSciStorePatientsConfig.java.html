package config.scedule;

import java.text.ParseException;

import org.quartz.JobDetail;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.scheduling.quartz.CronTriggerFactoryBean;
import org.springframework.scheduling.quartz.JobDetailFactoryBean;

import com.pkb.common.config.PhrConfig;
import com.pkb.service.integration.SyncNewPatientFromSciStoreJob;

@Configuration
public class SyncNewSciStorePatientsConfig {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final String SYNC_NEW_SCI_STORE_PATIENTS_JOB_DETAIL = "syncNewSciStorePatientsJobDetail";

    public static final String SCISTORE_JOB_DETAIL_KEY = "SciStoreJobDetailKey";
    public static final String SCISTORE_GROUP = "SciStoreGroup";

    @Autowired
    private PhrConfig config;

    @Bean(name = SYNC_NEW_SCI_STORE_PATIENTS_JOB_DETAIL)
    public JobDetailFactoryBean syncNewSciStorePatientsJobDetail() {
        JobDetailFactoryBean jobDetailFactory = new JobDetailFactoryBean();
        jobDetailFactory.setJobClass(SyncNewPatientFromSciStoreJob.class);
        jobDetailFactory.setDurability(true);
        jobDetailFactory.setDescription("Invokes SciStore integration job");
        jobDetailFactory.setGroup(SCISTORE_GROUP);
        jobDetailFactory.setName(SCISTORE_JOB_DETAIL_KEY);
        return jobDetailFactory;
    }

    @Bean
    public CronTriggerFactoryBean syncNewSciStorePatientsTrigger(@Qualifier(SYNC_NEW_SCI_STORE_PATIENTS_JOB_DETAIL) JobDetail syncNewSciStorePatientsJobDetail)
            throws ParseException {
        CronTriggerFactoryBean ctfb = new CronTriggerFactoryBean();
        ctfb.setJobDetail(syncNewSciStorePatientsJobDetail);
        LOGGER.info("Setting cron expression to: [{}]", config.getScriStorePatientSyncCron());
        ctfb.setCronExpression(config.getScriStorePatientSyncCron());
        return ctfb;
    }

}
