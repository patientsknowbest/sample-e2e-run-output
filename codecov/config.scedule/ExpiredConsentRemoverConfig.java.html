package config.scedule;

import com.pkb.common.config.PhrConfig;
import com.pkb.service.emailmessage.IAddressInfoProvider;
import com.pkb.service.emailmessage.ITemplateParamsFactory;
import com.pkb.service.emailmessage.impl.EmailManager;
import com.pkb.service.patientconsent.ExpiredConsentRemover;
import com.pkb.service.patientconsent.ExpiredConsentRemoverJob;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.user.impl.UserAccessManager;
import com.pkb.service.user.impl.UserManager;
import org.quartz.JobDetail;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.scheduling.quartz.CronTriggerFactoryBean;
import org.springframework.scheduling.quartz.JobDetailFactoryBean;

import java.text.ParseException;

@Configuration
public class ExpiredConsentRemoverConfig {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private static final String EXPIRED_CONSENT_REMOVER_JOB_DETAIL = "expiredConsentRemoverJobDetail";

    @Autowired
    private PhrConfig config;

    @Bean
    public ExpiredConsentRemover expiredConsentRemover(UserAccessManager userAccessManager,
                                                       EmailManager emailManager,
                                                       UserManager userManager,
                                                       ITemplateParamsFactory templateParamsFactory,
                                                       IAddressInfoProvider addressInfoProvider,
                                                       PatientConsentManager patientConsentManager) {
        return new ExpiredConsentRemover(userAccessManager, emailManager, userManager, templateParamsFactory, addressInfoProvider, patientConsentManager);
    }

    @Bean(name = EXPIRED_CONSENT_REMOVER_JOB_DETAIL)
    public JobDetailFactoryBean expiredConsentRemoverJobDetail() {
        JobDetailFactoryBean jobDetailFactory = new JobDetailFactoryBean();
        jobDetailFactory.setJobClass(ExpiredConsentRemoverJob.class);
        jobDetailFactory.setDurability(true);
        return jobDetailFactory;
    }


    @Bean
    public CronTriggerFactoryBean expiredConsentRemoverCronTrigger(@Qualifier(EXPIRED_CONSENT_REMOVER_JOB_DETAIL) JobDetail expiredConsentRemoverJobDetail)
            throws ParseException {
        CronTriggerFactoryBean ctfb = new CronTriggerFactoryBean();
        ctfb.setJobDetail(expiredConsentRemoverJobDetail);
        LOGGER.info("Setting cron expression to: [{}]", config.getExpiredConsentRemoverCron());
        ctfb.setCronExpression(config.getExpiredConsentRemoverCron());
        return ctfb;
    }



}
