package config.scedule;

import java.text.ParseException;

import org.quartz.JobDetail;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.scheduling.quartz.CronTriggerFactoryBean;
import org.springframework.scheduling.quartz.JobDetailFactoryBean;

import com.pkb.common.config.PhrConfig;
import com.pkb.service.integration.PollSciStoreNotificationJob;

@Configuration
public class PollForSciStoreNotificationsConfig {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final String POLL_FOR_SCI_STORE_NOTIFICATIONS_JOB_DETAIL = "pollForSciStoreNotificationsJobDetail";

    public static final String SCISTORE_POLL_JOB_DETAIL_KEY = "SciStorePollJobDetailKey";
    public static final String SCISTORE_GROUP = "SciStoreGroup";

    @Autowired
    private PhrConfig config;

    @Bean(name = POLL_FOR_SCI_STORE_NOTIFICATIONS_JOB_DETAIL)
    public JobDetailFactoryBean pollSciStoreJobDetail() {
        JobDetailFactoryBean jobDetailFactory = new JobDetailFactoryBean();
        jobDetailFactory.setJobClass(PollSciStoreNotificationJob.class);
        jobDetailFactory.setDurability(true);
        jobDetailFactory.setDescription("Invokes SciStore poll for notifications job");
        jobDetailFactory.setGroup(SCISTORE_GROUP);
        jobDetailFactory.setName(SCISTORE_POLL_JOB_DETAIL_KEY);
        return jobDetailFactory;
    }

    @Bean
    public CronTriggerFactoryBean pollForSciStoreNotificationsTrigger(@Qualifier(POLL_FOR_SCI_STORE_NOTIFICATIONS_JOB_DETAIL) JobDetail pollForSciStoreNotificationsJobDetail)
            throws ParseException {
        CronTriggerFactoryBean ctfb = new CronTriggerFactoryBean();
        ctfb.setJobDetail(pollForSciStoreNotificationsJobDetail);
        LOGGER.info("Setting cron expression to: [{}]", config.getSciStoreNotificationCron());
        ctfb.setCronExpression(config.getSciStoreNotificationCron());
        return ctfb;
    }

}
