package config.scedule;

import org.quartz.CronTrigger;
import org.quartz.JobDetail;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.scheduling.quartz.CronTriggerFactoryBean;
import org.springframework.scheduling.quartz.JobDetailFactoryBean;

import com.pkb.common.config.PhrConfig;
import com.pkb.service.dataupload.emis.esv2.EmisJob;

@Configuration
public class EmisSchedulingConfig {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    private PhrConfig config;

    public static final String EMIS_JOB_DETAIL_KEY = "EmisJobDetailKey";
    public static final String EMIS_GROUP = "emisGroup";
    private static final String EMIS_JOB_DETAIL = "emisJobDetail";

    @Bean(name = EMIS_JOB_DETAIL)
    public JobDetailFactoryBean emisJobDetail() {
        JobDetailFactoryBean jobDetailFactory = new JobDetailFactoryBean();
        jobDetailFactory.setJobClass(EmisJob.class);
        jobDetailFactory.setDescription("Invokes EMIS integration job");
        jobDetailFactory.setDurability(true);
        jobDetailFactory.setGroup(EMIS_GROUP);
        jobDetailFactory.setName(EMIS_JOB_DETAIL_KEY);
        return jobDetailFactory;
    }

    @Bean
    public CronTriggerFactoryBean emisTrigger(@Qualifier(EMIS_JOB_DETAIL) JobDetail jobDetail) {
        CronTriggerFactoryBean ctfb = new CronTriggerFactoryBean();
        ctfb.setJobDetail(jobDetail);
        ctfb.setMisfireInstruction(CronTrigger.MISFIRE_INSTRUCTION_DO_NOTHING);
        LOGGER.info("Setting cron expression to: [{}]", config.getEmisEsJobCron());
        ctfb.setCronExpression(config.getEmisEsJobCron());
        return ctfb;
    }
}
