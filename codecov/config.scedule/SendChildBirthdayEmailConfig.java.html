package config.scedule;

import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.service.emailmessage.IAddressInfoProvider;
import com.pkb.service.emailmessage.ITemplateParamsFactory;
import com.pkb.service.emailmessage.impl.ChildBirthdayEmailSender;
import com.pkb.service.emailmessage.impl.ChildBirthdayEmailSenderJob;
import com.pkb.service.emailmessage.impl.EmailManager;
import com.pkb.service.reminders.ReminderManager;
import org.quartz.JobDetail;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.scheduling.quartz.CronTriggerFactoryBean;
import org.springframework.scheduling.quartz.JobDetailFactoryBean;

import java.text.ParseException;

@Configuration
public class SendChildBirthdayEmailConfig {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final String SEND_CHILD_BIRTHDAY_EMAIL_JOB_DETAIL = "sendChildBirthdayEmailJobDetail";

    @Autowired
    private PhrConfig config;

    @Bean
    public ChildBirthdayEmailSender childBirthdayEmailSender(
            ReminderManager reminderManager,
            EmailManager emailManager,
            IAddressInfoProvider formatter,
            ITemplateParamsFactory emailTemplateParamsFactory,
            DateTimeService dateTimeService) {
        return new ChildBirthdayEmailSender(reminderManager, emailManager, formatter, emailTemplateParamsFactory, dateTimeService);
    }

    @Bean(name = SEND_CHILD_BIRTHDAY_EMAIL_JOB_DETAIL)
    public JobDetailFactoryBean childBirthJobDetail() {
        JobDetailFactoryBean jobDetailFactory = new JobDetailFactoryBean();
        jobDetailFactory.setJobClass(ChildBirthdayEmailSenderJob.class);
        jobDetailFactory.setDurability(true);
        return jobDetailFactory;
    }

    @Bean
    public CronTriggerFactoryBean sendChildBirthdayEmailCronTrigger(@Qualifier(SEND_CHILD_BIRTHDAY_EMAIL_JOB_DETAIL) JobDetail sendChildBirthdayEmailJobDetail)
            throws ParseException {
        CronTriggerFactoryBean ctfb = new CronTriggerFactoryBean();
        ctfb.setJobDetail(sendChildBirthdayEmailJobDetail);
        LOGGER.info("Setting cron expression to: [{}]", config.getChildBirthNotificationCron());
        ctfb.setCronExpression(config.getChildBirthNotificationCron());
        return ctfb;
    }
}
