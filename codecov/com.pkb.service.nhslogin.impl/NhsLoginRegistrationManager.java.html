package com.pkb.service.nhslogin.impl;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.IdentityVerification;
import com.pkb.datamodel.ImmutableIdentityVerification;
import com.pkb.datamodel.ImmutableRegistrationEventNew;
import com.pkb.domain.IdentityVerificationService;
import com.pkb.domain.RegistrationEventService;
import com.pkb.entities.enums.PhotoIdType;
import com.pkb.entities.enums.RegistrationMethod;
import com.pkb.entities.enums.Route;
import com.pkb.entities.enums.SponsorshipStatus;
import com.pkb.entities.enums.VerificationStatus;
import com.pkb.institute.entity.Team;
import com.pkb.kms.client.core.Kms;
import com.pkb.kms.shared.representation.KmsError;
import com.pkb.kms.shared.representation.LoginRequest;
import com.pkb.kms.shared.representation.NhsLoginRequest;
import com.pkb.model.PKBPersonDTO;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import io.prometheus.client.Counter;
import io.vavr.control.Either;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;

import java.util.Base64;
import java.util.Optional;
import java.util.UUID;

public class NhsLoginRegistrationManager extends TransactionManager {

    private final Kms kmsClient;
    private final PatientConsentManager patientConsentManager;
    private final RegistrationEventService registrationEventService;
    private final TeamUserManager teamUserManager;
    private final UserManager userManager;
    private final IdentityVerificationService identityVerificationService;

    private static final Counter PATIENT_TEAM_AFFILIATION = Counter.build()
            .name("pkb_nhs_login_registrations_team_affiliation")
            .labelNames("hasTeamAffiliation")
            .help("Number of NHS Login patient full registrations with or without team affiliation")
            .register();

    public NhsLoginRegistrationManager(
            PhrConfig config,
            TolvenBeanFactory beanFactory,
            DateTimeService dateTimeService,
            UUIDProvider uuidProvider,
            Kms kmsClient,
            PatientConsentManager patientConsentManager,
            RegistrationEventService registrationEventService,
            TeamUserManager teamUserManager,
            UserManager userManager,
            IdentityVerificationService identityVerificationService) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.kmsClient = kmsClient;
        this.patientConsentManager = patientConsentManager;
        this.registrationEventService = registrationEventService;
        this.teamUserManager = teamUserManager;
        this.userManager = userManager;
        this.identityVerificationService = identityVerificationService;
    }

    @Immutable
    @Style(stagedBuilder = true)
    public interface NhsLoginPatientMetadata {

        PKBPersonDTO patient();

        String password();

        String securityQuestion();

        String securityQuestionAnswer();

        Optional<Team> team();

        String nhsLoginCredentials();

        Route route();

    }
    /**
     * Full patient registration using information from NHS Login:
     *   - patient and credentials creation
     *   - patient authentication to provide information for subsequent requests
     *   - password backup creation based on provided security question (and answer)
     *   - account creation/activation and, if team affiliaion exists, creation of patient consent
     *   - creation of registration event
     *   - addition of NHS Login credentials
     *   - addition of external credential recovery (for password reset) if no team affiliation
     *
     * @param context
     *          {@link EHRRequestContext}
     * @param metadata
     *          {@link NhsLoginPatientMetadata}
     */
    public void registerPatient(EHRRequestContext context, NhsLoginPatientMetadata metadata) {

        PKBPerson registeredPerson = transactional(() -> {
            PKBPerson person = userManager.registerUser(context, metadata.patient());
            userManager.backUpPassword(person.getId(), metadata.password(), metadata.securityQuestion(), metadata.securityQuestionAnswer());
            var registeringContext = context.withRegisteringUser(person.getId());
            metadata.team().ifPresentOrElse(team -> {
                userManager.activatePerson(registeringContext, person, team.getId(), metadata.password());
                grantAccessToTeam(context, metadata.route(), person, team);
                identityVerificationService.save(buildIdentityVerificationWithTeamAffiliation(person, team));
            }, () -> {
                userManager.activatePerson(registeringContext, person, null, metadata.password());
                identityVerificationService.save(buildIdentityVerification(person));
            });

            PATIENT_TEAM_AFFILIATION.labels(Boolean.toString(metadata.team().isPresent())).inc();

            registrationEventService.create(ImmutableRegistrationEventNew.builder()
                    .registrationMethod(RegistrationMethod.NHS_LOGIN)
                    .registeredOn(dateTimeService.now())
                    .registeredOnConfirmed(true)
                    .personId(person.getId())
                    .build());
            return person;
        });

        String requestId = context.getCorrelationId().toString();
        kmsClient.addNhsLoginCredentials(requestId, new NhsLoginRequest(registeredPerson.getPublicId(), Base64.getDecoder().decode(metadata.nhsLoginCredentials())));
        if (metadata.team().isEmpty()) {
            addExternalCredentialRecovery(requestId, registeredPerson.getPublicId());
        }
    }

    /**
     * Grants default consent to team and sets sponsorship status to active.
     *
     * @param context {@link EHRRequestContext}
     * @param route the route the request is coming from
     * @param person the patient giving access to the team
     * @param team the team the patient is giving access to
     */
    public void grantAccessToTeam(EHRRequestContext context, Route route, PKBPerson person, Team team) {
        patientConsentManager.grantDefaultConsentToTeam(context, person, team, route);
        teamUserManager.updateStatus(team.getId(), person.getId(), SponsorshipStatus.ACTIVE);
    }

    /**
     * Add external credential recovery for a non-affiliated patient
     * @param password
     * @param personPublicId
     *              {@link UUID} user's public id
     */
    public void addExternalCredentialRecovery(char[] password, UUID personPublicId) {
        String requestId = UUID.randomUUID().toString();
        addExternalCredentialRecovery(requestId, personPublicId)
                .mapLeft($ -> kmsClient.login(requestId, new LoginRequest(personPublicId, password, false))
                        .peek($$ -> addExternalCredentialRecovery(requestId, personPublicId)));
    }

    private Either<KmsError, Void> addExternalCredentialRecovery(String requestId, UUID personPublicId) {
        return kmsClient.addExternalCredentialRecovery(requestId, personPublicId);
    }

    @VisibleForTesting
    protected IdentityVerification buildIdentityVerification(PKBPerson person) {
        return ImmutableIdentityVerification.builder()
                .signedConsentForm(false)
                .verificationDate(dateTimeService.now())
                .status(VerificationStatus.VERIFIED)
                .photoIdType(PhotoIdType.NHS_LOGIN)
                .photoIdNumber("Vouched for by NHS Login")
                .personId(person.getId())
                .build();
    }

    @VisibleForTesting
    protected IdentityVerification buildIdentityVerificationWithTeamAffiliation(PKBPerson person, Team team) {
        return ImmutableIdentityVerification.copyOf(buildIdentityVerification(person))
                .withVerifyingTeamId(team.getId());
    }
}
