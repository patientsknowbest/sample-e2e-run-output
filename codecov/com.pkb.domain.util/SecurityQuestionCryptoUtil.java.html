package com.pkb.domain.util;

import com.pkb.crypto.CryptoHelper;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import javax.crypto.SecretKey;

import static com.pkb.crypto.CryptoConstants.PBE_SHORT_NAME;

public class SecurityQuestionCryptoUtil {

    private SecurityQuestionCryptoUtil() {
    }

    @NotNull
    public static byte[] encryptPassword(char[] password, byte[] salt, int iterationCount, String securityQuestionAnswer) {
        try {
            SecretKey secretKey = CryptoHelper.secretKey(securityQuestionAnswer.trim().toLowerCase().toCharArray(), PBE_SHORT_NAME);
            return CryptoHelper.encryptPassword(password, salt, iterationCount, secretKey);
        } catch (Exception e) {
            throw new RuntimeException("Could not convert and encrypt password from char[] to bytes", e);
        }
    }

    @NotNull
    public static Either<Exception, char[]> decryptPassword(byte[] encryptedPassword, int iterationCount, byte[] salt, char[] securityQuestionAnswer) {
        try {
            SecretKey secretKey = CryptoHelper.secretKey(securityQuestionAnswer, PBE_SHORT_NAME);
            return Either.right(CryptoHelper.decryptPassword(encryptedPassword, salt, iterationCount, secretKey));
        } catch (Exception e) {
            return Either.left(e);
        }
    }
}
