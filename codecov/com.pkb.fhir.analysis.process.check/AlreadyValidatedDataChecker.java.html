package com.pkb.fhir.analysis.process.check;

import com.pkb.fhir.analysis.persistence.targetdb.TransformedEhrDataRepository;

import java.util.Arrays;
import java.util.List;

import static com.pkb.fhir.common.enums.TransformedEhrDataStatus.VALIDATED;

public class AlreadyValidatedDataChecker implements AlreadyProcessedDataChecker {
    private TransformedEhrDataRepository transformedEhrDataRepository;

    public AlreadyValidatedDataChecker(TransformedEhrDataRepository transformedEhrDataRepository) {
        this.transformedEhrDataRepository = transformedEhrDataRepository;
    }

    @Override
    public List<Long> findExistingSucceededData(List<Long> idsToProcess) {
        return transformedEhrDataRepository.findAllIdsByIdWithStatusIn(idsToProcess,
                Arrays.asList(VALIDATED));
    }

}
