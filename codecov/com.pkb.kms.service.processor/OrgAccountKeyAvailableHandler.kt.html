package com.pkb.kms.service.processor

import com.pkb.kms.service.cache.OrgAccountKeyCacheService
import com.pkb.kms.service.routing.KeyAvailableRouteBuilder
import com.pkb.kms.shared.structures.pubsub.KeyAvailable
import com.pkb.kms.shared.structures.pubsub.accountKeyAvailable
import com.pkb.kms.shared.structures.pubsub.orgPrivateId
import io.micrometer.core.annotation.Timed
import org.apache.camel.Consume
import org.apache.camel.Produce
import org.apache.camel.ProducerTemplate
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

@Service
class OrgAccountKeyAvailableHandler(
        private val orgAccountKeyCacheService: OrgAccountKeyCacheService,
        @Produce(KeyAvailableRouteBuilder.OUTBOUND_ACCOUNT_KEY_AVAILABLE) private val accountKeyAvailableProducer: ProducerTemplate,
) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    @Timed("org_key_available_handler", histogram = true, percentiles = [0.5, 0.75, 0.9, 0.95, 0.99, 0.999])
    @Consume(KeyAvailableRouteBuilder.INBOUND_ORG_KEY_AVAILABLE)
    fun process(keyAvailable: KeyAvailable) {
        log.debug("got a org key available pubsub message {}", keyAvailable)
        val orgPrivateId = keyAvailable.orgPrivateId()
        orgAccountKeyCacheService.triggerForAllOrgAccessibleAccounts(orgPrivateId) {
            accountKeyAvailableProducer.sendBody(accountKeyAvailable(it))
        }
    }

}
