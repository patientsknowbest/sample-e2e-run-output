package com.pkb.util.jsp;

import com.opensymphony.xwork2.util.ValueStack;
import com.pkb.action.TimelineDisplayGateway;
import com.pkb.util.metrics.SpaMetrics;
import org.apache.struts2.components.Component;
import org.apache.struts2.views.jsp.ComponentTagSupport;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

public class AngularTag extends ComponentTagSupport implements BeanAware {

    private static final long serialVersionUID = 1L;

    private String module;

    @Override
    public Component getBean(ValueStack stack, HttpServletRequest req, HttpServletResponse res) {
        return new Angular(stack);
    }

    @Override
    protected void populateParams() {
        super.populateParams();

        Angular tag = (Angular) component;
        tag.setModule(module);
        tag.setAngularGateWay(findBean(TimelineDisplayGateway.class));
        tag.setMetrics(findBean(SpaMetrics.class));

    }

    public void setModule(String module) {
        this.module = module;
    }
}
