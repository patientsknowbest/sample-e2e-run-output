package com.pkb.util.jsp;

import com.pkb.common.config.PhrConfig;
import com.pkb.util.security.HtmlMessageFilter;
import org.apache.commons.lang3.StringUtils;
import org.owasp.encoder.Encode;

import java.util.regex.Pattern;

public class JstlFunctions {

    private static HtmlMessageFilter MESSAGE_FILTER;

    private static final Pattern NEW_LINES = Pattern.compile("(?:\r\n|\n|\r)");

    public static String escapeHtml(String input) {
        return Encode.forHtml(input);
    }

    public static String escapeJavaScript(String input) {
        return Encode.forJavaScript(input);
    }

    public static  void initMessageFilter(PhrConfig config) {
        MESSAGE_FILTER = new HtmlMessageFilter(config.getBaseURL());
    }

    public static String filterDiaryEntry(Boolean isHtmlAllowed, String input) {
        if (isHtmlAllowed) {
            if (MESSAGE_FILTER != null) {
                return MESSAGE_FILTER.filter(input);
            }
            return input;
        } else {
            if (StringUtils.isEmpty(input)) {
                return "<p></p>";
            }
            String htmlContent = "<p>" + escapeHtml(input) + "</p>";
            return NEW_LINES.matcher(htmlContent).replaceAll("</p><p>");
        }
    }

    public static String escapeMessageForHtml(String input) {
        String htmlContent = escapeHtml(input);
        return NEW_LINES.matcher(htmlContent).replaceAll("<br/>");
    }
}
