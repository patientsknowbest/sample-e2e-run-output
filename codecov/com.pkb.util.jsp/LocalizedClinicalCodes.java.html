package com.pkb.util.jsp;

import com.opensymphony.xwork2.TextProvider;
import com.opensymphony.xwork2.TextProviderFactory;
import com.opensymphony.xwork2.inject.Inject;
import com.opensymphony.xwork2.util.ValueStack;
import com.pkb.util.MedicationDTO;
import org.apache.struts2.components.Component;
import org.apache.struts2.views.annotations.StrutsTag;
import org.apache.struts2.views.annotations.StrutsTagAttribute;
import org.owasp.encoder.Encode;
import org.slf4j.Logger;

import java.io.IOException;
import java.io.Writer;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

import static java.lang.invoke.MethodHandles.lookup;
import static org.apache.commons.lang3.StringUtils.isNotBlank;
import static org.slf4j.LoggerFactory.getLogger;

@StrutsTag(name = "localizedClinicalCodes", tldBodyContent = "empty", tldTagClass = "com.pkb.util.jsp.LocalizedClinicalCodesTag", description = "")
public class LocalizedClinicalCodes extends Component {
    private static final Pattern MEDICATION_CODE_PATTERN = Pattern.compile(MedicationDTO.MEDICATION_CODE, Pattern.LITERAL);
    private static final Pattern DOSE_UNIT_CODE_PATTERN = Pattern.compile(MedicationDTO.DOSE_UNIT_CODE, Pattern.LITERAL);

    private static final Logger LOGGER = getLogger(lookup().lookupClass());

    private TextProvider textProvider;
    private String clinicalCodes;

    public LocalizedClinicalCodes(ValueStack stack) {
        super(stack);
    }

    @StrutsTagAttribute(description = "Codes to be localized")
    public void setClinicalCodes(String clinicalCodes) {
        this.clinicalCodes = clinicalCodes;
    }

    @Inject
    public void setTextProvider(TextProviderFactory textProviderFactory) {
        this.textProvider = textProviderFactory.createInstance(LocalizedClinicalCodes.class);
    }

    @Override
    public boolean start(Writer writer) {
        boolean result = super.start(writer);

        try {
            if (isNotBlank(clinicalCodes)) {
                String localized = DOSE_UNIT_CODE_PATTERN
                        .matcher(MEDICATION_CODE_PATTERN.matcher(clinicalCodes)
                                .replaceAll(Matcher.quoteReplacement(textProvider.getText("manageMedications.txt.medicationCode"))))
                        .replaceAll(Matcher.quoteReplacement(textProvider.getText("manageMedications.txt.doseUnitCode")));

                writer.write("<span class=\"span_clinicalCodes\">" +
                        "<br/>" +
                        Encode.forHtml(localized) +
                        "</span>");
            }
        } catch (IOException e) {
            LOGGER.error("Failed to write localized clinical codes.",e);
        }

        return result;
    }
}
