package com.pkb.util.jsp;

import com.opensymphony.xwork2.util.ValueStack;
import com.pkb.interceptor.CSRFTokenHelper;
import org.apache.struts2.components.UIBean;
import org.apache.struts2.views.annotations.StrutsTag;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.util.Map;

@StrutsTag(name = "csrftoken", tldTagClass = "com.pkb.util.jsp.CSRFTokenTag", description = "CSRF token")
public class CSRFToken extends UIBean {

    public static final String TEMPLATE_NAME = "csrftoken";
    private static final String TEMPLATE_TOKEN_VALUE_PARAMETER = "csrftoken";
    private static final String TEMPLATE_TOKEN_NAME_PARAMETER = "csrfname";

    private CSRFTokenHelper tokenHelper;

    public CSRFToken(ValueStack stack, HttpServletRequest request, HttpServletResponse response, CSRFTokenHelper tokenHelper) {
        super(stack, request, response);
        this.tokenHelper = tokenHelper;
    }

    @Override
    protected String getDefaultTemplate() {
        return TEMPLATE_NAME;
    }

    @Override
    protected void evaluateExtraParams() {
        Map parameters = getParameters();
        parameters.computeIfAbsent(TEMPLATE_TOKEN_VALUE_PARAMETER, t -> buildToken());
        parameters.putIfAbsent(TEMPLATE_TOKEN_NAME_PARAMETER, CSRFTokenHelper.TOKEN_FIELD);
    }

    private String buildToken() {
        return (String) stack.getContext().computeIfAbsent(CSRFTokenHelper.TOKEN_FIELD, t -> tokenHelper.setTokenIfAbsent());
    }
}
