package com.pkb.util.jsp;

import com.opensymphony.xwork2.util.ValueStack;
import com.pkb.action.TimelineDisplayGateway;
import com.pkb.util.metrics.SpaMetrics;
import org.apache.struts2.components.Component;
import org.apache.struts2.views.annotations.StrutsTag;
import org.apache.struts2.views.annotations.StrutsTagAttribute;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.IOException;
import java.io.Writer;

@StrutsTag(name = "angular", tldBodyContent = "empty", tldTagClass = "com.pkb.util.jsp.AngularTag", description = "Loads an angular module to the page")
public class Angular extends Component {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private String module;

    private TimelineDisplayGateway angularGateWay;

    private SpaMetrics metrics;

    public Angular(ValueStack stack) {
        super(stack);
    }

    @StrutsTagAttribute(description = "The module to be loaded")
    public void setModule(String module) {
        this.module = module;
    }

    void setAngularGateWay(TimelineDisplayGateway angularGateWay) {
        this.angularGateWay = angularGateWay;
    }

    void setMetrics(SpaMetrics metrics) {
        this.metrics = metrics;
    }

    @Override
    public boolean start(Writer writer) {
        boolean result = super.start(writer);

        try {
            writer.write(angularGateWay.obtainHTMLFragment(this.module));
            if (obtainsPatientBanner()) {
                metrics.patientBannerCalled();
            }
        } catch (IOException e) {
            LOGGER.error("Could not print out value", e);
        }

        return result;
    }

    private boolean obtainsPatientBanner() {
        return "patient-summary".equals(this.module) || "default/".equals(this.module);
    }
}
