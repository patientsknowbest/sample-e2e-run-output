/**
 * FORMERLY USED FOR CAPTCHA:
 * This class is added to accommodate CAPTCHA related changes
 * Now CAPTCHA should be shown if user login attempts failed for consecutive three times
 * This is specific to session, if user closes the browser and opens it again, CAPTCHA will not be shown.
 * Currently used to set some session variables and cookies after authentication.
 * <p>
 * ALSO: registration pathway... if "register_icode" is set in the session, this will call
 * assignInstituteAdmin() as long as the userTYep isn't SUPERADMIN
 */
public class PreAuthenticationSuccessHandler extends SavedRequestAwareAuthenticationSuccessHandler {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    private static final String NUMBER_OF_ATTEMPTS = "numberOfAttempts";

    private static final Set<Route> RESTRICTED_SSO_LOGIN_ROUTES = Sets.immutableEnumSet(Route.WEBAPP_EMIS_PORTAL);

    private static final String INACTIVE_ACCOUNT_ERROR_PAGE = "/nhsError.action?error=out.of.area";

    public static final String REDIRECT_URI_PARAMETER = "redirect_uri";

    private final LoggedInUserUtil loggedInUserUtil;

    @Autowired
    private PhrConfig config;

    public PreAuthenticationSuccessHandler(LoggedInUserUtil loggedInUserUtil) {
        this.loggedInUserUtil = loggedInUserUtil;
        this.setTargetUrlParameter(REDIRECT_URI_PARAMETER);
    }

    @Override
    public void onAuthenticationSuccess(HttpServletRequest request, HttpServletResponse response, Authentication authentication) throws ServletException, IOException {

        LOGGER.trace("Enter onSuccessfulAuthentication : resetting number of attempts in session");
        HttpSession session = request.getSession();
        if (authentication.getDetails() instanceof BaseAuthenticationDetails) {
            Route route = ((BaseAuthenticationDetails) authentication.getDetails()).getRoute();
            session.setAttribute(PKBConstants.ROUTE, route);
        }
        session.removeAttribute(NUMBER_OF_ATTEMPTS);

        Optional<String> maybeRedirectPath = loggedInUserUtil
                .getNhsLoginAuthenticationDetailsFromPkbAuthenticationDetails(Optional.ofNullable(authentication.getDetails()))
                .flatMap(NhsLoginInfo::getRedirectPath);
        if (maybeRedirectPath.isPresent()) {
            String redirectPath = maybeRedirectPath.get();
            response.sendRedirect(URLDecoder.decode(redirectPath, UTF_8));
        }

        // formerly in RedirectHomeAction.java
        try {
            TeamUserManager teamUserManager = (TeamUserManager) PKBApplicationContext.getBean("teamUserManager");
            UserManager userManager = (UserManager) PKBApplicationContext.getBean("userManager");

            PersonAuthInfo user = (PersonAuthInfo) authentication.getPrincipal();

            if (user != null) {
                long userId = user.getId();
                PKBPerson person = userManager.getPKBPerson(userId);
                if (user.isPatient() && person.getDefaultAccountId() == null) {
                    getRedirectStrategy().sendRedirect(request, response, INACTIVE_ACCOUNT_ERROR_PAGE);
                    return;
                }


                session.setAttribute(I18nInterceptor.DEFAULT_SESSION_ATTRIBUTE, new Locale(person.getLanguageCode())); // WW_TRANS_I18N_LOCALE
                // set/update the PKB locale cookie
                CookieUtil.setCookie(request, response, UserLoginInterceptor.PKB_LOCALE_COOKIE,
                        person.getLanguageCode(), CookieUtil.DURATION_30_DAYS);

                Route route = (Route) request.getSession().getAttribute(PKBConstants.ROUTE);
                if (RESTRICTED_SSO_LOGIN_ROUTES.contains(route)) {
                    CookieUtil.setCookie(request, response, "pkbSSO", String.valueOf(true), CookieUtil.DURATION_SESSION, true);
                }
            } else {
                LOGGER.error("null auth user");
            }
        } catch (Exception e) {
            LOGGER.error("Error while getting the institute user or institutes", e);
        }

        request.getSession().setMaxInactiveInterval(this.config.getMaxInactiveIntervalSeconds());

        LOGGER.trace("Exit onSuccessfulAuthentication : ");

        super.onAuthenticationSuccess(request, response, authentication);
    }

}
