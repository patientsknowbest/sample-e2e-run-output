package com.pkb.servlets.filter;

import com.pkb.app.entity.PersonAuthInfo;
import com.pkb.dto.PersonSubjectCredentialsDTO;
import org.springframework.security.authentication.AbstractAuthenticationToken;
import org.springframework.security.core.GrantedAuthority;

import java.util.Collection;

public class NhsLoginAuthenticationToken extends AbstractAuthenticationToken {

    private final PersonAuthInfo user;

    public NhsLoginAuthenticationToken(PersonAuthInfo user, PersonSubjectCredentialsDTO credentials, Collection<? extends GrantedAuthority> authorities) {
        super(authorities);
        this.user = user;
        super.setAuthenticated(true);
        super.setDetails(credentials);
    }

    @Override
    public Object getCredentials() {
        return ((PersonSubjectCredentialsDTO) getDetails()).getUserPrivateKey();
    }

    @Override
    public Object getPrincipal() {
        return user;
    }
}
