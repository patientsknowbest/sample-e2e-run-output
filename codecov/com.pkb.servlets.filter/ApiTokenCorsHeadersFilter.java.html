package com.pkb.servlets.filter;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.servlet.Filter;
import javax.servlet.FilterChain;
import javax.servlet.FilterConfig;
import javax.servlet.ServletException;
import javax.servlet.ServletRequest;
import javax.servlet.ServletResponse;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;

public class ApiTokenCorsHeadersFilter implements Filter {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Override
    public void init(FilterConfig filterConfig) throws ServletException {
    }

    /**
     * The steroids.js framework for app development requires CORS Headers for cross domain POST requests.
     * This filter makes sure that only the apiToken.action response sends CORS headers in order to allow
     * the app to get the access and refresh token.
     */
    @Override
    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
            throws IOException, ServletException {
        LOGGER.trace("Enter doFilter");
        HttpServletRequest httpRequest = (HttpServletRequest) request;
        HttpServletResponse httpResponse = (HttpServletResponse) response;

        String URI = httpRequest.getRequestURI();
        if ((URI != null) && URI.endsWith("apiToken.action")) {
            httpResponse.setHeader("Access-Control-Allow-Origin", "*");
            httpResponse.setHeader("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE");
            httpResponse.setHeader("Access-Control-Allow-Headers", "Content-Type");
        }
        chain.doFilter(request, response);

        LOGGER.trace("Exit doFilter");
    }

    @Override
    public void destroy() {
        //		log.info("Enter destroy");
        //		log.info("Exit destroy");
    }
}
