package com.pkb.servlets.filter;

import javax.ws.rs.container.ContainerRequestContext;
import javax.ws.rs.container.ContainerResponseContext;
import javax.ws.rs.container.ContainerResponseFilter;
import javax.ws.rs.core.MultivaluedMap;
import javax.ws.rs.ext.Provider;
import java.io.IOException;

/**
 * Our REST API is based on Jersey.
 * We need to add headers to permit cross-domain API calls via AJAX; this is how to insert that.
 * <p/>
 * This class is connected to Jersey via servlet init-param -- see enunciate.xml, Jersey module.
 *
 * This functionality has now been ported to Jersey 2 with no need to declare the filter in enunciate
 *
 * @see com.pkb.util.RestApplication for registration
 *
 * @author rob
 * @author Viktor Orekoya <viktor@patientsknowbest.com>
 * @see http://blog.usul.org/cors-compliant-rest-api-with-jersey-and-containerresponsefilter/
 */
@Provider
public class ApiCorsHeadersFilter implements ContainerResponseFilter {

    @Override
    public void filter(ContainerRequestContext containerRequestContext, ContainerResponseContext containerResponseContext)
            throws IOException {

        MultivaluedMap<String, Object> headers = containerResponseContext.getHeaders();

        headers.add("Access-Control-Allow-Origin", "*");
        headers.add("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE");
        headers.add("Access-Control-Allow-Headers", "Authorization, Content-Type"); // just these, for now

    }
}
