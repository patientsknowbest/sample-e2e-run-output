package com.pkb.servlets.filter;

import org.apache.commons.lang3.StringUtils;
import org.springframework.security.web.DefaultRedirectStrategy;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;

public class TeamAwareRedirectStrategy extends DefaultRedirectStrategy {

    @Override
    public void sendRedirect(HttpServletRequest request, HttpServletResponse response, String url) throws IOException {
        String teamCode = request.getParameter("team"); // Comes from login.jsp - may be empty but never null
        if (StringUtils.isNotBlank(teamCode)) {
            // default URL ends with a param already...
            super.sendRedirect(request, response, url + "&team=" + teamCode);
        } else {
            super.sendRedirect(request, response, url);
        }

    }
}
