package com.pkb.service.dataupload.emis.esv2;

import com.google.common.collect.ImmutableList;
import com.pkb.common.config.PhrConfig;
import com.pkb.service.dataupload.emis.esv2.converters.EmisLsEntryToEmisFileConverter;
import com.pkb.service.dataupload.emis.esv2.model.EmisFile;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileSet;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileSftpSet;
import com.pkb.service.dataupload.emis.esv2.model.ImmutableEmisFileSftpSet;
import org.jetbrains.annotations.NotNull;

import java.nio.file.Path;
import java.time.LocalDate;
import java.util.Collection;
import java.util.Comparator;
import java.util.List;
import java.util.Optional;

/**
 * This class inherits EmisBatchBuilder class and is to ensure proper validations against Emis files and file structure applies in verifying Emis data.
 */
public class EmisVerifyBatchBuilder extends AbstractEmisBatchBuilder{


    public EmisVerifyBatchBuilder(@NotNull EmisLsEntryToEmisFileConverter emisLsEntryToEmisFileConverter, PhrConfig config, EmisEsDatestampSkipService emisEsDatestampSkipService) {
        super(emisLsEntryToEmisFileConverter, config, emisEsDatestampSkipService);
    }

    @NotNull
    public List<EmisFileSftpSet> buildEmisFileSftpSets(@NotNull Collection<EmisLsEntry> remoteList, String sftpUniqueId, Path serverDownloadDir) {
        return remoteList.stream()
                .map(emisLsEntry ->  makeEmisFileSftpset(emisLsEntry, sftpUniqueId, serverDownloadDir))
                .flatMap(Optional::stream)
                .sorted(Comparator.comparing(efss -> efss.emisFileSet().dateStamp()))
                .collect(ImmutableList.toImmutableList());
    }

    private Optional<EmisFileSftpSet> makeEmisFileSftpset(EmisLsEntry dirEntry, String sftpUniqueId, Path serverDownloadDir) {
        Optional<EmisFileSet> efs = makeFileset(dirEntry, sftpUniqueId);
        return efs.map(emisFileSet -> ImmutableEmisFileSftpSet.of(emisFileSet, serverDownloadDir, sftpUniqueId));
    }

    /**
     * Spec says there should only be bulks OR deltas in a directory, sanity check that is the case
     */
    @Override
    protected void validateConsistentBulksOrDeltas(List<EmisFile> batchFiles, LocalDate datestamp) {
        batchFiles.stream()
                .reduce((emisFile, emisFile2) -> {
                    if (emisFile.isBulk() != emisFile2.isBulk()) {
                        throw new RuntimeException("Mismatch of bulk and deltas in " + datestamp + " directory");
                    }
                    return emisFile;
                });
    }

    @Override
    protected boolean nonRegularFileHandler(EmisLsEntry lsEntry, LocalDate datestamp) {
        if (!lsEntry.isReg()) {
            LOGGER.error("Unexpected non-regular file {} found in datestamped directory, skipping {}", lsEntry, datestamp);
            return false;
        }
        return true;
    }

    @Override
    protected AbstractEmisFileSetBuilder getEmisFileSetBuilder(Path filesetPath, LocalDate datestamp) {
        return new EmisVerifyFileSetBuilder(filesetPath, datestamp);
    }

    @Override
    protected void checkFileTimestamp(EmisFile emisFile, LocalDate datestamp) {
        if (!emisFile.timeStamp().toLocalDate().equals(datestamp)) {
            throw new RuntimeException("emis file timestamp mismatch with parent directory, file: " + emisFile);
        }
    }

}
