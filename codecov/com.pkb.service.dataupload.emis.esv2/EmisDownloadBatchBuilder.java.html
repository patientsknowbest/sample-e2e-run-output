package com.pkb.service.dataupload.emis.esv2;

import com.pkb.common.config.PhrConfig;
import com.pkb.service.dataupload.emis.esv2.converters.EmisLsEntryToEmisFileConverter;
import com.pkb.service.dataupload.emis.esv2.model.EmisFile;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileSet;
import io.vavr.control.Try;
import org.jetbrains.annotations.NotNull;

import java.nio.file.Path;
import java.time.LocalDate;
import java.util.List;
import java.util.Optional;

/**
 * This class inherits EmisBatchBuilder class and is to ensure uninterrupted download of Emis files by avoiding validations .
 */
public class EmisDownloadBatchBuilder extends AbstractEmisBatchBuilder{


    public EmisDownloadBatchBuilder(@NotNull EmisLsEntryToEmisFileConverter emisLsEntryToEmisFileConverter, PhrConfig config, EmisEsDatestampSkipService emisEsDatestampSkipService) {
        super(emisLsEntryToEmisFileConverter, config, emisEsDatestampSkipService);
    }

    @Override
    protected Optional<EmisFileSet> makeFileset(EmisLsEntry dirEntry, String sftpUniqueId) {
        if (!dirEntry.isDir()) {
            LOGGER.error("Found unexpected non-directory file, skipping {} ", dirEntry.getFilename());
            return Optional.empty();
        }
        Optional<LocalDate> datestampIfValid = getDatestampIfValid(dirEntry);

        return Try.of(() -> datestampIfValid.map(datestamp -> buildEmisFileset(dirEntry, datestamp)))
                .getOrElseGet(throwable -> {
                    throw new RuntimeException("unable to make fileset for " + dirEntry.getFilename(), throwable);
                });
    }

    @Override
    protected void validateConsistentBulksOrDeltas(List<EmisFile> batchFiles, LocalDate datestamp) {

    }

    @Override
    protected boolean nonRegularFileHandler(EmisLsEntry lsEntry, LocalDate datestamp) {
        if (!lsEntry.isReg()) {
            LOGGER.warn("Unexpected non-regular file {} found in datestamped {} directory, proceeding with download.", lsEntry, datestamp);
        }
        return true;
    }

    @Override
    protected AbstractEmisFileSetBuilder getEmisFileSetBuilder(Path filesetPath, LocalDate datestamp) {
        return new EmisDownloadFileSetBuilder(filesetPath, datestamp);
    }

    @Override
    protected void checkFileTimestamp(EmisFile emisFile, LocalDate datestamp) {
        if (!emisFile.timeStamp().toLocalDate().equals(datestamp)) {
            LOGGER.warn("Emis file {}, timestamp mismatch with parent directory, proceeding with download.", emisFile);
        }
    }


}
