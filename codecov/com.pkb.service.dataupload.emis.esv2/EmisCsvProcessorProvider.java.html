package com.pkb.service.dataupload.emis.esv2;

import com.google.common.collect.ImmutableMap;
import com.google.common.collect.Maps;
import com.pkb.service.dataupload.emis.esv2.batch.BatchDataProcessor;
import com.pkb.service.dataupload.emis.esv2.model.BaseEmisEsDomainModel;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileType;
import org.jetbrains.annotations.NotNull;

import java.util.EnumMap;
import java.util.Set;
import java.util.stream.Collectors;

import static java.util.function.Function.identity;

public class EmisCsvProcessorProvider {

    private final ImmutableMap<EmisFileType, BatchDataProcessor<BaseEmisEsDomainModel>> dataTypeProcessorMap;

    public EmisCsvProcessorProvider(@NotNull Set<BatchDataProcessor<BaseEmisEsDomainModel>> dataTypeProcessors) {
        this.dataTypeProcessorMap = Maps.immutableEnumMap(dataTypeProcessors.stream().filter(dtp -> dtp.canProcess(dtp.getSupportedType())).collect(Collectors.toMap(BatchDataProcessor::getSupportedType, identity(), (l, r) -> {
            throw new IllegalArgumentException("Duplicate keys " + l + "and " + r + ".");
        }, () -> new EnumMap<>(EmisFileType.class))));
    }

    public ImmutableMap<EmisFileType, BatchDataProcessor<BaseEmisEsDomainModel>> get() {
        return dataTypeProcessorMap;
    }

    public BatchDataProcessor<BaseEmisEsDomainModel> getProcessor(EmisFileType type) {
        return dataTypeProcessorMap.get(type);
    }

    public Class<BaseEmisEsDomainModel> getInterface(EmisFileType type) {
        return getProcessor(type).interfaceClass();
    }
}
