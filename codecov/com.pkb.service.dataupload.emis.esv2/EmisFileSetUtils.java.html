package com.pkb.service.dataupload.emis.esv2;

import com.pkb.service.dataupload.emis.esv2.ImmutableEmisInternalResult.Builder;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileSet;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileSftpSet;
import com.pkb.service.dataupload.emis.esv2.model.VerifiedEmisFileSet;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.time.LocalDate;
import java.util.ArrayList;
import java.util.Comparator;
import java.util.List;
import java.util.function.Function;
import java.util.stream.Collectors;

public class EmisFileSetUtils {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    /**
     * Filter out EMIS batches by {@link EmisFileSet#dateStamp()} which were already processed.
     *
     * @param emisBatch
     *            EMIS batches to filter
     * @param lastBatchProcessed
     *            The time stamp of last processed batch
     * @return {@link List} of {@link EmisFileSet}s which time stamp is after lastBatchProcessed, empty list otherwise.
     */
    public <T> @NotNull List<T> filterBatchesAlreadyProcessed(@NotNull List<T> emisBatch,
                                                                        @NotNull LocalDate lastBatchProcessed,
                                                                        Function<T, EmisFileSet> efsProvider) {
        List<T> newBatch = new ArrayList<>(emisBatch.size());
        List<LocalDate> oldBatch = new ArrayList<>(emisBatch.size());
        for (T batch : emisBatch) {
            var efss = efsProvider.apply(batch);
            if (efss.dateStamp().isAfter(lastBatchProcessed)) {
                newBatch.add(batch);
            } else {
                oldBatch.add(efss.dateStamp());
            }
        }

        if (!oldBatch.isEmpty()) {
            LOGGER.info("Skipping EMIS deltas because they were already processed: {}", oldBatch);
        }
        LOGGER.info("There are {} new EMIS deltas available.", newBatch.size());
        return newBatch;
    }

    /**
     * Returns all complete EMIS batches.
     *
     * @param emisBatch
     *            EMIS batches to filter
     * @return {@link List} of complete {@link EmisFileSet}s, empty list otherwise.
     */
    public <T> @NotNull EmisInternalResult<List<T>> getAllBatches(@NotNull List<T> emisBatch, Function<T, EmisFileSet> efsProvider) {
        Builder<List<T>> result = ImmutableEmisInternalResult.builder();
        OrganizedBatches<T> ob = getOrganizedBatches(emisBatch, efsProvider);

        List<T> allBatches = new ArrayList<>();

        allBatches.addAll(ob.completeBatches);
        allBatches.addAll(ob.skippedBatches);

        if (ob.incompleteBatches.isEmpty()) {
            result.stepResult(EmisStepResult.FINISHED);
        } else {
            if (ob.lastIncomplete) {
                result.stepResult(EmisStepResult.FINISHED);
                LOGGER.info("Skip last batch {} as it is not complete.", efsProvider.apply(ob.incompleteBatches.get(0)).dateStamp());
            }else{
                allBatches.addAll(ob.incompleteBatches);
                result.stepResult(EmisStepResult.FINISHED_WITH_ERRORS);
                    LOGGER.warn("There are {} incomplete batches, manual correction needed!",
                            toTimestampList(ob.incompleteBatches, efsProvider));
            }

        }

        allBatches.sort(Comparator.comparingInt(emisBatch::indexOf));

        result.content(allBatches);
        return result.build();
    }

    /**
     * Returns complete EMIS batches up to the first error, everything after that is skipped.
     * 
     * @param emisBatch
     *            EMIS batches to filter
     * @return {@link List} of complete {@link EmisFileSet}s, empty list otherwise.
     */
    public <T> @NotNull EmisInternalResult<List<T>> getBatchesUpToTheFirstNotCompleteBatch(@NotNull List<T> emisBatch, Function<T, EmisFileSet> efsProvider) {
        Builder<List<T>> result = ImmutableEmisInternalResult.builder();
        OrganizedBatches<T> ob = getOrganizedBatches(emisBatch, efsProvider);

        result.content(ob.completeBatches);

        if (ob.incompleteBatches.isEmpty()) {
            result.stepResult(EmisStepResult.FINISHED);
        } else {
            //no need to check for last incomplete batch as it is already filtered out during download
                if (ob.completeBatches.isEmpty()) {
                    result.stepResult(EmisStepResult.FAILED);
                } else {
                    result.stepResult(EmisStepResult.FINISHED_WITH_ERRORS);
                }
                LOGGER.error("There are incomplete: {} and skipped: {} batches, manual correction needed!",
                        toTimestampList(ob.incompleteBatches, efsProvider), toTimestampList(ob.skippedBatches, efsProvider));

        }
        return result.build();
    }




    public <T>  @NotNull List<LocalDate> toTimestampListVerified(@NotNull List<T> input, Function<T,VerifiedEmisFileSet> vefsProvider) {
        return input.stream().map(vefs -> vefsProvider.apply(vefs).dateStamp()).collect(Collectors.toList());
    }

    public <T> @NotNull List<LocalDate> toTimestampList(@NotNull List<T> input, Function<T,EmisFileSet> efsProvider) {
        return input.stream().map(efss -> efsProvider.apply(efss).dateStamp()).collect(Collectors.toList());
    }

    /**
     * Returns batches in datestamp order and sftp server order
     */
    public @NotNull List<EmisFileSftpSet> getBatchesSortedByDayAndServer(@NotNull List<EmisFileSftpSet> emisBatch) {
        return emisBatch.stream().sorted(Comparator.comparing((EmisFileSftpSet efss) -> efss.emisFileSet().dateStamp()).thenComparing(EmisFileSftpSet::sftpUniqueId)).collect(Collectors.toList());
    }

    /**
     * Returns batches organized in lists of: completed, incomplete and skipped
     */
    private <T> OrganizedBatches getOrganizedBatches(@NotNull List<T> emisBatch, Function<T, EmisFileSet> efsProvider){
        OrganizedBatches ob = new OrganizedBatches<T>();
        for (T batch : emisBatch) {
            var efs = efsProvider.apply(batch);
            if (efs.isComplete()) {
                if (ob.incompleteBatches.isEmpty()) {
                    ob.completeBatches.add(batch);
                } else {
                    ob.skippedBatches.add(batch);
                }
            } else {
                ob.incompleteBatches.add(batch);
            }
        }

        if (ob.skippedBatches.isEmpty() && ob.incompleteBatches.size() == 1) {
            ob.lastIncomplete = true;
        }

        return ob;
    }

    private class OrganizedBatches<T>{
        List<T> completeBatches = new ArrayList<>();
        List<T> skippedBatches = new ArrayList<>();
        List<T> incompleteBatches = new ArrayList<>();
        boolean lastIncomplete;
    }

}
