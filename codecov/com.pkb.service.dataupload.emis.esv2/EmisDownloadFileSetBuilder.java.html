package com.pkb.service.dataupload.emis.esv2;

import com.pkb.service.dataupload.emis.esv2.model.EmisFile;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileType;
import org.jetbrains.annotations.NotNull;

import java.nio.file.Path;
import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.Set;

public class EmisDownloadFileSetBuilder extends AbstractEmisFileSetBuilder{

    public EmisDownloadFileSetBuilder(Path filesetPath, LocalDate datestamp) {
        super(filesetPath, datestamp);
    }

    protected EmisDownloadFileSetBuilder(@NotNull Set<EmisFileType> mandatoryTypes, @NotNull Set<EmisFileType> optionalTypes, Path filesetPath, LocalDate datestamp) {
        super(mandatoryTypes, optionalTypes, filesetPath, datestamp);
    }

    @Override
    protected void addToMap(Map<EmisFileType, List<EmisFile>> map, EmisFile file) {
        map.computeIfAbsent(file.type(), l -> new ArrayList<>()).add(file);
    }
}
