package com.pkb.service.dataupload.emis.esv2;

import com.pkb.common.config.PhrConfig;
import com.pkb.dto.EmisDiagnosticInformationDto;
import com.pkb.dto.ImmutableEmisDiagnosticInformationDto;
import com.pkb.dto.ImmutableEmisDiagnosticInformationDto.Builder;
import io.vavr.Tuple2;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.Collection;
import java.util.List;
import java.util.Map;
import java.util.Map.Entry;
import java.util.stream.Collectors;

public class EmisDiagnosticService {

    private final EmisConfigurationUtils emisConfigurationUtils;
    private final PhrConfig configuration;

    public EmisDiagnosticService(@NotNull EmisConfigurationUtils emisConfigurationUtils, @NotNull PhrConfig configuration) {
        this.emisConfigurationUtils = emisConfigurationUtils;
        this.configuration = configuration;
    }

    public @NotNull List<EmisDiagnosticInformationDto> getInformation() {
        return getRemoteFileList().entrySet().stream().map(this::convertToInformationDto).collect(Collectors.toList());
    }

    private EmisDiagnosticInformationDto convertToInformationDto(Entry<ConnectionData, Either<Exception, List<String>>> entry) {
        Builder builder = ImmutableEmisDiagnosticInformationDto.builder()
                .uri(entry.getKey().uri())
                .uniqueTag(entry.getKey().uniqueTag());
        Either<ImmutableEmisDiagnosticInformationDto, ImmutableEmisDiagnosticInformationDto> result = entry.getValue().map(l -> builder.connected(true).remoteFiles(l).build())
                .mapLeft(e -> builder.connected(false).error(e).build());
        if (result.isLeft()) {
            return result.getLeft();
        }
        return result.get();
    }

    public @NotNull Map<ConnectionData, Either<Exception, List<String>>> getRemoteFileList() {
        List<? extends ConnectionData> configs = emisConfigurationUtils.getEmisConnectionConfiguration();
        return configs.stream().map(s -> new Tuple2<>(s, getRemoteFileList(s))).collect(Collectors.toMap(t -> t._1, t -> t._2));
    }

    private Either<Exception, List<String>> getRemoteFileList(ConnectionData config) {
        try (IEmisDownloadConnection connection = config.connection()) {
            connection.connect();
            Collection<EmisLsEntry> directoryContent = connection.listDirectory();
            return Either.right(directoryContent.stream().map(EmisLsEntry::getFilename).collect(Collectors.toList()));
        } catch (Exception e) {
            return Either.left(e);
        }
    }
}
