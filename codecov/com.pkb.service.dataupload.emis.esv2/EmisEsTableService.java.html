    private <T, K> List<T> batchSave(List<T> entities, Function<T, K> getKey, Function<Collection<K>, List<T>> lookupMethod, Function<List<T>, List<T>> saveMethod) {
        var keys = entities.stream().map(getKey).collect(Collectors.toSet());
        var existingMap = lookupMethod.apply(keys).stream().collect(toMap(getKey, identity()));
        List<T> result = new ArrayList<>(entities.size());
        List<T> toPersist = new LinkedList<>();
        for (T actual : entities) {
            T existing = existingMap.get(getKey.apply(actual));
            if (existing == null) {
                toPersist.add(actual);
            } else {
                if (existing.equals(actual)) {
                    result.add(actual);
                } else {
                    toPersist.add(actual);
                }
            }
        }

        result.addAll(saveMethod.apply(toPersist));
        return result;
    }

    @Override

    private <T extends Identifiable, K> T save(T entity, Function<K, Long> findKeyByGuid, Function<T, K> getGuid, JpaRepository<T, Long> repository) {
        if (entity.getId() == null) {
            K guid = getGuid.apply(entity);
            Long id = guidToKeyCache.getIfPresent(guid);
            if (id == null) {
                id = findKeyByGuid.apply(guid);
                if (id != null) {
                    guidToKeyCache.put(guid, id);
                }
            }
            if (id != null) {
                entity.setId(id);
                return repository.saveAndFlush(entity);
            } else {
                var result = repository.save(entity);
                guidToKeyCache.put(guid, result.getId());
                return result;
            }
        } else {
            return repository.save(entity);
        }
    }

    private <T extends Identifiable> List<T> saveBatch(List<T> entities, Function<T, T> saveMethod, int batchSize, JpaRepository<T, Long> repository) {
        List<T> result = new ArrayList<>(entities.size());
        int cur = 0;
        for (T entity : entities) {
            result.add(saveMethod.apply(entity));
            if ((++cur % batchSize) == 0) {
                repository.flush();
            }
        }

        if ((cur % batchSize) != 0) {
            repository.flush();
        }

        return result;
    }
