package com.pkb.service.dataupload.emis.esv2;

import com.google.common.collect.Sets;
import com.pkb.service.dataupload.emis.esv2.model.EmisFile;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileSet;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileType;
import com.pkb.service.dataupload.emis.esv2.model.ImmutableEmisFileSet;
import org.jetbrains.annotations.NotNull;

import java.math.BigInteger;
import java.nio.file.Path;
import java.time.LocalDate;
import java.util.Collections;
import java.util.EnumMap;
import java.util.List;
import java.util.Map;
import java.util.Set;
import java.util.stream.Collectors;

import static com.pkb.service.dataupload.emis.esv2.model.EmisFileType.ADMIN_ORGANISATION;
import static com.pkb.service.dataupload.emis.esv2.model.EmisFileType.ADMIN_PATIENT;
import static com.pkb.service.dataupload.emis.esv2.model.EmisFileType.ADMIN_USER_IN_ROLE;
import static com.pkb.service.dataupload.emis.esv2.model.EmisFileType.CARE_RECORD_OBSERVATION;
import static com.pkb.service.dataupload.emis.esv2.model.EmisFileType.CARE_RECORD_PROBLEM;
import static com.pkb.service.dataupload.emis.esv2.model.EmisFileType.CODING_CLINICAL_CODES;
import static com.pkb.service.dataupload.emis.esv2.model.EmisFileType.CODING_DRUG_CODES;
import static com.pkb.service.dataupload.emis.esv2.model.EmisFileType.PRESCRIBING_DRUG_RECORD;

public abstract class AbstractEmisFileSetBuilder {

    private final static Set<EmisFileType> MANDATORY_FILES = Sets.immutableEnumSet(ADMIN_ORGANISATION,
            ADMIN_USER_IN_ROLE, ADMIN_PATIENT, CODING_CLINICAL_CODES,
            CODING_DRUG_CODES, CARE_RECORD_OBSERVATION,
            CARE_RECORD_PROBLEM, PRESCRIBING_DRUG_RECORD);

    private final static Set<EmisFileType> OPTIONAL_FILES = Collections.emptySet();

    private final Set<EmisFileType> mandatoryFileTypes;
    private final Set<EmisFileType> optionalFileTypes;
    private final Set<EmisFileType> supportedFileTypes;
    private final Map<EmisFileType, List<EmisFile>> supported = new EnumMap<>(EmisFileType.class);
    private final Map<EmisFileType, List<EmisFile>> notSupported = new EnumMap<>(EmisFileType.class);


    private final Path filesetPath;
    private final LocalDate datestamp;

    public AbstractEmisFileSetBuilder(Path filesetPath, LocalDate datestamp) {
        this(MANDATORY_FILES, OPTIONAL_FILES, filesetPath, datestamp);
    }

    protected AbstractEmisFileSetBuilder(@NotNull Set<EmisFileType> mandatoryTypes, @NotNull Set<EmisFileType> optionalTypes, Path filesetPath, LocalDate datestamp) {
        this.mandatoryFileTypes = mandatoryTypes;
        this.optionalFileTypes = optionalTypes;
        this.filesetPath = filesetPath;
        this.datestamp = datestamp;
        supportedFileTypes = Sets.union(this.optionalFileTypes, this.mandatoryFileTypes);
    }

    public void add(@NotNull EmisFile file) {
        if (getSupportedFileTypes().contains(file.type())) {
            addToMap(getSupported(), file);
        } else {
            addToMap(getNotSupported(), file);
        }
    }

    @NotNull
    public EmisFileSet build() {
        boolean isComplete = supported.keySet().containsAll(mandatoryFileTypes);

        BigInteger size = supported.values()
                .stream()
                .flatMap(List::stream)
                .collect(Collectors.toList()).stream().map(e -> BigInteger.valueOf(e.size())).reduce(BigInteger.ZERO, BigInteger::add);

        return ImmutableEmisFileSet.of(supported, size, notSupported, isComplete, datestamp, filesetPath);
    }

    protected Set<EmisFileType> getSupportedFileTypes(){
        return supportedFileTypes;
    }

    protected Map<EmisFileType, List<EmisFile>> getSupported(){
        return supported;
    }

    protected Map<EmisFileType, List<EmisFile>> getNotSupported(){
        return notSupported;
    }

    protected abstract void addToMap(Map<EmisFileType, List<EmisFile>> map, EmisFile file);

}
