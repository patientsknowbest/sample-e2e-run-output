package com.pkb.service.dataupload.emis.esv2;

import com.google.common.annotations.VisibleForTesting;
import com.google.common.collect.ImmutableMap;
import com.google.common.collect.Maps;
import com.pkb.common.ClearableInternalState;
import com.pkb.service.dataupload.emis.esv2.batch.FileProcessor;
import com.pkb.service.dataupload.emis.esv2.model.AdminPatient;
import com.pkb.service.dataupload.emis.esv2.model.PKBPersonInfo;
import io.micrometer.core.instrument.DistributionSummary;
import io.micrometer.core.instrument.Metrics;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Arrays;
import java.util.EnumMap;
import java.util.Set;
import java.util.concurrent.ConcurrentHashMap;
import java.util.function.Function;

public class EmisPersonStatHelper implements ClearableInternalState {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final String PERSON_ID_METRIC_NAME = "pkb_phr_integration_emis_csv_batch_patient_identifiers_final";
    private static final String PERSON_ID_JOB_METRIC_NAME = "pkb_phr_integration_emis_csv_job_patient_identifiers_final";
    private static final String PERSON_ID_JOB_COUNTER_NAME = "pkb_phr_integration_emis_csv_job_patient_identifiers";
    private static final String PERSON_METRIC_TAG = "codeType";
    private static final String PROCESSOR_TYPE_TAG = "processorType";

    private static final String TYPE_GUID = "GUID";
    private static final String TYPE_PERSON_ID = "PERSON_ID";
    private static final String TYPE_PERSON_NOT_FOUND = "PERSON_NOT_FOUND";

    private final ImmutableMap<FileProcessor.ProcessorType, Set<String>> personGuidSeenInRun;
    private final ImmutableMap<FileProcessor.ProcessorType, Set<String>> personGuidSeenInBatch;
    private final ImmutableMap<FileProcessor.ProcessorType, Set<Long>> personIdSeenInRun;
    private final ImmutableMap<FileProcessor.ProcessorType, Set<Long>> personIdSeenInBatch;
    private final ImmutableMap<FileProcessor.ProcessorType, Set<String>> personGuidUnexpectedlyUnableToFindPatientForInRun;
    private final ImmutableMap<FileProcessor.ProcessorType, Set<String>> personGuidUnexpectedlyUnableToFindPatientForInBatch;

    public EmisPersonStatHelper() {
        personGuidSeenInRun = initializeMap();
        personGuidSeenInBatch = initializeMap();
        personIdSeenInRun = initializeMap();
        personIdSeenInBatch = initializeMap();
        personGuidUnexpectedlyUnableToFindPatientForInRun = initializeMap();
        personGuidUnexpectedlyUnableToFindPatientForInBatch = initializeMap();
    }

    @Override
    public void clearState() {
        personGuidSeenInRun.values().forEach(Set::clear);
        personGuidSeenInBatch.values().forEach(Set::clear);
        personIdSeenInRun.values().forEach(Set::clear);
        personIdSeenInBatch.values().forEach(Set::clear);
        personGuidUnexpectedlyUnableToFindPatientForInRun.values().forEach(Set::clear);
        personGuidUnexpectedlyUnableToFindPatientForInBatch.values().forEach(Set::clear);
    }

    private <T> ImmutableMap<FileProcessor.ProcessorType, Set<T>> initializeMap() {
        EnumMap<FileProcessor.ProcessorType, Set<T>> enumMap = new EnumMap<>(FileProcessor.ProcessorType.class);
        Arrays.stream(FileProcessor.ProcessorType.values()).forEach(processorType -> enumMap.put(processorType, ConcurrentHashMap.newKeySet()));
        return Maps.immutableEnumMap(enumMap);
    }

    public void reportBatchEnd() {
        report(personGuidSeenInBatch, processorType -> Metrics.summary(PERSON_ID_METRIC_NAME, PROCESSOR_TYPE_TAG, processorType.name(), PERSON_METRIC_TAG, TYPE_GUID));
        report(personIdSeenInBatch, processorType -> Metrics.summary(PERSON_ID_METRIC_NAME, PROCESSOR_TYPE_TAG, processorType.name(), PERSON_METRIC_TAG, TYPE_PERSON_ID));
        report(personGuidUnexpectedlyUnableToFindPatientForInBatch, processorType -> Metrics.summary(PERSON_ID_METRIC_NAME, PROCESSOR_TYPE_TAG, processorType.name(), PERSON_METRIC_TAG, TYPE_PERSON_NOT_FOUND));
    }

    public void reportJobFinished() {
        report(personGuidSeenInRun, processorType -> Metrics.summary(PERSON_ID_JOB_METRIC_NAME, PROCESSOR_TYPE_TAG, processorType.name(), PERSON_METRIC_TAG, TYPE_GUID));
        report(personIdSeenInRun, processorType -> Metrics.summary(PERSON_ID_JOB_METRIC_NAME, PROCESSOR_TYPE_TAG, processorType.name(), PERSON_METRIC_TAG, TYPE_PERSON_ID));
        report(personGuidUnexpectedlyUnableToFindPatientForInRun, processorType -> Metrics.summary(PERSON_ID_METRIC_NAME, PROCESSOR_TYPE_TAG, processorType.name(), PERSON_METRIC_TAG, TYPE_PERSON_NOT_FOUND));
        clearState();
    }

    private <T> void report(ImmutableMap<FileProcessor.ProcessorType, Set<T>> input, Function<FileProcessor.ProcessorType, DistributionSummary> metricSupplier) {
        input.forEach((processorType, set) -> {
            if (!set.isEmpty()) {
                metricSupplier.apply(processorType).record(set.size());
                set.clear();
            }
        });
    }

    public void reportPatient(FileProcessor.ProcessorType processorType, AdminPatient patient) {
        reportPatient(processorType, patient.personGuid());
    }

    private void reportPatient(FileProcessor.ProcessorType processorType, String personGuid) {
        if (personGuidSeenInRun.get(processorType).add(personGuid)) {
            Metrics.counter(PERSON_ID_JOB_COUNTER_NAME, PROCESSOR_TYPE_TAG, processorType.name(), PERSON_METRIC_TAG, TYPE_GUID).increment();
        }
        personGuidSeenInBatch.get(processorType).add(personGuid);
    }

    public void reportPatient(FileProcessor.ProcessorType processorType, PKBPersonInfo patient) {
        reportPatient(processorType, patient.id());
    }

    private void reportPatient(FileProcessor.ProcessorType processorType, Long personId) {
        if (personIdSeenInRun.get(processorType).add(personId)) {
            Metrics.counter(PERSON_ID_JOB_COUNTER_NAME, PROCESSOR_TYPE_TAG, processorType.name(), PERSON_METRIC_TAG, TYPE_PERSON_ID).increment();
        }
        personIdSeenInBatch.get(processorType).add(personId);
    }

    public void reportMissingPatient(FileProcessor.ProcessorType processorType, AdminPatient patient) {
        reportMissingPatient(processorType, patient.personGuid());
    }

    private void reportMissingPatient(FileProcessor.ProcessorType processorType, String personGuid) {
        if (personGuidUnexpectedlyUnableToFindPatientForInRun.get(processorType).add(personGuid)) {
            Metrics.counter(PERSON_ID_JOB_COUNTER_NAME, PROCESSOR_TYPE_TAG, processorType.name(), PERSON_METRIC_TAG, TYPE_PERSON_NOT_FOUND).increment();
        }
        personGuidUnexpectedlyUnableToFindPatientForInBatch.get(processorType).add(personGuid);
    }

    @VisibleForTesting
    ImmutableMap<FileProcessor.ProcessorType, Set<String>> getPersonGuidSeenInRun() {
        return personGuidSeenInRun;
    }

    @VisibleForTesting
    ImmutableMap<FileProcessor.ProcessorType, Set<String>> getPersonGuidSeenInBatch() {
        return personGuidSeenInBatch;
    }

    @VisibleForTesting
    ImmutableMap<FileProcessor.ProcessorType, Set<Long>> getPersonIdSeenInRun() {
        return personIdSeenInRun;
    }

    @VisibleForTesting
    ImmutableMap<FileProcessor.ProcessorType, Set<Long>> getPersonIdSeenInBatch() {
        return personIdSeenInBatch;
    }
}
