package com.pkb.service.dataupload.emis.esv2;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.dataupload.emis.esv2.EmisEsJobState;
import com.pkb.dataupload.emis.esv2.EmisJobStatus;
import com.pkb.service.dataupload.emis.esv2.repository.EmisEsJobStateRepository;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.data.domain.PageRequest;
import org.springframework.data.domain.Slice;
import org.springframework.data.domain.Sort;

import javax.transaction.Transactional;
import java.util.List;
import java.util.Optional;

import static org.springframework.util.CollectionUtils.isEmpty;

public class EmisEsJobStateService {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private final EmisEsJobStateRepository emisEsJobStateRepository;
    private final DateTimeService dateTimeService;

    public EmisEsJobStateService(EmisEsJobStateRepository emisEsJobStateRepository, DateTimeService dateTimeService) {
        this.emisEsJobStateRepository = emisEsJobStateRepository;
        this.dateTimeService = dateTimeService;
    }

    public Optional<EmisEsJobState> find(long jobId) {
        return emisEsJobStateRepository.findById(jobId);
    }

    public Slice<EmisEsJobState> findLastBatch() {
        return emisEsJobStateRepository.findJobs(PageRequest.of(0, 200, Sort.Direction.DESC, "id"));
    }

    public Optional<EmisEsJobState> findLastSuccessfulJob() {
        return getSingleResult(emisEsJobStateRepository.findSuccessfulJobsDescendingByStarted(PageRequest.of(0, 1)));
    }

    public Optional<EmisEsJobState> findLatestJob() {
        return getSingleResult(emisEsJobStateRepository.findJobsDescendingByStarted(PageRequest.of(0, 1)));
    }

    @NotNull
    public EmisEsJobState save(@NotNull EmisEsJobState job) {
        EmisEsJobState persistedJobState = emisEsJobStateRepository.save(job);
        emisEsJobStateRepository.flush();
        return persistedJobState;
    }

    @NotNull
    public EmisEsJobState create() {
        EmisEsJobState defaultJob = createDefaultJob();
        return save(defaultJob);
    }

    public EmisEsJobState create(@NotNull EmisEsJobState lastJob) {
        LOGGER.info("Previous job {}", lastJob.getId());
        EmisEsJobState defaultJob = createDefaultJob();
        EmisEsJobState result = save(defaultJob);
        LOGGER.info("Current job {}", result.getId());
        return result;
    }

    @Transactional
    public int abortRunningJobs() {
        return emisEsJobStateRepository.bulkUpdateStatus(EmisJobStatus.RUNNING, EmisJobStatus.ABORTED);
    }

    public Optional<EmisEsJobState> findFirstFailedJob() {
        return getSingleResult(emisEsJobStateRepository.findFailedJobsDescendingByStarted(PageRequest.of(0, 1)));
    }

    private EmisEsJobState createDefaultJob() {
        EmisEsJobState job = new EmisEsJobState();
        job.setStarted(dateTimeService.now());
        job.setRetryPermitted(true);
        job.setStatus(EmisJobStatus.RUNNING);
        return job;
    }

    private Optional<EmisEsJobState> getSingleResult(List<EmisEsJobState> input) {
        if (isEmpty(input)) {
            return Optional.empty();
        }
        return Optional.ofNullable(input.get(0));
    }
}
