package com.pkb.service.dataupload.emis.esv2;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.dataupload.emis.esv2.EmisEsJobState;
import com.pkb.dataupload.emis.esv2.EmisJobStatus;
import org.jetbrains.annotations.NotNull;
import org.quartz.DisallowConcurrentExecution;
import org.quartz.Job;
import org.quartz.JobExecutionContext;
import org.quartz.JobExecutionException;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.time.Instant;
import java.time.temporal.ChronoUnit;
import java.util.Optional;

@DisallowConcurrentExecution
@Component
public class EmisJob implements Job {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private EmisEsJobStateService emisEsJobStateService;
    private EmisIntegrationService emisIntegrationService;
    private EmisEsBatchStateService emisEsBatchStateService;
    private PhrConfig configuration;
    private DateTimeService dateTimeService;

    @Override
    public void execute(JobExecutionContext context) throws JobExecutionException {
        if (configuration.isEmisEsEnabled()) {
            LOGGER.info("EMIS job started");
            try {
                clearAbortedJobs();
                Optional<EmisEsJobState> maybeLatestJob = emisEsJobStateService.findLatestJob();
                if (maybeLatestJob.isPresent()) {
                    EmisEsJobState latestJob = maybeLatestJob.get();
                    additionalRun(latestJob);
                } else {
                    firstRun();
                }
                LOGGER.info("EMIS job finished");
            } catch (Exception e) {
                LOGGER.error("Error during EMIS job execution, job failed!", e);
            }
        } else {
            LOGGER.info("Skipping EMIS job as it is turned off.");
        }
    }

    @Autowired
    public void setConfiguration(@NotNull PhrConfig configuration) {
        this.configuration = configuration;
    }

    @Autowired
    public void setEmisEsJobStateService(@NotNull EmisEsJobStateService emisEsJobStateService) {
        this.emisEsJobStateService = emisEsJobStateService;
    }

    @Autowired
    public void setEmisEsBatchStateService(@NotNull EmisEsBatchStateService emisEsBatchStateService) {
        this.emisEsBatchStateService = emisEsBatchStateService;
    }

    @Autowired
    public void setEmisIntegrationService(@NotNull EmisIntegrationService emisIntegrationService) {
        this.emisIntegrationService = emisIntegrationService;
    }

    @Autowired
    public void setDateTimeService(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    private boolean isNextRunAllowed(EmisEsJobState latestJob) {
        return latestJob.getStatus() == EmisJobStatus.COMPLETED || latestJob.isRetryPermitted();
    }

    private void firstRun() {
        run(emisEsJobStateService.create(), false);
    }

    @VisibleForTesting
    private void additionalRun(EmisEsJobState latestJob) {
        boolean downloadOnly = !isNextRunAllowed(latestJob);
        EmisEsJobState job = emisEsJobStateService.create(latestJob);
        int movedBatches = emisEsBatchStateService.moveBatchStateToNewJob(latestJob.getId(), job.getId());
        LOGGER.info("Moved {} batch state from job {} to {}.", movedBatches, latestJob.getId(), job.getId());
        run(job, downloadOnly);
    }

    private void run(EmisEsJobState currentJob, boolean downloadOnly) {
        try {
            EmisStepResult result = emisIntegrationService.run(currentJob.getId(), downloadOnly);
            handleResult(result, currentJob);
        } catch (Exception e) {
            LOGGER.error("Error during EMIS job execution; job failed!", e);
            EmisEsJobState actualState = emisEsJobStateService.find(currentJob.getId())
                    .orElseThrow(() -> new IllegalStateException("EMIS job can't be found by id " + currentJob.getId() + "!"));
            actualState.setFinished(dateTimeService.now());
            actualState.setStatus(EmisJobStatus.FAILED);
            emisEsJobStateService.save(actualState);
        }
    }

    private void handleResult(EmisStepResult result, EmisEsJobState currentJob) {
        if (EmisStepResult.FINISHED == result) {
            handleSuccessResult(currentJob);
        } else {
            handleErrorResult(currentJob);
        }
    }

    private void handleSuccessResult(EmisEsJobState jobState) {
        jobState.setFinished(dateTimeService.now());
        jobState.setStatus(EmisJobStatus.COMPLETED);
        jobState.setRetryPermitted(true);
        emisEsJobStateService.save(jobState);
        LOGGER.info("EMIS job id:{} finished successfully", jobState.getId());
    }

    private void handleErrorResult(EmisEsJobState jobState) {
        Instant current = dateTimeService.now();
        jobState.setFinished(current);
        jobState.setStatus(EmisJobStatus.FAILED);
        jobState.setRetryPermitted(!isAfterFailureDeadline(current));
        emisEsJobStateService.save(jobState);
        LOGGER.error("EMIS job id:{} failed!", jobState.getId());
    }

    private boolean isAfterFailureDeadline(Instant current) {
        Instant lastTime = emisEsJobStateService.findLastSuccessfulJob().map(EmisEsJobState::getStarted)
                .orElseGet(() -> emisEsJobStateService.findFirstFailedJob().map(EmisEsJobState::getStarted).orElse(null));
        if (lastTime == null) {
            return false;
        }
        int gracePeriodHours = configuration.getEmisFailureGracePeriodHours();
        Instant deadline = lastTime.plus(gracePeriodHours, ChronoUnit.HOURS);
        boolean result = current.isAfter(deadline);
        if (result) {
            LOGGER.error("Disabling EMIS processing, download of files will continue. Last successful run more than {} hours ago; manual intervention needed!", gracePeriodHours);
        }
        return result;
    }

    private void clearAbortedJobs() {
        int updateAbortedJobs = emisEsJobStateService.abortRunningJobs();
        if (0 < updateAbortedJobs) {
            LOGGER.warn("Cleared {} stuck jobs.", updateAbortedJobs);
        }
    }
}
