package com.pkb.service.dataupload.emis.esv2;

import com.fasterxml.jackson.core.type.TypeReference;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.dataformat.yaml.YAMLMapper;
import com.fasterxml.jackson.datatype.guava.GuavaModule;
import com.fasterxml.jackson.datatype.jdk8.Jdk8Module;
import com.google.common.collect.ImmutableList;
import com.google.common.collect.ImmutableMap;
import com.pkb.common.config.PhrConfig;
import com.pkb.dataupload.emis.esv2.csv.EmisEsAdminOrganisation;
import com.pkb.institute.OrgMergeHelper;
import com.pkb.institute.entity.Org;
import com.pkb.service.dataupload.emis.esv2.model.AdminPatient;
import com.pkb.util.StringUtil;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.io.IOException;
import java.io.InputStream;
import java.nio.file.Files;
import java.util.List;
import java.util.Optional;
import java.util.Set;
import java.util.stream.Collectors;

import static com.pkb.service.dataupload.emis.esv2.model.Constants.CONFIG_WHITELIST_KEY_NHS_NUMBER;
import static com.pkb.service.dataupload.emis.esv2.model.Constants.CONFIG_WHITELIST_KEY_PATIENT_NUMBER;
import static org.apache.commons.collections4.CollectionUtils.isEmpty;

public abstract class AbstractEmisEsWhitelistService implements EmisEsWhitelistService {

    private static final String WHITELIST_ALL = "*";

    private static final ObjectMapper YAML_MAPPER = new YAMLMapper().registerModule(new GuavaModule()).registerModule(new Jdk8Module());

    protected final PhrConfig config;

    AbstractEmisEsWhitelistService(PhrConfig config) {
        this.config = config;
    }

    private void validateOdsCode(String odsCode) {
        // ODS code: 1-8 chars, alpha uppercase + digits
        if (odsCode.length() < 1 || odsCode.length() > 8 || !StringUtils.containsOnly(odsCode, StringUtil.ALPHA_UPPER + StringUtil.NUMERIC)) {
            throw new IllegalArgumentException("Invalid ODS code in whitelist entry: " + odsCode);
        }
    }

    private void validateIdList(List<String> ids) {
        if (ids.contains(WHITELIST_ALL) && ids.size() > 1) {
            throw new IllegalArgumentException("Whitelist contains ALL wildcard plus other entries");
        } else if (!ids.contains(WHITELIST_ALL)) {
            ids.forEach(id -> {
                if (id.length() < 4 || id.length() > 10 || !StringUtils.containsOnly(id, StringUtil.NUMERIC)) {
                    throw new IllegalArgumentException("Invalid ID in whitelist entry: " + id);
                }
            });
        }
    }

    private void validateIdKeys(Set<String> keys) {
        List<String> unrecognisedKeys = keys.stream().filter(k -> !k.equals(CONFIG_WHITELIST_KEY_NHS_NUMBER) & !k.equals(CONFIG_WHITELIST_KEY_PATIENT_NUMBER))
                .collect(Collectors.toList());
        if (!unrecognisedKeys.isEmpty()) {
            throw new IllegalArgumentException("Unrecognised ID keys: " + String.join(",", unrecognisedKeys));
        }
    }

    private boolean isWildcard(List<String> whitelist) {
        return whitelist != null
                && whitelist.size() == 1
                && WHITELIST_ALL.equals(whitelist.get(0));
    }

    protected ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> fromExternalWhitelistConfig() {
        if (config.isEmisEsWhiteListEnabled()) {
            return config.getEmisEsWhitelistYamlPath().map(ymlPath -> {
                try (InputStream configInputStream = Files.newInputStream(ymlPath)) {
                    return readYamlConfig(configInputStream);
                } catch (IOException e) {
                    throw new RuntimeException("Failed loading Emis Whitelist Config", e);
                }
            }).orElse(ImmutableMap.of());
        } else {
            return ImmutableMap.of();
        }
    }

    protected ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> readYamlConfig(InputStream configInputStream) throws IOException {
        ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> parsedWhitelist = YAML_MAPPER.readValue(configInputStream,
                new TypeReference<>() {
                });
        parsedWhitelist.forEach((key, value) -> {
            validateOdsCode(key);
            validateIdKeys(value.keySet());
            value.values().forEach(this::validateIdList);
        });
        return parsedWhitelist;

    }

    @Override
    public boolean isAllowedPatient(@NotNull AdminPatient adminPatient, @NotNull EmisEsAdminOrganisation emisEsAdminOrganisation) {
        return isAllowedPatient(getMap(), adminPatient, emisEsAdminOrganisation);
    }

    @Override
    public boolean isAllowedPatient(@NotNull AdminPatient adminPatient, @NotNull OrgMergeHelper orgMergeHelper) {
        if (!config.isEmisEsWhiteListEnabled()) {
            return true;
        }
        return isAllowedPatient(getMap(), adminPatient, orgMergeHelper);
    }

    protected boolean isAllowedPatient(ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> map, @NotNull AdminPatient adminPatient, @NotNull OrgMergeHelper orgMergeHelper) {
        if (!config.isEmisEsWhiteListEnabled()) {
            return true;
        }
        return isAllowedPatient(map, adminPatient, orgMergeHelper.getOdsCode());
    }

    protected boolean isAllowedPatient(ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> map, @NotNull AdminPatient adminPatient,
                                       @NotNull EmisEsAdminOrganisation org) {
        if (!config.isEmisEsWhiteListEnabled()) {
            return true;
        }
        return isAllowedPatient(map, adminPatient, org.getOdsCode());
    }

    private boolean isAllowedPatient(ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> map, @NotNull AdminPatient adminPatient, @NotNull String odsCode) {
        if (adminPatient.nhsNumber().isEmpty() && adminPatient.patientNumber().isEmpty()) {
            return adminPatient.isHandleAsDischargedOrDeleted() && map.containsKey(odsCode);
        }
        return isAllowedId(adminPatient.nhsNumber(), getNhsIdsForOdsCode(map, odsCode)) ||
                isAllowedId(adminPatient.patientNumber().map(Object::toString), getPatientIdsForOdsCode(map, odsCode));
    }

    private boolean isAllowedId(Optional<String> maybeId, @Nullable List<String> whitelistOrNull) {
        Optional<List<String>> maybeWhitelist = Optional.ofNullable(whitelistOrNull);
        return maybeId.flatMap(id -> maybeWhitelist.map(whitelist -> isWildcard(whitelist) || whitelist.contains(StringUtils.trimToNull(id)))).orElse(false);
    }

    /**
     * @return true if whitelist filtering is enabled, and if this org is not listed with a "*" whitelist entry
     */
    @Override
    public boolean isFilteredByPatientWhitelist(@NotNull Org org) {
        return isFilteredByPatientWhitelist(getMap(), org);
    }

    protected boolean isFilteredByPatientWhitelist(ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> map, @NotNull Org org) {
        if (!config.isEmisEsWhiteListEnabled()) {
            return false;
        }

        // by default: whitelisting is enabled for all orgs; unless we have a * (all) value for their whitelist
        Set<String> odsCodes = org.getOdsCodes().collect(Collectors.toSet());

        if (isEmpty(odsCodes)) {
            return true;
        }

        // yes, filtered by whitelist if neither property is the wildcard
        return odsCodes.stream().anyMatch(s -> !isWildcard(getNhsIdsForOdsCode(map, s))
                && !isWildcard(getPatientIdsForOdsCode(map, s)));
    }

    @Override
    public Set<String> getOdsCodes() {
        return getOdsCodes(getMap());
    }

    protected Set<String> getOdsCodes(ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> map) {
        return map.keySet();
    }

    @Override
    public void setYamlConfig(String configYaml) {
        throw new IllegalStateException("This shouldn't ever be called!");
    }

    private ImmutableList<String> getNhsIdsForOdsCode(ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> map, String odsCode) {
        return map.getOrDefault(odsCode, ImmutableMap.of()).getOrDefault(CONFIG_WHITELIST_KEY_NHS_NUMBER, ImmutableList.of());
    }

    private ImmutableList<String> getPatientIdsForOdsCode(ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> map, String odsCode) {
        return map.getOrDefault(odsCode, ImmutableMap.of()).getOrDefault(CONFIG_WHITELIST_KEY_PATIENT_NUMBER, ImmutableList.of());
    }

    @Override
    public void clearState() {
        //NO-OP as this supposed to be immutable
    }

    abstract protected ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> getMap();

    @Override
    public boolean isWhitelistedOrg(Org o) {
        if (!config.isEmisEsWhiteListEnabled()) {
            return true;
        }
        return o.getOdsCodes().anyMatch(s -> isWhitelistedOrg(s));
    }

    @Override
    public boolean isWhitelistedOrg(String odsCode) {
        if (!config.isEmisEsWhiteListEnabled()) {
            return true;
        }
        return getOdsCodes().contains(odsCode);
    }
}
