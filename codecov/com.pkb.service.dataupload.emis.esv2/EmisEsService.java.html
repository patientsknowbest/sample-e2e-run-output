package com.pkb.service.dataupload.emis.esv2;

import com.pkb.dataupload.emis.esv2.EmisEsJobState;
import com.pkb.dto.EmisJobInformationDto;
import com.pkb.mappers.EmisEsJobStateToEmisInformationDtoMapper;
import io.vavr.Tuple2;
import org.jetbrains.annotations.NotNull;
import org.quartz.JobKey;
import org.quartz.Scheduler;
import org.quartz.SchedulerException;
import org.springframework.data.domain.Slice;

import java.io.IOException;
import java.nio.file.FileVisitResult;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.SimpleFileVisitor;
import java.nio.file.attribute.BasicFileAttributes;
import java.util.Collections;
import java.util.List;
import java.util.stream.Collectors;

import static config.scedule.EmisSchedulingConfig.EMIS_GROUP;
import static config.scedule.EmisSchedulingConfig.EMIS_JOB_DETAIL_KEY;
import static java.util.stream.Collectors.toSet;

public class EmisEsService {

    private final EmisEsJobStateService emisEsJobStateService;
    private final EmisEsBatchStateService emisEsBatchStateService;
    private final Scheduler scheduler;
    private final EmisEsJobStateToEmisInformationDtoMapper mapper;
    private EmisConfigurationUtils emisConfigurationUtils;

    public EmisEsService(@NotNull EmisEsJobStateService emisEsJobStateService, @NotNull Scheduler scheduler, EmisEsBatchStateService emisEsBatchStateService,
                         EmisEsJobStateToEmisInformationDtoMapper mapper, EmisConfigurationUtils emisConfigurationUtils) {
        this.emisEsJobStateService = emisEsJobStateService;
        this.scheduler = scheduler;
        this.emisEsBatchStateService = emisEsBatchStateService;
        this.mapper = mapper;
        this.emisConfigurationUtils = emisConfigurationUtils;
    }

    public List<EmisJobInformationDto> getEmisJobInformation() {
        Slice<EmisEsJobState> jobStates = emisEsJobStateService.findLastBatch();
        var jobIds = jobStates.stream().map(EmisEsJobState::getId).collect(toSet());
        var jobs = emisEsBatchStateService.findByJob(jobIds);
        return
                jobStates.stream()
                        .map(e -> new Tuple2<>(e, jobs.getOrDefault(e.getId(), Collections.emptyMap())))
                        .map(mapper::toDto)
                        .collect(Collectors.toList());
    }

    public void triggerEmisJob() throws SchedulerException {
        scheduler.triggerJob(JobKey.jobKey(EMIS_JOB_DETAIL_KEY, EMIS_GROUP));
    }

    public void clearEmisDownloadDirectory() {
        Path cleanLocalDownload = emisConfigurationUtils.getLocalDownloadDirectory();
        try {
            Files.walkFileTree(cleanLocalDownload,
                    new SimpleFileVisitor<>() {
                        @Override
                        public FileVisitResult visitFile(Path path, BasicFileAttributes basicFileAttributes) throws IOException {
                            Files.delete(path);
                            return FileVisitResult.CONTINUE;
                        }

                        @Override
                        public FileVisitResult postVisitDirectory(Path path, IOException e) throws IOException {
                            if (!path.equals(cleanLocalDownload)) {
                                Files.delete(path);
                            }
                            return FileVisitResult.CONTINUE;
                        }
                    });
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
    }
}
