package com.pkb.service.dataupload.emis.esv2;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.common.base.Error;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.emis.es.entity.EmisEsObservationDelta;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Component;

import java.util.List;

@Component
public abstract class EmisEsDeltaManager<T> extends TransactionManager {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    public static final String ERROR_APPLYING_DELTA = "Error while applying EMIS ES delta";

    private final EmisEsManager emisEsManager;

    public EmisEsDeltaManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider, EmisEsManager emisEsManager) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.emisEsManager = emisEsManager;
    }

    /**
     *
     */
    public void applyEmisEsDeltas(@NotNull LoggedInEHRRequestContext requestContext, long patientId,
            long patientAccountId) {
        LOGGER.info("Started processing deltas for patientAccountId={}", patientAccountId);
        LoggedInEHRRequestContext piggybackContext = requestContext.withPiggyback(true);

        // check for any deltas to be processed via EMIS ES
        List<EmisEsObservationDelta> queue = emisEsManager
                .getObservationDeltas(piggybackContext, patientAccountId, getDeltaType());
        for (EmisEsObservationDelta delta : queue) {
            transactional(() -> {
                        Either<Error, T> result = applyDelta(piggybackContext, patientId, patientAccountId, delta);
                        if (result.isRight()) {
                            emisEsManager.deleteEmisEsObservationDeltaById(delta.getId());
                        } else {
                            LOGGER.error("Failed applying delta to uniqueid={} patientId={}", delta.getParentUniqueId(), patientId);
                        }
                    }
            );
        }
    }

    public abstract Either<Error, T> applyDelta(@NotNull LoggedInEHRRequestContext ehrRequestContext, long patientId, long patientAccountId,
            EmisEsObservationDelta delta);

    public abstract EmisEsObservationDelta.DeltaType getDeltaType();
}
