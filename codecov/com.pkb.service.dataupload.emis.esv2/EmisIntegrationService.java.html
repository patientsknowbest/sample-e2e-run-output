package com.pkb.service.dataupload.emis.esv2;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.dataupload.emis.esv2.EmisEsBatchState;
import com.pkb.service.dataupload.emis.esv2.model.ImmutableVerifiedEmisFileSetLists;
import com.pkb.service.dataupload.emis.esv2.model.VerifiedEmisFileSetLists;
import io.micrometer.core.instrument.Counter;
import io.micrometer.core.instrument.Metrics;
import io.micrometer.core.instrument.Timer;
import name.neuhalfen.projects.crypto.bouncycastle.openpgp.keys.keyrings.KeyringConfig;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.IOException;

import static java.util.Collections.emptyList;

public class EmisIntegrationService {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private static final String TAG_SOURCE = "source";
    private static final String TAG_PHASE = "phase";
    private static final String INTEGRATION_TIMER_NAME = "pkb_phr_integration_emis_csv_timings";
    private static final Timer INTEGRATION_TIMER = Timer.builder(INTEGRATION_TIMER_NAME)
            .tags(TAG_SOURCE, "", TAG_PHASE, "")
            .description("Time needed to finish EMIS ES integration steps")
            .publishPercentiles(0.5, 0.95)
            .publishPercentileHistogram()
            .percentilePrecision(2)
            .register(Metrics.globalRegistry);

    private static final String SOURCE_RESULT_COUNTER_NAME = "pkb_phr_integration_emis_csv_source_results";
    private static final String SOURCE_DOWNLOAD_RESULT_COUNTER_NAME = "pkb_phr_integration_emis_csv_source_download_results";
    private static final String TAG_RESULT = "result";
    private static final Counter SOURCE_RESULT_COUNTER = Counter.builder(SOURCE_RESULT_COUNTER_NAME)
            .tags(TAG_SOURCE, "", TAG_RESULT, "")
            .description("Emis ES source results")
            .register(Metrics.globalRegistry);

    private static final EmisInternalResult EMPTY_FAILED_RESULT = ImmutableEmisInternalResult.of(ImmutableVerifiedEmisFileSetLists.of(emptyList(), emptyList()), EmisStepResult.FAILED);

    private final EmisDownloader emisDownloaderService;
    private final EmisBatchVerifier emisBatchVerifier;
    private final EmisProcessor emisProcessor;
    private final EmisConfigurationUtils emisConfigurationUtils;
    private final EmisEsBatchStateService emisEsBatchStateService;
    private final EmisEsMissingCodeHelper missingCodeHelper;
    private final EmisEsManager emisEsManager;
    private final EmisEsStatusService emisEsStatusService;
    private final EmisOrgHelper emisOrgHelper;
    private final EmisPersonStatHelper emisPersonStatHelper;

    public EmisIntegrationService(@NotNull EmisDownloader emisDownloaderService,
                                  @NotNull EmisBatchVerifier emisBatchVerifier,
                                  @NotNull EmisProcessor emisProcessor,
                                  @NotNull EmisConfigurationUtils emisConfigurationUtils,
                                  @NotNull EmisEsBatchStateService emisEsBatchStateService,
                                  @NotNull EmisEsMissingCodeHelper missingCodeHelper,
                                  @NotNull EmisEsManager emisEsManager,
                                  @NotNull EmisEsStatusService emisEsStatusService,
                                  EmisOrgHelper emisOrgHelper,
                                  EmisPersonStatHelper emisPersonStatHelper) {
        this.emisDownloaderService = emisDownloaderService;
        this.emisBatchVerifier = emisBatchVerifier;
        this.emisProcessor = emisProcessor;
        this.emisConfigurationUtils = emisConfigurationUtils;
        this.emisEsBatchStateService = emisEsBatchStateService;
        this.missingCodeHelper = missingCodeHelper;
        this.emisEsManager = emisEsManager;
        this.emisEsStatusService = emisEsStatusService;
        this.emisOrgHelper = emisOrgHelper;
        this.emisPersonStatHelper = emisPersonStatHelper;
    }

    public EmisStepResult run(long jobId, boolean downloadOnly) {
        try {
            emisEsStatusService.reportStart();
            Path localDirectory = emisConfigurationUtils.getLocalDownloadDirectory();
            LOGGER.info("EMIS integration service for job id:{} started for {}.", jobId, downloadOnly ? "download only" : "full processing" );

            List<? extends ConnectionData> emisSftpConfiguration = emisConfigurationUtils.getEmisConnectionConfiguration();
            emisEsStatusService.reportUniqueTags(emisSftpConfiguration.size());

            Map<String, EmisEsBatchState> batchStateBySftpTag = initializeBatchStateBySftpTag(jobId, emisSftpConfiguration);

            Map<String, EmisStepResult> downloadResults = emisSftpConfiguration.stream()
                    .collect(Collectors.toMap(ConnectionData::uniqueTag,
                            s -> downloadFromServer(s, localDirectory, batchStateBySftpTag.get(s.uniqueTag()).getLastProcessedBatchDatestamp(), jobId)));

            var verifiedBatches = downloadOnly ? EMPTY_FAILED_RESULT : verifyBatches(localDirectory, jobId, batchStateBySftpTag);

            EmisStepResult processingResult = downloadOnly ?  EmisStepResult.FAILED : processBatches(verifiedBatches, jobId, batchStateBySftpTag);

            List<EmisStepResult> stepResults = List.of(getFinalResultOfSteps(downloadResults.values()), verifiedBatches.getStepResult(), processingResult);

            EmisStepResult result = stepResults.stream()
                    .min(comparingInt(EmisStepResult::getValue))
                    .orElse(EmisStepResult.FAILED);


            LOGGER.info("EMIS integration service for job id:{}, executing: {} finished with result: {}, \nfull results: {}", jobId, downloadOnly ? "download only" : "full processing", result, stepResults);
            emisEsStatusService.reportServerResult(result);

            return result;
        } finally {
            try {
                //make sure we always clear down any caches between runs
                missingCodeHelper.logAnyEmisCodeProblemsAndThenClearCaches(jobId);
                emisOrgHelper.jobFinished(jobId);
                emisPersonStatHelper.reportJobFinished();
            } finally {
                emisEsManager.clearState();
                emisEsStatusService.clearState();
                emisOrgHelper.clearState();
                emisPersonStatHelper.clearState();
            }
        }
    }

    private Map<String, EmisEsBatchState> initializeBatchStateBySftpTag(long jobId, List<? extends ConnectionData> emisSftpConfiguration){
        Map<String, EmisEsBatchState> batchStateBySftpTag = emisEsBatchStateService.findByJob(singletonList(jobId)).getOrDefault(jobId, new HashMap<>());
        return emisSftpConfiguration.stream()
                .collect(Collectors.toMap(ConnectionData::uniqueTag,
                        s -> {
                            EmisEsBatchState currentBatchState = batchStateBySftpTag.get(s.uniqueTag());
                            if (currentBatchState == null) {
                                currentBatchState = emisEsBatchStateService.createDefault(jobId, s.uniqueTag());
                                batchStateBySftpTag.put(s.uniqueTag(), currentBatchState);
                            }
                            return currentBatchState;
                        }));
    }

    private EmisStepResult downloadFromServer(ConnectionData connection, Path localDirectory, LocalDate lastProcessedBatchDatestamp, long jobId) {
        LOGGER.info("EMIS download for job {} for server: {} started.", jobId, connection.uniqueTag());
        Path serverDownloadDir = localDirectory.resolve(connection.uniqueTag());
        createDirectory(serverDownloadDir);

        emisEsStatusService.reportCurrentUniqueTag(connection.uniqueTag());

        EmisStepResult result = downloadBatches(connection, serverDownloadDir, lastProcessedBatchDatestamp, jobId);


        LOGGER.info("EMIS download for job {} and server: {} finished with result: {}", jobId, connection.uniqueTag(), result);

        incrementDownloadCounter(connection.uniqueTag(), result);
        return result;
    }

    private EmisInternalResult<VerifiedEmisFileSetLists> verifyBatches(Path localDirectory, long jobId , Map<String, EmisEsBatchState> batchStateBySftpTag) {
        LOGGER.info("EMIS verify for job {} started.", jobId);
        EmisInternalResult<VerifiedEmisFileSetLists> verifiedBatches = EMPTY_FAILED_RESULT;

        try {
            KeyringConfig keyringConfig = emisConfigurationUtils.getDecryptionKey();
            verifiedBatches = recordTime("emisVerify", "allServers", () -> emisBatchVerifier.verifyBatches(localDirectory, keyringConfig, batchStateBySftpTag));

        } catch (Exception e) {
            LOGGER.error("Error during running EMIS verify for job {} , Reason:", jobId, e);
        } finally {
            emisEsManager.clearStatelessAdminPatientOverrides();
        }

        LOGGER.info("EMIS verify for job {} finished with result: {}", jobId, verifiedBatches.getStepResult());
        return verifiedBatches;
    }


    private EmisStepResult processBatches(EmisInternalResult<VerifiedEmisFileSetLists> verifiedBatches, long jobId, Map<String, EmisEsBatchState> batchStateBySftpTag ) {
        LOGGER.info("EMIS process for job {} started.", jobId);
        EmisStepResult processResult = EmisStepResult.FAILED;
        try {
            KeyringConfig keyringConfig = emisConfigurationUtils.getDecryptionKey();
            EmisInternalResult<Map<String, EmisEsBatchState>> processInternalResult = recordTime("emisProcessing", "allServers", () -> emisProcessor.processBatches(verifiedBatches.getContent(), keyringConfig, batchStateBySftpTag));
            processResult = processInternalResult.getStepResult();

        } catch (Exception e) {
            LOGGER.error("Error during running EMIS process for job {}. Reason:", jobId, e);
        } finally {
            emisEsManager.clearStatelessAdminPatientOverrides();
        }

        EmisStepResult finalResult = getFinalResultOfSteps(List.of(verifiedBatches.getStepResult(), processResult));

        LOGGER.info("EMIS process for job {} finished with result: {}", jobId, finalResult);
        return finalResult;
    }

    private EmisStepResult downloadBatches(ConnectionData connection, Path serverDownloadDir, LocalDate lastProcessedBatchDatestamp, long jobId) {
        EmisStepResult downloadResult = EmisStepResult.FAILED;
        try {
            downloadResult = recordTime("emisDownload", connection.uniqueTag(), () -> emisDownloaderService.startDownload(connection, lastProcessedBatchDatestamp, serverDownloadDir));
        } catch (Exception e) {
            LOGGER.error("Error during running EMIS download job {} for connection: {}, Reason:", jobId, connection.uniqueTag(), e);
        }
        return downloadResult;
    }



    private Path createDirectory(Path input) {
        try {
            return Files.createDirectories(input);
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
    }

    @VisibleForTesting
    EmisStepResult getFinalResultOfSteps(Collection<EmisStepResult> stepResults) {
        // FINISHED is the highest value
        return stepResults.stream().min(comparingInt(EmisStepResult::getValue)).orElse(EmisStepResult.FAILED);
    }

    private <T> T recordTime(String phase, String emisConnectionUniqueTag, Callable<T> toMeasure) {
        try {
            return Metrics.timer(INTEGRATION_TIMER_NAME, TAG_SOURCE, emisConnectionUniqueTag, TAG_PHASE, phase).recordCallable(toMeasure);
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }

    private static <T> void incrementDownloadCounter(String emisConnectionUniqueTag, EmisStepResult result) {
        Metrics.counter(SOURCE_DOWNLOAD_RESULT_COUNTER_NAME, TAG_SOURCE, emisConnectionUniqueTag, TAG_RESULT, result.toString()).increment();
    }

}
