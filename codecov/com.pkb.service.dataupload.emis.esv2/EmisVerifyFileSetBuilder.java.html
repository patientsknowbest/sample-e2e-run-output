package com.pkb.service.dataupload.emis.esv2;

import com.pkb.service.dataupload.emis.esv2.exception.EmisFilePatternException;
import com.pkb.service.dataupload.emis.esv2.model.EmisFile;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileType;
import org.jetbrains.annotations.NotNull;

import java.nio.file.Path;
import java.time.LocalDate;
import java.util.Arrays;
import java.util.List;
import java.util.Map;
import java.util.Set;

public class EmisVerifyFileSetBuilder extends AbstractEmisFileSetBuilder{

    public EmisVerifyFileSetBuilder(Path filesetPath, LocalDate datestamp) {
        super(filesetPath, datestamp);
    }

    protected EmisVerifyFileSetBuilder(@NotNull Set<EmisFileType> mandatoryTypes, @NotNull Set<EmisFileType> optionalTypes, Path filesetPath, LocalDate datestamp) {
        super(mandatoryTypes, optionalTypes, filesetPath, datestamp);
    }

    @Override
    protected void addToMap(Map<EmisFileType, List<EmisFile>> map, EmisFile file) {
        boolean typeExists = map.containsKey(file.type());
        if(typeExists){
            throw new EmisFilePatternException(String.format("Duplicate EMIS file type '%s' in file set!", file.type()));
        }else{
            map.put(file.type(), Arrays.asList(file));
        }
    }
}
