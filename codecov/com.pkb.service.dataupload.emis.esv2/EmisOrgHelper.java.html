package com.pkb.service.dataupload.emis.esv2;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.pkb.common.ClearableInternalState;
import com.pkb.dataupload.emis.esv2.EmisArtifactType;
import com.pkb.dataupload.emis.esv2.EmisJobArtifact;
import com.pkb.dataupload.emis.esv2.csv.EmisEsAdminOrganisation;
import com.pkb.service.dataupload.emis.esv2.model.AdminOrganisation;
import com.pkb.service.dataupload.emis.esv2.repository.EmisJobArtifactRepository;
import io.micrometer.core.instrument.Metrics;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Collection;
import java.util.HashSet;
import java.util.Map;
import java.util.Optional;
import java.util.Set;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.ConcurrentMap;
import java.util.function.Function;

public class EmisOrgHelper implements ClearableInternalState {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final ObjectMapper OBJECT_MAPPER = new ObjectMapper();

    private static final String ORG_ID_METRIC_NAME = "pkb_phr_integration_emis_csv_org_identifiers_final";
    private static final String ORG_METRIC_TAG = "codeType";

    private static final String ORG_ID_COUNT_METRIC_NAME = "pkb_phr_integration_emis_csv_org_identifier_types";
    private static final String TYPE_UNKNOWN_ODS = "UnknownODS";
    private static final String TYPE_UNKNOWN_GUID = "UnknownGUID";
    private static final String TYPE_GUID = "GUID";
    private static final String TYPE_ODS = "ODS";

    private final EmisJobArtifactRepository emisJobArtifactRepository;

    private ConcurrentMap<String, String> orgOdsCodeSeen = new ConcurrentHashMap<>();
    private ConcurrentMap<String, String> orgGuidSeen = new ConcurrentHashMap<>();
    private ConcurrentMap<String, String> unknownOdsCodes = new ConcurrentHashMap<>();
    private ConcurrentMap<String, Set<String>> messageAndGuidSeen = new ConcurrentHashMap<>();

    public EmisOrgHelper(EmisJobArtifactRepository emisJobArtifactRepository) {
        this.emisJobArtifactRepository = emisJobArtifactRepository;
    }

    public void logIfOrgWasntSeenYet(EmisEsAdminOrganisation emisOrg, String message, Object... parameters) {
        logIfOrgWasntSeenYet(emisOrg, EmisEsAdminOrganisation::getOrganisationGuid, EmisEsAdminOrganisation::findOdsCode, message, parameters);
    }

    public void logIfOrgWasntSeenYet(AdminOrganisation emisOrg, String message, Object... parameters) {
        logIfOrgWasntSeenYet(emisOrg, AdminOrganisation::organisationGuid, AdminOrganisation::odsCode, message, parameters);
    }

    private <T> void logIfOrgWasntSeenYet(T input, Function<T, String> guidExtractor, Function<T, Optional<String>> odsExtractor, String message, Object... parameters) {
        reportOrgGuid(guidExtractor.apply(input));
        Metrics.counter(ORG_ID_COUNT_METRIC_NAME, ORG_METRIC_TAG, TYPE_UNKNOWN_GUID).increment();
        odsExtractor.apply(input).ifPresent(ods -> {
            reportOdsCode(ods);
            if (!unknownOdsCodes.containsKey(ods)) {
                unknownOdsCodes.put(ods, ods);
                Metrics.counter(ORG_ID_COUNT_METRIC_NAME, ORG_METRIC_TAG, TYPE_UNKNOWN_ODS).increment();
            }
        });

        if (messageAndGuidSeen.computeIfAbsent(message, k -> new HashSet<>()).add(guidExtractor.apply(input))) {
            LOGGER.info(message, parameters);
        }
    }

    public void reportOrgGuid(String orgGuid) {
        if (!orgGuidSeen.containsKey(orgGuid)) {
            Metrics.counter(ORG_ID_COUNT_METRIC_NAME, ORG_METRIC_TAG, TYPE_GUID).increment();
            orgGuidSeen.put(orgGuid, orgGuid);
        }
    }

    public void reportOdsCodes(Collection<String> odsCodes) {
        odsCodes.forEach(this::reportOdsCode);
    }

    public void reportOdsCode(String odsCode) {
        if (!orgOdsCodeSeen.containsKey(odsCode)) {
            Metrics.counter(ORG_ID_COUNT_METRIC_NAME, ORG_METRIC_TAG, TYPE_ODS).increment();
            orgOdsCodeSeen.put(odsCode, odsCode);
        }
    }

    public void jobFinished(long jobId) {
        if (!orgGuidSeen.isEmpty()) {
            Metrics.summary(ORG_ID_METRIC_NAME, ORG_METRIC_TAG, TYPE_GUID).record(orgGuidSeen.size());
        }
        if (!orgOdsCodeSeen.isEmpty()) {
            Metrics.summary(ORG_ID_METRIC_NAME, ORG_METRIC_TAG, TYPE_ODS).record(orgOdsCodeSeen.size());
        }
        if (!unknownOdsCodes.isEmpty()) {
            Metrics.summary(ORG_ID_METRIC_NAME, ORG_METRIC_TAG, TYPE_UNKNOWN_ODS).record(unknownOdsCodes.size());
        }
        if (!orgGuidSeen.isEmpty() || !orgOdsCodeSeen.isEmpty() || !unknownOdsCodes.isEmpty()) {
            try {
                String resultJson = OBJECT_MAPPER.writer().writeValueAsString(Map.of(TYPE_GUID, orgGuidSeen.keySet(), TYPE_ODS, orgOdsCodeSeen.keySet(), TYPE_UNKNOWN_ODS, unknownOdsCodes.keySet()));
                emisJobArtifactRepository.save(new EmisJobArtifact(jobId, EmisArtifactType.ORG_RESULT, resultJson));
            } catch (Exception e) {
                throw new RuntimeException(e);
            }
        }
        clearState();
    }

    @Override
    public void clearState() {
        orgOdsCodeSeen.clear();
        orgGuidSeen.clear();
        messageAndGuidSeen.clear();
        unknownOdsCodes.clear();
    }
}
