package com.pkb.service.dataupload.emis.esv2;

import com.pkb.service.dataupload.emis.esv2.repository.EmisJobArtifactRepository;
import com.pkb.util.CsvStreamService;

import javax.persistence.Tuple;
import javax.transaction.Transactional;
import java.io.Writer;
import java.util.List;
import java.util.Map;
import java.util.function.Function;

import static java.util.stream.Collectors.toMap;

public class EmisJobArtifactService {

    private static final List<String> UNMATCHED_CODES_COLUMN_HEADERS = List.of(
            "external_id", "code_id", "code_system", "code_description_id", "term", "category_description", //populated columns
            "datatype_mapping", "privacy_mapping", "exclude", "loinc_test_translation" //blank placeholders
    );
    private static final Function<String, Function<Tuple, Object>> COLUMN_SPECIFIC_TUPLE_LOOKUP_FUNCTION = columnName -> tuple -> tuple.get(columnName);
    private static final Map<String, Function<Tuple, Object>> UNMATCHED_CODES_COLUMN_VALUE_PROVIDER = UNMATCHED_CODES_COLUMN_HEADERS.stream()
            .collect(toMap(columnName -> columnName, COLUMN_SPECIFIC_TUPLE_LOOKUP_FUNCTION));

    private EmisJobArtifactRepository emisJobArtifactRepository;
    private CsvStreamService csvStreamService;

    public EmisJobArtifactService(EmisJobArtifactRepository emisJobArtifactRepository, CsvStreamService csvStreamService) {
        this.emisJobArtifactRepository = emisJobArtifactRepository;
        this.csvStreamService = csvStreamService;
    }

    @Transactional
    public Void retrieveUnmatchedEmisCodesAndWriteToOutputAsCsv(Writer outputWriter) {
        csvStreamService.streamToCsvConsumer(outputWriter, UNMATCHED_CODES_COLUMN_HEADERS, UNMATCHED_CODES_COLUMN_VALUE_PROVIDER, true)
                .accept(emisJobArtifactRepository.getMostRecentUnmatchedCodes());
        return null;
    }
}
