    /**
     * @return <code>true</code> when processing was successful without any error, <code>false</code> otherwise.
     */
    boolean processBatch(@NotNull VerifiedEmisFileSftpSet batch, @NotNull KeyringConfig keyringConfig, @NotNull Long batchId) {
        Map<EmisFileType, EmisEsProcessedRow> previousStates = emisEsProcessedRowService.findByJob(batchId);
        try {
            boolean processingResult = batch.verifiedEmisFileSet()
                    .files()
                    .entrySet()
                    .stream()
                    .collect(Collectors.groupingBy(file -> PROCESSING_GROUPS.get(file.getKey())))
                    .entrySet()
                    .stream()
                    .sorted(Map.Entry.comparingByKey(Comparator.naturalOrder()))
                    .map(Map.Entry::getValue)
                    .map(processingGroup -> processingGroup.stream()
                            .map(typeAndFile -> processFile(typeAndFile.getKey(), typeAndFile.getValue(), keyringConfig,
                                    batchId, previousStates.get(typeAndFile.getKey()), batch))
                            .map(t -> updateStateMap(t, previousStates))
                            .reduce((l, r) -> l && r) // overall result := result1 && result2 && .... && resultN
                            .orElse(true))
                    .filter(result -> !result)
                    .findFirst() // short-circuits, so processing stops when the first processing group fails
                    .orElse(true);

            //before clearing any caches , perform a postProcess step if the processing went OK (currently only needed for CRP processing)
            if (processingResult) {
                boolean postProcessResult = emisCsvProcessorProvider.get().values().stream().map(p -> postProcess(p, previousStates.get(p.getSupportedType()))).reduce((l, r) -> l && r).orElse(true);
                if (postProcessResult) {
                    emisEsProcessedRowService.deleteStatesForBatch(batchId);
                }
                processingResult &= postProcessResult;
            }
            return processingResult;
        } finally {
            batchProcessingCaches.forEach(c -> clearCache(c, previousStates));
        }
    }

    private static final Set<EmisFileType> STATELESS_PROCESSING_REQUIRED = Sets.immutableEnumSet(ADMIN_ORGANISATION, CODING_CLINICAL_CODES, CODING_DRUG_CODES, ADMIN_PATIENT);
    private static final Set<EmisFileType> SCAN_FOR_MISSING_CODES = Sets.immutableEnumSet(CARE_RECORD_OBSERVATION, PRESCRIBING_DRUG_RECORD);

    /**
     * @return <code>true</code> when processing was successful without any error, <code>false</code> otherwise.
     */
    boolean processBatchValidateEmisCodes(@NotNull VerifiedEmisFileSet batch, @NotNull KeyringConfig keyringConfig, @NotNull Path localDirectory) {
        boolean result = true;
        for (EmisFileType actual : STATELESS_PROCESSING_REQUIRED) {
            VerifiedEmisFile codesEmisFile = batch.files().get(actual);
            result &= processCodesStatelessly(codesEmisFile, keyringConfig, localDirectory);
        }

        for (EmisFileType actual : SCAN_FOR_MISSING_CODES) {
            VerifiedEmisFile observationsEmisFile = batch.files().get(actual);
            result &= processCodeValidation(observationsEmisFile, keyringConfig, localDirectory);
        }
        return result;
    }

    private void clearCache(BatchProcessingCache cache, Map<EmisFileType, EmisEsProcessedRow> previousStates) {
        EmisEsProcessedRow previousState = previousStates.get(cache.getSupportedType());
        if (previousState != null) {
            cache.clearCached(previousState.getBatchId());
        }
    }

    private boolean updateStateMap(Tuple3<Boolean, EmisFileType, EmisEsProcessedRow> input, Map<EmisFileType, EmisEsProcessedRow> states) {
        states.put(input._2(), input._3);
        return input._1;
    }

    private boolean postProcess(BatchDataProcessor<BaseEmisEsDomainModel> processor, EmisEsProcessedRow previousFileState) {
        boolean result = true;
        if (processor.isPostProcessingRequired()) {
            LOGGER.info("Starting post processing {}", processor.interfaceClass().getSimpleName());
            try {
                result = recordTime(POST_PROCESSOR_TIMER_NAME, processor.getClass().getSimpleName(), () -> processor.postProcess(previousFileState));
                LOGGER.info("Successfully post processed {}", processor.interfaceClass().getSimpleName());
            } catch (Exception e) {
                LOGGER.error("Post processing failed!", e);
                result = false;
            }
        }
        return result;
    }

    @VisibleForTesting
    Tuple3<Boolean, EmisFileType, EmisEsProcessedRow> processFile(EmisFileType emisFileType,
                                                                  VerifiedEmisFile emisFile,
                                                                  KeyringConfig keyringConfig,
                                                                  Long batchId,
                                                                  EmisEsProcessedRow previousFileState,
                                                                  VerifiedEmisFileSftpSet parentSftpSet) {
        return recordTime(FILE_PROCESSING_TIMER_NAME, emisFileType.name(), () -> {
            LocalDate datestamp = parentSftpSet.dateStamp();

            Either<Exception, EmisEsProcessedRow> validatedStateOrError = validateAndInitFileStateOrError(emisFileType, emisFile, batchId, parentSftpSet, datestamp, previousFileState);
            if (validatedStateOrError.isLeft()) {
                LOGGER.error("Error validating: ", validatedStateOrError.getLeft());
                return new Tuple3<>(false, emisFileType, previousFileState);
            }
            EmisEsProcessedRow fileState = validatedStateOrError.get();

            if (configuration.isEmisEsStateResumeEnabled() && isPreviouslyProcessed(fileState, emisFile)) {
                LOGGER.info("Skip processing of {} as it was previously processed. Hash: {}", emisFile.relativePath(), fileState.getFileHash());
                BatchDataProcessor<BaseEmisEsDomainModel> selected = emisCsvProcessorProvider.getProcessor(emisFileType);
                incrementCounter(FILE_SKIP_COUNTER_NAME, selected);
                return new Tuple3<>(true, emisFileType, fileState);
            } else {
                return buildStreamAndProcess(emisFileType, emisFile, keyringConfig, parentSftpSet, fileState);
            }
        });
    }

    private Either<Exception, EmisEsProcessedRow> validateAndInitFileStateOrError(EmisFileType emisFileType, VerifiedEmisFile emisFile, Long batchId, VerifiedEmisFileSftpSet parentSftpSet, LocalDate datestamp, EmisEsProcessedRow fileState) {
        if (fileState == null) {
            return Either.right(emisEsProcessedRowService.createDefault(batchId, emisFileType, datestamp));
        } else if (!fileState.getFilesetDatestamp().equals(datestamp)) {
            //previous state is for a different datestamp. This is only ok if the previous state is for a date
            // we've since decided to skip (e.g. EMIS sent us broken files, we partially parsed, now want to skip)
            if (emisEsDatestampSkipService.shouldSkipDateStamp(fileState.getFilesetDatestamp(), parentSftpSet.sftpUniqueId())) {
                LOGGER.info("Skipping previous state for {} due to datestamp skip", emisFile.relativePath());
                emisEsProcessedRowService.clearProgress(fileState);
                fileState.setFilesetDatestamp(datestamp);
                return Either.right(fileState);
            } else {
                return Either.left(new IllegalStateException("Abort processing, something is wrong, datestamp does not match previous run. " +
                        "Current: " + datestamp + " Previous: " + fileState.getFilesetDatestamp() + " File: " + emisFile.relativePath()));

            }
        }
        return Either.right(fileState);
    }

    @NotNull
    @VisibleForTesting
    Tuple3<Boolean, EmisFileType, EmisEsProcessedRow> buildStreamAndProcess(EmisFileType emisFileType, VerifiedEmisFile emisFile, KeyringConfig keyringConfig, VerifiedEmisFileSftpSet parentSftpSet, EmisEsProcessedRow fileState) {
        LOGGER.info("Starting processing {} for server {}", emisFile, parentSftpSet.sftpUniqueId());
        BatchDataProcessor<BaseEmisEsDomainModel> selected = emisCsvProcessorProvider.getProcessor(emisFileType);
        if (selected == null) {
            LOGGER.error("Could not find suitable data type processor for {} ", emisFileType);
            return new Tuple3<>(false, emisFileType, fileState);
        }

        RowId<Long> previousProcessingState = configuration.isEmisEsStateResumeEnabled() ? convert(fileState) : null;

        Path filePath = parentSftpSet.serverDownloadDir().resolve(emisFile.relativePath());
        try (var entityStream = csvReader.buildStream(filePath, keyringConfig, selected.interfaceClass(), emisFile, previousProcessingState)) {
            emisEsStatusService.reportCurrentFile(selected.getProcessorType(), emisFile);

            String currentHash = emisFile.gpgHash();
            LOGGER.info("Current file {} for server {} hash: {}", emisFile.relativePath(), parentSftpSet.sftpUniqueId(), currentHash);

            if (StringUtils.isBlank(fileState.getFileHash())) {
                fileState = setHashAndSave(fileState, currentHash);
            } else if (!currentHash.equals(fileState.getFileHash())) {

                LOGGER.error("Abort processing, something is wrong, file hash does not match previous run. Current: {}, Previous: {}, File: {}", currentHash,
                        fileState.getFileHash(),
                        emisFile.relativePath());
                return new Tuple3<>(false, emisFileType, fileState);
            }

            Tuple2<Boolean, EmisEsProcessedRow> processResult = selected.process(entityStream.getRecordStream(), fileState);
            fileState = processResult._2;
            if (processResult._1) {
                if (!selected.isPostProcessingRequired()) {
                    fileState.setFullyProcessed(true);
                    emisEsProcessedRowService.save(fileState);
                }
                LOGGER.info("Successfully processed {} for server {}", emisFileType, parentSftpSet.sftpUniqueId());
                incrementCounter(SUCCESS_COUNTER_NAME, selected);
            } else {
                LOGGER.error("Failed to process {} for server {}", emisFileType, parentSftpSet.sftpUniqueId());
                incrementCounter(FAILURE_COUNTER_NAME, selected);
            }

            return new Tuple3<>(processResult._1, emisFileType, processResult._2);

        } catch (Exception e) {
            LOGGER.error("Failed streaming EMIS CSV {}", filePath, e);
            return new Tuple3<>(false, emisFileType, fileState);
        }
    }

    private EmisEsProcessedRow setHashAndSave(EmisEsProcessedRow fileState, String currentHash) {
        fileState.setFileHash(currentHash);
        fileState = emisEsProcessedRowService.save(fileState);
        return fileState;
    }


    /**
     * Used only for scanning future batches, loading the codes into our DB so that we can check EMIS observations for missing codes.
     * Temporary code.
     */
    @VisibleForTesting
    boolean processCodesStatelessly(VerifiedEmisFile emisFile, KeyringConfig keyringConfig, Path localDirectory) {
        LOGGER.info("Stateless processing starting for {}", emisFile);
        BatchDataProcessor<BaseEmisEsDomainModel> codeProcessor = statelessProcessors.get(emisFile.type());
        if (codeProcessor == null) {
            LOGGER.error("Stateless processing: no stateless processor of type {}", emisFile.type());
            return false;
        }

        Path filePath = localDirectory.resolve(emisFile.relativePath());
        try (var entityStream = csvReader.buildStream(filePath, keyringConfig, codeProcessor.interfaceClass(), emisFile, null)) {
            emisEsStatusService.reportCurrentFile(codeProcessor.getProcessorType(), emisFile);

            LOGGER.info("Stateless processing: current file {}", emisFile.relativePath());

            EmisEsProcessedRow unusedFileState = new EmisEsProcessedRow();
            Tuple2<Boolean, EmisEsProcessedRow> processResult = codeProcessor.process(entityStream.getRecordStream(), unusedFileState);

            if (processResult._1) {
                LOGGER.info("Stateless processing: successfully processed {}", codeProcessor.getSupportedType());
                incrementCounter(SUCCESS_COUNTER_NAME, codeProcessor);
            } else {
                LOGGER.error("Stateless processing: failed to process {}", codeProcessor.getSupportedType());
                incrementCounter(FAILURE_COUNTER_NAME, codeProcessor);
            }

            return processResult._1;
        } catch (Exception e) {
            LOGGER.error("Stateless processing: failed streaming EMIS CSV {}", filePath, e);
            return false;
        }
    }

    /**
     * Used only for scanning future batches, loading the codes into our DB so that we can check EMIS observations for missing codes.
     * Temporary code.
     */
    @VisibleForTesting
    boolean processCodeValidation(VerifiedEmisFile emisFile, KeyringConfig keyringConfig, Path localDirectory) {
        Path filePath = localDirectory.resolve(emisFile.relativePath());

        LOGGER.info("Code validation: starting processing {}", emisFile);
        AbstractBatchCodeValidator<BaseEmisEsDomainModel> codeValidator = codeValidators.get(emisFile.type());
        try (var entityStream = csvReader.buildStream(filePath, keyringConfig, codeValidator.interfaceClass(), emisFile,null)) {
            emisEsStatusService.reportCurrentFile(codeValidator.getProcessorType(), emisFile);
            LOGGER.info("Code validation: current file {}", emisFile.relativePath());

            codeValidator.process(entityStream.getRecordStream());

            LOGGER.info("Code validation: successfully processed validation {}", emisFile.type());

            return true;
        } catch (Exception e) {
            LOGGER.error("Code validation: failed streaming EMIS CSV {}", filePath, e);
            return false;
        }
    }

    private boolean isPreviouslyProcessed(EmisEsProcessedRow fileState, VerifiedEmisFile emisFile) {
        return fileState.isFullyProcessed() && emisFile.type().ordinal() == fileState.getEmisFileType();
    }

    private RowId<Long> convert(EmisEsProcessedRow fileState) {
        if (fileState.getLastProcessedRow() == null) {
            return null;
        }
        return new RowId<>(fileState.getLastProcessedRow());
    }

    private <T> T recordTime(String timerName, String name, Callable<T> toMeasure) {
        try {
            return Metrics.timer(timerName, METRIC_DATATYPE_TAG, name).recordCallable(toMeasure);
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }
