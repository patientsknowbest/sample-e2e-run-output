package com.pkb.service.dataupload.emis.esv2;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.dataupload.emis.esv2.EmisEsBatchState;
import com.pkb.service.dataupload.emis.esv2.repository.EmisEsBatchStateRepository;
import org.jetbrains.annotations.NotNull;
import org.springframework.data.domain.Sort;

import javax.transaction.Transactional;
import java.time.LocalDate;
import java.util.Collection;
import java.util.Collections;
import java.util.List;
import java.util.Map;
import java.util.Optional;

import static java.util.function.Function.identity;
import static java.util.stream.Collectors.groupingBy;
import static java.util.stream.Collectors.toMap;
import static org.apache.commons.collections4.CollectionUtils.isEmpty;

public class EmisEsBatchStateService {
    private final EmisEsBatchStateRepository emisEsBatchStateRepository;
    private final DateTimeService dateTimeService;

    public EmisEsBatchStateService(EmisEsBatchStateRepository emisEsBatchStateRepository, DateTimeService dateTimeService) {
        this.emisEsBatchStateRepository = emisEsBatchStateRepository;
        this.dateTimeService = dateTimeService;
    }

    public Optional<EmisEsBatchState> findByJobAndSftpTag(long jobId, String sftpUniqueTag) {
        return emisEsBatchStateRepository.findByJobAndSftpTag(jobId, sftpUniqueTag);
    }

    public List<EmisEsBatchState> findAll() {
        return emisEsBatchStateRepository.findAll(Sort.by("sftpUniqueId"));
    }

    public Map<Long, Map<String, EmisEsBatchState>> findByJob(Collection<Long> jobIds) {
        if (isEmpty(jobIds)) {
            return Collections.emptyMap();
        }
        return emisEsBatchStateRepository.findByJob(jobIds)
                .stream()
                .collect(groupingBy(EmisEsBatchState::getJobId, toMap(EmisEsBatchState::getSftpUniqueId, identity())));
    }

    @Transactional
    public int moveBatchStateToNewJob(long oldJobId, long newJobId) {
        return emisEsBatchStateRepository.moveBatchStateToNewJob(oldJobId, newJobId);
    }

    public EmisEsBatchState save(@NotNull EmisEsBatchState batch) {
        return updateRecordedDateAndSave(batch);
    }

    public EmisEsBatchState createDefault(long jobId, String sftpUniqueId) {
        EmisEsBatchState result = new EmisEsBatchState();
        result.setJobId(jobId);
        result.setSftpUniqueId(sftpUniqueId);
        result.setLastProcessedBatchGuid("");
        result.setLastProcessedBatchDatestamp(LocalDate.EPOCH);

        return updateRecordedDateAndSave(result);
    }

    @NotNull
    private EmisEsBatchState updateRecordedDateAndSave(EmisEsBatchState result) {
        result.setRecordedDate(dateTimeService.now());
        return emisEsBatchStateRepository.save(result);
    }
}
