package com.pkb.service.dataupload.emis.esv2;

import com.google.common.collect.ImmutableList;
import com.google.common.collect.ImmutableMap;
import com.pkb.common.config.PhrConfig;
import com.pkb.dataupload.emis.esv2.csv.EmisEsAdminOrganisation;
import com.pkb.entities.enums.OrgExternalIdType;
import com.pkb.institute.OrgMergeHelper;
import com.pkb.institute.entity.Org;
import com.pkb.service.dataupload.emis.esv2.model.AdminPatient;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.ByteArrayInputStream;
import java.io.IOException;
import java.util.Set;

public class EmisEsMutableWhiteList extends AbstractEmisEsWhitelistService {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final EmisEsImmutableWhitelist emisEsImmutableWhitelist;
    private ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> odsCodeToNhsAndPatientNumbers;
    private boolean isOverridden;

    public EmisEsMutableWhiteList(EmisEsImmutableWhitelist emisEsImmutableWhitelist, PhrConfig config) {
        super(config);
        this.emisEsImmutableWhitelist = emisEsImmutableWhitelist;
        this.isOverridden = false;
        this.odsCodeToNhsAndPatientNumbers = ImmutableMap.of();
    }

    @Override
    public boolean isAllowedPatient(@NotNull AdminPatient adminPatient, @NotNull EmisEsAdminOrganisation org) {
        if (isOverridden) {
            LOGGER.warn("Checking overridden whitelist for patient [pn:{} nid:{}], ods:[{}]", adminPatient.patientNumber(), adminPatient.nhsNumber(), org.getOdsCode());
            return emisEsImmutableWhitelist.isAllowedPatient(getMap(), adminPatient, org);
        }
        LOGGER.warn("Checking normal whitelist for patient [pn:{} nid:{}], ods:[{}]", adminPatient.patientNumber(), adminPatient.nhsNumber(), org.getOdsCode());
        return emisEsImmutableWhitelist.isAllowedPatient(adminPatient, org);
    }

    @Override
    public boolean isAllowedPatient(@NotNull AdminPatient adminPatient, @NotNull OrgMergeHelper orgMergeHelper) {
        if (isOverridden) {
            LOGGER.warn("Checking overridden whitelist for patient [pn:{} nid:{}], ods:[{}]", adminPatient.patientNumber(), adminPatient.nhsNumber(), orgMergeHelper.getOdsCode());
            return emisEsImmutableWhitelist.isAllowedPatient(getMap(), adminPatient, orgMergeHelper);
        }
        LOGGER.warn("Checking normal whitelist for patient [pn:{} nid:{}], ods:[{}]", adminPatient.patientNumber(), adminPatient.nhsNumber(), orgMergeHelper.getOdsCode());
        return emisEsImmutableWhitelist.isAllowedPatient(adminPatient, orgMergeHelper);
    }

    @Override
    public boolean isFilteredByPatientWhitelist(@NotNull Org org) {
        if (isOverridden) {
            LOGGER.warn("Checking overridden whitelist for org ods:[{}]", org.getExternalIdByType(OrgExternalIdType.ODS));
            return emisEsImmutableWhitelist.isFilteredByPatientWhitelist(getMap(), org);
        }
        LOGGER.warn("Checking normal whitelist for org ods:[{}]", org.getExternalIdByType(OrgExternalIdType.ODS));
        return emisEsImmutableWhitelist.isFilteredByPatientWhitelist(org);
    }

    @Override
    public Set<String> getOdsCodes() {
        if (isOverridden) {
            return emisEsImmutableWhitelist.getOdsCodes(getMap());
        }
        return emisEsImmutableWhitelist.getOdsCodes();
    }

    @Override
    public void setYamlConfig(String configYaml) {
        try {
            odsCodeToNhsAndPatientNumbers = readYamlConfig(new ByteArrayInputStream(configYaml.getBytes()));
            isOverridden = true;
            LOGGER.warn("Set whitelisting for these Orgs (ods): {}", String.join("", odsCodeToNhsAndPatientNumbers.keySet()));
        } catch (IOException e) {
            throw new RuntimeException("Failed setting EMIS whitelist config!", e);
        }
    }

    @Override
    public void clearState() {
        isOverridden = false;
        odsCodeToNhsAndPatientNumbers = ImmutableMap.of();
    }

    @Override
    protected ImmutableMap<String, ImmutableMap<String, ImmutableList<String>>> getMap() {
        return odsCodeToNhsAndPatientNumbers;
    }
}
