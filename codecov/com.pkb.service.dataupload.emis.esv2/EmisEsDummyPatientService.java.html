package com.pkb.service.dataupload.emis.esv2;

import com.pkb.dataupload.emis.esv2.EmisEsDummyPatient;
import com.pkb.service.dataupload.emis.esv2.repository.EmisEsDummyPatientRepository;
import org.jetbrains.annotations.NotNull;

import java.util.Collections;
import java.util.Map;
import java.util.Set;

import static java.util.function.Function.identity;
import static java.util.stream.Collectors.toMap;

public class EmisEsDummyPatientService {

    private final EmisEsDummyPatientRepository repository;

    public EmisEsDummyPatientService(EmisEsDummyPatientRepository repository) {
        this.repository = repository;
    }

    public boolean isPatientDummy(@NotNull Long personId) {
        return repository.findByPersonId(personId).isPresent();
    }

    public Map<Long, Boolean> arePatientsDummy(@NotNull Set<Long> personIds) {
        if (personIds.isEmpty()) {
            return Collections.emptyMap();
        }

        var personIdToEmisEsDummyPatient = repository.findAllByPersonIdIn(personIds).stream()
                .collect(toMap(EmisEsDummyPatient::getPersonId, identity()));

        return personIds.stream()
                .collect(toMap(identity(), personIdToEmisEsDummyPatient::containsKey));
    }

    public EmisEsDummyPatient createDummyPatientRecord(Long personId, String adminPatientGuid) {
        EmisEsDummyPatient dummyPatient = new EmisEsDummyPatient();
        dummyPatient.setPersonId(personId);
        dummyPatient.setPatientGuid(adminPatientGuid);
        return repository.save(dummyPatient);
    }
}
