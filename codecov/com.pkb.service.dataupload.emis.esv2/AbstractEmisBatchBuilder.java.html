package com.pkb.service.dataupload.emis.esv2;

import com.google.common.collect.ImmutableList;
import com.pkb.common.config.PhrConfig;
import com.pkb.service.dataupload.emis.esv2.converters.EmisLsEntryToEmisFileConverter;
import com.pkb.service.dataupload.emis.esv2.model.EmisFile;
import com.pkb.service.dataupload.emis.esv2.model.EmisFileSet;
import io.vavr.control.Try;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.nio.file.Path;
import java.time.LocalDate;
import java.util.Collection;
import java.util.Comparator;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;

import static com.pkb.service.dataupload.emis.esv2.model.EmisFileSet.DIRECTORY_DATESTAMP_FORMAT;

public abstract class AbstractEmisBatchBuilder {

    protected static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final EmisLsEntryToEmisFileConverter emisLsEntryToEmisFileConverter;

    private final PhrConfig config;

    protected final EmisEsDatestampSkipService emisEsDatestampSkipService;

    private final Map<LocalDate, AbstractEmisFileSetBuilder> batch = new HashMap<>();

    public AbstractEmisBatchBuilder(@NotNull EmisLsEntryToEmisFileConverter emisLsEntryToEmisFileConverter, PhrConfig config, EmisEsDatestampSkipService emisEsDatestampSkipService) {
        this.emisLsEntryToEmisFileConverter = emisLsEntryToEmisFileConverter;
        this.config = config;
        this.emisEsDatestampSkipService = emisEsDatestampSkipService;
    }

    @NotNull
    public List<EmisFileSet> buildFilesets(@NotNull Collection<EmisLsEntry> remoteList, String sftpUniqueId) {
        return remoteList.stream()
                .map(emisLsEntry ->  makeFileset(emisLsEntry, sftpUniqueId))
                .flatMap(Optional::stream)
                .sorted(Comparator.comparing(EmisFileSet::dateStamp))
                .collect(ImmutableList.toImmutableList());
    }

    protected Optional<EmisFileSet> makeFileset(EmisLsEntry dirEntry, String sftpUniqueId) {
        if (!dirEntry.isDir()) {
            LOGGER.error("Found unexpected non-directory file, skipping {} ", dirEntry.getFilename());
            return Optional.empty();
        }

        Optional<LocalDate> datestampIfValid = getDatestampIfValid(dirEntry);
        return Try.of(() -> datestampIfValid.map(datestamp -> buildEmisFileset(dirEntry, datestamp)))
                .getOrElseGet(throwable -> {
                    if (emisEsDatestampSkipService.shouldSkipDateStamp(datestampIfValid.orElseThrow(), sftpUniqueId)) {
                        LOGGER.warn("Exception processing fileset {} but skipping as shouldSkipDateStamp", dirEntry.getFilename(), throwable);
                        return Optional.empty();
                    }
                    throw new RuntimeException("unable to make fileset for " + dirEntry.getFilename(), throwable);
                });
    }

    @NotNull
    protected EmisFileSet buildEmisFileset(EmisLsEntry dirEntry, LocalDate datestamp) {
        if (batch.containsKey(datestamp)) {
            throw new RuntimeException("Processing same directory twice, directory = " + dirEntry.getFilename());
        }

        AbstractEmisFileSetBuilder emisFileSetBuilder = getEmisFileSetBuilder(dirEntry.getRelativePath(), datestamp);

        addFilesFromDirectory(dirEntry, datestamp, emisFileSetBuilder);

        batch.put(datestamp, emisFileSetBuilder);
        return emisFileSetBuilder.build();
    }

    protected Optional<LocalDate> getDatestampIfValid(EmisLsEntry entry) {
        return Try.of(() -> LocalDate.parse(entry.getFilename(), DIRECTORY_DATESTAMP_FORMAT))
                .toEither()
                .peekLeft(e -> {
                    LOGGER.error("Found unexpected non-datestamp directory {}", entry.getFilename());
                    if (config.getEmisEsHaltForUnexpectedSftpFiles()) {
                        throw new RuntimeException("Unable to parse EmisLsDirEntry datestamp " + entry.getFilename(), e);
                    }
                })
                .toJavaOptional();
    }

    protected void addFilesFromDirectory(EmisLsEntry dirEntry, LocalDate datestamp, AbstractEmisFileSetBuilder emisFileSetBuilder) {

        List<EmisFile> batchFiles = dirEntry.getChildren().stream()
                .filter(lsEntry -> nonRegularFileHandler(lsEntry,datestamp))
                .map(this::convert)
                .flatMap(Optional::stream)
                .peek(emisFile ->  checkFileTimestamp(emisFile, datestamp))
                .collect(Collectors.toUnmodifiableList());

        validateConsistentBulksOrDeltas(batchFiles, datestamp);

        batchFiles.forEach(emisFileSetBuilder::add);
    }

    protected Optional<EmisFile> convert(EmisLsEntry lsEntry) {
        try {
            return Optional.of(emisLsEntryToEmisFileConverter.convert(lsEntry));
        } catch (Exception e) {
            LOGGER.error("Can't convert '{}' to EMIS file! Reason:", lsEntry, e);
            return Optional.empty();
        }
    }

    protected abstract void validateConsistentBulksOrDeltas(List<EmisFile> batchFiles, LocalDate datestamp);

    protected abstract boolean nonRegularFileHandler(EmisLsEntry lsEntry, LocalDate datestamp);

    protected abstract AbstractEmisFileSetBuilder getEmisFileSetBuilder(Path filesetPath, LocalDate datestamp);

    protected abstract void checkFileTimestamp(EmisFile emisFile, LocalDate datestamp);

}
