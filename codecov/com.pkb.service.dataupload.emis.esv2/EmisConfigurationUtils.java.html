package com.pkb.service.dataupload.emis.esv2;

import static org.apache.commons.lang3.StringUtils.isBlank;

import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.Set;
import java.util.stream.Collectors;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.dataformat.yaml.YAMLMapper;
import com.fasterxml.jackson.datatype.guava.GuavaModule;
import com.fasterxml.jackson.datatype.jdk8.Jdk8Module;
import com.fasterxml.jackson.datatype.jsr310.JavaTimeModule;
import com.google.common.annotations.VisibleForTesting;

import com.pkb.common.config.PhrConfig;
import com.pkb.service.dataupload.emis.esv2.gcp.GcpConnectionData;
import com.pkb.service.dataupload.emis.esv2.sftp.SftpConnectionData;

import name.neuhalfen.projects.crypto.bouncycastle.openpgp.keys.callbacks.KeyringConfigCallbacks;
import name.neuhalfen.projects.crypto.bouncycastle.openpgp.keys.keyrings.KeyringConfig;
import name.neuhalfen.projects.crypto.bouncycastle.openpgp.keys.keyrings.KeyringConfigs;

public class EmisConfigurationUtils {

    private final PhrConfig configuration;
    private static final ObjectMapper YAML_MAPPER = new YAMLMapper().registerModule(new GuavaModule()).registerModules(new Jdk8Module(), new JavaTimeModule());

    public EmisConfigurationUtils(PhrConfig configuration) {
        this.configuration = configuration;
    }

    public Path getLocalDownloadDirectory() {
        String downloadDir = configuration.getEmisEsDownloadBaseDir();
        if (isBlank(downloadDir)) {
            throw new RuntimeException("Emis download directory can't be null / empty!");
        }
        Path result = getPath(downloadDir);
        if (!Files.isDirectory(result)) {
            throw new RuntimeException("Emis download directory '" + downloadDir + "' must exist!");
        }
        return result;
    }

    private static <T> List<T> readEmisList(Path path, Class<T> type) {
        try {
            return YAML_MAPPER.readValue(Files.readString(path), YAML_MAPPER.getTypeFactory().constructParametricType(List.class, type));
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
    }

    List<ConnectionData> getEmisConnectionConfiguration() {
        List<ConnectionData> result = new ArrayList<>();
        configuration.getEmisEsGcpConfigYamlPath()
                .map(path -> readEmisList(path, GcpConnectionData.class))
                .ifPresentOrElse(
                        gcpConnections -> result.addAll(gcpConnections),
                        () -> {
                            List<SftpConnectionData> sftpConnections = readEmisList(configuration.getEmisEsSftpConfigYamlPath(), SftpConnectionData.class);
                            sftpConnections.forEach(sftpConnectionData -> checkFileExists(getPath(sftpConnectionData.privateKey())));
                            result.addAll(sftpConnections);
                        });

        Set<String> dupedTags = result.stream().collect(Collectors.groupingBy(ConnectionData::uniqueTag))
                .entrySet().stream().filter(e -> e.getValue().size() > 1)
                .map(Map.Entry::getKey)
                .collect(Collectors.toSet());

        if (!dupedTags.isEmpty()) {
            throw new IllegalStateException("The following tags are not uniqe in the configuration: " + dupedTags);
        }

        return result;
    }

    KeyringConfig getDecryptionKey() {
        Path privateKeyPath = checkFileExists(getPath(configuration.getEmisEsPgpPrivateKeyFile()));
        return KeyringConfigs.withKeyRingsFromFiles(privateKeyPath.toFile(),
                privateKeyPath.toFile(), KeyringConfigCallbacks.withPassword(configuration.getEmisEsPgpPrivateKeyFilePassphrase()));
    }

    @VisibleForTesting
    Path getPath(String path) {
        return Paths.get(path);
    }

    @VisibleForTesting
    Path getPath(Path path) {
        return path;
    }

    private Path checkFileExists(Path filePath) {
        if (!Files.isRegularFile(filePath)) {
            throw new RuntimeException("File [" + filePath + "] must exist!");
        }
        return filePath;
    }
}
