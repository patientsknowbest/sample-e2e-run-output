package com.pkb.service.dataupload.emis.esv2;

import com.fasterxml.jackson.databind.DeserializationFeature;
import com.fasterxml.jackson.dataformat.csv.CsvMapper;
import com.fasterxml.jackson.datatype.jdk8.Jdk8Module;
import com.fasterxml.jackson.datatype.jsr310.JavaTimeModule;

public class EmisCsvMapperFactory {

    private EmisCsvMapperFactory() {
    }

    public static CsvMapper getEmisCsvMapper() {
        CsvMapper csvMapper = new CsvMapper();
        csvMapper.registerModule(new Jdk8Module());
        csvMapper.registerModule(new JavaTimeModule());
        csvMapper.enable(DeserializationFeature.ACCEPT_EMPTY_STRING_AS_NULL_OBJECT);
        return csvMapper;
    }
}
