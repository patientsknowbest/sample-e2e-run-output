package com.pkb.service.dataupload.emis.esv2;

import com.pkb.dataupload.emis.esv2.EmisEsDatestampSkip;
import com.pkb.service.dataupload.emis.esv2.repository.EmisEsDatestampSkipRepository;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.time.LocalDate;
import java.util.Optional;

import static com.pkb.service.dataupload.emis.esv2.model.EmisFileSet.DIRECTORY_DATESTAMP_FORMAT;

public class EmisEsDatestampSkipService {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final EmisEsDatestampSkipRepository repository;

    public EmisEsDatestampSkipService(EmisEsDatestampSkipRepository repository) {
        this.repository = repository;
    }

    public boolean shouldSkipDateStamp(LocalDate dateStamp, String serverUniqueTag) {
        String dateStampStr = dateStamp.format(DIRECTORY_DATESTAMP_FORMAT);
        Optional<EmisEsDatestampSkip> skipRecord = repository.findSkipForDatestampMatchingTagOrAny(dateStampStr, serverUniqueTag);
        if (skipRecord.isPresent()) {
            LOGGER.warn("matched datestamp to skip {}", dateStampStr);
            return true;
        }
        return false;
    }
}
