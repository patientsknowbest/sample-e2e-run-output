package com.pkb.service.dataupload.emis.esv2;

import com.pkb.dataupload.emis.esv2.EmisEsCareRecordProcessingState;
import com.pkb.service.dataupload.emis.esv2.repository.EmisEsCareRecordProcessingStateRepository;
import org.springframework.stereotype.Service;

import javax.transaction.Transactional;
import java.util.HashSet;
import java.util.Set;

@Service
public class EmisEsCareRecordProcessingStateService {
    private final EmisEsCareRecordProcessingStateRepository emisEsCareRecordProcessingStateRepository;

    public EmisEsCareRecordProcessingStateService(EmisEsCareRecordProcessingStateRepository emisEsCareRecordProcessingStateRepository) {
        this.emisEsCareRecordProcessingStateRepository = emisEsCareRecordProcessingStateRepository;
    }

    public Set<EmisEsCareRecordProcessingState> getAll(Long batchId) {
        return new HashSet<>(emisEsCareRecordProcessingStateRepository.getAllByBatchId(batchId));
    }

    public EmisEsCareRecordProcessingState save(Long batchId, String observationGuid, String patientGuid) {
        EmisEsCareRecordProcessingState state = new EmisEsCareRecordProcessingState();
        state.setBatchId(batchId);
        state.setObservationGuid(observationGuid);
        state.setPatientGuid(patientGuid);
        return emisEsCareRecordProcessingStateRepository.save(state);
    }


    @Transactional
    public void clearAll(Long batchId) {
        emisEsCareRecordProcessingStateRepository.clearAllByBatchId(batchId);
    }
}
