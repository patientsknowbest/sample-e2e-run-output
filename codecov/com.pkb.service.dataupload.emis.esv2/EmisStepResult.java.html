package com.pkb.service.dataupload.emis.esv2;

public enum EmisStepResult {

    // ints in descending order from success to failure
    // the min value from all processing steps will be the overall result of the sync
    FINISHED(3), FINISHED_WITH_ERRORS(2), FAILED(1), FAILED_WITH_EMIS_CODING_ERRORS(0);

    private final int value;

    EmisStepResult(int value) {
        this.value = value;
    }

    public int getValue() {
        return value;
    }
}
