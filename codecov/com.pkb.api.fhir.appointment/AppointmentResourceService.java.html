package com.pkb.api.fhir.appointment;

import static com.pkb.authz.ImmutableTargetMeta.targetMeta;
import static com.pkb.authz.action.Action.SEARCH;
import static com.pkb.authz.context.Path.FHIR;
import static com.pkb.authz.data.ImmutableAuthorizationInputs.inputs;
import static com.pkb.domain.repository.ImmutableAppointmentQuery.appointmentQuery;
import static io.vavr.API.$;

import static io.vavr.API.Match;
import static io.vavr.API.Set;
import static io.vavr.API.Some;
import static io.vavr.Predicates.instanceOf;
import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;

public class AppointmentResourceService extends AbstractResourceService<Appointment> {
    private final AppointmentService appointmentService;
    private final AppointmentMapper appointmentMapper;
    private final AuthorizationDataService authorizationDataService;
    private final ProvenanceMapper provenanceMapper;
    private final FhirApiProperties fhirConfig;

    public AppointmentResourceService(ServerResponseExceptionMapper exceptionMapper,
                                      AppointmentService appointmentService,
                                      AppointmentMapper appointmentMapper,
                                      AuthorizationDataService authorizationDataService,
                                      ProvenanceMapper provenanceMapper,
                                      FhirApiProperties fhirConfig) {
        super(exceptionMapper);
        this.appointmentService = appointmentService;
        this.appointmentMapper = appointmentMapper;
        this.authorizationDataService = authorizationDataService;
        this.provenanceMapper = provenanceMapper;
        this.fhirConfig = fhirConfig;
    }

    public Either<BaseServerResponseException, FhirCriteriaResult<Appointment>> search(AuthenticatedIdentity authenticatedIdentity,
                                                                                       UUID patientPublicId,
                                                                                       TemporalRangeRestriction<LocalDate> localDateRangeRestriction,
                                                                                       List<IncludeResource> includes) {
        // Fetch auth data one level above calling the PKB domain services.
        AuthorizationInputs authInputs = inputs()
                .authenticatedIdentity(authenticatedIdentity)
                .action(SEARCH)
                .addTargets(targetMeta(com.pkb.datamodel.Appointment.class, Set(), Some(patientPublicId), true))
                // If I was going to call multiple PKB domain services, I could add all the targetMetas in one go here
                // then reuse the AuthorizationData instance for all the PKB domain service calls.
                .path(FHIR)
                .build();

        AuthorizationData authorizationData = authorizationDataService.fetch(authInputs);
        AppointmentQuery query = appointmentQuery()
                .dateRestriction(localDateRangeRestriction)
                .patientId(patientPublicId)
                .authData(authorizationData)
                .maxNumberOfResources(fhirConfig.getAppointmentMaxNumberOfResources())
                .build();
        return appointmentService.search(query)
                // All authz exceptions are ignored and we just return an empty result for search operations.
                // Except for SystemUserTypeNotPermittedException, which returns an error response to the caller
                // https://pkbdev.atlassian.net/wiki/spaces/TECHNOLOGY/pages/127795254/FHIR+API+Roadmap+Appointment
                .fold(error -> Match(error).of(
                        Case($(instanceOf(SystemUserTypeNotPermittedException.class)), nae -> left(toSystemUserTypeNotPermitted())),
                        Case($(), right(FhirCriteriaResult.empty()))),
                        appointments -> right(mapResult(appointments, includes)));
    }

    private FhirCriteriaResult<Appointment> mapResult(CriteriaResult<com.pkb.datamodel.Appointment> result, 
                                                      List<IncludeResource> includes) {
        List<Appointment> appointments = result.getPage().map(appointmentMapper::toAppointment);
        List<Provenance> provenances = includes.contains(IncludeResource.PROVENANCE) ? 
                result.getPage().map(provenanceMapper::toProvenance) :
                List();

        return ImmutableFhirCriteriaResult.<Appointment> fhirCriteriaResult()
                .total(result.getTotal())
                .page(appointments)
                .includes(List.narrow(provenances))
                .build();
    }
}
