package com.pkb.filesanitizer;

import io.vavr.Tuple;
import io.vavr.Tuple2;
import io.vavr.control.Either;
import org.apache.commons.lang3.StringUtils;
import org.apache.tika.config.TikaConfig;
import org.apache.tika.detect.Detector;
import org.apache.tika.metadata.Metadata;
import org.apache.tika.mime.MediaType;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.ByteArrayInputStream;
import java.io.InputStream;
import java.util.Arrays;
import java.util.List;
import java.util.function.Function;

import static com.pkb.filesanitizer.Result.CANNOT_VERIFY;
import static com.pkb.filesanitizer.Result.FAIL;
import static com.pkb.filesanitizer.Result.OK;

/***
 * Utility class that verifies whether a file (e.g. attachment) contains some possible threat
 */
public class FileSanitizer {

    private static final String MEDIA_TYPE_SAFE = "application/octet-stream";

    private static final String MEDIA_TYPE_PDF = "application/pdf";

    private static final String MEDIA_TYPE_DOCX = "application/vnd.openxmlformats-officedocument.wordprocessingml.document";

    private final static Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final List<SanitizationRule> rules;
    private final TikaConfig config;

    private static final SanitizationRule[] DEFAULT_SANITIZATION_RULES = new SanitizationRule[] {
            new PDFSanitizationRule()
    };

    public FileSanitizer() {
        this(DEFAULT_SANITIZATION_RULES);
    }

    public FileSanitizer(@NotNull SanitizationRule... rules) {
        this.rules = Arrays.asList(rules);
        this.config = TikaConfig.getDefaultConfig();
    }

    /**
     * Tries to detects media type based on the stream.
     */
    public Either<Exception, String> detectMediaType(@NotNull InputStream stream) {
        try {
            Metadata metadata = new Metadata();
            Detector detector = config.getDetector();
            MediaType mediaType = detector.detect(stream, metadata);
            return Either.right(mediaType.toString());
        } catch (Exception e) {
            return Either.left(e);
        }
    }

    /**
     * Tries to recognize file type and apply sanitization rules
     * 
     * @param fileBytes
     *            the file contents
     * @return Either an exception during media type detection or the result of the check and the detected media type
     */
    public Either<Exception, Tuple2<Result, String>> check(@NotNull byte[] fileBytes) {
        ByteArrayInputStream stream = new ByteArrayInputStream(fileBytes);
        Either<Exception, String> mediaType = detectMediaType(stream);
        return mediaType
                .map(type -> rules
                        .stream()
                        .filter(rule -> rule.handles(type))
                        .map(rule -> rule.check(fileBytes))
                        .reduce(CANNOT_VERIFY, this::sanityCheckReducer))
                .map(result -> Tuple.of(result, mediaType.get()));
    }

    public @NotNull String fixMediaType(@NotNull byte[] content, String originalMediaType) {
        Either<Exception, Tuple2<Result, String>> checkResult = check(content);
        String logMessage = checkResult
                .map($ -> "Media type: " + $._2 + ", verified: " + $._1)
                .mapLeft(Throwable::toString)
                .fold(Function.identity(), Function.identity());
        LOGGER.info("Sanitization result: {} ", logMessage);

        String defaultReturnValue = StringUtils.isNotBlank(originalMediaType) ? originalMediaType : MEDIA_TYPE_SAFE;

        if (checkResult.isRight()) {
            if (MEDIA_TYPE_PDF.equalsIgnoreCase(originalMediaType) || MEDIA_TYPE_PDF.equalsIgnoreCase(checkResult.get()._2)) {
                return fixPDFMediaType(checkResult, defaultReturnValue);
            }
            if (MEDIA_TYPE_DOCX.equalsIgnoreCase(originalMediaType) || MEDIA_TYPE_SAFE.equalsIgnoreCase(checkResult.get()._2)) {
                return defaultReturnValue;
            } else {
                LOGGER.info("Media type override to {}", checkResult.get()._2);
                return checkResult.get()._2;
            }
        }
        return defaultReturnValue;
    }

    private String fixPDFMediaType(Either<Exception, Tuple2<Result, String>> checkResult, String defaultReturnValue) {
        switch (checkResult.get()._1) {
        case FAIL:
            // PDF-specific override: check sanitization status and set to octet-stream on failure (so that users will just download it, not view)
            LOGGER.info("PDF failed sanity check, override to {}", MEDIA_TYPE_SAFE);
            return MEDIA_TYPE_SAFE;
        case CANNOT_VERIFY:
        case OK:
            if (MEDIA_TYPE_SAFE.equalsIgnoreCase(checkResult.get()._2)) {
                return defaultReturnValue;
            } else {
                LOGGER.info("PDF media type override to {}", checkResult.get()._2);
                return checkResult.get()._2;
            }
        default:
            return defaultReturnValue;
        }
    }

    @NotNull
    private Result sanityCheckReducer(Result l, Result r) {
        // fail if any rules failed,
        // pass if none of the rules failed but at least one rule passed
        if (r == FAIL || l == FAIL) {
            return FAIL;
        } else if (r == OK || l == OK) {
            return OK;
        } else {
            return CANNOT_VERIFY;
        }
    }
}
