package com.pkb.filesanitizer;

import com.itextpdf.text.pdf.PdfArray;
import com.itextpdf.text.pdf.PdfDictionary;
import com.itextpdf.text.pdf.PdfName;
import com.itextpdf.text.pdf.PdfReader;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Optional;
import java.util.stream.IntStream;

public class PDFSanitizationRule implements SanitizationRule {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Override
    public boolean handles(String mimeType) {
        return "application/pdf".equalsIgnoreCase(mimeType);
    }

    @Override
    public Result check(byte[] bytes) {
        try {
            PdfReader reader = new PdfReader(bytes);
            String jsCode = reader.getJavaScript();
            boolean seemsToContainJs = IntStream
                    .range(0, reader.getXrefSize())
                    .mapToObj(reader::getPdfObject)
                    .filter(pdfObj -> pdfObj instanceof PdfDictionary)
                    .map($ -> (PdfDictionary) $)
                    .map(dict -> (PdfName.JAVASCRIPT.toString().equalsIgnoreCase(Optional.ofNullable(dict.get(PdfName.S)).map(
                            Object::toString).orElse(null)) || dict.contains(PdfName.JS)))
                    .reduce(false, (l, r) -> l || r);
            if (jsCode == null && !seemsToContainJs) {
                PdfDictionary root = reader.getCatalog();
                PdfDictionary names = root.getAsDict(PdfName.NAMES);
                PdfArray namesArray = null;
                if (names != null) {
                    PdfDictionary embeddedFiles = names.getAsDict(PdfName.EMBEDDEDFILES);
                    if (embeddedFiles != null) {
                        namesArray = embeddedFiles.getAsArray(PdfName.NAMES);
                    }
                }
                return ((namesArray == null) || namesArray.isEmpty()) ? Result.OK : Result.FAIL;
            }
            return Result.FAIL;
        } catch (Exception e) {
            LOGGER.error("Failed to sanitize PDF", e);
            return Result.CANNOT_VERIFY;
        }
    }
}
