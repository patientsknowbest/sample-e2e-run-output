package com.pkb.questionnaireservice.entities

import com.fasterxml.jackson.databind.annotation.JsonDeserialize
import com.fasterxml.jackson.databind.annotation.JsonSerialize
import com.pkb.questionnaireservice.QuestionnaireId
import com.pkb.questionnaireservice.TeamPublicId
import com.pkb.questionnaireservice.modelv2.QuestionnaireTeam
import java.io.Serializable
import java.time.Instant
import javax.persistence.Column
import javax.persistence.Embeddable
import javax.persistence.Entity
import javax.persistence.Id
import javax.persistence.IdClass
import javax.persistence.JoinColumn
import javax.persistence.ManyToOne
import javax.persistence.Table

@Entity
@Table(name="questionnaire_team", schema="ques")
@IdClass(QuestionnaireTeamId::class)
@JsonDeserialize(`as`= QuestionnaireTeamEntity::class)
@JsonSerialize(`as`= QuestionnaireTeamEntity::class)
class QuestionnaireTeamEntity(
        @ManyToOne
        @JoinColumn(name = "questionnaire_id", nullable = false, updatable = false)
        @Id
        override val questionnaire: QuestionnaireEntity,

        @Id
        @Column(name = "team_id")
        override val teamId: TeamPublicId,

        @Column
        override var hidden: Boolean = false,

        @Column(name = "updated_date")
        override var updatedDate: Instant
) : QuestionnaireTeam


@Embeddable //This isn't really meaningful but it forces the kotlin compiler JPA plugin to generate a no-arg constructor
data class QuestionnaireTeamId(
        val questionnaire: QuestionnaireId,
        val teamId: TeamPublicId
) : Serializable {
    companion object {
        private const val serialVersionUID: Long = 1
    }
}
