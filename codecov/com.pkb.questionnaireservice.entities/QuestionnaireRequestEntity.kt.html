package com.pkb.questionnaireservice.entities

import com.fasterxml.jackson.annotation.JsonCreator
import com.fasterxml.jackson.databind.annotation.JsonDeserialize
import com.fasterxml.jackson.databind.annotation.JsonSerialize
import com.pkb.entities.enums.Route
import com.pkb.kms.shared.representation.PersonPublicId
import com.pkb.questionnaireservice.QuestionnaireRequestId
import com.pkb.questionnaireservice.TeamPublicId
import com.pkb.questionnaireservice.modelv2.QuestionnaireRequest
import org.hibernate.annotations.Immutable
import org.hibernate.annotations.Type
import java.time.Instant
import javax.persistence.CascadeType
import javax.persistence.Column
import javax.persistence.Entity
import javax.persistence.EnumType
import javax.persistence.Enumerated
import javax.persistence.FetchType
import javax.persistence.Id
import javax.persistence.JoinColumn
import javax.persistence.ManyToOne
import javax.persistence.OneToMany
import javax.persistence.Table

@Entity
@Table(name="questionnaire_request", schema="ques")
@Immutable
@JsonDeserialize(`as`= QuestionnaireRequestEntity::class)
@JsonSerialize(`as`= QuestionnaireRequestEntity::class)
class QuestionnaireRequestEntity
@JsonCreator(mode = JsonCreator.Mode.PROPERTIES)
constructor(

        @Id
        @Type(type = "pg-uuid")
        override val id: QuestionnaireRequestId,

        @ManyToOne(fetch = FetchType.EAGER)
        @JoinColumn(name = "questionnaire_id")
        override val questionnaire: QuestionnaireEntity,

        @Column(nullable = false, name="created_date")
        override val createdDate: Instant,

        @Column(nullable = false, name="requestor_person_id")
        @Type(type = "pg-uuid")
        override val requestorPersonId: PersonPublicId,

        @Column(nullable = false, name = "requesting_team_id")
        @Type(type = "pg-uuid")
        override val requestingTeamId: TeamPublicId,

        @Column(nullable = false, name="patient_id")
        @Type(type = "pg-uuid")
        override val patientId: PersonPublicId,

        @Column(nullable = false, name = "route")
        @Enumerated(EnumType.STRING)
        @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
        override val route: Route,
        
        @OneToMany(fetch = FetchType.EAGER, targetEntity = QuestionnaireRequestIdentifierEntity::class, mappedBy = "questionnaireRequestId", cascade = [CascadeType.ALL])
        override val identifiers: List<QuestionnaireRequestIdentifierEntity>
) : BaseEntity(), QuestionnaireRequest
