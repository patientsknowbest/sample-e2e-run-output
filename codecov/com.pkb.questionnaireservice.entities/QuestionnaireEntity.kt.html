package com.pkb.questionnaireservice.entities

import com.fasterxml.jackson.annotation.JsonCreator
import com.fasterxml.jackson.annotation.JsonIdentityInfo
import com.fasterxml.jackson.annotation.ObjectIdGenerators
import com.fasterxml.jackson.databind.annotation.JsonDeserialize
import com.pkb.entities.enums.PrivacyFlag
import com.pkb.kms.shared.representation.PersonPublicId
import com.pkb.questionnaireservice.QuestionnaireId
import com.pkb.questionnaireservice.modelv2.Questionnaire
import com.pkb.questionnaireservice.modelv2.QuestionnaireComponent
import com.pkb.questionnaireservice.repositories.ques.EntityManagerResolver
import org.hibernate.annotations.Immutable
import org.hibernate.annotations.Type
import java.time.Instant
import javax.persistence.Column
import javax.persistence.Entity
import javax.persistence.EnumType
import javax.persistence.Enumerated
import javax.persistence.Id
import javax.persistence.Table

@Entity
@Immutable
@Table(name="questionnaire", schema="ques")
@JsonDeserialize(`as`= QuestionnaireEntity::class)
@JsonIdentityInfo(generator = ObjectIdGenerators.PropertyGenerator::class, property = "id", resolver= EntityManagerResolver::class, scope=QuestionnaireEntity::class)
class QuestionnaireEntity

@JsonCreator(mode = JsonCreator.Mode.PROPERTIES)
constructor(

        @Id
        @Type(type = "pg-uuid")
        override val id: QuestionnaireId,

        @Column(nullable = false)
        override val name: String,

        @Column(nullable = false, name="creator_person_id")
        @Type(type = "pg-uuid")
        override val creatorPersonId: PersonPublicId,

        @Type(type = "jsonb")
        @Column(nullable = false, columnDefinition = "jsonb")
        override val components: List<QuestionnaireComponent>,

        @Column(nullable = false, name="created_date")
        override val createdDate: Instant,

        @Column(nullable = false, name="default_privacy_flag")
        @Enumerated(EnumType.STRING)
        @Type(type = "com.vladmihalcea.hibernate.type.basic.PostgreSQLEnumType")
        override val defaultPrivacyFlag: PrivacyFlag
): BaseEntity(), Questionnaire
