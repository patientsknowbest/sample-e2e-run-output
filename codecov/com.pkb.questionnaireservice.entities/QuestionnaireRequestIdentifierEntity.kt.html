package com.pkb.questionnaireservice.entities

import com.fasterxml.jackson.annotation.JsonCreator
import com.fasterxml.jackson.databind.annotation.JsonDeserialize
import com.fasterxml.jackson.databind.annotation.JsonSerialize
import com.pkb.questionnaireservice.QuestionnaireRequestId
import com.pkb.questionnaireservice.QuestionnaireRequestIdentifierId
import com.pkb.questionnaireservice.modelv2.QuestionnaireRequestIdentifier
import org.hibernate.annotations.Immutable
import org.hibernate.annotations.Type
import javax.persistence.Column
import javax.persistence.Entity
import javax.persistence.Id
import javax.persistence.Table

@Entity
@Immutable
@Table(name="questionnaire_request_identifier", schema="ques")
@JsonDeserialize(`as`= QuestionnaireRequestIdentifierEntity::class)
@JsonSerialize(`as`= QuestionnaireRequestIdentifierEntity::class)
class QuestionnaireRequestIdentifierEntity
@JsonCreator(mode = JsonCreator.Mode.PROPERTIES)
constructor(

        @Id
        @Type(type = "pg-uuid")
        override val id: QuestionnaireRequestIdentifierId,

        @Type(type = "pg-uuid")
        @Column(nullable = false, name="questionnaire_request_id")
        override val questionnaireRequestId: QuestionnaireRequestId,

        @Column(nullable = false, name="system")
        override val system: String,

        @Column(nullable = false, name="value")
        override val value: String
): BaseEntity(), QuestionnaireRequestIdentifier
