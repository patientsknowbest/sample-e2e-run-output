package com.pkb.questionnaireservice.entities

import com.pkb.kms.shared.representation.AccountPublicId

import org.hibernate.annotations.Immutable

import javax.persistence.Column

import javax.persistence.FetchType

import javax.persistence.ManyToOne

@Entity
@Table(name="questionnaire_response", schema="ques")
@Immutable
@TypeDef(name = "jsonb", typeClass = JsonBinaryType::class)
class QuestionnaireResponseEntity(

        @Id
        @Type(type = "pg-uuid")
        override val id: QuestionnaireResponseId,

        @ManyToOne(fetch = FetchType.EAGER)
        @JoinColumn(name = "questionnaire_request_id")
        override val questionnaireRequest: QuestionnaireRequestEntity,

        @Column(nullable = false, name="created_date")
        override val createdDate: Instant,
        
        @Column(nullable = true, name="completed_by")
        override val completedBy: PersonPublicId?,

        @Column(nullable = false, name = "patient_account_id")
        @Type(type = "pg-uuid")
        override val patientAccountId: AccountPublicId,

        @Column(nullable = false)
        internal val nonce: ByteArray,

        @Column(name = "dpl_key_id", nullable = false)
        @Type(type = "pg-uuid")
        internal val dplKeyId: DplKeyId?,

        @Column(name = "dpl_encrypted_answers", nullable = false)
        internal val dplEncryptedAnswers: ByteArray?
) : BaseEntity(), QuestionnaireResponseBase


/**
 * This is the intermediate representation that we deserialize a decrypted response to before converting to an entity with
 * encrypted data. Correspondingly this is used as an intermediate representation for serializing a decrypted response
 * back to the client.
 */
@JsonSerialize(`as`= DecryptedQuestionnaireResponse::class)
@JsonDeserialize(`as`= DecryptedQuestionnaireResponseSerialize::class)
open class DecryptedQuestionnaireResponse
@JsonCreator(mode = JsonCreator.Mode.PROPERTIES)
constructor(
        override val id: QuestionnaireResponseId,
        override val questionnaireRequest: QuestionnaireRequestEntity,
        override val createdDate: Instant,
        override val completedBy: PersonPublicId?,
        override val patientAccountId: AccountPublicId,
        override  val answers: List<Answer>
) : QuestionnaireResponse {
    constructor(entity: QuestionnaireResponseEntity, answers: List<Answer>) : this(entity.id, entity.questionnaireRequest, entity.createdDate, entity.completedBy, entity.patientAccountId, answers)
}

class DecryptedQuestionnaireResponseSerialize
@JsonCreator(mode = JsonCreator.Mode.PROPERTIES)
constructor(
        id: QuestionnaireResponseId,
        @JsonIdentityInfo(generator = ObjectIdGenerators.PropertyGenerator::class, property = "id", resolver= EntityManagerResolver::class, scope=QuestionnaireRequestEntity::class)
        questionnaireRequest: QuestionnaireRequestEntity,
        createdDate: Instant,
        completedBy: PersonPublicId?,
        patientAccountId: AccountPublicId,
        answers: List<Answer>
) : DecryptedQuestionnaireResponse(id, questionnaireRequest, createdDate, completedBy, patientAccountId, answers)
