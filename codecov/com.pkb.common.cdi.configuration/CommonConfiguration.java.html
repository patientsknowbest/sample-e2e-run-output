package com.pkb.common.cdi.configuration;

import com.pkb.common.config.ConfigStorage;
import com.pkb.common.config.ConfigStorageFactory;
import com.pkb.common.config.KmsClientConfigImpl;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.config.PhrConfigImpl;
import com.pkb.common.config.PkbPluginConfig;
import com.pkb.common.config.PkbPluginConfigImpl;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.common.datetime.DefaultDateTimeService;
import com.pkb.common.datetime.FakeDateTimeService;
import com.pkb.common.testlogging.DefaultDetailLoggingProvider;
import com.pkb.common.testlogging.DetailLoggingProvider;
import com.pkb.kms.client.core.KmsClientConfig;
import com.pkb.service.uuid.UUIDProvider;
import io.micrometer.core.instrument.Metrics;
import io.micrometer.core.instrument.binder.okhttp3.OkHttpMetricsEventListener;
import okhttp3.OkHttpClient;

import javax.enterprise.context.ApplicationScoped;
import javax.enterprise.inject.Produces;

@ApplicationScoped
public class CommonConfiguration {

    @Produces
    @ApplicationScoped
    public ConfigStorage configStorage() {
        return ConfigStorageFactory.getConfigStorage();
    }

    @Produces
    @ApplicationScoped
    public PkbPluginConfig pluginConfig(ConfigStorage configStorage) {
        return new PkbPluginConfigImpl(configStorage);
    }

    @Produces
    @ApplicationScoped
    public KmsClientConfig kmsClientConfig(ConfigStorage configStorage) {
        return new KmsClientConfigImpl(configStorage);
    }

    @Produces
    @ApplicationScoped
    public PhrConfig phrConfig(ConfigStorage configStorage) {
        return new PhrConfigImpl(configStorage);
    }

    @Produces
    @ApplicationScoped
    public DetailLoggingProvider defaultTestLoggingService() {
        return DefaultDetailLoggingProvider.INSTANCE;
    }

    @Produces
    @ApplicationScoped
    public DateTimeService dateTimeService(PkbPluginConfig config) {
        if (config.isFakeDateTimeServiceEnabled()) {
            return new FakeDateTimeService();
        } else {
            return new DefaultDateTimeService();
        }
    }

    @Produces
    @ApplicationScoped
    public UUIDProvider getUUIDProvider() {
        return new UUIDProvider();
    }

    @Produces
    @ApplicationScoped
    public OkHttpClientHolder httpClient() {
        return new OkHttpClientHolder(new OkHttpClient.Builder()
                .eventListener(OkHttpMetricsEventListener.builder(Metrics.globalRegistry, "okhttp.requests").build())
                .build());
    }
}
