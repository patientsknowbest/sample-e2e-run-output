package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4Organization
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.reference.extract.OrganizationExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts organization references from [org.hl7.fhir.r4.model.Organization] resources
 */
@Service
class OrganizationOrganizationExtractor : OrganizationExtractor {

    override fun extract(resources: List<IDomainResource>) =
        resources.mapNotNull { it as? R4Organization }
            .mapNotNull { extractSourceIdAndOrganisation(it) }
            .toSet()

    private fun extractSourceIdAndOrganisation(organization: R4Organization): UpstreamResource? {
        val source = organization.getSourcedReference()
        return if (organization.hasPartOf()) {
            UpstreamResource(source.upstreamSource, organization.partOf.reference)
        } else {
            null
        }
    }
}
