package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4Encounter
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.reference.extract.OrganizationExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts organization references from [org.hl7.fhir.r4.model.Encounter] resources
 */
@Service
class EncounterOrganizationExtractor : OrganizationExtractor {

    override fun extract(resources: List<IDomainResource>) =
        resources.mapNotNull { it as? R4Encounter }
            .mapNotNull { extractSourceIdAndOrganisation(it) }
            .toSet()

    private fun extractSourceIdAndOrganisation(encounter: R4Encounter): UpstreamResource? {
        val source = encounter.getSourcedReference()
        return if (encounter.hasServiceProvider()) {
            UpstreamResource(source.upstreamSource, encounter.serviceProvider.reference)
        } else {
            null
        }
    }
}
