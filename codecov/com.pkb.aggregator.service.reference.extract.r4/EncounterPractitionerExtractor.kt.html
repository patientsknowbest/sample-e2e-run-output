package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4Encounter
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.reference.extract.PractitionerExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts practitioner references from [org.hl7.fhir.r4.model.Encounter] resources
 */
@Service
class EncounterPractitionerExtractor : PractitionerExtractor {

    override fun extract(resources: List<IDomainResource>) =
        resources.mapNotNull { it as? R4Encounter }
            .flatMap { extractSourceIdAndPractitioner(it) }
            .toSet()

    private fun extractSourceIdAndPractitioner(encounter: R4Encounter): Set<UpstreamResource> {
        val source = encounter.getSourcedReference()
        return if (encounter.hasParticipant()) {
            encounter.participant
                .map { UpstreamResource(source.upstreamSource, it.individual.reference) }
                .toSet()
        } else emptySet()
    }
}
