package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4Appointment
import com.pkb.aggregator.R4Reference
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_PRACTITIONER
import com.pkb.aggregator.service.reference.extract.PractitionerExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts practitioner references from [org.hl7.fhir.r4.model.Appointment] resources
 */
@Service
class AppointmentPractitionerExtractor : PractitionerExtractor {

    override fun extract(resources: List<IDomainResource>) =
        resources.mapNotNull { it as? R4Appointment }
            .flatMap { extractSourceIdAndPractitioner(it) }
            .toSet()

    private fun extractSourceIdAndPractitioner(appointment: R4Appointment): Set<UpstreamResource> {
        val source = appointment.getSourcedReference()
        return appointment.participant
            .map { R4Reference(it.actor.reference) }
            .filter { it.referenceElement.resourceType == FHIR_TYPE_PRACTITIONER }
            .map { UpstreamResource(source.upstreamSource, it.reference) }
            .toSet()
    }
}
