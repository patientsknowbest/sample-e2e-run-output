package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4AllergyIntolerance
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.reference.extract.EncounterExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts encounter references from [org.hl7.fhir.r4.model.AllergyIntolerance] resources
 */
@Service
class AllergyIntoleranceEncounterExtractor : EncounterExtractor {

    override fun extract(resources: List<IDomainResource>) =
        resources.mapNotNull { it as? R4AllergyIntolerance }
            .mapNotNull { extractSourceIdAndEncounter(it) }
            .toSet()

    private fun extractSourceIdAndEncounter(allergyIntolerance: R4AllergyIntolerance) =
        if (allergyIntolerance.hasEncounter()) {
            val source = allergyIntolerance.getSourcedReference()
            UpstreamResource(source.upstreamSource, allergyIntolerance.encounter.reference)
        } else null
}
