package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4Element
import com.pkb.aggregator.R4Practitioner
import com.pkb.aggregator.R4Reference
import com.pkb.aggregator.configuration.PhrMigrationConfiguration
import com.pkb.aggregator.configuration.PhrMigrationConfiguration.Companion.FROZEN_BY_TEAM
import com.pkb.aggregator.configuration.PhrMigrationConfiguration.Companion.SOURCE_ORG
import com.pkb.aggregator.configuration.PhrMigrationConfiguration.Companion.SOURCE_TEAM
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.reference.PhrReferenceHelper
import com.pkb.aggregator.service.reference.extract.OrganizationExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts organization references from [org.hl7.fhir.r4.model.Practitioner] resources
 */
@Service
class PractitionerOrganizationExtractor(
    private val migrationConfiguration: PhrMigrationConfiguration,
    private val phrReferenceHelper: PhrReferenceHelper,
) : OrganizationExtractor {

    override fun extract(resources: List<IDomainResource>) =
        resources.mapNotNull { it as? R4Practitioner }
            .flatMap { extractSourceIdAndOrganisation(it) }
            .toSet()

    private fun extractSourceIdAndOrganisation(practitioner: R4Practitioner): Set<UpstreamResource> {
        val source = practitioner.getSourcedReference()

        // Check identifiers for assigning orgs
        val assigners = practitioner.identifier
            .filter { it.hasAssigner() }
            .map { UpstreamResource(source.upstreamSource, it.assigner.reference) }

        // Patients migrated from PHR may have some references stored in custom extensions
        val frozenByTeam = findFrozenByTeamReferences(source, practitioner)
        val identifierSources = findSourceReferences(source, practitioner.identifier)
        val telecomSources = findSourceReferences(source, practitioner.telecom)

        return assigners
            .plus(frozenByTeam)
            .plus(identifierSources)
            .plus(telecomSources)
            .toSet()
    }

    private fun findFrozenByTeamReferences(source: UpstreamResource, practitioner: R4Practitioner) =
        practitioner
            .getExtensionsByUrl(migrationConfiguration.extensionSystems[FROZEN_BY_TEAM])
            .map { it.value as R4Reference }
            .map { UpstreamResource(source.upstreamSource, it.reference) }

    private fun findSourceReferences(source: UpstreamResource, elements: List<R4Element>) =
        elements
            .flatMap { element ->
                element.extension
                    .filter { extension ->
                        // We're only interested in extensions with one of two specific urls
                        extension.url == migrationConfiguration.extensionSystems[SOURCE_TEAM]
                                || extension.url == migrationConfiguration.extensionSystems[SOURCE_ORG]
                    }
            }
            .mapNotNull { phrReferenceHelper.findPhrValueReference(it) }
            .map { UpstreamResource(source.upstreamSource, it.toString()) }
}
