package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4Patient
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_PRACTITIONER
import com.pkb.aggregator.service.reference.extract.PractitionerExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts practitioner references from [org.hl7.fhir.r4.model.Patient] resources
 */
@Service
class PatientPractitionerExtractor : PractitionerExtractor {

    override fun extract(resources: List<IDomainResource>) =
        resources.mapNotNull { it as? R4Patient }
            .flatMap { extractSourceIdAndPractitioner(it) }
            .toSet()

    private fun extractSourceIdAndPractitioner(patient: R4Patient): Set<UpstreamResource> {
        val source = patient.getSourcedReference()

        // Get GPs if present
        val generalPractitioners = if (patient.hasGeneralPractitioner()) {
            patient.generalPractitioner
                .mapNotNull {
                    when (it.referenceElement.resourceType) {
                        FHIR_TYPE_PRACTITIONER -> UpstreamResource(source.upstreamSource, it.reference)
                        else -> null // this is a practitioner extractor; ignore non-practitioners
                    }
                }
        } else {
            listOf()
        }

        return generalPractitioners.toSet()
    }
}
