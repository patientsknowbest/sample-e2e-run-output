package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4Provenance
import com.pkb.aggregator.R4Reference
import com.pkb.aggregator.configuration.PhrMigrationConfiguration
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_ORGANIZATION
import com.pkb.aggregator.service.reference.extract.OrganizationExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts organization references from [org.hl7.fhir.r4.model.Provenance] resources
 */
@Service
class ProvenanceOrganizationExtractor(
    private val migrationConfiguration: PhrMigrationConfiguration
) : OrganizationExtractor {

    override fun extract(resources: List<IDomainResource>) =
        resources.mapNotNull { it as? R4Provenance }
            .flatMap { extractSourceIdAndOrganization(it) }
            .toSet()

    private fun extractSourceIdAndOrganization(provenance: R4Provenance): Set<UpstreamResource> {
        val source = provenance.getSourcedReference()

        val connectingOrganization = provenance.getExtensionsByUrl(migrationConfiguration.extensionSystems[PhrMigrationConfiguration.CONNECTING_ORG])
            .map { it.value as R4Reference }

        val agentReferences = provenance.agent.flatMap { agent ->
            mutableListOf<R4Reference>().apply {
                add(agent.who)
                add(agent.onBehalfOf)
                agent.onBehalfOf.getExtensionsByUrl(migrationConfiguration.extensionSystems[PhrMigrationConfiguration.SOURCE_TEAM])
                    .map { it.value as R4Reference }
                    .let { addAll(it) }
                agent.onBehalfOf.getExtensionsByUrl(migrationConfiguration.extensionSystems[PhrMigrationConfiguration.SOURCE_ORG])
                    .map { it.value as R4Reference }
                    .let { addAll(it) }
            }
        }

        return connectingOrganization.plus(agentReferences)
            .filter { it.referenceElement.resourceType == FHIR_TYPE_ORGANIZATION }
            .map { UpstreamResource(source.upstreamSource, it.reference) }
            .toSet()
    }
}
