package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4Appointment
import com.pkb.aggregator.R4Meta
import com.pkb.aggregator.R4Reference
import com.pkb.aggregator.configuration.PhrMigrationConfiguration
import com.pkb.aggregator.configuration.PhrMigrationConfiguration.Companion.PRIVACY_OVERRIDE_ORG
import com.pkb.aggregator.configuration.PhrMigrationConfiguration.Companion.SOURCE_ORG
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.reference.extract.OrganizationExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts organization references from [org.hl7.fhir.r4.model.Appointment] resources
 */
@Service
class AppointmentOrganizationExtractor(
    private val migrationConfiguration: PhrMigrationConfiguration,
) : OrganizationExtractor {

    override fun extract(resources: List<IDomainResource>): Set<UpstreamResource> =
        resources.mapNotNull { it as? R4Appointment }
            .flatMap { extractSourceIdAndOrganization(it) }
            .toSet()

    private fun extractSourceIdAndOrganization(appointment: R4Appointment): Set<UpstreamResource> {
        val source = appointment.getSourcedReference()

        // Source org information may be present as an extension of the meta field, e.g.
        // meta:
        //     extension:
        //       - url: >-
        //           http://fhir.patientsknowbest.com/structuredefinition/source-organisation
        //         valueReference:
        //           reference: Organization/5d9587af-260d-4198-bb95-2953027d3805
        val sourceOrgs = if (appointment.hasMeta()) {
            findSourceReferences(source, appointment.meta, migrationConfiguration.extensionSystems[SOURCE_ORG])
        } else {
            emptyList()
        }

        val privacyOverrideOrgs = if (appointment.hasMeta()) {
            findSourceReferences(source, appointment.meta, migrationConfiguration.extensionSystems[PRIVACY_OVERRIDE_ORG])
        } else {
            emptyList()
        }

        return sourceOrgs
            .plus(privacyOverrideOrgs)
            .toSet()
    }

    private fun findSourceReferences(source: UpstreamResource, meta: R4Meta, url: String?) =
        url?.let {
            meta.getExtensionsByUrl(url)
                .map { it.value as R4Reference }
                .map {
                    UpstreamResource(source.upstreamSource, it.reference)
                }
        } ?: emptyList()
}
