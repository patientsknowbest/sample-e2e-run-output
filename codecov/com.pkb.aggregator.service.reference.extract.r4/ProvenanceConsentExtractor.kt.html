package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4Provenance
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_CONSENT
import com.pkb.aggregator.service.reference.extract.ConsentExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts consent references from [org.hl7.fhir.r4.model.Provenance] resources
 */
@Service
class ProvenanceConsentExtractor : ConsentExtractor {

    override fun extract(resources: List<IDomainResource>) =
        resources.filterIsInstance<R4Provenance>()
            .flatMap { extractSourceIdAndConsent(it) }
            .toSet()

    private fun extractSourceIdAndConsent(provenance: R4Provenance): Set<UpstreamResource> {
        val source = provenance.getSourcedReference()

        val target = provenance.target

        return target
            .filter { it.referenceElement.resourceType == FHIR_TYPE_CONSENT }
            .map { UpstreamResource(source.upstreamSource, it.reference) }
            .toSet()
    }
}
