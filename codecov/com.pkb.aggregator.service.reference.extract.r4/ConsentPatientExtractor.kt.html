package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4Consent
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.reference.extract.PatientExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts patient references from [org.hl7.fhir.r4.model.Consent] resources
 */
@Service
class ConsentPatientExtractor : PatientExtractor {

    override fun extract(resources: List<IDomainResource>) =
            resources.filterIsInstance<R4Consent>()
                    .mapNotNull { extractSourceIdAndPatient(it) }
                    .toSet()

    private fun extractSourceIdAndPatient(consent: R4Consent): UpstreamResource? {

        val source = consent.getSourcedReference()
        return if (consent.hasPatient()) UpstreamResource(source.upstreamSource, consent.patient.reference)
        else null
    }
}
