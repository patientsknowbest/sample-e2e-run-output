package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4Consent
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_ORGANIZATION
import com.pkb.aggregator.service.reference.extract.OrganizationExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts organization references from [org.hl7.fhir.r4.model.Consent] resources
 */
@Service
class ConsentOrganizationExtractor : OrganizationExtractor {

    override fun extract(resources: List<IDomainResource>) =
            resources.filterIsInstance<R4Consent>()
                    .flatMap { extractSourceIdAndOrganization(it) }
                    .toSet()

    private fun extractSourceIdAndOrganization(consent: R4Consent): Set<UpstreamResource> {
        val source = consent.getSourcedReference()

        val organizations = consent.organization

        val organizationActorReferences = consent.provision.actor.map { it.reference }
                .filter {it.referenceElement.resourceType == FHIR_TYPE_ORGANIZATION }



        return organizations.plus(organizationActorReferences)
                .map { UpstreamResource(source.upstreamSource, it.reference) }
                .toSet()
    }


}
