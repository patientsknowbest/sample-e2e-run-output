package com.pkb.aggregator.service.reference.extract.r4

import com.pkb.aggregator.R4Provenance
import com.pkb.aggregator.R4Reference
import com.pkb.aggregator.extension.getSourcedReference
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_PATIENT
import com.pkb.aggregator.service.reference.extract.PatientExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.springframework.stereotype.Service

/**
 * Extracts patient references from [org.hl7.fhir.r4.model.Provenance] resources
 */
@Service
class ProvenancePatientExtractor : PatientExtractor {

    override fun extract(resources: List<IDomainResource>) =
        resources.mapNotNull { it as? R4Provenance }
            .flatMap { extractSourceIdAndPatient(it) }
            .toSet()

    private fun extractSourceIdAndPatient(provenance: R4Provenance): Set<UpstreamResource> {
        val source = provenance.getSourcedReference()

        val targets = provenance.target

        val agentReferences = provenance.agent.flatMap { agent ->
            mutableListOf<R4Reference>().apply {
                add(agent.who)
                add(agent.onBehalfOf)
            }
        }

        return targets.plus(agentReferences)
            .filter { it.referenceElement.resourceType == FHIR_TYPE_PATIENT }
            .map { UpstreamResource(source.upstreamSource, it.reference) }
            .toSet()
    }
}
