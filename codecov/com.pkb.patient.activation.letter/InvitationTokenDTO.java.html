package com.pkb.patient.activation.letter;

import static com.google.common.base.Preconditions.checkArgument;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

import java.io.Serializable;

import org.immutables.value.Value.Check;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Parameter;

/**
 * <pre>
 * While inviting a Patient via Letter Invitation flow two token is generated.
 * 1. A token that makes a connection between a medical record and
 * 2. A registration token that can be:
 * 2. a) Only used once
 * 2. b) can be expired/revoked.
 * This token pair can be used securely to activate a Patient in PKB.
 * </pre>
 */
@Immutable
public interface InvitationTokenDTO extends Serializable {
    /**
     * <pre>
     * For security reasons we are not sending out internal Medical Record ID as a plain text to Patients.
     * Instead, a token is assigned (in a one to one relation) to a Medical Record ID when token gets created.
     * </pre>
     */
    @Parameter
    String getDecryptToken();

    @Parameter
    String getInvitationCode();

    @Check
    default void checkContent() {
        checkArgument(isNotBlank(getDecryptToken()), "Decrypt Token cannot be null, neither effectively empty.");
        checkArgument(isNotBlank(getInvitationCode()), "Invitation Token cannot be null, neither effectively empty.");
    }
}
