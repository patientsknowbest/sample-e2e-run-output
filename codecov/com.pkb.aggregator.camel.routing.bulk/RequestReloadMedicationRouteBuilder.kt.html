package com.pkb.aggregator.camel.routing.bulk

import com.pkb.aggregator.camel.MEDICATION_PAGE_LINKS_PROPERTY
import com.pkb.aggregator.camel.MEDICATION_PROPERTY
import com.pkb.aggregator.camel.bean.process.MedicationDeleter
import com.pkb.aggregator.camel.bean.process.MedicationProcessor
import com.pkb.aggregator.camel.bean.search.MedicationSearch
import com.pkb.aggregator.camel.beanMethod
import com.pkb.aggregator.camel.parallelFhirSearch
import com.pkb.aggregator.camel.routing.AbstractAggregatorRouteBuilder
import com.pkb.aggregator.camel.routing.createFhirSearchRoute
import com.pkb.aggregator.camel.routing.createFhirTransactionRoute
import com.pkb.aggregator.camel.strategy.BodyMutableListBundleAggregationStrategy
import com.pkb.aggregator.configuration.FhirConfiguration
import com.pkb.aggregator.service.FHIR_TYPE_MEDICATION
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import org.hl7.fhir.instance.model.api.IAnyResource
import org.springframework.stereotype.Component

/**
 * Defines camel routes for performing full reloads of medication resources.
 */
@Component
class RequestReloadMedicationRouteBuilder(
    private val fhirConfiguration: FhirConfiguration,
    private val bundleAggregationStrategy: BodyMutableListBundleAggregationStrategy,
    pubSubNamespaceService: PubSubNamespaceService,
) : AbstractAggregatorRouteBuilder(pubSubNamespaceService) {

    companion object {
        const val REQUEST_MEDICATION_RELOAD = "direct:aggregator/reload/medication"
        
        private const val ROUTE_ID = "medication-reload"
    }

    override fun configure() {

        // Register routes for loading medication resources from upstream (blocking and non-blocking)
        val upstreamRoutes = fhirConfiguration.upstream.endpoints.map { createFhirSearchRoute(it) }

        from(REQUEST_MEDICATION_RELOAD).routeId(ROUTE_ID)
            // Set placeholder properties
            .setProperty(MEDICATION_PROPERTY) { mutableListOf<IAnyResource>() }.id("$ROUTE_ID-create-resource-exchangeproperty")
            .setProperty(MEDICATION_PAGE_LINKS_PROPERTY) { mutableListOf<String>() }.id("$ROUTE_ID-create-resource-pagelinks-exchangeproperty")
            // Construct search query from all known codings
            .beanMethod(MedicationSearch::createSearchUrl).id("$ROUTE_ID-identifier-search")
            // Perform organization search in parallel across all upstreams using criteria from previous step
            .parallelFhirSearch(simple("$FHIR_TYPE_MEDICATION?code=\${body}&_sort=code"), upstreamRoutes, bundleAggregationStrategy)
                .id("$ROUTE_ID-upstream-search")
            // Get extra pages if needed
            .dynamicRouter(beanMethod(MedicationSearch::paginate)).id("$ROUTE_ID-paginate")
            // Aggregate all the information into a single bundle
            .beanMethod(MedicationProcessor::aggregate).id("$ROUTE_ID-aggregate")
            // Remove all data from downstream prior to writing the new info
            .beanMethod(MedicationDeleter::hardDelete).id("$ROUTE_ID-downstream-delete")
            // Post transaction bundle to downstream
            .to(createFhirTransactionRoute(fhirConfiguration.downstream.endpoint)).id("$ROUTE_ID-downstream-push")
            // Log info
            .beanMethod(MedicationProcessor::logOutcome).id("${ROUTE_ID}-log-outcome")
    }
}
