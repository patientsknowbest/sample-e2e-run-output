package com.pkb.aggregator.camel.routing.bulk

import org.hl7.fhir.instance.model.api.IAnyResource
import org.springframework.stereotype.Component

/**
 * Defines camel routes for performing full reloads of organization resources.
 */
@Component
class RequestReloadOrganizationRouteBuilder(
    private val fhirConfiguration: FhirConfiguration,
    private val bundleAggregationStrategy: BodyMutableListBundleAggregationStrategy,
    pubSubNamespaceService: PubSubNamespaceService,
) : AbstractAggregatorRouteBuilder(pubSubNamespaceService) {

    companion object {
        const val REQUEST_ORGANIZATION_RELOAD = "direct:aggregator/reload/organization"
        
        private const val ROUTE_ID = "organization-reload"
    }

    override fun configure() {

        // Register routes for loading organization resources from upstream (blocking and non-blocking)
        val upstreamRoutes = fhirConfiguration.upstream.endpoints.map { createFhirSearchRoute(it) }

        val orgSearch = simple("$FHIR_TYPE_ORGANIZATION?identifier=\${body}" +
                "&_sort=name" +
                "&_revinclude=$FHIR_TYPE_PRACTITIONER_ROLE:organization" +
                "&_include:iterate=$FHIR_TYPE_PRACTITIONER_ROLE:practitioner")

        from(REQUEST_ORGANIZATION_RELOAD).routeId(ROUTE_ID)
            // Set placeholder properties
            .setProperty(ORGANIZATION_PROPERTY) { mutableListOf<IAnyResource>() }.id("$ROUTE_ID-create-resource-exchangeproperty")
            .setProperty(ORGANIZATION_PAGE_LINKS_PROPERTY) { mutableListOf<String>() }.id("$ROUTE_ID-create-resource-pagelinks-exchangeproperty")
            // Construct search query from all known identifiers
            .beanMethod(OrganizationSearch::createSearchUrl).id("$ROUTE_ID-identifier-search")
            // Perform organization search in parallel across all upstreams using criteria from previous step
            .parallelFhirSearch(orgSearch, upstreamRoutes, bundleAggregationStrategy).id("$ROUTE_ID-upstream-search")
            // Get extra pages if needed
            .dynamicRouter(beanMethod(OrganizationSearch::paginate)).id("$ROUTE_ID-paginate")
            // Aggregate all the information into a single bundle
            .beanMethod(OrganizationProcessor::aggregate).id("$ROUTE_ID-aggregate")
            // Remove all data from downstream prior to writing the new info
            .beanMethod(OrganizationDeleter::hardDelete).id("$ROUTE_ID-downstream-delete")
            // Post transaction bundle to downstream
            .to(createFhirTransactionRoute(fhirConfiguration.downstream.endpoint)).id("$ROUTE_ID-downstream-push")
            // Log info
            .beanMethod(OrganizationProcessor::logOutcome).id("${ROUTE_ID}-log-outcome")
    }
}
