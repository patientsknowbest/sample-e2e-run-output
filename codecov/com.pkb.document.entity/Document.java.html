package com.pkb.document.entity;

public class Document extends DocumentMetadata {

    public Document() {}

    public Document(DocumentMetadata metadata, byte[] content) {
        setId(metadata.getId());
        setStorage(metadata.getStorage());
        setIdInStorage(metadata.getIdInStorage());
        setAccountId(metadata.getAccountId());
        setAuthorPersonId(metadata.getAuthorPersonId());
        setMediaType(metadata.getMediaType());
        setFinalSubmitTime(metadata.getFinalSubmitTime());
        setEncryptedKeyInline(metadata.getEncryptedKeyInline());
        setAlgorithm(metadata.getAlgorithm());
        setSymmetricEncryptionKeyId(metadata.getSymmetricEncryptionKeyId());
        setNonce(metadata.getNonce());
        setDatapointLevelEncrypted(metadata.isDatapointLevelEncrypted());
        setDatapointLevelKeyId(metadata.getDatapointLevelKeyId());
        setContent(content);
    }

    private byte[] content;

    public byte[] getContent() {
        return content;
    }

    public void setContent(byte[] content) {
        this.content = content;
    }
}
