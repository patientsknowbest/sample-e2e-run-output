package com.pkb.document.entity;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Map;
import java.util.Optional;
import java.util.UUID;

public class DocumentsByIdAndAccount implements Serializable {

    private Map<UUID, Map<Long, UUID>> map;

    private DocumentsByIdAndAccount(Map<UUID, Map<Long, UUID>> map) {
        this.map = map;
    }

    public static Builder builder() {
        return new Builder();
    }

    public Optional<UUID> get(UUID docId, Long accountId) {
        UUID copyId = null;
        Map<Long, UUID> accountIdMap = map.get(docId);
        if (accountIdMap != null) {
            copyId = accountIdMap.get(accountId);
        }
        return Optional.ofNullable(copyId);
    }


    public static class Builder {

        private Map<UUID, Map<Long, UUID>> map;

        private Builder() {
            map = new HashMap<>();
        }

        public void add(UUID originalDocId, Long accountId, UUID docCopyId) {
            if (!map.containsKey(originalDocId)) {
                map.put(originalDocId, new HashMap<>());
            }

            map.get(originalDocId).put(accountId, docCopyId);
        }

        public DocumentsByIdAndAccount build() {
            return new DocumentsByIdAndAccount(map);
        }

    }

}
