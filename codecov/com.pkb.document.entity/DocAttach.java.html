package com.pkb.document.entity;

import org.hibernate.annotations.GenericGenerator;
import org.hibernate.annotations.Type;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.persistence.Temporal;
import javax.persistence.TemporalType;
import java.io.Serializable;
import java.time.Instant;
import java.util.Date;
import java.util.UUID;

import static com.pkb.infrastructure.hibernate.id.enhanced.ResettablePooledLoOptimizer.OPTIMIZER_CANONICAL_NAME;

/**
 * PKB-controlled entity for Tolven's doc.Document table.
 * We need better control over this.
 * <p>
 * Throw out Tolven's type inheritance, and make it easy to attach to EHRData (not just parent documents).
 * <p>
 * <p>
 * Table "doc.attachment"
 * Column | Type | Modifiers
 * ---------------------+-----------------------------+-----------
 * id | bigint | not null
 * description | character varying(1000) |
 * media_type | character varying(255) |
 * upload_time | timestamp without time zone |
 * account_id | bigint |
 * tolvenauthor_id | bigint |
 * attacheddocument_id | bigint |
 * parentdocument_id | bigint |
 * Indexes:
 * "attachment_pkey" PRIMARY KEY, btree (id)
 * Foreign-key constraints:
 * "fka7e14523508c6f07" FOREIGN KEY (tolvenauthor_id) REFERENCES core.account_user(id)
 * "fka7e14523c3f6c9c5" FOREIGN KEY (attacheddocument_id) REFERENCES doc.document(id)
 * "fka7e14523fd9d3eff" FOREIGN KEY (parentdocument_id) REFERENCES doc.document(id)
 * "fka7e14523ffbab5a8" FOREIGN KEY (account_id) REFERENCES core.account(id)
 * <p>
 * FOREIGN KEY and/or index to add: (parentehrdata_id) REFERENCES app.menu_data(id)
 *
 * @author robwhelan
 */
@Entity(name = "DocAttach")
@Table(schema = "doc", name = "attachment")
public class DocAttach implements Serializable {
    private static final long serialVersionUID = 1L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Id
    @GenericGenerator(name = "seq_doc_gen", strategy = "enhanced-sequence",
            parameters = {
                    @org.hibernate.annotations.Parameter(
                            name = "optimizer",
                            value = OPTIMIZER_CANONICAL_NAME),
                    @org.hibernate.annotations.Parameter(
                            name = "schema",
                            value = "doc"),
                    @org.hibernate.annotations.Parameter(
                            name = "sequence_name",
                            value = "seq_doc"),
                    @org.hibernate.annotations.Parameter(
                            name = "increment_size",
                            value = "50")
            }
    )
    @GeneratedValue(
            strategy = GenerationType.SEQUENCE,
            generator = "seq_doc_gen")
    private long id;

    @Column(name = "account_id")
    private long accountId;

    @Column(name = "parent_document_metadata_id")
    @Type(type = "pg-uuid")
    private UUID parentDocumentMetadataId;

    @Column(name = "menudata_id")
    private Long menuDataId;
    // better name: parentEHRDataId

    @Column(name = "attached_document_metadata_id")
    @Type(type = "pg-uuid")
    private UUID attachedDocumentMetadataId;

    @Column(name = "author_id")
    private String authorUid;

    @Column(length = 1000)
    private String description;

    @Column(name = "media_type")
    private String mediaType;

    @Temporal(TemporalType.TIMESTAMP)
    @Column(name = "upload_time")
    private Date uploadTime;

    /**
     * Hibernate needs an empty constructor; but use the other one preferably!
     */
    public DocAttach() {
    }

    /**
     * doesn't set description
     */
    public DocAttach(long accountId, String authorUid, UUID parentDocumentId, Long parentEHRDataId,
                     UUID attachedDocumentId, String mediaType, Instant uploadTime) {
        this.accountId = accountId;
        this.authorUid = authorUid;
        this.parentDocumentMetadataId = parentDocumentId;
        this.menuDataId = parentEHRDataId;
        this.attachedDocumentMetadataId = attachedDocumentId;
        this.mediaType = mediaType;

        // default value
        setUploadTime(Date.from(uploadTime));
    }

    /**
     * id check only; "true" if neither has an id set.
     */
    @Override
    public boolean equals(Object obj) {
        if (!(obj instanceof DocAttach)) {
            return false;
        }
        if (this.getId() == ((DocAttach) obj).getId()) {
            return true;
        }
        return false;
    }

    @Override
    public String toString() {
        return "DocAttachment: " + getId();
    }

    @Override
    public int hashCode() {
        if (getId() == 0) {
            throw new IllegalStateException("id not yet established in DocAttachment object");
        }
        return Long.valueOf(getId()).hashCode();
    }

    /**
     * The unique and primary key of the document.
     */
    public long getId() {
        return id;
    }

    public void setId(long id) {
        this.id = id;
    }

    public String getMediaType() {
        return mediaType;
    }

    public void setMediaType(String mediaType) {
        this.mediaType = mediaType;
    }

    public long getAccountId() {
        return accountId;
    }

    public void setAccountId(long accountId) {
        this.accountId = accountId;
    }

    public String getDescription() {
        return description;
    }

    public void setDescription(String description) {
        this.description = description;
    }

    public Date getUploadTime() {
        return uploadTime;
    }

    public void setUploadTime(Date uploadTime) {
        this.uploadTime = uploadTime;
    }

    public UUID getParentDocumentMetadataId() {
        return parentDocumentMetadataId;
    }

    public void setParentDocumentMetadataId(UUID parentDocumentMetadataId) {
        this.parentDocumentMetadataId = parentDocumentMetadataId;
    }

    public UUID getAttachedDocumentMetadataId() {
        return attachedDocumentMetadataId;
    }

    public void setAttachedDocumentMetadataId(UUID attachedDocumentMetadataId) {
        this.attachedDocumentMetadataId = attachedDocumentMetadataId;
    }

    public Long getMenuDataId() {
        return menuDataId;
    }

    public void setMenuDataId(Long menuDataId) {
        this.menuDataId = menuDataId;
    }

    public String getAuthorUid() {
        return authorUid;
    }

    public void setAuthorUid(String authorUid) {
        this.authorUid = authorUid;
    }
}
