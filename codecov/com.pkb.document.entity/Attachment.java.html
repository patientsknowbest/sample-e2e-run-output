package com.pkb.document.entity;

import com.pkb.util.FileUtil;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.Nullable;

import java.io.Serializable;
import java.util.Date;
import java.util.UUID;

/**
 * A DTO class to represents a attachment to a Message
 *
 * @author pravinam
 */
public class Attachment implements Serializable {

    /**
     * Version number for this entity object
     */
    private static final long serialVersionUID = 1L;

    private Long id;

    private Long accountId;

    private UUID docMetadataId;

    private Long parentMessageId;

    private Long userId;

    private String filename;

    private String extension;

    private String mediaType;

    private Date uploadTime;

    private byte[] content;

    private byte[] thumbnailContent;

    public boolean isBrowserImage() {
        if ("image/jpeg".equals(mediaType) || "image/png".equals(mediaType) || "image/gif".equals(mediaType)
                || "jpeg".equals(mediaType) || "jpg".equals(mediaType) || "png".equals(mediaType) || "gif".equals(mediaType)) {
            return true;
        }

        // don't trust media type; try file name
        String fileName = StringUtils.trimToEmpty(filename).toLowerCase();
        if (fileName.endsWith(".jpg") || fileName.endsWith(".jpeg") || fileName.endsWith(".png") || fileName.endsWith(".gif")) {
            return true;
        }

        return false;
    }

    public boolean isPdf() {
        return FileUtil.isPdf(filename, mediaType);
    }

    public boolean isConvertibleToPdf() {
        return FileUtil.isConvertibleToPdf(filename, mediaType);
    }

    public boolean isViewable() {
        return FileUtil.isViewable(filename);
    }

    public boolean isStreamable() {
        return FileUtil.isStreamable(filename);
    }

    public String getVideoType() {
        return FileUtil.getVideoType(filename);
    }

    public Long getId() {
        return id;
    }

    public String getFilename() {
        return filename;
    }

    public void setFilename(String filename) {
        this.filename = filename;
    }

    public Date getUploadTime() {
        return uploadTime;
    }

    public void setUploadTime(Date uploadTime) {
        this.uploadTime = uploadTime;
    }

    public String getMediaType() {
        return mediaType;
    }

    public void setMediaType(String mediaType) {
        this.mediaType = mediaType;
    }

    public@Nullable String getMediaPrefix() {
        if (getMediaType() == null) {
            return null;
        }
        String[] parts = getMediaType().split("/");
        return parts[0];
    }

    public @Nullable String getMediaSuffix() {
        if (getMediaType() == null) {
            return null;
        }
        String[] parts = getMediaType().split("/");
        return parts[1];
    }

    public Long getAccountId() {
        return accountId;
    }

    public void setAccountId(Long accountId) {
        this.accountId = accountId;
    }

    public Long getParentMessageId() {
        return parentMessageId;
    }

    public void setParentMessageId(Long parentMessageId) {
        this.parentMessageId = parentMessageId;
    }

    public byte[] getContent() {
        return content;
    }

    public void setContent(byte[] content) {
        this.content = content;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public Long getUserId() {
        return userId;
    }

    public void setUserId(Long userId) {
        this.userId = userId;
    }

    public UUID getDocMetadataId() {
        return docMetadataId;
    }

    public void setDocMetadataId(UUID docMetadataId) {
        this.docMetadataId = docMetadataId;
    }

    public byte[] getThumbnailContent() {
        return thumbnailContent;
    }

    public void setThumbnailContent(byte[] thumbnailContent) {
        this.thumbnailContent = thumbnailContent;
    }

    public boolean isReportable(boolean isPatientInOwnAccount, boolean reportableMessage) {
        return isPatientInOwnAccount && reportableMessage && isPdf();
    }

    public String getReportCssClass(boolean hasBeenReported) {
        String cssClass = "btn btn-primary report-link";
        if (hasBeenReported) {
            cssClass += " disabled";
        }
        return cssClass;
    }

}
