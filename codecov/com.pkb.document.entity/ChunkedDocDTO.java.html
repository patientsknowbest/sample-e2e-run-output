package com.pkb.document.entity;

import java.io.Serializable;
import java.util.UUID;

/**
 * A common DTO class with basic attributes for every document which needs to uploaded/downbload in chunks
 */
public class ChunkedDocDTO implements Serializable {

    private static final long serialVersionUID = 3752705865909534747L;

    private UUID metadataId;

    private UUID parentMetadataId; // if main document then parent id will be null

    private byte[] content;

    private String mediaType;

    private String name;

    private boolean sanitized;

    public UUID getMetadataId() {
        return metadataId;
    }

    public void setMetadataId(UUID metadataId) {
        this.metadataId = metadataId;
    }

    public UUID getParentMetadataId() {
        return parentMetadataId;
    }

    public void setParentMetadataId(UUID parentMetadataId) {
        this.parentMetadataId = parentMetadataId;
    }

    public byte[] getContent() {
        return content;
    }

    public void setContent(byte[] content) {
        this.content = content;
    }

    public String getMediaType() {
        return mediaType;
    }

    public void setMediaType(String mediaType) {
        this.mediaType = mediaType;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    /**
     * @return Whether the document has gone through MIME detection and corresponding sanitization
     */
    public boolean isSanitized() {
        return sanitized;
    }

    public void setSanitized(boolean sanitized) {
        this.sanitized = sanitized;
    }

}
