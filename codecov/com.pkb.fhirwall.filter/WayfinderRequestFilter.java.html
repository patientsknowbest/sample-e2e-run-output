package com.pkb.fhirwall.filter;

import com.pkb.fhirwall.config.FhirwallConfig;
import org.springframework.stereotype.Component;

import javax.servlet.*;
import java.io.IOException;
import java.util.Objects;

/**
 * PHR-11330
 * Until Wayfinder moves off their mtls + id token security model, we are supporting 2 hosts. 1 is for wayfinder (so we can
 * perform mtls) and the other is for all other clients (which will use oauth2). <i>Almost</i> everything else in the app should
 * be the same (note com.pkb.fhirwall.service.resource.wayfinder.AppointmentResourceRemover).
 *
 * This filter adds an attribute to the request so we can easily tell later on whether the request is a wayfinder-specific
 * request or not.
 */
@Component
public class WayfinderRequestFilter implements Filter {

    public static String WAYFINDER_REQUEST = "WAYFINDER_REQUEST";
    private FhirwallConfig fhirwallConfig;

    public WayfinderRequestFilter(FhirwallConfig fhirwallConfig) {
        this.fhirwallConfig = fhirwallConfig;
    }
    @Override
    public void doFilter(ServletRequest servletRequest, ServletResponse servletResponse, FilterChain filterChain) throws IOException, ServletException {
        var wayfinderHost = fhirwallConfig.getWayfinderHost();
        var isWayfinderRequest = Objects.equals(wayfinderHost, servletRequest.getServerName());
        servletRequest.setAttribute(WAYFINDER_REQUEST, isWayfinderRequest);
        filterChain.doFilter(servletRequest, servletResponse);
    }
}
