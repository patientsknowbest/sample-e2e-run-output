package com.pkb.kms.client.core;

import com.pkb.common.config.BaseConfig;

import static com.pkb.kms.shared.ConstantsKt.KMS_CLIENT_ID_DEFAULT;

public interface KmsClientConfig extends BaseConfig {

    // Implemented in java because the handling of default methods is easier

    @Override
    default String getBaseURL() {
        return getConfigStorage().getString("kms.baseurl", "http://kms:8080");
    }

    default int getAccountKeysBatchSize() {
        return getConfigStorage().getInt("kms.key.query.batch.size", 75);
    }

    /**
     * @deprecated This is now ignored, the KMS cannot be disabled.
     */
    @Deprecated
    default boolean enabled() {
        return getConfigStorage().getBoolean("kms.key.loading.enabled", true);
    }

    default String getClientId() {
        return getConfigStorage().getString("kms.clientId", KMS_CLIENT_ID_DEFAULT);
    }

    default long httpClientConnectTimeout() {
        return getConfigStorage().getLong("kms.http.connect.timeout.millis", 10000L);
    }

    default long httpClientReadTimeout() {
        return getConfigStorage().getLong("kms.http.read.timeout.millis", 30000L);
    }

    default long httpClientWriteTimeout() {
        return getConfigStorage().getLong("kms.http.read.timeout.millis", 10000L);
    }
}
