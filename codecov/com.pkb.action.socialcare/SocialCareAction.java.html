package com.pkb.action.socialcare;

import com.pkb.action.BaseAction;
import com.pkb.integration.socialcare.model.ErrorDetail;
import com.pkb.integration.socialcare.model.SocialCareData;
import com.pkb.datamodel.Email;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.NationalId;
import com.pkb.user.entity.PKBPerson;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.List;
import java.util.Optional;
import java.util.UUID;

import static com.pkb.entities.enums.NationalIdType.NHS_NUMBER;
import static org.apache.commons.collections4.CollectionUtils.isEmpty;

public class SocialCareAction extends BaseAction {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final String HOME = "home";
    protected static final String NO_DATA = "socialCare.txt.no_data";
    protected static final String NO_NHS_NUMBER = "socialCare.txt.no_nhs_number";
    protected static final String CAN_NOT_CONNECT = "socialCare.txt.can_not_connect";
    protected static final String UNAUTHORIZED = "socialCare.txt.unauthorized";

    private final String tab = "socialcare";

    private List<SocialCareData> socialCareDataList;
    private String infoText = NO_DATA;

    @Override
    public String execute() {
        if (showSocialCareForPatient()) {
            return SUCCESS;
        }

        return HOME;
    }

    public String fetchSocialCareData() {
        if (!showSocialCareForPatient()) {
            return SUCCESS;
        }

        PKBPerson patient = loggedInUserUtil.getContextOrAccessingUser(getLoggedInEHRRequestContext());
        UUID userPublicId = patient.getPublicId();
        Email userEmail = patient.getEmail();
        Optional<NationalId> maybeNationalId = patient.getNationalIdAnyTypeValidForSameCountryAsType(NHS_NUMBER);
        if (maybeNationalId.isEmpty()) {
            infoText = NO_NHS_NUMBER;
            return SUCCESS;
        }

        try {
            socialCareDataList = userManager.getPathfindersData(maybeNationalId.get().getValue(), userPublicId, userEmail);
        } catch (Exception e) {
            LOGGER.error("Can't get social care data for user {}", userPublicId, e);
        } finally {
            if (socialCareDataList == null) {
                infoText = CAN_NOT_CONNECT;
            } else if (hasUnathorizedError()) {
                infoText = UNAUTHORIZED;
            }
        }

        return SUCCESS;
    }

    @Override
    public String getCustomStyle() {
        return MOBILE_OPTIMIZED_STYLE;
    }

    public boolean hasSocialCareData() {
        if (isEmpty(socialCareDataList)) {
            return false;
        }

        return socialCareDataList.stream().anyMatch(SocialCareData::hasData);
    }

    public String getTab() {
        return tab;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public List<SocialCareData> getSocialCareDataList() {
        return socialCareDataList;
    }

    public String getInfoText() {
        return infoText;
    }

    protected boolean hasUnathorizedError() {
        return socialCareDataList.stream()
                .filter(SocialCareData::hasError)
                .map(SocialCareData::getError)
                .anyMatch(ErrorDetail::isUnauthorized);
    }

    private boolean showSocialCareForPatient() {
        return isSocialCareVisibile(false);
    }

}
