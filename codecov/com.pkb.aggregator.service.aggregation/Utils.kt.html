package com.pkb.aggregator.service.aggregation

import com.pkb.aggregator.R4Coding
import com.pkb.aggregator.R4DomainResource
import com.pkb.aggregator.R4IdType
import com.pkb.aggregator.R4Identifier
import com.pkb.aggregator.extension.findAllSourcedReference
import com.pkb.aggregator.extension.simpleDownstreamResourceId
import com.pkb.aggregator.extension.simpleResourceId
import com.pkb.aggregator.persistence.entity.embeddable.MapEntryKey
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamIdentifier
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource

fun R4Identifier.toMapEntryKeys(): List<MapEntryKey> {
    return this.findAllSourcedReference().map {
        MapEntryKey(
            upstreamResource = UpstreamResource(it.upstreamSource, it.upstreamResourceId),
            identifier = UpstreamIdentifier(this.system, this.value))
    }
}

fun R4Coding.toMapEntryKeys(): List<MapEntryKey> {
    return this.findAllSourcedReference().map {
        MapEntryKey(
            upstreamResource = UpstreamResource(it.upstreamSource, it.upstreamResourceId),
            identifier = UpstreamIdentifier(this.system, this.code))
    }
}

fun <T> updateResourceId(resources: List<T>, idGenerator: (T) -> String)
        where T : R4DomainResource =
    resources.associateBy {
        updateResourceId(it, idGenerator)
    }

fun <T> updateResourceId(resources: List<T>) where T : R4DomainResource =
    updateResourceId(resources) {
        it.simpleDownstreamResourceId()
    }

fun <T> updateResourceId(resource: T, idGenerator: (T) -> String): String where T : R4DomainResource {
    // Set new, deterministic resource id
    val newId = idGenerator(resource)
    resource.setId(R4IdType(newId))
    return resource.simpleResourceId()
}
