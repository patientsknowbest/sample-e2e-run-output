package com.pkb.aggregator.service.aggregation

import com.pkb.aggregator.R4Extension
import com.pkb.aggregator.persistence.entity.embeddable.UpstreamResource
import com.pkb.aggregator.service.FHIR_TYPE_ORGANIZATION
import com.pkb.aggregator.service.reference.PhrReferenceHelper
import com.pkb.aggregator.service.reference.ReferenceExtractor
import org.hl7.fhir.instance.model.api.IDomainResource

/**
 * Interface defining the operations required to merge multiple upstream resources representing the same real-world
 * 'thing' (person, organisation etc) into a single resource for the downstream store.
 *
 * The basic process for doing this is:
 *
 * 1. Figure out what the downstream resource id is going to be
 * 2. Construct an empty resource instance and iterate through the upstream resources to copy them over
 * 3. Store any new identifiers (national IDs, org IDs, etc) for the resource that were detected during processing
 * 4. Update all fhir references so that it all lines up in the downstream (e.g. downstream patient resource correctly
 *    refers to downstream org resource, rather than still pointing to the upstream resource id which won't be valid
 *    in the downstream store)
 */
interface ResourceAggregator<T> where T : IDomainResource {

    fun merge(resources: List<IDomainResource>, currentDownstreamId: String?, references: ReferenceExtractor.IdentifierMap): T {

        val downstreamId = deriveDownstreamResourceId(currentDownstreamId, resources)
        val downstreamResource = createDownstreamInstance(downstreamId)

        resources.forEach {
            mergeResource(downstreamResource, it, references)
        }

        mergeIdentifiers(downstreamResource)
        updateReferences(downstreamResource, references)

        return downstreamResource
    }

    fun merge(
        resources: List<IDomainResource>,
        currentDownstream: T?,
        currentDownstreamId: String?,
        references: ReferenceExtractor.IdentifierMap
    ): T {

        if (currentDownstream == null) {
            return merge(resources, currentDownstreamId, references)
        }

        resources.forEach {
            mergeResource(currentDownstream, it, references)
        }

        mergeIdentifiers(currentDownstream)
        updateReferences(currentDownstream, references)

        return currentDownstream
    }

    fun createTranslatedPhrExtensionReferences(
        source: UpstreamResource,
        extensions: List<R4Extension>,
        references: ReferenceExtractor.IdentifierMap,
        referenceHelper: PhrReferenceHelper,
    ): MutableList<R4Extension> {

        return extensions
            .map { upstreamExtension ->
                val downstreamExtension = upstreamExtension.copy()

                // If it's not a PHR custom reference, do nothing and return the unmodified copy
                val phrValueReference = referenceHelper.findPhrValueReference(upstreamExtension)
                if (phrValueReference != null) {
                    val needle = UpstreamResource(source.upstreamSource, phrValueReference.toString())
                    val newReference = when (phrValueReference.resourceType) {
                        FHIR_TYPE_ORGANIZATION -> references.getOrganizationReference(needle)
                        else -> throw RuntimeException("Unexpected resource type ${phrValueReference.resourceType}")
                    }
                    downstreamExtension.setValue(newReference)
                }
                downstreamExtension
            }
            .toMutableList()
    }

    fun createDownstreamInstance(downstreamId: String): T

    fun mergeResource(
        downstreamResource: T,
        upstreamResource: IDomainResource,
        references: ReferenceExtractor.IdentifierMap
    )

    fun stripResource(downstreamResource: T, sourceTag: R4Extension)

    fun mergeIdentifiers(downstreamResource: T)

    fun updateReferences(downstreamResource: T, references: ReferenceExtractor.IdentifierMap) {}

    fun deriveDownstreamResourceId(currentDownstreamId: String?, resources: List<IDomainResource>): String
}
