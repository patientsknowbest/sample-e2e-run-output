package com.pkb.domain.exception;

import java.util.stream.Stream;

import static java.lang.String.format;

public class OrgNetworkException extends RuntimeException {

    public static final String ORG_ALREADY_IN_NETWORK_CODE = "org.already.in.network";
    public static final String ORG_ALREADY_IN_NETWORK = "Org %d is already in a network";
    public static final String INVITED_ORG_NO_COORD_CODE = "org.no.coord";
    public static final String INVITED_ORG_NO_COORD = "Org %d does not have any org coords with access to the org key";

    private final String errorCode;
    private final String[] arguments;

    public OrgNetworkException(String errorCode, String message, Object... arguments) {
        super(format(message, arguments));
        this.errorCode = errorCode;
        this.arguments = Stream.of(arguments).map(Object::toString).toArray(String[]::new);
    }

    public String getErrorCode() {
        return errorCode;
    }

    public String[] getArguments() {
        return arguments;
    }
}
