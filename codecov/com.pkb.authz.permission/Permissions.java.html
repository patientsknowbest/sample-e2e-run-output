package com.pkb.authz.permission;

import com.pkb.authz.action.Action;
import io.vavr.collection.HashMap;
import io.vavr.collection.List;
import io.vavr.collection.Map;
import io.vavr.collection.Set;
import io.vavr.collection.Stream;
import org.immutables.value.Value;
import org.immutables.value.Value.Derived;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Parameter;
import org.immutables.value.Value.Style.ImplementationVisibility;
import org.immutables.vavr.encodings.VavrEncodingEnabled;
import org.jetbrains.annotations.NotNull;

import static com.pkb.authz.permission.ImmutableDenied.denied;
import static io.vavr.API.List;
import static io.vavr.API.Set;
import static java.lang.String.format;
import static java.util.function.Function.identity;
import static java.util.stream.Collectors.toMap;

@Immutable
@VavrEncodingEnabled
@Value.Style(
        of = "permissions",
        visibility = ImplementationVisibility.PUBLIC,
        defaults = @Value.Immutable(builder = false, copy = false))
public abstract class Permissions {
    private static final String AMBIGIOUS_PERMISSION_DEFINITION_FOR_ACTION = "Ambigious permission definition for Action=[%s]: Permission-a=[%s] vs Permission-b=[%s]";

    @Parameter
    public abstract Set<Permission> permissions();

    public final @NotNull Set<Action> allowedActions() {
        return permissions()
                .filter(Permission::allowed)
                .map(Permission::action);
    }

    public final @NotNull Set<Action> deniedActions() {
        return permissions()
                .filter(Permission::denied)
                .map(Permission::action);
    }

    public final boolean allowToPerform(@NotNull Set<Action> requestedActions) {
        return requestedActions.diff(allowedActions()).isEmpty();
    }
    
    public final boolean allowToPerform(@NotNull Action requestedAction) {
        return allowToPerform(Set(requestedAction));
    }

    public final boolean isDeniedToPerformActions(@NotNull Set<Action> requestedActions) {
        return !allowToPerform(requestedActions);
    }

    public final boolean isDeniedToPerformAction(@NotNull Action requestedAction) {
        return !allowToPerform(Set(requestedAction));
    }

    public final @NotNull List<Permission> permissionsForActions(@NotNull Set<Action> requestedActions) {
        List<Permission> justificationsForAllowedActions = requestedActions.intersect(allowedActions())
                .flatMap(permissionsByActions()::get)
                .toList();
        return allowToPerform(requestedActions) ? justificationsForAllowedActions : collectDiscrepancies(requestedActions);
    }

    public final @NotNull List<Justification> justifications(@NotNull Action action) {
        var sAction = Set(action);
        return (allowToPerform(sAction) ? 
                permissionsForActions(sAction) :
                collectDiscrepancies(sAction))
                .map(Permission::justifications).flatMap(i -> i);
    }

    @Derived
    public @NotNull Map<Action, Permission> permissionsByActions() {
        return HashMap.ofAll(permissions()
                .collect(toMap(Permission::action, identity(), (permissionA, permissionB) -> {
                    String actionName = permissionA.action().getClass().getSimpleName();
                    throw new IllegalStateException(format(AMBIGIOUS_PERMISSION_DEFINITION_FOR_ACTION, actionName, permissionA, permissionB));
                })));
    }

    private List<Permission> collectDiscrepancies(@NotNull Set<Action> requestedActions) {
        Map<Action, Permission> permissionsByActions = permissionsByActions();

        Stream<Permission> deniedPermissions = requestedActions.intersect(deniedActions())
                .toStream()
                .flatMap(permissionsByActions::get);

        Set<Action> declaredActions = permissions().map(Permission::action);

        Stream<Permission> undeclaredPermissions = Stream.ofAll(requestedActions.diff(declaredActions))
                .map(action -> denied(action,
                        List(ImmutableTextualJustification
                                .justification(String.format("No permission is assigned for action=[%s]. Deny by default policy is applied.", action)))));

        return Stream.concat(deniedPermissions, undeclaredPermissions).toList();
    }
}
