package com.pkb.authz.permission;

import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;
import org.immutables.vavr.encodings.VavrEncodingEnabled;
import org.jetbrains.annotations.NotNull;

import static java.util.stream.Collectors.joining;

/**
 * The positive result of an authorization check. Indicates the the operation should proceed.
 */
@Immutable
@Style(of = "allowed", passAnnotations = { NotNull.class })
@VavrEncodingEnabled
@SuppressWarnings("varargs")
public abstract class Allowed implements Permission {

    private static final long serialVersionUID = 1L;

    @Override
    public final boolean allowed() {
        return true;
    }

    @Override
    public String toString() {
        return new StringBuilder("Action=[")
                .append(action())
                .append("] is allowed: ")
                .append(justifications().map(Justification::stringify).collect(joining(" and ", "[", "]")))
                .toString();
    }
}
