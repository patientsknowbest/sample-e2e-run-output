package com.pkb.authz.permission;

import com.pkb.authz.action.Action;
import io.vavr.collection.List;
import org.immutables.value.Value.Parameter;
import org.jetbrains.annotations.NotNull;

import java.io.Serializable;

/**
 * Represents the outcome of {@link com.pkb.authz.policy.Policy} check, when all of the conditions are met.
 */
public interface Permission extends Serializable {

    boolean allowed();

    default boolean denied() {
        return !allowed();
    }

    /**
     * Action that was granted/denied.
     */
    @Parameter
    Action action();

    /**
     * We need to provide reasons why execution of an action was allowed/forbiden.
     */
    @Parameter
    @NotNull
    List<Justification> justifications();
}
