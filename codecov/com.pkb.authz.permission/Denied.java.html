package com.pkb.authz.permission;

import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;
import org.immutables.vavr.encodings.VavrEncodingEnabled;
import org.jetbrains.annotations.NotNull;

import static java.util.stream.Collectors.joining;

/**
 * The negative result of an authorization check. Indicates the the operation should <em>not</em> proceed.
 */
@Immutable
@Style(of = "denied", passAnnotations = { NotNull.class })
@VavrEncodingEnabled
@SuppressWarnings("varargs")
public abstract class Denied implements Permission {

    private static final long serialVersionUID = 1L;

    @Override
    public final boolean allowed() {
        return false;
    }

    @Override
    public String toString() {
        return new StringBuilder("Action=[")
                .append(action())
                .append("] is denied because: ")
                .append(justifications()
                        .map(Justification::stringify)
                        .collect(joining(", ", "[", "]")))
                .toString();
    }

}
