package com.pkb.action.sso.systmone;

import com.pkb.action.BaseAction;
import com.pkb.domain.SSOService;
import org.springframework.beans.factory.annotation.Autowired;

import javax.validation.constraints.NotBlank;
import javax.validation.constraints.Size;
import java.time.ZoneId;
import java.time.format.DateTimeFormatter;
import java.util.Locale;

public class GenerateTestUrl extends BaseAction {

    private String content;

    @NotBlank
    private String nhsNumber;

    @Size(min = 20, max = 20)
    private String ssoUsername;

    @Size(min = 20, max = 20)
    private String ssoPassword;

    @Size(min = 32, max = 32)
    private String key;

    @Size(min = 32, max = 32)
    private String iv;

    @NotBlank
    private String externalId;

    protected SSOService ssoService;

    @Override public String input() throws Exception {
        if (!config.isAllowSsoTestingPages()) {
            return NOT_FOUND;
        }

        return super.input();
    }

    @Override
    public String execute() throws Exception {
        if (this.isRequestNotValid()) {
            return input();
        }

        try {
            DateTimeFormatter formatter = DateTimeFormatter.ofPattern("yyyy-MM-dd'T'HH:mm:ss").withLocale(Locale.UK).withZone(ZoneId.of("UTC"));
            String timeString = dateTimeService.nowLocalDateTime().format(formatter);
            String toEncrypt = "username=" + ssoUsername + "&nhsnumber=" + nhsNumber + "&password=" + ssoPassword + "&timestamp=" + timeString;
            content = ssoService.encryptSystmOneUrlContent(toEncrypt, key, iv);
        } catch (Exception e) {
            //noinspection StrutsI18nError,SSBasedInspection
            addActionError(e.getMessage());
            return input();
        }

        return SUCCESS;
    }

    public String getNhsNumber() {
        return nhsNumber;
    }

    public void setNhsNumber(String nhsNumber) {
        this.nhsNumber = nhsNumber;
    }

    public String getSsoUsername() {
        return ssoUsername;
    }

    public void setSsoUsername(String ssoUsername) {
        this.ssoUsername = ssoUsername;
    }

    public String getSsoPassword() {
        return ssoPassword;
    }

    public void setSsoPassword(String ssoPassword) {
        this.ssoPassword = ssoPassword;
    }

    public String getKey() {
        return key;
    }

    public void setKey(String key) {
        this.key = key;
    }

    public String getIv() {
        return iv;
    }


    public void setIv(String iv) {
        this.iv = iv;
    }

    public String getExternalId() {
        return externalId;
    }

    public void setExternalId(String externalId) {
        this.externalId = externalId;
    }

    public String getContent() {
        return content;
    }

    @Autowired
    public void setSsoService(SSOService ssoService) {
        this.ssoService = ssoService;
    }
}
