package com.pkb.action.sso.systmone;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.domain.SSOService;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.entities.enums.SSOType;
import com.pkb.service.user.impl.UserManager;
import com.pkb.sso.SsoMetrics;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.PersonAmbiguityHandler;
import io.prometheus.client.Summary.Timer;
import io.vavr.control.Option;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;
import org.springframework.beans.factory.annotation.Autowired;

import java.time.LocalDateTime;
import java.time.ZoneId;
import java.time.format.DateTimeFormatter;
import java.util.HashMap;
import java.util.Locale;
import java.util.Map;
import java.util.Optional;
import java.util.regex.Pattern;

import static org.apache.commons.lang3.StringUtils.isBlank;

public class SystmOneSSOSessionRequestAction extends BaseAction {

    private static final Pattern SPLIT_URL_INDIVIDUAL_PARAMS = Pattern.compile("&");
    private static final Pattern SPLIT_URL_INDIVIDUAL_PARAM_VALUE = Pattern.compile("=");

    static int URL_REPLAY_TIMEOUT_SECONDS = 20;

    private SSOService ssoService;

    private SsoMetrics ssoMetrics;

    private String password;
    private String username;

    private String content;
    private String externalId;

    public String processRequest() {
        if (!config.isSystmOneSsoEnabled()) {
            return handleError("ssoSystmOneError.err.systmone_sso_disabled");
        }
        Timer timer = ssoMetrics.startTimerForProcessingSessionRequest(SSOType.SYSTMONE);
        try {
            if (isBlank(externalId)) {
                return handleError("ssoSystmOneError.err.missing_external_id");
            }

            if (isBlank(content)) {
                return handleError("ssoSystmOneError.err.missing_content");
            }

            return ssoService.decryptUrlContent(externalId, content)
                    .fold(this::handleError, this::processDecryptedContent);
        } finally {
            timer.observeDuration();
        }

    }

    private String processDecryptedContent(@NotNull String decryptedContent) {
        Map<String, String> map = getUrlParamsMap(decryptedContent);
        username = map.get("username");
        if (isBlank(username)) {
            return handleError("ssoSystmOneError.err.missing_username");
        }

        password = map.get("password");
        if (isBlank(password)) {
            return handleError("ssoSystmOneError.err.missing_password");
        }

        String nhsnumber = map.get("nhsnumber");
        if (isBlank(nhsnumber)) {
            return handleError("ssoSystmOneError.err.missing_nhs_number");
        }

        try {
            Option<PKBPerson> maybePerson = NationalIdType.NHS_NUMBER.getValidNationalIdAndType(nhsnumber)
                    .map(userManager::getPKBPersonByNationalId)
                    .map(e -> PersonAmbiguityHandler.getMaybePersonOrNone(e, "nationalId"))
                    .orElse(Option.none());

            if (maybePerson.isEmpty()) {
                return handleError("ssoSystmOneError.err.patient_not_found");
            }

            contextUserId = maybePerson.get().getId();
        } catch (Exception ignored) {
            return handleError("ssoSystmOneError.err.patient_not_found");
        }

        String timestamp = map.get("timestamp");
        if (isBlank(timestamp)) {
            return handleError("ssoSystmOneError.err.missing_timestamp");
        }

        return processTimestamp(timestamp)
                .map(this::handleError)
                .orElseGet(this::checkAuthRecord);
    }

    private String checkAuthRecord() {
        if (ssoService.authRecordExists(username, SSOType.SYSTMONE)) {
            return Action.SUCCESS;
        }

        return handleError("sso.err.general_authentication_error");
    }

    private Optional<String> processTimestamp(@NotNull String timestamp) {
        //Incoming timestamp format: 2020-03-30T08:51:20
        DateTimeFormatter formatter = DateTimeFormatter.ofPattern("yyyy-MM-dd'T'HH:mm:ss").withLocale(Locale.UK).withZone(ZoneId.of("UTC"));
        LocalDateTime datetime;
        try {
            datetime = LocalDateTime.parse(timestamp, formatter);
        } catch (RuntimeException ignored) {
            return Optional.of("ssoSystmOneError.err.invalid_timestamp");
        }

        //You get a grace period of X seconds from the point of clicking the button in SystmOne to us processing the request
        // This is to prevent the replay of URLs which have already been generated.
        // Also if the timestamp is in the future then something has messed up at the generating end and the URL is
        // unsafe to use as it would allow for replay, so reject as well.
        LocalDateTime adjustedDateTime = datetime.plusSeconds(URL_REPLAY_TIMEOUT_SECONDS);
        LocalDateTime now = LocalDateTime.now(ZoneId.of("UTC"));
        if (datetime.isAfter(now)) {
            return Optional.of("ssoSystmOneError.err.future_timestamp");
        } else if (adjustedDateTime.isBefore(now)) {
            return Optional.of("ssoSystmOneError.err.expired_timestamp");
        }

        return Optional.empty();
    }

    @NotNull
    private Map<String, String> getUrlParamsMap(@NotNull String urlParams) {
        String[] params = SPLIT_URL_INDIVIDUAL_PARAMS.split(urlParams);
        Map<String, String> map = new HashMap<>();
        for (String param : params) {
            String name = SPLIT_URL_INDIVIDUAL_PARAM_VALUE.split(param)[0].trim().toLowerCase();
            try {
                String value = SPLIT_URL_INDIVIDUAL_PARAM_VALUE.split(param)[1].trim();
                map.put(name, value);
            } catch (Exception ignored) {
                //If we have a parameter name, but no value at all, then add empty for the value
                //This is so we can get more specific handling of what's wrong later on, and deliver a more
                // useful error message to the user.
                if (StringUtils.isNotBlank(name)) {
                    map.put(name, "");
                }
            }
        }
        return map;
    }

    private String handleError(@NotNull String errorCode) {
        addActionError(getText(errorCode));
        ssoMetrics.incrementUnSuccessfulSessionRequest(SSOType.SYSTMONE, errorCode);
        return Action.ERROR;
    }

    private String handleError(@NotNull SSOService.DecryptionError errorCode) {
        return handleError(errorCode.getMessageKey());
    }

    @Autowired
    public void setSsoService(SSOService ssoService) {
        this.ssoService = ssoService;
    }

    @Autowired
    public void setSsoMetrics(SsoMetrics ssoMetrics) {
        this.ssoMetrics = ssoMetrics;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public String getPassword() {
        return password;
    }

    public String getUsername() {
        return username;
    }

    public void setPassword(String password) {
        this.password = password;
    }

    public void setUsername(String username) {
        this.username = username;
    }

    public String getContent() {
        return content;
    }

    public void setContent(String content) {
        this.content = content;
    }

    public String getExternalId() {
        return externalId;
    }

    public void setExternalId(String externalId) {
        this.externalId = externalId;
    }
}
