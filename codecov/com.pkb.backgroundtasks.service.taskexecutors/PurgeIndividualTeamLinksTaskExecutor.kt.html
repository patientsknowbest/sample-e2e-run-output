package com.pkb.backgroundtasks.service.taskexecutors

import com.pkb.backgroundtasks.AccountPrivateId
import com.pkb.backgroundtasks.AccountPublicId
import com.pkb.backgroundtasks.OrgPrivateId
import com.pkb.backgroundtasks.entity.AccountTaskDetails
import com.pkb.backgroundtasks.entity.AccountTaskType
import com.pkb.backgroundtasks.repository.AccountUserRepository
import com.pkb.crypto.dto.AccountKeysDTO
import com.pkb.entities.enums.AccountUserStatus
import org.springframework.stereotype.Component
import org.springframework.transaction.annotation.Transactional

@Component
class PurgeIndividualTeamLinksTaskExecutor(private val accountUserRepository: AccountUserRepository) : AccountTaskExecutor<PurgeIndividualTeamLinksTaskExecutor, PurgeIndividualTeamLinksTaskDetails>(
    PurgeIndividualTeamLinksTaskDetails::class.java
) {
    @Transactional
    override fun performTask(
        accountPublicId: AccountPublicId,
        accountKeysDTO: AccountKeysDTO,
        accountPrivateId: AccountPrivateId,
        taskDetails: PurgeIndividualTeamLinksTaskDetails
    ): Boolean {
        accountUserRepository.findInterTeamLinks(accountPrivateId, taskDetails.orgPrivateIds)
            .forEach { au -> au.status = AccountUserStatus.INACTIVE }
        return true
    }
}

data class PurgeIndividualTeamLinksTaskDetails(val orgPrivateIds: List<OrgPrivateId>) : AccountTaskDetails<PurgeIndividualTeamLinksTaskDetails>(
    AccountTaskType.PURGE_INDIVIDUAL_TEAM_LINKS
) {
    override fun getSelf(): PurgeIndividualTeamLinksTaskDetails = this
}
