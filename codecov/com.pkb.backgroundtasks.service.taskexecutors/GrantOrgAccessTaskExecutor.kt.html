package com.pkb.backgroundtasks.service.taskexecutors

import com.pkb.backgroundtasks.AccountPrivateId
import com.pkb.backgroundtasks.AccountPublicId
import com.pkb.backgroundtasks.OrgPublicId
import com.pkb.backgroundtasks.entity.AccountTaskDetails
import com.pkb.backgroundtasks.entity.AccountTaskType
import com.pkb.backgroundtasks.util.RequestIdUtil
import com.pkb.crypto.dto.AccountKeysDTO
import com.pkb.kms.client.core.Kms
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Component
import java.lang.invoke.MethodHandles

@Component
class GrantOrgAccessTaskExecutor(
    private val kms: Kms,
    private val requestIdUtil: RequestIdUtil
) : AccountTaskExecutor<GrantOrgAccessTaskExecutor, GrantOrgAccessTaskDetails>(GrantOrgAccessTaskDetails::class.java) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    override fun performTask(
        accountPublicId: AccountPublicId,
        accountKeysDTO: AccountKeysDTO,
        accountPrivateId: AccountPrivateId,
        taskDetails: GrantOrgAccessTaskDetails
    ): Boolean =
        taskDetails.orgPublicIds.map { orgPublicId ->
            kms.grantOrgAccess(requestIdUtil.getRequestId(), accountPublicId, orgPublicId)
                .peekLeft { log.error("Failed to grant org {} access to account {}. Error: {}", orgPublicId, accountPublicId, it) }
                .isRight
        }
            .all { it }
}


data class GrantOrgAccessTaskDetails(val orgPublicIds: List<OrgPublicId>) : AccountTaskDetails<GrantOrgAccessTaskDetails>(AccountTaskType.GRANT_ORG_ACCESS) {
    override fun getSelf(): GrantOrgAccessTaskDetails = this
}
