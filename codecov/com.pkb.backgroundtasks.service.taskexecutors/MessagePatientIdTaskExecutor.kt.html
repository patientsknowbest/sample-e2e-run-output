package com.pkb.backgroundtasks.service.taskexecutors

import com.pkb.backgroundtasks.AccountPublicId
import com.pkb.backgroundtasks.PersonPrivateId
import com.pkb.backgroundtasks.entity.AccountTaskDetails
import com.pkb.backgroundtasks.entity.AccountTaskType
import com.pkb.backgroundtasks.repository.AppMenuDataRepository
import com.pkb.backgroundtasks.repository.PKBPersonRepository
import com.pkb.crypto.decryptionhelper.DecryptionHelperAppMenuData
import com.pkb.crypto.dto.AccountKeysDTO
import com.pkb.entities.app.AppMenuData
import com.pkb.entities.enums.MenuDataType
import org.springframework.stereotype.Component
import java.util.UUID

/**
 * Messages are supposed to have a related patient id. Some early messages don't have this populated.
 * For this we have to get the first and last messages of the conversation and look at various fields
 * like sender, recipient and participants that _might_ contain a patient we can use. Otherwise
 * we set a boolean flag to say we've looked and can't find anything
 */
@Component
class MessagePatientIdTaskExecutor(
    private val menuDataRepo: AppMenuDataRepository,
    private val decryptionHelper: DecryptionHelperAppMenuData,
    private val personRepo: PKBPersonRepository
) : AccountTaskExecutor<MessagePatientIdTaskExecutor, MessagePatientIdTaskDetails>(MessagePatientIdTaskDetails::class.java) {


    override fun performTask(accountPublicId: AccountPublicId, accountKeysDTO: AccountKeysDTO, accountPrivateId: Long, taskDetails: MessagePatientIdTaskDetails): Boolean {
        menuDataRepo.findByDataTypeInAndAccountIdOrderByDate01(listOf(MenuDataType.encounterEvent, MenuDataType.message), accountPrivateId)
            .map { it to decryptionHelper.decryptAndDeserializeAppMenuDataWithAccountKeys(it, accountKeysDTO) }
            .groupBy { it.conversationId ?: UUID.randomUUID() }
            .filter { (_, messages) -> messages.any { it.patientId == null && it.patientNotFound != true } }
            .entries.fold(mutableMapOf<PersonPrivateId, Boolean>(), { patientIdCache, (_, messages) -> findAndPopulatePatientId(messages, patientIdCache) })
        return true
    }

    private fun findAndPopulatePatientId(messages: List<Message>, patientIdCache: PatientIdCache): PatientIdCache {
        val patientId = messages
                .filter { it.messageType != "ENCOUNTER_DETAILS" }
                .let(this::getFirstAndLastOrNullIfEmpty)
                ?.let { (first, last) -> findReferencedPatientForConversation(first, last, patientIdCache) }

        messages
                .map { it.fixPatientIdAndReturnAppMenuData(patientId) }
                .let(menuDataRepo::saveAll)

        //The map is mutable so this is just to keep the fold method happy
        return patientIdCache
    }

    private fun findReferencedPatientForConversation(firstMessage: Message, lastMessage: Message, patientIdCache: PatientIdCache): Long? {
        return lastMessage.patientId ?: getPossiblePatientIdLists(firstMessage, lastMessage)
                .mapNotNull { findSinglePatientFromGroup(it, patientIdCache) }
                .firstOrNull()
    }


    private fun getFirstAndLastOrNullIfEmpty(messages: List<Message>): Pair<Message, Message>? {
        return if (messages.isEmpty()) {
            null
        } else {
            messages.first() to messages.last()
        }
    }

    /**
     * Returns a list of lists of ids that should be searched in order to see if we can find a patient
     * If there's more than one patient in any group of ids, its ambiguous, so we can't use any of them
     * and should try the next group.
     */
    private fun getPossiblePatientIdLists(firstMessage: Message, lastMessage: Message) =
            sequenceOf(listOfNotNull(firstMessage.senderId), listOfNotNull(firstMessage.recipientId))
                    .plus(linkedSetOf(firstMessage, lastMessage).map(this::getParticipants))


    private fun findSinglePatientFromGroup(personIdList: List<Long>, patientIdCache: PatientIdCache): Long? =
            filterToPatientIds(personIdList, patientIdCache)
                    .let { filteredList ->
                        if (filteredList.size > 1) listOf() else filteredList // If more than one patient in group, ignore
                    }
                    .firstOrNull()

    private fun filterToPatientIds(personIds: List<PersonPrivateId>, patientIdCache: PatientIdCache): List<PersonPrivateId> {
        personIds.subtract(patientIdCache.keys)
                .let { unknownPersonIds ->
                    val newPatients = if (unknownPersonIds.isNotEmpty()) personRepo.filterToPatientIds(unknownPersonIds) else unknownPersonIds
                    unknownPersonIds.associate { it to newPatients.contains(it) }
                }
                .also { newPersonsMap -> patientIdCache.putAll(newPersonsMap) }


        return personIds.filter { patientIdCache[it]!! }
    }


    private fun getParticipants(message: Message): List<Long> =
            message.participants
                    .flatMap { participantString -> participantString.split(',') }
                    .map(String::trim)
                    .filter(String::isNotBlank)
                    .map(String::toLong)
}

//Little bit of sugar
typealias Message = Pair<AppMenuData, MutableMap<String, Any>>
typealias PatientIdCache = MutableMap<PersonPrivateId, Boolean>

private fun Message.fixPatientIdAndReturnAppMenuData(possiblePatientId: Long?): AppMenuData {
    if (possiblePatientId != null) {
        patientId = possiblePatientId
    } else {
        patientNotFound = true
    }
    return this.first
}

private val Message.messageType: String?
    get() = first.string05
private val Message.conversationId: UUID?
    get() = first.uuid01
private val Message.recipientId: Long?
    get() = first.string01?.toLong()
private val Message.senderId: Long?
    get() = first.string02?.toLong()
private var Message.patientNotFound: Boolean?
    get() = first.boolean02
    set(value) {
        first.boolean02 = value
    }
private var Message.patientId: Long?
    get() = first.string03?.toLong()
    set(value) {
        first.string03 = value?.toString()
    }

@Suppress("UNCHECKED_CAST")
private val Message.participants: List<String>
    get() = second["participants"] as List<String>

class MessagePatientIdTaskDetails : AccountTaskDetails<MessagePatientIdTaskDetails>(AccountTaskType.MESSAGE_PATIENT_ID) {
    override fun getSelf(): MessagePatientIdTaskDetails = this
}
