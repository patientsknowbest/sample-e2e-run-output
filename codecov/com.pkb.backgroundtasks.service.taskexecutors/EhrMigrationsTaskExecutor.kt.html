package com.pkb.backgroundtasks.service.taskexecutors

import com.pkb.entities.enums.MenuDataType
import org.slf4j.Logger
import org.slf4j.LoggerFactory
import org.springframework.data.domain.PageRequest
import org.springframework.data.domain.Slice
import java.lang.invoke.MethodHandles
import java.util.UUID

abstract class EhrMigrationsTaskExecutor<SELF : EhrMigrationsTaskExecutor<SELF, MD>, MD : EhrMigrationsTaskDetail<MD>>(
    private val phrClient: PhrClient,
    private val menuDataRepo: AppMenuDataRepository,
    private val requestIdUtil: RequestIdUtil,
    private val properties: BackgroundTasksProperties,
    migrationDetailClass: Class<MD>
) :
    AccountTaskExecutor<SELF, MD>(migrationDetailClass) {

    private val log: Logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    protected abstract val dataTypes: List<MenuDataType>

    override fun performTask(accountPublicId: AccountPublicId, accountKeysDTO: AccountKeysDTO, accountPrivateId: Long, taskDetails: MD): Boolean {
        val versions = getEhrMigrationVersions()
        return dataTypes
            .flatMap { dataType ->
                val targetVersion = versions[dataType]!!
                generateSequence({ nextSlice(accountPrivateId, dataType, targetVersion) }) { nextSlice(accountPrivateId, dataType, targetVersion, it) }
                    .filter { it.hasContent() }
                    .map { EhrDataMigrationRequestDTO(accountPrivateId, dataType, it.content) }
                    .map {
                        phrClient.triggerEhrMigrations(requestIdUtil.getRequestId(), it).execute()
                            .apply {
                                if (!isSuccessful) {
                                    log.error("Error during EHR migration for account {}. Code: {} Error: {}", accountPublicId, code(), errorBody()?.string())
                                }
                            }
                            .isSuccessful
                    }
            }
            .all { it }
    }

    override fun getSkippableTasks(accountsToProcess: List<AccountTaskPending>, accountIdLookup: Map<UUID, Long>): Set<AccountTaskPending> {
        val accountsWithResults = menuDataRepo.findAccountsWithDataOfType(accountIdLookup.values, dataTypes)
        return accountsToProcess.filter { !accountsWithResults.contains(accountIdLookup[it.accountPublicId]) }.toSet()
    }

    private fun nextSlice(accountPrivateId: AccountPrivateId, dataType: MenuDataType, targetVersion: Int, prev: Slice<AccountPrivateId>? = null) =
        nextPageable(prev)?.let { menuDataRepo.getIdsNeedingMigrationForAccountAndDataType(accountPrivateId, dataType, targetVersion, it) }

    private fun nextPageable(prev: Slice<AccountPrivateId>?) =
        when {
            prev == null -> PageRequest.of(0, properties.ehrMigrationsMigrator.pageSize)
            prev.hasNext() -> prev.nextPageable()
            else -> null
        }

    private fun getEhrMigrationVersions() =
        phrClient.getEhrMigrationVersions().execute().run {
            if (!isSuccessful) throw RuntimeException("Error getting EHR Migration versions from PHR: ${errorBody()?.string()}")
            body()!!.versions.toMap()
        }
}
