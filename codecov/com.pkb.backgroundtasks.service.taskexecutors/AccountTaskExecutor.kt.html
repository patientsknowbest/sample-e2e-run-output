package com.pkb.backgroundtasks.service.taskexecutors

import com.pkb.backgroundtasks.AccountPrivateId
import com.pkb.backgroundtasks.AccountPublicId
import com.pkb.backgroundtasks.entity.AccountTaskDetails
import com.pkb.backgroundtasks.entity.AccountTaskPending
import com.pkb.crypto.dto.AccountKeysDTO
import org.springframework.stereotype.Component
import java.util.UUID

abstract class AccountTaskExecutor<Self : AccountTaskExecutor<Self, MD>, MD : AccountTaskDetails<MD>>(open val detailsClass: Class<MD>) {

    abstract fun performTask(accountPublicId: AccountPublicId, accountKeysDTO: AccountKeysDTO, accountPrivateId: AccountPrivateId, taskDetails: MD): Boolean

    /**
     * Subclasses should override this if they can identify accounts that don't need task execution without having crypto
     * keys from the KMS. In theory it should always be possible to calculate this up-front when populating
     * the pending tasks table using Liquibase, but in practice doing so may involve some impractically slow
     * queries, in which case it will be more practical to filter these out batch by batch here.
     */
    open fun getSkippableTasks(accountsToProcess: List<AccountTaskPending>, accountIdLookup: Map<UUID, Long>): Set<AccountTaskPending> = emptySet()

}


@Component
class TaskExecutorRegistry(taskExecutors: List<AccountTaskExecutor<*, *>>) {

    private val taskExecutorsByTaskDetailsClass = taskExecutors.associateBy { it.detailsClass }

    fun <MD : AccountTaskDetails<MD>, M : AccountTaskExecutor<M, MD>> taskExecutorFor(taskDetails: MD): M? {
        @Suppress("UNCHECKED_CAST")
        return taskExecutorsByTaskDetailsClass[taskDetails.javaClass] as M?
    }
}
