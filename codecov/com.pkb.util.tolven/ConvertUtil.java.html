package com.pkb.util.tolven;

import com.pkb.VersionDetails;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.emis.es.entity.EmisEsPersonGuid;
import com.pkb.entities.core.CoreSecurityQuestion;
import com.pkb.model.PKBPersonDTO;
import com.pkb.model.PKBSecurityQuestion;
import com.pkb.user.entity.PKBPerson;
import com.pkb.user.entity.PKBPersonHelper;
import com.pkb.user.entity.PKBPersonProperty;

import java.time.Instant;
import java.util.ArrayList;
import java.util.List;

/**
 * A utility class to convert the Tolven entities into PKB DTOs and vice versa
 */
public final class ConvertUtil {

    private final DateTimeService dateTimeService;
    private final PKBPersonHelper pkbPersonHelper;

    public ConvertUtil(DateTimeService dateTimeService, PKBPersonHelper pkbPersonHelper) {
        this.dateTimeService = dateTimeService;
        this.pkbPersonHelper = pkbPersonHelper;
    }

    /**
     * Convert the Tolven security questions into PKB security questions
     *
     * @param list
     * @return
     */
    public List<PKBSecurityQuestion> getPKbSecurityQuestions(
            List<CoreSecurityQuestion> list) {
        List<PKBSecurityQuestion> pkbQuestionList = new ArrayList<>();
        PKBSecurityQuestion pkbQuestion;
        for (CoreSecurityQuestion question : list) {
            pkbQuestion = new PKBSecurityQuestion();
            pkbQuestion.setId(question.getId());
            pkbQuestion.setQuestion(question.getQuestion());
            pkbQuestionList.add(pkbQuestion);
        }
        return pkbQuestionList;
    }

    /**
     * Convert PKBPerson DTO to PKBPerson entity
     *
     * @param user
     * @return
     */
    public PKBPerson getPKBPerson(VersionDetails versionDetails, PKBPersonDTO user) {
        PKBPerson person = new PKBPerson();
        person.setId(user.getId());
        person.setTitle(user.getTitle());
        person.setJobTitle(user.getJobTitle());
        person.setFirstName(user.getFirstName());
        person.setMiddleName(user.getMiddleName());
        person.setLastName(user.getLastName());
        person.setAddress1(user.getAddress1());
        person.setAddress2(user.getAddress2());
        person.setCity(user.getCity());
        person.setCountry(user.getCountry());
        person.setDateOfBirthString(user.getDateOfBirthString());
        pkbPersonHelper.validateDeathTimestamp(user.getDeathTimestamp() == null ? null : user.getDeathTimestamp().toInstant());
        person.setDeathTimestamp(user.getDeathTimestamp());
        person.setGender(user.getGender());
        if (user.getEmailId() != null) {
            person.setPrimaryEmail(user.getEmailId(), versionDetails);
        }
        person.setPassword(user.getPassword());
        person.setPostalCode(user.getPostalCode());
        person.setState(user.getState());
        person.setOrganizationName(user.getOrganizationName());
        person.setOrganizationUnit(user.getOrganizationUnit());
        person.setProperties(user.getProperties());
        person.setStatus(user.getStatus(), dateTimeService.now());
        person.setUserType(user.getUserType());
        person.setPhone(user.getPhone());
        person.setSkypeId(user.getSkypeId());
        person.setLanguageCode(user.getLanguageCode());
        person.setTimeZoneId(user.getTimeZoneId());
        person.setNotificationOption(user.getNotificationOption());
        if (person.getProperties() != null) {
            for (PKBPersonProperty property : person.getProperties()) {
                property.setPerson(person);
            }
        }
        if (user.getEmisEsPersonGuid() != null) {
            EmisEsPersonGuid guid = new EmisEsPersonGuid();
            guid.setValue(user.getEmisEsPersonGuid());
            guid.setPerson(person);
            person.setEmisEsPersonGuid(guid);
        }

        return person;
    }

    public PKBPerson getPKBPersonWithNationalIds(EHRRequestContext context, PKBPersonDTO user) {
        Instant now = dateTimeService.now();
        PKBPerson person = getPKBPerson(VersionDetails.of(context, now), user);
        user.getNationalIds().forEach(nationalId -> person.addOrUpdateNationalId(context, nationalId, now));
        return person;
    }

    public PKBPersonDTO getPKBPersonFromPlugin(PKBPerson user) {
        PKBPersonDTO person = new PKBPersonDTO();
        person.setId(user.getId());
        person.setTitle(user.getTitle());
        person.setJobTitle(user.getJobTitle());
        person.setUserName(user.getUserName());
        person.setFirstName(user.getFirstName());
        person.setLastName(user.getLastName());
        person.setAddress1(user.getAddress1());
        person.setAddress2(user.getAddress2());
        person.setCity(user.getCity());
        person.setCountry(user.getCountry());
        person.setDateOfBirthString(user.getDateOfBirthString());
        person.setGender(user.getGender());
        person.setEmailId(user.getEmail());
        person.setPassword(user.getPassword());
        person.setPostalCode(user.getPostalCode());
        person.setState(user.getState());
        person.setOrganizationName(user.getOrganizationName());
        person.setOrganizationUnit(user.getOrganizationUnit());
        person.setPhone(user.getPhone());
        person.setUserType(user.getUserType());
        person.setStatus(user.getStatus());
        person.setLanguageCode(user.getLanguageCode());
        person.setTimeZoneId(user.getTimeZoneId());
        person.setHumanUUID(user.getHumanUUID());
        return person;
    }
}
