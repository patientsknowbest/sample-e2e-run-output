import javax.naming.InitialContext;

public class TransactionManager implements ITransactionManager {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    public static final int DEFAULT_TRANSACTION_TIMEOUT = 0;
    //this field is modified in arq tests to disable explicit tx management
    protected static String USER_TRANSACTION_JNDI = "java:jboss/UserTransaction";

    protected final PhrConfig config;
    protected final TolvenBeanFactory beanFactory;
    protected DateTimeService dateTimeService;
    protected final UUIDProvider uuidProvider;

    public TransactionManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider) {
        this.config = config;
        this.beanFactory = beanFactory;
        this.dateTimeService = dateTimeService;
        this.uuidProvider = uuidProvider;
    }

    //Integration tests fiddle with it....
    @VisibleForTesting
    public void setDateTimeService(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    public TolvenBeanFactory getBeanFactory() {
        return beanFactory;
    }

    protected PhrConfig getConfig() {
        return config;
    }

    private class DefaultTransactionOpened implements TransactionOpened {

        @Override
        public void close() {
            rollbackIfNotCommitted();
        }
    }

    /**
     * @deprecated use {@link TransactionManager#transactional(Runnable, Runnable...)}
     * or {@link TransactionManager#transactional(Callable, Runnable...)} instead.
     */
    @Deprecated
    @Override
    public TransactionScope beginTransaction() {
        return beginTransaction(DEFAULT_TRANSACTION_TIMEOUT);
    }

    @Override
    public ITransactionManager.TransactionScope beginTransaction(int timeoutSeconds) {
        try {
            UserTransaction userTransaction = lookupUserTransaction();
            if (Status.STATUS_ACTIVE == userTransaction.getStatus()) {
                traceThread("Trying to call begin on userTransaction that is already active");
                return new NoTransactionOpened(); // somewhere a kitten dies
            }
            if (timeoutSeconds > 0) {
                LOGGER.info("Updating transaction timeout to {}s", timeoutSeconds);
                userTransaction.setTransactionTimeout(timeoutSeconds);
            }
            userTransaction.begin();
            LOGGER.debug("Began transaction");
            return new DefaultTransactionOpened();
        } catch (NotSupportedException | SystemException e) {
            throw new RuntimeException(e);
        }
    }

    /**
     * @deprecated use {@link TransactionManager#transactional(Runnable, Runnable...)}
     * or {@link TransactionManager#transactional(Callable, Runnable...)} instead.
     */
    @Deprecated
    @Override
    public void commitTransaction() {
        try {
            UserTransaction userTransaction = lookupUserTransaction();
            traceThread(String.format("Trying to call commit on userTransaction[status=%s]", userTransaction.getStatus()));
            userTransaction.commit();
            LOGGER.debug("Committed transaction");
        } catch (SystemException | HeuristicMixedException | HeuristicRollbackException | RollbackException e) {
            // temporary awful hack 20161121 mvarga (not that it is worse than how it used to be :/)
            LOGGER.error("Could not roll back transaction", e);
            // throw new RuntimeException(e);
        }
    }

    /**
     * Call from finally block -- if commitTransaction() didn't get called,
     * that means something broke and we should rollback.
     * Catches and logs any exception to avoid needing any external error handling.
     *
     * @deprecated use {@link TransactionManager#transactional(Runnable, Runnable...)}
     * or {@link TransactionManager#transactional(Callable, Runnable...)} instead.
     */
    @Deprecated
    @Override
    public void rollbackIfNotCommitted() {
        try {
            UserTransaction userTransaction = lookupUserTransaction();
            if (userTransaction != null) {
                int status = userTransaction.getStatus();
                if (Status.STATUS_ACTIVE == status ||
                        Status.STATUS_MARKED_ROLLBACK == status ||
                        Status.STATUS_ROLLEDBACK == status) {
                    userTransaction.rollback();
                    traceThread("rolled-back transaction[status=" + status + "]");
                }
            }
        } catch (Exception e) {
            LOGGER.error("error checking transaction status, or rolling back", e);
        }
    }

    @Override
    public void ensureValidTransactionStatus() {
        //throws NotSupportedException, SystemException, NamingException {
        try {
            UserTransaction userTransaction = lookupUserTransaction();

            LOGGER.info("TEST TRANSACTION: status: {}", userTransaction.getStatus());
            if (userTransaction.getStatus() == Status.STATUS_MARKED_ROLLBACK) {
                userTransaction.rollback();
                LOGGER.info("TEST TRANSACTION: forced rollback; new status: {}", userTransaction.getStatus());
            } else if (userTransaction.getStatus() == Status.STATUS_ROLLEDBACK) {
                LOGGER.info("TRANSACTION: is rolled-back; attempting commit");
                try {
                    userTransaction.commit();
                    LOGGER.info("...committed rolled-back transaction: {}", userTransaction.getStatus());
                } catch (Exception e) {
                    LOGGER.info("TRANSACTION: commit of rolled-back trans FAILED: {}", userTransaction.getStatus(), e);
                }
            }
        } catch (Exception e) { // wrap, to simplify interface
            throw new PKBException(e);
        }
    }

    public UserTransaction lookupUserTransaction() {
        try {
            InitialContext ctx = new InitialContext();
            try {
                return ((UserTransaction) (ctx).lookup(USER_TRANSACTION_JNDI));
            } finally {
                ctx.close();
            }
        } catch (Exception e) { // wrap, to simplify interface
            throw new RuntimeException(e);
        }
    }

    private void traceThread(String msg) {
        Thread current = Thread.currentThread();
        StackTraceElement[] traceArr = current.getStackTrace();
        LOGGER.debug("TX ({}): {} - from: {}", current.getName(), msg, traceArr[3].toString());
    }

    /**
     * Finds and rolls back any 'dangling' unfinished transactions (either because a a commit/rollback command is missing
     * or they timed out and were aborted by transaction reaper ) .
     *
     * @param transactionSite
     */
    @Override
    public void cleanupTransaction(String transactionSite) {

        UserTransaction userTransaction = null;

        try {

            userTransaction = lookupUserTransaction();

            if (userTransaction != null &&
                    (userTransaction.getStatus() == Status.STATUS_ROLLEDBACK ||
                            userTransaction.getStatus() == Status.STATUS_ACTIVE)
                    &&
                    userTransaction.getClass().getCanonicalName().equals("org.jboss.tm.usertx.client.ClientUserTransaction")) {

                Object context = userTransaction.getClass().getDeclaredMethod("getTransactionPropagationContext").invoke(userTransaction);
                if (context != null) {
                    LOGGER.warn("Rolling back potentially unfinished transaction for {}", transactionSite);
                    userTransaction.rollback();
                }
            }
        } catch (IllegalStateException ise) {
            forceClearOfTransactionContext(userTransaction, ise);
        } catch (Exception e) {
            //ignore
            LOGGER.error("Error during cleanup", e);
        }
    }

    private void forceClearOfTransactionContext(UserTransaction userTransaction, IllegalStateException ise) {

        try {
            Object threadInfo = invokePrivate(userTransaction, "getThreadInfo");
            invokePrivate(threadInfo, "pop");
        } catch (InvocationTargetException | NoSuchMethodException | IllegalAccessException e) {
            LOGGER.error("Error during cleanup", e);
        }
    }

    private Object invokePrivate(Object o, String methodName)
            throws NoSuchMethodException, InvocationTargetException, IllegalAccessException {
        Method m = o.getClass().getDeclaredMethod(methodName);
        if (!m.isAccessible()) {
            m.setAccessible(true);
        }
        return m.invoke(o);
    }

    @Override
    public void transactional(Runnable runnable, int timeout, Runnable... errorHandlers) {
        transactional(() -> {
            runnable.run();
            //noinspection ReturnOfNull
            return null;
        }, timeout, errorHandlers);
    }

    @Override
    public <V> V transactional(Callable<V> callable, Runnable... errorHandlers) {
        return transactional(callable, DEFAULT_TRANSACTION_TIMEOUT, errorHandlers);
    }

    @Override
    public void transactional(Runnable runnable, Runnable... errorHandlers) {
        transactional(runnable, DEFAULT_TRANSACTION_TIMEOUT, errorHandlers);
    }

    @Override
    public <V> V transactional(Callable<V> callable, int timeout, Runnable... errorHandlers) {
        boolean newTransactionStarted = false;
        try {
            TransactionScope transactionScope = beginTransaction(timeout);
            newTransactionStarted = transactionScope instanceof TransactionOpened;
            V v = callable.call();
            if(newTransactionStarted){
                commitTransaction();
            }
            return v;
        } catch (RuntimeException exception) {
            Arrays.stream(errorHandlers).forEach(Runnable::run);
            throw exception;
        } catch (Exception exception) {
            Arrays.stream(errorHandlers).forEach(Runnable::run);
            throw new RuntimeException(exception);
        } finally {
            if(newTransactionStarted){
                rollbackIfNotCommitted();
            }
        }
    }
}
