package com.pkb.util.tolven;

import com.pkb.core.DatapointSymmetricKeyRemote;
import com.pkb.data.EHRRemote;
import com.pkb.encounter.HealthRecordAccessRemote;
import com.pkb.reminders.ReminderRemote;
import com.pkb.tolven.PKBAccountDAORemote;
import com.pkb.user.PKBPersonRemote;
import org.jboss.ejb.client.EJBClient;
import org.jboss.ejb.client.EJBLocator;
import org.jboss.ejb.client.StatelessEJBLocator;
import org.springframework.beans.factory.InitializingBean;

/**
 * This factory class takes care of providing the beans to Manager classes
 *
 * @author pravinam
 */
public class TolvenBeanFactory implements InitializingBean {
    private PKBPersonRemote pkbPersonBean;

    private ReminderRemote reminderBean;

    private static TolvenBeanFactory beanFactory;

    private PKBAccountDAORemote accountDaoBean;

    private DatapointSymmetricKeyRemote datapointSymmetricKeyBean;

    private EHRRemote ehrRemote;

    private HealthRecordAccessRemote healthRecordAccessRemote;

    private TolvenBeanFactory() {
    }

    /**
     * DO NOT USE THIS, USE DEPENDENCY INJECTION!
     *
     * @return
     * @deprecated USE DI!
     */
    @Deprecated
    public static synchronized TolvenBeanFactory getFactory() {
        if (beanFactory == null) {
            beanFactory = new TolvenBeanFactory();
        }
        return beanFactory;
    }

    public PKBPersonRemote getPKBPersonBean() {
        return pkbPersonBean;
    }

    @SuppressWarnings("unchecked")
    public <RI> RI lookup(String beanName, Class<RI> interfaceClass) {

        return lookup(new StatelessEJBLocator<>(interfaceClass, "phr", "tolvenEJB", beanName, ""));
    }

    @SuppressWarnings("unchecked")
    public <RI> RI lookupStateful(String beanName, Class<RI> interfaceClass) {
        try {
            return lookup(EJBClient.createSession(interfaceClass, "phr", "tolvenEJB", beanName, ""));
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }

    @SuppressWarnings("unchecked")
    public <RI> RI lookup(EJBLocator<RI> locator) {
        return EJBClient.createProxy(locator);
    }

    public ReminderRemote getReminderBean() {
        return reminderBean;
    }

    public PKBAccountDAORemote getAccountDaoBean() {
        return accountDaoBean;
    }

    public DatapointSymmetricKeyRemote getDatapointSymmetricKeyBean() {
        return datapointSymmetricKeyBean;
    }

    public EHRRemote getEhrRemote() {
        return ehrRemote;
    }

    public HealthRecordAccessRemote getHealthRecordAccessRemote() {
        return healthRecordAccessRemote;
    }


    @Override
    public void afterPropertiesSet() throws Exception {
        pkbPersonBean = lookup("PKBPersonBean", PKBPersonRemote.class);
        reminderBean = lookup("ReminderBean", ReminderRemote.class);
        accountDaoBean = lookup("AccountDAOBean", PKBAccountDAORemote.class);
        datapointSymmetricKeyBean = lookup("DatapointSymmetricKeyBean", DatapointSymmetricKeyRemote.class);
        ehrRemote = lookup("EHRBean", EHRRemote.class);
        healthRecordAccessRemote = lookup("HealthRecordAccess", HealthRecordAccessRemote.class);
    }
}
