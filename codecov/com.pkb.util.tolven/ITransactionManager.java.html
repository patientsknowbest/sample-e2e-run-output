package com.pkb.util.tolven;

import java.util.concurrent.Callable;

public interface ITransactionManager {

    void transactional(Runnable runnable, int timeout, Runnable... errorHandlers);

    void transactional(Runnable runnable, Runnable... errorHandlers);

    <V> V transactional(Callable<V> callable, int timeout, Runnable... errorHandlers);

    <V> V transactional(Callable<V> callable, Runnable... errorHandlers);

    interface TransactionScope extends AutoCloseable {
        @Override
        void close();
    }

    interface TransactionOpened extends TransactionScope {
    }

    class NoTransactionOpened implements TransactionScope {
        @Override
        public void close() {
            // no-op
        }
    }

    ITransactionManager.TransactionScope beginTransaction();

    ITransactionManager.TransactionScope beginTransaction(int timeoutSeconds);

    void commitTransaction();

    /**
     * Call from finally block -- if commitTransaction() didn't get called,
     * that means something broke and we should rollback.
     * Catches and logs any exception to avoid needing any external error handling.
     */
    void rollbackIfNotCommitted();

    void ensureValidTransactionStatus();

    void cleanupTransaction(String transactionSite);
}
