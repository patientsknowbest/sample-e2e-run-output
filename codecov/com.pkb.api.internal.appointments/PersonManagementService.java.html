package com.pkb.api.internal.appointments;

import com.pkb.api.internal.model.requests.PersonSearcRequestDto;
import com.pkb.api.internal.validator.PersonSearchByEmailRequestValidator;
import com.pkb.api.util.PKBCommonApiParams;
import com.pkb.datamodel.Email;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.user.entity.PersonContact;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.Collection;
import java.util.List;
import java.util.Map;
import java.util.Set;
import java.util.stream.Collectors;

import static com.pkb.datamodel.ImmutableDisclosingEmail.disclosingEmail;

@Service
public class PersonManagementService {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    private UserManager userManager;

    @Autowired
    private PersonSearchByEmailRequestValidator personSearchByEmailRequestValidator;

    public Map<Email, List<PKBPerson>> getPersonsByEmails(PersonSearcRequestDto request, PKBCommonApiParams commonParams) {
        personSearchByEmailRequestValidator.validate(request);
        var personsByEmails = userManager.getPKBPersonsByConfirmedOrPrimaryEmails(request.getValues().stream().map(v -> (Email)disclosingEmail(v)).collect(Collectors.toList()),
                PersonContact.Lazy.PERSON_W_CONTACTS);
        LOGGER.info("Non sesitive person information was retrieved by person id: {} for person ids: {}.", commonParams.getPerson().getId(),
                getPersonIds(personsByEmails.values()));
        return personsByEmails;
    }

    private Set<Long> getPersonIds(Collection<List<PKBPerson>> persons) {
        return persons.stream().flatMap(List::stream).map(PKBPerson::getId).collect(Collectors.toSet());
    }
}
