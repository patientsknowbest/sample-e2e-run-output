package com.pkb.aggregator.service.aggregation.reload

import com.pkb.aggregator.R4Bundle
import com.pkb.aggregator.extension.addTransactionEntry
import com.pkb.aggregator.service.aggregation.r4.OrganizationAggregator
import com.pkb.aggregator.service.aggregation.r4.PractitionerAggregator
import com.pkb.aggregator.service.aggregation.r4.PractitionerRoleAggregator
import com.pkb.aggregator.service.reference.ReferenceExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.hl7.fhir.r4.model.Bundle
import org.springframework.stereotype.Service

/**
 * Handles full reload/re-aggregation of a single organization and associated practitioners
 */
@Service
class OrganizationReloadAggregator(
    private val organizationAggregator: OrganizationAggregator,
    private val practitionerAggregator: PractitionerAggregator,
    private val practitionerRoleAggregator: PractitionerRoleAggregator,
    private val referenceExtractor: ReferenceExtractor,
) {

    data class AllOrganizationResources(
        val organizations: List<IDomainResource>,
        val practitioners: List<IDomainResource>,
        val practitionerRoles: List<IDomainResource>,
    ) {
        fun allResources() = organizations
            .plus(practitioners)
            .plus(practitionerRoles)
    }

    fun aggregate(currentDownstreamId: String?, resources: AllOrganizationResources): R4Bundle {

        val bundle = R4Bundle().also {
            it.type = Bundle.BundleType.TRANSACTION
        }

        val references = referenceExtractor.extract(resources.allResources())

        organizationAggregator.merge(resources.organizations, currentDownstreamId, references).also {
            bundle.addTransactionEntry(it)
        }

        // Practitioners
        practitionerAggregator.mergeAll(resources.practitioners, references).onEach {
            bundle.addTransactionEntry(it)
        }

        // Roles
        practitionerRoleAggregator.mergeAll(resources.practitionerRoles, references).onEach {
            bundle.addTransactionEntry(it)
        }

        return bundle
    }
}
