package com.pkb.aggregator.service.aggregation.reload

import com.pkb.aggregator.R4Bundle
import com.pkb.aggregator.R4DomainResource
import com.pkb.aggregator.extension.addTransactionEntry
import com.pkb.aggregator.service.aggregation.r4.MedicationAggregator
import com.pkb.aggregator.service.reference.ReferenceExtractor
import org.hl7.fhir.r4.model.Bundle
import org.springframework.stereotype.Service

/**
 * Handles full reload/re-aggregation of a single medication
 */
@Service
class MedicationReloadAggregator(
    private val medicationAggregator: MedicationAggregator,
    private val referenceExtractor: ReferenceExtractor,
) {

    fun aggregate(currentDownstreamId: String?, baseResources: List<R4DomainResource>): R4Bundle {

        val bundle = R4Bundle().also {
            it.type = Bundle.BundleType.TRANSACTION
        }

        val references = referenceExtractor.extract(baseResources)

        medicationAggregator.merge(baseResources, currentDownstreamId, references).also {
            bundle.addTransactionEntry(it)
        }

        return bundle
    }
}
