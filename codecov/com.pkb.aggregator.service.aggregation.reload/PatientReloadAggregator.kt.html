package com.pkb.aggregator.service.aggregation.reload

import com.pkb.aggregator.R4Bundle
import com.pkb.aggregator.R4IdType
import com.pkb.aggregator.extension.addTransactionEntry
import com.pkb.aggregator.service.FHIR_TYPE_PERSON
import com.pkb.aggregator.service.aggregation.r4.PatientAggregator
import com.pkb.aggregator.service.aggregation.r4.PersonAggregator
import com.pkb.aggregator.service.copier.r4.AllergyIntoleranceDownstreamCopier
import com.pkb.aggregator.service.copier.r4.AppointmentDownstreamCopier
import com.pkb.aggregator.service.copier.r4.ConditionDownstreamCopier
import com.pkb.aggregator.service.copier.r4.EncounterDownstreamCopier
import com.pkb.aggregator.service.copier.r4.MedicationStatementDownstreamCopier
import com.pkb.aggregator.service.copier.r4.ObservationDownstreamCopier
import com.pkb.aggregator.service.reference.ReferenceExtractor
import org.hl7.fhir.instance.model.api.IDomainResource
import org.hl7.fhir.r4.model.Bundle
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.lang.invoke.MethodHandles

/**
 * Handles full reload/re-aggregation of a single patient account
 */
@Service
class PatientReloadAggregator(
    private val patientAggregator: PatientAggregator,
    private val personAggregator: PersonAggregator,
    private val encounterDownstreamCopier: EncounterDownstreamCopier,
    private val observationDownstreamCopier: ObservationDownstreamCopier,
    private val medicationStatementDownstreamCopier: MedicationStatementDownstreamCopier,
    private val appointmentDownstreamCopier: AppointmentDownstreamCopier,
    private val conditionDownstreamCopier: ConditionDownstreamCopier,
    private val allergyIntoleranceDownstreamCopier: AllergyIntoleranceDownstreamCopier,
    private val referenceExtractor: ReferenceExtractor,
) {

    private val log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass())

    data class AllPatientResources(
        val patient: List<IDomainResource>,
        val person: List<IDomainResource>,
        val encounters: List<IDomainResource>,
        val observations: List<IDomainResource>,
        val medicationStatements: List<IDomainResource>,
        val appointments: List<IDomainResource>,
        val conditions: List<IDomainResource>,
        val allergies: List<IDomainResource>,
    ) {
        fun allResources() = patient
            .asSequence()
            .plus(person)
            .plus(encounters)
            .plus(appointments)
            .plus(observations)
            .plus(medicationStatements)
            .plus(conditions)
            .plus(allergies)
            .toList()
    }

    fun aggregate(currentDownstreamId: String?, resources: AllPatientResources): R4Bundle {

        if (resources.patient.isEmpty()) {
            throw RuntimeException("No patient to aggregate")
        }

        val bundle = R4Bundle().also {
            it.type = Bundle.BundleType.TRANSACTION
        }

        // Get every reference in the upstream resources, cross-referenced with downstream equivalents
        val references = referenceExtractor.extract(resources.allResources())

        val patient = patientAggregator.merge(resources.patient, currentDownstreamId, references).also {
            bundle.addTransactionEntry(it)
        }

        // Only process person resources if we have any
        if (resources.person.isNotEmpty()) {
            // Note the relationship between the patient/person resource id - this is valid while we only have Person
            // resources from the EHR migrator, but may not be sufficient when we have customers sending us Person
            // resources that don't make the same assumptions
            val personResourceId = R4IdType(FHIR_TYPE_PERSON, patient.idElement.idPart)
            personAggregator.merge(resources.person, personResourceId.toString(), references).also {
                bundle.addTransactionEntry(it)
            }
        }

        // Encounters
        encounterDownstreamCopier.copy(resources.encounters, references).onEach {
            bundle.addTransactionEntry(it.value)
        }

        // Observations
        observationDownstreamCopier.copy(resources.observations, references).onEach {
            bundle.addTransactionEntry(it.value)
        }

        // MedicationStatements
        medicationStatementDownstreamCopier.copy(resources.medicationStatements, references).onEach {
            bundle.addTransactionEntry(it.value)
        }

        // Appointments
        appointmentDownstreamCopier.copy(resources.appointments, references).onEach {
            bundle.addTransactionEntry(it.value)
        }

        // Conditions
        conditionDownstreamCopier.copy(resources.conditions, references).onEach {
            bundle.addTransactionEntry(it.value)
        }

        // Allergies
        allergyIntoleranceDownstreamCopier.copy(resources.allergies, references).onEach {
            bundle.addTransactionEntry(it.value)
        }

        // We explicitly check the log level here so as not to summarise the bundle if it isn't needed
        if (log.isDebugEnabled) {
            val stats = bundle.entry.groupBy { it.resource.resourceType }.mapValues { it.value.size }
            log.debug("Preparing transaction bundle with resources: {}", stats)
        }

        return bundle
    }
}
