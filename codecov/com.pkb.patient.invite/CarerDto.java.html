package com.pkb.patient.invite;

import com.pkb.datamodel.Email;
import io.vavr.control.Option;

public final class CarerDto {
    private final NameDto name;
    private final Email email;

    public CarerDto(String firstName, String lastName, Email email) {
        this.name = NameDto.name(firstName, lastName);
        this.email = email;
    }

    public String getFirstName() {
        return name.getFirstName();
    }

    public String getLastName() {
        return name.getLastName();
    }

    public Email getEmail() {
        return this.email;
    }

    public Option<Email> findEmail() { return Option.of(this.email); }
}
