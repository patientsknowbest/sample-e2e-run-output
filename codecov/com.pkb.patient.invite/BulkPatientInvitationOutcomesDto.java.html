package com.pkb.patient.invite;

import com.google.common.collect.ImmutableSet;
import com.pkb.datamodel.Email;

import java.util.List;
import java.util.Set;

public final class BulkPatientInvitationOutcomesDto {
    private List<BulkPatientCreationOutcomeDto> outcomes;

    public BulkPatientInvitationOutcomesDto(List<BulkPatientCreationOutcomeDto> outcomes) {
        this.outcomes = outcomes;
    }

    public List<BulkPatientCreationOutcomeDto> getOutcomes() {
        return outcomes;
    }

    public void setOutcomes(List<BulkPatientCreationOutcomeDto> outcomes) {
        this.outcomes = outcomes;
    }

    public static class BulkPatientCreationOutcomeDto {
        private final Set<String> DOWNLOADABLE_STATUSES = ImmutableSet.of("SUCCEEDED", "PARTIALLY_SUCCEEDED", "FAILED");

        private final String initiationDataTime;
        private final String initiatorsName;
        private final String link;
        private final Email userName;
        private final String status;

        public BulkPatientCreationOutcomeDto(String initiationDataTime, String initiatorsName, String link, Email userName, String status) {
            this.initiationDataTime = initiationDataTime;
            this.initiatorsName = initiatorsName;
            this.link = link;
            this.userName = userName;
            this.status = status;
        }

        public String getInitiationDataTime() {
            return initiationDataTime;
        }

        public String getInitiatorsName() {
            return initiatorsName;
        }

        public String getLink() {
            return link;
        }

        public Email getUserName() {
            return userName;
        }

        public String getStatus() {
            return status;
        }

        public boolean getIsOutcomeDownloadable() {
            return DOWNLOADABLE_STATUSES.contains(status);
        }
    }
}
