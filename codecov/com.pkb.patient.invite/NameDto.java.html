package com.pkb.patient.invite;

import java.util.Optional;

public class NameDto {
    private final String title;
    private final String firstName;
    private final String lastName;

    public static NameDto name(String firstName, String lastName) {
        return new NameDto(null, firstName, lastName);
    }

    public static NameDto nameWithTitle(String title, String firstName, String lastName) {
        return new NameDto(title, firstName, lastName);
    }

    private NameDto(String title, String firstName, String lastName) {
        this.title = title;
        this.firstName = firstName;
        this.lastName = lastName;
    }

    public Optional<String> findTitle() {
        return Optional.ofNullable(title);
    }

    public String getFirstName() {
        return firstName;
    }

    public String getLastName() {
        return lastName;
    }
}
