package com.pkb.patient.invite;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.service.patientconsent.PatientConsentManager;
import com.pkb.service.team.TeamUserManager;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;

public class RequestBoundNewPatientRegisterFactory {
    private final TeamUserManager teamUserManager;
    private final PatientConsentManager patientConsentManager;

    public RequestBoundNewPatientRegisterFactory(TeamUserManager teamUserManager, PatientConsentManager patientConsentManager) {
        this.teamUserManager = teamUserManager;
        this.patientConsentManager = patientConsentManager;
    }

    public RequestBoundNewPatientRegister create(@NotNull PKBPerson coordinator,
            @NotNull LoggedInEHRRequestContext requestConsent) {
        return new RequestBoundNewPatientRegister(teamUserManager,patientConsentManager)
                .withCoordinator(coordinator)
                .withRequestContext(requestConsent);
    }
}
