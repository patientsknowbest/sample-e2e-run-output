package com.pkb.patient.invite;

import com.pkb.entities.enums.ValidNationalId;

import java.util.Optional;

import static com.google.common.base.Preconditions.checkArgument;

public final class InviteeIdsDto {
    private final ValidNationalId nationalId;
    private final String teamLevelId;
    private final String orgLevelId;

    public InviteeIdsDto(ValidNationalId nationalId, String teamLevelId, String orgLevelId) {
        checkArgument(nationalId != null || teamLevelId != null || orgLevelId != null, "At least one ID must not be present");
        this.nationalId = nationalId;
        this.teamLevelId = teamLevelId;
        this.orgLevelId = orgLevelId;
    }

    public Optional<ValidNationalId> findNationalId() {
        return Optional.ofNullable(nationalId);
    }

    public Optional<String> findTeamLevelId() {
        return Optional.ofNullable(teamLevelId);
    }

    public Optional<String> findOrgLevelId() {
        return Optional.ofNullable(orgLevelId);
    }

}
