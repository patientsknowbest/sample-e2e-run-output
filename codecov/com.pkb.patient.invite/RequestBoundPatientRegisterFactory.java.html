package com.pkb.patient.invite;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.institute.entity.Team;
import com.pkb.patient.InvitationNationalIdValidator;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;

public class RequestBoundPatientRegisterFactory {
    private final UserManager userManager;
    private final RequestBoundExistingPatientRegisterFactory existingPatientRegisterFactory;
    private final RequestBoundNewPatientRegisterFactory newPatientRegisterFactory;
    private final RequestBoundCarerRegisterFactory carerRegisterFactory;

    public RequestBoundPatientRegisterFactory(
            UserManager userManager,
            RequestBoundExistingPatientRegisterFactory existingPatientRegisterFactory,
            RequestBoundNewPatientRegisterFactory newPatientRegisterFactory,
            RequestBoundCarerRegisterFactory carerRegisterFactory) {
        this.userManager = userManager;
        this.existingPatientRegisterFactory = existingPatientRegisterFactory;
        this.newPatientRegisterFactory = newPatientRegisterFactory;
        this.carerRegisterFactory = carerRegisterFactory;
    }

    public PatientRegister create(
            @NotNull PKBPerson inviter,
            @NotNull Team invitingTeam,
            @NotNull NationalIdType nationalIdType,
            @NotNull InvitationNationalIdValidator invitationNationalIdValidator,
            @NotNull LoggedInEHRRequestContext requestContext) {
        return new RequestBoundPatientRegister(
                userManager,
                existingPatientRegisterFactory.create(),
                newPatientRegisterFactory.create(inviter, requestContext),
                carerRegisterFactory.create(),
                invitationNationalIdValidator)
                        .withInviter(inviter)
                        .withInvitingTeam(invitingTeam)
                        .withNationalIdType(nationalIdType)
                        .withRequestContext(requestContext);
    }
}
