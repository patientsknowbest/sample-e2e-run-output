package com.pkb.patient.invite;

import com.google.common.collect.ImmutableList;
import com.pkb.common.base.Error;
import com.pkb.datamodel.consent.PatientConsentUpdateTemplate;
import com.pkb.user.entity.PKBPerson;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import java.util.Collection;
import java.util.List;

import static com.pkb.patient.invite.PatientRegister.FailureReason.EXISTING_USER_IS_NOT_A_PATIENT;
import static com.pkb.patient.invite.PatientRegister.FailureReason.FAILED_TO_INVITE_EXISTING_USER_WITHOUT_EMAIL;
import static com.pkb.patient.invite.PatientRegister.FailureReason.FAILED_TO_INVITE_EXISTING_USER_WITH_EXISTING_EMAIL;
import static com.pkb.patient.invite.PatientRegister.FailureReason.FAILED_TO_PARSE;
import static com.pkb.patient.invite.PatientRegister.FailureReason.ORG_DOES_NOT_HAVE_EXACTLY_ONE_ORG_LEVEL_ID_TYPE;
import static com.pkb.patient.invite.PatientRegister.FailureReason.PATIENT_IS_OPTED_OUT;
import static com.pkb.patient.invite.PatientRegister.FailureReason.TEAM_DOES_NOT_HAVE_EXACTLY_ONE_TEAM_LEVEL_ID_TYPE;
import static com.pkb.patient.invite.PatientRegister.FailureReason.UNKNOWN;
import static com.pkb.patient.invite.PatientRegister.FailureReason.USER_WITH_NATIONAL_ID_EXISTS;
import static com.pkb.patient.invite.PatientRegister.FailureReason.USER_WITH_ORG_LEVEL_ID_EXISTS;
import static com.pkb.patient.invite.PatientRegister.FailureReason.USER_WITH_TEAM_LEVEL_ID_EXISTS;
import static java.util.Collections.emptyList;
import static java.util.Collections.singletonList;
import static org.apache.commons.collections4.CollectionUtils.isEmpty;

public interface PatientRegister {
    /**
     * @return Either a registered User or a model describing errors why registration has failed.
     */
    Either<FailedInvitation, SuccessfulInvitation> registerAndActivate(@NotNull InvitationDto patientInvitation,
            PatientConsentUpdateTemplate patientConsent);

    /**
     * @param invitations
     *         bulk of Patient invitations to be executed.
     * @param patientConsent
     * @return Full Registration results.
     */
    List<Either<FailedInvitation, SuccessfulInvitation>> registerAndActivateInBulk(@NotNull Collection<InvitationDto> invitations,
            PatientConsentUpdateTemplate patientConsent);

    enum SuccessType {
        ALREADY_IN_TEAM,
        ADDED_TO_TEAM,
        INVITED_TO_REGISTER,
        ADDED_TO_TEAM_DEFERRED,
        NEW_ACCOUNT_CREATED,
        NEW_ACCOUNT_CREATED_AND_INVITATION_SENT
    }

    /**
     * Inviting a patient is the main objective.
     */
    final class SuccessfulInvitation {
        private final SuccessType successType;
        private final PKBPerson user;
        /**
         * <pre>
         * - {@code null} means, inviting carer one was not attempted.
         * - An empty list means that carer one was successfully invited.
         * - A list with items in it means inviting carer one has failed and list contains failure reason(s).
         * </pre>
         */
        private @Nullable List<Error> carerOneInvitationErrors;
        /**
         * <pre>
         * - {@code null} means, inviting carer two was not attempted.
         * - An empty list means that carer was was successfully invited.
         * - A list with items in it means inviting carer two has failed and list contains failure reason(s).
         * </pre>
         */
        private @Nullable List<Error> carerTwoInvitationErrors;

        public SuccessfulInvitation(@NotNull SuccessType successType, @NotNull PKBPerson user) {
            this.successType = successType;
            this.user = user;
        }

        public SuccessType getSuccessType() {
            return successType;
        }

        public PKBPerson getUser() {
            return user;
        }

        public SuccessfulInvitation withCarerOneInvitationErrors(List<Error> errors) {
            this.carerOneInvitationErrors = ImmutableList.copyOf(errors);
            return this;
        }

        public SuccessfulInvitation withCaresTwoInvitationErrors(List<Error> errors) {
            this.carerTwoInvitationErrors = ImmutableList.copyOf(errors);
            return this;
        }

        public boolean isCarerOneSuccessfullyInvited() {
            return carerOneInvitationErrors != null && carerOneInvitationErrors.isEmpty();
        }

        public List<Error> getCarerOneInvitationErrors() {
            return carerOneInvitationErrors == null ? emptyList() : carerOneInvitationErrors;
        }

        public boolean isCarerTwoSuccessfullyInvited() {
            return carerTwoInvitationErrors != null && carerTwoInvitationErrors.isEmpty();
        }

        public List<Error> getCarerTwoInvitationErrors() {
            return carerTwoInvitationErrors == null ? emptyList() : carerTwoInvitationErrors;
        }

        public boolean isPartialSuccess() {
            return !isEmpty(carerOneInvitationErrors) || !isEmpty(carerTwoInvitationErrors);
        }
    }

    enum FailureReason {
        FAILED_TO_PARSE,
        USER_WITH_NATIONAL_ID_EXISTS,
        USER_WITH_ORG_LEVEL_ID_EXISTS,
        USER_WITH_TEAM_LEVEL_ID_EXISTS,
        ORG_DOES_NOT_HAVE_EXACTLY_ONE_ORG_LEVEL_ID_TYPE,
        TEAM_DOES_NOT_HAVE_EXACTLY_ONE_TEAM_LEVEL_ID_TYPE,
        FAILED_TO_INVITE_EXISTING_USER_WITHOUT_EMAIL,
        FAILED_TO_INVITE_EXISTING_USER_WITH_EXISTING_EMAIL,
        EXISTING_USER_IS_NOT_A_PATIENT,
        PATIENT_IS_OPTED_OUT,
        UNKNOWN
    }

    final class FailedInvitation {
        public static FailedInvitation failedOnOrgLevelIdType(@NotNull List<Error> errors) {
            return new FailedInvitation(ORG_DOES_NOT_HAVE_EXACTLY_ONE_ORG_LEVEL_ID_TYPE, errors);
        }

        public static FailedInvitation failedOnTeamLevelIdType(@NotNull List<Error> errors) {
            return new FailedInvitation(TEAM_DOES_NOT_HAVE_EXACTLY_ONE_TEAM_LEVEL_ID_TYPE, errors);
        }

        public static FailedInvitation failedOnNationalId(@NotNull List<Error> errors) {
            return new FailedInvitation(USER_WITH_NATIONAL_ID_EXISTS, errors);
        }

        public static FailedInvitation failedOnOrgLevelId(@NotNull List<Error> errors) {
            return new FailedInvitation(USER_WITH_ORG_LEVEL_ID_EXISTS, errors);
        }

        public static FailedInvitation failedOnTeamLevelId(@NotNull List<Error> errors) {
            return new FailedInvitation(USER_WITH_TEAM_LEVEL_ID_EXISTS, errors);
        }

        public static FailedInvitation failedOnUserTypeMismatch(@NotNull List<Error> errors) {
            return new FailedInvitation(EXISTING_USER_IS_NOT_A_PATIENT, errors);
        }

        public static FailedInvitation failedOnOptedOutCheck(@NotNull List<Error> errors) {
            return new FailedInvitation(PATIENT_IS_OPTED_OUT, errors);
        }

        public static FailedInvitation failureForUnknownReason(@NotNull List<Error> errors) {
            return new FailedInvitation(UNKNOWN, errors);
        }

        public static FailedInvitation invitingExistingUserWithoutEmail(@NotNull Error error) {
            return new FailedInvitation(FAILED_TO_INVITE_EXISTING_USER_WITHOUT_EMAIL, singletonList(error));
        }

        public static FailedInvitation invitingExistingUserWithExistingEmail(@NotNull Error error) {
            return new FailedInvitation(FAILED_TO_INVITE_EXISTING_USER_WITH_EXISTING_EMAIL, singletonList(error));
        }

        public static FailedInvitation parsingFailure(@Nullable List<Error> patient, @Nullable List<Error> carerOne, @Nullable List<Error> carerTwo) {
            return new FailedInvitation(FAILED_TO_PARSE,
                    patient == null ? emptyList() : patient,
                    carerOne == null ? emptyList() : carerOne,
                    carerTwo == null ? emptyList() : carerTwo);
        }

        private FailureReason failureReason;
        private List<Error> patientInvitationErrors;
        private List<Error> carerOneInvitationErrors;
        private List<Error> carerTwoInvitationErrors;

        private FailedInvitation(@NotNull FailureReason failureReason, @NotNull List<Error> patientInvitationErrors) {
            this(failureReason, ImmutableList.copyOf(patientInvitationErrors), ImmutableList.of(), ImmutableList.of());
        }

        public FailedInvitation(@NotNull FailureReason failureReason, @NotNull List<Error> patient, @NotNull List<Error> carerOne, @NotNull List<Error> carerTwo) {
            this.failureReason = failureReason;
            this.patientInvitationErrors = patient;
            this.carerOneInvitationErrors = carerOne;
            this.carerTwoInvitationErrors = carerTwo;
        }

        public @NotNull FailureReason getPatientInvitationFailureReason() {
            return failureReason;
        }

        public @NotNull List<Error> getPatientInvitationErrors() {
            return patientInvitationErrors;
        }

        public @NotNull List<Error> getCarerOneInvitationErrors() {
            return carerOneInvitationErrors;
        }

        public @NotNull List<Error> getCarerTwoInvitationErrors() {
            return carerTwoInvitationErrors;
        }
    }
}
