package com.pkb.patient.invite;

import com.pkb.datamodel.Email;
import org.jetbrains.annotations.NotNull;

import java.time.LocalDate;
import java.util.Optional;

public class InviteeDto {
    private final NameDto name;
    private final InviteeIdsDto inviteeIds;

    private final Email email;
    private final AddressDto address;
    private final LocalDate dateOfBirth;

    public InviteeDto(@NotNull NameDto name, @NotNull InviteeIdsDto inviteeIds) {
        this(name, inviteeIds, null, null, null);
    }

    public InviteeDto(@NotNull NameDto name, @NotNull InviteeIdsDto inviteeIds, Email email) {
        this(name, inviteeIds, email, null, null);
    }

    public InviteeDto(@NotNull NameDto name, @NotNull InviteeIdsDto inviteeIds, AddressDto address) {
        this(name, inviteeIds, null, address, null);
    }

    public InviteeDto(@NotNull NameDto name, @NotNull InviteeIdsDto inviteeIds, LocalDate dateOfBirth) {
        this(name, inviteeIds, null, null, dateOfBirth);
    }

    public InviteeDto(@NotNull NameDto name, @NotNull InviteeIdsDto inviteeIds, Email email, AddressDto address) {
        this(name, inviteeIds, email, address, null);
    }

    public InviteeDto(@NotNull NameDto name, @NotNull InviteeIdsDto inviteeIds, Email email, LocalDate dateOfBirth) {
        this(name, inviteeIds, email, null, dateOfBirth);
    }

    public InviteeDto(@NotNull NameDto name, @NotNull InviteeIdsDto inviteeIds, Email email, AddressDto address, LocalDate dateOfBirth) {
        this.name = name;
        this.inviteeIds = inviteeIds;
        this.email = email;
        this.address = address;
        this.dateOfBirth = dateOfBirth;
    }

    @NotNull
    public NameDto getName() {
        return name;
    }

    @NotNull
    public InviteeIdsDto getInviteeIds() {
        return inviteeIds;
    }

    @NotNull
    public Optional<Email> findEmail() {
        return Optional.ofNullable(email);
    }

    @NotNull
    public InviteeDto withEmail(Email email) {
        return new InviteeDto(name, inviteeIds, email, address, dateOfBirth);
    }

    @NotNull
    public Optional<AddressDto> findAddress() {
        return Optional.ofNullable(address);
    }

    @NotNull
    public InviteeDto withAddress(AddressDto address) {
        return new InviteeDto(name, inviteeIds, email, address, dateOfBirth);
    }

    @NotNull
    public Optional<LocalDate> findDateOfBirth() {
        return Optional.ofNullable(dateOfBirth);
    }

    @NotNull
    public InviteeDto withDateOfBirth(LocalDate dateOfBirth) {
        return new InviteeDto(name, inviteeIds, email, address, dateOfBirth);
    }
}
