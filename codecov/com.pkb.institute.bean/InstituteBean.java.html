package com.pkb.institute.bean;

import com.pkb.institute.InstituteLocal;
import com.pkb.util.DbUtil;
import org.jetbrains.annotations.NotNull;
import org.jetbrains.annotations.Nullable;

import javax.ejb.Local;
import javax.ejb.Stateless;
import javax.persistence.EntityManager;
import javax.persistence.PersistenceContext;
import javax.persistence.Query;
import javax.persistence.TypedQuery;
import java.util.Collection;
import java.util.List;
import java.util.Map;

import static com.pkb.util.DbUtil.getExactlyOneUniqueResultOrNull;
import static java.util.stream.Collectors.toMap;

// The remainder of this bean can be dropped at the same time as EHRBean is removed
@Stateless(mappedName = "InstituteBean")
@Local(InstituteLocal.class)
@Deprecated
public class InstituteBean implements InstituteLocal {

    @PersistenceContext
    private EntityManager em;

    @Override
    public @Nullable String getOrgName(long orgId) {
        TypedQuery<String> query = em.createQuery("SELECT name FROM Org o WHERE o.id = :orgId", String.class);
        query.setParameter("orgId", orgId);
        return getExactlyOneUniqueResultOrNull(query);
    }

    @Override
    public @Nullable String getTeamName(long teamId) {
        TypedQuery<String> query = em.createQuery("SELECT name FROM Team t WHERE t.id = :teamId", String.class);
        query.setParameter("teamId", teamId);
        return getExactlyOneUniqueResultOrNull(query);
    }

    @Override
    public @NotNull Map<Long, String> getOrgNames(@NotNull Collection<Long> orgIds) {
        return DbUtil.getInStreamOfBatches(orgIds, ids -> {
                    Query query = em.createQuery("SELECT id,name FROM Org o WHERE o.id IN (:orgIds)");
                    query.setParameter("orgIds", ids);
                    return (List<Object[]>) query.getResultList();
                })
                .collect(toMap(t -> (Long) t[0], t -> (String) t[1]));
    }

    @Override
    public @NotNull Map<Long, String> getTeamNames(@NotNull Collection<Long> teamIds) {
        return DbUtil.getInStreamOfBatches(teamIds, ids -> {
                    Query query = em.createQuery("SELECT id,name FROM Team t WHERE t.id IN (:teamIds)");
                    query.setParameter("teamIds", ids);
                    return (List<Object[]>) query.getResultList();

                })
                .collect(toMap(t -> (Long) t[0], t -> (String) t[1]));
    }
}
