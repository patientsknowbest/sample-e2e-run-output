package com.pkb.integrations.emis.model.restcalls;

import java.time.ZonedDateTime;
import java.util.stream.Collectors;

import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import com.fasterxml.jackson.annotation.JsonInclude;
import com.fasterxml.jackson.annotation.JsonPropertyOrder;
import com.fasterxml.jackson.databind.util.StdConverter;
import com.google.common.collect.ImmutableList;
import com.google.common.collect.ImmutableSet;

import com.pkb.integrations.emis.model.EmisApiStyle;
import com.pkb.integrations.emis.model.EmisAppointment;
import com.pkb.integrations.emis.model.EmisAppointmentSession;
import com.pkb.integrations.emis.model.EmisClinician;
import com.pkb.integrations.emis.model.EmisJacksonFixes;
import com.pkb.integrations.emis.model.EmisLocation;

@EmisApiStyle
@Value.Immutable
@JsonPropertyOrder({ "AppointmentsFromDateTime", "Appointments", "Locations", "SessionHolders", "Sessions" })
@SuppressWarnings("NullableProblems")
public interface EmisAppointmentsResponseRaw extends EmisResponse {

    @NotNull
    ImmutableList<EmisLocation> getLocations();

    @NotNull
    ImmutableList<EmisClinician> getSessionHolders();

    @NotNull
    ImmutableList<EmisAppointmentSession> getSessions();

    @NotNull
    ZonedDateTime getAppointmentsFromDateTime();

    @NotNull
    @JsonInclude(JsonInclude.Include.NON_NULL)
    ImmutableList<EmisAppointment> getAppointments();

    class DeserConverter extends StdConverter<ImmutableEmisAppointmentsResponseRaw, EmisAppointmentsResponse> {

        /**
         * This is ugly, see {@link EmisJacksonFixes} for more details
         */
        @NotNull
        @Override
        public EmisAppointmentsResponse convert(@NotNull ImmutableEmisAppointmentsResponseRaw value) {
            return ImmutableEmisAppointmentsResponse.builder()
                    .appointmentsFromDateTime(value.getAppointmentsFromDateTime())
                    .appointments(value.getAppointments().stream().map(EmisJacksonFixes::fixAppointment).collect(Collectors.toList()))
                    .build();
        }

    }

    class SerConverter extends StdConverter<EmisAppointmentsResponse, EmisAppointmentsResponseRaw> {

        /**
         * This is ugly, see {@link EmisJacksonFixes} for more details
         */
        @NotNull
        @Override
        public EmisAppointmentsResponseRaw convert(@NotNull EmisAppointmentsResponse value) {
            ImmutableSet<EmisAppointmentSession> sessions = value.getAppointments().stream().map(EmisAppointment::getSession).collect(ImmutableSet.toImmutableSet());
            return ImmutableEmisAppointmentsResponseRaw.builder()
                    .appointmentsFromDateTime(value.getAppointmentsFromDateTime())
                    .appointments(value.getAppointments().stream().map(EmisJacksonFixes::fixAppointment).collect(Collectors.toList()))
                    .sessions(sessions)
                    .locations(sessions.stream().map(EmisAppointmentSession::getLocation).collect(Collectors.toSet()))
                    .sessionHolders(sessions.stream().flatMap(session -> session.getClinicians().stream()).collect(Collectors.toSet()))
                    .build();
        }

    }

}
