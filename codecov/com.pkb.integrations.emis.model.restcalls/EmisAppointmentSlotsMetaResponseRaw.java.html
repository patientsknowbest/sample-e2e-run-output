package com.pkb.integrations.emis.model.restcalls;

import java.util.stream.Collectors;

import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import com.fasterxml.jackson.annotation.JsonPropertyOrder;
import com.fasterxml.jackson.databind.util.StdConverter;
import com.google.common.collect.ImmutableList;

import com.pkb.integrations.emis.model.EmisApiStyle;
import com.pkb.integrations.emis.model.EmisAppointmentSession;
import com.pkb.integrations.emis.model.EmisClinician;
import com.pkb.integrations.emis.model.EmisJacksonFixes;
import com.pkb.integrations.emis.model.EmisLocation;


@EmisApiStyle
@Value.Immutable
@JsonPropertyOrder({ "Locations", "SessionHolders", "Sessions" })
@SuppressWarnings("NullableProblems")
public interface EmisAppointmentSlotsMetaResponseRaw extends EmisResponse {

    @NotNull
    ImmutableList<EmisLocation> getLocations();

    @NotNull
    ImmutableList<EmisClinician> getSessionHolders();

    @NotNull
    ImmutableList<EmisAppointmentSession> getSessions();

    class DeserConverter extends StdConverter<ImmutableEmisAppointmentSlotsMetaResponseRaw, EmisAppointmentSlotsMetaResponse> {

        /**
         * This is ugly, see {@link EmisJacksonFixes} for more details
         */
        @NotNull
        @Override
        public EmisAppointmentSlotsMetaResponse convert(@NotNull ImmutableEmisAppointmentSlotsMetaResponseRaw value) {
            return ImmutableEmisAppointmentSlotsMetaResponse.builder()
                    .sessions(value.getSessions().stream().map(EmisJacksonFixes::fixSession).collect(Collectors.toList()))
                    .build();
        }

    }

    class SerConverter extends StdConverter<EmisAppointmentSlotsMetaResponse, EmisAppointmentSlotsMetaResponseRaw> {

        /**
         * This is ugly, see {@link EmisJacksonFixes} for more details
         */
        @NotNull
        @Override
        public EmisAppointmentSlotsMetaResponseRaw convert(@NotNull EmisAppointmentSlotsMetaResponse value) {
            return ImmutableEmisAppointmentSlotsMetaResponseRaw.builder()
                    .sessions(value.getSessions())
                    .locations(value.getSessions().stream().map(EmisAppointmentSession::getLocation).collect(Collectors.toSet()))
                    .sessionHolders(value.getSessions().stream().flatMap(session -> session.getClinicians().stream()).collect(Collectors.toSet()))
                    .build();
        }

    }

}
