package com.pkb.integrations.emis.model.restcalls;

import static org.springframework.http.HttpMethod.POST;

import java.time.LocalDate;
import java.util.Optional;

import org.immutables.value.Value;
import org.jetbrains.annotations.NotNull;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;

import com.pkb.integrations.emis.model.EmisApiStyle;

@JsonDeserialize(as = ImmutableEmisAccountLinkRequest.class)
@EmisApiStyle
@Value.Immutable
@SuppressWarnings("NullableProblems")
public abstract class EmisAccountLinkRequest extends EmisRequest<EmisAccountLinkResponse> {

    EmisAccountLinkRequest() {
        super(EmisAccountLinkResponse.class, "/me/applications", POST);
    }

    @NotNull
    public abstract Optional<String> getSurname();

    @NotNull
    public abstract Optional<LocalDate> getDateOfBirth();

    @NotNull
    public abstract LinkageDetails getLinkageDetails();

    @JsonDeserialize(as = ImmutableLinkageDetails.class)
    @Value.Immutable
    @EmisApiStyle
    public interface LinkageDetails {

        @NotNull
        String getAccountId();

        @NotNull
        String getLinkageKey();

        @NotNull
        String getNationalPracticeCode();
    }
}
