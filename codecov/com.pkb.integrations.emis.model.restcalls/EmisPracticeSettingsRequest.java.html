package com.pkb.integrations.emis.model.restcalls;

import static org.springframework.http.HttpMethod.GET;

import org.immutables.value.Value;

import com.fasterxml.jackson.annotation.JsonIgnore;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;

@JsonDeserialize(as = ImmutableEmisPracticeSettingsRequest.class)
@Value.Immutable
@SuppressWarnings("NullableProblems")
public abstract class EmisPracticeSettingsRequest extends EmisRequest<EmisPracticeSettingsResponse> {

    EmisPracticeSettingsRequest() {
        super(EmisPracticeSettingsResponse.class, "/practices/{nationalPracticeCode}/settings", GET);
    }

    public abstract String getPracticeCode();

    @Value.Derived
    @JsonIgnore
    @Override
    public String getEndPointPath() {
        //noinspection DynamicRegexReplaceableByCompiledPattern
        return super.getEndPointPath().replace("{nationalPracticeCode}", getPracticeCode());
    }

    @JsonIgnore
    @Value.Derived
    @Override
    public boolean serialiseToQueryString() {
        return false;
    }

}
