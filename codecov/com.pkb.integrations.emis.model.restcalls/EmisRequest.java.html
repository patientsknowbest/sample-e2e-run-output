package com.pkb.integrations.emis.model.restcalls;

import static org.springframework.http.HttpMethod.GET;

import org.immutables.value.Value;
import org.springframework.http.HttpMethod;

import com.fasterxml.jackson.annotation.JsonIgnore;

public abstract class EmisRequest<T extends EmisResponse> {

    private final Class<T> responseClass;
    private final String endPointPath;
    private final HttpMethod method;

    EmisRequest(Class<T> responseClass, String endPointPath, HttpMethod method) {
        this.responseClass = responseClass;
        this.endPointPath = endPointPath;
        this.method = method;
    }

    @JsonIgnore
    @Value.Derived
    public Class<T> getResponseClass() {
        return responseClass;
    }

    @SuppressWarnings("WeakerAccess")
    @JsonIgnore
    @Value.Derived
    public String getEndPointPath() {
        return endPointPath;
    }

    @SuppressWarnings("WeakerAccess")
    @JsonIgnore
    @Value.Derived
    public HttpMethod getMethod() {
        return method;
    }

    @JsonIgnore
    @Value.Derived
    public boolean includeBody() {
        return getMethod() != GET;
    }

    @JsonIgnore
    @Value.Derived
    public boolean serialiseToQueryString() {
        return getMethod() == GET;
    }

}
