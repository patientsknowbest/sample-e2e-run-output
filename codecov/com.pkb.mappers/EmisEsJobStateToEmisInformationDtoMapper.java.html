package com.pkb.mappers;

import com.pkb.dataupload.emis.esv2.EmisEsBatchState;
import com.pkb.dataupload.emis.esv2.EmisEsJobState;
import com.pkb.dto.EmisBatchInformationDto;
import com.pkb.dto.EmisJobInformationDto;
import com.pkb.dto.ImmutableEmisBatchInformationDto;
import com.pkb.dto.ImmutableEmisJobInformationDto;
import com.pkb.dto.ImmutableEmisJobInformationDto.Builder;
import io.vavr.Tuple2;

import java.time.LocalDateTime;
import java.time.ZoneId;
import java.util.Date;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

public class EmisEsJobStateToEmisInformationDtoMapper {

    public EmisJobInformationDto toDto(Tuple2<EmisEsJobState, Map<String, EmisEsBatchState>> input) {
        EmisEsJobState job = input._1;
        Map<String, EmisEsBatchState> batches = input._2;
        Builder builder = ImmutableEmisJobInformationDto.builder()
                .id(job.getId())
                .status(job.getStatus())
                .startDate(Date.from(job.getStarted()))
                .retryPermitted(job.isRetryPermitted());
        if (job.getFinished() != null) {
            builder = builder.endDate(Date.from(job.getFinished()));
        }
        List<EmisBatchInformationDto> batchInfos = batches.values().stream().map(this::toDto).collect(Collectors.toList());
        builder.addAllBatchInfo(batchInfos);
        return builder.build();
    }

    public EmisBatchInformationDto toDto(EmisEsBatchState batch) {
        return ImmutableEmisBatchInformationDto.builder()
                .lastProcessedBatchGuid(batch.getLastProcessedBatchGuid())
                .lastProcessedBatchDatestamp(batch.getLastProcessedBatchDatestamp())
                .lastProcessedBatchBulk(batch.isLastProcessedBatchBulk())
                .recordedDate(LocalDateTime.ofInstant(batch.getRecordedDate(), ZoneId.of("UTC")))
                .uniqueId(batch.getSftpUniqueId())
                .build();
    }
}
