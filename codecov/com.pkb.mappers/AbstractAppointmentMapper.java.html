package com.pkb.mappers;

import com.google.common.collect.ImmutableMap;
import com.pkb.calendar.entity.AppointmentDTO;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.AppointmentMessagingSettings;
import com.pkb.domain.appointment.AppointmentMessagingSettingsService;
import com.pkb.dto.ActionItemDto;
import com.pkb.dto.ActionsDto;
import com.pkb.dto.ImmutableUrlDto;
import com.pkb.dto.UrlDto;
import com.pkb.entities.enums.AppointmentStatus;
import com.pkb.service.datapoint.DataWithUIPermissions;
import com.pkb.util.PKBServiceUtil;

import java.util.ArrayList;
import java.util.List;
import java.util.Optional;
import java.util.UUID;

import static com.pkb.util.UrlUtil.buildHref;
import static com.pkb.util.WebNamespace.DIARY;
import static java.lang.String.valueOf;

public abstract class AbstractAppointmentMapper<D extends ActionsDto> implements EntityMapper<AppointmentDTO, D, AppointmentEntityDependency<AppointmentDTO>> {

    protected PKBServiceUtil pkbServiceUtil;
    private PhrConfig config;
    private DateTimeService dateTimeService;
    private AppointmentMessagingSettingsService appointmentMessagingSettingsService;

    AbstractAppointmentMapper(PKBServiceUtil pkbServiceUtil, PhrConfig config, DateTimeService dateTimeService, AppointmentMessagingSettingsService appointmentMessagingSettingsService) {
        this.pkbServiceUtil = pkbServiceUtil;
        this.config = config;
        this.dateTimeService = dateTimeService;
        this.appointmentMessagingSettingsService = appointmentMessagingSettingsService;
    }

    @Override
    public ActionItemDto buildEditActionItem(AppointmentDTO entity, EntityDependency<AppointmentDTO> dependency) {
        return buildAction("edit-data", buildEditAnchor(entity, dependency));
    }

    @Override
    public ActionItemDto buildDeleteActionItem(AppointmentDTO entity, EntityDependency<AppointmentDTO> dependency) {
        return buildAction("delete-data", buildDeleteAnchor(entity, dependency));
    }

    @Override
    public ActionItemDto buildPrivacyLabelActionItem(AppointmentDTO entity, EntityDependency<AppointmentDTO> dependency) {
        return buildAction("change-privacy", buildPrivacyChangeAnchor(entity, dependency));
    }

    @Override
    public ActionItemDto buildDiscussActionItem(AppointmentDTO entity, EntityDependency<AppointmentDTO> dependency) {
        throw new UnsupportedOperationException("Discussion is not applicable for Appointment entries");
    }

    protected ActionItemDto buildEmisCancelActionItem(AppointmentDTO entity, EntityDependency<AppointmentDTO> dependency) {
        return buildAction("cancellation", buildEmisCancelAnchor(entity, dependency));
    }

    private UrlDto buildEditAnchor(AppointmentDTO entity, EntityDependency<AppointmentDTO> dependency) {
        return ImmutableUrlDto.builder()
                .href(buildHref(DIARY, "editAppointment", ImmutableMap.of("uniqueId", String.valueOf(entity.getBaseFields().getUniqueId()),
                                "returnToAppointment",entity.isReturnToAppointment()?"true":"false"),
                        getContextOrAccessingUserId(dependency)))
                .content(pkbServiceUtil.getMessage("appointment.txt.edit_appointment", dependency.getLocale()))
                .build();
    }

    private UrlDto buildDeleteAnchor(AppointmentDTO entity, EntityDependency<AppointmentDTO> dependency) {
        return ImmutableUrlDto.builder()
                .href(buildHref(DIARY, "deleteAppointmentConfirm", ImmutableMap.of("uniqueId", String.valueOf(entity.getBaseFields().getUniqueId()),
                                "returnToAppointment",entity.isReturnToAppointment()?"true":"false"),
                        getContextOrAccessingUserId(dependency)))
                .content(pkbServiceUtil.getMessage("appointment.txt.delete_appointment", dependency.getLocale()))
                .build();
    }

     UrlDto buildPrivacyChangeAnchor(AppointmentDTO entity,EntityDependency<AppointmentDTO> dependency) {
        return ImmutableUrlDto.builder()
                .href(buildHref(DIARY, "editAppointmentPrivacy", ImmutableMap.of("uniqueId", String.valueOf(entity.getBaseFields().getUniqueId()),
                                                                                            "returnToAppointment",entity.isReturnToAppointment()?"true":"false"), getContextOrAccessingUserId(dependency)))
                .content(pkbServiceUtil.getMessage("datapoint.txt.edit_privacy", dependency.getLocale()))
                .build();
    }

    private UrlDto buildEmisCancelAnchor(AppointmentDTO entity, EntityDependency<AppointmentDTO> dependency) {
        return ImmutableUrlDto.builder()
                .href(buildHref(DIARY, "cancelEmisAppointmentForm", ImmutableMap.of("slotId", valueOf(entity.getEmisSlotId())), getContextOrAccessingUserId(dependency)))
                .content(pkbServiceUtil.getMessage("appointment.txt.cancel_appointment", dependency.getLocale()))
                .build();
    }


    @Override
    public List<ActionItemDto> buildActionItems(DataWithUIPermissions<AppointmentDTO> dto, AppointmentEntityDependency<AppointmentDTO> dependency) {

        List<ActionItemDto> actionItems = new ArrayList<>();

        buildManageUrl(dto, dependency).ifPresentOrElse(
                url -> actionItems.add(buildAction("edit-data", url)),
                () -> {
                    buildRescheduleUrl(dto, dependency).ifPresent(url -> actionItems.add(buildAction("change-appt", url)));
                    buildCancelUrl(dto, dependency).ifPresent(url -> actionItems.add(buildAction("cancellation", url)));
                }
            );

        actionItems.addAll(buildBaseActionItems(dto, dependency));
        if (dto.isPrivacyFlagEditable()) {
            actionItems.add(buildPrivacyLabelActionItem(dto.getData(), dependency));
        }

        return actionItems;
    }

    List<ActionItemDto> buildBaseActionItems(DataWithUIPermissions<AppointmentDTO> dto, AppointmentEntityDependency<AppointmentDTO> dependency) {
        AppointmentDTO entity = dto.getData();
        List<ActionItemDto> actionItems  = new ArrayList<>();

        if (dto.isEditOrDeletable() && !hasManagementPartnerUrl(dto, dependency)) {
            actionItems.add(buildEditActionItem(entity, dependency));
        }

        if (dto.isEditOrDeletable() && !hasManagementPartnerUrl(dto, dependency)) {
            actionItems.add(buildDeleteActionItem(entity, dependency));
        }

        return actionItems;
    }

    Optional<UrlDto> buildCancelUrl(DataWithUIPermissions<AppointmentDTO> dto, EntityDependency<AppointmentDTO> dependency) {
        AppointmentDTO entity = dto.getData();
        if (dto.isEditEmisIntegratedAppointment()) {
            if (entity.isEmisAppointmentLive()) {
                return buildEmisCancelActionItem(entity, dependency).getAnchor();
            }
        } else if (isAppointmentChangeable(dependency, entity)) {
            return Optional.of(buildAppointmentRequestUrl(entity.getBaseFields().getUniqueId(), dependency,
                    "appointment.txt.request_cancellation", "cancel"));
        }

        return Optional.empty();
    }

    Optional<UrlDto> buildRescheduleUrl(DataWithUIPermissions<AppointmentDTO> dto, EntityDependency<AppointmentDTO> dependency) {
        if (!dto.isEditEmisIntegratedAppointment() && isAppointmentChangeable(dependency, dto.getData())) {
            return Optional.of(buildAppointmentRequestUrl(dto.getData().getBaseFields().getUniqueId(), dependency,
                    "appointment.txt.request_change", "reschedule"));
        }
        return Optional.empty();
    }

    Optional<UrlDto> buildManageUrl(DataWithUIPermissions<AppointmentDTO> dto, AppointmentEntityDependency<AppointmentDTO> dependency) {
        if (hasManagementPartnerUrl(dto, dependency)) {
            return Optional.of(ImmutableUrlDto.builder()
                    .href(buildHref(DIARY, "appointmentManagementRedirect",
                            new ImmutableMap.Builder<String, String>()
                                    .put("dataPointUniqueId", String.valueOf(dto.getData().getBaseFields().getUniqueId()))
                                    .build(),
                            getContextOrAccessingUserId(dependency)))
                    .content(pkbServiceUtil.getMessage("appointment.txt.manage", dependency.getLocale()))
                    .build());
        }
        return Optional.empty();
    }

    private boolean hasManagementPartnerUrl(DataWithUIPermissions<AppointmentDTO> dto, AppointmentEntityDependency<AppointmentDTO> dependency) {
        return dependency.hasManagementPartnerUrl(dto.getData().getBaseFields().getUniqueId());
    }

    private boolean isAppointmentChangeable(EntityDependency<AppointmentDTO> dependency, AppointmentDTO entity) {
        boolean messagingEnabled = config.isAppointmentMessagingEnabled() &&
                appointmentMessagingSettingsService.getAppointmentMessaging(entity.getSource().getOrgId())
                        .map(AppointmentMessagingSettings::isAppointmentMessagingEnabled)
                        .orElse(false);

        return dependency.loggedInRequestContext().isPatient() &&
                dependency.loggedInRequestContext().isUserInOwnAccount() &&
                entity.getStartTimestamp().isAfter(dateTimeService.now()) &&
                entity.getStatus() == AppointmentStatus.ACTIVE &&
                entity.getAppointmentRequestConversationId() == null &&
                messagingEnabled;
    }

    private UrlDto buildAppointmentRequestUrl(UUID uniqueId,
                                              EntityDependency<AppointmentDTO> dependency,
                                              String textKey,
                                              String requestType) {
        return ImmutableUrlDto.builder()
                .href(buildHref(DIARY, "introduceAppointmentRequest",
                        new ImmutableMap.Builder<String, String>()
                                .put("uniqueId", String.valueOf(uniqueId))
                                .put("requestType", requestType)
                                .put("tab", "diary")
                                .put("subTab", "calendar")
                                .build(),
                        getContextOrAccessingUserId(dependency)))
                .content(pkbServiceUtil.getMessage(textKey, dependency.getLocale()))
                .build();
    }

}
