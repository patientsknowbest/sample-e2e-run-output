package com.pkb.mappers;

import com.pkb.dto.ImmutablePersonDto;
import com.pkb.dto.PersonDto;
import com.pkb.entities.enums.Gender;
import com.pkb.user.entity.PKBPerson;
import com.pkb.user.entity.PKBPersonHelper;
import org.jetbrains.annotations.Nullable;

import java.time.LocalDate;
import java.util.Date;
import java.util.Optional;

import static java.util.Collections.emptySortedSet;

// TODO Mapstruct: once 3.1+ is out, Vavr and Mapstruct 3.0 doesn't play ball, code generation ends in StackOverflow
// But there is no Mapstruct support of Immutables before 3.0.....
// https://github.com/mapstruct/mapstruct/issues/1716
// https://github.com/mapstruct/mapstruct/pull/1807
// https://www.gitmemory.com/issue/mapstruct/mapstruct/1751/495925020
public class PKBPersonToPersonDtoMapper {

    private final PKBPersonHelper pkbPersonHelper;

    private final OrgLevelIdMapper orgLevelIdMapper;

    private final NationalIdMapper nationalIdMapper;

    public PKBPersonToPersonDtoMapper(PKBPersonHelper pkbPersonHelper,
                                      OrgLevelIdMapper orgLevelIdMapper,
                                      NationalIdMapper nationalIdMapper) {
        this.pkbPersonHelper = pkbPersonHelper;
        this.orgLevelIdMapper = orgLevelIdMapper;
        this.nationalIdMapper = nationalIdMapper;
    }

    @Nullable
    public PersonDto personToPersonDto(PKBPerson person) {
        if (person == null) {
            return null;
        }
        ImmutablePersonDto.Builder builder = ImmutablePersonDto.builder()
                .address1(person.getAddress1())
                .address2(person.getAddress2())
                .city(person.getCity())
                .country(person.getCountry())
                .defaultAccountId(person.getDefaultAccountId())
                .emailConfirmed(person.isEmailConfirmed())
                .firstName(person.getFirstName())
                .frozenByTeam(person.isFrozenByTeam())
                .frozenByTeamId(person.getFrozenByTeamId())
                .gender(Gender.getById(person.getGender()))
                .humanUUID(person.getHumanUUID())
                .id(person.getId())
                .jobTitle(person.getJobTitle())
                .languageCode(person.getLanguageCode())
                .lastName(person.getLastName())
                .middleName(person.getMiddleName())
                .optoutDetails(person.getOptoutDetails())
                .postalCode(person.getPostalCode())
                .sharingEnabled(person.isSharingEnabled())
                .state(person.getState())
                .status(person.getStatus())
                .statusDate(person.getStatusDate())
                .title(person.getTitle())
                .userType(person.getUserType())
                .deathTimestamp(Optional.ofNullable(person.getDeathTimestamp()).map(Date::toInstant))
                .addressLong(person.getAddressLong())
                .phone(person.getPhone())
                .publicID(person.getPublicId());

        Optional<LocalDate> maybeDob = person.findDateOfBirth();
        if (maybeDob.isPresent()) {
            builder.age(pkbPersonHelper.calculateAge(person))
                    .dateOfBirth(maybeDob)
                    .under18(pkbPersonHelper.isUnderAge(18, person));
        }

        if (person.hasContactsLoaded()) {
            builder.contactable(person.isContactable())
                    .hasContact(person.hasContact())
                    .primaryEmail(person.getEmail())
                    .userName(person.getUserName());
        } else {
            builder.contactable(false)
                    .hasContact(false);
        }

        if (person.hasOrgLevelIdsLoaded()) {
            builder.orgLevelIds(orgLevelIdMapper.orgLevelIdToOrgLevelIdDTO(person.getOrgLevelIds()));
        } else {
            builder.orgLevelIds(emptySortedSet());
        }

        if (person.hasNationalIdsLoaded()) {
            builder.nationalIds(nationalIdMapper.nationalIdsToNationalIdDTOs(person.getNationalIds()));
        }

        return builder.build();

    }
}
