package com.pkb.mappers;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.SourceDetails;
import com.pkb.datamodel.consent.ImmutableSourceDTO;
import com.pkb.datamodel.consent.SourceDTO;
import org.springframework.stereotype.Component;

/**
 * Maps between old and new constructs for holding source information while we transition these
 * definitions into datamodel from remote-client
 *
 * SourceDTO -> SourceDetails
 * EHRRequestContext -> SourceDTO
 */
@Component
public class SourceDetailsMapper {

    //Map from new style DTO -> old style details
    public SourceDetails mapFromSourceDTO(SourceDTO sdto) {

        SourceDetails details = new SourceDetails();
        sdto.connectingOrgId().ifPresentOrElse(details::setConnectingOrgId, () -> sdto.orgId().ifPresent(details::setConnectingOrgId));
        sdto.hl7PartnerId().ifPresent(details::setHl7PartnerId);
        sdto.sourceText().ifPresent(details::setText);
        sdto.orgId().ifPresent(details::setOrgId);
        sdto.personId().ifPresent(details::setPersonId);
        details.setRoute(sdto.route());
        sdto.teamId().ifPresent(details::setTeamId);
        sdto.maybeResponsibleOrganisationIdentifier().ifPresent(details::setResponsibleOrganisationIdentifier);

        return details;
    }

    //Map from EHRRequestContext -> new style DTO
    public SourceDTO mapFromEHRRequestContext(EHRRequestContext requestContext) {
        return ImmutableSourceDTO.builder()
                .route(requestContext.getAccessRoute())
                .personId(requestContext.getMaybeAccessingUserId())
                .teamId(requestContext.getTeamId())
                .orgId(requestContext.getOrgId())
                .connectingOrgId(requestContext.getMaybeConnectingOrgId())
                .hl7PartnerId(requestContext.getMaybeHl7PartnerId())
                .maybeResponsibleOrganisationIdentifier(requestContext.getMaybeResponsibleOrganisationIdentifier())
                .build();
    }
}
