package com.pkb.mappers;

import com.pkb.dto.ImmutableOrgLevelIdDTO;
import com.pkb.dto.OrgLevelIdDTO;
import com.pkb.user.entity.OrgLevelId;
import org.jetbrains.annotations.NotNull;

import java.util.Optional;
import java.util.SortedSet;
import java.util.TreeSet;

import static java.util.stream.Collectors.toCollection;

/**
 * Maps {@link OrgLevelId} entity to {@link OrgLevelIdDTO}  service layer representation.
 */
public class OrgLevelIdMapper {

    private OrgMapper orgMapper;

    private OrgLevelIdTypeMapper orgLevelIdTypeMapper;

    public OrgLevelIdMapper(OrgMapper orgMapper, OrgLevelIdTypeMapper orgLevelIdTypeMapper) {
        this.orgMapper = orgMapper;
        this.orgLevelIdTypeMapper = orgLevelIdTypeMapper;
    }

    SortedSet<OrgLevelIdDTO> orgLevelIdToOrgLevelIdDTO(SortedSet<OrgLevelId> orgLevelIds) {
        return orgLevelIds.stream().map(this::orgLevelIdToOrgLevelIdDTO).collect(toCollection(TreeSet::new));
    }

    private OrgLevelIdDTO orgLevelIdToOrgLevelIdDTO(@NotNull OrgLevelId orgLevelId) {
        return ImmutableOrgLevelIdDTO.builder()
                .org(Optional.ofNullable(orgLevelId.getOrg()).map(orgMapper::orgToOrgDTO))
                .value(Optional.ofNullable(orgLevelId.getValue()))
                .type(Optional.ofNullable(orgLevelId.getType()).map(orgLevelIdTypeMapper::orgLevelIdTypeToOrgLevelIdTypeDTO))
                .personId(Optional.ofNullable(orgLevelId.getPersonId()))
                .build();
    }
}
