package com.pkb.mappers;

import com.google.common.collect.ImmutableMap;
import com.pkb.app.entity.SourceDetails;
import com.pkb.calendar.entity.AppointmentDTO;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.domain.appointment.AppointmentMessagingSettingsService;
import com.pkb.dto.ActionItemDto;
import com.pkb.dto.AppointmentViewDto;
import com.pkb.dto.ImmutableAppointmentViewDto;
import com.pkb.dto.ImmutableUrlDto;
import com.pkb.dto.UrlDto;
import com.pkb.service.datapoint.DataWithUIPermissions;
import com.pkb.util.PKBServiceUtil;

import java.util.List;
import java.util.Optional;

import static com.pkb.util.UrlUtil.buildHref;
import static com.pkb.util.WebNamespace.DIARY;
import static org.apache.commons.lang3.StringUtils.defaultIfBlank;

public class AppointmentViewMapper extends AbstractAppointmentMapper<AppointmentViewDto> {

    public AppointmentViewMapper(PKBServiceUtil pkbServiceUtil, PhrConfig config, DateTimeService dateTimeService, AppointmentMessagingSettingsService appointmentMessagingSettingsService) {
        super(pkbServiceUtil, config, dateTimeService, appointmentMessagingSettingsService);
    }

    @Override
    public AppointmentViewDto entityToDto(DataWithUIPermissions<AppointmentDTO> dto, AppointmentEntityDependency<AppointmentDTO> dependency) {
        AppointmentDTO entity = dto.getData();
        SourceDetails source = entity.getSource();
        List<ActionItemDto> actionItems = buildActionItems(dto, dependency);
        Optional<UrlDto> editPrivacyUrlDto = dto.isPrivacyFlagEditable() ? Optional.of(buildPrivacyChangeAnchor(dto.getData(), dependency)) : Optional.empty();
        return ImmutableAppointmentViewDto.builder()
                .sourceSimpleDisplay(defaultIfBlank(source.getDisplayText(), pkbServiceUtil.getMessage("source.not_set", dependency.getLocale())))
                .sourceFullDisplay(source.getDisplayTextFull())
                .baseFields(entity.getBaseFields())
                .dropdownId(buildDropdownId(entity))
                .cancelUrl(buildCancelUrl(dto, dependency))
                .changeUrl(buildRescheduleUrl(dto, dependency))
                .addAllActions(actionItems)
                .sourceType(buildSourceType(source))
                .route(buildRoute(dto, pkbServiceUtil, dependency.getLocale()))
                .moreNhsUrl(buildMoreNhsUrl(actionItems, dto, dependency))
                .editPrivacyUrl(editPrivacyUrlDto)
                .hasManagementPartnerUrl(dependency.hasManagementPartnerUrl(entity.getBaseFields().getUniqueId()))
                .build();
    }

    private Optional<? extends UrlDto> buildMoreNhsUrl(List<ActionItemDto> actionItems, DataWithUIPermissions<AppointmentDTO> dto, EntityDependency<AppointmentDTO> dependency) {
        if (actionItems.isEmpty() || dto.isEditEmisIntegratedAppointment() || !dto.isEditOrDeletable()) {
            return Optional.empty();
        }

        AppointmentDTO entity = dto.getData();
        return Optional.of(
                ImmutableUrlDto.builder()
                .href(buildHref(DIARY, "nhsAppointmentMenu", ImmutableMap.of("uniqueId", String.valueOf(entity.getBaseFields().getUniqueId()),"returnToAppointment",entity.isReturnToAppointment()?"true":"false"),
                        getContextOrAccessingUserId(dependency)))
                .content(pkbServiceUtil.getMessage("appointmentView.txt.more", dependency.getLocale()))
                .build());
    }
}
