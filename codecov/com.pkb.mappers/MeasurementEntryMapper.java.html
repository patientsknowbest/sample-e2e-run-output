package com.pkb.mappers;

import com.google.common.collect.ImmutableMap;
import com.pkb.app.entity.SourceDetails;
import com.pkb.dto.ActionItemDto;
import com.pkb.dto.AmountDto;
import com.pkb.dto.ImmutableAmountDto;
import com.pkb.dto.ImmutableMeasurementEntryDto;
import com.pkb.dto.ImmutableUrlDto;
import com.pkb.dto.MeasurementEntryDto;
import com.pkb.dto.UrlDto;
import com.pkb.service.datapoint.DataWithUIPermissions;
import com.pkb.test.entity.MeasurementDTO;
import com.pkb.util.PKBServiceUtil;

import java.util.ArrayList;
import java.util.List;
import java.util.Optional;

import static com.pkb.util.UrlUtil.buildHref;
import static com.pkb.util.WebNamespace.TEST;
import static java.lang.String.valueOf;
import static java.util.stream.Collectors.toList;
import static org.apache.commons.lang3.StringUtils.defaultIfBlank;
import static org.apache.commons.lang3.StringUtils.isNotEmpty;

public class MeasurementEntryMapper implements EntityMapper<MeasurementDTO, MeasurementEntryDto, EntityDependency<MeasurementDTO>> {

    private PKBServiceUtil pkbServiceUtil;

    public MeasurementEntryMapper(PKBServiceUtil pkbServiceUtil) {
        this.pkbServiceUtil = pkbServiceUtil;
    }

    @Override
    public List<MeasurementEntryDto> entitiesToDtos(List<DataWithUIPermissions<MeasurementDTO>> dtos, EntityDependency<MeasurementDTO> dependency) {
        return dtos.stream()
                .map(dto -> entityToDto(dto, dependency))
                .sorted((dto1, dto2) -> dto2.getMeasureDate().compareTo(dto1.getMeasureDate()))
                .collect(toList());
    }

    @Override
    public MeasurementEntryDto entityToDto(DataWithUIPermissions<MeasurementDTO> dto, EntityDependency<MeasurementDTO> dependency) {
        MeasurementDTO entity = dto.getData();
        SourceDetails source = entity.getSource();
        return ImmutableMeasurementEntryDto.builder()
                .sourceSimpleDisplay(defaultIfBlank(source.getDisplayText(),
                        pkbServiceUtil.getMessage("source.not_set", dependency.getLocale())))
                .sourceFullDisplay(source.getDisplayTextFull())
                .baseFields(entity.getBaseFields())
                .dropdownId(buildDropdownId(entity))
                .amount(buildAmount(entity))
                .measureDate(entity.getMeasureDate())
                .id(Optional.ofNullable(entity.getId()))
                .addAllActions(buildActionItems(dto, dependency))
                .sourceType(buildSourceType(source))
                .route(buildRoute(dto, pkbServiceUtil, dependency.getLocale()))
                .build();
    }

    @Override
    public List<ActionItemDto> buildActionItems(DataWithUIPermissions<MeasurementDTO> dto,
                                                EntityDependency<MeasurementDTO> dependency) {
        MeasurementDTO entity = dto.getData();
        List<ActionItemDto> actionItems = new ArrayList<>();

        if (dto.isEditOrDeletable() && entity.getId() != null) {
            actionItems.add(buildDeleteActionItem(entity, dependency));
        }

        actionItems.add(buildDiscussActionItem(entity, dependency));
        return actionItems;
    }

    private AmountDto buildAmount(MeasurementDTO entity) {
        return ImmutableAmountDto.builder()
                .primary(Optional.of(entity)
                        .filter(dto -> isNotEmpty(dto.getValueFormatted()))
                        .map(dto -> formatAmount(entity.getValueFormatted(), entity.getMeasurementType().getFormattedUnitOrUnitAsDefault())))
                .secondary(Optional.of(entity)
                        .filter(dto -> isNotEmpty(dto.getValue2Formatted()))
                        .map(dto -> formatAmount(dto.getValue2Formatted(), dto.getMeasurementType().getFormattedUnit2OrUnit2AsDefault())))
                .build();
    }

    private String formatAmount(String value, String unit) {
        return value + " " + unit;
    }

    @Override
    public ActionItemDto buildDeleteActionItem(MeasurementDTO entity, EntityDependency<MeasurementDTO> dependency) {
        return buildAction("delete-data", buildDeleteAnchor(entity, dependency));
    }

    private UrlDto buildDeleteAnchor(MeasurementDTO entity, EntityDependency<MeasurementDTO> dependency) {
        return ImmutableUrlDto.builder()
                .href(buildHref(TEST,
                        "measurementHistory!deleteMeasurement",
                        ImmutableMap.of(
                                "deleteId", valueOf(entity.getId()),
                                "measurementTypeId", entity.getMeasurementType().id().toString()),
                        getContextOrAccessingUserId(dependency)))
                .content(pkbServiceUtil.getMessage("datapoint.txt.delete", dependency.getLocale()))
                .build();
    }

    @Override
    public ActionItemDto buildDiscussActionItem(MeasurementDTO entity, EntityDependency<MeasurementDTO> dependency) {
        return buildAction("discuss", buildDiscussAnchor(entity, dependency));
    }

    private UrlDto buildDiscussAnchor(MeasurementDTO entity, EntityDependency<MeasurementDTO> dependency) {
        ImmutableMap.Builder<String, String> parameters = new ImmutableMap.Builder<String, String>()
                .put("measurementTypeId", entity.getMeasurementType().id().toString());
        // Calculated BMI measurement has no ID but it is still valid
        if (entity.getId() != null) {
            parameters.put("selectId", valueOf(entity.getId()));
        }
        return ImmutableUrlDto.builder()
                .href(buildHref(TEST,
                        "measurementHistory!discussMeasurements",
                        parameters.build(),
                        getContextOrAccessingUserId(dependency)))
                .content(pkbServiceUtil.getMessage("datapoint.txt.discuss", dependency.getLocale()))
                .build();
    }

    @Override
    public ActionItemDto buildEditActionItem(MeasurementDTO entity,
            EntityDependency<MeasurementDTO> dependency) {
        throw new UnsupportedOperationException("Discussion is not applicable for Measurement entries");
    }

    @Override
    public ActionItemDto buildPrivacyLabelActionItem(MeasurementDTO entity,
            EntityDependency<MeasurementDTO> dependency) {
        throw new UnsupportedOperationException("Discussion is not applicable for Measurement entries");
    }
}
