package com.pkb.mappers;

import com.pkb.dto.I18nTextDTO;
import com.pkb.dto.ImmutableMeasurementGraphsDto;
import com.pkb.dto.ImmutableMeasurementOverviewDto;
import com.pkb.dto.ImmutableMeasurementValueDto;
import com.pkb.dto.ImmutableUrlDto;
import com.pkb.dto.MeasurementGraphsDto;
import com.pkb.dto.MeasurementOverviewDto;
import com.pkb.dto.MeasurementValueDto;
import com.pkb.test.entity.MeasurementDTO;
import com.pkb.test.entity.MeasurementHistoryDTO;
import com.pkb.test.entity.MeasurementType;
import com.pkb.util.PKBServiceUtil;
import com.pkb.util.UrlUtil;
import io.vavr.collection.LinkedHashMap;

import java.util.ArrayList;
import java.util.List;
import java.util.Locale;
import java.util.Map;
import java.util.Optional;

import static com.pkb.util.WebNamespace.TEST;
import static java.lang.String.valueOf;
import static java.util.stream.Collectors.toList;
import static org.apache.commons.lang3.StringUtils.isNotEmpty;

public class MeasurementMapper {

    private static final String ACTION = "measurementHistory";

    private SparkLineGraphMapper sparkLineGraphMapper;

    private PKBServiceUtil pkbServiceUtil;

    public MeasurementMapper(PKBServiceUtil pkbServiceUtil, SparkLineGraphMapper sparkLineGraphMapper) {
        this.pkbServiceUtil = pkbServiceUtil;
        this.sparkLineGraphMapper = sparkLineGraphMapper;
    }

    public List<MeasurementOverviewDto> historyToOverviewDtos(List<MeasurementHistoryDTO> dtos, Long contextUserId,
            Locale locale) {
        return dtos.stream()
                .map(historyDTO -> historyToOverviewDto(historyDTO, contextUserId, locale))
                .collect(toList());
    }

    private MeasurementOverviewDto historyToOverviewDto(MeasurementHistoryDTO historyDTO, Long contextUserId, Locale locale) {
        MeasurementType measurementType = historyDTO.getMeasurementType();
        MeasurementDTO latestMeasurement = historyDTO.getLatestMeasurement();
        List<I18nTextDTO> warnings = Optional.ofNullable(latestMeasurement)
                .map(MeasurementDTO::getWarnings)
                .orElse(List.of());
        List<String> warningsText = warnings.stream()
                .map(w -> pkbServiceUtil.getMessage(w.getProperty(), w.getParams(), locale))
                .collect(toList());

        var isInternalType = historyDTO.getMeasurementType().id().asInternalId().isDefined();

        return ImmutableMeasurementOverviewDto.builder()
                .title(isInternalType ? pkbServiceUtil.getMessage(measurementType.displayName(), locale) : measurementType.displayName())
                .graphs(buildGraphs(historyDTO))
                .displayCategory(measurementType.getDisplayCategory())
                .detailUrl(buildDetailUrl(measurementType, contextUserId))
                .addAllWarnings(warningsText)
                .addAllContent(latestMeasurement != null ? buildContent(latestMeasurement, measurementType) : List.of())
                .build();
    }

    private MeasurementGraphsDto buildGraphs(MeasurementHistoryDTO historyDTO) {
        if (historyDTO.getXData().size() > 1 || historyDTO.getXData2().size() > 1 ) {
            return sparkLineGraphMapper.measurementHistoryToGraphDto(historyDTO);
        }

        //one point does not constitute a trend so don't show the graph
        return ImmutableMeasurementGraphsDto.builder().build();
    }

    private List<MeasurementValueDto> buildContent(MeasurementDTO latestMeasurement,
            MeasurementType measurementType) {
        ArrayList<MeasurementValueDto> content = new ArrayList<>(List.of(ImmutableMeasurementValueDto.builder()
                .value(latestMeasurement.getValueFormatted())
                .unit(measurementType.getFormattedUnitOrUnitAsDefault()).build()));
        if (isNotEmpty(latestMeasurement.getValue2Formatted())) {
            content.add(ImmutableMeasurementValueDto.builder()
                    .value(latestMeasurement.getValue2Formatted())
                    .unit(measurementType.getFormattedUnit2OrUnit2AsDefault()).build());
        }
        return content;
    }

    private ImmutableUrlDto buildDetailUrl(MeasurementType measurementType, Long contextUserId) {
        Map<String, String> parameters = LinkedHashMap.of("measurementTypeId", valueOf(measurementType.id())).toJavaMap();
        return ImmutableUrlDto.builder()
                .href(UrlUtil.buildHref(TEST, ACTION, parameters, contextUserId))
                .isEscapeAmp(true)
                .build();
    }
}
