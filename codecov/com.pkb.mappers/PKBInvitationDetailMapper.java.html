package com.pkb.mappers;

import com.google.common.annotations.VisibleForTesting;
import com.pkb.app.entity.SourceDetails;
import com.pkb.encounter.entity.PKBInvitationDetail;
import com.pkb.entities.doc.DocInvitation;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.Nullable;
import org.mapstruct.Mapper;

// TODO PHR-11035 - https://pkbdev.atlassian.net/browse/PHR-11035
// addUserId should be saved into a different column in doc.Invitation table
// After that this class could be deleted
@Mapper
public abstract class PKBInvitationDetailMapper {

    public PKBInvitationDetail map(DocInvitation invitation) {
        PKBInvitationDetail detail = new PKBInvitationDetail(new SourceDetails());
        detail.setAddUserId(extractAddUserId(invitation));
        return detail;
    }

    @Nullable
    @VisibleForTesting
    String extractAddUserId(DocInvitation invitation) {
        String userId = "";
        String invitationType = invitation.getDocInvitationType();
        String[] invitationTypeFields = invitationType.split("#");
        if (invitationTypeFields.length == 2) {
            userId = invitationTypeFields[1];
            if (userId.startsWith(PKBInvitationDetail.USER_TO_ADD)) {
                userId = userId.substring(PKBInvitationDetail.USER_TO_ADD.length());
                if (StringUtils.isNumeric(userId) && StringUtils.isNotBlank(userId)) {
                    return userId;
                }
            }
        }
        return null;
    }
}
