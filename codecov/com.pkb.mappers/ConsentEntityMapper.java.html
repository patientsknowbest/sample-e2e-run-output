package com.pkb.mappers;

import com.pkb.datamodel.consent.PatientConsentDTO;
import com.pkb.institute.entity.Team;
import com.pkb.util.PKBServiceUtil;

import java.util.ArrayList;
import java.util.List;
import java.util.Locale;
import java.util.Map;

public interface ConsentEntityMapper {

    String PRIVACY_GENERAL = "consentcategory.generalhealth.badge";
    String PRIVACY_SEXUAL = "consentcategory.sexualhealth.badge";
    String PRIVACY_MENTAL = "consentcategory.mentalhealth.badge";
    String PRIVACY_SOCIAL = "consentcategory.socialcare.badge";
    String ONLY_DATA_ADDED_BY_TEAM = "consentcategory.only_data_added_by_team";

    default List<String> buildTeamConsents(Team team,
                                           Map<Long, PatientConsentDTO> consents,
                                           PKBServiceUtil pkbServiceUtil,
                                           Locale locale) {

        List<String> teamConsents = buildPrivacyFlags(team.getId(), consents, pkbServiceUtil, locale);
        if (teamConsents.isEmpty()) {
            teamConsents.add(pkbServiceUtil.getMessage(ONLY_DATA_ADDED_BY_TEAM, new String[] { team.getName() }, locale));
        }
        return teamConsents;
    }

    default List<String> buildPrivacyFlags(long id,
                                           Map<Long, PatientConsentDTO> consents,
                                           PKBServiceUtil pkbServiceUtil,
                                           Locale locale) {
        List<String> privacyFlags = new ArrayList<>();
        PatientConsentDTO patientConsent = consents.get(id);
        if (patientConsent != null) {
            if (patientConsent.generalHealthConsentGranted()) {
                privacyFlags.add(pkbServiceUtil.getMessage(PRIVACY_GENERAL, locale));
            }
            if (patientConsent.sexualHealthConsentGranted()) {
                privacyFlags.add(pkbServiceUtil.getMessage(PRIVACY_SEXUAL, locale));
            }
            if (patientConsent.mentalHealthConsentGranted()) {
                privacyFlags.add(pkbServiceUtil.getMessage(PRIVACY_MENTAL, locale));
            }
            if (patientConsent.socialCareConsentGranted()) {
                privacyFlags.add(pkbServiceUtil.getMessage(PRIVACY_SOCIAL, locale));
            }
        }

        return privacyFlags;
    }
}
