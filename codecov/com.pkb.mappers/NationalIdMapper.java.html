package com.pkb.mappers;

import com.pkb.dto.ImmutableNationalIdDTO;
import com.pkb.dto.NationalIdDTO;
import com.pkb.user.entity.NationalId;
import org.jetbrains.annotations.NotNull;

import java.util.Optional;
import java.util.Set;

import static java.util.stream.Collectors.toSet;

/**
 * Maps {@link NationalId} entity to {@link NationalIdDTO} service layer representation.
 */
public class NationalIdMapper {

    Set<NationalIdDTO> nationalIdsToNationalIdDTOs(Set<NationalId> nationalIds) {
        return nationalIds.stream().map(this::nationalIdToNationalIdDTO).collect(toSet());
    }

    private NationalIdDTO nationalIdToNationalIdDTO(@NotNull NationalId nationalId) {
        return ImmutableNationalIdDTO.builder()
                .type(Optional.ofNullable(nationalId.getType()))
                .value(Optional.ofNullable(nationalId.getValue()))
                .build();
    }
}
