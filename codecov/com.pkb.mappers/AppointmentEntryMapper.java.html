package com.pkb.mappers;

import com.google.common.collect.ImmutableMap;
import com.pkb.app.entity.SourceDetails;
import com.pkb.calendar.entity.AppointmentDTO;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.domain.appointment.AppointmentMessagingSettingsService;
import com.pkb.dto.AppointmentEntryDto;
import com.pkb.dto.ImmutableAppointmentEntryDto;
import com.pkb.dto.ImmutableUrlDto;
import com.pkb.dto.UrlDto;
import com.pkb.service.datapoint.DataWithUIPermissions;
import com.pkb.util.PKBServiceUtil;

import java.util.Locale;

import static com.pkb.entities.enums.AppointmentStatus.ACTIVE;
import static com.pkb.entities.enums.Route.EMIS_PFS;
import static com.pkb.util.UrlUtil.buildHref;
import static com.pkb.util.WebNamespace.DIARY;
import static java.lang.String.valueOf;
import static java.util.Optional.ofNullable;
import static org.apache.commons.lang3.StringUtils.defaultIfBlank;

public class AppointmentEntryMapper extends AbstractAppointmentMapper<AppointmentEntryDto> {

    public AppointmentEntryMapper(PKBServiceUtil pkbServiceUtil, PhrConfig config, DateTimeService dateTimeService, AppointmentMessagingSettingsService appointmentMessagingSettingsService) {
        super(pkbServiceUtil, config, dateTimeService, appointmentMessagingSettingsService);
    }

    @Override
    public AppointmentEntryDto entityToDto(DataWithUIPermissions<AppointmentDTO> dto, AppointmentEntityDependency<AppointmentDTO> dependency) {
        AppointmentDTO entity = dto.getData();
        SourceDetails source = entity.getSource();
        return ImmutableAppointmentEntryDto.builder()
                .sourceSimpleDisplay(defaultIfBlank(source.getDisplayText(), pkbServiceUtil.getMessage("source.not_set", dependency.getLocale())))
                .sourceFullDisplay(source.getDisplayTextFull())
                .baseFields(entity.getBaseFields())
                .dropdownId(buildDropdownId(entity))
                .startTimestamp(entity.getStartTimestamp())
                .status(getStatusText(entity, dependency.getLocale()))
                .notActive(entity.getStatus() != ACTIVE)
                .detailUrl(buildDetailUrl(entity, dependency))
                .subject(ofNullable(entity.getSubject()))
                .location(ofNullable(entity.getLocation()))
                .updatedSpeciality(ofNullable(entity.getUpdatedSpeciality()))
                .addAllActions(buildActionItems(dto, dependency))
                .nonPkbParticipants(entity.getNonPkbParticipantsNames())
                .sourceType(buildSourceType(source))
                .route(buildRoute(dto, pkbServiceUtil, dependency.getLocale()))
                .hasManagementPartnerUrl(dependency.hasManagementPartnerUrl(entity.getBaseFields().getUniqueId()))
                .build();
    }

    private UrlDto buildViewEmisAnchor(AppointmentDTO entity, EntityDependency<AppointmentDTO> dependency) {
        return ImmutableUrlDto.builder()
                .href(buildHref(DIARY, "viewEmisAppointment", ImmutableMap.of("slotId", valueOf(entity.getEmisSlotId())), getContextOrAccessingUserId(dependency)))
                .build();
    }

    private UrlDto buildViewAnchor(AppointmentDTO entity, EntityDependency<AppointmentDTO> dependency) {
        return ImmutableUrlDto.builder()
                .href(buildHref(DIARY, "viewAppointment", ImmutableMap.of("uniqueId", valueOf(entity.getBaseFields().getUniqueId())), getContextOrAccessingUserId(dependency)))
                .build();
    }

    private String getStatusText(AppointmentDTO entity, Locale locale) {
        String textKey;
        switch (entity.getStatus()) {
            case DNA:
                textKey = "appointmentEntry.txt.status.did_not_attend";
                break;
            case CANCELLED:
                textKey = "appointmentEntry.txt.status.cancelled";
                break;
            case ACTIVE:
                textKey = "appointmentEntry.txt.status.active";
                break;
            default:
                throw new IllegalArgumentException("Unexpected appointment status " + entity.getStatus());
        }

        return pkbServiceUtil.getMessage(textKey, locale);
    }

    private UrlDto buildDetailUrl(AppointmentDTO entity, EntityDependency<AppointmentDTO> dependency) {
        return entity.getSource().getRoute() == EMIS_PFS ? buildViewEmisAnchor(entity, dependency) : buildViewAnchor(entity, dependency);
    }
}
