package com.pkb.mappers;

import com.google.common.collect.Maps;
import com.pkb.dto.ImmutableTeamResearchPartnerDto;
import com.pkb.dto.TeamResearchPartnerDto;
import com.pkb.entities.enums.ResearchScope;
import com.pkb.entities.pub.PublicTeamResearchPartner;
import io.vavr.control.Option;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

@Component
public class TeamResearchPartnerMapper {
    private TeamResearchPartnerDto entityToDto(PublicTeamResearchPartner partner) {
        return ImmutableTeamResearchPartnerDto.builder()
                .name(Option.of(partner.getName()))
                .infoUrl(Option.of(partner.getInfoUrl()))
                .build();
    }

    public Map<ResearchScope, TeamResearchPartnerDto> entitiesToDtoMap(List<PublicTeamResearchPartner> partners) {
        return partners.stream().collect(Maps.toImmutableEnumMap(PublicTeamResearchPartner::getResearchScope,this::entityToDto));
    }

    public List<PublicTeamResearchPartner> dtoMapToEntities(Long teamId, Map<ResearchScope, TeamResearchPartnerDto> dtoMap) {
        return dtoMap.entrySet().stream()
                .map(entry -> {
                    var entity = new PublicTeamResearchPartner();
                    entity.setResearchScope(entry.getKey());
                    entity.setTeamId(teamId);
                    entry.getValue().name().peek(entity::setName);
                    entry.getValue().infoUrl().peek(entity::setInfoUrl);

                    return entity;
                }).collect(Collectors.toList());
    }
}
