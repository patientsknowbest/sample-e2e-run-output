package com.pkb.mappers;

import com.pkb.dto.ImmutableOrgDTO;
import com.pkb.dto.OrgDTO;
import com.pkb.institute.entity.Org;
import org.jetbrains.annotations.NotNull;

import java.util.Optional;

/**
 * Maps {@link Org} entity to {@link OrgDTO} service layer representation.
 */
public class OrgMapper {

    OrgDTO orgToOrgDTO(@NotNull Org org) {
        return ImmutableOrgDTO.builder()
                .publicId(org.getPublicId())
                .name(org.getName())
                .code(org.getCode())
                .crmLink(Optional.ofNullable(org.getCrmLink()))
                .hl7NewPatientTeamCode(Optional.ofNullable(org.getHl7NewPatientTeamCode()))
                .country(org.getCountry())
                .hl7AutoCreateRecords(org.getHl7AutoCreateRecords())
                .orgAccountId(org.getOrgAccountId())
                .hl7CheckNhsNumberStatus(org.getHl7CheckNhsNumberStatus())
                .timeZoneId(org.getTimeZoneId())
                .logoFileName(Optional.ofNullable(org.getLogoFileName()))
                .patientViewSuspended(org.isPatientViewSuspended())
                .build();
    }
}
