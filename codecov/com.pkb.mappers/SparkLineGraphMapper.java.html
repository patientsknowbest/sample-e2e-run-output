package com.pkb.mappers;

import com.pkb.common.datetime.DateTimeService;
import com.pkb.dto.ImmutableMeasurementGraphsDto;
import com.pkb.dto.MeasurementGraphsDto;
import com.pkb.test.entity.MeasurementHistoryDTO;
import com.pkb.util.SparkPathLineGraphTag;
import org.jetbrains.annotations.NotNull;

import java.util.ArrayList;
import java.util.List;

public class SparkLineGraphMapper {

    private static final String GRAPH_COLOR_YELLOW = "9c8e11";
    private static final String GRAPH_WIDTH_100 = "100";
    private static final String GRAPH_HEIGHT_20 = "20";

    private final DateTimeService dateTimeService;

    public SparkLineGraphMapper(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    public MeasurementGraphsDto measurementHistoryToGraphDto(MeasurementHistoryDTO historyDTO) {
        var latestMeasurement = historyDTO.getLatestMeasurement();
        if (latestMeasurement == null) {
            return ImmutableMeasurementGraphsDto.builder().build();
        }

        var graphs = new ArrayList<String>();
        if (latestMeasurement.getValue() != null) {
            graphs.add(buildGraphPath(historyDTO.getXData(), historyDTO.getYData()));
        }
        if (latestMeasurement.getValue2() != null) {
            graphs.add(buildGraphPath(historyDTO.getXData2(), historyDTO.getYData2()));
        }

        return ImmutableMeasurementGraphsDto.builder().addAllGraphPaths(graphs).build();
    }

    @NotNull
    public String buildGraphPath(List<Double> xData, List<Double> yData) {
        var tag = new SparkPathLineGraphTag();
        tag.setX(xData);
        tag.setY(yData);
        tag.setForceXMin(String.valueOf(dateTimeService.nowZonedDateTime().minusMonths(6L).toInstant().toEpochMilli()));
        tag.setForceXMax(String.valueOf(dateTimeService.now().toEpochMilli()));
        tag.setColor(GRAPH_COLOR_YELLOW);
        tag.setLastColor(GRAPH_COLOR_YELLOW);
        tag.setRangeColor(GRAPH_COLOR_YELLOW);
        tag.setHeight(GRAPH_HEIGHT_20);
        tag.setWidth(GRAPH_WIDTH_100);
        return tag.constructImagePath();
    }
}
