package com.pkb.mappers;

import com.pkb.dto.ImmutableOrgLevelIdTypeDTO;
import com.pkb.dto.OrgLevelIdTypeDTO;
import com.pkb.user.entity.OrgLevelIdType;
import org.jetbrains.annotations.NotNull;

import java.util.Optional;

/**
 * Maps {@link OrgLevelIdType} entity to {@link OrgLevelIdTypeDTO}  service layer representation.
 */
public class OrgLevelIdTypeMapper {

    private OrgMapper orgMapper;

    public OrgLevelIdTypeMapper(OrgMapper orgMapper) {
        this.orgMapper = orgMapper;
    }

    OrgLevelIdTypeDTO orgLevelIdTypeToOrgLevelIdTypeDTO(@NotNull OrgLevelIdType orgLevelIdType) {
        return ImmutableOrgLevelIdTypeDTO.builder()
                .id(orgLevelIdType.getId())
                .org(Optional.ofNullable(orgLevelIdType.getOrg()).map(orgMapper::orgToOrgDTO))
                .name(Optional.ofNullable(orgLevelIdType.getName()))
                .hl7AssigningAuthority(Optional.ofNullable(orgLevelIdType.getHl7AssigningAuthority()))
                .hl7TypeCode(Optional.ofNullable(orgLevelIdType.getHl7TypeCode()))
                .build();
    }
}
