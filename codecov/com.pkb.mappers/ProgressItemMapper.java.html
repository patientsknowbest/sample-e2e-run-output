package com.pkb.mappers;

import com.pkb.dto.ImmutableProgressItemDto;
import com.pkb.dto.ProgressItemDto;
import com.pkb.dto.ProgressItemDto.CssClass;
import com.pkb.util.PKBServiceUtil;

import java.util.ArrayList;
import java.util.List;
import java.util.Locale;

public class ProgressItemMapper {

    private PKBServiceUtil pkbServiceUtil;

    public ProgressItemMapper(PKBServiceUtil pkbServiceUtil) {
        this.pkbServiceUtil = pkbServiceUtil;
    }

    public List<ProgressItemDto> toProgressItemDtos(int teams,
                                                    boolean isWelcomeStepPresent,
                                                    int teamIdx,
                                                    Locale locale) {
        ArrayList<ProgressItemDto> progressItemDtos = new ArrayList<>();
        int overallIndex = 0;
        if (isWelcomeStepPresent) {
            overallIndex +=1;
            progressItemDtos.add(buildWelcomeItem(teams, teamIdx, locale));
        }

        if (teams == 1) {
            overallIndex +=1;
            progressItemDtos.add(buildSingleTeamItem(isWelcomeStepPresent, teamIdx, overallIndex, locale));
        } else {
            for (int i = 0; i < teams; i++) {
                overallIndex +=1;
                progressItemDtos.add(buildTeamItem(teams, teamIdx, overallIndex, i, locale));
            }
        }

        return progressItemDtos;
    }

    private ProgressItemDto buildTeamItem(int teams, int teamIdx, int overallIndex, int teamIndex, Locale locale) {
        CssClass cssClass = CssClass.DISABLED;
        if (teamIdx >= 0) {
            if (teamIndex == teamIdx) {
                cssClass = CssClass.ACTIVE;
            } else if (teamIndex < teamIdx) {
                cssClass = CssClass.VISITED;
            }
        }

        return buildItem("consentProgress.txt.team", List.of(overallIndex, teamIndex + 1, teams),
                cssClass, locale);
    }

    private ProgressItemDto buildSingleTeamItem(boolean hasWelcome, int teamIdx, int overallIndex, Locale locale) {
        if (hasWelcome) {
            return buildItem("consentProgress.txt.team_only", List.of(overallIndex),
                    teamIdx < 0 ? CssClass.DISABLED : CssClass.ACTIVE, locale);
        } else {
            return buildItem("consentProgress.txt.team_uncounted", CssClass.ACTIVE, locale);
        }
    }

    private ProgressItemDto buildWelcomeItem(int teams, int teamIdx, Locale locale) {
        String welcomeTextKey = "consentProgress.txt.welcome";
        if (teams == 0) {
            welcomeTextKey = "consentProgress.txt.welcome_uncounted";
        }
        return buildItem(welcomeTextKey, teamIdx < 0 ? CssClass.ACTIVE : CssClass.VISITED, locale);
    }

    private ProgressItemDto buildItem(String textKey, CssClass cssClass, Locale locale) {
        return ImmutableProgressItemDto.builder()
                .text(pkbServiceUtil.getMessage(textKey, locale))
                .cssClass(cssClass).build();
    }

    private ProgressItemDto buildItem(String textKey, List args, CssClass cssClass, Locale locale) {
        return ImmutableProgressItemDto.builder()
                .text(pkbServiceUtil.getMessage(textKey, args.toArray(), locale))
                .cssClass(cssClass).build();
    }
}
