package com.pkb.mappers;

import com.pkb.dto.ImmutableTeamDTO;
import com.pkb.dto.TeamDTO;
import com.pkb.institute.entity.Team;
import org.jetbrains.annotations.NotNull;

import java.util.Optional;

/**
 * Maps {@link Team} entity to {@link TeamDTO} service layer representation.
 */
public class TeamMapper {

    private OrgMapper orgMapper;

    public TeamMapper(OrgMapper orgMapper) {
        this.orgMapper = orgMapper;
    }

    public TeamDTO teamToTeamDTO(@NotNull Team team) {
        return ImmutableTeamDTO.builder()
                .id(team.getId())
                .publicId(team.getPublicId())
                .name(team.getName())
                .teamBasedMessagingEnabled(team.isTeamBasedMessagingEnabled())
                .org(orgMapper.orgToOrgDTO(team.getOrg()))
                .code(team.getCode())
                .country(team.getCountry())
                .specialtyId(Optional.ofNullable(team.getSpecialtyId()))
                .teamCRMLink(Optional.ofNullable(team.getTeamCRMLink()))
                .adminCRMLink(Optional.ofNullable(team.getAdminCRMLink()))
                .logoFileName(team.getLogoFileName())
                .headerFileName(team.getHeaderFileName())
                .footerFileName(team.getFooterFileName())
                .mandatoryNationalId(team.isMandatoryNationalId())
                .patientIdSupport(team.getPatientIdSupport())
                .sendSymptomReminders(team.isSendSymptomReminders())
                .symptomAlarmThreshold(Optional.ofNullable(team.getSymptomAlarmThreshold()))
                .symptomAlarmAmberThreshold(Optional.ofNullable(team.getSymptomAlarmAmberThreshold()))
                .symptomAlarmDays(Optional.ofNullable(team.getSymptomAlarmDays()))
                .teamAccountId(Optional.ofNullable(team.getTeamAccountId()))
                .timeZoneId(Optional.ofNullable(team.getTimeZoneId()))
                .hl7AutoCreateRecords(team.getHl7AutoCreateRecords())
                .cancerTeam(team.isCancerTeam())
                .privacyFlags(team.getPrivacyFlags())
                .defaultRequestGeneral(team.isDefaultRequestGeneral())
                .defaultRequestSexualHealth(team.isDefaultRequestSexualHealth())
                .defaultRequestMentalHealth(team.isDefaultRequestMentalHealth())
                .defaultRequestSocialCare(team.isDefaultRequestSocialCare())
                .languageCode(team.getLanguageCode())
                .defaultNewProsContactable(team.isDefaultNewProsContactable())
                .build();
    }
}
