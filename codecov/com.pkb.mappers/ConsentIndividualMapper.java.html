package com.pkb.mappers;

import com.pkb.datamodel.consent.PatientConsentDTO;
import com.pkb.dto.ConsentIndividualDto;
import com.pkb.dto.ImmutableConsentIndividualDto;
import com.pkb.dto.ImmutableUrlDto;
import com.pkb.dto.UrlDto;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.PKBServiceUtil;
import com.pkb.util.UrlUtil;
import org.apache.commons.lang3.StringUtils;

import java.util.List;
import java.util.Locale;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;
import java.util.stream.Stream;

import static com.pkb.util.WebNamespace.PATIENT;
import static java.lang.String.valueOf;
import static java.util.stream.Collectors.toList;

public class ConsentIndividualMapper implements ConsentEntityMapper {

    private static final String ACTION = "viewConsentIndividual";

    private PKBServiceUtil pkbServiceUtil;

    public ConsentIndividualMapper(PKBServiceUtil pkbServiceUtil) {
        this.pkbServiceUtil = pkbServiceUtil;
    }

    public List<ConsentIndividualDto> individualsToConsentIndividualDtos(List<PKBPerson> individuals,
                                                                         Long contextUserId,
                                                                         Map<Long, PatientConsentDTO> consents,
                                                                         Locale locale) {
        return individuals.stream()
                .map(individual -> individualToConsentIndividualDto(individual, contextUserId, consents, locale))
                .collect(toList());
    }

    private ConsentIndividualDto individualToConsentIndividualDto(PKBPerson individual,
                                                                  Long contextUserId,
                                                                  Map<Long, PatientConsentDTO> consents,
                                                                  Locale locale) {

        return ImmutableConsentIndividualDto.builder()
                .name(buildName(individual))
                .editUrl(buildEditUrl(individual.getId(), contextUserId))
                .jobTitle(Optional.ofNullable(individual.getJobTitle()))
                .addAllPrivacyFlags(buildPrivacyFlags(individual.getId(), consents, pkbServiceUtil, locale))
                .build();
    }

    private UrlDto buildEditUrl(Long id, Long contextUserId) {
        Map<String, String> parameters = Map.of("clinicianId", valueOf(id));
        return ImmutableUrlDto.builder()
                .href(UrlUtil.buildHref(PATIENT, ACTION, parameters, contextUserId))
                .build();
    }

    private String buildName(PKBPerson individual) {
        return Stream.of(individual.getTitle(),
                       individual.getFirstName(),
                       individual.getLastName()).filter(StringUtils::isNotBlank)
                .collect(Collectors.joining(" "));
    }
}
