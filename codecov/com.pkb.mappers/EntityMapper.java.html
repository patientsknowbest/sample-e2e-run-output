package com.pkb.mappers;

import com.pkb.app.entity.SourceDetails;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.dto.ActionItemDto;
import com.pkb.dto.ImmutableActionItemDto;
import com.pkb.dto.UrlDto;
import com.pkb.entities.enums.Route;
import com.pkb.service.datapoint.DataWithUIPermissions;
import com.pkb.util.PKBServiceUtil;

import java.util.ArrayList;
import java.util.List;
import java.util.Locale;
import java.util.Optional;

import static java.util.stream.Collectors.toList;

public interface EntityMapper<E, D, T extends EntityDependency<E>> {

    default List<D> entitiesToDtos(List<DataWithUIPermissions<E>> dtos, T dependency) {
        return dtos.stream()
                .map(dto -> entityToDto(dto, dependency))
                .collect(toList());
    }

    D entityToDto(DataWithUIPermissions<E> dto, T dependency);

    default List<ActionItemDto> buildActionItems(DataWithUIPermissions<E> dto, T dependency) {
        E entity = dto.getData();
        List<ActionItemDto> actionItems = new ArrayList<>();

        if (dto.isEditOrDeletable()) {
            actionItems.add(buildEditActionItem(entity, dependency));
        }

        if (dto.isPrivacyFlagEditable()) {
            actionItems.add(buildPrivacyLabelActionItem(entity, dependency));
        }

        if (dto.isEditOrDeletable()) {
            actionItems.add(buildDeleteActionItem(entity, dependency));
        }

        actionItems.add(buildDiscussActionItem(entity, dependency));
        return actionItems;
    }

    default ActionItemDto buildAction(String cssClass, UrlDto anchor) {
        return ImmutableActionItemDto.builder()
                .iconCssClass(cssClass)
                .anchor(anchor)
                .build();
    }

    default Optional<String> buildSourceType(SourceDetails source) {
        return Optional.ofNullable(source.getSourceType())
                .map(Enum::name)
                .map(String::toLowerCase);
    }

    default Optional<String> buildRoute(DataWithUIPermissions<E> dto, PKBServiceUtil pkbServiceUtil, Locale locale) {
        return Optional.ofNullable(((IBaseDTO) dto.getData()).getSource().getRoute())
                .filter(Route::isCaptured)
                .map(route -> {
                    String routeKey = "com.pkb.app.entity.Route." + route;
                    return pkbServiceUtil.getMessage("source.via", new String[] { pkbServiceUtil.getMessage(routeKey, locale) }, locale);
                });
    }

    default Long getContextOrAccessingUserId(EntityDependency<E> dependency) {
        return dependency.loggedInRequestContext().getContextOrAccessingUserId();
    }

    default String buildDropdownId(E entity) {
        return "options-" + ((IBaseDTO) entity).getBaseFields().getUniqueId();
    }

    ActionItemDto buildEditActionItem(E entity, EntityDependency<E> dependency);

    ActionItemDto buildDeleteActionItem(E entity, EntityDependency<E> dependency);

    ActionItemDto buildPrivacyLabelActionItem(E entity, EntityDependency<E> dependency);

    ActionItemDto buildDiscussActionItem(E entity, EntityDependency<E> dependency);

}
