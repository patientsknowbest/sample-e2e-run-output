package com.pkb.mappers;

import com.pkb.encounter.entity.PKBInvitation;
import com.pkb.entities.doc.DocInvitation;
import com.pkb.entities.enums.InvitationStatus;
import com.pkb.user.entity.PKBPerson;
import org.mapstruct.AfterMapping;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.mapstruct.MappingTarget;

import javax.inject.Inject;
import java.util.Map;

@Mapper
public abstract class PKBInvitationMapper {

    @Inject
    private PKBInvitationDetailMapper pkbInvitationDetailMapper;

    @Mapping(target = "targetUser", ignore = true)
    @Mapping(target = "accountKeyAccessToken", ignore = true)
    @Mapping(target = "invitationDetail", ignore = true)
    @Mapping(target = "invitationType", ignore = true)
    @Mapping(target = "accountKeyRequired", ignore = true)
    @Mapping(target = "accountId", source = "accountId")
    @Mapping(target = "authorUserId", source = "authorPersonId")
    @Mapping(target = "invitationStatus", source = "status")
    @Mapping(target = "grantedConsents", ignore = true)
    public abstract PKBInvitation map(DocInvitation invitation);

    public PKBInvitation map(DocInvitation invitation, Map<Long, PKBPerson> targetUserMap) {
        PKBInvitation result = map(invitation);
        result.setTargetUser(targetUserMap.get(invitation.getTargetPersonId()));
        return result;
    }

    /**
     * Don't let mapstruct use default enum parsing for InvitationStatus; we use special logic
     */
    protected InvitationStatus map(String value) {
        return InvitationStatus.value(value);
    }

    @AfterMapping
    protected void calculateSubject(DocInvitation invitation, @MappingTarget PKBInvitation result) {
        String invitationType = invitation.getDocInvitationType();
        result.setInvitationType(invitationType.substring(0, invitationType.indexOf('#')));
        result.setInvitationDetail(pkbInvitationDetailMapper.map(invitation));
    }

    @AfterMapping
    protected  void calculateGrantedConsents(DocInvitation invitation, @MappingTarget PKBInvitation result) {
        invitation.getGrantedConsents().ifPresent(result::setGrantedConsents);
    }

    @AfterMapping
    protected void calculateAccountKeyRequired(DocInvitation invitation, @MappingTarget PKBInvitation result) {
        result.setAccountKeyRequired(invitation.getAccountPrivateKey() != null);
    }
}
