package com.pkb.mappers;

import com.google.common.collect.ImmutableMap;
import com.pkb.app.entity.SourceDetails;
import com.pkb.diary.entity.DiaryEntry;
import com.pkb.dto.ActionItemDto;
import com.pkb.dto.DiaryEntryDto;
import com.pkb.dto.ImmutableDiaryEntryDto;
import com.pkb.dto.ImmutableUrlDto;
import com.pkb.dto.UrlDto;
import com.pkb.service.datapoint.DataWithUIPermissions;
import com.pkb.util.AttachmentUtil;
import com.pkb.util.PKBServiceUtil;

import java.util.Map;
import java.util.Optional;

import static com.pkb.util.UrlUtil.buildHref;
import static com.pkb.util.WebNamespace.DIARY;
import static java.lang.String.valueOf;
import static org.apache.commons.lang3.StringUtils.defaultIfBlank;

public class DiaryEntryMapper implements EntityMapper<DiaryEntry, DiaryEntryDto, EntityDependency<DiaryEntry>> {

    private static final Map<String, String> EXTENDED_PARAMETERS = ImmutableMap.of(
            "tab", "diary",
            "subTab", "journal"
    );
    private PKBServiceUtil pkbServiceUtil;
    private AttachmentUtil attachmentUtil;

    public DiaryEntryMapper(PKBServiceUtil pkbServiceUtil, AttachmentUtil attachmentUtil) {
        this.pkbServiceUtil = pkbServiceUtil;
        this.attachmentUtil = attachmentUtil;
    }

    @Override
    public DiaryEntryDto entityToDto(DataWithUIPermissions<DiaryEntry> dto, EntityDependency<DiaryEntry> dependency) {
        DiaryEntry entity = dto.getData();
        SourceDetails source = entity.getSource();
        return ImmutableDiaryEntryDto.builder()
                .sourceSimpleDisplay(defaultIfBlank(source.getDisplayText(), pkbServiceUtil.getMessage("source.not_set", dependency.getLocale())))
                .sourceFullDisplay(source.getDisplayTextFull())
                .baseFields(entity.getBaseFields())
                .dropdownId(buildDropdownId(entity))
                .htmlContentAllowed(entity.isHtmlContentAllowed())
                .id(entity.getId())
                .subject(Optional.ofNullable(entity.getSubjectOfEntry()))
                .content(Optional.ofNullable(entity.getEntryText()))
                .addAllAttachments(attachmentUtil.buildAttachments(entity.getAttachmentList(), EXTENDED_PARAMETERS, dependency))
                .addAllActions(buildActionItems(dto, dependency))
                .sourceType(buildSourceType(source))
                .route(buildRoute(dto, pkbServiceUtil, dependency.getLocale()))
                .build();
    }

    @Override
    public ActionItemDto buildEditActionItem(DiaryEntry entity, EntityDependency<DiaryEntry> dependency) {
        return buildAction("edit-data", buildEditAnchor(entity, dependency));
    }

    private UrlDto buildEditAnchor(DiaryEntry entity, EntityDependency<DiaryEntry> dependency) {
        return ImmutableUrlDto.builder()
                .href(buildHref(DIARY, "editDiaryEntry", ImmutableMap.of("id", valueOf(entity.getId())), getContextOrAccessingUserId(dependency)))
                .content(pkbServiceUtil.getMessage("datapoint.txt.edit", dependency.getLocale()))
                .build();
    }

    @Override
    public ActionItemDto buildDeleteActionItem(DiaryEntry entity, EntityDependency<DiaryEntry> dependency) {
        return buildAction("delete-data", buildDeleteAnchor(entity, dependency));
    }

    private UrlDto buildDeleteAnchor(DiaryEntry entity, EntityDependency<DiaryEntry> dependency) {
        return ImmutableUrlDto.builder()
                .href(buildHref(DIARY, "diaryEntry!deleteDiaryEntry", ImmutableMap.of("selectId", valueOf(entity.getId())), getContextOrAccessingUserId(dependency)))
                .content(pkbServiceUtil.getMessage("datapoint.txt.delete", dependency.getLocale()))
                .build();
    }

    @Override
    public ActionItemDto buildPrivacyLabelActionItem(DiaryEntry entity, EntityDependency<DiaryEntry> dependency) {
        return buildAction("change-privacy", buildPrivacyChangeAnchor(entity, dependency));
    }

    private UrlDto buildPrivacyChangeAnchor(DiaryEntry entity, EntityDependency<DiaryEntry> dependency) {
        return ImmutableUrlDto.builder()
                .href(buildHref(DIARY, "editDiaryPrivacy", ImmutableMap.of("dataPointId", valueOf(entity.getId())), getContextOrAccessingUserId(dependency)))
                .content(pkbServiceUtil.getMessage("datapoint.txt.edit_privacy", dependency.getLocale()))
                .build();
    }

    @Override
    public ActionItemDto buildDiscussActionItem(DiaryEntry entity, EntityDependency<DiaryEntry> dependency) {
        return buildAction("discuss", buildDiscussAnchor(entity, dependency));
    }

    private UrlDto buildDiscussAnchor(DiaryEntry entity, EntityDependency<DiaryEntry> dependency) {
        return ImmutableUrlDto.builder()
                .href(buildHref(DIARY, "discuss", ImmutableMap.of("id", valueOf(entity.getId())), getContextOrAccessingUserId(dependency)))
                .content(pkbServiceUtil.getMessage("datapoint.txt.discuss", dependency.getLocale()))
                .build();
    }

}
