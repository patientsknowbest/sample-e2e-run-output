package com.pkb.mappers;

import com.pkb.crypto.javasecurityutil.KeyUtil;
import com.pkb.entities.embeddables.EncodedEncryptedPrivateKeyInfo;
import com.pkb.entities.embeddables.EncodedEncryptedPrivateKeyInfoWithSecretDeriv;
import org.tolven.security.key.AccountPrivateKey;
import org.tolven.security.key.AccountSecretKey;

import javax.crypto.SecretKey;
import javax.crypto.spec.SecretKeySpec;
import java.io.IOException;
import java.security.GeneralSecurityException;
import java.security.PrivateKey;

/**
 * This is a temporary class to handle the bidirectional mapping between an AccountPrivateKey (remote-client) embeddable and
 * the EncodedEncryptedPrivateKeyInfo (database-entities) embeddable.
 *
 * This is needed during the migration from the old remote-client entities to the new database-entities, to bridge
 * the gaps which the transition is in place. Once we've fully converted over (i.e. EJBs are dead), this class can
 * be deleted, as it will no longer serve any purpose.
 *
 * This class deliberately does as little crypto as possible, and what there is, is not accessible outside this class
 * Fundamentally this is a mapper from new <-> old style, not a crypto object, and almost all the crypto is still
 * done in AccountPrivateKey/AccountSecretKey. This sometimes makes the code a bit "inefficient", as you'll go
 * new -> old -> new, but it's worth it to not propagate and copy paste the crypto around.
 */
public class AccountPrivateKeyMapper {

    private static final String NOT_INITIALIZED = "AccountPrivateKey not initialized";

    public AccountPrivateKeyMapper() {}

    /**
     * To from an old style AccountPrivateKey, with AccountSecretKey initialised, to a new style EncodedEncryptedPrivateKeyInfo
     */
    public EncodedEncryptedPrivateKeyInfo getEncodedEncryptedPrivateKeyInfoFromAccountPrivateKey(AccountPrivateKey oldStylePrivateKey) {

        AccountSecretKey oldStyleSecretKey = oldStylePrivateKey.getSecretKey();
        if (oldStyleSecretKey == null) {
            throw new IllegalStateException(NOT_INITIALIZED);
        }
        return new EncodedEncryptedPrivateKeyInfo(
                oldStylePrivateKey.getEncodedEncryptedPrivateKeyInfo(), oldStylePrivateKey.getKeySize(), oldStylePrivateKey.getAlgorithm(),
                oldStyleSecretKey.getEncryptedKey(), oldStyleSecretKey.getAlgorithm());
    }

    /**
     * To go from an old style AccountPrivateKey, with AccountSecretKey initialised, to a new style EncodedEncryptedPrivateKeyInfoWithSecretDeriv
     */
    public EncodedEncryptedPrivateKeyInfoWithSecretDeriv getEncodedEncryptedPrivateKeyInfoWithSecretDerivFromAccountPrivateKey(AccountPrivateKey oldStylePrivateKey) {

        AccountSecretKey oldStyleSecretKey = oldStylePrivateKey.getSecretKey();
        if (oldStyleSecretKey == null) {
            throw new IllegalStateException(NOT_INITIALIZED);
        }
        return new EncodedEncryptedPrivateKeyInfoWithSecretDeriv(
                oldStylePrivateKey.getEncodedEncryptedPrivateKeyInfo(), oldStylePrivateKey.getKeySize(), oldStylePrivateKey.getAlgorithm(),
                oldStyleSecretKey.getEncryptedKey(), oldStyleSecretKey.getAlgorithm(), oldStyleSecretKey.getSecretKeyDerivAlgorithm(),
                oldStyleSecretKey.getSecretKeyDerivDataLength(), oldStyleSecretKey.getSecretKeyDerivationSalt(), oldStyleSecretKey.getSecretKeyDerivIterations());
    }

    /**
     * To go from a new style EncodedEncryptedPrivateKeyInfoWithSecretDeriv to an "old style" AccountPrivateKey with initialised AccountSecretKey
     */
    public AccountPrivateKey getAccountPrivateKeyFromEncodedEncryptedPrivateKeyInfoWithSecretDeriv(EncodedEncryptedPrivateKeyInfoWithSecretDeriv newStylePrivateKey,
                                                                                                   String token)
            throws GeneralSecurityException, IOException {

        PrivateKey privateKey = getPrivateKey(newStylePrivateKey, token);
        AccountPrivateKey oldStylePrivateKey = AccountPrivateKey.getInstance();
        oldStylePrivateKey.init(privateKey, newStylePrivateKey.getPrivateKeySize(), newStylePrivateKey.getSecretKeyAlgorithm(), token,
                newStylePrivateKey.getSecretKeyDerivAlgorithm(), newStylePrivateKey.getSecretKeyDerivDataLength(),
                newStylePrivateKey.getSecretKeyDerivIterations(), newStylePrivateKey.getSecretKeyDerivSaltInline());
        return oldStylePrivateKey;
    }

    public static PrivateKey getPrivateKey(EncodedEncryptedPrivateKeyInfoWithSecretDeriv accountKey, String token) {
        try {
            SecretKey derivedKey = KeyUtil.deriveSecretKey(
                    accountKey.getSecretKeyDerivAlgorithm(),
                    token.toCharArray(),
                    accountKey.getSecretKeyDerivSaltInline(),
                    accountKey.getSecretKeyDerivIterations(),
                    accountKey.getSecretKeyDerivDataLength());

            byte[] clearKeyBytes = derivedKey.getEncoded();
            SecretKey secretKey = new SecretKeySpec(clearKeyBytes, accountKey.getSecretKeyAlgorithm());

            return KeyUtil.decryptPrivateKeyWithSecretKey(accountKey.getEncryptedPrivateKeyInfo(), secretKey);
        } catch (Exception ex) {
            throw new RuntimeException("Failed to decrypt with a PrivateKey", ex);
        }
    }
}
