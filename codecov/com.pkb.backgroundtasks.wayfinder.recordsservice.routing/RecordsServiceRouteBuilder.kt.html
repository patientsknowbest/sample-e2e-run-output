package com.pkb.backgroundtasks.wayfinder.recordsservice.routing

import com.pkb.backgroundtasks.config.BackgroundTasksProperties
import com.pkb.backgroundtasks.config.WayfinderRecordsServiceProperties
import com.pkb.backgroundtasks.routing.DownstreamChangeNotification
import com.pkb.backgroundtasks.wayfinder.recordsservice.service.RecordsServiceNotifierService
import com.pkb.common.testcontrol.camel.route.AbstractNamespaceAwareRouteBuilder
import com.pkb.common.testcontrol.services.PubSubNamespaceService
import io.micrometer.core.instrument.MeterRegistry
import org.apache.camel.Exchange
import org.apache.camel.component.jackson.JacksonDataFormat
import org.springframework.stereotype.Component

/**
 * We must nofify the Wayfinder Records Service when we receive appointments for a patient.
 * We'll send them a notification with the NHS number of the patient if we detected a new appointment in our downstream
 * / aggregated FHIR storage.
 *
 * https://pkbdev.atlassian.net/browse/PHR-9944
 */
@Component
class RecordsServiceRouteBuilder(
        backgroundTasksProperties: BackgroundTasksProperties,
        private val pubSubNamespaceService: PubSubNamespaceService,
        private val recordsServiceNotifierService: RecordsServiceNotifierService,
        private val meterRegistry: MeterRegistry
) : AbstractNamespaceAwareRouteBuilder() {
    companion object {
        const val INBOUND_NEW_APPOINTMENT = "direct:handleNewAppointment"
        const val ROUTE_ID = "new-appointment"
    }

    private val properties: WayfinderRecordsServiceProperties = backgroundTasksProperties.recordsService

    override fun configure() {
        onException(Exception::class.java)
                .process { logFailure(it) }
                .handled(false)

        from(INBOUND_NEW_APPOINTMENT)
                .routeId(ROUTE_ID)
                .process { if (!properties.enabled) it.isRouteStop = true }
                .filter { hasNewAppointment(it) }
                .filter { isForOnboardedTrust(it) }
                .bean(recordsServiceNotifierService, "findNhsNumber").id("$ROUTE_ID-find-nhs-number")
                .bean(recordsServiceNotifierService, "buildHttpMessage").id("$ROUTE_ID-build-http-message")
                .marshal(JacksonDataFormat(RecordsServiceNotifierService.NotificationRequest::class.java))
                .to("netty-http:${properties.requestUrl}").id("$ROUTE_ID-http-records-service")
                .process { logSuccess(it) }
    }

    override fun camelNamespaceService(): PubSubNamespaceService = pubSubNamespaceService

    override fun throwExceptionForNamespaceMismatch(): Boolean = false

    private fun hasNewAppointment(exchange: Exchange): Boolean {
        val notification = exchange.message.body as DownstreamChangeNotification
        val action = notification.attributes["action"]
        val resourceType = notification.attributes["resourceType"]
        return action.equals("CreateResource") && resourceType.equals("Appointment")
    }

    private fun isForOnboardedTrust(exchange: Exchange): Boolean {
        val odsCode = recordsServiceNotifierService.findOdsCode(exchange.message.body as DownstreamChangeNotification, exchange)
        //If the ods code is empty, and we haven't thrown an exception, that means we didn't have a source org, and
        // this is a patient-entered appointment. These DO NOT come across
        return !odsCode.isNullOrEmpty() && properties.wayfinderOnboardedTrustOdsCodes.contains(odsCode)
    }

    private fun logSuccess(exchange: Exchange) {
        meterRegistry.counter("wayfinder_appointment_records_service_notification_success").increment()
        log.info("NHS number sent to Wayfinder Appointment Records Service for ${exchange.properties["resourceId"]}")
    }

    private fun logFailure(exchange: Exchange) {
        meterRegistry.counter("wayfinder_appointment_records_service_notification_error").increment()
        log.error("Failed to send NHS number to Wayfinder Appointment Records Service", exchange.exception)
    }
}
