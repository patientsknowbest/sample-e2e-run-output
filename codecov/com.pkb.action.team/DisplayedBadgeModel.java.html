package com.pkb.action.team;

public enum DisplayedBadgeModel {

    GENERAL("general-health-consent", "consentcategory.generalhealth.badge"),
    MENTAL("mental-health-consent", "consentcategory.mentalhealth.badge"),
    SEXUAL("sexual-health-consent", "consentcategory.sexualhealth.badge"),
    SOCIAL("social-care-consent", "consentcategory.socialcare.badge");

    private final String cssClass;

    private final String text;

    DisplayedBadgeModel(String cssClass, String text) {
        this.cssClass = cssClass;
        this.text = text;
    }

    public String getCssClass() {
        return cssClass;
    }

    public String getText() {
        return text;
    }
}
