package com.pkb.action.team;

import com.google.common.collect.ImmutableMap;
import com.pkb.dto.PatientAccessLogDTO;
import com.pkb.dto.PatientConsentChangeReasonDTO;
import com.pkb.entities.enums.PrivacyFlag;

import java.time.Instant;
import java.util.Collection;
import java.util.Date;
import java.util.Map;
import java.util.function.Function;
import java.util.function.Predicate;

import static java.util.Collections.emptyList;
import static java.util.stream.Collectors.toList;

public class PatientAuditLogViewModel {

    private static final Instant UNKNOWN_ACCESS_TIME = Instant.parse("1900-01-01T00:00:00Z");

    private static Predicate<PatientConsentChangeReasonDTO> handleNullAsFalse(Function<PatientConsentChangeReasonDTO, Boolean> original) {
        return reason -> original.apply(reason) == Boolean.TRUE;
    }

    private static final ImmutableMap<Predicate<PatientConsentChangeReasonDTO>, DisplayedBadgeModel> CONSENT_PREDICATES = ImmutableMap.of(
            handleNullAsFalse(PatientConsentChangeReasonDTO::getGeneralHealthConsentGranted), DisplayedBadgeModel.GENERAL,
            handleNullAsFalse(PatientConsentChangeReasonDTO::getMentalHealthConsentGranted), DisplayedBadgeModel.MENTAL,
            handleNullAsFalse(PatientConsentChangeReasonDTO::getSexualHealthConsentGranted), DisplayedBadgeModel.SEXUAL,
            handleNullAsFalse(PatientConsentChangeReasonDTO::getSocialCareConsentGranted), DisplayedBadgeModel.SOCIAL
    );

    private static final ImmutableMap<PrivacyFlag, DisplayedBadgeModel> PRIVACY_FLAGS_TO_BADGE = ImmutableMap.of(
            PrivacyFlag.GENERAL, DisplayedBadgeModel.GENERAL,
            PrivacyFlag.MENTAL_HEALTH, DisplayedBadgeModel.MENTAL,
            PrivacyFlag.SEXUAL_HEALTH, DisplayedBadgeModel.SEXUAL,
            PrivacyFlag.SOCIAL_CARE, DisplayedBadgeModel.SOCIAL
    );

    private PatientAccessLogDTO dto;

    public PatientAuditLogViewModel(PatientAccessLogDTO dto) {
        this.dto = dto;
    }

    public Date getAccessTime() {
        return dto.getAccessTime();
    }

    public boolean getHasAccessTime() {
        return dto.getAccessTime().toInstant().isAfter(UNKNOWN_ACCESS_TIME);
    }

    public String getAccessedBy() {
        return dto.getAccessedBy();
    }

    public String getAccessType() {
        return dto.getAccessType();
    }

    public String getAccessedVia() {
        return dto.getAccessedVia();
    }

    public Collection<DisplayedBadgeModel> getConsentBadges() {
        if (dto.isConsentChangeEvent()) {
            return CONSENT_PREDICATES.entrySet().stream()
                    .filter(entry -> entry.getKey().test(dto.getConsentChangeReason()))
                    .map(Map.Entry::getValue)
                    .collect(toList());
        } else if (!dto.newFlags().get().isEmpty()) {
            return dto.newFlags()
                    .get()
                    .stream()
                    .map(PRIVACY_FLAGS_TO_BADGE::get)
                    .collect(toList());
        } else {
            return emptyList();
        }
    }

}
