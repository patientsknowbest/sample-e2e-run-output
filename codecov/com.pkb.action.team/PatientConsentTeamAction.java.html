package com.pkb.action.team;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.action.PatientConsentTeamDto;
import com.pkb.action.PatientConsentTeamService;
import com.pkb.action.PatientConsentTeamService.PatientConsentLinksDto;
import com.pkb.datamodel.consent.PatientConsentDTO;
import com.pkb.exception.PKBException;
import com.pkb.user.entity.PKBPerson;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.Optional;

import static com.pkb.action.BaseAction.NhsAppPage.HEALTH_RECORDS;

/**
 * Shows all other users that currently have access to this patient's account,
 * along with their granted consent. This will supersede
 * {@link PatientTeamAction} when NWL goes live
 *
 * @author @author Russ Gray <russ.gray@patientsknowbest.com>
 *
 */
public class PatientConsentTeamAction extends BaseAction {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final long serialVersionUID = 7457582749039473511L;

    // Only used if the logged-in user is a carer. This is used to restrict
    // carers to only grant their own level of consent to new carers or pros
    private PatientConsentDTO loggedInCarerConsent;

    private String from;

    private boolean canGrantGeneralHealthConsent;
    private boolean canGrantSexualHealthConsent;
    private boolean canGrantMentalHealthConsent;
    private boolean canGrantSocialCareConsent;

    private PatientConsentLinksDto patientConsentLinksDto;
    private PatientConsentTeamDto patientConsentTeamDto;

    @Autowired
    private PatientConsentTeamService patientConsentTeamService;

    @Override
    public String execute() {
        setNhsAppPage(HEALTH_RECORDS.name());

        String result = Action.SUCCESS;
        try {
            try {
                // Don't bother loading links if patient has opted-out
                if (isSharingEnabled()) {
                    loadLinks();
                }
            } catch (Exception e) {
                LOGGER.warn("could not find public id for patient", e);
            }
        } catch (Exception ex) {
            result = Action.ERROR;
            LOGGER.error("Error while getting team list", ex);
        }

        setPatientConsentTeamDto(patientConsentTeamService.buildPatientConsentTeamDto(getLoggedInEHRRequestContext()));

        LOGGER.info("Exiting execute result : {}", result);
        return result;
    }

    public String viewAddConsentIndividual() {
        long patientId = contextUserId == null ? loggedInUserUtil.getLoggedInUserId() : contextUserId;

        // Logged-in user is a carer if a patient is viewing a different patient
        boolean isCarer = isLoggedInUserPatient() && patientId == loggedInUserUtil.getLoggedInUserId();
        if (isCarer) {
            // TODO: MFA - Might need to refactor this action class.
            loggedInCarerConsent = getEHRRequestContext().getConsentStatus().getConsent().orElse(null);
        }

        populateData();
        return Action.SUCCESS;
    }

    // Set read-only variables
    private void populateData() {
        PKBPerson loggedInUser = loggedInUserUtil.getLoggedInPerson();
        PKBPerson contextUser = contextUserUtil.getContextUser();

        // Normally we can grant any consents (patient granting consent)
        canGrantGeneralHealthConsent = true;
        canGrantSexualHealthConsent = true;
        canGrantMentalHealthConsent = true;
        canGrantSocialCareConsent = true;
        // If we're a _clinician_ viewing this page, we shouldn't be able to bootstrap more consent than we have
        if (contextUser != null
                && loggedInUser.isPro()
                && !loggedInUser.getId().equals(contextUser.getId())) {
            // Clinicians can only grant consents they already have. No bootstrapping.
            Optional<PatientConsentDTO> consentOptional = getEHRRequestContext().getConsentStatus().getConsent();
            if (consentOptional.isPresent()) {
                PatientConsentDTO consent = consentOptional.get();
                canGrantGeneralHealthConsent = consent.generalHealthConsentGranted();
                canGrantSexualHealthConsent = consent.sexualHealthConsentGranted();
                canGrantMentalHealthConsent = consent.mentalHealthConsentGranted();
                canGrantSocialCareConsent = consent.socialCareConsentGranted();
            } else {
                LOGGER.error("Clinician {} has no consent, but is trying to grant access to another clinician", loggedInUser.getId());
            }
        }
    }

    public String listPatientTeam() {
        String result = Action.SUCCESS;
        try {
            loadLinks();
        } catch (Exception ex) {
            result = Action.ERROR;
            LOGGER.error("Error while getting team list", ex);
        }

        LOGGER.info("Exiting execute result : {}", result);
        return result;
    }

    private void loadLinks() {
        try {
            patientConsentLinksDto = patientConsentTeamService.buildPatientConsentLinksDto(getLoggedInEHRRequestContext(), getLocale());
        } catch (PKBException ex) {
            LOGGER.error("error loading teams for {}", getLoggedInEHRRequestContext().getContextOrAccessingUserId(), ex);
        }
    }

    public boolean isUserInOwnAccountAndHasTeams() {
        return getEHRRequestContext().isUserInOwnAccount() && !patientConsentLinksDto.getConsentTeamDtos().isEmpty();
    }

    public PatientConsentDTO getLoggedInCarerConsent() {
        return loggedInCarerConsent;
    }

    public String getFrom() {
        return from;
    }

    public void setFrom(String from) {
        this.from = from;
    }

    @Override
    public String getCustomStyle() {
        return MOBILE_OPTIMIZED_STYLE;
    }

    public boolean isCanGrantGeneralHealthConsent() {
        return canGrantGeneralHealthConsent;
    }

    public boolean isCanGrantSexualHealthConsent() {
        return canGrantSexualHealthConsent;
    }

    public boolean isCanGrantMentalHealthConsent() {
        return canGrantMentalHealthConsent;
    }

    public boolean isCanGrantSocialCareConsent() {
        return canGrantSocialCareConsent;
    }

    public PatientConsentTeamDto getPatientConsentTeamDto() {
        return patientConsentTeamDto;
    }

    public void setPatientConsentTeamDto(PatientConsentTeamDto patientConsentTeamDto) {
        this.patientConsentTeamDto = patientConsentTeamDto;
    }

    public PatientConsentLinksDto getPatientConsentLinksDto() {
        return patientConsentLinksDto;
    }
}
