package com.pkb.action.team;

import com.pkb.action.BaseAction;
import com.pkb.consent.model.NoConsentsRequired;
import com.pkb.entities.enums.UserType;
import com.pkb.exception.PKBException;
import com.pkb.institute.entity.Team;
import com.pkb.notification.entity.Activity.Action;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * For a team professional to freeze / unfreeze the record of an under-18 patient.
 * <p>
 * For freeze: verify current user is an active team clinician, with team-level access to patient (post-consent)
 * and that patient is under 18 and not already frozen.
 * <p>
 * For unfreeze -- ditto but don't check patient's age
 *
 * @author robwhelan
 */
public class PatientRecordFreezeAction extends BaseAction {
    private static final long serialVersionUID = 1L;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private PKBEmailMessageManager pkbEmailMessageManager;

    public String freeze() {
        UserType userType = loggedInUserUtil.getLoggedInUserType();
        Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();

        PKBPerson patient = contextUserUtil.getContextUser();

        // front-end bugs or people messing around:
        if (patient == null) {
            throw new PKBException("null contextuser");
        }
        if ((team == null) || !loggedInUserUtil.isLoggedInUserActiveTeamMember()) {
            throw new PKBException("current user not active team user: " + team);
        }
        if (userType != UserType.REG_CLINICIAN && userType != UserType.INSTITUTE_ADMIN) {
            throw new PKBException("current user can't freeze patients: " + userType);
        }

        if (!personHelper.isUnderAge(18, patient).orElse(true)) {
            throw new PKBException("patient must be under 18! found: " + personHelper.calculateAge(patient).orElse(0) + "; dob " + patient.getDateOfBirthString());
        }

        // failed validation that may happen in regular flows

        if (patient.isFrozenByTeam()) {
            addActionError(getText("patientRecordFreezeAction.err.already_frozen"));
            return SUCCESS;
        }

        try {
            userManager.beginTransaction();

            // update patient and notify (email)/record activity

            patient.setFrozenByTeamId(team.getId());
            userManager.updateUser(getEHRRequestContext(), patient);

            // Log the event and notify
            logActivityAndNotify(patient, team, true/*freeze*/);

            userManager.commitTransaction();
            addActionMessage(getText("patientRecordFreezeAction.msg.freeze_success"));
        } catch (Exception e) {
            throw new PKBException("failed freezing patient account for patient " + patient.getId(), e);
        } finally {
            userManager.rollbackIfNotCommitted();
        }

        return SUCCESS;
    }

    public String unfreeze() {
        UserType currentUserType = loggedInUserUtil.getLoggedInUserType();
        Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();

        PKBPerson patient = contextUserUtil.getContextUser();

        // front-end bugs or people messing around:
        if (patient == null) {
            throw new PKBException("null contextuser");
        }
        if ((team == null) || !loggedInUserUtil.isLoggedInUserActiveTeamMember()) {
            throw new PKBException("current user not active team user: " + team);
        }
        if (currentUserType != UserType.REG_CLINICIAN && currentUserType != UserType.INSTITUTE_ADMIN) {
            throw new PKBException("current user can't unfreeze patients: " + currentUserType);
        }

        // failed validation that may happen in regular flows

        if (!patient.isFrozenByTeam()) {
            addActionError(getText("patientRecordFreezeAction.err.already_unfrozen"));
            return SUCCESS;
        }

        try {
            userManager.beginTransaction();

            // update patient and notify (email)/record activity

            patient.setFrozenByTeamId(null);
            userManager.updateUser(getEHRRequestContext(), patient);

            // Log the event and notify
            logActivityAndNotify(patient, team, false/*freeze*/);

            userManager.commitTransaction();
            addActionMessage(getText("patientRecordFreezeAction.msg.unfreeze_success"));
        } catch (Exception e) {
            throw new PKBException("failed UNfreezing patient account for patient " + patient.getId(), e);
        } finally {
            userManager.rollbackIfNotCommitted();
        }

        return SUCCESS;
    }

    public String confirm() {
        return SUCCESS;
    }

    public String cancel() {
        return SUCCESS;
    }

    /**
     * @param freeze (false for unfreeze)
     */
    private void logActivityAndNotify(PKBPerson patient, Team actingTeam, boolean freeze) {

        Action action = (freeze) ? Action.RECORD_FROZEN : Action.RECORD_UNFROZEN;
        logActivityByOthers(action, dateTimeService.now(), false/*notify*/, new NoConsentsRequired());
        try {
            sendNotification(patient, actingTeam, freeze);
        } catch (PKBException e) {
            LOGGER.error("Exception while notifying patient of frozen {} account for patient {}", freeze, patient.getId(), e);
        }
    }

    private void sendNotification(PKBPerson patient, Team actingTeam, boolean freeze)
            throws NumberFormatException {

        // TODO: new templates?
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public void setPkbEmailMessageManager(PKBEmailMessageManager pkbEmailMessageManager) {
        this.pkbEmailMessageManager = pkbEmailMessageManager;
    }
}
