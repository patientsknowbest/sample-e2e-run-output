package com.pkb.patient.invite.bulk.csv;

import com.pkb.common.base.Error;
import com.pkb.common.csv.ColumnMapper;
import com.pkb.datamodel.Email;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.institute.entity.Org;
import com.pkb.patient.invite.AddressDto;
import com.pkb.patient.invite.InviteeDto;
import com.pkb.patient.invite.NameDto;

import java.time.LocalDate;
import java.util.List;
import java.util.Optional;

public class InviteeMapperFactory {
    private final InviteeIdsColumnsMapperFactory inviteeIdsMapperFactory;
    private final ColumnMapper<List<Error>, NameDto> nameMapper;
    private final ColumnMapper<List<Error>, Optional<Email>> emailMapper;
    private final ColumnMapper<List<Error>, Optional<LocalDate>> birthdayMapper;
    private final ColumnMapper<List<Error>, Optional<AddressDto>> addressMapper;

    public InviteeMapperFactory(
            ColumnMapper<List<Error>, NameDto> nameMapper,
            InviteeIdsColumnsMapperFactory inviteeIdsMapperFactory,
            ColumnMapper<List<Error>, Optional<Email>> emailMapper,
            ColumnMapper<List<Error>, Optional<LocalDate>> birthdayMapper,
            ColumnMapper<List<Error>, Optional<AddressDto>> addressMapper) {
        this.nameMapper = nameMapper;
        this.inviteeIdsMapperFactory = inviteeIdsMapperFactory;
        this.emailMapper = emailMapper;
        this.birthdayMapper = birthdayMapper;
        this.addressMapper = addressMapper;
    }

    public ColumnMapper<List<Error>, InviteeDto> create(NationalIdType nationalIdType, Org organization) {
        return new InviteeMapper(nameMapper, inviteeIdsMapperFactory.create(nationalIdType, organization), emailMapper, birthdayMapper, addressMapper);
    }
}
