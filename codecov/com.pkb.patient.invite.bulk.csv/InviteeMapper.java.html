package com.pkb.patient.invite.bulk.csv;

import com.pkb.common.base.Error;
import com.pkb.common.csv.ColumnMapper;
import com.pkb.common.csv.CsvRecord;
import com.pkb.datamodel.Email;
import com.pkb.patient.invite.AddressDto;
import com.pkb.patient.invite.InviteeDto;
import com.pkb.patient.invite.InviteeIdsDto;
import com.pkb.patient.invite.NameDto;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.time.LocalDate;
import java.util.List;
import java.util.Optional;

import static com.pkb.common.base.Eithers.combine;

public final class InviteeMapper implements ColumnMapper<List<Error>, InviteeDto> {
    private final ColumnMapper<List<Error>, NameDto> nameMapper;
    private final ColumnMapper<List<Error>, InviteeIdsDto> inviteeIdsMapper;
    private final ColumnMapper<List<Error>, Optional<Email>> emailMapper;
    private final ColumnMapper<List<Error>, Optional<LocalDate>> birthdayMapper;
    private final ColumnMapper<List<Error>, Optional<AddressDto>> addressMapper;

    public InviteeMapper(
            ColumnMapper<List<Error>, NameDto> nameMapper,
            ColumnMapper<List<Error>, InviteeIdsDto> inviteeIdsMapper,
            ColumnMapper<List<Error>, Optional<Email>> emailMapper,
            ColumnMapper<List<Error>, Optional<LocalDate>> birthdayMapper,
            ColumnMapper<List<Error>, Optional<AddressDto>> addressMapper) {
        this.nameMapper = nameMapper;
        this.inviteeIdsMapper = inviteeIdsMapper;
        this.emailMapper = emailMapper;
        this.birthdayMapper = birthdayMapper;
        this.addressMapper = addressMapper;
    }

    @NotNull
    @Override
    public Either<List<Error>, InviteeDto> map(@NotNull CsvRecord row) {
        Either<List<Error>, InviteeDto> invitee = combine(
                nameMapper.map(row),
                inviteeIdsMapper.map(row),
                InviteeDto::new
        );


        Either<List<Error>, InviteeDto> inviteeWithEmail = combine(
                invitee,
                emailMapper.map(row),
                (i, maybeEmail) -> maybeEmail.map(i::withEmail).orElse(i));

        Either<List<Error>, InviteeDto> inviteeWithDateOfBirth = combine(
                inviteeWithEmail,
                birthdayMapper.map(row),
                (i, maybeDateOfBirth) -> maybeDateOfBirth.map(i::withDateOfBirth).orElse(i));

        return combine(
                inviteeWithDateOfBirth,
                addressMapper.map(row),
                (i, maybeAddress) -> maybeAddress.map(i::withAddress).orElse(i));
    }
}
