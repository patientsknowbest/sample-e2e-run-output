package com.pkb.patient.invite.bulk.csv;

import com.pkb.common.base.Error;
import com.pkb.common.csv.ColumnMapper;
import com.pkb.common.csv.RowMapper;
import com.pkb.common.csv.RowValidator;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.institute.entity.Org;
import com.pkb.patient.invite.CarerDto;
import com.pkb.patient.invite.InvitationDto;

import java.util.List;
import java.util.Optional;

public class BulkInvitationRowMapperFactory {
    private RowValidator<Error> rowValidator;
    private InviteeMapperFactory inviteeMapperFactory;
    private ColumnMapper<List<Error>, Optional<CarerDto>> invitedCarerOneMapper;
    private ColumnMapper<List<Error>, Optional<CarerDto>> invitedCarerTwoMapper;

    public BulkInvitationRowMapperFactory(
            RowValidator<Error> rowValidator,
            InviteeMapperFactory inviteeMapperFactory,
            ColumnMapper<List<Error>, Optional<CarerDto>> invitedCarerOneMapper,
            ColumnMapper<List<Error>, Optional<CarerDto>> invitedCarerTwoMapper) {
        this.rowValidator = rowValidator;
        this.inviteeMapperFactory = inviteeMapperFactory;
        this.invitedCarerOneMapper = invitedCarerOneMapper;
        this.invitedCarerTwoMapper = invitedCarerTwoMapper;
    }

    public RowMapper<InvitationParsingError, InvitationDto> create(NationalIdType nationalIdType, Org organization) {
        return new InvitationDtoMapper(
                rowValidator,
                inviteeMapperFactory.create(nationalIdType, organization),
                invitedCarerOneMapper,
                invitedCarerTwoMapper
        );
    }
}
