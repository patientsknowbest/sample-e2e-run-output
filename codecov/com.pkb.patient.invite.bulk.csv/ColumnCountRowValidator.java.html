package com.pkb.patient.invite.bulk.csv;

import com.pkb.common.base.Error;
import com.pkb.common.csv.CsvRecord;
import com.pkb.common.csv.RowValidator;
import org.jetbrains.annotations.NotNull;

import java.util.ArrayList;
import java.util.List;

import static com.pkb.common.base.ErrorBuilder.error;

public final class ColumnCountRowValidator implements RowValidator<Error> {
    private static final String COLUMN_COUNT_ERROR_PATTERN_CODE = "patient.bulk.invite.column.count.mismatch";
    private static final String COLUMN_COUNT_ERROR_PATTERN = "Record %s: has %s fields instead of %s";

    private final int columnCount;

    public ColumnCountRowValidator(int columnCount) {
        this.columnCount = columnCount;
    }

    @NotNull @Override
    public List<Error> validate(@NotNull CsvRecord row) {
        List<Error> errorCollector = new ArrayList<>();

        if (row.columnCount() != columnCount) {
            errorCollector.add(columnCountMismatch(row, row.logicalLineNumber()));
        }

        return errorCollector;
    }

    private Error columnCountMismatch(CsvRecord row, long lineNumber) {
        return error(COLUMN_COUNT_ERROR_PATTERN_CODE, COLUMN_COUNT_ERROR_PATTERN).withArguments(lineNumber, row.columnCount(), columnCount).build();
    }
}
