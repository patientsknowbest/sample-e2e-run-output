package com.pkb.patient.invite.bulk.csv;

import com.pkb.common.base.Error;
import com.pkb.common.csv.ColumnMapper;
import com.pkb.common.csv.CsvRecord;
import com.pkb.datamodel.Email;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.List;
import java.util.Optional;

import static com.pkb.common.base.ErrorBuilder.errorOf;
import static com.pkb.datamodel.ImmutableDisclosingEmail.disclosingEmail;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_EMAIL;
import static java.util.Collections.singletonList;

public class EmailMapper implements ColumnMapper<List<Error>, Optional<Email>> {
    private static final String INVALID_EMAIL_CODE = "patient.bulk.invite.invalid.email.for.invitee";
    private static final String INVALID_EMAIL = "Invalid email '%s'";

    @NotNull
    @Override
    public Either<List<Error>, Optional<Email>> map(@NotNull CsvRecord row) {
        return row.findColumn(INVITEE_EMAIL)
                .map(this::validateEmail)
                .orElse(Either.right(Optional.empty()));
    }

    private Either<List<Error>, Optional<Email>> validateEmail(String email) {
        try {
            Email typedEmail = disclosingEmail(email);
            return Either.right(Optional.of(typedEmail));
        } catch (RuntimeException ignored) {
            return Either.left(singletonList(errorOf(INVALID_EMAIL_CODE, INVALID_EMAIL, email)));
        }
    }
}
