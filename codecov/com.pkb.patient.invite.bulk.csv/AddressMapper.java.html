package com.pkb.patient.invite.bulk.csv;

import com.pkb.common.base.Error;
import com.pkb.common.csv.ColumnMapper;
import com.pkb.common.csv.CsvRecord;
import com.pkb.patient.invite.AddressDto;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.List;
import java.util.Optional;

import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_ADDRESS_PART_ONE;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_ADDRESS_PART_TWO;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_CITY;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_COUNTY;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_POST_CODE;
import static java.util.Optional.empty;

public class AddressMapper implements ColumnMapper<List<Error>, Optional<AddressDto>> {
    @NotNull
    @Override
    public Either<List<Error>, Optional<AddressDto>> map(@NotNull CsvRecord row) {
        Either<List<Error>, Optional<AddressDto>> result;

        Optional<String> maybeAddressPartOne = row.findColumn(INVITEE_ADDRESS_PART_ONE);
        Optional<String> maybeAddressPartTwo = row.findColumn(INVITEE_ADDRESS_PART_TWO);
        Optional<String> maybeCity = row.findColumn(INVITEE_CITY);
        Optional<String> maybeCounty = row.findColumn(INVITEE_COUNTY);

        if (maybeAddressPartOne.isPresent() || maybeAddressPartTwo.isPresent() || maybeCity.isPresent() || maybeCounty.isPresent()) {
            AddressDto address = new AddressDto();
            maybeAddressPartOne.ifPresent(addressPartOne -> {
                address.setAddressPartOne(addressPartOne);
                row.findColumn(INVITEE_POST_CODE).ifPresent(address::setPostCode);
            });
            maybeAddressPartTwo.ifPresent(address::setAddressPartTwo);
            maybeCity.ifPresent(address::setCity);
            maybeCounty.ifPresent(address::setCounty);
            result = Either.right(Optional.of(address));
        } else {
            result = Either.right(empty());
        }

        return result;
    }
}
