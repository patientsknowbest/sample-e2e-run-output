package com.pkb.patient.invite.bulk.csv;

import com.google.common.collect.ImmutableList;
import com.google.common.collect.ImmutableList.Builder;
import com.pkb.common.base.Error;
import com.pkb.common.csv.ColumnMapper;
import com.pkb.common.csv.CsvRecord;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.entities.enums.ValidNationalId;
import com.pkb.exception.ApiCallMalformedException;
import com.pkb.institute.entity.Org;
import com.pkb.patient.invite.InviteeIdsDto;
import com.pkb.util.PKBServiceUtil;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.List;
import java.util.Optional;

import static com.pkb.common.base.ErrorBuilder.error;
import static com.pkb.common.base.ErrorBuilder.errorOf;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_NATIONAL_IDENTIFIER;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_ORGANIZATION_LEVEL_IDENTIFIER;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_TEAM_LEVEL_IDENTIFIER;
import static java.util.Collections.singletonList;

public final class InviteeIdsColumnsMapper implements ColumnMapper<List<Error>, InviteeIdsDto> {
    private static final String INVALID_NATIONAL_ID_CODE = "patient.bulk.invite.invalid.national.id";
    private static final String INVALID_NATIONAL_ID = "Invalid %s '%s'";

    private static final String MUST_HAVE_AT_LEAST_ONE_ID_CODE = "patient.bulk.invite.must.have.at.least.one.id";
    private static final String MUST_HAVE_AT_LEAST_ONE_ID = "Must have at least one type of id (national, org level or team level id)";

    private final NationalIdType nationalIdType;
    private final Org organization;

    public InviteeIdsColumnsMapper(NationalIdType nationalIdType, Org organization) {
        this.nationalIdType = nationalIdType;
        this.organization = organization;
    }

    @NotNull
    @Override
    public Either<List<Error>, InviteeIdsDto> map(@NotNull CsvRecord row) {
        Either<List<Error>, InviteeIdsDto> result;

        ImmutableList.Builder<Error> errorCollector = ImmutableList.builder();

        Optional<ValidNationalId> maybeNationalId = findNationalId(row, errorCollector);
        Optional<String> maybeOrganizationLevelId = row.findColumn(INVITEE_ORGANIZATION_LEVEL_IDENTIFIER);
        Optional<String> maybeTeamLevelId = row.findColumn(INVITEE_TEAM_LEVEL_IDENTIFIER);

        List<Error> errors = errorCollector.build();
        if (errors.isEmpty()) {
            if (maybeNationalId.isPresent() || maybeOrganizationLevelId.isPresent() || maybeTeamLevelId.isPresent()) {
                result = Either.right(new InviteeIdsDto(maybeNationalId.orElse(null), maybeTeamLevelId.orElse(null), maybeOrganizationLevelId.orElse(null)));
            } else {
                result = Either.left(singletonList(errorOf(MUST_HAVE_AT_LEAST_ONE_ID_CODE, MUST_HAVE_AT_LEAST_ONE_ID)));
            }
        } else {
            result = Either.left(errors);
        }

        return result;
    }

    private Optional<ValidNationalId> findNationalId(CsvRecord row, Builder<Error> errorCollector) {
        return row.findColumn(INVITEE_NATIONAL_IDENTIFIER)
                .flatMap(nid -> {
                    Optional<ValidNationalId> result = nationalIdType.getValidNationalIdAndType(nid);
                    if (result.isEmpty()) {
                        errorCollector.add(invalidNationalId(row));
                    }

                    try {
                        String normalizedNationalId = result.map(ValidNationalId::value).orElse(nationalIdType.cleanInput(nid));
                        PKBServiceUtil.checkSandboxNationalID(organization, nationalIdType, normalizedNationalId);
                    } catch (ApiCallMalformedException ignored) {
                        errorCollector.add(invalidNationalId(row));
                    }

                    return result;
                });
    }

    private Error invalidNationalId(CsvRecord row) {
        return error(INVALID_NATIONAL_ID_CODE, INVALID_NATIONAL_ID)
                .withArguments(nationalIdType.getName(), row.getColumn(INVITEE_NATIONAL_IDENTIFIER))
                .build();
    }
}
