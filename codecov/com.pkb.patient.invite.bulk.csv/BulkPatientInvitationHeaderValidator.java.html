package com.pkb.patient.invite.bulk.csv;

import com.pkb.common.base.Error;
import com.pkb.common.csv.HeaderValidator;

import java.util.ArrayList;
import java.util.List;
import java.util.Set;

import static com.google.common.collect.Sets.symmetricDifference;
import static com.pkb.common.base.ErrorBuilder.error;
import static com.pkb.common.base.ErrorBuilder.errorOf;

public class BulkPatientInvitationHeaderValidator implements HeaderValidator {
    private static final String INCORRECT_HEADER_COUNT_CODE = "patient.bulk.invite.invalid.header.column.count";
    private static final String INCORRECT_HEADER_COUNT = "The header row does not have %s columns. Is this a proper CSV file?";
    private static final String INCORRECT_HEADER_CODE = "patient.bulk.invite.invalid.header";
    private static final String INCORRECT_HEADER = "This file does not begin with the header from the template. Is this a proper CSV file?";

    private final Set<String> expectedColumnNames;

    public BulkPatientInvitationHeaderValidator(Set<String> expectedColumnNames) {
        this.expectedColumnNames = expectedColumnNames;
    }

    @Override public List<Error> validateHeader(Set<String> actualColumnHeaders) {
        List<Error> errorCollector = new ArrayList<>();

        int expectedColumnCount = expectedColumnNames.size();
        if (actualColumnHeaders.size() == expectedColumnCount) {
            if (!symmetricDifference(expectedColumnNames, actualColumnHeaders).isEmpty()) {
                errorCollector.add(errorOf(INCORRECT_HEADER_CODE, INCORRECT_HEADER));
            }
        } else {
            errorCollector.add(error(INCORRECT_HEADER_COUNT_CODE, INCORRECT_HEADER_COUNT).withArguments(expectedColumnCount).build());
        }

        return errorCollector;
    }
}
