package com.pkb.patient.invite.bulk.csv;

import com.pkb.app.entity.LoggedInEHRRequestContext;
import com.pkb.common.csv.RowMapper;
import com.pkb.entities.enums.NationalIdType;
import com.pkb.institute.entity.Org;
import com.pkb.institute.entity.Team;
import com.pkb.patient.InvitationNationalIdValidator;
import com.pkb.patient.invite.InvitationDto;
import com.pkb.patient.invite.PatientRegister;
import com.pkb.patient.invite.RequestBoundPatientRegisterFactory;
import com.pkb.user.entity.PKBPerson;
import org.jetbrains.annotations.NotNull;

public class BulkInvitationCsvProcessorFactory {
    private final BulkInvitationRowMapperFactory rowMapperFactory;
    private final RequestBoundPatientRegisterFactory patientRegisterFactory;
    private final InvitationResultConverter invitationConverter;

    public BulkInvitationCsvProcessorFactory(
            BulkInvitationRowMapperFactory rowMapperFactory,
            RequestBoundPatientRegisterFactory patientRegisterFactory,
            InvitationResultConverter invitationConverter) {
        this.rowMapperFactory = rowMapperFactory;
        this.patientRegisterFactory = patientRegisterFactory;
        this.invitationConverter = invitationConverter;
    }

    public BulkInvitationCsvProcessor create(
            @NotNull LoggedInEHRRequestContext requestContext,
            @NotNull PKBPerson inviter,
            @NotNull InvitationNationalIdValidator invitationNationalIdValidator,
            @NotNull Team invitingTeam) {

        NationalIdType nationalIdType = NationalIdType.getPrimaryNationalIdTypeFromCountryCode(invitingTeam.getCountry()).orElseThrow();
        Org invitingOrganization = invitingTeam.getOrg();
        RowMapper<InvitationParsingError, InvitationDto> rowMapper = rowMapperFactory.create(nationalIdType, invitingOrganization);
        PatientRegister patientRegister = patientRegisterFactory.create(
                inviter,
                invitingTeam,
                nationalIdType,
                invitationNationalIdValidator,
                requestContext);

        return new BulkInvitationCsvProcessor(rowMapper, invitationConverter, patientRegister);
    }
}
