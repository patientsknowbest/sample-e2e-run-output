package com.pkb.patient.invite.bulk.csv;

import com.google.common.base.Joiner;
import com.google.common.collect.Iterators;
import com.pkb.common.csv.CsvRecord;
import com.pkb.common.csv.CsvRecordConverter;
import io.vavr.collection.List;
import org.jetbrains.annotations.NotNull;

import static com.google.common.collect.Iterators.singletonIterator;

public class CsvContentConverter {
    private final CsvRecordConverter csvRecordConverter;
    private final String lineSeparator;

    public CsvContentConverter(@NotNull CsvRecordConverter csvRecordConverter) {
        this("\n", csvRecordConverter);
    }

    public CsvContentConverter(@NotNull String lineSeparator, @NotNull CsvRecordConverter csvRecordConverter) {
        this.lineSeparator = lineSeparator;
        this.csvRecordConverter = csvRecordConverter;
    }

    public String convert(List<CsvRecord> enrichedLines) {
        StringBuilder csvContent = new StringBuilder();

        return Joiner.on(lineSeparator)
                .appendTo(csvContent, Iterators.concat(singletonIterator(csvRecordConverter.getHeaderRow()), enrichedLines.map(csvRecordConverter::convertRecord).iterator()))
                .toString();

    }
}
