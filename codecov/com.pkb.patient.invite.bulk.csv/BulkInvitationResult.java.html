package com.pkb.patient.invite.bulk.csv;

import com.pkb.patient.invite.bulk.LineNumberedInvitationResult;
import io.vavr.collection.List;
import org.jetbrains.annotations.NotNull;

public class BulkInvitationResult {
    private int successCount;
    private int failureCount;
    private List<LineNumberedInvitationResult> results;

    public BulkInvitationResult(int successCount, int failureCount, @NotNull List<LineNumberedInvitationResult> results) {
        this.successCount = successCount;
        this.failureCount = failureCount;
        this.results = results;
    }

    public int getSuccessCount() {
        return successCount;
    }

    public int getFailureCount() {
        return failureCount;
    }

    public List<LineNumberedInvitationResult> getResults() {
        return results;
    }
}
