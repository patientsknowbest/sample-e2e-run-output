package com.pkb.patient.invite.bulk.csv;

import com.google.common.collect.ImmutableList;
import com.pkb.common.base.Error;

import java.util.List;
import java.util.Optional;

import static java.util.Optional.ofNullable;

public class InvitationParsingError {
    private final long lineNumber;
    private List<Error> patientParsingErrors;
    private List<Error> carerOneParsingErrors;
    private List<Error> carerTwoParsingErrors;

    public static InvitationParsingError patientParsing(long lineNumber, Error error) {
        return patientParsing(lineNumber, ImmutableList.of(error));
    }

    public static InvitationParsingError patientParsing(long lineNumber, List<Error> errors) {
        return new InvitationParsingError(lineNumber, errors, null, null);
    }

    public InvitationParsingError(long lineNumber, List<Error> patientParsingErrors, List<Error> carerOneParsingErrors, List<Error> carerTwoParsingErrors) {
        this.lineNumber = lineNumber;
        this.patientParsingErrors = patientParsingErrors;
        this.carerOneParsingErrors = carerOneParsingErrors;
        this.carerTwoParsingErrors = carerTwoParsingErrors;
    }

    public long getLineNumber() {
        return lineNumber;
    }

    public Optional<List<Error>> findPatientParsingErrors() {
        return ofNullable(patientParsingErrors);
    }

    public Optional<List<Error>> findCarerOneParsingErrors() {
        return ofNullable(carerOneParsingErrors);
    }

    public Optional<List<Error>> findCarerTwoParsingErrors() {
        return ofNullable(carerTwoParsingErrors);
    }
}
