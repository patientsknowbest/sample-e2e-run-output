package com.pkb.patient.invite.bulk.csv;

import com.google.common.collect.ImmutableMap;
import com.pkb.common.csv.CsvRecord;
import com.pkb.common.csv.ImmutableCsvRecord;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.BulkInvitationOutcome;
import com.pkb.datamodel.consent.PatientConsentUpdateTemplate;
import com.pkb.institute.entity.Team;
import com.pkb.patient.invite.bulk.LineNumberedInvitationResult;
import com.pkb.service.emailmessage.impl.PKBEmailMessageManager;
import com.pkb.service.user.BulkInvitationOutcomesManager;
import com.pkb.user.entity.PKBPerson;
import io.vavr.collection.Iterator;
import io.vavr.collection.List;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.enterprise.concurrent.ManagedExecutorService;
import javax.naming.InitialContext;
import java.lang.invoke.MethodHandles;
import java.time.LocalDateTime;
import java.util.Map;
import java.util.concurrent.Callable;
import java.util.concurrent.Future;
import java.util.function.BinaryOperator;

public class BulkInvitationFacade {
    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    public interface ParallelismStrategy {
        BulkInvitationResult processCsvRows(io.vavr.collection.List<CsvRecord> originalRows, BulkInvitationCsvProcessor csvProcessor, PatientConsentUpdateTemplate consent);

    }

    public static final ParallelismStrategy SYNCHRONOUS_EXECUTION_STRATEGY = (originalRows, csvProcessor, patientConsent) -> csvProcessor
            .processRows(originalRows, patientConsent);

    public static ParallelismStrategy parallelStrategy() {
        return new ParallelismStrategy() {

            private static final int PARALLELISM = 16;

            BinaryOperator<BulkInvitationResult> reducer = (r1, r2) -> new BulkInvitationResult(r1.getSuccessCount() + r2.getSuccessCount(),
                    r1.getFailureCount() + r2.getFailureCount(),
                    r1.getResults().appendAll(r2.getResults()));

            private ManagedExecutorService getManagedExecutorService() {
                try {
                    javax.naming.InitialContext ctx = new InitialContext();
                    ManagedExecutorService mes = (ManagedExecutorService)
                            ctx.lookup("java:jboss/ee/concurrency/executor/default");
                    return mes;
                } catch (Exception e) {
                    throw new RuntimeException(e);
                }
            }
            
            private Callable<BulkInvitationResult> task(List<CsvRecord> rows, BulkInvitationCsvProcessor csvProcessor, PatientConsentUpdateTemplate patientConsent){
                return () -> csvProcessor.processRows(rows, patientConsent);
            }
            
            @Override
            public BulkInvitationResult processCsvRows(List<CsvRecord> originalRows, BulkInvitationCsvProcessor csvProcessor, PatientConsentUpdateTemplate patientConsent) {
                ManagedExecutorService executorService = getManagedExecutorService();
                try {
                    Iterator<io.vavr.collection.List<CsvRecord>> rowGroups = originalRows.grouped(1 + originalRows.size() / PARALLELISM);
                    java.util.List<Callable<BulkInvitationResult>> tasks = rowGroups.map(rows -> task(rows, csvProcessor, patientConsent)).toJavaList();

                    java.util.List<Future<BulkInvitationResult>> result = executorService.invokeAll(tasks);

                    return result.stream().map((future) -> {
                        try {
                            return future.get();
                        } catch (Exception e) {
                            throw new RuntimeException(e);
                        }
                    }).reduce(new BulkInvitationResult(0, 0, List.empty()),
                            reducer);

                } catch (Exception e) {
                    throw new RuntimeException(e);
                }
            }
        };
    }

    private final DateTimeService dateTimeService;
    private final BulkInvitationOutcomesManager outcomesManager;
    private final BulkInvitationCsvProcessor csvProcessor;
    private final BulkInvitationResultProcessor resultProcessor;
    private final CsvContentConverter csvContentConverter;
    private final PKBEmailMessageManager pkbEmailMessageManager;
    private static final String MESSAGE_PATTERN = "Bulk Patient Invitation (Coordinator id: '{}', Team id: '{}') {}, because:";

    public BulkInvitationFacade(
            PKBEmailMessageManager pkbEmailMessageManager,
            @NotNull BulkInvitationResultProcessor resultProcessor,
            @NotNull CsvContentConverter csvContentConverter,
            @NotNull BulkInvitationOutcomesManager outcomesManager,
            @NotNull DateTimeService dateTimeService,
            @NotNull BulkInvitationCsvProcessor csvProcessor) {
        this.pkbEmailMessageManager = pkbEmailMessageManager;
        this.resultProcessor = resultProcessor;
        this.csvContentConverter = csvContentConverter;
        this.outcomesManager = outcomesManager;
        this.dateTimeService = dateTimeService;
        this.csvProcessor = csvProcessor;
    }

    public void invitePatientsInBulk(PKBPerson initiator, Team invitingTeam, List<CsvRecord> csvContent,
            @NotNull ParallelismStrategy parallelismStrategy, PatientConsentUpdateTemplate patientConsent,
            long actorId) {
        LocalDateTime initiatedAt = dateTimeService.nowLocalDateTime();

        boolean couldProcess = false;
        try {
            BulkInvitationOutcome outcome = outcomesManager.registerBulkInvitationStart(initiator, invitingTeam, initiatedAt);
            BulkInvitationResult result = parallelismStrategy.processCsvRows(csvContent, csvProcessor, patientConsent);

            String csvEnrichedWithOutcomes = csvContentConverter
                    .convert(enrichOriginalRowsWithOutcomes(csvContent, result.getResults()));

            outcomesManager.markAsFinished(outcome, csvEnrichedWithOutcomes, result.getSuccessCount(), result.getFailureCount());

            couldProcess = true;
        } catch (Exception cause) {
            // TODO PKB-1741: should we generate an incident number?
            LOGGER.error(MESSAGE_PATTERN, initiator.getId(), invitingTeam.getId(), "has failed", cause);
            couldProcess = markOutcomesAsFailedSilently(initiator, invitingTeam, initiatedAt);
        } finally {
            if (couldProcess) {
                pkbEmailMessageManager.notifyCoordinatorBulkAccountCreationCompleted(initiator, invitingTeam, actorId);
            } else {
                pkbEmailMessageManager.notifyCoordinatorBulkAccountCreationFailed(initiator, invitingTeam, actorId);
            }
        }
    }

    private boolean markOutcomesAsFailedSilently(PKBPerson initiator, Team invitingTeam, LocalDateTime uploadedAt) {
        boolean couldMarkAsFailed = true;

        try {
            outcomesManager.markAsFailed(initiator, invitingTeam, uploadedAt);
        } catch (Exception cause) {
            LOGGER.error(MESSAGE_PATTERN, initiator.getId(), invitingTeam.getId(), "failed to mark outcomes as failed", cause);
            couldMarkAsFailed = false;
        }
        return couldMarkAsFailed;
    }

    private List<CsvRecord> enrichOriginalRowsWithOutcomes(List<CsvRecord> originalRows, List<LineNumberedInvitationResult> results) {
        return originalRows.zipWith(results,
                (originalRow, invitationResult) -> {
                    Map<String, String> process = resultProcessor.process(originalRow, invitationResult);

                    ImmutableMap<String, String> columns = ImmutableMap.<String, String> builder()
                            .putAll(originalRow.toMap())
                            .putAll(process)
                            .build();

                    return new ImmutableCsvRecord(originalRow.logicalLineNumber(), columns);
                }).collect(List.collector());
    }
}
