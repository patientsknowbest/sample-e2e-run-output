package com.pkb.patient.invite.bulk.csv;

import com.pkb.common.base.Error;
import com.pkb.common.csv.ColumnMapper;
import com.pkb.common.csv.CsvRecord;
import com.pkb.patient.invite.NameDto;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.List;

import static com.pkb.common.base.Eithers.combine;
import static com.pkb.common.base.ErrorBuilder.error;
import static com.pkb.patient.invite.NameDto.name;
import static com.pkb.patient.invite.NameDto.nameWithTitle;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_FIRST_NAME;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_LAST_NAME;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_TITLE;
import static java.util.Collections.singletonList;

public class NameDtoMapper implements ColumnMapper<List<Error>, NameDto> {
    private static final String RECORD_MUST_HAVE_CODE = "patient.bulk.invite.missing.value.for.mandatory.column";
    private static final String RECORD_MUST_HAVE = "Missing mandatory column: '%s'";

    @NotNull
    @Override
    public Either<List<Error>, NameDto> map(@NotNull CsvRecord row) {
        Either<List<Error>, String> validatedFirstName = row.findColumn(INVITEE_FIRST_NAME)
                .map(Either::<List<Error>, String>right)
                .orElse(Either.left(singletonList(
                        error(RECORD_MUST_HAVE_CODE, RECORD_MUST_HAVE).withArguments(INVITEE_FIRST_NAME).build())));


        Either<List<Error>, String> validatedLastName =
                row.findColumn(INVITEE_LAST_NAME)
                        .map(Either::<List<Error>, String>right)
                        .orElse(Either.left(singletonList(
                                error(RECORD_MUST_HAVE_CODE, RECORD_MUST_HAVE).withArguments(INVITEE_LAST_NAME).build())));

        return combine(validatedFirstName, validatedLastName, (firstName, lastName) ->
                row.findColumn(INVITEE_TITLE)
                        .map(title -> nameWithTitle(title, validatedFirstName.get(), validatedLastName.get()))
                        .orElseGet(() -> name(validatedFirstName.get(), validatedLastName.get())));
    }

}
