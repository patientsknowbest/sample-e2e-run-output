package com.pkb.patient.invite.bulk.csv;

import com.pkb.common.base.Error;
import com.pkb.common.csv.ColumnMapper;
import com.pkb.common.csv.CsvRecord;
import com.pkb.datamodel.Email;
import com.pkb.patient.invite.CarerDto;
import io.vavr.control.Either;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;

import java.util.List;
import java.util.Optional;
import java.util.function.Function;

import static com.pkb.common.base.ErrorBuilder.error;
import static com.pkb.datamodel.ImmutableDisclosingEmail.disclosingEmail;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_FIRST_NAME;
import static com.pkb.patient.invite.bulk.csv.BulkInvitationConstants.INVITEE_LAST_NAME;
import static java.util.Collections.singletonList;
import static java.util.Optional.empty;

public final class InvitedCarerMapper implements ColumnMapper<List<Error>, Optional<CarerDto>> {
    private static final String INVALID_EMAIL_CODE = "patient.bulk.invite.invalid.email";
    private static final String INVALID_EMAIL = "Record %s (%s %s): invalid email: '%s'";

    private final String firstNameColumnName;
    private final String lastNameColumnName;
    private final String emailColumnName;

    public InvitedCarerMapper(String firstNameColumnName, String lastNameColumnName, String emailColumnName) {
        this.firstNameColumnName = firstNameColumnName;
        this.lastNameColumnName = lastNameColumnName;
        this.emailColumnName = emailColumnName;
    }

    @NotNull
    @Override
    public Either<List<Error>, Optional<CarerDto>> map(@NotNull CsvRecord row) {
        return row.findColumn(emailColumnName)
                .map(StringUtils::trimToEmpty)
                .map(StringUtils::lowerCase)
                .filter(StringUtils::isNoneBlank)
                .map(validateEmail(row))
                .orElse(Either.right(empty()));
    }

    private Function<String, Either<List<Error>, Optional<CarerDto>>> validateEmail(CsvRecord row) {
        return email -> {
            try {
                return Either.right(createInvitedCarer(disclosingEmail(email), row.getColumn(firstNameColumnName), row.getColumn(lastNameColumnName)));
            } catch (RuntimeException ignored) {
                return Either.left(singletonList(invalidEmail(row, email)));
            }
        };
    }

    private Optional<CarerDto> createInvitedCarer(Email email, String firstName, String lastName) {
        return Optional.of(new CarerDto(firstName, lastName, email));
    }

    private Error invalidEmail(CsvRecord row, String email) {
        return error(INVALID_EMAIL_CODE, INVALID_EMAIL)
                .withArguments(row.logicalLineNumber(), row.getColumn(INVITEE_FIRST_NAME), row.getColumn(INVITEE_LAST_NAME), email)
                .build();
    }
}
