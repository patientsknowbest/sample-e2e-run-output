package com.pkb.patient.invite.bulk.csv;

import com.pkb.common.base.Error;
import com.pkb.common.csv.ColumnMapper;
import com.pkb.common.csv.CsvRecord;
import com.pkb.common.csv.RowMapper;
import com.pkb.common.csv.RowValidator;
import com.pkb.patient.invite.CarerDto;
import com.pkb.patient.invite.InvitationDto;
import com.pkb.patient.invite.InviteeDto;
import io.vavr.control.Either;
import org.jetbrains.annotations.NotNull;

import java.util.List;
import java.util.Optional;

import static com.pkb.common.base.ErrorBuilder.errorOf;
import static com.pkb.patient.invite.bulk.csv.InvitationParsingError.patientParsing;

public final class InvitationDtoMapper implements RowMapper<InvitationParsingError, InvitationDto> {
    private static final String FAILED_DUE_TO_UNEXPECTED_ERROR_CODE = "patient.bulk.invite.failed.due.to.unexpected.error";
    private static final String FAILED_DUE_TO_UNEXPECTED_ERROR = "failed due to unexpected error %s";

    private final RowValidator<Error> rowValidator;
    private final ColumnMapper<List<Error>, InviteeDto> inviteeMapper;
    private final ColumnMapper<List<Error>, Optional<CarerDto>> invitedCarerOneMapper;
    private final ColumnMapper<List<Error>, Optional<CarerDto>> invitedCarerTwoMapper;

    public InvitationDtoMapper(
            @NotNull RowValidator<Error> rowValidator,
            @NotNull ColumnMapper<List<Error>, InviteeDto> inviteeMapper,
            @NotNull ColumnMapper<List<Error>, Optional<CarerDto>> invitedCarerOneMapper,
            @NotNull ColumnMapper<List<Error>, Optional<CarerDto>> invitedCarerTwoMapper) {
        this.rowValidator = rowValidator;
        this.inviteeMapper = inviteeMapper;
        this.invitedCarerOneMapper = invitedCarerOneMapper;
        this.invitedCarerTwoMapper = invitedCarerTwoMapper;
    }

    @Override
    public @NotNull Either<InvitationParsingError, InvitationDto> map(@NotNull CsvRecord row) {
        Either<InvitationParsingError, InvitationDto> result;

        try {
            result = attemptToMapRow(row);
        } catch (Exception cause) {
            result = Either.left(patientParsing(row.logicalLineNumber(), errorOf(FAILED_DUE_TO_UNEXPECTED_ERROR_CODE, FAILED_DUE_TO_UNEXPECTED_ERROR, cause.getMessage())));
        }

        return result;
    }

    // ========== Helper methods below. ==========

    private Either<InvitationParsingError, InvitationDto> attemptToMapRow(CsvRecord row) {
        List<Error> rowValidationErrors = rowValidator.validate(row);
        return rowValidationErrors.isEmpty() ?
                processRow(row) :
                Either.left(patientParsing(row.logicalLineNumber(), rowValidationErrors));
    }

    private Either<InvitationParsingError, InvitationDto> processRow(CsvRecord row) {
        Either<List<Error>, InviteeDto> parsedInvitee = inviteeMapper.map(row);
        Either<List<Error>, Optional<CarerDto>> parsedCarerOne = invitedCarerOneMapper.map(row);
        Either<List<Error>, Optional<CarerDto>> parsedCarerTwo = invitedCarerTwoMapper.map(row);

        return parsedInvitee.flatMap(
                invitee -> parsedCarerOne.flatMap(
                        maybeCarerOne -> parsedCarerTwo.map(
                                maybeCarerTwo -> {
                                    InvitationDto invitation = new InvitationDto(invitee);
                                    maybeCarerOne.ifPresent(invitation::setInvitedCarerOne);
                                    maybeCarerTwo.ifPresent(invitation::setInvitedCarerTwo);
                                    return invitation;
                                }
                        )
                ))
                .mapLeft(e -> new InvitationParsingError(
                        row.logicalLineNumber(),
                        parsedInvitee.isLeft() ? parsedInvitee.getLeft() : null,
                        parsedCarerOne.isLeft() ? parsedCarerOne.getLeft() : null,
                        parsedCarerTwo.isLeft() ? parsedCarerTwo.getLeft() : null));
    }

}
