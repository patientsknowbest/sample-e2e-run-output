/**
 * Used internally by the FHIR application to force immediate synchronization of any data that hasn't been
 * uploaded yet.
 * <p>
 * Note this is a bridge from new to old code, which is unusual and you probably shouldn't copy this.
 */
public class SynchUploadedDataAction extends BaseAction {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    @Autowired
    private TeamManager teamManager;
    @Autowired
    private TestManager testManager;
    @Autowired
    private RadiologyManager radiologyManager;
    @Autowired
    private CalendarManager calendarManager;
    @Autowired
    private EncounterManager encounterManager;
    @Autowired
    private MedicationManager medicationManager;
    @Autowired
    private DiagnosisManager diagnosisManager;
    public String synchUploadedData() {
        return Try(() -> {
            ImmutableSynchUploadedDataDTO dto = deserialize(request.getInputStream());
            LoggedInEHRRequestContext requestContext =
                    dto.isBgTask() ?
                            buildLoggedInBgTaskEhrRequestContext(dto.authenticatedUserId(), dto.userPrivateKey()) :
                            buildLoggedInUserEhrRequestContext(dto.authenticatedUserId(), dto.teamId().getOrElse((Long) null), dto.userPrivateKey());

            SynchableManager synchableManager = Match(dto.synchUploadType()).of(
                    Case($(RADIOLOGY), radiologyManager),
                    Case($(TEST_RESULT), testManager),
                    Case($(APPOINTMENT), calendarManager),
                    Case($(ENCOUNTER), encounterManager),
                    Case($(DOCUMENT), encounterManager),
                    Case($(MEDICATION_UPDATE), medicationManager),
                    Case($(DIAGNOSIS_UPDATE), diagnosisManager));
            synchableManager.syncUploadedData(requestContext, dto.targetPersonIds().asJava());
            return Action.SUCCESS;
        }).getOrElseGet(t -> {
            LOGGER.error("Error synchronizing uploaded data", t);
            return Action.ERROR;
        });
    }

    //TODO Temporary Path: remove when uploaded data phase 2 is implemented (OMNI-63)
    private LoggedInEHRRequestContext buildLoggedInBgTaskEhrRequestContext(Long authenticatedUserId, @NotNull UserPrivateKey fakeUserPrivateKey) {
        PKBPerson authenticatedPerson = userManager.getPKBPerson(authenticatedUserId);
        return ImmutableLoggedInBgTaskEHRRequestContext.builder()
                .correlationId(UUID.randomUUID())
                .accountLinkType(EHRRequestContext.AccountLinkType.ORG)
                .accessRoute(Route.SCHED_TASK)
                .consentStatus(noConsentRequired())
                .accessingUser(PersonAuthInfo.of(authenticatedPerson))
                .piggyback(true).build();
    }

    private LoggedInEHRRequestContext buildLoggedInUserEhrRequestContext(Long authenticatedUserId, @Nullable Long teamId, @NotNull UserPrivateKey userPrivateKey) {
        PKBPerson authenticatedPerson = userManager.getPKBPerson(authenticatedUserId);

        EHRRequestContext.AccountLinkType linkType = teamId != null ? EHRRequestContext.AccountLinkType.ORG : EHRRequestContext.AccountLinkType.INDIVIDUAL;

        Builder builder = ImmutableLoggedInUserEHRRequestContext.builder()
                .correlationId(UUID.randomUUID())
                .accountLinkType(linkType)
                .accessRoute(Route.WEBAPP)
                .consentStatus(noConsentRequired())
                .accessingUser(PersonAuthInfo.of(authenticatedPerson));


        Option.of(teamId)
                .peek(builder::teamId)
                .map(teamManager::getTeam)
                .map(Team::getOrg)
                .peek(builder::org);
        // Use piggyback mode as it bypasses any access control.
        // We're not retrieving any data, just processing outstanding stuff now that a key is available.
        return builder.piggyback(true).build();
    }
}
