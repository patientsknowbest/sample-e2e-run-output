package com.pkb.action.dataupload;

import com.pkb.action.BaseAction;
import com.pkb.service.dataupload.hl7.HL7ParsingManager;
import org.slf4j.Logger;
import org.springframework.beans.factory.annotation.Autowired;

import static io.vavr.control.Either.left;
import static io.vavr.control.Either.right;
import static java.lang.invoke.MethodHandles.lookup;
import static org.slf4j.LoggerFactory.getLogger;
import static org.springframework.security.crypto.bcrypt.BCrypt.checkpw;

/**
 * Checks login details for HL7 users
 */
public class Hl7AuthenticateAction extends BaseAction {
    private static final Logger LOGGER = getLogger(lookup().lookupClass());

    @Autowired
    private HL7ParsingManager hl7ParsingManager;

    private String username;
    private String password;

    public String authenticate() {
        return hl7ParsingManager.lookupOrgOrTeamOrPartnerFromUsername(username)
                .flatMap(orgTeamOrPartner -> orgTeamOrPartner.getHl7PasswordHash().toEither("no password set"))
                .flatMap(orgTeamPasswordHash -> checkpw(password, orgTeamPasswordHash) ? right("success") : left("password is incorrect"))
                .peek(success -> LOGGER.info("Hl7 authentication check for {} success", username))
                .peekLeft(error -> LOGGER.info("HL7 authentication check for {} error: {}", username, error))
                .getOrElseGet(error -> "error");
    }

    public String getUsername() {
        return username;
    }

    public void setUsername(String username) {
        this.username = username;
    }

    public String getPassword() {
        return password;
    }

    public void setPassword(String password) {
        this.password = password;
    }
}
