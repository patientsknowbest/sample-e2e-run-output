package com.pkb.action.registration;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.dto.OutOfAreaActivationNotificationDTO;
import com.pkb.exception.PatientNotFoundException;
import com.pkb.patient.PatientActivationManager;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import java.io.IOException;
import java.lang.invoke.MethodHandles;

import static io.vavr.API.Try;

public class OutOfAreaDeferredActivationNotificationAction extends BaseAction {

    private static final Logger LOGGER = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

    @Autowired
    private PatientActivationManager patientActivationManager;

    @Override
    public String execute() {
        return Try(this::processRequest)
                .recover(PatientNotFoundException.class, ex -> {
                    LOGGER.warn("Deferred activation for non-existent patient", ex);
                    return "NOT_FOUND";
                })
                .getOrElseGet(throwable -> {
                    LOGGER.error("Error sending out-of-area deferred activation notification", throwable);
                    return ERROR;
                });
    }

    private String processRequest() throws IOException {
        OutOfAreaActivationNotificationDTO dto = OBJECT_MAPPER.readValue(request.getInputStream(), OutOfAreaActivationNotificationDTO.class);
        patientActivationManager.notifyOutOfAreaPatientAccountActivation(dto.getPatientPublicId());
        return Action.SUCCESS;
    }
}
