    /**
     * API is used to load patient registration form
     * If the registration link contains UID then prefill the email address of that user and make it non editable
     * The link will contain the uid when the patient registered using the invitation link
     * else load the form without prefilling the email address
     */
    @Override
    public String input() throws SetContextNotAuthorisedException {
        LOGGER.info("Entering input");

        throwIfAnyUserLoggedIn();

        String result = SUCCESS;
        if (uid == null) {
            selfRegistered = true;

            team = null;

            if (StringUtils.isBlank(icode)) {
                return MISSING_ICODE;
            } else {
                try {
                    team = teamManager.getTeamByCode(icode);
                    if (team != null) {
                        instituteName = team.getName();
                        instituteCountry = team.getCountry();
                        team.setLogoFileName(getConfig().getInstituteLogoBaseDir() + team.getLogoFileName());
                    } else {
                        addActionMessage(getText("patientPersonalInfoAction.err.invalid_institute_code"));
                        return ERROR;
                    }
                } catch (PKBException e) {
                    LOGGER.error("failed getting Team for icode {}", icode, e);
                }
            }
            if (user == null) {
                user = new PKBPersonDTO();
            }

            if (team != null) {
                user.setTimeZoneId(team.getTimeZoneId());
                user.setCountry(instituteCountry);

            } else {
                user.setTimeZoneId(PKBConstants.DEFAULT_TIMEZONE_ID);
            }
        } else {
            selfRegistered = false;
            result = inputForInvitedPatient();
        }
        return result;
    }

    private String inputForInvitedPatient() {
        try {
            invitedPatient = userManager.getPKBPerson(uid, PKBPerson.Lazy.CONTACTS);
            if (invitedPatient == null) {
                // probably deleted after failed registration problems...
                addActionError(getText("patientPersonalInfoAction.err.invalid_link"));
                return ERROR;
            }
            if (invitedPatient.getStatus() == UserStatus.EMAIL_CONFIRMED) {
                addActionMessage(getText("baseRegisterAction.err.already_completed_registration", new String[] { config.getSupportNewTicketUrl() }));
                return LOGIN;
            }

            team = null;

            if (StringUtils.isNotBlank(icode)) {
                team = teamManager.getTeamByCode(icode);
                if (team != null) {
                    instituteName = team.getName();
                    instituteCountry = team.getCountry();
                    team.setLogoFileName(getConfig().getInstituteLogoBaseDir() + team.getLogoFileName());
                } else {
                    addActionError(getText("patientPersonalInfoAction.err.invalid_institute_code"));
                    return ERROR;
                }
            } else {
                instituteName = null;
            }

            if (user == null) {
                user = new PKBPersonDTO();
            }
            user.setId(uid);

            // verify the temp password -- past this step, we're safe showing info on this patient
            // because this is *not* just someone fishing with sequential ID numbers...

            // note that as long as we allow a missing temp password (for invitations w/o account) this is
            // still a security hole (so set minimal fields in that case!)
            if (StringUtils.isNotBlank(temp)) {
                PersonAuthInfo loggedUser = verifyCredentials(invitedPatient.getId(), temp);
                if (loggedUser == null) {
                    addActionError(getText("patientPersonalInfoAction.err.invalid_link"));
                    return ERROR;
                }

                // The previous screen should have validated the DOB. Check again here just in
                // case someone is attempting to jump directly into this action, without knowing the DOB
                Optional<LocalDate> maybeDob = invitedPatient.findDateOfBirth();
                if (maybeDob.isPresent()) {

                    String knownDay = pkbWebUtil.getWebDate(maybeDob.get());
                    String knownMonth = pkbWebUtil.getWebMonth(maybeDob.get());
                    String knownYear = pkbWebUtil.getWebYear(maybeDob.get());

                    if (knownDay.equals(getDay()) && knownMonth.equals(getMonth()) && knownYear.equals(getYear())) {
                        // All good
                    } else {
                        // No need to be too helpful with the error message, only misbehaving
                        // users will see this.
                        addActionError(getText("patientPersonalInfoAction.err.invalid_link"));
                        return ERROR;
                    }
                }

                // if clinician invited patient, we get: title, first, last
                // an HL7-created patient can have almost everything filled out
                populateDTOFromPerson(user, invitedPatient, team);

                Gender genderConst = Gender.getById(invitedPatient.getGender());
                gender = StringUtils.capitalize(genderConst.name().toLowerCase()); // i18n?
            } else {
                // no password provided -- this is a security hole!  keep it small
                user.setEmailId(invitedPatient.getEmail());
                user.setCountry(instituteCountry);
                user.setTimeZoneId(PKBConstants.DEFAULT_TIMEZONE_ID);
            }
            return SUCCESS;
        } catch (Exception e) {
            LOGGER.error("failed getting PKBPerson for uid {}", uid, e);
            addActionError(getText("patientPersonalInfoAction.err.invalid_link"));
            return ERROR;
        }
    }

    /**
     * On form validation error, the action method is not called but we
     * need some variables on the page that the action method populates.
     * <p>
     * We move some of this processing to the prepare method which is called
     * before the validation intercepter and so can take care of this.
     */
    @Override
    public void prepare() throws Exception {

        if (session.get("icode") != null) {
            // if we got it from the session then this is a jsp refresh on form validation error
            // or the person has navigated back and then stepped through the registration page again

            try {
                team = teamManager.getTeamByCode((String) session.get("icode"));
                if (team != null) {
                    // these are the page vars we need set on validation error also
                    instituteName = team.getName();
                    team.setLogoFileName(getConfig().getInstituteLogoBaseDir() + team.getLogoFileName());
                }
            } catch (PKBException e) {
                LOGGER.error("failed getting Team for icode {}", session.get("icode"), e);
            }
        }
    }

    /*
     * This API takes the patient's personal information as input -> Checks for the
     * given email id if exists sends an error
     */
    @Override
    public String execute() {
        String result = Action.ERROR;
        try {
            LOGGER.info("Entering-> processUserPersonalInfor");
            user.setUserName(user.getEmailId());
            // hard code user type: don't just accept what the JSP provides
            user.setUserType(UserType.PATIENT);
            Locale sessionLocale = (Locale) session.get(I18nInterceptor.DEFAULT_SESSION_ATTRIBUTE); // WW_TRANS_I18N_LOCALE
            user.setLanguageCode(sessionLocale != null ? sessionLocale.getLanguage() : "en");

            if (StringUtils.isBlank(user.getTimeZoneId())) {
                user.setTimeZoneId(PKBConstants.DEFAULT_TIMEZONE_ID);
            }

            if (uid != null) {
                invitedPatient = userManager.getPKBPerson(uid, Lazy.PROPERTIES, Lazy.CONTACTS, Lazy.NATIONAL_AND_LOCAL_IDS);
                user.setId(uid);
            } else if (userManager.pkbPersonUsernameExists(user.getUserName())) {
                // TODO: MULTIPLE-TEAMS - safe to assume we'll get a patient back at this point?
                invitedPatient = userManager.getSinglePersonByEmail(
                        user.getUserName(),
                        PersonContact.Lazy.PERSON_W_CONTACTS,
                        PersonContact.Lazy.PROPERTIES);
            }

            if (invitedPatient != null) {
                //Maintain the user type of the invited employee
                if (invitedPatient.getUserType() == UserType.EMPLOYEE) {
                    user.setUserType(UserType.EMPLOYEE);
                }
                if (UserStatus.INVITED == invitedPatient.getStatus()) {
                    user.setStatus(UserStatus.INVITED);
                } else if (UserStatus.CREATED == invitedPatient.getStatus()) {
                    user.setStatus(UserStatus.CREATED);
                } else if (UserStatus.NOCONTACT == invitedPatient.getStatus()) {
                    user.setStatus(UserStatus.NOCONTACT);
                } else if (UserStatus.EMAIL_CONFIRMED == invitedPatient.getStatus()) {
                    // User has confirmed email already
                    // It could be because of simultaneous submits or simply trying to re-register by mistake
                    return Action.SUCCESS;
                } else {
                    addActionError(getText("patientPersonalInfoAction.err.unable_to_register"));
                    LOGGER.error("Unsupported user state");
                    // Not supported user state
                    return Action.INPUT;
                }

                if (invitedPatient.getGender() != Gender.UNKNOWN.getId()) {
                    user.setGender(invitedPatient.getGender());
                }
                if (StringUtils.isNotBlank(invitedPatient.getDateOfBirthString())) {
                    user.setDateOfBirthString(invitedPatient.getDateOfBirthString());
                }
                if (invitedPatient.getEmail() != null) {
                    user.setEmailId(invitedPatient.getEmail());
                    user.setUserName(invitedPatient.getEmail());
                }
                // and the notification option (email throttling)
                user.setNotificationOption(invitedPatient.getNotificationOption());
            }
            addUserProperties(user, invitedPatient);
            session.put("user", user);
            session.put(PKBConstants.NEW_USER_EMAIL_ID, user.getEmailId());

            result = registerUser(invitedPatient);
            if (result.equals(SUCCESS)) {
                addActionMessage(getText("patientPersonalInfoAction.msg.successfully_registered"));
            }

        } catch (UserAlreadyExistsException ignored) {
            addFieldError("user.emailId", getText("patientPersonalInfoAction.err.email_used"));
            result = Action.INPUT;
        } catch (InvalidEmailAddressException ignored) {
            addFieldError("user.emailId",
                    getText("patientPersonalInfoAction.err.invalid_email"));
            result = Action.INPUT;

        } catch (Exception e) {
            result = Action.INPUT;
            LOGGER.error("general error recording personal info for user: {}", user, e);
            addActionError(getText("patientPersonalInfoAction.err.unable_to_register"));
        }
        LOGGER.info("Exiting-> processUserPersonalInfor");
        return result;
    }

    /*
     * Custom validation for attributes gender, date of birth must be valid date, etc.
     */
    @Override
    public void validate() {
        if (!selfRegistered) {
            invitedPatient = userManager.getPKBPerson(uid);
        }

        try {
            PKBPerson person = Option.of(getUser())
                    .flatMap(PKBPersonDTO::findEmailId)
                    .flatMap(email -> Option.of(userManager.getSinglePersonByEmail(email)))
                    .getOrNull();
            if ((person != null) && selfRegistered) {
                addFieldError("user.emailId", getText("patientPersonalInfoAction.err.email_used"));
            }
        } catch (IllegalReuseOfEmailAddressException e) {
            LOGGER.error("patient {} has multiple accounts!", getUser().getEmailId(), e);
            addActionError(getText("patientPersonalInfoAction.err.unable_to_register"));
            return;
        }

        getUser().setUserType(UserType.PATIENT);
        applyErrors(performRegistrationValidationSetDobAndGender(getUser()));
    }

    /* To check if the patient is already invited
     * If the patient exists in the database with the 'INVITED' status update the same row with the given details
     * and the user should be able to activate his account from the success page which contains an activation  link
     * else add this user as new user and the success page should intimate user to check his inbox in order to activate the account
     * Make an entry in InstituteUser table; no patient can register without the institute code
     */
    private String registerUser(PKBPerson person) {

        //Long instituteId=(Long)request.getSession().getAttribute("instituteId");
        Long instituteId = null;
        Team team = null;
        if (StringUtils.isNotBlank(icode)) {
            team = teamManager.getTeamByCode(icode);
            instituteId = team != null ? team.getId() : null;
        }

        boolean passwordChanged = false;
        boolean registrationCompleted = false;
        PKBPerson modPerson = null;
        String result = SUCCESS;
        try {
            userManager.beginTransaction();

            UserStatus invited = UserStatus.INVITED;
            UserStatus noContact = UserStatus.NOCONTACT;
            UserStatus created = UserStatus.CREATED;
            if ((person != null) && (person.getStatus() == invited || person.getStatus() == noContact
                    || person.getStatus() == created)) {
                modPerson = convertUtil.getPKBPerson(VersionDetails.of(getEHRRequestContext(), dateTimeService.now()), user);
                modPerson.setId(person.getId());
                modPerson.setContacts(person.getContacts());
                modPerson.setDefaultAccountId(person.getDefaultAccountId());
                modPerson.setPublicId(person.getPublicId());
                // modPerson.setUserType(person.getUserType());
                // modPerson.setStatus(UserStatus.INVITED);
                modPerson.setNationalIds(person.getNationalIds());
                modPerson.setOrgLevelIds(person.getOrgLevelIds());
                modPerson.setTeamLevelIds(person.getTeamLevelIds());

                RegisteringUserEHRRequestContext requestContext = getEHRRequestContext().withRegisteringUser(person.getId());
                if (userManager.getDefaultAccountId(user.getId()) == null) {
                    // user has PKBPerson but still needs
                    // accounts, etc. to be setup This is true of INVITED users
                    // like patients switching email for carer
                    Long teamId = team == null ? null : team.getId();
                    var accountsIds = userManager.registerUserFullSetupAccounts(requestContext, person.getId(), confirmPassword, teamId);
                    modPerson.setDefaultAccountId(accountsIds.getId());
                }

                // this includes changing the password from temp to modPerson.password
                // IMPORTANT: the registration link won't work again after this point
                userManager.updateRegistrationDetails(requestContext, modPerson, temp);
                passwordChanged = true;

                uid = modPerson.getId();
                if ((!teamUserManager.isTeamUser(uid)) && (StringUtils.isNotBlank(icode))) {
                    teamUserManager.createInstituteUser(uid, instituteId);
                }

                backupPassword(person.getId(), icode, Optional.empty(), Optional.empty());
            } else {
                person = userManager.registerUser(getEHRRequestContext().withRegisteringUser(Optional.empty()), user);
                uid = person.getId();
                if (!teamUserManager.isTeamUser(uid)) {
                    teamUserManager.createInstituteUser(uid, instituteId);
                }
                // send "confirm your email" message as last thing in transaction
                pkbEmailMessageManager.sendActivationMessage(person, team, null); // null specialtyId for patient
                result = PASSWORD_PENDING;
            }
            userManager.commitTransaction();

            if (!teamUserManager.isActiveTeamUser(uid) && person.isPatient()) {
                nhsLoginRegistrationManager.addExternalCredentialRecovery(confirmPassword.toCharArray(), person.getPublicId());
            }

            registrationCompleted = true;
        } catch (PKBException e) {
            // TODO: Do we need to log name here? Would PKB ID suffice?
            LOGGER.error("Error while registering patient {} {} with team-{}", person.getFirstName(), person.getLastName(), instituteId, e);
            result = INPUT;
        } finally {
            if (passwordChanged && !registrationCompleted) {
                // try to change pw back to temp, because LDAP has changed and won't roll back...
                try {
                    LOGGER.warn("attempting to change user password back to temp value {}", person.getId());
                    String currentPassword = modPerson.getPassword();
                    person.setPassword(temp); // set password back to temp in LDAP
                    userManager.updateRegistrationDetails(getEHRRequestContext(), person, currentPassword);
                } catch (Exception e) {
                    LOGGER.warn("failing restoring temp password... reg link will be broken for {}", person.getId(), e);
                }
            }

            userManager.rollbackIfNotCommitted();
        }
        return result;
    }
