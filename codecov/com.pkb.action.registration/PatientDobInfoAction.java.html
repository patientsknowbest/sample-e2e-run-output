package com.pkb.action.registration;

import com.pkb.datamodel.ImmutableRegistrationFailedAttemptLogDto;
import com.pkb.datamodel.RegistrationFailedAttemptLogDto;
import com.pkb.domain.RegistrationFailedAttemptLogService;
import com.pkb.exception.SetContextNotAuthorisedException;
import com.pkb.model.PKBPersonDTO;
import com.pkb.user.entity.PKBPerson;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.inject.Inject;
import java.time.Instant;
import java.time.temporal.ChronoUnit;
import java.util.UUID;

import static com.pkb.entities.pub.PublicRegistrationFailedAttemptLog.RegistrationField.DATE_OF_BIRTH;

public class PatientDobInfoAction extends BaseRegisterAction {

    private static final long serialVersionUID = 1L;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private PKBPersonDTO user;

    private String temp;

    private String title;

    private String firstName;

    private String lastName;

    @Inject
    private RegistrationFailedAttemptLogService registrationFailedAttemptLogService;

    @Override
    public String input() throws SetContextNotAuthorisedException {

        throwIfAnyUserLoggedIn();

        try {
            if (uid == null) {
                addActionError(getText("patientPersonalInfoAction.err.invalid_link"));
                return ERROR;
            }

            PKBPerson patient = userManager.getPKBPerson(uid, PKBPerson.Lazy.CONTACTS);
            if (patient == null) {
                addActionError(getText("patientPersonalInfoAction.err.invalid_link"));
                return ERROR;
            }

            title = patient.getTitle();
            firstName = patient.getFirstName();
            lastName = patient.getLastName();

            return SUCCESS;
        } catch (Exception e) {
            LOGGER.error("failed getting PKBPerson for uid {}", uid, e);
            addActionError(getText("patientPersonalInfoAction.err.invalid_link"));
            return ERROR;
        }
    }

    @Override
    public String execute() {
        try {
            if (uid == null) {
                addActionError(getText("patientPersonalInfoAction.err.invalid_link"));
                return ERROR;
            }

            PKBPerson patient = userManager.getPKBPerson(uid, PKBPerson.Lazy.CONTACTS);
            if (patient == null) {
                addActionError(getText("patientPersonalInfoAction.err.invalid_link"));
                return ERROR;
            }

            if (hasTooManyFailedRegistrationAttempts(patient.getPublicId())) {
                addFieldError("user.dob", getText("patientDobInfoAction.err.dob_unsuccessful"));
                return INPUT;
            }

            // Set here so they still get displayed if we return INPUT
            title = patient.getTitle();
            firstName = patient.getFirstName();
            lastName = patient.getLastName();

            user = new PKBPersonDTO();
            user.setId(uid);

            return patient.findDateOfBirth().map(maybeDoB -> {
                String knownDay = pkbWebUtil.getWebDate(maybeDoB);
                String knownMonth = pkbWebUtil.getWebMonth(maybeDoB);
                String knownYear = pkbWebUtil.getWebYear(maybeDoB);

                if (getDay().equals(knownDay) && getMonth().equals(knownMonth) && getYear().equals(knownYear)) {
                    logDobComparison(knownDay, knownMonth, knownYear, getDay(), getMonth(), getYear(), true);
                    registrationFailedAttemptLogService.deleteRegistrationFailedAttempts(patient.getPublicId(), DATE_OF_BIRTH);
                    return SUCCESS;
                } else {
                    logDobComparison(knownDay, knownMonth, knownYear, getDay(), getMonth(), getYear(), false);
                    addFieldError("user.dob", getText("patientDobInfoAction.err.dob_does_not_match"));
                    RegistrationFailedAttemptLogDto dto = ImmutableRegistrationFailedAttemptLogDto.builder()
                            .personPublicId(patient.getPublicId())
                            .registrationField(DATE_OF_BIRTH)
                            .registrationTime(dateTimeService.now())
                            .build();
                    registrationFailedAttemptLogService.save(dto);
                    return INPUT;
                }
            }).orElse(SUCCESS); // Since we don't know their DOB, just accept the date they provide.
        } catch (Exception e) {
            LOGGER.error("failed getting PKBPerson for uid {}", uid, e);
            addActionError(getText("patientPersonalInfoAction.err.invalid_link"));
            return ERROR;
        }
    }

    private void logDobComparison(String knownDay,
            String knownMonth,
            String knownYear,
            String day,
            String month,
            String year,
            boolean matched) {
        LOGGER.info("User id {} comparing DoB for registration. Expected [ {} {} {} ] Provided [ {} {} {} ] Matched: {}",
                uid,
                knownDay,
                knownMonth,
                knownYear,
                day,
                month,
                year,
                matched);
    }

    @Override
    public void validate() {
        user = new PKBPersonDTO();
        applyErrors(processDateOfBirth(getUser()));
    }

    private boolean hasTooManyFailedRegistrationAttempts(UUID publicPersonId) {
        Instant recentTime = dateTimeService.now().minus(config.getDefinitionOfFailedRegistrationAttemptInSeconds(), ChronoUnit.SECONDS);
        return registrationFailedAttemptLogService.findCountOfRegistrationFailedAttempts(publicPersonId, DATE_OF_BIRTH, recentTime) >= config
                .getNumberOfFailedRegistrationAttemptsAllowed();
    }

    public PKBPersonDTO getUser() {
        return user;
    }

    public void setUser(PKBPersonDTO user) {
        this.user = user;
    }

    public String getTemp() {
        return temp;
    }

    public void setTemp(String temp) {
        this.temp = temp;
    }

    public String getTitle() {
        return title;
    }

    public void setTitle(String title) {
        this.title = title;
    }

    public String getFirstName() {
        return firstName;
    }

    public void setFirstName(String firstName) {
        this.firstName = firstName;
    }

    public String getLastName() {
        return lastName;
    }

    public void setLastName(String lastName) {
        this.lastName = lastName;
    }
}
