package com.pkb.action.registration;

import com.pkb.VersionDetails;
import com.pkb.action.registration.letter.PatientActivationDetailsDTO;
import com.pkb.action.registration.nhslogin.NhsLoginFullPatientRegistrationDTO;
import com.pkb.action.registration.nhslogin.NhsLoginPatientActivationDTO;
import com.pkb.entities.enums.ValidNationalId;
import com.pkb.model.PKBPersonDTO;
import com.pkb.patient.activation.letter.AccountDetailsDTO;
import com.pkb.patient.activation.letter.ImmutableAccountDetailsDTO;
import com.pkb.patient.activation.letter.ImmutableInvitationTokenDTO;
import com.pkb.patient.activation.letter.InvitationTokenDTO;
import com.pkb.patient.activation.nhslogin.ImmutableNhsLoginAccountDetailsDTO;
import com.pkb.patient.activation.nhslogin.NhsLoginAccountDetailsDTO;
import com.pkb.security.NhsLoginAuthenticationDetails;
import com.pkb.user.entity.NationalId;
import com.pkb.user.entity.PKBPerson;
import io.vavr.Tuple;
import io.vavr.Tuple2;
import org.apache.commons.lang3.StringUtils;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.time.LocalDate;
import java.util.List;

import static com.pkb.entities.enums.NationalIdType.NHS_NUMBER;
import static com.pkb.entities.enums.UserType.PATIENT;
import static com.pkb.patient.activation.ImmutableAccountCredentialsDTO.accountCredentials;
import static com.pkb.patient.activation.ImmutableSecurityQuestionDTO.securityQuestion;
import static java.lang.String.format;
import static org.apache.commons.lang3.StringUtils.stripToNull;

public class PatientActivationDetailsDTOConverter {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    public Tuple2<InvitationTokenDTO, AccountDetailsDTO> convert(PatientActivationDetailsDTO activationDetails) {
        return Tuple.of(createInvitationTokenFrom(activationDetails), createNhsLoginAccountDetailsFrom(activationDetails));
    }

    public NhsLoginAccountDetailsDTO createNhsLoginAccountDetailsFrom(NhsLoginPatientActivationDTO activationDetails) {
        return ImmutableNhsLoginAccountDetailsDTO.builder()
                .accountCredentials(accountCredentials(activationDetails.getEmail(), activationDetails.getPassword().toCharArray()))
                .securityQuestionWithAnswer(
                        securityQuestion(activationDetails.getSecurityQuestionId(), activationDetails.getAnswerForSecurityQuestion(), activationDetails.getSecurityQuestion()))
                .build();
    }

    public PKBPersonDTO createPersonDtoFrom(NhsLoginFullPatientRegistrationDTO activationDetails, NhsLoginAuthenticationDetails authenticationDetails) {
        PKBPersonDTO person = new PKBPersonDTO();
        person.setUserType(PATIENT);
        person.setFirstName(authenticationDetails.getFirstName());
        person.setLastName(authenticationDetails.getLastName());
        person.setPassword(activationDetails.getPassword());
        person.setEmailId(activationDetails.getEmail());
        person.setAddress1(activationDetails.getAddress1());
        person.setAddress2(activationDetails.getAddress2());
        person.setCity(activationDetails.getCity());
        person.setState(activationDetails.getState());
        person.setPostalCode(activationDetails.getPostalCode());
        person.setGender(activationDetails.getGender());
        person.setNationalIds(retrieveNationalIdFromNhsLogin(authenticationDetails));
        person.setDateOfBirthString(authenticationDetails.getDateOfBirth());
        person.setCountry(activationDetails.getCountry());
        person.setTimeZoneId(activationDetails.getTimeZoneId());
        person.setLanguageCode(activationDetails.getLanguageCode());
        return person;
    }

    /**
     * Copies gender, address1, address2, city, postalCode, state fields from {@link PKBPerson} to
     * {@link AbstractPatientActivationDTO}.
     *
     * @param from the source object of the copy operation
     * @param to   the destination object of the copy operation
     */
    public void copyGenderAndAddressFields(@NotNull PKBPerson from, @NotNull AbstractPatientActivationDTO to) {
        to.setGender(from.getGender());
        to.setAddress1(from.getAddress1());
        to.setAddress2(from.getAddress2());
        to.setCity(from.getCity());
        to.setPostalCode(from.getPostalCode());
        to.setState(from.getState());
    }

    /**
     * Copies gender, address1, address2, city, postalCode, state fields from {@link AbstractPatientActivationDTO}
     * to {@link PKBPerson} if the data in their correspoinding fields differ.
     *
     * @param from the source object of the copy operation
     * @param to   the destination object of the copy operation
     * @return true if any fields needed to be updated because they were different, false otherwise
     */
    public boolean copyGenderAndAddressFields(@NotNull AbstractPatientActivationDTO from, @NotNull PKBPerson to) {
        boolean anyFieldsUpdated = false;
        if (to.getGender() != from.getGender()) {
            anyFieldsUpdated = true;
            to.setGender(from.getGender());
        }
        if (!StringUtils.equals(to.getAddress1(), from.getAddress1())) {
            anyFieldsUpdated = true;
            to.setAddress1(from.getAddress1());
        }
        if (!StringUtils.equals(to.getAddress2(), stripToNull(from.getAddress2()))) {
            anyFieldsUpdated = true;
            to.setAddress2(stripToNull(from.getAddress2()));
        }
        if (!StringUtils.equals(to.getCity(), from.getCity())) {
            anyFieldsUpdated = true;
            to.setCity(from.getCity());
        }
        if (!StringUtils.equals(to.getPostalCode(), from.getPostalCode())) {
            anyFieldsUpdated = true;
            to.setPostalCode(from.getPostalCode());
        }
        if (!StringUtils.equals(to.getState(), from.getState())) {
            anyFieldsUpdated = true;
            to.setState(from.getState());
        }
        return anyFieldsUpdated;
    }

    private AccountDetailsDTO createNhsLoginAccountDetailsFrom(PatientActivationDetailsDTO activationDetails) {
        LocalDate claimedDateOfBirth = LocalDate.of(activationDetails.getYearOfBirth(), activationDetails.getMonthOfBirth(), activationDetails.getDayOfBirth());
        return ImmutableAccountDetailsDTO.builder()
                .accountCredentials(accountCredentials(activationDetails.getEmail(), activationDetails.getPassword().toCharArray()))
                .securityQuestionWithAnswer(securityQuestion(activationDetails.getSecurityQuestionId(), activationDetails.getAnswerForSecurityQuestion(), activationDetails.getSecurityQuestion()))
                .claimedDateOfBirth(claimedDateOfBirth)
                .build();
    }

    private InvitationTokenDTO createInvitationTokenFrom(PatientActivationDetailsDTO activationDetails) {
        return ImmutableInvitationTokenDTO.of(activationDetails.getDecryptToken(), activationDetails.getInvitationCode());
    }

    private List<NationalId> retrieveNationalIdFromNhsLogin(NhsLoginAuthenticationDetails details) {
        ValidNationalId validNationalId = NHS_NUMBER.getValidNationalIdAndType(details.getNhsNumber())
                .orElseThrow(() -> new RuntimeException(format("Invalid NHS number '%s' in NHS login", details.getNhsNumber())));
        return List.of(new NationalId(validNationalId, VersionDetails.of(details.getEHRRequestContext())));
    }
}
