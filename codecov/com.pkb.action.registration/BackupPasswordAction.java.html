    /**
     * Called as part of registration (no user currently signed in)
     * <p>
     * API will activate the user based on the user type If the user is an
     * institute user he can activate his account using the activation link on
     * the registration success page without having invitation id If the
     * clinician registers without an institute he has to activate his account
     * using the activation link from the activation email If the account gets
     * activated successfully then save the security questions and answers
     * entered by the user If the user is an institute user update the status as
     * "ACTIVE" in instituteUser table
     */
    @Override
    public String execute() {
        LOGGER.info("Entering");
        try {
            // TODO: MULTIPLE-TEAMS Safe to do this for just the first user when there are multiple, since all use same password?
            PKBPerson user = userManager.getPKBPerson(uid, PKBPerson.Lazy.CONTACTS);
            if (user == null) {
                LOGGER.info("adding action error, returning input");
                addActionError(getText("backupPasswordAction.err.sorry_no_record"));
                return input();
            }

            // user found... and icode is valid, and user signing in matches UID
            // if auth fails, null currentUser value (and actionerror is added).
            PersonAuthInfo currentUser = verifyCredentials(user.getId());


            if (currentUser == null) {
                return input();
            } else {
                LoggedInEHRRequestContext context = ImmutableLoggedInUserEHRRequestContext.builder().from(getEHRRequestContext().afterUserLoggedIn(currentUser)).accessRoute(Route.WEBAPP).build();
                boolean isInstituteAdmin = user.isTeamCoordinator();
                // team user?  check if already activated!  if so, pass error message through sign-in?
                boolean isInstituteUser = teamUserManager.isTeamUser(user.getId());
                Team team = teamManager.getTeamByCode(icode);

                try {
                    userManager.beginTransaction();

                    Long defaultAccountId = userManager.getDefaultAccountId(user.getId());
                    if (defaultAccountId != null) {
                        // they already have the account, etc. -- just activate
                        user.setStatus(UserStatus.EMAIL_CONFIRMED, dateTimeService.now());
                        // confirm the primary and the only contact
                        user.getPrimaryContact().setConfirmed(true);
                        userManager.updateUser(context, user);
                    } else {
                        // user has keys (and can authenticate) but still needs accounts, etc. to be setup
                        userManager.activatePerson(context, user, team.getId(), j_password);
                    }
                    userManager.backUpPassword(user.getId(), j_password, securityQuestion, answer);

                    doPostConfirmationTasks(user,
                            isInstituteAdmin, team, context);
                    userManager.commitTransaction();
                } finally {
                    userManager.rollbackIfNotCommitted();
                }

                // Send the notification to team coordinator to activate the newly registered user
                if (isInstituteUser && !isInstituteAdmin) {
                    try {
                        teamManager.notifyTeamCoordToActivateUser(user, team);
                    } catch (Exception e) {
                        //Log the error; Rest of the steps should still proceed
                        LOGGER.info("Error while notifying the team coordinator-{}", e.getMessage());
                    }
                }
            }
        } catch (Exception e) {
            addActionError(getText("backupPasswordAction.err.could_not_update_security"));
            LOGGER.error("Error while saving backup j_password; returning input", e);
            return input();
        } finally {
            LOGGER.info("Exiting");
        }

        LOGGER.info("forwarding to /j_spring_security_check");
        return "loginSubmit";
    }

    private void doPostConfirmationTasks(PKBPerson user, boolean isInstituteAdmin, Team team, LoggedInEHRRequestContext context) {

        if (!isInstituteAdmin) {
            acceptPendingInvitations(user, context);

            if (user.isPatient()) {
                // create consent record
                PatientConsent pc = new PatientConsent.Builder(team.getId(), user.getId(), user.getId(),
                        AccessingEntityType.TEAM, dateTimeService.now()).route(Route.WEBAPP)
                        .setAllCategories(team, ConsentReason.EXPLICIT_CONSENT)
                        .build();

                userWebBean.updateOrCreateTeamConsentRecord(context, uid, team.getId(), pc);
            }
        }

        // save specialty (clinicians only)
        if (user.isPro()) {

            // sid will be populated if this is from an emailed URL during clinician registration
            String specialtyId = StringUtils.trimToEmpty(sid);
            if (StringUtils.isNumeric(specialtyId) && !StringUtils.isEmpty(specialtyId)) {
                userManager.saveClinicianSpecialty(user.getId(), Long.parseLong(specialtyId));
            }
        }

        registrationEventService.create(ImmutableRegistrationEventNew.builder()
                .registrationMethod(RegistrationMethod.SELF_REGISTRATION)
                .registeredOn(dateTimeService.now())
                .registeredOnConfirmed(true)
                .personId(user.getId())
                .build());
    }

    private void acceptPendingInvitations(PKBPerson currentUser, LoggedInEHRRequestContext context) {
        // this gets invitations TO and FROM current user -- we only want invitations TO this user
        List<PKBInvitation> invitations = invitationManager.getInvitations(currentUser.getId());

        if (invitations.isEmpty()) {
            LOGGER.info("No invitations found to be accepted by user: {}", currentUser.getId());
        }

        for (PKBInvitation invitation : invitations) {
            // only process new invitations where this user is target
            if (invitation.getInvitationStatus() == NEW
                    && currentUser.findEmail().map(email -> email.equals(invitation.getTargetEmail())).getOrElse(false)) {

                // for invitations that fail, notify the user and cancel them
                PKBPerson author = userManager.getPKBPerson(invitation.getAuthorUserId(), PKBPerson.Lazy.CONTACTS);
                PKBPerson invitee = invitation.getTargetUser();
                PKBPerson accountOwner = invitation.getAccountId() == null ? null
                        : userManager.getAccountOwner(invitation.getAccountId(), PKBPerson.Lazy.CONTACTS);
                try {

                    userWebBean.acceptInvitationAndNotify(context, invitation.getPublicId(), accountOwner, author, invitee, currentUser);

                } catch (PKBException ignored) {

                    if (author != null) {
                        List<Object> args = new ArrayList<>();
                        args.add(author.getFirstName());
                        args.add(author.getLastName());
                        args.add(author.getEmail());
                        addActionMessage(getText("backupPasswordAction.err.invitation_cancelled", args));
                    }
                    // really do expire it, so they can be re-invited
                    invitationManager.updateInvitationStatus(invitation.getPublicId(), EXPIRED);
                }
            }
        }
    }

    /**
     * API is used to authenticate the user with respect to the given
     * credentials
     */
    private PersonAuthInfo verifyCredentials(long id) {
        PersonAuthInfo authUser = null;
        try {
            // note: this is not an error; this method takes id, NOT username
            authUser = userManager.authenticateUser(id, j_password);

            if (StringUtils.isNotBlank(icode)) {
                Team team = teamManager.getTeamByCode(icode);
                if (team == null) {
                    addActionError(getText("backupPasswordAction.err.sorry_unknown_code"));
                    authUser = null;
                }
            }

        } catch (PKBException e) {
            LOGGER.error("Exception occur in backup j_password : invalid password for {}", j_username, e);
            addActionError(getText("backupPasswordAction.err.email_and_password_not_match"));
        }
        return authUser;
    }

    @Override
    public String input() {
        LOGGER.info("Enter : Input");

        // pre-populate email - we have this in the (clinician registration) URL:
        // http://localhost:8081/backupPasswordForm.action?uid=2750&icode=smoketest&sid=1900
        // (and do we actually need the icode in the URL?)


        if (loggedInUserUtil.isUserLoggedIn()) {
            PKBPerson user = loggedInUserUtil.getLoggedInPerson();
            setJ_username(user.findEmail().map(Email::address).getOrNull());
            userLoggedIn = true;
        } else {
            userLoggedIn = false;
            if (getUid() != null) {
                try {
                    PKBPerson registeringPerson = userManager.getPKBPerson(getUid(), PKBPerson.Lazy.CONTACTS);
                    if (registeringPerson == null) {
                        LOGGER.info("invalid uid-{}", uid);
                        addActionError(getText("backupPasswordAction.err.sorry_unknown_user"));
                        return INPUT;
                    }
                    setJ_username(registeringPerson.findEmail().map(Email::address).getOrNull());
                    if (registeringPerson.getStatus() == UserStatus.EMAIL_CONFIRMED) {
                        addActionMessage(getText("backupPasswordAction.err.already_registered"));
                        return LOGIN;
                    }

                } catch (PKBException e) {
                    LOGGER.error("error pre-populating email for registering user {}; ignoring", getUid(), e);
                }
            }
        }
        answer = "";

        LOGGER.info("Exit : Input");
        return INPUT;
    }

    /**
     * Called when updating this info for the existing, signed-in user
     *
     * @return
     */
    public String update() {
        LOGGER.info("Enter : Update");
        PKBPerson user = loggedInUserUtil.getLoggedInPerson();
        setJ_username(user.findEmail().map(Email::address).getOrElse(""));
        String result = Action.INPUT;
        try {
            userManager.authenticateUser(user.getId(), j_password);
            userManager.backUpPassword(user.getId(), j_password,
                    securityQuestion, answer);
        } catch (Exception e) {
            addActionError(getText("backupPasswordAction.err.invalid_password"));
            LOGGER.info("Error while saving backup j_password", e);
        }
        customizeSuccessMessage();
        result = input();
        LOGGER.info("Exiting update");
        return result;
    }

    private void customizeSuccessMessage() {
        if (!hasErrors()) {
            if (loggedInUserUtil.isUserLoggedIn()) {
                addActionMessage(getText("backupPasswordAction.msg.backup_success"));
            } else {
                addActionMessage(getText("backupPasswordAction.msg.backup_success_now_login"));
            }
        }
    }
