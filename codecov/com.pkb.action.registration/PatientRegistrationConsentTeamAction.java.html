package com.pkb.action.registration;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.datamodel.consent.ModifiablePatientConsentDTO;
import com.pkb.dto.AdditionalConsentsDto;
import com.pkb.interceptor.ConsentInterceptor;
import org.jooq.tools.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.Arrays;
import java.util.List;
import java.util.Objects;
import java.util.UUID;

import static java.util.Collections.emptyList;
import static java.util.Optional.ofNullable;
import static java.util.stream.Collectors.toList;
import static org.jooq.tools.StringUtils.isBlank;

public class PatientRegistrationConsentTeamAction extends BaseAction {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private PatientRegistrationConsentTeamDto consentDto;
    private AdditionalConsentsDto additionalConsentsDto;
    private ModifiablePatientConsentDTO patientConsent;
    private String redirectURL;
    private UUID teamPublicId;
    private String visitedTeamPublicIds;

    @Autowired
    private PatientRegistrationConsentService consentService;

    @Override
    public String execute() {
        try {
            var request = ImmutablePatientRegistrationConsentTeamServiceReadRequest.builder()
                    .eHRRequestContext(getEHRRequestContext())
                    .userId(loggedInUserUtil.getLoggedInUserId())
                    .editMode(ConsentInterceptor.ReviewConsentStatus.NOT_REQUIRED == getSession().get("ReviewConsentStatus"))
                    .locale(getLocale())
                    .contextOrLoggedInUserId(contextUserUtil.getContextOrLoggedInUserId())
                    .nhsLoginAuthenticationDetails(loggedInUserUtil.getNhsLoginAuthenticationDetails())
                    .homeResult(loggedInUserUtil.getLoggedInUserHomeResult())
                    .teamPublicId(teamPublicId)
                    .visitedTeamPublicIds(getUuidListFromString(getVisitedTeamPublicIds()))
                    .backText(getText("global.txt.nav.back"))
                    .build();
            consentDto = consentService.getPatientRegistrationConsentTeamDto(request);
            consentDto.getContextUserId().ifPresent(this::setContextUserId);
            consentDto.getRedirectUrl().ifPresent(this::setRedirectURL);
            return consentDto.getActionResult();
        } catch (RuntimeException ex) {
            LOGGER.error("Unable to get patient registration consent team DTO", ex);
            return Action.ERROR;
        }
    }

    public String saveConsentReviewForTeam() {
        try {
            setVisitedTeamPublicIds(updateVisitedTeamPublicIds());
            consentService.saveConsentReviewForTeam(ImmutablePatientRegistrationConsentReviewSaveRequest.builder()
                    .eHRRequestContext(getEHRRequestContext())
                    .userId(loggedInUserUtil.getLoggedInUserId())
                    .additionalConsents(ofNullable(getAdditionalConsentsDto()).orElse(new AdditionalConsentsDto()))
                    .patientConsent(getPatientConsent())
                    .teamPublicId(teamPublicId)
                    .visitedTeamPublicIds(getUuidListFromString(visitedTeamPublicIds))
                    .build())
                    .ifPresent(this::setTeamPublicId);
            return Action.SUCCESS;
        } catch (RuntimeException ex) {
            LOGGER.error("Unable to save consent review for team", ex);
            return Action.ERROR;
        }
    }

    public PatientRegistrationConsentTeamDto getConsentDto() {
        return consentDto;
    }

    public String getRedirectURL() {
        return redirectURL;
    }

    public void setRedirectURL(String redirectURL) {
        this.redirectURL = redirectURL;
    }

    public AdditionalConsentsDto getAdditionalConsentsDto() {
        return additionalConsentsDto;
    }

    public void setAdditionalConsentsDto(AdditionalConsentsDto additionalConsentsDto) {
        this.additionalConsentsDto = additionalConsentsDto;
    }

    public ModifiablePatientConsentDTO getPatientConsent() {
        return patientConsent;
    }

    public void setPatientConsent(ModifiablePatientConsentDTO patientConsent) {
        this.patientConsent = patientConsent;
    }

    @Override
    public String getCustomStyle() {
        return NO_NAVIGATION + " " + NO_TEAM_FOOTER;
    }

    public UUID getTeamPublicId() {
        return teamPublicId;
    }

    public void setTeamPublicId(UUID teamPublicId) {
        this.teamPublicId = teamPublicId;
    }

    public String getVisitedTeamPublicIds() {
        return Objects.requireNonNullElse(visitedTeamPublicIds, StringUtils.EMPTY);
    }

    public void setVisitedTeamPublicIds(String visitedTeamPublicIds) {
        this.visitedTeamPublicIds = visitedTeamPublicIds;
    }

    public boolean hasTeamInformation() {
        return !isBlank(Objects.requireNonNull(consentDto.getTeam()).getPatientTeamInformation());
    }

    private List<UUID> getUuidListFromString(String uuids) {
        if (isBlank(uuids)) {
            return emptyList();
        }
        return Arrays.stream(uuids.split(",")).map(UUID::fromString).collect(toList());
    }

    private String updateVisitedTeamPublicIds() {
        if (isBlank(visitedTeamPublicIds)) {
            return teamPublicId.toString();
        }
        return String.join(",", visitedTeamPublicIds, teamPublicId.toString());
    }
}
