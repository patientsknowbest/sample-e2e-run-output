package com.pkb.action.registration;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import com.pkb.action.NhsLoginResource;
import com.pkb.exception.PKBException;
import com.pkb.institute.entity.Team;
import com.pkb.service.team.TeamManager;
import com.pkb.service.team.TeamUserManager;
import com.pkb.util.TermsAndConditionsConfigDTO;
import org.apache.http.client.utils.URIBuilder;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.net.URISyntaxException;

import static org.apache.commons.lang3.StringUtils.isBlank;
import static org.apache.commons.lang3.StringUtils.isNotBlank;

public class RegisterPatientStartAction extends BaseAction implements NhsLoginResource {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private static final long serialVersionUID = 1L;

    private transient TeamManager teamManager;
    private transient TeamUserManager teamUserManager;

    private String icode;
    private Long uid;

    private boolean consented = false;
    public TermsAndConditionsConfigDTO termsAndConditionsConfig;

    /**
     * temporary password for clinician-created patient
     */
    private String temp;

    /**
     * true if the user has already chosen a password and security info
     */
    private boolean securityInfoSet = false;

    private Team institute;

    @Override
    public String execute() {
        if (uid != null) {
            if (teamUserManager.isUserRegistered(uid)) {
                return "patientAlreadyRegistered";
            } else if (isConsented()) {
                return "invitedRegConsented";
            } else {
                return "consentForm";
            }
        } else {
            if (loadTeamIfValid(icode)) {
                if (isConsented()) {
                    return "selfRegConsented";
                } else {
                    return "selfRegStart";
                }
            } else {
                return "input";
            }
        }
    }

    public String showConsentForm() {
        // iCode is not required for carer invitations

        termsAndConditionsConfig = buildTermsAndConditionsConfig();

        if (getUid() != null) {
            return Action.SUCCESS;
        }

        if (loadTeamIfValid(icode)) {
            return Action.SUCCESS;
        } else {
            return Action.ERROR;
        }
    }

    /**
     * used to load user input form for institute code
     *
     * @Params : none
     * @Return type : String
     */
    @Override
    public String input() {
        return Action.SUCCESS;
    }

    @Override
    public String getNhsLoginLink() {
        try {
            var nhsLoginLink = new URIBuilder(config.getNhsIntegrationUrl());
            nhsLoginLink.addParameter("isSelfReg", "true");
            if (institute != null) {
                return nhsLoginLink.addParameter("teamCode", institute.getCode()).toString();
            }

            return nhsLoginLink.toString();
        } catch (URISyntaxException exception) {
            throw new RuntimeException(exception);
        }
    }

    public boolean isNhsRegistrationButtonVisible() {
        return config.isNhsLoginWebUiButtonEnabledForSelfRegistration();
    }

    /**
     * Validate the institute code with the database
     *
     * @Params : institute code as String
     * @Return type : boolean
     * Returns true : when given institute code exists in the database
     * Returns false : when given institute code does not exists in the database
     */
    private boolean loadTeamIfValid(String code) {
        if (isNotBlank(code)) {
            try {
                institute = teamManager.getTeamByCode(code);
            } catch (PKBException exception) {
                LOGGER.error("error getting institute for code {}", code, exception);
            }
        }

        if (institute == null) {
            if (code == null) {
                addActionMessage(getText("registerPatientStartAction.err.need_code"));
            } else if (isBlank(code)) {
                addActionMessage(getText("registerPatientStartAction.err.sorry_need_code"));
            } else { // there's a code, but it's wrong
                addActionMessage(getText("registerPatientStartAction.err.unknown_code"));
            }
            return false;
        } else {
            // valid institute
            institute.setLogoFileName(getConfig().getInstituteLogoBaseDir() + institute.getLogoFileName());
            session.put("instituteId", institute.getId());
            session.put("icode", institute.getCode());
            session.put("uId", uid);
            return true;
        }
    }

    public String getIcode() {
        return icode;
    }

    public void setIcode(String icode) {
        this.icode = icode;
    }

    public TeamManager getTeamManager() {
        return teamManager;
    }

    public void setTeamManager(TeamManager teamManager) {
        this.teamManager = teamManager;
    }

    public Team getInstitute() {
        return institute;
    }

    public void setInstitute(Team institute) {
        this.institute = institute;
    }

    public Long getUid() {
        return uid;
    }

    public void setUid(Long uid) {
        this.uid = uid;
    }

    public String getTemp() {
        return temp;
    }

    public void setTemp(String temp) {
        this.temp = temp;
    }

    public TeamUserManager getTeamUserManager() {
        return teamUserManager;
    }

    public void setTeamUserManager(TeamUserManager teamUserManager) {
        this.teamUserManager = teamUserManager;
    }

    public boolean isSecurityInfoSet() {
        return securityInfoSet;
    }

    public void setSecurityInfoSet(boolean securityInfoSet) {
        this.securityInfoSet = securityInfoSet;
    }

    public boolean isConsented() {
        return consented;
    }

    public void setConsented(boolean consented) {
        this.consented = consented;
    }

    public TermsAndConditionsConfigDTO getTermsAndConditionsConfig() {
        return termsAndConditionsConfig;
    }

    public void setTermsAndConditionsConfig(TermsAndConditionsConfigDTO termsAndConditionsConfig) {
        this.termsAndConditionsConfig = termsAndConditionsConfig;
    }
}
