package com.pkb.action.registration;

/**
 * Verify the email address that a user has entered is valid
 */
public class ForgotPasswordEmailInfoAction extends ForgotPasswordActionSupport implements ServletRequestAware {
    private static final long serialVersionUID = 2L;
    private static final Logger LOGGER = getLogger(lookup().lookupClass());

    public static final String RESTART = "restart";

    private static final String WRONG_DETAILS = "forgotPasswordPersonalInfoAction.err.wrong_details";
    private static final String LINK_EXPIRED = "forgotPasswordPersonalInfoAction.err.expired_link";
    private static final String NOT_YET_REGISTERED = "forgotPasswordPersonalInfoAction.err.not_yet_registered";
    private static final String PASSWORD_ALREADY_RESET = "forgotPasswordPersonalInfoAction.err.password_already_reset";
    private static final String EMPTY_EMAIL = "forgotPassword.error.email_empty";
    private static final String INVALID_EMAIL = "forgotPassword.error.email_invalid";

    @Autowired
    private PhrConfig config;
    @Autowired
    private PKBEmailValidator emailValidator;
    @Autowired
    private PasswordResetService passwordResetService;

    @Override
    public void validate() {
        //noinspection StrutsI18nError,SSBasedInspection
        emailValidator.validEmail(resetPassObj.findEmailId().map(Email::address).getOrNull()).toEither().peekLeft(failure -> Match(failure).of(
                Case($(instanceOf(Blank.class)), () -> Some(Tuple.of("resetPassObj.emailId", getText(EMPTY_EMAIL)))),
                Case($(instanceOf(Invalid.class)), invalid -> {
                    if (nonNull(invalid.cause())) {
                        LOGGER.warn("Email validation failed. Root cause:", Invalid.class.cast(invalid).cause());
                    }
                    return Some(Tuple.of("resetPassObj.emailId", getText(INVALID_EMAIL)));
                }))
                .forEach($ -> addFieldError($._1, $._2)));
    }

    /**
     * Step one - part one.
     * Display a form where User can enter her/his email address to start the process.
     */
    @Override
    @SkipValidation
    public String input() {
        LOGGER.info("Forgot Password Screen is opened");
        return SUCCESS;
    }

    /**
     * Step one - part two.
     * On successful form submission:
     * Use the provided email address to try to send a password reset link to user.
     * We don't want leak any info (is email is used in our systems or not), hence
     * we always display the same page after form submission.
     * Users should receive an email if the address belongs to a registered email address.
     */
    public String send() {
        LOGGER.info("Submitted data via Forgot Password form");

        return emailValidator.validEmail(resetPassObj.findEmailId().map(Email::address).getOrNull()).fold(
                failure -> {
                    LOGGER.warn("User input=[{}] is invalid ({})", resetPassObj.getEmailId(), failure.message());
                    return ERROR;
                }, email -> {
                    passwordResetService.startPasswordReset(email);
                    return SUCCESS;
                });
    }

    /**
     * Step two - part one (next is: {@link ForgotPasswordPersonalInfoAction#execute()})
     * When User opens link we sent via email then we render a form.
     */
    @Override
    public String execute() {
        LOGGER.info("Password Reset link is opened");

        return validateUserInput().fold(
                validationError -> {
                    if (isNull(resetPassObj)) {
                        LOGGER.warn("Mandatory request parameters are missing. Possible request tempering?");
                    } else {
                        var input = format("email=[%s],expiry date=[%s],reset code=[%s]", resetPassObj.findEmailId().map(Email::address).getOrNull(), resetPassObj.getExpiryDate(), resetPassObj.getResetCode());
                        LOGGER.warn("User input=[{}] is invalid ({})", input, validationError);
                    }
                    return ERROR;
                },
                details -> passwordResetService.findVerificationCriteria(details).fold(
                        failure -> Match(failure).of(
                                Case($(instanceOf(LinkExpired.class)), () -> {
                                    addActionError(getText(LINK_EXPIRED));
                                    return RESTART;
                                }),
                                Case($(instanceOf(AlreadyResetByTeamMember.class)), () -> {
                                    addActionError(getText(PASSWORD_ALREADY_RESET));
                                    return INPUT;
                                }),
                                // Below cases are invalid at this stage and should be handled such:
                                // Indicates request tempering or some bug in our system.
                                Case($(instanceOf(ResetCodeMismatch.class)), () -> {
                                    addActionError(getText(WRONG_DETAILS));
                                    return INPUT;
                                }),
                                // Indicates request tempering or some bug in our system.
                                Case($(instanceOf(NoUserByEmail.class)), ERROR),
                                Case($(instanceOf(NotRegistered.class)), () -> {
                                    addActionError(getText(NOT_YET_REGISTERED));
                                    return ERROR;
                                })),
                        verifiableBits -> {
                            setVerifiableBits(verifiableBits);
                            resetPassObj.setQuestion1(verifiableBits.question());
                            resetPassObj.setUnaffiliatedPatient(verifiableBits.unaffiliatedPatient());
                            return SUCCESS;
                        }));
    }

    private Validation<String, PasswordResetDetails> validateUserInput() {
        if (isNull(resetPassObj)) {
            return invalid("resetPassObj is null");
        }

        Validation<String, Instant> validatedExpiryDate = Try(() -> Instant.parse(resetPassObj.getExpiryDate())).toValidation()
                .mapError(Throwable::getMessage);

        Validation<String, Email> validatedEmail = emailValidator.validEmail(resetPassObj.findEmailId().map(Email::address).getOrNull())
                .mapError(EmailValidationFailure::message);

        return Validation.combine(validatedExpiryDate, validatedEmail)
                .ap((expiryDate, email) -> (PasswordResetDetails) ImmutablePasswordResetDetails.builder()
                        .expiryDate(expiryDate)
                        .email(email)
                        .resetCode(resetPassObj.getResetCode())
                        .build())
                .mapError(errors -> String.join(",", errors));
    }

    public String getHospitalMapIframeSrc() {
        return config.getHospitalMapIframeSrc();
    }
}
