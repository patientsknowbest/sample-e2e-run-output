package com.pkb.action.registration;

import com.pkb.app.entity.EHRRequestContext;
import com.pkb.util.ContextUtil;
import org.apache.struts2.interceptor.ServletRequestAware;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import java.io.Serializable;

/**
 * Allow the user to choose the new password based on answer to security
 * question
 *
 * @author pravinam
 *
 */
public class ForgotPasswordAction extends ForgotPasswordActionSupport implements Serializable, ServletRequestAware {

    private static final long serialVersionUID = 1L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    private ContextUtil contextUtil;

    @Override
    public String execute() {
        LOGGER.debug("In ForgotPasswordAction");
        String result = SUCCESS;
        if (resetPassObj != null) {
            if (resetPassObj.isUnaffiliatedPatient()) {
                result = resetPasswordUsingRecoveryCredentials();
            } else {
                result = resetPasswordUsingSecurityAnswer();
            }
        }
        if (SUCCESS.equals(result)) {
            addActionMessage(getText("forgotPassword.txt.successfulPasswordReset"));
        }
        return result;
    }

    private String resetPasswordUsingRecoveryCredentials() {
        return userManager.transactional(() -> {
            try {
                EHRRequestContext context = contextUtil.getEHRRequestContext(request);
                userManager.resetPasswordUsingExternalRecovery(context.getCorrelationId(), resetPassObj.getUid(), resetPassObj.getPassword());
                return SUCCESS;
            } catch (Exception e) {
                LOGGER.error("Failed reset password for user with ID=[{}]", resetPassObj.getUid(), e);
                return ERROR;
            }
        });
    }

    private String resetPasswordUsingSecurityAnswer() {
        return userManager.transactional(() -> {
            try {
                userManager.resetPassword(resetPassObj.getUid(), resetPassObj.getQuestion1(), resetPassObj.getAnswer1(), resetPassObj.getPassword());
                return SUCCESS;
            } catch (RuntimeException e) {
                addActionError(getText("forgotPasswordAction.err.wrong_details"));
                LOGGER.error("Failed reset password for user with ID=[{}]", resetPassObj.getUid(), e);
                return "inputDetails";
            } catch (Exception exception) {
                LOGGER.error("Failed reset password for user with ID=[{}]", resetPassObj.getUid(), exception);
                return ERROR;
            }
        });
    }
}
