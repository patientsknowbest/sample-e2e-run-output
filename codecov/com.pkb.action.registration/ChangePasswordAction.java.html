package com.pkb.action.registration;

import com.opensymphony.xwork2.Action;
import com.opensymphony.xwork2.ActionContext;
import com.pkb.action.BaseAction;
import com.pkb.app.entity.PersonAuthInfo;
import com.pkb.exception.PKBException;
import com.pkb.model.PKBSecurityQuestion;
import com.pkb.service.team.TeamUserManager;
import com.pkb.service.user.impl.UserManager;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.security.authentication.UsernamePasswordAuthenticationToken;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.context.SecurityContextImpl;

import java.util.List;
import java.util.Map;

import static com.pkb.util.PKBWebUtil.SPRING_SECURITY_CONTEXT;

/**
 * This action changes user password and this is accessed by only logged in user
 *
 * @author mahendera
 */
public class ChangePasswordAction extends BaseAction {

    /**
     *
     */
    private static final long serialVersionUID = 1L;

    private String oldPassword;
    private String newPassword;
    private String confirmPassword;
    private String securityQuestion;
    private String answer;
    private final String tab = "myaccount";
    private final String subTab = "changePassword";
    private List<PKBSecurityQuestion> secretQuestions;

    private TeamUserManager teamUserManager;

    /**
     * This action changes user password
     */
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Override
    public String execute() {
        String result;
        LOGGER.info("Entering->execute");

        try {
            // this will fail if old password is wrong (before getting into the
            // changePassword() call)
            userManager.authenticateUser(loggedInUserUtil.getLoggedInUserId(), oldPassword);

            userManager.changePassword(loggedInUserUtil.getLoggedInUserId(), oldPassword, newPassword, securityQuestion, answer);

            // Login the user with new password
            PersonAuthInfo user = userManager.authenticateUser(loggedInUserUtil.getLoggedInUserId(), newPassword);
            loginWithNewPassword(user);
            addActionMessage(getText("changePasswordAction.msg.changed_password"));
            result = Action.SUCCESS;
            answer = "";
        } catch (PKBException ex) {
            LOGGER.error("Error while changing password", ex);

            addFieldError("oldPassword", getText("changePasswordAction.err.old_password_wrong"));
            result = Action.INPUT;
        } catch (Exception ex) {
            LOGGER.error("Error while changing password", ex);
            addActionError(getText("changePasswordAction.err.sorry_could_not_update"));
            result = Action.ERROR;
        }
        LOGGER.info("Exiting->execute");
        return result;
    }

    @Override
    public void validate() {
        if (this.securityQuestion != null) {
            if ("-1".equals(this.securityQuestion.trim())) {
                addFieldError("securityQuestion",
                        getText("changePasswordAction.err.enter_security_question"));
            }
        }
    }

    private void loginWithNewPassword(PersonAuthInfo user) {
        Map<String, Object> session = ActionContext.getContext().getSession();

        SecurityContextImpl userSession = null;
        if (session != null) {
            userSession = (SecurityContextImpl) session.get(SPRING_SECURITY_CONTEXT);
            if (userSession != null) {
                Authentication auth = userSession.getAuthentication();
                if (auth != null) {
                    UsernamePasswordAuthenticationToken newToken = new UsernamePasswordAuthenticationToken(
                            user, newPassword, auth.getAuthorities());
                    newToken.setDetails(auth.getDetails());
                    userSession.setAuthentication(newToken);
                }
            }
        }
    }

    @Override
    public String input() {
        answer = "";
        return Action.INPUT;
    }

    public String getOldPassword() {
        return oldPassword;
    }

    public void setOldPassword(String oldPassword) {
        this.oldPassword = oldPassword;
    }

    public String getNewPassword() {
        return newPassword;
    }

    public void setNewPassword(String newPassword) {
        this.newPassword = newPassword;
    }

    public String getConfirmPassword() {
        return confirmPassword;
    }

    public void setConfirmPassword(String confirmPassword) {
        this.confirmPassword = confirmPassword;
    }

    public UserManager getUserManager() {
        return userManager;
    }

    public String getSecurityQuestion() {
        return securityQuestion;
    }

    public void setSecurityQuestion(String securityQuestion) {
        this.securityQuestion = securityQuestion;
    }

    public String getAnswer() {
        return answer;
    }

    public void setAnswer(String answer) {
        this.answer = answer;
    }

    public List<PKBSecurityQuestion> getSecretQuestions() {
        if (secretQuestions == null) {
            secretQuestions = pkbWebUtil.getSecretQuestions(getLocale());
        }
        return secretQuestions;
    }

    public void setSecretQuestions(List<PKBSecurityQuestion> secretQuestions) {
        this.secretQuestions = secretQuestions;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public String getTab() {
        return tab;
    }

    public String getSubTab() {
        return subTab;
    }

    public TeamUserManager getTeamUserManager() {
        return teamUserManager;
    }

    public void setTeamUserManager(TeamUserManager teamUserManager) {
        this.teamUserManager = teamUserManager;
    }

}
