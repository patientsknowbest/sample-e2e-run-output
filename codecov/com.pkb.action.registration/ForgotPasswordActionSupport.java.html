package com.pkb.action.registration;

import com.opensymphony.xwork2.ActionSupport;
import com.pkb.action.registration.reset.VerifiableBits;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.dto.ResetPasswordDTO;
import com.pkb.service.user.impl.UserManager;

import javax.servlet.http.HttpServletRequest;
import java.io.Serializable;
import java.util.Map;

import static com.pkb.PKBConstants.EARLIEST_ACCEPTED_YEAR_OF_BIRTH;
import static io.vavr.collection.Stream.rangeClosed;
import static io.vavr.collection.Stream.rangeClosedBy;
import static java.util.function.Function.identity;

/**
 * Get the common methods for the Forgot Password flow into one place
 *
 * @author keith
 */
public class ForgotPasswordActionSupport extends ActionSupport implements Serializable {

    private static final long serialVersionUID = 1L;

    protected UserManager userManager;
    protected HttpServletRequest request;
    protected DateTimeService dateTimeService;
    protected ResetPasswordDTO resetPassObj = new ResetPasswordDTO();

    private static final Map<String, String> DAYS = rangeClosed(1, 31)
            .map(dayIndex -> Integer.toString(dayIndex))
            .toLinkedMap(identity(), identity())
            .toJavaMap();

    private static final String[] MONTH_NAMES = {"global.txt.january", "global.txt.february", "global.txt.march", "global.txt.april",
            "global.txt.may", "global.txt.june", "global.txt.july", "global.txt.august", "global.txt.september", "global.txt.october",
            "global.txt.november", "global.txt.december"};

    public UserManager getUserManager() {
        return userManager;
    }

    public void setUserManager(UserManager userManager) {
        this.userManager = userManager;
    }

    public HttpServletRequest getRequest() {
        return request;
    }

    public void setRequest(HttpServletRequest request) {
        this.request = request;
    }

    public void setServletRequest(HttpServletRequest request) {
        this.request = request;
    }

    public ResetPasswordDTO getResetPassObj() {
        return resetPassObj;
    }

    public void setResetPassObj(ResetPasswordDTO resetPassObj) {
        this.resetPassObj = resetPassObj;
    }

    public Map<String, String> getDays() {
        return DAYS;
    }

    public Map<String, String> getMonths() {
        return rangeClosed(1, 12)
                .toLinkedMap(String::valueOf, index -> getText(MONTH_NAMES[index - 1]))
                .toJavaMap();
    }

    public Map<String, String> getYears() {
        int currentYear = dateTimeService.today().getYear();
        return rangeClosedBy(currentYear, EARLIEST_ACCEPTED_YEAR_OF_BIRTH, -1)
                .map(dayIndex -> Integer.toString(dayIndex))
                .toLinkedMap(identity(), identity())
                .toJavaMap();
    }

    public void setDateTimeService(DateTimeService dateTimeService) {
        this.dateTimeService = dateTimeService;
    }

    protected void setVerifiableBits(VerifiableBits bits) {
        this.resetPassObj.setVerifyLastName(bits.verifyLastName());
        this.resetPassObj.setVerifyDateOfBirth(bits.verifyDateOfBirth());
        this.resetPassObj.setVerifyPostalCode(bits.verifyPostalCode());
        this.resetPassObj.setVerifySecurityAnswer(bits.verifySecurityQuestion());
    }
}
