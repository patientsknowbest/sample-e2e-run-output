package com.pkb.action.registration;

import com.opensymphony.xwork2.Action;
import com.pkb.action.BaseAction;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

public class PatientRegistrationConsentIntroAction extends BaseAction {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private PatientRegistrationConsentIntroDto consentDto;

    @Autowired
    private PatientRegistrationConsentService consentService;

    @Override
    public String execute() {
        try {
            var request = ImmutablePatientRegistrationConsentIntroServiceReadRequest.builder()
                    .eHRRequestContext(getEHRRequestContext())
                    .userId(loggedInUserUtil.getLoggedInUserId())
                    .editMode(false)
                    .locale(getLocale())
                    .build();
            consentDto = consentService.getPatientRegistrationConsentIntroDto(request);
            return Action.SUCCESS;
        } catch (RuntimeException ex) {
            LOGGER.error("Unable to build consent DTO", ex);
            return Action.ERROR;
        }
    }

    public PatientRegistrationConsentIntroDto getConsentDto() {
        return consentDto;
    }

    @Override
    public String getCustomStyle() {
        return NO_NAVIGATION + " " + NO_TEAM_FOOTER;
    }
}
