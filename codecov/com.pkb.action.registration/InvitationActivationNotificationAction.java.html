package com.pkb.action.registration;

import com.pkb.action.BaseAction;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.LoggedOutEHRRequestContext;
import com.pkb.dto.InvitationActivationNotificationDTO;
import com.pkb.exception.MailException;
import com.pkb.institute.entity.Team;
import com.pkb.service.invitation.impl.InvitationManager;
import com.pkb.service.team.TeamManager;
import com.pkb.service.user.impl.UserManager;
import com.pkb.user.entity.PKBPerson;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import java.io.IOException;

import static com.pkb.app.entity.ImmutableLoggedOutEHRRequestContext.copyOf;
import static com.pkb.entities.enums.Route.SCHED_TASK;
import static io.vavr.API.Try;
import static java.lang.invoke.MethodHandles.lookup;


public class InvitationActivationNotificationAction extends BaseAction {

    private static final Logger LOGGER = LoggerFactory.getLogger(lookup().lookupClass());

    @Autowired
    private InvitationManager invitationManager;
    @Autowired
    private TeamManager teamManager;

    @Override
    public String execute() {
        return Try(this::processRequest)
                .getOrElseGet(throwable -> {
                    LOGGER.error("Error processing invitation activation notification", throwable);
                    return ERROR;
                });
    }

    private String processRequest() throws IOException, MailException {
        InvitationActivationNotificationDTO dto = OBJECT_MAPPER.readValue(request.getInputStream(), InvitationActivationNotificationDTO.class);

        PKBPerson person = userManager.getPKBPerson(dto.patientId());
        if (person == null) {
            throw new RuntimeException("Patient not found by id: " + dto.patientId());
        }

        Team team = teamManager.getTeam(dto.teamId());
        if (team == null) {
            throw new RuntimeException("Team not found by id: " + dto.teamId());
        }

        EHRRequestContext requestContext = copyOf((LoggedOutEHRRequestContext) getEHRRequestContext()).withAccessRoute(SCHED_TASK);

        try {
            invitationManager.sendInvitation(team, dto.patientId());
            userManager
                    .recreateOwnerAccessToAccount(requestContext.getCorrelationId().toString(),
                            userManager.getDefaultAccountWithActiveAccountUser(person.getId()).getPublicId(), person.getPublicId())
                    .throwIfFailed();
        } catch (UserManager.RecreateOwnerAccessToAccountResult.Exception exception) {
            throw new RuntimeException("Account access failed to restore after new credentials for patient id-" + dto.patientId(), exception);
        }

        return SUCCESS;
    }
}
