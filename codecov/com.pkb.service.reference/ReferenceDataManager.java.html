package com.pkb.service.reference;

import com.google.common.collect.Lists;
import com.pkb.app.interfaces.IBaseDTO;
import com.pkb.common.config.PhrConfig;
import com.pkb.common.datetime.DateTimeService;
import com.pkb.datamodel.ReferenceDatum;
import com.pkb.domain.ReferenceDatumService;
import com.pkb.entities.enums.ReferenceDataType;
import com.pkb.service.uuid.UUIDProvider;
import com.pkb.util.tolven.TolvenBeanFactory;
import com.pkb.util.tolven.TransactionManager;
import org.apache.commons.collections4.MapUtils;
import org.apache.commons.lang3.StringUtils;
import org.apache.commons.lang3.builder.HashCodeBuilder;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.HashMap;
import java.util.HashSet;
import java.util.LinkedHashMap;
import java.util.Map;
import java.util.Objects;
import java.util.function.Function;
import java.util.stream.Collectors;

import static com.pkb.entities.enums.PrivacyFlag.GENERAL;
import static com.pkb.entities.enums.ReferenceDataType.HOSPITAL_SERVICE;
import static com.pkb.entities.enums.ReferenceDataType.ORDER_SERVICE;

/**
 * Reference Data is a collection of customer-specific code sets.
 *
 * Each customer (organisation) can configure sets of ID/Text pairings, for
 * several data types (see ReferenceDatum.ReferenceDataType) and can optionally
 * specify privacy labels where they make sense for that data type.
 *
 * This reference data can be used for a few purposes, but in the first
 * instance it allows us to apply privacy labels to a data point based on pre-
 * configured, customer-specified values for e.g. their order service codes.
 *
 * See http://dev.patientsknowbest.com/home/hl7-api/code-sets for a little
 * more information.
 */
public class ReferenceDataManager extends TransactionManager {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private final ReferenceDatumService referenceDatumService;

    public ReferenceDataManager(PhrConfig config, TolvenBeanFactory beanFactory, DateTimeService dateTimeService, UUIDProvider uuidProvider,
                                ReferenceDatumService referenceDatumService) {
        super(config, beanFactory, dateTimeService, uuidProvider);
        this.referenceDatumService = referenceDatumService;
    }

    public static class RefDatumOrgAndData {
        private Long orgId;
        private ReferenceDataType dataType;
        private String dataId;

        public RefDatumOrgAndData(Long orgId, ReferenceDataType dataType, String dataId) {
            this.orgId = orgId;
            this.dataType = dataType;
            this.dataId = dataId;
        }

        public Long getOrgId() {
            return orgId;
        }

        public ReferenceDataType getDataType() {
            return dataType;
        }

        public String getDataId() {
            return dataId;
        }

        @Override
        public boolean equals(Object o) {
            if (!(o instanceof RefDatumOrgAndData)) {
                return false;
            }

            RefDatumOrgAndData refDatumOrgAndData = (RefDatumOrgAndData) o;
            return Objects.equals(this.orgId, refDatumOrgAndData.orgId) &&
                    this.dataType == refDatumOrgAndData.dataType &&
                    Objects.equals(this.dataId, refDatumOrgAndData.dataId);
        }

        @Override
        public int hashCode() {
            return new HashCodeBuilder()
                    .append(orgId)
                    .append(dataType)
                    .append(dataId)
                    .toHashCode();
        }
    }

    public <T, R extends IBaseDTO> Map<T, ReferenceDataManager.RefDatumOrgAndData> getReferenceDatumMap(Collection<T> inputCollection, Function<T, R> getIbaseDto,
            ReferenceDataType dataType, Function<T, String> dataId) {
        return inputCollection.stream()
                .collect(Collectors.toMap(
                        Function.identity(),
                        r -> convertToRefDatum(getIbaseDto.apply(r), dataType, dataId.apply(r)),
                        (u, v) -> {
                            throw new IllegalStateException(String.format("Duplicate key %s", u));
                        },
                        LinkedHashMap::new));
    }

    private <R extends IBaseDTO> RefDatumOrgAndData convertToRefDatum(R iBaseDto, ReferenceDataType dataType, String dataId) {
        if (iBaseDto != null && iBaseDto.getSource() != null) {
            return new ReferenceDataManager.RefDatumOrgAndData(iBaseDto.getSource().getOrgId(), dataType, dataId);
        }
        return new ReferenceDataManager.RefDatumOrgAndData(null, dataType, dataId);
    }

    public <T> Map<RefDatumOrgAndData, String> processReferenceDatumBatch(Map<T, RefDatumOrgAndData> refDatumMap, Collection<ReferenceDataType> dataTypes) {
        if (MapUtils.isEmpty(refDatumMap)) {
            return Collections.emptyMap();
        }
        Map<RefDatumOrgAndData, String> refDatumToTextMap = new HashMap<>();
        Lists.partition(new ArrayList<>(refDatumMap.entrySet()), getConfig().getDBQueryBatchSize()).forEach(
                batch -> {
                    Collection<Long> batchOrgIds = new HashSet<>();
                    Collection<String> batchDataIds = new HashSet<>();

                    batch.forEach(entry -> {
                        Long orgId = entry.getValue().getOrgId();
                        String dataId = entry.getValue().getDataId();
                        if (orgId != null) {
                            batchOrgIds.add(orgId);
                        }
                        if (dataId != null) {
                            batchDataIds.add(dataId);
                        }
                    });

                    refDatumToTextMap.putAll(referenceDatumService.getReferenceData(batchOrgIds, dataTypes, batchDataIds).stream()
                            .filter(Objects::nonNull)
                            .collect(Collectors.toMap(
                                    datum -> new RefDatumOrgAndData(datum.getOrg().getId(), datum.getDataType(), datum.getDataId()),
                                    ReferenceDatum::getDataText)));
                });
        return refDatumToTextMap;
    }

    /**
     * Applies the correct privacy label to an {@code IBaseDTO} data point,
     * using the available information to search for relevant reference data.
     *
     * See http://dev.patientsknowbest.com/home/privacy-labels for more details.
     *
     * @param dataPoint
     *            The data point to which the privacy label should be applied.
     */
    public void applyPrivacyLabels(IBaseDTO dataPoint) {
        applyPrivacyLabels(dataPoint, "");
    }

    /**
     * Applies the correct privacy label to an {@code IBaseDTO} data point,
     * using the available information to search for relevant reference data.
     *
     * Will also search for the reference data by orderServiceId, if provided.
     *
     * See http://dev.patientsknowbest.com/home/privacy-labels for more details.
     *
     * @param dataPoint
     *            The data point to which the privacy label should be applied.
     * @param orderServiceId
     *            The order service ID associated with the data point.
     */
    public void applyPrivacyLabels(IBaseDTO dataPoint, String orderServiceId) {

        // Firstly, default the data point to GENERAL
        dataPoint.setPrivacyFlag(GENERAL);

        // Org required to lookup reference data
        Long orgId = dataPoint.getSource().getOrgId();
        if (orgId != null) {
            // Then use any order service mappings
            if (StringUtils.isNotBlank(orderServiceId)) {
                applyPrivacyLabels(dataPoint, referenceDatumService.getReferenceDatum(orgId, ORDER_SERVICE, orderServiceId).getOrNull());
            }

            // TODO: Set privacy labels based on default privacy label for source team / org

            // Finally, use any hospital service mappings
            String specialty = dataPoint.getBaseFields().getHospitalServiceCode();
            if (StringUtils.isNotBlank(specialty)) {
                applyPrivacyLabels(dataPoint, referenceDatumService.getReferenceDatum(orgId, HOSPITAL_SERVICE, specialty).getOrNull());
            }
        }
    }

    private void applyPrivacyLabels(IBaseDTO dataPoint, ReferenceDatum datum) {
        if (datum != null) {
            LOGGER.info("Setting privacy label based on {} reference datum", datum.getDataType());
            dataPoint.getBaseFields().setGeneralHealthConsentRequired(datum.getGeneralHealthConsentRequired())
                    .setMentalHealthConsentRequired(datum.getMentalHealthConsentRequired())
                    .setSexualHealthConsentRequired(datum.getSexualHealthConsentRequired())
                    .setSocialCareConsentRequired(datum.getSocialCareConsentRequired());
        }
    }
}
