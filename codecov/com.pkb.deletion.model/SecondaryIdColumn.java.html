package com.pkb.deletion.model;

import com.pkb.deletion.model.errors.DeletionServiceError;
import com.pkb.deletion.repository.DeleteOperationRepository;
import io.vavr.control.Either;
import org.immutables.value.Value;

import java.util.List;
import java.util.Optional;
import java.util.UUID;
import java.util.stream.Collectors;

/**
 * A secondary id value on a table against which inbound references are made above and beyond
 * the primary key. Most common example: 'public_id'
 */
@Value.Immutable
public abstract class SecondaryIdColumn<TYPE, TABLE_PK_TYPE> implements IdColumn<TYPE, TABLE_PK_TYPE> {

    /**
     * We don't support - because we don't have to - self-references using anything other than the primary key
     * of a table.
     */
    @Override
    public Either<DeletionServiceError, Integer> expandSelfReferences(UUID operationId, String parentTableName, DeleteOperationRepository queryExecutor) {
        return Either.right(0);
    }

    @Override
    public String getIdJoinTableClause(String parentTableName) {
        return ", " + parentTableName + " id_join_table";
    }

    @Override
    public String buildFkReferenceWhereConditions(List<String> fkColumnNames, Optional<Class<?>> nonStandardForeignKeyDataType) {
        //Hack: we should really pass down the details of the parent table pk to get the correct column name and datatype here
        //Fortunately we don't have instances of tables with secondary ids where the primary key is anything other than
        //`id bigint primary key`
        String prefix = "doi.id_to_delete::bigint = id_join_table.id AND (";
        return fkColumnNames.stream().map(fkColumnName -> "id_join_table." + getName() + " = primary_table." + fkColumnName + getColumnCast(nonStandardForeignKeyDataType)).collect(Collectors.joining(" OR ", prefix, ")"));
    }

    public static <TABLE_PK_TYPE> SecondaryIdColumn<Long, TABLE_PK_TYPE> secondaryId(String columnName, List<InboundForeignKeyReference<?, Long>> inboundForeignKeyReferences) {
        return secondaryId(columnName, Long.class, inboundForeignKeyReferences);
    }

    public static <TYPE, TABLE_PK_TYPE> SecondaryIdColumn<TYPE, TABLE_PK_TYPE> secondaryId(String columnName, Class<TYPE> columnType, List<InboundForeignKeyReference<?, TYPE>> inboundForeignKeyReferences) {
        return ImmutableSecondaryIdColumn.<TYPE, TABLE_PK_TYPE>builder().name(columnName).inboundForeignKeys(inboundForeignKeyReferences).columnTypeClass(columnType).build();
    }
}
