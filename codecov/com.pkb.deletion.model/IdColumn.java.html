package com.pkb.deletion.model;

import com.pkb.deletion.model.errors.DeletionServiceError;
import com.pkb.deletion.repository.DeleteOperationRepository;
import io.vavr.control.Either;
import org.apache.commons.lang3.StringUtils;

import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.UUID;

/**
 * An ID column is a column that contains a unique(ish) value of some sort that
 * can be referenced by a foreign key from another table. -ish because there are some odd
 * cases like the terribly named "uniqueid" on menu_data which we reference from various places
 * but which can actually resolve to several rows in the table.
 * @param <TYPE> The type of the column
 * @param <TABLE_PK_TYPE> The type of the principal PK of the parent table (may be the same in the case of the
 *                       pk column itself). Only maintained to enforce type-safety in the graph.
 */
public interface IdColumn<TYPE, TABLE_PK_TYPE> {

    Map<Class<?>, String> CAST_STRING_LOOKUP = Map.of(
            UUID.class, "::uuid",
            Long.class, "::bigint"
    );

    String getName();

    Class<TYPE> getColumnTypeClass();

    /**
     * Inbound references from other tables to this id value that we will need to follow
     * and delete.
     */
    List<InboundForeignKeyReference<?, TYPE>> getInboundForeignKeys();

    default Either<DeletionServiceError, Integer> writeDependentDeletionLists(DeleteOperation parentDelete, DeleteOperationRepository queryExecutor) {
        return Either.sequenceRight(io.vavr.collection.Stream.ofAll(getInboundForeignKeys().stream().map(fk -> fk.writeDependentDeletionOperations(parentDelete, this, queryExecutor))))
                .map(childCounts -> childCounts.fold(0, Integer::sum));
    }

    /**
     * In some cases, tables contain references to themselves (e.g. sourcepersonid on pkbperson) We have to handle these
     * specially because of the recursion this entails.
     */
    Either<DeletionServiceError, Integer> expandSelfReferences(UUID operationId, String parentTableName, DeleteOperationRepository queryExecutor);

    /**
     * The delete operation table only references records by the primary key of the table. When we're searching for
     * records in a dependent table that references via a secondary id, we have to add a join to the parent table
     * to get the secondary id value to match against.
     */
    String getIdJoinTableClause(String parentTableName);

    @SuppressWarnings("OptionalUsedAsFieldOrParameterType")
    String buildFkReferenceWhereConditions(List<String> fkColumnNames, Optional<Class<?>> nonStandardForeignKeyDataType);

    /**
     * Since the delete operation table stores all ids as strings for simplicity, we have to cast when joining
     * to the primary table. To handle the few special cases where we put fk values in columns with the wrong
     * datatype, this method can be provided with an override if the referencing column has a type that differs
     * from the id column it points at.
     */
    @SuppressWarnings("OptionalUsedAsFieldOrParameterType")
    default String getColumnCast(Optional<Class<?>> nonStandardDataType) {
        return StringUtils.defaultString(CAST_STRING_LOOKUP.get(nonStandardDataType.orElse(getColumnTypeClass())));
    }

    default String getColumnCast() {
        return getColumnCast(Optional.empty());
    }

}
