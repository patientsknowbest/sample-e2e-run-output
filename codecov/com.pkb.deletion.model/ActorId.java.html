package com.pkb.deletion.model;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import org.immutables.value.Value;

import java.util.UUID;

@Value.Immutable
@JsonSerialize(as = ImmutableActorId.class)
@JsonDeserialize(as = ImmutableActorId.class)
public interface ActorId {
    UUID getId();
    Type getType();

    /**
     * We need to store the identity of the user that triggered the deletion.
     * Until we move to keycloak for all auth, there will be some (Phoenix especially)
     * users who can potentially trigger  deletions who don't exist in the main PHR database
     * as PKBPerson records, and the rest will be users authenticated by PHR. Both can be
     * identified by a UUID but it will be useful to know which database to look in to
     * resolve the UUID.
     */
    enum Type {
        KEYCLOAK,
        PKBPERSON
    }
}
