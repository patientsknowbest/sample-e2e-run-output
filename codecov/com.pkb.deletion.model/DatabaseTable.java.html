package com.pkb.deletion.model;

import com.fasterxml.jackson.annotation.JsonIgnore;
import com.pkb.deletion.model.errors.DeletionServiceError;
import com.pkb.deletion.repository.DeleteOperationRepository;
import io.vavr.control.Either;
import org.immutables.gson.Gson;
import org.immutables.value.Value;

import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;
import java.util.stream.Stream;

import static com.pkb.deletion.model.PkColumn.pk;
import static java.util.Collections.emptyList;

/**
 * Represents a database table from which we might want to delete records
 */
@Value.Immutable
@Gson.TypeAdapters
public abstract class DatabaseTable<PK_TYPE>  {

    private static final Map<String, DatabaseTable<?>> TABLE_BY_NAME = new HashMap<>();

    public abstract String getTableName();
    @JsonIgnore @Value.Redacted public abstract Class<PK_TYPE> getPkClass();
    @JsonIgnore @Value.Redacted public abstract PkColumn<PK_TYPE> getPkColumn();

    /**
     * In addition to the primary key column, against which most database references will be made,
     * some tables also have other, secondary unique ids. The most common example is "public_id" that
     * exists on various tables. We most follow references to these as well and delete dependent data.
     */
    @JsonIgnore @Gson.Ignore @Value.Redacted public abstract List<IdColumn<?, PK_TYPE>> getSecondaryIdColumns();

    @Value.Derived
    @JsonIgnore @Gson.Ignore @Value.Redacted public List<IdColumn<?, PK_TYPE>> getAllIdColumns() {
        return Stream.concat(Stream.of(getPkColumn()), getSecondaryIdColumns().stream())
                .collect(Collectors.toList());
    }

    /**
     * Some dependent services will have a bunch of objects that reference e.g. pkbperson.public_id. Rather than
     * having to enumerate and process these all separately, this flag allows us to define a "virtual" pkbperson table
     * for those services to hold the person ids that tie all the other objects together, but no attempt will be made
     * to validate. delete or rollback this table
     */
    @Value.Default
    public boolean isVirtual() { return false; }

    @Value.Check
    void postConstruct() {
        if(TABLE_BY_NAME.put(this.getTableName(), this) != null) {
            throw new IllegalStateException("More than one table defined for name " + this.getTableName());
        }
    }

    static DatabaseTable<?> forName(String name) {
        return TABLE_BY_NAME.get(name);
    }

    public Either<DeletionServiceError, Integer> buildDependentDeletionGraph(DeleteOperation deleteOperation, DeleteOperationRepository queryExecutor) {
        return Either.sequenceRight(io.vavr.collection.Stream.ofAll(Stream.concat(
                getAllIdColumns().stream().map(idColumn -> idColumn.expandSelfReferences(deleteOperation.getId(), getTableName(), queryExecutor)),
                getAllIdColumns().stream().map(idColumn -> idColumn.writeDependentDeletionLists(deleteOperation, queryExecutor))
        ))).map(descendantCounts -> descendantCounts.fold(0, Integer::sum));
    }




    public static DatabaseTable<Long> table(String name) {
        return  table(name, emptyList());
    }

    public static DatabaseTable<Long> table(String name, List<InboundForeignKeyReference<?, Long>> inboundForeignKeyReferences) {
        return  ImmutableDatabaseTable.<Long>builder().tableName(name).pkColumn(pk(Long.class, inboundForeignKeyReferences)).pkClass(Long.class).build();
    }


    public static DatabaseTable<Long> multiIdTable(String name, PkColumn<Long> pkColumn, List<IdColumn<?, Long>> idColumns) {
        return ImmutableDatabaseTable.<Long>builder().tableName(name).pkColumn(pkColumn).secondaryIdColumns(idColumns).pkClass(Long.class).build();
    }


    public static <PK_TYPE> ImmutableDatabaseTable.Builder<PK_TYPE> builderFor(Class<PK_TYPE> pkClass) {
        return ImmutableDatabaseTable.<PK_TYPE>builder().pkClass(pkClass);
    }

    public static ImmutableDatabaseTable.Builder<Long> builderForLong() {
        return ImmutableDatabaseTable.<Long>builder().pkClass(Long.class);
    }


}
