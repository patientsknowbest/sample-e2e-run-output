package com.pkb.deletion.model;

import com.pkb.deletion.model.errors.DeletionServiceError;
import com.pkb.deletion.repository.DeleteOperationRepository;
import io.vavr.collection.Stream;
import io.vavr.control.Either;
import org.immutables.value.Value;

import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.UUID;
import java.util.function.Function;
import java.util.function.Predicate;
import java.util.stream.Collectors;

/**
 * A primary key column on a table against which inbound foreign key references
 * may be made.
 */
@Value.Immutable
public abstract class PkColumn<TYPE> implements IdColumn<TYPE, TYPE> {

    @Override
    @Value.Default
    public String getName() {
        return "id";
    }

    abstract List<String> getSelfReferenceColumnNames();



    @Override
    public Either<DeletionServiceError, Integer> expandSelfReferences(UUID operationId, String parentTableName, DeleteOperationRepository queryExecutor) {

        if (getSelfReferenceColumnNames().isEmpty()) {
            return Either.right(0);
        }

        String schema = queryExecutor.getDeletionServiceSchema();

        String sqlPrefix = "INSERT INTO " + schema + ".delete_operation_id SELECT :operationId, primary_table." + getName() + "::text FROM " + parentTableName + " primary_table INNER JOIN " + schema + ".delete_operation_id doi ON doi.operation_id = :operationId AND (";

        String sqlSuffix = ") LEFT OUTER JOIN " + schema + ".delete_operation_id doi_existing ON primary_table." + getName() + " = doi_existing.id_to_delete" + getColumnCast() + " AND doi_existing.operation_id = :operationId WHERE doi_existing.id_to_delete IS NULL;";
        String sql = getSelfReferenceColumnNames().stream().map(selfRefColumn -> "primary_table." + selfRefColumn + " = doi.id_to_delete" + getColumnCast()).collect(Collectors.joining(" OR ", sqlPrefix, sqlSuffix));


        Function<Integer, Either<DeletionServiceError, Integer>> insertOperation = (prevCount) -> queryExecutor.runUpdateDeleteQuery(sql, Map.of("operationId", operationId));
        Function<DeletionServiceError, Either<DeletionServiceError, Integer>> foldErrorToZeroCount = err -> Either.right(0);
        Predicate<Either<DeletionServiceError, Integer>> resultIsZeroCount = result -> result.map(count -> count == 0).getOrElse(false);

        var recursionStream = Stream.iterate(insertOperation.apply(0),
                (prev) -> prev.fold(foldErrorToZeroCount, insertOperation)
        );

        return Either.sequenceRight(recursionStream.takeUntil(resultIsZeroCount))
                .map(seq -> seq.fold(0, Integer::sum));
    }


    /**
     * Not required. For the primary key the id values to be deleted are stored directly in
     * delete_operation_id.id_to_delete, so we don't have to join against the original table when looking for
     * dependent rows in referring tables, we can just join against delete_operation_id on its own.
     */
    @Override
    public String getIdJoinTableClause(String parentTableName) {
        return "";
    }

    @Override
    public String buildFkReferenceWhereConditions(List<String> fkColumnNames, Optional<Class<?>> nonStandardForeignKeyDataType) {
        return fkColumnNames.stream().map(fkColumnName -> "primary_table." + fkColumnName + " = doi.id_to_delete" + getColumnCast(nonStandardForeignKeyDataType)).collect(Collectors.joining(" OR ", "(", ")"));
    }



    public static PkColumn<Long> pk(List<InboundForeignKeyReference<?, Long>> inboundForeignKeyReferences) {
        return pk(Long.class, inboundForeignKeyReferences);
    }

    public static <TYPE> PkColumn<TYPE> pk(Class<TYPE> columnType, List<InboundForeignKeyReference<?, TYPE>> inboundForeignKeyReferences) {
        return ImmutablePkColumn.<TYPE>builder().inboundForeignKeys(inboundForeignKeyReferences).columnTypeClass(columnType).build();
    }

    public static PkColumn<Long> pk(List<InboundForeignKeyReference<?, Long>> inboundForeignKeyReferences, List<String> selfReferenceColumnNames) {
        return pk(Long.class, inboundForeignKeyReferences, selfReferenceColumnNames);
    }

    public static <TYPE> PkColumn<TYPE> pk(Class<TYPE> columnType, List<InboundForeignKeyReference<?, TYPE>> inboundForeignKeyReferences, List<String> selfReferenceColumnNames) {
        return ImmutablePkColumn.<TYPE>builder().inboundForeignKeys(inboundForeignKeyReferences).selfReferenceColumnNames(selfReferenceColumnNames).columnTypeClass(columnType).build();
    }


}
