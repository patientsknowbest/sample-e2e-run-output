package com.pkb.deletion.model;

import com.fasterxml.jackson.annotation.JsonIgnore;
import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.pkb.deletion.model.errors.DeletionServiceError;
import com.pkb.pubsub.aggregator.models.AggregateResponse;
import org.immutables.value.Value;

import java.util.List;
import java.util.Optional;
import java.util.UUID;
import java.util.stream.Collectors;

@Value.Immutable
@Value.Style(stagedBuilder = true, get = { "is*", "get*" })
@JsonSerialize(as=ImmutableAggregateDeleteOperationResponse.class)
@JsonDeserialize(as=ImmutableAggregateDeleteOperationResponse.class)
public interface AggregateDeleteOperationResponse extends AggregateResponse<AggregateDeleteOperationResponse, DeleteOperationResponse> {

    @Value.NonAttribute
    @JsonIgnore
    default List<DeletionServiceError> getAggregateErrors() {
        return getResponses().stream().map(DeleteOperationResponse::getError).filter(Optional::isPresent).map(Optional::get).collect(Collectors.toList());
    }

    @Value.Default
    default UUID getRequestId() {
        return getResponses().stream().findFirst().map(DeleteOperationResponse::getRequestId).orElseThrow();
    }

    @Value.Default
    default ActorId getActor() {
        return getResponses().stream().findFirst().map(DeleteOperationResponse::getActor).orElseThrow();
    }

    @Value.Default
    default DeleteOperationType getOperationType() {
        return getResponses().stream().findFirst().map(DeleteOperationResponse::getOperationType).orElseThrow();
    }



    @Value.Derived
    @JsonIgnore
    default DeleteRequestStatus getDeletionStatus() {
        switch(getStatus()) {
            case INCOMPLETE:
                return getOperationType().getIncompleteStatus();
            case TIMED_OUT:
                return getOperationType().getTimeoutStatus();
            default:
                return getResponses().stream().map(DeleteOperationResponse::getDeletionStatus).reduce(DeleteRequestStatus::combine).orElseThrow();
        }
    }


    @Override
    default AggregateDeleteOperationResponse combine(AggregateDeleteOperationResponse other) {
        return ImmutableAggregateDeleteOperationResponse.builder().addAllResponses(getResponses()).addAllResponses(other.getResponses()).addAllRequiredResponseServices(getRequiredResponseServices()).timedOut(isTimedOut() || other.isTimedOut()).build();
    }



}
