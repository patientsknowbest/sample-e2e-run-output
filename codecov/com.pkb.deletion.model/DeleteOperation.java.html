package com.pkb.deletion.model;

import org.jetbrains.annotations.NotNull;

import javax.annotation.Nullable;
import javax.annotation.ParametersAreNonnullByDefault;
import java.time.Instant;
import java.util.UUID;

/**
 * This is effectively an entity class but we're not using hibernate/jpa
 * to manage it because it create a whole bunch of performance issues around
 * persisting the delete_operation_id rows that reference it, particularly in the
 * context of the highly recursive tree structure that it represents, which we build
 * using a whole bunch of reduce operations over lots of nested streams.
 */
@ParametersAreNonnullByDefault
public class DeleteOperation {

    private final UUID id;
    private final UUID requestId;
    private final UUID parentOperationId;
    private final String tableName;
    private Instant createdDate;

    private final UUID requestingUserId;

    private final ActorId.Type requestUserIdType;

    /**
     * @see InboundForeignKeyReference#isOwning()
     */
    private final boolean owned;

    /**
     * This (and nextChildOrdinal) are used to ensure that we process sibling nodes in the deletion tree in the same
     * order that we create them. This is important because the same row in some tables is reachable at multiple
     * different points in the tree. If we don't delete in the same order we build the tree we can potentially end up
     * violating referential integrity constraints.
     */
    private final int ordinal;

    private int nextChildOrdinal = 0;

    public DeleteOperation(UUID id, UUID requestId, @Nullable UUID parentOperationId, String tableName, ActorId requestingUser, int ordinal, boolean owned) {
        this.id = id;
        this.requestId = requestId;
        this.parentOperationId = parentOperationId;
        this.tableName = tableName;
        this.requestingUserId = requestingUser.getId();
        this.requestUserIdType = requestingUser.getType();
        this.ordinal = ordinal;
        this.owned = owned;
    }

    public DeleteOperation(UUID id, UUID requestId, @Nullable UUID parentOperationId, Instant createdDate, String tableName, ActorId requestingUser, int ordinal, boolean owned) {
        this.id = id;
        this.requestId = requestId;
        this.parentOperationId = parentOperationId;
        this.tableName = tableName;
        this.requestingUserId = requestingUser.getId();
        this.requestUserIdType = requestingUser.getType();
        this.ordinal = ordinal;
        this.owned = owned;
        this.createdDate = createdDate;
    }


    @Nullable
    public UUID getParentOperationId() {
        return parentOperationId;
    }

    @NotNull
    public UUID getId() {
        return id;
    }

    public UUID getRequestId() {return requestId;}

    public String getTableName() {
        return tableName;
    }

    public Instant getCreatedDate() {
        return createdDate;
    }

    public void setCreatedDate(Instant createdDate) {this.createdDate = createdDate;}

    public boolean isOwned() {
        return owned;
    }

    public int getOrdinal() {
        return ordinal;
    }

    public ActorId getRequestingUser() {
        return ImmutableActorId.builder().id(requestingUserId).type(requestUserIdType).build();
    }

    DeleteOperation spawnChild(String tableName, boolean owned) {
        return new DeleteOperation(UUID.randomUUID(), getRequestId(), getId(), tableName, getRequestingUser(), nextChildOrdinal++, owned);
    }

    @SuppressWarnings("unchecked")
    public <T> DatabaseTable<T> getTableDefinition() {
        return (DatabaseTable<T>) DatabaseTable.forName(getTableName());
    }

}
