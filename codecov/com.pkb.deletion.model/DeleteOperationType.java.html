package com.pkb.deletion.model;

public enum DeleteOperationType {
    CREATE(DeleteRequestStatus.CREATING, DeleteRequestStatus.INVALID),
    EXECUTE(DeleteRequestStatus.EXECUTING, DeleteRequestStatus.FAILED),
    ROLLBACK(DeleteRequestStatus.ROLLING_BACK, DeleteRequestStatus.FAILED_TO_ROLL_BACK);


    private final DeleteRequestStatus incompleteStatus;
    private final DeleteRequestStatus timeoutStatus;

    private DeleteOperationType(DeleteRequestStatus incompleteStatus, DeleteRequestStatus timeoutStatus) {
        this.incompleteStatus = incompleteStatus;
        this.timeoutStatus = timeoutStatus;
    }

    DeleteRequestStatus getIncompleteStatus() {
        return incompleteStatus;
    }

    DeleteRequestStatus getTimeoutStatus() {
        return timeoutStatus;
    }
}
