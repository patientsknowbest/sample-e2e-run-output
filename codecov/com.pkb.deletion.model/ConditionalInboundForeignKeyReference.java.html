package com.pkb.deletion.model;

import org.immutables.value.Value;

/**
 * Represents an inbound foreign key reference to a table that is conditional on some other value in the referring table.
 * The primary example of this is menu_data, where some columns contain id references for some datatypes, so we have
 * to filter by datatype when searching for rows that refer to the parent table.
 */
@Value.Immutable
public interface ConditionalInboundForeignKeyReference<PK_TYPE, FK_TYPE> extends InboundForeignKeyReference<PK_TYPE, FK_TYPE> {

    String getFilterColumnName();
    String getFilterColumnValue();


    static <PK_TYPE, FK_TYPE> ConditionalInboundForeignKeyReference<PK_TYPE, FK_TYPE> conditionalFk(DatabaseTable<PK_TYPE> referencingTable, String filterColumnName, String filterColumnValue, String... fkColumnNames) {
        return ImmutableConditionalInboundForeignKeyReference.<PK_TYPE, FK_TYPE>builder().referencingTable(referencingTable).filterColumnName(filterColumnName).filterColumnValue(filterColumnValue).addFkColumnNames(fkColumnNames).isOwning(false).build();
    }

    static <PK_TYPE, FK_TYPE> ConditionalInboundForeignKeyReference<PK_TYPE, FK_TYPE> conditionalFk(DatabaseTable<PK_TYPE> referencingTable, String filterColumnName, String filterColumnValue, Class<?> nonStandardDataType, String... fkColumnNames) {
        return ImmutableConditionalInboundForeignKeyReference.<PK_TYPE, FK_TYPE>builder().referencingTable(referencingTable).filterColumnName(filterColumnName).filterColumnValue(filterColumnValue).addFkColumnNames(fkColumnNames).isOwning(false).nonStandardDataType(nonStandardDataType).build();
    }

    @Override
    default String getAdditionalWhereConditions() {
        return " AND " + getFilterColumnName() + "='" + getFilterColumnValue() + "'";
    }

}
