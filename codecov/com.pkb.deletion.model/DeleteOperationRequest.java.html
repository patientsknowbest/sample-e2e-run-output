package com.pkb.deletion.model;

import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
import com.fasterxml.jackson.databind.annotation.JsonSerialize;
import com.pkb.pubsub.aggregator.models.AggregatableRequest;
import com.pkb.pubsub.aggregator.models.AggregateResponse;
import org.immutables.value.Value;

import java.util.Collection;
import java.util.Set;
import java.util.UUID;

@Value.Immutable
@Value.Style(stagedBuilder = true)
@JsonSerialize(as=ImmutableDeleteOperationRequest.class)
@JsonDeserialize(as=ImmutableDeleteOperationRequest.class)
public interface DeleteOperationRequest extends DeleteOperationMessage, AggregatableRequest<AggregateDeleteOperationResponse, DeleteOperationResponse>, DeleteOperationRequestBase {

    @Override
    @Value.Default
    default UUID getAggregatableRequestId() { return UUID.randomUUID(); }
    DeleteOperationType getOperationType();

    @Override
    default AggregateDeleteOperationResponse getAggregateResponse(Collection<DeleteOperationResponse> responses, Set<String> requiredServices, boolean timedOut){
        return responses.stream()
                .map(it -> it.getAggregateResponse(requiredServices, timedOut))
                .reduce(AggregateResponse::combine)
                .orElseGet(() -> ImmutableAggregateDeleteOperationResponse.builder().requestId(getRequestId()).aggregatableRequestId(getAggregatableRequestId()).operationType(getOperationType()).actor(getActor()).addAllRequiredResponseServices(requiredServices).timedOut(timedOut).build());
    }
}
