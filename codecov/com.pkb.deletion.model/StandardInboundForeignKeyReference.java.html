package com.pkb.deletion.model;

import org.immutables.value.Value;

import java.util.List;

/**
 * A normal (non-conditional) foreign key reference
 */
@Value.Immutable
@Value.Style(get = {"is*", "get*"})
public interface StandardInboundForeignKeyReference<PK_TYPE, FK_TYPE> extends InboundForeignKeyReference<PK_TYPE, FK_TYPE> {

    static <PK_TYPE, FK_TYPE> StandardInboundForeignKeyReference<PK_TYPE, FK_TYPE> standardFk(DatabaseTable<PK_TYPE> referencingTable, String... fkColumnNames) {
        return ImmutableStandardInboundForeignKeyReference.<PK_TYPE, FK_TYPE>builder().referencingTable(referencingTable).fkColumnNames(List.of(fkColumnNames)).owning(false).build();
    }

    static <PK_TYPE, FK_TYPE> StandardInboundForeignKeyReference<PK_TYPE, FK_TYPE> owningFk(DatabaseTable<PK_TYPE> referencingTable, String... fkColumnNames) {
        return ImmutableStandardInboundForeignKeyReference.<PK_TYPE, FK_TYPE>builder().referencingTable(referencingTable).fkColumnNames(List.of(fkColumnNames)).owning(true).build();
    }

}
