package com.pkb.deletion.model;

public enum DeleteRequestStatus {
    CREATING, //Request received, scope of the deletion is being calculated
    VALIDATING, //Deletion tree is being validated
    VALID, //Deletion tree found valid, ready for execution
    INVALID, //Deletion tree invalid, deletion cannot proceed
    EXECUTING, //Deletion running now
    TO_CANCEL, //Deletion has been marked for rollback during execution
    SUCCEEDED, //Deletion completed successfully
    CANCELED, //Deletion was rolled back by external request before completing
    FAILED, //Deletion failed during execution
    ROLLING_BACK, //Deletion completed, but is now being rolled back manually
    ROLLED_BACK, //Deletion has successfully rolled back
    FAILED_TO_ROLL_BACK //Rollback was attempted but failed
    ;

    //Allows for reducing lots of statuses to a single, aggregate status. Statuses latter in the list
    //trump earlier ones.
    public DeleteRequestStatus combine(DeleteRequestStatus other) {
        return this.compareTo(other) > 0 ? this : other;
    }
}
