package com.pkb.tolven.entity;

import java.io.Serializable;
import java.util.Base64;
import java.util.Optional;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.persistence.Transient;

import org.jetbrains.annotations.NotNull;

@Entity
@Table(name = "user_credentials")
public class UserCredential implements Serializable {

    @Id
    @Column(name = "user_id")
    private Long userId;

    @Column(name = "certificate_base64")
    private String certificateBase64;

    @Column(name = "pkcs12_base64")
    private String pkcs12Base64;

    @Transient
    private transient String tolvenPassword;

    /**
     * PKB person id or principal name
     * 
     * @param id
     */
    public Long getUserId() {
        return userId;
    }

    public void setUserId(Long id) {
        this.userId = id;
    }

    public String getCertificateBase64() {
        return certificateBase64;
    }

    public byte[] getCertificateBytes() {
        return Base64.getDecoder().decode(getCertificateBase64() );
    }

    public void setCertificate(byte[] certificate) {
        setCertificateBase64(Base64.getEncoder().encodeToString(certificate));
    }

    public void setCertificateBase64(String certificate) {
        this.certificateBase64 = certificate;
    }

    public String getPkcs12Base64() {
        return pkcs12Base64;
    }

    public byte[] getPkcs12Bytes() {
        return Base64.getDecoder().decode(getPkcs12Base64());
    }

    public void setPkcs12Base64(String pkcs12Base64) {
        this.pkcs12Base64 = pkcs12Base64;
    }

    public void setPkcs12(byte[] pkcs12) {
        setPkcs12Base64(Base64.getEncoder().encodeToString(pkcs12));
    }

    public void setTolvenPasswordByString(@NotNull String tolvenPasswordString) {
        this.tolvenPassword = tolvenPasswordString;
    }

    /**
     * @return Plain text password, if set
     */
    public Optional<String> getTolvenPassword() {
        return Optional.of(tolvenPassword);
    }

    /**
     * Clear the plaintext password field of the object.
     * The hashed value ({@link #getTolvenPasswordHash()}) remains unchanged.
     */
    public void clearTolvenPasswordString() {
        this.tolvenPassword = null;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (!(o instanceof UserCredential)) {
            return false;
        }

        UserCredential that = (UserCredential) o;

        if (userId != null ? !userId.equals(that.userId) : that.userId != null) {
            return false;
        }
        if (certificateBase64 != null ? !certificateBase64.equals(that.certificateBase64) : that.certificateBase64 != null) {
            return false;
        }
        return pkcs12Base64 != null ? pkcs12Base64.equals(that.pkcs12Base64) : that.pkcs12Base64 == null;
    }

    @Override
    public int hashCode() {
        int result = userId != null ? userId.hashCode() : 0;
        result = 31 * result + (certificateBase64 != null ? certificateBase64.hashCode() : 0);
        result = 31 * result + (pkcs12Base64 != null ? pkcs12Base64.hashCode() : 0);
        return result;
    }
}
