package com.pkb.messagehasher.kmsclient;

import com.pkb.crypto.dto.AccountKeysDTO;
import com.pkb.kms.client.core.Kms;
import com.pkb.kms.shared.representation.KmsError;
import com.pkb.messagehasher.MetricsHelper;
import com.pkb.messagehasher.configuration.KmsConfig;
import io.vavr.control.Either;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.Optional;
import java.util.UUID;

import static java.util.Collections.singletonList;

@Service
public class KmsClientHelper {
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    KmsConfig kmsConfig;

    @Autowired
    Kms retrofitKms;

    @Autowired
    MetricsHelper metricsHelper;


    public Optional<AccountKeysDTO> getAccountKeys(UUID publicAccountId) {

        String requestId = UUID.randomUUID().toString();

        LOGGER.debug("KMS request {} publicAccountId {}", requestId, publicAccountId);
        Either<KmsError, List<AccountKeysDTO>> accountKeysTrusted = retrofitKms.getAccountKeysTrusted(requestId, singletonList(publicAccountId));

        return accountKeysTrusted
                .peekLeft(kmsError -> {
                    metricsHelper.kmsKeyHitCounter("error", kmsError.getType().toString()).increment();
                    LOGGER.debug("Got Error from KMS for request {} publicAccountId {} : {}", requestId, publicAccountId, kmsError);
                })
                .map(accountKeys -> {
                    metricsHelper.kmsKeyHitCounter("found", "").increment();
                    LOGGER.debug("KMS request {} accountKeysdtos {}", requestId, accountKeys);
                    return Optional.of(accountKeys.get(0));
                })
                .getOrElse(Optional.empty());
    }
}
