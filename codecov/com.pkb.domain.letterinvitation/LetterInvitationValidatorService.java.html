package com.pkb.domain.letterinvitation;

import com.pkb.entities.pub.PublicPKBPerson;
import io.vavr.control.Option;

import static com.pkb.entities.enums.UserStatus.PATIENTS_REGISTRATION_NOT_ALLOWED;
import static java.text.MessageFormat.format;
import static org.apache.commons.lang3.StringUtils.isBlank;

public class LetterInvitationValidatorService {

    Option<String> validatePerson(PublicPKBPerson person) {
        String validationError;

        if (person.isPatient()) {

            if (PATIENTS_REGISTRATION_NOT_ALLOWED.contains(person.getStatus())) {
                validationError = format("Registration not allowed for a patient with status: {0}", person.getStatus());
            } else {
                if (isBlank(person.getDateOfBirthString())) {
                    validationError = "Record missing a date of birth field";
                } else {
                    return Option.none();
                }
            }
        } else {
            validationError = format("Invitation is not allowed for this user type: {0}.", person.getUserType());
        }
        return Option.of(validationError);
    }
}
