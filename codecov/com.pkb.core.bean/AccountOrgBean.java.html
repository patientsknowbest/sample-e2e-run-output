package com.pkb.core.bean;

import com.google.common.collect.Table;
import com.pkb.core.AccountOrgLocal;
import com.pkb.core.entity.AccountOrg;
import com.pkb.util.CorrelationIdAroundAdvice;
import com.pkb.util.DbUtil;
import org.jetbrains.annotations.Nullable;

import javax.ejb.Local;
import javax.ejb.Stateless;
import javax.interceptor.Interceptors;
import javax.persistence.EntityManager;
import javax.persistence.PersistenceContext;
import javax.persistence.TypedQuery;
import java.util.Collection;
import java.util.List;
import java.util.Set;
import java.util.function.BiFunction;
import java.util.function.Function;

/**
 * Replacing (bit by bit) Tolven's beans for working with accounts and account keys.
 * First task: work with AccountOrg records.
 *
 * @author robwhelan
 */
@Stateless(mappedName = "AccountOrgBean")
@Interceptors(CorrelationIdAroundAdvice.class)
@Local(AccountOrgLocal.class)
public class AccountOrgBean implements AccountOrgLocal {

    @PersistenceContext
    private EntityManager em;

    /**
     * @deprecated DO NOT USE.
     * <p>
     * If you require this functionality use AccountOrgManager.findAccountOrg(...)
     */
    @Override
    @Nullable
    @Deprecated
    public AccountOrg findAccountOrg(long accountId, long orgId) {
        TypedQuery<AccountOrg> query = em.createQuery(
                        "SELECT ao FROM AccountOrg ao\n" +
                                "WHERE ao.accountId = :accountId\n" +
                                "AND ao.orgId = :orgId",
                        AccountOrg.class)
                .setParameter("accountId", accountId)
                .setParameter("orgId", orgId);

        return DbUtil.getExactlyOneResultOrNull(query);
    }

    @Override
    public Table<Long, Long, AccountOrg> findAccountOrgs(Set<Long> accountIds, Set<Long> orgIds, Function<AccountOrg, Long> key1, Function<AccountOrg, Long> key2) {
        BiFunction<Collection<Long>, Collection<Long>, List<AccountOrg>> q = (accountIdBatch, orgIdBatch) -> em.createQuery(
                        "SELECT ao FROM AccountOrg ao\n" +
                                "WHERE ao.accountId IN (:accountIds)\n" +
                                "AND ao.orgId IN (:orgIds)",
                        AccountOrg.class)
                .setParameter("accountIds", accountIdBatch)
                .setParameter("orgIds", orgIdBatch)
                .getResultList();

        return DbUtil.getInBatches(accountIds, orgIds, q, key1, key2);
    }
}
