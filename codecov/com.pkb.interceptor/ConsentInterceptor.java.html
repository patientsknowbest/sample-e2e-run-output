package com.pkb.interceptor;

import com.google.common.annotations.VisibleForTesting;
import com.google.common.collect.ImmutableMap;

import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
import com.pkb.app.entity.EHRRequestContext;
import com.pkb.app.entity.EHRRequestContext.AccountLinkType;

import org.apache.commons.lang3.StringUtils;
import org.apache.struts2.StrutsStatics;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.annotation.Nullable;

import java.util.UUID;

import static com.pkb.app.entity.EHRRequestContext.AccountLinkType.NO_LINK;
import static com.pkb.consent.model.AbstractConsentStatus.defaultConsent;
import static com.pkb.entities.enums.PropertyName.CONSENT_REVIEW_COMPLETED;
import static com.pkb.entities.enums.Route.NOT_CAPTURED;
import static com.pkb.util.Constants.NHS_LOGIN_CORRELATION_ID;

public class ConsentInterceptor extends MethodFilterInterceptor {

    private static final long serialVersionUID = -7761222164940252450L;
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @VisibleForTesting
    static final String REVIEW_CACHE_KEY = "ReviewConsentStatus";

    private static final Map<String, Tuple3<Route, AccountLinkType, Boolean>> ACTION_TO_ROUTE_LINKTYPE_CONSENTREQUIRED_MAP = ImmutableMap.of(
            "hl7Upload", Tuple.of(Route.HL7_API, AccountLinkType.ORG, Boolean.TRUE),
            "coreDevicesNotificationCallback", Tuple.of(Route.CORE_DEVICES, AccountLinkType.NO_LINK, Boolean.FALSE));

    @Autowired
    private EHRRequestContextManager ehrRequestContextManager;

    @Autowired
    private ContextUserUtil contextUserUtil;

    @Autowired
    private LoggedInUserUtil loggedInUserUtil;

    @Autowired
    private ContextUtil contextUtil;

    @Autowired
    private PKBWebUtil pkbWebUtil;

    @Autowired
    private AppIdentityUtil appIdentityUtil;

    @Autowired
    private CorrelationIdUtil correlationIdUtil;

    public enum ReviewConsentStatus {
        REQUIRED, SUPPRESSED, NOT_REQUIRED
    }

    @Override
    protected String doIntercept(ActionInvocation invocation) throws Exception {
        ActionContext context = invocation.getInvocationContext();
        Map<String, Object> session = context.getSession();
        HttpServletRequest request = (HttpServletRequest) context.get(StrutsStatics.HTTP_REQUEST);
        Route route = appIdentityUtil.isNhsAppSession(request) ? Route.WEBAPP_NHS_APP : Route.WEBAPP;

        PersonAuthInfo principal = loggedInUserUtil.getPrincipal().orElse(null);
        Long contextUserId = contextUserUtil.getContextUserId();

        if (request.getSession().getAttribute(PKBConstants.ROUTE) != null) {
            route = (Route) request.getSession().getAttribute(PKBConstants.ROUTE);
        }

        UUID correlationId = correlationIdUtil.getCorrelationIdFromStringOrGenerateNewOne(getCorrelationIdFromRequest(request));
        EHRRequestContext requestContext = createEHRRequestContext(principal, contextUserId, invocation, route, correlationId);
        contextUtil.setEHRRequestContext(request, requestContext);

        if (isConsentModelActive(principal, contextUserId)) {
            checkConsentReviewStatusForLoggedInUser(session);
        }

        if (requestContext != null) {
            correlationIdUtil.putIfAbsent(correlationId);
        }

        try {
            return invocation.invoke();
        } finally {
            correlationIdUtil.remove();
        }
    }

    private boolean isConsentModelActive(PersonAuthInfo loggedInUser, Long contextUserId) {
        return (loggedInUser != null && loggedInUser.isPatient()) || contextUserId != null;
    }

    // TODO WARNING this is (mostly) copied into RestUtil.createEHRRequestContext()
    private EHRRequestContext createEHRRequestContext(PersonAuthInfo loggedInUser, Long contextUserId,
                                                      ActionInvocation invocation, Route route, UUID correlationId) {

        if (loggedInUser == null) {
            // No logged in user, but this is expected for HL7 and other uploads
            return Optional
                    .ofNullable(ACTION_TO_ROUTE_LINKTYPE_CONSENTREQUIRED_MAP.get(invocation.getInvocationContext().getName()))
                    .map(r -> ImmutableLoggedOutEHRRequestContext.of(r._2, r._1, ConsentStatus.builder().required(r._3).build(), correlationId))
                    .orElse(ImmutableLoggedOutEHRRequestContext.of(NO_LINK, (route == null ? NOT_CAPTURED : route), ConsentStatus.builder().build(), correlationId));
        }


        switch (loggedInUser.getUserType()) {
            case PATIENT:
                if ((contextUserId == null) || contextUserId.equals(loggedInUser.getId())) {
                    // Patient's own account
                    return ehrRequestContextManager.createEHRRequestContextForOwner(loggedInUser,
                            correlationId, route);
                } else /* context user is set, and is not the logged-in user */ {
                    // Carer
                    return ehrRequestContextManager.createEHRRequestContextForCarer(loggedInUser,
                            contextUserId, correlationId, route);
                }

            case REG_CLINICIAN:
            case INSTITUTE_ADMIN:
                Team team = loggedInUserUtil.getLoggedInUserPrimaryTeam();
                boolean breakTheGlassActive = pkbWebUtil.hasBreakTheGlassForPatient(contextUserId);
                if (team == null) {
                    return ehrRequestContextManager.createEHRRequestContextForIndividualClinician(loggedInUser,
                            contextUserId, correlationId, route);
                } else {
                    EHRRequestContext ehrRequestContext = ehrRequestContextManager.createEHRRequestContextForTeamClinician(loggedInUser, team,
                            contextUserId, correlationId, route, breakTheGlassActive);

                    return ehrRequestContext;
                }
            case ORG_COORD:
                return ehrRequestContextManager.createEHRRequestContextForOrgCoord(loggedInUser,
                        correlationId, route);

            case SUPER_ADMIN:
                // superadmin has decrypt access for employees, and potentially non-team clinicians to reset passwords.
                // that's NYI
                return ehrRequestContextManager.createEHRRequestContextForPkbAdmin(loggedInUser,
                        correlationId, route);
            case PRIVACY_OFFICER:
                return ehrRequestContextManager.createEHRRequestContextForPrivacyOfficer(
                        loggedInUser, correlationId, route);

        default:
            // pkb admin, employee
            return ImmutableLoggedInUserEHRRequestContext.of(NO_LINK, route, defaultConsent(), correlationId, loggedInUser);
        }
    }

    private void checkConsentReviewStatusForLoggedInUser(Map<String, Object> session)
            throws ConsentReviewRequiredException {
        ReviewConsentStatus status;
        // Check the session to see if we have existing status
        Object val = session.get(REVIEW_CACHE_KEY);
        if (val != null) {
            status = (ReviewConsentStatus) val;
        } else {
            // Unknown status - check the person properties
            if (!loggedInUserUtil.isUserLoggedIn()) {
                // e.g. on logout.  short-circuit; nothing to do here!
                return;
            }

            PKBPerson person = loggedInUserUtil.getLoggedInPerson();
            // property will return false unless we find the property and it's set "true"
            boolean reviewRequired = person.isPatient()
                    && !person.checkPropertyFlag(CONSENT_REVIEW_COMPLETED);

            if (reviewRequired || isSelfRegistrationViaNHSLogin()) {
                status = ReviewConsentStatus.REQUIRED;
            } else {
                status = ReviewConsentStatus.NOT_REQUIRED;
            }

            session.put(REVIEW_CACHE_KEY, status);
        }

        if (status == ReviewConsentStatus.REQUIRED) {
            // Suppress the check so we don't go into a loop (this interceptor
            // will run again on the first wizard page, so it mustn't throw again)
            session.put(REVIEW_CACHE_KEY, ReviewConsentStatus.SUPPRESSED);
            throw new ConsentReviewRequiredException("Suppressed consent check");
        }
    }

    @Override
    protected boolean applyInterceptor(ActionInvocation invocation) {
        String actionName = invocation.getProxy().getActionName();

        for (String excludeActionName : excludeMethods) {
            LOGGER.debug("checking {} against current action {}", excludeActionName, actionName);
            if (StringUtils.equalsIgnoreCase(actionName, excludeActionName)) {
                return false;
            }
        }

        return super.applyInterceptor(invocation);
    }

    private boolean isSelfRegistrationViaNHSLogin() {
        return loggedInUserUtil.getNhsLoginAuthenticationDetails()
                .flatMap(NhsLoginInfo::isSelfReg)
                .orElse(false);
    }

    public EHRRequestContextManager getEhrRequestContextManager() {
        return ehrRequestContextManager;
    }

    public void setEhrRequestContextManager(
            EHRRequestContextManager ehrRequestContextManager) {
        this.ehrRequestContextManager = ehrRequestContextManager;
    }

    @Nullable
    private String getCorrelationIdFromRequest(HttpServletRequest request) {
        return (String) request.getSession().getAttribute(NHS_LOGIN_CORRELATION_ID);
    }
}
