package com.pkb.interceptor;

import com.opensymphony.xwork2.ActionInvocation;
import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
import com.pkb.entities.enums.UserType;
import com.pkb.util.LoggedInUserUtil;
import org.slf4j.Logger;
import org.springframework.beans.factory.annotation.Autowired;

import java.lang.invoke.MethodHandles;
import java.util.Arrays;
import java.util.List;
import java.util.regex.Pattern;

import static java.util.Collections.emptyList;
import static java.util.stream.Collectors.toList;
import static org.slf4j.LoggerFactory.getLogger;

/**
 * Allows filtering of actions so that only certain users are allowed to access them
 * Introduced for PHR-4694
 */
public class UserTypeInterceptor extends MethodFilterInterceptor {

    private static final Logger LOGGER = getLogger(MethodHandles.lookup().lookupClass());
    private static final Pattern COMMA_SEPARATED_LIST_PATTERN = Pattern.compile("[ ]*,[ ]*");

    private LoggedInUserUtil loggedInUserUtil;
    private List<UserType> allowedUserTypes;


    @Override
    protected String doIntercept(ActionInvocation actionInvocation) throws Exception {
        UserType userType = loggedInUserUtil.getLoggedInUser().getUserType();
        if(!allowedUserTypes.contains(userType)){
            return "redirectHomeResult";
        }
        return actionInvocation.invoke();
    }
    private List<UserType> generateUserTypesFromString(String val) {
        if (val == null) {
            return emptyList();
        } else {
            String[] list = COMMA_SEPARATED_LIST_PATTERN.split(val);
            return Arrays.stream(list).map(UserType::valueOf).collect(toList());
        }
    }


    @Autowired
    public void setLoggedInUserUtil(LoggedInUserUtil loggedInUserUtil) {
        this.loggedInUserUtil = loggedInUserUtil;
    }

    public void setAllowedUserTypes(String allowedUserTypes) {
        this.allowedUserTypes = generateUserTypesFromString(allowedUserTypes);

    }


}
