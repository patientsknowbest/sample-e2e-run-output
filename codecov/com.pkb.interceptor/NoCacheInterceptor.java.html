package com.pkb.interceptor;


import com.opensymphony.xwork2.ActionContext;
import com.opensymphony.xwork2.ActionInvocation;
import com.opensymphony.xwork2.ObjectFactory;
import com.opensymphony.xwork2.Result;
import com.opensymphony.xwork2.config.entities.ResultConfig;
import com.opensymphony.xwork2.inject.Inject;
import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
import org.apache.struts2.StrutsStatics;
import org.apache.struts2.result.StreamResult;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.servlet.http.HttpServletResponse;
import java.util.Map;

/**
 * We were doing this in a filter, but there was no way to detect a file download from there,
 * and if we set no-cache headers for a file download, it can't be read by IE over SSL.
 */
public class NoCacheInterceptor extends MethodFilterInterceptor {
    private static final long serialVersionUID = -1426629727063490768L;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    private ObjectFactory objectFactory;

    @Override
    public String doIntercept(ActionInvocation invocation) throws Exception {
        Object action = invocation.getAction();

        LOGGER.debug("RedirectMessageInterceptor: before call into {}", action.getClass());

        //
        // execute this before handling the result (which would commit the stream...)
        //
        invocation.addPreResultListener((invocation1, resultCode) -> {
            LOGGER.debug("Handling no-cache headers in PreResultListener");
            try {

                // need to get the actual class to be used by the result, to also catch custom subclasses
                Result result = null;
                Map<String, ResultConfig> results = invocation1.getProxy().getConfig().getResults();
                ResultConfig resultConfig = results.get(resultCode);
                if ((resultConfig != null) && (resultConfig.getClassName() != null)) {
                    result = (Result) objectFactory.buildBean(resultConfig.getClassName(), null, false);
                }

                if ((result instanceof StreamResult)) {
                    // set short cache time?
                } else {
                    // no cache
                    ActionContext context = invocation1.getInvocationContext();
                    HttpServletResponse response = (HttpServletResponse) context.get(StrutsStatics.HTTP_RESPONSE);
                    response.setCharacterEncoding("UTF-8");
                    response.setHeader("Cache-Control", "max-age=0, no-store, no-cache, must-revalidate");
                    response.setHeader("Pragma", "no-cache");
                    response.setDateHeader("Expires", 0); // prevents caching at the proxy server
                    // also: strongly encourage standards mode in old IE browsers:
                    response.setHeader("X-UA-Compatible", "IE=Edge");
                }

            } catch (Exception e) {
                LOGGER.error("non-fatal: error in NoCacheInterceptor with result {} to action {}",
                        resultCode, invocation1.getAction().getClass(), e);
                // DO NOT rethrow; this is not essential processing
            }
        });

        LOGGER.debug("invoking action: {}", action);
        String resultCode = invocation.invoke();

        LOGGER.debug("NoCacheInterceptor: after call into {}: {}", action.getClass(), resultCode);
        return resultCode;
    }

    @Inject
    public void setObjectFactory(ObjectFactory objectFactory) {
        this.objectFactory = objectFactory;
    }

}
