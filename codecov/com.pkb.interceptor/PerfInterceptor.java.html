package com.pkb.interceptor;

import com.opensymphony.xwork2.ActionInvocation;
import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
import com.pkb.util.LoggedInUserUtil;
import io.prometheus.client.Histogram;
import io.prometheus.client.Summary;
import org.apache.commons.lang3.ArrayUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;

import java.util.Arrays;

public class PerfInterceptor extends MethodFilterInterceptor {

    private static final long serialVersionUID = -1;

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    private LoggedInUserUtil loggedInUserUtil;

    private static final Histogram actionHistograms = Histogram.build()
            .linearBuckets(0, 0.25, 40)
            .name("actionHistogram")
            .help("Histogram of actions")
            .labelNames("action", "method")
            .register();

    private static final Summary actionSummaries = Summary.build()
            .name("webActionDuration")
            .help("Latencies of Struts/Tomcat actions")
            .labelNames("action", "method")
            .quantile(0.5, 0.05)
            .quantile(0.9, 0.01)
            .quantile(0.95, 0.005)
            .quantile(0.99, 0.001)
            .register();

    @Override
    public String doIntercept(ActionInvocation invocation) throws Exception {
        String actionId = invocation.getProxy().getActionName() + "." + invocation.getProxy().getMethod();

        // timer() factory method works as a getOrAdd, so it's safe to call it
        // and it will either create a new timer (1st time) or get an existing one
        String resultCode;
        String[] labels = Arrays.copyOfRange(ArrayUtils.addAll(actionId.split("\\."), "notProvided", "notProvided"), 0, 2);
        Summary.Timer prometheusTimer = actionSummaries.labels(labels).startTimer();
        Histogram.Timer histogram = actionHistograms.labels(labels).startTimer();
        try {
            resultCode = invocation.invoke();
        } finally {
            prometheusTimer.observeDuration();
            histogram.observeDuration();
        }
        return resultCode;
    }
}
