package com.pkb.interceptor;

import com.opensymphony.xwork2.ActionInvocation;
import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
import com.pkb.exception.ConsentDeclarationRequiredException;
import com.pkb.util.ContextUtil;
import org.slf4j.Logger;
import org.springframework.beans.factory.annotation.Autowired;

import javax.servlet.http.HttpServletRequest;
import java.lang.invoke.MethodHandles;

import static org.apache.commons.lang3.StringUtils.equalsIgnoreCase;
import static org.apache.struts2.StrutsStatics.HTTP_REQUEST;
import static org.slf4j.LoggerFactory.getLogger;


public class ConsentRedirectInterceptor extends MethodFilterInterceptor {

    private static final long serialVersionUID = -3231054273000170002L;

    private static final Logger LOGGER = getLogger(MethodHandles.lookup().lookupClass());

    private ContextUtil contextUtil;

    @Override
    protected boolean applyInterceptor(ActionInvocation invocation) {
        return isNotOnExcludedActionList(invocation.getProxy().getActionName()) && super.applyInterceptor(invocation);
    }

    @Override
    protected String doIntercept(ActionInvocation invocation) throws Exception {
        HttpServletRequest request = (HttpServletRequest) invocation.getInvocationContext().get(HTTP_REQUEST);
        contextUtil.getEHRRequestContext(request).getLoggedInContext().ifPresent(requestContext -> {
            if ((requestContext.isPatient() || requestContext.getContextUserId().isPresent()) &&
                    !requestContext.getConsentStatus().canViewData()) {
                throw new ConsentDeclarationRequiredException("Can't proceed setting user permissions: consent status is not valid: "
                        + requestContext.getConsentStatus());
            }
        });

        return invocation.invoke();
    }


    @Autowired
    public void setContextUtil(ContextUtil contextUtil) {
        this.contextUtil = contextUtil;
    }

    private boolean isNotOnExcludedActionList(String actionName) {
        return excludeMethods.stream()
                .noneMatch(actionNameToBeExcluded -> {
                    LOGGER.trace("checking {} against current action {}", actionNameToBeExcluded, actionName);
                    return equalsIgnoreCase(actionName, actionNameToBeExcluded);
                });
    }

}
