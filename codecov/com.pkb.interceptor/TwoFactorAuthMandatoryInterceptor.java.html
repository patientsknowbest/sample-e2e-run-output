package com.pkb.interceptor;

import com.opensymphony.xwork2.ActionInvocation;
import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
import com.pkb.exception.TwoFactorAuthenticationMandatoryException;
import com.pkb.security.PKBAuthenticationDetails;
import com.pkb.util.LoggedInUserUtil;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

@Component
public class TwoFactorAuthMandatoryInterceptor extends MethodFilterInterceptor {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    @Autowired
    private LoggedInUserUtil loggedInUserUtil;

    @Autowired
    private PKBInterceptorHelper pkbInterceptorHelper;

    @Override
    protected String doIntercept(ActionInvocation invocation) throws Exception {
        if (isMandatorySecondFactorAuthenticationRequired()) {
            LOGGER.debug("Madatory two factor authentication detected, redirecting to enable 2fa");
            throw new TwoFactorAuthenticationMandatoryException();
        }
        return invocation.invoke();
    }

    @Override
    protected boolean applyInterceptor(ActionInvocation invocation) {
        boolean result = pkbInterceptorHelper.applyInterceptorByActionName(invocation,
                excludeMethods) && super.applyInterceptor(invocation);
        LOGGER.debug("Apply interceptor: {}.", result);
        return result;
    }

    private boolean isMandatorySecondFactorAuthenticationRequired() {
        boolean result = false;
        Object authDetails = loggedInUserUtil.getAuthenticationDetails();
        if (PKBAuthenticationDetails.class.isInstance(authDetails)) {
            PKBAuthenticationDetails pkbAuthDetails = PKBAuthenticationDetails.class.cast(authDetails);
            result = pkbAuthDetails.twoFactorAuthMandatory() && !pkbAuthDetails.twoFactorAuthDone();
        }
        return result;
    }
}
