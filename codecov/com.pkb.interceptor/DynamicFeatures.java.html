package com.pkb.interceptor;

import com.pkb.util.AppIdentityUtil.AppIdentity;
import org.immutables.value.Value.Default;
import org.immutables.value.Value.Immutable;
import org.immutables.value.Value.Style;

import static com.pkb.util.AppIdentityUtil.AppIdentity.PKB_WEB;

@Immutable
@Style(allParameters = true, stagedBuilder = true)
public interface DynamicFeatures {
    boolean allowLogout();

    boolean allowPatientSwitch();

    boolean allowTeamSwitchAtLogin();

    boolean allowTeamSwitchAfterLogin();

    boolean allowPatientBanner();

    boolean allowCustomTeamColors();

    boolean allowInviteWhenComposingMessage();

    boolean allowNHSAppAttachmentDownload();

    boolean forceNHSRedirectHomeAfterTimeout();

    boolean showForgotPasswordLinkOnLoginScreen();

    boolean renderTeamFooter();

    boolean renderNavigation();

    boolean renderNhsBanner();

    boolean renderNhsWayfinderBanner();

    boolean renderCarePlanHeadingLinks();

    boolean renderPlanOutgoingLinks();

    boolean renderNhsStyle();

    boolean renderNhsFooter();

    /**
     * This string is be used to identify in what context our pages are getting renderred.
     * Currently our app is displayed in:
     * - EMIS SSO sessions,
     * - embedded into NHS App and
     * - PKB Web (normal).
     */
    @Default
    default AppIdentity appIdentity() {
        return PKB_WEB;
    }
}
