package com.pkb.interceptor;

import com.google.common.annotations.VisibleForTesting;
import com.opensymphony.xwork2.ActionContext;
import org.apache.struts2.ServletActionContext;
import org.apache.struts2.dispatcher.HttpParameters;
import org.apache.struts2.dispatcher.Parameter;
import org.apache.struts2.util.TokenHelper;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.Map;
import java.util.Optional;

public class CSRFTokenHelper {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    public static final String TOKEN_FIELD = "pkb.csrf.token";
    public static final String HEADER_TOKEN_FIELD = "Pkb-Csrf-Token";

    public Optional<String> getToken() {
        HttpParameters params = getParameters();
        Parameter parameter = params.get(TOKEN_FIELD);

        if (!parameter.isDefined()) {
            Optional<String> tokenFromHeader = Optional.ofNullable(getTokenFromHeader());
            if (!tokenFromHeader.isPresent()) {
                LOGGER.warn("Could not find CSRF token mapped to token name: {}", TOKEN_FIELD);
            }
            return tokenFromHeader;
        }
        return Optional.of(parameter.getValue());
    }

    public boolean isValidToken() {
        return getToken().map(this::handleToken).orElseGet(this::handleMissingToken);
    }

    public String generateCSRFToken() {
        return TokenHelper.generateGUID();
    }

    public String setTokenIfAbsent() {
        return setTokenIfAbsent(TOKEN_FIELD);
    }

    public String setTokenIfAbsent(String tokenName) {
        Map<String, Object> session = getSession();
        try {
            return (String) session.computeIfAbsent(tokenName, t -> generateCSRFToken());
        } catch (IllegalStateException e) {
            throw new IllegalArgumentException(
                    "Error creating HttpSession due response is committed to client. You can use the CreateSessionInterceptor or create the HttpSession from your action before the result is rendered to the client: ",
                    e);
        }
    }

    private boolean handleToken(String token) {
        Map<String, Object> session = getSession();
        String sessionToken = (String) session.get(TOKEN_FIELD);
        if (!token.equals(sessionToken)) {
            LOGGER.warn("CSRF token does not match required one. Expected: {}, got: {}", sessionToken, token);
            return false;
        }
        return true;
    }

    private boolean handleMissingToken() {
        LOGGER.warn("No token found for token name {} -> Invalid token ", TOKEN_FIELD);
        return false;
    }

    @VisibleForTesting
    Map<String, Object> getSession() {
        return ActionContext.getContext().getSession();
    }

    @VisibleForTesting
    HttpParameters getParameters() {
        return ActionContext.getContext().getParameters();
    }

    @VisibleForTesting
    String getTokenFromHeader() {
        return ServletActionContext.getRequest().getHeader(HEADER_TOKEN_FIELD);
    }
}
