package com.pkb.interceptor;

import com.google.common.annotations.VisibleForTesting;
import com.opensymphony.xwork2.ActionContext;
import com.opensymphony.xwork2.ActionInvocation;
import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
import com.pkb.common.config.PhrConfig;
import com.pkb.entities.enums.Route;
import com.pkb.util.AppIdentityUtil;
import com.pkb.util.SessionUtil;
import org.springframework.beans.factory.annotation.Autowired;

import javax.servlet.http.HttpServletRequest;
import java.util.Optional;

import static com.opensymphony.xwork2.ActionContext.getContext;
import static com.pkb.util.PKBConstants.ROUTE;
import static org.apache.struts2.StrutsStatics.HTTP_REQUEST;

/**
 * Intercept request to set appropriate info in Session to later determine what functionality should be turned on/off Example : For EMIS SSO view, logout and back-to-patients links
 * need to be disabled
 */
public class FeatureInterceptor extends MethodFilterInterceptor {

    private final PhrConfig config;
    private final SessionUtil sessionUtil;
    private final AppIdentityUtil appIdentityUtil;

    @Autowired
    public FeatureInterceptor(
            PhrConfig config,
            SessionUtil sessionUtil,
            AppIdentityUtil appIdentityUtil) {
        this.sessionUtil = sessionUtil;
        this.appIdentityUtil = appIdentityUtil;
        this.config = config;
    }

    @Override
    protected String doIntercept(ActionInvocation actionInvocation) throws Exception {
        ActionContext context = getActionContext();
        HttpServletRequest request = (HttpServletRequest) context.get(HTTP_REQUEST);
        boolean nhsAppSession = nhsAppSession(request);
        boolean wayfinderSession = isWayfinderSession(request);
        boolean nhsSession = nhsAppSession || wayfinderSession;

        // only render the Wayfinder-specific banner if the nhsStyle parameter is present but it is not an NHS App (native) session
        boolean renderWayfinderBanner = wayfinderSession && !nhsAppSession;

        DynamicFeatures features = ImmutableDynamicFeatures.builder()
                .allowLogout(notEmisPortalSession(context) && !nhsAppSession)
                .allowPatientSwitch(notEmisPortalSession(context))
                .allowTeamSwitchAtLogin(notSsoPortalSession(context) && !nhsAppSession && !isOtpSession(context))
                .allowTeamSwitchAfterLogin(notSsoPortalSession(context) && !nhsAppSession)
                .allowPatientBanner(notSsoPortalSession(context) && config.isPatientBannerEnabled())
                .allowCustomTeamColors(!nhsAppSession)
                .allowInviteWhenComposingMessage(!nhsAppSession)
                .allowNHSAppAttachmentDownload(nhsAppSession)
                .forceNHSRedirectHomeAfterTimeout(nhsAppSession)
                .showForgotPasswordLinkOnLoginScreen(!nhsAppSession)
                .renderTeamFooter(!nhsAppSession)
                .renderNavigation(!nhsAppSession)
                .renderNhsBanner(nhsSession)
                .renderNhsWayfinderBanner(renderWayfinderBanner)
                .renderCarePlanHeadingLinks(!nhsAppSession)
                .renderPlanOutgoingLinks(!nhsAppSession)
                .renderNhsStyle(wayfinderSession)
                .renderNhsFooter(nhsSession)
                .appIdentity(appIdentityUtil.resolve(request))
                .build();

        sessionUtil.setDynamicFeatures(features, context);

        return actionInvocation.invoke();
    }

    private boolean isWayfinderSession(HttpServletRequest request) {
        if (sessionUtil.getDynamicFeatures() != null) {
            return sessionUtil.getDynamicFeatures().renderNhsStyle() || "true".equals(request.getParameter("nhsStyle"));
        } else {
            return "true".equals(request.getParameter("nhsStyle"));
        }
    }

    private boolean isOtpSession(ActionContext context) {
        return Optional.ofNullable(context.getSession()).map(session -> (Route) session.get(ROUTE)).filter(Route.WEBAPP_OTP::equals).isPresent();
    }

    private boolean notEmisPortalSession(ActionContext context) {
        HttpServletRequest request = (HttpServletRequest) context.get(HTTP_REQUEST);
        return !appIdentityUtil.isEmisPortalSession(request);
    }

    private boolean notSsoPortalSession(ActionContext context) {
        HttpServletRequest request = (HttpServletRequest) context.get(HTTP_REQUEST);
        return !appIdentityUtil.isSsoPortalSession(request);
    }

    private boolean nhsAppSession(HttpServletRequest request) {
        return appIdentityUtil.isNhsAppSession(request);
    }

    @VisibleForTesting
    ActionContext getActionContext() {
        return getContext();
    }
}
