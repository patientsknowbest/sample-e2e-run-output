package com.pkb.interceptor;

import com.opensymphony.xwork2.ActionInvocation;
import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
import com.pkb.service.user.impl.PersonContactManager;
import org.apache.struts2.ServletActionContext;
import org.springframework.beans.factory.annotation.Autowired;

/**
 * Cleans up dangling transaction (ones that have not been commited or rolled back properly )
 * This prevents some parts of the code from affecting whole application
 * (because a timed out/unfinished transaction is associated with thread until it's killed ).
 */
public class TransactionCleanupInterceptor  extends MethodFilterInterceptor {

    private static final long serialVersionUID = -8544255300012313L;

    //Used only to be able to lookup current transaction
    @Autowired
    private PersonContactManager personContactManager;

    @Override
    protected String doIntercept(ActionInvocation invocation) throws Exception {
        try {

            return invocation.invoke();
        }
        finally
        {
            if (!Boolean.TRUE.equals(ServletActionContext.getServletContext().getAttribute("pkb.unit.test.execution"))) {
                String transactionSite = invocation.getProxy().getConfig().getClassName() + "." + invocation.getProxy().getMethod();
                personContactManager.cleanupTransaction(transactionSite);
            }
        }
    }
}
