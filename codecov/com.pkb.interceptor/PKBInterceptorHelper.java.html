package com.pkb.interceptor;

import com.opensymphony.xwork2.ActionInvocation;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Component;

import java.util.Set;

import static org.apache.commons.lang3.StringUtils.equalsIgnoreCase;

/**
 * Helper class which contains common login for PKB interceptors.
 */
@Component
public class PKBInterceptorHelper {

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());

    /**
     * Checks whether the interceptor should be applied for the action or not.
     *
     * @param invocation
     *            The current action.
     * @param excludeMethods
     *            The actions which should be skipped.
     * @return <code>true</code> when the action is not in the skip list, false otherwise.
     */
    public boolean applyInterceptorByActionName(ActionInvocation invocation,
                                                Set<String> excludeMethods) {
        boolean result = true;
        String actionName = invocation.getProxy().getActionName();
        for (String excludeActionName : excludeMethods) {
            LOGGER.trace("Checking {} against current action {}.", excludeActionName, actionName);
            if (equalsIgnoreCase(actionName, excludeActionName)) {
                result = false;
                break;
            }
        }
        return result;
    }
}
