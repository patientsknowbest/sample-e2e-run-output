package com.pkb.interceptor;

import com.google.common.annotations.VisibleForTesting;
import com.opensymphony.xwork2.ActionInvocation;
import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
import com.pkb.datamodel.Email;
import com.pkb.datamodel.user.PersonIdsWithNhsLoginStatus;
import com.pkb.kms.client.core.Kms;
import com.pkb.kms.shared.representation.LoginRequest;
import com.pkb.kms.shared.representation.NhsLoginRequest;
import com.pkb.security.BaseAuthenticationProvider.TypedAuthentication;
import com.pkb.security.NhsLoginAuthenticationDetails;
import com.pkb.service.nhslogin.impl.NhsLoginRegistrationManager;
import com.pkb.service.team.TeamManager;
import com.pkb.user.entity.PKBPerson;
import com.pkb.util.ContextUtil;
import com.pkb.util.LoggedInUserUtil;
import io.vavr.control.Option;
import io.vavr.control.Try;
import org.apache.struts2.StrutsStatics;
import org.jetbrains.annotations.NotNull;
import org.slf4j.Logger;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.security.core.context.SecurityContextImpl;

import javax.servlet.http.HttpServletRequest;
import java.lang.invoke.MethodHandles;
import java.util.Base64;
import java.util.UUID;

import static com.pkb.datamodel.ImmutableDisclosingEmail.disclosingEmail;
import static com.pkb.util.Constants.NHS_LOGIN_IDS;
import static com.pkb.util.Constants.NHS_LOGIN_INFO;
import static com.pkb.util.PKBWebUtil.SPRING_SECURITY_CONTEXT;
import static java.util.Objects.nonNull;
import static org.slf4j.LoggerFactory.getLogger;


public class NhsLoginInterceptor extends MethodFilterInterceptor {

    private static final Logger LOG = getLogger(MethodHandles.lookup().lookupClass());

    @Autowired
    private Kms kmsClient;

    @Autowired
    private ContextUtil contextUtil;

    @Autowired
    private LoggedInUserUtil loggedInUserUtil;

    @Autowired
    private PKBInterceptorHelper pkbInterceptorHelper;

    @Autowired
    private TeamManager teamManager;

    @Autowired
    private NhsLoginRegistrationManager nhsLoginRegistrationManager;

    @Override
    protected boolean applyInterceptor(ActionInvocation invocation) {
        return pkbInterceptorHelper.applyInterceptorByActionName(invocation, excludeMethods)
                && super.applyInterceptor(invocation);
    }

    @Override
    protected String doIntercept(ActionInvocation invocation) throws Exception {

        HttpServletRequest request = (HttpServletRequest) invocation.getInvocationContext().get(StrutsStatics.HTTP_REQUEST);
        SecurityContextImpl securityContext = (SecurityContextImpl) request.getSession().getAttribute(SPRING_SECURITY_CONTEXT);

        TypedAuthentication<NhsLoginAuthenticationDetails> info = (TypedAuthentication<NhsLoginAuthenticationDetails>) request.getSession().getAttribute(NHS_LOGIN_INFO);
        PersonIdsWithNhsLoginStatus ids = (PersonIdsWithNhsLoginStatus) request.getSession().getAttribute(NHS_LOGIN_IDS);

        if (nonNull(info) && nonNull(ids) && loggedInUserUtil.isUserLoggedIn() && nonNull(securityContext)) {
            PKBPerson principal = loggedInUserUtil.getLoggedInPerson();
            // Make sure the emails match!
            if (emailMatches(principal, disclosingEmail(info.getName()))) {

                String requestId = Try.of(() -> contextUtil.getEHRRequestContext(request).getCorrelationId().toString())
                        .getOrElse(() -> UUID.randomUUID().toString());
                String credentials = (String) securityContext.getAuthentication().getCredentials();

                LoginRequest loginRequest = new LoginRequest(ids.getPublicId(), credentials.toCharArray(), false);
                NhsLoginRequest nhsLoginRequest = new NhsLoginRequest(ids.getPublicId(), Base64.getDecoder().decode(info.getCredentials()));

                kmsClient.login(requestId, loginRequest)
                        .flatMap($ -> kmsClient.addNhsLoginCredentials(requestId, nhsLoginRequest))
                        .peek(result -> {
                            request.getSession().removeAttribute(NHS_LOGIN_INFO);
                            request.getSession().removeAttribute(NHS_LOGIN_IDS);
                        })
                        .peekLeft(err -> LOG.warn("Found nhs credentials but failed to create in kms: {}", err));

                var authDetails = info.getDetails();
                if (authDetails.isSelfReg().orElse(false)) {
                    authDetails.getTeamCode()
                            .map(teamManager::getTeamByCode)
                            .ifPresent(team -> nhsLoginRegistrationManager
                                    .grantAccessToTeam(contextUtil.getEHRRequestContext(request), authDetails.getRoute(),
                                            principal, team));
                }
            }
        }

        return invocation.invoke();
    }

    @VisibleForTesting
    protected boolean emailMatches(PKBPerson principal, @NotNull Email email) {
        var targetEmail = Option.of(email);
        return principal.getContacts().stream()
                .anyMatch(pc -> targetEmail.equals(pc.getContactIfEmail()));
    }
}
