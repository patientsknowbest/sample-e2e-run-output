package com.pkb.interceptor;

import com.opensymphony.xwork2.ActionContext;
import com.opensymphony.xwork2.ActionInvocation;
import com.opensymphony.xwork2.inject.Inject;
import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
import com.pkb.util.CookieUtil;
import com.pkb.util.LoggedInUserUtil;
import com.pkb.util.PKBWebUtil;
import com.pkb.util.jsp.beans.SupportedLanguages;
import org.apache.struts2.dispatcher.HttpParameters;
import org.apache.struts2.interceptor.I18nInterceptor;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.slf4j.MDC;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.security.web.savedrequest.SavedRequest;
import org.springframework.security.web.savedrequest.SavedRequestConstant;

import javax.servlet.http.Cookie;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.util.Arrays;
import java.util.List;
import java.util.Map;
import java.util.Optional;
import java.util.stream.Collectors;

import static com.pkb.action.HttpParametersRebuilder.modify;
import static java.util.Collections.emptyList;
import static org.apache.commons.lang3.StringUtils.isNotBlank;
import static org.apache.struts2.ServletActionContext.getRequest;
import static org.apache.struts2.ServletActionContext.getResponse;

/**
 * This interceptor handles a few minor post-login tasks like
 * synching locale and handling redirect URLs. It should probably
 * be split into several more focussed interceptors
 */
public class UserLoginInterceptor extends MethodFilterInterceptor {

    public static final String PKB_LOCALE_COOKIE = "pkbLocale";

    private static final long serialVersionUID = 1L;
    private static final String LOGGED_IN_USER_MDC_KEY = "loggedInUserId";

    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private static final String DEFAULT_LANG = "en";
    public static final String ACCEPT_LANGUAGE = "Accept-Language";

    @Autowired
    private PKBWebUtil pkbWebUtil;

    @Autowired
    private LoggedInUserUtil loggedInUserUtil;

    @Inject
    private SupportedLanguages supportedLanguages;

    @Override
    public String doIntercept(ActionInvocation invocation) throws Exception {
        Map<String, Object> session = invocation.getInvocationContext().getSession();

        if (loggedInUserUtil.isUserLoggedIn()) {
            MDC.put(LOGGED_IN_USER_MDC_KEY, Long.toString(loggedInUserUtil.getLoggedInUserId()));
        }

        syncLocale(invocation, supportedLanguages);

        if (session.containsKey(SavedRequestConstant.SPRING_SECURITY_SAVED_REQUEST_KEY)) {
            pkbWebUtil.copyRedirectUrl((SavedRequest) session.get(SavedRequestConstant.SPRING_SECURITY_SAVED_REQUEST_KEY));
        }

        try {
            return invocation.invoke();
        } finally {
            MDC.remove(LOGGED_IN_USER_MDC_KEY);
        }
    }

    private static void syncLocale(ActionInvocation invocation,
                                   SupportedLanguages supportedLanguages) {
        ActionContext ac = invocation.getInvocationContext();

        HttpParameters params = ac.getParameters();
        //Parameter is never null. It is always returned first value from array converted to string
        String langCode = params.get(I18nInterceptor.DEFAULT_PARAMETER).getValue();// request_locale
        HttpServletRequest request = getRequest();

        if (isNotBlank(langCode)) {
            // add/update cookie
            HttpServletResponse response = getResponse();
            LOGGER.info("locale requested: {}; setting to cookie", langCode);
            CookieUtil.setCookie(request, response, PKB_LOCALE_COOKIE, langCode, CookieUtil.DURATION_30_DAYS);
        } else {
            // No param in this request
            // is there an existing cookie locale we can set into the session?
            Cookie localeCookie = CookieUtil.readCookie(request, PKB_LOCALE_COOKIE);
            if (localeCookie != null) {
                // the I18nInterceptor picks this up and set it in the session
                modify(params).put(I18nInterceptor.DEFAULT_PARAMETER, localeCookie.getValue());
            } else {
                modify(params)
                        .put(I18nInterceptor.DEFAULT_PARAMETER, getBrowserLanguageOrDefault(request, supportedLanguages));
            }
        }
    }

    private static String getBrowserLanguageOrDefault(HttpServletRequest request, SupportedLanguages supportedLanguages) {
        List<String> supportedLangNames = getSupportedLanguages(supportedLanguages);
        Optional<String> matchingLang = getPreferredLanguages(request).stream()
                .filter(supportedLangNames::contains)
                .findFirst();
        return matchingLang.orElse(DEFAULT_LANG);
    }

    private static List<String> getPreferredLanguages(HttpServletRequest request) {
        String acceptLanguage = request.getHeader(ACCEPT_LANGUAGE);
        if (acceptLanguage != null) {
            return Arrays.stream(acceptLanguage.split(","))
                    .map(UserLoginInterceptor::getLanguageWithoutVariant)
                    .collect(Collectors.toList());
        } else {
            return emptyList();
        }
    }

    private static List<String> getSupportedLanguages(SupportedLanguages supportedLanguages) {
        return supportedLanguages.getLanguages()
                .stream()
                .map(locale -> locale.getLocale().getLanguage())
                .collect(Collectors.toList());
    }

    private static String getLanguageWithoutVariant(String val) {
        return val.split(";")[0]
                .split("-")[0];
    }

}
