package com.pkb.interceptor;

import com.google.common.annotations.VisibleForTesting;
import com.google.common.collect.Sets;
import com.opensymphony.xwork2.ActionContext;
import com.opensymphony.xwork2.ActionInvocation;
import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
import com.pkb.common.config.PhrConfig;
import org.apache.struts2.ServletActionContext;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpSession;
import java.util.Set;

import static org.apache.struts2.StrutsStatics.HTTP_REQUEST;

public class CsrfTokenInterceptor extends MethodFilterInterceptor {

    public static final String INVALID_TOKEN_CODE = "invalid.token";
    private static final Logger LOGGER = LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
    private final static Set<String> CSRF_PROTECTED_METHODS = Sets.newHashSet("POST", "PUT", "DELETE", "PATCH");

    private final PhrConfig config;

    private final CSRFTokenHelper csrfTokenHelper;

    public CsrfTokenInterceptor(CSRFTokenHelper csrfTokenHelper, PhrConfig config) {
        this.config = config;
        this.csrfTokenHelper = csrfTokenHelper;
    }

    @Override
    protected boolean applyInterceptor(ActionInvocation invocation) {
        if (!config.isCSRFProtectionEnabled()) {
            return false;
        }
        ActionContext actionContext = invocation.getInvocationContext();
        HttpServletRequest httpRequest = (HttpServletRequest) actionContext.get(HTTP_REQUEST);
        String method = httpRequest.getMethod();
        return CSRF_PROTECTED_METHODS.contains(method) && super.applyInterceptor(invocation);
    }

    @Override
    protected String doIntercept(ActionInvocation invocation) throws Exception {
        LOGGER.debug("Intercepting invocation to check for valid CSRF token.");
        return handleToken(invocation);
    }

    protected String handleToken(ActionInvocation invocation) throws Exception {
        if (csrfTokenHelper.isValidToken()) {
            return invocation.invoke();
        } else {
            return INVALID_TOKEN_CODE;
        }
    }

    @VisibleForTesting
    HttpSession getSession() {
        return ServletActionContext.getRequest().getSession(true);
    }
}
