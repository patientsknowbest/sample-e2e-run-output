package com.pkb.interceptor;

import com.opensymphony.xwork2.ActionInvocation;
import com.opensymphony.xwork2.interceptor.AbstractInterceptor;
import com.pkb.common.config.PhrConfig;
import org.apache.logging.log4j.LogManager;
import org.apache.logging.log4j.Logger;

import javax.servlet.http.HttpServletRequest;
import java.util.Arrays;
import java.util.Enumeration;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

import static org.apache.struts2.ServletActionContext.getRequest;

public class NhsParamLeakInterceptor extends AbstractInterceptor {

    private static final String NHS_APP_IOS = "nhsapp-ios";
    private static final String NHS_APP_ANDROID = "nhsapp-android";

    private final PhrConfig config;
    private static final Logger LOG = LogManager.getLogger(NhsParamLeakInterceptor.class);

    public NhsParamLeakInterceptor(PhrConfig config) {
        this.config = config;
    }

    @Override
    public String intercept(ActionInvocation invocation) throws Exception {
        checkForNhsUrlLeaks();
        return invocation.invoke();
    }

    private static Set<String> userAgentsToCheck = new HashSet<>(Arrays.asList(NHS_APP_IOS,NHS_APP_ANDROID));

    private static Set<String> stemWhitelist = new HashSet<>(List.of("/test/nodecorate_"));

    private static Set<String> parameterWhitelist = new HashSet<>(Arrays.asList("id", "tab", "subTab", "contextUserId", "clearContext",
            "patientId", "PKBID", "code", "inviteeId", "userId", "call", "title", "firstName", "lastName",
            "orgId", "referer", "upcomingPageCount", "pastPageCount", "consultTeamId", "page_id", "eui64Code", "deviceId",
            "invitationId", "error", "conversationId", "contentDisposition","sourceFileId", "dataPointId", "uniqueId",
            "msg", "searchIdType", "searchId", "searchLevel", "measurementTypeId", "month","year", "planId",
            "selectedPlanUniqueId", "selectId", "teamPublicId", "visitedTeamPublicIds", "testResultTypeId", "messageId",
            "fileUploadErrorMessage", "rangeWanted", "rangeName", "period", "periodsToDisplay", "periodsBack", "periodsForward", "reload",
            "editingId","deleteId","deletePersonalLibraryId", "testResultTypeIdSource", "selectResultId", "gender",
            "encounterEventUniqueId", "nhsStyle", "e", "requestType", "request_locale", "tkid", "teamId", "reportDocumentId",
            "externalMessageId"));

    private void checkForNhsUrlLeaks() throws Exception {
        HttpServletRequest request = getRequest();
        String userAgent = request.getHeader("User-Agent");
        String xRequestedWith = request.getHeader("X-Requested-With");
        if (!"XMLHttpRequest".equals(xRequestedWith) && "GET".equals(request.getMethod()) &&
                shouldCheckUserAgent(userAgent) && !stemIsWhitelisted(request.getRequestURI()) ) {
            Enumeration<String> paramNames = request.getParameterNames();
            while (paramNames.hasMoreElements()) {
                String paramName = paramNames.nextElement();
                if (!parameterWhitelist.contains(paramName)) {
                    String message = "PHR-7601: 'GET "+request.getRequestURL()+"' leaking parameter :'"+paramName+"'";
                    if (config.isSecurityCheckUrlParametersEnabled()) {
                        throw new RuntimeException(message);
                    } else {
                        LOG.error(message);
                    }
                }
            }
        }
    }

    private boolean shouldCheckUserAgent(String userAgent) {
        return config.isSecurityCheckAllUserAgents() ||  userAgentsToCheck.contains(userAgent);
    }

    private boolean stemIsWhitelisted(String stem) {
        return stemWhitelist.stream().anyMatch(stem::startsWith);
    }


}
